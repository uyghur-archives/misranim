-ۋۇ ئاناڭنى ،قاپلان يۈرۈكىدە ناشتىلىق يەۋالغان ئوخشىمامسەن قانجۇق ؟!
-نۇمۇسسىز ئوغرى، بىر قىز بالىنىڭ نەرسىسىنى ئوغرىلاپ تۇرۇپ يەنە خىجىل بولماي قول تەگكۈزۈۋاتامسەن . ۋىجدانىنى ئىت يەپ كەتكەن ئەبلەخ ......
    بۇ قىزنىڭ مەسۇم چىرايىدا ،كىيىكنىڭكىدەك شوخ كۆزلىرىدە بىرخىل جاسارەت ئەكىس ئىتىپ تۇراتتى .شۇ جاسارەتنىڭ تۈپەيلىمۇ ياكى ئىيول قۇياشنىڭ قىزدۇرىشىمۇ ،مەڭزى  كەچ كۈزدە قىزىرىپ پىشقان ئاناردەك قىپقىزىل بۇلۇپ كەتكەن ئىدى . قۇندۇزدەك قاپقارا قاشلىرى ئوتتۇرسىدا ،بادامدەك بۇرنىنىڭ ئۈستىدە بىلىنەر بىلىنمەس تەر تامچىسى پارقىرايتتى .
    بۇ گەپنى ئاڭلاپ مۇستافا «پىخ......»قىلىپ كۈلۈپ  سالدى -دە ، ئەتراپقا ئوبدان بىر قارىۋەتكەندىن كىيىن ،تاماكىسىنى يەرگە تاشلاپ خۇرۇم بەتىنكىسىدە ئىزىپ يەنجىۋەتتى .بۇنىڭدىن ئىلياس بىر نىمىنى سەزگەندەك قىلىپ قىزنى قۇيۋېتىپ ،ئارقىسىغا بۇرۇلۇپ ماڭغان مۇستافانىڭ كەينىدىن كۆندۈرۈلگەن پىستە ئىتتەك ئەگەشتى .
   -ئاكا،مەن ئۇ قىز بالىنى قاۋاقخانىغا تاپشۇرۇپ بىرەيمۇ يا........؟ ئۆزىغۇ خېلى چىرايلىق بىر نىمە ئىكەن .كىرىمىمۇ خېلى ياخشى بۇلىدىغاندەك قىلىدۇ.،ئاۋۋال سىز كۆڭلىڭىزنى ئېچىۋېلىڭ بولمىسا،يا........-ئەمدى ئۇ سۆزلىرىنى داۋاملاشتۇرالمىدى . ئۇنىڭ پۇتلىرى يەردىن كۈتۈرۈلۈپ ئىنچىكە بوينى مۇستافانىڭ كۈچلۈك قوللىرى ئارىسدا قالدى. مۇستافانىڭ ئۇزۇن چاچلىرى يۆگۈۋالغان كۆزلىرىدىن بىرخىل غەزەپ ئۇچقۇن يالقۇنجان تۇراتتى .
-ئېغىزىڭنى يۇم ،-ئۇنىڭ تەمكىن چىققان سۆزلىرىدىن غەزەپ نەپرىتى ئىپادىلىنىپ تۇراتتى .بۇرھان مۇستافانىڭ دۇلىسىغا ئاستا بىرنى مۇشلاپ مەنىلىك بىرنى قاراپ قۇيۇپ يولىنى داۋام ئەتتى .
   تۇيۇقسىز بۇرھانغا تېلفۇن كەلدى :
-بۇرھان ئاكا ،خەلق مەيدانىنىڭ ئۇدۇلىدىكى دىسكوخانىدا جىدەل چىقتى يەتتە سەككىز ھارام تاماق بۇ يەرنى قالايمىقان قىلىۋەتتى ...................
  - نان قېپىلار ،شۇنچىلىك ئىشنىمۇ قاملاشتۇرالماي ماڭا تېلفۇن قىلىشتىڭما ؟؟؟ھەممىگە مەن چاپامدىم- دەپلا تېلفۇننى قۇيۋەتتى .ئاندىن كەينىگە بۇرۇلۇپ ئارقىسىدىن كىلىۋاتقان يىگىرمە نەچچە ئادەمنى ئۇ قاۋاقخانىغا بۇيرۇپ قۇيۇپ ئۆزى مۇستافا بىلەن يولىنى داۋام ئەتتى .
     مۇستافا ،مەردان ،بۇرھان ئۈچەيلەن بىرگە تاماق يىيىش ئۈچۈم مۇستافانىڭ داچىسىغا قايتتى .بۇلار مۇستافانىڭ خىيالچان ھالىتىگە ،سۇرالغان سۇئاللارغا ھە ياكى ياق دەپلا پارۋاسىز جاۋاپ بىرىشلىرىگە ھەيران قالغان بولسىمۇ،بىراق مۇستافادىن بولغان ئەھۋالنى سۇراشنى توغرا تاپمىدى .
      تاماق ۋاقتىدا بۇرھان مۇستافانىڭ جىمجىتلا ئولتۇرۋېلىشىدىن ئىچى سىقىلىپ :
-ھەي ئاغىنە ،ئەجەپ يۈزى ئېچىلمىغان قىزدەكلا نازلىنىپ ئولتۇرسەنغۇ ؟؟؟-ئۇ يەنە مۇستافانىڭ جىمجىت ئولتۇغانلىقىنى كۈرۈپ كۆڭلى بىر نىمىنى سەزدى بولغاي ،مۇستافانى جىمجىت قالدۇرۇشنى ئويلاپ ئۈنلۈك ئاۋازدا تەنە ئارلاش ،-بۈگۈنكى تاماق سەل تۇزلۇغ بۇلۇپ قالدىمۇ نېمە ؟؟؟ قۇپە مەردان تاماقنى سىرتتىن يەيلى ،-دەپ سىرتقا ماڭغاندا مۇستافانىڭ خوشلاشقىلىمۇ چىقماي غەمكىن ئولتۇرغانلىقىنى كۈرۈپ ئىشىكنى ئاستا يېپىپ قۇيۇپ جىمجىتلا چىقىپ كەتتى .
    مۇستافا ھوجرىسىدىكى كارۋاتتا ئاسمانغا يۈلىنىپ ،قۇلىنى بېشىغا قۇيۇپ دېرىزىدىن چۈشىۋاتقان ئاي نۇرىغا قاراپ خىياللار قاينىمىغا چۈكۈپ كەتتى . ئۇنىڭ مەقسەتسىزلار كۈلۈشلىرىدىن نىمىلەرنى ئويلاۋاتقانلىقىنى بىلگىلى بولمايتتى :«ھەقىقەتەن گۈزەل شۇ قىياقتەك قاشلار شەھلا كۆزلەر .... يەنە تېخى ئۇنىڭ ئۇغۇللارغا خاس غەيرەت جاسارىتى چۇ تېخى ،ئۇ زادى كىمدۇر ؟ئۇنىڭ كىملىكىنى نىمىشقىمۇ سۇرىۋالمىغان بولغىيتىم ........ئەگەر ئۇنى قايتا ئۇچراتسام ........»-ئۇ شۇلارنى خىيال قىلغىنىچە چۈشلەر قوينىغا كىرىپ كەتتى .
     ئەتتىگندە بۇرھان ئۇ ياتقان ئۆيگە كىرىپ كەلدى .دەل شۇ ۋاقىتتا ئۇنىڭ تېلفۇنى ئەنسىزلا سايراپ كەتتى .بۇرھان تېلفۇندا قىسقىلا پاراڭلاشقاندىن كىيىن ئەھۋالنى مۇستافاغا چۈشەندۈردى .ئەسلىدە ئاخشام جىدەل چىقارغانلار بىلەن ئۇرۇشقان بۇرادەرلەر ئچىدىن بەش ئادەم ئېغىر يارلىنىپ بىر ئادەم ئۆلگەن ئىدى .بۇنى ئاڭلىغان مەردان بۇ ئىشلارغا مۇستافانىڭ ئارامىنى بۇزۇشنى توغرا كۆرمەي بىر نەچچەيللەنى ئەگەشتۈرۈپ تەڭ كىچىدىلا قاۋاقخانىغا باردى ،بىراق بۇ كەسكىن جىدەل چوڭىيىپ مەرداننىڭ چەكلەش دائىرسىدىن ھالقىپ كەتكەندىن كىيىن ساقچىلار بۇ ئىشقا ئارلىشىپ مەردان ۋە ئۇنىڭ كەينىدىكى بىر نەچچەيلەننى تۇتۇپ كەتكەن .تەكشۈرۈشتە ئىنىقلىنىشىچە ئاخشامقى جىدەلدە ئىككى تەرەپتىن يارلانغانلارنىڭ سانى 50دىن ئېشىپ كەتكەن 6ئادەم ئۆلگەن ئىدى.
    بۇ گەپلەرنى ئاڭلاپ مۇستافا ئەتتىگەنلىك تاماقنىمۇ يىمەي ئۇدۇل ساقچى باشلىقىنىڭ ئۆيى تەرپكە ماڭدى ،يول بويى ئۇنىڭ كاللىسىنى بۇ ئىشلارنى قانداق ھەل قىلىش كېرەك دىگەن خىيال چىرمىۋالغان ئىدى .ساقچى باشلىقىنىڭ ئۆيىگە يىتىپ كەلگەندە ھاقچى باشلىقى .ئۆيىدە يوق بۇلۇپ چىقتى .بۇرھان ساقچى باشلىقىنىڭ مەكتەپكە كىتىپ بارغان 18ياشلار ئەتراپىدىكى ئوغلىنى كۈرۈپ كاللىسىغا بىر ئەقىل كەلدى -دە،مۇستافانىڭ قۇلىقىغا بىر نىمىلەرنى پىچىرلىدى .
       مۇستافا ساقچى باشلىقىغا تېلفۇن قىلدى .
      - ئەتتىگەنلىكىڭىز خەيرىلىك بولسۇن ساقچى باشلىقى ئەپەندى ،ئەتتىگەن يىگەن تامىقىڭىز سىڭىدىغاندەك قىلامدۇ نېمە؟ تاماق خېلى ئوخشىغاندەك قىلىدۇ بىراق يېرىمىنى ئاشۇرۇپ قۇيۇپسىزغۇ؟تېتىمىغان بولسا مەن قايتا ئەتكۈزۈپ بىرەي -مۇ؟
        ساقچى باشلىقى مۇستافانىڭ ئۆيىگە بېرىپ بولغانلىقىنى ھېس قىلدى -دە،جىددىيلىك بىلەن مۇستافادىن سورىدى:
       -ئىنىم ئۆيگە بېرىپ ئاۋارە بوپسەن ئەمەسمۇ ؟مەن ئەتتىگەنلىك خىزمەتكە كەلگەن ئىدىم ،ئۆيگە كەچ تەرەپلەردە قايتارمەنمىكىن بەك ئالدىراش بولمىساڭ مىنى كەچتە ئىزدەپ كەلگىن !!!
       -ياق ساقچى باشلىقى ، مۇستافا ئەزەلدىىن بىكارغا ئاۋارە بۇلۇپ يۈرمەيدۇ ،ئۇنداقتا مەن ساقلاپ تۇراي ئايالىڭ ھەقاچان بىرەر سائەتتىن كىيىن قايتىپ كىلەرمىكىن ،ھەقىقەتەن چىرايلىق ئىكەن جۇمۇ ئۇ خوتۇن ، ھەقاچان پەيزى باردۇ ؟
        -مىنى ساقلاپ تۇر مۇستافا ھازىرلا بارىمەن -بۇ نۆۋەت ساقچى باشلىقى جىددىيلىك بىلەن جاۋاپ بەردى .
         ساقچى باشلىقى ئۆيىگە كىرىپلا كىرسلودا ئولتۇرغان مۇستافا بىلەن بۇرھاننى كۈرۈپ ھەيران قالدى .
      -ئىشىك قۇلۇپلاقلىق ئىدىغۇ ئۆيگە قانداق كىردىڭلا ،-ھەيرانلىق بىلەن سورىدى ساقچى باشلىقى .
      -ئۆيىڭىزنىڭ ئىشىكى بەكلە ئاجىز ئىكەن باشلىق ئەپەندى .بىكار ۋاقىت تاپقاندا ئالماشتۇرۋەتكۈزەرسىز !!!
      -بۇنچە ئەتتىگەندە نىمىگە ئىزدەپ كەلگەن ئىدىڭىز خەۋەر قىلىۋەتسىڭىزلا ئالدىڭىزغا بارار ئىدىم .
      -ياق ساقچى باشلىقى سىزنى ئاۋارە قىلىپ يۈرمەي دىدىم شۇ ، ئاخشام دوسلىرىم چاقچاقلىشىۋاتسا يېقىن كۈرۈپ مىھمان قىلىمەن دەپ قېشىڭىزغا ئەكىلىۋاپتىكەنسىز ،بىزنى يات كۆرمەپسىز سىزگە كۆپتىن -كۆپ تەشەككۈر ،زىياپەت ئاخىرلاشقان بولسا ئۇلار ئۈچۈن ھۆرمىتىمنى بىلدۈرۋالسام ،بىز مىھمان بولغىلى قايتا بىر كەلسەك خالىسىوڭىز دوستلىرىمنى قايتۇرۇپ كىتەيمىكىن دېگەن ئىدىم ؟-ساقچى باشلىقى مۇستافانىڭ مەقسىتىنى چۈشەندى بولغاي :
      - ئۇلار ئادەم ئۆلتۈرۈپتۇ ،مىنىمۇ بۇ خىزمەتنىڭ نېنىنى يىگىلى قۇيۇڭ ،بۇ ئىش مىنىڭ قۇلۇمدىن كەلمەيدۇ .ئەگەر ئۆلگەنلەرنىڭ ئائىلە تاۋابىئاتلىرى ئەرىز سۇنسا ئۇنىڭ ھۆددىسىدىن ......
      -ئەنسىرمەڭ ساقچى باشلىقى ھەممە ئىشلارنى ئۆزۈم ھەل قىلىمەن !!!
      -بىراق بۇ ئىش راسلا قۇلۇمدىن كەلمەيدۇ .
       بۇ گەپنى ئاڭلاپ مۇستافا بۇ قاقباشنىڭ ئاسانلىقچە ئىرمەيدىغانلىقنى ھېس قىلدى -دە،بىر دەم ئويلىنىۋالغاندىن كىين ،-چۇقۇم سىزنى رازى قىلىمەن پۇلدىن ئەنسىرمەڭ .....ھە راس تۇلۇق ئوتتۇردا ئوقۇيدىغان چىرايلىق بىر ئوغلىڭىز باركەن ھە ؟؟؟ ئۇنىڭ ئوقۇشى ياخشىدۇ ؟ئاڭلىسام بۇ يىل ئىمتاھان بىرىدىكەن .
       بۇ گەپنى ئاڭلىغان ساقچى باشلىقى گەپنىڭ تىگىگە يىتىپ مۇستافا بىلەن قارشلاشماسلىق نىيتىگە كەلدى ،مۇستافا بۇ شەھەردىكى تەسىر كۈچى چوڭ ئىككى كاتتاباشنىڭ بىرى ئىدى ئۇنىڭ نۇرغۇن يەردە دىسكوخانا ،قاۋاقخانا ،ئەركىن تاماششاخانا...... لىرى بار ئىدى .ئۇنىڭ قىلىۋاتقان قانۇنلۇق ھەم قانۇنسىز سودىلىرىنىڭ ھەممىسى باشقىلارغا ئايان بولسىمۇ بىراق ئۇنىڭ بىلەن قارشىلىشىنىڭ ئاقىۋىتى ياخشى بولمىغاچقا نۇرغۇن ساقچىلار مۇستافاغا چېتىشلىق دىلولارنى پاش قىلىش نىيتىدىن ۋاز كەچكەن ئىدى .ئەگەر ساقچى باشلىقى ھۇشىنى تاپمىسا ئۇنۇڭغا ئائىلىسىدىكىلەرنى ئېلىپ بۇ يۇرتتىن كىتىشتىن باشقا ئامال يوق ئىدى .
       -  بۇ ئىشتا چۇقۇم سىزگە  ياردەم قىلىمەن مۇستافا. سىز ئاۋۋال قايتىپ تۇرۇڭ ئادەملىرىڭىزنى ساقچى ماشىنىسى بىلەن ئۆز جايىغا يەتكۈزۈپ قۇياي !
       -رەخمەت سىزگە ،ئاۋارە قىلىدىغان بولدۇم .ئەتە ئەتتىگەن بانكا ھېساۋاتىڭىزنى سۈرۈشتە قىلىپ بېقىڭ!-مۇستافا كۈلۈمسىرەپ قۇيۇپ بۇرھان بىلەن چىقىپ كەتتى .

-ۋۇ ئاناڭنى ،قاپلان يۈرۈكىدە ناشتىلىق يەۋالغان ئوخشىمامسەن قانجۇق ؟!
-نۇمۇسسىز ئوغرى، بىر قىز بالىنىڭ نەرسىسىنى ئوغرىلاپ تۇرۇپ يەنە خىجىل بولماي قول تەگكۈزۈۋاتامسەن . ۋىجدانىنى ئىت يەپ كەتكەن ئەبلەخ ......
    بۇ قىزنىڭ مەسۇم چىرايىدا ،كىيىكنىڭكىدەك شوخ كۆزلىرىدە بىرخىل جاسارەت ئەكىس ئىتىپ تۇراتتى .شۇ جاسارەتنىڭ تۈپەيلىمۇ ياكى ئىيول قۇياشنىڭ قىزدۇرىشىمۇ ،مەڭزى  كەچ كۈزدە قىزىرىپ پىشقان ئاناردەك قىپقىزىل بۇلۇپ كەتكەن ئىدى . قۇندۇزدەك قاپقارا قاشلىرى ئوتتۇرسىدا ،بادامدەك بۇرنىنىڭ ئۈستىدە بىلىنەر بىلىنمەس تەر تامچىسى پارقىرايتتى .
    بۇ گەپنى ئاڭلاپ مۇستافا «پىخ......»قىلىپ كۈلۈپ  سالدى -دە ، ئەتراپقا ئوبدان بىر قارىۋەتكەندىن كىيىن ،تاماكىسىنى يەرگە تاشلاپ خۇرۇم بەتىنكىسىدە ئىزىپ يەنجىۋەتتى .بۇنىڭدىن ئىلياس بىر نىمىنى سەزگەندەك قىلىپ قىزنى قۇيۋېتىپ ،ئارقىسىغا بۇرۇلۇپ ماڭغان مۇستافانىڭ كەينىدىن كۆندۈرۈلگەن پىستە ئىتتەك ئەگەشتى .
   -ئاكا،مەن ئۇ قىز بالىنى قاۋاقخانىغا تاپشۇرۇپ بىرەيمۇ يا........؟ ئۆزىغۇ خېلى چىرايلىق بىر نىمە ئىكەن .كىرىمىمۇ خېلى ياخشى بۇلىدىغاندەك قىلىدۇ.،ئاۋۋال سىز كۆڭلىڭىزنى ئېچىۋېلىڭ بولمىسا،يا........-ئەمدى ئۇ سۆزلىرىنى داۋاملاشتۇرالمىدى . ئۇنىڭ پۇتلىرى يەردىن كۈتۈرۈلۈپ ئىنچىكە بوينى مۇستافانىڭ كۈچلۈك قوللىرى ئارىسدا قالدى. مۇستافانىڭ ئۇزۇن چاچلىرى يۆگۈۋالغان كۆزلىرىدىن بىرخىل غەزەپ ئۇچقۇن يالقۇنجان تۇراتتى .
-ئېغىزىڭنى يۇم ،-ئۇنىڭ تەمكىن چىققان سۆزلىرىدىن غەزەپ نەپرىتى ئىپادىلىنىپ تۇراتتى .بۇرھان مۇستافانىڭ دۇلىسىغا ئاستا بىرنى مۇشلاپ مەنىلىك بىرنى قاراپ قۇيۇپ يولىنى داۋام ئەتتى .
   تۇيۇقسىز بۇرھانغا تېلفۇن كەلدى :
-بۇرھان ئاكا ،خەلق مەيدانىنىڭ ئۇدۇلىدىكى دىسكوخانىدا جىدەل چىقتى يەتتە سەككىز ھارام تاماق بۇ يەرنى قالايمىقان قىلىۋەتتى ...................
  - نان قېپىلار ،شۇنچىلىك ئىشنىمۇ قاملاشتۇرالماي ماڭا تېلفۇن قىلىشتىڭما ؟؟؟ھەممىگە مەن چاپامدىم- دەپلا تېلفۇننى قۇيۋەتتى .ئاندىن كەينىگە بۇرۇلۇپ ئارقىسىدىن كىلىۋاتقان يىگىرمە نەچچە ئادەمنى ئۇ قاۋاقخانىغا بۇيرۇپ قۇيۇپ ئۆزى مۇستافا بىلەن يولىنى داۋام ئەتتى .
     مۇستافا ،مەردان ،بۇرھان ئۈچەيلەن بىرگە تاماق يىيىش ئۈچۈم مۇستافانىڭ داچىسىغا قايتتى .بۇلار مۇستافانىڭ خىيالچان ھالىتىگە ،سۇرالغان سۇئاللارغا ھە ياكى ياق دەپلا پارۋاسىز جاۋاپ بىرىشلىرىگە ھەيران قالغان بولسىمۇ،بىراق مۇستافادىن بولغان ئەھۋالنى سۇراشنى توغرا تاپمىدى .
