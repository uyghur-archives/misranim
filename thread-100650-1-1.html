                                                       لۇتپۇللادەك ياشاپ باققۇم بار
               ئەركەك بولۇپ ياشاپ باققۇم بار،
               يايلاقلاردا چېپىپ باققۇم بار.
               دەريا بولۇپ ئۆركەشلەپ دائىم،
               يىراقلارغا ئېقىپ باققۇم بار.

              مەشئەل بولسام دەيمەن يالقۇنلۇق،
              دىللارغا ئوت يېقىپ باققۇم بار.
              ئادەم بولۇپ مەنمۇ يۈرەكلىك،
              لۇتپۇللادەك ياشاپ باققۇم بار!
                            لۇتپۇللا مۇتەللىپ تەۋەلللۇتىنىڭ 90 يىللىقىغا بېغىشلانغان!   (تۇرسۇن ئارسلان)

               بىزدىمۇ مۇنداق ياشاشنى ئۈمىد قىلىدىغان ئەقلى ئويغاق قىزلىرىمىزمۇ بار،بۇ شئېرىمنى پەزىلەتلىك قىزلىرىمىزغا،يەنە نەۋرۇز ئېيىنىڭ 8-كۈنىدىكى ئاتالمىش ئاياللار بايرىمىنى تەبرىكلەش ئۈچۈن،بالىلىرىنى ئاچ،ئەرلىرىنى يالغۇز قويۇشقا تەييارلانغان بىر قىسىم ئانىلىرىمىزغا بېغىشلايمەن!
                                         نۇزۇگۇمدەك ياشاپ باققۇم بار
        
                   زىمىن سۆيگەن ئىللىق قۇياشتەك،
                    نۇرلىرىمنى چېچىپ باققۇم بار.
                    مۇھەببەتنىڭ يامغۇرى بولۇپ،
                    بارچە ئەلگە يېغىپ باققۇم بار.
                     
                     ئەۋلاد ئۈچۈن بارى ئاتالغان،
                     ئۇلۇغ ئانا بولۇپ باققۇم بار.
                     پاكلىق ئۈچۈن جاندىن ھەم كىچىپ،
                     نۇزۇگۇمدەك ياشاپ باققۇم بار!
                                              ئاينۇر مۇھەممەدتۇرسۇن چىن قەلىمىدىن.

ئەسسالامۇئەلەيكۇم :

شىئېرىمغا تەققاس قىلىپ يازغان شىئېرىڭىز  پەيزى چىقىپتۇ ، ئۆزۈمنى ئېلىپ ئېيىتسام ئەزەلدىن شىئېر يېزىپ باقايچۇ دەپ شىئېر يازغان بەندە ئەمەسمەن ، بەزىدە ئىچىم ئۆرتىنىدۇ ، قەلەمنى ياكى كۇنۇپكا تاختىسىنى تۇتىمەن ، بىردەمدىن كىيىن قارىسام بىر ئىككى شىڭگىل شىئېر پەيدا بولۇپ قالىدۇ ، بەزىدە خۇشال بولىمەن ، ئورتاقلىشىدىغان ئادەم بولمىغىنىدا قەلەمنى تۇتىمەن ، يەنە بىر ئىككى شىڭگىل بالۇن قاپىيە مەيدانغا كىلىپ قالىدۇ (ئۆزۈمنى شىئېر يازىمەن دەۋېلىشتىن ئۇيىلىمەن).بەزىدە شىئېر يېزىشتىن قورقىمەن ، شىئېر بىر ئوت ، بۇ ئوت ئەڭ ئاخىرىدا شائىرنى كۆيدۈرۈپ ئاندىن تۈگەيدۇ ، رىئاللىقىمىزدا شائىرلارنى كۆيدۈرىدىغان بۇنداق ئوتلار كۆپىيىۋاتىدۇكى چىن شائىرلىرىمىز ئازلاپ قەدەھ ئەھلى كىشىلەرشائىر ئاتىلىپ قېلىۋاتىدۇ .
نەۋايىدەك تاھارەت بىلەن ئولتۇرۇپ شىئېر يازىدىغانلارنى شائىر دەيتۇق، شاراپ سورۇنىدىن دەلدەڭشىپ چىقىپ قەلەم تۇتىدىغانلارمۇ شائىر ئاتالغان كۈنلەردە ئاپىرىدە بولىۋاتقان ئاچچىق ئۈچەيدەك ئەشئارلارنىڭ نىمىلىكىدىن سىز ماڭا مان سىزگە قاراشماقتىمىز ، دىمەككى شىئېرلىرىمىز ۋىجدانىيلىقتىن ، خەلقچىلىكتىن ، رىئاللىقتىن ئايرىلىپ قېلىۋاتىدۇ ، بۇ بەلكىم شىئېرىيەتنىڭ ئۇۋالچىلىق كۈنلىرى دەپ ئاتالسا مۇۋاپىق بولىدىغاندۇ دەپمۇ قالىمەن . قىممىتىنى بىلىدىغان ئوقۇرمەنگە يولۇققان شىئېر بەخىتلىك شىئېردۇر ، شىئېرلىرىڭىزنىڭ داۋاملىق بەخىتلىك شىئېرلاردىن بولپ قېىشىنى ئۆمىد قىلىمەن .
