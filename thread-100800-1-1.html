ئەسسالامۇ ئەلەيكۇم قېرىنداشلار ، مەن گۇاڭجۇدىكى ئالىدەرىجىلىك ماشىنىلار رىمونت مەركىزىدە ئىنجىنىرلىق ۋەزىپىسىنى ئۆتەپ تۇرغۇچى ، ئابلىمىت ئەخمەت بولىمەن ، تور ۋاستىسى بىلەن كەڭ قېرىنداشلار بىلەن تونۇشقانلىقىمدىن ئىنتايىن خۇشال مەن ، شۇنداقلا كەڭ ئاپتوموبىل ھەۋەسكارلىرى ۋە ئاپتوموبىل رىمونتىغا قىزىققۇچى ھەۋەسكارلارغا چىن دىلىمدىن ئوتلۇق سالام يوللايمەن.
   قېرىنداشلار، بۈگۈنكىدەك دەۋىر ئۇچقاندەك تەرەققى قىلىۋاتقان كۈندە ، بىزنىڭ ئاپتوموبىل مۇلازىمەت ساھەيىمىز كەسپى رىمونت ئىنجىنىرلىرى ۋە رىمونت خادىملىرىنىڭ زىيادە كەمچىل بولىشى ۋە ئىپتىدائىي ئۇستاز تۇتۇپ ئۈگۈنۈش شەكلىدە يېتىشىپ چىققان ئۇستىلار ۋە شاگىرتلار نىڭ بۈگۈنكى ئىلغار كومپىيوتېر تېخنىكىسى بىرلەشتۈرۈپ ئىشلەنگەن ئاپتوموبىللارنىڭ ئالدىدا داڭ قېتىپ نىمە قىلارىنى بىلمەي قېلىشتەك سەۋەبلىرىدىن تولىمۇ بىچارە، ۋە ئېچىنىشلىق ھالەتتە قالدى. كەڭ قېرىنداشلىرىمىزنىڭ ئۆزىمىزنىڭ رىموتخانىلىرىغا بولغان ئىشەنچىسى زور دەرىجىدە تۆۋەنلەپ ، ئىشەنچىسىز بولۇپ قېلىش گىرداۋىغا بېرىپ قالدى. بۇنداق بولىشىدىن سەۋەب، ئىلگىركى ئىپتىداھى ھالەتتىكى ئۇستا ۋە شاگىرتتىن تەركىپ تاپقان كونا رىمونتچىلىق قۇرۇلمىسى ھازىرقى ئاپتوموبىللارنىڭ ئۇچقاندەك تەرەققى قىلىۋاتقان ئىلغار تېخنىكىسىغا يېتىشەلمىگەنلىكىدىن بولىۋاتىدۇ، چۈنكى بۇنداق قارا ھالەتتە رىمونت كەسپىنى ئۈگۈنۈش يولىدا ھېچقانداق ئۆزلىگىدىن ئۈگۈنۈش ۋە بىلىم-تېخنىكا دائىرىسىنى كېڭەيتىش ئاساسى تىكلەنمىگەن بولىدۇ ، شۇڭا يېڭى تېخنىكىغا جەڭ ئېلان قىلىشقا ھېچقانداق ئىقتىدارى بولمايدۇ. ھەتتا ھەر جەھەتتىن پىشقان كونا رىمونت ئۇستىلىرىمۇ بۇ خىل ھالەتتىن مۇستەسنا ئەمەس. ئەلۋەتتە بۈگۈنكى كۈندىكى رىمونتخانىلىرىمىزدا ئاددى بولغان بەزى مەشخۇلاتلار ئېلىپ بېرىلىدۇ، پىشقان رىمونتچى ئۇستىلىرىمىز يېڭى ئىلغار تېخنىكىلار ئىشلىتىلگەن ئاپتوموبىل رىمونتىدا تەلەي سىناش ئۇسۇلى ئارقىلىق ئۇ يەر- بۇ يەرنى كوچىلاپ باقىدۇ، كاشىلا ئاددى يەردە بولۇپ كاشىلىنى بايقاپ قالسا قالدى، ئەگەر بايقىيالمىسا خېرىدارنىڭ يېرىم كۈن ۋاقتىنى ئېلىپ بولۇپ، خېرىدارنى باشقا رىمونتخانىغا بېرىشقا تەۋسىيە قىلىدۇ ياكى خېرىدار چۈشەنمەيدىغان ئابىستىراكىت ئۇقۇم ۋە كەسپى تىللار بىلەن خېرىدارنى قايمۇقتۇرۇپ قويۇپ يولغا سالىدۇ. مانا يۇقارقىدەك ئەجەللىك نۇقتىلار بىزنىڭ رىمونت خانىرىلىمىزنىڭ ئىشەنچىسىز بولۇپ قېلىشىدىكى ۋە قېرىنداشلىرىمىزنىڭ كەينى چىقامايدىغان رىمونت دېڭىزىغا غەرىق بولۇپ ، بىزار بولىشىدىكى سەۋەبلەردۇ.  مۇشۇ ئامىللار تۈپەيلىدىن ئاز كۈنگە قالماي بىزنىڭ ماشىنا مۇلازىمەت كەسپىمىز جەمىئىيەت سەھنىسىدىن سۈپۈرۈپ چۈشۈرۋېتىشى مۈمكىن . شۇڭا بىزنىڭ ماشىنا مۇلازىمەت كەسپىمىز بىر تۈركۈم كەسپى خادىملارغا مۇقتاج بولىۋاتىدۇ، مەنمۇ يۇرتىمىزنىڭ ھەرقايسى ماشىنا مۇنازىرە تور بېكەتلىرىدە مۇنازىرە ئېلىپ بېرىش ۋە مەسلىھەت بېرىش جەريانىدا بۇ جەھەتتىكى كەمچىلىكىمىزنى چوڭقۇر ھېس قىلىپ يەتتىم . نۇرغۇنلىغان قېرىنداشلىرىمىزنىڭ بۇ كەسىپنى كەسپى ئۇسۇلدا ئۈگۈنۈشتىكى ئوتتەك قىزغىنلىغى مېنى تولىمۇ تەسىرلەندۈردى. ئەپسۇسلىنارلىقى ، بىزدىكى كەسپى بىلىملەرگە بولۇپمۇ ماشىنا مۇلازىمىتىگە ئائىت ئۇيغۇر تىلىدىكى ماتىرياللارنىڭ يوق دىيەرلىك بولىشى ياكى ماتىر ياللارنىڭ تېخنىكا سالمىغى ئاز ياكى ۋاقتى ئۆتكەن كونا ماتېرياللار بولۇپ ھازىرقى ئاپتوموبىل تېخنىكىسى بىلەن بىردەك بولماسلىقى تۈپەيلى قېرىنداشلىرىمىز بۇ ئارزۇسىغا يېتەلمەي ۋاتىدۇ.ھەتتا مەلۇم بىر مەكتەپنىڭ مۇدىرىدىن ئىگەللىشىمچە « ماشىنا رىمونت كەسپى ئوقۇغۇچىلارنى ئۇستاملارغا بىۋاستە شاگىرقا بېرىش ئۇسۇلى بىلەن قارا ھالەتتە ئۈگۈتىلىدىكەن»بۇنداق بولسا ھەرقانچە يوغان دېپلوم ئالغان بىلەنمۇ يەنىلا قارا رىمونتچى بولۇپ قالىدۇدە....... بۇ كىشىنى تولىمۇ ئېچىندۇرىدۇ ئەلۋەتتە.
يۇقارقى سەۋەبلەر تۈپەيلىدىن مەن گۇۋاڭجۇدىكى ئەڭ داڭلىق كەسپى رىمونتچىلىق مەكتەپلىرىنىڭ ئوقۇتۇش ماتېرياللىرىنى تەرجىمە قىلىپ ۋە مەزمۇنلىرىنى تېخىمۇ بىيىتىپ  تەھرىرلەپ مۇشۇ ماتېرياللار ئاساسىدا ئۆزەمنىڭ ئۇزۇن يىللىق ماشىنا مۇلازىمەت كەسپىدىكى تەجىربە يەكۈنلىرىمنى بىرلەشتۈرۈپ بىرتۈركۈم ماشىنا مۇلازىمەت كەسپىدىكى كەسپى خادىملارنى يېتىشتۈرۈپ چىقىپ، مىللىتىمىزنىڭ ماشىنا مۇلازىمەت كەسپىدىكى يۇقارقىدەك پاسسىپ ھالەتنى ئۆزگەرتىش ئارزۇسىدا گۇاڭجۇدىن ئۈرۈمچىگە قايتىپ كەسپى تەربىيلەش كۇرسى ئېچىشنى قارار قىلدىم.
قېرىنداشلىرىمىز نەزىرىدە رىمونتچىلىق بىر تەس كەسىپ ئۈگۈنۈش ئۈچۈنمۇ چوقۇم ئۈچ تۆت يىل ۋاقىت كېتىدۇ دەپ قارايدۇ، ئەمىلىيەتتە بۇ كەسىپنى ئۈگۈنۈش ئۈچۈن پەقەتلا بىر يىل ۋاقىت بولسا كۇپايە قىلىدۇ، دۆلەت ئىگىلىكىدىكى مەكتەپلەرنىڭ دەرىس سائەت سانىنى توشقۇزۇشتەك مەجبۇرىيتى بولغانلىقى ئۈچۈن باشقا قوشۇمچە دەرىسلىكلەر بىلەن قوشۇلۇپ نەچچە يىلغا سوزۇلۇپ كېتىدۇ، بۇ سەۋەبلىك ئوقۇغۇچىلار ئۈگەنگەن بىلىملىرىنى ئۆز ۋاقتىدا ئەمىلىيەتتىن ئۆتكۈزەلمىگەنلىكى ئۈچۈن ۋاقىتنىڭ ئۇزىرىشىغا ئەگىشىپ ئالدىدا ئۈگەنگەنلىرىنى ئۇنتۇپ مەكتەپ پۈتتۈرگەندە ھېچنىمىنى بىلمەيدىغان ھالەتكە بېرىپ قالىدۇ. ھازىر ئىچكىرنىڭ ھەرقايسى ئۆلكىلىرىدىكى داڭلىق كەسپى رىمونتچىلىك مەكتەپلىرىدە رىمونتچىلىق كەسپىنىڭ بىر يىلدىن ئېشىپ كېتىدىغىنى يوق پەقەت تولۇق كۇرۇس ۋە ئالى تېخنىكوم كەسىپلىرى دۆلەتلىك ئىمتىھانغا قاتنىشىدىغان بولغاچقىلا ۋاقتى 3يىلغا ئۇزارتىلغان. ئەگەر نۇقۇل ھالدا كەسىپ ئۈگەنمەكچى بولساق بۇنداق ئۇزۇن ۋاقىت ۋە ئىقدىسادنى ئىسراپ قىلىشنىڭ ھېچقانداق زۆرۆرىيتى يوق ئەلۋەتتە. شۇڭا مەن شۇنداق ۋەدە بېرەلەيمەنكى، ئوقۇغۇچىلار بىر يىللىق ئوقۇشنى تاماملاپ بولغاندىن كېيىن ئۆز ئالدىغا كاشىلىغا دىئاگىنوز قويۇش ۋە رىمونت قىلىش ئىقتىدارىنى ھازىرلىيالايدۇ
ئوقۇغۇچىلار تەربىيلىنىش جەريانىدا نەزىيە بىلىملىرىنى پۇختا ئىگەللىگەندىن باشقا ئۆزۈمنىڭ يىتەكچىلىكىدە بارلىق ئەمىلى مەشخۇلاتنى ئېلىپ بارىدۇ. تەربىيلەش ئۆلچىمى 4S دۇكىنى ئۆلچىمى بويىچە بولىدۇ. كەسىپ دەرىسىدىن باشقا ھېچقانداق قوشۇمچە دەرىسلىك ئۆتۈلمەيدۇ. ئوقۇتۇش ئۈنىمى ئوقۇغۇچىلارنى ئۆز ئالدىغا چوڭ كىچىك تىپتىكى بارلىق رىمونتلارنى مۇستەققىل تاماملىيالايدىغان بولۇش ۋە دىئاگىنوز قويۇش ئىقتىدارىنى يېتىلدۈرۈش بىلەن بىرگە ھەرخىل شىفىر يەشكۈ ۋە كومپىيوتېرلىق سايمانلارنى ئىشلىتىشنى بىلىش ۋە ئالى دەرىجىلىك رىمونت خادىمى ساپاسىنى يېتىلدۈرۈش  نىشانىغا يېتىش ئۆلچەم قىلىنىدۇ.
ئوقۇغۇچى قوبۇل قىلىش ۋاقتى 3-ئاينىڭ 20-كۈنىگىچە داۋاملىشىدۇ، ئوقۇغۇچى قۇبۇل قىلىش توختىتىلغاندىن كېيىن بىردەك ئوقۇغۇچى قۇبۇل قىلىنمايدۇ. مەزكۈر كۇرۇسمۇ پەقەت بىر قېتىم ئورۇنلاشتۇرىلىدۇ.
ئوقۇش مۇددىتى 1يىل بولۇپ دۈشەنبىدىن جۈمەگىچە پۈتۈن كۈنلۈك دەرىس ئۆتۈلىدۇ .دەم ئېلىش ، بايرام ۋە تەتىللەر دۆلەتلىك مەكتەپلەر بىلەن بىردەك بولۇپ ،  دەرىس 5-ئاينىڭ 6-كۈنى باشلىنىدۇ.
بىرىنچى قېتىملىق ئوقۇشتا ئۈگۈنەلمىگەنلەر ئىككىنچى قېتىم قايتا ھەقسىز ئوقۇتىلىدۇ.
ياتاق ھەققى 1000يۈەن، يوتقان-كۆپە،كۈندىلىك تۇرمۇش بۇيۇملىرى ۋە خىزمەت كىيمى ھەقسىز تەمىنلىنىدۇ.
تاماق : تەربىيلەش ئورنىمىزنىڭ خاس ئاشخانىسى بار
ئەسكەرتىش: ئوقۇغۇچى قوبۇل قىلىش  سانى 30 نەپەر
تىزىملاتقان ئالدىنقى 10 ئوقۇغۇچىغا باھادا 5پىرسەنت ئېتىبار قىلىنىدۇ  تورداش قېرىنداشلارنىڭمۇ ياردەمدە بولۇپ تېخىمۇ كۆپلىگەن ئوقۇشنى خالايدىغان قېرىنداشلىرىمىزنى بۇ چەكلىك پۇرسەتتىن خەۋەردار قىلىشىنى ، مىللى ئىگىلىك تەرەقىياتىمىز ئۈچۈن بىر كىشلىك تۆھپىسىنى قوشۇشىنى ئۈمىد قىلىمىز.
تەھرىرلەنگەن ئوقۇتۇش ماتىرياللىرىنىڭ كېلىش مەنبەسى
ئوقۇتۇش ماتىرياللىرىمىزنىڭ كېلىش مەنبەسى گۇاڭجۇدىكى داڭلىق كەسپى رىمونتچىلىق مەكتەپلىرىنىڭ ئىچكى ئوقۇتۇش ماتېرياللىرى بولۇپ، باشقا يېڭى تېخنىكىلارغا ئائىت مەزمۇنلار قوشۇلغان.
1) ئاپتوموبىل ماتور قۇرۇلمىسى ۋە رىنمونتى
2) ئاپتوموبىل تەگلىك قۇرۇلمىسى ۋە رىمونتى   
3)          ئاپتوموبىل توك يولىدىن ئاساس      
4) ئاپتوموبىل ئېلىكتىر تېخنىكىسى
5) ئاپتوموبىل ئىلىكتىرىك كونتىرول تېخنىكىسى                     
6) ئاپتوماتىك سۈرئەت ئۆزگەرتىش كورۇپكىسى مىخانىزىمى ۋە زىمونتى                  
يولغا قويۇلغان كەسىپ تۈرى:
1) مىخانىك ۋە ئىلىكتىرون بىر گەۋدە قىلىنغان ئاپتوموبىل رىمونت ئۇستىسى كەسپى
