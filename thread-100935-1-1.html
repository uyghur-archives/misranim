نورۇز بايرىمى بارلىق تۈركى مىللەتلەرنىڭ جۈملىدىن ئۇيغۇرلارنىڭ كونا كالېندارى بويىچە 3-ئاينىڭ 21-كۇنى ئۆتكۇزىلىدىغان مىللى ئەنئەنىۋى يېڭى يىل بايرىمى. قومۇلدا بۇ بايرام "سەھراخەتمە" دەپ ئاتىلىپ ھەر يىلى "نورۇز بايرىمى" ئۆتكۇزۇلىدىغان كۇنگە توغۇرلىنىپ دىنىي تۈس ئالغان ھالدا ئۆتكۇزىلىدۇ. ئۇيغۇر خەلقى ئارىسىدا، يېقىنقى دەۋرلەرگىچە "نورۇز بايرىمى"، "نورۇز مەشرەپلىرى" ساقلانغان. كېيىن شىڭ شىسەي ھاكىمىيىتى غەرەزلىك ھالدا ئەمەلدىن قالدۇرغان، نۇرغۇن تارىخى پاكىتلار ئۇيغۇر "ئون ئىككى مۇقامى" تەركىبىدە "نورۇز مۇقامى" دەپ ئاتالغان بىر مۇقامنىڭ بارلىغىنى كۆرسىتىدۇ. ئەنە شۇ "نورۇز مۇقامى" تۈركى خەلقلەردە ھەر خىل شەكىلدە گەۋدىلەنگەن ۋە ئۈستۇنلىكنى ئالغان ياكى تەسىر كۆرسەتكەن ، رايۇنلارغا كەڭ تارقالغان،ھەمدە ئۇنىڭ "نورۇز بوزرۇك"، "نوئۇرز ھۆسەينى"،"نوئۇرز بۇ سېلىڭ"،"نورۇز راست"،"نورۇز سۇلتانى" قاتارلىق كۆپلىگەن ۋارىيانتلىرى بولغان. بىراق، بۇ قەدىمى مۇزىكا گۇرۇپىلىرىنىڭ روشەن نەمۇنىلىرى ساقلانمىغان، گەرچە قومۇل مۇقاملىرىدىمۇ "نوئۇرز مۇقامى" دەپ ئاتالغان تىپىك مۇقام گۇرۇپىسى ساقلانمىغان بولسىمۇ، ئەمما خەلق ئىچىدە نورۇز بايرىمىدا ئېيتىلىدىغان قوشاقنىڭ ئاز بىر قىسىم ئۈلگىلىرى ساقلانغان. ئەپسۇسكى،يوقۇرىدا ئېيتىلغان نورۇز قوشىقىنىڭ ئۆزىگە خاس ئاھاڭى يوقىلىپ كەتكەن. شۇنداق بولسىمۇ بۇ نورۇز قوشىقىنى ئەينى دەۋىرلەردىكى "نورۇز مۇقامى" نىڭ بىر بۆلىكى بولۇشى مۇمكىن دېگەن پەرەزدىن يەنىلا ۋاز كېچەلمەيمىز. بۇ نۇقتا يېقىن كەلگۇسىدە ئەلۋەتتە ئىسپاتلانغۇسى. 80-يىللارغا كەلگەندە ئۇيغۇرلار ئۆزىنىڭ بۇ قەدىمكى بايرىمىنى قايتىدىن ھەر يىلى تەبرىكلەپ كەلمەكتە!

2013- يىلى 3- ئاينىڭ 9 -كۈنى شىنجاڭ سەنئەت ئىنىستۇتى زالىدا مېھرىگىياھ نورۇز مەشرىپى ئۆتكۈزۈلدى، مەشرەپكە ئەل سۆيگەن سەنئەتكارلىرىمىز قاتناشتى،شۇنداقلا يېڭى ئىجادىيەتلىرىنى خەلقىمىزگە سۇندى.



مەركەزنىڭ ئاپتونۇم رايونىمىزغا قاراتقان ئەۋزەل سىياسەتلىرىدىن ئۈنۈملۈك پايدىلىنىپ كېلىۋاتقان خېجىڭ ناھىيىسى ئۆز ئالدىغا ئىگىلىك تىكلىمەكچى بولغان ھەر مىللەت قېرىنداشلارغا ئالاھىدە كۆڭۈل بۆلۈپ كېلىۋاتقان بولۇپ، خېجىڭ ناھىيىسىدە تۇغۇلۇپ چوڭ بولغان ئۇيغۇر كارخانىچى مېھرىگۈل خانىم، ناھىيەنىڭ پۇخرالارغا قاراتقان ئەۋزەل سىياسىتىدىن بەھرىمان بولىۋاتقان ، ئەڭ مۇھىمى ئۆز دىيارىنىڭ رايون ئەۋزەللىكىدىن ، ئەجدادلىرىمىز نەچچە ئەسىرلەر بويى ئىشلىتىپ پايدىلىنىپ كېلىۋاتقان يەرلىك مەھسۇلاتلىرىمىزنىڭ ئالاھىدىلىكىدىن پايدىلنىپ تۇرۇپ ئىگىلىك تىكلىگەن ئىقتىدارلىق ئايال بولۇپ ئۇنىڭ باشچىلىقىدا 2008- يىلى 6- ئايدا قۇرۇلغان شىنجاڭ مېھرىگىياھ گىرىم بۇيۇملىرى چەكلىك شىركىتى خېجىڭ ناھىيىسىدىكى ئۇيغۇر قېرىنداشلىرىمىز پەخىرلنىدىغان، باشقا قېرىنداش مىللەتلەر ئەتىۋارلاپ ئىشلىتىدىغان، ئۈنۈمى يۇقىرى بىر يۈرۈش تەبىئىي مەھسۇلاتلىرى بىلەن ئاپتونۇم رايونىمىزنىڭ ئىچى ھەم سىرتىغا تونۇلىۋاتقان ھەتتا تەسىرى ئوتتۇرا ئاسىيادىكى قوشنا ئەللەرگىچە كىڭىيىۋاتقان شىركەتلەرنىڭ بىرى بولۇپ قالدى. تەبئىيلىك پۈتكۈل ئىنسانىيەتنىڭ ئاساسىي ئىھتىياجىغا ئايلىنىۋاتقان بۈگۈنكى كۈندە كىشىلەرنىڭ گىرىم بۇيۇملىرىغا نىسبەتەن تەلىۋى بارغانسىرى زىيانسىز، تەبىئىي، ھېچقانداق خىمىيىۋىي بۇيۇملار ئارلاشتۇرۇلمىغان ساھاگە قاراپ يۈرۈش قىلىۋاتىدۇ. بۇنىڭ نەتىجىسىدە زىيانسىز ، تەبىئىي خام ئەشيالار ئاساسىدا ئىشلەنگەن گىرىم بۇيۇملىرى جەمئىيىتىمىز ئۈچۈن ناھايىتى زۆرۈر  تەخىرسىز ئىھتىياج ھەم تەلەپ بولۇپ قالغان ئىدى.


مۇشۇ خىل تەخىرسىز ۋەزىيەتنى كۆزدە تۇتقان ‹‹شىنجاڭ مېھرىگىياھ گىرىم بۇيۇملىرى چەكلىك شىركىتى›› قۇرۇلغاندىن ھازىرغىچە ئەجدادلىرىمىزنىڭ گىرىم بۇيۇملىرى ئىشلىتىش تەجىربلىرىنى يەكۈنلىگەن ئاساستا ھازىرغىچە 7 تۈرلۈك مەھسۇلاتنى ئىشلەپ چىقىرىپ بازارغا سالدى. مېھرىگىياھ مەھسۇلاتلىرى ئەجدادلىرىمىز ئۇزاق زامانلاردىن بۇيان ئەتىۋارلاپ ئىشلىتىپ كەلگەن شاپتۇل مىغىزى، ئۈستىقۇددۇس ،زەنجىۋىل قاتارلىق دورىلىق خام ئەشيالارنى خام ماتىريال قىلىنىپ  ئىشلىنىپ زامانىمىز كىشىلىرنىڭ تۇرمۇش شارائىتى، تەبىئىي مۇھىتى،يۈرۈش تۇرۇشى قاتارلىق ئەھۋاللىرى بىلەن زىچ بىرلەشتۈرۈلگەن ھالدا ئىشلەپچىقىرىلغان بولۇپ شۇ ئاساستا بازارغا كىرگەنىدى. شۇڭا بازارغا كىرگەندىن ھازىرغىچە ئىستىمالچىلار ياقتۇرۇپ ئىستىمال قىلىدىغان بىر-بىرىگە سوۋغا قىلىدىغان ئىشەنچىلىك ، ئەتىۋارلىق مەھسۇلات بولۇپ قالدى.                  

  شىنجاڭ مېھرىگىياھ گىرىم بۇيۇملىرى چەكلىك شىركىتى قۇرۇلغاندىن ھازىرغىچە توختىماي تەتقىقات ئىلىپ بېرىش، ئىشلەپچىقىرىش ، مەھسۇلات تۈرىنى كۆپەيتىش ۋە سۈپىتىنى ياخشلاش جەھەتلەردە كۆپ كۈچ سەرىپ قىلىشتىن سىرت مەھسۇلاتنى سىتىپ چىقىرىش بىلەن بىرگە بازار ئېچىش قاتارلىق نۇقتىلاردا كۆپ باش قاتۇرۇپ ئەقىللىق ئاتا بوۋىلىرىمىزدىن قالغان ‹ ھەمكارلاشساڭ تاغ ئاشارسەن ئەقىدىسى ›بويىچە ‹‹ ئۈرۈمچى جەۋلانى مىللىي مەدەنىيەت تەرەققىيات چەكلىك شىركىتى›› بىلەن ھەمكارلىق ئورنۇتۇپ ئۇيغۇر كارخانچىلىرىنىڭ ھەمكارلىق سەھنىسىدە نامايەندە تىكلىدى.


ئۈرۈمچى جەۋلانى مىللىي مەدەنىيەت تەرەققىيات چەكلىك شىركىتىمۇ ئاپتونۇم رايونىمىز تەۋەسىدە مىللىي مەدەنىيەت جۇلالىرىنى مەملىكەت ئىچى  سىرتىغا نامايەندە قىلىپ بىرىۋاتقان سەركەردە شىركەتلەرنىڭ بىرى بولۇش سۈپىتى بىلەن مەھسۇلات سۈپىتى، قىممىتى،ئىشەنچىلىك دەرىجىسىگە ئالاھىدە ئەھمىيەت بىرىپ ئۆزىمىزنىڭ يەرلىك مەھسۇلاتلىرمىزنىڭ قىممىتىنى نامايەندە قىلىپ خەلقىمىزنىڭ ئىشەنچىسىگە ئىرىشكەن شىركەتلەرنىڭ بىرى، شۇنداق بولغانلىقى ئۈچۈن ئۈرۈمچى جەۋلانى مىللىي مەدەنىيەت تەرەققىيات چەكلىك شىركىتى بىلەن شىنجاڭ مېھرىگىياھ گىرىم بۇيۇملىرى چەكلىك شىركىتى ـ مېھرىگىياھ مەھسۇلاتلىرىنى سىتىپ چىقىرىش، بازار ئېچىش جەھەتتە ئۆزگىچە ھەمكارلىق مۇناسىۋىتى ئورناتتى.  



  مانا بۈگۈنكى كۈندە مېھرىگىياھ مەھسۇلاتلىرى مېھرىگىياھ شىركىتىنىڭ خېجىڭ ناھىيىسىدىكى كۆلەملەشكەن ، 50 تىن ئارتۇق ياشلار ئىشلەيدىغان ، زامانىۋىي ئۈسكۈنىلەر بىلەن قوراشتۇرۇلغان زاۋۇتقا ئايلاندى.  شىنجاڭ مېھرىگىياھ  گىرىم بۇيۇملىرى چەكلىك شىركىتى ئۆز خاسلىقى ۋە ئەۋزەللىكى بىلەن داڭلىق ماركا ، ئىسىل ئوبراز تىكلەش جەھەتلەردە تىخىمۇ ئىلگىرلىگەن ھالدا تىرىشچانلىق كۆرسىتىۋاتقان بۈگۈنكىدەك  ئالتۇن دەۋرانلاردا بىز ئانا يۇرت مۇھەببىتى ، ئەجداد ئەنئەنىسىنى ئۆزلىرىگە كۈچلۈك ئەقىدە قىلىپ ئىلگىرلەۋاتقان بۇ ئىككى شىركەتكە ئۇتۇق تىلەش بىلەن بىرگە بىز ئىككى شىركەتنى قۇتلۇقلاپ خېجىڭ ناھىيىسىدىن كەلگەن ئەزىز مېھمانلار بىلەن مەركىزىي شەھەر ئۈرۈمچىدە جەم بولۇپ مېھرىگىياھ مەشرەپ داستىخىنىغا داخىل بولدۇق
