يوللىغان ۋاقتى 2013-3-10 11:09:25 PM |ھەممە قەۋەتنى كۆرۈش |
    كىشلىك دۇنيا شۇقەدەر ئەگرى توقاي ۋە جاپالىق بولىدىكەن، بەزىلەر سۇلتان بولسا يەنە بەزىلەر گاداي بولىدىكەن،خاياتنىڭ ماھىيتىنى چۈشىنىش ئەسلا مۇمكىن بولمايدىكەن،،بەزىلەر تەقدىرنىڭ ئېغىر سىناقلىرىنى باشتىن كەچۈرسە بەزىلەر ھاياتتا مەغرۇر كېرىلىپ ئۆزىنىڭ پۇل ۋە مال دۇنياسى بىلەن خۇشال ھەم بەخىتلىك ياشايدىكەن بىراق كىممۇ ئويلاپ باققان بولسۇن بۇ تەڭسىز دۇنيا ھېچكىمگە ھەقىقى بەخىتنى ھەقىقى بىر گۈزەل ھايات خاتىمىسىنى قالدۇرۇپ باققان ئەمەس مانا بۇ ھاياتتىكى بىر گەۋدە.     ھاياتنىڭ ھەقىقى مەنىسىنى ھەقىقى بەخىتنىڭ زادى نىمە ئىكەنلىكىنى زادى كىم بىلىدىغاندۇ ھە؟ ھازىرقى جەمئيىتىمىز بۇيىچە ئېيىتقاندا  نۇرغۇنلىغان كىشىلەر ھەقىقى بەخىتنى پۇل ۋە مال دۇنيا دەپ ھېساپلايدىغان بولدى شۇنى بىلىش كىرەككى ھەقىقى بەخىت ئۆزىنىڭ ھالال تەر ئەمگىكى ۋە ئۆزىنىڭ ھەقىقى تىرىشچانلىقى بەدىلىگە كەلگەن بەخىتتۇر.      شۇنداق دەپ پەرەز قىلىشقىمۇ بولىدۇ،،سىز بۇ دۇنيادا شۇنچە بىخىللىق ۋە شۇنچە ئىچى تارلىق بىلەن ياشاپ ئۆتكەن بولسىڭىز ئۇ ھالدا سىز ئەڭ بىچارە ۋە ئەڭ چاكىنا ئادەمگە ئايلىنىپ قالغان بولىسىز. چۈنكى بۇنداق بىچارىلەرچە ئۆتكەن ئادەمىنىڭ مەڭگۈ قەدىر-قىممىتى ۋە يۈز ئابرۇيى بولمايدۇ چۈنكى ئۇنىڭ بۇ پۇل مال-دۇنيالىرى ئۇنى بايدەك كۆرسەتكىنى بىلەن ئۇنىڭ ھەقىقى ئەخلاق -پەزىلىيتى .ئىستىلى ناچار، بىر تىيىن ئۈستىدىن تىك تۇرىدىغان ،خەلىققە ھېچقانداق پايدىسى تەگمەيدىغان ئىنسانلار ئۈچۈن ئېيىتقاندا بۇ شۇ كىشلەرنىڭ نەقەدەر ئاجىز ۋە زەيىپ ئىكەنلىكىنى چۈشەندۈرۈپ بىرەلەيدۇ؟      سىزدىن شۇنداق دەپ سۇرغۇم كىلۋاتىدۇ.سىز ھەردائىم كوچا-كويلاردىكى تار-بۇلۇڭ ۋە تار كوچىلاردا غەيۋەت قىلىشىپ ئولتۇرغان شۇ كىشلەرنى كۆرگىنىڭىزدە شۇ كىشلەرنىڭ نەقەدەر بىچارە ئىكەنلىكىنى ئويلاپ باقتىڭىزمۇ؟ چۈنكى كۈن بويى سىرىتتا غەيۋەت قىلىپ ئولتۇرغان چاغلىرىڭىز سىزنىڭ ۋاقىتنى ئارتۇقچە زايە قىلغانلىقىڭىزدىن دىرەك  بىرىدۇ،،سىز شۇ ۋاقىتلاردا نۇرغۇنلىغان ئىشلارنى قىلىپ بولىشىڭىز مۇمكىن.      سىزدىن شۇنداق دەپ سورغۇم كىلىۋاتىدۇ،سىز باينى كۆرۈپ ئۇنىڭغا خۇشامەت قىلىپ ئاغزى -ئاغزىڭىزغا تەگمەي خۇشامەت قىلىسىز.كەينىدىن ئۇنىڭ بايلىقىغا ھەسەت قىلىپ يامان قارغىش بىلەن ئۇنى ھاقارەتلەيسىز ، بىراق بىرەر كەمبەغەلنى كۆرۈپ <<سەن گاداينىڭ نىمەڭ باتتى>> دىگەندەك قىلىپ ئۇنىڭغا ئازار بىرىسىز.باينى تىللايسىز كەمبەغەلنى ياراتمايسىز سىزدىن سۇراي بۇ سىزدىكى غورۇرمۇ؟ سىزدىكى ھەقىقى مىھرىبانلىقمۇ؟      سىزدىن شۇنداق دەپ سورغۇم كىلىۋاتىدۇ،،سىز ھەردائىم باشقىلارنىڭ كىچىككىنە ئىشلىرىدىن قۇسۇر تېپىشقا تەييار تۇرىسىز قېنى سىزدىن سوراي...سىز ئۆز قېرىندىشىڭىزنىڭ شۇنچە كۆپ ياخشى تەرەپلىرىنى كۆرىسىز،،بىراق ئۇنىڭ شۇنچە كۆپ ياخشىلىقلىرى كۆزىڭىزگە كۆرۈنمەيدۇ ئەكىسچە ئۇنىڭ كىچىككىنە ئەيىپ-نوقسانلىرى كۆزىڭىزگە كۆرۈنۈپ كىتەمدۇ؟ سىزدگە شۇنداق دىگۈم كىلىۋاتىدۇ ،،ئەجەبا سىز بۇ چاغقا قەدەر خاتالىشىپ باقمىغانمۇ؟     سىزدىن شۇنداق سۇرغۇم كىلۋاتىدۇ،،سىز ھەمىشە ئۆزىڭىزنىڭ پەرزەنىتلىرىڭىزنى باشقا بالىلار بىلەن سېلىشتۇرىسىز .بىراق شۇنى ئويلاپ بىراق سىزنىڭ شۇ بالىلىرىڭىزمۇ ھەرگىز ئەسكى ئەمەس،سىز شۇنداق چاغلاردا بالىلىرىڭىزنىڭ قانچىلىك ئوڭايسىز ئەھۋالدا قالدىغانلىقىنى ۋە قانچىلىك كۆڭلى پەرىشان بولدىغانلىقىنى تەسەۋۇر قىلىپ باقتىڭىزمۇ؟ يەنە سىزگە شۇنداق دىگۈم كىلىۋاتىدۇ،،،ئەجەبا سىزنىڭ بالىلىق دەۋرىڭىز دە سىز پۈتۈنلەي پاك ۋە گەپ ئاڭلايدىغان بىرى بولغۇيمىتىڭىز؟        سىزدىن شۇنداق سورغۇم كىلىۋاتىدۇ،،،سىز گەپ قىلىشقا توغرا كەلسە ھېچ قالماي تەرىپلەيسىز بىراق سىز ئۆزىڭىزنىڭ ئۆيىڭىزگە قاراپ باقتىڭىزمۇ؟ چەتئەلدە بۇنداقكەن ئۇنداقكەن دىگەنلىرىڭىزدە بۇ خىل ئىسلاھاتنى ئۆزىڭىزنىڭ ئائىلىڭىزدىن ئەتراپىڭىزدىن باشلاش كىرەكلىكىنى ئويلاپ باقتىڭىزمۇ؟ سىزگە شۇنداق دىگۈم كىلىۋاتىدۇ..چەتئەل ياۋۇرپا سىياسەتلىرىنى يېڭى كەشپىياتلىرىنى بىرمۇ -بىر تەرىپلىگەن چاغلىرىڭىزدا ئۆزىڭىز شۇنچە تەرىپلەيدىغان شۇ چەتئەل كىشلىرىدىكى ئاڭ ۋە سەۋىيە بىلەن بىز ئۇيغۇر كىشلىرىدىكى ئاڭ -سەۋيەنىڭ نەقەدەر پەرىقلقى ئىكەنلىكىنى ئويلاپ باقتىڭىزمۇ؟ ناۋادا بىزمۇ تىرىشىدىكەنمىز بىزمۇ ھەرگىز بۇ دەۋىرنىڭ كەينىدە قالمايمىز ھەم شۇ تەرىپىنى قىلىدىغان چەتئەلدىكى يېڭى كەشپىياتلاردىن بىز ئۆزلىرىمىزمۇ تولۇق پايدىلىنىپ يېڭى كەشپىياتلارنى بارلىققا كەلتۈرەلەيمىز ئەمەسمۇ؟          سىزدىن شۇنداق دەپ سورغۇم كىلىۋاتىدۇ،،سىز ھەمىشە ئۆزىڭىزنىڭ يىقىن ئادەملىرى شۇنداقلا كىملا بۇلسۇن بىرى بىلەن ياقا تارتىشىپ سوقۇشقانلىرىڭىزدا ئۆزىڭىزنىڭ قىلغانلىرى كۆزىڭىزگە كۆرۈنمەستىن شۇ كىشىلەرنىڭ قىلغانلىرىدىن رەنجىپ ئۇلارغا ئادەۋەت ساقلايسىز .سىزگە شۇنداق دىگۈم كىلىۋاتىدۇ،سىز ھەمىشە يىقىنلىرىڭىزدىن رەنجىپ ئادەۋەت ساقلىغانلىرىڭىزدا ئۆزىڭىزنىڭ سەۋىر-خاتالىقلىرى ئۈستىدە ئويلىنىپ باقتىڭىزمۇ سىزدىن سوراي ئىنساننىڭ ئەڭ ئالىي خىسلىتىنىڭ ئەپۇچانلىق ئىكەنلىكىنى ئۇنتۇپ قالدىڭىزمۇ؟ ئەجەبا سىزنىڭ ھەقىقى مىھرىڭىز ۋە ئىنسانىي قەدىر-قىممىتىڭىز قەيەرگە كەتتى؟         سىزدىن شۇنداق سورىغۇم كىلىۋاتىدۇ،،ھەرقېتىم بىراۋنى ئىشەك كالا دەپ چاقىرغاندا ئۆزىڭىزنىڭ شۇ ئىشەك ۋە شۇ كالىلاردىنمۇ ھاماقەت ئىكەنلىكىڭىزنى ئۇنتۇپ قالدىڭىزمۇ؟چۈنكى كالا ۋە ئىشەكلەر بىر قېتىم پېتىپ قالغان ئېرىقتىن ھۆركىيدۇ؟ ئىگىلىرىنىڭ قىلغان ياخشىلىقلىرىنى ئۇنۇتمايدۇ؟بىراق بىز ئىنسانلارچۇ؟بىرقىتىم يىقىلغان ئورۇنغا قايتا قەدەم باسىمىز ..بىراۋنىڭ قىلغان ياخشىلىقلىرىنى ئۇنتۇپ ئۇنىڭغا ياخشىلىق ئورنىدا يامانلىق قىلىدىغان كىشلەرمۇ ئاز ئەمەس،،،سىزگە شۇنداق دىگۈم كىلىۋاتىدۇ،،،ئەجەبا ھەقىقى ئەقىللىق دەپ سانىلدىغانلار ئىنسانلار بولسا بۇ ھالدا ئىنسانلاردىكى ئەقىل ۋە پاراسەت قەيەرلەرگە كەتتى؟        سىزدىن شۇنداق سورىغۇم كىلىۋاتىدۇ..سىز ھەردائىم ئۆزىڭىزنىڭ  شەخسى سىرلىرىڭىزنى دوستۇم دەپلا خالىغانلا ئادەمگە ئېيتىپ قېلىشىڭىز مۇمكىن بىراق كەينىدىنلا ئۇ ئادەمنىڭ شەخسى سىرلىرىڭىزنى باشقىلارغا دىگىنىنى بىلىپ قاتتىق غەزەپلىنىشىڭىز مۇمكىن بىراق سىزدىن سوراپ باقاي ئۆزىڭىزنىڭ ئىچىگە پاتمىغان گەپ باشقىلارنىڭ ئىچىگە پاتسۇنمۇ؟ سىز ئۆزىڭىز قوبۇل قىلىشقا ئاجىز كەلگەن شۇ رىئاللىقنى باشقىلار قانداقمۇ قوبۇل قىلالىسۇن؟        سىزدىن شۇنداق سورغۇم كىلىۋاتىدۇ،،بىزنىڭ بەزى بىرقىسىم كىشىلىرىمىز قىزلارنى مەكتەپتە ئوقۇتۇش ئەخمەقلىق دەپ قارايدۇ،،بۇ  تۈپتىن خاتا قاراش چۈنكى قىزلار بىر مىللەتنىڭ تەربىيلىگۈچىسى شۇنداقلا بىر مىللەتنىڭ ئانىسى،قىزلارنىڭ بىلىمى بولمىسا قانداقمۇ ئۆز بالىلىرىنى ياخشى ۋە ئەخلاقلىق ھەقىقى 21-ئەسىر ئۆسمۈرلىرىدىن قىلىپ تەربىيلەپ چىققىلى بولسۇن؟سىزگە شۇنداق دىگۈم كىلىۋاتىدۇ قىزبالا بولۇش گۇناھ ئەمەس چۈنكى بىز قىزلار بۇلغاچقىلا بۇ دۇنيا شۇنچە گۈزەل ھەم پاكىز ئەمەسمۇ....      سىزدىن شۇنداق سورىغۇم كىلىۋاتىدۇ،،سىز بەزى چاغلاردا بەزى كۆڭۈلسىزلىكلەرگە يولۇققانلىرىڭىزدا ھاراق ئىچسەم كەيىپ قىلسام قايغۇمنى ئۇنتۇپ كىتىمەن دەپ ئويلىشىڭىز مۇمكىن بىراق بۇ ۋاقىتلىق ئۇنتۇلدۇرۇشنىڭ كوچىلاردا دەلدەڭشىپ مېڭىپ يۈرۈشنىڭ نەقەدەر خىجىللىق ئىش ئىكەنلىكىنى سىزنى كوچىدا كۆرگەن كىشىلەرنىڭ سىزگە قانچىلىك يىرگىنىش نەزىرىدە قارايدىغانلىقىنى چۈشىنىپ يىتەلەمسىز؟سىز بۇ دۇنيادىكى كىچىككىنە ئازاپقا بەرداشلىق بىرەلمىسىڭىز سىز نىمانچە ئىرادىسىز ۋە بىچارە بىر ئىنسان ھە.  سىزدىن شۇنداق سورىغۇم كىلىۋاتىدۇ،،سىز بەزى چاغلاردا شۇنچە زىرىكىشلىك ۋە شۇنچە بىئارامچىلىق ھېس قىلىشىڭىز مۇمكىن بىراق سىز مۇشۇ چاغلاردا ئۆزىڭىزگە قانچىلىك تەسەللى ۋە قانچىلىك كۆيىنىش مۇھتاج ئىكەنلىكىڭىزنى چۈشەنگىنڭىزدە ھەقىقەتەن ئۆزىڭىزگە ھەقىقى مىھىر ۋە ھەقىقى كۆيىنىدىغان ئادەملىرىڭىزنىڭ قانچىلىك ئىكەنلىكىنى ھېس قىلىپ يەتمىدىڭىزمۇ؟شۇنداقتىمۇ سىز دوسلىرىڭىزنىڭ كۆپلىكىدىن مەغرۇرلىنىپ يۈرىۋىرەمسىز؟ شۇنى بىلىڭكى چىشىنىڭ ئېقىنى چىقىرىپ تۇرغان ھەممىلا ئادەمنىڭ سىزگە يىقىن دوسىت بولۇپ كىتىشى ناتايىن ... سىزدىن شۇنداق دەپ سورغۇم كىلىۋاتىدۇ...سىز ھەردائىم كېرىلىپ ئۆزىڭىزنى چوڭ تۇتۇپ يۈرگەنلىرىڭىزدە سىزنىڭ بۇنچە مەغرۇرلىقىڭىزنىڭ مەڭگۈلۈك بولمايدىغانلىقىنى چۈشىنىپ يەتتىڭىزمۇ؟شۇنى بىلىپ قېلىڭكى سىزنىڭ بۇنچە قەد كۆتۈرۈپ مەغرۇر ياشاشلىرىڭىزمۇ ھەرگىز مەڭگۈلۈك بولمايدۇ ...ھامان بىر كۈنى قەددىڭىزنى ئېگىدىغان شۇنداقلا سىزنى يەر بىلەن يەكسان قىلىدىغان بىرى چىقماي قالمايدۇ،،چۈنكى چەينەك ھېچقاچان ئۆزىندىن تۆۋەن پىيالىگە ئىگىلمەي قالمايدۇ،،،، سىزدىن شۇنداق دەپ  سورغۇم كىلىۋاتىدۇ...سىز ھەرقېتىم باشقىلاردىن ئەيىپ نۇقسان ئىزدەپ ئۆزىڭىزنى چوڭ تۇتۇپ يۈرگەن چاغلىرىڭىزدا ئۆزىڭىزنىڭ قانچىلىك بىر ئادەم ئىكەنلىكىڭىزنى ئۇنتۇپ قالدىڭىزمۇ؟ھەرقېتىم قاتار چاي ۋە سورۇنلارغا قاتراپ يۈرگەنلىرىڭىزدە سىزنى تۆت كۆزى بىلەن
