ئەسسالامۇ ئەلەيكۇم !بۇرادەرلەر يېقنىدىن بۇيان كۆپلىگەن كىشىلىرىمىزنىڭ ئاغزىدا مۇسۇلمانلار ‹‹قالاق ››دەيدىغان گەپ بولۇپ قالدى ،يېقىنىقى ئەسىرلەردىن بىرى خەلىپىلىكنىڭ يوقۇلىشى دۇنيا ئىسلام كۈچلىرنىڭ ئارقىمۇ- ئارقا مەغلۇپ بولىشى بىر قىسىم مۇسۇلمانلارنىڭ
قۇرئان ۋە سۈننەتتىن يىراقلىشىپ ئۆزىنى بىلەلمەسلىكى تۈپەيلدىن دۇنيا مىقياسىدا مۇسۇلمانلارنىڭ سىياسسى ۋە ئىقتىزادى تەسىرى كۈندىن كۈنگە ئاجىزلاپ باردى ، شۇنىڭدەك يەھۇدى ۋە خىرستىيان كۈچلىرنىڭ خىلمۇ -خىل نەيرەڭلىرى بىلەن، ئەقىدىسى ئاجىز خېلى بىر قىسىم مۇسۇلمانلار، ئاللاھنىڭ ‹‹سىراتەل مۇستەقىم››يولىدىن ئازغانلارنىڭ يولىغا قاراپ مېڭشقا باشلىدى .
ئىسلامنىڭ ھەقىقىتى ،دىنسىزلارغا قېتىلىپ كەتكەن بۇ ئازغۇن كىشلەرنىڭ  ۋە دۇنيادىكى بارلىق دىنسىزلارنىڭ كۈچىشى بىلەن خۇنۈكلەشتۈرۈلدى ، شۇنىڭ بىلەن بىرگە ئۇلار ، بىز ۋە تەقدىرى بىزگە ئوخشايدىغان بارلىق مۇسۇلمانلارنى دۇنيا سەھنىسىدىكى ئەڭ ئەسكى كىشلەردىن قىلىپ كۆرسىتىشكە تىرشماقتا،‹‹كاپىرلار ھېلە- مىكىر قىلىدۇ››
مانا بۈگۈن 4خەلىپە ۋە ھارۈن رەشىد، مۇئاۋىيە قاتارلىق ئىسلام سەردارلىرنىڭ تاپىنغا سۆيۈپ ياشىغان ئىت كەبى خارلانغان بۇ دىنسىزلار ئاداۋەتخورلىقىدىن  بۈگۈنكى كۈندە بىز مۇسۇلمانلارنى دۇنيادىكى ئەڭ قالاق ،ئەڭ بىچارە قەۋىم ۋە ئەڭ ئەسكى دىننىڭ ئىگىسى قىلىپ كۆرسىتىشكە ئۇرۇنماقتا.
ئۇنىڭ ئىچىكى ئەڭ گەۋدىلىك بولغىنى ‹‹مۇسۇلمانلار تەرەققى تاپمىغانلا ،››‹‹ مۇسۇلمانلاق بۈگۈنكى دەۋىردىكى ئەڭ قالاق كىشلەر››،‹‹تەرەققياتقا توسقۇنلۇق قىلغۇچىلار››. دىگەنگە ئوخشاش سەپسەتىلەرنى كۆتۈرۈپ چىقىپ ئەۋلاتلىرىمىزغا ئىسلامنى دۈشمەن قىلىپ كۆرسىتىش ھېلىسدۇر.
پېقىر بىلمىم چەكلىك، تەپەككۇرۇم ئاددى بولىشىغا قارماي بۇھەقتە قېرىنداشلار بىلەن ئويلىغانلىرىمنى  ئازراق بولسىمۇ ئورتاقلىشىش مەقسىتىدە بۇ تىمنى يوللۇدۇم، قېرىنداشلارنىڭ ئەقىل كۆزى بىلەن تەپەككۈر قىلىپ، ئىلمى پىكىر يۈرگۈزۈشىنى ئۈمۈد قىلىمەن.
خوش! ئۇنداقتا ‹‹تەرەققىيات››دىگەن نىمۇ؟
بۇ سۆز نۇرغۇن مەزمۇنلارنى ئۆز ئىچىگە ئالىدۇ ،ئەمما بىز  بۇيەردە دەۋاتقان تەرەققىيات دىنسىزلارنىڭ كۆز قارىشى بويىچە مۇنداق بىر ئۇقۇمنى ئىپادىلەيدۇ.
تەرەققىيات ــــــــــ دۇنيا مەنپەتلىرىدىن ئەڭ كۆپ ،ئەڭ قۇلاي ھالەتتە پايدىلنشىتۇر.
قاراڭكى،دىنسىزلار ماختاۋاتقان ،مەدىھىيلەۋاتقان ۋە بىزنىڭ بىر قىسىم مۇسۇلمانلىرىمىزمۇ بۇنى بۈيۈك دىنمىزدىن ئۈستۈن كۆرىۋاتقان ، ئاتالمىش ‹‹تەرەققىياتقا››،
مانا بۇ ،رەببىمىز راھمان تائالاننىڭ ئالدىدا چىۋىننىڭ قانىتىچىلىك ئىتبارغا ئىگە بولمىغان ،باقى دۇنيادىكى ماددى يۈكسىلىشتۇر ،
ئۇنداقتا بۇ ماددى يۈكسەكلىكنىڭ ئىگىسى بولمىش دىنسىزلار راستىنلا تەرەققى قلغانمۇ ؟
مەنچە ياق . ئۇلار ئەڭ ئەشەددى قالاقتۇر ، چۈنكى ياراتقۇچىمىز ئالاھ تائالا ‹‹ئىنسان ۋە جىننى ماڭلا ئىبادەت قىلىش ئۈچۈن ياراتتىم››دىگەن ئايەتنى نازىل قىلىش ئارقىلىق ئىنسانىيەت ۋە جىننىيەتنىڭ تەرققىيات نىشانىنى بەلگىلەپ  بىرىپ بولغان ، ئۇنداقتا  پەقەتلا ئۆزىنىڭ نەپسى ۋە شەھۋىتى ئۈچۈن‹‹ تەرەققى ›› قىلغان ئاللاھنىڭ بۈيۈك قانۇنى ۋە كۆرسەتمىسىگە بويۇنتاۋلىق قىلىپ ئۈستۈنلىك تالاشقۇچى ،ئاسى ئىنسان قانداقمۇ تەرەققى قىلغان
بولسۇن !
ئۇلار كۆرەڭلىكتىن، ئالاھ تائالاننىڭ ‹‹ھېچ كىشى ئۇنىڭغا(ئاللاھقا) تەڭداش بولالمايدۇ›› دىگەن خىتابىغا ئىتراز بىلدۈرگۈچىلەردۇر.
ئۇلار ئاللاھنىڭ زىمىنىدا، ئاللاھنىڭ نئېمەتلىرىدىن پايدىلنىپ تۇرۇقلۇق ئاللاھقا شۇكۈر ئېيتمىغۇچى كاززاپلاردۇر.

ئەمدى مۇسۇلمانلارغا كەلسەك ،مۇسۇلمان ،ئاللاھقا تەسلىم بولغۇچىدۇر ، ھەققىقى بىر مۆئمىن مۇسۇلمان ھەرگىزمۇ ئاللاھنىڭ ئىتائىتىدىن چىقمايدۇ ۋە ئاللاھنىڭ نىئمەتلىرنى ئىنكار قىلمايدۇ ، ئەمما بۈگۈنكىدەك كۇپرى ھەدىدىن ئاشقان ، دىنسىزلار ئىنسانىيەت تارىخى بۈگۈنگىچە كۆرۈپ باقمىغان دەرىجىدە ماددى دۇنيا مەنپەتلىرى ئۈچۈن ئىتپاقلاشقان ،كۆپلىگەن مۇسۇلمانلار ئۆزلىرنىڭ توغرا يولىدا مۇستەھكەم تۇرالمىغانلىقى سەۋەبىدىن ئالاھنىڭ ياردىمدىن ئايرىلىپ قېلىۋاتقان ،ياكى ئاللاھنىڭ ئېغىر سىناقلىرىدا سىنلىۋاتقان، بىر دەۋىرگە دۈچ كەلدى .
شۇ ۋەجىدىن ئۇلار تۈرلۈك سۈيقەستچىلەرنىڭ زىيانكەشلىكىگە ۋە چەكلىمىسىگە ئۇچىراپ كەلدى، ھەمدە ماددى دۇنياسىنى بېيتىشقا پۇرسەتسىز قالدى ،مانا بۇ مۇسۇلمانلارنىڭ دۇنيا تەرەققىياتدا ئارقىدا قېلىشىنىڭ تۈپكى سەۋەپلىرىدۇر، ھەرگىزمۇ ئىسلامنىڭ خاتالىقى ئەمەس . ئىسلام مۇسۇلمانلارنىڭ شەرپىدۇر.
بۇ يەردە شۇنىمۇ ئېيتىپ ئۆتۈش كىرەككى توغرا يولدىن كەلگەن ماددى بايلىق ھەرگىزمۇ ئىنساننىڭ دۈشمىنى ئەمەس، بەلىكى ئۇ مۇسۇلمانلار ئۇچۈن ئاللاھ كۆرسەتكەن تەرەققىيات نىشانىغايىتىشتىكى بىر ۋاستىدۇر .
دىنىمىز ھېچقاچان توغرا بولغان ماددى دۇنيا تەرەققىياتنى چەكلىگەن ئەمەس، بەلكى ئىلھاملاندۇرغان ،
مۇبارەك پەيغەمبىرىمىز مەدىنىگە ھېجرەت قىلىپ كەلگەندىن كىيىن ساھابىلەرنى سودا قىلىشقا ، ھۈنەر ئۈگۈنىشكە ،ئۆزىنىڭ تۇرمۇشىنى ئۆزلىرى قامداشقا ھەمدە پۇل تېپىپ، تاپقان پۇللىرىدىن مىسكىنلەر ئۈچۈن زاكات بىرىشكە دەۋەت قىلغانلىقى . ھەمدە ساھابىلەرنىڭ دۆلەت ئاتلاپ ئىلىم ئۈگۈنىشى، ئۈگۈتىشى ،سودا قىلىشى .ئىسلامنىڭ تەرەققىياتقا بولغان نەمۈنىلىك ئۈلگىللىرىدۇر،  خەلىپىلىك دەۋرى ۋە1200 يىللىق ئۆتكەنكى تارىخ مۇسۇلمانلارنىڭ دۇنيادىكى  ئىقتىزاتتا ۋە ئىلىم- پەندە يىتەكچىلىك رول ئويناپ كەلگەن شانلىق نەتىجىللىرى بىلە بىزەلگەندۇر ، بۇنىمۇ ئۇنتۇپ قالمايلى!

ئەمما دۇنيا تىركچىلىكىنى ئاخىرەت ئەمەللىرىدىن ئارتۇق كۆرۈش، يەنى تەرەقىياتنىڭ ئاساسى نىشانىنى ئۇنتۇپ قېلىش بۇ ھەرگىمۇ تەرەققى قىلغانلىق بولمايدۇ .
بۈگۈنكى مۇسۇلمانلار، يۇقرىدا دىيىلگەندەك تۈرلۈك چەكلىمە ۋە سەۋەپلەر تۈپەيلدىن بىز دەۋاتقان ‹‹تەرەققىياتنىڭ›› كەينىدە قالغان بولسىمۇ ئۇلار ھەرگىزمۇ ‹‹قالاق›› ئەمەس، بەلكى بىر ئاددى مۆئمىن مۇسۇلمان ، ئاللاھنىڭ نەزىرىدە كىشلەر ‹‹ئانداقچى ئالىم ،مۇنداقچى ئېنژىنىر ›› دەۋاتقانلاردىن نەچچە ھەسسە ئۈستۈن بولغان ‹‹تەرەققىيات ››ساھىبىدۇر .
بىلىشىمىز كىرەككى، ئىنسانىيەتنىڭ تەرەققىيات مەنبەسى دەل بىز مۇسۇلمانلاردۇر ، مەسلەن ئادەم ئەلەيھى سالام دەۋرىدە دېھقانچىلىق يۈكسەك تەرەققى قىلغان بولۇپ كىشلەر ئەتتىگەن تېرىپ كەچتە يىغۋالاتتى . داۋۇت ئەلەيھى سالامنىڭ دەۋرىدە تۆمۈرچىلىك ھەم ئالاھىدە تەرەققى قىلغان بولۇپ داۋۇت ئەلەيھى سالام توقۇغان ساۋۇتلار بۈگۈنكى دەۋىلەردە تېخى توقۇلغىنى يوق، سالىھ ئىنسان ،ئىسكەندەرنىڭ دەۋرىدىكى قۇيمۇچىلىق سانائىتى ھازىرچە بىزدىن يىراق ، سەمۇدنىڭ قەۋمى ياسىغان ئەل -ئىھراملانىڭ سىرى تېخى ئېچلمىدى ، ئۇلار مۇسۇلمان تەرەقىياتچى پىشىۋالاردۇر ، ۋاھا كازالار ،
قىسقىسى ئاللاھ خالىقتۇر ، ئۇ خالىغان شەيئىنى ياراتقىچىدۇر ،خالىغان كىشنىڭ قولى بىلەن خالىغان نەرسىنى بەرپا قىلغۇچىدۇر ، كاپىر ھەم ئاللاھنىڭ مەخلۇقىدۇر ،ئاللاھ مەلۇم نەرسىلەرنى ئۇلارنىڭ قولى بىلەن يارتىپ يەنە شۇلارنىڭ زىينىغا مۇسۇلمانلارنىڭ پايدىسىغا قىلىشقا ھەم قادىردۇر ، ئاللاھ ھېكمەت بىلەن ئىش قىلغۇچىدۇر .

خۇلاسە ، رەببىمىز ئالاھ! ئىنسان باللىرنى ھۆرمەتلىك قىلدى ،  ئالەمدىكى بارلىق شەيئىلەرنى ئىنسانىيەتنىڭ مەنپئەتى ئۈچۈن يارتىپ بەردى .
مەقسەت ،كەڭ مۇسۇلمان قېرىنداشلار ،دىنسىزلارنىڭ قايمۇقتۇرىشىغا قايمۇقۇپ كەتمەڭلار ! ۋە ئۆزۈڭلەرنى تۆۋەن كۆرۈپ دىنسىزلارنى ماختاپ ئۇچىرۋەتمەڭلار ! كىمنىڭ تەرەققى قىلغانلىقىنى ئاللاھ ئوبدان بىلىدۇ ،
جەننەت بىزنىڭ تەرەقىيات بىكتىمىزدۇر ، مۇسۇلمان جەننەتكە بېرىش ئۈچۈنلا تەرەققى قىلىدۇ ، ئىلىم- پەن ھەم ئاللاھنىڭ ھىكمىتىدۇر ، ئۇ توغرا پايدىلانغۇچىلارغا ياخشىلىق ، خاتا پايدىلانغۇچىغا زىيان ئېلىپ كەلگۈسىدۇر.

رەبىمىز ئاللاھ ،ھەق يولىدا بولساقلا ھامان بىزگە كۆپلىگەن تەرەققيات پۇرسەتلىرىنى يارتىپ بەرگۈسى ،
ۋە ئىنسانىيەتنىڭ يىتەكچى كۈچلىردىن قىلغۇسى . مۇسۇلمان ئىنسانىيەتنىڭ جەۋھىرىدۇر، ئىنسانى تەرەقىياتنىڭ يۈكسەك چوقۇسى مۇھەممەد سەللەلاھۇ ئەلەيھى ۋەسسەللەم ھەر بىر مۇسۇلماننىڭ تەرەققىيات ئۇستازىدۇر .

ئىنشائالاھ ! بارلىق مۇسۇلمانلار ھامان بىر كۈنى دۇنيا تەرەقىياتىدىكى يىتەكچلىك ئورۇننى قايتۇرۇپ ئالغۇسى،
ۋەئىنسانىيەتنىڭ مەنپەتى ئۈچۈن ئاۋالقىدەك كۆپلىگەن پايدىلىق تەرەققىياتلارنى قولغا كەلتۈرۈپ دۇنيا ۋە ئاخىرەتتە پايدا ئالغۇسى ،

ئىلمى پىكىر قىلىڭ!

(ئادەم گۈشىنىڭ باھاسىنى تۆۋەن چاغلىماڭ ! ئۇجەننەتنى ياكى دوزاخنى ئەكىلىپ بىرىدۇ.)
