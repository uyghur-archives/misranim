مەن ئۇنى كۆرۈپ خوشال بولدۇم، نىملا دىگەن بىلەن qq رىمدىكى بىر تاللا كىشى ئەمەسمۇ،

ھە راس ئۇ ماڭا نەچچە كۈندىن كېيىن سۆرەت ئەۋەتىشكە ۋەدە بەرگەن، سوراپ باقاي ، لېكىن مەن ئاشۇ - سىز ماڭا سۆرەت ئەۋەتىشنى ۋەدە قىلغان ، ھازىر ئەۋەتىپ بېرەمسىز دىگەن بۇ گەپنى خەنزۇچە قانداق يازدۇ بىلمەيمەن ، ئاھ خۇدا مىنى مۇشۇنداقلا بىلىمسىز يارتتىڭمۇ دىدىم،  ئۇلۇغ ئاللاھ چوقۇم تىرىشقان كىشىگە بېرىدۇ، ئۆگەنگەننى ئاللا ھەرگىز بىكارچى قىلمايدۇ، چوقۇم ئۆگۈنۈمەن دەپ خەنزۇچە كەشلىك كۇرۇسقا تىزىملاتتىم.
خەنزۇچىدە ھېشقانداق ساۋادى بولمىغان بىر كىشىنىڭ بىراقلا كۇرۇسقا كىرىپ ئۆگۈنۈشى ئەلۋەتتە ئاستا بۇلدىكەن، ياخشى ئوقالمىسام، مەيلى ساۋاقداشلىرىمنىڭ ،مەيلى باشقىلارنىڭ كەمسىتىشلىرىگە ئۇچرىسام ، خەنزۇچە ئۆگۈنۈش نىيتىمنىڭ مۇستەھكەملىكى ماڭا كۈچ مەدەت بېرەتتى.
  
كۈندۈزى ئېتېز ئىشى ، كەشتە بولسا خەنزۇچە كەچلىك كۇرۇس ، كەشلىك كۇرۇستىن قايتىپ ئۇنىڭ بىلەن پاراڭلىشىش بۇلاتتى ، ئۆيگە كىلىمەن كومپىيوتۇرنى ئېچېپ ئۇنىڭ بىلەن پاراڭلاشسام دەيتتىم،
بۈگۈن ئۇ توردا باركەن، مەنمۇ خەنزۇچە كەشلىك كۇرۇسقا كىرۋاتقىلى خېلى بولدى، ئۆزەم بىر سوراپ باقاي رەسىمنى دەپ خەت باشلىدىم،
مەن- ھەي سىز بارمۇ يە،
ئۇ- بارغۇ ،
مەن - ماڭا ۋەدە قىلغان رەسىمچۇ؟
ئۇ- ھە بار ، qqرىمنىڭ بوشلىقىغا چىقىرىپ قويدۇم، سىزگە ھازىر ئەۋەتىپ بېرەي ،
مەن- ھە ماقۇل
ئۇ- شۇنداق دىگۈنۈچە بىر نەچچە رەسىمنى ئەۋەتتى،
ئۇ رەسىمنى قانداق تەسۋىرلىسەم بۇلا، ئاپاراتنى ئىگىز كۈتۈرۈپ ئاسمانغا يانتۇ قاراپ چۈشۈپتۇ، مودا مىنكوخەن خېنم خېلى ياخشى چۈشۈپتۇ ،  راستنى دىسەم ئۇماق قىزكەن
مەن- بۇ راس سىزما؟
ئۇ - مەن بولماي كىمۇ، ئەلۋەتتە مەن،
مەن- ھەقىقەتەن ئۇماقكەنسىز، چىرايلىقكەنسىز ،
ئۇ - رەھمەت
ئۇ - كەچۈرۈڭ ،ئىشىم چىقىپ قالدى، مەن چۈشتۈم خوش، كېيېن پاراڭلىشايلى .

مەن خوش دەپ بولغۇچە ئۇ توردىن چۈشۈپ بوپتۇ،  ھەي ئۇنىڭ ئوماق سۆرەتلىرىگە قاراپ ئولتۇرۇپ خىيال سۈرۈپ كىتىپتىمەن، خۇددى ئۇنى ياخشى كۆرۈپ قالغاندەكلا، بەلكىم شۇنداقتۇ،ئۇنىڭ بىلەن پاراڭلىشىش تەس كىلىپ خەنزۇچە كۇرۇسقا كىرۋاتىمەنغۇ ، بەلكىم شۇنىڭ ئۈچۈن ، بەلكىم ئۆزەم ئۈچۈن ، ئۇنى بىلمەيتتىم.
مەن داۋاملىق كۇرۇسقا كىردىم ، خەنزۇچىدە نەتىجەم خېلى ئىلگىرلەش بولدى، ئىنىمنىڭ ياردىمىسىزمۇ ، بىمالال خەت يېزېشنى يېتېلدۈردۈم، شۇ كۈنلەردە ئۇ تورغا چىقماس بۇلۋالدى، qqرىمدا ئىلگىركىدەكلا پەقەت مۇشۇ قىزلا باتتى ، باشقا ئادەم قاشلاشنىمۇ ئويلاپ باقمىدىم، مانا كۇرۇسقا كىرىپ توپتوغرا بىر ئاي بوپتۇ، خەنزۇ يولداشلار بىلەن ئۇچرۇشۇپ قالساممۇ تەمتىرمەستىن خېلى گەپلىشەلەيدىغان بولدۇم، بۈگۈن ئۇنىڭ تورغا چىقىشىنى شۇنچىلىك ئارزۇ قىلدىم ، چۈشتىن باشلاپ ، قاراڭغۇ چۈشكۈچە ساقلىدىم. ئەپسۇس ئۇ كۆرۈنمەيتى، مەن ئۇ چوقۇم چىقسا ئۆزەمنىڭ ئىش ئىزلىرىنى ، قانداق ئادەم ئىكەنلىكىمنى، خەنزۇچىنى شۇنچىلىك راۋان سۆزلەشكە ، خەتلەرنى بىمالال يېزېشقا ئۆزىنىڭ تۈرتكە بولغانلىقىنى دەيمەن دەپ نىيەت قىلدىم. كەشقۇرۇن خۇپتەن نامىزىدىن كېيېنرەك ئۇ توردا پەيدا بولدى.

مەن- ياخشىمۇ سىز؟
ئۇ-ياخشى سىزچۇ?
مەن- ياخشى ، ھەجەپ يۇقاپلا كەتتڭىزغۇ ، تورغىمۇ تۈزۈك چىقمايسىز؟
ئۇ-ئۆگۈنۈش بىلەن ئالدىراش،كۈندە تەكرار - كۈندە تاپشۇرۇق ئىشلەپ شۇ،
مەن- سىزگە بىر گەپ دەيمۇ ؟
ئۇ- ھە دەۋېرىڭ ،
مەن ئەسلى خەنزۇچە بىلمەيتىم،
ئۇ- نىمە؟ مانا يېزۋېتېپسىزغۇ ، شۇنداق ياخشى ، گەپلىرىڭىزنى چۈشەنمىدىم،
مەن - 1ئاي بۇرۇنقى مەن ئەمەس ، ئۇ ۋاقىتتا مەن دەپ بەرگەن سۆزلەرنى ئىنىم يازدى ، مەن خەنزۇچىدە ھېچقانداق خەتنى بىلمەيتتىم، مەن ھەم شوپۇرمۇ ئەمەس، مەن بىر ئاددى دېھقان بالىسى ، مەكتەپتە ياخشى ئوقۇمىغان،  قۇلۇمدا تۇلۇقسىز مەكتەپنىڭ دېپلومى بار تۇرۇپ، خەنزۇچىنى بىلمەيدىغان  بىر ياش،   مىنىڭ خەنزۇچە ئۆگىنىشىمگە سىز زور تۈرتكە بولدىڭىز، سىز بىلەن ياخشى دوست بولساق بۇلامدۇ؟

منىكوخەن قىزچاق بىر ھازا تۇرۋاتقاندىن كېيېن، qq ئويناپ سىزدەك تورداشنى تۇنجى ئۇچرىتىشىم، خەنزۇچە ئۆگۈنۈشتىكى بۇ روھىڭىزدىن تەسىرلەندىم، ماقۇل ياخشى دوست بولۇپ قالايلى،
مەن- سىز تورغا چىقمايدىكەنسىز ، ئۆگۈنۈش بىلەن ئالدىراش دەيسىز، تېلپون نۇمۇرىڭىزنى قالدۇرسىڭز ، مەن تېلپون ئۇراي قانداق،
ئۇ- كەچۈرۈڭ ھازىر تېلفونىم يوقتى ، نەچچە كۈندىن كېيىن تېلفون تۇتقان چاغدا دەپ بېرەي، ياكى qq رىڭزغا قالدۇرۇپ قوياي،
مەن - ھە ماقۇل ، ئۇنتۇپ قالماڭ،
ئۇ - ھەي خوش مەن چۈشتۈم، ئۆگۈنۈش قىلمىسام بولمايدۇ ، دەپ چۈشۈپ كەتتى

ئۇ شۇ چۈشكەنچە تورغا چىقمىدى، مانا بىر ھەپتە بۇلاپتۇ، يەنە يوق،  كەشلىك كۇرۇستىن يېنېپ qqرىمنى ئاشتىم ، ئۇ تېلفۇن نۇمۇرىنى قالدۇرۇپ قويۇپتۇ، خوشاللىقىمدا ئۇنىڭ ئاۋازىنى ئاڭلايدىغانلىقىمدىن ھاياجانلىنىپ نۇمۇرنى تەستە بېسېپ ، ئۇلاش كۇنۇپكىسىنى باستىم، ئەپسۇس تېلپون ئىتۋاپتۇ،
ئەتىسى يەنە نۇمۇرنى بېسىپ ئەگەر ئالسا نىم دەي ؟ دەپ ئويلاپ قۇلۇمنىڭ ئۇلاش كۇنۇپكىسىغا تىگىپ كەتكىنىنى بىلمەي قاپتىمەن، قارىسام ئۇلاندى  ، مۇزىكىلق قۇڭغىراققا ،  كەچۈرەلمەيمەن دىگەن ناخشىنىڭ كوبلىتلىرى چىقۋاتاتتى

                                                                         يالغان بولدى بەرگەن ۋەدىلەر.

                                                                         ئۇنتالىسام سىنى ناۋادا،

مەن- ئەسسالامۇ ئەلەيكۇم ، دىلراباما سىز؟
ئۇ- ئەسسالامۇ ئەلەيكۇم ھە مەن، سىز ئۆتكۈرمۇ؟
مەن - ۋەئەلەيكۇم كومسالاي ، ئەسسالامۇ ئەلەيكۇم دىسەم  يەنە ئەسسالامۇ ئەلەيكۇم دەيسىزغۇ ۋەئەلەيكۇم ئەسسالام دىمەي دىدىم.
ئۇ پاراققىدە كۈلۋەتتى،مانا شۇنىڭ بىلەن بىزنىڭ تېلفونلىشىشىمىز باشلاندى، ئۇ باشلانغۇچتىن تارتىپ خەنزۇچىدە ئوقۇغان ، ئۇيغۇرچە خەتلەرنى يېزىش ، ئوقۇشنى بىلمەيدىكەن، بۇ يىلقى تەتىلدە چوقۇم ئانا تىلىمنى يېزېشنى ، چوقۇم ئانا تىلىمنى ئىشلىتىشنى ، خەتلەرنى ئوقۇشنى ئۆگۈنىمەن دەپ بەردى.
