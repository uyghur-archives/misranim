ئەسسالام دوسلار ،بۇ تىما خېلى قىزىپ قاپتىغۇ مەنمۇ ئىككى كەلىمە سۆز قىستۇرۇپ قوياي:
1853-يىلى ئامرىكىنىڭ بىر فىلوتى ياپۇنيە پورتىغا بېسىپ كىرىپ سودا ئالاقە قىلىشنى(ئىلگىرى بىكىنمىچىلىك سىياسىتىدە) تەشەببۇس قىلدى.ئۇنداق بولمايدىكەن ئۇرۇش قىلدىغانلىقىنى بىلدۇردى ،شۇنىڭ بىلەن ئامالسىز شوگىنات(سانغۇننىڭ ھاكىميەت ئورگىنى)ھاكىميىتى پورتلارنى ئېچىۋەتتى ،شۇنىڭ بىلەن باشقا دۆلەتلەرمۇ شۇنداق تەلەپلەرنى قويۇپ ھاكىميەتكە ئېغىر تەھدت سالدى.ھاكىميەت سەكراتقا چۈشۈپ قالدى.بىر قىسىم ئوتتۇرا تۆۋەن دەرىجىلىك سامۇرايلار غەرىپنىڭ ئىلغار تېخنىكىسىنى،ئىلغار ئىدىيىسىنى قوبۇل قىلىپ قورال كۈچ ئارقىلىق مۇستەبىت ھاكىميەتنى ئاغدۇرۇپ تاشلاشنى تەشەببۇس قىلدى ھەمدە غەلبە قىلىپ 1867-يىلى مىيجى يىلدىكى يېڭىلىققا كۈچۈش ئىسلاھاتىنى ئېلىپ باردى .ئىسلاھاتتا:
ناھىيە تەسىس قىلىش ،ھوقۇقنى مەركەزلەشتۈرۈش.غەرىپنىڭ ئىلغار تېخنىكىسىنى كىرگۈزۈش،سانائەتنى تەرەقى قىلدۇرۇش،تەرەقىپەرۋەرلىشىشنى تەشەببۇس قىلدى.ئەڭ مۇھىمى ياۋرۇپا ئامرىكىلىقلاردىن ئۆگىنىپ تىرىشىپ مائارىپنى تەرەقى قىلدۇرۇش .
مەجبۇريەت مائارپ سىياسىتىنى يۈرگۈزۈپ بىر تۇتاش ئوقۇتۇش تۈزۈمىنى قوللاندى .
ئەسكەرلىككە ئېلىش تۈزۈمىنى قوللىنىپ «سامۇراي »روھىنى قەدىرلەيدىغان ،تېننوغا سادىق ئارمىيە قورۇشقا باشلىدى.بۇ ئارمىيە ناھايىتى تىز ياپۇنيەنىڭ سىرىتقا قارىتا كېڭەيمىچىلىك قىلدىغان قورالىغا ئايلاندى .يېرىم ئەسىردىلا ياپۇنيە كۈچلۇك دۆلەت قاتارىدىن ئورۇن ئالدى .

ئەمدى ئۆزۈ مىزنىڭ گىپىگە كىلەيلى مانا بىزمۇ ياپۇنلاردىن قىلىشمۇغىدەك مىللەت، مەن ئالى مەكتەپتە ئوقۇغان ۋاقتىمدا، كىتاپقا شۇنداق ھىرىسمەن ئىدىم، بىركۈنى مەكتەپ كۈتۈپخانىسىنى ئارلاۋىتىپ ناھايتى كونا بىر جوڭگو تارىخى كىتاپقا كۆزۈم چۈشتى،بۇكىتاپ خەنتىلچە ئاپتۇرىمۇ خەنمىللىتىدىن ئىدى، مەنبۇ كىتاپنى ئارىيەتكە ئىلىپ ئوقۇدۇم ، بۇكىتاپتا جۇڭگو تارىخى چڭنشىخۇاڭ دەۋرىدىن باشلاپ سۆزلەنگەن ئىدى، ئەينى ۋاقىتتا چڭنشىخۇاڭ نۇرغۇن قىتىم قەسىتكە ئۇچۇرغان. ئۇنۇرغۇن مىللەتتىن ئۆزىگە قوغدۇغۇچى قويۇپ سىنىغان، لىكىن دىگىدەك كۆڭلىدىكىدەك ياخشى قوغدىيالمىغان، كىيىن ئۈمۈدسىزلەنگەن يەنە بىرمىللەتنى ئۆزىگەقوغدىغۇچى قىلىپ سىنىغان ، بۇمىللەت چنشىخۇاڭنى ناھايتى ياخشى قوغدىغان، كىيىن چىنشىخۇاڭ بۇمىللەتنىڭ قورۇقماس قىززىق قان باتۇرانە جاسا رىتىنى كۆرۈپ بۇمىللەتتىن نۇرغۇن ئەسكەر ئالغان، كىيىن بۇمىللەت چىن دۆلىتى ئۈچۈن نۇرغۇن تۆھپىلەرنى قوشقان ، بۇمىللەت كىمدۇ؟ مانا دەل بىز ئۇيغۇرلار. كىيىن چىنشىخۇاڭنىڭ بەزىبىر ھەققانى ئادىل ئەمەس سىياسەت قانۇنلىرىغا  دىگىدەك  ئۇيغۇرسەركەردىلىرىنىڭ ھەممىسى قايىل بولمىغان ھەمدە سىياسەتلىرىگەئەمەل قىلمىغان . بۇنى بايقىغان چىنشىخۇاڭ كىيىن ھەممىسى بىرلىشىپ قارشى چىقىشتىن ھەم شاھلىق ئورنىدىن مەھرۇم قىلىشتىن قورققان چىنشىخۇاڭ ئۇيغۇر سەركەردىلىرىنى يۇرتىغا قايتۇرۋەتكەن ، بەزىلىرىنى مەخپى ئۆلتۈرۋەتكەن، بەزىلىرىنى ئەمىلى ئۆسمەس ئىشىك باقاۋۇلىىقىغا سالغان ، ھەم ئاددى ئەسكەر ئورۇنىغا چۈشۈرۈپ قويغان.
مانا ھازىرمۇ جۇڭگودا ئۇيغۇر ئەسكەر يوق دىيەرلىك ،ئۇيغۇر ئەرلىرىمىزنىڭ جەڭگىۋارلىقى خاراپ قىلىندى، دىياۋيۈ ئارىلى مەسلىسى چىققاندىن بۇيان خېلى كۆپ ئۇيغۇرئەسكەر ئالدى، تېخى ئەسكەر قوبۇل قىلىش ئىلانىنى تىلىۋىزۇردا ئىلان قىلغىنىنى دەيمەن.تىخى بەش ئالتە ئاينىڭ ئالدىدا بىرئۇيغۇر ئاتا بالىسىنى ئەسكەرلىككە ئالدۇرالماي، پارا بىرىش نىتىگىمۇ كەلگەن ئىدى. لىكىن ھازىردۆلىتىمىز تاس قالدى يىگىتلىرىمىزنى  مەجبۇرى ئەسكەرلىككە ئىلىپ ماڭغىلى.
ھازىر  ھەقىقى ئۇيغۇر تارىخى يىزىلغان كىتاپ گىرمانىيە مۇزىخانىدا،
تارىخلىرىمىز يىزىلغان كىتاپلارنىڭ ھەممىسى 50يىل ئالدىدا باشقىلار تەرىپىدىن كۆيدۈرۋىتىلگەن،
ئەينى ۋاقىتتا شىنجاڭغابىسىپ كىرگەن شىڭسىسەي ۋە سىتالىن لار بىر ئۇيغۇرلارنى ماختاپ كەتكەن. لىكىن بىزدە ھازىر جەڭگىۋاردىن ئەسەرمۇ قالمىدى.
