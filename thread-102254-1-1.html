ئەسسالامۇ ئەلەيكۇم مۇنبەرداشلار ھەممىڭىزلەرگە ئاللاھنىڭ رەخمەتلىرى بولسۇن،
بىر نەچچە كۈننىڭ ئالدىدا مەھەلىمىزدە بىر مىيىت نامىزى بۇلۇپتۇ دەپ ئاڭلدىم، ئەنە شۇ مىيىت نامىزىغا قاتنىشىپ ئۈلۈمنى بىر ئەسلەش بىلەن ئاللاھتىن قورقىدىغان قەلبىمنى ، ئىمانىمنى كۈچەيتىش ئۈچۈن، ئەنە شۇ نامازنى چۈشۈرگىلى كەلگەن ئالىملارنىڭ ۋەز نەسىھەتلىرىنى ئاڭلاش ئۈچۈن ئۇ مىيىت نامىزىغا قاتنىشىىشنى نىيەت قىلدىم، جانابى ئاللانىڭ ئىرادىسى بىلەن شۇ كۈنى پىشىن نامىزىدىن كېيېن مىيىتنى نامىزىغا ئۈلگۈرەلىدىم.
  مەسچىتكە بېرىپ پىشىن نامىزىنى ئۈتەپ بولغاندىن كېين ، مىيىتنى چۈشۈرۈش ئۈچۈن  مەسچىتنىڭ ئالدىغا مىيىت ئېلېپ كىلىندى، جامائەت كەتمەي مىيىت نامىزىغا قاتنىشىپ بېرىڭلار ، دەپ بىر كىشى توۋلىدى. كىشىلەر خېلى كۆپ بۇلۇپ مەسچىتنىڭ ئالدى توشقان ئىدى.

بىزنىڭ بۇ يەردە مىيىت بۇلۇپ قالسا ، بىر ئاخۇن بار بۇلۇپ شۇ ئاتايىن كېلىپ ، ئىسقات تارقىتىش  ، نەزىر قىلىش دىگەندەك بىرمۇنچە سۆزلىۋېتىپ مەيلى ۋاپات تاپقان كىشى ھاراقكەش بولسۇن، ۋە ياكى ناماز ئوقۇمايدىغان كىشى بولسۇن، ھەممىسىنى بىر ئامال قىلىپ جەننىتى قىلىپ سۆزلەپ ، تارقاتقان فىدىيە پۇلىنى، ئېسىل ماشىنىلاردا كىلىپ يانچۇقىغا سېلېپ ئېلېپ كىتەتتى،  بۇگۈن يەنە شۇ كىشى نامازنى چۈشۈردىغان بولدى دەپ ئويلىسام،  ئۇ مۇللام ئەمەس باشقا بىر موللام كەلدى.

ئۇ كىشى  نامازنى چۈشۈرۈشنىڭ ئالدىدا ئازراق ۋەز قىلدى،

                                                  ناھايتى شەپقەتلىك مېھرىبان ئاللاھنىڭ ئىسمى بىلەن باشلايمەن
ئالدى بىلەن جانابى ئاللاھقا چەكسىز ھەمدۇ سانالار بولسۇن، ئول زاتتىن ئۇلۇغراق بىر زات يوقتۇر ،ھەقىقەتەن ئاللا ھەممىدىن ئۇلۇغدۇر ، ۋە پەرۋەردىگارىمىز پادىشاھلارنىڭ پادىشاھىدۇر ، ھەقىقەتەن شاھىنشاھتۇر ،  ۋە پەرۋەردىگارىمىزنىڭ ھەبىبى ، بىزنىڭ پەيغەمبىرىمىز مۇھەممەد ساللاھۇئەلەيھى ۋەسەلمگە دۇرۇت سالاملار، ۋە ئاللاھنىڭ رەخمەتلىرى بولسۇن، ۋە ئۇ كىشىنىڭ دوسلىرى، ۋە ئائىلە -تاۋاباتلىرىغا ، ۋە ساھابىلىرىگە ، تاكى قىيامەت قايىم بولغۇچە ئىمانىدا مۇستەھكەم تۇرغان مۇسۇلمان ۋە مۇمىنلەرگە ئاللاھنىڭ رەخمەت-ھىدايەتلىرى بولسۇن،

ئېيتتى پەيغەمبىرىمىز ، يەتكۈزۈپ قويۇڭلار مەندىن بىرە ئايەت ، ياكى بىرە ھەدىس بىلسەڭلا يەتكۈزۈپ قۇيۇڭلار، ئەنە شۇنى يەتكۈزۈپ ئەمەل قىلماق دۇنيانىڭ بارچە لەززەتلىرىدىن  ياخشىراقتۇر،

بۈگۈن مانا ئالدىمىزدا مەرھۇم تۇرۇپتۇ،  جانابى ئاللا قۇئان كەرىمدە، ھەرقانداق بىر جان ئۈلۈمنىڭ تەمىنى تېتغۇچىدۇر ، دەپ خەۋەر بەرگەن،
مانا بۇ دۇنيادا مەن ھايات قالىمەن دىگەن بىر ئىنسان يوق ،  ئادەم ئەلەيسالامدىن تارتىپ مانا بۈگۈنگىچە تالاي ئىنسانلار جانلىرىنى  ئاللاھقا تەسلىم قىلماي ئىلاجى بولمىدى، مانا ئارىمىزىدىكى ھېچ بىرىمىز مەن ئەبەدى ياشايمەن دىيەلمەيمىز، ھەممىمز ئاللاھنىڭ دەرگاھىغا كەتكۈچى،

ئەبۇ ھۈرەيرە رەزىياللاھۇ ئەنھۇدىن مۇنداق رىۋايەت قىلىندۇ، پەيغەمبەر ساللاھۇ ئەلەيھى ۋە سەللەم -
كىمىكى مىيىتنىڭ نامىزى چۈشۈرۈلۈپ بولغىچە تۇرسا، ئۇنىڭغا بىر قىرات ئەجىر بۇلىدۇ، كىمىكى دەپنە قىلىنىپ بولغۇچە تۇرسا ، ئۇنىڭغا ئىككى قىرات ئەجىر كىلىدۇ- دېۋىدى
ئىككى قىرات دىگەن نىمە - ؟ دەپ سورالدى
- پەيغەمبەر ئەلەيسالام - چوڭ ئىككى تاغقا ئوخشايدۇ دىدى.


ئاندىن ئۇ موللام جامائەتلىرىمىز ئازراق ۋاقىت چىقىرىپ مانا بۇ مىيىتنى قەبرىستانغىچە چىقىپ دەپنە قىلىىشىپ بېرىشىڭىزلارنى ، ئەنە رەسۇلۇللاھ ئېيتقاندەك ، تاغچىلىك ساۋاپقا مۇيەسسر بۇلىشىڭىزلارنى ئۈمۈت قىلىمەن.

ئەمدى بۇ ۋاپات تاپقان   كىشىنى تۇنۇشتۇرۇپ ئۆتسەك، بۇ كىشى مانا 50 ياشتىن ئاشقان كىشىدۇر،  

( بۇ كىشىنى باشقىلار ناماز ئوقۇمامسىز دىسە، ئەتە ئوقۇيمەن دەيدىغان كىشى ئىدى، ئەگەر ئاللا خالىسا ، ئاللا ئىگەم بۇ كىشىنىڭ جانلىرىنى ئالمىغان بولسا ، بەلكىم بۇ كىشى ئەتىدىن باشلاپ نامازنى ئوقۇشى مۇمكىن ئىدى ،  بىراق ئەجەل يېتىپ ، ۋاپات بۇلۇپ قەبرىستانغا كىرگەندە ھەقىقەتەن پۇشايمان قىلساق كىچىكىمىز، بۇ بىر ئىمتىھان دۇنيادۇر  )  

مانا مۇشۇ ماڭا ئەڭ ياققان شىجائەتلىك بىر جۈملە سۆزنى قىلغاندىن كېيىن  ،
  ئاللاھ ھەر بىرىمىزنى دۇنيا ۋە ئاخىرەتتە ئەمىن قىلسۇن، ھەممىمز  ئاللاغا دۇئا قىلايلى، بۇ كىشىنى ھەم بارلىق مۇسۇلمانلارنى  پەرۋەردىگارىمز ئاللاھ گۇناھلىرىنى كەچۇرسۇن، ئەمدى مىيىت نامىزنى باشلىساق، جامائەتلەر سەپ بۇلۇپ تۇرايلى جامائەت،  ئاللاھۇ ئەكبەر....

مىيىت نامىزىنى  چۈشۈرۈپ بۇلۇپ قەبرىستانغىچە ئۇزۇتىشىپ بېرىپ ، تاكى ئۆيگە كەلگىنىمدە ھەم ئۇ ئالىمنىڭ
( بۇ كىشىنى باشقىلار ناماز ئوقۇمامسىز دىسە، ئەتە ئوقۇيمەن دەيدىغان كىشى ئىدى، ئەگەر ئاللا خالىسا ، ئاللا ئىگەم بۇ كىشىنىڭ جانلىرىنى ئالمىغان بولسا ، بەلكىم بۇ كىشى ئەتىدىن باشلاپ نامازنى ئوقۇشى مۇمكىن ئىدى ،  بىراق ئەجەل يېتىپ ، ۋاپات بۇلۇپ قەبرىستانغا كىرگەندە ھەقىقەتەن  پۇشايمان قىلساق كىچىكىمىز، بۇ بىر ئىمتىھان دۇنيادۇر )
مۇشۇ سۆزلىرى يادىمغا يەتسىلا بەدەنلىرىم تىترەك ئولىشىپ كىتىدۇ، چۈنكى بىزنىڭ كۆپ ئوغۇل قىزلىرىمىزنى نامازغا چاقىرلسا ھازىر ئويناپ چوڭ بولغاندا ئوقۇيمەن ،دەيدىغانلارنى  كۆپ ئۇچرۇتۇپ قالىمىز ، ئەتە ئوقۇيمەن ، چوڭ بولغاندا ئوقۇيمەن دىمەڭ، ئاللا ياشلىقىمىزدا قىلغان ئىبادەتنى، ئەڭ ياخشى كۆرىدۇ،  چۈنكى ياشلىقىمىزدا بىزدىكى كۈچ -قۇۋەت ئوغۇرلۇق قىلىشقا، ناچار يامان بۇزۇق ئىش قىلىشقا ھەم يىتىدۇ، ئەنە شۇ يامان ئىشتىن قايتىپ ئىبادەتكە كەلسەك ، ئاللا بىزدىن كۆپ رازى بۇلىدۇ، شۇڭا قېرىغاندا ئوقۇيمەن ، چوڭ بولغاندا ئوقۇيمەن ، ئۇيۇننى ئويناپ بۇلۇپ ئوقۇيمەن دىمەي ھەر بىر مىنوتنى قەدىرلەيلى ،
  ئەنە شۇ ھاياتلىق ۋاقتىمىزنى چىڭ تۇتۇپ ، جانابى ئاللاھقا بەندىچىلىكنى جايىدا ھەقىقى مۇسۇلماندەك قىلىشنى ئاللا نىسىپ قىلسۇن، ئاللا بىزنىڭ ئۈلۈملىرىمىزنى خەيىرلىك قىلسۇن، زالالەتتە يۈرگەن ئۇيغۇر قىز ئوغۇللىرمىزغا ئاللاھنىڭ ھىدايەتلىرى بولسۇن،  ئامىن.
