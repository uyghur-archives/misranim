باھارنىڭ ئىللىق ھېدى دىماقلىرىمغا ئۇرۇلۇپ تۇراتتى شۇ تاپتا مەن شۇنچە جىمجىت ھەم تىمتاس ئورمان ۋادىسىدا تىك قىيالارغا سەپ-سېلىپ قاراپ تۇراتتى قانداقتۇر بىر ئېچىنىشلىق ئۆتمۈش ئۇنىڭ ھاياتىنى قان-زار بىلەن ئاياقلاشتۇردىغاندەك بىلىنەتتى ،تۇيۇقسزلا ئۇنىڭ قولىدىكى تىلفۇن جىرىڭلاپ كەتتى،بىراق فىرۇزە بۇ تىلفۇننى ئىلىشىنى رەت قىلىپ تىلفۇننى ئىتىۋەتتى،شۇ تاپتا ئۇ ئۆزىگە شۇنداق دەپ پىچىرلىماقتا ئىدى،
-مەن ساراڭ ئەمەس ...مەن راستىنلا ساراڭ ئەمەس...
فىرۇزە شۇنداق خىياللارنى قىلىپ دىڭىزنىڭ قىرغىقىغا كىلىپ ئولتۇرغاندىن كىيىن كۆزىدىكى ياشلارنى سۈرىتتى ئۇنىڭ ئەلەمگە تولغان بۇ قەلبى ئۇنى تىخىمۇ ئازاپلىماقتا ئدى.
                                مەسچىد ئالدىدىن تېپىۋىلىنغان بوۋاق
بىر كۈنى سەھەردىلا مەھەللىدىكى جامائەت بامدات نامىزىدىن ئۇزاپ ماڭماقتا ئىدى.نۇردۇن ئاكىمۇ قولىدىكى قول چىراغنى پۇلاڭلاتقىنىچە كىتىپ باراتتى تۇيۇقسىزلا كىمدۇر بىر كىچىك بۇۋاقنىڭ ئىڭراپ يىغلىغان ئاۋازى ئاڭلاندى نۇردۇن ئاكا يۈگۈرۈپ بىرىپ ئۇ بۇۋاققا قارىغىندا شۇنچە ئۇماق بىر قىز بۇۋاق ئىدى،نۇردۇن ئاكا خۇشال بولۇپ ئۇ قىز بۇۋاقنى كۆتۈرۈپ ئۆيگە ئېلىپ كەلدى،
-خوتۇن..تىزرەك ئورنۇڭلاردىن تۇرۇڭلار...
نۇردۇن ئايالى گۈلنۇرنى چارقىرىشقا باشلىدى.
-ئەتىگەندە نىمە بولغاندۇ بۇ ئادەمگە،،،ئوھۇش،،
گۈلنۇر كىرىلىپ ئورنىدىن تۇردى ۋە بۇۋاققا قاراپ چۆچۈپ كەتتى،
-ۋاي ..خۇدايىم ،،كىم بۇۋاق بۇ ئەمدى؟ نەدىن تېپىپ كىردىڭىز؟كىمنىڭ بالىسى؟
-گەپنى ئاز قىل بۇمۇ ئاللاھنىڭ بىزگە بەرگەن نىممىتى،..بۇ بۇۋاق يول بويىدىلا يىغلاپ ياتقان ئىكەن ئېلىپ كىردىم،بىز قىز قېلىپ بىقىۋالايلى دەيمىنا..
-نىمانداق گەپ قىلىسىز؟خولۇم-خوشنىلار سورىسا نىمە دەيمىز بۇ بۇۋاقنى؟
-شۇ مەن بېقىۋالدىم دىمەمسەن؟كىيىن چوڭ قىلىپ قويساق ئۆزىگىمىزگىمۇ ياخشى مانا تۇي قىلغىنىمىزغا 6 يىل بوپتۇ بىراق بىرەر پەرزەنتىمىز يوق بۇ قىزنى قىز قىلپ بىقىۋالايلى،
-ئوغۇ...شۇ....ئۇلۇغ ئاللاھ مۇشۇ بۇۋاقىنى بىزگە تەقىدىر قىلغان بولسا كىرەك ..بۇ قىزغا نىمە دەپ ئىسىم قويىمىز؟
-مەن ئاۋال جامائەتكە بىر خەۋەر قىلاي..بۇ قىزنىڭ ئىسىمنى قويۇپ ھەقلىتىپ قويمىساقمۇ قاملاشماس...بالىنىڭ ئاپىسى سەن بولغانكەن ئىسىمنى ئۆزۈڭ ئويلاشقىن
-توۋا ..مەنمۇ ئانا بولۇپ قالدىم دە ..ئەمدى..بىراق بۇ بىچارە نارسىدە بالىنى تاشلىۋەتكەن قايسى نائەھلى بولغۇيتى ...ھەي..بىر بالىنى 9 ئاي 9 كۈن قورساق كۆتۈرۈپ تۇغۇپ ئەمدى تاشلاپ قويۇشقا قانداقمۇ كۆڭلى ئۇنىغان بولغۇيتى ھە...
-بولدى قىل خوتۇن ،،،بۇ نارسىدە قىزنى بىز بېقىپ چوڭ قىلىپ قاتارغا قوشساق ياخشى ئەمەسمۇ..ساۋاپ بىرىدۇ.
نۇردۇن شۇنداق دەپ قويۇپ چىقىپ كەتتى،گۈلنۇر ھەدە بۇ ئۇماق قىزنىڭ بىشىنى سىلاپ ئۇ قىزنى تاتلىققىنە سۆيۈپ قويدى كەچتە مەھەللىدىكى جامائەتلەر كىرىپ ئۇ قىزغا فىرۇزە دىگەن ئىسىمنى قويدى،

                         گۈدەك فىرۇزە،
ئارىدىن 2 يىل ئۆتۈپ فىرۇزە ئەقىل تېپىپ 2 ياشقا كىردى.شۇنداق كۈنلەرنىڭ بىرىىدە گۈلنۇر ھەدىنىڭ بويىدا قالدى بۇ خىل خۇشاللىق نۇردۇن ئاكا ۋە ئائىلىسىنى شۇنداق شاتلاندۇردى.گۈلنۇرھەدىمۇ بۇنىڭدىن بەكلا خۇشال ئىدى،كۈنلەر شۇنداق ئۆتۈپ گۈلنۇر ھەدىنىڭ بوشىنىدىغان كۈنى يىتىپ كەلدى دوختۇرلار جىددى قۇتقۇزۇش ئۆيىگە كىرىپ كەتكىنىچە خېلى ئۇزۇنغىچە چىقمىدى.بىر ئازدىن كىيىن بىر دوختۇر تۇغۇت بۆلۈمىدىن چىقىپ نۇردۇنغا شەرەت قىلىپ ئىشخانىسىغا كىرىپ كەتتى.
-بۇ قەغەزگە قول قۇيۇڭ...
-نىمە قەغەز بۇ؟ نىمە قىلىمەن ؟مەن خەنزۇچە خەت ئوقالمايمەن ،،،
-ھازىر ئايالىڭىزنىڭ ئەھۋالى ئېغىر،،،پەقەت ئانا -بالا ئىككىسىنىڭ بىرىنى قۇتۇلدۇرۇپ قالغىلى بولىدۇ كىمنى قۇتۇلدۇرايلى؟
-بولدى ئايالىمنى قۇتۇلدۇرۇپ قېلىڭلار،،،بالا دىگەننى كىيىن تۇغۇپ  بەرسىمۇ بولىدۇ...
نۇردۇن شۇنداق دەپ دوختۇرنىڭ قۇلىدىكى قەغەزگە ئەمدىلا ئىمزا قوياي دەپ تۇرىشىغا بىر سىستىرا قىز يۈگۈرگىنىچە دوختۇرنىڭ ئىشخانىسىغا كىردى،
-دوختۇر بىمارنىڭ ئەھۋالى ئېغىرلاپ كەتتى...بالىنى قۇتۇلدۇرغىلى بولىدۇ ئەمما ئاپىسىغا ئامال قىلالمايمىزكەن،،،قانداق قىلىمىز؟
دوختۇر ئايالمۇ يۈگۈرۈپ تۇغۇت بۆلۈمىگە كىرىپ كەتتى..نۇردۇن تېخىمۇ تىتىلدىغىنىچە زال ئىچىدە بىردەم ئۇياققا بىردەم بۇياققا مېڭىپ دىگەندەك شۇنچە قۇرقۇنۇش ئىچىدە چۆرگىلەپ يۈرەتتى،بىر ئازدىن كىيىن دوختۇر ئايال بىشىنى تۆۋەن سېلىپ ئىغىر ئۇھ تارتىپ قۇيۇپ تۇغۇت بۆلۈمىدىن چىقىپ كەلدى،
-دوختۇر ئايالىمنىڭ ئەھۋالى قانداقراق؟بالىنىڭچۇ؟
-كەچۈرۈڭ ،،قولىمىزدىن كەلگەننىڭ ھەممىنى قىلدۇق بىراق بىزدىمۇ ئامال يۇق ...سىزنىڭ بالىڭىز قۇتۇلۇپ قالىدى ئەمما ئايالىڭىز قۇتقۇزۇش ئۈنۈم بەرمەي ئۆلۈپ كەتتى،،،يەنە كىلىپ ئايالىڭىز بىر قىز بىر ئوغۇل تۇغدى.
-نىمە ئۆلۈپ كەتتى دەمسىز؟ئاھ...خۇدا..بۇ ماڭا نىمە كەلگۈلۈك ئەمدى...
نۇردۇن كۆز ئالدىدىكى بۇ رىئاللىقىنى قوبۇل قىلماي ئىلاجى بولمىغان ئىدى يەنە كىلىپ فىرۇزە شۇنداقلا گۈلنۇرنىڭ قوشكىزەك باللىرى بىر نۇردۇننىڭ تەربىيسىگە قالغان ئىدى
3 بالا بىلەن قالغان نۇردۇنننىڭ كاللىسىنى تۈگىمەس خىياللار چىرمىۋالغان ئىدى.ئۇ كۆز ئالدىدىكى 3 نارسىدە بالىغا ئىچ ئاغرىتىپ قارايتى يەنە كىلىپ فىرۇزە خەقنىڭ تاشلاندۇق بالىسى بولغانلىقى ئۈچۈن يىڭىلا 2 بۇۋاقنىڭ ئاتىسى بولغان نۇردۇن ئۈچۈن فىرۇزەنى بېقىش ئېغىر كىلىپمۇ قالغان ئىدى
گۈلنۇرنىڭ ئاخىرەتلىك ئىشلىرى ئاخىرلىشىپ بولغاندىن كىيىن نۇردۇن بۇ 3 بالىغا دادا ھەم ئاپا بولمىسا بولمايتى ئۇ قىسقىغىنە 5 ئاينى شۇنچە تەستە ئۆتكۈزدى چۈنكى ئۇنىڭغا بۇ 3 بالىنىڭ ھالىدىن خەۋەر ئېلىش ئېغىر كىلپ قالغان ئىدى.
نۇردۇن قىزىغا سەبىھە ئوغلىغا سەردار دەپ ئىسىمنى قويۇپ بولغاندىن كىيىن باللىرىنى ئائىلىگە بالا باققۇچى تەكلىپ قىلىپ باقتۇردى  بىراق بۈگۈنكى بۇ كۈنگە كەلگەندە فىرۇزەنى بېقىش ئۇنىڭغا ئارتۇقچە يۈك ئورنىدا بولۇپ قالغان ئدى ،نۇردۇن بۇنى ئاۋارىچىلىق دەپ قارايتى ،،
نۇردۇن بىر نەچچە كۈنلەپ فىرۇزەنى قانداق قىلىپ بۇ ئۆيدىن چىقىرۋىتىش ئۈستىدە ياخشى ئۇيلاندى بىراق ئۇ ئويلىغان بۇ ئامال فىرۇزەنىڭ پۈتكۈل بىر ھاياتىنى شۇنداقلا پۈتكۈل بىر ھاياتىنى ۋەيران قىلىپ ئۇنىڭ ساراڭ قىز دىگەن نامغا ئىگە بولۇپ قېلىشىغا سەۋەپچى بولغان ئىدى.

ياخشىلىق قىلغان ئىكەن ئاخىرىغىچە قىلسۇن ،بولامدۇ.
