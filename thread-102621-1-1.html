                                                                                                                                                                                         ئەسسالامۇئەلەيكۇم قېرىنداشلار مەن سىلەرگە يېقىندا راس بولغان ئاچچىق ھەقېقەتتىن بىرنى دەپ بىرەي؛ مېنىڭ بىر ئاچام بولۇپ 17 يىل بۇرۇن قاغىلىق ناھيە ئورمانچىلىق ئېدارىسىنىڭ كەينىدە ئولتۇرۇشلۇق ئابدىقەيۇم سۇلتان دېگەن كىشى بىلەن توي قىلغان ئېدى؛ مەزكۇر ئابدىقەيۇم سۇلتان ئاچام بىلەن توي قىلىپ 8 يىل بولغاندا يەنى 2004-يىلنىڭ ئاخىرلىرى سېياسېغا چېتلىپ قېلىش سەۋەپلىك مۇددەتسىز قاماق جازاسىغا ھۆكۈم قىلىنغان ھازىر ئۈرۈمچى شەھەرلىك 4-تۇرمىدە جازا ئۆتەۋاتىدۇ؛بۇلارنىڭ ئۈچ بالىسى بار ئاچامنىڭ يولدىشى تۈرمىگە كىرىپ كەتكەندىن كىيىن ھازىرغىچە يولدىشىنىڭ ئاتا-ئانىسىنىڭ ئۆيىدە تۇرىۋاتقان ئېدى؛ يېقىندا يولدىشىنىڭ ئېنسى كىچىككىنە بىر خاتالىقى ئۈچۈن ئاچامنىڭ بالىسىنى ئۇرۇپ كەتكەن ئاچام بالىسىغا چېدىماي ئارلاشقان ئاندىن كىيىن ئانىسى بىلەن بالىسىنى قوشۇپ راسا دۇمبالىغان؛ ئۇرۇپ ھاردۇق يەتكەندىن كىيىن مېنىڭ قېشىىمغا چىقىپتۇ ھىلقى دەيۈز ئۆيگە كىرىپ ئاچىڭىزنى ئېلىپ چىقىپ كېتىڭ دەپ بىر ئىش بولغانلىقىنى سەزدىم كىرمىدىم؛ چۈنكى كىرسەم جودا چوڭىيىپ كېتشى تۇرغانلا گەپ مەنمۇ ئوغۇل بالا بولغاندىن كىيىن قاراپ تۇرمايمەندە ئاچامنىڭ تاياق يېيشىگە؛چوڭلارنى كىرگۈزدۈم ئاچامنى بالىلىرى بىلەن ئېلىپ يېنىپ چىقىپتۇ ئارلىقتا ئىككى ئاي ئەتىراپىدا ئۆتكەندىن كىيىن ئادەم ئەۋەتىپتۇ ئۈچ بالىنى بەرسۇن دەپ يولدىشىنىڭ ئۆيىگە بالىلىرى پاتىدىكەنۇ ئانىسى پاتمايدىكەن قوشۇلمىدۇق سەۋەبى بىرىنچىدىن دادىسىدىن ئايرىلغان شۇ نارسىدىلەرنى ئانىسىدىنمۇ ئايرىغىمىز كەلمىدى؛ ئىككىنچىدىن ياخشى ئانا پەرزەنىتلىرى ئۈچۈن بىر مەكتەپتۇر؛ ئۇلارمۇ بوش كەلمىدى سەتلەشتۇق تىللاشتۇق يەنىلا ھەل بولمىدى؛ ئاچام 17 يىل ئولتۇرۇپتۇ شۇ ئۆيدە بىر تال يىڭنە چاغلىق بىر نەرسە بەرمىسىمۇ مەيلى پەقەت ئۈچ بالىسىنى ئېلىپ يېنىپ چىقسۇن دېسەك يەنىلا قايىل بولمىدى؛ ئۇنداق بولسا ئاچام بالىلىرى بىلەن سىلەرنىڭ قېشىڭلاردا بىللە تۇرسۇن دادىسىدىن ئايرىلغان بۇ نارسىدلەرنى ئانىسىدىنمۇ ئايرىمايلى دېسەم ئاچاڭنى باقالمىساڭ ھەيدىۋەت؛ ئەرگە بەر دېگەنگە ئوخشاش گەپلەرنى قىلىپ ھاقارەت قىلدى؛ يەنە تېخى ئاچامنىڭ يولدىشى تۈرمىگە كىرىپ كەتكەندىن بۇيان بۇ ئۈچ بالا بىلەن ئانىسىنى بېقىپ بولالماي نۇرغۇن قەرىزگە بوغۇلۇپ كېتىپتىمىش بۇنداق يالغان گەپ قىلمىسىمۇ بىز بىر نەرسە تەلەپ قىلمىدۇققۇ؛ تونۇيدىغانلار بىلىسىلەر ئاكىسنىڭ مەبلىغى بىلەن ئاكىسى ئېچىپ بەرگەن ئاشلىق-ماي دۇكىنىدا ناھيىلىك پىلانلىق تۇغۇتنىڭ قارشى تەرىپىدە جېنىنى بېقىۋاتىدۇ ئانچە بەك تۇرمىشى ناچار ئادەملەرمۇ ئەمەس قول ئېلىكىدە بار تۇرۇپ قەرزىمىز بار دەپ تۇرغىنىدىن قارىغاندا بالىلارنى بۇلارغا بەرسەك دادىسىنى تىلەمچىلىككە سالىدىغان ئۈچ ئوغۇلنىڭ كېنوسىدەك بۇ نارسىدلەرنى تىلەمچىلىككە ياكى شۇنىڭغا ئوخشاش پۇل تاپىدىغان ئىشلارغا سېلىشتىن يانمىغىدەك؛ جەمىيتىمىزدە ئوقۇش يىشىدىكى نارسىدىلەرنى ئالداپ ئىنسان قېلىپىدىن چىققان ئادەم بېدىكلىرىگە سېتۋىتىدىغان نائەھلىلەرمۇ پۇل ئۈچۈن قېلىدۇ ئەمەسمۇ؛ كۆرەش داۋاملىشۋاتىدۇ بالىلار مېنىڭ قېشىمدا بۇ ئىشنى تورغا يوللاپ نېمە مەقسىتى باردۇ دەپ قېلشىڭىز مۇمكىن شۇ كىچىككىنە نارسېدلەرنىڭ بېشىغا كەلگەن قىسمەتلەرگە ئىچىم ئاغرىپ يوللاپ قويدۇم مېنىڭ ئورنۇمدا سىز بولسىڭىز قانداق قىلاتتىڭىز؛

بۇ تېمىنى يېزىشمدىكى مەقسەت قىرىنداشلارنىڭ توغىرا مەسلەھەت بېرىشىنى سورايمەن مۇشۇ نارسىدلەرنى ئانىسىدىن ئايرىۋەتمەي مىللىتىمىز ئۈچۈن تۆھپە قوشالىغۇدەك تەربىيلەپ قاتارغا قوشۇش ئۈچۈن مېنىڭ تەربىيەمگە ئۆتۈشىنى تەلەپ قىلىشىم ئورۇنلۇقمۇ ؟
قېرىنداشلار مۇشىنىڭغا ئوخشاش يولدىشى تۇرمىدا ياكى ئۆلۈپ كەتكەن ئائىلىسىدىكىلەر تەرپىدىن خورلىنۋاتقان ۋە چەتكە قېقلىش سەۋەپلىك ناتوغىرا يولغا مېڭىپ قېلۋاتقان ئانىلارمۇ ئاز ئەمەس بۇ ئانىلارنىڭ قولىدا چوڭ بولغان كەلگۈسىنىڭ ئېگلىرى مىۋىسىز دەرەختۇر؛ بۇنىڭدا ماۋۇ تېمىدا تىلغا ئېلىنغان ئاياللارنىڭ دۈشمىنى ئاياللارنى دۇمبالاشتىن باشقىنى بىلمەيدىغان ئاۋۇ ھەزلەككە ئوخشاش تۆت پۇتلۇق ئەرلەرنىڭ بىر كىشىلىك تۆھپىسى بار دەپ قارايمەن؛
