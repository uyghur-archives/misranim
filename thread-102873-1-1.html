ئەسسالامۇ ئەلەيكۇم مىسرانىم مۇنبىرىدىكى مۇنبەرداشلار،سىلەرگە ئاللاھنىڭ سالىمى  بولسۇن.
      ئاللاھنىڭ ھىممىتى بىلەن بۇ دۇنياغا كۆز ئاچقاندىن بىرى،22قىتىم بۇغداي پىشىغى بولۇپ،22قىتىم غازاڭ سارغىيىپ،يەنە 22قىتىم قەھرىتان سوغۇقلاردا يەر توڭلاپتۇ.مانا بۈگۈن ئاللاھنىڭ ھىدايىتى بىلەن مۇنۇ قۇرلارنى يىزىپ ئولتۇرۇپتىمەن.(ئاللاھنىڭ مۇشۇنداق كۈنلەرنى ئاتا قىلغىنىغا مىڭ مەرتىۋە شۈكرى)
     ھەممىمىزگە ئايان بولغىنىدەك، مۇكەممەللىك پەقەت ئاللاھقىلا خاس بولۇپ، ئىنسان مۇكەممەللىكتىن يىراق ئىكەن.گەرچە نۇرغۇن قىتىملاپ تېرىشساقمۇ يەنىلا خاتالىق ئۆتكۈزۈشتىن مۇستەسنا بولغىلى بولمايدىكەن.بۇنداق دېگىنىم ھەرگىزمۇ بىز بەرىبىر مۇكەممەل بولمىغاندىكىن بۇنداق خاتالىقلارنىڭ كۆرۈلىشى،بىلىپ بىلمەي يامان ئىشلارنى قىلىپ قويىشىمىزنى كەڭ قورساقلىق بىلەن نەزەردىن ساقىت قىلىۋەتسەك بولىدىكەن دېگەنلىك ئەمەس،ياشىغاندىكىن مۇكەممەللىكنى نىشانلاپ ،ئاللاھنىڭ ئەھكاملىرىنى تولۇق بەجا كەلتۈرۈپ ياشىشىمىز كېرەك ئەلۋەتتە.مانا مەنمۇ ئاق سۈت ئەمگەن بەندە، تۇغۇلۇپ ئەقلىمگە كىلىپ نۇرغۇنلىغان خاتالىقلارنى بەزىدە بىلىپ بەزىدە بىلمەي تۇرۇپ ئۆتكۈزدۈم، بەزىدە قىلغانلىرىمغا ئۆكۈندۈم،پۇشايمان قىلدىم.بەزىدە قىلغانلىرىمدىن يىرگىنىپ كەتتىم،شۇندىمۇ ئۈمۈدىمنى ئۈزمەي توغرا يوللارنى ئىزدەپ ياشاپ كەلدىم.
     ئازمايدىغان ئىنسان يوق ئوخشايدۇ.بەزىلەر ئازىدىكەنيۇ يەنىلا ئىمانى غالىپ كىلىپ ،ئەقلى- ھۇشىنى يىغىپ توغرا يوللادا ماڭىدىكەن.يەنە بەزىلەر ئازىدىكەن،تىڭىرقاش ئىچىدە ئازىدىكەن،توغرىمىدۇ خاتامىدۇ دېگەن ئويلادا ئازىدىكەن،بۇلار ئاللاھنىڭ كالامىنى ئادا قىلسا،ئىمانى چىڭىسا يەنە توغرا يولغا قايتىدىكەن.يەنە بەزىلەر ئازىدىكەن،ئازغىنىنى بىلمەي ئازىدىكەن،توغرىمىدۇ خاتامىدۇ دېگەنلەنى ئويلىماستىن ئازىدىكەن.بۇنداقلار قانداق ۋاقىتتا توغرا يولىنى تاپىدۇ ،بۇنىسى ماڭا نامەلۇم،شۇنداقتىمۇ بۇنداقلارنىڭ توغرا يول تىپىۋىلىشىنى ئۈمۈد قىلىمەن ئەلۋەتتە.
    جەمىيىتىمىز تەرەققىي قىلىپ،شەھەرلىشىش ئەۋج ئالغان بۈگۈنكى كۈندە ،كىشىلىرىمىزنىڭ قىممەت ئۆلچىمىدە ،كىشىلىك قارىشىدا ھەتتا ئۆزىگە بولغان تونۇشىدا ئۆزگىرىش بولماقتا.بۇنداق ئۆزگىرىشلەر بەزىدە ياخشى ئاقىۋەتلەرنى كەلتۈرۈپ چىقارسىمۇ،يەنە كۆپ ئەھۋاللاردا كىشىلەرنى ئۆز يولىدىن چىقارماقتا.بۇ يەردە مېنىڭ زۇلپىقار بارات ئۆزباشنىڭ «بىز قانداق چاكىنىلىشىمىز»دېگەن بىر ئەسىرىنى ئوتتۇرىغا قويغۇم كىلىۋاتىدۇ.بۇ ئەسەردە زۇلپىقار بارات ئۆزباش بىزنىڭ ئۇيغۇرلا جەمىيىتىمىزدە بۇلۇنىۋاتقان بىر قىسىم ھادىسىلەنى مىسال ئىلىپ،ئۇلارنىڭ قانداق كىلىپ چىقىش ۋە ئاقىۋەتلىرىنى ناھايتىمۇ قايىل قىلارلىق قىلىپ كۆرسىىتپ بىرىپتۇ.بۇ ماقالىدە بىر ئوقۇم تىلغا ئىلىنغان،ئۇ بولسىمۇ «جەمىئەتلىشىۋاتقان چاكىنىلىق.»ئوقۇپ باققان قېرىنداشلىرىمىز بەلكىم بىلىدۇ،جەمىئەتلىشىۋاتقان چاكىنىلىق دېگىنىمىز بىر ئىجتىمائى توپتا بىردەك ھىمايە قىلىنىدىغان،ئورتاقلىق تۇيغۇسى ،مەسئۇلىيەت ئېڭى پەيدا قىلىدىغان قىممەت قاراشلىرى ،ئىجتىمائى مىزانلارغا خىلاپ ھالدا ئومۇملاشقان ،ئەمما ئەخلاقسىزلىق ھىساپلانمايدىغان ،جىنايەت ئۆلچىمىگىمۇ توشمايدىغان،ۋاقىتنىڭ ئۆتۈشى بىلەن شۇ ئىجتىمائى توپ ئەزالىرى ئارىسىدا ھەيرانلىق ھىس قىلىنمايدىغان ئىجتىمائى بىنۇرماللىق.(شۇ ماقالىدىن ئىلىندى)
    ئۇنداقتا بۇنداق بىنۇرماللىقنىڭ كىلىپ چىقىشىنىڭ سەۋەبى نىمە دېگەندە ،زۇلپىقار بارات ئۆزباش شەھەرلىشىش ،زامانىۋىلىشىش،يىڭى خىرىس،يىڭى خىرىسقا قارىتا ئەل بولۇش ياكى ئۇنى ئۆزگەرتىش قاتارلىق نوقتىلاردىن تۇرۇپ تەھلىل قىلغان.يىڭى خىرىسقا دۇچ كەلگەندە،بىزنىڭ مەدەنىيىتىمىز،بەزى ئۆرپ-ئادەتلىرىمىز يېڭى ئىقىپ كىرگەن مەدەنىيەتنىڭ ھۇجۇمىغا ئۇچرايدۇ.بۇنداق ۋاقىتتا بەزى كىشىلەر سىرتتىن كەلگەن مەدەنىيەتنى ئىنكار قىلىپ،ئۆز مەدەنىيىتىنى ساقلاپ قىلىشقا تېرىشىدۇ ھەم ساقلاپ قالىدۇ.يەنە بەزىلەر ئۆزگەرتىشنى ئويلىماستىنلا ئەل بولىدۇ،بۇنداق قىلمىشلار دەككىلەنمىسە باشقا ئىجتىمائى ئەزالار تەرىپىدىن رەت قىلىنمىسا بۇنداق ئەھۋاللار ئەۋج ئىلىپ،جەمىئەتلىشىۋاتقان چاكىنىلقنى كەلتۈرۈپ چىقىدۇ،يەنى بىزنىڭ ئېسىل ئۆرپ ئادەتلىرىمىز ،ئۇزۇن تارىخقا ئىگە مەدەنىيەتلىرىمىز پاسسىپ ئورۇنغا چۈشۈپ قالىدۇ،يەنە نۇرغۇن مۇسۇلمانلار قائىدىسىگە ئۇيغۇن بولمىغان ئىشلار يۈز بىرىدۇ.بىر كىچىكىدىن باشلاپ خەنزۇ مەكتەپتە ئوقۇغان قىزنىڭ، كىيىنچە توي قىلغاندىمۇ شۇنداق خەنزۇ مەكتەپتە ئوقۇپ چوڭ بولغان ئۇيغۇر يىگىتىنى تاللايدىغانلىقى،بالىسى بولسىمۇ كىچىكىدىن باشلاپ خەنزۇ مەكتەپتە ئوقۇتىدىغانلىقى،يەنە بىر قىزنىڭ ئورىنىپ ئەمەس،ياغلىق چىگىشنىمۇ خالىمايدىغانلىقى،ئۆزىنىڭ مۇشۇنداق چىچىنى« ئاپتاپقا سىلىپ» ئەركىن يۈرۈشكە كۆنۈپ قالغانلىقىدەك مىساللار بۇنىڭ تىپىك مىسالى دەپ ئويلايمەن.(ئۆز كۆزۈم بىلە كۆرۈپ ئۆز قۇلىقىم بىلەن ئاڭلىغان)
     سەن بۇ ماۋزۇغا ياشىغىم بار دەپ،بېشىغا ئىزىش توغۇرلۇق يىزىپ،يەنە تۇرۇپ جەمىئەتتىن چۈشۈپسەن،بۇلارنىڭ قانداق باغلىنىشلىقى بار،تاغدىن-باغدىن سۆزلەۋاتامسەن قانداق دەپ سورىغىڭلا كىلىۋاتامدۇ قانداق؟
    يۇقىرىقىدەك ئۆز يولىدىن ئىزىش ،يامان يوللاغا مىڭىپ قىلىش دېگەندەكلەرنى جەمىئەتتىن ئايرىپ قارىساق بولمايدۇ ئەلۋەتتە،يۇقىرىدا سۆزلەپ ئېيتىلغاندەك،جەمىئىتىمىزنىڭ بۈگۈنكىدەك ھالغا يىتىشى،نۇرغۇن كىشىلەرنىڭ بىلىپ بىلمەي خاتا يولغا مىڭىپ قىلىشىغا سەۋەپچى بولغان،ئۇنداقتا خاتالىقىمىزنىڭ ھەممىسىنى جەمىئەتكە ئارىتپ قويامدۇق ،خاتالىقنى ئۆتكەزگۈچى بىز تۇرساق دىمەكچىمۇ؟جەمىئەتنىڭ مۇشۇنداق ھالغا يىتىشى جەمىئەتتە ياشاۋاتقان ئىجتىمائى ئەزالارنىڭ« تۆھپىسى» ئەمەسمۇ؟شۇڭا نۇرغۇن ئىشلارنى بىز قىلىۋاتىمىز،جەمىئەتتە ئەكىس ئىتىۋاتىدۇ.مەلۇم بىر كىشى خاتا ئىشتىن بىرنى قىلىدۇ،ئاق -قارىنى پەرق ئىتىپ بولالمىغان يەنە بىر نەچچە كىشى ئەگىشىدۇ،شۇنىڭ بىلەن جەمىئىتىمىزدىكى ناچار ئىللەتلەر كۆپىيىدۇ.
    مانا مەنمۇ مۇشۇنداق جەمىئەتتە ياشاۋاتقان ،مۇشۇ جەمىئەتنىڭ بەزى چايقاشلىرىدا بىلىپ بىلمەي چايقىلىپ كىتىۋاتقان،بەزىدە بۇران چاپقۇنلاردا يىقىلىپ،ئورنۇمدىن تۇرۇشقا ئۇرۇنىۋاتقان،خاتا ئىش قىلىپ،يەنە تۆۋبە قىلىپ،گۇناھىمنى تىلەپ،ياخشى يوللارنى تىلەپ كىلۋاتقان ئاددى بىر بەندە.ئۆزۈم نۇرغۇن ئىشلارنى ۋايىغا يەتكۈزۈپ قىلىپ بولالمىدىم،مۇكەممەللىكتىن يىراقمەن،شۇڭا بۈگۈن بۇنداق بىر تېمىنى يىزىپ باشقىلارنى ئەيىبلەشكە ئاجىزمەن.مۇشۇنداق بىر جەمىئەتنى ئاز تولا بولسىمۇ تونۇپ يىتەلىگىنىمدىن خۇرسەنمەن ھەمدە توغرا يوللارنى تىپىپ مېڭىشقا كۈچۈمنىڭ يىتىشىچە تېرىشىمەن.
   ئاخىرىدا گەرچە نۇرغۇن ئشىلارنى توغرا قىلىپ بولالمىغان بولساممۇ،ياشىغىم بار،بۇندىن كىيىن توغرا ئىشلار بىلەن مەشغۇل بولۇپ ياشىغىم بار.ياشىغىم بار،جەمىئەتنىڭ بۇران چاپقۇنلىرىدا تىك تۇرۇپ،ئۆزۈمنى بىلىپ ياشىغىم بار.ياشىىغم بار،مەھمۇدنىڭ(دادام)نىڭ ئوغلىدەك ياشىغىم بار.ياشىىغم بار،ئۇيغۇر ئوغلانىدەك ياشىغىم بار.ياشىغىم بار،بۇ دۇنيادا ئەڭ ئەۋۋەل ئۇ دۇنيالىقىم ئۈچۈن ياشىغىم بار......
