ئەسسالامۇ ئەلەيكۇم ئەھلى ئىسلام مۇسۇلمان بۇرادەرلەر،مەن بۇ تېمىنى يوللۇدۇم.مەنمۇ  بىر ئۇيغۇر مۇسۇلماننىڭ ئەۋلادى.مېنىڭ بۇ تېمىنى يوللاشتىكى مەقسىدىم قانداقتۇر ئۆزىمىزنىڭ ئىتنىك ئۆرپ-ئادەتلىرىدىن بىرى بولغان مىللىتىمنىڭ قائىدە-يۇسۇنلىرىنى بۇزۇش ياكى تەپرىقىچىلىك قىلىش ئەمەس.قەبرىستانغا مەنمۇ چىقىمەن.لىكىن قەبرىستانلىققا چىقىشتىكى مەقسەت نېمە؟پەقەت ئىبرەت ئېلىش،ئۆلۈمنى ئەسلەش.بەزىلەر ئۇ قەبرىستانلىققا چىققانلار ئۆلۈكلەردىن بىر نەرسە تىلەپتىمۇ دەپ سوراپتۇ.ئالدى بىلەن شۇلاردىن سوراپ باقاي سىز بىر ئەھلى مۇسۇلمان .ئاللاھنىڭ كۆرۈپ تۇرغۇچى،ئاڭلاپ تۇرغۇچى ئىكەنلىكىنى بىلىسىز. ئەجىبا مەخسۇس بىر كۈننى بەلگىلەپ شۇ ئورۇننىڭ ئۆزىگە بېرىپ دۇئا-تەكبىر قىلىشىڭىزنىڭ ئورنى بارمۇ؟مەن بارماڭ دىمەيمەن لىكىن «پەيغەمبىرىمىزنىڭ ئىشلارنىڭ ئەڭ ياخشىسى ئوتتۇرا ھال بولغىنى»دىگەندەك بارغاندىمۇ ئىسلام پىرىنسىپلىرىدىن چەتنەپ كەتمەڭ دىمەكچى.مەسىلەن ئېسىلغان ئاق لاتىلار،تېرىلەر قاتارلىقلار.مانا مەن يوللىغان تېمىدىكى ئەڭ بەك كۆزدە تۇتقان ئىش مۇشۇ.مېنى تىللىغانلار چىقتى،كەمسىتكەنلەر چىقتى.ئەمما بىرەر ئايەت-ھەدىسلەر ئارقىلىق مۇشۇ ئىشنىڭ ئىسلامنىڭ نەرىدە بارلىقىنى،تۇغ-ئەلەملەرنى ،تېرىلەرنى ئېسىشنىڭ سەۋەبىنى،ساۋابىنى دەپ چۈشەندۈرىدىغانلار چىقمىدى.ئەگەر ئېسىشقا توغرا كەلسە پەيغەمبىرىمىزنىڭ ،ساھابىلەرنىڭ قەبرىلىرى ئېسىشقا ئۇلاردىنمۇ بەك ھەقلىققۇ دەيمەن.نېمىشقا ئۇلارغا ئېسىلمايدۇ؟ھە دسىلا ئۇلار سەنچىلىك بىلمەمتى،ئەجىبا ئاتا-بوۋىلىرىىمىز بۇرۇندىن باشلاپ قىلىپ كەپتىكەنغۇ؟دىگەندەك مەزمۇندىكى سۆزلەرلا كۆپ سالماقنى ئىگىلىدى.خاتالاشمىسام بىزنىڭ ئىسلام كىلىشتىن ئىلگىركى ئاتا-بوۋىلىرىمىز بۇددا،شامان دىنغا ئىتىقاد قىپتىكەنغۇ.ئۇلار ئۆزلىرىنىڭ قىلغانلىرىنى خاتا دىسە شۇ باتىل دىنلارغا ئىتىقاد قىلار بولغىيمىدى؟ئويلاپ باقايلى قېرىنداشلار،تۆت چاھاريارنىڭ بىرى بولغان ئۇلۇغ ساھابە ئۆمەر رەزىيەللاھۇ ئەنھۇ خەلىپىلىك دەۋرىدە «ئەگەردە خاتالاشسام مېنى تۈزىتىپ قويۇڭلار »دىگىنىدە سەلمان پارىسى رەزىيەللاھۇ ئەنھۇ خاتالاشساڭ قىلىچ بىلەن تۈزۈتىپ قويىمىز دىگەنىكەن.شۇنچىلىك تەقۋادار ئۇلۇغ كىشىلىرىمىزمۇ مەن ھەرگىز خاتالاشمايمەن دىيەلمىگەن .توغرا مەن خاتالاشقاندىمەن.ئەمما بىرەر قېرىندىشىمىزنىڭ قولىدىن سىنىڭ مۇشۇ ئىشىڭ ئايەت-ھەدىسنىڭ مۇنۇ يىرىگە زىت دەپ تۇرۇپ تۈزىتىش بەرگەندىن بىرەرسىمۇ چىقمىدى.چىقتى دىگەندىمۇ تىل-ئاھانەت قىلىشلار چىقتى.مەن بۇنى ئۇلۇغ ئىگەم ئاللاھقا ئامانەت قويۇپ قويدۇم.تېما مەزمۇنىدىن شۇ كىشىلەرنى مەسىخىرە قىلدى دېگۈچىلەر خاتا چۈشىنىپ قالماڭلار مەن ئەسلا ئاكا-ئۇكا،ئاچا-سىڭىل دىنى قېرىنداشلىرىمنى مەسخىرە قىلغىنىم يوق.ئىمام ئاسىم ئسلامغا تۆھپە قوشقان،ئىسلام ئۈچۈن جان پىدا قىلغان بىر ئىسلام جەڭچىسى.لىكىن بۇ كشىنىڭ كىلىپ چىقىشى،ئىسلام يولىدا ئېلىپ بارغان ئىش-پائالىيەتلىرى توغرىسىدا بىلىدىغانلار قانچىلىك؟بۇنى بىز ھەممىمىز ئۆزىمىزدىن سوراپ باقساق بولىدۇ.ئىسلام دىنى شىنجاڭغا قاچان كىردى؟ئىمام ئاسىم قاچان،قەيەردە ياشىدى؟بۇنى چوقۇم شۇ تۇغ-ئەلەملەرنى ئاسقۇچىلار بىلىشى مۇمكىن.مەنمۇ سىلەرنىڭ قېرىندىشىڭلار،مىللەتدىشىڭلار،بۇ سوئالىمغا تىل ياغدۇرىۋەتمەي.بىر چۈشەنچە بەرسەڭلار. دىن-ئىسلام ئۈچۈن  يوللانغان  بۇتېمىنىڭ ئۆز قېرىنداشلىرىم تەرىپىدىن يولسىز تىل-ئاھانەتلەرگە قىلىشىدىن ،  تېمىنىڭ رەددىيەسىنىڭ قايتا بىر تېما بولۇپ يوللانغانلىقىدىن ئەپسۇسلانغان بولساممۇ،يۇقىردىكى مېنىڭ قاپاق كاللامدىن ئۆتمىگەن ئىسلامنىڭ ھېچقانداق ئەمىر-پەرمانلىرىدا قەيت قىلىنمىغان ئىشلاردىن سورىغان سوئالىمغا  بىر جاۋاپ بەرگەن بولساڭلار ،مەن ۋە يەنە ماڭا ئوخشاش بىلمىگەن قېرىنداشلىرىمىز بىر چۈشىنىۋالسا.يەنە تەكىتلەيمەن تۇغ-ئەلەملەرنى ئېسىشنىڭ سەۋەبى نېمە؟«ئىمام ئاسىم»دىگەن كىم؟ پاكىت-ئىسپاتلار بىلەن بولسۇن.ئۆز خىيالى بويىچە لىكسىيە سۆزلەيدىغانلار ئىنكاس يازمىساڭلارمۇ بولىدۇ.

تېخى بەزىلەر باركى قايسى دىنغا ئىتىقاد قىلىسەن دىسە ، ئىسلام دىنىغا دەيدۇ ، نىمە ئۈچۈن بۇ دىنغا ئىشىنىسەن دىسە ، ئاتا -بۇۋىلىرىمىزمۇ مۇشۇ دىنغا ئىشىنىپتىكەن دىيىشىدۇ ، بەزىلەرنىڭ كاللىسى بەكلا ئاددى ، بەزى ئىشلارنىڭ ماھىيىتىنى چوڭقۇرلاپ ئويلاپ باقمايدۇ ، كاللا دىگەننى ئاللا نىمە دەپ ياراتتى ، ئويلۇنۇڭ، تەپەككۈر قىلىڭ ، ئۈگۈنۈڭ دەپ ياراتقاندۇ . بۇ تىمىمۇ شۇ مازار تاۋاپ قىلىشنى توغرا دىيىشنى ياقلايدىكەن ، ھەتتا 5-6-ئايلاردا تاۋاپ قىلىدىغان ئادەمنىڭ  بەك جىقلىقىنى دىيىشىپ كىتىپتۇ بەزىلەر ، ئۇلار نىمە مەقسەتتە تاۋاپ قىلغىنى كەلگىنى بىلەن كارى يوق ، ئەڭ مۇھىمى ئاۋۇ تۇغلارغا بىر چۈشەنچە بەرسە بولاتتى .
