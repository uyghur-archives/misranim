ئەسسالامۇئەلەيكۇم  قىرىنداشلار  مۇشۇ بىر نەچچە يىلدىن بۇيان جەميىتىمىزدىكى بىر بىرسىنى  قۇيۇش ،  ئالداش ، ماللىرىنى ئوغۇرلاش ، ئاتا ئانىسىنى  قاقشىتىش ، راس مالنى كۆرسىتىپ يالغاننى سېتىش ، ئوغۇللا قىزلانى ئالداش  ، قىزلا ئۆچ ئېلىش مەقسىتىدا ئوغۇللانى ئالداش ياكى ئۆزىدىكى بىر قىسىم كېسەللىك بىلەن  ئوغۇللاغا يوقتۇرۇش قاتارلىق بىرقىسىم جەمىيەتتىن ،ئۆيدىن ، ھەتتا دوسلىرىدىن ئوچ ئېلىش پىخسىكىسى شەكىنلىنىپ ئۆز مىللىتىگە ئۆچ بۇلۇش ، ئۆز مىللىتىدىن ئۆچ ئېلىش ، يەنە بىر بىر سىگە ئېشەنمەسلىك ، ئەتراپىدىكىلە يامانلىق قىلسا قىلىدىكى ياخشىلىقنى ئويلاپمۇ قۇيماسلىق  قاتارلىق يامان ئېللەتلەر تارقالماقتا . گەپنىڭ نەق ئۆزىنى قىلسام كىشىلەر ئارا  كۇيۇمچانلىق ، مېھرىبانلىق ، ئىتىپاقلىق ، بىرلىشىش ، ئۆملۇك ،بىر نىيەت بىر مەخسەتتە بولماسلىق  قاتارلىق ئېشلار يۇز بەرمەكتە ،بۇنداق دېسەم ھەممىلا كىشى ئۇنداق ئەمەس پەقەت ئاز بىر قىسىم كىشلىرىمىز يەنىلا شۇنداق ، ئەلۋەتتە كۇيۇمچان ئاتىلىرىمىز مېھرىبان ئانىلىرىمىزمۇ يوق ئەمەس ئەلۋەتتە .
    ئەمدى ئۇيلىغانلىرىم كىچىك تىپتىكى بىر فىلىم ئېشلىسەك  بۇ فىلىمدا بىر ياش بالا كومپىيۇتېر ئوغىرلاپ قاچىدۇ بۇنى بايقىغان كىشلەر دەرھال ئۇنى قۇغلايدۇ ۋە بىر ساقچىخانىنىڭ ئالدىدا تۇتۇۋالىدۇ ھەم بىر نەچچىسى ئۆزىنى تۇتۇۋالاماي ئۇنى ئۇرىدۇ ، ۋاراڭ چۇرۇڭنى ئاڭلىغان ساقچى ئەمەلدارى ۋە ئۇنىڭ بىلەن پاراڭلىشىۋاتقان دوستى بىر قانچە ساقچىلار ئالدىراپ چىقىدۇ دۋە تاياق يەۋەتقان بالىنى ئايرىۋېلىپ كىشلەرنى نىمىش بولغىننى سوراپ بولغىچە بۇ ئۇغرىنى سولاش كېرەك دەپ ئەتراپىدىكىلەر ۋاقىرايدۇ بۇنى ئاڭلاپلا ساقچى ئەمەلدارى نىمىش بولغىننى بىلىپلا ئۇنى سۇتلاشقا باشلايدۇ باياتىن بېرى بولغان سۆزلەنى ئاڭلىغان ساقچى ئەمەلدارىنىڭ يېقىن دۇستى بۇ ئېشقا ئارلىشىپ ياش بالىنىڭ ئالدىغا بېرىپ ئۇنىڭ كۆزىگە بىر قارىۋەتكەندىن كىيىن   ھېچ خېجىل بولماستىن  ئاستا زوڭزىيىپ ئۇنىڭ ئايىغىنى  ئېيتقاش ئەتراپىدىكىلەگە مۇنداق دەيدۇ  
بىر ئاياق مەينەت بوپ كەتتى دەپلە ئۇنى تاشلىۋېتىشكە كىم چىدايدۇ ھەم كىم تاشلېۋېتىدۇ  ھەتتە يىتلىپ كەتسىمۇ ياماپ كەيسە بولىدىغۇ بىرە مىڭ كۇيلۇق نەرسىنى دەپ ئۆزىمىزنىڭ مىللىتىنى تۇتۇپ سولاپ بېرەمدۇق ئاياق مەينەت بوپ كەتسە تازلاپ مايلاپ كەيسەك يېتلىپ كەتسە ياماپ كەيسەك بولىدىغۇ خاتالاشمايدىغان كىم بار  ئۇنى ئۇرۇپ تىللىشاق ياكى سولىساق بۇ مەسلە ھەل بولامدۇ  ياق  ئۇ يەنە ئۇغۇرلۇق قىلىشى مۇمكىن بەلكى بۇنىڭدىن چۇڭراق نەرسنىنى ياكى قىممەترەك نەرسىلەنى ئۇغۇرلىشى مۇمكىن ئۇنىڭدىن كىم  نىمىشقا شۇنداق قىلغانلىقىنى سوراپ باقتى يەنە كىم بۇنداق قىلسا ئېغىر جىنايەت ھەم گۇناھ بولدىغانلىقىنى سوراپ باقتى بۇنداق ياشاشنىڭ ئۇيغۇرغا خاس ئەمەسلىكىنى ، تارىختىن بۇيان ھەتتا يولۋاسنىمۇ ھەۋەستە قۇيغان بۇ ئۇيغۇرمىللىتىگە زادى نىم بولدى ................................................................................................................................................؟  ئۇ ئادەم شۇنداق دېگەچ بالىنى ساقچىخانىغا ئېلىپ كىرىپ كېتىدۇ ئەتراپتىكى كىشلەر ئۇزىنىڭ خاتالىقىنى تۇنۇپ بىر ھازا ئۆزىنىڭ نىمىش قىلغانلىقىنى بىر كومپىيۇتېرنى دەپ بىر بالىنى بىر قىرىندىشىنى ئەركىنلىككە زار بولىدىغان گۇندىخانىغا سولاپ قۇيماقچى بولغانلىقىنى ئۆيلاپ ئۇ بالىنى ئىزدەپ كەينىدىن كىرىدۇ  
ھېلىقى ياخشى ئادەم بۇ بالىدىن نىمىشقا شۇنداق قىلغانلىقىنى سۇرىغاندا ئۇ بالا ئانىسىنىڭ ئېغىر كىسەللىكىنى شۇنى داۋالاشقا پول لازىملىقىنى دېيشىگە كومپىيۇتېرچى ئادەم كىرىپ سورايدۇ ھەم بولغان ئەھۋالنى بىلىپ ھېلىقى ياخشى ئادەم بىلەن بىرلىكتە ئۇ بالىنىڭ ئۆيىگە بارىدۇ ئەمما ئۇنىڭ بايامقى گەپلىرى يالغان چىقىدۇ ئۇ ئەمدى راسنى دېمىسە بولمايدىغانلىقىنى بىلىپ راس گەپنى دەيدۇ ئەسلىدە ئېسىنى بىلسە بىر ئايال بىلەن تىلەمچىلىك قىلىدىكەن كىيىن ئۇ چوڭ بولغاندا ئۇ ئايال ئۇغۇرلۇققا ساپتۇ كىيىن ئۇنى ساقچىلار تۇتۇۋېلىپ سوراق قىلغاندا ئۇ ئايالنىڭ شۇنداق قىلغۇزغانلىقىنى دەيدۇ ساقچىلار تەكشۇرۇپ ئۇ ئايالنى تۇتۇپ سولايدۇ بۇ بالىنىڭ ئاتا ئانىسى چەت بىر يېزىدىن تاپىدۇ ھەمدە ئاتا ئانىسى ئۇ بالىنى يوقاپكەتكەنلىكىنى دەپ ئېلىپ قالىدۇ ئۇ بالا ئۇ ئۆيدە كۇندە تىل ئاھانەت تاياق دومبا يەيدۇ ئەسلىدە بۇ بالىنى ئايلىقىنى 1500 كويدىن ئېجارىگە بەگەنلىكىنى ئەمما ئۇ بالا نان بېرىۋاتقان ئۇ ئانىنى ساقچىلاغا تۇتۇپ بەگەنلىكىنى دەپ ئۇنى كۇندە ئېغىر ئېشلاغا سالغاچ ئېچكىردىكى كىشىلار بېلەن ئالاقىلاشىۋاتقانلىقىنى ئاڭلىغان بالا ئاتا ئانىسىغا ئۆچمەنلىكى كىلىپ يەنە ئۇرۇمچىگە قاچىدۇ ۋە جان بېقىش يولىغا كىرىشكە باشلايدۇ  ۋە بۇكىشلەگە ئۇچرايدۇ  بولغان ئىشلانى ئاڭلىغان بۇ ئىككى كىشى بۇ بالىغا ياردەم قىلىدۇ ھەمدە كىچىك بىر دۇكان ئېچىپ بېرىدۇ   


ئەمدى مەخسەتكە  كەلسەم  كىشلەر مۇشۇنداق ئۇغرىنى كۆرگەندە ئۇرماي تىللىماي ئۇنىڭغا نىمىشقا شۇنداق قىلغانلىقىنى  قىلاي دېسە نۇرغۇن ياخشى ئىشلانىڭ بارلىقىنى ۋە چىرايلىق تەربىيە قىلىشنىڭ زۇرۇرلىكىنى بىلسىكەن  مۇشۇنداق قىلىۋەسە ئاخىرىدا ئۇمۇ باشقىلاغا ياردەم قىلىشنى ئۇيلايدۇ ھەم ئاستا ئاستا نۇرغۇن كىشىلار ياخشلىق قىلىشنى ئۇيلايدىغان بالىلار بولسا ئاتا ئانىسىنىڭ قىلغىننى كۆرۇپ ئۆگىندىغان بولامىكىن  دېگەن ئۇمىد تە بۇنى يېزىپ تۇرداشلا بۇنى كۆرۇپ ئازدۇر كوپتۇر كوچ چىقىرىپ  بىرقىسىم ناچار قىلمىشلاغا قاراپ شۇنى سۆكوپ ئاز ئەممە ساز كېنۇلانى تۇرلادا ئېلان قىلىپ تۇرسا كېنۇ قىلىم شېركەتلىرى بولمىسىمۇ توي فېلىم ئالىدىغانلا بىلەن كۇرۇشۇپ بىرنەچچە پىلانلىغۇچى بىلەن قىلسمۇ بولىدۇ دەپ ئۇيلىدىم  بۇ پەقەت مېنىڭ شەخسى ئۇيلىغانلىرىم  باشقا كىشلەنىڭ ئۇيلىشىپ بېقىشىڭلاغا تەۋسىيە
