ۋەي . ئەسسالامۇ ئەلەيكۇم.ۋەئەلەيكۇم ئەسسالام بوگۇن شەنبە ۋاقتىڭىز بارمۇ ؟ياق بۇگۇن زىشى بار ،سىزكىمۇ دىدى قىز ،بىرسىگە بەرگەن ۋەدىڭىز ئىسىڭىزدەبارمۇ ؟نىمەۋەدە ؟ئەگەردە كىم كىمنى بالدۇر تونىۋالسا تۇنىيالمىغان تەرەپ ،تونۋالغان تەرەپنى بىركۇن شەھەرنى تاماششا قىلدۇرىدىغان ۋەدىڭىزچۇ ؟ھە، با،سىز باھادىرمۇ ؟ ھەئە مەن ،قەشقەرگە بۇگۇن كىلىشىم نەرسە كىرەكلىرىمنى بىتەرەپ قىلىۋىتىپلا سىزگە تىلفۇن قىلىشىم كىلەلەمسىز ؟ ئەلۋەتتە قەرشقەر گە منى دەپ كەلگەن تۇرسىڭىز چىقمىسام قانداق بولىدۇ ،ئەمسە كىلىڭ ياكى مەن باراي ،ياق بولدى ساقلاپتۇرۇڭ مەنلا باراي نەدەكورشىمىز دىدى قىز ،شىنخۇاكىتىپخانسىدا سىزنى ساقلايمەن ،ھەراس يىنىمدا كوپ پۇلمۇ يۇق كورۇشكەندىن كىيىنكى ئىشلار ۋەدىمىز بويىچە بولسۇن ،چاتاقيۇق شاھزادەم مەن مۇ شۇنى دەي دەپ تۇراتتىم ساقلاپ تۇرۇڭ ھازىرلا يىتىپ بارىمەن دىدى قىزمۇچاقچاق ئارلاش تەلەپپۇزدا ،بولىدۇ ،ھەراس مىنىڭ كوپ ۋاختىم يوق بىر سائەت كىچە كىپبولالمىسىڭىز كىتىپ قالىمەن ئۇچاغدا مىنى كورەلمەي ئارماندا قالىسىز پۇرسەت غەنىمەت پۇرسەت كۇتكەنلەرگە ئەمەس بەلكى تىرىشقانلارغا مەنسۇپ ئالدىراڭ ھە باھادىرمۇ چاققاق قىلىپ ،شۇنداق مۇ ئەمسە مەن ئۇچۇپلاباراي ساقلاپ تۇرۇڭ دىگەچ قىزمۇ تىلىفۇننى قويدى
باھادىرمۇ تىلىفۇننى قۇيغاچ ئىككىسىنىڭ تونۇشۇش جەريانىنى بىر قۇر ئەسلەپ ئوتتى .بۇ مەزگىلدە باھادىر ،نەۋرە ئاكىسىسى قاتارلىق بىريۇتلۇق بىر نەچچەيلەن قەشقەرگە قاراشلىق مەلۇم ناھىيەدە قۇرلۇش ئورنىدا قارا ئىشچى بولۇپ ئىشلەۋاتاتتى .باھادىر نەۋرە ئاكىسى بىلەن بىرگە تۇراتتى ،بۇ جۇمە كۇنى بولۇپ بۇكۇندە ئاكا -ئۇكا ئىككىسى ھارقان چە مۇھىم ئىش بولسىمۇ قىلمايتى .ئاكا تىلىفۇنىڭنى ئەكىلە دىدى باھادىر ،نىممە قىلىسەن .بىردەم چىچىرىمنى ئىچىۋالاي ھە مانا ھەراس مىنىڭ چېچېرىمىنى تاقىۋەت تىلىفۇن كەلسە ئالما ھە دىگەچ سىتقا قاراپ ماڭدى نەۋرە ئاكىسى .ھەبۇلىدۇ ساڭا كەلگەن تىلىفۇننى ئالمايمەن ماڭا كەلسە ئالاي ،باھادىر،ئاكسىنىڭ چېچې توپىغا كوز يۇگۇرتۇپ ئۇلگىردى ،ۋاي ۋۇي نىمە دىگەن كوپ ئادەم بۇ ئەۋرە ئاكام مۇشۇ ئادەملەرگە قانداقمۇ خەت يىزىپ بولدىغاندۇ ؟ئۇشۇنداق دىگەچ قىزلار توپىغا كوزيۇگۇرتتى ،شەبنەم .ئايگۇل گۇلى يەنە ئالقانداق ئىسىملارنى قويىۋالغان قىزلارمۇ كوپ ئىدى .تۇيۇقسىز ئۇنىڭ كوزى ئىسىمىنىڭ باش ھەرىپنى ق ئوتتۇرسى خەنزۇچە ئاياغ ھەرىىنى نۇر دەپ قويىۋالغان بىرقىزغا كوزى چۇشتى ،بۇخەنزۇمىدۇ ؟ياكى ؟ئاۋال ماتىريالىنى كورۇپ باقايچۇ ،ئۇقىزنىڭ ماتىرماتىريالى مۇنداق ئىدى يىشى 20 ياش.راس ئىسمىدىگەن يەردە قەلبىنۇر .شەخىسنىڭ سوزىدە [ئاللاھتىن بەختىمنى تىلەيمەن دەپ يىزىقلىق ئىدى] باھادىر قەلبىنۇرنىڭ چېچې نۇمرىنى يادلىۋىلىپ ئوزىنىڭ چېچېسىنى ئاچتى ھەمدە يادلىۋالغان نۇمۇرنى قاچىلىۋالدى .ئارىدىن ئۇزۇن ئوتمەي قەلبىنۇرمۇ باھادىرنى قوشىۋالدى .ئۇ ئىككىسى تەخمىنەن يىرىم سائەتلەرچە پاراڭلاشتى مەلۇم بولدىكى قەلبىنۇر قەشقەر تىببىي ئۇنۋىرسىتتا ئوقۇيدىكەن ھازىر ئوقۇۋاتقىلى 4 يىل بوپ تۇ كېلەر يىلى ئوقۇش پۇتتۇردىكەن ،شۇكۇنى باھادىر جۇمە نامزىغىمۇ ئۇلگۇرەلمىدى ، مانا ئۇ ئىكىكىسىگە چې چې دا تونۇشقىنىغىمۇ بىر ئايدىن ئىشىپتۇ .ئۇلارنىڭ چې چې دا قىلىشقان گەپ سۇزلىرى كوزقاراشلىرى جەميەت توغۇرلۇق پاراڭلىرى ئاساسەن بىريەر دىن چىقاتتى ، بۇگۇن قەلبىنۇر ،باھادىرنىڭ تىلىفۇن نۇمرىنى سورىدى باھادىر مۇ تىللىفۇن نوررىنى دەپ بەردى .بىر ئازدىن كىيىن قەلبىنۇر دىن تىلىفۇن كەلدى باھادىر تىلىفۇننى ئەسسالامۇ ئەلەيكۇم ياخشىمۇسىز دەپ ئالدى ،ۋەئەلايكۇم ئەسسالام قانداق ئەھۋالىڭىز ياخشىمۇسىز ،ياخشى ئوزىڭىز چۇ نىمە ئىشلار بىلەن مەشغۇل بوپكەتتىڭىز ئوقۇشىڭىز چۇ ؟نەتىجىڭىز ياخشىمۇ؟ ،ياخشى بۇگۇن دەم ئىلىش بولغاچقا ئويگە چىققان سىزچۇ؟ نىمە قىلىپ ؟ئويدىكىلىرىڭىز سالامەت تۇرۇپتىمۇ ؟مەنمۇ بىكا شۇسىزنى قاچان چې چې سىنى ئاچاركىن دەپ ئولتۇرۇشۇم دىدى باھادىر ،ئۇلارمۇ سالامەت تۇرۇپتۇ سىزنىڭ ئويدىكىلەرچۇ ؟ سورىدى قەلبىنۇر ،بىلمەيمەن بەلكىم ئۇلارمۇ ئاللانىڭ ھىممىتىدە ياخشى تۇرغاندۇ .ئويىڭىزگە قايمىدىڭىزما؟ياق دىدى باھادىر ،ھە قەشقەر تەرەپلەرگە كەلمەمسىز باري دىگەن قەشقەردە يابىرەر تونۇشۇممۇ يوق شۇڭا ،ھە مۇنداقدەڭ مەن تونىشىڭىز ئەمەسما؟ قەلبىنۇر رەنجىگەن تەلەپپۇزدا.مەن سىزنى دوستۇ دەپ يۇرسەم بۇنىمە دىگىنىڭىز ،باھادىر ئويلىمايلا قىپ قويغان گىپىدىن ئوڭايسىز لىنىپ بىردەم تۇرىۋالدى .ياق ئۇنداق دىگىنىم يۇق بىز بۇرۇن كورشۇپ باقمىغان .قەشقەرگە بىرىپلا سىزنى ئۇچرىتالىشىم ناتايىن ،شۇڭا ماڭا بىرپارچا رەسىمىڭزنى ئەۋەتىپ بەرسىڭىز بولامدۇ ؟ ھە بولىدۇ ،سەل ساقلاپ تۇرۇڭ بىر ئازدىن كىيىن تىلىفۇن قىلاي دەپ تىلىفۇننى قويدى باھادىر بولساتىخىچە ئويلىماي دەپ قويغان گىپىگە پۇشايمان قىلىپ ئولتىراتتى ،تىلىفۇنىغا رەڭلىك ئۇچۇر كەلدى ھە قاچان قەلبىنۇر ئەۋەتتى بولغاي باھادىر دەرھال كەلگەن ئۇچۇرنى كورۇشكە باشلىدى ،كەلگەن ئۇچۇردا ئوخشاش بىر قىزنىڭ ئىككى خىل كىيىنگەن رەسىمى جىلمىيىپ خۇددى باھادىرغا قاراۋاتقاندەك بۇلاقتەك كوزلىرىنى يوغان ئىچىپ قاراپ تۇراتتى .نىمە دىگەن چىرايلىق ھە دىۋەتتى باھادىر ،سۇرەتىكى قەلبىنۇر بىشىغا گىلەم دۇپپا كىيگەن ئۇستى بىشىغا ئەتلەستىن پۇتۇن قىلىپ كوڭلەك كىيگەن بولۇپ ھەقىيقى ئۇيغۇر قىزىغا خاس سالاپەت چىقىپ تۇراتتى .يەنە بىر پارچىسى ئوخشاشلا بىشىغا گىلەم دوپپا كىيگەن ئۇستى بىشىغا بولسا ئوقۇغىچىلار فۇرمىسى كىيگەن سۇرەت ئىدى ،باھادىر سۇرەتكە قاراپ ئولتۇرۇپ ئوزىگە كەلگەن تىلىفۇننىمۇ سەزمەي قالدى ئەگەر قەلبىنۇر يەنە قايتىلاپ تىلىفۇن ئۇرمىغان بولسا ئۇ سۇرەتتىكى قىزغا قاراپ قاچانغىچە ئولتىراتتىكىن تىلىفۇننىڭ ئەنسىز جىرىڭلىشى ئۇنىڭ خىيالىنى بۇزدى ،باھادىر كەلگەن تىلىفۇن نومرىغا قارىماستىن كىمۇ ئادەمنىڭ خىيالىنى بۇزۇپ قاملاشمىغان .ئۇ غۇدرىغىنىچە تىلىفۇننى شۇنداق دەپ ئالدى مەن شاھزادە بىردەم دە نىمبولدى ئۇستازىدىن گەپ ئىشكەن شاگىرتلاردەك غۇدىراپلا قاپسىزغۇ ھە سىز مىدىڭىز كەچۇرۇڭ ،شۇ سىزنىڭ رەسىمدىكى جامالىڭىزغا قاراپ مۇڭلىنىپ قاپتىمەن ،سىز بەك چىرايلىكەنسىز قەلبىنۇر ،شۇنداقمۇ؟ھەئە راستىنلا چىرايلىقكەنسىز خۇددى بىركورۇپلا ئاشىقبولغۇدەك دەرىجىدە گۇزەلكەنسىز ،بۇرەسىمنى قەيەردە چۇشكەن باشتىكىسى ياشلا بايرىمىدا كىيىنىكسى مەكتەپتە ،سىزمۇ بىرەر پارچە رەسىمىڭىزنى ئەۋەتمەمسىز ،ھە بولىدۇ سەل ساقلاڭ باھادىر تىلىفۇننى قويۇپلا دوسلىرىنىڭ مەن رەسىمگە چۇشمەيمەن دىگەنلىرىگە ئۇنىماي خاتىرە بوپقالىدۇ ئاداش دەپ زۇرلاپ چۇشۇرگەن بىر پارچە رەسىمىنى ئەۋەتتى قەلبىنۇرمۇ باھادىرنىڭ ئەۋەتكەن رەسىمىنى تاپشۇرۇپ ئالدى باھادىر رەسىمدە ئاپئاق كوڭلەك قارا ئىشتاننى ياراشتۇرۇپ كىيگەن پۇتىدا توشەي چاچلىرى رەتلىك تارالغان بارماقتەك تۇم ئۇلانما قاشلىق كىرپىكلىرى خۇددى قىزلارنىڭكىدەك ئۇزۇن يوغان كوزلىرىنى بىرنۇقتىغاتىكىپ خۇددى بىرسىنى كوزتىۋاتقاندەك قاراپ تۇراتتى قەلبىنۇرمۇ نىمە دىگەن قاملاشقان يىگىت ھە دەۋەتتى بىرئازدىن كىيىن باھادىرتىلىفۇن قىلدى تاپشۇرۇپ ئالدىڭىز مۇ دوسلۇرۇم چۇشمەيمەن دىسەم ئۇنىماي زۇرلاپ دىگۇدەك چۇشكەن تىم ،ياخشى تارتىغان سۇرەتكەن باھادىر سىزنىڭ قىشىڭىز كوزلىرىڭىز ھەم قىزلاردىمۇ كەم ئۇچرايدىغان ئۇزۇن كىرپىكلىرىڭىز ئادەمنى ئوزىگە مەپتۇن قىلىدىكەن ،ناۋادا مەن ئوغۇل بالابولۇپ سىز قىزبالا بولغان بولسىڭىز سىزنى ھەرقانچە بەدەل تۇلىسەمۇ قوغلىشاتتىمكەن شۇنداق چىرايلىق قىزبالا بولاتتىڭىزكەن لىكىن پۇتىڭىزدىكى توشەي ياراشماپتۇ ھى ھى دىدى قەلبىنۇر چاخچاق قىلىپ ،شۇنداق مۇ ؟شۇڭا خۇدايىم بۇبالىنى قىز قىلىپ يارتىپ قويسام قەلبىنۇر بۇقىزنى قوغداپ بۇلالماي قىرىپ كەتمىسۇن دەپ مىنى سىزنى قوغداش ئۇچۇن ئوغۇل قىلىپ يارتىپ تىكەن باھادىرمۇ چاخچاق ئارلاش شۇنداق دىدى ،شۇنداق مۇ ؟ھەئە شۇڭا مەن سىزنى بىرئومۇر قوغداش قارارىغاكەلدىم قەلبىنۇر تەلىۋىمنى ئوبدانراق ئويلۇشۇپ بىقىشىڭىزنى ئۇمۇت قىلىمەن سىزنى ياخشى كورۇپقالدىم قەلبىنۇر سىزنى ئومۇرلۇك ھەمراھىم قىلىپ تاللىسام سىز قۇبۇل قىلارسىزمۇ ؟قەلبىنۇر بىردەم جىمىپ كەتتى بىر ئازدىن كىيىن مەن ئويلۇشۇپ كورەي ئاۋال سىزقەشقەرگە كىلىڭ ئاندىن جاۋابىنى بىرەي دىگەن ئىدى مانا شۇ سەۋەپتىن باھادىر قەشقەرگەكەلدى. بۇگۇن بۇئىككەيلەن شىنخۇاكىتاپخانىسىدا كورۇشمەكچى كىم كىمنى بالدۇر تونىۋالار قەلبىنۇر بالدۇر تونىۋالارمۇ ياكى باھادىرمۇ ؟ داۋامى بار بۇ ئەسەر ئۆزەمنى تۇنجى
