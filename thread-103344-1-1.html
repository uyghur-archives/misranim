ئەسسالامۇ ئەلەيكۇم مىسرانىم مۇنبىرىنىڭ ئەزىزلىرى ھەممىڭلارغا سالاملار يار بولغاي !

كۈن ، ئاينىڭ بەرىكىتى قالمايۋاتقان بۇ كۈنلەردە يىراق - يىقىندىكى ئۇرۇق - تۇققانلار ، دوسىت - يارەنلەر ، قىز - يىگىتلەر بىريەرگە كىلىپ ھال - مۇڭ بولۇپ ، كۆڭلىدىكى گەپلىرىنى ئېيتىشقا مۇيەسسەر بولالمايۋاتىدۇ .
تىلىفۇندا دىيىشسە تىلىفۇن ھەققى ئارتىپ كەتكەن ، پوچتا يوللانمىسى قىلاي دىسە ۋاقىت ئارتىپ كەتكەن بۇ ۋەزىيەتتە مەزكۇر مۇنبەر ۋاقىت ، ئىقتىساد جەھەتتىن مەلۇم دەرىجىدە قولايلىقلارنى يارىتىپ بىرەلەيدۇ ، ئەلۋەتتە بۇنىڭدىن باشقىمۇ ئۇيغۇر توربەت ، مۇنبەرلىرى بار ، بىراق ھەر مۇنبەرنىڭ ئۆزىگە چۇشلۇق مىھمىنى ۋە ئەزالىرى بولىدۇ ، شۇڭا مەزكۇر تىما ئارقىلىق ئۇرۇق - تۇققانلارئارا ھال سورىشىدىغان ، دوست - يارەنلەر ئارا پىكىرلىشىپ تۇرىدىغان ، قىز - يىگىتلەر ئارا تاتلىق پاراڭلارنى ( چەكتىن ئېشىپ كەتمىگەن ) قىلىش ئارقىلىق ئۆزىنىڭ بىرەر كۈنلۈك ئەھۋالىنى بايان قىلىپ قارشى تەرەپ ۋە كۆپچىلىكنى خەۋەردار قىلىش ، ئورتاقلىشىش مەقسىتىدە مەزكۇر تىما يوللاندى .

تىمىغا قارىتا قۇرۇق چىراي ئىپادىسى ۋە مۇناسىۋەتسىز ئىنكاسلار چۈشۈشنىڭ ئالدىنى ئېلىش ئۈچۈن ئالدىن دەپ ئۆتۈلدى .



بۈگۈن 2013 - يىلى 4 - ئاينىڭ 7 - كۈنى           ( يەكشەنبە )



تور دۇنياسىدىن نۇرغۇن خەۋەر ۋە تىمىلارنى كۆردۈم ، ئەڭ تەسىر قىلغىنى شۇكى ئىچكىرى ئۆلكىلەردە توخۇلارغا تارقالغان يۇقۇملۇق كىسەلنىڭ ئادەم جېنىغا زامىن بولغىنى بولدى .
مەن قۇش زۇكۇمى دىگەنلەرگە ئانچە دىققەت قىلىپ كەتمەيىتتىم ، بىراق ئىچكىرى ئۆلكىلەردە مۇشۇ يۇقۇم بىلەن 6 نەپەر جۇڭگو پۇقراسىنىڭ ئۆلگەنلىكى بۇ ئىشقا سەل قارىماسلىق لازىملىقىنى ئەسكەرىتكەندەك بولدى ، نىمىشقا دىگەندە ئادەتتە رايۇنىمىزدا ئىستىمال قىلىدىغان توخۇ گۆشى ۋە توخۇ تۇخۇمىنى ئىچكىرى ئۆلكىلەردىن يۆتكەپ كىلىدۇ دەپ كىسىپ ئېيتالمىساممۇ ، يۆتكەپ كىلىش ئېھتىماللىقى بار .
شۇڭا بۇ يەردە قوشۇمچە قىلىپ ، بۇ كۈنلەردە تورداشلارنىڭ بازاردىن توخۇ ، توخۇ تۇخۇمى ۋە توخۇ گۆشى سېتىۋالغاندا سۈپەتلىك ، ئۆزىگە ئىشەنچىلىك بولغانلىرىنى تاللاپ سېتىۋىلىشنى دەپ ئۆتۈپ كىتەي .

قېنى ئەمسە مەرھەممەت  ، بۇ تىما ئارقىلىق كۆڭۈلدىكى گەپلەرنى ئورتاقلىشايلى ، ئىتىپاقلىشايلى !

بۇرۇننىڭ  بۇرۇننىسىدا  ،  تۇخۇنىڭ  تۇرۇنىسىدا ، بىر مۇنبەر بوپتىكەن ، ئۇ مۇنبەر دە ھەرخىل ھەر قىسما كىشىلەر ياشايدىكەن ، بىر بىرىنىڭ گىپىنى ئاڭلىمايدىكەن ، بىر بىرىنى ھۆرمەت قىلمايدىكەن ،  بىر بىرىنىڭ گىپىگە كىرمەيدىكەن ، بىر ئىشتا دائىم بىرلىككە كىلەلمەيدىكەن ،  بىراق ،يەيدىغان تامىقى ، قىلىدىغان ئىبادىتى ، سۆزلەيدىغان تىلى ،كىيىدىغان تۇمىقى ئوخشايدىكەن ...بىراق شۇنىسى ئەممە لىكىنزە  ...ئىنتىلىشى ، ئەڭ ئاخىرقى مەقسىتىمۇ ئوپ ئوخشاشكەن .........  شۇنىڭ بىلەن  شۇنداق قىلىپ ......ئەڭ ئاخىرقى ئىتتىپاقلىشايلى  دىگەن  سۆز تەسىر قىلدى  ماڭا !  سەل  تاققا تۇققۇ سۆزلەپ قويدۇم جىما ....


مۇنازىرە يۆلىنىشىنى ئۆزگەرتۋەتمەي يەنە    باشقا  گەپ قىلاي ، قايتا  كىلەي ھە  ...


چۈشتە ، ئاغىيلىنىرىم بىلەن مۇسۇلمانلار كوچىسىغا باغان ،  شىئەن دە تۇرغىلىمۇ 4 يىل بوپ كىتىپتۇ ، يەنە بىرنەچچە ئايدىن كىيىن ئۆيگە كىتىدىكەنمىز ، ئاڭلىساق ئىخلاس تاللا بازىرى ئېچىپپتىكەن ، بېرىپ باقماپتۇق ،  ئىزدەپ كىتىپ  باساق نەدىدۇ   نەدىدۇ  دىيىشىپ ، بىژ ساقاللىق ئۇيغۇر بوۋاي  توكسىكىلت مىنىپ يېنىمىزدىلا پەيدا بۇلۇپ قالدى ،  ئەسسالام ئەلەيكۇم ئۇكۇلۇرۇم نەگە قانداق ماڭىمىز دەيسىلە؟ دەپتى  ئىخلاس تاللا بازىرى مۇشۇ يىقىندا دەپ ئاڭلىغان  دىسەك،بوۋاي قۇپۇپ  ھە  مەن  شۇ يەرگە ماڭغام ، ئوڭغا قەرىلسەڭلا  بارىسىلە دەپ توكسىكىلىتىگە مىنىپ كىتىپ قالدى ......مەن باللىغا  مۇشۇ  ئەدەم  بىز  باساق ئىخلاسنىڭ ئىچىدە  ئولتۇراپ  كەتسە ھە  ئۇكۇلۇرۇم  نىمە ئالىسىلە  دەپ .....دىگەندەك كۈلۈشمەك بىردەمدە  بېرىپتۇق ......شۇندا   قەيلىسام   ھېلىقى  بوۋاي  پوكەينىڭ  يېنىدا  تۇمامدا ......باللىغا   بايام  مەن  نىم دىگەن دىسەم  .،،،،، ئۈنىمىزنى چىقاماي تازا كۈلۈپتۇق ،...  .......ناۋاتنىمۇ  مۇسۇلمانچىمىدۇ  دەپ  ئالمايتىم  ئىچكىردە  ، قارىسام ناۋات باكەن ئېلىۋالدىم ،........قارىسام  زەمزەم  سۈيىمۇ  بار ئىكەن ....  قايتىدىغاندا  خۇدايىم بۇيرىسا  ئۆيدىكىلەرگە ئالغاچ  كەتمىسەم ،  ئالدىنقى قېتىم  قايتقاندا   دادام  ئاز ئازدىن ئىستىمال قىلىۋېتىپتىكەن ........بۇدا  ئۈزەم  ئاپىرىپ  بەرمىسەم......
