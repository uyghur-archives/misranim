ئەسسالامۇئەلەيكۇم مىسرانىمنىڭ ئىشتىراكچىلىرى . مۇنبەر كۆرۈپ تۇرغىلىمۇ خىلى بوپتۇ . تور بىكەت ۋە كۆرگەن ئاڭلىغانلىرىممۇ خىلى بوپتۇ . تور بىكەت ۋە دىتاللار قانداق پۇل تاپىدۇ شۇ توغرسىدا ئاز تولا  بىلگىنىمنى ئورتاقلىشاي دەيمەن .
ئاساسلىق تور بىكەتتىن كىرىدىغان كىرىم ئىلاندىن كىلىدىغان بولۇپ ، ئىلانلاردا ياخشىراق بولىدىغىنى مەسھۇلاتلارنىڭ ئىلانلىرى .ئۇندىن قالسا باشقا تور بىكەتنىڭ تەشۋىقات ئىلانى بولۇپ بۇ ئىلان يىڭى تور بەتلەرنى تونۇلۇشتىن بۇرۇن ئاز تولا مەبلەغ سىلىپ ئۆز تور بىكىتىنى داڭلىق تور بەتلەرنىڭ كىشىلەر كۆرىلىدىغان يەرلەرگە رەسىملىك ئۇلىنىش كىرگۈزۈپ تەشۋىق قىلىدۇ . لىكىن تونۇلمىغان تور بىكەتلەر نىمىدىن پايدىلىنىپ پۇل تاپىدۇ . ئەلۋەتتە ئىلان سۇپىسىدىن . يەنى ئاساسلىق كىرىم جىق يەنى دوللار كىرىدىغان سۇپىدىن گوگول ئىلان سۇپىسى ئەڭ ئالدىنقى قاتاردا بولۇپ ھەقىقى ھاردۇق چىقىدىغان كىرىم مەنبەسى بولىدۇ . لىكىن بىكەت خاسلىقىنى ساقلاشتا يەنى چوڭ تور بەتلەرنىڭ خاسلىقىنى بۇزۇپ قويىدىغان بولۇپ ئۇيغۇرچە بىكەتتە خەنزۇچە ئىنگىلىزچە مەزمۇن ئارلىشىپ قالغاندەك چوڭ تور بەتلەردە كۆپ ئۇچرىتالمايسىز  .ئۇلار كىچىك تور بەتنىڭ تەشۋىق ئىلانىدىن كىچىك تور بەتلەر گوگول ئىلان سۇپىسىدىن پۇل تاپىدۇ .
ئەمدى بىز كىرسەك پۇل كىرمەمدۇ ، ئەزا بولسىچۇ دىيىلگەندە بۇنىڭدا تور بىكەتنىڭ قىممىتى ئاشىدۇ خالاس . لىكىن پۇل كىرمەيدۇ . يەنى قانچىكى جىق ئەزاسى بولسا كۆرلىشى جىق ئاشسا شۇ تور بەتنىڭ زىيارەت مىقدارى ئاشىدۇ . شۇنىڭ بىلەن بىر تور نامىنىڭ قىممىتى ئاشىدۇ . لىكىن ئۇ تور بەتنى ساتمىسا يەنىلا داڭقى چىققاننىڭ پايدىسى يوق . داڭقى چىقتى دىسەكمۇ ساتمىسا پايدىسى يوق .
يەنى تور بىكەتنىڭ ئاساسلىق كىرىم قىلىدىغان ئۇسۇلىنى بىلىۋالدىڭلار . ھازىرچە ئۇيغۇر تور بەتلىرى مۇشۇ ئارقىلىق پۇل تاپىدۇ . خەنزۇچە بىكەتلەر يەنى فىلىم يوللاش تور بىكىتى مەسىلەن  يوۇكۇ   تۇدو تورى . ياكى چاقماق چۇشۇرگۈچ كىنو تورى دىگەندەكلار ئۆز مۇلازىمىتىنى يەنى ئۆز تىخنىكىسىنى ئايلىق پۇلغا ئالىي ئىقتىدار ئىچىپ تىزىملاتقان ئەزالارغا شۇ ئارقىلىق پۇل تاپىدۇ . لىكىن ئۇيغۇرچە كىنو تور بەتلىرى ئاساسەن خەنزۇچە تورغا كىنو چىقىرىپ ئاندىن تور بىكىتىگە چاپلاپ ئاچىدۇ . فىلىم تورىنىڭ بىشىدا نەچچە سىكونتلىق ئىلان چىقىدۇ بۇمۇ شۇ خەنزۇچە تور بىكەتلەرنىڭ پۇل تىپىشىدىكى بىر ئۇسۇل .

ئەمدى بەزى بىر دىتال توغرىسىدا مەسىلەن 360 بىخەتەرلىك قاراۋۇلى دىگەندەك ھەقسىز دىتاللار بۇلار مەڭگۈ ھەقسىز دەيدۇ . بەزىلار ھەقسىز نەرسىگە نىمە كىلىدۇ . ئۇنداق نەرسە ساختا دەيدۇ . لىكىن بىكاغا بىر كىم كىكىرىپمۇ بەمەيدۇ . ئۇلار قانداق پۇل تاپىدۇ . 360 نىڭمۇ ئالىي ئەزالىق مۇلازىمىتى بارغۇ دەيمەن ئىنىق بىلمىدىم .شۇ ئارقىلىق بولمىسا بىخەتەرلىك قاراۋۇلىدا  كومپۇيۇتۇردىكى كاشىلىنى يىراقتىن كونتۇرۇل قىلىپ سىزگە ۋاكالىتەن باشقۇرۇپ ئوڭشاپ بىرىدىغان ياكى يۇيۇلۇپ كەتكەن نەرسىلەرنى ئەسلىگە كەلتۇرىدىغان نەچچە ئون خىل تۇرلىرى تۇۋىنى 10 يۇئەندىن 30 يۇئەنگىچە مۇلازىمەت ھەققى بىلەن قىلىدۇ . سىز شۇ يەرنى بىلسىڭىز كىرىپ باقسىڭىز ھازىرغىچە قانچىلىك كىشىگە مۇلازىمەت قىلغىنىنىڭ سانىنى كۆرتىپ تۇرىدۇ . نەچچە مىليون قىتىم . بىرىلگەن باھامۇ ناھايىتى ياخشى . شۇنداق ھىساپلاپ بىقىڭ 10 يۇئەندىن 1 مىليون قىتىم مۇلازىمەت قىلسا قانچە پۇل . تىخى بىر تارمىقىلا يەنى 10 نەچچىسى بار . بۇلار تىخى مەخسۇس شىركەت مۇلازىمىتى ۋىروسلىرىنى ئۆلتىرىدىغان ياكى باشقۇرىدىغان مۇلازىمەتنى ئۇستىگە ئىلىپ خىزمەرلەرنىمۇ قىلىشى مۇمكىن . ياكى باشقا مۇلازىمەتلىرىمۇ بولىشى مۇمكىن . ئىشقىلىپ  ھازىر جەمىيىتىمىزدىن تارتىپ تورلارغىچە ھەممە نەرسە پۇل تىپىش كويىغا چۈشۈگلۇك . سىز بىكارغا بولسىلا ئەزان گوشنىڭ شورپىسى يوق دەپ ئۇ تور بىكەت ياكى دىتالنى كەمسىتمەڭ . سىز تاش سانىغۇچە ئۇلار قۇم ساناپ بولىدۇ . بىكارلىق نەرسسى پەقەت سىزنى باشلىۋالىدىغان تانىسى . سىز ئۇنىڭغا كۆنگەندە ئۇلار پۇلغا ساتمەن دىسە باشقا يەردىن ئالماي يانا شۇ يەردىن ئالغاندەكلا ئىش  . گوگولدەك دۇنيادىكى بىرىنچى باي تور بىكەتنىڭمۇ ھەقسىز ئىقتىدارلىرى ئادەمنىڭ ئەقلىگە كەلمەيدىغان ئاجايىپ ئىقتىدارلار بار . ھەممىسى ھەقسىز . شۇلار گوگول دىگەن نامنىڭ نامىنى چىقىرىدۇ . ئاندىن ھەقلىق مۇلازىمىتىنى ھەممە ئىشلىتىدۇ . ھازىرمۇ مىسرانىم ئالىي ئەزالىق ئىقتىدارىنى يولغا قويۇپ نادىر مەزمۇنلارنى نادىر فىلىملەرنى كۆرىدىغان قىلسا 100مىڭ ئەزادىن 1000 ئاچسا ئايلىقى 10 يۇئەن بولسىمۇ ئايدا 10 يۇەن دىگەن گەپ . شۇنىڭ ئۈچۈنمۇ خىلى جىق كۈچ تىخىنىكا كىتىدۇ ، فىلىم سۇزۇك نادىر فىلىم بولىشى مەزمۇن يىڭى ھىچ يەردە يوق كىشىگە مەنىۋى زوق ئاتا قىلالىشى كىرەك ئەلۋەتتە .

ئاخىرىدا مىنىڭ مۇشۇ ئۇيغۇر تورچىلىقىدا مەبلەغ سالسا ياخشى بولىدۇ دەپ قارايدىغان بىر نەچچە تەكلىۋىم بار ئوتتۇرغا سۇنۇپ باقاي .
ھازىر ئۇيغۇر تورچىلىقىدا كىنو بىكەتلىرى ھەممىسى ئوخشاش بىر كىنو . 2 ئادەم 5 ئادەمگە ئاۋاز بەرگەن كىنولام بار . تىلۋىزىيە ئىستانىسى تەرجىمە قىلغاندىن باشقىنى ياقتۇرمايمەن . شۇڭا مۇشۇ بىرەر  كىنو بىتى قۇرۇلۇپ تىلۋىزىيە ئىستانىسى بىلەن ھەمكارلىشىپ  چاقماق چۈشۈرگۈچنىڭ كىنو قويغۇچى ۋە تور بىتىنى تەرجىمە قىلىپ ۋاكالىتەن ئىشلىتىش ئارقىلىق ھەمكارلىق ئورنۇتۇپ تىلۋىزىيە ئىستانىسىنىڭ بۇرۇن تەرجىمە قىلغان كىنولىرىنى قويىدىغان بىر كىنو بىكتى ئاز دەپ قارايمەن .
يەنە بىرىسى  بەيدۇنىڭ بىلىمەن يانى سۇئال جاۋاپ بىتىدەك بىر تور بىكەت بولسا ھەر قانداق مەسىلا ھەر قانداق سۇئال يوللىنىدىغان ئەزا بولغان كىشى شۇنىڭغا جاۋاپ بىرىپ ھۆرمەتكە دەرىجىگە ئىرىشەلەيدىغان توغرا قانائەتلىنەرلىك جاۋاپ بەرگەنلار مۇكاپاتلىنىدىغان يەنى  بەيدۇدىن بىرەر سۇئال سورىغاندا چىقىدىغان بەتتىكى سىستىمىغا ئوخشاش بىر نەت قۇرۇلغان بولسا  بۇمۇ بىر ھىسابتا ياخشى ئىدى . ساملاملىق، ئويۇن، تىخىنىكا ھەر قايسى سەھىپىلەردە بىلمىگگننى ئانا تىلدا يىزىپ ئانان تىلدا جاۋاپ بىرىدىغان بىر بەت بولسا نىمە دىگەن ياخشى .

ئەمدى خەتلىرىم توغرىسىدا ئىملا خاتا دەيدىغان ئىش بولسا كۆزەينىكىمنى تاقىماي يازغان ئىكرانغا قارىماي . توغرا چۈشىنەسىلەن .  لىكىن مۇنازىرە تەكلىپ پىكىلەردە ئورتاقلاشساق ياق دىمەيمىز ئەمدى .

سالىھ بىگ بىلوگى سىزلەرنى قارشى ئالىدۇ .
