  ئەسسالامۇ ئەلەيكۇم بارلىق ئۇيغۇر قېرىنداشلار ، <بۇززۇۋان> دىگەن بۇ ئاتالغۇ كۆپچىلىككە ناتونۇش بولمىسا كېرەك. چۈنكى ، جەمئىيىتىمىزنىڭ ئۇچقاندەك تەرەققىياتىغا ئەگىشىپ، بىز ئۇيغۇرلانىڭ ئانا تىلىمىز بولغان ئۇيغۇر تىلىنىڭ ،ئۆز ئۇيغۇرلىرىمىز تەرىپىدىن <تەرەققىي> قىلىپ ، يات بىر تىلغا ئايلىنىپ كېتىۋاتقانلىقىنى ھەممىلا يەردە كۆرۈپ ، ئۇچرىتىپ تۇرىۋاتىمىز.  تۆۋەندە مۇنبەرداشلار بىلەن  بىر ساۋاقدىشىمنىڭ مۇشۇ تېمىدا يازغان ، تەنقىدى ئەھمىيەتكە ئىگە ماقالىسىنى ئورتاقلاشماقچىمەن :
                                                   <بۇززۇۋان>
    بۇززۇۋان دېگەن  57 -مىللەت بولغان مىنكوخەندىن قالسىلا،58-مىللەت بولىدۇ . ئۇلارنىڭ تىلىنى ساپ ئۇيغۇرمۇ، ساپ خەنزۇمۇ چۈشىنەلمەيدۇ،يېرىمىنى ئۇيغۇر ، يېرىمىنى خەنزۇ چۈشىنىدۇ. دىگەن گېپىنى مۇكەممەل بىر مىللەت چۈشىنەلمەيدۇ. شۇڭا ئۇلار گېپىنى سۆزلەپ ، سۆزلەپ توختاپ قالىدۇ.يا ، ئىچى پۇشمايدىكەن بۇخەقنىڭ ،بىر چىرايلىق تىلىمىزنى بۇزۇپ . ھەر قېتىم ئۇيغۇر تىلى بىلەن خەنزۇ تىلىنى ئارلاشتۇرۇپ سۆزلىگەنلەرنى كۆرسەم،ئۇلارنىڭ ئۇيغۇر چىمۇ ئەمەس، خەنزۇچىمۇ ئەمەس گەپ قىلغان زۇۋانىغا تۈكۈرۈپ، پۇتۇمنى تىقىۋەتكىم كېلىدۇ. توۋا دەپ قالىمەن ، بۇ خەقنىڭ تېگى .ئۇيغۇرمىدۇ ياكى خەنزۇمىدۇ؟ ئۇيغۇر دەي دىسەم خەنزۇچە سۆزلەۋاتقان ، خەنزۇ دەي دىسەم ئۇيغۇرچە سۆزلەۋاتقان ، ئاخىرىدا بۇ قازاق ئوخشايدۇ دەپ قالىمەن. بۇنداق دېسەم قازاقلارغىمۇ ئۇۋال بولغۇدەك .ھەي ......
      ئاخىرىدا  پۈتۈن ساپ ئۇيغۇرلارغا شۇنى دېگۈم  كېلىۋاتىدۇ: ئەي ئۇيغۇرلار ! ئۆزۇڭلارنى ھەرگىزمۇ خەنزۇغا ئوخشاتماڭلار! سىلەر دىگەن ئۇيغۇر بالىسى،خەنزۇ بالىسى ئەمەس ! ئانا تىلىمىزنى ئۆزىمىز قوغدىمىساق ، باشقا مىللەت ھەرگىزمۇ بىز گە تىلىمىزنى قوغداپ بەرمەيدۇ.
قوشۇمچە:بۇ ماقالىنىڭ قۇرۇلمىسى ئاددى بولغىنى بىلەن ، تەربىيىۋى ئەھمىيىتى ئىنتايىن  چوڭ ئىكەن. شۇڭا كۆپچىلىك بىلەن ئورتاقلىشىشنى لايىق تاپتىم.مۇشۇ ھەقتە كۆپچىلىكنىڭ قانداق پىكرى بولسا ،مۇنبەرداشلار بىلەن پىكىر يۈرگۈزۈپ، ئانا تىلىمىزنى قوغداپ قېلىش ئۈچۈن(گەرچە ئانا تىلىمىز تېخى يوقىلىش گىردابىغا كەلمىگەن بولسىمۇ، مۇشۇنداق بۇززۇۋانلار تىلىمىزنى نىڭ تىلىمىزنى يوقىتىۋېتىشىدىن قوغدىنىش ئۈچۈن)  ئازراق بولسىمۇ ، ھەسسىمىزنى قوشۇشنى ئۈمۈد قىلىمەن.

ئۆگەي ئانامغۇ يوقتى بولمىسا،
نەچچە تىلىم بار بىلمىدىم توۋا.
بىلىپ بولالماي ئەسلى تىلىمنى،
“زۇۋانىم قانداق ؟”دەيمەن گاھىدا .



ئانام بەرگەنغۇ قۇشقاچ تىلىنى ،
سايرىغاندىمەن نېچۈن تورغايچە ؟
خۇدۇمنى بىلمەي خاسلىقنى ئۇنتۇپ ،
ماھارەت دېسە گول بوپ ئۆزەمچە.



ئۇنتۇلارمۇ شۇ سەۋەب بىزدىن ؟
«ئورماندا بار»دەپ «قۇشقاچ دېگەن قۇش»
قەپەزگە كىرىپ چىقالماي ئاندىن ،
بىزگە ناتونۇش بولارمۇ ئۇچۇش ؟

ئەجداد ئىزىنى ئېچىپ بىرلىكتە،
قۇرايلى ئىناق بەختىيار ماكان.

مەسىلىنى ئىلمى نۇقتىدىن تەھلىل قىلىپ  مۇۋاپىق بولمىغان تەرەپلەرگە ئىلمى رەددىيە بېرىشىڭىزنى،تەنقىدى پىكىر بېرىشىڭىزنى،ھەل قىلىش چارىلىرىنى ئوتتۇرغا قويۇشىڭىزنى ئۈمۈت قىلىمىز،ھەرگىزمۇ باشقىلارغا ھاقارەت قىلىشقا بولمايدۇ،ھېسسىياتقا تايىنىپ ئاغىزمىزغا كەلگەننى دېسەك ئۆزىمىزنىڭ بىلىم سەۋىيىمىزنىڭ تولىمۇ چولتا ئىكەنلىگىنى بىلدۈرۈپ قويۇش بىلەن بىرگە،ھېچقانداق مەسىلە ھەل بولمايدۇ،سىز دېگەندەك تەربىيىۋى ئەھمىيىتىدىن تۈگۈل چولتا دېيىلگەن گەپلىرىڭىز ئۈچۈن ئۆزىڭىز ئاۋارىچىلىققا ئۇچراپ قېلىشىڭىز،باشقىلارنىڭ مەسخىرىسىگە قېلىشىڭىز مۈمكىن.
مەسىلىنى ئۆزىمىزنىڭ مەنتىقىسى بويىچىلا ئانالىز قىلىپ،باشقىلارنى قايىل قىلالىغۇدەك پىكىرنى ئوتتۇرغا قويالمىسىڭىز،ھەرقانچە كۈچەپ كەتسىڭىز بىكار.بۇ مېنىڭ سىزگە بەرگەن سەمىمى پىكرىم،قوبۇل قىلىش قىلماسلىغىڭىز يەنىلا ئۆز ئىختىيارىڭىزدا.
