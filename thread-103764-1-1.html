ئەسسالامۇ ئەلەيكۇم كۆڭلىگە ئوتتەك ئىشتىياق ، قىزغىنلىق ۋە سەنئەت تۇيغۇسىنى پۈككەن ، مۇشۇ ئاينىڭ 20 - كۈنى << چۈش ، تەبىئەت >> نامىدىكى ھازىرقى زامان ماي بويان ئەسەرلىرى كۆرگەزمىسنى ئېچىش تەييارلىقىنى قىلىۋاتقان  نۇرمۇھەممەد ئىبراھىم ( شەۋقى ) ۋە روزىمەت تۇرسۇن ئەپەندى ، ئاۋۋال ھەر ئىككىڭلارنىڭ دىيارىمىزنىڭ گۈزەل - سەنئەت ساھەسىدىكى ئىزدىنىش ۋە خەلققە ھەقسىز كۆرگەزمە ئۇيۇشتۇرۇپ بىرىشتەك خىسلىتىڭلارغا رەھمەت ئېيتىمەن ، ئۇنىڭدىن كىيىن مەزكۇر كۆرگەزمىنىڭ مىسرانىم ئىلان كۆزنىكىدىكى  پائالىيەتنىڭ چۈشەندۈرلىشى ۋە مۇناسىۋەتلىك بىر قانچە رەسىملەر توغرىسىدا ئازراق ئويلىغانلىرىم بار ئىدى ، شۇنى ئورتاقلىشىش مەقسىتىدە بۇ مەزمۇندا تىما يوللىدىم .

گەرچە ئىككىڭلارنىڭ باشقا ئەسەرلىرىڭلارنى ئانچە كۆرۈپ باقمىغان بولساممۇ ئەمما https://uyghur-archive.com/misranim/elan/qux/ بۇ ئادىرىسقا كىرگۈزگەن ئەسىرىڭلارنى كۆرۈپ ئولتۇرۇپتىمەن ، ئەسەرلەردە زامانىۋىلىق بىلەن مىللىي ئەنئەنە بىرلىشىپلا قالماي يەنە مىللىمىزنىڭ رەڭ كۆز قارىشى ، تۇرمۇش ئۇسلۇبى دىگەنلەر ناھايىتى ئۇستىلىق بىلەن يارىتىلىپتۇ .
نۇرمۇھەممەت ئەپەندىنىڭ رىئالىزىم بىلەن خىيالىزىم بىرلەشتۈرۈلگەن ئەسەرلىرى ، روزىمۇھەممەت ئەپەندىنىڭ ئاددىي شەكىللەر ئارقىلىق ئادەمنى ئويغا سېلىپ قويىدىغان ئەسەرلىرى ئادەمگە باشقىچە تۇيغۇ بىرىدىكەن ، شۇ تاپتا يەنە بىرخىل ئوي ، يەنى نۇرمۇھەممەت ئەپەندىنىڭ ئىككى رەسىمى ماڭا ئالاھىدە تەسىر قىلدى .
بۇنىڭ بىر رەسىم مۇقام ئۇستازى ، شائىرە ئاماننىساخاننىڭ رىئالىزىم بىلەن خىيالىزىم ئارلاشما ئۇسلۇپ قىلىنىپ سىىزپ چىقىلغان رەسىمى .

كۆرىنىپ تۇرۇپتىكى بۇ رەسىمدە كۆپ دۆلەت ، كۆپ خىل مەدەنىيەت ئارقا كۆرىنىش قىلىنىش بىلەن بىرگە ئۇيغۇر مەدەنىيتى ، ئۇيغۇر مۇقامى ئالاھىدە  يورۇتۇپ بىرىلگەن .
يەنى رەسىمنىڭ ئارقا كۆرىنىشگە سەددىچىن سىپىلى ( جۇڭگو ) ، ئايال ئەركىنلىك ھەيكىلى ( ئامرىكا ) ، ئېففېل تۆمۈر مۇنارى ( فىرانسىيە ) ، رىم تىياتىرخانىسى ( ئىتالىيە ) ، فۇجى تېغى ( ياپونىيە ) ، ئەلئىھرام ( مىسىر ) قاتارلىق دۆلەتلەرنىڭ ۋەكىل خارەكتىرلىك قۇرۇلۇش ، مەنزىرە رايۇنىنى رەسىمگە قىستۇرۇش ئارقىلىق رەسىمنىڭ تىلى ئىنتايىن كىڭەيتىلگەن بولۇپ ، بۇنىڭدىن بىر قارىسا  رەسسامنىڭ ئۇيغۇر مەدەنىيتى ، ئۇيغۇر مۇقاملىرىنىڭ دۇنياغا يۈزلىنىشىنى ، دۇنياغا نامايەن بولىشىنى پەرەز قىلساق ، يەنە بىر قارىسا ئۇيغۇر مەدەنىيتى ، ئۇيغۇر مۇقاملىرىنىڭ دۇنيادىكى باشقا مەدەنىيەت ، مۇقاملاردىن گۈزەللىكى ، يېقىشچانلىقى بىلەن ئۈستۈن تۇرىدىغانلىقنى پەرەز قىلىش ئارقىلىق ئاپتۇرنىڭ كەڭ تەرەپلىمىلىك ئوي پىكىرنى بۇ رەسىمگە مۇجەسسەملەپ كۆرۈرمەنلەرگە تېخىمۇ مۇرەككەپ ، ئالاھىدە بولغان زوقنى ئاتا قىلىشى مەقسەت قىلغانلقىنى كۆرىۋالماق تەس ئەمەس .
ئەمدى ماڭا يەنە بىر ئالاھىدە تەسىر قىلغان رەسىم يەنى تۆۋەنكى رەسىم ئۈستىدە ئويلىغانلىرىمنى بايان قىلىپ ئۆتىمەن .
بۇ رەسىمگە قارىساملا رەسىم مەزمۇنىدا بىر - بىرىگە زىت بولغان كۆرىنىش ، ئىپادىلەش بارمىكىن دەيمەن .
ئۇيغۇر كىيىم - كىچەكلىرى جۈملىدىن ئۇيغۇر باش كىيمى بولغان دوپپا شۇنداق سىپتا ھەم گۈزەل ، شۇڭىمىكىن بىز ئۇنى قانداقلا كىيىم كىچەك بىلەن كىيسەك يارىشىدۇ ، ئۇيغۇر دوپپىسىنى تەرىپلەشكە تىلىم ئاجىزلىق قىلىدۇ ھەم بۇ يەردە مەزمۇندىن چەتنەپ كىتىدۇ ، ئۇيغۇر دوپپىسى دىمەك ئۇيغۇر دىمەكتۇر دەپ چۈشىنىمەن ، ئەمدى بۇ رەسىمدىكى قىزغا كەلسەك بۇ قانداقتۇر زامانىۋى مەدەنىيەت ۋە زامانىۋى ئۇسلۇپتا سىزىلغان ، ئوبراز يارىتىلغان قىز ئىكەن .
دىمەك بۇ رەسىمدە ئۇيغۇر مەدەنىيتى بىلەن زامانىۋى مەدنىيەت ئۇچرىشىپ ، زامانىۋى ئارلاش مەدەنىيەت ئۇسلۇبى يارىتىلدى دىگەن گەپ .
بەلكىم بۇرەسىمدە  شەۋقى ئەپەندىمنىڭ  مەن بىلىپ يىتەلمىگەن ئوي پىكىر ۋە يىڭىلىق يارىتىش روھى باردۇ ئەمما مەن قانچە قاراپمۇ بۇ رەسىمدە زادى نىمە دىيىلمەكچى ئۇنى بىلىپ يىتەلمىدىم .
رەسىمدىكى ئۇيغۇر ئەر ، ئوغۇللىرىنىڭ باش كىيمى بىلەن كۆككۆز ، كىيىمسىز ھالەتتىكى قىز ئوتتۇرسىدا زادى قانداق باغلىنىش بار ؟ بۇ رەسىمدىكى مەركىزى ئىدىيە زادى نىمە ؟  مەن ھىچبىر ئۆزۈمگە لايىق بولغان جاۋاپقا ئىرىشەلمىدىم .
رەسىمدىكى ئۇيغۇر باش كىيمى ئورنىغا باشقا مىللەتنىڭ ( تۈركىي مىللەت ) باش كىيمىنى خىيالەن قويۇپ ئويلاپ باقتىم ، نەتىجىسى مەنسىتكەنلىك ئارلاشقان كۈلكە بولدى ،  قىزنىڭ ئورنىغا چىرايلىق بىر مىللىيچە كىيىنگە ئۇيغۇر قىزىنى خىيالەن قويۇپ ئويلاپ باقتىم ، نەتىجىسى بىر قىزنىڭ ئۆز مىللىتىنىڭ ئوغلانىغا بولغان سېغىنىش ، ئەسلەش ئەكىس ئەتتۈرۈلگەن رەسىم بولۇپ چىقتى ، دىمەك ماڭا مىنڭ ھازىرقى ئوي پىكىرىمگە كىيىنكىسى بىرئاز توغرا كەلدى .
بىراق بۇ رەسىمنىڭ ئالدىمىزدا ئۆتكۈزۈلمەكچى بولغان كۆرگەزمىگە قويۇلۇپ ( مەن قويۇلدىغان قويۇلمايدىغانلىقىنى ئۇقمايمەن ) ماڭا ئوخشاش كەسپى  ساھەدە ئەمەس ، ئوي - پىكىرى تۆۋەنرەك بولغان قېرىنداشلارغا بولۇپمۇ باشقامىللەتتىكى قېرىنداشلارغا ئەكىس تەسىر بىرىپ قويىشىدىن ئەندىشە قىلىۋاتىمەن .
ئەكىس تەسىرى دىگىنىم ، رەسىمدىكى قىزنى ئۇيغۇر قىزى دەپ چۈشىنىپ قېلىپ ئۇيغۇر قىزلىرىنىڭ شەرمىي ھايا ، ئەخلاقىغا مۇخالىپ بولغان كىيىمسىز ھالەتتە بىراۋنى سېغىنىشى ، ئۇنى ئويلىشى مومكىنمۇ ؟ بۇ ئۇيغۇر قىزلىرىدا بولۇشقا تىگىشلىك ئەخلاق پەزىلەتمۇ ؟ ئۇيغۇر قىزلىرى رەسساملارنىڭ قولىدا خەلقى ئالەمگە مۇشۇنداق كۆرگەزمە قىلىنامدۇ ؟....دىگەندەك خاتا چۈشىنىشلەر ۋە سۇئاللار شەكىللنىپ قالمىسۇن دىمەكچى ...
ئەگەر بۇ رەسىمدىكى قىزنى چەتئەللىك قىز دەپ پەرەز قىلساق بۇ يەنىلا ئۇيغۇر مىللى ئۆرۈپ ئادىتى ، ئەنئەنىسىگە  مۇناسىپ كەلمەيدىكەن .
قارىغانچە ئادەمگە چەتئەللىك قىز بىلەن بىر ئۇيغۇر يىگىتنىڭ قانداقتۇر ئىجتىمائىي تۇرمۇش ۋە باشقا جەھەتتە قويۇق ئالاقىسى باردەك ...
بۇ رەسىمنىڭ زامانىۋى سەنئەت ئۇسلۇبىدا يارىتىلغان رەسىم ئىكەنلىكىنى بىلىپ تۇرساممۇ بىراق يەنىلا مۇشۇ تەرەپلىرىنى ئويلاپ قوبۇل قىلالمايۋاتىمەن .
بىر مىللەت ۋە رايۇن ئۈچۈن زامانىۋىلىق بىلەن ئەنئەنە ، ئۆرۈپ ئادەتنىڭ ماس قەدەمدە ئوڭ تاناسىپلىق ھالەت بىلەن ئىلگىرلەپ مېڭىشى ئىنتايىن مۇھىم دەپ قارايمەن ، شۇڭىمىكىن بۇ رەسىمدىكى زامانىۋى سەنئەت شەكلىدە قانداقتۇر تەرەققىيات ، زامانىۋىلىق ۋە ئەنئەنە ، ئۆرۈپ ئادەت ئوتتۇرسىدا  ئازراق زىت بولغان مەزمۇن شەكىللىنىپ قالدىمىكىن دەيمەن .
بەلكىم مەن بۇندىن بۇرۇن ئۇيغۇر رەسساملىرى ياراتقان  بۇنداق زامانىۋى ئۇسلۇپتىكى رەسىملەرنى ئانچە كۆرۈپ باقمىغىنىم ئۈچۈن دەماللىققا بۇ رەسىمگە ساددىلارچە پىكىر - تەكلىپ بەرگەندىمەن ، بىراق ئۆزۈم كەسپى رەسسام بولمىغان بىلەن ئاز تولا گۈزەل سەنئەت تۇيغۇسى بار دەپ قارايمەن ، ئۇنىڭ ئۈستىگە كومپيۇتېردا سۈرەت لايھىلەش جەھەتتە ئازراق بىلىمىم بار ، بۇنداق دەپ ئۆزۈمنى ئوخشىتۋاتقىنىم يوق ، مەن پەقەت بۇ رەسىم ئۈستىدە ئويلىغان شەخىسىي تەكلىپ - پىكىرىمنى ئوتتۇرغا قويدۇم .  
بۇ سۆزلىرىمنىڭ ئۆزۈم ئۈچۈن توغرا بولسىمۇ ، سىز ( شەۋقى ئەپەندى ) ۋە كەڭ تاماششىبىنلار ئۈچۈن ئەرزىمەس بىر سۆز يەنى ئىتىراپقا ئېلىنمايدىغان سۆزلەر بولىشىمۇ مومكىن ، شۇنداقتىمۇ بۇر رەسىم ئۈستىدە قايتا بىر تەپەككۇر قىلىپ ، بۇ رەسىمنىڭ يەنە كىملەر ئالدىدا خاتا چۈشەنچە پەيدا قىلىپ قويىشنى كۆزدە تۇتۇپ كۆرگەزمىگە قويۇش - قويماسلىقنى ئويلىنىپ بېقىشىڭىزنى ئالاھىدە ئۈمىد قىلىمەن .
مۇناسىۋەتلىك ئادرىستىكى تىلىفۇنغا تىلىفۇن قىلىپ بۇ مۇددىئايىمنى ئېيتىشنىمۇ ئويلاشقان ، بىراق گەپنى قەيەردىن باشلاش ، قايسى يۇسۇندا قىلىشنى تازا بىلەلمىدىم ، ئۇنىڭ ئۈستىگە يۇقارقى سۆزلىرىمنى قىسقا ۋاقىت ئىچىدە قىلىپ بولغىلى بولماي قېلىپ ، قىممەتلىك ۋاقتىڭىزنى ئېلىپ قويۇشنى نامۇۋاپىق كۆرۈپ شۇڭا مۇنبەردە بايان قىلدىم .

ئەڭ ئاخىردا سىزلەرنىڭ بۇ قېتىملىق پائالىيتىڭلارنىڭ ئوڭۇشلۇق ، قىزغىن كەيپيات ئىچىدە ئاخىرلىشىشنى ، ياخشى باھا ، قوللاشلارغا ئىرىىششىڭلارنى ئۈمىد قىلىش بىلەن بىرگە سىزلەرنىڭ ئەجرىڭلارغا يەنە بىر قېتىم رەھمەت ئېيتىمەن .

بۇ تەكلىپ - پىكىر تۈسىنى ئالغان تىمىنى يوللاشتىكى مەقسىدىم قانداقتۇر بۇ قېتىملىق پائالىيەتنىڭ قىزغىنلىقىغا سوغۇق سۇ سىپىش ئەمەس ، پەقەت شەۋقىي ئەپەندىگە بىرىلگەن ئازراق تەكلىپىم .
شۇڭا تورداشلارنىڭ بۇ تىما مەزمۇنىنى ۋە تىمىدىكى مۇناسىۋەتلىك رەسىم ئۈستىدە قالايمىقان پىكىر يۈزگۈزۈشىنى خالىمايمەن .
تورداشلار تىمىدىكى مۇناسىۋەتلىك ئادرىسقا كىرسە بۇ پائالىيەت ۋە مەزكۇر شەخىسنىڭ ئەھۋالىدىن خەۋەردار بولالايدۇ ، يەنە ماۋۇ رەسىملىك ئىلاننى چەكسىمۇ شۇ ئادرىسقا ئۇلىنىدۇ .
