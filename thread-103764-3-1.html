پىكاسسو ، دالى قاتارلىقلار ياشىغان ئىجىتىمائى تۇرمۇش بىلەن بىز ياشاۋاتقان ئىجىتىمائىي تۇرمۇش ئوخشامدۇ ؟

ياۋرۇ ، ئامرىكا قىتئەلىرىدىكى خەلقلەر ئالىقاچان سەنئەتنى زامانىۋىلىققا يۈزلەندۈرۈپ بولغان ،، ئۇلار ئۈچۈن بۈنچىلىك رەسىملەر ھىچنىمە ئەمەس، بىراق بىز ئۈچۈن بۇ زامانىۋى سەنئەت شەكلىنىڭ تېخى تۇنجى قەدەملىرى ...
شۇڭا مەن ئۇ سىز دىگەن پىكاسسو ، دالى لارنىڭ رەسىملىرىنى چۈشەندۈرۈپ بىرىشكە سەل ئاجىزلىق كىلىمەن ، ئۇنى سىزگە ئوخشاش سەنئەتنىڭ ھەرقانداق تەرىپىنى قوبۇل قىلالايدىغان تورداشلار چۈشەندۈرۈپ بىرەلىشى مومكىن .
بەلكىم سىز ياشاۋاتقان ئىجتىمائىي تۇرمۇش بىلەن مەن ياشاۋاتقان ئىجتىمائىي تۇرمۇشتىمۇ چوڭ پەرقلەر بولىشى مومكىن ، شۇڭا بۇ يەردىكى كۆز قاراش پىكىرىمىز بىر يەردىن چىقمىغان گەپ ..

ئۇ رەسىمگە ئاۋۇ يەردە دىلبەر ئىسىملىك قىز تورداش ئۇيغۇر قىزىنىڭ ئۆز سۆيگىنىنى چىن ئىخلاسى بىلەن كۈتۈشى ، سۆيىشى دىگەندەك ئىزاھلار بىلەن شەرھىلەپ بىرىپتۇ ..شۇنداقمۇ دەيلى ، شۇنداق بولغان ھالەتتىمۇ بىر ئۇيغۇر قىزىنى كىيىمسىز يالىڭاچ ھالەتتە ئۆز سۆيگىنى ، يارىنى كۈتۈش ، سېغىنىشىنى يالىڭاچ سىزىش شۈنچە زۆرۈرۈمۇ ؟
ئۇنىڭغا ئاز تولا بولسىمۇ يېپىنچا سىزىپ قويسمۇ مەقسەت مۇددىئانى بىلگىلى بولاتتى .

قالغىنىغا بۇ رەسىمنى كۆرگەزمىگە ئاسقاندىن كىينكى نەق مەيداندىكى ۋە باشقا تاراتقۇلاردىكى كۆرۈرۈمەنلەرنىڭ باھاسى ئايدىڭلىشىدۇ .

پىكاسسو ، دالى قاتارلىقلار ياشىغان ئىجىتىمائى تۇرمۇش ...

دېگىنىڭىز توغرا ، دەپ ئۆتتۈمغۇ ، بۇ ئوخشىمايدىغان قائىدە ، ئوخشىمايدىغان دۇنيا قاراش ، ئوخشىمايدىغان جەمئىيەتتە ياشىغان كىشىلەرنىڭ كاللىسىدىكى ئوي پىكىرمۇ ئوخشىمايدىغان ھالەتتە تەپەكۇر قىلىدۇ . شۇڭا گۈزەل-سەنئەتتە ئىپادىلىگەن مەزمۇننى نوقۇل بىر ئادەمنىڭ ئىدىيىسىگە مەجبۇرىلاشقا بولمايدۇ ، بۇنىڭغا بەزىدە رېئاللىقتىن ھالقىغان تەسەۋۋۇر كۈچى تەلەپ قىلىدۇ .

مەن كىنو رېجىسورلىقىدا ئوقىۋاتىمەن ، ۋەتەندە قانداق دەرىسلەر بار بىلمىدىم ، لېكىن بىزگە نوقۇل كىنوچىلىق بىلىملىرىلا ئەمەس ، مۇشۇنداق مەۋھۇم ئۇقۇم بېرىدىغان گۈزەل-سەنئەت ، فىلىملارنى چۈشەندۈرىدۇ ، بۇ خېلى مۇھىم دەرىسىمىز ھېسابلىنىدۇ . سىز كۆرۈپ قىزىققان ھوللىۋۇد كىنولىرىمۇ شۇنداق مەۋھۇم ھەم باي تەسەۋۋۇر كۈچىدىن كەلگەن .

شۇڭا بۇنى رېئاللىقتىن چەتنەپ چۈشەنسىڭىزمۇ بولىدۇ . سەنئەتتە ماۋۇ پۈتۈنلەي خاتا ، ماۋۇ پۈتۈنلەي توغرا دەيدىغان ئىش يوق ، سىزنىڭكىنىمۇ خاتا دېيىشكە بولمايدۇ ، چۈنكى سىزنىڭ پىكىر قىلىش ئۇسۇلىڭىز باشقىچە بولىدۇ .

بۇنى شۇ كۆرگەزمىدە شۇ رەسىمنى سىزغان كىشىدىن سوراپ باقسىڭىز جاۋابى چىقىدۇ . لېكىن سىز يەنىلا ئۆزىڭىزنىڭ كۆز قارىشىنى ساقلاپ قالسىڭىز بولىدۇ ، چۈنكى بايا دەپ ئۆتكەندەك سەنئەتنىڭ ئوخشىمىغان كاللىدا ئەكىس ئەتكەن ئۇقۇمىمۇ ئوخشىمايدۇ ، بۇ تەبىئىي پەن ئەمەس ، بىر بولسا بىر ، ھەرگىز ئىككى ئەمەس دېيىلمەيدۇ .

يوللىغان ۋاقتى 2013-4-15 02:28:24 AM |ھەممە قەۋەتنى كۆرۈش |
     ھېلىقى يالىڭاچ بۇينىنى قىسىپ، يۈكۈنۈپ ئولتۇرغان ،قېشىدا تاشلاپ قويۇلغان دوپپا بار قىزنىڭ رەسىمى  رەسساملىقتىكى ياۋرۇپالىقلارغا كونا بىزلەرگە نىسپەتەن يىڭى بىر ئۇسلۇپ ...   (تېخنىكومنىڭ رەسىم، مۇزىكا كەسپىدە ئۇقۇۋاتقان چاغىلىرمىزدا بىزمۇ تەھلىل قىلىپ كىتەتتۇق... ھازىر ئۇنتۇپ كىتىپتىمەن... نى زامان رەسىم سىزمىغىلى...) ئەمما بۇ قانداق ئوسلۇپ بولسۇن سەنئەت بولۇش سۈپىتى بىلەن ئۆز-مىللىتىنىڭ قىممەت ئۆلچىمى ۋە مەدىنيەت-سەنئەت تەلىۋىدىن يىراقلىشىپ كەتسە خەلققە نەپ بەرمەيلا قالماستىن ئەكسىچە شۇمبۇيا بوپ قېلىشى مۇمكىن... ...
   مۇشۇنىڭغا قىستۇرۇپ ئازراق <<تاققا-تۇققا >>تەھلىللەرنى ئېلىپ بارغۇم كىلىۋاتىدۇ:
    رەسساملىق(ماي بوياق رەسىم) دەسلەپ شەرىقتە بۇددا دىنى مەدىنىيتى داۋامدا رىئالىزىم ئۇسلۇبتا ئۆلچەملىشىپ  كىيىن ياۋرۇپادىكى ئەدەبىيات-سەنئەتنىڭ قايتا گۈللىنىشى داۋامىدا غەرىپلىكلەرنىڭ قۇلىغا ئۈتۈپ تاكى بۈگۈنكى كۈنگە قەدەر شۇلار بىكىتكەن ئەندىزە، قېلىپ ، لاھىيە ۋە ئۇسلۇپلار بۇيچە داۋاملىشىپ كىلىۋاتىدۇ... بۇلۇپمۇ ياۋرۇپادىكى چوڭ مۇزىيخانلار، كۆرگەزمەخانىلار ۋە سەنئەت سارايلىرىدىكى كىم ئارتوق قىلىپ سېتىلىۋاتقان رەسىملەرنىڭ باھاسىنىڭ دەھشەت يۇقۇرىلىقى ، سىزىلغان رەسىمدىكى مۇجمەللىك ، ھەرخل ئۇسلۇپلارنىڭ يۇغۇرۇلما ھالەتتە ئىپادىلىنىش دەرىجىسىنىڭ يۇقىرى-تۈۋەنلىك بۇيچە ۋاقتى كەلگەندە ئادەتتىكى ئاۋامدىن كۆرە كەسىپ ئىگىلىرىمۇ قىلچە بىر نەرسىلەرنى ھېس قىلىپ يەتكىلى بولمايدىغان ئەسەرلەرنىڭ تۇيۇقسىزلا مەشھۇر بوپ كېتىشى كىشىنى بەكمۇ ھەيران قالدۇرىدۇ...
     ئاندىن بۇنداق ھادىسە مۇۋاپىقىيەت قازىنىش يولىدا ئۆز ئۇسلۇبىغا قارتا تەڭشەش ئېلىپ بېرىۋاتقان سەنئەتكارغا نىسپەتەن يىڭى بىر قىممەت ئۆلچىمى ياكى ئوسلۇپ يارتىشتىكى ئۈلگە بوپ قالىدۇ... شۇنداق قىلىپ ھەرقايسى ئەللەردە غەرىپنىڭ سەنئەت قارىشىنى چىقىش قىلغان يىڭى-يىڭى مەسلەك، ئۇسلۇپ، ئېقىملار بارلىققا كىلىدۇ...
     مىنىڭ ھېس قىلىشىمچە  دۇنيادىكى مەدىنيەتلەرنىڭ يەرشارىلىشىشى ۋە  ئۆز-ئارا بىر-بىرسىنى يۇتىۋېلىشى داۋامدا مەدىنيەت-سەنئەتنى يىتەكلەپ ماڭىدىغان ،مۇئەييەنلەشتۈردىغان... ئاڭ، قېلىپ، ئۆلچەملەرنى بىكىتىش سالاھىيتى تەبىئىيلا سانائىتى ۋە پەن-تېخنىكىسى تەرەققى قىلغان ئەل خەلىقلىرى قولىدابولىدىكەن...مەسىلەن: بۇ ھوقۇق ھازىر پەن-تېخنىكىسى غايەت زور تەرەققى قىلغان ياۋرۇ-ئامىركانلارنىڭ قولىدا...
     بەلكىم بۇ رەسىمدە  ئاپتۇرنىڭ كەسپى ئىزدىنشلىرى داۋامىدا ئۆز سوبىكتىغا تەسىر كۆرسەتكەن ئاشۇ ئوسلۇپ ياكى ئېقىملار ئازدۇر-كۆپتۇر ئىپادىلىنىپ قالغان بولۇشى مۇمكىن...     (يۇقارقىلار مىنىڭ ئاددى تەھلىلىم خالاس)
