باھادىر قەلبىنۇرنى ساقلاپ يىرىم سائەتتەك تۇردى ،قەلبىنۇر كەلمىگەچ ئىككىنچى قەۋەتلەرنى ئايلاندى ،باھادىر پەسكە چۇشكەندە بولسا دىيشكەن ۋاقىتمۇ توشۇپ قالغاندى قەلبىنۇرمۇ كىلىپ بولغاندى باھادىر ،قەلبىنۇرنى كورۇپلا تونىۋالدى ،بىرىپلا ئۇنى مەن بالدۇر تونۋالسام ياخشى بولماس دەپ ئويلىدى باھادىر .قەلبىنۇر بولسا كىتاپ جازىلىرى ئارسىدىن ئوزىنى يۇشۇرۇپ باھادىرنى ئىزدەۋاتاتتى ،باھادىر تەمتىرىمەيلا قەلبىنۇرنىڭ يىنىغا باردى ،ۋە قەلبىنۇرنىڭ ئالدىدىن بىر كىتىاپنى ئىلىپلا كورۇشكە كىرىشتى ،قەلبىنۇر باھادىرنىڭ كورىۋاتقان كىتابىنى كورۇپ كۇلىۋەتتى ۋە سىز ئوغۇلبالا تۇرۇپ قىزلاغا ئائىت كىتاپلارنىمۇ كوردىكەن سىزھە ،
شۇنداق ئانچە مۇنچە كورۇپ قويىمەن سەۋابى كەلگۇسىدە سىز بىلەن توي قىلسام ھالىڭىزدىن خەۋەر ئىلىشىم ئۇچۇن پايدىسى تىگىدۇ ئەمەسمۇ ؟

قەلبىنۇرنىڭ شۇ زامانلا قاپاقلىرى تۇرىلدى دە سوزلەپلا كەتتى ،قانداق قاملاشمىغان تەربىيە كورمىگەن بالا سىز كىم سىزبىلەن تويقىلىمەن دەپتۇ .بىز تونۇشقان بولساقتى كاشكى قاملاشمىغان .ھازىر سىلەر ئوغۇللارغا گەپ قىلغىلىمۇ بولمايدىكەن .قەلبىنۇر ،باھادىرنى تونيالمىغان ئىدى ،

شۇنداقمۇ . ئەمسە مەن قەشقەرگە بىكارغىلا كەپتىمەندە قەلبىنۇر شۇنداقمۇ دىدى باھادىر ،

بۇنىمە قىلغىنىڭىز بۇگۇن مەن مىھمان قىلىمەن دەپ تۇرۋالدى قەلبىنۇر ،

سىزنىڭ مىنىڭكى مىنىڭكى سىزنىڭ ئەمەسمۇ ،يەنە كىلىپ سىلەر ئوقۇغۇچىلارنىڭ باشقا تاپاۋىتىڭلار يۇق پەقەت ئاتا -ئاناڭلارنىڭ قولىغا قاراشلىق يىنىڭىزدىكى پۇلغا بۇگۇن مىنى مىھمان قىلىۋەتسىڭىز ئەتە نىمە يەيسىز ياتاققا بىنىۋىلىپ مەكتەپنىڭ تامىقىغىلا قارشلىق بوپقالىسىز باشقا ساۋاقداشلىرىڭىزنىڭ ئالدىدا بويۇن قىسىپ قىلىشىڭىز نى خالىمايمەن ،

مەن دىگەن ئوغۇلبالا پۇل تاپالايمەن ،ئەمما سىزچۇ قىز لار دىگەن پۇلسىز قالسا بولمايدۇ ،ماقۇل دەڭ بۇگۇنكى ھىساۋات مەندىن بولسۇن .سىزگە كىيىن مۇ پۇرسەت با ئۇچاغدا مىھمان قىلماي مەن دىسىڭىزمۇ ھەرگۇز ئۇنىمايمەن ،

باھادىرنىڭ سوزلىرى ئورۇنلۇق ئىدى دىمىسىمۇ ئوقۇغۇچىلارنىڭ بولۇپمۇ قىز ئوقۇغۇچىلار شۇنداق ،يەنە كىلىپ قىزلارنىڭ دۇشمىنى كوپ بولىدۇ شۇڭا قىزلارنىڭ يىنىدىن پۇل ئۇزۇلۇپ قالسا قەتئىي بولمايدۇ ،

شۇنداق قىلىپ .ساقچىلار بىزنى يالاپ قاماقخانىغا ئەكىلىپ تاپشۇرۇپ بەردى .قاماقخانا بەئەينى تىرىك دۇنيادىكى دەۋزەقنىڭ ئۆزى ئىدى .مەن دەك يىڭى كىرگەنلەرنى قاماقخانا خادىملىرى قاتتىق تەكشۇرگەندىن كىيىن كامىرلار غا سولاپ قويۇپ كونا مەھبۇسلارغا بۇنىڭغا ئوبدان قاراپ قۇيۇڭلار تۇزۇمنى ئۈگۇتۇپ قويۇشنى ئۇنۇتماڭلار دەپ قويۇپ چىقىپ كىتەتتى ،شۇنىڭ بىلەن كونا مەھبۇسلار مەندەك يېڭى كىرگەنلەرنى بىرقاتار نىمەئش بىلەن كېرگىنىمنى ساقچىلاردەك سوراق قىلۋەتكەندىن كىيىن تازا ئۇرۇپ پۇخادىن چىقىۋالغاندىن كىيىن ئۇزۇن بوپىتىكەن بىركىمنى ئۇرۇپباقمىغىلى خىلى چىداملىقكەنسەن ھە بۈگۈندىن باشلاپ كامىر تازلىقى ساڭا قالدى كامىرنىڭ ھەممە تازلىقىنى سەن قىلىسەن تاكى يېڭىدىن بىرسى كىرگۇچە چۇشەندىڭما ھە چۇشەندىم دىمىسىڭىز ياكى كۆزىڭىزنى يۇغانراق ئىچىپ قارسىڭىز يەنە ئۇرشاتتى ،ئاماليۇق

كۇنلۇرۇم ئەنە شۇنداق زۇلۇم ئۈستىگە زۇلۇم بولۇپ ئۆتىۋاتاتتى بۇ زۇلۇم ئازكەلگەندەك بۇيەرنىڭ تامىقىغا قورسامۇ تويمايدۇ

جان بەكمۇ قىينىلاتتى. ئارىدىن بىرنەچچە ئاي ئۆتۇپ تەپتىش مەھكىمىسى بىزنى سوتقا تارتتى ،مەن سوتقا چىقىشىمنى ئويلاپمۇ باقمىغان ئۆزۈمنى گۇناھسىز چىقىپكىتىمەن دەپلا ئويلايتىم .نەدىكىنى قارلىغۇچى جىنايەت ئۆتكۇزۇش سەۋەبىنى بىر-بىرلەپ سوتچىغا بايان قىلىدى .سوتچى بىزگە بىرقارىۋەتكەندىن كىيىن بىزگە ئۆزۇڭلارنى ئاقلاش سۆزقىلىڭلار دەپ ئۆزىمىزنى ئاقلاش نۆۋتىنى بىزگە بەردى ،ئۇئىككىسى ئۆزلىرىنى ئاقلىمايدىغان لىقىنى ئەيىپلىگۇچىنىڭ سۆزىگە قۇشۇلىدىغانلىقىنى بىلدۇرۇپ ،سوتچى باھادىر گۇناھسېز ئۇنى قويۋىتىڭلار ئۇنىڭ بۇ ئىشلاردىن قىلچە خەۋرى يۇق بۇ جىنايەتنى بىز ئۆتكۇزگەن بۇلاڭچىلىق ۋە زەخمىلەندۇرۇش ئىشىغا ئۇنى قوشۇپ قويماڭلار ئۇراسلا گۇناھسىز ،

شۇنداقمۇ؟بۇنى بىزقارار قىلىمىز ھە يەنە ئويلۇشۇپ بىقىشقا بولىدۇ .سەن قانداق قارايسەن باھادىر ئۆزۇڭنى ئاقلاش سۆزۇڭ يۇقمۇ؟

ئەلۋەتتە بار دوسلۇرۇمنىڭ دىگەنلىرى راس مەن پەقەت گۇلشەننى قوغداش ئۇچۇنلا ئۇ لۇكچەكلەر بىلەن سوقۇشقان باشقا ئىشلاردىن خەۋرىم يۇق كىيىنكى ئىشلار مەن گۇلشەننى مەكتەپكە ئىلىپ كەتكەندىن كىيىن يۇزبەرگەب بۇ ئىشلاردىن راستلا خەۋرىم يۇق

بولىدۇ بىز مۇرەسسە قىلىش ئارقىلىق ئويلۇشۇپ كۆرەيلى ؟

بۇلارنى ئەپكەتسەڭلار بولىدۇ .شۇنىڭ بىلەن سوت ئاخىرلاشتى يەنە شۇ ترىك دەۋزاخ قاماقخانىغا قايتۇرۇپ كىلىندۇق ،ئازاپلىق كۈنلەر يەنە باشلاندى ،يا قوساق تۇيمىغان ھازىرغۇ كامىرداشلىرىم بوزەك قىلمايدۇ ئۇلار بىلەنمۇ ئاكا ئۇكىدەك بوپكەتتۇق ،بولۇپمۇ شىرئەلى ئىسىملىك كامىردىشىم بىز بەك يىقىن بوپكەتتۇق مىنڭ ئۆيۇمدىن يوقلاپ كېلىدىغان ئادەم بولمىغاچقا شىر ئەلىنىڭ ئۆيدىكىلىرى سىلىپ بەرگەن پۇلغا بىرگە تاماق ئىلىپ يەيتتۇق ،بۇيەردە ئۆيدىن كەلمىگەنلەر قورساقتىن بەكلا قىينىلاتتى كۈنىگە 3تال مۇما بىلەن بىر چۈمۈش كۆكتات شورپىسى بىرىلىدىغان بولۇپ تاماقلارنىڭ تازلىقى بەكلا ناچار بەزىدە كۆكتات شور پىلىرىدىن قۇرتمۇ چىقىپقالاتتى ئاماليۇق ئىلىۋىتىپ ئىچەتتۇق جەميەتتە بولسىڭىز بىر نەچچە كۈن ئاچقالسىڭىزمۇ يىمەيسىز ئادەتتە لېكىن ۋاققىتلىق بولسىمۇ ھاياتنى ساقلاپ قىلىش مۇھىمدە ئاماليۇق يەيمىز شۇنى .بۇ تاماق پەقەت بىزنىڭ ئۆلۇپ قالماسلىقىمىزنىڭ دورىسى بولالايتى خالاس ،سوتتىن قايتىپ كىرىپ 10 كۈن بولغان كۈنى سوت خادىملىرى ئىزدەپكەلدى .

مانا بۇ ساڭا چۈشۈرۈلگەن ھۆكۈم ئوقۇپ بىرەيلىمۇ ياكى ئۆزۇڭ كۆرىۋالامسەن

؟خىيالىم چىگىش ،گادىرماچ خىياللار ئىچىدە تىڭىرقاپ قالدىم قانداق جازا بەرگەندۇ شۇنداققىلىپ تۇرمىگە يول ئالامەنمۇ ؟ياكى؟

قويۋەتكەنمىدۇ ؟جازالىغان تەقدىردىمۇ .جازا مۇددىتىمنى جەميەتتە تۇگۇتۇش ھۆكمى بولغايدى ئىلاھىم شۇنداق بولسا گۇلشەن بىلەن بۇرۇنقىدەك ئەركىن كۆرشەلەيمەن ئۇمەكتەپنى پۈتتۈرسىلا خاتىرجەم توي قىلالايمىز،ئۇمۇ بۇيل مەكتەپ پۈتتۈرىدۇ .بىز شۇنداق ۋەدە بىرىشكەن ،قاچان گۈلشەن كەپتەپتىن چىقسىلا تويقىلاتتۇق .

گەپ قىل ،سوت خادىمىنىڭ قوپال مۇئامىىلىسى خىيالىمنى چىچىۋەتتى .نىمە دىيشىمنى بىلمەيتىم ،قانداق بىز ئوقۇپ بىرەمدۇق ،كۆڭلۇڭ يىرىم بولىۋاتىدۇ ،بۇنى بىزمۇ ھېس قىلىپ تۇرۇپتىمىز قىرىندىشىم ،لېكىن ئوغۇلبالىدەك گۇناھ ئۆتكۈزگەندىكىن ئوغۇل بالىدەك جازاسىنى تارتىشىڭ ھەقلىق ،كېم سىنى قالايمىقانچىلىق چىقارسۇن خەقلەرنى بۇلىسۇن ئۇرسۇن يەنە كېلىپ مەڭگۇلۇك مىيىپ قىلىپ قۇيسۇن دەپتۇ ھەي گۇي باشقىلاردا ئاناڭنىڭ ھەققى بامىتى بۇ جازامۇ ئازلىققلىدۇ سەنلەرگە .سوت خادىمىنىڭ قىرىندىشىم دىگەن گەپلىرى شۇنداق يىقىملىق بىلىنگەن بولسىمۇ كېيىنكى گەپلىرىگە چىدىمىدىم ،شۇنداقتىمۇ ئۆزۇمنى بىسىۋىلىپ ئەپەندىم سىز بولسىڭىزمۇ ماڭا تۆھمەت قىلماڭ ،توغرا باشقىلار بىلەن ئۇرۇشقىنىم راس بۇنى ئىنكار قىلمىدىم ،لېكىن باشقىلار لايىقىمغا پەسكەشلىك قىلماقچى بولسا قاراپ تۇرامدىم ئەجەبا مىنىڭ لايىقىمنى بولغۇسى ئايالىمنى قوغداش مەجبۇريىتىم ۋە ھوقۇقۇممۇ يۇقما ؟ناۋادا ئۇسىزنىڭ قىزىڭىز بولسا قانداق قىلاتتىڭىز ؟مىنىڭ ئورنۇمدا سىز بولسىڭىز قاراپ

تۇرامتىڭىز ؟باشقىلار سىزنىڭ قىزىڭىزغا پۇخۇرلۇق قىلسا ئۇنى كۆزئالدىڭىزدىن تارتىپ ئەكەتسە دادا بولغان ئەركەك بولغان سىزقول قوۋۇشتۇرۇپ قاراپ تۇرامىڭىز ؟دەۋىتىپتىمەن ئاچچىقىمدا ،

نىمە دىدىڭ ئۇ ئالدىمغا دىۋەيلەپ كېلىشى ئۇنىڭ بىلەن بىرگە كەلگەن ئايال خادىم ئۇنى تۇتىۋىلىپ ئۇلگۇردى ،

بولدى قىلىڭ ئۇنىڭ بۇيەردە تارتىۋاتقان دەرت- ئازابى ئازمۇ شۇمۇ يىتىپ ئاشدۇ، كېمنىڭ بۇيەرگە كىرىپ قالغۇسى بار ،ئۇنىڭ ئورنىدا نەقمەيداندا سىز بولسىڭىزمۇ شۇنداق قىلاتتىڭىز .ئۇ باشتىن ئاخىر لالىقىنى قوغداش ئۇچۇن باشقىلار بىلەن سوقۇشقانلىقىدا چىڭ تۇرۇپكەلدى بەلكىم ئۇنىڭ ئىىتقانلىرى راستۇ شىرىكلىرىمۇ شۇنداق دپ كىلىۋاتىدۇ ،مەنچە بولسا ئۇنى قەھرىمان دەپ تەبرىكلەيتىتىم ئەمما قانۇننىڭ بەزىدە كۆزى كۆرمەس بوپقالىدۇ شۇ سەۋەپتى گۇناھسىز ئادەممۇ گۇناھكار بوپقالىدۇ ،بەلكى ئۇ راسلا گۇناھسىزدۇ ئۇنىڭ كۆڭلىگە ئازار بەرمەڭ .

بۇپتۇ سىزنىڭ يۇزىڭىزنى قىلاي

يەنىلا ئانىلارنىڭ قەلبى يۇمشاق كىلىدۇ ئاۋۇ ئەر مىنى ئۇرماقچىدى ،بۇ ئايال خىزمەتدىشى مىنى قۇتقۇزۇپ قالدى .بىز بۇرۇن تونۇشمايتۇق بولمىسا .پەقەت سۇتقا چىقىش ۋە ئەكرىش جەريانىدىلا بىر نەچچە قىتىم كۆرۇشكەن خالاس .رەھمەت سېزگە ئانامنى ئەسلەتتىڭز بۇ ئايال 40 ياشلار ئەپچۇرىسىدىكى ئوتتۇرا ياشلىق ئايال ئىدى ،

ئەرزىمەيدۇ مىنىڭ مۇ سىنىڭ دەك باللىرىم بار ،قىنى ئىتە ئوقۇپ بىرەيمۇ ؟

بولدى ھەدە سىزنى ئاۋارە قىلماي ،ھۆكۇم ۋاراقچىسىنى ئىلىپ كامىرغا قايتىپ كىردىم ،كامىرداشلىرىم خۇددىي مىنى قويۋەتكەندەك خۇشال بولۇشۇپ مىنى ئورىۋىلىشتى ئاداش قانداق ھۆكۇم قىپتۇ قويۋىتەمدىكەن يا ئىلاھىم شۇنداق بولغىيتى ئاداش سەن بولساڭمۇ بۇ ئازاپ دۇنياسىدىن قۇتۇلۇپ ،ئەركىن جەميەتنىڭ ئەركىن ھاياتنىڭ پەيزىنى سۇر سەڭ بىزمۇ خۇش بولىمىز ئاداش .دىيشىپ كۆڭلۇمنى ئازراق بولسىمۇ خۇشقىلاتتى .بۇجايدا بەلكىم مۇھەببەت بار دىسە باشقىلار ئىشەنمەسلىكى مۇمكىن ئەمما ھەقىقى مىھىر -مۇھەببەت .ھەم رەزىللىك مۇ شۇشۇ يەردە بار .قىنى دىگىنە ئاداش قانداق ھۆكۇن چىقرىپتۇ قويۋىتەمدىكەن ،بىلمىدىم دوسلار بىلمىدىم مەن تىخى ئۇنەرسىنى كۆرۇپمۇ باقمىدىم .

ھۆكۈمىڭ چىقماپتىما ۋاھ ئاداش تەلىيىڭ باركەن ئۆتكەندە بىرسىگىمۇ شۇنداق بولغان ھۆكۈم چىقماي قويۋەتكەنتى سىنىمۇ قويۋەتكۈدەك ئاداش .

ئىلاھىم شۇنداق بولغاي دۇسلار لېكىن ھۆكۈم چىقىپتۇ مانا ئۇنەرسە مەن كۆرۇشكە جۈرئەت قىلالمىدىم ،

ئەكىلە ئاداش مەنلا ئوقۇپ بىرەي دەپ قۇلۇمدىكى رەسۋا خالتامنى .ئۇيەردە سوت ھۆكىمىنى شۇنداق ئاتايمىز ئەمەلىيەتتە ئۇ مەن ئۇچۇن رەسۋا خالتىسى ئەمەس مىنىڭ كەگۈسۈمنى بەلگۇلەيدىغان قەغەز ئىدى .تارتىپ ئالدى كامىردىشىم ھەممىنىڭ يىقىن دۇستۇم شىر ئەلىي [ شىر ئەلىي بىلەن ھازىرغا قەدەر ئىزدىشىپ ئۆتىۋاتىمىز ]ئاندىن ئۇ ئۈنلىك قىلىپ ئوقۇشقا باشلىدى جاۋاپكار باھادىر مەلۇم ئاينىڭ مەلۇم كۈنى دوسلىرى بىلەن بىرلىشىپ شەھرىمىزنىڭ مەلۇم رايۇنىدا بۇلاڭچىلى قىلىش قەستە ئادام يارلاندۇرۇش ھەم ئۆمۇرلۇك مىيىپ قىلىپ قويۇشتەك جىنايەتلەرنى سادىر قىلغان گەرچە جاۋاپكار بۇلاڭچىلىق زەخمىلەندۇرۇش قاتارلىق جىنايەتلەرگە بىۋاستە قاتناشمىغان بولسىمۇ ،بۇ ئۇرۇش ماجرانىڭ باشلىنىشىغا سەۋەپچى بولغان شۇ تۇپەيلى يۇقارقىدەك جىنايەتلەرنىڭ سادىر بولشىدا مەسئۇلىيىتى بار دەپ قارلىپ دولىتىمىز ئاساسى قانۇنىغا ۋە جىنايى ئىشلار قانۇنىغا ئاساسەن توۋەندىكىچە ھۆكۈم چىقاردۇق .

جاۋاپكار باھادىر باشقىلارنى بۇلاش مىيىپقىلىپ قويۇش قاتارلىق جىنايەتلەرگە بىۋاستە قاتناشمىغان بولسىمۇ يۇقارقى جىنايەتلەرنىڭ سادىر بولىشىدا ،مەسئۇلىيىتى بار، يەنە جەميەتنىڭ ئامانلىقىغا زىيان يەتكۇزگەن دەپ قارلىپ تۇۋەندىكىدەك جازا بىرىلدى .جاۋاپكارغا مۇددەتلىك 4 يىللىق قاماق جازاسى بىرلىپ قوشۇمچە جوشۇمچە 3000يۈەن جەرىمانە قويۇلسۇن .ئەگەر جازا قايىل بولمىسا 15 كۈن ئىچىدە نارازىلىق ئەرزى بەرسە بولىدۇ .تامام .

4 يىل بۇ گەپنى ئاڭلاپ بىشىمغا بىرسى بازغان بىلەن ئۇرغاندەك تاسلا قالدىم ھۇشۇمدىن كەتكىلى ،نېمىشقا شۇنداق بولىدۇ 4يىل ئاڭلىماققا ئىتماققا ئاسان لېكىن 4 يىلنى بۇنداق جايدا ئەركىنلىكتىن مەھرۇم ھالدا تۈگۈتىپ بولماق ئاسان گەپ ئەمەس .تۇرۇپ دۇسلۇرۇمغا ئاچچىقىم كەلدى .خوپبوپتۇمۇ دىدىم ئۇلارمۇ 15 ؛10 .يىللىقتىن كىسىلگەن ئىدى .تۇرۇپ ئۇلارنىڭ بۇ ھالغا قىلىشىغا مەن سەۋەپچىدەكمۇ ھېس قىلىپ قالدىم ئەمەلىيەتتە ئۇلار ماڭا ياردەم قىلىمىز دەپ ئاخىردا ماڭا ۋە گۇلشەنگە زىيانكەشلىك قىلغاندى ،مەن يا دوسلىرىمغا بىز كەتكەندىن كىيىن ئۇ لۈكچەكلەرنىڭ پۇت -قوللىرىنى سۇندۇرۋىتىڭلار ،دىمىگەن ئۇ ئىككىسى ماڭا ياردەم قىلىمىز دەپ ئاقىۋەت مىنى ئەركەنلىكتىن يەنە ھاياتىم بولغان تۇنجى سۆيگۈنىم گۇلشەندى ئايرىۋەتتى .يەنە بۇيەرگە سۆرەپ كىردى ،بىچارە گۈلشىنىم ئەھۋالىڭ قانداقراقتۇ ،مەن بۇيەرگەكىرىپ قالغان دىن بىرى سەنبىلەن كۆرۇشۇپ باقمىدىم ،

بولدى ئاداش كۆڭلىڭنى يىىرىم قىلما دىدى كامىردىشىم شىر ئەلى دوسلىرىڭ غا ئاچچىقلانما ئۇلارمۇ تىگىشلىك جازاسىنى تارتىپتۇ تىخى سىنىڭ ئازكەن 4 يىل تىزلا ئۆتۇپ كىتىدۇ گۈلشەن بىلەنمۇ كۆرشەلەيسەن غەيرەتلىك بول ،تۈرمىگە بىرگە كېتىدىكەنمىز 4 يىللىق جاپانى سە بىلەن بىرگە تارتىمەن ئاداش جاپادىمۇ ھالاۋەتتىمۇ بىرگە بولىمىز ،ئەركىنرەك بول .

قانداق ئەركىن بولىمەن ئاداش سەن مۇشۇ يەردە ئەركىنلىك ھېس قىلىۋاتامسەن ،ياق ئاداش مەن ئەرىز قىلىمەن ،قايتىدىن سوت ئاچسۇن بۇ ئادىللىق ئەمەس مەن قايىل ئەمەس .

بولدى قىلغىن باھادىر سەۋىرقىل باشقىلارمۇ بىر ئىغىزدىن تەسەللى بەرگەنمۇ بولدى ،بۇيەردەە ئەڭ يىقىن ئادىمىم يەنىلا شىر ئەلى ئىدى بىز كۆپ سىردىشاتتۇق يەنە بىر ئورۇندا يىتىپ قوپاتتۇق .نۆۋەتچىلىكتىمۇ بىرگە تۇراتتۇق ،ئۇمۇ جازا ھۆكىمىنى ساقلاپ تۇراتتى ،شىر ئەلىمۇ مەسلىكىدە باشقىلار بىلەن ئۇرشۇش جەريانىدا بۇلاڭچىلىق شەكىللەندۇگەن دەپ قارلىپ قولغا ئىلىنغان ئىدى ،

بولىدۇ ئاداش ئەرىز قىلىپ باققىن بەلكى پايدىسى بوپقالار ،

قايتا ئەرىزمۇ بەردىم ،لېكىن نەتىجىسى ئوخشاش بولدى يەنىلا 4 يىل ئىككىنچى قىتىملىق جازا ھۆكىمىنى ئىلىپ 12 كۈن بولغاندا ئەمگەك بىلەن ئۆزگەرتىش مەيدانى يەنى {劳改队} گە يۆيكەپ ئىلىپ ماڭدى بىرگە ماڭغانلار قاتارىدا دوسلۇرۇم مىنى سۆرەپ كىىرگەن دوستلۇرۇم .كامىردىشىم ئەڭ يىقىن دوستۇم شىر ئەلىمۇ باتى ،ماششىنىدا كىتىپ بارىمىز ھىچكىم بىر ئىغىز گەپمۇ قىلمىدى پەقەت بىزنى يالاپ كىتىۋاتقان ساقچىلارلا بىزگە تۇرمە ئەھۋالىنى سۆزلەپ يۇل كۆرسىتىپ ماڭاتتى .تۇرمىگە بارغاندىن كىيىن قائىدە تۇزۇمگە رىئايە قىلىڭلا ياخشى ئىشلەپ كەچچۇرۇم ئىلىپ بالدۇرراق چىقىشىڭلارنى ئۈمۈت قىلىمىز ،ھەراس ئۇيەرگە بارغاندا تۈمە ساقچىلىرى قايسى يۇل بىلەن كەلدىڭلا بىلەمسىلە دىسە ياق بىلمەيمىز دەپ جاۋاپ بىرەسىلە بولمىسا سىلەرنى قاتتىق نازارە ئاستىغا ئالىدۇ ،بۇگەپلەرنىڭ ھېچقايسىىسى قۇلۇقۇمغا كىرمەيتتى قىچىپ كېتەيمۇ ؟بىچارەم، ھاياتىم ،گۈلشىنىم نىمقىلىۋاتقاندۇ ؟ئەھۋالى قانداقراقتۇ؟ مەنا توپ توغرا 6 ئاي بولدى سەن بىلەن كۆرشۇپ باقمىدىم ھالىڭ نىچۇك دەپ ئەھۋالمۇ سوريالمىدىم ،مىنى ئىزدەپ نەلەرگە بارغانسەن ؟،سەن سىز ياشىيالمايمەن سىنى يوقۇتۇپ قويۇشنى خالىمايمەن .ناۋادا مىنى تاشلاپ كەتسىڭىز ،چەتئەلگە چىكەتسىڭىزمۇ سىزنى ئىزدەپ تاپىمەن ،دەپ ۋەدىلەر بەرگەنتىڭ مىنى نەلەردە ئىزدەپ يۇرىدىغانسەن؟ جىنىم ،ياكى مەندىن رەنجىگنسەنمۇ؟ ياق سەن مەندىن رەنجىپ باقمىغان ھەم رەنجىپمۇ يۇرمەيسەن ،قانچىلىك ئازاپ تارتىۋاتقانلىقىڭنى بىلىپ تۇرۇپتىمەن ،6 ئايدىن بىرى سىنى ئويلىمىغان سىغىنمىغان بىر كۈنۈم بولمىدى بىرسىكىنۇت بولمىدى .ھەربىر تىنىقىم سىنى دەپلا سوقىۋاتىدۇ ،دىلىممۇ تىلىممۇ، سىنىڭ ئىسمىڭنى زىكىر قىلىۋاتىدۇ ،جىنىم مەن يىراققا كەتمىدىم مەن سىنىڭ يىنىڭدىلا .سەن مەنسىز ياشىيالمايسەن مەن مۇ ھەم شۇنداق ،تەغدىر بىزنى بىزنىڭ مۇھەببىتىمىزنى سىناۋاتىدۇ ، بىزگە بولمىغۇر چاقچاق قىلىۋاتىدۇ ،بەرداشلىق بەرگىن جىنىم بۇكۇنلەرمۇ تىزلا ئۆتۇپ كىتىدۇ مەن چوقۇم سەن ئۇچۇن ياشايمەن پەقەت سەن ئۇچۇن ›سۆزلىرىمنى ئاڭلاۋاتامدىغان سەن جىنىم نىدالىرىم ساڭا ئاڭلىنىۋاتامدىغاندۇ سەن چۇقۇم ئاڭلاۋاتىسەن ،ھەم ئاڭلىيالايسەن ،مەن تۈرمىگە بىرىپلا ساڭا خەت يازىمەن ئۆزۈڭنى كۆپ ئاسرا خەيىر ئاللاھتىن ساڭا ئامانلىق تىلەيمەن .!
