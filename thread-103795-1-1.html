    ئەسسالامۇئەلەيكۇم مۆھتىرەم قېرىنداشلار ھەممىڭلارنىڭ ئەزىز تېنى سالامەت ، ئائىلەڭلار خاتىرجەم  ، تۇرمۇشۇڭلار بەختلىك بولسۇن !!!
گەرچە مەن ساخاۋەتچى بۇلالمىساممۇ ، لىكىن  ئۆزەم تۇرۇشلۇق ئىلى ئوبلاستىغا قاراشلىق ئىككى شەھەر ، سەككىز ناھىيە تەۋەسىدىكى  ئىقتىسادى قىيىنچىلىق سەۋەبىدىن جىددى ئاغرىقلارنى داۋالىتىش ئۈچۈن  قىيىنچىلىقتا قىلىۋاتقان ئاجىز كىشلەر توپىنىڭ نىداسىنى  تور ئارقىلىق باشقا قېرىنداشلارغا يەتكۈزۈش ئارقىلىق ئاق كۆڭۈل ، ساخاۋەتچى كىشلىرىمىزنى جىددى ياردەم ۋە قۇتقۇزۇشقا مۇھتاج كىشلىرىمىزگە ئىللىقلىق يەتكۈزۈشكە چاقىرىق قىلىشىنى يۈكسەك بۇرچ ۋە مەسئولىيەت بىلىپ كىلىۋاتىمەن .
       بۈگۈنكى بۇ قېرىندىشىمىز بولسا ئىلى ئوبلاستى قورغاس ناھىيە چىڭسىخوزا بازىرى تەڭرىتاغ يولى 205-قورادا ئولتۇرۇشلۇق مەلىكە ئادىلخۇجا(ئايال، قازاق ، 34ياش ، يولدىشى ئۇيغۇر ) بۇلۇپ ، بۇ ئاجىزە كىچىكىندىنلا ئاتا-ئانىسىدىن يىتىم قالغاندىن كىيىن ، تەقدىرنىڭ تۈرلۈك ئاچچىق-چۈچۈكىنى كۆپ تارتقان ، كىيىن ئۇيغۇر يىگىتى پۇلات بىلەن توي قىلىپ ھالال تەرىگە تايىنىپ ئەمدىلا تۇرمۇشى خاتىرجەم بۇلۇپ ھايات جىلۋىسى پارلىغان ئىللىق ئائىلىسىدە تۇيۇقسىز ئۇچۇق ئاسماندا چاقماق چاققاندەك مەلىكە جىددى خاراكتىرلىك ئاق قان راكىغا گىرىپتار بۇلۇپ قېلىپ دەسلەپ شىنجاڭ تىببى ئونۋېرىستىتى قارمىقىدىكى 1-دۇختۇرخانىدا داۋالانغان بولسىمۇ يۇقىرى داۋالىنىش خىراجىتىنى تۆلەش تۈگۈل كۈندىلىك خىراجەت قىلغۇدەكمۇ ئىقتىساد بولمىغانلىقى ئۈچۈن جىددى ئوپىراتىسيە قىلمىسا ھاياتىغا خەۋپ يىتىش ھالىتىدە تۇرسىمۇ ، ئىلى قازاق ئاپتۇنۇم ئوبلاستلىق دوستلۇق دۇختۇرخانسىغا كىلىپ ئادەتتىكى تاشقى كىسەللىكلەر بۆلۈمىدە داۋالانماقتا ، ئاجىزە كىسەل كارۋىتىدا ئاغرىق ئازابىدا پۇچىلانسا ئۇماق پەرزەنتى مىھرىبان ئانىسىغا تەلمۈرۈپ يىغلىماقتا ، يولدىش پۇلات ئۆزىنىڭ نامراتلىقىدىن ئايالىنى داۋالىتالمىغانلىقى ئۈچۈن ئاھ ئۇرماقتا ، ئەزىزى قېرىنداشلار بۇ ئاجىزە ئۈچۈن ، ھىچ بولمىغاندا ئۇنىڭ ئۇماق پەرزەنتى ئۈچۈن بولسىمۇ ئاز تۇلا ياردەم قۇلىمىزنى سۇنۇپ قۇيايلى ، ئاللاھ ھەممىزنى ئۆز پاناھىدا ساقلىسۇن ، بۇ ئاجىزەنىڭ كىسىلىگە شىپالىق بەرسۇن !
      مەلىكە ھازىر ئىلى قازاق ئاپتۇنۇم ئوبلاستلىق دوستلۇق دۇختۇرخانسىغا ئادەتتىكى تاشقى كىسەللىكلەر بۆلۈمىدە داۋالىنىۋاتىدۇ ، تۈۋەندىكىلەر  مەلىكەنىڭ كىسەللىك دىئاگنۇز قەغىزى ۋە مۇناسىۋەتلىك ئىسپات ماتىرىياللىرى
