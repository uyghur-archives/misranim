                                           سەن بار يەرلەر.........

                          سەن بار يەرلەر گويا بىر جەننەت،
                          ئوخشىمايدۇ ھىچقانداق جايغا.
                          شۇڭا  دائىم  بارغۇم  كىلىدۇ،
                         ھەر ۋاقىت يار سېنىڭ يىنىڭغا.


                        سەن بار يەرلەر باھار ھەمىشە،
                        زىمىستانمۇ كەلمەيدۇ مەڭگۇ.
                        بولمىساڭ گەر يايرىم يىنىمدا،
                        زىمىستانمۇ كەتمەيدۇ مەڭگۇ.


                       يايرىم سەن بار باغلارغا كىرسەم،
                       قۇشلارمۇ ھەم چۈشۇپتۇ  ناۋاغا.
                       جامالىڭغا  قارىدىم مەن  ئۇزاق،
                       تامام ئەسىر بولدۇم مەن ساڭا.


                     سەن بار يەرلەر بولۇپ سەيلىگاھ،
                     ئۆزگىلەرنى قىپتۇ مەھلىيا.
                     <<سەيلىگاھ>>قا يىغىلغانلار ھەم،
                     ئەگىپ كىتەلمەي قاپتۇ ئۇندا.

                    باراي دەيمەن سەن بار يەرلەرگە،
                    شۇنچە يىراق بولسىمۇ تامام.
                    دىدارىڭغا بىر قىتىم قېنىپ،
                    ھەممىلەرنى ئۇنتۇپلا كەتسەم.


                                      سەن يوق


                     ھاياتىمغا ھەمراھ ئىدىڭ،
                     ئازابىمغا داۋا ئىدىڭ.
                     خۇشاللىقىمغا ھەم جور ئىدىڭ،
                     يىللار ئۆتتى يىنىمدا سەن يوق.


                   سەن كىتىپ، قالدىم يەككە ــ يىگانە،
                   ھاياتلىق يولۇمدا يوقتۇ نىشانە.
                   ئۇزۇلدى ئۇمۇد ھەممىدىن يەنە،
                   يىللار ئۆتتى يىنىمدا سەن يوق.


                   ئاي چىھرىڭگە كۈلۇپلا باققان،
                   كەڭ قۇچاقنى ساڭىلا ئاچقان.
                   مۇھەببىتىم پارلاق نۇر چاچقان،
                   يىللار ئۆتتى يىنىمدا سەن يوق.


                   مۇھەببەتتە باردۇر داۋان،
                   بىز داۋاندىن تەڭلا ئاشقان.
                  ھەممە بىزگە بەخىت تىلىگەن،
                  يىللار ئۆتتى يىنىمدا سەن يوق.


                  بۇلبۇل گۇلگە ئاشىق بولغاندەك،
                  كاككۇك زەينەپنى چاقىرغاندەك.
                  ئۆتكەنىدۇق مەڭگۇ دۇنيادا بىردەك،
                  يىللار ئۆتتى يىنىمدا سەن يوق.


                  ھەر قېتىم ئۆتسەم قاراپ قالىمەن،
                  بىللە ئولتۇرغان ئاشۇ ئورۇنغا.
                  ھاياتنىڭ باھارى مۇھەببەت ئىكەن،
                  يىللار ئۆتتى يىنىمدا سەن يوق.


                  مۇھەببەت دىدۇق،يەنە شۇ مۇھەببەت،
                  ھاياتىمدىن كەتكىن،كەلمىگىن پەقەت.
                  ئەسلىملىرىم قىپ قالسۇن مەڭگۇ تائەبەت،
                  يىللار ئۆتتى يىنىمدا سەن يوق.........


                                    سەۋدايىلىق

                    سەۋدايىلىق ئىلكىدە ئازاپلار تارىتتىم،
                     ئۆتكەنكى ئېشلارنى ئەسكە ئالدىم.
                     مۇھەببەت يولىغا كىچىكلا كىردىم،
                     ئېشەنگىن پەقەتلا ئۇنتۇيالمىدىم.


                  قاراشلىرىڭ يۇرەكنى ئىزىپلا قويدى،
                  ئەسلىملەر قوينىغا باشلاپلا كىردى.
                  ھەممىلەر بىزلەرگە قاراپلا قالدى،
                  ئېشەنگىن پەقەت ئۇنتۇيالمىدىم.


                  سەۋدا بولۇپ سىنىڭ ئوتۇڭدا،
                  كۆيۇپلا يۇردۇم يار پىراقىڭدا.
                  تاشلاپلا كەتتىڭ مېنى يىراققا،
                  ئېشەنگىن پەقەت ئۇنتۇيالمىدىم.

                              *                                   *                    *

                    سەۋدالىق ئىلكىدە ئۆتتى كۈنلىرىم،
                    كۈتۇپ سىنى ھەر كۈن،ھەر تۇنى.
                    سەۋدالىق ئىلكىدە تەلمۇرۇپ يۇرۇپ،
                    سۆيۇپ يۇردۇم خۇشۋاخ ھىدىڭنى.


                    ئېشقىڭدا چۆللەرنى كىزىپلا يۇردۇم،
                    بىر تامچە سۇيۇڭگە ئىنتىزار بولۇپ.
                    نە ئىلاج سەن ئۆزۇڭ تاشلاپلا كەتتىڭ،
                   بۇ پاك قەلىبنى ئازاپتا قويۇپ.


                   كۈنلەرنى سەنسىز ئۆتكۇزەلىسەممۇ،
                   تۇنلەرنى سەنسىز ئۆتكۇزەلمىدىم.
                   ئازاپتا قويغان سەندەك بىرسىنى،
                   ئېشەنگىن پەقەتلا ئۇنتۇيالمىدىم.


                                  مەنبە: ئۆز قەلىمىم.
                                    يوللىغۇچى: توغراق شاھزادە.

                        زىمىستانمۇ كەلمەيدۇ ئەبەد .

                       سەن بار گۈزەل  باغلارغا كىرسەم،

كىمدۇر قىسسا شەيتان كۆزىنى .

                    باراي دەيمەن سەن بار يەرلەرگە،
                    شۇنچە يىراق بولسىمۇ تامام.
                    دىدارىڭغا قېنىپ ھەسسىلەپ
                    يۈرىكىڭدە ئالسامكەن ئارام .
