ما تىمىنى كۆرمىسەم بوپتىكەن .ئۇيغۇر يۇقالمايدۇ قاراڭ .لىكىن مىللەت يۇقىلىدۇ ،ىسز مىللەتلەرنىڭ ئاسمىلاتسىيەلىششى دىگەن گەپنى ئاڭلىغانمۇ ؟ۋېيكو ،تاكو خەنكو ،خاكو.......دىگەن .«民考汉»دىگەن بۇ گەپنىڭ نەخ تەرجىمىسى يەنە ئۇيغۇرچە ئاتىلىشى «خەنزۇۋان »دىگەن بۇلىدۇ ،باشقا قىزغىز ،قازاق ،تاجىك ....لارمۇ ئۇنداقلارنى «خەنزۇۋان »دەپ ئاتايدۇ .بۇ دىگەن ئۆزى مىللى ،تىلى خەنزۇتىلى .يەنى ئانا تىل بىلمەيدۇ دىگەن گەپ .لىكىن ھازىر نۇرغۇن خەنزۇۋانلار ئۆزلىكىدىن ئانا تىلنى ئۆگىنىشكە باشلىدى .سىزنىڭ بۇ تەپەككۇرىڭىز يەنە باشقىلارنى قايمۇقتۇرىدۇ قاراڭ .خەنزۇۋانلار ھەرگىزمۇ ئۆزىنىڭ ئارۇسى بۇيچە خەنزۇۋان بولغان ئەمەس بەلكى سىز،بىزگە ئوخشاش ئاتا-ئانىلارنىڭ زورى بىلەن شۇنداق بولغان .سىزنىڭ يەتتە ياشقا كىرگەن بالىڭىز مەن ئۇيغۇرچە ئوقۇيمەن ،خەنزۇچە ئوقۇمايمەن دەپ تەلەپ قۇيالامدۇ ،ئۇنداق ئەمەس ،ئاددى مىسال«سۇ بۇىيدا »رۇمانى قاچان ئۇيغۇرچىغا تەرجىمە قىلىنغان ؟شۇ ۋاقىتتا خەنزۇۋانلار ھازىرقىدەك كۆپمىدى ،ھەرگىز ئۇنداق ئەمەس ،لىكىن دۇنيانى ھەيران قالدۇرغىدەك تەرجىمە بولغان ،ئۆزبىك ،قازاق ....ۋە باشقا تىللارغا «سۇ بۇيىدا »رۇمانى خەنزۇچىدىن ئەمەس ئۇيغۇرچە تەرجمىسىدىن پايدىلىنىپ تەرجىمە قىلىنغان .بىزنىڭ بالىلىرىمىز كچىكىدىنلا خەنزۇتىلىنى ئۆگەنمىسمۇ يەنىلا ئۆگىنىپ كىتەلەيدۇ ،شۇڭا پەرزەنتىللىرىمىزدە ئەڭ ئالدى بىلەن ئۇيغۇرچە تەپەككۇر بۇلۇشى كىرەك ،ئۆزۈمنى مىسالغا ئېلىپ ئېيتسام ئوتتۇرا تېخنىكۇم مەكتىپىگە كەلگەن ۋاختىمدا ئىسمىمنى خەنزۇچە يازالمايتتىم ،ئۈچ يىل ئۇقۇش جەريانىدا خەنزۇتىلىنى شۇنداق ياخشى ئۆگەندىم ،ئاخىردا خىزمەتچى قۇبۇل قىلىش ئىمتىھانىمۇ پۈتۈنلەي خەنزۇچە بىرىپ لايەقەتلىك ئۆتتۈم .بۇ يەردە ئۆزۈمنى ماختىماقچى ئەمەسمەن ،لىكىن شۇنى دىمەكچىمەنكى بىزنىڭ پەرزەنتىللرىمىز نىمىنىڭ مۇھىم ئىكەنلكىنى ئۆزلىكىدىن تۇنۇپ يەتكەندە ناھايتى تىزلا ئۆگىنۋالىدۇ ،ھازىرقى جەمئىيەتتە خەنزۇتىلى ئۆگىنىش ئىنتايىن مۇھىم ،بۇنى ھەممەيلەن بىلىمىز ،باشلانغۇچتا ھەتتا ئۇيغۇرچە ساۋادىمۇ تۇلۇق چىقماي مەكتەپتىن چىقىپ كەتكەن كىشلىرىمىز ،بىر ئۆمۇر دېھقانچىلىق بىلەن شۇغۇللىنىپ ،مەكتەپ يۇزى كۆرۈپ باقمىغان دېھقانلىرىمىز تۇرمۇش سەۋەبلىك شەھەرگە ينىپ چىقىپ ،خەنزۇلار بىلەن ئارلىشش جەرياندا شۇنچە تىز خەنزۇچە سۆزلەشنى ئۆگىنۋالدى .بۇ ئەمەلىيەت ،شۇنداقكەن ،بىز يەنە نىمىدىن ئەنسىرەيمىز .بۇنىڭ ئەكىسچە بىز مەجبۇرى ئۆگىنىشنى تەلەپ قىلىۋاتقان بالىلىرىمىز خەنزۇچىنى قانچىلىك ئۆزلەشتۈرۋاتىدۇ .بۇ ھەممىزگە ئايان .شۇڭا ئۇنى -بۇنى يوللاپ ۋاختمىزنى ئىسراپ قىلماي .«ئاناتىل »دەپ ئاۋازىمىزنى قۇيۋىتىپ توۋلايلى .ھازىرقى قوشتىلمۇ ئەسلى تۈپ يىلتىزىدىن ئانا تىل ئۇقۇتۇشىنى ئاساس خەنزۇتىلى ئوقۇتۇشىنى قۇشۇمچە قىلغان مائارىپ ،پەقەت يەرلىك سىياسەتچىلەرلا بىزنىڭ پەرزەنتىللىرىمىز «ئانا تىلنى بىلمىسە مەيلى لىكىن خەنزۇتىلىنى چۇقۇم بىلىشى كىرەك »دەپ «قوشتىل »مائارىپىنى «تاق تىل »مائارىپىغا ئايلاندۇرۇپ قۇيىۋاتىدۇ .قانۇن ئۆگىنەيلى
