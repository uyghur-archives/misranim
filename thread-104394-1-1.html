ئەسسالامۇ ئەلەيكۇم تورداش تۇرىنىڭ بارلىق ئەزىىز ئىشتىراكچىلىرى قانداق ئەھۋالىڭلار ؟ سالامەت تۇردۇڭلارمۇ ؟

مانا تۈرمىگە كېرگىلىمۇ ھەش -پەش دېگۈچە 2 يىل ئۆتۇپتۇ ،بۇ 2 يىلدا نۇرغۇن نەرسىلەرگە قىيىن قىستاق سوئال سوراقلارغا ھەتتا مەن ئەزەلدىن نامىنى ئاڭلىساممۇ تىنىم شۈركىنىدىغان كېرىپ قىلىشنى ئەسلا خالىمايدىغان بۇ ھاياتى دوۋزاخ تۈرمىگىمۇ كۆنۈپ قالدى .ئادەم ھەممىگە كۆنىدىكەن ،گاھ خالاپ، گاھ مەجبۇررىي دىگەندەك .ئەركىنلىكتىن مەھرۇم قىلىش،قاتمۇ -قات بىسىم، تۈگىمەس يوقلىماقىلىنىش ،ئىسسىق -سوغۇق دىمەستىن ئىلىپ بىرىلغان ھەربىي مەشىق،ئىغىر ئەمگەك ،جاپالىق تۇرمۇش، بەلكېم كونىلارنىڭ ئېت كۆنمىگەن تۇرمۇشقا ،ئادەم كۆنىدۇ دىگىنى راست بولسا كېرەك ؟

مانا بىزمۇ كۆنۈپ قالدۇق. ۋاقىتنىڭ ئۆتىشى ئادەمنى ھەممىگە كۆندۈردىكەن. ۋەھشىي يىرىتقۇچ ھايۋان شىر ،يولۋاسلارنىمۇ قەپەسكە سولاپ قۇيسا يوۋاشلاپ قالغانغا ئوخشاش.جاھاننىڭ شاھى ،جاھا ئەركىسى ،ئىنسانىيەتنىڭ باشقۇرغۇچىسى بولغان ئەركەك زاتىنىمۇ سولاپ قويسا يوۋشلاپ قالدىكەن .خۇددىي قايسى بىر ناخشىدا ئىتىلغانغا ئوخشاش،جاھانغا پاتمىغان سەگە قازانغا قانداق پاتقانسەن ،قازانغا پاتساڭ پاتقانسەن تۇۋاقنى قانداق ياپقانسەن دىگەنگە ئوخشاش .جەمىيەتتە نىي -نىي قىنىغا پاتماي يوۋاشنى بوزەك قىلىپ ياماندىن چاچقان مۇشۇكتىن قورقۇپ يۇرگەندەك ياشىغان نوچىلارمۇ بۇ جايغا كېرسە يوۋاشلاپلا قالدىكەن مەنغۇ جەميەتتە باشقىلارنى بوزەك قىلمىغان دادىل ياشىغان ئەزەلدىن مەردانا ئىدىم راستنى راسىت ،يالغاننى يالغان پىتى راتسچىللىق بىلەن ياشاشنى ھاياتلىقتا ياشاش مىزانىم قىلىپ ياشىغان . تۈرمىگە كېرگەندىن كېيىنمۇ شۇنداق ياشاپ كەلدىم ئەپسۇس مىنىڭ راسچىللىقىم دائىم مىنى پۇتلاپ كولدۇرلىتىپ كەلدى ،يەنى ئۆز پۇتۇمغا چۈشەك بولۇپ كەلدى.لېكىن راستچىللىقمدىن ئەزەلدىن پۇشايمان قىلمىدىم .مىنىڭ راستچىللىقىم ھامان بىركۈنى مىنى بەخىتكە ئىرىشتۈرەتتى . ئەپسۇس يەنە شۇ راستچىللىقىم مىنى كچىكلا مىھرىبان ئاپامدىن ئايرىۋەتتى ،ماڭا بولمىغۇر چاقچاق قىلدى ،ئائىلىدە بولسا ئاتامنىڭ مۇھەببىتەگە ئىرشەلمىدىم، جەميەتتە جەميەتنىڭ مۇھەببىتىگە ئىرۈشتۈرمىدى ،ئەمدى گۈلشەننىڭ مۇھەببىتىگە ئىرىشىپ ھاياتنىڭ پەيزىنى سۈرەي دىگەندە بۇمۇ يوق بولدى ،تەقدىرىم مېنى يەنە بىرقىتىم ئالدىدى كولدۇرلاتتى ، يەنە كېلىپ مىنى ئەركىنلىكتىن ۋاقىتلىق مەھرۇم قىلدى، ئەمما مەن ھىلىھەممۇ راستچىللىقتىن ۋازكەچمىدىم ھەم مەڭگۈ شۇنداق قىلىمەن !

تۈرمىگە يىڭى كېرگەن چاغلىرىم ھىلىمۇ ئىسىمدە ،تۈرمە ساقچىلىرى بىزنى، قاماقخانا ساقچىلىرىدىن ئۆتكۈزۋالغان شۇكۈنى ،بىز تۆتىمىزنى ،ما ئىسىملىك دەرىزە كۆز ‹管柱›يەنى مەسئۇل خادىم دىگەن مەنىدە بىلەن ئەركىن ئىسىملىك ‹大组长› يەنى چوڭ دىجۇرنىي ياكى باشقۇرغۇچى جىنايەتچىنى تۈرمىدە شۇنداق ئاتايدۇ ،قاتارلىقلار ئۆتكۈزۋالغاندىن كېيىن ھەممىمىزنى ‹‹晾衣房 ›› كېيىم قۇرتۇش ئۆينى كۆرسىتىدۇ]دا ھەممىمىزنى ئانىدىن تۇغما قىلىپ ‹‹检查››تەكشۇرۇش مەنىسىدە تەكشۈرۈپ بولغاندىن كېيىن ما ئىسىملىك گۈەنجۇ سوئال سوراشقا باشلىدى .ئىسمىڭ ؟شىرئەلىي بۇ قاماقخانىدا تۇنۇشقان ھەمدە ماڭا كۆپ ياردەمدە بولغان دوستۇم شىرئەلىي ئىدى ،جىنايىتىڭ ؟ بۇلاڭچى ،يىشىڭ؟19ياش،شۇنداق سۇراق نۆۋتى ئەڭ ئاخىردا تۇرغان ماڭا كەلگەندە قانداق جاۋاپ بىرشىمنى بىلمەيتتىم،يەنە كېلىپ مەن خەنزۇچىنى بىلمىگەچ ئۇنىڭ نىمە دەۋاتقىنىنى نەدىن بىلەي ،ئاخىردا ئەركىن دازۇجاڭ تەرجىمان بولدى ،

ھە گۇناھسىز ئادەم تۈرمىگە نىمىشقا كېردىڭ دەيدۇ ؟

مەن باشتى ئاياق بولغان ئىشلارنى بايان قىلدىم .ئەركىنمۇ ئىرىنمەي تەرجىمە قىلىپ تۇردى .

سەن جىنايىتىڭگە قايىل ئەمەس شۇنداقمۇ دەيدۇ ؟

ئەلۋەتتە قايىل ئەمەس مەن نىمە گۇناھ قىلدىم ؟ئەجەبا مېنىڭ ئۆز لايىقىمنى قوغداش ،ئۇنى ئاسراش،ئۇنىڭغا كۆيۈنۈش ھوقۇقۇممۇ يوقمۇ ؟ئۇنى قوغدىيالمىسام ،ئۇنى قانات ئاستىغا ئىلىشقا ،ئۇنى ئاسراشقا جۇرئەت قىلالمىسام،ئۇنداقتا مىنىڭ ياشىغىنىمنىڭ نىمە ئەھمىيتى بولسۇن .ئىتىڭە باشلىق بىزنىڭ ئۆز ئادىمىمىزنى قوغدىيالمىساق ئۇنى ئاسراش ھوقۇقىمىزمۇ بولمىسا .ئۇنداقتا بىزنى ئەركەك دىگىلى بولامدۇ؟ بىزنىڭ نېمە رولىمىز قالىدۇ ؟ئوز ئادىمىمىزنى يەنى كەلگۈسىدىكى ئايالىمىزنى قوغداش ھوقۇقۇم بولمايدىكەن مېنىڭ ياشىغىنىمنىڭ نېمە زۆرۈريىتى ؟

‹‹把嘴关掉 ››يەنى بولدى بەس ياكى ئاغزىڭنى يۇم دىگەن مەنىدە ،{带他床上压过去}يەنى ئەچىقىپ كارۋاتقا بىسىۋەت ياكى قامىۋەت دىگەن مەنىدە .

،شۇنداق قىلىپ قاتتىق باشقۇرۇشقا قامالدىم .شۇ قامالغانچە ساق 3 ئاي تۈرمىگە كىرىش تەربىيسىنى قاتتىق باشقۇرۇشتا ئۆتكۈزدۈم.قاتتىق باشقۇرۇش كامىردا مەن بىلەن قوشۇپ13 نەپەرھەرخىل خاتالىق بىلەن قامالغانلار بار ئىدى .بىز باشقا جىنايەتچىلەرگە قارىغاندا ئالاھىدە باشقۇرلاتتۇق .ئەتتىگىنى باشقا جىنايەتچىلەردىن 1 سائەت بالدۇر ئورنىمىزدىن تۇراتتۇق . بىز ئورنىمىزدىن تۇرۇپ تازلىقىمىزنى قىلىپ ئۈگۈنۈشكە چۈشكەندە ،باشقا جىنايەتچىلەر شىرىن چۈش كۆرشۇپ ياتاتتى.تازلىق قىلىش ئورنىدىن تۇرۇش ۋاقتى بەك قىسقا يەنى پەقەت 10 مېنۇت بولۇپ بۇ ۋاقىت شەخسى تازلىق ،ياتاق تازلىقىنى ئۆز ئىچىگە ئالاتتى .مۇشۇ 10 مېنۇتتا تازلىقنى قىلىپ بولالمىساق ،يىرىم كۈنلەپ ئۆرە تۇرۇش جازاسى بىرلەتتى .تەخمىينەن باشقا جىنايەتچىلەر ئورۇنلىرىدىن تۇرشۇپ تازلىقىنى قىلىپ ،تاماق ۋاقتى بولغۇچە ،بىز باشقىلاردىن 2 سائەت بالدۇر تۇرغان بولاتتۇق .تامىقىمىزمۇ باشقىلاردىن پەرىقلىنەتتى .باشقا جىنايەتچىلەر ،سەي مۇما ،ياكى نان چاي .ئچسە ،بىزگە ،مۇما بىلەن لۇبۇ يەنى تۇرۇپتا قىلىنغان سوغۇق سەي بىرلەتتى ،ئىشقىلىپ بىزنىڭ ھەركۈنى ئەتتىگەنلىك تامىقىمىز تۇرۇپتا قىلىنغان خام ئاچچىق سوغۇق سەي بولاتتى .يىمەي تۆكۈۋىتىشكە قەتئىي بولمايدۇ .تۈرمە دىگەن بۇ جايدا ھەممە جىنايەتچى بىر- بىرنىڭ پىيدىلا يۈرشىدۇ بىرسىنىڭ خاتالىقىنى مەلۇم قىلسا ،مەلۇم قىلغۇچىغا مۇكاپات بىرلىدىغان بولغاچ چوڭراق ئاۋازدا گەپلەشكىلىمۇ بولمايدۇ دىگەن گەپ ، ناۋادا يىمەي تۆكىۋەۋتىڭىزمۇ بولدى تاكى گۇناھىڭىزنى تونۇپ سىزگە مەسئۇل كادىر تەسىرلەنگۈدەك تۆۋە نامىدىن بىرنى يىزىپ چىقمىسىڭىز تاماق يوق، نەچچە كۈنلەپ ئاچ قالىدىغان گەپ، بەرىبىر دىگەن يىرگە ئەكىلىدۇ .ئەگەر باشقا جىنايەتچى كامىردىشىڭىزنىڭ سىزگە ئىچى ئاغرىپ تاماق بىرىپ قويدىمۇ بولدى، بىچارە ئۇمۇ بىردەك قوشۇلۇپ سىزدىنمۇ بەكرەك جازالىناتتى .ئامال يوق بىرسىنىڭ ئازاپ چىكىۋاتقىنىنى كۆرۈپ تۇرۇپمۇ ياردەم قىلغىلى بولمايدۇ .چۈشلىكى باشقا جىنايەتچىلەر ،‹‹自由活动›› يەنى ئەركېن پالىيەت مەنىسىدە ،قىلسا بىز ياكى ياتاقتا ئۈگۈنۈش قىلاتتۇق ،ياكى سىرىتتا سوغوقتا 11ئاي مەزگىلى خلى سوغوق بولىدۇ ھەربىي مەشىق قىلاتتۇق .مەشىق قىلسىغۇ ئادەم توڭلىمايدىكەن لېكىن داق سىمېنۇت يەردە رايىپلاردەك ئولتۇرغۇزۇپ قويدىغان بولسا سوغوق ،ئۇ جىنىمىزدىن ، بۇ جىنىمىزغا ئۆتۇپ كېتەتتى .ئۇنى ئازدەپ سازايى قىلغاندەك ،كىېيگەن كېيمىمىزنىڭ دۈمبى قىسمىغا قىزىل خەتتە يوغان قىلىپ ‹1级严管罪› يەنى 1- دەرىجىلىك قاتتىق باشقۇرۇش دىگەن مەنىدە .خەت چاپلاپ قويلاتتى .كەچلىكى باشقا جىناتەتچىلەر 9،9.30 .10 لاردا‹ 点名›يوقلىما قىلىنىپ يىتىش بۇيرىقى بىرىلسە بىز قاتتىق باشقۇرۇشتىكىلەر ياكى ياتاقتا .يابولمىسا ‹‹学习班›› يەنى ئۈگۈنۈش سىنىپىدا ئارلىقىمىز 1 مىتىردىن ئارلىق قالدۇرلۇپ ئۈگۈنۈش قىلاتتۇق ،ئۆز- ئارا پاراڭ سىلىشقا بولماتتى .پاراڭسىلىشقىنىمىزنى بەنجاڭلار ھەربىر ئەترەتتە 6 دا زۇجاڭ بولۇپ 2 - مەخسۇس بىزگە قاراشقا مەسئۇل ئىدى .كۆرۈپلا قالسا كادىرلارنىڭ رۇخسىتىسىز كىلىپلا ئۇرۇپ كىتەتتى ياكى بولمىسا نەچچە سائەتلەپ يەنە ئۆرە تۇراتتۇق يىتىش ۋاقتىمىزكەچلىكى2 سائەت كىيىن بولاتتى .بۇ قاتتىق باشقۇرۇش ئەترىتى بىلەن يىڭىدىن تۈرمىگە كېرگەنلەرنى باشقۇرۇش ئەترىتى بىر بولسىمۇ بىز ئايرىپ باشقۇرلاتتۇق .قاتتىق باشقۇرۇش ئەترىتىگە ئاساسلىقى ھەر قايسى ئەترەتلەردىن ، قىچىشقا ئۇرۇنغان ،ئۆلىۋالماقچى بولغان .كادىرلار بىلەن تەڭ تۇرغان ،پېتنە -ئىغۋا تارقاتقان ،ناماز ئوقۇغان ،شايكا تۇپلاپ يۇرتۋازلىق قىلغان ،ئاگاھلاندۇرۇش سىزىقىدىن چىقىپ كەتكەن ،باشقىلار بىلەن قەستەن شۇقۇشقان قاتارلىق ئىشقىلىپ قىلمىشى ‹‹ئا›› دەرىجىسىگە تۇشقانلارنىڭ ھەممىسى مۇشۇ قاتتىق باشقۇرۇش ئەترىتىگە يۆتكەپ ئاگاھلاندۇرلۇش بىرلەتتى .بۇنداق ئاگاھلاندۇرۇش ياكى ،قارڭغۇ ئۆيگە قامالغانلار ئاساسلىقى ياتاق باشلىقى دىجۇرنى دىگەندەك تۇرمىدە ئەمەل تۇتقان ئامالسىزلىقتىن جازا يۇرگۈزۈلگەنلار بولاتتى ،بولمىسا بىردەك قاتتىق باشقۇرۇشقا يوللىناتتى .مەن ساق ئىككى يىلدا 3 قىتىم قاتتىق باشقۇرۇشقا يوللاندىم ئەڭ ئاز بولغىنى 1 ئايلىق جازا . 8 قىتىم قاراڭغۇ ئۆينى تاماششا قىلدىم ئەڭ ئۇزۇن تاماششا قىلغان ۋاقتىم 1 ئاي بولدى ماڭا ئاساسلىقى 小تۇرغۇن ئىسىملىك ئۆرۇمچىلىك دۇيجاڭ ئار بولاتتى . شۇ تۇرغۇن دۇيجاڭنىڭ پايدىسىدىن ماڭا يىل قوشۇلۇپ قالمىدى بۇيەردە شۇ دۇيجاڭغا ئالاھىدە رەھمىتىمنى بىلدۈرۈپ ئۆتۇپ كىتىمەن سىزنىڭ تۈرمىدە ماڭا كۆپ ياردەمدە بولغىنىڭىزنى مەڭگۈ ئۇنتۇپ قالمايمەن !قاراڭغۇ ئۆيگە قامالغان ،ياكى قاتتىق باشقۇرۇشقا يوللانغان ھەرقانداق بىر جىنايەتچىنىڭ ئۆيدىن كەلگەن خەت -خالتىلىرى ،يوقلاپ كەلگەن ئۇرۇق -تۇققانلىلىرى بىردەك بىرىلمەتتى ھەم كۆرۈشتۈرۈلمەتتى .

لېكىن مەن قاتمۇقات جازالانغىنىم ئۈچۈن كادىرلار تۈزۈلۇپ قالىدۇ دەپ قاردىمۇ ئەيتاۋۇر مىنىڭ خەت خالتامنى يەتكۈزۈپ برىپتۇردى .بىر قىتىم دادام ۋە ئاچام يوقلاپ كەلدى، يەنە بىرقىتىم گۈلشەن يوقلاپ كەلدى ، بۇ 2 يىل جەريانىدا قانچە قىتىملاپ كادىرلار بىلەن ‹‹教育课››تەلىم تەربىيە بۆلمىدىكى ئايال ساقچىلار بىلەن سۆھبەت ئىلىپ بارغىنىمنى بىلمەيمەن .ئىشقىلىپ ھەپتىدە 3-4 قىتىملاپ ئاز بولغاندىمۇ 2 قىتىملاپ سۆھبەت ئلىپ باراتتىم .ئەترەت رەھبەرلىرى بىلەن ئىلىپ بارغان سۆھبەت بۇنىڭ سىرتىدا .ئۇلارنىڭ ۋەزىپىسى بىز جىنايەتچىلەرنى ئۆزگەرتىش بولغاچقا ئىرىنمەي تەربىيە قىلاتتى .ھەرقىتىم سۆھبەت ئىلىپ بارسام بىر قەدەر ياخشى بولۇپ قالاتتىم .شۇنىڭ ئۈچۈن بولسا كېرەك قاتتىق باشقۇرۇشتا پەقەت مىنىڭلا خەت -خالتام ۋاقتىدا قولۇلغا تىگىپ تۇراتتى .مەن تۇنجى قىتىم قاتتىق باشقۇرۇشتىن چىقىپ باشقا ئەترەتكە بۆلۇنگەن كۈنۈم گۈلشەنگە خەت يازدىم ،خىتىم كېتىپ 27 كۈن بولغان كۈنى گۈكشەندىن جاۋاپ كەلدى .بۇيەردە ئۇ خەتلەرنى يىزىپمۇ يۈرمەي .شۇندىن ئىتىبارەن ھەر ئايدا گۈلشەندىن 2 پارچە خەت بىر خالتا 50 يۈەن پۇل ئۈزۈلمەي كىلىپ تۇردى .مانا ھەش پەش دىگۈچە 2 يىل تىزلا ئۆتۈپ كەتتى نۇرغۇن ئازاپ تارتتىم، قىينالدىم ،ئەمما ئۆلۈپ قالمىدىم .ھەرقىتىم كۆڭلۈم يىرىم بولسا گۈلشەندىن كەلگەن خەتلەرنى يىنىش -يىنىشلاپ ئوقۇپ كۆڭلۈمگە تەسەللىي تاپاتتىم .

ئۇنىڭ باھادىر ئۈمۈتۋار بولۇڭ !مەن مەڭگۈ سىزنىڭكى !مەن مەڭگۈ سىزنى ساقلايمەن!بۇ بىزنىڭ مۇھەببىتىمىزنى سىنايدىغان سىناق !مەن مەڭگۈ بىر سىزنىلا سۆيمەن ! سىز ئۈچۈن يارالدىم ! سىزمىنىڭ مۇھەببىتىممۇ ! جىسمىممۇ ! پەقەت ۋە پەقەت بىر سىزگىلا- سىزگىلا مەنسۇپ !بىزنى ھىچكېم ئايرىۋىتەلمەيدۇ !مەن پەقەت بىر سىز ئۈچۈنلا يارالغان ھاتىممۇ پەقەت بىر سىز بىلەنلا ئاخىرلىشىدۇ! مەن سىزنى مەڭگۈ ساقلايمەن غەيرەتلىك بولۇڭ ! مۇشۇ قۇرلارنى ئوقۇغۇنىمدا كۆڭلۈم كۆتۈرلۇپ ئىچىلىپ- قالاتتىم .بۇ سۆزلەر مىنى داۋاملىق ئۈمۈتلۈك ياشاشقا ئۈندەيتتى .قىرىشقاندەك مەن كەچۈرۈم ئىلىپ بالدۇر چىقىپ كىتىشكە تىرىشقانسىرى ،بالدۇر چىقىپ گۈلشەننى بەخىتتن بەھرىمان قىلىشقا شۈنچە ئالدىرسام .ئالدىرغان يولدا قالار دىگەندەك .خەنزۇ جىنايەتچىلەر بىلەن ئۇيغۇر جىنايەتچىلەر بىر ئەترەتتە ئارلاش بولغاچقا يەنى كىلىپ مىنىڭ خەنزىچىدىن قىلچىلىكمۇ خەۋرىم يوق بولغاچ ، خەنزۇلار بىلەن تولا ھاللاردا ئۇرشۇپ قالاتتىم .كېم ئۇرۇشسا ھەرئىككىلى تەرەپ ئوخشاش جازاغا تارتىلاتتى .ئۇرۇشقان ئىككىلىمىز تەڭلا قاتتىق باشقۇرۇشقا پالىناتتۇق ،ياكى بولمىسا قاراڭغۇ ئۆينى تاماششا قىلىپ تازا قىينىلىپ 1 ئايدىن 15 كۈن ئەتراپىدى كۈنمۇ كۆرمەيدىغان قاراڭغۇ ئۆيدە تازا پۇخادىن چىققۇچە ياشىۋالاتتۇق .ئىشقىلىپ كۈنلۈرۈم بەزىدە خۇشاللق بىلەن كۆپۈنچە جازالىنىش بىلەن دىگەندەك 2 يىلنى ئۆتـكۈزدۈم .ئەمدى بولسا بۇنداق جازالاشقىمۇ كۆنۈپ قالدىم .ئادەمنى قايتا قايتا قىيناپ جازالاۋەرسىمۇ تاياق ئۆتمەيدىغان ،تەپسە ئەۋرىمەيدىغان ھەرقانداق جازامۇ تەسىر قىلمايدىغان ،بوپقالدىكەن .دەسلەپ جازاغا ئۇچرىغاندا{电棒} توك كالتىكى ياقسا چىقىراپ جىنىم چىقىپ كەتكىلى تاس قالاتتى ھازىر توك كالتىكى ياقسا چىۋىن قونغاندەك تەسىر قىلمايدىغان بوپ قالغاندىم . تەلىيىمگە يارشا ئەترەتتكە تەقسىم قىلغاندا دوستۇم شىر ئەلىي بىلەن بىر ئەترەتكە تەڭ تەىسىم قىلىنغان، ئەمگەك ۋەزىپەمگە ئۇ ياردەمدە بولاتتى.ئاخىرى مەنمۇ ئۆزگەردىم ئەسلى ئادەمگە تاياق توقماقتىن ياخشى گەپ بەكرەك تەسىر قىلدىكەن .كادىرلارنىڭ تەربىيسى، گۈلشەننىڭ خىتىدىكى سۆزلىرى ،تەلىم تەربىيە بۆلمىدىكى ئايال ساقچىلارنىڭ قايتا -قايتا تەربىيە ئىشلەش تۇرۇتكىسىدە ، مەنمۇ ئۆزگەردىم تىرشىپ -تىرمىشىپ يۇرۇپ خەنزۇچىنى ئۈگەندىم .خەنزۇلار بىلەنمۇ ئۇرۇشماس بولدۇم .ئۇلار ئادەتتىمۇ بىزدىن بىر قەدەر ئالاھىدە ئىمتىيازغا ئىگە ،قارىماققا ئوخشاش مۇئامىلىدە بولغاندەك تۇرسىمۇ ،كادىرلارمۇ ئۇلارغا كۆپۈنچە يان باسىدۇ ،مىللەت سۆيەر كادىرلارنىڭ قولىدا ھوقۇق يوق .ھوقۇق بارلىرىنىڭ ۋىجدانى يوق .ئازراق بولسىمۇ ۋىجدانلىقلارنىڭ ھوقۇقى بولسا يارلىۋىدە ساڭگىلاپ تۇرغان سەللا قىڭغىر كەتسە مۇللاق ئىتىشدىن قورقۇپ بىزلەرنى قوغدىيالمىغان،ئىشقىلىپ بىلىپ بولمايدۇ پەقەتۈبىزنى ھىچنىمىگە ئەرزىمەس سىسىق جىنايەتچى بىلىدۇ خالاس .شۇنىڭدىن بولسا كىرەك مىللىي كادىرلارمۇ كۆپۈنچە ھاللاردا بىزلەرنى باساتتى .دەل مۇشۇنداق ياخشىلىنىشقا قاراپ يۇزلىنىۋاتقان كۈنلەردە گۈلشەننىڭ
