مۇشۇ يالقۇن روزى بىر رۇمان يىزىپ باقسا ، بىز ئوبزور يىزىپ باقساق قانداق بولار!؟

يالقۇن رۇزى ھەمىشە يازغۇچىلىرىمىزنى مۇكەممەل قىلماپتۇ، دەپلا قاخشايدىكەن
مۇكەممەللىك بىر خۇداغىلا خاس نەرسە: ئادەم خاتادىن خالى بولغىلى بولمايدۇ
كىتاپخان كىتاپنى ئوقۇپ بولغاندىن كىيىن ئۆز تەسراتىنى سۆزلەيدۇ
بىر رۇماننى پۈتكۈزۈش يالقۇن رۇزى بىر ئوبزور يىزىپلا تىللاپ تاشلىغاندەك
ئاسان بولسىچۇ -كاشكى!

توغرا ئۇيغۇر ئەدەبىياتى تەنقىدكە مۇھتاج: لىكىن تەنقىتتىن بەكرەك قوللاشقا مۇھتاج
تويۇتا شىركىتى مەسلىسى بار ئاپتۇمۇبىللارنى يىغىۋالغاندەك ،
يالقۇن رۇزىنىڭ بىر ئىغىزگىپى بىلەن مەمتىمىن ھۇشۇر  ھەممە كىتاپنى
يىغىۋىلىپ قايتا يازسۇنمۇ؟

شۇنداق دەيمەنكى: ئۇيغۇر يازغۇچىلىرى ئەسەرلىرىنى پۈتكۈزۈپ بولۇپلا
يالقۇن رۇزىغا بەرسە، يالقۇن رۇزى ئۆزى خالىغانچە تەھرىرلەپ بەرسە ئاندىن ئۇيغۇر
ئەدەبىياتى تاياق-توقماقتىن خالاس بولارمىكىن - تاڭ!
جەمىيەتنى كۆزىتىشتە ھەركىمنىڭ ھىس قىلغىنى ئوخشاش بولمايدۇ
يالقۇن رۇزىنىڭ كۆزىگە نىمە كۆرۈنسە ئۇيغۇر يازغۇچىلىرىنىڭ كۆزىگىمۇ شۇ
نەرسە كۆرۈنۈشى: ھەتتا يالقۇن رۇزىنىڭ كۆزىگە پاشا كىرسە مەمتىمىن ھۇشۇرنىڭ كۆزىدىن
ياش چىقىشى كىرەكمىدۇ!؟



بۇ روماننى ھازىر خوتەن خەلق رادىئۇ ئىستانسىنىڭ ئەدەبىي ئاڭلىتىش پىروگرامىسىدا ئاڭلىتىۋاتىدۇ
شۇڭلاشقىمۇ  ئامدىنقى ھەپتە  پشتىكەش ئەكەلگەن گىزشىت ژۇرناللارنىڭ ئارسىدىن بوينىنى قىسىپ قالغان تارىم ژۇرنىلىنى قۇلۇمغا ئىلىپ ئادىتىم بۇيىچە مۇقاۋا سۈرەتتىن ھۇزۇر ئالغاندىن كىيىن (تارىم ژۇرنىلىدا مۇقاۋىدىن باشقا ياشى نەرسە قالمىغاندەك) مۇندەرىقىسىنى كۆرىۋىتىپ يالقۇن رۇزىنىڭ مەمتىمىن ھوشۇر ھەققىدىكى قاخشاشلىرىغا كۆزۈم چۈشتى
دەررۇ قايتىدىن ئورنۇمنى تىپىپ ئاساسەن دىگۈدەك ئوقۇپ چىقتىم : ۋەزنى بىر پوۋسىتنىڭ ئورنىنى ئالغان بۇ ئوبزۇردا قاخشاشتىن باشقا يەنە نىمە بار دەيسىز. مەن ئۇيغۇر ئەدەبىياتىدا تەنقىت بولۇش كىرەك دەپ قارىساممۇ ئۇنى قوللاپ كەتمەيمەن، چۈنكى بۇ ئەمدىلەتتىن گۈللىنىۋاتقان ئۇيغۇر ئەدەبىياتى ئۈچۈن سۇ سىپىپ ئوت ئۆچۈرۈشتىن باشقا نەرسە ئەمەس

باشقا سۆزلىرىمنى خالايىقنىڭ ئىنكاسىغا قاراپ داستىخانغا چىقىراي بۇ يازمىنى ئاخىرىدا   arginakun تەھرىرلىگەن. ۋاقتى  2013-4-23 04:59 AM
