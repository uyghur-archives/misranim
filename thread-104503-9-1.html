    مۇشۇ باشقۇرغۇچىلار نىمە قىلىدىغاندۇ مۇشۇنداق ۋاقىتتا . مۇنبەرنىڭ ساغلاملىقى ھەممىدىن مۇھىم ئەمەسمۇ قېرىنداشلىرىم، نىمە قىلىمىز بىر- بىرىمىزنى غاجاپ. بىز ئەمەلىيەتتە ئۇنداق مىللەت ئەمەس، بەلكى مۇشۇ تېمىدىكى بىر قىسىم تېتىقسىز لاۋىزا ھېجىقىزلاردەك ناچار پىسپاسلىنىڭ پوقىغا تېيىلىپ شۇنداق نامدا قېلىۋاتىمىز. ئادەم دىگەن كەمتەر ھەم كەم سۆز بولغاننى كۆرگەن، قۇرۇق تۈڭدەك تاراقشىپ بىر مەھەللىنى بىر ئالغان كىشى مىنىڭچە قولىدىن پوقمۇ كەلمەيدۇ ، پەقەت ئېغىزىدا لاپلا خالاس. ئاز بولسىمۇ ساز ئىنكاس يازىدىغان بۇرۇنقى ئەزالار بەلكىم مۇشۇنداق لاتا غىلاپلىنىڭ سېسىق پۇرىقىدىن قېچىپ چەتكە چىقىۋالغان بولسا كېرەك، تىزىلىپ كەتكەن ئىنكاسلارنىڭ بىر قىسىملىرىدىن مەنمۇ بەرداشلىق بېرەلمەي ئاخىرى ساسلىققا كىرىشكە مەجبۇر بولدۇم. تېخى بەزى ئەزالار بۇ تىمىغا مۇناسىۋەتسىز بولغان سۆزلىرى ئاز بولۇپ قالغاندەك باشقا تېمىلاردىكى پوقلارنىمۇ چېچىپ يۈرۈپتۇ. ئەجەبا سىزنىڭ ئىچىڭىز شۇنداق تارمۇ ؟ ، ئاشۇنداق ۋاستىلەر بىلەن يېنىڭىزدا كۈندە بار بولغان مۇنبەردېشىڭىزغا ھاقارەت قىلىۋالسىڭىز، ھەرگىزمۇ سەمىرىپ قالمايسىز. سۆزلەڭ دىسە ئىماندىن چۈشىسىز، ئەمەلىيەتتە غەيۋەتنىڭ سورىقى دوزاق ئىكەنلىكىنى بىلمەيسىز.
    تېما ئېگىسى، تېمىڭىزغا كىرىپ قالايمىقان سۆزلەپ كەتتىم كەچۈرۈڭ، ئەسلىدە ئۆز چۈشەنچىمىز بىلەن ئىلمى سۆزلەر ئارقىلىقمۇ تېمىدىكى ئاساسى ئىدىيەنى يورۇتۇپ بىرەلەيتۇق ، ئەمما تېخى ئېچىلىپ بولالمىغان ئەقىلسىز باشلارنىڭ قاقمىسىدا ماۋۇ تېما باشقا بىر سورۇنغا ئايلىنىپ كىتىپتۇ. بىر ئەزانىڭ سۆزى ماڭا تەسىز قىلدى، ھەم توغرا. ئۇنىڭ دىگىنىگە ئوخشاش بىر قانچە نامدا كىرىدىغان بىر ۋە بىرقانچە ئەزا ئەنە شۇنىڭغا ئوخشاش ئارقىمۇ- ئارقا تېتىقسىز ۋاستىلار بىلەن كىشىنى بىزار قىلىدۇ ھەم قارا چاپلاشقا ئورۇنىدۇ. بۇنى ھېلىقى چاپلانغان باشقا مەزمۇندىكى 10 ئەزانىڭ باھا سۆزىدىن كۆرۈۋېلىشقا بولىدۇ. ھازىر تور دىگەن شوپىلەينىڭ ئۈيىدىمۇ بار، ماۋۇنىڭدەك ئىش قىلىشنى بىلمەيدىغان ھارام تاماقلار ئەلۋەتتە مۇشۇنداق قىلىپ ۋاقىتنى ئۆتكۈزمىسە باشقا ئامالى يوق. قولىدىن بىرەر ئىش كەلگەن بولسا بۇنداق تالاش- تارتىش قىلىشقا نەدىكى ۋاقىت.
   سەمىڭمە شۇنداق سېلىپ قوياي ساقلاندىلا، ماڭا ئەگىشىپ يېڭى مەزمۇن ھاسىل قىلىشقا ئورۇنۇشما. مەن پەقەت ئاچچىقىمدىلا بۇ بىر پارچە ئىنكاسنى يېزىشقا ۋاقتىمنى ئاجىراتتىم، ئەمەلىيەتتە ۋاقتىم ھازىرچە ئالتۇندىنمۇ قىممەتلىك، سەن قۇرۇڭ تۈڭلەر بىلەن تاراقشىشقا ۋاقتىم بولغان بولسا ئەڭ پاسكىنا سۆزلەر بىلەن يىرتىۋەتكەن بولاتتىم، چۈنكى ھەممە ئادەم سۆزلەشنى بىلىدۇ.
    ھەرقانداق بىر تېما ئۈستىدە ئەتىراپلىق ئويلۇنۇپ سۇغۇققانلىق بىلەن سۆزلەشنى يېتىلدۈرەيلى تورداشلار، ئۆزىمىزگە ئۆيەننىڭ قەيەردىن كېلىدىغانلىقىنى بىلەيلى. ئاۋۇ ناخشىغا كەلسەك مەن تېخى ئاڭلاپ باقمىدىم، دەسلەپتىكىسىنىمۇ بىر قېتىم ئوزۇن يوللۇق ماشىنىدا كىتىپ بېرىپ كۆرگەن ئىدىم. ۋاقتىم كۆپ بولغان بولسا ئەلۋەتتە مەنمۇ كۆرەر بولغىيتىم بىراق نىمە مەنىسى بارلىقىنى بىلىپ بولغۇچە ھەممە يەردە مودا بولۇپ كەتكىنىنى بىلدىم. تېخى ئەمدى چوڭ بولىۋاتقان 5 ياشتىن 15 ياشقىچە بولغان بالىلارنىڭ ئاغزىدا شۇنداق بىر ئاھاڭنى ئاڭلايمەن، ئۇلار نەدە يۈرۈپ بۇنداق مودىغا بىرىكىپ كەتتى؟ - ، بىز ئۈنى بىلىشكە ئاجىزمىز. شۇنىڭ ئۈچۈن ھەرقانداق بىر تېما ئۈستىدە قالايمىقان سۆزلەشنىڭ ئورنىغا ئەھمىيەتلىك سۆزلەر بىلەن بىر بىرىمىزگە ياردەم قىلايلى. ياخشى كىشى ئەلۋەتتە تەنقىد ۋە تەكلىپنى قۇبۇل قىلالايدۇ، شاكىلىنى ئىلغاپ مېغىزىنى يىيىش ئۈچۈن ئەلۋەتتە بىر- بىرىمىزگە يان- يۆلەك بولىمىز، چۈنكى كىشى يۈزىدىكى قارىنى كۆرۈش ئۈچۈن ئەينەككە مۇھتاج بولىدۇ ياكى يېنىدىكى كىشى كۆرسۈتۈپ قويىدۇ. بىز بۇلارنى توغرا يول بىلەن قۇبۇل قىلىشىمىز كېرەك.
   خەيىر دوستلار ئەمدى ئىشىمنى قىلاي، ئىنكاس يازماي زىيارەتتە بولىۋاتقىلى بىر يىلچە بولۇپ قاپتىكەن، بۇگۈن پۇخادىن چىققىچە يېزىۋالدىم. كىيىنكى قېتىم يېڭى تېمام ئۈستىدە كۆرىشەرمىز.

تىللىشىۋاتقانلارنىمۇ توغرا چۈشىنەي بوپتۇ ، بۇنىڭ تىلىنى بىلمىگەندىكىن . ئۇلارنىڭ نېمە ئۈچۈن بۇ ناخشىنى چىقاردى دەپ ئويلىغۇدەك كاللىسىمۇ يوق . خەقنى ئېغىزىنى بۇزۇپ تىللايدۇ يۇ ، ئۆزىنى پاك ، خەقنى مەينەت نەرسە كۆرىدىغان پەس كۆرىدۇ ... تېمىغا مۇناسىۋەتلىك ئىنكاستىن تىللىغان ئىنكاس جىقكەن .

تىللىغان كىشى يا بىكارچىلىقتا ئىچى پۇشۇپ قالغان ، يا بولمىسا ئۆزىنىڭ تۇرمۇشىدىمۇ ئىناۋىتى يوق ئېغىزى بېزەپ ئۇرۇشقاق كىشى .
