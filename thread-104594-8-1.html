مەزكۇر قانۇنغا ئاساسەن ، مەزكۇر دۆلەت قانۇن ئارقىلىق ئېدارە قېلىنىدىغان ھەقىقىي قانۇن دۆلىتى بولىدىغان بولسا ، مەسچىت سېلىشقا تەستىق بەرمىگەن ئەمەلدارلار . دۆلەت ئاساس قانۇننى دەپسەندە قىلغانلىقى ئۈچۈن تېگىشلىك جازاغا تارتىلىپ،مەسچىت قۇرۇلۇشىغا دەرھال ئىجازەت بېرىلىشى كىرەك!

بۇ مەسچىتلەر ھۆكۈمەت تەرىپىدىن ئارخىپ تۇرغۇزۇلغان ،قانۇنلۇق دىنىي پائالىيەت سورۇنى بولغاندىن كېيىن ،خەتەرلىك بولۇپ قالغاندا چوقۇم رىمونىت قىلش كېرەك ،بولمىسا ئىتقادچى ئاممىنىڭ ھاياتىغا خەۋپ يەتمەمدۇ ؟
يەنە دىسەك يەنەشۇ ،مەركەزدىن ھەرگىزمۇ يىلدا پەقەت مانچە مەسچىتنى رېمونىت قىلىڭلار دەپ بۇيرۇق كەلمەيدۇ. ھازىر ھۆكۈمەت ھەممىدە ئادەمنى ئاساس قىلىش ، ئامممىغا مۇلازىمەت قىلىشنى تەكىتلەۋەتقاندا ،بۇنداق خەتەرلىك مەسچىتلەردە ناماز ئوقۇۋاتقان مۇسۇلمان ئاممىنىڭ مەنپەئەتىنى چوقۇم ئويلىشىدۇ. قانچىنى رېمونىت قىلشقا تېگىشلىك بولسا ،شۇنچىنى رېمونىت قىلىدۇ، بەلكىم ئايرىم مەبلەغمۇ ئاجرىتىدىغاندۇ..ئەلۋەتتە بەزى يۇرتلاردىكىدەك بىرئوبدان مەسچىتنى چۇۋۇپ ،باشقا يۇرت -مەھەلللىەر بىلەن بەسلىشىپ كاتتا مەسچىت سېلىش ئىشلىرىنى ھۆكۈمەتلا ئەمەس ، دىنىمىزدىمۇ  رىياكارلىق دەپ چەكلەيدۇ.
ئەمدى مارلبېشىدىكى ئىشلارغا كەلسەك ،ھەممىسى تۆۋەندىكى پوق قوساق ئەمەلدارلارنىڭ خىەلقنى  يالۋۇتۇش ئارقىلىق ،ئاددىي ئىشنى مۇرەككەپ كۆرسىتىپ ، مەسئۇليەتنى يۇقىرىغا ئىتتىرىپ قويۇپ  ئۆزىگەتېخىمۇ  باش ئېگىدىغان قىلىش ، شۇئارقىلىق ئۆزلىرىنى شۇنداق بىر تەس ئىشنى ھەل قىلىپ  خەلقنىڭ ھاجىتىدىن چىققان قىلىپ كۆرسىتىش ئۈچۈن ، ئۆزلىرنىڭ ئاتالمىش ئىناۋەت -ئابرويىنى تېخىمۇ ئاشۇرۇش ، ھوقۇق- ھەيۋىسىنى كۆز-كۆزقىلىش ئۈچۈن ئويناۋاتقان ئويۇنى خالاس .
ئاللا ئىشلىرىمىزنى ئاسان قىلار.
مەسلەكداشلىق مىللەتداشلىقتىن ئەلا!
