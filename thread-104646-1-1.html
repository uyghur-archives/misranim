ئەسسالاممۇ ئەلەيكۇم مۆھتىرەم تورداشلار ،مۇنبەردىكى  ھۆرمەتكە سازاۋەر يازغۇچىلار ۋە شائىرلار ،كەمىنىدىن سىلەرگە ئىھتىرام بولغاي !!

پىقىر مۇنبەردىكى بىر ئەزا بولۇش سۇپىتىم بىلەن ئەزىز يازغۇچىلىرمىزغا كىچىككىنە بىر ئىشنى ئىلتىجا قىلماقچىمەن !

     مەن ئەدەبىي ئەسەرلەرگە ئالاھىدە قىزىقىدىن بىر گوي  ،بولۇپمۇ پىرورا ئەسەرلەرگە،گەرچە ئۆزۇم ئانچە يىزىپ كەتمىسەممۇ ،ھەم يازغانلىرىمنى مۇنبەردە ئىلان قىلىشقا سالايىتىم توشمىسۇ ،لىكىن مۇنبەردىكى قىرىنداشلىرمىزنىڭ ئەسەرلىرنى قىزىقپ ئىشتىياق بىلەن ئوقۇيمەن !

ئەپسۇس ،مۇشۇ بىر نەچچە پوۋستنى كۆرگەندىن كىيىن كۆڭلۇم بىر قىسمىلا بولۇپ قالدى ،كۆپىنچە چوڭ ھەجىمىلىك ئەسەرلەرنى ئوقۇغۇنۇمدا تازا قىزىپ ئوقۇپ ئوقۇپ ئەسەرنىڭ ئىچىگە كىرىپ كىتىمەن ،ناھايتى ئەپسۇس ،كۆپىنجى ھاللاردا <داۋامى بار >دىگەن خەتنى كۆرىمەن ،خۇددى شىرىن چۇش كۆرىۋاتقاندا بىراۋ تەرپىدىن ئويغۇتۇلۇپ كەتكەندەك ،چىۋىن يەۋالغاندەك بوپ قالىمەن .كىچىك چاغلىرىمدا تىلىۋۇزۇر كۆرەۋىتىپ ئەڭ ياخشى يىرىگە كەلگەندە ياكى ئىلان ياكى خەۋەر باشلىۋالاتتى ،خەۋەردىن كىيىن كۆرىمەن دەپ تىلىۋۇزۇرغا قاراپ ئولتۇرۇپ قالاتتىم ،نەتىجىدە  ئۇقۇم كىلىپ ئورنۇمدىلا ئۇخلاپ قالاتتىم .مانا ،مۇشۇنىڭغا ئوخشاش <داۋامى يىزىلىۋاتىدۇ >ياكى <داۋامى بار >دىگەن ئەسەرلەرنى كۆرسەم بىر قىسمىلا بولۇپ قالىمەن ،يەنە ئاستىدىكى <داۋامى قاچان چىقار،داۋامى يىزىلىپ بولغاندا ئوقاي >دىگەندك بىر نەچچە ئىنكاسلارنى كۆرگىنىمدە خۇددى ئەنە ئاشۇ ياخشى ڧىلىملەر ئارىسىغا قىستۇرۇلغان ئەدەمنى بىزار قىلىدىغان ئىلان ئۇچۇرلارنى كۆرگەندەك بولىمەن .مىنىڭچە بولغاندا ئىنكاس دىگەن ئەسەرگە بىرىلگەن باھا ،يازغۇچىغا بولغان ئىلھام بولۇشى ،ھەم يازغۇچىلىرمىزنىڭ بىرىلگەن تەكلىپ پىكىر ،ئەسەردىن ئالغان ھىسسي تۇيغۇلار بولۇشى كىرەك ئىدى (ئەلۋەتتە بۇ مىنىڭ شەخسى قارشىم ،ئومومىيلىققا ۋەكىل بولالماسلىقى مومكىن ).بەلكىم ئاپتۇرنىڭ ئەسەرلەرنى تۇركۇمگە بولۇپ يوللىشىدا ئۆزىنىڭ دانا ئوي خىياللىرى باردۇر ،ئوقۇرمەن بىلەن بالدۇرراق يۇز كۆرۇشسۇن ،داۋامى چىققۇچە ئۆزى مۆلچەرلەپ باقسۇن دەيدىغاندۇ ،ئەلۋەتتە بۇمۇ خاتا ئەمەس ،بىراق قانداق بولۇشتىن قەتئىي نەزەر ،ھۆرمەتلىك ئاپتۇرلىرمىزنىڭ ھەجىمى چوڭراق پىروزا ئەسەرلەرنى يوللىغاندا تولۇق يىزىلىپ بولغاندا يوللۇشىنى ئىلتىجا قىلىمەن .

  ئاتاگاچىلىق قىلىپ قويدۇم ،ئەيىپكە بۇيرۇمىغايسىلەر !!

ئاپتۇرلىرمىزنىڭ تىنىگە سالامەتلىك ئىشلىرغا ئۇتۇق تىلەيمەن !ئىلھامىڭلار ئۆلمىگەي ،سالامەت بولۇڭلار !
