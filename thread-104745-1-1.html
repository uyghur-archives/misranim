بۈگۈن بىر خەۋەر كۆرۈپ قالدىم . خىنەندىكى بىر ساماۋەر شورپىخانىسى مۇسۇلمانچە ۋىۋىسكىسى ئىسىپ ، چوشقا گۆشى سىتىپتۇ . شۇ يەردىكى مۇسۇلمانلار بۇلارنىڭ ئاشخانىسى ئالدىدا تۇرۋىلىپ ئاشخاننى تاقاتقىزىۋىتىپتۇ . ئەمدى چوشقا گۆشى چوقۇم ھارام ئۇنى ھەربىر مۇسۇلمان بولغان ئادەم بىلىدىغان ئاددى بىر ئىش . ئەمما قويچۇ ؟
قوينى ئاللاتالا ئىنسانلارغا بەرگەن كاتتا نىئىمەتنىڭ بىرسى . بۇنىڭمۇ ئەلۋەتتە ھالال ۋە ھارىمىي بولىدۇ .
ئاللا قوينى جەننەتتىن چىقىرىپ بەردى . ئەمما ئۇنىڭ گۆشىنى قوي ئۆلتۈرگەن چاغدا بىسمىللا ئاللاھۇ ئەكبەر دىگەن تەكبىر سۆزى قوشۇلۇپ ئۆلتۈرۈلگەن بولسا بۇنى تىخىمۇ ھالال بولدى . تىخى نۇرغۇن قىرىنداشلارغا تىپىشماق شەكلىدە قوي ھالالمۇ دىسە ئىكىلەنمەي ھالال دىيىشىدۇ . بۇمۇ بىر ئاددى تۈز كۆڭۈللۈك ، ساددىلىق . قوي ھەقىقەتەن جەننەتتىن چىققان بۇنىڭدا شەك يوق . ئەمما تەكبىرسىز ئۆلتۈرگەن قوي بۇ قەتئىي ھارام . شۇ قوينىڭ يۇڭىمۇ ھارام ، تىرىسىمۇ ھارام . ئەمدى قوي گۆشى ئىكەن دەپ يىقىندىن بىرى ھەرقايسى چوڭ شەھەرلەرنىڭ كوچا دوقمىشلىرىدا خۇيزۇ ، خەنزۇ ، ئۇيغۇرلارنىڭ ئاچقان ھىلىقى مالاچۇرسىنى كۆرىمىز . بولۇپمۇ چوڭ شەھەر ئۈرۈمچىدە . سىلەرنىڭچە شۇلارنىڭ ساتقان گۆشى ھالالمىدۇ ؟ شۇ گۆش قوينىڭ بولغان بىلەن تەكبىر ئىيىتىپ ئۆلتۈرگەن قويمۇ ياكى تەكبىرسىز ئۆلتۈرىلگەن قوينىڭ گۆشىمىدۇ ؟
شۇڭا تورداشلار مۇشۇ ئىشقا بەك دىققەت قىلساڭلار ، مۇشۇ بىر تىما توغۇرلۇق مۇنازىرەلىشىپ باقساق .
