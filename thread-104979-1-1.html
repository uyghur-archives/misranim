                                                  ۋىژداننىڭ سۇئالى

                            كۆز ئاچسام دۇنياغا بولۇپ مەن ئىنسان،
                                                  قانداق ياشايسەن؟ دەپ سورىدى ۋىژدان،
                                                   چوڭ  بولدۇم، ياشىدىم خېلى  ياشقىچە،
                                                    ھاياتنىڭ  يولىدا  داۋان  ئاشقىچە،
                                                     تاپمىدىم دەل جاۋاپ بولغاچ مەن نادان.

                                                      ساقالمۇ رەڭ تۈزۈپ ئۆسكەچ ئىڭەكتە،
                                                      بارماقلار سەپ تۈزۈپ تۇرغاچ بىلەكتە،
                                                       ۋىژدانىم قايتىلاپ قىلدى كۆپ سوراق.
                                                       جاۋاپسىز ئويلاندىم ئۇزاقتىن-ئۇزاق،
                                                       بىلدىمكى ساپ ئەقىل تۇرغاچ كىتەكتە.

                                                       ئۆمۈرنىڭ دەپتىرى ئېچىلدى شۇ ئان،
                                                      كۆرۈندى رەتمۇ-رەت مەن تۈزگەن پىلان،
                                                      مەنىسىز  ناخشىنى  رېتىمسىز  توۋلاپ،
                                                       ياشلىغىم ئۆتۈپتۇ ئۆزۈمنى ئالداپ،
                                                       خاماندا بىر دان يوق كۆرۈنەر سامان.

                                                        دىدىكى  ۋىژدانىم  قايتا- قايتىلاپ،
                                                        غەپلەتتىن تىز ئويغان،قەددىڭنى رۇسلاپ،
                                                        ۋەتەن دەپ ئىزدەنگىن بەخت ۋە نىجات،
                                                        سەندە بار مۇقەددەس تەپەككۇر قانات،
                                                        قەدەم ئال ھاياتقا مەنزىلنى بويلاپ.

                                                        نادانلىق ئىلكىدىن بولدۇم مەن ئازات،
                                                        چۈشەندىم ئادەمگە قىممەتكىن ھايات،
                                                       ئەل ئۈچۈن جان پىدا دەپ تۈزدىم پىلان،
                                                        سۇئالنى ۋىژدانىم چۈشەندى ھامان،
                                                         شۇ ھايات يولىدا ؛-يېڭلىق يارات!

                                                                   مەنبە؛ئۆزەمنىڭ ئىجادىيتى
