دەۋادىكى دەلىل ئىسپات كۈچىگە سەل قارىغان دەۋاگەررنىڭ دەۋا تەلىۋى رەت قىلىندى.

ئەكبەر ئابدۇشۈكۈر


بۇ مەزمۇنلار پۈتۈنلەي مىسرنىم مۇنبىرىدىن كۆچۈرۈلگەن @) Z" R; o2 x3 }# e2 R

نۆۋەتتىكى جەمىيەت تەرەققىياتغا ئەگىشىپ، پۇقرالار ئارسىدا قانۇن ساۋاتلىرى قەدەممۇ قەدەم ئۇمۇلىشىۋاتقان بولسىمۇ يەنىلا بىر قىسىم كىشلەرنىڭ قانۇن ئارقىلىق ئۆز ھۇقۇق مەنپئەتىنى قوغداش ئىڭى تۆۋەن بۇلۇشتەك ئەھۋاللار يەنىلا مەۋجۇت. شۇڭا ھەر قانداق كىشى ئۆز ھۇقۇق مەنپئەتىنى قانۇن ئارقىلىق قوغداشنى ئۈگۈنۈشى. شۇنىڭ بىلەن بىر چاغدا  قانۇن ئورۇنلىرىغا دەۋا قىلىش ئالدىدىن ئۆز دەۋا تەلىۋىگە قارتا دەلىل ئىسپات كۆرسۈتۈش لازىملىقىدەك ھەقىقەتنى ئىسىدە تۇتۇشى كىرەك. ئەگەر ئۇنداق بولمايدىكەن دەلىل ئىسپاتسىز دەۋا تەلىۋى قانۇن ئالدىدا پۇت تىرەپ تۇرالمايدۇ. تۆۋەندە مەلۇم بىر جايدىكىتوك سوقىۋىتىشتىن كىلىپ چىققان جىسمانى تۆلەم ماجرا دىلوسىدا دەۋاگەرلەرنىڭ  ئۆز دەۋا تەشەببۇسىدىكى دەلىل ئىسپات كۈچىگە سەل قارىغانلىقتىن دەۋا تەلىۋى رەت قىلىنىش جەرياننى سەمىڭىزلەرگە سۇنىمەن.

2012-يىلى 5-ئاينىڭ 29-كۈنى مەزكۇر دەۋادىكى جاۋاپكار ئەسەت ئۆزىنىڭ ئۆگزىسىگە لاي سالماقچى بولۇپ، ئۆگزىگە لاي سېلىشنى  مەزكۇر دەۋادىكى قۇشۇمچە جاۋاپكار دولقۇنغا 600 يۈەنگە كۆتۈرە بەرگەن، جاۋاپكار دولقۇن قازا قىلغۇچى مەخەم ، ياقۇپ ، ئىمىن، رەخمان، يۇنۇس قاتارلىق 5 كىشىنى ياللاپ جاۋاپكار ئەسەتنىڭ ئۆيىگە بارغان، جاۋاپكار دولقۇن ئىش ئۈنىمىنى يۇقۇرى كۆتۈرۈش ئۈچۈن  قۇرباندىن بىردانە لاي تارتىش ماشىنىسىنى بىر كۈنلىكىنى 50 يۈەنگە ئىجارىگە ئېلىپ قۇراشتۇرۇشتۇرۇپ، ماشىنا ئارقىلىق لاي تارتقان، ئۇلار ئىش باشلاشتىن بۇرۇن ئىش تەقسىم قىلىپ، قازا قىلغۇچى مەخەم تۆۋەندە تۇرۇپ غالتەككە ئۇسۇلغان لاينى، لاي تارتىش ماشىنىسىنىڭ تارتىش ئىلمىكىگە ئىلىپ بىرىدىغان بولۇپ، ئىش باشلاپ مەشخۇلات ئېلىپ بېرىپ، شىنجاڭ ۋاقتى سائەت 3:30 لار ئەتراپىدا، مەخەم لاي ئۇسۇلغان قول ھارۋىنى لاي تارتىش كىرانىنىڭ ئىلمىكىگە ئىلىۋاتقاندا تۇيۇقسىزلا ھارۋىغا يۆلىنىپ يىقىلىپ چۈشكەن، جاۋاپكارلار مەخەمنى توك سوقىۋەتكەن ئوخشايدۇ دەپ قاراپ، ئائىلە-تاۋاباتلىرىغا خەۋەر قىلىشقا ئۇلگۇرمەي يېزىلىق شىپاخانىغا ئېلىپ بارغان، دوختۇرخاندىكىلەر ئۆزلىرنىڭ ئامال قىلالمايدىغانلىغنى، چوڭراق دوختۇرخانغا ئاپرىپ كۆرسىتىشنى ئىيىتقان. ئۇلار دەرھال ئازادلىق ئارمىيە دوختۇرخانىسغا ئىلىپ كەلگەن دوختۇرخاندىكى خادىملار مەخسۇس ئاپىراتقا سىلىپ تەكشۇرمەستىن، سىرتقى جەھەتتىن كۇزتۇپ تەكشۇرۇش ئارقىلىق جاۋاپكارلاردىن نىمە ئىش بولغانلىغىنى سورىغان ئۇلار توك سوقىۋەتكەنلىك گۇمانىنى دىگەن، دوختۇرلار مەزكۇرنىڭ «تنقى توختاپ بولۇپتۇ،توك سوقىۋەتكەن بولسا بەدەننىڭ سىرتقى جەھەتتىن بىرەر توك سوققانلىقتەك كويۇك ئالامەت كورۇلمەيدۇ، تەكشۇرتىمىز دىسەڭلار مىڭ يۇەندىن ئارتۇق پۇل كىتىدۇ، بولمىسا ئىلىپ كەتسەڭلارمۇ بولىدۇ»دىگەنلىكتىن تەكشۇرتمەي ئىلىپ كەتكەن، قازا قىلغۇچىنىڭ ئاكىسى يۇقارقى ئەھۋادىن خەۋەر تاپقاندىن كىيىن 2012-يىلى 5-ئاينىڭ 29-كۇنى سائەت  9.30 دە ساخچىخانسىغا دىلو مەلۇم قىلغان. يىزىنىڭ سىياسى قانۇن ئىشلارغا مەسئۇل رەھبرى دىلو ئەھۋالىدىن خەۋەر تىپىپ، يېزىلىق توك باشقۇرۇش پونكىتى رەھبەرلىكىگە «8-كەنىتتە بىرسىنى توك سوقىۋىتىپ ئۆلۈپ كىتىپتۇ، سىلەر بېرىپ تەكشۇرۇپ بىقىڭلار» دەپ ئورۇنلاشتۇرغان، يىزلىق توك باشقۇرۇش پونكىتىدىن ئىككى كىشى بىرگە بىرىپ تەكشۇرۇش بىلەن ناھيىلىك توك باشقۇرۇش ئىدارىسىگە خەۋەر قىلىش ئارقىلىق بىخەتەرلىك بۆلىمىدىن خادىملار كىلىپ خىزمەت ھوقوقى دائىرسىدە تەكشۇرۇپ، لاي تارتىش ماشىنىسىنىڭ سىرىتقى قېپىدىن توك ئۆتۈپ كەتكەن ئەھۋالنىڭ يوقلىقى، سىرىتقى سىمدىن لاي تارتىش ماشىنىسىنىڭ ئىچىگىچە بولغان سىمنىڭ ساق ئىكەنلىكىنى ئىغىزاكى مۇئەييەنلەشتۇرگەن، مادانىڭ ئىچكى قىسمىدىن توك ئۆتۈپ كىتىش كەتمەسلىك ئەھۋالىنى باھالاش سالاھىيتى بولمىغانلىقى ئۈچۈن مادانى ئىچىپ تەكشۇرۇش ئىلىپ بارمىغان. يېزىلىق ساقچىخانا دىلو مەلۇماتىنى تاپشۇرۇپ ئالغاندىن كىيىن دەرھال نەخ مەيدانغا يىتىپ بىرىپ، تەكشۇرۇش ئىلىپ بارغان. دەۋاگەرنىڭ ئائىلىسىدىكىلەر مەخەمنى توك سوقىۋىتىپ ئۆلۈپ كەتتى دىگەن، ساقچخانىدىكىلەر ‹قازا قىلغۇچىنىڭ ئۆلۈش سەۋەبىدە باشقا گۇمانىڭلار بار بولسا قانۇن دوختۇرىغا باھالىتڭلار، ئەگەر توك سوقىۋىتىپ ئۆلدى دەپ قارىساڭلار ھەق تەلەپ دەۋاسى قىلساڭلارمۇ بولىدۇ ›دىگەن ۋە ناھيىلىك ج خ ئىدارىسى قانۇن دوختۇرلۇق ئىشخانىسىگە مەلۇم قىلغان ئۇلار يىتىپ كىلىشتىن ئىلگىرى قازا قىلغۇچىنىڭ تۇققانلىرىدىن قازا قىلغۇچىنىڭ دادىسى،  ئۈچ ئاكىسى، ۋە ئايالى قاتارلىقلارنىڭ ئوز-ئارا قوشۇلۇش ئارقىلىق«بىزنىڭ قىرىندىشمىز مەخەم بۇگۇن چۇشتىن كىيىن (2012-يىلى5-ئاينىڭ19-كۇنى) 8-كەنىت4-مەھەللىدە قۇرلۇش ئورنىدا ئىشلەش جەريانىدا، توك تۇتۋىلىپ ئۆلۇپ كەتكەن. بۇ ئەھۋالغا قارتا ج خ ئىدارىسنىڭ قانۇن دوختۇرى جەسەتنى تەكشۇرمىسىمۇ بولىدۇ، قانۇن دوختۇرىغا تەكشۇرتمەسلىكتىن كىلىپ چىققان ئاقۋەتكە ئۆزىمىز ئىگە» دىگەن مەزمۇندا ھوججەت يىزىپ بىرىپ جەسەتنى قانۇن دوختۇرىغا تەكشۇرمەستىن يەرلىككە قويۋەتكەن. يۇقارقى ھادىسسىدىن كىيىن قازا قىلغۇچىنىڭ ئۆلۇم، نەزىر چىراق ئىشلىرنى ئۇزتۇشقا جاۋاپكار دولقۇن 5000 يۇەن پۇل 1750 يۈەن قىممىتىدىكى بىر تۇياق قوي، ئەسەت 5000 يۇەن نەخ پۇل بەرگەن.

قازا قىلغۇچى مەخەمنىڭ نەزىرە چىراق ئىشلىرى تۈگىگەندىن كىيىن ئۇنىڭ  تۇققانلىرى خەلق سوت مەھكىمىسىگە « « 2012-يىلى 5-ئاينىڭ 29-كۈنى جاۋاپكار دولقۇن مەرھۇم مەخەم.قاتارلىق 5 كىشىنى ياللاپ جاۋاپكار ئەسەتنىڭ ئۆيىنىڭ ئۆگزىسىگە لاي سېلىش ئەمگىكىگە قاتنىشىپ، شىنجاڭ ۋاقتى 3:30 لاردا مەرھۇم مەخەم قول ھارۋىغا لاينى ئۇسۇپ، قول ھارۋىنى لاي تارتىش كىرانىنىڭ ئىلمىكىگە ئىلىۋاتقاندا، توك سوقىۋىتىپ نەخ مەيداندا قازا قىلغان، بۇ ۋەقەنى يېزىلق ساقچىخانىسىغا مەلۇم قىلغاندىن كىيىن ساقچىلار نەخ مەيدانغا بېرىپ ئەھۋالنى ئېنىقلاپ مۇھەممەت ئېزىزنىڭ توك سوقىۋىتىپ ئۆلۈپ كەتكەنلىكىنى مۇئەييەنلەشتۈرگەن، توك باشقۇرۇش شىركىتىدىكىلەرمۇ نەخ مەيدانغا كىلىپ تەكشۈرۈش ئېلىپ بېرىپ، لاي تارتىش كىرانىنىڭ ئەتتىگەندىن كەچكىچە مەشخۇلات قىلىش سەۋەپلىك بېسىمى ئېشىپ كىتىپ،  ماداسى قىزىپ لاي ھارۋىسىغا توك ئۆتۈپ كەتكەنلىكىنى ئېنىقلاپ چىقتى، جاۋاپكار دولقۇن قازا قىلغۇچىنى ياللاپ ئىشلەتكەن شۇڭا ئۇلاردا ياللاش ياللىنىش مۇناسىۋىتى بار، جاۋاپكار ئەسەتنىڭ بۇھادىسسە بىلەن مۇناسىۋىتى بولمىسىمۇ، ئۇ بىۋاستە مەنپەئەتلەنگۈچى شۇڭا تۆلەم جاۋاپكارلىقىنى ئۈستىگە ئېلىشى كىرەك،  شۇنىڭ ئۈچۈن سوت مەھكىمىسىنىڭ جاۋاپكارلاردىن بىزنىڭ ئۇمۇمى زىيىنىمىز 220706.5 يۈەننى تۆلىتىپ ئېلىپ بىرىشىنى تەلەپ قىلىمەن. » دەپ دەۋا قىلدى.

خەلق سوت مەھكىمىسى بۇ دىلوغا سوت ئىچىپ قاراپ چىقىپ، پۇخرالار ساغلام ياشاش ھوقۇقغا ئىگە، پۇقرالارنىڭ تىنىنى زەخمىلەندۇرگەنلەر دەۋالىنىش راسخۇتىنى، ئىشقا چىقالماي ئازىيىپ كەتكەن كىرمىنى، مىيىپ بولغۇچىنىڭ تۇرمۇش ياردەم پۇلىنى ۋە شۇنىڭغا ئوخشاش راسخۇتلىرىنى تۆلىشى كىرەك، ئۆلۇشكە سەۋەپچى بولغانلار ئۆلگۇچىنىڭ دەپنە پۇلىنى، ئۆلگۇچى ھايات ۋاقتىدا بىقىپ كەلگەن كىشىلەرنىڭ  زورۇر تۇرمۇش خىراجىتىنى ۋە شۇنۇڭغا ئوخشاش راسخۇتلارنى بىرىشى لازىم دەۋاگەرلەرنىڭ، جاۋاپكارلاردىن مەرھۇم مەخەمنىڭ دەپنە خىراجىتى 16180.5 يۈەن، ئۆلەم تۆلىمى ئۇچۇن يىلىغا 5442 يۇەندىن20 يىللىقى  108840 يۈەن، مىيىت ئۇزۇتۇش جەريانىدىكى ئىشتىن قېلىش ھەققى 2595 يۈەن قازا قىلغۇچىنىڭ پەرزەنتىلىرىنىڭ بېقىلىش ھەققى 83562 يۈەن، قازا قىلغۇچىنىڭ ئاتا-ئانىسىنىڭ بېقىلىش ھەققى 9529 يۈەن بولۇپ جەمئى 220706.5 يۈەننى تۆلىتىپ ئېلىپ بەرسە دىگەن دەۋا تەلىۋىنى قانۇن دائىرسىدە مۇۋاپىق  قوللاشقا تىگىشلىك بولسىمۇ، بىراق سوتتا مەرھۇم مەخەمنىڭ توك سوقىۋىتىپ ئۆلۇپ كەتكەنلىكىدەك پاكىتنى ئىسپاتلىيالمىدى ، يېزىلىق ساقچىخانا ئەينى ۋاقىتتا دىلو مەلۇماتىنى تاپشۇرۇپ ئىلىپ نەخ مەيدانغا بىرىپ تەكشۇرۇپ، ئۇلارغا« قازا قىلغۇچىنىڭ ئۆلۈش سەۋەبىدە باشقا گۇمانىڭلار بار بولسا قانۇن دوختۇرىغا باھالاتساڭلار» بولىدۇ دىگەن ۋە ناھيىلىك ج خ ئىدارىسى قانۇن دوختۇرلۇق ئىشخانىسىگە مەلۇم قىلغان ئۇلار يىتىپ كىلىشتىن ئىلگىرى قازا قىلغۇچىنىڭ تۇققانلىرى ئوز-ئارا قوشۇلۇش ئارقىلىق« ج خ ئىدارىسنىڭ قانۇن دوختۇرى جەسەتنى تەكشۇرمىسىمۇ بولىدۇ، قانۇن دوختۇرىغا تەكشۇرتمەسلىكتىن كىلىپ چىققان ئاقۋەتكە ئۆزىمىز ئىگە» دىگەن مەزمۇندا ھوججەت يىزىپ بىرىپ جەسەتنى يەرلىككە قويۋەتكەن، يىزلىق ۋە ناھيىلىك توك باشقۇرۇش ئورنىدىن لاي تارتىش ئۈسكىنىسىنى تەكشۇرۇپ توك ئۆتۇپ كەتكەنلىكىدەك ئەھۋالنى بايقىمىغان، دوختۇرلارمۇ مەخەمنىڭ جەسىتىنى تەكشۇرۇپ توك سوقۇۋەتكەنلىكىدەك ئەھۋالنىڭ يوقلىقىنى دىگەن، سوتتا دەۋاگەر ۋە دەۋاگەرلەرنىڭ ئۇرۇق-تۇققانلىرىغا يەرلىكنى ئىچىپ جەسئەتنى قانۇن بويىچە ئەدلىيلىك باھالىتىش توغرىسىدا  قانۇن چۇشەندۇرۇش ۋە ئويلۇنۇپ كورۇش توغرىسىدا ۋاقىت بەرگەن بولساقمۇ باھالىتىشنى ئىلتىماس قىلمىدى، شۇ سەۋەپلىك تەرەپلەر ئوتتۇرسىدا مەخەمنىڭ نىمە سەۋەپتىن 1. دەۋاگەرلەرنىڭ، جاۋاپكارلاردىن مەرھۇمنىڭ دەپنە خىراجىتى 16180.5 يۈەن، ئۆلەم تۆلىمى ئۇچۇن يىلىغا 5442 يۇەندىن20 يىللىقى 108840 يۈەن، مىيىت ئۇزۇتۇش جەريانىدىكى ئىشتىن قېلىش ھەققى 2595 يۈەن قازا قىلغۇچىنىڭ پەرزەنتىلىرىنىڭ بېقىلىش ھەققى 83562 يۈەن، قازا قىلغۇچىنىڭ ئاتا-ئانىسىنىڭ بېقىلىش ھەققى 9529 يۈەن بولۇپ جەمئى 220706.5 يۈەننى تۆلىتىپ ئېلىپ بەرسە دىگەن دەۋا تەلىۋى رەت قىلىنسۇن.دەپ ھۆكۈم چىقاردى.
