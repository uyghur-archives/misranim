يوللىغان ۋاقتى 2013-5-4 07:18:18 PM |ھەممە قەۋەتنى كۆرۈش |
ھۆرمەتلىك مۇنبەرداشلىرىم، مەن سىلەرگە ئۆز تەسىراتلىرىمنى سۆزلەپ بېرەي.قىممەتلىك ئىنكاسىڭلارنى ئايىماسلىقىڭلارنى ئۈمىت قىلىمەن.
مەن كىچىك چاغلىرىمدا ياغلىق چىگەتتىم،كېيىن ئوقۇش پۈتكۈزۈپ جەمىيەت تەسىرىمۇ قانداق ؟يالاڭباش يۈرىدىغان ،...ئۆتكەن يىلى ھامىلدار بولغۇنۇمدىن باشلاپ ياغلىق چەگدىم،(بىراق كەينىدىن)چېگەتتىم، نىمىشقا دەمسىز،؟ياغلىق چەگسەم گويا سەتلىشىپ گويا قېرى  مومايلارغىلا  ئوخشاپ قالغاندەك ھېس قىلاتتىم، قوشكېزەك ئوغۇللىرىم تۇغۇلۇپ چاچراپ كېتىپ يېرىم يىل  بولغان بولسىمۇ ياغلىقنى قەتئى بېشىمدىن ئېلىپ باقمىغان ئىدىم،بۇ يىل 3-ئاينىڭ مەلۇم بىر كۈنى ئۆي تازلايمەن دەپ چاچلىرىم بەك مەينەت بوپ كەتكەچكە ،ماگزىنىمغا كەلگەنلەرنىڭ توختىماي تىلفۇن قىلىشىدىن ،ياغلىقنى ئوراپ چىگىۋېلىپ  ماگزىنغا چىقتىم،ماگزىنىمنىڭ ئەتراپىدا ماگزىن ئاچىدىغان ،پۈتۈن ئورىنىپ  يۈرىدىغان ،نامازنىڭ ۋاقتىنى بىر قېتىممۇ كەم قىلمايدىغان، ئاجايىپ تەقۋا بىر نەچچە خوشنام بار ئىدى،  بۇ بىر نەچچەيلەن بەك ئىناق بولۇپ ،تاپقاننى تەڭ يەپ دىگۈدەك ئۆتىدىغان، بىرلىكى بار ئاياللار ئىدى.ئۇلارغا كۈچلۈك ھۆرمىتىم بار ئىدى،خوشنىلىرىم بىر دەمدىلا مېنى مۇبارەكلەپ كىرىشتى،ۋە ياغلىقىمنىڭ ماڭا ئىنتايىن ياراشقانلىقىنى ئەمدى ھەرگىزمۇ ئېلىۋەتمەسلىكىمنى ئىسلامدا ئايال كىشىنىڭ بوينى،كۆكسى ،قول بېغىشى،پاچاقلىرى  قاتارلىق ئورۇنلىرىنىڭ  ئوراش ئىچىگە كىرىدىغانلىقىنى سۆزلەپ چىقىپ كېتىشتى، مەن كۈلۈپ كەتتىم،كۆڭلۈمدىكى راس گەپنى قىلغاندا ئۇلارنىڭ بىرلىشىپ يېنىمغا كىرىپ ئالاھىدە مۇبارەكلىشى مېنى سۆيۈندۈرگەنلىكتىن كۈلگەن ئىدىم،
ئۇلار چىقىپ كەتكەندىن كېيىن مەن ئەينەككە ئۇزاق قاراپ كەتتىم، ياغلىقىنىڭ ماڭا ياراشقانلىقىنى ئەمدى كۆرىۋاتاتتىم،ئادەتتە  پەرداز ۋە زىبۇ -زىننەت بىلەن چىرايلىق كۆرۈنگەن چېھرىمىز،ياغلىق ئىچىگە كىرسە  تېخىمۇ چىرايلىق بوپ كېتىدىكەن ئەمەسمۇ؟ نىمىشقا دەمسىز؟ھۆسۈن تۈزەش ئورۇنلىرىغا بېرىپ چاچ ياسىتىش يوق،ئىڭىكىم ساڭگىلاپ كەتتى دەپ ئوپراتسىيەخانىلارغا بېرىپ ئىڭەكنى تارتقۇزۇپ ياسىتىش يوق، بوينۇمغا داغ چۈشتى دىمەيسىز،بوينۇم ئورۇق دىمەيسىز،سەت دىمەيسىز،ئالتۇن ھالقا سالالمىدىم دىمەيسىز،ئالتۇن زەنجىر سالالمىدىم دىمەيسىز،پەقەت ياغلىقتا ئورانسىڭىز ئىككى قېشىڭىز ۋەكۆزىڭىز بەك چىرايلىق ۋە يارقىن گەۋدىلىنىپ سىزنى تېخىمۇ چىرايلىق قىلىۋېتىدىكەن، يۈزلىرىڭىز تېخىمۇ كىچىك ئوماق كۆرسىتىدىكەن ئەمەسمۇ؟؟؟؟؟؟
بۇلار مېنىڭ ئاشۇ ئەينەك ئالدىدا ئولتۇرۇپ قىلغان ئەڭ ئاددى تەسەۋۇرلىرىم ۋە كۆز قاراشلىرىم ئىدى خالاس...

ئەتىسى ماگزىنغا چىقىپ ،بىر قۇر تازلاپ تۇرىشىمغا تۆنۆگۈن كىرگەن خوشنىلىرىمدىن  ئىسلام كىيىملىرى ساتىدىغان بىر خوشنام كىرىپ مېنىڭ بىلەن ئەھۋاللاشقاندىن كېيىن ئىككى قۇر ئىسلامچە كۆڭلەڭ بىلەن ئىككى دانە ئىسلامچە چوڭ ياغلىقنى كۆتىرىپ كىرىپ ،بۇلارنىڭ ماڭا قىلغان سوۋغىسى ئىكەنلىكىنى،بۇندىن كېيىن ھەر بىر قېتىملىق نامىزىمدا ئۆزى ئۈچۈن دۇئا قىلىپ قويسىلا كۇپايە ئىكەنلىكىنى ئېيتىپ ئۆز قولى بىلەن كىچىك ياغلىقىمنى ئېلىۋېتىپ ياغلىقنىڭ بىرسىنى شەكىل چىقىرىپ سېلىپ قويۇپ چىقىپ كەتتى، مەن يەنە ئەينەككە قارىدىم،ياغلىق بۇ قېتىم ماڭا ئاجايىپ چىرايلىق ياراشقان بولۇپ ئەينەكتىكى ئەكسىمگە ئۇزاققىچە قاراپ تۇرۇپ قالدىم،كۆزۈمدىن سىرغىپ چۈشكەن ياشلىرىم ياغلىقىمغا تامچىلاشقا باشلىدى،بۇ ياش تامچىلىرىنىڭ سەۋەبى؛بۇ ئايال خوشنامنىڭ قىلىۋاتقانلىرى ئىسلامنىڭ قانچىلىق ئۇلۇقلىغىنى، ئىسلامنىڭ پاكلىغى،ئۇلۇغلىنىشى ئۈچۈن بېرىلگەن  ھەقسىز سوۋغا ئىكەنلىكىدىن ئىدى،....
ئاز ۋاقىت ئۆتمەي  يەنە ئىككى  خوشنام بازاردىن ماڭا  ئۈچ يۈرۈش ئىسلام كىيىم-كىچەكلىرى ۋە ياغلىقلىرىنى سېتىۋېلىپ ئەكىلىپ مۇنداق دىدى،
-سەن قوشكېزەك تۇققان ئايال، كۆڭلۈڭنى يېرىم قىلما، ئاللاھ خالىسا ساڭا يەنە پەرزەنت ئاتا قىلىدۇ،ئۈمىتلىك بولۇپ ھەممە نەرسىنى ئاللاھقا سېغىنىش كېرەك، قىيامەتتە ئاياللارنى ئاللاھقا ئەڭ يېقىن قىلىدىغىنى ياغلىق، بىز سېنىڭ ئاللاھقا تېخىمۇ يېقىن بولغانلىقىڭنى تەبرىكلەپ كىردۇق، بىز بالا كىيىمى ساتقاچقا ،بازار ئارلاپ  ساڭا ئەڭ چىرايلىق دەپ قارىغانلىرىمىزنى سېتىۋالدۇق دەپ ، ئاللاھ ھەر قانداق مۇرادىڭغا يەتكۈزسۇن بۇ بىزنىڭ ئاللاھ ۋە ئىسلام يولىدا قىلغان ئازراق كۆڭلىمىز دەپ بىر مۇنچە تەسەللى كۆڭۈل ياساشلاردىن كېيىن چىقىپ كەتتى......مەن ئۈنسېلىپ يىغلاپ تاشلىدىم، دىمىسىمۇ مەن ھەممىنى بىلىمەن،مەيلى ناماز ئۆتەش،مەيلى قۇرئان ئوقۇش،بىراق نىمىشقا موشۇ ئاياللاردەك بولالمايمەن؟ نىمىشقا  ھەممىلا نەرسىگە ئېرىنچەكلىك بىلەن بىپەرۋا مۇئامىلە قىلىمەن؟ مەنمۇ ئۇلارغا ئوشاشلا مۇسۇلمان ئايال،ئىسلام دىنى پۇقراسىغۇ؟ بۇلار قىلغاننى مەن نىمىشقا قىلمايمەن؟ نىمىشقا قىلالمايمەن؟ئۇلار نىمىشقا ماڭا بۇنچە قىپ كېتىدۇ؟
-بۇ دەل ياغلىقنىڭ ھۆرمىتى،ئىسلام دىنىنىڭ قائىدە يوسونلىرىنىڭ ھۆرمىتى، ئاللاھنى خۇش قىلىدىغان ھۆرمەت،ئاللاھنىڭ ئەڭ سادىق پەيغەمبىرى مۇھەممەت ئەلەيھىسسالامنىڭ -ئى مۇسۇلمانلار .سىلەر ئىسلامدا ئۆز-ئارا قېرىنداش كېلىسىلەر،شۇنىڭ ئۈچۈن ئىناق ئۆتۈڭلار،بىر-بىرىڭلارنى ياخشى يوللارغا باشلاڭلار،يامان كۈندە قالغانلارغا ھېسداشلىق قىلىڭلار،يىتىملارغا خەيرى-ئېھسان قىلىڭلار،بولۇپمۇ ئاياللارنى ياخشى ئەمەللەرگە، ياخشى يوللارغا  باشلاڭلار،ئۇلارنىڭ يولىدا قىلغان ھەر بىر ھەركىتىڭلارنىڭ ساۋابىنى ھەممىنى كۆرۈپ تۇرغۇچى ئاللاھ ھەسسىلەپ بېرىدۇ،دىيىلگەن ھەدىسنى يادىمغا ئالدىم،
ئەمەلىيەتتە بۇ كيىملەرنى كىيەلەيتتىم،سېتىۋالالايتتىم،بىراق نىمە ئۈچۈن كەيمىدىم؟نىمە ئۈچۈن سېتىۋالمىدىم؟
نىمە ئۈچۈن قايتىدىن پەرزەنت ئاتا قىلىشنى ھەممىگە قادىر بىر ئۇلۇغ ئاللاھدىن تىلىمىدىم،بەك ئۇزاق ياش تۆكتۈم،ھەممىسىنىڭ  ئاللاھقا بولغان ئۇلۇقلىشىغا،ئىسلامغا بولغان ھىدايىتىگە ئىمان ئېيتتىم،ئۇلارغا چەكسىز سالامەتلىك ھەم ئۇزۇن ئۆمۈر تىلەپ دۇئا قىلدىم،
مانا شۇنىڭدىن باشلاپ ئۆز ئېتىقادىمنى مۇستەھكەملەش ئۈچۈن تىرىشتىم،ئاللاھدىن ماڭا يەنە پەرزەنت  ئاتا قىلىشنى،مېنى ئانا بولۇش ھوقوقۇمغا ئىگە قىلىشنى،بارلىق مۈشكۈلاتلىرىمنى يېڭىشىمغا مەدەت بېرىشنى، يىغلىماي، ئاھ ئۇرماي خوشال ،ئۈمىتلىك  ياشاشنى ئۆتۈنۈپ تىلىدىم، كۈنلىرىم خېلى مەنىلىك، ئۆز نۆۋىتى بىلەن ئۆتىۋاتاتتى.سودام بەركەتلىك،تۇرمۇشۇم خوشال مەنىلىك بولۇپ ئۆزگىرەۋاتقانلىقىنى ھېس قىلىشقا باشلىدىم،ھازىر ئۆزەمنى ياغلىق بىلەن ئىسلامچە كىيىم-كىچەكلەر بىلەن تېخىمۇ چىرايلىق بوپ كېتىۋاتقاندەك ھېس قىلىپ زوقلىنىدىغان بوپ قالدىم،قىزىق يېرى ھەر قانداق چاغدىكى ئاياللارنىڭ ئالاھىدە زوقلىنىپ قاراشلىرىغا، ھۆرمەتلەشلىرىگە،ئالاھىدە ئىززەتلەشلىرىگە ئېرىشكىنىمدە ياغلىقنىڭ ھۆرمىتىنى،ئىسلامنىڭ ھۆرمىتىنى شۇنچىلىك چوڭقۇر ھېس قىلىشقا باشلىدى، بۇرۇن كوچىلاردا ماڭسام بەزى ئىسلامچە كىيىنگەن قىز-ئاياللىرىمىز مېنىڭمۇ كۆزۈمگە ئاجايىپ ئىسسىق-ھەم ئىسىل كۆرىنەتتىكى، خۇددى ياغلىق ئۈچۈنلا يارالغاندەك كۆزۈمنى ئۈزەلمەيتتىم، ئۇلارغا زوقلىنىپ قارايتتىميۇ-ئۆزەمگە ئۇنچىلىك ياراشمايدىغاندەك ھېس قىلغان ئىدىم،مانا ئەمدى مەنمۇ باشقىلارغا مەن زوقلانغاندەك، باشقىلارمۇ ماڭا زوقلىنىدىغان ئىسلام گۈزىلىگە ئايلانغان ئىدىم،(بۇ پەقەت شەخسى كۆز قارىشىم جۇمۇ.....ئۆزەمنى ماختىغىنىم ئەمەس)
تۆنۆگۈن بىزنىڭ خوشنا  ماگزىندىكى گىرىمچىك خوشنام ماڭا قىممىتى بەشيۈز يۈەندىن ئاشىدىغان بىر يۈرۈش ئىسلامچە كىيىم ۋە ئىسلامچە ياغلىق ئەكىرىپتۇ،ئۇ مۇنداق دىدى؛
-ھەدە ،سىز قوشكېزەك تۇققان ئايال، سىزنى بەك ھۆرمەتلەيمەن،سىز موشۇ يولدا ناھايىتى ياخشى كېتىۋاتىسىز، مېنىڭ كەسپىم گىرىمچىك بولغاچقا ،ھازىرچە ئامال يوق،ناماز بامداتتىن باشقىغا يىتىشەلمەيۋاتىمەن،ئاللاھ خالىسا بىر كۈنى بىزمۇ سىزدەك يۈرەرمىز، بۇ كىيىملەرنى سىزنىڭ بويىڭىزغا قاراپ سېتىۋالدىم، ھەر بىر كەيگىنىڭىزدە مېنى يادقا ئېلىپ دۇئا قىلىپ قويۇڭ، بۇ مېنىڭ ئازراق كۆڭلۈم،دەپ پۇل سۇنساممۇ ئۇنىماي چىقىپ  كەتتى،
- ھۆرمەتلىك مۇنبەرداشلىرىم، ئەمدى ئېيتىڭلارچۇ؟ بۇ ياغلىقنىڭ ھۆرمىتىنى،ئىسلامنىڭ ھۆرمىتىنى   ئۇلۇغلايدىغانلاردىن  موشۇ دۇنيادا يەنە نەچچىسى باردۇ؟ مەن ئەگەر بارلىق مۇسۇلمان قىز-ئاياللىرىمىزمۇ موشۇنداق ئۇيغۇر مىللىتىنىڭ ئەنئەنىسى بويىچە مىللىچە،ۋە ئىسلامچە كىيىم -كېچەكلىرىمىزنى كىيىپ، ئۆز ئېتىقادىمىزدا چىڭ تۇرۇپ،ھەقىقى ئۇيغۇرغا  ئەخلاقلىق،غورورلۇق خاس ياشىغان بولساق  نىمە دىگەن ياخشى بولاتتى-ھە؟موشۇ توختىماي يامان بولىدۇ دەيدىغان، بىزنى يامانلىقلاردىن توسۇپ تۇرىدىغان  بۇۋا-مومىلىرىمىز ئازلاپ كەتتىمۇ-نىمە؟
مەن دەيمەنكى:-بۇرۇن بىخوتلۇقىمىزدا قانداقلا بەڭۋاشلىق قىلغان بولساق، ياۋرۇپا ئەنئەنىسىگە ئەگەشكەن بولساق،ئۆتكەن ئىشقا سالىۋات،ھەممىگە قادىر ئاللاھ،ھەممىنى كۆرۈپ تۇرىدىغان كارامەتلىرى بىلەن ھەممىمىزنىڭ  گۇناھلىرىنى ئەپۇ قىلىپ،  سىلەرنى ئۆز شاپائىتىدىن بەھرىمەن قىلىدۇ، قېنى مۇسۇلمان قىز-ئاياللار سىلەر دۇنيانىڭ ئانىسى،سىلەرنىڭ پەزىلىتىڭلار دۇنيانى ئۆزگەرتىدۇ، سىلەرنىڭ سەۋەنلىكىڭلار دۇنيانى ھالاك قىلىدۇ، سىلەرنىڭ ئەخلاقىڭلار دۇنيانى  يۈكسەلدۈرىدۇ، قېنى بىزنىڭ ئۇيغۇر،مۇسۇلمان دىگەن مەدىنىيىتىمىز بويىچە ياشاپ كۆرۈڭ، ياغلىقنى ئوراپ سېلىپ كۆرۈڭ،ئىسلامچە،ئۇيغۇرنىڭ مەدىيىتى بويىچە ياشاپ كۆرۈڭ، باشقا دىندىكىلەر ئۇيغۇرنىڭ مەدىنىيىتىنى شۇنچىلىك قەدىرلەيدىكەنۇ- بىز نىمىشقا ئۆز مەدىنىيىتىمىزنى قەدىرلىمەيمىز،؟ياغلىقنىڭ،ياق ئىسلامنىڭ ھۆرمىتىنى قىلىڭ،شۇ چاغدا قەدىر-قىممىتىڭىز ،ئورنىڭىز، نەسلىڭىز،ئىسىللىقىڭىز، مەرتىۋىڭىز،تۇرمۇشتىكى مېھرى -مۇھەببىتىڭىزنى، ھالال بەركىتىڭىزنى  ھەقىقى ئۇلۇغلىيالايسىز،ئاللاھنىڭ   ھۆرمىتىگە ئېرىشەلەيسىز.
مەنبە؛-ئۆز قەلىمىم،ئۆز تەسىراتىم.
