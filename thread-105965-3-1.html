مۇشۇ تىمدا مەنمۇ ئەمن ئۆي توغىرسدىكى دەرتلىرىمنى تۆكلىۋىلاي-----------بىزمۇ 2010يىلى ،كونائۆيمىزنى كەلكۈن ‹‹سەل››ئۆرىۋىتپ يىڭى ئۆي سالدىغان بولغان، ئەملىيەتتە ئۆتلەرچاققىدەك دەرىجدە بۇزۇلۇپمۇ كەتمىگەن ،يىزدىن ئاپەتكەئۇچىرغان  ئۆيلەرنى مەجبۇرى چىقىڭلار،دىدى ،بەزى كەم ئەقل كىشلىرىمزكادىرلارنىڭ گپگە كىرىپ ساق ئۆيلىرنمۇ چاقتى ، بەزىلەرگە پۇرسەت بولوپ كونائادەم ئولتۇرمايدىغان ئۆيلەرنمۇ تىزىملككە كىرگۈزۈپ چاقتى ، بىزنمۇ شۇقاتاردا چىقىڭلار دىگەندى ،بىزنىڭ ئۆيدىكلەرئىككى قاش بولوپ قالدۇق،ئاكام بىلەن ئانام ئۆينى چاقمىز،ھۆكۈمەت كۈزگىچە سىلپ بىردۇ،دىدى ، مەن بىلەن دادام چاقماسلقتاچىڭ تۇردۇق ، مەن يىزاباشلقى بىلەن تىللشپمۇ قالدىم ،مىنى 15كۈنلۈك سولايدىغانمۇ بولدى،مەن ئۆزەمنىڭ ئۆي چاقمەنمۇ چاقمايمانمۇ ئۆزەمنىڭ ئىشى دىدىم ، ئۇلارسىلپ بىرىمىزدىگەندى ،مەن يامان بولساڭ يىڭى ئۆينى سىلپ بول ،ئاندىن چاقمەن ،دىدىم ،‹‹سۇنى كۆرمەي ئىشتان سىلىش››بىلەن خوشۇم يوق دەپ تۇرۋالدىم ،مىنى سولماقچى بولدى ،مەن قانۇنغاخىلاپلق قىلمدىم ،سولساڭ ياتمەن ،چىقسام ئۈستڭىدىن ئەرىزقىلمەن ،ياتقان كۈنۈمگە سەندىن پۇل ئالمەن .دىدىم ،ئاقۋەت ،3ئىغىزئۆينى چاقمايدىغان ،يىڭى ئۆي پۈتسە ئۆزىمزچاقدىغان بولدۇق ، ئەگەر قىشقچە ئۆي پۈتسە مەن شەڭجاڭنى مھمان قىلدىغان ،ئۆي پۈتمسە ئىشخانسدا ياتدىغان بولدۇم ، كۈزبولغچە ئانام ماڭاتولاقاقشدى ،سەن چاققىلى قۇيماي ،دەپ ،دەرۋەقە قىشقىچە ئۆي پۈتمىدى ،ئۆينى نوچلق بىلەن چاققانلارقىشنى خەقنىڭ ،تۇققانلىرنىڭ ئۆيدە چىقاردى ، بىزئۆيدە چىقاردۇق ،ئاناممۇ ماڭاقايل بولدى ، 2011يىلى ئەتيازدا قۇرۇلۇش باشلاپ يازدا ئاران پۈتتى ، بىزمۇ يىل ئاخىردا ئۆيگە يىزنىڭ مەجبۇرلۇشى بىلەن كۆچتۇق، بىزنىڭ 2تام ئىدى ،شۇيەردە 80دىن ئارتۇق ئائلك بار،ئەمماتوكنى ئۆزىمىزباشقلارنىڭ ئۆيدىن ئاچىقتۇق ‹‹كىلۋىتى 1كوي››،سۇنى تۇڭدا توشۇپ ئىچىۋاتمىز، ئىككى يىل بولدى ،سۇ،توك ،گاز،پار،خالا.مۇنچا،تور،بار،مەھەللەيولغاچىراق يىقپ بىرىمز، يولنى ماييول قىلپ بىرمىزدىگەن ھىچنمە قىلپ بەرگنى يوق ، بىزنىڭ ئىككى ئويگە جەمئى 80مڭغايىقىن كەتكەن بۇنڭىدىن ئىككى ئۆيگە 15مىڭنى بىزيىنمىزدىن چىقارغان ، ئەمماكىين ھىساپ قىلپ بىزگە 14200يۈۋەن ياندىغان بولغان ئىدى ،ئەمماھازىرغىچە بەرمدى ،يابەرمەيمىزدىمىدى ، يىزا باشلقمۇ يۆتكىلەلمەي ،ناھىيدىن ئەمن ئۆيلەرنىڭ ھىساباتنى 0چۈشۈرۈش،دىھقانلارنى رازى قىلمسڭىزيۆتكمەيمىزدەپتكەن ،شۇنىڭ بىلەن 2يىل بولدى ،ياھەل بولمدى ،شۇپۇل ھەل بولسا ئالمغان بەرمگەن بولاتتۇق ،ياندىن پۇل چىقىم كىچىك ۋاقىمتدا قىيغىتىپ ئوينايدىغان، نۇرغۇن باللىق ئە ...

ھەي................ يامان تاتلىق ئىشلارنى ئىسىمگە سېلىپ قويدىڭىز ئەمەسمۇ ؟

كىچىك ۋاقىتلىرىمدا مىنىڭمۇ خامان باققىنىم ئىسىمدە ، يامغۇر ياغىدىغان بولسا بۇغداي ، باشاقلارنى يۆگەپ  ساتمىنىڭ ئاستىغا كىرىپ شار - شار قىلغان ئاۋازلارنى ئاڭلاپ كىتەتتۇق .
كالىلىرىمىزنىڭ  كەينى مەھەللىسىگە خالتا كەيدۈرۈپ قويۇپ ۋالغا سالغىنىممۇ ئىسىمدە ...كىيىنچە تىراكتور ، سۆپەكلەر پەيدا بولدى .
ھەي ، سامان توشۇيدىغان ۋاقىتتا ئويۇن بوپ بىرەتتىدە ئەمما ... سامانلىققا سەكرەپ ئويناپ ئارقىدىن ئۆستەڭلەرگە سەكرىگەن .
تاۋۇزلۇقتىكى ساتمىلارغا  يۈزىدە شەبنەم يالتىراپ تۇرغان مۇزدەك تاۋۇزلاردىن نەچسىنى ئۈزۈپ ئەكىرىپ قويۇپ كاسىداپ يەپ كىتەتتىم ، ئۆز قولى بىلەن ئۈزۈپ يىگەن تاۋۇز بىلەن بازاردىكى تىلىملاپ ساتقان تاۋۇزنىڭ تەمى ئوخشىسىمۇ لەززىتى باشقىچە بولىدىكەن ، ئۇنىڭ ئۈستىگە مەن كاسىداپ يىيىشكە ئامراق ئىدىم .
قىش پەسلىمۇ كۆڭۈللۈك ئۆتەتتى ، چۇككالتەك دەيدىغان بىر ئويۇن بولىدىغان ئويناپ كىتدىىغان ، يەنە ھېلىقى پاينەك دەيدىغان ئويۇنمۇ بولدىغان پاينەك تالىشىمەن دەپ ھوشۇقلارغا تاياق يەيدىغان ....





ھەي ................ بۇنداق مۇھىت ، بۇنداق ئويۇن ، بۇنداق ھىسىياتنى بۇ يەردە تورغا چىقىۋاتقانلاردىن قانچىسى بىلەركىن . بىراق مىنىڭ شۇنداق خۇشال ، بىغەم ، راھەت كۈنلىرىم ئاشۇ ئىكەن .
ھازىر بولسا ئۇ كۈنلەرگە قايتاي دىسەم مۇھىت ، شارائىتنى تاپقىلى بولسىمۇ بىراق ئاشۇ چاغلاردىكى تاتلىق ھىسىياتلارنى تاپىقىلى بولمايدۇ .
ئون كوي پۇلنى تاشلىۋەتكەن كىشىنى كۆرمىدىم ، ئەمما ئون يىللىق دوستىنى تاشلىۋەتكەن كىشىنى كۆردۈم .

ئەستا ، مەنمۇ شەھەرلىك ئەمەس ، يېزا نوپۇس ئۆيۈممۇ ھەم كىسەك تاملىق ئۆي .
شىمالىي شىنجاڭ ، شەرقى شىنجاڭدىكى يېزا ئۆيلىرى بىلەن جەنۇبىي شىنجاڭنىڭ يېزا ئۆيلىرى مۇئەييەن پەرىقلەرگە ئىگە ئىكەن .
بىر مىسالى : جەنۇبىي شىنجاڭنىڭ يېزا ئۆيلىرىدە ئۆينىڭ يورۇقلۇقىغا ئانچە ئەھمىيەت بىرىلمەيدۇ ،  ئادەتتە تاۋدان ياكى ھاۋادان دەپ بىر كىۋادىرت مىتىرچە چوڭلۇقتىكى دەرىزىنى تورۇسقا يىقىن ( كىشىنىڭ قولى يەتمەيدۇ ) قويىدۇ .
قىش پەسلىدىمۇ كۆپۈنچە يېزا ئۆيلىرى ئوتۇن ياكى كۆمۈر بىلەن قىشنى چىقىرىدۇ . شۇڭا قىشلىق ئۆيلەر قاراڭغۇ بولغىننى ئاز دەپ يەنە ئىسنىڭ تەسىرىگە ئۇچرايدۇ .
مەن مۇشۇلارنى قىسقارتىپ بايان قىلىپ قويغان ئىدىم ، سىز مىنى بىنا ئۆيى بار شەھەرلىككەن دەپ قاپسىزدە ...
ئون كوي پۇلنى تاشلىۋەتكەن كىشىنى كۆرمىدىم ، ئەمما ئون يىللىق دوستىنى تاشلىۋەتكەن كىشىنى كۆردۈم .
