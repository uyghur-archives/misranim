ھەرقايسى دۆلەتلەرنىڭ تالانت مۇسابىقىلىرىنى كۆرىۋاتقانلار مېنىڭچە ئاز ئەمەس . ناۋاغا قارىسام مىللىي سەنئىتىمىزگىغۇ گەپ كەتمەيدۇ ، بىراق باشقىلىرىدا ھېچقانداق يېڭىلىق يوق ، ھەممىسى ناي ناي لاي لاي ...

يېڭى ‹‹ ناۋا ›› دا دەپتىغۇ ، ‹‹ نېمە قىلسا بولىدۇ ››. بىرىنجى ئۆتكەلدە چوقۇم ناخشا تەلەپ قىلغان بولسا ناخشا ئېيتىپ ئۆتكەندىن كېيىن ، كېيىنكى قەدەملەردە كىشىنى جەلپ قىلغۇدەك يېڭىلىق قىلىش كېرەك ، بۇ يېڭىلىققا ئەلۋەتتە بىزدە كەم بولىۋاتقان ، جۇڭگونىڭ ئۆلۈك مائارىپىغا ماسلىشىپ رودمېنتلىشىپ كەتكەن ، ‹‹ بۇنداق قىلسام خەق نېمە دەپ قالار ›› دەيدىغان خەقنىڭ كۆزىگە قاراپ يېڭىلىق يارىتىشتىن قورقىدىغان كاللىدا ‹‹ imagination ، تەسەۋۋۇر ›› دېگەن نەرسە يوق ، تالانت دېگەن ناخشا-ئۇسسۇلدىلا بولمايدۇ ، نېمىدە تالانتى بولسا شۇنى قىلىش كېرەك .

بىرىنجى ناۋا دا بىر قىز كىيىم لايىھەلەپ ئەپ كېلىپتۇ ، بۇمۇ بىر تالانت ، ئوتتۇرا مەكتەپ ئوقۇغۇچىلىرى ئۆزلىرى ئىجاد قىلغان راپ، خىپ-خوپ ئېلىپ كەپتۇ بۇمۇ بىر تالانت ، كېيىنكىسىدە سەل ئىچىۋاغاندەك چىققىنى بىلەن ھەركەت بىلەن قوشۇپ يۇمۇر ئېلىپ كەپتۇ ، بۇمۇ بىر تالانت، بىر قىز كىيىمنى سەللا ئۆزگەرتسە ئۇيغۇرچە-ھىندىچە-ئەرەبچە بولۇپ ئۆزگىرەيدىغان كىيىمنى كىيىپ ھەرخىل ئۇسسۇلنى ئويناپ بەردى ، بۇمۇ بىر تالانت .

ئەمدى ناۋا يېڭى باشلاندى ، مەنچە تېخىمۇ جىق تۈردىكى تالانتلار چىقىدۇ ، مەسىلەن: كۈچتۈڭگۈرلۈك بەستلىشىش ، سېرىك ، سېھرىگەرلىك ، بەدەن ماھارىتى، ئاۋاز ماھارىتى ( ھەرخىل ئاۋازنى دوراش ياكى ئېغىزدا مۇزىكا چېلىش ) ھەرخىل تىلدا ناخشا ئېيتىش ، رەسساملىق ......... دېگەندەك ، ھەتتا بىزنىڭ خىيالىمىزغىمۇ كىرمىگەن ھەرخىل تالانتلار ....

ئاندىن توردا باھا بەرگەن تورداشلار ، سىلەرگە بىر تەلەپ ھەم تەكلىپ :
بۇ تالانت مۇسابىقىسى بولغاچقا ، سەھنىگە چىققان تالانتلىقلار ئۈستىدە توردا ئولتۇرۇپ غەيۋەت قىلمىسىڭىز ياكى تىللىمىسىڭىز . ئەگەر سىزنىڭ تالانتىڭىز بولسا ، تەسسەۋۋۇر كۈچىڭىز بولسا سىزمۇ قىلسىڭىز بولىدۇ ، قانداقلا بولمىسۇن ، سەھنىگە چىققانلا ئەمەس ، ھەرقانداق كىشىنىڭ تەسسەۋۋۇرىنى بوغۇش ھوقۇقىڭىز يوق ، شۇڭا كىشىگە باھا بەرگەندە ، قانچلىك تالانتى باركەن ، شۇنىڭغا قاراپ پىكىر بەرسڭىز بولىدۇ ، ئەمما ، يۇقارقى ئىنكاسلاردەك كىشىنىڭ زىتىغا تېگىدىغان سۆزلەرنى قىلمىسىڭىز .

مۇشۇ يەردە خەت يېزىپ ئولتۇرغان ھەممە كىشىنىڭ ‹‹ ئۆلۈك مائارىپ ›› تا تەربىيەلەنگىنىنى بىلىمەن . مەنمۇ بۇرۇن شۇنداق مائارىپتا بىلىم ئېلىپ ، ئۆزۈمنىڭ كۆزىگە سىغمىغان ، سەل كۆزگە كۆرۈنگەن ئىش ياكى سۆزلەرنى قىلغانلارنى ‹‹ قائىدىسىز ›› دەپ تىللايتىم . ھازىر خېلى يىل بولدى مەن چەتنىڭ تەسەۋۋۇر كۈچىگە باي مائارىپىدا ئوقۇۋاتىمەن ، ھەم شۇنىڭدىن باشلاپ ‹‹ تەسسەۋۋۇر قىلىش ، يېڭىلىق يارىتىش ›› كۈچىنىڭ بىزدە قانچىلىك كاملىقىنى ، قانچىلىك دەرىجىدە خەقنىڭ كۆزىگە قاراپ ، خەق بىر دېسە ئىككى دېيىشكە جۈرئەت قىلماي ياكى ئۆزىمىز ئۆزىمىزنى بوغۇپ تۇرىۋاتقىنىمىزنى بىلىپ يەتتىم . مېنىڭ كەسپىمدىمۇ ( كىنو رېجىسورلىقى ) تەسسەۋۋۇر، يېڭىلىق يارىتىش ئەڭ مۇھىم ئورۇندا تۇرىدۇ . خەق قىپ بولغاننى قىلسىڭىز نەتىجىڭىز نۆل . چەتنىڭ ئېلانلىرىنى كۆرىسىزغۇ ، بۇ تەسسەۋۋۇرغا بايلىقىنىڭ نامايەندىسى .

مېنىڭچە تەسەۋۋۇرىنى ئەركىن قويىۋېتىپ يېڭىلىق يارىتالايدىغان ياشلىرىمىز نۇرغۇن ، ناۋا چوقۇم بىزگە يېڭىلىق ئەپ كېلەلەيدۇ .

توختىخاچام بۇرۇتلۇق ، قاق جىگدىسى قۇرۇتلۇق. تور ئىسمىم بەك مەتو .

مىنىڭچە بۇ ناۋانى مۇسابىقە شەكلىدە ئېلىپ بېرىپ نەق مەيداندىكى تاماشىبىنلار ۋە باھالىغۇچىلارنىڭكىنى ئاساس قىلىپلا نەتىجىسىنى يەكەنلەش لىللا بولماپتۇ .
ئەگەر نەق مەيداندىكى تاماشىبىنلار ئىچىدە ماھىرنىڭ ئۇرۇق تۇققان ، دوسلىرى كۆپ بولسا ئەلۋەتتە بىلەت سانى ئېشىپ كىتىدۇدە ...
باھالىغۇچىلارمۇ تازا يۇمشاق باشكەن ، ئۇنىمۇ ياخشى ، بۇنىمۇ ياخشى دىگەن خۇددى خامىلىوندەك ...تەنقىدى پىكىرلەرنىمۇ بىرىش كىرەكتە .
باھالىغۇچىلارغا پاشا ئىشان ، مەريەم ناسىر ، رەيھان ئابلىز ، ئىززەت ئىلياس ، تۇرسۇنئاي ئىبرايىم ، سەنەۋەر تۇرسۇن ، ئايشەم قىييۇم ، رابىيە مۇھەممەت ، دىلنار ئابدۇللا ، رىسالەت ھاپىز ، گۈلجەننەت ھۈسەن ،
ھوشۇر قارى ، ئابدۇللا ئابدۇرىھىم ، داۋۇتجان ناسىر ، مۆمىنجان ئابلكىم ، دىلشات ئىبراھىم ، ئەسقەر رەقىپ ، تىلمان قادىر ، كۆرەش رەجەپ ، تاشمۇھەممەت باتۇر ، ئۆمەرجان ئالىم ، قاتارلىق سەنئەت ساھەسىگە كونا ، پىشقان ئارتىسلارنى تەكلىپ قىلسا باھالاش ئۈنۈمى ياخشى بولىدۇ . تۈنۈگۈن تېخى ئېغى ئوچۇق ئىشتان كىيىپ ئوينايدىغان ناخشىچىلارنى ئەكىلىپ باھالاشقا قويسا ئۇلار ماھىرلارنىڭ سىرىتقى قىياپىتىگە بەكرەك ئەھمىيەت بىرىدىكەن .

يەنە نەخ مەيداندا سىرىتتىن باھالاشنى يولغا قويسا بولىدۇ مەسىلەن تور ، تىلىفۇن ئابونىتلىرىغىمۇ باھالاش پۇرسىتى يارىتىپ بەرسە مۇسابىقە ، بىر قەدەر توغرا ھەم ئادىل بولىدۇ .
