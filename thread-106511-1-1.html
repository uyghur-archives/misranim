رەسىمدىكى كۆرۈنۈش ، غۇلجا شەھىرىدىكى مەلۇم ئويۇنچۇقخانىنىڭ ئىچى . ئاشۇ توپلۇشۇپ تۇرغان ياشلارنىڭ ھەممىسى دىگۈدەك ياشلىق مەزگىلى ئۇرغۇپ تۇرغان ئۇيغۇر ياشلىرى . بۇنداق ئويۇنچۇقخانىلار غۇلجا شەھىرىنىڭ ھەممىلا يېرىدا بار ، بۇ ئويۇنچۇقخانىلار پات پات تاقىلىپ قېلىپ يەنە بىر مەزگىلدىن كېيىن يەنە ئېچىلىپ تۇرىدۇ . ئاشۇ ئۆرە تۇرغان ئۇيغۇر ياشلىرى شۇنچە تەستە تاپقان پۇللىرىنى مۇشۇ يەردە يوق قىلىپ ، ئىقتىسادىنى ، ۋاقتىنى خورۇتۇپ بارماقتا . ئۇلار ئويناۋاتقان بۇ ئويۇن پۇلغا نومۇر سېتىۋىلىپ ، شۇ نومۇرنى كۆپەيتىدۇ ياكى ئازايتىدۇ . بۇنداق ئويۇنچۇقخانىلارنى خېلى كۆپ ئارىلاپ چىقتىم ، شۇنداقلا بۇ ئورۇنغا داخىل بولغۇچىلارنىڭ %95 دىن ئارتۇقىنىڭ ئۇيغۇر ياشلىرى ئىكەنلىگىنى بايقىدىم . بۇ ياشلارغا مۇشۇ يەردىىن باشقا جايلاردا تىرىكچىلىك يوللىرى تېپىلمىغاندۇ ؟ ئۇلار بۇ جايدا نۇرغۇن پۇللىرىنى يوق قىلىش بىلەن بىرگە ئالتۇندەك ياشلىق مەزگىلىنى غازان قىلماقتا .
سۆيۈملۈك ئۇيغۇر ياشلىرى بۇ جايدىن نېرى تۇرۇڭ ، بۇ جاي سىزگە مەنپەئەت بەرمەيدۇ ، بۇ جاي يانچۇقىڭىزدىكى تۆت تەڭگىڭىزنى شۇنداق ئۇستۇلۇق بىلەن ئېلىۋالىدىكى بۇنى سىز يېنىڭىز قۇرۇغلانغاندا ئاندىن سېزىسىز . شۇنۇڭ ئۈچۈن سەگەك بولۇپ ، ئۆزىڭىزنىڭ قىممەتلىك ۋاقتىنى كەلگۈسىڭىز ئۈچۈن سەرىپ قىلىڭ !  بىز ، ئاق قارىنى خېلى ئوبدان پەرق ئېتەلەيدىغان خەلق ، سىلەرنىڭ شۇنداق قىلالايدىغانلىقىڭلارغا ئىشىنىمەن .

بىكارچىلىق.ئۇيۇنچۇقخانىغا نىمە بار دەپ بارىدۇ.بارىدىغانلار ئاساسەن يىزا ياشلىرى،بۇنداق ئورۇندا ئۇينىمىسىمۇ پۇلنى يىغىپ ئۆي ئالامدۇ،خوتۇن ئالامدۇ،ئاتا ئانىسىنى ھەجگە ئاپىرامدۇ،يا قۇرساققا ئىككى كىلوگۆش ئىلىپ دادىسىغا يىگۈزەمدۇ شۇنى قىلمامدۇ .بۇنداق ئۇيۇنچۇق نامىدىكى قۇرۇلغان جايلارنىڭ ھەممىسىلا دىگۈدەك قىمار تۈسىنى ئالغان دىيەلەيمەن.ئاڭلىسام بۇنداق ئورۇننىڭ ئىجارىسىمۇ بازار تەۋەسىدە خىلىلا ئۆرە بولىدىكەن،ئۇ ئورۇننى ئاچقان ئادەم شۇنچىۋالا دەسمىينى سىلىپ ئويۇنچۇق قىلامدىكەن؟ساقچى باشقۇرمامدۇ دىيىشىمىز مۇمكىن،مەدىنىيەت ئىدارىسى باشقۇرمامدۇ دىيىشىمىز مۇمكىن،يەنە ئاللىكىملەر ھۆكۈمەتتىنمۇ ئاغرىنىشىمىز مۇمكىن.بىراق ھەر قايسى قانۇن ئىجىرا تارماقلىرىنىڭمۇ ھۇقۇقى چەكلىك،تولىمۇ چەكلىك.نىمىشقا بۇ يەرگە كىردىڭ دەپ كىرگەن ئادەمنى ئىتىۋىتىش ياكى سولىۋىتىش ھۇقۇقى بولمايدۇ ساقچىدا،نىمىشقا بۇ يەرنى ئاچتىڭ دەپ مال مۆلۈكىنى مۇسادىرە قىلىپ پىچەتلەپ تاقىۋىتەلمەيدۇ مەدىنىيەت ئىدارىسىدىكىلەر.بۇ جەمىيەت شۇنداق جەمىيەت،ھەممىلا ئادەم ئاڭلىق،ھەممىلا ئادەم پۇل تاپاغا تۇرۇشسا،ھەممىلا ئادەم قانۇننى بىلىپ رىئايە قىلىپ كەتسە ساقچىنىڭمۇ،مەدىنىيەت ئىدارىسىنىڭ قانۇن ئىجىراچىللىرىمۇ ياستۇقنى قىرلاپ ئوخلىسا بولىدۇ.لىكىن ئادەمدە ساپا ،مىللەت غۇرۇرى،ئىشەنىچ تۇيغۇسى بولمىسا نىمە ئامال.
ئادەمنى باشقۇرۇشتىنمۇ تەس ئىش يۇق دەيمەن.شۇنداق ئورۇننىڭ ئالدىغا بىرۋىلىپ ئۇيۇنچۇقخانىغا كىرىپ كىتىۋاتقانلارغا:ئۇينىماڭلار،ئىسىت تاپقان پۇلۇڭلار........................دەپ سۆزلەپ بىقىڭ،سۆزىڭىزنى ئاڭلايدىغانلار قانچىلىك،ئاڭلىمايدىغانلار قانچىلىك،تىخى تاياق يەپ قىلىشىڭىزمۇ مۇمكىن.ئادەم دىگەنگە بىر نىمە دىگىلى بولمايدۇ.ئەقلى بارلار كىرمەڭلار،كىرگەنلەر كىيىن پۇشايمان قىلىدۇ بەرىبىر،

يىزا يوللىرىدا توپ توپ ئولتۇرۇپ كەتكەن ياشلارغا قاراپ ئادەم ئىچىنىپ قالىدىكەن،ئۇلارمۇ يىزا ياشلىرى،تولىسى يا ئىتىز ئىشى خۇش ياقمايدىغان،يا سىرىتتا ئىشلەش خۇش ياقمايدىغان،قۇڭى بوش باللار ئاز ئەمەس.ئىتىز ئىشىغا چىدىمايدۇ،قىرى ئاتا ئانىسىنى ئىتىز ئىشىغا سىلىپ قۇيۇپ ئۆزى سىرىتتا ،يا سىرىتتا،بازاردا خەققە ئىشلىمىگەن،ئىشلىسىمۇ تىخى ھاكاۋۇر دەڭلا ئۇنداق باللار.بۇنداق ياشلار پۇل تىپىشنى خىيال قىلىدۇيۇ،تىرىشىشنى تازا خالاپ كەتمەيدۇ،ئاز دەسمىي بىلەن كۆپ پۇل تىپىشنى ئارزۇ قىلىشىدۇ،نەدە شۇنداق ئىش باركەن.بازاردىكى پۇزۇر ياسانغان تىجارەتچى باللارغا،سىرىتتىنكىرگەن تىجارەتچىلەرگە قاراپ تىجارەت قىلسىلا پۇلنى شۇنداق ئاسان تاپقىلى بولىدىغان ئوخشايدۇ دەپ قالامدىكىن.شۇنىڭ بىلەن ئۇنى قىلىشقا تەمشىلىپمۇ باقىدۇ،لىكىن ئۇزاق ئۆتمەي يالتىيۋالىدۇ.ۋاي،،،،،،،،،ھەممە ئىشنىڭ دەردى بار دەيمەن.ئۇلارمۇ ئاندىن ئاتا ئانىسىدىن ئالغان پۇلنى قاۋاققا بىرىپ ھاراقنى كاللىغا ئىچىۋالغاندا ئۇيۇنچۇقخانىغا ماڭىدۇ دە مەرتلىگى تۇتۇپ ئويناۋىردۇ.خۇجايىنىمۇ خۇش خۇش دەپ پۇلنى ئىلىۋىردۇ،پۇلى تۈگىگەنلەر ئۆيگە قايتار پۇلى يۇق شۇ قارا قۇرساق خۇجايىنلارغا يالۋۇرۇپ يۈرگەن چىدىمىسا جىدەل چىققان،ئاخىردا نىمە بولىدۇ،تاياقنى يىتىم يەڭ،تاماقنى ھىكىم دەپ بار بولغىلۇق شۇ ياشلىرىمىزغا بولىدۇ.
