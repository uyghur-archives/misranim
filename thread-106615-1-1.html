كۆزلىرىم ئىنتىزارلىق بىلەن سانسىز ئادەملەر توپى ئىچىدىن ئۇنىڭ گەۋدىسىنى ئىزدەيتتى،لىكىن تاپالماي ئۇمۇدسىزلىك بىلەن باشقا ياقلارغا تەلمۇرەتتى...
شۇنداق ئىزدەش بىلەن يېرىم سائەت ئۆتۈپ كەتكەن بولسىمۇ ئۇ تۇنۇش سىماھنى ئۇچرىتالماي،ئۈمۈدسىزلىك بىلەن لەۋلىرىم ئۆمچەيگىنىچە يولۇچى كۈتۈش زالىدىن سىرتقا ماڭدىم،شۇ ۋاقىتتا نۆۋەتچى ئايرۇپىلاننىڭ ئۇچقانلىغى ئىلان قىلىندى...يۈرۈگۈم كۆكرەك قەپىزىمنىڭ ئىچىگە سىغمايۋاتقاندەك نەپەس ئالالماي قالدىم،نۆۋەتچى ئايرۇپىلان ئۇنىلا ئەمەس ،پۇتۇن ئارزۇ -ئارمانلىرىمنى بىرگە ئېلىپ كۆككە كۆتۈرۈلگەن ئىدى.كۆڭلۈم شۇنچىلىك يېرىم بولدى،ئۇنىڭ ماڭا چۈشەندۇرۈش پۇرسىتىمۇ بەرمەي،مىنى ۋە ھۇجۇدۇمدىكى  ئىسسىق بىر جاننى تاشلاپ كەتكىنى ماڭا تولىمۇ ئەلەم بولدى.ئەجىبا ئۇنىڭ مىنى ياخشى كۆرىمەن دىگەنلىرى،مەن ئۈچۈن ھەممىدىن كىچىشكە رازى  دىگەنلىرى يالغانمىدۇ؟تويدىن كىيىن ماڭا شۇنچە كۈيۈنۈشلىرى ،مەن ئۈچۈن پايپىتەك بۇلۇشلىرى ،سائەت ئاتلىماي تېلىفۇن ئۇرۇپ ئەھۋال سوراشلىرى ...ھەمممىسى ساختىلىقمۇ؟مىنى مۇشۇنداق ئازاپتا قالدۇرۇش ئۈچۈن قۇرۇلغان قىلتاقمۇ؟
قەلبىم زىددىيەتلىك خىياللار بىلەن تولغان ئىدى ،زالدىن قانداق چىققىنىمنى بىلمەيمەن ،ئازراق مېڭىپلا مەس ئادەمدەك دەلدەڭشىپ كەتتىم-دە،ئانچە يىراق بولمىغان جايغا قويۇلغان سۈگەت ئاستىدىكى تاش ئورۇندۇققا بىرىپ ئولتۇرۇپ ئېغىر تىندىم،ھۇجۇدۇمدىكى ئۇ سەبى جان بۇ كىلىشمەسلىكنى سەزدىمۇ ،بوشقىنە تەۋرەپ قويدى،بۇ تۇنجى قېتىم تەۋرىشى ئىدى،ھەم خوشاللىق،ھەم قايغۇ ئارلاشقان غەلىتە ھىسىيات بىلەن قورسۇغۇمنى ئاستا سىلاپ قويدۇم....


تۈنۈگۈن ئۇ كىچە ئىشلەيدىغان بولغاچقا مىنى ئاپامنىڭ ئۆيىگە ئاپىرىپ قويغان ئىدى،سائەت 10بولاي دىسىمۇ مىنى ئالغىلى بارمىدى،شۇنىڭ بىلەن ئۆزۈم ئۆيگە بېرىشقا مەجبۇرى بولدۇم،ئويگە كەلگىنىمدە ئۇ يوق ئىدى،لىكىن ئۈستەل ئۈستىدە مىنىڭ كۈندىلىك خاتىرەم ۋە خاتىرە ئاستىغا قىستۇرۇلغان بىر پارچە خەت تۇراتتى....
يۈرۈگۈم جىغلا قىلىپ قالدى،ھەم بۇ خاتىرىنى قانداق تاپقانلىغىغا ھەيران قالدىم،چۈنكى ئۇ ئەزەلدىن مىنىڭ نەرسىلىرىمنى قالايمىقان ئاختۈرمايتتى،مىنىڭ شەخسىيىتىمگە ھۆرمەت قىلاتتى،چىگىش خىياللار ئىچىدە خەتنى قولۇمغا ئالدىم.

خەت مۇنداق باشلانغان ئىدى

مەن كەتتىم،گەرچە ساڭا،ۋە سىنىڭ ھۇجۇدۇڭدىكى مۇھابىتىمىزنىڭ مەسئۇلى بولغان پەرزەنتىمگە قىيالمىساممۇ سىنىڭ ئازاۋىڭ بەدىلىگە ئۆزۈمنىڭ كۆڭلىنى خوش قىلىپ ياشىغۇم كەلمىدى،سىنى قانچىلىك ياخشى كۆرىدىغىنىمنى سەزگەنسەن؟ساڭا ئىرىشىش بۇ دۇنيادىكى ئەڭ زور خوشاللىغىم ئىدى،سەن مىنىڭ نەزىرىمدە ئۇماق بىر پەرىشتە ئىدىڭ،سېنى سۈيەتتىم،سىنىڭ بالىلارچە ئۇماق قىلىقلىرىڭ،تاتلىق سۆزلىرىڭ ماڭا ئالەمچە ھۇزۇر بېغىشلايتتى،بىر ئىشلارنى قاملاشتۇرۇپ قىلالماي تەمتىرەشلىرىڭمۇ شۇنچىلىك سۈيۈملۈك ئىدى،ئىسىڭدىمۇ شورپىنىڭ تۇزىنى تەڭشىيەلمەي شاكراپ قىلىۋەتكىنىڭدىمۇ ئۇنى شۇنداق مەزە قىلىپ سىنىڭ بولدى يىمەڭ دىگەنلىرىڭگە قارىماي،مەزە قىلىپ يىگەنلىرىم....
ئىشقىلىپ سەن ھاياتىمنىڭ بىر قىسمى ئىدىڭ،سەندىن ئايرىلىشنى تەسەۋۇر قىلالمايتتىم،سەن تويىمىزنىڭ ئالدىدا باشقا بىرىنى ياخشى كۆرىدىغانلىغىڭنى دىگەن ۋاقتىڭدىمۇ ئۇنىڭغا پەرۋا قىلمىغان ئىدىم،بىزنىڭ ئاتا-ئانىمىز تۇنۇشتۇرغان بولسىمۇ،بىر كۆرۈپلا سىنى ياخشى كۆرۈپ قالغان،مىنىڭ ياخشى كۆرىدىغىنىم ھەرگىزمۇ سىنىڭ ئۇماق چىرايىڭ ئەمەس،بەلكى بالىلارچە مىجەزىڭ ئىدى،ئويلىغىنىڭنى ئۇدۇللا دەيتتىڭ،شۇنچىلىك ئۇچۇق -يۇرۇق ئىدىڭ،قىسقىسى سەندىكى ئۇ مىجەزنى باشقا قىزلاردىن تاپقىلى بولمايتتى...
شۇڭلاشقا تويدىن كىيىن چوقۇم بارچە مىھرىمنى ئۇنىڭغا بىرىپ باشقىلارنى ئۇنتۇلدۇرىمەن دەپ خام-خىيال قىلغان ئىكەنمەن،ئەسلىدە ئۇنىڭ قەلبىڭدىكى مەڭگۈ تاشتەك قاتقان ئابىدىسىنى،مىنىڭ بۇ مۇھابەتكە ئىنتىزار قەلبىم يۇيۇپ چىقىرىشقا ئامالسىز ئىكەن ئەمەسمۇ؟مەن چۈشەندىم ،ھەممىنى چۈشەندىم،كۈندۈلۈك خاتىرەڭ ماڭا ھەممىنى سۆزلەپ بەردى،بىز توي قىلغاندىن بۇيان ئازاپ ئىچىدە ياشاۋېتىپسەن،مەن بۇنىڭغا چىدىمايمەن،مەن چىن دىلىمدىن سۆيگەن ئادىمىمنىڭ ئازاپ چىكىشىگە قاراپ تۇرالمايمەن،مەن كەتتىم...بىرەر ئاي سىرتلاردا يۈرۈپ مىڭەمنى سەگىتىپ كىلەي،ئۆزەمگە ھەم ساڭا ئويلىنىش پۇرسىتى قالدۇراي،مەن قايتىپ كەلگەندە ئۇنىڭسىز ياشاشقا كۆنسەڭ بوينىمغا ئېسىل سەن ئۈچۈن باغرىمنى كەڭ ئاچاي،ئەگەردە ئۇنداق بولمىسا......

مىڭەمدىن تۇتۇن چىقىپ كەتتى،ئۇ ئالدىراش ماڭغان بولسا كىرەك،كۆمپىيوتىرنىمۇ ئۆچۈرمىگەن ئىدى،ئۇنىڭدىكى ئۇچۇرلاردىن ئۇنىڭ  ئۇرۇمچىگە ئۇچىدىغان ئايرۇپىلان بىلىتى زاكاس قىلغانلىغىنى بىلدىم-دە،ھېچنىمىگە قارىماي ئايرۇپىلان ئىستانسىسىغا تاكسى توسۇپ ماڭدىم....

ئەمدى كۆز ياشلىرىم مارجاندەك تۆكۈلۈشكە باشلىدى،دەسىۋىدە ئۈنسىز يىغلىغان بولسام ،بارا بارا ئۆكسگەن ئاۋازىم سىرتىغا چىقىپ كەتتى،بىر چاغدا قوللىرىمنىڭ ئاراچلىرىدىن ئۇنىڭ ئاياقلىرىنى كۆردۇم-دە،ئىتتىكلا بېشىمنى كۆتۈردۇم،ئۇ ماڭا قاراپ تۇراتتى،ئۇنىڭ كۆزلىرىدىمۇ ياش تامچىلىرى پىلىلداپ قالغان ئىدى...
مەن ئورنىمدىن تۇرۇپلا ئۇنىڭ باغرىغا ئۆزۈمنى ئاتتىم،گويا ئۇنى ئەمدى قويىۋەتسەم مەڭگۈ يۈتتۈرۈپ قويىدىغاندەك مەھكەم قۇچاقلىدىم،شۇ ھالدا بىر دەم تۇرغاندىن كىيىن ئۇ بىلەكلىرىمدىن تۇتۇپ مىنى ئۆزىگە قاراتتى ۋە كۆزلىرىمگە مىختەك قادىلىپ تۇرۇپ-ماڭا ۋەدە بەر،سەندىن باشقا ھەممە ئىشلارنى ئۇنتۇپ كىتىمەن دەپ ۋەدە بەر-،دېدى.
مەن بېشىمنى بوشقىنە لىڭشىتتىم،ئۇ كۆز ياشلىرىمنى سۈرتۈپ مىنى قايتىدىن باغرىغا باستى.


مەن ئاتا-ئانامنىڭ مەجبۇرلىشى بىلەن،ئۆزوم شۇنچىلىك ياخشى كۆرىدىغان ئادەمدىن ئايرىلىپ ،ماڭا ناتۇنۇش بولغان بىرسى بىلەن توي قىلماقچى بولدۇم،بىز تويدىن بۇرۇن پەقەت 3قېتىملا كۆرۈشتۇق،3-قېتىم كۆرۈشكەن ۋاقتىمىزدا ئۇنىڭغا،كۆڭلۈمدە باشقا بىرىنىڭ بارلىغىنى ئېيتتىم،لىكىن ئۇ پەرۋا قىلىدىغاندەك ئەمەس ئىدى،ئۇ ئاز كەلەگەندەك ئۆزىنىڭمۇ  شۇنداق شىرىن مۇھاببەتلىك ۋاقىتلارنى باشتىن كەچۈرگەنلگىنى ھازىر پەقەت مىنىلا ياخشى كۆرىدىغانلىغىنى ئېيتتى،ئۇنىڭ بۇ سۆزلىرى مىنىڭ كۆڭلۈمنى تېىخىمۇ يېرىم قىلدى،مىنىڭ قارشىلىق كۆرسىتىشلىرىم،نالە-پەريادلىرىم كارغا كەلمىدى ،قانچىلىك قارشى تۇرماي بۇ توي ئاخىرى بولدى،ئۇلار مىنىڭ ئۇقۇشۇمنى تاماملاشقا ئۈلگۈرمەيلا توينى قىلىۋەرگەن ئىدى.

ئەتە توي بولىدىغان كىچە پەقەتلا ئۇيقۇم كەلمىدى،قەلبىمدىكى ئازاپلىق تۇيغۇلارنى خاتىرەمگە پۇتۇپ چىقتىم،بۇرۇندىنلا يېزىقچىلىققا قىزىقاتتىم،ئۆزۈمنىڭ ھىس تۈيغۇلىرىنى خاتىرە قىلىپ ساقلاشقا ئادەتلەنگەن ئىدىم،قانداق چاغلاردا ئۇخلاپ قالدىمكىن،سەھەردە ئاپام ئوغۇل تەرەپنىڭ نىكاھقا كەلگەنلىگىنى ئېيتىپ ئويغاتتى.

مەن ناھايىتى روھسىز ھالەتتە ئورنۇمدىن تۇردۇم،قاپاقلىرىم ئىششىپ كەتكەن ئىدى.
توي باشلاندى،كۆز ئالدىمدا بولىۋاتقان بۇ ئىشلارنىڭ رىئاللىق بولماسلىقىنى ،كۆزۈمنى ئاچساملا ئويغۇنۇپ كىتىدىغان بىر چۈش بولۇپ قېلىشىنى ئۈمۈد قىلاتتىم،لىكىن بۇ رىئاللىق ئىدى،مەن بۇگۈن ئۇنىڭدىن مەڭگۈ ئايرىلىپ قالغانلىغىمغا ئىشەندىم،يۈرۈگۈم پارە-پارە بولۇپ كەتكەن ئىدى،بىز ئالاقىلاشمىغىلى ئۇزۇن بولغان بولسىمۇ،بۇگۈن ئۇنىڭ يۈرىكى تۇيدىمۇ ،ياكى باشقىلاردىن توي خەۋرىمنى ئاڭلاپ قالدىمۇ ،ئىشقىلىپ ئۇ ئارقا-ئارقىدىن تېلىفۇن ئۇرىۋەردى،مەندە تېلىفۇننى ئالغۇدەك جۇرئەت،ئۇنىڭ ئاۋازىنى ئاڭلىغۇچىلىك تاقەت قالمىغان ئىدى،تېلىفۇنىمنى باغرىمغا باسقىنىمچە يىغلاپ كەتتىم...
ئاخىرى ئۇ قىسقا ئۇچۇر ئەۋەتتى.
كەچۈرگىن،قۇتلۇق كۈنۈڭنى تېلىفۇن قىلىپ تەبرىكلەي دىگەن ئىدىم،لىكىن ئالدىراش ئوخشايسەن،بەخىتلىك كۈنلەر ئەمەسمۇ؟مەن توغرا چۈشۈنىمەن،شۇڭلاشقا ئۇچۈر ئارقىلىق بولسىمۇ تەبرىكلەپ قوياي،بەختىڭگە كۆز تەگمىسۇن،مەڭگۈ بەخىتلىك ياشا،مەن ئۆز بەختىمنىمۇ ساڭا قوشۇپ ئەۋەتتىم،سەن بىر ئۆمۇر مىنىڭ بەختىمگە قەرىزدار!!!
ئۇنىڭ بۇ ئۇچۇرىنى كۆرۈپ ،دىلىمدىكى يارا ئېغىز ئېلىپ قاناپ كەتكەن ئىدى،تاكى توي يۆتكىلىپ بولغانغا قەدەر كۆز ياشلىرىم قۇرىمىدى،باشقىلار بۇنى توي قىلغان قىزلاردىكى رەسمىيەت ئۈچۈن يىغلاش دەپ چۈشەنگەندۇ بەلكىم،لىكىن مىنىڭ ئۇ ئىشلارغا كۆڭۈل بولگۈچىلىك مادارىم يوق ئىدى.
مىھمانلار ئۇزاپ ئۇنىڭ دوستلىرى بىزنى ئۆزىمىزنىڭ ئۆيىگە ئېلىپ كەلدى،ئۇلار بىردەم كۈلكە چاخچاقلار بىلەن ئولتۇرغاندىن كىيىن،ھەممىسى قايتىشتى،ئۇ دوستلىرىنى ئۇزۇتۇپ قويغىلى پەسكە چۈشۈپ كەتكەن ئىدى،مەن ئۆيدە يالغۇز قالدىم،يەڭگىلەرمۇ ئۇنىڭ چوڭ ئۆيىدە قالغان ئىدى،مەن بىر خىل ئەنسىزلىك ئىچىدە يۈرەك سوقۇشلىرىمنى ئاڭلاپ ئولتىراتتىم،ئىشىكنىڭ ئېچىلغان ئاۋازىدىن چۈچۈپ بېشىمنى كۆتۈردىم،كۆزلىرىمىز ئۇچرىشىپ قالدى،ئۇ ماڭا قاراپ كۈلۈمسىرىدى،مەن تىزلا كۆزلىرىمنى باشقا ياققا بولىۋالدىم،يۈرۈگۈم تېىخىمۇ ئەنسىز دوپۇلدەشكە باشلىدى...
ئۇبىر ئېغىزمۇ گەپ قىلمىدى،تىلۋۈزۈرنى ئېچىپ قويۇپ يېنىمغا كىلىپ ئولتۇردى،ئارىدىن قانچىلىك ۋاقىت ئۆتتىكىن،ئۇ ئاخىر ئېغىز ئاچتى.
-بۇ تويغا نارازى ئىكەنلىگىڭىزنى بىلىمەن،گەرچە ئاپامنىڭ ئارزۇسى بويىچە توي قىلغان بولساممۇ ،لىكىن سىزنى ياخشى كۆرىمەن،مەڭگۈ قەدىرلەيمەن،ئۆتكەن كۈنلەرنى ئۇنتىماقنىڭ ئۇنچە ئاسان ئەمەسلىگىنى بىلىمەن،لىكىن چوقۇم ئۇنى ئۇنتۇپ كىتەلەيدىغىنىڭىزغا ئىشىنىمەن،مىنىڭ سىزگە بولغان مۇھابىتىم چوقۇم قەلبىڭىزنى ئىرىتەلەيدۇ،ئۇ شۇگەپلەرنى دىگەچ،قوللىرىمنى ئالقانلىرى ئارىسىغا ئالدى...گوييا قورقۇنۇچلۇق مەخلۇققا يولۇققان ناتىۋاندەك پۇتۇن بەدەنلىرىم تىكەنلىشىپ كەتتى ۋە قولۇمنى تىزلا تارتىۋالدىم.
