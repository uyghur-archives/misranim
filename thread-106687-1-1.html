  قەدىرلىك قېرىنداشلىرىم ، توردۇنياسى ئارقىلىق تۇنۇشۇپ ، بىلمىگەنلىرىمىزنى بىلدۇق ھەم بىلدۈردۇق ، بۇلۇپمۇ يىمەك ئىچمەك جەھەتتە بەكرەك ئەھمىيەت بېرىدىغان بۇلۇپ قالدۇق . بازارلاردىكى توخو گۈشى ، بېلىق گۈشى ، توخو تۇخۇمى قاتارلىقلارنى كۆرسەك بەكرەك دىققەت قىلىدىغان ، دوست يارەنلەرنىڭكىگە مىھمان بۇلۇپ بارغاندا بولسا نەدە بېقىلغان توخو ، قايسى دەريادا بېقىلغان بېلىق دەپ سوراپ تۇرۇپ يەيدىغان بولدۇق . لېكىن رىستۇرانلاردىكى توخو ، بېلىقلارنىچۇ ؟ رىستۇران ئۈستىلىدىن چىقىرىۋەتكىلى بولماسمۇ ؟ نىمىشقا ھەرقېتىم توي تۈكۈنلەرگە كىرسەك ئىشەنچسىز توخو ، بېلىقلار بىزىرىپ تۇرىۋىرىدۇ ھەم بىزلەر يەۋىرىمىز؟ نىمىشقا كاللىمىزنى سىلكىۋەتمەيمىز ؟ تېخىچىلا توخو قورۇمىسى يىدۇق ، بېلىق كاۋىپى يىدۇق دەپ يۈرىۋىرىمىز ؟
   ئۆتكەندە ئىشخانىمىزدىكى خىزمەتداشلار رىستۇراندا ئولتۇرماقچى بولدۇق . مىنىڭ تەلىپىم بۇيىچە ، خىزمەتداشلىرىم توخو گۈشى بىلەن بېلىق گۈشىنى يىمەسلىكنى قارار قىلدى . ئۆزەم رىستۇران خۇجايىنىغا توخو گۈشى بىلەن بېلىق گۈشىنى يىمەيدىغانلىقىمىزنى ، ئۇنىڭ ئورنىغا باشقا قورۇمىلارنى چىقىرىپ بېرىشنى ئېيتتىم . ئۇ ماڭا ئورنىغا گۆش قورۇمىسى قىلسا تەننەرقى ئېشىپ كىتىدىغانلىقىنى ئېيتتى . مەن گۆشنى سەل جىقراق چىقىرىپ بېرىشنى ، قىيمىلىق تائامنى بىر تەخسە ئارتۇق چىقىرىپ بېرىشنى تەلەپ قىلدىم . شۇنىڭ بىلەن ھەرھالدا ئىشەنچلىك تاماقلارنى يىدۇق . لېكىن قارىسام ئەتراپتىكى ئۈستەللەرنىڭ ھەممىسىدە ئاشۇ بىز يىمەيمىز دىگەن قورۇمىلار بىزىرىپ تۇرىدۇ . ھەممىسى ئىشتىھا بىلەن يەپ ئولتۇرۇپتۇ . بۇنىڭغا نىمە دەيمىز ئەمدى ؟
  بىر تۇرۇپ مەن بەك ئېشىپ كىتىۋاتىمەنمۇ نىمە ؟ دەپ ئۆز ئۆزەمدىن گۇمانلىنىپ كىتىمەن . ئەتراپىمدىكىلەرگە يىمەك ئىچمەك توغرىسىدا ھارماي سۆزلىدىم . قۇبۇل قىلغانلارمۇ بولدى . قۇبۇل قىلماي ، مىنى مەنسىتمىگەنلەرمۇ بولدى . ھەتتا ئوقۇمۇشلۇق ئادەملەرمۇ ماڭا غەيرى پوزىتسىيە تۇتتى . بەزىلەر تېخى تور دىگەندە خالىغانچە يېزىۋىرىدۇ ، ئۇنچىلىك سەگەكلىشىپ كەتمەڭ ، يەيدىغان نەرسە تاپالماي ، ئۈلۈپ قالماڭ يەنە ... دىگەنلەرمۇ بولدى .
   مەن بىر ئوقۇتقۇچى ، بازارلاردىكى ئىسمى ئۇيغۇرچە ، زاۋۇتى خەنزوچە مەھسۇلاتلار ، ئىچىملىكلەر ، خالتىلىق سۈتلەر ، ھەرخىل گۆشلەر ، خالتىلىق چۆپ ، لازا قاتارلىق ئەخلەت يىمەكلىككە ئائىت رەسىم ، سۆرەت ، يېزىق كۆرۈنۈشلىرىنى توردىن ئىزدەپ يۈرۈپ ئوقۇغۇچىلىرىمغا دەرسكە بىرلەشتۈرۈپ ھارماي ، قايتا قايتا كۆرسەتتىم ، شۇنىڭ بىلەن كۆپ قىسىم ئوقۇغۇچىلىرىم دىققەت قىلىدىغان بولسىمۇ ، بەزىلىرى يەنىلا يەۋاتىدۇ . مىنى كۆرگەندە يۇشۇرۇپ ، قېچىپ تۇرۇپ يەۋاتىدۇ . شۇنداقتىمۇ مەن تا ھازىرغىچە سۆزلەشتى توختاتمىدىم ھەم توختاتمايۋاتىمەن . ئۆتكەندە بىر سىنىپقا دەرسكە كىرىشىم بىلەنلا ئوقۇغۇچىلىرىم تەنەپۇستا يەۋاتقان نانلىرىنى يۇتۇپ بۇلالماي ئالدىراپ ئورۇنلىرىغا بېرىپ ئولتۇردى . ئۇلارنىڭ دىيىشىچە ، مىنىڭ تەسىرىمدىن ئۆيلىرىدىن نان ئەكىلىۋېلىپ ، شۇنى يەۋىتىپتۇ . بۇ قېتىم مەن ئۇلاردىن شۇنچىلىك سۈيۈنۈپ كەتتىم . نىمىلا بولمىسۇن مىنىڭ تەربىيەم ئۇلارغا سىڭىپتۇ . ئەمگىكىم بىكار كەتمەپتۇ . ھەريىلى يېڭى ئوقۇغۇچىلارغا دەرسكە كىرىمەن . شۇ ئوقۇغۇچىلارغا تەسىر قىلالىسام ، ئۇلار يىمەك ئىچمەك جەھەتتىن ئىھتىيات قىلسا دىگەن ئىدىيەدە يىمەك ئىچمەك جەھەتتە ھەردائىم ئىزدىنىپ تۇرۇۋاتىمەن .
  قەدىرلىك قېرىنداشلىرىم ، سىلەرچە مىنىڭ دىگەنلىرىم توغرىمۇ ؟ قاچانغىچە شۈبھىلىك يىمەكلىكلەرنى بىلەلمەي يۈرىۋىرىمىز ؟
