مۇبادا سىزنى ئۇقۇشماسلىقتىن ساراڭلار دوختۇرخانىسىغا ئەكىرىپ قويسا ئۆزىڭىزنىڭ ساراڭ ئەمەسلىكىنى قانداق ئىسپاتلايسىز ؟

بۇنىڭدىن بىر مەزگىل ئىلگىرى ئامرىكىدا ، روھىي كىسەللەرنى تۇشۇشقا مەسئۇل بىر شوپۇر ،يولدا ئىھتىياتسىزلىقتىن 3نەپەر روھىي كىسەلنى قاچۇرۇپ قويغان ،خىزمىتىدىن ئايرىلىپ قالماسلىق ئۈچۈن بىر ئاپتۇبۇس بىكىتىگە بېرىپ ھەقسىز ئاپىرىپ قۇيۇشقا ۋەدە قىلىپ نەپەر يولۇچىنى ماشىنىغا سېلىپ ئۇدۇللا روھىي كىسەللىكلەر دوختۇرخانىسىغا ئاپىرىپ قويغان .

كىيىن بىر مۇخبىر بۇلارنىڭ قانداق ئۇسۇل بىلەن ئۆزىنىڭ نورمال ئادەم ئىكەنلىكىنى ئىسپاتلاپ دوختۇرخانىدىن چىقىپ كەتكەنلىكىگە قىزىقىپ ئۇلارنى رىياسەت قىلغان .

مۇخبىر : سىز دوختۇرخانىغا ئېلىپ كېلىنگەندىن كىيىن قانداق ئامال بىلەن چىقىپ كىتىشنى ئويلىدىڭىز ؟

بىرىنچى ئادەم : مەن بۇيەردىن چىقىپ كىتىش ئۈچۈن ، مەن چۇقۇم ئالدى بىلەن ئۆزەمنىڭ روھىي كىسىلىم يوقلىقىنى ئىسپاتلىشىم كىرەك دەپ ئويلىدىم .

مۇخبىر : سىز قانداق ئىسپاتلىدىڭىز ؟

بىرىنچى ئادەم : مەن ئۇلارغا «يەر شارى يۇمىلاق » دىدىم ، بۇ بىر ھەقىقىتقۇ ، مەن ھەقىقەتنى سۆزلىسەم باشقىلار مىنى ساراڭ دەپ ئويلىمايدۇ دەپ ئويلۇدۇم .

مۇخبىر : ئاخىرى سىز مۇۋەپپەقىيەت قازاندىڭىزمۇ ؟

بىرىنچى ئادەم : مەن بۇ گەپنى 14-قېتىم دىگەن ۋاقىتتا ،سېستىرا ماڭا پەللەك ئۇكىلىدىن بىرنى سالدى .


مۇخبىر : سىز  روھىي كىسەللىكلەر دوختۇرخانىسىدىن قانداق قۇتۇلۇپ چىقتىڭىز ؟

ئىككىنچى ئادەم : ئەمىلىيەتتە مەن ۋە بىرىنچى ئادەم ھەر ئىككىلىمىزنى ئۈچۈنچى ئادەم قۇتقۇزۇپ چىقتى ، يەنى ئۈچۈنچى ئادەم قۇتۇلۇپ چىققاندىن كىيىن ساقچىغا مەلۇم قىلىپ بىزنى قۇتقۇزىۋالدى .

مۇخبىر : ئۇنداق بولسا سىز ئەينى ۋاقىتتا قايسى ئامال بىلەن ئۆزىڭىزنىڭ نورمال ئادەم ئىكەنلىكىنى ئىسپاتلىماقچى بولدىڭىز ؟

ئىككىنچى ئادەم : مەن ئۇلارغا ئۆزەمنىڭ بىر جەمىئەتشۇناس ئىكەنلىكىمنى ئېيتتىم ، ۋە ئامرىكىنىڭ سابىق پىرىزدىنتنىنىڭ كىلىنتۇن ئىكەنلىكىنى ، ئەنگىلىيەنىڭ سابىق باش ۋەزىرىنىڭ بىلايىر ئىكەنلىكنى ئېيتتىم ، مەن جەنۇبي تىنچ ئوكياندىكى ھەرقايسى ئارال دۆلەتلەرنىڭ داھىلىرىنىڭ ئىسمىنى دەۋاتقان چېغىمدا ،سېستىرالار ماڭا ئۇكۇل ئۇردى ، مەن داۋاملىق سۆزلەشكە پېتىنالمىدىم .


مۇخبىر : پەقەت سىزلا مۇۋەپپەقىيەتلىك ھالدا نىرۋا كىسەللىكلەر

دوختۇرخانىسىدىن چىقىپ كىتىپسىز ، سىز قانداق قىلدىڭىز ؟

ئۈچۈنچى ئادەم : مەن دوختۇرخانىغا ئېلىپ كىلىنگەندىن كېيىن

ئارتۇقچە گەپ قىلىپ ئۆزەمنىڭ نورمال ئادەم ئىكەنلىكىنى ئىسپاتلاشقا ئۇرۇنمىدىم .تاماق يەيدىغان ۋاقىتتا تاماق يىدىم ، ئۇخلاش ۋاقتى كەلگەندە ۋاقتىدا ئۇخلۇدۇم ، سېستىرالار مىنىڭ ساقىلىمنى ئېلىپ قويغاندا مەن ئۇلارغا رەھمەت ئېيتتتىم ،مۇشۇنداق داۋاملاشتۇرۇپ  28-كۈنىگە كەلگەندە ئۇلار مىنى دوختۇرخانىدىن چىقىرىۋەتتى .

شۇڭا سىز ئۆزىڭىزنىڭ قانداق ئادەم ئىكەنلىكىڭىزنى ئىسپاتلىشىڭىزنىڭ زۇرۈرىيىتى يوق ،  ھەممە ئىشنىڭ تەبئى بولغىنى ياخشى .

ئەگەر سىز ئاشۇ ئۈچ ئادەمنىڭ بىرەسى بۇلۇپ قالغان بولسىڭىز ، قانداق قىلاتتىڭىز ؟

مەنبە : مىكرو ئۇچۇر ئاممىۋى سۇپىسىدىن ئابلەت ئەيسا تەرجىمىسى .
