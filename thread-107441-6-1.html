بۇ قۇرلۇشنى ھەققەتەنمۇ ئۇيغۇر قول ھۇنەر ،مىللى مىمارچىلىقنىڭ ئەڭ يۇقىرى پەلىسى ئەكىس ئەتكەن دىسەك ئارتۇق كەتمەيدۇ،ھىققەتەنمۇ شۇنچىلىك پۇل كەتكىنى راس،پۇتۇن قۇرلۇش ئۇيغۇر بىناكارلىقىنىڭ ئەڭ يارقىن ئوبىرازى بىلەن ھازىرقى زامان تەسۋىر تىخنىكسى قولۇنۇپ قورۇلغان ،بۇ جايدا ئۇيغۇر تائاملىرىنىڭ ئوبىرازىنى يەنمۇ يۇكسەلدۇرۇش بىلەن بىرگە ئۇيغۇر خانلىقىنىڭ ئەينى دەۋىردىكى خاسلىقى نامايەن قىلىنغان ،بۇ قۇرلۇشنىڭ بىزىلىشى ئىچىدىكى مىللى خاسلىق كەلگەن ساياھەتچىلەرگە  مىللىتمىزنىڭ ئەينى زاماندىكى ھەيۋىتىنى ھەققى نامايەن قىلالايدۇ. ئۇيغۇر ئۇن ئىككى مۇقام بىلەن مەشھۇر ئۇيغۇر قامۇسى قۇتادغۇبىلىكنىڭ  تۇرۇكلەرگە ئۇستىلىق بىلەن ئۇيۇپ نەقىشلىنىشى  يوقۇلۇش ئالدىدا تۇرغان يازمىلارنىڭ ،ئەجداتلىرمىزنىڭ تارىخلىرى  تەسۋىرلەنگەن سۇرەتلەرنىڭ كۇرۇنۇپ تۇرۇشى ئۇيغۇر مىللىتىنىڭ باتۇرلىقىنى ھەققى نامايەن قىلغان . بۇ قۇرلۇشقا ھەرقايسى جايلاردىكى  تاماقخانلارنى ،تۇرۇكچە تاماخخانىلارنى ھەم ئۇرۇمچىدىكى داڭلىق تاماخانىلارنى  ھەرگىزمۇ  بۇ قۇرلۇشقا سىلشتۇرغىلى  بولمايدۇ، بۇ ھەققى مىللى ئەنئەنى ئۇزىگە خاسلاشتۇرغان قۇرلۇش ،بۇنىىغا ھازىرچە دۆلىتىمىزدە تىخى بۇ خىل ئۇسلۇپدا ياسالغان بىرمۇ ئۇرۇن يوق ! جوڭگۇدا بىرلا دىسەك ھەققىتەنمۇ ئارتۇق  كەتمەيدۇ، ئىستىمال سەۋىيسىگە كەلسەك بۇنى قۇرلۇشنى كۇرۇش ئەينى دەۋىردىكى مىلىتمىزنىڭ ھەيۋىسى ھىس قىلالايسىز ، قۇرلۇشنىڭ ھەيۋىتىنى بىلدۇرۇش ھەم ئۇنى ياخشى ساقلاش ئۇچۇن ھەتتا كۇپ ساياھەتچى ھەم مىھمانمۇ قوبۇل قىلمايدۇ! مانا بۇ ئۇيغۇرنىڭ ھەيۋىتى !  بۇ قۇرلۇشننى . بىر ئۇمۇر چەتئەلدە سودا قىلغان مىللەت ئۇتىدا كۇيگەن يۇرتنى مىللىتىنى سىغىنغان  ئۇيغۇر ساخاۋاتچى سودىگەرلىمىز مۇستەقىل مەبلەخ سىلىپ ئۇچ يىل سەرىپ قىلىپ سەرەمجانلاشتۇرۇپ چققان ،مۇ يەردىكى ھەربىر نەرسىدىن ئۇيغۇرلارنىڭ نامايەندىسىنى تاپالايسىز !  شۇڭا بىلمىگەنلەر سەۋەپسىزلا سىلشتۇرماڭلار !مۇمكىن بولسا ئۇز كۇزۇڭلار بىلەن كۇرۇپ ئاندىن سىشتۇرغىدەك بولسا ئاندىن سەۋەپ بىلەن سۆزلەڭلار! ئۇلارمۇ تاپققان پۇلىرىنى يۇرتىدىكلەر ئۇچۇن ئايىماي بىرىۋاتىدۇ ! مۇبادا يۇز يۇئەن تاپشۇرۇپ كۇرۇپ مىلىتىڭىزنىڭ  ھەيۋىتى ھىس قىللالماي پۇشايمان بىلەن يىنىپ چىقسىڭىز ئۇشۇ پۇلىڭىزنى مەن سىزگە تۆلەپ بىرىشكە رازى !
