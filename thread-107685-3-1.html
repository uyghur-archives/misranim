ئۇيغۇر يولۋىسى دىگەن بالا ئىنكاس يازسا بەك ھاياجانلىنىپ كىتىدىكەن ، ئۇنىڭ ئۈستىگە بىر تىما ئىچىدە بەك كۆپ مەشغۇلات قىلىدىكەن ، ئادەتتە بىر تىما ئىچىدە ئايرىم دىيىشكە تىگىشلىك بولغان گەپلەرنى ۋە ياكى باھالاشقا تىگىشلىك بولمىغان ئىنكاسلارنىمۇ باھالاپ ئەمدى پەسىيىپ قالغان دولقۇننى ئولغايىتقاندەك ئىش قىلىدىكەن  .
ھەي ، ھېلىقى ئوتتۇرغا سەكرەپ چۈشۈپلا تىللاپ چۈشۈپ كەتكەن يەنە بىر ئەزا مىنىڭچە بىرەرسىنىڭ كىچىك ئىسمىكەن ، يېپىشىپلاۋالدىغۇ ئۇ تىمىغا يائاللا ... بۇ يەردە ئۇلارنىڭ يامان گېپىنى قىلىپ بىرىۋاتقىنىم يوق ، بىراق ئۇ تىمىدا ئىچىم سىقىلىپ قالدى .
ئىپپەت قۇربانلىرى دىگەننى ئوقىدىم ئانچە مۇنچە ئەمما تولۇق ئەمەس ، ئۇنداق چوڭ ھەجىملىك تىمىلارنى ئادەم ئوقىغاندا ئۈزمەي ئوقىسا بولىدۇ ، شۇڭا يىتەرلىك ۋاقىت چىققاندا ئاندىن ئوقۇيمەن .

بىزنىڭ شۇ پېتىچە قانچە سائەت ئولتۇرغىنىمىزنى بىلمەيمەن، ئەمما كۈتكۈچى قىزنىڭ
-ھەدە خاپا بولماي ھىساۋات قىلۋەتكەن بولساڭلا ، بەك كەچ بوپ كەتتى ، ھېلىمۇ ئىشتىن چۈشەممەيۋاتىمىز
دىگەن سۈزى بىلەن ئۇ ئاستا بېشىنى كۈتىرىپ ماڭا قارىدى .
-ماقۇل سىڭلىم ھازىرلا ، كەچۈرۈڭ
دىدىم سىپايە تەلەپپۇزدا ھەم دەرھال ئۆزەمنى ئوڭشاپ ئۇنىڭ توسقىنىغا قارىماي ھىساۋەت قىلۋەتتىم ، بىلەمسىزكىن مەن ئەزەلدىن ئۇنىڭ ماڭا سوۋغا ئىلىپ بىرىشىنى ۋەياكى ياخشى بىر مىھمان قىلىشىنى ئويلاپمۇ باقمىغان ، ئادەتتە بىز پەقەت سىنىڭ مىنىڭ دىيىشمەيتتۇق ، مەن تۇلىراق ئۇنىڭ پۇل خەجلىشىنى خالىمايتتىم ، چۈنكى بىز  ئۇنداق ئاددى يەگە بارمايدىغان بولغاچقا چىقىممۇ سەل كۆپرەك بۇلاتتى ، ئەڭ مۇھىمى مەن ئۇنىڭ ئائىلە تۇرمۇشىنىڭ قىيىنچىلىقتا قېلىشىنى خالىمايتتىم ، چۈنكى بۇ ئائىلىنىڭ يۈكىنىڭ ئۇنىڭ زىممىسىدە ئىكەنلىگىنى ئوبدان بىلەتتم .......................


      شۇنداق قىلىپ بىز سىرتقا چىقتۇق ، خوشلىشىدىغان ۋاقىتمۇ كەلدى ، بىز بىر- بىرىمىزنىڭ كۈزىگە قاراپ نىمە ئويلىغىنىمىزنى بىر قاراپلا بىلەلەيتتۇق ، گەرچە ئۇ مىنى يېنىدا بىرئاز ئېلىپ قالالغان بولسىمۇ مىنىڭ بۈگۈن ئۇنىڭدىن چۇقۇم ئايرىلىپ كىتىدىغىنىمنى كۆڭلى تۇيغان بولسا كىرەك ، ئۇمىنى پەقەتلا ماڭغىلى قويمىدى ،ئۇ يە مىنى تۇتۋالمىدى ،يۇلۇمنى تۇسۇۋالمىدى پەقەت ياشلانغان بىچارە كۈزىنى ماڭا مىختەك تىكپ تۇرۇپ ، بۇلىدۇ كەت لېكىن مەنمۇ سەن بىلەن نەگە باساڭ شۇ يەگە بارىمەن دەپ چىڭ تۇرۋالدى ،ئەمدى ئۇنىڭغا ھېچ بىر جاۋاپ بىرەلمىدىم .
-ئۈتىنەي مەردان ، يۇلىمنى توسمىسىڭىز ؟بىز قاچانغىچە مۇشۇنداق ئۈتىمىز ، سىز بىر مۇۋاپىق ھەل قىلىش چارىسى تاپالمىدىڭىز ، مىنى تۇلىمۇ بىچارە ھالغا چۇشۇرۇپ قويدىڭىز ، سىزنى خيالىمدىنن يۈرۈگۈمدىن چىقىرۋىتەلىگەن بولسام تۇلىمۇ خۇشال بولغان بۇلاتتىم ، ئۈتىنەي مىنى ئارامىمىدا قويسىڭىز›
-باھار ، جېنىم گىپىمنى ئاڭلا ، سەن مىنى چۈشەنمەيدىغان يەدە ئەمەس ، سىنى نەقەدەر سۈيىدىغىنىمنى ئوبدان بىلىسەن ؟مىنىڭ شۇ تاپتا قانچىلىك ئازاپلىنۋاتقىنىم ئۈزۈمگە ئايا ن ، مەن شۇ تاپتا ئۈزۈمنى بىر نىمە قىلۋالغۇدەك بوپ كەتتىم ، ئايالىممۇ يامان ئەمەس ياخشى ئايال ،پەقەت ئۇنى سۈيەلمىدىم  سەن ئۈزەڭنى ئەيىپلىمە ، مەن ئەزەلدىن ئۇنى سۈيۈپ باقمىغان ، بەزىدە ئۇنىڭغا بەكلا ئىچىم ئاغرىيدۇ ، ئۈزۈمگە شۇنداق ئۆچ بوپ كىتىمەن ، ئۈزۈمنى بۇ تۇرمۇشقا شۇنداق زورلاپ باقساممۇ داۋام ئىتەلمىىگۈدەكمەن، ئەمما ئانامنىڭ كىسىلىمۇ ساڭا ئايان ، بۇ خەۋەرنى ئاڭلىسا ئۇ بەرداشلىق بىرەلمەيدۇ ، ئۆمۈر بۇيى پۇشايماندا قېلىشتىن قورقىمەن ، مەن نىمە قىلىشىمنى پەقەتلا بىلەلمىدىم ، بىردىنبىر بىلىدىغىنىم سىنى سۈيىمەن ، سەندىن ۋاز كەچمەيمەن ، يابىر ئۆلمۈدۈم ،بۇنداق ياشاشتىن جاق تويدۇم.

      ئۇشۇنداق دىۋىدى تاق ئۇنىڭ ئاغزىنى تۇتىۋالدىم دە خۇدا ساقىلىسۇن ، دەپ تۈفى تۈفى دىگىنىمچە ئۇنىڭ ئاغزىغا يىنىك ئۇرۇپ قويدۇم.
-باھار جېنىم
-قۇلۇقىم سەندە
-سەن بۈگۈن كەتسەڭ چۇقۇم مەندىن ئايىلىپ كىتىمەن دەپ تۇرۋالىسەن ، سەن داۋام ئۆزىڭىزنى ئىسپاتلاڭ دەيتىڭغۇ ، بۈگۈن مەن ساڭا ئۈزۈمنى ئىسپاتلاي .
-كىرەك ئەمەس ، مەن قايتاي ، سىزمۇ ئۇنىڭ تۇغۇلغان كۈنىنى كۈڭۈلدۈكىدەك ئۆتكۈزۈپ بىرىڭ دىگىنىمچە يۈگەگەن پېتىم ئۇنىڭدىن يىراقلاشتىم ، ئەمما ئۇمۇ كەينىمدىن يۈگۈرەپ كىلىپ مىنىڭ بىلىكىمدىن تۇتۋالدى دە
-بىللە بارىمىز
-نىمە؟ ئېلىشىپ قالمىغانسىز ؟سائەت نەچچە بولدى؟توۋا ئۆيگە قانداق كىرىمىز ، ئۇ مىنى تىللاپ كەتسىچۇ ؟گۇمانلىنىپ قالسىچۇ ؟
-خاتىرجەم بول قالغىنىنى ماڭا قوي ، ئىشقىلىپ مەن بىلەن بىللە بارىسەن ، بولمىسا مەنمۇ ھېچ نەگە بارمايمەن
-ھىم بۇلىدۇ .
مىنىڭ جاۋابىم ئۇنى قاتتىق خۇشال قىلۋەتتى ، مىنىڭ ئىچىمگە تۇيۇقسىز جىن كىردى ،مەندە بىردىنلا  ئۈزۈم سۆيگەن ئادەمنىڭ ئايالىنى كۈرۈپ بېقىش ئىستىكى قوزغالدى دە ھېچ ئىككىلەنمەيلا ئۇنىڭ كەينىدىن ماڭدىم دە تورت يەنە بىرمۇنچە يىمەكلىك ئېلىپ سىلەنىڭ ئۆيگە قاراپ ماڭدۇق .

     قوڭغۇراقىنى بېسىشىمىزغا سىزمۇ خۇددى بىزنى ساقلاپ تۇرغاندەك ئىشكنى دەررۇ ئاچتىڭىز ۋە ئىككىمىزنى كۈرۈپ سەل تىڭىرقاپ قالدىڭىز يۇ يەنىلا خۇشال ھالەتتە مىنى ئۆيگە تەكلىپ قىلدىڭىز ، شۇنداق قىزغىن ھالەتتە ئاستىمغا كۆرپە سېلىپ مىنى يۇقىرغا باشلاپ يۈگۈرۈپ يۈرۈپ ئالدىمغا چاي قۇيۇپ يىمەكلىكلەرنى بىردەمدىلا تىزىپ بولدىڭىز .
شۇتاپتا يۈرۈگۈم قاتتىق ئاغرىماقتا ، خۇددى سىز ماڭا ھازىرلا ئېتىلىدىغاندەك ، ئۇزۈمدىن ئۈزۈمگە يۈزلىرىم قىزىرىپ شۇنداق كالامپايلىشىپ كەتتىم ، ئۇمىنى ئەڭ يېقىن ئاغىنەمنىڭ سىڭلىسى بۇ يەگە خىزمەتكە كەپتىكەن شۇڭا باشلاپ كەلدىم دەپ تۇنۇشتۇردى ،سىزمۇ ئارتۇق گەپ قىلمىدىڭىز ............
ئەمدى كىچىك قىزىڭىزنىڭ يىغىسى ھەممىمىزنىڭ دىققىتىنى تارتتى ،سىز ئۇنى بىز با ئۆيگە كۈتۈرۈپ كىرىپ پەس قىلدىڭىز ھەم مىنى كۆرسىتىپ ئاچا دەڭە ، ئاچىنى سۈيۈپ قۇيىڭە قىزىم دىيىشىڭىزگە ئۇ مەندىن قېچىپ قىقاسلاپ يىغلاپ كەتتى ..................

      شۇتاپتا مەنمۇ ئۈزۈمنى باسالماي كىچىك بالىدەك يىغلاۋاتقان ئىدىم  ، قىزىڭىزنى كۈرۈپ مىنىڭ دىل يارام قوزغىلىپ كەتتى ، چۈنكى ئۇ تۇغۇلغان كۈنى مەردان ئىككىمىزنىڭ يۈرەك پارىمىز ، مىنىڭ جېنىمنىڭ پارچىسى بۇيۇمدىن ئاجراپ  كىتىپ بۇ ئۆمرۈمدىكى ئەڭ ئازاپلىق بىر كۈننى ئۆتۈزگەن ئىدىم .......................
سىز ھېچنىمىنى ئاڭقىرالماي بىر ئاز تۇرۇپ قالدىڭىز ، مەردان ماڭا تىكىلگىنىچە توختىماي ئاقا ئارقىدىن بىر نەچچە تال تاماكىنى چىكىپ بولدى .
ئەمدى بۇ ئۆيدە پەقەت تۇرغۇم كەلمىدى ، سىز بالىنى يەنە بىر ئۆيگە ئېلىپ چىقىپ كىتىۋىدىىڭىز
مەردان ئاستا يېنىمغا كىلىپ قۇلۇمنى تۇتتى ،
-مەن كىتىمەن
-ياق ، مۇمكىن ئەمەس ، سىنى ھېچ نەگە ماڭدۇرمايمەن ، مۇشۇ يەردە ئۇخلايسەن
-ماقۇل سەن بىلەن بىر ئۆيدە ئۇخلايمەن ، قىلالامسىز ؟
-چاتاق يوق
ئەسلىدە مەن ئۇنى نىيىتىدىن ياندۇرماقچى ئىدىم.............

          ئەمما  ئۇ  تام تۈۋىگە كۆرپە سېلىپ ماڭا ياستۇقنى ئۇزاتتى دە يېنىمدا كۆرپىمۇ سالماي يەردىلا ياتتى ، سىزمۇ ئارتۇق گەپ قىلماي ئۇنىڭ يېنىغا كىلىپ ياتتىڭز ، مەن تام تەرەپكە ، ئۇ ماڭا ، سىز ئۇنىڭغا قارىغان پېتى شۇنداق ياتتۇق ، مەن ھېچ كۆز ياشلىرىمنى باشقۇراممىدىم ، سىزنى ئۆز ئورنىمغا قۇيۇپ سىزگە شۇنداق ئىچىم ئاغرىدى ، ئەمما ماڭىچۇ ، سىز يەنىلا مەندىن بەخىتلىك ، سۆيگەن ئادىمىڭىز بىلەن بىللە ياشىيالايسىز ، ئەمما مەنچۇ ؟
شۇ تاپتا سىزىپ تۇرۇپتىمەن ھېچ قايسىمىزنىڭ كۈزىدە ئۇيۇق يوق ............................
مەن شۇنداق ئۇسساپ كىتىپ ئاستا ئورنىمدىن تۇرۇپ ئاشخانا ئۆيگە سۇ ئىچىش ئۈچۈن چىقتىم ، ئەمما كۆرگىنىم سىزنىڭ قاراڭغۇ بۇلۇڭدا ئولتۇرۇپ ئاغزىڭىزنى تۇتۇپ ئىسەدەپ يىغلىغىنىڭىز بولدى ، ئەمدى ئاستا كەينىمگە ياندىم دە يان تىلفۇنۇمنى قۇلۇمغا ئالدىم ۋە ئەڭ يېقىن دوستۇمغا ماڭا دەرھال تىلفۇن قىلىشنى سۈزۈمگە ماسلىشىشنى ئېېيتپ ئۇچۇر يوللىدىم
.
-ۋەي ، گەپ قىلە ، ھە ئاكامما قاچان كەپتۇ ؟كەچتىما
-ھە ھ ھە ،شۇ شۇنداق
-نىمە دەيدۇ ، مىنى دەرھال يېنىمغا كەلسۇن ، نىمە نىمىشقا ماڭا تىلفۇن قىلماپتۇ ،ماقۇل بىلدىم ، ھازىرلا باراي.
دىگىنىمچە ئورنۇمدىن تۇردۇم دە كىيىملىرىمنى كىيىپ ماڭماقچى بولدۇم ، ئەمما سىز ماڭا كۈيۈنۈپ
-سىڭلىم بەك كەچ بوپ كەتتى ، مەردان ئاپىرىپ قويسۇن
دىگەن بولسىڭىزمۇ ، رەخمەت ھاجىتى يوق دىگىنىمچە ئۇنىڭغا قاراشقىمۇ پېتىنالماي ئۆيدىن چىقپ  ئۈزۈمنى سىرتقا ئاتتىم ۋە  ھا ھا ھا دەپ ئاۋازىمنى بۇلىشىغا قۇيۋېتىپ يىغلاپ چوڭ يولدا يۈگۈرگەن پېتىم قاراڭغۇلۇققا سىڭىپ كەتتىم ...................
