       يۈگرەۋاتىمەن ،كۆز يېشىمنى باشقۇراممغان ھالدا ئاجايىپ ئىچىنىشلىق يىغلىغان پېتى يۈگۈرەۋاتىمەن، يولدىن  ئاندا ساندا ئۈتۈپ كىتىۋاتقان ماشىنىلارنى ھىساپقا ئالمىغاندا كۇچا تىىمتاس،مۇشۇ يۈگەگەنچە جاھاننىڭ ئۇ چېتىگە كىتىشىنى ، بۇ يامغۇر كەبى ياشلىرىم بىلەن بارلىق دەت ئەلەم ، ئازابىمنى ، تۇنجى سۈيگۈنۈمنى ئۇنتۇپ كىتىشنى ،مىنى كۆيدۇرۇپ كۈل قىلغان ، باغرىمنى يارا زىدە قىلىپ مىنى نەپەسمۇ ئالامماس قىلىپ قويغان ، ئاشىق يۈرۈگۈمنى قان زەرداپقا توشتۇرۋەتكەن ، مىنى ئۇ نىڭدىن باشقىنى قۇبۇل قىلاممىغۇدەك ، جاھاندا باشقا بىرسىنى كۈرەلمۈگۈدەك دەرىجىدە ئەقىلمنى ، كۈزۈمنى ، دىلىمنى كور قىلغان بۇ ساراڭلاچە ئاشىقىلقىمنى چۆرۈپ تاشلىۋىتىشنى ، مۇمكىن بولسىدى كىچە كۈندۈز ۋىسال ئىزلەپ ، مۇھببىتىمگە بىر قانالمىغان بۇ بىمار ، سۆيگۈ قەلەندەرى يۈرۈگۈمنى باغرىمدىن يۇلۇپ ئېلىپ تاش بىلەن ئۇرۇپ چەيلەپ دەسسىۋەتكۈم ئۇنى پارە-پارە قىلۋەتكۈم   ، ئۈزۈمنى قايتىدىن روسلاشنى ، ھەممىنى يېڭىدىن باشلاشنى شۇنچىلىك بىچچارىلارچە ئۈمىت قىلساممۇ ئۇنداق قىلاممىدىم ..........

    ئاچا مەن بىر بىمار مەن ، ماڭا نىسىپ بولمايدىغان بىرسىگە ئاشىق بوپ قالغان داۋا تېپىلمايدىغان بىر بىمارمەن  ، مىنى ھەممىدىن قىينايدىغىنى ، مەن كۈزۈمنى يۇمۇپلا ئۇنى قوينىڭىزدىن باغرىڭىزدىن سۈكۈپ ئېلىپ كىتەلمەيمەن ، ئۇنداق قىلىش قۇلۇمدىن كەلمۈگۈدەك ، يا ئۇنىڭدىن ۋاز كىچەلمەيمەن ، سىزنىڭ نەچچە يىللاردىن بىرى ھېچ بولمىسا كۈنىڭىز خاتىرجەم ، بولمىغاندىمۇ بۈگۈنگىچە  ،ئەمما مىنىڭچۇ ؟ماڭا بۇ يەتتە يىل خۇددى يەتمىش يىل  ياق  يەتتە يۈز يىلدەك بىلىنىپ كەتتى ، كۈنى تۈن مىنى ئازاپ ئىلكىگە ئېلۋېلىپ مىنى ساراڭلادەك قىيناپ ھېچ ھالىمنى قويمىدى ، يە ئۇنىڭدىن ۋاز كىچەلمىسەم يە ئۇنىڭغا ئىرىشەلمىسەم ،بىر -بىرنى سۈيۈپ تۇرۇپ بۇنداق ئۇۋالچىلىق تارتىش بەك ئازاپلىق ئىكەن ، مەن چىدىمىدىم ، جېنىم ئاچا شۇ ۋاقىىتتا مەن پەقەت ئۈلۈمدىن باشقىنى ئويلىيالمىدىم ، مىنى تېخمۇ قىينايدىغىنى بىچارە مەرداننىڭ ئازابى ، ئۇنىڭ غەم مۇڭ باسقان كۆزلىرىگە قارىغىنىمدا ، ئۇنىڭ ياش قېتىپ قالغان يامغۇر كەبى ھۆللەنگەن يۈزلىرىنى ئالقىنىمغا ئالغىنىمدا يۈرەك باغرىم تىلىنىپ ، خۇددى قوقاستا پۇچۇلىغاندەك بۇلۇپ ئۈزۈمنى يۇقۇتۇپ قۇيىمەن ..................

       مەن ئەنە شۇ كۈنىدىن باشلاپ بۇ چەمبىرەكتىن بۇ ئۈچ بۇرجەكلىك مۇھەببەت چەمبىرىكىدىن چىكىنىپ چىقىشنى تاللىدىم ، چىكىنگىنىم ھەرگىزمۇ ۋاز كەچكىنىم ئەمەس ، ئۇنى ئەبەدى ، ئاداققىچە سۈيىمەن ،  بىز گەرچە ئىككى يەردە تۇرساقمۇ ، ئالاقىلاشمىساقمۇ ، بىر بىرىمىزنى كۆرمىسەكمۇ ئىككىمىزنىڭ بىللە بولغاندىكى شۇ گۈزەل مىنۇتلىرىمىز ، ئۇنىڭ ماڭا بولغان ساپ مۇھەببىتى ، مىنىڭ گۈزەل چىن ئەقىدەم ماڭا مەڭگۈلۈك ھەمرا ، مەن ئۇنى ئۇ ئالەمدىمۇ كۈتىمەن ، ئەگەر ئىككىنچى قېتىم يارىلىش نىسىپ  بۇلۇپ قالسىدى ، خۇدادىن پەقەت ، پەقەت ئۇنىلا تەلەپ قىلغان بۇلاتتىم .

     ئاچا مەن شۇ ۋاقتنىڭ ئۈزىدە بۇ يۇرتتىن كەتمەكچى ، سىلەگە چىن دىلىمدىن بەخىت تىلىمەكچى ، سىزنىڭمۇ ئاشىقىلىق يۇلىدا مەندەك بىر بىچچارە ئىكەنلىگىڭىزنى كۆزدە تۇتۇپ كۆز ئالدىڭلادىن مەڭگۈلۈك يۇقىماقچى بولدۇم دە شۇنداق تەسلىكتە بىر تاكسى توسۇدۇم ۋە ئۇزۇن يوللۇق بىكەتكە كىلىپ كىچىك ماشىنىدىن بىرنى بولاپ ئۆز يۇرتۇم بولغان باشقا شەھەرگە يۈرۈپ كەتتىم ، مىنىڭ ئەمدى ئۇنى ئىككىنچى رەت كۆگۈممۇ ، ئىشلىگىممۇ ئىشقىلىپ ھېچ ئىش قىلغۇم ھېچكىمنى كۆرگۈم يوق ، ئۈمىت قىلىدىغىنىم تىزراق بۇ يەردىن ئۇنىڭ كۆز ئالدىدىن يۇقۇلۇش ، ئەمما مەن ئۇنى باشقا بىر جايدا پىنھان ، يالغۇز ، شۇنچە بىچارىلاچە سۈيىمەن ، ئۇنىڭ يوللىرىغا تاكى چاچلىرىمنى قۇراۋ باسقىچە ، كۆزلىرىمگە ئاقلار چۈشكىچە ، بەللىرىم تال كەبى ئىگۈلگىچە ،كۆز نۇرۇم يۇقالغىچە ھەتتا ئەڭ ئاخىرقى بىر نەپىسىم قالغۇچە ئۇنىڭ يۇلىغا تەلمۈرىمەن ، ئۇنى زارىقىپ كۈتىمەن ...........

     بۇ قۇرلانى ئۇقۇۋېتىپ ئۇ قىزغا بولغان ئۆچلىگىم ، غەزىپىم تۇيۇقسىز يۇقاپ كەتتى ، ئۇنى تۇيۇقسز كۆز ئالىدىمغا ئەكىلىپ سالدىم ، بىچارە قوللىرم تىترەپ كۆز ياشلىرىمنى سۈرۈتكىدەك ھالىمۇ يوق ئىدى شۇنىڭغا باقماي مەن تېخچە كۆز ياشلىرىمنى باشقۇرالمايۋاتاتتىم ، خەتنىڭ ئاخىرنى ئۇقۇماق بۇلۇپ ئىتتىك داۋامىغا كۆز يۈگۈرتتۈم يۇ ، تۇيۇقسىز دوستۇم لەيلى ئىسىمگە كىچىپ ئەتراپقا قارىدىم ، تىلفۇن قىلسام ئىتىك ، ئىشى چىقىپ قالغان بولسا كىرەك دەپ ئويلىدىم دە بۇ يەردە ئولتۇرۇشنى مۇۋاپىق كۆرمىدىم ،شۇ تاپتا كۆڭلىم بىر تىمتاس يەرنى خالايتتى ،ئاۋازىمنى بۇلىشىغىچە قۇيۇپ بىرىپ ھالىمدىن كەتكۈدەك بىر قېنىپ يىغلىۋالغۇم بار ئىدى ..............

    ئورنىمدىن چاققانلىق بىلەن تۇردۇم دە خەتنى سومكامغا سالدىم ۋە تىلفۇنۇمنى ئەتتىم ، تاكسىدىن بىرنى تۇسۇپ دەريا بۇيىغا ئاپىرىپ قۇيۇڭ دىگەن گەپنى ئارانلا قىلالدىم دە بېشىمنى قامانلىغىنىمچە ئىسەدەپ يىغلاپ خىىيالغا چۆكتۈم.....................
