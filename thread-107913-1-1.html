بىمارلىقىمغا باقماي ، بىر نەچچە كۈن سوتكىسىغا 15 سائەتتىن ئارتۇق ئىشلەپ چارچاپ كەتكەنلىكتىنمۇ ياكى توردىكى بىر قىسىم ئۇرۇنسىز تالاش -تارتىشلار ۋە ئەتىراپىمدا يۇز بىرىۋاتقان ئىشلاردىن رايىم قايتىپ ،قۇلۇمغا باشقا ئىش كەلمىدى . شۇڭا ، مىڭەمنى ئازراق ئارام ئالدۇرۇش مەقسىتىدە ، مەزكۇر تىمىنى يوللىدىم ، ھەممىسىنى بۇگۇن يىزىپ بۇلالمايمەن . كوپچىلىكنى بۇ تىمىنى بىرگە تاماملاشقا  ئالى ئېھتىرام بىلەن تەكلىپ قىلىمەن ،.
تورداشلارنىڭ مەزكۇر تىمىنىڭ داۋاملىشىشىنى نەزەرگە ئالغان ئاساستا ، ئىلمى پىكىر يورگۇزۇشىنى ئوز-ئارا بىر -بىرىنى (جوملىدىن مىنى) تىللاپ كەتمەسلىكىنى ئۇمىت قىلىمەن ، مەن كەسىپ ئەھلى ئەمەس ، شۇڭا ، بىر قىسىم خاتا كەتكەن تەرەپلەر بولسا ، دوستلارنىڭ  قىممەتلىك تەكلىپ - پىكىر بىرىشىنى ئۇمۇت قىلىمەن ، شارائىتى يار بەرگەن دوستلارنىڭ بۇھەقتە  ئىزدىنىپ بېقىشىنى ، جوملىدىن ، مۇشۇ ساھەدىكى  خادىملارنىڭ مەزكۇر تىما ئاستىدا ئېلىپ بېرىلماقچى بولغان بىر قىسىم پىكىرى ئوتكۇر ، تەپەككۇرى ئىجادى  بولغان مۇنازىرىچىلىرىمىزنىڭ قاراشلىرىدىن ئىلھام ئېلىپ ، يىڭى نەتىجىلەرنى قولغا كەلتۇرۇشنى ، بىلگەنلەرنىڭ بۇ ھەقتىكى بىلىملەرنى يوللىشىنى ئۇمىت قىلىمەن !
كىلىڭلار ، دوستلار ،بىرگە ئوگىنەيلى !
1. يەر رايىشۇناسلىق  دىگەن نىمە؟
مەن بۇ ھەقتە ھىچقانداق ماتىرىيال كورمىگەن ، ئەمەليەتتە ، بونداق ، بىر ئىلىمنىڭ بارلىقىدىن ئوزەمنىڭمۇ خەۋىرى يوق. ھازىر ھاۋارايى ، سۇرايى......دىگەندەك  ئىلىملەر بار ،  بۇ ئىلىمنىڭ  مىنىڭچە يەرتەۋرەشتىن ئالدىن مەلۇمات بىرىش توغرىسىدىكى تارمىقى مەۋجۇت . ئەگەر بار بولسا ، كوپچىلىكنىڭ ئۇچۇر بىرىشىنى ئۇمىت قىلىمەن .
ئېنىقلىما: يەر رايىشوناسلىق -ئاسمان جىسىملىرى ھەركىتى ، ئاتمۇسفىرا ، سۇ ، جانلىقلارنىڭ ھەركىتى ۋە يەرنىڭ پوستى قەۋىتى ، ئوتتۇرا قەۋىتى ۋە يادروسىنىڭ مىخانىكىلىق ، ئىسىسقلىق  ، ئىلىكتىر - ماگىنىتلىق ھەركىتى يادرو (يادرو يىغىلىش ۋە يىمىرىلىش رىئاكسىيىنى كورسىتىدۇ)تۇرلۇك ھەرىكەت ئالاھىدىلىكلىرى ۋە ئۇلارنىڭ يەر ھەركىتى يەنى يانار تاغ پارتلاش، يەر تەۋرەش، يەر يېرىلىش، يەر كۇچۇش ، يەر ئىسسىقلىقى قاتارلىقلاربىلەن مۇناسىۋىتىنى تەتقىق قىلىش ئاساسدا ، ماتىماتىكىلىق سېستىما مودىلى تورغۇزۇپ ، مودىل ۋە ئالاقىدار ئولچەش - سىزىش نەتىجىسىگە ئاساسەن ، ئۇمۇمى دائىرىلىك ياكى قىسمەن دائىرىلىك يەر ھەركىتىگە قارىتا (مەسلەن ، يەر تەۋرەش، يانار تاغ پارتلاش) ئالدىن مەلۇمات بىرىدىغان ، چېتىلىش دائىرىسى كەڭ  ۋە مۇرەككەپ بولغان ، سىياسى ، ئىجتىمائى ، ئىقتىسادى ۋە ئىكۇلۇگيىلىك ئەھمىيىتى زور بولغان ئونۋېرسال پەندۇر.
بۇگۇن مۇشۇ يەرگىچە يىزىپ توراي ، （خېلى ھاردىقىم چىقىپ قالدىيا）، قېنى ، تورداشلار ، داۋامىنى يىزىشقا قاتناشمامسىلەر ! ، ھەممىمىز بىرلىشىپ ، يىڭى بىلىملەرگە ئىرىشىپ قالساق ئەجەپ ئەمەس . بۇ - ئۇرۇنسىز تالاش -تارتىش قىلغان ۋە ئوز-ئارا  ئۇرۇش قاندىن كوپ ئۇرۇنلۇق !.

2.يەر رايىشوناسلىقنىڭ تەتقىقات مەقسىتى ۋە دائىرىسى
يەر رايىشوناسلىقنىڭ تەتقىقات مەقسىتى - تۇرلۇك يەر ھەرىكەتلىرى (تاغ كۇچۇش ، يەر يېرىلىش ،يانار تاغ پارتلاش ،يەر تەۋرەش قاتارلىق ھادىسىلەر)نىڭ يۇز بىرىش سەۋبىنى ئېنىقلاش ۋە ئۇنى تارىختا يۇز بەرگەن ھادىسلەر ھادىسىلەر ئانالىزىغا قۇشۇپ بىرلىكتە ئانالىز قىلىش ئاساسىدا تەتقىقات ئۇبىكتى قىلىنغان رايۇننىڭ تۇرلۇك ئالاقىدار ئولچەش سانلىق مەلۇماتلىرىغا ئاساسەن ، شۇ رايۇننىڭ ياكى دائىرىنىڭ  يەر ھەركىتى قانۇنيەتلىرىنى تېپىپ چىقىپ مۇشۇقانۇنىيەتلەر ئاساسىدا، مەزكۇر رايۇننىڭ كەلگۇسىدىكى مەلۇم ۋاقىت بۇلەكلىرى مەسلەن بىر كۇنلۇك، 3كۇنلۇك ، بىر ھەپتىلىك ، بىر ئايلىق ھەتتا ئۇنىڭدىن ئارتۇق مەزگىللىك يەر ھەرىكەتلىرىگە ئالدىن مەلۇمات بىرىشتۇر.

2.يەر رايىشوناسلىقنىڭ تەتقىقات مەقسىتى ۋە دائىرىسى
يەر رايىشوناسلىقنىڭ تەتقىقات مەقسىتى - تۇرلۇك يەر ھەرىكەتلىرى (تاغ كۇچۇش ، يەر يېرىلىش ،يانار تاغ پارتلاش ،يەر تەۋرەش قاتارلىق ھادىسىلەر)نىڭ يۇز بىرىش سەۋبىنى ئېنىقلاش ۋە ئۇنى تارىختا يۇز بەرگەن ھادىسلەر ھادىسىلەر ئانالىزىغا قۇشۇپ بىرلىكتە ئانالىز قىلىش ئاساسىدا تەتقىقات ئۇبىكتى قىلىنغان رايۇننىڭ تۇرلۇك ئالاقىدار ئولچەش سانلىق مەلۇماتلىرىغا ئاساسەن ، شۇ رايۇننىڭ ياكى دائىرىنىڭ  يەر ھەركىتى قانۇنيەتلىرىنى تېپىپ چىقىپ مۇشۇقانۇنىيەتلەر ئاساسىدا، مەزكۇر رايۇننىڭ كەلگۇسىدىكى مەلۇم ۋاقىت بۇلەكلىرى مەسلەن بىر كۇنلۇك، 3كۇنلۇك ، بىر ھەپتىلىك ، بىر ئايلىق ھەتتا ئۇنىڭدىن ئارتۇق مەزگىللىك يەر ھەرىكەتلىرىگە ئالدىن مەلۇمات بىرىشتۇر.
يەر رايۇنىشوناسلىقنىڭ تەتقىقات مەقسىتى قىسقارتىپ ئېيتقاندا ، يەر ھادىسىلىلىرىگە ئالدىن مەلۇمات بىرىپ ، كىلىپ چىقىشى مۇمكىن بولغان ئاپەتنىڭ تەسىرىنى ئازايتىش ۋە زۇرۇر تېپىلغاندا خۇددى سۇنئى يامغۇر ياغدۇرغانغا ئوخشاش سۇنئى ئۇسۇلدا تەسىر كورسىتىش ئارقىلىق ئاپەتنى كونتىرول قىلىشتۇر.
يەر رايىشوناسلىقىنىڭ دائىرىسى ناھايىتى كەڭ بۇلۇپ ، بۇ ئاساسلىقى يەر ھەركىتىنى كەلتۇرۇپ چىقىرىدىغان ئامىللارنىڭ كوپلىگى تەرىپىدىن بەلگىلەنگەن.
يەررايۇنىشوناسلىقىنىڭ تەتقىقات ئوبىكتى مەلۇم بىر رايۇننىڭ يەر ھەركىتى بولغاچقا ، شۇ رايۇن يەر ھەركىتىگە تەسىر كورسىتىدىغان بارلىق ئامىللار ۋە ئۇنىڭ  شۇ رايۇن يەر ھەركىتىگە كورسىتىدىغان تەسىرىدۇر .
مەلۇم بىر رايۇننىڭ ھەركىتىگە تەسىر كورسىتىدىغان ئاساسلىق ئامىللارنى توۋەندىكىلەرگە ئايرىش مومكىن:
1)ئامىللارنىڭ شەكىللىنىشىگە ئاساسەن تەبئى ئامىللار ۋە سۇنئى ئامىللار ، ئارىلاشما ئامىللار دەپ ئۈچكە ئايرىش مۇمكىن.
تەبئى ئامىللار :شۇ رايۇننىڭ ئۇزىدىكى فىزىكىلىق ئامىللار مەسلەن ، زىچلىقى ،تەركىبى ، جوغراپىيىلىك ئورنى، يەر ماگىنت تەسىرىگە ئۇچراش ئەھۋالى ، يەر ئىسىسقلىقى ،يەر تۇزۇلۇشى،  مىخانىكىلىق ھەرىكەت ئەھۋالى ، ئۇسۇملۇك يېپىنچىسى سۇرايى ، ھاۋارايى،.....ۋە يۇقارقىلارنىڭ تارىخى ئوزگىرىشى قاتارلىقلار.
ئاستىرونۇمىيىلىك ئامىللار ئاساسلىغى يەر ھەركىتىگە تەسىرى كۇچلۇك بولغان ، قۇياش ۋە ئاي قاتارلىقلارنىڭ شۇرايۇنغا كورسىتىدىغان تۇرلۇك فىزىكىلىق تەسىرى ، مىقتارى ۋە تەسىرى ھەم ئۇلارنىڭ دەۋرىيلىك دەۋرىيىسى ئوزگىرىشى قاتارلىقلار .
بۇگۇن ، مۇشۇ يەر گىچە يىزىپ تۇراي ، قېنى كوپچىلىك مەرھەمەت!
مۇھاكىمە ھەممەيلەننىڭ قەلەم تەۋرىتىشىنى كۆتىدۇ!!

يەرشارىنىڭ كىلىماتى كۆپ خىل ئامىللارنىڭ تەسىرىگە ئۇچرايدۇ . باشقا پىلانىتانىڭ تەسىرى تۈپەيلىدىن ، يەرشارىنىڭ قۇياشنى ئايلىنىپ ھەركەت قىلىش ئوربىتىسىدا ھەر 19 مىڭ يىلدىن 26 مىڭ يىل بولغان ئارلىقتا ئاز-تولا دەۋرىيلىك ئۆزگۈرۈش بولۇپ تۇرىدۇ . يەرشارىنىڭ ئېغىش بۇلۇڭى ھەر 41 مىڭ يىلدا 22 گىرادۇس بىلەن  24.6 گىرادۇس ئارلىقىدا دەۋرىيلىك ئۆزگۈرۈش بولىدۇ . بۇ ، يەرشارىنىڭ قۇياش نۇرىنى قۇبۇل قىلىشىغا قانچىلىك تەسىر يەتكۈزسە ، يەرشارىنىڭ تىمپىراتۇرىسىمۇ شۇنچىلىك تەسىرگە ئۇچراپ ، يەر شارىدا دەۋرىيلىك مۇز دەۋرى مەزگىلى كۆرىلىدۇ .
يەنە يانار تاغنىڭ پارتىلىشى كۆپ مىقداردىكى گۈڭگۈرتلەشكەن ماددا ۋە توزاننى ئاتموسفىراغا چىقىرىپ ، قۇياش رادىئاتسىيىسىنى توسۇپ ، يەرشارى تېمپىراتۇرىسىنى تۆۋەنلىتىۋېتىدۇ . شۇنىڭ بىلەن بىرگە ، قۇرۇقلۇقنىڭ لەيلەپ سىلجىشى كۆپ ھاللاردا دېڭىز سۈيىنىڭ ئېقىش يۆلىنىشىنى ئۆزگەرتىۋېتىدۇ . ئىسسىق بەلۋاغدىكى ئىللىق دېڭىز سۈيى توسقۇنلۇققا ئۇچرا قۇتۇبقا بارالمىسا ، يۇقىرى پاراللېللىق دېڭىز-ئوكيان يىل بويى مۇز تۇتىدۇ . مۇز قارنىڭ قاپلاش دائىرىسى كېڭىيىپ ، قۇياش نۇرىنىڭ قايتىشى كۈچىيىپ كەتكەندە يەرشارى قۇبۇل قىلىدىغان قۇياش نۇرى ئازلاپ كىتىپ ئۇخشاشلا مۇز دەۋرىگە سۆرەپ كىرىدۇ .
