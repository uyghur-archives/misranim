مەن قانداقتۇر بۇ تېمىنى كۆپچىلىككە دەردىمنى تۆكۈش ئۈچۈن ئەمەس بەلكىم كۆپچىلىكنىڭ ماڭا ھەقىقى بىر توغرا يول كۆرسىتىشنى ئاساسى مەقسەت قىلىش سۈپىتدە يوللىدىم ................
  مەن جەنۇبى شىنجاڭدىكى مەلۇم ناھىينىڭ ئاساسى قاتلامدا ئىشلەۋەتقان ئادەتتىكى بىر يېزا كادىرى ، قىز دوستۇم بولسا مەن بىلەن خوشنا يېزىدىكى مەركىزى بالىلار باغچىسىدا ئىشلەيدۇ . ئىككىمىزنىڭ يۈرىۋاتقىنىمىزغىمۇ مانا ھەش پەش دىگۈچە 1 يىلدەك ۋاقىت بۇلۇپ قالدى .مۇناسىۋىتىمىز شۇنداق ياخشى بۇلۇپ ، بىر بىرىمىزنى بىردەم كۆرمىسەك چىيالمىغۇدەك دەرىجىدە ئىدى ،تالاي كېچىلەرنى گۈزەل كېلەچىكىمىز ئۈچۈن شىرىن ھەمسۆھبەتتە ئۆتكۈزگەن ،بىر بىرىمىزگە شۇنچىلىك ۋاپادار ،شۇنچىلىك سەمىمى ئىدۇق ،ئۇنىڭ ئۈچۈن ھەتتا جېنىمدىنمۇ كىچىشكە ھەر ۋاقىت تەييار ئىدىم .بىر بىرىمىزدىن ھەر قانداق سىر تۇتۇشقا ئارىمىزدا يول قۇيۇلمايتى ،مۇشۇنداق گۈزەل ياشلىقنىڭ ئالتۇندەك شىرىن پەسىللىرىدە ئارىلىقتىكى بەزى كۆڭۈلسىزلىكلەر بىزنىڭ بۇ مىلودىيىمىزگە خاتىمە بېرىش ئالدىدا تۇرىۋاتىدۇ .........
نەق گەپكە كەلسەم ، قىز دوستۇم بىلەن بىرگە ئۆتۈشنىڭ ئالدىدا ئۇنىڭ بۇرۇن بىر يىگىت بىلەن يۈرگىنى ماڭا مەلۇم ئىدى ،لىكىن ئۇنى چىن كۆڭلۈمدىن سۆيگەنلىكىم ئۈچۈن ئۇنىڭ بۇرۇنقى ئىشلىرىنى پەقەت ياشلىقتىكى قىزىقىش دەپلا چۈشىنىپ كەلدىم ۋە بۇ ئىشلارنى بەكمۇ كۆڭلۈمگە ئېلىپ كەتمىدىم . كىيىن بارا بارا مۇناسىۋىتىمىزنىڭ ھازىرقى دەرىجىگە بېرىپ يىتىشى بىلەن ،ئۇنىڭ كەيپىياتىدا مۇئەييەن دەرىجىدە ئۆزگىرىش بۇلۇشقا باشلىدى ،مۇناسىۋىتىمىز بارغانچە سۇسلىشىپ كىتىۋاتقاندەك ھېس قىلاتتىم ،بىر كۈنى ئۇنىڭ بىلەن كۆرۈشىدىغانلىقىمنى ئېيتىپ ،زادى نىمە ئىش بولغانلىقىنى سوراپ باققۇم كەلدى .دەسلىپىدە خېلى ئۇزۇنغىچە بىر نەرسە دىمەي كۆز يېشى قىلىشقا باشلىدى ، كىيىن بولغان ھەقىقى ئەھۋالنى ماڭا سۆزلەپ بەردى . ئەسلىدە ئۇنىڭ بۇرۇنقى يىگىتىگە يۈكسەك ئىشنىپ ،ئۆزىنىڭ جېنىدىنمۇ ئەزىز بولغان پاكلىقىنىڭ بۇرۇنقى يىگىتى  تەرىپىدىن نابۇت قىلىنغانلىقى ۋە كىيىن بۇنىڭ بىلەن كارى بولماي باشقا قىزلار بىلەن كىتىپ قالغانلىقىدەك ئەھۋالنى ماڭا تۆكۈلۈپ تۇرۇپ سۆزلەپ بەردى .شۇ ۋاقىتتىكى ھېسياتىمنى بەلكىم بەزى تورداشلار چۈشىنىشى مۇمكىن ،مۇنداقچە ئېيىتقاندا خېلى خېلى ئىشلارغا چىداشلىق بولغان مەننىڭ كۆزۈمدىن چۈشكەن تامچىلارنى يۇشۇرۇشقا ئامالسىز قالدىم ،ئەزەلدىن بىر بىرىمىزدىن سىر يۇشارماسلىقنى دىگەن ،ئاخىرى ئۇ ماڭا راس گىپىنى قىلغانىدى .گېپىنى دەپ بۇلۇپ ماڭا " سەن مۇشۇ ئىشلارنى قۇبۇل قىلالىساڭ ،تاكى تىنىقىمنىڭ ئاخىرىغىچە ساڭا ئۆمۈرلۈك ۋاپادار بۇلۇپ ئۆتۈشكە قەسەم قىلىمەن " دەپ نۇرغۇن ۋەدىلەرنى بەردى ، لىكىن ئۇ چاغدا بۇ گەپلەر ھىچبىر كاللامغا تەسىر قىلىدىغاندەك ئەمەس ئىدى . بۇرۇنقى يىگىتىدىن زارلىنىپ شۇنداق دەرت تۆكۈپ كەتتى ،ھەم ئۆزىنىڭ ھەر قانداق بىر يىگىتنى ياخشى كۆرۈش ھوقوقىنىڭ يوقلىقىنى ،مىنىڭ بۇ ئىشلارنى قۇبۇل قىلالمىسام ،ماڭا ئۇلۇغ ئاللادىن بەخىت تىلەيدىغانلىقىنى ،ئاغرىنىشقىمۇ ھەققى يوقلىقىنى بىلدۈرۈپ ئۆتكۈزگەن گۇناھى ئۈچۈن ئۇلۇغ ئاللاغا نالە قىلىپ يېغلاشقا باشلىدى ،شۇ ۋاقىتتا ئۆزەمنى تۇتىۋالالماي مەنمۇ يىغلاپ تاشلىدىم .بىر تەرەپتىن بۇرۇنقى ۋىجدانسىزنىڭ پەسلىكىگە نەپرەت ئۇقىسام ، يەنە بىر تەرەپتىن ئۇنىڭ شۇ ۋاقىتتىكى ھالىغا قاراپ شۇنداق ئىچىم سىيرىلىپ كەتتى ،ھېسداشلىقىم قوزغالدى ، يەنە بىر تەرەپتىن ئۇنىڭدىن ئايرىلىپ قېلىشقا قەتتىلا چىدىمايتىم .لىكىن بۇ ئىشلارغا مىنىڭ يىگتلىك غورورۇم قەتئىيلا يول قويمايتى .
مانا ئارىمىزدىن شۇ ئىشلار بولغىنىغىمۇ 1 ئايدەك ۋاقىت بولدى ، مۇناسىۋىتىمىز ئەلۋەتتە بۇرۇنقىدىن بەكلا سۇسلىشىپ كەتكەنىدى ،بەزىدە تېلىفۇن قىلسام ئۈزىۋىتىدۇ ،بەزىدە ئۆزى تېلىفۇن قىلىپ كۆرۈشكۈسى بارلىقىنى دەيدۇ ۋە كۆرۈشۈپ ئۇزاق بولمايلا يەنە خوشلىشىپ كىتىپ قالىدۇ ، ئويلاپما قالىمەن مەيلىمۇ يە ، ياشلىقتا ئازمايدىغان كىم بار دەپ ، لىكىن ئۇنىڭغا قەتئىيلا ۋىجدانىم يول قويمايۋاتىدۇ .
مانا شۇنداق قىلىپ ئىزدىشىپ ئىزدەشمەي دىگەندەك 1 ئايچە بۇلۇپ قالدى ، نىمە قىلارىمنى قەتئىيلا بىلمىدىم ،ئەگەر ئۇنى مەن قۇبۇل قىلالمىسام ئۇنىڭ بۇندىن كىيىنكى تەقدىرى زادى قانداق بۇلۇپ كېتەر ، بۇ ئىشلارمۇ كاللامدىن ھىچ بىر ئەگىپ كىتەلمەيدۇ .         بولسا كەڭ تورداشلارنىڭ مىنىڭ بۇ سەمىمى دەردىمنى باشقىچە نىيەتتە چۈشەنمەي ، ماڭا قانداق قىلىش توغىرىلىق بىر يول كۆرسەتكەن بولساڭلار ، ئازراق بولسىمۇ ئۆزەمگە تەسەللى تېپىپ قالامەنمىكىن دەيمەن .
  ئەڭ ئاخىرىدا دەيدىغىنىم ، ئېي ئۇلۇغ ئاللا ، سەندىن مىللىتىمىزنىڭ يىگىتلىرىگە ۋاپا ،قىزلىرىگە ھايا تىلەيمەن .....................

ھۆل سۆگەتنى ئوتۇن قىلما ،
ئىپپەتسىزنى  خۇتۇن .

يېڭى  ماقال ـ تەمسىل ، ئۆزۈمنىڭ ئىجادىيەتلىرىدىن .
ئادەم مۇشۇ  زاماندا  بىر قىزغىمۇ ئىشەنگۈسى كەلمەيدۇ .  شۇڭا، نېمە ئۇ قۇرۇق مۇھەببەت
ئاخونۇم نېكاھنى ئوقۇپ ، بىر ياستۇققا باش قويغاندا ئاندىن ئىشەنمىسە بىكاركەن .

مەن يولۇققان ئىشقا سىزمۇ يولۇقىپسىز . كېرەك يوق ، ئوغۇل بالا بولغاندىكىن قەددىڭىزنى
تىك تۇتۇپ مەغرۇر ياشاڭ .  ئۆلگۈر  قىزلار  ، ئادەم ئارلاشماستا ، ياخشى كۆرمەستە
مۇناسىۋىتى چۇڭقۇرلاشماستا دېمەيدىكەن .   ئادەمنى ئازابلاپ مۇھەببەت تەرەققىي قىلدىغاندا
ئۆيىگە ئەلچى كىرىشكە ئاز قالغاندا ، مەن  پاك ئەمەستىم ، قولۇم خېمىر دا ئىدى دەپ
ھەر كىم قىلغانلىرى ئۈچۈن بەدەل تۆلىشى كېرەك . مۇھەببەت ، تۇرمۇش دېگەنلەر كۆز
ياشلىرىڭىزغا ئىشەنمەيدۇ .
بەربىر   ئۇ ئايالنى ئالسىڭىز مۇ كىيىن    ئۇ ئىشلارنى ئۇنتۇپ كېتەلمەيسىز .
كىيىن يەنە ئۆزىڭىزنى ئازابلايدىغان ئىش چىقىدۇ .   ھەم ئۇ ئايالمۇ سىزنى ھەقىقىي
سۆيەلمەسلىكى مۇمكىن .   
ھازىرقى    مايماقلار ،  ئانچىكى    ئىش ئۈچۈن   ئىشتانبېغىنى يېشىپ بېرىپ كىيىن
ئۆزىنىڭ ئەيبىنى يېپىش ئۈچۈن باھانە تېپىشقا  ئۇستا ، خۇددى ئوقۇغاندەك ...
ھازىرقى قىزلارغا چىن ئەقىدە ، ۋاپا قىلسا بىلمەيدۇ ، ئويناپ تاشلىسا كەينىدىن يىغلاپ
يۈرىيدۇ .مانا مۇھەببەت مۇشۇنداق ...




















































































مۇۋەپپىقىيەت  قازىنىش  ، مەن  توغۇلۇش  بىلەنلا  خۇدانىڭ
ماڭا   يۈكلىگەن   مەسئۇلىيىتىدۇر .
