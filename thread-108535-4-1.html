   ئەخلاققا ئۇيغۇن بولمىغان سۆيگۈ ھېچكىمگە بەخت ئاتا قىلالمايدۇ .

مۇشۇ سىزنىڭ نادىر ئىمزايىڭىزكەنغۇ  ؟؟؟  ھىڭ     كۆپكۆك كۆكىرىپ كەتەكن ئىمزاكەنغۇ بۇ

  
ئىمزا دىگەننى مۇنداق قىشىڭىزدەك قۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇۇپ              



     قارا رەڭدە يازسىڭىز  خەتنىڭ رەڭگى دىگەن نىگىرنىڭ يۈزىدەك ياكى مۈشۈك يالاپ قويغان شىبلىتنىڭ تۇمۇشقۇدەك پا-پار ،پار-پار پاقىراپ تۇرسا ئاندىن تېما ئېگىسىنىڭ كۆزلىرىنى چىقىپ دىققىتىنى تارتىدۇ .

   نىمە بۇ خۇددى قارا سىياھ تاپالمىغاندەك  ، 15 -يىلنىڭ ئالدىدىكى مەدىنىيەت ئىنقىلابىدا ئىشلەپ چىقىرىلغان سىياھتا ئىمزا يازسسا مانا مەندىن باشقا كىم قارايدۇ.

   ئەسلى مىنىڭمۇ كۆزۈمگە چىلىقمايدىكەن ئىمزايىڭىز ، ئالىدراش بازارغا كىتىپ بارسام ، سىزنىڭ ئىشىك ئالددىن ئۆتۈپ قاپتىمەن ،  كۆزۈمگە كۆك رەك  رېئاكسىيە قىلغاچ ئىشىك بىشىغا يىزىپ قويغان  كۆك رەڭلىك نادىر ئىمزايىڭىز كۆزۈمنى چىقىپ كەتتى قارىسام  ئەخلاققا ئۇيغۇن بولمىغان سۆيگۈ ھىچكىمگە بەخىت ئاتا قىلالمايدۇ دىگەن بىر جۇملا نادىر ئىمزا بولدى .
   مەن بۇ ئىمزانى ئۇيغۇرچىغا تىخىمۇ يىقىنلاشتۇرۇپ
     رەڭگى ئارلاشما رەڭ ئىرقىغا تەۋە بولغان ئارغۇن رەڭدە يىزىلغان ئىمزالار ھىچكىمنىڭ دىققىتىنى تارتالمايدۇ دەپ تەرجىمە قىلغان  باغاقچىنى ئىشىكنىڭ بىشىغا قىستۇرۇپ قويغان ئىدىم كۆرگەنسىزمۇ يا ؟
