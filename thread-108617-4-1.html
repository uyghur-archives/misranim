ئاغزىمغا كەلدى دەپ دەۋەرمەڭلار !! مۇشۇنداق ئۇلۇغ كۈنلەردىمۇ يالغاننى راسىت دەپ سۆزلەپ يۈرمەسمىز ، بەك ئىشەنمەيمىز دەپ كەتسەڭلار ، يوقىرىدا مەن دېگەن يانفۇنلارنى ئالدىمغا ئەپ كېلىڭلار ، مايەدە بىرسى ئۆزىنى ئۇنداق قىلالايمەن ، بۇنداق قىلالايمەن دەپ داڭلاۋاتقىنى يوق ، پەقەت كۆز قارىشىمنى دەپ باقتىم شۇ .......

ئۇنداق دەپ كېتىشكىمۇبولماسمىكىن،مەن ئەزەلدىن مىللى كارخانچىلارنى قوللىغۇچىلارنىڭ بىرى.لازىملىق نەرسىلەرنىڭ ئۇيغۇرچىسىنى تاپالمىغانداباشقىسىنى،ئۇيغۇرنىڭ دۇكىنىدىن تاپالمىغانداباشقادۇكاندىن ئالىمەن،بۇمېنىڭ تۇرمۇشۇمنىڭ بىرقىسمى.لېكىن مىللى كارخانىلارنىڭ سېتىشتىن كېيىنكى مۇلازىمىتى ئادەمنى سەل ئۈمۈدسىزلەندۈرۈپ قويىدۇ‹‹ياتنىڭ ياندىن،تۇققاننىڭ جاندىن ئۆتەر››دەيدىغان گەپ بار،باشقىسىنى قويۇپ تۇرۇپ يېقندىكىنى ئېيىتساممۇ6.ئاينىڭ20.كۈنلىرى ئەتراپىدايۇرتىمىزغاتوك يەتكۈزۈلدى،بىزمۇشۇكۈندىن باشلاپ زامانىۋى ئېلىكتىرئۈسكىنىلىرىنى ئىشلىتەلەيدىغان بولدۇق،ئۆزۈمگە،بىرتۇققانلارغابولۇپ مەشرەپ شىركىتى بازارغاسالغان توڭلاتقۇدىن تۆتنى ئالغان ئېدىم،ئېلىشتائۈرۈمچىدىكى باش دۇكىنىدىن ئالغان،ئۇلاردىن تەپسىلى سوراپ ئۈرۈمچىدىن ئالساممۇكېيىن بىرەمەسىلەكۆرۈلسەئىگەبولامسىلەرتوڭلاتقۇنى قاغىلىقتائىشلىتەتتۇق دىسەم ۋايچاتاق يوق،قاغىلىقتاۋاكالەتچىلىرىمىزبارشۇلاربىرتەرەپ قىلىپ بىرىدۇدەپ شۇنداق چىرايلىق چۈشەنچەبەردى.
شۇنىڭ بىلەن تۆت دانەتوڭلاتقۇنىڭ پۇلىنى تۆلەپ قاغىلىققايەتكۈزۈپ بېرىشكەدىيىشىپ قويۇپ يېنىپ كەلدىم،بىرەھەپتىلەردەتوڭلاتقۇمۇيېتىپ كەلدى تاشقى قېپىداھىچقانداق ئۇرۇلغان سوقۇلغان دىگەندەك بىنورمال ئەھۋاللاركۆرۈلمىگەندىكىن ساق سالامەت تاپشۇرۇپ ئالغىنىمغاخۇشال بولۇپ ئۆيگەئاپىرىپ چىرايلىق تىزىپ قويۇپ ئەتەرامىزان دىگەن كۈنى ئاخشىمى چوڭ ئۈمۈدلەربىلەن توڭلاتقۇنى توكقاچېتىپ قويۇپ ئەتىسى قارىساق توڭلاتقۇداھىچقانداق ئۆزگىرىش يوق،ئاخشىمى قانداق بولسائەتىسىمۇبىركىچەتوكقاچېتىپمۇيەنەئاخشىمى قانداق بولسائەتىسىمۇشۇپېتى تۇرۇپتۇ،ئاڭغىچەيەنەبىرتوڭلاتقۇمۇشۇنداق بولغانلىقى توغرۇلۇق تېلىفۇن كەلدى،مەشرەپ شىركىتىدىكىلەربىلەن ئالاقىلاشسام قاغىلىقتىكى رىمونىت ئورنىغائاپىرىشىمى ئېيىتتى.ئۇيەرگەئاپارسام بىرسىنىڭ بىرتال تۇرۇبىسى ئۈزۈك،يەنەبىرسىنىڭ ئىچىدەگازيوق بولۇپ چىقتى،ئۇلارئوڭشىماق بولۇپ 60كويدىن رىمونىت ھەققى تەلەپ قىلدى،يەنەشىركەتتىكىلەربىلەن ئالاقلىشىپ ئالماشتۇرۇپ بېرىشنى ئېىيسام بۇئىش بىزنىڭ ئالماشتۇرۇش دائىرىمىزدەئەمەس،رىمونىت قىلدۇرۇپ بېرىمىز،زاپچاسلىرى ھەقسز،رىمونىت ھەققى تۆلەيسىزدىدى،ئاچچىقمدابىرمۇنچەتاكاللىشىپمۇقالدىم،ئاخىرى ئاران دىگەندەبىرنى ئالماشتۇرىدىغان بىرنى رىمونىت قىلدۇرىدىغانغاكېلىشتۇق،ھەممىدىن بەك ئاچچىقمنى كەلتۈرگېنى رىمونىتچىنىڭ قاغىلىقتىن ئالماي ئۈرۈمچىدەخ....بارمىدى،ئەمدى نەگەبارساڭ باردىگىنى بولدى.شۇكۈندىن باشلاپ مۇشۇمىللى كارخانىلارنىڭ بولۇپمۇمۇشۇمەشرەپ شىركىتنىڭ مەھسۇلاتلىرى،مۇلازىمەتلىرىگەنارازى بولۇشقاباشلىدىم.
مۇشۇنداق ئىشقايولۇقۇپلاقالمىساق مىللى كارخانىلارنىڭ ھەرقانداق مەھسۇلاتلىرىنى پۈتۈ كۈچىمىزبىلەن قوللايمىز.
