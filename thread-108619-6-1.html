بۇرۇن ماڭا بىر خېنەنلىك خەنزۇ شوپۇر دۇڭ سۇڭرۇي توغۇرلۇك مۇنداق بىر گەپنى قىلىپ بەرگەن .  دۇڭ سۇڭرۇي شۇنداق ئاددىي بىر دېھقان بۇلۇپ ئەسكەرلىككە كىرگەندە  قورال _ ياراق توغۇرلۇق ھىچ قانداق ئاددىي چۇشەنچىگە ئىگە بولماي تۇرۇپلا بىۋاستە جەڭگە قاتنىشىپتىكەن  ، ھېلىقى كۆۋرۈكنى پارتىلتىدىغان چاغدا ئۇنىڭ خېنەنلىك بەنجىڭى كۆۋرۈك ئاستىغا بىللە كىرىپ بومبىنى تىرەپ قۇيىدىغانغا نەرسە تاپالماي دۇڭ سۇڭرۇيغا : سەن بومبىنى كۆۋرۈككە تىيەپ تۇتۇپ  تۇر ، مەن تىرەك تېپىپ كىلەي دەپ بومبىنىڭ پىلتىسىنڭ تارتىۋېتىپلا قېچىپ كىتىپتىكەن .دۇڭ سۇېڭرۈي دېگەن ھاماقەت  ساراڭلارچە بومبىنى تىيەپ تۇرۇپ  پارتلاپ ئۈلۈپتىكەن ، شۇنىڭدىن كېيىن خېنەنلىكلەرنى ئالدامچىلار دېگەن گەپ پەيدا بوپتىكەن .
ھېلىقى خەنزۇ يەنە ماڭا 25000 يوللۇق ئۇزۇن سەپەر توغۇرسىدا توختۇلۇپ  : ئاشۇ ئۇزۇن سەپەر جەريانىدا قىزىل ئارمىيە جەڭچىلىرى قارلىق تاغلاردا ئاشلىق مەسىلىسىنى ھەل قىلىش ئۈچۇن  ياۋا كۆكتات ، ئوت -چۆپلەرنىڭ يىلتىزى ۋە تاسما كەمەرلىرىنى پۇشۇرۇپ يەپ ھاياتىنى ساقلىغان دېگەن گەپلەرگە ئىشىنەمسەن دەپ سورىدى . مەن بىزنىڭ تارىخ دەرىسىدە شۇنداق سۆزلەنگەن دېسەم  ئۇ :  تارىخ دەرىسىڭگە ئىشەنمە ، ئويلاپ باق ئاشۇ قېلىن قار ياققان تاغدا نەدىمۇ ياۋا كوكتات ۋە ئوت - چوپلەرنىڭ يىلتىزىنى تاپقىلى بولسۇن . ئۇلارنىڭ تاسما كەمەرلىرىنى پىشۇرۇپ يىدى دىگەن گېپىنى راس دىسەكمۇ بىر تال تاسما كەمەر بىر ئادەمگە قانچىلىك ئۇزۇق بولماقچىتى . ئۇلار پەقەت ئاشۇ قارلىق تاغدا ئاچلىق ۋە كېسەللىك سەۋەبىدىن ئۆلگەن سەبداشلىرىنىڭ جەسىدىنى يەپ جان ساقلاپ بۇ سەپەرنى تۈگەتكەن دەيدۇ . ئويلاپ باقسام بۇ گەپنىڭ ھەقىقەتەن ئاساسى بار ئىكەن .
ئۇ يەنە ھازىرقى زاماندا ئۇرۇش بۇلۇپ قالسا ئاشۇ يالغان تارىقتىكى خۇاڭ جى گۇاڭ ، چى شىياۋيۈن ، دۇڭ سۇڭ رۈيدەك قەخىرمانلار مەيدانغا چىقارمۇ ؟ مۈمكىن ئەمەس مەڭگۈ مۈمكىن ئەمەس دىگەن ئىدى .
مەن يېقىندا بىر خەۋەردا ئامىركا تارقاتقۇلىرىدا جۇڭگۇ ئەسكەرلىرىنىڭ جەڭگىۋارلىغى يوق ، ئەگەر ئۇرۇش بۇلۇپ قالسا ۋەتەن ئۈچۈن جېنىنى ئاتايدىغانلار چىقماسلىغى مۈمكىن . چۈنكى ئازاتلىق ئارمىيە جەڭچىلىرىنىڭ كوپ قىسىمى يالغۇز پەرزەنىتلىك ئائىلىدىن كەلگەن ، يالغۇز پەرزەنىتلەر جۇڭگۇدا بەك ئەتىۋارلىنىدۇ ئۇلار ھاياتىنى ۋەتەن ئۇچۈن بېغىشلىيالمايدۇ دىگەن خەۋەرنى كۆرگەنتىم . ئىشقىلىپ بۇ جۇڭگۇچىلىقتا ھەممە ئىشلار يۈز بېرىدىكەن . . . . . . .
بولغاچقىلا سەندالۇ -بازغان، ئارىسىدىن قىلىچ يارالغان .
