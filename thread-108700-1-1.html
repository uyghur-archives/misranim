باشقا يۇرتلاردا قانداقكىن بىلمىدىم ،بىز ھەر يىلى تولۇقسىز ئوتتۇرا مەكتەپتىن ئوقۇش پۈتتۈرگەن ئوقۇغۇچىلارنى ئوتتۇرا تىخنىكوم ،ۋە كەسپى مەكتەپلەرگە يەتكۈزۈپ بىرىش ئۈچۈن يازلىق تەتىلدە (ۋەزىپىگە)تەشۋىقات ئۈچۈن كەنىتلەرگە چۈشىمىز.بۇ جەرياندا كۆرگەن  بىر ئىش منى تولىمۇ بىئارام قىلدى.
بىركۈنى بىر  خىزمەتدىشىم بىلەن بىللە يەنە شۇ تەشۋىقات خىزمىتىگە كەنىتلەرگە باردۇق،مەلۇم بىر كەنىتتىكى بۇ يىل ئوقۇش پۈتتۈرگەن بىر ئوقۇغۇچىمىزنىڭ ئۆيىگە كىردۇق.ئىشىكنى خىلى بىر ھازا چەككەن بولساقمۇ ھىچ بىر تىۋىش ئاڭلانمايتى،دەرۋازىنىڭ ئوڭ تەرپىگە ھۆكۈمەت ياردىمى بىلەن سىلىنغان پىششىق كىسەك  قۇرۇلمىلىق ئۆي سىلىنغان بولۇپ تىخى نەچچە كۈن ئىلگىرىلا يىپىلىپ ئىشىك دەرىزىللىرى ئورنىتىلغان .دەرۋازىنىڭ ئۇدۇلىدا بولسا كونا خام خىش قۇرۇلمىلىق ئۈچ ئىغىزلىق يەنە بىر يۈرۈش ئۆي تۇراتتى.
مەن خىزمەتدىشىمىز بىلەن بىرگە ئۆي ئىگىسىنىڭ ئىسمىنى دېھقان پەدىسىگە توۋلاپ ھويلىغا كىرىپ كەلدۇق.ھويلىغا كىرىپ بىر نەچچىنى توۋلىغاندىن كىيىن ئۆي ئىچىدىن بىر بوۋاينىڭ سۇس ئاۋازى ئاڭلاندى.
،-ياسىنكا بارمۇ سىلە؟!
-ۋوي، كىلىڭلا بالام كىلىڭلار،ئۆي ئىچىدىن ئۈستىگە ئاق يەكتەك ،تامبال كەيگەن،7-ئاينىڭ تومۇز ئىسسىقىدا باشنى ئىسسىق تۇتقۇلۇق دىگەنمۇ ئەيتاۋۇر، سەل كونىراپ قالغان سەرپۇش تۇماقنى چۆكۈرۈپ دىگىدەك كىيۋالغان  60-70ياشلار ئەپچۆرىسىدىكى ياداڭغۇ بىر بوۋاي چىقىپ كەلدى،قائىدە بويىچە بىز بوۋاي بىلەن كۆرۈشۈپ بۇ ئائىلنىڭ خاتىرجەملىكىگە دۇئا قىلغاندىن كىيىن گىپىمىزنى باشلىدۇق .
-ياسىنكا،سىلنىڭ بىر قىز نەۋرىللىرى باركەن بۇ يىل ئوقۇش پۈتتۈرگەن شۇنداقمۇ؟ھە ئەمدى بۇ نەۋرىللىرى يۇقۇرلاپ ئوقۇش ئىمتاھانىغا قاتنىشىپتىكەن ،ئەگەر نەۋرىللىرى تولۇق ئوتتۇرغا ئۆتەلىسە تولۇق ئوتتۇرا مەكتەپلەردە ئوقۇيدۇ،ئەگەر ئۆتەلمىسە ،ھەر قايسى ئاپتونۇم رايۇن، ۋىلايەت ،ناھىيە دەرىجىلىك كەسپى تىخنىكوم مەكتەپلەرگە بارمىسا بولمايدۇ،ئەمدى سىللە بۇ توغرىسىدا ئۆيدىكىلەر بىلەن قانداقراق مەسلىھەت قىلىشتىلىكىن شۇنى بىلىپ باقايلى دەپ كەلگەن ئىدۇق.
-ھە،بۇغۇ ياخشى گەپكەن باللىرىم ،ئەمدى بار ئەھۋالنى سىلەرمۇ كۆردۇڭلار مانا ئۆيۈمنىڭ بار ئەھۋالى مۇشۇ ،بۇ بالنىڭ دادىسى بالدۇرلا ئۆلۈپ كىتىپ ئانسى باشقا بىرىگە ئەرگە تىگىىپ كەتكەن ،بۇ بالىمۇ ماڭا كىچىكلا قىلىپ مەن بىقىپ چوڭ قىلغان ،بەش يىل بولدى خوتۇنۇم رەھمەتلىك ئۆلۈپ كىتىپ مەھەللەكوي يىراق بىر يۇتتىن بىر خوتۇننى نىكاھلاپ قويغان ،بۇ خوتۇنمۇ ئىككى-ئۈچ كۈندە بىر كىتىۋىلىپ ئۆيدە شۇ مۇشۇ بالام بىلەن يالغۇز قالاتتۇق ،قايسى كۈنى خوتۇن يوقىلاڭ ئىشلار بىلەن بالامغا قىيداپ كىتىپ قالدى،بالاممۇ ھازىر تاغىسىنىڭ ئۆيىدە تۇرۋاتىدۇ ،تاغىللىرى ئوقۇتمايمەن دەۋاتاتتى ،مىنىڭ بالىنىڭ كىيىنلىكىگە ئانداق ياكى مانداق دىگۈدەك ئەھۋالىم يوق مانا كۆردۈڭلا،قانداق قىلساق بولدىكىن ئەمدى،بالىغۇ ئوقۇيمەن دەپلا تۇرىدۇ.
بوۋاي مۇشۇ گەپلەرنىمۇ خىلى تەستە قىينىلىپ دەۋاتاتتى .بۇۋاينىڭ بۇ ھالىتىگە قاراپ بىزمۇ ئارتۇقچە گەپ قىلىشنى لايىق كۆرمىدۇق .
-ھە.....،ئانداق بولسا بۇ گەپلەرنى بالىنىڭ تاغىسى بىلەن بىر نىمە دىيشەيلى ،ماقۇل ئەمسە سىلىنىڭمۇ ئارامىللىرىنى بۇزۇپ قويدۇق ،ماقۇل خوش ئەمسە .بىز بوۋايدىن قىزنىڭ تاغسىنىڭ ئۆيىنى سورىۋىلىپ بوۋاي بىلەن خوشلىشىپ قايتىپ چىقتۇق.
تاغىسىنىڭ ئۆيى يىقىنلا يەردە ئىدى،بىز دەرۋازىسى ،ھويلىسىمۇ يوق ،پىشايۋانغا رەخىت بىلەن توسۇپ قويۇلغان يەنە بىر ئۆيگە كەلدۇق.پىشايۋان ئىچىدە بىر ئىككى ئايالنىڭ پاراڭلىرى ئاڭلىنىپ تۇراتتى.
-ئەسسالامۇ ئەلەيكۇم !بىز توسۇپ قويۇلغان رەخىتنى قايرىپ پىشايۋانغا ئۆتتۇق،-كىلىشسىلە، ۋۇي مۇئەللىم تىنىچلىقمۇ تىنىچ ئامان تۇردۇڭلارمۇ؟،ئايالنىڭ تەكەللۇپىدىن خىلى قائىدىلىكلىگى چىقىپ تۇراتتى.بۇ ئۆي ئادەتتىكىچە خام كىسەكتىن سىلىنغان خىلىلا كونىراپ قالغان ئۆي ئىدى.ئايال 30-35ياشلارنىڭ قارسىنى ئالغان ياش ئايال ئىدى ،يىنىدا بىر بۆشۈكتە بالسى بولسا كىرەك ،پۇشۇلداپ ئۇيقىغا كەتكەن،بۆشۈك يىنىدا بۇ ئايال بىلەن تەڭ دىمەتلىك يەنە بىر ئايال ئولتۇراتتى.
-چىمەنگۈل مۇشۇ ئۆيدە تۇرۋاتامدۇ ؟بىز ئوتتۇرا مەكتەپتىن كەلگەن ،چىمەنگۈلنىڭ ئوقۇتقۇچىللىرى بولمىز.
ئايال ،-ھە بار ،چىمەنگۈل ماياققا چىق، مۇئەللىملىرىڭ كەپتۇ.چىمەنگۈل ئۇيقۇچان كۆزلىرىنى ئۇۋلاپ چىقىپ بىز بىلەن ئاددىيغىنە سالاملاشتى.
پىشايۋان ئاستىدا بىز ئولتۇرغىدەك ئەھۋالنىڭ يوقلىقىنى بىلىپ بىز ئۆرە تۇرۇپلا گىپىمىزنى باشلىدۇق،ئايال گەپلىرىمىزنى ئاڭلىغاندىن كىيىن ئەلپازى سەل ئۆزگەردى.
-بۇ بالا ئۇزۇن بولدى بىزنىڭ ئۆيدە تۇرۋاتىدۇ،بىقىپ چوڭ قىلغان چوڭ دادىسىنىڭ ئەھۋالىنىمۇ كۆرگەنسىلە ھەقىچان سىلەرگە راسىت گەپنى قىلساق مۇشۇ ۋاققىچە بۇ بالىنى مۇشۇنچىلىك ئوقۇتتۇق ،يەنە داۋاملىق ئوقۇتىمىز دىسەك بىزنىڭ ئەھۋالىمىز ئۇنىڭغا يار بەرمەيدۇ،بۇ بالا يەنە ئوقۇيمەن دىگىنى بىلەن بۇ بالىنىڭ خەجى خىراجىتىنى تارتىپ تامىقىنى ئىتىپ ئىش كۈشلىرىنى ئورۇنلاشتۇرۇپ مۇشۇ كۈنگە كەلدۇق،يەنە داۋاملىق ئوقۇتالمايمىز .
مۇشۇ گەپلەرنى دىگۈچە مەن باشقىلاردىن ئاڭلايدىغان ئۆگەي ئانا ھەققىدىكى خىلمۇ خىل گەپلەر خىيالىمدىن كىچىپ ئۆتتى.
بۇ ئايال ھەر قانچە بولسىمۇ بۇ بالىنىڭ ئالدىدا ،قىلغىنىغا مىننەت قىلىپ بۇنداق دەپ كەتمىسىمۇ بولاتتىغۇ دىگەنلەرنى ئويلىدىم.بىچارە چىمەنگۈل بۇ گەپلەرنى ئاڭلىغىچە بىشىنى كۆتۈرمەي يىغلامسىراپ قاراپ تۇراتتى ،ئىچىم سىرىلىپ كەتتى .
خىزمەتدىشىممۇ مەندەك ئويلىدى بولغاي ،
-،شۇنداقتىمۇ سىلە مۇشۇكۈنگىچە بىقىپ چوڭ قىپسىلە بۇ بالىمۇ قىلغىنىڭلارنى بىر كۈن ياندۇرار ،بالنىڭ ئوقۇش ئارزۇسى بولغانئىكەن يەنە چىداپ بىر ئوقۇتۇپ قويۇڭلار ،بۇ بالا ئەمدى تىخى15-16 نىڭ قارسىنى ئالغان بالىكەن،  ئوقۇماي يۈرسە يەنە بىر ئىككى يىلدىن كىيىن بۇ بالنىڭمۇ تۇرمۇشنى ئورۇنلاشتۇرمىساق بولمايدۇ ھەقىچان،جىقراق ئوقىغان ئادەمنىڭ كۆز قارشى ،قىلغان ئىشى باشقىچە بولدۇ ،دەپ چۈشەندۈرۈشكە باشلىدى .
ئەمما ھىلقى ئايال گەپنى  ھىچ چۈشەنگەندەك ئەمەس ،-كىم ئوقتىدۇ؟ قانداق ئوقتىدۇ؟ نىمە بىلەن ئوقۇيدۇ بۇ؟ ،ھۆكۈمەت ھەقسىز ئوقۇتتى دىگەن بىلەن ئۇنىڭ يەنە ئۇششاق چۈششەك چىقىملىرى تۇرسا .................
گەپنىڭ قىسقىسى بىز ئوقۇتالمايمىز .نىمە دىيشىمىزنى بىلەلمىدۇق ،ئايالنىڭ ئەلپازىدىن قارىغاندا يەنە سۆزلىسەك تىخىمۇ چالۋاقاپ ئۆينى بىشىغا كىيدىغاندەك قىلاتتى.
-،ماقۇل شۇنداقتىمۇ يولداشلىرى كەلگەندە يەنە بىر ئويلىشىپ مەسلىھەت قىلغاچ تۇرشالا ،بىز ئەتىگىچە يەنە كىلىمىز .
ئۆيدىن قايتىپ چىقتۇق .

ھەي بىچارە ! ئىچىمدە بۇ بىچارە قىزغا ئىچىم ئاغرىپ قالدى ،چىمەنگۈل مەكتەپتە ئوقۇشى خىلىلا ياخشى قىز ئىدى ،مەكتەپكە ئامراق ئىدى ،تۈرلۈك ئىجتىمائى پائالىيەتلەردە خىلى ئاكتىپ، بولۇپمۇ سەنئەت پائالىيەتلىرىدە ھەر دائىم سەھنىدىن قالمايتى.ئۇنىڭ نۇرغۇن ئارزۇ ئارمانلىرى بار بولغىيتى .بەلكىم ئۇنىڭ كاتتا دوختۇر بولۇپ چوڭ دادسىنى داۋالىغۇسى،داڭلىق ئارتىس بولۇپ ئۆزىنى ئاشۇنداق تۆۋەن كۆرگەنلەرگە ،ئۆزىنى ياقتۇرمىغانلارغا ئۆزىنى بىر كۆرسىتىپ قويغىسى بار بولغىيتى ئىسىت...... ئىسىت........
ئەتراپىمىزدا چىمەنگۈلگە ئوخشاش يەنە قانچىلغان ئوقۇشقا تەشنا ئوقۇغۇچىللىرىمىز يۇقۇرلاپ ئوقۇيالماي ھەتتا تولۇقسىز ئوتتۇرا مەكتەپ ئوقۇشىنىمۇ خاتىرجەم ئوقۇيالماي جەمىيەتكە قىتىلىپ كىتىۋاتقاندۇ،يەنە قانچىلغان ئۆسمۈرلىرىمىز، پەرزەنىتلىرىمىز ئارزۇ نىمە غايە نىمە ھىس قىلىپ بولالماي جەمىيەتكە چىقىپ جەمىيەتكە ئائىلگە يۈك بولۇپ ئاجىز جانلىرىنى قىيناپ يۈردىغاندۇ؟؟؟؟؟؟
