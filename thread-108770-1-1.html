ئەسسالامۇ ئەلەيكۇم مىسرانىم مۇنبىرىدىكى ئەزىزلەر ، ماڭا ئازراق ياردەم لازىم بوپ قالغانتى ، گەپنى ئەگىتمەي ئۇدۇللا دەۋەرسەم ، مەن ھەپتە بۇرۇن يېرىم كېچىدە قورسىقىم قاتتىق ئاغرىپ كېتىپ ئۇدۇللا ۋىلايەتلىك دوختۇرخانىغا بارسام بۆرەككە تاش چۈشكەن دەپ دىئاگىنوز قويدى . ئوپىراتسىيە قىلدۇرسام 10 مىڭ يۈەندىن كۆپرەك پۇل كېتىدىكەن . شۇنىڭ بىلەن  مەن ياخشى نىيەتلىك مەلۇم بىر دوختۇرنىڭ مەسلىھەت بېرىشى بىلەن بىر خەنزۇ يولداشنىڭ شەخسى دوختۇرخانىسىدا 2000 تۆلەپ مەلۇم بىر ئۈسكىنە ئارقىلىق تاشنى ئاتقۇزۇپ پارچىلاپ چۈشۈرگۈزدۈم . ھەپتىگىچە ئىچىدىغان دورا بەردى .ھەپتە بولغاندا بېرىپ تەكشۈرتسەم تاش تامامەن تۈگەپ بوپتۇ ، لېكىن يەنە بىر بۆرەكتە سەل ئاز مىقداردا باركەن ، ئۇنىڭغىمۇ 2000 يۈەن تاپشۇرساڭ ھازىر ئېتىپ پارچىلاپ چۈشۈرىمىز دەيدۇ . لېكىن ھازىر قورسىقىمدا قىلچە ئاغرىق يوق . ئۇ دوختۇرنىڭ پۇل تېپىش ئۈچۈن دىگەن گىپىمۇ بۇنى بىلمەيمەن ، ھەم پۇلنى سەل ئازراق ئاپىرىپ قالغانلىقىم سەۋەبىدىن ماقۇل دەپ قويۇپلا دوختۇرخانىدىن يېنىپ چىقتىم . شۇڭا مەن مۇنبەرداشلاردىن مەسلىھەت سوراپ باقماقچى . بۆرەككە تاش چۈشكەن ئادەملەر ئادەتتە يېمەكلىك جەھەتتە قانداق يېمەكلىككە دىققەت قىلىشى كېرەك؟ نىمىلەرنى ئىستىمال قىلىپ نىمىلەرنى ئىستىمال قىلماسلىقى كېرەك؟ ئوپىراتسىيە قىلدۇرماي ياكى ھېلىقىدەك شەخسى دوختۇرخانىلارغا بارماي ئائىلىدە مىللىيچە داۋالاش ئارقىلىق پارچىلاپ چۈشۈرگىلى بولامدۇ؟ ، كۆپنىڭ كۆزى كۆپ دىگەندەك بۇ مۇنبەردە دوختۇرلار ياكى تۇرمۇش تەجرىبىسى كۆپ ئەزالار خېلى كۆپ سالماقنى ئىگەللەيدۇ ، شۇڭلاشقا بۇ مۇنبەرگە يوللاشنى لايىق كۆردۈم ،مۇشۇ ھەقتە بىلىدىغانلار بولسا ياردىمىڭلارنى ئايىمىساڭلار ....(ئالدىنئالا رەھمەت)
