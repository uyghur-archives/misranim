قەدىر كىچىسى توغرىسىدا ھەدىسلەر


ئەنەس رەزىيەللاھۇ ئەنھۇدىن، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم مۇنداق دىگەن: قەدىر كېچىسى بولغاندا، جىبرىئىل ئەلەيھىسسالام بىر توپ پەرىشتىلەر جامائىتى بىلەن چۇشۇپ، ھەربىر قىيامدا تۇرۇپ ناماز ئۇقۇۋاتقان ياكى ئولتۇرۇپ اللھ نى ئەسلەۋاتقان بەندىگە رەھمەت يوللاپ، مەغپىرەت سورايدۇ، ئۇلارنىڭ رۇزا ھېيىت كۈنى بولغاندا، اللھ تائالا ئۇلارنى پەرىشتىلەرگە باھالاپ، ماختاپ بىرىپ ؛ ئى پەرىشتىللىرىم! ئۆز ئەمىلىنى تۇلۇق تامام قىلغان ئىشچىنىڭ ئىش ھەققى مۇكاپاتى قانداق بۇلىدۇ؟ دەيدۇ، ئۇلار: ئى رەببىمىز! ئۇنىڭ مۇكاپاتى ، ئىش ھەققى تۇلۇق بىرىلىشى كىرەك دەيدۇ، اللھ تائالا: ئى پەرىشتىلىرىم! مىنىڭ ئەر-ئايال بارلىق بەندىللىرىم ئۆزلىرىنىڭ ئۇستىدىكى مىنىڭ پەرزىمنى تۇلۇق ئادا قىلىپ بولۇپ، ئاندىن دۇئاغا ئاۋازلىرىنى، قوللىرىنى ئۇستۇن كۆتۇرشۇپ چىقىشتى، ئىززىتىم بىلەن، ئۇلۇغلۇقۇم، سېخيلىقىم، ئۇستۇنلۈكۈم،(ئۆزۇمگە لايىق) ئورنۇمنىڭ ئۈستۈنلىكى بىلەن قەسەم قىلىمەنكى، مەن چۇقۇم ئۇلارنىڭ دۇئاسىنى ئىجابەت قىلىمەن، ئىبادىتىنى قۇبۇل قىلىمەن، ئى بەندىللىرىم! ئەمدى قايتىڭلار، مەن سىلەرگە مەغپىرەت قىلدىم، يامانلىقلىرىڭلارنى ياخشىلىققا ئالماشتۇرۇپ بەردىم، دەيدۇ. ئۇلار مەغپىرەت قىلىنغان ھالدا ئۆيلىرىگە قايتىشىدۇ.بەيھەقى توپلىغان.

ئۇبادەت ئىبنى سامىت رەزىيەللاھۇ ئەنھۇدىن، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم بىزگە قەدىر كېچىسىدىن خەۋەر بىرىش ئۈچۈن چىقسا، مۇسۇلمانلاردىن ئىككى كىشى دە – تالاش ، جىدەل، مۇنازىرە قىلىشىپ قالدى، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم: ھەقىقەتەن مەن سىلەرگە قەدىر كېچىسىنى خەۋەر قىلىپ بېرىش ئۈچۈن چىققان ئىدىم، لېكىن، پالانى بىلەن پالانى دە – تالاش ، جىدەل، مۇنازىرە قىلىشىپ، ئۇ مەندىن كۆتۈرۈلۈپ كەتتى، كۆتۈرۈلۈپ كېتىشىمۇ، ئۇنتۇپ قېلىشىممۇ سىلەرگە ياخشى بولۇشى مۇمكىن، قەدىر كېچىسىنى، رامىزان ئېيىنىڭ ئاخىرقى توققۇزىدا،يەتتىسىدە، بەشىنچى كېچىللىرىدە ئىزدەڭلار،دىدى. ئەھمەد، بۇخارى توپلىغان، سەھىھ.

ئىبنى ئۇنەيس رەزىيەللاھۇ ئەنھۇ مۇنداق دەيدۇ؛ مەن ئى اللھ نىڭ ئەلچىسى! مىنىڭ يىزىدا تۇرۇشلۇق ئورنۇم بار،اللھ قا شۇكىرىكى، مەن ئۇ جايدا اللھ قا ھەمدە ئېيتىش بىلەن بىرگە ناماز ئۇقۇيمەن، قەدىر كىچىسىنى تېپىش ئۈچۈن مىنى بىركېچىگە بۇيرىسىلا، مەن شۇ كېچىسى بۇ مەسجىدكە چۇشەي دىسەم، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم ماڭا؛ 23-كېچىسى چۇشكىن دىدى، ئۇنىڭ ئوغلىغا: داداڭ قانداق قىلاتتى؟ دىيىلدى، ئۇ: دادام شۇكۈنى ئەسىرنى ئۇقۇپلا مەسجىدكە كىرىپ كەتسە، بامداتنى ئۇقىمىغىچە ھېچقانداق ھاجەتكە چىقمايتتى، بامدات نامىزىنى ئۇقۇپ، مەسجىدنىڭ ئىشكى ئالدىدىكى ئۇلىغىغا مىنىپ، ئۆزى تۇرۇشلۇق يىزىغا كىتەتتى، دىدى. ئەبۇداۋۇد توپلىغان.

ئىبنى ئۆمەر رەزىيەللاھۇ ئەنھۇما مۇنداق دەيدۇ؛ رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەمدىن قەدىر كېچىسى توغرىسىدا سورالغاندا؛ قەدىركېچىسى ھەر بىر رامىزاندا باردۇر دىگەن. (يەنى قىيامەتكىچە ھەممە رامىزاندا باردۇر). ئەبۇداۋۇد توپلىغان، ئۇ:بۇ ھەدىس ئىبنى ئۆمەر رەزىيەللاھۇ ئەنھۇمانىڭ ئۆزىنىڭ سۆزى دىگەن.

ئەبۇ بۇكرەت رەزىيەللاھۇ ئەنھۇدىن، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم مۇنداق دىگەن: قەدىر كېچىسىنى ئاخىرقى ئون كۈننىڭ قالغان توققۇزدا، يەتتىسىدە، بەشىدە، ئۈچىدە، ئەڭ ئاخىرقى تاق كېچىللىرى ئىزدەڭلار. ئەھمەد، تىرمىزى، ھاكىم، بەيھەقى توپلىغان، سەھىھ.

ئائىشە رەزىيەللاھۇ ئەنھا مۇنداق دەيدۇ؛ مەن؛ ئى اللھ نىڭ ئەلچىسى!ئەگەر بىلسىلە قېنى دەپ بەرسىلە، مەن قەدىر كېچىسى نىمە دەيمەن؟ دىسەم، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم ماڭا؛ «ئى ئاللاھ! ھەقىقەتەن سىلى ئەپۇ قىلغۇچى، ئەپۇ قىلىشنى ياخشى كۆرىلا، مەندىن ئەپۇ قىلسىلا»دىگىن،دىدى.ئەھمەد، تىرمىزى، ئىبنى ماجە، ھاكىم توپلىغان، سەھىھ، بۇ دۇئا قەدىر كېچىسى ئوقۇلىدۇ.

ئائىشە رەزىيەللاھۇ ئەنھادىن، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم رامىزاننىڭ قېلىپ قالغان ئون كۈنى كىرگەندە، ئىشتىنىنى مەھكەم چىگىپ، بۇ ئاخىرقى ئون كۈننىڭ كېچىلىرى تۈنەيتتى، ئەھلىنى ئويغۇتاتتى. بەيھەقى، بۇخارى،مۇسلىم، ئەبۇ داۋۇد، نەسەئى، ئىبنى ماجە توپلىغان، سەھىھ.

ئائىشە رەزىيەلاھۇ ئەنھادىن، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم رامىزاننىڭ ئاخىرقى ئون كۈنىدە، ئۇ كۈنلەردىن باشقا چاغدا كۆرسەتمىگەن، قىلىپ باقمىغان ئاجايىپ تېرشىچانلىقلارنى كۆرسىتەتتى،(يەنى ئىبادەتتە قاتتىق چىڭ تۇراتتى،كېچىلىرى تۈنەيتتى، ئەھلىنى ئويغۇتاتتى). ئەھمەد، مۇسلىم، تىرمىزى توپلىغان، سەھىھ.

زەرر ئىبنى ھۇبەيش رەزىيەللاھۇ ئەنھۇ مۇنداق دەيدۇ؛ مەن ئۇبەي ئىبنى كەئبى رەزىيەللاھۇ ئەنھۇدىن سوئال سوراۋېتىپ، ئۇنىڭغا؛ ھەقىقەتەن قېرىندىشىڭ ئىبنى مەسئۇد رەزىيەللاھۇ ئەنھۇ؛ كىم بىر يىل مۇكەممەل كېچىسى قىيامدا تۇرۇپ تۇنىسە، ئاندىن قەدىر كېچىسىنى تاپالايدۇ، دەيدۇ دىسەم، ئۇماڭا: اللھ ئۇنىڭغا رەھمەت قىلسۇن، ئۇ كىشىلەرنىڭ يۇلىنىۋالماسلىقىنى مەقسەت قىلىپتۇ، لىكىن ئۇ قەدىر كېچىسىنىڭ رامىزاندا، رامىزاننىڭ ئاخىرقى ئونىدا، بۇلۇپمۇ 27-كېچىسى ئىكەنلىكىنى بىلىدۇ دەپ، ئاندىن قەدىر كېچىسىنىڭ رامىزاننىڭ 27-كېچىسى ئىكەنلىكىگە ئىستىسنا قىلمايلا يەنى اللھ خالىسا دىمەيلا قەسەم قىلدى، مەن: ئى ئەبامۇنزىر!قايسى نەرسە بىلەن شۇنداق جەزىمە قىلىپ، 27-كېچىسى دەيسەن؟ دىسەم، ئۇماڭا: رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم بىزگە خەۋەر قىلغان ئالامەت بىلەن دەپ؛ قۇياش شۇكۈنى نۇرسىز چىقىدۇ دىدى. مۇسلىم توپلىغان سەھىھ.

ئەبۇ سەئىد رەزىيەللاھۇ ئەنھۇ مۇنداق دەيدۇ؛ رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم رامىزاننىڭ ئەۋۋەلقى ئونىدا ئىتكاپ قىلدى، ئاندىن ئوتتۇرىدىكى ئونىدا، تىرىدىن بولغان گۇمبەز چىدىردا ئېتكاپ قىلىپ،ئاندىن بېشىنى چىقىرىپ مۇنداق دىدى: ھەقىقەتەن مەن مۇشۇ قەدىر كېچىسىنى ئىزدەپ، رامىزاننىڭ ئەۋۋەلقى ئونىدا ئىتكاپ قىلدىم، ئاندىن ئوتتۇرىدىكى ئونىدا ئېتكاپ قىلدىم، ئاندىن ماڭا پەرىشتەئارقىلىق قەدىر كېچىسى رامىزاننىڭ ئاخىرقى ئونىدا دىيىلدى، كىم مەن بىلەن ئېتىكاپ قىلماقچى بولسا، رامىزاننىڭ ئاخىرقى ئونىدا ئېتكاپ قىلسۇن. ماڭا بۇ قەدىر كېچىسى چۈشۈمدە كۆرسىتىلگەن، ئاندىن كېيىن ئۇنتۇلدۇرۇلدۇم، ھەقىقەتەن مەن ئۆزۈمنى قەدىر كېچىسىنىڭ ئەتىگىنى سۇ بىلەن لايغا سەجدە قىلىپ كۆردۈم، (يەنى ئالامىتى قار – يامغۇردۇر) ئۇنى رامىزان ئېيىنىڭ ئاخىرقى ئون كۈن ئىچىدىكى ھەربىر تاق كېچىلەردە ئىزدەڭلار. ئەبۇ سەئىد رەزىيەللاھۇ ئەنھۇ مۇنداق دەيدۇ؛ شۇكېچىسى يامغۇر ياغدى، مەسجىد خورما شاخلىرى بىلەنلا يېپىلغان بۇلۇپ، مەسجىدتىن يامغۇر ئۇتۇپ كەتتى، مەن رامىزاننىڭ 21-كۈنى ئەتىگەندە، ئىككى كۆزۇم بىلەن رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەمنىڭ پىشانىسىدا يامغۇر سۇيى بىلەن لاينىڭ ئەسىرنى كۆردۇم. مالىك، ئەھمەد، بەيھەقى،بۇخارى،مۇسلىم، نەسەئى، ئىبنى ماجە توپلىغان، سەھىھ.

ئىبنى ئابباس رەزىيەللاھۇ ئەنھۇدىن، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم مۇنداق دىگەن: قەدىر كېچىسىنى رامىزاننىڭ ئاخىرقى ئونىدا ئىزدەڭلار، قېلىپ قالغان توققۇزىنچى، يەتتىنچى، بەشىنچى كېچىسى ئىزدەڭلار، بۇخارى، ئەھمەد، ئەبۇ داۋۇد توپلىغان، سەھىھ. ئەبۇ بۇكرەت رەزىيەللاھۇ ئەنھۇدىن، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم مۇنداق دىگەن: قەدىر كېچىسىنى ئاخىرقى ئون كۈننىڭ قالغان توققۇزدا، يەتتىسىدە، بەشىدە، ئۈچىدە ئەڭ ئاخىرقى تاق كېچىللىرى ئىزدەڭلار. ئەھمەد، تىرمىزى، ھاكىم، بەيھەقى توپلىغان، سەھىھ.

ئىبنى ئۆمەر رەزىيەللاھۇ ئەنھۇدىن، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەمنىڭ ساھابىللىرىدىن بىركىشىنىڭ چۇشىدە، ئۇنىڭغا قەدىر كېچىسىنىڭ (رامىزاننىڭ)ئاخىرقى يەتتىسىدە ئىكەنلىكى كۆ رسىتىلىپتۇ، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم ئۇنى ئاڭلاپ: مەن چۈشۈڭلارنىڭ ئاخىرقى يەتتىسىگە مۇۋاپىق كەلگىنىنى كۆردۈم، كىم قەدىر كېچىسىنى ئىزدىمەكچى بولسا، رامىزان ئېيىنىڭ ئاخىرقى يەتتىسىدە ئىزدىسۇن،دىدى. مالىك، ئەھمەد، بەيھەقى، بۇخارى،مۇسلىم توپلىغان، سەھىھ

ئائىشە رەزىيەللاھۇ ئەنھادىن، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم مۇنداق دىگەن: قەدىر كېچىسىنى رامىزاننىڭ ئاخىرقى ئون كۈنىدىكى تاق كېچىلەردىن ئىزدەڭلار. ئەھمەد، بەيھەقى،بۇخارى، تىرمىزى توپلىغان، ئىبنى ئۆمەر رەزىيەللاھۇ ئەنھۇدىن، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم مۇنداق دىگەن: قەدىر كېچىسىنى ئىزدەڭلار، ئۇنى ئىزدىمەكچى بولغان كىشى، رامىزاننىڭ 27 – كېچىسى ئىزدىسۇن. ئەھمەد توپلىغان، ئىبنى ئۆمەر رەزىيەللاھۇ ئەنھۇدىن، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم مۇنداق دىگەن: قەدىر كېچىسىنى ئاخىرقى يەتتىسىدە ئىزەدەڭلار، مالىك، مۇسلىم، ئەبۇ داۋۇد توپلىغان، سەھىھ. ئابدۇللاھ ئىبنى ئۇنەيىس رەزىيەللاھۇ ئەنھۇدىن، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم مۇنداق دىگەن: قەدىر كېچىسىنى رامىزاننىڭ 23 – كېچىسى ئىزدەڭلار. تەبرانى توپلىغان، سەھىھ.(ھەدىسلەرنىڭ بىر – بىرىگە ئوخشىماي كېلىشى، بۇ كېچە مەخپى قىلىنغان توغرىسىنى ئاللاھ بىلىدۇ، لېكىن، قەدىر كېچىسى رامىزاننىڭ ئاخىرقى ئونىدا،تاق كېچىسى چوقۇم بار،بۇ سەھىھ 27 – كېچىسى سەھىھ دىگەنلەرمۇ بار.)
