ئەسسالامۇ ئەلەيكۇم مىسرانىم مۇنبىرىدىكى بارلىق تورداشلار ياخشىمۇ سىلەر؟
نەچچە كۈن بۇرۇن بېكىتىمىزدە شىنجاڭ ئونۋېرسىتىتى فىزكا پەن ئاسپىرانتى،شىنجاڭ مەمۇرلۇق ئىمتىھان تەتقىقاتچىسى،مەمۇرلۇق ئىمتىھان سۆرەتلىك دەرىسلىكلىرىنى ئىشلەپ باللانىڭ ياخشى باھاسىغا ئىرىشىپ كىلىۋاتقان ئابدۇغېنى مۇئەللىم ۋە بېكىتىمىز (www.7student.com) ھەمكارلىشىپ 2013-يىللىق شىنجاڭ مەمۇرلۇق ئىمتىھانىغا توردا تەييارلىق قىلىش(yy  قانىلىمىزدا دەرىس سۆزلەش) توغرىسىدا پىلان ۋە تەكلىپ تېمىسى يوللىغان ئىدۇق.باللانىڭ قىزغىن قوللىشى ۋە تەكلىپلىرى بىلەن قۇرۇشنى ھەقىقىي قارار قىلىپ تىزىملاشنى باشلىغان ئىدۇق،ئەسلى پىلاندا پەقەت 50 كىشىلىك قىلىشىنى قارار قىلغان ئىدۇق،لىكىن بۇ 50 سان ئىككى كۈن ئىچىدىلا توشقان بىلەن تىزىملىتىدىغانلارنىڭ ئايىغى ئۈزۈلمىدى،شۇ مۇناسىۋەت بىلەن ئويلىشىش ئارقىلىق ساننى 100 گە كۆپەيتتۇق،دىمەك يەنە 50 ئادەم تىزىملايمىز،ئۇنىڭدىن جېق تىزىملىمايمىز.شۇ مۇناسىۋەت بىلەن تىزىملىتىش ئارزۇسى بارلارغا يەنە بىر ئۇقتۇرىۋېتەيلى دىگەن مەقسەتتە بۇ تېمىنى مىسرانىمغا ئېلىپ كەلدۇق.تۆۋەنچىلىك بىلەن باشقۇرغۇچىلارنىڭ تەستىقلاپ قۇيۇشىنى ئۈمىد قىلىمىز!
دەرۋەقە ئالاھىدە شىتات ئوقۇتقۇچىلىق ئىمتىھان نەتىنجىسى چىققان بولسىمۇ ئۆتكەنلىرىمىز پەقەت ئاز بىر قىسىمىلا،ھەم تېخى يۈز تۇرانە ئىمتىھانلار بار.ئۆتەلمىگەن دوستلىرىمىزمۇ ئاز ئەمەس،ھەم ئۆتەلمىگەنلىك بىلەن بوشىشىپ،ئىرادىمىزنى يوقاتساق تېخىمۇ بولمايدۇ،شۇڭا ۋاقىتنى چىڭ تۇتۇپ باشقا ئىمتىھانلارغا بۇلۇپمۇ شىنجاڭ مەمۇرلۇق ئىمتىھانىغا ياخشى تەييارلىق قىلىشىڭىزنى تەۋسىيە قىلىمەن!چۈنكى بۇمۇ بىر مۇھىم بىر پۇرسەت،ۋاقىتمۇ ئاز قېلىۋاتىدۇ.
yy بولسا ئىقتىدارلىرى qq غا ئوخشاپ كېتىدىغان،لىكىن ئۇنىڭدىن ئىلغار بولغان بىرخىل ئالاقىلىشىش قورالى.yy دا قانداق دەرىس ئاڭلايمىز دىگەندە مۇئەللىم ئاغزاكى دەرىس سۆزلەيدۇ،باشقىلار ئاڭلاپ تۇرىمىز ھەم مۇئەللىمنىڭ تەييارلىغان يازمىچە ۋە سۈرەتلىك دەرىسلىكلىرىنى كۈرۈپ تۇرىمىز،قانداق سۇئال تۇغۇلسا شۇ yy توپىمىزدا خەت يېزىش ئارقىلىق سورىسىڭىز بولىدۇ.ئىقتىدارلىرى ئىنتايىن مۇكەممەل ۋە قولايلىق،بۇنى بەلكىم yy دا دەرىس ئاڭلاپ باققان دوستلىرىمىز بىلىشى مۇمكىن.قاتناشقانلارغا كومپىيۇتېرىڭىزنى يىراقتىن تىزگىنلەش ئارقىلىق قانداق ئىشلىتىش ۋە دەرىس ئاڭلاشنى ئۆگىتىپ قويىمىز!(ئىشلىتىش ئۇسۇلى ھەم بېكىتىمىزدە سۈرەتلىك دەرىسلىك قىلىپ يولىنىدۇ.)

تىزىملىتىش ۋاقتى:2013- يىل 7-ئاينىڭ 22-كۈنىدىن 30-كۈنىگىچە (سان توشقان ھامان توختىتىلىدۇ)

تىزىملتىش ئۇسۇلى: qq دا تۆۋەندىكى نۇمۇرنى قاچلاپ ئالاقىلىشىپ ياكى بىۋاستە تېلفۇن قىلىپ،قارشى تەرەپكە سورالغان مۇناسىۋەتلىك ئۇچۇرلىرىڭىزنى تىزىملىتىش ئارقىلىق تىزىملىتىسىز.

دەرىس باشلاش ۋە ئاخرلىشىش ۋاقتى:2013-يىل 8-ئاينىڭ 1-كۈنىدىن 31-كۈنىگىچە

دەرىس سۆزلەش ۋاقتى ۋە سائىتى:8-ئاينىڭ 1-كۈندىن باشلاپ ھەر كۈنى كەچلىگى شىنجاڭ ۋاقتى سائەت 7:30 دىن 9:30 غىچە ئىككى سائەت دەرىس سۆزلىنىدۇ.ھەپتىدە 6 كۈن،ئۇمۇمىي دەرىس سائىتى 50 سائەت

2013-يىللىق مەمۇرلۇق ئىمتىھان مۆلچەر سۇئاللىرى(بايان ۋە مۇھاكىمە ،مەمۇرىي كەسپىي ئىقتىدار سىناش)

ئوقۇش پۇلى:200 يۈەن (ئادەتتىكى مەمۇرلۇق كۇرۇسلىرىنىڭ ئوقۇش پۇلىنىڭ ئوندىن بىرى)
بۇ ھەم مەمۇرلۇق كۇرۇسلىرىغا بېرىپ ئوقۇشقا ئىقتىساد،ۋاقىت ۋە باشقا شارائىتلىرى يار بەرمەيدىغان،ئۆز ئۆگىنىشتە قىيىنچىلىققا يۇلۇققان دوستلار ئۈچۈن ياخشى بىر پۇرسەت!

سىزدە بۇلۇشقا تېگىشلىك شەرتلەر:ھەر كۈنى كەچلىگى بەلگىلەنگەن ۋاقىتتا تورغا چىقىش شارائىتىڭىز بۇلۇشى كېرەك،تېلفۇن ئارقىلىق ئاڭلايمەن دىسىڭىز تېلفۇنىڭىزنىڭ ئېقىم مىقتارىغا كاپالەتلىك قىلىسىز(لىكىن كومپىيۇتېردا تورغا چىقىشىڭىزنى تەۋسىيە قىلىمىز!)

پۇل تاپشۇرۇش ئۇسۇلى: بانكا كارتىسىغا پۇل سېپ بېرىش ياكى يۈتكەپ بېرىش ئارقىلىق تاپشۇرسىڭىز بولىدۇ(بانكا كارتا نۇمۇرى 26-چىسىلا چىقىرىلىغاندىن كىيىن ئاندىن پۇل تاپشۇرۇسىڭىز بولىدۇ.ئاۋال تىزىملىتپ قۇيۇپ تۇرىسىز!)

شۇڭا بۇمۇ ھەم مەمۇرلۇق كۇرۇسلىرىغا بېرىپ ئوقۇشقا قۇربى ۋە شارائىتى يەتمەيۋاتقان مەمۇرلۇق ئىمتىھان قاتناشچىللىرى ئۈچۈن ياخشى پۇرسەت.شۇڭا قاتنىشىشنى خالايدىغانلار تىزىدىن ئالاقىلىشىپ تىزىملىتىۋېلىڭ!سان چەكلىك،سان توشقان ھامان تىزىملاش توختىتىلىدۇ.
