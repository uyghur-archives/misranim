يۇرتىمىز سۇ ناھايىتى قىممەتلىك بىر رايون قەدىمدىن بىرى ئەجدادلىرىمىز قۇم -بوران چۆللۆك بىلەن كۈرەش قىلىپ پارچە -پارچە بوستانلىقلاردا سۇ قوغلىشىپ ئاساسلىقى سۇ بىلەن تەبىئەتنى ئۆزگەرتىپ كەلدى .لېكىن ھۆل يىغىننىڭ ئازبولۇشى ،مۇھىتنىڭ ناچارلىشىشى ،نوپۇس ۋە تىرىلغۇ يەرنىڭ كۆپىيىشى سەۋەبلىك سۇ ۋە كىشىلىرىمىز ئارىسىدىكى راستىنى ئىيتقاندا سۇبىلەن دېھقانلىرىمىز ئارىسىدىكى زىددىيەت كۈنسايىن كۈچەيمەكتە ،ئاساسلىقى جەنۇبى شىنجاڭدىكى خوتەن، قەشقەر شەرقى شىنجاڭدىكى تـۇرپان، قۇمۇل رايونلىرى رۇشەن سۇ كىرزىسىغا ئۇچرىماقتا .

ئەمدى نەق گىپىمىزغا كەلسەك ،مەن يىقىندا 90-يىللاردا تېلۋۇزورلاردا قويۇلۇپ كەڭ خەلق ئاممىسىنىڭ قىزغىن ئالقىشىغا ئىرىشكەن 6قىسىملىق تېلۋىزىيە تىياتىرى <<ئەڭ ئاخىرقى كۆل >>فىلىمىنى كۆردۈم ،گەرچە بۇ فىلىمنى مەن كىچىك ۋاقىتلىرىمدا قويۇلغاندا كۆرگەن بولساممۇ ۋەقەلىكى تازا ئىسىمدە قالماپتىكەن ،بۇ قىتىم بۇ فىلىمنى كۆرۈپ ئىنتايىن تەسىرلەندىم بولۇپمۇ باش رولدىكى نۇسرەتنىڭ ئۆلىمى ، سىدىقنىڭ قەيسەرلىكى ئاۋام ئۈچۈن جان كۆيەرلىكى بولۇپمۇ فىلىم ئاخىرىدا نۇسرەتنىڭ دادىسىنىڭ قىلغان نالىسى، ئەلۋەتتە ئادەمنى تەسىرلەندۈرمەي قالمايدۇ .
فىلىم داڭلىق يازغۇچىمىز ئەختەم ئۆمەر ئاكىنىڭ <<قۇرتلاپ كەتكەن كۆل >>ناملىق پوۋىستىغا ئاساسەن سىنارىيەلەشتۈرۈلگەن بولۇپ فىلىمدا80-يىللارنىڭ ئاخىردىكى تەكلىماكاننىڭ گىرۋىكىدىكى قۇم قۇيرۇق كەنتدە يۈزبەرگەن ئىچىنىشلىق تىراگىدىيە باش نۇقتاقىلىپ تۇرۇپ تەسۋىرلەنگەن .ئاساسلىق رولدىكى سەيدىن ئائىلىسىدىن باشلىنىپ ،قۇم قۇيرۇق كەنتىگە قۇدۇق سۈيى كەلگەن جەرياندىكى ۋەقەلىك تىراگىدىيە ،مۇتەسسىپلەر بىلەن يېڭىلىق تەرەپدارلىرىنىڭ كۈرەشلىرى رۇشەن ئېچىپ بىرىلگەن ،ئاساسلىقى باپكارلار كۆلىنىڭ مەينەت سۈيىنى ئىچىپ قورسىقى يوغىناپ قالغان ،نۇسرەتنىڭ تارتقان ئازاب -كۈلپەتلىرى نۇسرەتكە چاپلانغان بوھتان پىتنە -پاساتلار ،ئاخىرىدا نۇسرەتنىڭ پاجىئەلىك ئۆلىمى ،شۇ ئارىدا كەنىت سىكىرتارى بىلەن سەمەت غوجامنىڭ ئاشكارا يوشۇرۇن كۈرەشلىرى ،ئاخىرىدا سىدىق تەرەپتارلىرىنىڭ غەلىبە قىلىپ قۇدۇق كولاپ سۇ يەتكۈزۈشتىن ئىبارەت خەلققە پايدىلىق ئىشى خىلى ئەتراپلىق تەشۋىق قىلىنغان .

كۆرۈرمەنلەرنىڭ ياقتۇرىشىغا ئىرىشكەن ئاساسلىق تەرەپلەر خىلى قويۇق كۆرسىتىلگەن دىنى پۇراق ،قۇم قۇيرۇق كەنتىدىكىلەرنىڭ نادانلىقى ،خۇراپاتلىق ،يېڭىلققا ئىنتىلىش روھى ،پاكىز سۇغا بولغان تەشنالىقى يېزىلىمىزدا بولىۋاتقان سۇ قىس مەسلىسى ئەينەن تەسۋىرلىنىشى بولۇپمۇ نۇسرەتنىڭ پاجىئەلىك ئۆلىمى ،سەمەت غوجامنىڭ رەزىللىكى قاتارلىق ۋەقەلىكلەر ،كۆرۈرمەنلەرنىڭ دىققىتىنى تارتىپ شۇ دەۋرلەردە 90-يىللاردا خىلى تەسىر قوزغىغان فىلىم ھىسابلىنىدۇ .
مەن بۇ فىلىمنى كۆرگەندىن كېيىن فىلىمنىڭ ئەسلىسى بولغان ئەختەم ئۆمەر ئاكا يازغان <<قۇرتلاپ كەتكەن كۆل >>ناملىق پوۋىستنى ئوقۇپ بىقىش ئىستىكى تۇغۇلدى ،راستىمنى ئىيتقاندا بۇ پوۋىستىنى ئۇقۇپ باقمىغانىدىم ،ئاخىرى ئىزدەپ يۈرۈپ <<ئاھ ،رەھىمسىز دەريا >>ناملىق پوۋىستلار توپىلىمىدىن ئۇ پوۋىستنى ئوقۇدۇم ،ئوقۇش جەريانىدا پوۋىستتا تەسۋىرلەنگەن ۋەقەلىك بىلەن ،فىلىمدا كۆرسۈتۈلگەن مەزمۇننىڭ  خىلى كۆپ يەرلىرىنىڭ ئوخشىمايدىغانلىقىنى ئەختەم ئۆمەر ئاكىنىڭ ھېسىياتىغا ،ئەسەرنىڭ چىنلىقىغا ھۆرمەت قىلىنمىغانلىقىن ھېس قىلدىم .
فىلىمگە قۇدۇق كولاش ئەترىتى ۋە شۇنىڭغا مۇناسىۋەتلىك كىشى ئىشلار مەجبۇرى قوشۇلغان ئەملىيەتتە پوۋىستنىڭ ئاخىرغىچە قۇم قۇيرۇق كەنتىگە (پوۋىستتا ئاياغقۇم )ھەتتا قايتا چىپىلىدىغان ئۆستەڭ سۈيىمۇ يىتىپ كېلەلمىگەن ،گەرچە كەنت كادىرلىرى ئۆستەڭ چىپىشنى ئويلاشقان بولسىمۇ لېكىن قۇم قۇيرۇق كەنتىدىكىلەر يەنىلا <<قۇرتلاپ كەتكەن كۆل >>سۈيىنى ئىچىشتىن ئىبارەت قىسمەتتىن قۇتۇلالمىغان ،فىلىم پىلانلىغۇچى ،سىنارىست، رىژىسسورلار قەستەن بۇۋەقەلىكنى فىلىمگە كېرگۈزگەن ،بەلكىم پارتىيەنىڭ سىياسىتىنى تەشۋىق قىلماقچى بولغاندۇ ؟ لېكىن شۇ ۋاقىتتا ئەمەس ،فىلىم قويۇلغۇچە بەلكىم تاھازىرغىچە قۇم قۇيرۇق كەنتىگە پاكىز سۇنىڭ بارغانلىقى بۇ تېخىچە سىر .
باش رولدىكى نۇسرەت بىلەن ئامانگۈلنىڭ قۇدۇق كولىغۇچىلار بىلەن بىرىپ كېلىش جەريانىدىكى ئۇلارنى بىرخىل يېڭىلىپ ئىلىپ كەلگۈچى،شاپائەتچى سۈپىتىدە ئىشىنىشى ،ھەم ئۇلارنىڭ تەسىرگە ئۇچراپ يۈزىنى ئىچىشى ،گىرىم قىلىشى پوۋىستتا قەتئى يىزىلمىغان ،پوۋىستتا ئۆز قىزلىق ئىپپىتىنىڭ  بوھتان پىتنە -پاساتقا ئۇچرىغىنى كۆرگەن نۇسرەتنىڭ ئۆزىنىڭ ئۆلتۈرىۋىلىش ئارقىلىق پاكل؛ىقنى ئىپادىلىشى بىلەن فىلىمدىكى ئەخمىقانا ھەركەتلەر قەتئى ماس كەلمىگەن .ئۇ ۋەقەلىكلەر ئەمەلىيەتكە ئۇيغۇن ئەمەس قەستەن باشقا يىتەكلەنگەن .

پوۋىستتىكى مۇھەممەت سەيدىن بىلەن فىلىمدىكى سەيدىن ،پوۋىستتىكى ياسىن مەدەك بىلەن فىلىمدىكى سىدىق ئوبرازى .
پوۋىستتىكى مۇھەممەت سەيدىن توغرا سۆزلۈك ،ھەققانىيەتچى پاك ،ھەم غورۇرلۇق مۇسۇلمان سۈپىتىدە ئوتتۇرىغا چىقىرىلغان ،ھەم پوۋىستتىكى بىردىنبىر ئادالاتسىزلىككە قارشى تۇرغۇچى سۈپىتىدە ئوتتۇرىغا چىقىرىلىدۇ  ،پوۋىستتا مەيلى ئابدۇرىشىت ئىمام بولسۇن ،مەيلى مىراب ،كەنت باشلىقى، سىكىرتار،بولسۇن ھەممىسى ئىغىز بۇرۇن يالىشىپ مۇھەممەت سەيدىن ئۆزى يالغۇزلا قالغان ،يەنى فىئودال خوجىلار بىلەن تۆۋەننى بىسىپ يۇقىرنى كۆتۈرىدىغان جانباقتى كادىرلارنىڭ چىكىدىن ئاشقان ھەركەتلىرىگە قارشى تۇرغان قەھرىمان سۈپىتىدە  مەدىھلىنىدۇ .ئەپسۇس ئاخىرىدا نۇسرەتنىڭ ناھەق ئۆلىمى سەۋەبلىك ئۆزىنى گۇناھكار ھىسابلاب ساراڭ بولۇپ قىلىشى بىلەن <<قۇرتلاپ كەتكەن كۆل>>گە يالىڭاچ سەكرەيدۇ .
فىلىمدىكى سەيدىنمۇ گەرچە بىر مۇسۇلمان سۈپىتىدە ھەققانىيەتنى ياقلىغۇچى سۈپىتىدە ئوتتۇرىغا چىقسىمۇ ئۇنىڭغا يەنە بىر سالاھىيەت پىشقەدەم كەنىت كادىرلىق سالاھىيەت قىتىلىپ قالىدۇ ،سەيدىننىڭ ھەققانىيەت ئۈچۈن قىلغانلىرى بىر ھەققىقى ساپ مۇسۇلمانلىق سۈپىتىدە ئەمەس ،پارتىيە ئەزاسىلىق سۈپىتىدە قىلغان دېگەندەك ئىددىيە ئالغا سۈرىلىدۇ .
فىلىمدىكى يەنە بىر ۋەقەلىك يېزىنىڭ سۇكادىرى ئامانگۈلنى مەجبۇرى خوتۇنلۇققا بىرىلىشى ،فىلىمدا نۇرغۇنلىغان ئەگرى- توقايلىقلاردىن كېيىن ئامانگۈل بۇ قىسمەتتىن قۇتۇلۇپ قالىدۇ ،قۇتۇلۇپ قىلىشىدا سىدىق ،سەيدىن ھەم قۇدۇق كولىغۇچىلارئالاھىدە رول ئوينايدۇ .
ئەپسۇس پوۋىستتا بولسا ھاشىرخان خۇراپاتلىقنىڭ قۇربانى سۈپىتىدە ئابدۇرىشىت ئىمامغا مەجبۇرى خوتۇنلۇققا بىرىلىپ دەپسەندە قىلىشغا ئۇچرايدۇ ،مۇھەممەت سەيدىندىن ئىبارەت بۇ ئاقكۆڭۈل كىشى يالغۇز كۈچى ئاجىزكېلىپ قىلىپ توسۇشقا كۈچى يەتمەي قالىدۇ .
فىلىمدە بولسا قانۇن -تۈزۈم پارتىيەنىڭ سىياسىتىن ئومۇملىشىۋاتقانلىقى تەكىتلەنسىمۇ پوۋىسستا بولسا ئەمەلىيەتتە شۇ ۋاقىتتا جاھالەت ،خۇراپاتلىق ۋە چىرىكلىكتىن ئىبارەت ئۈچ چوڭ تاغنىڭ ئاستىدا زۇلمەتلىك كۈن ئىلىۋاتقان يېزا دېھقانلىرىمىزنىڭ ئەھۋالى ئەينەن تەسۋىرلەنگەن .
پوۋىستتا ياسىن مەدەكتىن ئىبارەت بىر توي قىلغان ، بالىسى بار، قىمارۋاز بىر ئوبراز نۇسرەتنىڭ ياخشى كۆرگىنى قىلىپ تەسۋىرلەنگەن بولسا  .
فىلىمدە سىدىقتىن ئىبارەت بىر كەنىت سىكىرتارى نۇسرەتنىڭ ياخشى كۆرگىنى قىلىپ تەسۋىرلەنگەن ،
بۇ يەردە ئەختەم ئۆمەر ئاكىنىڭ چىن مۇھەببەت تەسۋىرى ھېسىياتىغا يەنە بىر قىتىم زەربە بەرگەن ،ئەختەم ئۆمەر ئاكىنىڭ تەسۋىرلىمەكچى بولغىنى چىن مۇھەببەت ھەرگىزمۇ پۇل ،ئورۇن ،مەرتىۋە ياش سۈرۈشتىرىدىغان ساختىلىق بولماستىن ھەقىقى چىنلىق ،ساختىلىق ئارىلاشتۇرۇلمىغان گەرچە قوپال چاقچاق ،ئىسكەتسىز ھەل بىرىلمىگەن چىن يېزا تۇرمۇشى چىن مۇھەببەت ئىدى ،ياسىن مەدەك گەرچە بىر توي قىلىپ بالىسى بار ،ئارىلاپ قىمار ئوينايدىغان ،سوتوشۇپ جان باقىدىغان ئادەتتىكى سەھرا يىگىتى بولسىمۇ نۇسرەت ئۇنىڭغا كۆڭۈل بەرگەن ،ئۇنىڭ بىلەن شېرىن ۋەدىلەر بىرىشكەن مانا بۇ پوۋىستتىكى ۋەقەلىكلەر .
فىلىمدە بولسا سىدىقتىن ئىبارەت بىر كەنت سىكىرتارى نۇسرەتنىڭ ياخشى كۆرگىنى قىلىپ تەسۋىرلەنگەن ،بۇ ئادەمگە زورمۇزۇ تىڭىپ قويغاندەك تۇيغۇ بىرىدۇ ،ئەمەلىيەتنى چىقىش قىلىپ ئويلاپ باقساق بولىدۇ ئادەتتىكى يېزا قىزنىڭ بىر كەنىت سىكىرتارىنى ياخشى كۆرىشى ،ئۇنىڭ بىلەن شۇ يېزىدا مۇھەببەت ئىزھار قىلىپ يۈرىسى مۇمكىنمۇ ؟بەلكىم مۇمكىن بوپ قالار .لېكىن پوۋىستتا تەسۋىرلەنگەن ھالەت بويىچە كەنت يېزا كادىرلىرى سۇۋان مىراب ،مامۇت نۇر ،سىدىق بايچەكلەر ھەرگىزمۇ فىلىمدە كۆرسىتىلگەندەك خەلقنى ئويلايدىغان كادىرلار بولماستىن ئۆز نەپسىگە چوغ تارتىدىغان ،پارتىيىمىزنىڭ شەنىگە داغ تەككۈزىدىغان مىتە قۇرتلار خالاس ،ھەم ئۇيغۇرلار ئىشلىگەن فىلىم يازغان ئەسەرلەردە بولمىسۇن ھىچقاچان كەنىت سىكىرتارى دېگەنلەر بىر ئاددى يېزا قىزى بىلەن مۇھەببەتلىشىپ تارتىشىپ سۆرىشىپ يۈرگەن ئەمەس بىر بولسا ئىرىنى ھاشارغا ھەيدەپ ئايالىغا چىقىلغان بىربولسا مەجبۇرلاش ۋاستىسى بىلەن خوتۇنلۇققا ئالغان .
چۈشىنەلمىگىنىم سىدىق نېمىشقا ئادەتتىكى دېھقان بالىسى قىلىپ تەسۋىرلەنمەستىن كەنىت سىكىرتارى سۈپىتىدە ئوتتۇرىغا چىقىرىلىدۇ ،ئابدۇرىشىت ئابلىزدىن ئىبارەت بۇ سىدىقنىڭ رولىنى ئالغان ئاكىمىز بىر كەنىت سىكىرتارى سۈپىتىدە ئەمەس ،ئوتلۇق دېھقان يىگىتى سۈپىتىدە ئوتتۇرىغا چىققان بولسا ئۆزىنىڭ چىراي سالاپىتىگە بەك ياراشقان بولاتتى ھە!؟بىر ئاددى دېھقان يىگىتى نۇسرەتنى قوغدىيالماسمۇ ؟ئۇپارتىيە ئەزاسى بولمايمۇ ھەقىقەتنى قوغدىيالماسمۇ ؟
پوۋىسىتتا سىدىق بايچەك بىلەن ياسىن مەدەكنىڭ جىدىلىدىن كېيىن ،ئۆزى ياخشى كۆرمەيدىغان ئاجرىشىش ئالدىدا تۇرغان ئايالنىڭ ئىپپىتى ئۆزىنىڭ غورۇرى ئۈچۈن گەرچە ئەخمەقلىق بولسىمۇ ياسىن مەدەكنىڭ ئىسىلىپ ئۆلىۋىلىشى بىلەن ،نۇسرەتنىڭ قورسىقى يوغىناپ قالغان چاغدىكى سىدىقنىڭ ئۆز ئابرويى ئورنى ئۈچۈن ئۆزى ياخشى كۆرگەن نۇسرەتنى ئۆز ھالىغا تاشلاپ قويىشى ،ئۇنىڭ پاكلىقىغا ئىشەنمەي گۇمانسىرىشى ،ھەتتا ئەڭ ئاخىرىدا دوختۇرخانىدا تەكشۈتكەندىن كېيىنلا ئاندىن توي قىلايلى دېيىشى ،بۇنى ياسىن مەدەكنىڭ ئۆلىمىگە سىلىشتۇرغاندا سىدىقنىڭ ئوبرازى بەكلا خۈنۈكلەشتۈرىۋىتىلگەن .
فىلىمنىڭ نامى ئەختەم ئۆمەرنىڭ <<قۇرتلاپ كەتكەن كۆل >>نامى بۇ فىلىمدا <<ئەڭ ئاخىرقى كۆل >>دەپ ئۆزگەرتىلگەن ئويلاپ باقىدىغان بولساق باپكالار كۆلى ئەڭ ئاخىرقى كۆل بولالىدىمۇ؟ياق .فىلىمدە تەسۋىرلەنگىنى بويىچە 90-يىللاردا ئەمەس ھەتتا ئۇنىڭدىن يىگىرمە نەچچە يىل ئۆتكەن ھازىرمۇ جەنۇبى شىنجاڭنىڭ خىلى كۆپ يەرلىرى <<ئەڭ ئاخىرقى كۆل >>دىن قۇتۇلغىنى يوق .ئاددىسى مەن تەكلىماكاننىڭ چىتى ئەمەس قەشقەردىن 40كىلومىتىر يىراقلىقتىكى تۇغقانلىرىمنىڭ ئۆيىگە بىرىپ قالغىنىمدا ئۇلارنىڭ تاھازىرغىچە ئۆستەڭ سۈيى بىلەن ،كۆل سۈيى ئىچىۋاتقىنىنى كۆرىمەن ،تۇرۇبا سۈيى تېخى ئومۇميۈزلۈك ئومۇملاشقنىنى يوق ،گەرچە شەخسى ئائىلە قۇدۇقلىرى كولىغان بولسىمۇ بىرەر يىل ئۆتمەيلا قۇدۇق سۈيىنىڭ تارتىلىپ كېتىشى ئاچچىق بولۇپ قىلىشى سەۋەبلىك يەنە شۇ ئۆستەڭ بىلەن كۆل سۈيى ئۇلارنىڭ ئىچىملىك سۈيىدۇر ،ئۆستەڭلەرنى قاتتىقلاشتۇرۇلۇپ ،تەرەپ-تەرەپتىن  قۇدۇق كولىنىپ تىرىلغۇ سۇ مەسلىسى ھەل قىلىنغان بولسىمۇ ،يەر ئاستى سۈيىنىڭ تۆۋەنلەپ كېتىشى ،يەرنىڭ شورلىشىپ كېتىشى ،كىچىك قۇدۇقلاردىن سۇ چىقماسلىقتەك ئەھۋاللار كۆرۈلمەكتە .توغراق ،يۇلغۇنلارنى بۇزۇپ يەرئىچىش ،نوپۇس يەرنىڭ كۆپىيىشى سەۋەبلىك كېلىپ چىقىۋاتقان قانلىق سۇ تالىشىش پاجىئەلىرى بىزنىڭ دىققىتىمىزنى تارتماي تۇرالمايدۇ .
راستنى ئىيتقاندا <<قۇرتلاپ كەتكەن كۆل >>پاجىئەسى تېخى تۈگىگىنى يوق ،ئۇ كەڭ يېزىلىمىزدا تېخىمۇ ئەۋج ئالماقتا ،ئاياغقۇم كەنتىدەك سۇ قىس كەنتلەرلا ئەمەس ئىكولىگىيەلىك مۇھىتى ،تەبىئى شارائىتى ،ئىتىقاد چۈشەنچىلىرى ئاجىزلاۋاتقان ،بۇزۇلۇۋاتقان ،خارابلىشىقا يۈز تۇتىۋاتقان تالاي كەنت مەھەللىرىمىز<<قۇرتلاپ كەتكەن كۆل>>دىكى قۇرتلاردەك لۆمشىپ تەكلىماكان گىرۋەكلىرىدە مىغىلداپ يۈرمەكتە .ئەي <<قۇرتلاپ كەتكەن كۆل>>بىز سېنىڭ <<ئەڭ ئاخىرقى كۆل >>تىراگىدىيەڭدىن قاچانمۇ قۇتۇلارمىز .
