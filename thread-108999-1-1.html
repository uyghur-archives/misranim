ئەخمەد 16 ياشقا كىرگەن بولۇپ،ئۇنىڭ ئىككىى ئاكىسى ۋە بىر سىڭلىسى بار ئىدى.ئۇنىڭ ئاتا-ئانىسى بالىلىرىنى ھەقىقى ئىسلام پەرزەنتى قىلىپ چىقىش يولىدا بار تىرىشچانلىقىنى كۆرسىتىدىغان تەقۋا مۇسۇلمانلاردىن ئىدى.ئىسلامى تەربىيىنى ھەممىنىڭ ئالدىغا قوياتتى. ئۇلار مۇسۇلمان دوستلىرى ۋە باشقا كىشىلەرنىڭ ئەڭ ئاۋۋال بالىلىرىڭىزنىڭ دىنى تەربىيىسى ۋە ئاللاھنىڭ پەرزلىرىنى ئادا قىلىشى قانداق بولىۋاتىدۇ دەپ سوراشنىڭ ئورنىغا، بالىلىرىنىڭ مەكتەپتىكى ئەھۋالىنى سورىشىدىن نارازى بولاتتى.   
مىڭ ئەپسۇسكى ئەھمەد ئاتالمىش زامانىۋى تۇرمۇشقا- ئادەم قايمۇقتۇرىدىغان تۇرمۇش قاينىمىغا بېشىچە كىرىپ كەتتى.ئۇ دوستلىرى بىلەن كۈن- كۈنلەپ كىنوخانىدا ئولتۇرۇشقا، قىزلارنىڭ پارىڭىنى قىلىشقا رازى ئېدىكى، ناماز ئوقۇمايتتى.ئەھمەدنىڭ ئاتا ئانىسى ئۇنىڭ قىلىپ يۇرگەن ئىشلىرىنى بايقاپ،بىر كۈنى كەچقۇرۇن ئۇنى ئولتۇرغۇزۇپ ئىسلام ھەققىدە پاراڭلاشتى.   
-ئەھمەد،جېنىم ئوغلۇم،سەنمۇ بىلىسەن،مەن سىنى جېنىمدىنمۇ ئەزىز بىلىمەن.مەن سىنىڭ داداڭ،سىلەر مىنىڭ يۇرەك پارىلىرىم،مەن ھەمىشە سىلەرگە ياخشى بولسىكەن دەيمەن.--ئۇنىڭ دادىسى ناھايىتىمۇپەرىشان بىر ھالدا سۆزلىمەكتە ئىدى.   
-مەن بىلىمەن!--ئەھمەد سوغۇقلا جاۋاب بەردى   
-ئەھمەد،ئەگەر مەكتەپتە ئوقۇشتىن زىرىككەن بولساڭ،بولدى ئوقۇمىغىن،بىز تاكى ئاللاھنىڭ ھوزۇرىغا قايتقۇچە بىللە ياشايلى،بىللە ئىبادەت قىلايلى ئوغلۇم.   
-دادا!ماقۇل،ماقۇل!لىكىن مەن ئەمدى تىخى 16گە كىردىم،ۋاقىت دىگەن بارغۇ، مەن ۋەدە بىرەي،كەلگۈسىدە چوقۇم ياخشى ئادەم بولىمەن.   
-ئوغلۇم،سەن ئۆزۈڭنىڭ ئۆلۈم ۋاقتىڭنى بەلگىلىيەلەمسەن؟ئىگىز-پاكارلىقىڭ، ئاق-قارىلىقىڭ دىگەندەكلەرنى بەلگىلىيەلەمسەن؟-- سورىدى دادىسى.   
-ياق!.   
-ئەمىسە پۇتۇۋىتىلگەن ۋاقىتتا سىنىڭ ئىختىيارىڭسىز سىنى بۇ دۇنياغا ئاپىرىدە قىلغان ھەم ئاخىرىدا يەنە سىنىڭ ئىختىيارىڭسىز جىنىڭنى تىنىڭدىن جۇدا قىلىدىغان زاتچۇ؟ئۇ بەلگىلىيەلەمدۇ؟--دادىسى يەنە سورىدى.   
-ھەئە،مەن بىلىمەن دادا،لىكىن مەن ھازىر ئۆلۈپ كەتمەيدىغاندەكلا ھىس قىلىمەن.--   
-ئوغلۇم، خۇددى تۇغۇلىدىغان ۋاقىتتا ئۆزۈڭنىڭ تۇغۇلىدىغانلىقىڭنى بىلمىگەندەك،ئۆلۈپ كىتىشتىن بۇرۇنمۇ قاچان، قايسى ۋاقىتتا ئۆلۈپ كىتىشىڭنى بىلەلمەيسەن.   
-مەن ئاللاھنى ياخشى كۆرىمەن،مۇشۇنىڭ ئۈچۈن بولسىمۇ ئاللاھ مىنى كەچۈرىۋىتىدۇ!   
ئۇنىڭ ئانىسى يىغلاشقا باشلىدى ۋە بۇ پاراڭغا سۆز قوشتى:   
-سۆيۈملۈك ئوغلۇم،ئويلاپ باق، بىر بالا ئاتا ئانىسىنىڭ دىگىنىگە ئۇنىماي ئۇلارنى قاخشاتسا، ئۇ بالىنىڭ مەن سىلەرنى ياخشى كۆرىمەن دەپ قويغىنىغا ئاتا ئانىسى ئۇنىڭدىن رازى بوپ كىتەرمۇ؟   
-ياق!!   
-شۇنىڭدەك پۇتكۈل كائىناتنىڭ ياراتقۇچىسى بولغان ھەممىگە قادىر،بۈيۈك ئاللاھ بەندىسى بولغان بىزلەر ئۇنىڭ قانۇنىغا بويسۇنمىساق،ئاغزىمىزدا ياخشى كۆرىمەن دىگىنىمىزگىلا بىزدىن رازى بوپ كەتمەيدۇ-دېدى ئانا،-سەن چوقۇم ئاللاھقا ۋە پەيغەمبىرىمىزگە بۇيسۇنۇش ئارقىلىق ئاللاھقا بولغان مۇھەببىتىڭنى ئىپادىلىشىڭ كېرەك.   
-ئانا،ھايات نىمىشقا بۇنداق بولىدۇ؟نىمىشقا ئۇنى قىلساڭ بولمايدۇ بۇنى قىلساڭ بولمايدۇ،ئۇنى قىل بۇنى قىل دەيسىلەر؟مىنىڭ ھەممە ئىشتىن چەكلىنىپ ئۇنداق مەنىسىز تۇرمۇش كەچۈرگۈم يوق!-دەپ ۋاقىرىدى ئەھمەد.   
-ئەخمەد!ئۇ سىنىڭ ئاناڭ!ھۆرمەتسىزلىك قىلما!مۇھىمى سىنى ياراتقان،ساڭا سىنى ياخشى كۆرىدىغان مۇشۇنداق ئىسىل ئانىنى بەرگەن ئاللاھتىن قورققىن!ئىسلامدىمۇ خۇشاللىق ،ھۇزۇر-ھالاۋەت بار!--دېدى دادىسى كەسكىنلىك بىلەن.   
-ئاللاھ بەندىلىرىنى دوزاخقا كىرگۈزسە قانداقمۇ مھرىبان ئاتىلىدۇ؟-دېدى ئەھمەد قايتۇرۇپ.   
-سەن ھازىرمۇ ھايات ياشاۋاتقانسەن؟   
-ھەئە!- دىدى ئەھمەد ناھايىتى قوپاللىق بىلەن.   
-سەن مۇشۇنداق ئاسىيلارچە ياشاۋاتقان بولساڭمۇ ئاللاھنىڭ سىنى ھايات قالدۇرغىنى ئاللاھنىڭ مىرىبانلىقى بولماي نىمە؟-دېدى دادىسى قايتۇرۇپ.   
ئۇنىڭ ئاغزى تۇۋاقلىنىپ قالدى،ۋە ئورنىدىن تۇرۇپ ياتاق ئۆيىگە قاراپ ماڭدى.ۋە:«ھەممىڭلار بەك زىرىكىشلىككەنسىلەر!نىمىشقا بىز باشقا دىندىكىلەرگە ئوخشاش ھاياتنىڭ پەيزىنى سۈرەلمەيمىز؟ مەن ئۆزۇمنىڭ سالاھىيىتىمدىن نومۇس قىلىىۋاتىمەن!!»دەپ ۋارقىرىغىنىچە ھۇجرىسىغا كىرىپ كەتتى.   
بىچارە ئەھمەد ،بۇگۇنكى كۈننىڭ ئۆزىنىڭ ھاياتىنىڭ ئاخىرىدىن سانىغاندا 10-كۈنى ئىكەنلىكىنى ئازراق بولسىمۇ بىلسە ئىدى كاشكى……

ئەخمەد  ئۇيقۇغا غەرق بولغان ھالدا كارۋىتىدا ياتاتتى .  .يوتقاننىڭ تۇيۇقسىز پۇلاڭلىشىدىن چۈچۈپ ئويغىنىپ كەتتى . نامازپىشىم  ۋاقتى يېقىنلىشىپ قالغان ئىدى
ـ ئوغلۇم ،ئورنىڭدىن تۇرغىن ـ دادىسى ئەخمەدنى مىدىرلىتىپ  چاقىردى .
ـ ۋايجان ،نىمىشقا ؟ سائەت ئەمدى چۈش ۋاقتى بىر بولدىغۇ ؟ـ دەپ ۋارقىرغىنىچە ئۆزىنىڭ قوڭغۇراقلىق سائىتىگە قارىدى .
ـ چۈش ۋاقتى بىر ؟! ساقمۇ سەن ؟ ھەممە ئادەم ئورنىدىن تۇرۇپ بولدى ، خۇددى ئەتىگەن سائەت 6دە ئويغاتقاندەك گەپ قىلىسەن يا   ـ دېدى  دادىسى ئاچچىقلانغان ھالدا .
ـ   نىمىشقا مېنى ئويغۇتىسەن ؟ - ئەخمەد رەنجىپ سورىدى .
ـ بۇگۈن سەن مەن بىلەن مەسچىتكە بىرىپ ناماز ئوقۇيسەن ، سېنى ئۆمرىدە بىرەر قېتىم بولسىمۇ ناماز ئوقۇسا قانداق بولىدىغانلىقىنى بىلىپ باقسۇن دەيمەن .ـ دادىسى قەتئىيلىك بىلەن دېدى .
ئەخمەد تىركىشىپ يۇرۇپ ،مىڭ تەستە ئورنىدىن تۇردى .
ـ ئىككى مىنۇت ئىچىدە كارۋىتىڭدىن چۈشمىسەڭ ،مەن بىر ئىستىكان سوغۇق سۇنى ساڭا چىچىۋىتىمەن – دادىسى دەپ بولۇپ ئەخمەدنىڭ ياتاق ئۆيىدىن چىقىپ كەتتى .
ئەخمەد  تازا بىر ئاچچىغىنى چىقىرۋىلىش ئۈچۈن مۇنچىغا  يۇگۇردى .
20 مىنۇتۇتىن كېيىن ئەخمەد ۋە ئۇنىڭ دادىسى مەسچىت تەرەپكە قاراپ يول ئالدى .
ـ ئەخمەد ،مەن سېنىڭ ئاللاھ  ئالدىدا تۇرۇشقا نىمىشقا بۇنچىلىك قىيىقلىقىڭنى بىلەلمىدىم .
ـ  مەن ئۆيگە قايتىپ كومپۇيىتىر ئوينسام ياخشىراق ، 10سائەتلىك نامازغا تۇرۇپ بولالماي ئىچىم سىقىلىپ كىتىدۇ،ھىرىپ كىتىمەن .
ـ ئەخمەقمۇ سەن ؟ ئۇزاق بولغان تەقدىردىمۇ ،يىرىم سائەتكە بارمايدۇ ، سەن ھازىردىن باشلاپ ئاشۇ بولمىغۇر قىلىقلىرىڭنى تاشلىغىن ،
سېنىڭ قىلىمىشلىرىڭ خۇددى ماڭا قەۋزىيەت بولۇپ قالغان ئادەمنىڭ قىلىقىدەكلا تۇيۇلىدۇ .
ـ داد ،ئۇلار چوقۇم ئاڭلاپ قالدى ،سەن مېنى يەنە بىر قېتىم ئوڭايسىزلاندۇرماقچىمۇ ؟- ئەخمەد ۋاقىردى .
ـ سېنى ئوڭايسىزلاندۇردۇممۇ ؟! سېنىڭ قىلىمىشلىرىڭدىن مەن خىجىل بۇلىۋاتىمەن  ،  داداڭ................مۇشۇ يەرشۇ ،بىرەر قېتىم بولسىمۇ ئۆزۇڭگە پايدىلىق  ئىش قىلغىن .
ئەخمەد ۋە ئۇنىڭ دادىسى مەسچىتكە كىردى .دادىسى ئىنچىكىلىك بىلەن ئەخمەدنى كۈزەتتى ، ھەمدە مۇۋاپىق ئۇسۇلدا ئەخمەدكە تەرەت ئىلىشنى ئۈگىتىشكە تىرىشتى .
ئۇنىڭ تەرەت ئىلىشقا تۇتقان مۇئامىلىسى  خۇددى ئەتىگەندىن كەچكىچە ئىغىر ئەمگەك قىلىپ ھىرىپ كەتكەن ئادەمنىڭ مۇئامىلىسىگە ئوخشايتى .
ئۇلار ناماز ئوقۇيدىغان زالغا كىرگەندىن كېيىن ،ئالدىنقى سەپتىن ئورۇن ئالدى .مەزىن تەكبىر ئىيتىپ بولغاندىن كېيىن ھەممە ئادەم نامازغا تەييار بولدى .ھەمدە ئىمامغا ئەگىشىپ بىرلىكتە نامازنى باشلىۋەتتى .بۇ ئەخمەدنىڭ تۇنجى قېتىم ناماز ئوقۇشى ئىدى .
ناماز ۋاقتىدا ئەخمەدنىڭ سول تەرىپىدە تۇرغان دادىسى ئۇنىڭ يىغلىغان ئاۋازىنى ئاڭلىدى .  ئۇنىڭ ئوغلى يىغلاۋاتىدۇ  ! بىرخىل ھاياجانلىق ئىچىدىكى بەخت تۇيغۇسى دادىنىڭ بەدىنىنى سىيپاپ ئۆتكەندە ،ئۇنىڭ يۇزىدە تەبەسسۇم ،چۇنكى ئۇ ئوغلى بىلەن ئاللاھ ئالدىدا مۇرىنى مۇرىگە تىرەپ تۆۋەن باش ئىگىپ  تۇرماقتا .
نامازىدن كېيىن ئەخمەدنىڭ دادىسى ئۇنىڭغا قاراپ كۈلدى .
ـ ماشائاللاھ ،مېنىڭ ئەڭ سۇيۇملۇك ئوغلۇم ! يەنە ن ناماز ئوقۇشتىنمۇ سۆيۇملۇك نەرسە بارمىكەن   ؟! سەن تېخى يىغلاپ تۇرۇپ ئاللاھنى مەدىيلىدىڭ !

ـ نېمە ؟! سەن تېخى نادانكەنسەن ـ دادا ئۆزىنى ئاران تۇتۇۋالدى .
ـ ئوغلۇم ،بايا سەن ماڭا ئۆزۇڭنىڭ دۇستىڭنى ئۈمۈتسىزلەندۇرۇپ قويغانلىقىڭ ئۈچۈن يىغلىغانلىقىڭنى دىدڭ، ئۇنداقتا ئەگەر سەن سېنىڭ رەببىڭنى ئۈمۈتسىزلەندۇرۇپ قويساڭ ھىچ گەپ يوق دەپ ئويلامسەن ؟ سەندە ھېچقانداق تۇيغۇ بولمامدۇ ؟ ساڭا دىسەم ،ئاللاھ ساڭا ۋە ھەرقانداق ئادەمگە ھاجىتى بولمايدۇ ،شۇڭا سەن ئەڭ ياخشىسى ئۆز ھەركىتىڭ ئارقىلىق ئاللاھقا قۇلچىلىقىڭنى  ئىپادىلىگىن .چۇنكى سەن ئاللاھقا مۇھتاجسەن  .›› دېدى .ئاندىن ئاچچىقلانغان پىتى ئورنىدىن تۇرۇپ سىرىتقا چىقىپ بىر چوڭ   دەرەخنىڭ ئاستىغا كېلىپ  ئولتۇردى .ئۇنىڭ كۆزلىرى يەرگە تىكىلگەن بولۇپ ،كۆزىدىن ئاققان ياشلىرى يەرگە تامچە بولۇپ ئىقىپ تۇراتتى . بۇ كۆز تامچىلىرى  دادىنىڭ ئوغلىغا بولغان غەم قايغۇسى  ۋە سۆيگۇسى ئۈستىگە چۈشمەكتە .

ئەھمەدنىڭ دادىسى مۇھەممەد تاماق ئۈستىلى ئالدىدا چۆگۈندىن چىقىپ ھاۋاغا كۆتىرىلىۋاتقان ھوردىن كۆزىنى ئۈزمەي گەپ سۆزسىز ئولتۇراتتى.ئەھمەدنىڭ ئانىسى ئالىيە ئۇنىڭ قىشىغا كېلىپ ئولتۇردى.
-بالىلارنىڭ دادىسى،نىمىگە شۇنچە پەرىشان بوپ قالدىڭلار؟
-شۇ ئەھمەدنىڭ ئىشى بولمامدۇ،ئۇنى دەپ يۇرىكىم پۇچىلىنىپ كىتىۋاتىدۇ مەن نى-نى ئاچچىق كۈنلەرنى بىشىمدىن ئۆتكۈزگەن،ئەمما ئوغلۇمنى كۆڭلۈمدىكىدەك تەربىيلىيەلمىگىنىمدەك ماڭا ئېغىر كەلمىگەن ئىدى مەن گۇناھ ئىچىدە چوڭ بولدۇم،نۇرغۇن يىللار كۈنلىرىم ئاللاھنى تونىماي،ئۇنىڭغا ئىبادەت قىلماي ئۆتۈپ كەتتى،مىنىڭ ئىزچىل ئويلايدىغىنىم ئوغلۇمنى ئاللاھنىڭ يولىدا تەربىيلەپ ئاشۇ گۇمراھلىقتا ئۆتكەن كۈنلىرىمنىڭ ئورنىنى تولۇقلىسام،ئاللاھ ئالدىدا يۈزۈم يورۇق بولسا،ھېچكىم ھىچكىمغا ئەسقاتمايدىغان ئاشۇ دەھشىتى قىيامەتتە مانا مەن تەربىيلىگەن ئوغۇل دىيەلىسەم، ئۇلۇغ ئاللاھ بزلەرنى قول تۇتۇشۇپ ئەھلى جەننىتىلەر قاتارىدىن ئورۇن ئېلىشقا نىسىپ قىلسا،ھەي……لېكىن ئەمەلىيەتتىچۇ؟ئۇ مۇشۇ پىتى ئۆلۈپ كەتسە ھالى قانداق بولۇر؟مەن بۇ ئىچ ئاغرىقىدا قانداق ياشايمەن؟…مىنىڭ ئوغلۇم،يۈرەك پارەم!….-ئەھمەدنىڭ دادىسى بوغۇلۇپ يىغلاپ كەتتى، ئانىسىمۇ ئۆزىنى تۇتۇۋالالمىدى، ئەر-ئايال ئىككىسى بىر بىرىگە تەسەللىي بىرىپ ئۇزۇندىن ئۇزۇن يىغلاشتى..
ئىككى كۈن مانا مۇشۇنداق ئۆتۈپ كەتتى،ئەھمەنىڭ پەقەت 6كۈنىلا قالدى.
ئەھمەد قۇلىقىغا نائۇشنىك سېلىۋالغان بولۇپ مۇزىكا ئاڭلىغاچ كارىۋاتتا يىرىم ياتاتتى،كۆزى ۋە خىيالى بولسا قولىدىكى يىرىم يالىڭاچ چولپانلارنىڭ رەسىملىرى بىلەن تولغان ژورنالدا يۈگرەيتتى.
ۋاقىت ناھايىتىمۇ تىز ئۆتمەكتە ئىدى، مانا بىردەمدە قۇياشمۇ پاتتى،ئەھمەد كارۋاتتىن سەكرەپ چۈشۈپ،دوستلىرى بىلەن دىيىشىپ قويغان كەچلىك كۆرۈشۈشكە تەييارلاندى.ئۇ پەلەمپەيدىن چۈشىشىگە ئالدى ئىشىكتە تۇرغان ئانىسى ئۇنى توسىۋالدى..
-ئانا نېمە بولدى؟نېمە مىنى توسىۋالىسەن؟
-بۈگۈن سىنى چىققىلى قويمايمەن، قارا كەچ سائەت 9 بولدى،كەچلىك تاماقمۇ تەييار بولدى،بىرگە ئولتۇرۇپ كەچلىك تاماق يەيلى،گىپىمگە كىر بالام.
-نېمە؟!!مەن Mo بىلەن بۈگۈن كەچتە كۆرۈشىدىغانغا ۋەدە بىرىپ قويغان-دەپ ۋارقىرىدى ئەھمەد.
-ھازىر بەك كەچ بولۇپ كەتتى بالام،تاماقمۇ تەييار بولدى،بىرگە ئولتۇرۇپ تاماق يەيلى،قارا،ئاكىلىرىڭ سىڭلىڭ داداڭ ئىككىمىزنى پەقەت خاپا قىلمايدۇ،نىمىشقا سەنلا گىپىمىزگە كىرمەيسەن؟
-چۈنكى ئۇلار ياشاشنى بىلمەيدۇ!!
-بۇ گەپ ئاغزىڭغا قانداق كەلدى؟ئۇلار نامازلىرىنى جايىدا ئېلىپ مىڭىۋاتىدۇ،ھەممە ئىشلىرى شۇنداق جايىدا…ئۆزۈڭگە قاراپ باقە!قىلىۋاتقان ئىشىڭدىن ئۆزۈڭگە بىرەر نەپ تەگدىمۇ ؟ ئۆزۈڭنىڭ نېمە قىلىۋاتقىنىڭنى بىلەمسەن يوق؟
-بىر چەتكە تۇرە!!!
ئەھمەد ۋاقىرىغىنىچە ئاپىسى تەرەپكە بۇرۇلدى ۋە جەينىكى بىلەن ئىتتىرىۋىتىپ ئىشىكتىن ئوقتەك ئىتىلىپ چىقىپ كەتتى.ئانىسى سەنتۈرۈلۈپ بىرىپ تامغا سوقۇلدى ۋە يىقىلىپ چۈشتى.
-يا،ئۇلۇغ ئاللاھ!ئوغلۇمنى كەچۈرگىن،ئوغلۇمنى كەچۈرگىن!-ئانىسى ئاۋازىنى قويۇۋىتىپ يىغلاپ كەتتى.ئەھمەدنىڭ دادىسى يۈگۈرۈپ كېلىپ ئايالىنى يۆلىۋالدى،ئەھمەدكە بىر نېمە دەي دەپ بولغىچە ئەھمەد ئاللىبۇرۇن پەلەمپەيدىن چۈشۈپ كەتكەنىدى.دادىسى ئايالىنى يۆلەپ ئولتۇرغۇزدى، يىقىلغاندىكى ئاغرىق ئازابى ۋە ئوغلىنىڭ قىلغانلىرى قوشۇلۇپ ئانىنى ھالسىزلاندۇرىۋەتكەن ئىدى.توختىماي يىغلايتتى.
مۇھەمەد ئايالى ئالىيەنىڭ كۆزلىرىگە قارىدى،كۆز ياشلىرى يىپى ئۈزۈلگەن مارجاندەك تۆكۈلمەكتە،كۆزلىرى بولسا قىزىرىپ ئىششىپ كەتكەن:«بۇ نېمە ئۈچۈن؟زادى نېمە ئۈچۈن؟…..مەن سىنىڭ رازىلىقىڭ ئۈچۈن ئوغلۇمنى قۇتقۇزۇش ئۈچۈن جىنىمدىن كەچسەممۇ رازىمەن ئى ئاللاھ!ئۇ مىنىڭ بىر قىسمىم،مىنىڭ ئامراق ئوغلۇم.ئۇنىڭ بۇ قىلغانلىرى يۇرۇكۇمنى سۇغۇرىۋالغاندىنمۇ ئېشىپ كەتتى….مەن ئۇنى ياخشى كۈرىمەن ئاللاھىم! ئۇنى ھىدايەتكە باشلىغىن!...»دەپ يامغۇر يىغلاپ ئاللاھقا ئىلتىجا قىلماقتا ئىدى.
ئەھمەدنىڭ دادىسىمۇ چىدىيالماي قالدى،يۇرىكى پارە-پارە بولۇپ كەتكەندەك بىئارامچلىق ئىچىدە يۇم-يۇم يىغلايتتى.
ئەھمەد ئانىغا ئازار يەتكۈزۈشنىڭ نەقەدەر ئېغىر ئىش ئىكەنلىكىنى ئاللاھنىڭ جازاسىدىن قۇتۇلالمايدىغانلىقىنى ئازراق بولسىمۇ ئويلىسا ئىدى كاشكى…..

ئەھمەد ئۆيىدىن چىقىپلا دوستى Mo تەرەپكە قاراپ يۈگۈردى.
-ھوي! ئەھمەد!نىمە قىلدىڭ؟ھەجەپ ئۇزۇندا چىقتىڭغۇ؟.
-يەنە ئاتا-ئانام بولمامدۇ، بۈگۈن يەنە دىن توغرۇلۇق سۆزلەپ قۇلقۇمنى پاڭ قىلىۋەتتى.
-ھا ھا!ئۇلارغا ھايات كېرەك ئەمەسمىدۇ دەيمەن،دىمەكچى بولغىنىم ئاللاھ دىگەنغۇ كەچۈرىمەن دەپ،بىز چىن كۆڭلىمىزدىن ئاللاھقا ئىشەنسەك،باشقىلارغا زىيانكەشلىك قىلمىساقلا ئاللاھ بىزنى جەننەتكە كىرگۈزىدۇ.ئەمەسمۇ-دېدى Mo بىلەرمەنلىك قىلىپ.
Mo بىلەن ئەھمەد يانداشقان ھالدا ئۇلار داۋاملىق بارىدىغان كەچلىك كۇلۇبقا قاراپ ماڭدى…
ئۇلار بىر قانچە مىنۇتلاردىن كېيىن ئوغرى تۇتىدىغان ساقچىلارنىڭ بۇ يەردىن ئۆتىدىغانلىقىنى بىلمەيتتى.(بىلەلمەيدۇ ئەلۋەتتە)
بەش ساقچى ماشىنىسى ئوغرىلانغان ماشىنىنىڭ كەينىدىن ئىز قوغلاپ كىتىۋاتاتتى، ئوغرىلانغان ماشىنا ئۇدۇل ئەھمەدلەر بار كوچىغا قاراپ ماڭدى،بارغانچە يىقىنلاۋاتقان ساقچى ماشىنىسىنىڭ سىگنالى چاقچاق بىلەن كىتىۋاتقان ئەھمەد بىلەن Moنىڭ دىققىتىنى تارتتى.
ئىككىسىدىن 100 مېتىر يىراقلىقتا كىلىۋاتقان ئوغرىلانغان ماشىنا 40mphتىزلىك بىلەن كىلىۋاتاتتى،تۇيۇقسىز كونتروللىقىنى يوقىتىپ ياندىكى توختىتىپ قويۇلغان ماشىنىغا ئۈسسىدى ۋە ئۇنى تۈرتكىنىچە ئەھمەد ۋە Moبار يولغا ئاپاردى…
ئەھمەد دەماللىققا نېمە بولغىنىنى بىلمەي تۇرۇپ قالدى، Moبولسا چىقىرىغىنىچە كەينىگە بۇرۇلدى،ماشىنا خۇددى ئەھمەدنى سوقماقچىدەكلا يانغا بىر سىيرىلدى ۋە كۈچ بىلەن سوقۇپ نەچچە مېتىر ئارىلىققا ئېتىۋەتتى.ئۇ ئۇتتۇر بىرىپ بىر قىش دۆۋىسىگە چۈشتى.ۋە ھوشىدىن كەتتى Moۋارقىرىغىنىچە دوستىنىڭ قېشىغا يۈگۈرۈپ كەلدى.
-ۋاي خۇدايىم!نىمە ئىش بۇ!ماڭا ئۆلۈپ بەرمە ئادەم قورقۇتۇپ!- Moقورققىنىدىن ۋارقىراپ كەتتى ۋە ئەھمەدنى سىلكىشكە باشلىدى.
ساقچىلار يىتىپ كەلدى ۋە Moنىڭ ئۇنىڭ ھۇشسىز بەدىنىنى سىقىۋىرىپ يارىسنى تېخىمۇ ئىغىرلاشتۇرىۋىتىشىدىن ئەنسىرەپ تارتىپ بىر ياققا ئەكەتتى.
بىر قىسىم ساقچىلار بىرىپ ئوغرىلارنى تۇتتى ۋە بىر قىسىملىرى بولسا ئەھمەدنىڭ يىنىدا پايپىتەك بولۇشتى.
Mo قورقۇنچ ئىچىدە ئەھمەدنىڭ ئۆيىگە قاراپ چاپتى.ۋە ئۆيىنىڭ ئالدىغا كېلىپ ئىشىكنى ساراڭلارچە ئۇرۇشقا باشلىدى.
ئەھمەدنىڭ دادىسى ئەنسىزلىك بىلەن ئىشىكنى ئاچتى.
-Mo ؟نېمە ئىش بولدى؟!-ئەھمەدنىڭ دادىسى ئەنسىرەپ سورىدى.
-ئەھمەد…ئەھمەد ئۇ قاتناش ۋەقەسىگە ئۇچرىدى!- Moھاسىراپ ھۆمۈدەپ تۇرۇپ ئاران دېدى.
-مىنى باشلاپ بار!تىز بول!-ئەھمەدنىڭ دادىسى شۇنداق دىگەچ ئايىغىنى كەيدى،ئەھمەدنىڭ ئانىسىمۇ ئەگەشتى.
ئەھمەدنىڭ دادىسى بىلەن Moئەھمەدنىڭ قىشىغا كەلدى،ئۇنىڭ ئەتراپىنى بىر مۇنچە دوختۇرلار ئورىۋالغانىدى.
ئەھمەدنىڭ دادىسى بىر دوختۇردى سورىدى:
-ئۇ قانداقراق؟يارىسى ئېغىرمۇ؟
-ئۇ تېخى ھوشىغا كەلمىدى،مۈرە سۆڭىكى ئاجراپ كىتىپتۇ.ھازىر چوقۇم جىدىى ھالدا دوختۇرخانىغا ئاپىرىپ مىڭىسىدە قان چۈشۈش ياكى باشقا زەخمە ئەھۋالى بارمۇ-يوق تەكشۈرىشىمىز كېرەك.
دادىسى ئىختىيارسىز يىغلىۋەتتى…
ئەھمەدنىڭ ئاتا-ئانىسى جىددى قۇتقۇزۇش ماشىنىسنىڭ قىشىغا كەلدى
-مەنمۇ بارايچۇ!-دىدى Moئەھمەدنىڭ دادىسىغا.
-بولدى،ئۆيۈڭگە قايتىپ بۈگۈن يۈز بەرگەن ئىشلارنى ۋەنىمىشقا يۈز بەرگەنلىكىنى ئوبدان ئويلاپ باققىن..
جىددى قۇتقۇزۇش ماشىنىسىنىڭ ئىشىكى سېسترا تەرىپىدىن يىپىۋىتىلدى.ۋە تىز سۈرئەتتە ئەھمەد ۋە ئاتا ئانىسىنى دوختۇرخانىغا ئېلىپ ماڭدى.
Moبارغانچە يىراقلىشىۋاتقان جىددى قۇتقۇزۇش ماشىنىسىغا قارىغىنىچە مىدىر سىدىر قىلماي تۇراتتى.بىر ياندا بولسا ساقچىلار ۋەقە يۈز بەرگەن نەخ مەيداننى تەكشۈرىۋاتاتتى.
Moيانغا بۇرۇلدى، ئۆيىگە ئەمەس بەلكى ئەھمەد بىلەن بارماقچى بولغان كىچىلىك كۇلۇب تەرەپكە بۇرۇلدى.
«نېمىشقا ئۆيگە قايتىپ ئازابلىنىپ ئولتۇرغۇدەكمەن؟ھىچكىممۇ مىنىڭ بىچارە ئەھمەد بىلەن بىرگە ھوزۇرلانماقچى بولغان خۇشال كىچەمنى يوق قىلىمەن دەپ خام-خىيال قىلمىسۇن..» Moشۇنداق ئويلىغىنىچە كىچىلىك كۇلۇبقا قاراپ ماڭدى…

ئەھمەدنىڭ ئاتا ئانىسى كىچىچە ئەھمەدنىڭ ئەتراپىدا پەرۋانە بولۇپ ،ئۇنىڭ ھوشىغا كىلىشىنى ساقلىدى.
تۇيۇقسىز ئەھمەدنىڭ كۆز قارچۇغى مىدىرلاشقا باشلىدى،ۋە ئاستاغىنە كۆزىنى ئاچتى.
-ئاللاھۇ-ئەكبەر!!ئوغلۇم!ئاخىرى ھوشۇڭغا كەلدىڭ،قانداقراق بىر يەرلىرىڭ ئاغرىۋاتامدۇ؟-سورىدى ئانا ئۇنىڭ قوللىرىنى چىڭ تۇتۇپ ۋە يەنە ئىسەدەپ يىغلاشقا باشلىدى.
-ماشىنا ماڭا قاراپ كەلگەندە…مەن …ئەمدى…ئۆلۈپ كىتىدىغان بولدۇم دەپ ئويلىغانتىم…-دېدى ئەھمەد زەئىپ  ئاۋازدا.-بۇ يەردە تۇرغىنىڭلارغا قانچە ئۇزۇن بولدى؟
-11 سائەت بولدى،بىز بۇ يەردە توختىماي ناماز ئوقۇدۇق.سەن ئۈچۈن ئاللاھقا دۇئا قىلدۇق.-دېدى دادا ئۆزىنى تۇتىۋىلىپ.
«ئاتا-ئانام مەن ئۈچۈن ئاللاھقا دۇئا قىلىپ بىر كىچە ينىمدا تۇرۇپتۇ،مەنچۇ؟ ئۇلارنىڭ قىشىدا بىرەر لەھزە گۇناھسىز ھالەتتە تۇرۇپ باقماپتىمەن…»ئەھمەد خىيالغا چۆكۈشكە باشلىدى.
ئەھمەدنىڭ كۆزلىرى ياشقا تولدى،ئەھمەدنىڭ دادىسى ئۇنىڭ قىشىغا يېقىن كەلدى ۋە سورىدى:
-نىمىگە يىغلايسەن، ئەھمەد؟
-ھىم…ئۆزۈممۇ ئۇقمايمەن،بەلكىم ھايات قالغىنىمغا خۇشال بولىۋاتسام كېرەك-دەپ قويدى ئۇ ئويلاۋاتقانلرىنى دىيىشتىن خىجىل بولۇپ.
ئىككى كۈندىن كېيىن ئۇ بالىنىستىن چىقتى،ئۇنىڭ ئوڭ قولى بوينىغا ئېسىپ قويۇلغان بولۇپ،مۈرىسىمۇ تىڭىپ قويۇلغان ئىدى.ئۇ ئەمدى ئارام ئېلش ئۈچۈن يىتىپ تۇرىۋىدى، Moدىن تېلېفون كەپ قالدى.
-ھوي ئەھمەد!بالىنىستىن چىققىنىڭنى ئاڭلاپلا تېلېفون قىلىشىم.قانداقراق تۇرىسەن؟

-قارا ئەھمەد،ھىلىقى كۈنى بۇ ئىش بولۇپ بارالمىدىڭ،بۈگۈن كەچ كىلەلەمسەن؟ سەن كىلەلمەي بەك ئاچچىقىم كەلگەنتى،ئۇ كۈنى مەن مۇڭداشقان قىزلارنى كۆرگەن بولساڭ،ھەي…
-Mo،راستىمنى ئىيتسام بارغۇم يوق،دەپ باقە بۇ ئاللاھنىڭ ئاگاھلاندۇرىشىمىدۇ؟
-ئاللاھنىڭ ئاگاھلاندۇرۇشى؟؟سەن ساقمۇ؟سەنمۇ ئۇلارغا ئوخشاش رادىكال بوپ كەتمىگەنسەن بىر دەمدە؟سەنمۇ دېگەن!ئۇنداق ھاياتنىڭ قىلچە مەنىسى يوق!
-ئۇلارنىڭ ھازىرقى ھاياتىنى مەنىسىزمۇ دەيلى،بىراق ئۇلار ئۆلگەندىن كېيىن ھەممىگە ئىرىشەلەيدۇ.
-نىمىلەرنى دەيدىغانسەن،ئەھمەد!بىز ھەممىمىز مۇسۇلمان!بىز جەننەتكە كىرىمىز.بوپتۇ دوزاخقا كىردۇقمۇ دەيلى،لېكىن بىر مەزگىلدىن كېيىن ئوخشاشلا يەنە جەننەتكە قايتىپ چىقىمىز،شۇنداق تۇرسا يەنە نىمىگە ئەنسىرەيسەن؟ھەم ئاللاھ بەكمۇ مېھرىبان،بىز ئۇنىڭغا ھەقىقى ئىشەنسەكلا بىزنىڭ گۇناھلىمىزنى كەچۈرىۋىتىدۇ-دېدى بىچارە Mo يەنە بىلەرمەنلىك قىلىپ.
-ھىم…مەن بىلمەيمەن،مەن يەنە ئويلاپ باقاي.بۈگۈن ئىنىقلا بارالمايمەن،بىلىكىم تېخى پۈتۈنلەي ساقىيىپ كەتمىدى.ئەتە ساڭا جاۋابىنى بىرەي بولامدۇ؟
-ماقۇل،شۇنداق بولسۇن ئەمىسە.تېلفۇن قىلىشنى ئۇنۇتما-ھە!ھازىر مۇشۇ تاپتا بۇ زىرىكىشلىك ئۆيدە دادامنىڭ قۇرئان ئوقۇغىنىنى ئاڭلاپ ئولتۇرىۋاتىمەن،ئۇلارنىڭ تۇرمۇشى نىمانچە مەززىسىزدۇ؟شۇنداق تۇرۇپ يەنە كۈلەلىگىنىگە ھەيرانمەن.
ئەھمەد تېلفۇننى قويىۋەتكەندىن كېيىن شۇنداق چۈچۈپ كەتتى،بۇ ئۇنىڭ تۇنجى قېتىم ئاللاھتىن قورقۇش تۇيغۇسىدا بولۇشى ئىدى.ئاللاھ نېمە قىلىمەن دىسە قىلالايدۇ،ئۇنى بىر قېتىم ئۆلۈم گىرداۋىدىن  قۇتۇلدۇردى…، ياق!.. ياق! -ئەھمەدنىڭ كۆزىدىن يەنە ياش ئېقىشقا باشىلىدى.
-ئاللاھ مىنى ئەمدى قانداقمۇ ياخشى كۆرسۇن؟مەن نىمىلەرنى قىلدىم؟بىرەر سىكۇنتمۇ پۇتۇن دىققىتىم بىلەن ئاللاھقا يۈزلىنىپ باقماپتىمەن، ئانامنى يارىلاندۇردۇم،ئەمما ئۇ يەنە ئىزچىل مىنى ياخشى كۆرىدۇ.ئاللاھ مەندەك ئازغۇننى قانداقمۇ كەچۈرسۇن…؟-ئەھمەد ئاچچىق ئويلارغا پاتقان ئىدى.
يەنە بىر كىچە ئۆتۈپ كەتتى،ئۇنىڭ ئۆلۈم كۈنىگە پەقەت 4 كۈنلا قالدى.
ئەتىسى بىلىكىنىڭ ئاغرىشى يەنە قويۇپ بەرمىدى،ئۇ Moنىڭ كىچىلىك كۇلۇپقا بىرىش تەكلىپىنى رەت قىلدى، ئەھمەدنىڭ كۈنلىرى ھايات توغرىسىدا ئويلاش بىلەنلا ئۆتمەكتە ئىدى…
كۈنلەر ئۆتۈپ مانا ئەھمەدنىڭ ئەڭ ئاخىرقى بىر كۈنى كەپ قالدى.
ھىچكىممۇ ئۇنىڭ بۈگۈن بۇدۇنيا بىلەن خوشلىشىدىغانلىقىنى ئۇقمايتتى،ھەتتا پەرىشتىلەرمۇ…
ئۇ كۇن دەل پەيشەنبە ئىدى،سەھەرگە يېقىن ئەھمەد بىر قورقۇنچلىق چۈشتىن چۈچۈپ ئويغاندى ،قولىنىڭ ئاغرىقى سەللىمازا قويۇپ بەرگەن ئىدى.
ئەھمەھ ئاستاغىنا پەسكە چۈشتى ۋە تاماق ئۈستىلىنىڭ ئالدىغا كېلىپ ئولتۇردى، شىرە ئۈستىدىكى نەرسىلەرنى يۇغۇشتۇراي دەپ، دادىسىنىڭ ھۆججەتلىرنى ئىشكاپكا سىلىش ئۈچۈن مىڭىۋاتقاندا،بىر خاتىرە دەپتەر  سوغۇرۇلۇپ يەرگە چۈشۈپ كەتتى.ئۇ ھەممە نەرسىلەرنى جايىغا سىلىۋىتىپ بولۇپ قىزىقىش ئىچىدە ئۇ خاتىرىنى ئاچتى.
خاتىرىنىڭ تۇنجى بېتىگە  قۇرئاندىكى تۆۋەندىكى ئايەت يېزىلغان ئىدى:
(ئى مۇھەممەد!مىنىڭ تىلىمدىن)ئېيتقىنكى، «(گۇناھلارنى قىلىۋىرىپ)ئۆزلىرىگە جىنايەت قىلغان بەندىلىرىم!ئاللاھنىڭ رەھمىتىدىن ئۈمىدسىزلەنمەڭلار.ئاللاھ ھەقىقەتەن (خالىغان ئادەمنىڭ)جىمى گۇناھلىرىنى مەغپىرەت قىلىدۇ، شۈبھىسىزكى، ئاللاھ ناھايىتى مەغپىرەت قىلغۇچىدۇر،ناھايىتى مىھرىباندۇر.»
ئەھمەد تۇيۇقسىز بۇقۇلداپ يىغلاپ كەتتى.ئۇنىڭ قوللىرى لاغ لاغ تىترەپ، كۆز ئالدى غۇۋالىشىشىقا باشلىدى، ۋە تىزلىكتە خاتىرىنىڭ يەنە بىر بىتىنى ئۆرىدى:«مەن بارغانچە ئاللاھتىن يىراقلىشىپ ،بىلىپ بىلمەي گۇناھقا پىتىپ كەتكەنىدىم،ئاللاھ ماڭا ئۆزىنىڭ نۇرلۇق يولىنى كۆرسەتتى،ئىككى ئوغلۇم تۇغۇلغاندىن كېيىن مەن يەنە ئازدىم،ئاللاھنى ئۇنتۇدۇم،…»ئۇ يەنە بەت ئۆرىدى ۋە ئاجايىپ ھەيرانلىق ئىچىدە ئوقۇشقا باشلىدى:«مىنىڭ كىچىك ئوغلۇم ئەھمەد،  ئۈچىنچى ئوغلۇم،ھازىر 5 ياشقا كىردى،دەل ئۇ مىنى قايتىدىن ئاللاھ تەرەپكە قايتۇردى.مەن ئۆيۈمدە بىر تېلېۋىزىيە فىلىمى كۆرۈپ ئولتۇراتتىم،ئوغلۇم كىرىپ كەلدى ۋە :«دادا ئەگەر ھازىر بىر شەيخ ئويگە كىرسە قانداق قىلاتتىڭ؟»دېدى،مەن جىق ئويلانمايلا:«دەرھال تېلىۋىزۇرنى ئۆچۈرىۋىتەتتىم!»دەپ جاۋاب بەردىم.ئەھمەد:«لېكىن دادا!ئاللاھ ھەمىشە سېنى كۆرۈپ تۇرىدۇ ،ئاللاھ ھەم دۇنيادىكى ھەرقانداق شەيختىنمۇ ئۇلۇغ ئەمەسمۇ؟» مىنىڭ زۇۋانىم تۇتۇلۇپلا قالدى. مەن ئۆزۈمدىن نومۇس قىلىشقا باشلىدىم،مەن يەنە قايتىدىن ئاللاھ تەرەپكە قايتتىم،ئاللاھقا قىلغان قەسىمىمدە چىڭ تۇردۇم،ئايالىم ئالىيەمۇ ھەم ….»ئەھمەد داۋاملىق ئوقۇماقتا ئىدى…«ئوغلۇم ئەھمەد بىر مۇنچە ناتوغرا يولدىكى بالىلار بىلەن ئارىلىشىپ يۈرىدۇ، مەن ئۇنىڭ بارغانچە دىندىن يىراقلىشىپ كىتىۋاتقانلىقىنى سىزىۋاتىمەن. مەن ھەقىقەتەن ئامالسىز قالدىم.ئۇ ئاللاھنىڭ .ئىزنى بىلەن مىنى ئاللاھقا قايتۇرغان ئىدى،ئەمدى ئۇنى قانداق ئۆزگەرتىشىم كېرەك؟ قانداق قىلسام ئۇنى ئۆزگەرتەلەيمەن؟…»
ئەھمەد خاتىرىنى ياپتى،كۆز ياشلىرى يۈزلىرىنى يۇيماقتا ئىدى،ئۇ قولىغا قەلەم ئالدى:«دادا! مەن بۇ خاتىرەڭنى ئوقۇدۇم.مەن كىچىك چىغىمدا (ئاللاھنىڭ ئىزنى بىلەن) بىر سوئالىم ئارقىلىق سىنىڭ ئۆزگىرىشىڭگە سەۋەب بولغان ئىكەنمەن.سىنىڭ خاتىرەڭدىكى بۇ سۆزلەر ئاللاھنىڭ ئىزنى ۋە ئىرادىسى بىلەن مىنىڭ ئاللاھقا قايتىشىمغا سەۋەب بولدى.»ئۇ خاتىرىنىڭ ئەمدى ئوقۇپ بولغان ئاخىرقى بىتىگە بۇ خەتلەرنى يازدى..
ئەھمەد ئورنىدىن تۇردى ۋە تۇنجى قېتىم ئۆزى خالاپ تاھارەت ئالدى.كىيىملىرىنى ، ئايىقىنى كىيىپ مەسچىتكە قاراپ ماڭدى.
ئىنتايىن ئەپسۇسكى ئەھمەد دوستى  Moنىڭمۇ مۇشۇ يولدا كىتىۋاتقانلىقىنى بىلمەيتتى! Moيەنە ئەھمەدنى ئاشۇ تۇيۇق يولغا قايتۇرىۋىتەمدۇ يوق؟ياكى ئەھمەد دوستى Moنى ھەق يولغا قايتۇرۇپ كىلەمدۇ؟ئەھمەد ھازىر تاللىغان يولىدا مەزمۇت تۇرالامدۇ؟

.ئەھمەد بىر خىل پاك تۇيغۇلارغا چۈمۈلمەكتە،ئۇباشقىچە خۇشال ئىدى،بۇ خۇشاللىق پەقەت ئاللاھ ياخشى كۆرىدىغان يولدا مۇستەھكەم مىڭىپ،ئاللاھنى خۇشال قىلىدىغان ئەمەللەرنى قىلغاندىلا ھېس قىلغىلى بولىدىغان مەڭگۈلۈك خۇشاللىق ئىدى...
ئەھمەد يېقىملىق ئاۋازدا سايراۋاتقان،ئەمەلىيەتتە ئاللاھ ئاتا قىلغان.ۋە ئەڭ ياخشى بىلىدىغان يوسۇندا ئاللاھنى زىكرى قىلىۋاتقان قۇشلارنىڭ مۇڭلۇق سادالىرى ئارىسىدا خىيال بىلەن كىتىۋاتاتتى
كۈتۈلمىگەندە بىر بۇلۇڭدىن Mo چىقىپ كەلدى.ۋە ئەھمەدنى كۆرۈپ ۋارقىرىۋەتتى.
-ۋوي ئەھمەد!سىنى قايتا كۆرگىنىمدىن خۇشالمەن.
ئەھمەد كۈلۈمسىرىدى.
-نەگە ماڭغانتىڭ ؟-سورىدى Mo
-ھىم…،راستىمنى ئىيتسام Mo ،مەسجىدكە ماڭغانتىم.
-نېمە؟؟؟مەسجىدكە؟؟سەن ساراڭ بوپ قالمىغانسەن؟
-گىپىمنى ئاڭلا Mo،سەن كاللاڭنى سىلكىۋىتىپ،ئويلىنىپ باقمىغىچە چۈشەنمەيدىكەنسەن.
-سىنىڭ كاللاڭدىن ئۆتۈپ بولدى؟!
-ئىشقىلىپ مەن ئىزدىنىۋاتىمەن.
-ۋاھ!ساڭا دەپ قوياي،سەن بۇنىڭدىن كېيىن بۇرۇختۇملۇقتا ياشايسەن ئەھمەد،ئەمما قىيامەت كۈنى مەنمۇ ئوخشاشلا جەننەتكە كىرىمەن،سەن شۇ چاغدا نىمىشقا ئوينىۋالمىغان بولغىيتىم دەپ پۇشايمان قىلىسەن.
-Moئۆزۈڭنىڭ جەننەتكە كىرىدىغانلىقىڭغا شۇنداق ئىشىنەمسەن؟
-ئەلۋەتتە!بۇنى مەن ئەمەس،ئاللاھ ۋە ئۇنىڭ پەيغەمبىرى دېگەن،.
-نىمىلا دىمەيلى،بىز ھازىر تىخى ھەدىسنى چۈشىنەلەيدىغان  ۋەباشقىلارغا چۈشەندۈرەلەيدىغان ھالەتتە ئەمەس Mo،سەن راستىنلا جەننەت ئەھلىلىرى كېيىن گۇناھ ئىش قىلمىغىنىغا پۇشايمان قىلىدۇ دەپ ئويلامسەن؟ئەمەلىيەتتە پۈتۈنلەي ئەكىسچە.
-ئىددىيەڭنىڭ بۇنداق تىز ئۆزگىرىپ كەتكىنىگە ئىشىنەلمەي قىلىۋاتىمەن،ئەھمەد،ئەمما سىنىڭ ئەسلىڭگە قايتىدىغانلىقىڭغا ئىشىنىمەن،ھەم قاراپ تۇر مەن سىنى ئۆزگەرتىمەن.
ئەھمەد Moغا مىيىقىدا كۈلۈپ قويدى-دە ئۆز يولىغا ماڭدى.
«ياق…ياق!...ئۇ نېمە قىلماقچىدۇ؟….»ئەھمەد  ئويلىنىپ قالدى.ۋە يەنە يولىنى داۋاملاشتۇردى.
ئۇ مەسجىتكە يىتىپ كەلدى،مەسجىتنىڭ دەرۋازىسىدىن ئاتلىغان شۇ دەقىقىدە  ئەزەلدىن بولۇپ باقمىغان ھاياجانغا چۈمۈلدى.
«مۇشۇ جاي شۇ…»دەيتتى ئۇ ئۆز-ئۆ‍زىگە پىچىرلاپ.
مەسجىد ئىچىدە ھېچكىم يوق بولۇپ جىمجىت ئىدى.ئۇ مەسجىتنىڭ بىر بۇلۇڭىغا باردى ۋە قىبلىگە يۈزلىنىپ ئولتۇردى.
«ئى ئاللاھ! سەندىن قورققان ھالەتتە ساڭا يۈزلىنىپ تۇرۇپتىمەن، مىنى مەغپىرەت قىلغىن!ھازىرغىچە سەن ياقتۇرمايدىغان نۇرغۇن گۇناھ ئىشلارنى قىلدىم،سەن يەنىلا نىئمىتىڭنى ئايىمىدىڭ، جىنىمنى تىنىمدىن ئايرىمىدىڭ…»ئەھمەد ئىككى سائەتچە ئاللاھقا ئىلتىجا قىلىپ،گۇناھىنى تىلەپ يىغلىدى.يۈزلىرى كۆز ياشلىرى بىلەن يۇيۇلدى.كۆز ياشلىرى جەينامازنىمۇ ھۆل قىلىۋەتتى.
تۇيۇقسىز مەسچىتنىڭ ئىمامى كىرىپ كەلدى ۋە يىغلاۋاتقان ئەھمەدنى كۆرۈپ، گەپ قىلماي ئۇنىڭ دۇئاسىغا قۇلاق سىلىپ تۇردى.ئىماممۇ تەسىرلەنگىنىدىن يىغلاپ كەتتى ۋە ئاستاغىنا قىشىغا كېلىپ ئۇنىڭغا تەسەللىي بىرىشكە باشلىۋىدى،ئەھمەد قىچىشقا ئۇرۇندى.
-ئىمام،مەن…..مەن…مەن ھەقىقەتەنمۇ خىجىل….دىدى ئەھمەد ئوڭايسىزلانغان ھالدا.ئۇ تېخى گىپىنىڭ ئاخىرىنى دەپ بولالمايلا ئىمام گەپ باشلىدى.
-ئەھمەد، ھېچقىسى يوق.سەن ئىچ ئىچىڭدىن نومۇس قىلدىڭ،قىلغانلىرىڭغا پۇشايمان قىلدىڭ،ئەمەلىيەتتە بۇ ياخشى ئىش. سىنىڭ خىجىل بولۇشۇڭ،قىلغان ئەسكى ئىشلىرىڭدىن كىرىلىپ يۈرگىنىڭدىن كۆپ ياخشى ئەمەسمۇ.- دېدى ئىمام ناھايىتى مىھرىبانلىق بىلەن.-ئاتا-ئاناڭ چوقۇم سەندىن پەخىرلىنىدۇ،ھازىر ھەممىدىن مۇھىمى ئەمدى ئۆزۈڭنى پۈتۈنلەي ئاللاھقا تاپشۇرغىن.
-مەن ئاتا-ئانامغا ئازار بەرگەن ئىمام،ئەمدى قانداق قىلسام ئۇلارنى خۇشال قىلالايمەن؟
-ئۇلار سىنىڭ بۇ ھالىتىڭنى كۆرسە چوقۇمكى خۇشال بولىدۇ،داداڭ چوقۇم سەندىن پەخىرلىنىدۇ.
-دىمەكچى بولغىنىم،مەن ئۇلارغا سوۋغات سۈپىتىدە بىرەر نەرسە ئىپ بىرەيمىكىن دېگەن.
-ھىم،بولىدۇ،سەن ئاپاڭغا گۈل ياكى شاكىلات، داداڭغا ئەتىر ئىلىپ بەرسەڭ بولامۇ؟
ئەھمەد كۈلۈمسىرىدى ۋە:

ئەھمەد ئاللاھتىن مەغپىرەت تەلەپ قىلغاچ يولىنى داۋام قىلىۋاتاتتى…Moئەھمەدنى يىراقتىن كۆرۈپ قالدى ۋە ئۇنى بۇرۇنقى ھالىتىگە-گۇناھقا پاتقان ھالىتىگە قايتۇرۇپ كېلىش پىلانى ئۈچۈن ئۇدۇل ئەھمەد تەرەپكە قاراپ ماڭدى.ئۇ يىرىم يالىڭاچ بىر قىزنى يىتىلىۋالغان ئىدى. ئوتتۇرا ياشلاردىكى ھىجابلانغان بىر مۇسۇلمان ئايال يولنى كىسىپ ئۆتمەكچى بولۇپ تۇرغان ئىدى… ئەھمەدمۇ دەل يولنى كىسىپ ئۆتىۋاتاتتى…Moتىزلىكتە يولنىڭ ئوتتۇرىسىغا كەلدى

ئەھمەد يەرگە قارىغىنىچە يولدىن ئۆتىۋەردى.شەيتان ۋە ئۆلۈم تەڭلا ئەھمەدكە يوپۇرۇلۇپ كەلمەكتە…
«ئاۋۇ قىزغا قارا ئەھمەد!نىمىدىگەن چىرايلىق!ھىچ بولمىسا ئۇنىڭغا قاراپ بولسىمۇ ھوزۇرلانغىن،ئارماندا قالىسەن.»شەيتان ئۇنىڭ قۇلىقىغا پىچىرلايتتى.
يىراقتىن كىلىۋاتقان بىر يۈك ماشىنىسى تۇيۇقسىز كونتروللىقىنى يوقاتتى. تىز سۈرئەتتە ئۆزلىرىگە قاراپ كىلىۋاتقان ماشىنىدىن قىچىشىۋاتقان كىشىلەر قورققىنىدىن چىقىرايتتى.ئەھمەد ھىلىقى مۇسۇلمان ئايالنىڭ نەخ ماشىنىنىڭ ئۇدۇلىغا كەپ قالغانلىقىنى كۆردى!-ياق!-ئۇ سومكىسىنى تاشلاپ جىنىنىڭ بارىچە ئۇ ئايال تەرەپكە قاراپ يۈگۈردى.
«ئاللاھ ئۆزۈڭ ساقلىغايسەن،مەن ئۇنى قۇتقۇزاي،ئاللاھ يولىدا بىر ياخشى ئىش قىلغىنىم بوپ قالسۇن،ئاللاھنى رازى قىلاي…»ئەھمەد شۇنداق دىگىنىچە ئۇنىڭغا قىشىغا قاراپ يۈگۈردى ۋە ئىتىلىپ بىرىپلا ئۇنى ئىتىىرىۋەتتى. دەل مۇشۇ دەقىقىدە ماشىنا Moۋە يىنىدىكى قىزبالىنى قاتتىق سوقۇپ  نەچچە مېتىر ئارىلىققا ئېتىۋەتتى.ئەمدىكى نىشان ئەھمەد ئىدى،ئۇ ھىلىقى ئايالنى ئىتتىرىۋىتىپ  قىچىشقا ئۈلگۈرمەي ماشىنىغا تۇتۇلدى.سىكۇنتا يەتمىگەن ۋاقىت ئىچىدە بۇ ئۈچ بالا قانغا مىلىنىپ يىتىپ قالدى.
«ئى ئاللاھ!مەن قورقۇۋاتىمەن،مىنىڭ بۇ ھالەتتە ئۆلگۈم يوق ئىدى،مەن ھەتتا تولۇق بىر كۈن بەش نامازنى ئادا قىلىپ باقمىدىم،مىنىڭ بۇنداق تىزلا…»ئەھمەد ئۆز-ئۆزىگە پىچىرلىدى،كۆزلىرى ياشلاندى،باش قىسمىدىن توختىماي قان ئېقىۋاتاتتى.
بۇ ۋەقە قۇلاقتىن قۇلاققا يىتىپ،بىردەمدە ئادەم ئولىشىپ كەتتى، ئىماممۇ يىتىپ كەلدى ۋە ئۇتتۇر ئەھمەدنىڭ  قىشىغا قاراپ يۈگۈردى..
-ئەھمەد!-ئىمام  توۋلىغىنىچە ئۇنىڭ قىشىغا كەلدى ۋە ئۇنىڭ قانغا بويالغان قوللىرىنى ئالقىنىغا ئالدى..
-كەلىمە شاھادەتنى ئىيتقىن ئوغلۇم!
تۇيۇقسىز ئۇنىڭ بەدىنىنى سوغۇققىنە  شامال سىيپاپ ئۆتكەندەك بولدى،كۆز ئالدىدا ئەزەلدىن كۆرۈپ باقمىغان، تەسەۋۋۇرمۇ قىلىپ باقمىغان مەنزىرىلەرزاھىر بولۇشقا باشلىدى:
بىر توپ پەرىشتىلەر ئاسماندىن چۈشۈشكە باشلىدى،بىر قىسىم پەرىشتىلەر قاناتلىرىنى كەڭ ئىچىپ تۇرغان بولۇپ، قۇياشتەك نۇر چاچاتتى،چىھرىدىن ۋە پۈتۈن بەدىنىدىن گۈزەللىك تۆكۈلۈپ تۇراتتى.قالغان بىر قىسمى بولسا پۈتۈنلەي ئەكسىچە-يۈزلىرى ۋە پۇتۇن بەدىنى ئادەمنى قاپقارا، بەكمۇ قورقۇنچلىق ئىدى،يۈزلىرى ئادەمنى ئەيمەندۈرەتتى.ئادەم چىدىيالمىغۇدەك دەرىجىدە قورقۇنچلىق ئاۋاز چىقىراتتى.
-لا ئىلاھە ئىللەللا!-دېدى ئەھمەد قىينىلىپ تۇرۇپ.
ئىمامنىڭ چىرايىغا كۈلكە يۈگۈردى،ھاياجاندىن كۆز ياشلىرىنى تۇتالماي قالدى.ئەھمەدنىڭ پىشانىسىدىن پۇر-پۇر تەر كۆتۈرۈلدى.ئىمام بۇنىڭ ياخشى بېشارەت ئىكەنلىكىنى بىلەتتى.
ئەھمەد قايسى پەرىشتىلەرنىڭ كىمگە چۈشۈشىنى بىلمەيتتى.تۇيۇقسىز ئۇ ئۆزىنى بەكمۇ راھەت ھېس قىلدى،-ئۆمرىدە ھېس قىلىپ باقمىغان بىر يېقىملىق تۇيغۇ-شۇنداق گۈزەل  پەرىشتىلەر ئۇنىڭغا تەبەسسۇم بىلەن قاراپ تۇراتتى. ئەھمەد قولىنى ئۇلارغا سۇندى.
-بولدى يىغلىما ئەھمەد،كەلگىن.بىز سىنىڭ دوستۇڭ.بىىز سىنى ياخشى كۆرىمىز.بز سەن تاكى ئاللاھنى كۆرگىچە سىنى قوغدايمىز.-شۇنداق دىگەچ پەرىشتىلەر ئەھمەدنىڭ قولىدىن تۇتتى،ئۇلار ئەھمەدنىڭ روھىنى خۇددى ئانا يىڭى تۇغۇلغان بوۋىقىنى كۆتۈرگەندەك ئاۋايلاپقىنا ئىلىپ ماڭدى.
ئەھمەد بۇ مەنزىرىلەرنى كۆرۈپ ھەيرانلىقتىن،خۇشاللىقتىن يىغلىۋەتتى. بىر پەرىشتە  ئىلىپ كەلگەن ھەرخىل رەڭلىك نۇردىن ياسالغان بۈشۈكتەك نەرسىگە قاراپ بىر ئىشىنىپ بىر ئىشەنمەي تۇرۇپ قالدى، پەرىشتىلەر ئەھمەدنىڭ روھىنى بۆشۈكتەك نەرسىگە سىلىۋاتقاندا،باشقا پەرىشتىلەر ئەھمەدكە ئۇ ئەزەلدىن ھېچقانداق ئادەم زاتىدىن پۇراپ باقمىغان خۇشپۇراق بىر خىل ئەتىر چىچىشقا باشلىدى.
ئوخشاش ۋاقىتتا ھىلىقى قورقۇنچلۇق پەرىشتىلەر Moۋە يىنىدىكى قىزنى قىيىن قىستاققا ئالماقتا ئىدى.
پەرىشتىلەر ئۇ ئىككىسىنىڭ جىنىنى خۇددى يۇڭدىن تىكەن ئايرىغاندەك يۇلۇپ ئالدى.پەرىشتىلەر ئەھمەدكە ئەمىنلىك بولسۇن ئۈچۈن ئۇلارنىڭ قىينىلىپ ۋارقىرىغان ئاۋازىنى ئاڭلاشتىن توسۇپ قالدى.
-ياق!ئۆتۈنۈپ قالاي!مەن مۇسۇلمان!مەن مۇسۇلمان!-ۋارقىرىدى Moئازاپقا چىدىماي.
-مۇسۇلمان!مۇسۇلماننىڭ يۈزىنى چۈشۈرمە!سەن ئاللاھقا بويسۇنمىدىڭ،ئاللاھ ياقتۇرمايدىغان ئىشلار بىلەن ئۆمرۈڭنى زايا قىلدىڭ،ئەمدى سەن ئازاپلىنىسەن!مەڭگۈ!
-ئاللاھ جاجاڭنى بەرسۇن Mo،مىنى بۇ يولغا سەن باشلىدىڭ!ساڭا ئاللاھ ئىككى ھەسسە ئازاب بەرسۇن!-دەپ توۋلايتتى قىزچاق جان ئاغرىقىغا چىدىماي.
-مەن سىنى مەجبۇرلىمىغان،سەن ئۆزۈڭ خالاپ ماڭا ئەگەشكەن!
-تىنىچلىنىڭلار!بۇ داۋانى قىيامەت كۈنى قىلىشىڭلار!-دېدى پەرىشتىلەر.

ئەھمەدنىڭ دادىسى يىتىپ كەلدى،قانغا مىلىنىپ ياتقان ئەھمەدنى كۆرۈشى بىلەن ئۆزىنى تۇتىۋالالماي ئۇنىڭغا ئۆزىنى ئاتتى.
ئىمام ئەھمەدنىڭ دادىسىنى كۆردى ۋە تىزدىن  ئۇنى ئەھمەدنىڭ جەسىتىگە يىقىنلىشىشتىن توسىدى.
-نېمە ئىش بولدى؟ ئۇنىڭغا بىر ئىش بولمىغاندۇ؟-سورىدى يىغلامسىراپ.
-ئاللاھ ئۇنى دەرگاھىغا ئەكەتتى.-دېدى ئىمام توۋەن ئاۋازدا.
-يا ئاللاھ!ئوغلۇم!مىنىڭ ئوغلۇم!-دېدى دادا ۋە تۇرغان يىرىدە ئولتۇرۇپلا قالدى.
-مۇھەممەد،گىپىمنى ئاڭلا،ئوغلۇڭ ئاللاھقا قايتقانىدى.مەن ئۇنىڭ بۈگۈن مەسجىتتە يىغلاپ ئولتۇرغانلىقىنى كۆرگەن ئىدىم، ئۇ تەخمىنەن ئىككى سائەتچە ئاللاھتىن مەغپىرەت تىلەپ،ئاللاھقا يالۋۇرۇپ يىغلىدى. ئۇنىڭ بىلەن شۇنداق ياخشى پاراڭلاشتىم،ئۇ تېخى ئايالىڭ ئىككىڭلارغا سوۋغات ئىلىپ ئككىڭلارنى خۇشال قىلماقچى ئىدى.ئەڭ مۇھىمى،مۇھەممەد،ئۇ كەلىمە شاھادەت ئىيتىپ جان ئۈزدى، بۇلا ئەمەس،ئۇ بىر مۇسلىمەنىڭ ھاياتىنى قۇتقۇزۇش ئۈچۈن ھاياتىدىن ئايرىلدى.ئۇنىڭ نەپىسىى توختاش ئالدىدا ئۇنىڭ كۆككە قاراپ تەبەسسۇم قىلغىنىنى كۆردۈم.ھە راست،بۇنى ئالغىن.»دېدى ئىمام ۋە ئەھمەدنىڭ ئۇلارغا ئالغان سوۋغىتىنى سۇندى.
ئەھمەدنىڭ دادىسى سوۋغاتنى ئالدى،قوللىرى توختىماي تىترىمەكتە ئىدى.ئۇ ئالدىرىماي بولاقنى ئاچتى.ئىچىدە چىرايلىق بىر قۇتا ئەتىر ۋە بىر قۇتا شاكىلات بار ئىدى.ئۇ ئىچىدىكى بارغاقچىنى ئوقۇشقا باشلىدى.
«سۈيۈلۈك دادا،بۇ سىنىڭ،مەن ئاللاھنى ھەممىدىن ياخشى كۆرىمەن ۋە بۇنىڭدىن كېيىن ھەممە ئىشنى ياخشى قىلىپ سىنى ۋە ئاپامنى ھەممىدىن مۇھىمى ئاللاھنى خۇشال قىلىمەن.مەن ھىلى ناماز ۋاقتىدا سەن بىلەن بىللە مەسجىتكە بىرىپ ناماز ئوقۇيمەن ئىنشائاللاھ»
ئەھمەدنىڭ دادىسى ئۆزىنى تۇتالماي يىغلاپ كەتتى.
-ئوغلۇم،مەن سەندىن پەخىرلىنىمەن!ئاللاھۇ ئەكبەر!ئاللاھۇ ئەكبەر!مۇھەممەد ئورنىدىن تۇردى ۋە ۋە ئۆيدە ئوغلىنىڭ گۇناھىنى تىلەپ ئولتۇرغان ئايالىنىڭ قىشىغا قاراپ چاپتى.
-ئالىيە!ئالىيە!قارا ئەھمەدنىڭ بىزگە ئالغان سوۋغاتلىرىغا!-ئەھمەدنىڭ دادىسى تىزلىكتە ئەھمەد ئالغان شاكىلاتنى ئايالىغا سۇندى ۋە باغاقچىنى ئۇنىڭغا كۆرسەتتى:
«قەدىرلىك ئانا!بۇ شاكىلاتنى ساڭا ئالدىم.مەن ساڭا روھىي ۋە جسمانىي جەھەتتىن جىق ئازار بەردىم.،ئاللاھ بىلەن قەسەمكى بۇنداق ئىش ئەمدى قايتا يۈز بەرمەيدۇ. بۇندىن كېيىن سىنىڭ قوشۇماڭنى تۈرگۈزمەيمەن،سىنى خۇشال قىلىمەنكى خاپا قىلمايمەن.ئىنشائاللاھ بىز بىللە ئاللاھقا ئىبادەت قىلىمىز،ئاللاھ خالىسا ياندىشىپ جەننەتكە كىرىمىز.»
-سۇبىھاناللاھ!لا ئىلاھە ئىللەللا!مىنىڭ ئوغلۇم ئاللاھقا قايتىپتۇ-ئانا ھاياجاندىن توۋلىۋەتتى.
ھىچكىممۇ ئەھمەد ئىرىشكەن شەرەپنى ۋە گۈزەللىكنى ۋە ئۇنىڭ بىشىدىن كەچۈرگەنلىرىنى ئويلاپ يىتەلمەيتتى،شۇنداقلا Mo باشتىن كەچۈرگەن ئازاپنىمۇ ھېچكىم تەسەۋۋۇر قىلالمايتتى.
