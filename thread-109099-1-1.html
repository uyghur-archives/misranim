                                   خىيالىمغا بولىسەن تۇتقۇن

                             مەن ئېزلەيمەن بەخېت ئوچۇغى ،
                               پىشقانمىدۇ يارنىڭ پوچىغى .؟
                                ئورغاق ئايدا ئورىسام ئۇنى ،
                                 ئېچىلارمۇ يارنىڭ قۇچىغى .


                              ئەدراسماندا ئېسرىق سالسام ،
                              پاشا كىلىپ چاقالماس سىنى .
                             مەن قولومغا يەلپۈگۇچ ئالسام ،
                           ئەركەك چىۋىن قونپاقسۇن قىنى .


                               ساڭا بولاي يازدا يىپىنچا ،
                        قىش-قىروۋدىن ساقلايمەنغۇ ھەم .
                           مەخمەل بولوپ ساڭا سىلىنچا ،
                           يىتىلدۈرەي مۇھەببەتكە تەم .


                             ماڭا تەۋە يەككە-يىگانە ،
                         ئاسماندىكى ھۆر پەرى كەببى .
                           چولپانىمدەك چاقنا نۇرانە ،
                         بۇ ئاشقى چىن يۇرەك سەببى .


                       يوقاپ كەتمە شەپەق بىلەن تەڭ ،
                        دەرتلىرىڭنى يوشىرىپ مەندىن .
                       بۇ سۆيگۇگە سەن مۇۋاپىق ئەڭ ،
                    چىن مۇھەببەت ئايلانسۇن سەندىن .


                       يولتۇز يانار ئاسماندا يال- يال ،
                      كۆرسەم دەيمەن سەھەردە باھار .
                      سەن يۇرەكتە يانىسەن ۋال-ۋال ،
                      گۈل بەرگىدىن شەبنەملەر تامار .


                       كۆلۇپ تۇرار ئاسمىنىڭدا ئاي ،
                      سېخى نۇرنى ئالەمگە چىچىپ .
                       چىمەن ئارا بوزلايدۇ تورغاي ،
                    كىچىك-كىچىك ئېقىنلار كىچىپ .


                       تېتىرەيدۇ قولومدا بىر گۇل ،
                     ھەسرەتلىرىم سانجىلغانمىدۇ.؟
                      زىبا ئەيلەپ ماڭا بىرنى كۈل ،
                       باشقا ئاشىق يوقالغانمىدۇ.؟


                       خىيالىمغا بولېسەن تۇتقۇن ،
                     ئاي ياسىغان گۈزەل چەمبەردە .
                     ھەسرىتىمنى تىزرەك يۇلغۇن ،
                    ھالەك بولماي گۈزەل دىلبەردە .


                         مەرھابا.2013.7.29
