ئەسسالامۇ ئەلەيكۇم مىسرانىم ئەپەندىم ياخشىمۇ سىز؟ بۇ مۇنبەردە قانداق ساپاسىز باشقۇرغۇچىلارنى تەربىيلەۋاتىسىز؟ نىمە قىلغىنى بۇلارنىڭ. بىر دوستىم بار ئىدى.يېڭى تۇغۇتلۇق.ھازىر دوختۇرخانىدا تۇغۇت باقىدىغان ئادىمى يوق.ئاپىسى تۈگەپ كەتكەن. شۇ توغۇرلۇق دوستۇمغا ياردەم قىلاي دەپ تۇغۇت باققۇچى ئىزدەش توغىرىسىدا تىما يوللىغان ئىدىم. بىر تال ئەسكەرتىشمۇ بىرىپ قويماي ئۈچۈرۋىتىپسىلەر. بەك تېتىخسىز ناچار خەق باكەن بۇ مۇنبەردە. ئۇچۇر تورى دىگەندەك تور بىكەتلەرنىمۇ ئوبدان بىلەتتىم. ئەمما ئۇ تور بىكەتنىڭ ئىنكاسى بەكلا ئاستىكەن. بۇ تۇغۇت دىگەن قارغۇچىسس بومممىسا بەك جاپالىق نازۇك مەسىلىكەن .مۇنبەرگە چىقىدىغانلار كۆپ.ئىنكاسلارمۇ كۆپ بۇلىدۇ شۇڭا جىددى ياردەم سوراپ تىما يوللىغان ئىدىم.ئىسىت شۇنچىلىك ھالىڭلارغا خەت باسقان قوللۇرۇم. نەدىكى قوقۇنۇشلۇق ئالۋاستى دەمدۇ ئىش قىلىپ ئادەم ئالدايدىغان كۆڭۈل ئېچىش خارەكتىردىكى ئەخلەت پۈچەك  تىمىلار تەستىقلىنىپ نادىر تىمىلار بوپ كىتىپ مۇشۇ بىر مۇسۇلمان قېرىندىشى ئۈچۈن شۇ نەچچە قۇر ياردەم تىمىسنى تەستىقلىمىغان ھالىڭلارغا ۋاي.ئۇلۇغ رامزان ئايلىردا ئاللاھ ئىنساۋىڭلارنى بىرە. ئاللاھ يەنە سىلەردەك نان قېپىلارنىڭ ياردىمىسىز ئۇ دوستۇمنىڭمۇ ئىشلىرنى ئاسان قىلا.ئامىن. ئەڭ ئاخىردا سىزگە بىر ئىشنى ئەسكەرتىپ قۇياي مىسرانىم ئەپەندىم. ئەڭ ياخشىسى ھېلىقىدەك كارغا كەلمەيدىغان باشقىلارنىڭ يوللىغان تىمىسنى ئۇقا ئۇقىمايلا تەستىقلىماستىن ئۈچۈرۋىتىدىغان،باشقىلارنىڭ ئەمگىكىنى موچەنگە ئەرزىمەس قىلۋىتىدىغان ئۇخلاپ قۇپاپلا چاپىقىنى سۈرۈتمەي تۇرۇپ ئەخلەت پۈچەك تىملارنى تەستىقلايدىغان پوق قۇساقلارنى تىزدىن كۈتىگە تىپىڭ.بوممىسا ما مۇنبەر ئېغىلغا ئايلىنىپ كەتمىسۇن.چۈنكى ئۇنداق ئادەملەرنىڭ ئىشەكتىن نىمە پەرقى.قەلبىلە ھىسىداشلىق مىھرى مۇھاببەتتتىن ئەسەرمۇ قالمىغان ئادەمنى شۇنداق دىيىش ھەقلىق.

7-ئاينىڭ 27-كۈنى چۈشتىن كېيىن، خېنەن ئۆلكىسى نېيشياڭ ناھىيەسىنىڭ شەھەر ئاھالىسى شياۋجۇ بىرقانچە دوستى بىلەن شۇ يەردىكى باۋتيەنمەن مەنزىرە رايونىغا سۇ ئۈزگىلى بېرىپ، كەچ سائەت يەتتە يېرىملار ئەتىراپىدا قايتىپ كېلىۋاتقاندا، بىر ساقچى ماشىنىسى ھۇيقىتىپ چىقىرىغىنىچە كېلىپ ئۇلارنىڭ ئالدىنى توسىدۇ. ئارقىدىنلا ساقچى ماشىنىسىدىن پۇقراچە كىيىنگەن تۆت ئادەم چۈشۈپ، ئۇلارنىڭ ماشىنىسىنىڭ ئالدى تەرەپ شامال توسۇش ئەينىكىنى چېقىۋېتىدۇ، ھەمدە مەجبۇرىي ھالدا ماشىنىنىڭ ئىشكىنى ئاچماقچى بولىدۇ. تۇيۇقسىز يۈز بەرگەن ۋەقەدىن شياۋجۇ ۋە ئۇنىڭ دوستلىرى ئىنتايىن قورقۇپ كېتىپ، شۇئان ماشىنىنىڭ ئىشىكىنى چىڭ ئېتىۋېتىپ 110 غا تېلېفون قىلىپ ياردەم سورايدۇ. 30 مىنۇتتىن كېيىن 110 نىڭ ماشىنىسى نەق مەيدانغا يېتىپ كېلىدۇ. شۇ چاغدىلا ئۇ ۋە ئۇنىڭ دوستلىرى ماشىنىدىن چۈشۈشكە جۈرئەت قىلىدۇ. 110 ساقچىلىرى ئاۋال ئۇ ۋە ئۇنىڭ دوستلىرىنىڭ ھاراق ئىچكەن-ئىچمىگەنلىكىنى تەكشۈرىدۇ، ئۇلارنىڭ ھاراق ئىچمىگەنلىكىنى مۇئەييەنلەشتۈرگەندىن كېيىن، ئۇلارنى نېيشياڭ ناھىيەسى چىلىپىڭ يېزىلىق ساقچىخانىنىڭ بىر تەرەپ قىلىشىغا تاپشۇرۇپ بېرىدۇ. كەچ سائەت ئونلاردا ئۇ ۋە ئۇنىڭ دوستلىرى چىلىپىڭ يېزىلىق ساقچىخانىغا ئېلىپ بېرىلىپ، كېچىلەپ سوراق قىلىنىدۇ.






ساقچى ماشىنىسىنىڭ شوپۇرى جاڭxxنىڭ شياۋجۇغا ئېيتىشىچە، ئەينى ۋاقىتتا ساقچى ماشىنىسى ۋەزىپە ئىجىرا قىلىۋاتقان ئىكەن. ئۇنىڭ ماشىنىسى ساقچى ماشىنىسىنىڭ ئالدى تەرەپتە بولۇپ، ساقچى ماشىنىسى بىلەن ئىككىسى ئوخشاش يۆلۈنۈشكە قاراپ نەچچە كىلومېتىر مېڭىش جەريانىدا، ئۇ ساقچى ماشىنىسىغا يول بەرمىگەن، شۇ ئاچچىغىدا جاڭxxئۇنىڭ ماشىنىسىدىن ئېشىپ كەتكەندىن كېيىن ئالدىنى توسۇپ، موشتى بىلەن ئۇنىڭ ماشىنىسىنىڭ ئالدى تەرەپ شامال توسۇش ئەينىكىنى چېقىۋەتكەن. جاڭxxۋاقىتلىق ھاياجان تۈپەيلى چېقىۋەتكەن ماشىنا ئەينىكىنى ئۆز باھاسى بويىنچە تۆلەپ بېرىدىغانلىقىنى بىلدۈرگەن. ئارقىدىنلا نېيشياڭ ناھىيەلىك جامائەت خەۋپسىزلىك ئىدارىسىنىڭ مۇئاۋىن باشلىقى جاڭ لىن بىلدۈرۈپ : «ئەينى ۋاقىتتا بەنچاڭ يېزىلىق ساقچىخانىنىڭ ساقچى ماشىنىسى شۇ يەردىكى بىر جىنايەتچىنى يالاپ ئېلىپ ماڭغان، ساقچى ماشىنىسى يول بويى ساقچى چىرىقىنى يېقىپ گۈدۈك بېرىپ ماڭغان بولسىمۇ، ئالدى تەرەپتىكى بىر بىيادى ماشىنىسى يول بەرمىگەن، ساقچى ماشىنىسى ئۇ ماشىنىغا يېتىشىۋالغاندىن كېيىن، شوپۇر يېنىكلىك قىلىپ ماشىنا ئەينىكىنى چېقىۋەتكەن، نۆۋەتتە ساقچى تەرەپ تۆۋەندىكىدەك بىر تەرەپ قىلدى: بىرىنچى، ساقچى ماشىنىسى شوپۇرى ۋاقىتلىق ئىشچى بولغاچقا خىزمەتتىن توختىتىلدى. ئىككىنچى، بۇ ۋەقەنى چىلىپىڭ ساقچىخانىسىنىڭ بىر تەرەپ قىلىشىغا تاپشۇرۇپ بېرىلىدۇ، قولغا ئېلىنىدىغانلار قولغا ئېلىنىدۇ، بىر تەرەپ قىلىنىدىغانلار بىر تەرەپ قىلىنىدۇ، ئۈچىنچى، بەنچاڭ يېزىلىق ساقچىخانا ئالاقىدار كىشىلەردىن كەچۈرۈم سوراشقا بۇيرۇلدى» دېدى.
