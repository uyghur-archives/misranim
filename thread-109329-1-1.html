                                                         مۇقەددىمە

   تەڭرى دەرگاھىدا  ھەيۋەت - ھەشەمەت لەززەت ، بايلىق - باياشادلىق بىھېسابتۇر . سېنىڭ بۇنداق سوۋغاتلىرىڭدىن تەڭرى بىھاجەت ، تەڭرى دەرگاھىدا يوق نەرسە پەقەت ئەلەم ۋە ھەسرەت ، سەن شۇنى سوۋغات قىلىپ بار .  
                       ئەلشىر نەۋائى

      ئازاپ قوينىدا مۇڭلىنىپ تۇرۇپ تۆكۈلگەن كۆزياشلار نىمىدىگەن تاتلىق . ۋىسالغا كاۋاپ بولغان يۈرەك ئارقىلىق ئىنتىلىش غورورلۇق كىشىنى سۆيۈندۈرىدۈ . پاجىئە نەق ۋىسالغا يىتىشتىن باشلىنىدۇ .... يۈرەكتىن قۇيۇق ئىس - تۇتەك پۇرىغان ۋاقىت بەخىت ئۈچۈن ئىزدىنىۋاتقان چاغلاردۇر . تۈندىكى يالغۇز قارىياغاچ كۆز يىشى قىلىدۇ ، تۆكۈلىۋاتقان ياپراقلار بۇنىڭ دەلىلى ئەمەسمۇ ؟ !

                                            قاراڭغۇدا قارىياغاچ مېنىڭ سىردىشىم
                                           شىلدىرلىسا يوپۇرمىقى تۆكۈلەر يىشىم

    قارىياغاچقا چىڭ يىپىشىپ ھەسرەتلىنىۋاتقان يۈرەك ، شىلدىرلاۋاتقان ياپراقلار ، تۈن كىچە ، غەمكىن چىقىۋتقان شاماللار نېمىدىگەن لىرىكىلىق - ھە ! دۇنيا ھەسرەتكە لىق تولغان ....
   تەنھالىق قوينىدا ياش تۆكۇش ، چەكسىز خىياللاردا تەپەككۇر شىخىغا قونۇش ، يار ئىشقىدا ئوت بولۇپ يېنىشتىنمۇ ئارتۇق ھوزۇر بولارمۇ؟
   قەلىبنىڭ جاراھەتلەنگەن ، ئاچچىق ئازاپلاردا قىسىلغان مىنۇتلىرى - ئىنساننىڭ ئۆزىنى ئىزدىگەن ، بەخىت ئىشىكىنى چەككەن چاغلىرىدۇر .
  چېكىت ئادەمنى يارىتىدۇ . بىھۇدە ئۇرۇلغان چىكىت ئۆزىنى ھالاك قىلىش بىلەن بىللە ئانا زىمىنغا ھاقارەت كەلتۈرۈپمۇ قويىدۇ . چېكىت دىمەك باقىينىڭ ئاخىرىقى نوقتىسىدۇر .
   قار ياغقان كېچە دەرد - ئەلەم چەككۈچىلەرنىڭ ئۇنتۇلماس بىر كىچىسى ، شۇ كىچە ئۇلار ئۆزلىرى ھەققىدە ئويلىنىشلارغا مۇيەسسەر بولالايدۇ ، لېكىن ، ھەسرەت يەنىلا ئۇلاردىن ئايىرىلالمايدۇ .
  جۇدۇنلۇق كۈنلەر ئۇنىڭ روھىغا ئۇۋا سالدى . روھى مۇزلاپ تېنى قۇرۇدى . ئۇ مۇزغا ئايلانغان .ئادەم ...
  مۇڭلۇق تۆكۇلگەن كۆز ياش ، يالغۇز قارىياغاچ ، سۆيۇملۇك تەنھالىق ، ھۇزۇرلۇق كۈز ، قار ياغقان كېچە ، مۇزلىغان روھ ، ئاھ ، يەنە بار زەردەپقا ئۆرۇلگەن قان ....

                                                      مۇڭلانغان           خىيال
  كۆڭۇل كىمنى سۆيسە ، شۇ كۆز ئالدىدا  بولىدۇ . كۆز قايەرگە تىكىلسە ، شۇ يەردە ئۇنىڭ قىياپىتى ئۇچۇپ يۈرۈيدۈ.
                   -  يۈسۈپ خاس ھاجىپ

    ئىغىر خىياللار دا مۇڭلانغان نۇر ئەلى ھاياتلىق شامى ئۆچكەندەك سۈكۈتكە چۇمگەن ئۆيىدە ئاچچىق ئازاپلاردىن ئوتقا چۇشكەن قىلدەك تولغۇنۇپ ياتاتتى . يۈرىكى تۇز سۇيىگە چىلانغاندەك ئېچىشاتتى . ئوتلۇق تىنىقلىرى بۇرۇن تۆشۇكچىلىرىنى كۆيدۇرۇپ چىقاتتى . تەنھالىق قوينىدىكى ئاچچىق ھەسرەتتىن قاتقان قاپقارا ، يوغان كۆزلىرى ئالدىدا بىر پەرىزات تىنماي جىلۋە قىلاتتى - دە ، ۋۇجۇدى چاراسلاپ كۆيەتتى . ئۇ ئاشۇ كۇنلەرنى ئەسلىمەسلىككە تىرىشاتتىيۇ ، لىكىن قانداقتۇر بىر كۇچ ئۇنىڭ ئەسلىمە خاتىرىلىرىنى تىنىمسىز ۋاراقلايتتى . << سۆيگۇ ئەڭ مۇقەددەس بىر نۇر ، مەن يورۇقلۇق دۇنياسىدا ئۇنىڭغا ئىرىشەلمەيدىغانلىقمنى سىزىپ تۇرىۋاتىمەن . >> ئۇ بۇ دۇنيانى يوق دەپ قاراپ ، ئەبەدىيلىك گۇزەللىككە قېتىلىپ كىتىشنى ئارمانمۇ قىلىپ قالاتتى . ھامان بىر كۈنى بۇ دۇنيانىڭ پارچىلىنىپ كىتىدىغانلىقىغا شەكسىز ئىشەنگەندە ، ئۇنىڭ ئىچكى دۇنياسى ئايان بولاتتى . < سۆيگۇ بولمىسا بۇ دۇنيا قۇرۇق ، قاقشال نەرسە ، ئىنساننىڭ مەڭگۈلۈك مەۋجۇدلىقىنى ساقلاپ قالالايدىغان بىرلا نۇر بولىدۇ .ئۇ نۇر قەلىبتىن كۆتۇرلۇپ كەتسە ، ئىنساندىن ئارتۇق بالا - قازا كەلتۇرگۇچى بولماس . ئىنساننى بۇ دۇنياغا تارتۇشتۇرۇپ تۇرىدىغان نەرسىمۇ ئاشۇ نۇر . لىكىن ، ئىنسانلار شەيتاننىڭ ۋەسۋىسىدە بۇنى سىزەلمەي قالىدۇ .
ئىنسان .ئۆزىدىكى تەنھالىق روھىدىكى قاراڭغۇلۇق بىلەن ئاشۇ نۇرغا ئىنتىلىشى كىرەك ئىدى . ئەپسۇس ئۇلارنىڭ كۆزلىرى پەردىلىنىپ ، دىللىرى قارىيىپ بۇنى تونۇپ يىتەلمەيۋاتىدۇ . نۇرلار خىرەلەشتى . ئىنسان شۇڭا ئازغۇن . كۆپ ھاللاردا ھەقنى كۆرەلمەيدۇ . مەنچۇ ؟ ... >
      ئۇنىڭ روھى ئەسلىمە خاتىرلىرى يەنىلا ۋاراقلاندى .

             *      *     *      *       *      *         *         *       *        *          *        *       *            *

  قارلىق كىچە ، ھەممە يەر قارغا پۇركىنىپ ، گۇزەل بىر پاكلىق دۇنياسى نامايەن بولۇپ تۇراتتى . سوغۇق ئۆزكۈچىنى كۆرسىتۋاتاتتى . كوچا چىراغلىرىدا ئاندا - ساندا دولىلىرىنى قورۇشۇپ ، ئۆتۇشۇپ تۇرغان ئادەملەرنى ئۇچراتقىلى بولاتتى . بۇ رومانتىك تۇس ئالغان ، لىرىكىغا تويۇنغان شۇنداق بىر گۈزەل كىچە بولغانىدى .
     نۇر ئەلى تۇنجى مۇھەببەت سىگىنالىنى مۇشۇ كىچە قوينىدا ياڭراتقانلىقىدىن ئۆزىنى شۇنچىلىك بەختىيار سەزدى . بىراق قەلبىدە بىر خىل ۋەھىمە دولقۇنلايتتى . بەلكى ، بۇ تۇنجى سۆيگۇنىڭ تارتىنىش ، قورۇنۇش تۇيغۇلىرى بولسا كىرەك .  لۇتفىيە ئۇنىڭ ئەكىسچە بەكمۇ روھلۇق ، جۇشقۇن كۆرنەتتى . قۇيۇق كىرپىكلىرى ئاستىدىكى قارا كۆزلىرى سۆيگۈدىن سۈمۈرگەن نۇرلاردا قوياش سۆيگەن دېڭىزدەك نۇرلىنىپ تۇراتتى . ئىچ - ئىچىدىن قايناۋاتقان شېرىن ھېسلار ئۇنى  مەسىت قىلىپ ، ھاياتىنى ئۇنىڭغا تىكەنسىز ئەتىرگۇلدەك كۆرسىتەتتى . ئۇ ئۆزى ئىرىشمەكچى بولغان بەخىتكە ئېرىشىپ بولغاندەك ھېسياتتا ئىدى . قايغۇ - ئەلەم ، ھەسرەتلەر ئۇنىڭغا يات......
   - ۋيى ماۋۇ قورقۇنچاقنى ، - دېدى لۇتفىيە مۈرىسى بىلەن نۇرئەلىنى يىنىك سوقۇپ . نۇر ئەلى ھۇپپىدە قىزىرىپ كەتتى . كىچە ئۇنىڭ خىجىللىقنى يۇشۇرۋالدى .
  - نېمىدىن قورقاتتىم ، - نۇرئەلى شۇنداق دەپ ئۆزىنى روھلۇق تۇتۇشقا تىرىشتى . شۇنداقتىمۇ ئەتراپقا ئالاقىزادىلىك بىلەن قاراپ قوياتتى .
  بۇ كىچە ئۇلار يۇرەكلىرىدىلا سۆزلەشتى . بىر - بىرىگە ئوتلۇق مەنىلىك ، سېھىرلىك بىقىشلار بۇنى رۇشەن دەلىللەيتتى . ئاياغلىرى ئاستىدىن چىقىۋاتقان يىقىملىق مۇزىكىدەك غاسىلداشلار گويا ئۇلارنىڭ مۇھەببىتىنىڭ ئىشىكىنى ئاچقانلىقىغا قۇت تىلەۋەتقاندەك شۇنچىلىك يىقىملىق ئىدى . قەلىبلىرىدە مەۋج ئۇرۇۋاتقان گۇزەل كۇيلەر ئۇلارنى ئاجايىپ بىر گۇزەل ماكانغا يېتەكلەيتتى . ئۆزلىرىنى شۇ قەدەر بەخىتلىك ھېسابلىشاتتى .
  ئۇلار ئاياغلىرى ئاستىدا << بۇلبۇل >> سايراتقانچە كوچا ئايلىنىپ ، مەكتەپ قورۇسى ئىچىگە قايتىپ كىرىشتى - دە ، تاتلىق بىر ئازاپ ئىلكىدە خەيرىلىشىپ ، ئۇن - تىنسىز ئايرىلىشتى . ئىسسىق بىر ئېقىم نۇر ئەلىنىڭ تومۇر - تومۇرلىرىدا ئېقىپ تۇرۋاتقاچقا دېكابىرنىڭ سوغۇقىمۇ ئۇنى مۇزلىتالمىدى . ئۇ ياتاققا كىرىپ خىلى ۋاقىتقىچە شېرىن خىياللاردىن تولغۇنۇپ ئۇخلىيالمىدى . ياتاقداشلىرى بولسا ، بۇ دۇنيادىن ھېچبىر غېمى يوقتەك ئېزىلىپ ئۇخلاپ كىتىشكەندى .
