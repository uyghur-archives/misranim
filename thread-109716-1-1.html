                                             چۈنكى ئۇ ئانا


                                      غازان بولۇپ گۈزەل باغلىرىڭ،
                                      كېپىنەكلەر ئۇچۇپ كېتىشسە.
                                      ھىجرانلاردا قالغان چاغلىرىڭ،
                                      ۋىسال سېنى تەرك ئېتىشسە.
                                      پەقەت بىر زات قالۇر يېنىڭدا،
                                      ئۇ ئانادۇر،چۈنكى ئۇ ئانا.

                                      كېيىك ياتماس بولۇپ تاغلىرىڭ،
                                      ئۈلپەتلىرىڭ سەندىن كېچىسسە.
                                      سۆيدۈم دېگەن قەلەم قاشلىرىڭ،
                                      ئۆزگە باققا كۆزلەر تىكىشسە.
                                      پەقەت بىر زات قالۇر يېنىڭدا،
                                      ئۇ ئانادۇر،چۈنكى ئۇ ئانا.

                                      كۆككە يېتىپ پەرياد-ئاھلىرىڭ،
                                      كەڭرى ئالەم ھەم تار كېلىشسە.
                                      قىيناقلارغا قېلىپ ئىچ-باغرىڭ،
                                      پۈتۈن دۇنيا سېنى ئېزىشسە.
                                      پەقەت بىر قالۇر يېنىڭدا،
                                      ئۇ ئانادۇر،چۈنكى ئۇ ئانا.



                                               ھەقىقەت


                                   ئەزىزلەر، نان نېمە؟بۇ ھايات نېمە؟
                                   ئاشىققا جان نېمە؟ھەم مامات نېمە؟
                                   مەھكۇملۇق نېمىدۇر ھۆر قانات نېمە؟
                                   خىياللار قەلىبتە غەليان قىلىدۇ،
                                   ھەقىقەت مىھىرلىك كۈلۈپ تۇرىدۇ.

                                   ئەزىزلەر،قەيەردىن كەلگەي سائادەت،
                                   گۈزەللىك نەدىدۇر،نەدۇر ئادالەت.
                                   ئىنسانلىق نە قىلىپ تاپار كامالەت،
                                   تەشۋىشلەر قەلىبتە تېنەپ يۈرىدۇ،
                                   ھەقىقەت مىھىرلىك كۈلۈپ تۇرىدۇ.

                                   ئەزىزلەر،ئەلەمگە بولۇپ مۇپتىلا،
                                   تار بولۇپ تۇيۇلسا بۇ كەڭرى دۇنيا.
                                   تەسەللى قۇشىدىن ئۈزۈلسە سادا،
                                   ئاسماندا پارلىغان قۇياش ئۆچىدۇ،
                                   ھەقىقەت مىھىرلىك كۈلۈپ تۇرىدۇ.

                                   ئەزىزلەر،بۇ كۈنلەر باھاردۇر ئۆتكەي،
                                   باغلارنى جۇت بوران ۋەيرانە  ئەتكەي،
                                   ئىنساننىڭ دەردىگە ئېيتىڭ كىم يەتكەي،
                                   بىر كۈنى كۆزلەرگە ياشلار تولىدۇ،
                                   ھەقىقەت مىھىرلىك كۈلۈپ تۇرىدۇ.

                                   ئەزىزلەر،نىچۈن بۇ تېنەش تىڭىرقاش،
                                   نىچۈن بۇ چۈشكۈنلۈك،نىچۈن بۇ كۆز ياش،
                                   ئۇدۇممۇ بىزلەرگە غەپلەتتە ياشاش،
                                   بىر يول بار مۇستافا ئەپكەلگەن يول بۇ،
                                   ھەقىقەت مىھىرلىك كۈلۈپ تۇرىدۇ.

                                   ئالەمدە سەيياھمىز بىلىڭ ئەزىزلەر،
                                   كۈنۇ-تۈن ئىستىغفار ئېيتىڭ ئەزىزلەر،
                                   مەسىيەت -گۇناھتىن قايتىڭ ئەزىزلەر،
                                   بۇ دۇنيا ئۇ دۇنيا قىيامەتتىمۇ،
                                   ھەقىقەت مىھىرلىك كۈلۈپ تۇرىدۇ.
