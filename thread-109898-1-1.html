ئەي سۈيۈملۈك مىھرىبان دادا ياخشىمۇ سىز ! سىز مىنىڭ يۆلەنچۈكۈم ، سىز مىنىڭ يۈرۈكۈم !سىز بولغاچقا مەن بەخىتلىك ،سىز بولغاچقا مەن خاتىرجەم سىز بولغاچقا مىنىڭ ھەممە يەردە بېشىم ئۈستىن تۇرۇدۇ .
مەن سىزنى بۇرۇن مىھرىبان كۈيۈمچان ئانام بىلەن سېلىشتۇراتتىم . ئانام ماڭا كۈيۈنىدۇ ئانامنىڭ باغرى ئىنتايىن يۇمشاق دەيتتىم .ئەمماسىزنىڭمۇ باغرىڭىز ئانامنىڭكى بىلەن ئوخشاشكەن .پەقەت سىز مىھرىبانلىقىڭىزنى سىرتىڭىزغا چىقىرالمايدىكەنسىز .  ئىچىڭىزدە بولسا ماڭا بولغان مىھرى -مۇھەببىتىڭىز يالقۇنجاپ تۇرىدىكەن ،ھەردەم بالامنىڭ بوينى قىسىلمىسا ، كىشى ئالدىدا مەردانە ياشىسۇن دەيدىكەنسىز . بىر خىل دادىلىق يەنى ئەركەكلىك سالاپىتى سىزنى توسۇپ قويىدىكەن . شۇ ھىسياتىڭىزنى مەن بىلمەپتىمەن . ماڭا يىراقتىن كۈيۈنگەنلىرىڭىزنى ، سەن مىنىڭ جېنىم ، سەن مىنىڭ قانىتىم دىگەنلىرىڭىزنى بىلمەستىن ،چۈشەنمەستىن سىزدىن ئاغرىنىپ كەپتىمەن . مەن خاتا قىلىپتىمەن .سىزماڭا كۈيۈنىدىكەنسىز ، مىنى ئەتىۋارلايدىكەنسىز . نەچچە يىل نەچچە ئاي مە سىزنى خاپا قىلدىم . مىنى ئۇردىڭىز ،تىللىدىڭز ئەمما يۈرۈكىزدىن ئەمەس پەقەتلا شۇ مەسئۇلىيەت ئۈچۈن شۇنداق  قىلدىڭىز .مىنى ياخشى ئادەم بولسۇن دەپ شۇنداق قىلدىڭىز ، لىكىن سىزدىن ئاغرىندىم .سىزنى يالغۇز قويدۇم مىنى كەچۈرۈڭ دادا مەن ئېزىپتىمەن .



بۈگۈن كىچە بىر چۈش كۆردۈم چۈشۈمدە مەن  بىر ئات ھارۋىسدا مېڭىپتىمەن  يېنىمدا نۇرغىنلىغان دوسلۇرۇم بار .بىزشۇنداق خۇشال توپىلاڭ يولدا كىتىۋاتاتتۇق . تۇيۇقسىز بىر ئەگمىدىن ئۆتتۇق . ئالدىمىزدا شۇنداق سالاپەتلىك بىر ئادەم پىيادە كىتىۋاتاتتى .ھاۋا شۇنداق ئىسسىق ئەمما ئۇئادەم چاپان بىلەن ئىدى . بىز ھەرقانچە تىز ماڭساقمۇ ئۇ ئادەمگە يىتىشەلمىدۇق . ئاخىرى بولالماي بۇكىمدۇ .بىلىپ  باقىلى دەپ يەرگە چۈشۈپ يۈگۈردۇق ئالاھەزەل بىرەر سائەتتەك يۈگۈردۇق .دوسلۇرۇمنىڭ ھەممىسى كەينىمدە قەپ قالدى. ئۇ   ئادەم بولسىا ئاستا مېڭىۋاتاتتى ،مەن يۈگۈرۈپ -يۈگۈرۈپ ئاخىرى يىتىشىۋالدىم . لىكىن چىرايىنى كۆرۈپ...........
شۇنداق چۈچۈپ ئويغۇنۇپ كەتتىم .ئۇ ئادەم دەل سىز ئىدىڭىز دادا . مەن چۈشۈم توغرۇلۇق ئويلاندىم . ئاخىرى تاپقان جاۋابىم دادا مەن سىزنى تەنھا يەككە يىگانە يالغۇز قالدۇرۇپتىمەن .سىزنى شۇنداق غېرىپ قىلىپتىمەن . سىزنىڭ ئالدىڭىزدا ئاللا ئىگەم ئالدىدا ئوغۇللۇق مەجبۇرىيىتىمنى ئادا قىلالماپتىمەن .جېنىم دادا مىنى كەچۈرۈڭ .مىنى قايتا باغرىڭىزغا بېسىڭ . ئەمدى سىزنى يالغۇق قويمايمەن .سىزنىڭ مۇستەھكەم قانىتىڭىز بولىمەن دادا !جېنىم دادا !!!

بۇ ئەسەرنى دوسلۇرۇمنىڭ ئوقۇپ ياخشى بىر ئويلۇنۇپ بېقىشىنى ئاتا -ئانىسىنى خاپا قىلىۋاتقانلارنىڭ كاللىسىنى سىلكىۋېتىشىنى ئۈمۈت قىلىمەن .ئەسەردىكى چۈش راست يەنى مەن كۆرگەن چۈش .

‹‹   داداڭ رازى ، ئاناڭ رازى  ، رەببىڭ رازى   ››
