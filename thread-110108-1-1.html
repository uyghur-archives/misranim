     ئەسسالامۇ ئەلەيكۇم ئەزىزلەر ، ‹مىنى ئۇيغۇرئىكىس› توغرۇلۇق يېزىلغان تىمىنى ۋە ئىنكاسلارنى ئوقۇغاندىن كىيىن ، تورداشلارنىڭ خېلى كۆڭۈل بۈلىۋاتقانلىقىنى ھىس قىلدىم . شۇڭا مەنمۇ مىسرانىم بېتىدە بىلگىنىم بويىچە ئىز قالدۇرۇپ قويۇش نىيىتىگە كەلدىم .


     ئالدى بىلەن قېىرىنداشلارغا دەۋالماقچى بولغان سۆزۈم شۇكى ، يەرلىك كومپىيۇتېر بىلەن مۇناسىۋىتى  بولمىغان مەشغۇلاتلاردا ( يەنى كومپىيۇتېرغا پىرۇگىرامما قاچىلاش تەلەپ قىلىنمايدىغان توردىكى مەشغۇلاتلاردا ) ئۇبۇنتۇ مەشغۇلات سىستېمىسى ۋىندوۋىسنىڭ ئەڭ يېڭى نۇسخىسى بولغان ۋىندوۋىس8 دىنمۇ ياخشى . ( بولۇپمۇ ئۇيغۇر تىلىنى ئانا تىلىم دەپ قەدىرلىگۈچىلەر ئۈچۈن ).

   پەقەت ۋىندوۋىسنىڭ توركۆرگۈچىسى بولغان IE پىرۇگىراممىسىنىڭ ئۇبۇنتۇ ئۈچۈن تەمىنلەنگەن نەشرى بولمىغانلىق سەۋىۋىدىن IE توركۆرگۈگە بېقىنىدىغان ئازغىنە مەشغۇلاتلارنى نۇرمال قىلغىلى بولمايۋاتىدۇ . (بۇ دۆلىتىمىز ئىچىدىكى ھادىسە ).

ئۇندىن باشقا شەخسى كومپىيۇتېرلار ئۈچۈن يەنىلا ۋىندوۋىس 8 دىن ياخشى ھەم بىخەتەر .

نىمىشقا مۇشۇنداق قارىغانلىقىمنىڭ تەپسىلاتىنى تۈۋەندىكى چۈشەندۈرۈشلىرىمدىن بىلگەيسىز .


         ئۇيغۇرئىكىىس ھەققىدە چۈشەنچە


    1.ئۇيغۇرئىكىس دىگەن نىمە؟

    جاۋاپ :  ئۇيغۇرئىكىس -ئۇيغۇر تىلىنى ئاساس قىلىپ ئىشلىتىدىغان خەلىقلەرنىڭ ئىھتىياجىغا ماسلاشتۇرۇپ ياساپ چىقىلغان كومپىيۇتېر مەشغۇلات سېستىمىسى .

ئىنىقراق ئېيىتقاندا خەلىقئارادا ئىناۋەتلىك ئۇبۇنتۇ مەشغۇلات سېستىمىسىغا ئۇيغۇرتىلىنى قۇشۇش ئىشى ئەمەلگە ئاشۇرۇلغاندىن كىيىن ، ئۇيغۇرلار ئارىسىسدا تېخىىمۇ قۇلاي ئۆزلىشىشى ئۈچۈن ، زۈرۈر پىرۇگىراممىلارنى ۋە تەڭشەكلەرنى سېستىمىنى قاچىلاپلا ئىشلەتكىلى بولىدىغان قىلىپ ياساپ چىقىلغان .

ئۇبۇنتۇنىڭ ھازىرقى ئەڭ مۇنتىزىم نەشرى بولغان 12.4 نى ئاساس قىلغان بولغاچقا ، ناھايىتى ياخشى ئىشلەۋاتىدۇ ،ھازىر تۈنجى قېتىملىق سىناق نۇسخىسى ئۇيغۇرئىكىس 1.0 نامى بىلەن تارقىتىلدى ، ئىشلەتكۈچىلەرنىڭ ئىنكاس ۋە تەلەپ پىكىرلىرى ئاساسىسىدا مۇنتىزىم نۇسخىسى تېخىمۇ لاياقەتلىك ۋۇجۇتقا چىققۇسى .

   2.ئۇيغۇرئىكىسنى كىملەر كىم ئۈچۈن قانداق بارلىققا كەلتۈردى؟

   جاۋاپ:  خەلىقئارادا كەڭ ئومۇملىشىپ ئىشلىتىلىپ كىلىۋاتقان لىنوكىس سېستىمىسغا تەۋە بولغان ئۇبۇنتۇ دەپ ئاتالغان بىر سېستىما بولۇپ ، ئۇيغۇر كومپىيۇتىرچىلىقىدا زور كۈچ قۇشۇۋاتقان ئۇيغۇر كومپىيۇتېر ئىلمى جەمىيىتى ۋە بىلىك تورىنىڭ باشچىلىقىدا بىر تۈركۈم ئوت يۈرەك ، خەلىقپەرۋەر كىشىلىرىمىز بىرلىكتە

تەرجىمە قىلىشقا كىرىشكەن ، شۇنىڭ بىلەن بىرگە ئۇبۇنتۇ سېستىمىسىنى ئۇيغۇر تىلىنى ئىشلىتىدىغان خەلىقلەرنىڭ ئۆزلەشتۈرەلىشىگە قۇلايلىقلاشتۇرۇپ بىرىش ئۈچۈن ئۇيغۇرئىكىس نامىدىكى بۇ سېستىمىنى بارلىققا كەلتۈرگەن. بۇ جەريان پىداكارلىق بىلەن تاماملانغان ھەم يەنە مۇستەھكەملەش خىزمەتلىرىنى خالىس ئىشلەۋاتىدۇ .

    3.ئۇيغۇرئىكىسنىڭ قانداق ئالاھىدىلىگى بار ؟

    جاۋاپ :  ئۇيغۇرئىكىس مەشغۇلات سېستىمىسى ئۇيغۇر تىل -يېزىقىنى مۇكەممەل قوللايدۇ .

قۇشۇمچە باشقا دىتال قاچىلىماستىن بىمالال ئۇيغۇر تىلىدىكى يازما مەشغۇلاتلىرىنى قىلغىلى بولىدۇ ،

رەسىم ۋە سىن ئۈستىگە باشقا تىللارغا ئوخشاشلا ئۇيغۇرچىنى يېزىق شەكلىدە كىرىشتۈرگىلى بولىدۇ .

توردىكى ھەرقانداق يازما يازىدىغان رامكىلارغا ئۇيغۇرچىنى ئۆلچەملىك يازغىلى ۋە كۆرگىلى بولىدۇ .

ئىلخەت ۋە چ چ دىگەندەك ئالاقىلەردە ئۇيغۇرچىنى ئۆلچەملىك ئىشلەتكىلى بولىدۇ .

توردىكى مەشغۇلاتلاردا ۋىرۇس يۈكلەنگەن پىرۇگراممىلاردىن ئەنسىرەشنىڭ ھاجىتى يوق .

سېستىمىدا كۈرۈلگەن خاتالىقلارنى ئۆزلىگىدىن تۈزەش ئىقتىدارى بار .

ھەرقانداق خەت نۇسخىلىرى بىرگە قاچىلانغان ، ناھايىتى ئۆلچەملىك كۈرىنىدۇ .

دىتاللارنى قاچىلاش ۋە ئۈچۈرۈش باسقۇچلىرى ناھايىتى ئاددى ھەم بىخەتەر ،

پۈتۈنلەي ئۈزىڭىزنىڭ ئىختىيارىدا بىرتەرەپ قىلالايسىز ،

سىنبەلگۈلەر ۋە ئۇسلۇپ شەكىللىرى بەك كۆپ ھەم چىرايلىق ،

دىتاللار ھەقسىز .

توردىن دىتال قاچىلاش زۈرىرىيىتى كۆپ ئەمەس .تورغا ئۇلانغان كومپىيۇتىردا تەۋسىيىلىك تۇلۇقلىمىلارنى ئاپتۇماتىك تېپىپ ئورنىتىش ئىقتىدارى بار .

سېستىما ھۆججەتلىرىنىڭ يۇقاپ كىتىش مەسىلىسى يۈز بەرمەيدۇ ،

باشقا سېستىمىلار بىلەن بىللە ئىشلىتىشكىمۇ مۇۋاپىق كىلىدۇ .

كۆپ سىغىم تەلەپ قىلمايدۇ .

      4.ئۇيغۇرئىكىسنى كومپىيۇتىرىمغا قاچىلىسام ، ماڭا نىمىگە پايدىسى بار ؟

    جاۋاپ :   ئەگەر سىز ئۇيغۇر تىلىنى ئاساس قىلىپ ئىشلىتىۋاتقان ئادەم بولسىڭىز ، كۆرىۋاتقان كومپىيۇتىرىڭىزنىڭمۇ ئۇيغۇرچە بولغىنى توغرىراق . سىز خەڭسۇچە ۋە ئىنگىلىسچە ئىشلىتىپ ئادەتلەنگەن بولسىڭىز ، كەلگۈسىدىكى بالىلار بۇنداق قىلماي ئۇيغۇرچىنى ياقتۇرىدىغان چىقىپ قالسا ، سىزنىڭ بۇ ئۆمرىڭىز ئەجىرسىز ئۆتكەن بوپ قالمامدۇ . ئۇنىڭ ئۈستىگە ئۇيغۇر تىلى دىگەن تىلمۇ دۇنيادا خېلى ئىمتىيازلىق ، بالىلار چۇقۇم بىزدىن ئەقىللىق ، ئۇلارنىڭ قارغىشىغا قالماسلىق ئۈچۈن بولسىمۇ

ئانا تىلنى قەدىرلەش كېرەك .

      5.كومپىيۇتىرىمغا ئۇيغۇرئىكىسنى قاچىلىسام ، خەڭسۇچە ،ئىنگىلىسچە ۋە باشقا تىللارنىمۇ ئىشلىتەلەمدىم ؟

         جاۋاپ :   دۇنيادىكى ئىشلىتىلىپ كىلىۋاتقان ، ئاساسلىق بۇلۇش ئىناۋىتىدىن قالمىغان ھەرقانداق تىلنى  ئىشلەتكىلى بولىدۇ . سېستىمىنى قايتا قاچىلىمىسىڭىزمۇ ، ناھايىتى ئوڭايلا تىللارنى ئالماشتۇرۇپ ئىشلىتەلەيسىز .

     6.مەن خەڭسۇچە ۋە ئىنگىلىسچىغا ئۇستا ، ئۇيغۇرچىغا ھاجىتىم چۈشمەيدۇ ؟

     جاۋاپ :ھاجىتىڭىز چۈشمىسە ئىشلەتمىسىڭىزمۇ بولىدۇ ، ئەمما ئۇيغۇر بۇلۇپ تۇرۇش ئۈچۈن يەنە بىر مەسلىھەت قىلىپ كۈرۈڭ.

      7.ئۇيغۇرئىكىس ۋىندوۋىسنىڭ ئۇيغۇرچە قىلىنغىنىمۇ ؟

     جاۋاپ :  ئەمەس ، بۇ ۋىندوۋىسنىڭ خاتاسىنى تۈزەش ئۈچۈن ئۈلگە بۇلىۋاتقان ، لىنوكىس سېستىمىلىرىنىڭ تۇغۇندىسى بولغان ئۇبۇنتىنىڭ بالىسى ھەم ئۈزى.

ئىجتىمائىلشىشنىڭ ھەقىقى ئىپادىلىنىشى .

     8.ۋىندوۋىسقا كۈنۈپ قاپتىمەن ، ئۇيغۇرئىكىسنى ئىشلىتىپ بۇلالماي قالارمەنمۇ ؟

    جاۋاپ :  ئۇيغۇرئىكىسنى ئىشلىتىش ۋىندوۋىس ئىشلىتىشتىن كۆپ قۇلاي ھەم ئىلمى ،  ۋاقىت ئىسراپچىلىقىدىن خالى .

     9.لىنوكىس بىلەن ۋىندوۋىس قانداق پەرىقلىنىدۇ ؟ قايسىسى ياخشى ؟

     جاۋاپ :  لىنوكىسنى ھەركىم ئۈزىگە خاسلاشتۇرۇپ ئۆزلەشتۈرىۋالسا بولىدۇ ، كودلىرى ئاشكارە .

ئىجتىمائىلقنى پىرىنسىپ قىلىدۇ .ھەقسىز . يېڭىلىنىشى تىز ، كەلگۈسىدىكى يۈزلىنىشكە

ئەڭ ماس كىلىدۇ . كومپىيۇتىرنىڭ نىمىلىگىنى ئۇچۇق ئاشكارە چۈشەنگىلى بولىدۇ .

     10.لىنوكىس ، ئۇبۇنتۇ ، ئۇيغۇرئىكىس دىگەنلەرنىڭ قانداق مۇناسىۋىتى بار ؟

   جاۋاپ : لىنوكىس › ئۇبۇنتۇ › ئۇيغۇرئىكىس

    11.مەن كومپىيۇتىرىمنى قانداق ئىشلارغا ئىشلىتەلەيمەن ؟ قانداق ئىشلىتىشىم كېرەك ؟

   جاۋاپ : سىز تورسىز ئىشلەتسىڭىز سىن ، رەسىم ، ئۈن ھۆججەتلىرىنى قۇيالايسىز ، تەھرىرلىيەلەيسىز .     تېكىسلەرنى يازالايسىز ، كۈرەلەيسىز ، تەھرىرلىيەلەيسىز .

ھىساپلاشلارنى ، لاھىيىلەشلەرنى ئېلىپ بارالايسىز .ماتىرياللارنى ساقلاپ قۇيالايسىز .

توردا بولسىڭىز دۇنيانى بىلىپ تۇرالايسىز ، سودا ئالاقىلەرنى قىلالايسىز ، جاھاننىڭ ئۇ چېتىدىكى دوستىڭىز بىلەن بىر ئۆيدە كۈرۈشكەندەك كۈرۈشەلەيسىز . ئۈزىڭىزنىڭ بانكىسىنى كونتىرول قىلالايسىز .

قانداق ئىشلىتىش كېرەك دىگەندە ، ئاۋال نىمىگە ئىشلەتمەكچى بولغىنىڭىزنى جەزىم قىلىپ ، شۇ ئىشىڭىزغا ماس كىلىدىغان سېستىما ۋە دىتاللارنى تاللىشىڭىز كېرەك ، ئەگەر سىز توردىكى پۇل مۇئامىلە ئىشلىرىنى قىلمىسىڭىز باشقا ھەرقانداق ئىشلارنى ئۇيغۇرئىكىس ئارقىلىق قىلالايسىز ، زۈرۈر بولغاندا پۇل مۇئامىلە ئىشلىرىنىمۇ قىلغىلى بولىدىغان قىلىپ تەڭشەش مۇمكىن . سىزنىڭ ئەڭ ياخشى تاللىشىڭىز بۇلالايدۇ . قوش سېستىما قىلىپ ئىشلىتىپ ، قايسى سېستىمىنىڭ ياخشى ئىشلەيدىغانلىقىنى سېلىشتۇرۇپ ئايرىۋالسىڭىز بولىدۇ ، سىزگە قايسىنىڭ ئىشلەش ئىقتىدارى يارىغان بولسا ، شۇنى تاللىشىڭىز توغرا .

مۇھىمى سىز ۋە كىيىنكىلەر ئۈچۈن بىر قارارغا كىلىڭ .

    12.بالىلارغا كومپىيۇتېر ئۈگەتمىسەممۇ بۇلامدۇ ؟

   جاۋاپ : ئۇنداقتا مۇۋاپىق بولماس . كەلگۈسىدىكى ئىشلار كومپىيۇتىرغا تېخىمۇ باغلىنىپ قالىدۇ ، قىيىنچىلىق تارتىپ قېلىش مەسىلىسنى نەزەردىن ساقىت قىلساق بولماسمىكىن .

   13.كومپىيۇتېر سېتىۋالغاندا قانداق تاللىشىم كېرەك ؟

   جاۋاپ : كومپىيۇتېر سېتىۋالغاندا ئاۋال كومپىيۇتىرنىڭ قانداق نەرسە ئىكەنلىگىنى بىلىدىغان ئىشەنچىلىك دوستىڭىزدىن ياردەم سۇراڭ . ھەقىقى مەنىدىكى كومپىيۇتىر - قاتتىق جىسىملا ئەمەس . شۇڭا ئەسكىسىگە ئۇچراپ قېلىشتىن ساقلىنىڭ ، ھازىرقى ۋە كىيىنكى سەۋىيىەگە ماسلىشىش كېرەك دىگەننى شەرىت قىلىڭ .ئالدىنىپ قالماڭ . بەزى كىشىلەر سىزنىڭ پۇلىڭىزنى بولىۋالمىسىمۇ ، ئالداپ ئېلىۋالالايدۇ ، ئۈزىڭىزگە ئۈزىڭىز مەسئۇل بولمىسىڭىز بولمايدۇ .

5. ئۇبۇنتۇنى قاچىلىغاندىن كېيىن ، ئانا تىلىمىزنى كۆڭۈلدىكى تىل سۈپىتىدە تەڭشەش ۋە خەت نۇسىىلىرىنى بەلگۈلەش تۇلىمۇ ئاسان .


سېلىشتۇرما : ۋىندوۋىستا بولسا سەۋىيىسى تۈۋەن كىشىگە نىسبەتەن تۇلىمۇ قىيىن .


6. ئۇبۇنتۇنى قاچىلىغاندىن كېيىن ، قۇشۇمچە ئورنىتىدىغان دىتاللار كۆپ ئەمەس ، قاچىلاش ۋە قاچىلىماسلىقىزنى ئۈزىمىز بەلگۈلەيمىز ، بىرسىنى قاچىلىساق باشقىسىنى قاچىلامسىز دەپ سورىمايدۇ . ئۈچۈرۈشمۇ تىنىچ تامالىنىدۇ .


سېلىشتۇرما : ۋىندوۋىستا بولسا بىر دىتالنى قاچىلاش مەشغۇلاتى قىلغاندا ، باشقا دىتاللارنىمۇ قاچىلاشنى سۇرايدۇ ، سىزنىڭ ۋاختىڭىز بىلەن ھىساپلاشمايدۇ . سىزنى كىچىك بالىدەك ئەخمەق قىلىدۇ .سىز چەتئەل تىلىنى بىلمىسىڭىز خالىغانچە كىيىنكى قەدەمنى بېسىپ داۋاملاشتۇرۇشقا توغرا كىلىدۇ . كومپىيۇتىرىڭىزنىڭ يۈكى ئېغىرلاپ كىتىدۇ . ئۈچۈرمەكچى بولسىڭىز سىزدىن نىمە سەۋەپتىن ئۈچۈرمەكچى دەپ سۇراپ تۇرىۋالىدۇ ، سىزنىڭ ۋاقتىڭىزنىڭ قىلچە ئىتىۋارى بولمايدۇ . چەتئەل تىلىنى بىلمىگەن كىشىگە تېخىمۇ تەس .


8. ئۇبۇنتۇدا مەيلى يەرلىك ۋە مەيلى توردا بولسۇن ، ئانا تىلنى چىرايلىق يازغىلى ۋە كۆرگىلى بولىدۇ .ئارتۇق تەڭشەشلەرنى تەلەپ قىلمايدۇ .


سېلىشتۇرما : ۋىندوۋىستا بولسا بىز ئەڭ ياقتۇرۇپ ، ئەڭ كۆپ ئىشلىتىدىغان چ چ قۇرالىنىمۇ ئانا تىلىمىزدا ئىشلىتىش پۇرسىتىگە ئىرىشەلمەيۋاتىمىز .


9. ئۇبۇنتۇدا پىرۇگراممىلارنىڭ ئىگەللەيدىغان سىغىمى كىچىك ،


قاچىلىنىشى تىز تاماملىنىدۇ .پىرۇگىراممىلارنى توردىن ئىزدەش زۈرۈرىيىتى يوق دىيەرلىك . بولغاندىمۇ چۈشۈرۈش جەريانى ئاسان ۋە تىز .


سېلىشتۇرما : ۋىندوۋىستا بولسا ئىچىڭىزنى پۇشۇرىدىغان باسقۇچلار تۇلا . سىزنىڭ ۋاختىڭىزنى ئېلىپ ، تىرىكتۈرۈپ ، ھالىڭىزنى قويمايدۇ .




10. ئۇبۇنتۇدا ۋىرۇس دىگەن نەرسىگە يۇچۇق قالدۇرۇلمىغان ئوخشايدۇ . ۋىرۇسخور قاچىلاسنى تەلەپ قىلمايدۇ .


سېلىشتۇرما : ۋىندوۋىستا بولسا قاچانلا قارىسىڭىز ۋىرۇس تارقاتقۇچى ۋىرۇسخور پىرۇگراممىلىرىنىڭ ئىلانىدىن قۇتۇلالمايسىز .


11. ئۇبۇنتۇ سېستىمىسى قاچىلانغان قاتتىق دىسكىنى ھەرقانداق كومپىيۇتىردا ئوخشاشلا ئىشلەتكىلى بولىدۇ . كومپىيۇتىرنىڭ قاتتىق ئۈسكىنىلىرىگە ئورتاق بولغان قوزغاتقۇچلار تەمىنلەنگەن .


سېلىشتۇرما : ۋىندوۋىستا بولسا دەل ئەكسىچە ،


12. ئۇبۇنتۇ سېستىمىسى پۈتۈن دۇنيادىكى مەۋجۇتلىقىنى يۇقاتمىغان ھەممە ئادەمگە ئوخشاش مۇلازىمەت قىلىدۇ .


سېلىشتۇرما : ۋىندوۋىس بولسا سانى ۋە پۇلى كۆپ مىللەت كىشىلىرى ئۈچۈن مۇلازىمەت قىلىدۇ .


13. ئۇبۇنتۇ سېستىمىسى ئەڭ ئالدى بىلەن سىزنىڭ ۋاختىڭىز ۋە خىزمەت ئۈنىمىڭىزنى ئويلىشىدۇ ،


سېلىشتۇرما : ۋىندوۋىس بولسا شەخسىيەتچى كىشىلەر ئۈچۈن پۇرسەت يارىتىپ ، ۋاقىت ئىسراپچىلىقىغا ئازادە يول ئېچىپ بىرىدۇ .


14. ئۇبۇنتۇ سېستىمىسى كۈندە دىگۈدەك ماس قەدەمدە ھەقسىز يېڭىلىنىپ تۇرىدۇ .


سېلىشتۇرما : ۋىندوۋىس بولسا نەچچە يىلدا بىر قېتىم يېڭىلىنىپ ، بانكا كارتىڭىزنى ئىشلىتىشكە مەجبۇر قىلىدۇ .


مىكروسوفىت شىركىتىنىڭ ئاۋال ئويلايدىغىنى پۇل ، جۈملىدىن بايلار ئۈچۈن مۇلازىمەت قىلىشتىن ئىبارەت .


15. ئۇبۇنتۇ مەۋجۇت تىل ۋە يېزىقىنى تەرەققى قىلدۇرۇشنى تەۋىسسىيە قىلىدۇ ،


سېلىشتۇرما : ۋىندوۋىس بولسا سانى ۋە پۇلى كۆپ قەۋىملەرنىڭ


مەدىنىيىتىنى ساقلاپ قېلىپ ، باشقىسىنى يۇقىتىشنىڭ باشلامچىسى .




مۇشۇنچىلىك بوپ تۇرسۇن ، ئاساسەن ھىس قىلغانلىرىمنى يېزىپ بولغان ئوخشايمەن ، ئەمدى ئۇبۇنتۇدىكى ئىشلىتىشمىزگە ئەرزىيدىغان


بىر نەچچە ئالاھىدىلىكنى تەھلىل قىلىمەن :


1. ئۇبۇنتۇدا گوگىلنىڭ ئانا تىلىمىزغا ئۆزلەشتۈرۈلگەن تور كۆرگۈچىنى گوگىل ئىلخەت نامىمىز ئارقىلىق تيزىملىتىپ ئىشلەتسەك ، ئۇ بىز ئۈچۈن بەكمۇ قۇلايلىقلارنى ئېلىپ كىلىدۇ . بىزگە مۇناسىۋەتلىك تارىخنى ساقلايدۇ ، مەيلى قايسى كومپىيۇتىردا تورغا ئۇلانمايلى ، دائىملىق مەزمۇنلارنى ئۆزلۈگىدىن كۆرسىتىپ ، مەشغۇلات ئۈنىمىنى يۇقىرى كۈتىرىشىمىزگە ياردەم بىرىدۇ .

بۇ يەردە سىناق قىلىپ كۈرۈشكە ئەرزىيدىغان بىر ئىش بار ، يەنى بىز بىر گىمائىل خەت نامىدا گوگىل تور كۆرگۈچىسىگە تىزىملىتىپ ، دائىملىق ئاممىۋى مەزمۇن ۋە ئۇلىنىشلارنى رەتلەپ ۋە تەڭشەپ ، شۇ تىزىملاتقان گىمائىل خەتنىڭ ئادرىس نامى ۋە مەخپى نۇمۇرىنى ھەمدە باشقا بىخەتەرلىك تولدۇرۇشنى تەلەپ قىلىدىغان مەزمۇنلارنى دوستلۇرىمىزغا ئەۋەتىپ بەرسەك ، كۆپلىگەن ئادەم بىرلا ۋاقىتتا ، شۇ تىزىملىك بىلەن كىرىپ ، ئىنتايىن قۇلاي بولغان تور كۆرگۈگە ئىگە بۇلالىشى مۇمكىن .بىر قېتىم تىزىمغا كىرىپ بولغاندىن كىيىن ، قايتا تىزىملىتىشنىڭ زۈرىرىيىتى بولماسمىكىن دەيمەن ، مەن بۇنى بىرلا ۋاقىتتا ئۈچ كومپىيۇتىردا ئىشلىتىپ باقتىم زادى بىرلا ۋاقىتتا نەچچە كومپىيۇتىردا ياكى نەچچە رايۇندا ئىشلەتكىلى بۇلۇش ئىمكانىيىتى بارلىقى ھازىرچە ماڭا قاراڭغۇ ، ئەگەر مۇشۇ ئۇسۇلنى سىناپ بېقىشنى خالايدىغانلار بولسا مەن بىلەن كۆرۈشۈڭلار ، مەن سىلەرگە ئۈزەم تەييارلىۋالغان تىزىملىكنى ئەۋەتىپ بىرەي .

بۇ ئۇسۇلنى ئەلۋەتتە ۋىندوۋىس مۇھىتىدىمۇ ئىشلەكىلى بولىدۇ ، ( گەرچە خەتلەر ياخشى بولمىسىمۇ )


   2. ئۇبۇنتۇ مۇھىتىدا تۇرۇپ ، مۇڭداشقۇ قۇرالىنى ئىشلەتكەندە ئانا تىل يېزىقىمىزنى بىمالال چىرايلىق ئىشلىتەلەيمىز .

ئۇبۇنتۇدا ئەڭ ئىقتىدارلىق ، بىزگە ئەڭ مۇۋاپىق كىلىدىغان بىر مۇڭداشقۇ قۇرالىنى ئورتاق ئىشلەتسەك ، دوستلار ئارا ئانا تىلىمزنىڭ قىممىتىنى ھىس قىلغان بۇلار ئىدۇق .

ۋىندوۋىستىن ئايرىلىشنىڭ زۈرىرىيىتى يوق ، پەقەت ئۇبۇنتۇ سېستىمىسىنى ۋىندوۋىس ئىچىگە مەۋھۇم قىلىۋالساقمۇ بۇ نىشاننى ئىشقا ئاشۇرغىلى بولىدۇ ،

ئۇ چاغدا بۇ مۇڭداشقۇ بىزنىڭ ئەركىن - ئازادە پاراڭ قىلىدىغان سورۇنىمىز بوپ قالاتتى .


     3. ئۇبۇنتۇ مۇھىتىدا ئوخشاشلا بىر گوگىل ئىلخەت نامىدا ئۇبۇنتۇ يۇمشاق دىتال مەرگىزىگە تىزىملىتىۋالساق ، كومپىيۇتىرلارنى قەدەمداش قىلىش ئىقتىدارىدىن پايدىلىنىپ ، ئۈزىمىزنىڭ كومپىيۇتىرىغا قاچىلانغان يۇمشاق دىتاللار بىلەن باشقىلارنىڭ كومپىيۇتىرىدىكىنى سېلىشتۇرۇپ تۇرغىلى بولىدۇ . بۇ ئىقتىداردىن پايدىلىنىپ ، ئەڭ ياخشى پىرۇگراممىلارنى ۋاختىدا خەۋەردار بولۇپ ئىشلىتەلەيمىز . مەن بۇنى ئۈيۈمدە ئۈچ كومپىيۇتىردا سىناپ باقتىم .


    4. ئۇبۇنتۇنىڭ كۆڭۈلدىكى بىللە قاچىلانغان مۇڭداشقۇ قۇرالى ئارقىلىق ، پاراڭلاشقىلى بولغاندىن باشقا ، ھۆججەتلەرنى ئۆزئارا يوللىغىلىمۇ بولىدۇ . بۇ ماڭا كۆپ قۇلايلىق تۇغدۇرۇپ بەردى .


    5. ئۇبۇنتۇ سېستىمىسىدا ماڭا ئەڭ ياققان بىر ئالاھىدىلىك قىلماقچى بولغان مەشغۇلاتنى ناھايىتى تىزلا ۋۇجۇتقا چىقارغىلى بولىدۇ ، توردىكى دائىملىق پىرۇگراممىلارنى ئىكران يۈزىگە تىزلەتمە قىلىپ ساقلىغىلى ۋە تۈرلەر بۇيىچە قىسقۇچلارغا ئايرىپ ، رەتلىك ساقلاپ قويغىلى ، تىنىچ ئىجرا قىلغىلى بولىدۇ . مىسال ، ھەقسىز بوشلۇقلار ، تېلۋىزۇر ئۇلىنىشلىرى ، چ چ قۇرالى ، توردىكى ئۈندىدار ، لۇغەت ، ئىملا تەكشۈرگۈچ ، تور ئۇيۇنلىرى ... ۋەھاكازالارغا ئوخشاشلارنى تىزلەتمە قىلىش بەكمۇ قۇلاي .


    6. ئۇبۇنتۇدا تىللارنى ۋە ئىشلەتكۈچىلەرنى خالىغانچە تەڭشىگىلى بولغاچقا ، چەتئەل تىلىدا ئىشلىتىش زۈرۈرىيىتى تۇغۇلسىمۇ ، ئاسانلا ھەل قىلغىلى بولىدۇ .


    7. ئۇبۇنتۇنى بارماق دىسكىغا قاچىلىۋېلىپ ، ئەڭ نۇرمال تەڭشەشلەرنى ماس ھالدا ساقلاپ قويغاندا ، بىر كۈنلەردە سېستىمىنى ياكى كومپىيۇتىرنى ئالماشتۇرۇش توغرا كەلسە ، بىز تەييارلىغان بارماق كومپىيۇتىر بەك ئەسقاتىدۇ . دىسكا رايۇنلىرىنى تەڭشەش ، پىرۇگراممىلارنى ئەسلىگە كەلتۈرۈش ، تەڭشەشلەرنى ئەسلىگە كەلتۈرۈش قاتارلىق زۈرۈرىيەتلەردە بۇنىڭ رۇلىنى سەل قارىغىلى بولمايدۇ .




      ئەزىز قېرىنداشلار ، مەن بۇ يازمامدا ھەقىقى ئىشنىلا سۆزلىدىم ، شەخسى مۇددىئا دىگەنگە ئورۇن قويمىدىم . لېكىن شۇنىڭغا چوڭقۇر ئىشىنىمەنكى ، ھازىر بولمىسىمۇ ئۇزاققا قالماي بىزگە ئوخشاش بېقىندىلىقتا ياشاۋاتقان كىشىلەر ئۇبۇنتۇ بىلەن ۋىندوۋىسنىڭ پەرقىنى كۈرۈپ يىتىدۇ .


      مەن ياخشى نەرسىنى تەۋىسسىيە قىلماسلىقنى شەخسىيەتچىلىكنىڭ جۈملىسىدىن دەپ قارىغاچقا بۇ يازمىنى تەييارلىدىم . يازمامدا قېيىپ كەتكەن مەزمۇنلار بوپ قالغان بولسا توغرا چۈشىنىشىڭلارنى ئۇمىد قىلىمەن .

  قىممەتلىك ۋاختىڭىزنى چىقىرىپ تەھرىرلىۋەتسىڭىز تۇلىمۇ مەمنۇن بولغان بۇلار ئىدىم .
