     مەن باشقىلاردىن ئاڭلىغان تاكىراڭخانىغا مۇناسىۋەتلىك بىر ئىشنى دەپ بېرەي :
     ھەممەيلەن بىلىدىغان كەلپىن دىگەن بىر يۇرت بار ، 2000-يىللارنىڭ باشلىرىدا مۇنچا ، ساتراشخانا ،داپەنجىخانا نامى بىلەن قىزىل چىراقلىق بەزمىخانا ئېچىش مودا بولغاندىن كېيىن ، بىر يولداش بازار تەكشۈرۈپ قارىسا بىر كەلپىن دىگەن ناھىيەدە بىرمۇ بۇنداق يەر يوقكەن ، خوشاللىغىدا ساتراشخانىدىن بىرنى ئېچىپ داغدۇغىلىق تىجارەت باشلاپتۇ ، بىرەر  ئايدىن كېيىن بەزىبىر ياش بالىلار ئاز-تولا كىرىپ چىقىدىغان بولىۋاپتۇدەك . چوڭلار مەسلىھەتلىشىپتۇ ، تاقىتىشقا كۈچ يەتمەيدۇ ، ئۇرۇپ -چاقسا (ئۇلارنىڭ ئاساسەن ئۇچۇق ئاشكارا سېرىق مۇلازىمىتى قانۇنلۇق بولغان بىلەن)قانۇنغا خىلاپ ، ئاخىرىدا بۇلار مۇنداق ئامال تېپىپتۇ ، يەنى ھەر كۈنى مەھەللىدىكى ياشتا چوڭ ئاياللار نۆۋەت بىلەن شۇ  ساتراشخانىدىن يىراق بولىمىغان يەرگىلا چوڭ-كىچىك ئورۇندۇقلىرىنى قويۇشۇپ گازىر چاققاچ پاراڭلىشىپ ئولتۇرىدىغان بولۇپتۇ ،كارامەت ئىش بوپتۇ ،بۇ يەرگە كىرىپ-چىقىدىغانلار قارىسا كۈندىلا چوڭلار بار ،كىرسە چوڭلار كۆرۈپ قالىدۇ ،نىمە دىگەن سەتچىلىك ،ئاستا-ئاستا كىرىپ -چىقىدىغانلارنىڭ ئايىغى ئۈزۈلۈپتۇ ،سودا يوق ساتراشخانىمۇ ئۆزلۈكىدىن تاقىلىپتۇ .
