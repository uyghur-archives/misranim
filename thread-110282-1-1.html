                                             ئايالىم بولسا

                               ئاپتۇر-ھەسەنجان ئوسمان تۈرگۈنى


                                ماڭا كۈيۈپ يۈرگۈدەك ئايالىم بولسا ،
                               ھەرقىسمەتكە كۆنگۈدەك ئايالىم بولسا .
                               كەلسەم ئەگەرسەپەردىن ئىشىك تۈۋىدە ،
                               تاتلىق كۈلۈپ كۈتكۈدەك ئايالىم بولسا .

                              مەن كۆيگەندەك،كۆيگۈدەك ئايالىم بولسا،
                               مىنى بىر دەپ سۆيگۈدەك ئايالىم بولسا .
                                 دۇچار بولسام ئەگەردە پىتنە-پاساتقا ،
                                مىنى پاك دەپ بىلگۈدەك ئايالىم بولسا .

                               باش قويغۇدەك مۈرەمگە ئايالىم بولسا ،
                               باغرىم بېسىپ ياتقۇدەك ئايالىم بولسا ،
                                 ئاتاپ ماڭا بارىنى ساقلاپ قەدرىنى ،
                               مەيدەم كىرىپ يۈرگۈدەك ئايالىم بولسا.

                               خۇلقى تاتلىق قولى چىۋەر ئايالىم بولسا ،
                              گاھ مەسلىھەت ئۆزى چىچەن ئايالىم بولسا.
                                قاراپ شۇئان كۈزۈمگە يىتىپ ھالىمغا ،
                              جۆر بولغۇدەك دەردىمگە ئايالىم بولسا .

                              قۇلقى يۇمشاق ئىتائەتمەن ئايالىم بولسا ،
                              گويا مەندەك شىجائەتمەن ئايالىم بولسا .
                             گاھ يوقنىمۇ چاندۇرماي ئۆتۈپ كەتكۈدەك ،
                             شۇنداق شۈكرى قانائەتمەن ئايالىم بولسا .


                                مىنى تۈۋرۈك بىلگۈدەك ئايالىم بولسا ،
                            گاھ تەۋرەنسەم غەم باسقۇدەك ئايالىم بولسا .
                                  ساقلاپ نازى-خۇلقىنى دىيانىتىنى  ،
                               دائىم گۈلدەك يۈرگۈدەك ئايالىم بولسا .

                                   ئۆز بولساقمۇ ئۇياتچان ئايالىم بولسا ،
                                    يات ئالدىدا خىجىلچان ئايالىم بولسا .
                               ئۆيگە كىرسەم ئىسسىق چاي شۇنداق مىھرىبان،
                                ماڭا شۇنداق كۈيۈمچان ئايالىم بولسا .

                                مېھرى ئىسسىق ھارارەتلىك ئايالىم بولسا،
                                ئۆيگە پىششىق پاراسەتلىك ئايالىم بولسا .
                                   رازى قىلىپ مىنى ھەم ئاتا-ئانامنى  ،
                                دۇئا ئالغان ساخاۋەتلىك ئايالىم بولسا .

                                  يىلدا بىر ئاي رۇزىدار ئايالىم بولسا ،
                                  مەن مېھرىگە قەرىزدارئايالىم بولسا .
                               تۇنۇپ ھەقنى رەببىمنى،سۈيۈپ ئىسلامنى،
                                بەش نامازغا پەرىزدار ئايالىم بولسا .

                                 ئۆزىگە خاس غورورلۇق ئايالىم بولسا ،
                                كۆڭلى يۇمشاق ئىنساپلىق ئايالىم بولسا.
                                يۈز كەلگۈدەك ھەردائىم ۋىجدان ئالدىدا ،
                                شۇنداق مۇسلىم ئىمانلىق ئايالىم بولسا .

                                            مەنبە- ئۆزۈم   


باسقان ھەر بىر قەدىمىم سەن ئۈچۈن .
ھەقىقى ئەركەك ئاياللارنى قۇل قىلمايدۇ ھەم ئاياللارغا قۇل بولمايدۇ.

سۆيمەكچى ئىدىم

                                                  سۆيمەكچى ئىدىم

                                       ئاپتۇر-ھەسەنجان ئوسمان تۈرگۈنى


                                   ئەي تاتلىق مەن سىنى سۆيمەكچى ئىدىم ،
                                     مېھرىڭدە لاۋۇلداپ كۆيمەكچى ئىدىم .
                                     شامال بوپ يىراقلاپ كەتسەڭ ناۋادا ،
                                   پەي بۇلۇپ تەڭمۇ-تەڭ يۈرمەكچى ئىدىم.

                                   ئەي تاتلىق مەن سىنى سۆيمەكچى ئىدىم ،
                                    خالىساڭ خانىشتەك كۈتمەكچى ئىدىم .
                                   گاھ بىلىپ گاھ بىلمەي ئازار بەرسەڭمۇ ،
                                   سەۋرىم بار ھەممىگە كۆنمەكچى ئىدىم .

                                  ئەي تاتلىق مەن سىنى سۆيمەكچى ئىدىم ،
                                   بەختىڭگە جۆر بۇلۇپ كۈلمەكچى ئىدىم .
                                    كەلسە گەر كۈلپەتلە سىنىڭ بېشىڭغا ،
                                  سەن ئۇچۇن شۇ ئوتقا كىرمەكچى ئىدىم .

                                 ئەي تاتلىق مەن سىنى سۆيمەكچى ئىدىم ،
                                  چىكەمگە گۈل قىلىپ قىسماقچى ئىدىم .
                                    بولسىمۇ ھەر گۈلنىڭ ياندا تىكىنى  ،
                                سەن ئۇچۇن تىكەننى تۇتماقچى ئىدىم .

                                ئەي تاتلىق مەن سىنى سۆيمەكچى ئىدىم ،
                                سەن بولساڭ چۆلدىمۇ قالماقچى ئىدىم .
                                سورىساڭ قېنى دەپ مەندىن بىر بۇلاق ،
                                سەن ئۇچۇن مىڭ بۇلاق قازماقچى ئىدىم.

                               ئەي تاتلىق مەن سىنى سۆيمەكچى ئىدىم ،
                               سىنى دەپ نى يىراققا كەتمەكچى ئىدىم .
                               بۇ يوللار بەك ئۇزۇن بولسىمۇ مۈشكۈل ،
                               سەن بىلەن مەنزىلگە يەتمەكچى ئىدىم .

                              ئەي تاتلىق مەن سىنى سۆيمەكچى ئىدىم ،
                               بەخىتنىڭ پەيزىنى سۈرمەكچى ئىدىم .
                              نى يىللار ھېجراندا يۇردۈم تەمتىرەپ ،
                              سەن بىلەن قاقاقلاپ كۈلمەكچى ئىدىم .

                              ئەي تاتلىق مەن سىنى سۆيمەكچى ئىدىم ،
                               يىغلىساڭ يېشىڭنى سۈرتمەكچى ئىدىم .
                                دەردىڭگە بۇ جېنىم بولسا دەرمانە  ،
                             ئەركەكمەن سەن ئۇچۇن ئۆلمەكچى ئىدىم.
