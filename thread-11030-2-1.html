مەن بىرنىمىلەرنى دىمەيلا قۇياي ،لىكىن مەن بىر تىما يوللىغان قەشقەر مەنزىرىسى نامى بىلەن سۇرەتلىك ئىدى ،ئۇزۇمغۇ بىر قاپاق كاللا زادى مۇشۇ سۇرەتلىك فىلىملەرنى قانداق يوللاشنى بىلەلمەيلا كىتىۋاتىمەن ،ھەر قېتىم شۇنداق تىمىلارنى يوللىغاندا باشقۇرغۇچى بۇرادەرلەردىن تۇۋەنچىلىك بىلەن : ئامال بار فىلىم يوللىغاندا بىزدەكلەر يوللىساق تازا چىقمايدىكەن ،شۇڭا سىلەر بۇنى تەھرىرلەپ قۇيساڭلار ،ياكى بولمىسا ئۇگىتىپ قۇيساڭلار ،دىگەننى قىستۇرۇپ قۇيىمەن ،ئامال قانچە بىلمەسلىك ئەيىپ ئەمەس ئۈگەنمەسلىك ئەيىپ تە !!شۇڭا ئۇگىنىۋالايلا دىدىم لىكىن قاملاشمىدى ،شۇنىڭ بىلەن بۇگۇن ئىزىلىپ ئۇلتۇرۇپ قەشقەرنىڭ مەنزىرىللىرىنى يوللىغان  ئۇممۇ يۇق ؟ ياكى ئاتۇشلۇققا قەشقەرنىڭ سۇرەتلىرىنى يوللاش راۋا بولمايدۇ مۇ يە ؟؟ھەممىز باشقا-باشقا يۇرتلاردىن بولغاچ بېرىپ كۇرەلمىسەكمۇ بۇ مەۋھۇم تور دۇنياسىدا بولسىمۇ كۇرۇپ ئازراق بولسىمۇ يۇرتلىرىمىز ئارا چۇشۇنىشنى ئىلگىرى سۇرۇشسەك دىگەن مەخسىدىم بار ئىدى ،مەندىغۇ باشقا مەخسەتمۇ يوق ،شۇ ،ياخشى كۇڭلۇمگە ئىشەك تەپكەندەكلا بىر ئىش بولدى دىسە . لىكىن بۇنى باشقۇرغۇچى خادىملارنىڭ ئۇزىگە ئېلىپ كىتىپ قالماسلىغىڭلارنى تىلەيمەن ،چۇنكى سىلەرمۇ مۇشۇ مۇنبەر ئۈچۈن ئۇن - تىنسىز تەر ئاققۇزۇۋاتىسىلەر ،سىزلەرگە كۇپ تەشەككۇر !!! ھەم بۇ مۇنبىرىمىزنى ئەڭ ياخشى مۇنازىرە مۇنبىرى قىلىپ چىقىشقا كۇچەۋاتىسىلەر ،بۇ كۆز يۇمۇشقا بولمايدىغان بىر پاكىت ،پاكىت ئالدىدا پىل ئۇسۇرۇپتۇ دەيدىغان گەپ بار ئەمەسمۇ ؟؟؟ لىكىن بۇگۇنكىدەك تىمىڭىز چىقمىدى دەپلا پاسسىدە ئۇچۇرىۋەتمەي ،تىمىڭزىنى ئۇچۇرىمىز چىقمىغانلىق سەۋەبى بىلەن شۇڭا بۇنى قايتىدىن بىر تەكشۇرۇپ چىقىدىغىننى يوللاپ بىرىڭ دىگەننى ئەسكەرتىپ قۇيساڭلار ،ئاجايىپ مىننەتدار بوپ قالاتتىم دىمەكچىمەن .
ئادەممەن ئادەمدىن قاچمايمەن يىراق ئادەمنى ئادەمدىن ئادەمدىن ئايرىيمەن بىراق ئىزلەيمەن دۇسلارنى دۇسلار ئىچىدىن چىن يۇرەك قەلبىمدىن ياندۇرۇپ چىراق .
