                                                       ئانا ھايات تۇرۇپ بەر

                                                ئاپتۇر-ھەسەنجان ئوسمان تۈرگۈنى

بۇ شىئېرىمنى مەن ۋە ئاكام ئۈچۈن ،بارلىقىنى ۋە ياشلىقىنى قۇربان قىلىپ ،بىز ئۈچۈن ئۈن-تۈنسىز غەم يەپ كەلگەن، ھەردائىم، ھەر مىنۇت ھەر سىكۇنت بالام بالام ،،،،دەپ چېچىغا ئاق كىرگەن ،مېھىرگە ۋە نۇرغا تولغان ئاشۇ يۈزلىرىگە، بىز ئۈچۈن پەقەت ۋە پەقەت  بىز ئۈچۈن قورۇق چۈشكەن ،جاپاكەش ۋە قىلغان ئەجرى ۋە جاپاسىغا ئەزەلدىن ۋايساپ باقمىغان، بۇ دۇنيالىقتىكى جەننىتىم بولغان ئاشۇ مېھنەتكەش سۈيۈملۈك ئانام [باغدات ياسىن ] غا بېغىشلايمەن، ئانا جاپا چەكتىڭىز ، بىز ئۈچۈن ھەر بىر قىلغانلىرىڭىزغا رازى بۇلۇڭ، ئانا جېنىم ئانا سىزنى سۈيىمەن .  ئاخىرىدا بارلىق ئانىلارنىڭ تىنىنىڭ سالامەت ئۆمرىنىڭ ئۇزۇن بۇلشىنى تىلەيمەن.


                                         مۇمكىن بولسا كىچىكلەپ، يۈرىۋالسام ئەركىلەپ ،
                                          ئاتسام ئانا باغرىڭغا ،ئۇرۇپ قويساڭ پەپىلەپ .
                                         سەنسىز دۇنيا قاراڭغۇ، يۈرمىسەمكەن تەمتىرەپ ،
                                         قوللىرىڭغا سۆيگۈچە ،  ئانا ھايات تۇرۇپ بەر .

                                         گەرخالىساڭ كىچىكلەپ،مىڭىپ بەرسەم تايتاڭلاپ،
                                         يىقىلساممۇ قەستەنگە ، يۆلەپ قويساڭ  ئاۋايلاپ .
                                         ئەسلەپ شۇنداق ئۆتمۈشنى،تىزلىرىمنى سەن سىلاپ.
                                         مەن چېھرىڭگە قانغۇچە ، ئانا ھايات تۇرۇپ بەر .

                                       ئازمايدىغان ئىنسان يوق،خام سۈت ئەمگەن ئەمەسمۇ ،
                                            ئەپۇ قىلغان پەقەتلا ، ئانا بىر سەن ئەمەسمۇ .
                                        سۇئال قويدۇم ئۆزۈمگە ، بۇ تەقدىرمۇ سىناقمۇ ،
                                       شۇنداق ئەقىل تاپقۇچە ،  ئانا ھايات تۇرۇپ بەر .

                                       چۈشەنمەپتۇق قەلبىڭنى،سېغىنىپسەن ئەمەسمۇ ،
                                      قاراپ يولغا يېنىشلاپ ، تەلمۈرۈپسەن ئەمەسمۇ .
                                      شۈكۈر قىلدىڭ يەنىلا ، بىزلەر پەرزەنت ئەمەسمۇ ،
                                      ياشلىرىڭنى سۈرتكۈچە،  ئانا ھايات تۇرۇپ بەر .

                                    تەنبىھ بەرسەڭ چۈشەنمەي ، خاپا بوپتۇق ئەمەسمۇ ،
                                    كوتۇلدايدۇ دەپ ئويلاپ ، قېچىپ يۇردۇق ئەمەسمۇ .
                                    چۈشەنگەندە بىز سىنى ، بۇ كىچىككەنلىك بولماسمۇ ،
                                    مەن مەڭزىمنى ياققۇچە ،   ئانا ھايات تۇرۇپ بەر .

                                    كۆپ بولسىمۇ كەمچىللىك، سەن يۇشۇردۇڭ ئەمەسمۇ ،
                                    ياتقا بىزنى ياخشى دەپ ، كۆتۈرۈپ كەلدىڭ ئەمەسمۇ.
                                    يات ئالدىدا مەن ئەزىز ، چۈنكى سەن بار ئەمەسمۇ ،
                                       تۈزەلگىچە ئىللىتىم ، ئانا ھايات تۇرۇپ بەر .

                                      بالا ئىدۇق بالا بوپ ، زار يىغلاتتۇق ئەمەسمۇ ،
                                   كۆزلىرىڭنى نەملىتىپ ، غەم باستۇردۇق ئەمەسمۇ .
                                  چىداپ كەلدىڭ ھەممىگە،  ئانا ئۇلۇغ ئەمەسمۇ ،
                                  سەندىن دۇئا ئالغۇچە ،  ئانا ھايات تۇرۇپ بەر .

                                   سۆزۈڭ توغرا ھەقىقەت، كۆپنى كۆرگەن ئەمەسمۇ ،
                                     ئىتائەتتە ئىزلىرىم ،   يۇلۇم راۋان ئەمەسمۇ .
                                 ئىشلىرىمدىن ھەممە خوش، ئەل خاتىرجەم ئەمەسمۇ .
                                  نەتىجەمنى كۆرگىچە ،   ئانا ھايات تۇرۇپ بەر .

                                 گاھ غەم باسسا سەنمۇ ھەم ،ئىزتىراپتا ئەمەسمۇ ،
                                ھەر يادىڭغا مەن كىچىپ ، سەن پىغاندا ئەمەسمۇ .
                                  قىرىتقىنى يىل ئەمەس ، بالا دەردى ئەمەسمۇ ،
                                 مەن يەلكەڭنى تۇتقىچە ، ئانا ھايات تۇرۇپ بەر .

                                سۇئال قويسا بەخىت دەپ ، ئانام بەختىم ئەمەسمۇ ،
                                مىنىڭ ئۇچۈن ھەر دەتكە ، ئوتقا كىرگەن ئەمەسمۇ .
                                 تەرىپلىسەم مېھرىڭنى ،  دۇنيا ھەيران ئەمەسمۇ ،  
                                سىنى دۇنيا بىلگۈچە ،   ئانا ھايات تۇرۇپ بەر .

                                 ئاتا-ئانا قەدرىنى ، پەرزەنت تۇنۇپ يىتەرمۇ ،
                                ياكى ئۇمۇ ئۆي تۇتۇپ ، بالىلىق بولسا بىلەرمۇ .
                                 ھىس قىلغاندا ئاشۇ چاغ، پۇشايماندا قالۇرمۇ ،
                                پۇشايمانغا قاچا يوق ،  ئانا ھايات تۇرۇپ بەر .

                               سۆزلەپ بىرەي بالامغا ، بىلسە ياخشى ئەمەسمۇ ،
                               ئۇدۇم قالسا خېسلىتىڭ ، تىخى ياخشى ئەمەسمۇ .
                               قىنى دىسە چوڭ ئانام ،   بۇ سۇئالمۇ ئازاپمۇ ،
                               ئانا يەنەبىر قۇتقۇزغىن ، سەن نىجاتكار ئەمەسمۇ .
                                نەۋرەڭ سىنى كۆرگىچە ،  ئانا ھايات تۇرۇپ بەر .
