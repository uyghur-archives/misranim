ئەسسىلامۇ ئەلەيكۇم قېرىنداشلار ئاللاھتىن ھەممىڭلارنىڭ ئىشلىرىغا بەركەت ئامەت تىلەش بىلەن بىرگە سىلەرنىڭ ھەممە ئىشلاردا ئاللاھنىڭ غەزىپىدىن قورقۇپ ئىنساپ بىلەن ياشىشىڭلارنى ئالاھىدە ئۈمىت قىلىمەن چۈنكى نىمە دىمىگەن بىلەن ئاخىرىدا ئاخىرەتتە ئاللاھتىن ئىبارەت پۈتكۈل كائىنات ئىگىسى ئالدىدا ھىساپ كىتاپ بىرىدىغان گەپ شۇڭا مەيلى بۇ دۇنيادا قانچىلىك مۇغەمبەرلىك قىلىۋېلىىڭ يەنىلا ئۇنىڭ سۇرىقى بار .
بۇ يەردە گەپنىڭ بېشىدىلا ئاخىرەتتىكى سوئال سۇراغدىن گەپ باشلاپ قالغانلىقىمدىن ئەجەپلىنىپ قالغان بۇلىشىڭىز مۇمكىن ئەمدى كىيىنكى بايانلارنى ئاڭلىسىڭىز ياقىڭىزنى تۇتۇپ قالىسىز ئاخىرىدا ئىنسابىڭىز بولسا باشقىلارغا ئىبرەتكە يەتكۈزۈپ قۇيىسىز ئەگەر مىللەت ئۈچۈن كۈيىنىدىغان يۈرەك بولسا كۈچەپ تەشۋىق قىلىسىز .
ئەيدىز -مىللەتنىڭ دۈشمىنى ،بىراق ئەيدىز بۇلۇپ قالغانلار ئۇلار ياكى ئىچ ئاغرىتىپ بولغىلى بولمايدىغان ۋە ياكى ئۇلارنىڭ كىسىلىگە ھەمدەمدە بۇلۇپ تارتىشىپ بەرگىلى بولمايدىغان كىشىلەر توپى بۇلۇپ بۇ خىل كىشىلەر توپى پۇتكۈل دۆلەتلەردىكى بارلىق كىشىلەرنىڭ چۇڭقۇر ئۇيلىنىشىغا تىگىشلىك توپ بۇلۇپ قالماقتا .نۆۋەتتە بۇ توپ بارا بارا ئىنتايىن ئاق كۈڭۈل ئەسلىدە تارقىلىشقا تىگىشلىك بولمىغان كىشىلەر توپىغا قاراپ زەھەرلىك چىشلىرىنى چىقىرىپ يالماپ كەلمەكتە .
بىزنىڭ بىر دۇكان قۇشنىمىز بۇلىدىغان ئۇنىڭ بىر ئالى مەكتەپنى پۈتتۈرۈپ كەلگەن سىڭلىسى بۇلۇپ ئاڭلاشلارغا قارىغاندا ئۇ ئالى مەكتەپتە باشقىلاردىن ئەيدىزنى يۇقتۇرۇپ كەلگەن بۇلۇپ .ئەيدىزنى يۇقتۇرۇش سەۋەبى بەلكىم ئۇنىڭ ئەخلاقسىز قىلمىشىدىن بۇلغان بۇلىشىمۇ ياكى قان ئارقىلىق ۋە ياكى باشقا تەسىر تۈپەيلى يۇققان بۇلىشىمۇ مۇمكىن .مەيلى نىمىلا بولمىسۇ بۇ چىرايلىق دائىم گىرىم قىلىپلا يۈرىيدىغان يۈزىدىن ئۇپا ئەڭلىك كەتمەيدىغان قىز بۇلۇپ ئالى مەكتەپنىڭ بىرىنجى يىللىقىدا ئەيدىز نى يۇقتۇرۇۋالغاندىن كىيىن دائىما داۋالىنىپ دۇرىسىنى يىيىش ئارقىلىق ئۈزىنىڭ بەدەن جەھەتتىكى ئاجىزلىقىنى كىسەللىكىنى يۇشۇرۇپ كەلگەن .شۇڭا ھەۋىسى كۈچلۈك يىگىتلەر ئۇنىڭ چىرايلىقىغا قاراپ شۈلگەيلىرىنى ئېقىتىپ كەتكەن بۇلىشى مۇمكىنكى ئۇنىڭ ئەيدىزلىقىنى بىلەلمىگەن بۇلىشى مۇمكىن  ئۇنىڭ ئەيدىز بۇلۇپ قېلىش ئىھتىمالىنى ئۇيلىمىغان بۇلىشى مۇقەررەر .
ئۇ قىز ئۈزىنىڭ تۈرلۈك ھىيلە نەيرىڭى ئائىلىسىنىڭ ئۇساللىقى تۈپەيلى چ چ دا يىراق يېقىندىكى يىگىتلەرگە رەسىمىنى يۇللاپ يۈرۈپ چىرايلىق ناز كەرەشمىلىك قىلىقى بىلەن بىر يىل ئىلگىرى قاغىلىقتىن بىر ساقچى ئەيدىز بولمىغان يىگىت بىلەن توردا يەنى چ چ دا تۇي قىلىشقا پۈتۈشۈپ يۇرت ئارىلاپ يۈرۈپ چاي ئىچۈرۈشۈپ بولۇنغان قىزنىڭ ئائىلىسىدىكىلەر بۇنىڭدىن قىزنىڭ توي قىلالايدىغانلىقىدىن مەمنۇن بۇلۇپ خۇشال بۇلۇشقانكى ھەرگىزمۇ توي قىلغاندىن كىيىن كىسەل يىگىتنىمۇ يالمايدىغانلىقىنى ئەسلا ئۇيلىمىغان مەقسەت قىزى توي قىلسىلا بولدى دەپ قارىغان .بىراق تورداشلارنىڭ ياردىمىدە (چۈنكى مەن بۇنى مىسرانىم ۋە باشقا بىرقانچە تور بىكەتلەردە تارقاتقان ئىدىم .باشقىلارنى يىگىت تەرەپكە يەتكۈزۈشكە چاقىرىق قىلغان ئىدىم .)يىگىت تويدىن ئىلگىرىكى قىزنىڭ بارلىق يانداپ ئۈتۈپ كىتىش ھىيلىلىرىنى بەربات قىلىپ قىزنى قان تەكشۈرتكۈزۈپ ئۇنىڭ ئەيدىزلىقىنى .ئۇققان .شۇنىڭ بىلەن بىر نەچچە كۈنلا قالغان توي بۇزۇلۇپ بولماس بولغان، يىگىت قۇتۇلۇپ قالغان .
بىراق تۇلىمۇ ئەپسۇس .مانا بىر قانچە كۈندىن بۇيان قوشنىمىز دۇكىنىنى ئاچمىدى ئۇنىڭ دۇكاننى ئاچماسلىقتا يۈرەكنى ئىزىدىغان باشقا يەنە بىر سەۋەپ بولغاچقا باشقىلارنىڭ ئانچە دىققىتىنى تارتمىدى .بىراق ئۇشتۇمتۇت مۇنداق خەۋەرنى ئاڭلاپ قالدىم ئۇ ئەيدىز قىز يەنە ئۇخشاش ئۇسۇل بىلەن چ چ دا قەشقەرلىك ھازىر ئۈرۈمچىدە توپچىلىق قىلىدىغان ئىنتايىن موللام سۈپەت ھايالىق دىيانەتلىك ياۋاش يۇمشاق بىر بالا بىلەن تۇنۇشۇپ ئۇنى ئىندەككە كەلتۈرۈپتىمىش .شۇنىڭ بىلەن قىزنىڭ ئاتا -ئانىلىرى ئالدىنقى قېتىملىق تەجرىبە ساۋاقنى قۇبۇل قىلىپ دوختۇرلۇق تەكشۈرتۈشنى پارا بىرىپ ھەل قىلىپ مانا ئۇلار توي قىلىپ بوپتۇ .ئۇلار ھازىرمۇ ئۈرۈمچىدە بەلكىم توي ئۈرۈمچىدە بولغان ئۇخشايدۇ .مىنىڭچە قىز ئەيدىز بولغان بىر يىگىت بىلەن ياكى ئەيدىز بولسىمۇ ئېچىنمايدىغانراق بىرسى بىلەن توي قىلغان بولسىمۇ ئەپسۇسلانماس ئىدۇق بىراق يىگىت تەرىپلەشكە باشقىلارنىڭ ئاڭلىشىغا قارىغاندا بەكمۇ ئەخلاقلىق ھايالىق ئىنتايىن باي ئۇيغۇر مىللىي ئىگىلىكىمىزدە كەم بولسا بولمايدىغان كەلگۈسىنىڭ كارخانىچىلىرى بولمىش باي ئىكەن .بۇ نىمە كەلگۈلۈك .ئۇلارنىڭ مەخپى ئىشلىرى ھامان ئاشكارا بۇلۇپ مانا بۈگۈن بۇ ئىشنى سىلەرگە يوللاپ تۇرغان بولساقمۇ بىراق بىز كىچىكتۇق .
بىراق شۇنى قەتئىي بىلىپ قېڭلار توي قىلىشتىن ئىلگىرى مەيلى قىز بۇلۇڭ ياكى ئۇغۇل بۇلۇڭ چۇقۇم ھەرقانداق باھانەلەرنى قۇيۇپ دادىل تەكشۈرتۈشنى بىللە قىلىڭ ھەم نەتىجىنى ئىسپات بىلەن ئۈزىڭىز كۈرۈڭ .پەقەت ھايات سىزگە لازىم بولسا چۈنكى قانۇندا  ئەيدىزلارنىڭ توي قىلىشقا بۇلىدىغانلىقى بىراق بالىلىق بولسا بولمايدىغانلىق ئوتتۇرىغا قۇيۇلغان قارشى تەرەپ تەكشۈرگۈچى تۈرلۈك سەۋەپ ياكى بۇ خىل كىسەلنى يۇشۇرىشى مۇمكىن .ئەمما سىزنىڭ ئەيدىز بىلەن بىلىپ تۇرۇپ توي قىلىپ ئۈتكىڭىز بولسا ئۇ باشقا گەپ .ئەگەرۋىجدانىڭىز بولسا بۇنى باشقا بارلىق تور بەتلەردە يوللاپ قۇيۇڭ  .چ چ ياكى چ چ توپى ۋېيشىن -ئۈندىداردا تارقىتىپ قۇيۇڭ .
يەنە ئەڭ چوڭ ئادەمنىڭ ئىچىنى سىيرىيدىغان ئىش . ئۇلارنىڭ يېقىندا دۇكان ئاچماسلىقىدىكى باشقا سەۋەپ .
قوشنىمىزنىڭ ئاچقان دۇكىنى ئەسلى دۇكان ئايال تەرەپنىڭ (ئەر تەرەپ ئەيدىز قىزنىڭ ئاكىسى )بۇلۇپ قىز (ياكى ئەرنىڭ ئايالى )
بۇ دۇكاننى باشقىلا بىلەن بىرلىشىپ ئاچقان سەيپۇڭچىلىق دۇكىنى ئىدى قىز توي قىلىشتىن ئىلگىرى ئۇنىڭ 10نەپەر شاگىرتى بار ئىدى .ئىشى ئىنتايىن ياخشى .دادىسى كىچىكىدىلا قازا قىلىپ كەتكەن ئانىسى كىسەلچان بۇلۇپ 4يىل ئالدىدا قازا قىلىپ كەتكەن ئاڭغىچە قىز شۇ دۇكان بىلەن ئۈينىڭ يۈكىنى كۈتۈرۈپ كەلگەن .2007-يىلى بولسا كىرەك .قىز ناھايىتى ئۇسال  يىگىت ئاتالغان ھازىرقى ئەيدىزنىڭ ئاكىسى بىلەن توي قىلدى .بەلكىم يىگىتنىڭ چىرايى (ئۇ ھەقىقەتەن چىرايلىق )ئۇنى مەپتۇن قىلىۋالغان ياكى شىرىن شىكەر سۈزلىرىگە ئالدانغان بۇلىشى مۇمكىن .
شۇنداق قىلىپ تۇي قىلىپ بىر يىلغىچە يىگىتنى قىز باققان بولدى .بىر يىلدىن كىيىن باشقىلارنىڭ مەسلىھەتى قىزنىڭ قىزىقتۇرىشى ۋە قوللىشى بىلەن سەيپۇڭچىلىق ئىشىنى تاشلاپ دۇكاننى كىيىم كىچەككە ئۈزگەرتتى .دۇكان كىيىم كىچەك دۇكانلىرى ئارىسىدا شۇنداقلا ئەڭ ئاۋات يەردە بولغاچقا ئۇلارنىڭ ئىشى بارا بارا ئاقتى تەرەققىي قىلدى .يىگىتمۇ تىرىشچان ئىدى .
كىيىن ئۇلار بالىلىق بولدى .ئەپسۇس يىگىت نەچچە قېتىم تالاغا قاراپ بۇزۇلدى قىز لىۋىنى چىشلەپ ھەر كۈنى ئەتىگەن كىرىپ كەچتە چىكىتىپ يۈرۈپ بارلىق دۇكان ئىشى، ئائىلە ئىشى،يىگىتنىڭ ئاتا ئانىسىنى بېقىش ،بارلىق ئىشلارنى ئۈزى قىلدى يىگىت قارىشىپمۇ بەرمەيتتى .يىگىتنىڭ ئاتا ئانىسى شۇنداق قىلسىمۇ قىزدىن پەقتلا رازى ئەمەس ئەڭ ئېچىنىشلىق قىز قۇرسىقىدا بار ۋاقىتلاردىمۇ چوڭ ئۈينىڭ نېنىنى ئۈزى يېقىپ قىلاتتى قېيىن ئانا تۇنۇر بېشىغا يېقىن يۇلاپ قالمايتى .بۇ ھەقىقەتەنمۇ ئېغىر كىلىدىغان ئىش دە-ھازىرقى بەزى قىزلار ئۇ قىلغاننىڭ ئۇن مىڭنىڭ بىرىنىمۇ كۈتىرەلمەيدۇ.بەلكىم قىز بۇنداق ئىشلارنى ئۈز بىر تۇققانلىرىمغا دىسەم ئۇلاغا ئېغىرچىلىقمنى سالسام ئۇلار ئۈزىنىڭ تۇرمۇشىنى تارتىشارمۇ ياكى مىنىڭمۇ مەيلىلا كۈتىرەي چىداي ئۇڭلىنىپ قالار دەپ ئۇيلىغان بۇلىشى مۇمكىن .
بىراق يىگىت بۇزۇلۇپ ئۇ قىزدىن تارتىپ ئەكەلسە بىۇنىڭ بىلەن ئاپاق چاپاق بۇلۇپ ھېلى ئۇنىڭ بىلەن ھېلى بۇنىڭ بىلەن ئاپاق چاپاق بۇلۇپ يۈرۈپ قىزنىڭ قەلبىنى بەكلا يارا قىلدى .
بىراق قىز چىدىغانغىلا ئىش چىرايلىق يېپىق ئۈتۈپ كەتسە مەيلىدى بىراق ئۇنداق بۇلماي يىگىتنىڭ خۇرىكى ئۈسۈپ .
كىيىن يەنە بىرىنى تېپىپ مۇشۇنى ئالىمەن سەن ئۈيدىن چىق دەپ قىزنى ئۇرۇپ چىقىرىۋېتىپتىمىش .ئۇكىيىن بەدىنىنى باشقىلارغا كۈرسەتكىنىدە بىز بۇ ئەھۋالنى تۇلۇق بىلدۇق .بەدىنىدىكى يارىدىن ھەر قانچە باغرى تاش ئادەممۇ يىغلاپ سالار دەپ ئويلايمەن .
قىزنىڭ ئائىلىسىنى بۇزغۇچى تەرەپ قىزنى كۈرسىلا تىللاپ تالا مۈشۈكى ئۈي مۈشۈكىنى قوغلاپ يۈردى .ئەڭ ئېچىنىشلىق ئىش شۇكى  قىز ئاجرىشىپ كىتىشنى لايىق كۈردىسىمۇ .بىراق ئۇنىڭغا بىرسىنىت ئايرىپ بىرىلمەيدىكەن -(يىگىتنىڭ شەخسى قارىشىچە )تېخى شۇمۇ يەتمىگەندەك قىزغا بىرمۇنچە قەرىز ئارتىلىپتىمىش -ئۇلارنىڭ دىيىشىچە قىز يىگىت بىلەن تۇي قىلىپ يىگىتنى بىرمۇنچە قەرىزگە بۇغۇپ قۇيۇپتۇ مىش -بىراق ئەمەلىيەت دۇكان ئەتراپىدىكى بارلىق كىشىلەرگە ئايانكى قىز بەكلا ھايالىق ئەخلاقلىق بۇلۇپ بارلىق تىجارەت ئىشلىرىنى ئاساسەن ئۈزى قىلاتتى .يىگىت پەقەت مال ئەكىلىپ بەرسە ئەكىلىپ بىرەتتى بۇلمىسا ئۇنىمۇ قىز ئەكىلەتتى .پۈتۈن ئەتىگەن كەچ يىگىت ئاغىنە دەپ( پۇل تاپقاندا ئالغان ماشىنا بىلەن )چۈرگىلەپ يۈرەتتى -شۇنى ئالاھىدە دىيىش كىرەككى قىز يىگىت بىلەن تۇي قىلىش ئالدىدا يىگىتنىڭ ئاددىيسى بىر ئادەتتىكى پۇت بىلەن تىپىپ ھەيدەيدىغان ئۈچ چاقلىق ۋېلىسپىتقىمۇ پۇلى يۇق دىگۈدەك بۇلۇپ تۇي قىلىپ بىرەر يىلغىچە قىز يىگىتنىڭ توي قەرزىنى تۈلىگەن يىگىتنى بېيىتقان مۇتۇ كىيىنچە ماشىنا ئالغان .
بىراق قىز تەرەپنىڭ ھايالىقلىقى سەۋەبىدىن يېقىندىكى رامىزاندىمۇ ئۇلار قىزنىڭ ئۈيىگە چىقىپ ھەركۈنى ئادەم باشلاپ چىقىپ ئىشىك ئالدىدا قالايمىقان تۇۋلاپ تىللاپ يۈرگەن بىراق قىز تەرەپ رامىزاندا ئۈلۈپ كەتكۈدەك بولسىمۇ بىر نەرسە دىيىشمەيمىز دەپ سەۋىر قىلغان بىراق ئۇلار چىداپ چىداپ بۇلالماي .ھىيتتىن كىيىنكى -20كۈنلىرى بولسا كىرەك قاتتىق ئۈرۇشۇپ كەتتى .
پارا بەرسە ھەممىنى ئاستىن ئۈستۈن قىلىۋېتىدىغان پارىخورلار قاراپ تۇتۇپ شۇنچە ئىش تورسىمۇ يىگىت تەرەپكە يان بېسىپ تۇرغان .
قېنى دەپ بېقىڭلار قىزنىڭ دۇكاندا ھەققىي يوقمۇ .ئاجراشقاندا مال مۈلۈك قانداق ئايرىلىش كېرەك ياكى پارا بەرگەندىن كىيىن ھەممە ئاستىن ئۈستىن بولسا مەيلىمۇ .ئەڭ ئېچىنىشلىقى قىز ئۈزىنىڭ ھەققىي بار شۇ دۇكاننى باشقىلار ئالىمەن دىگەن دىگەن باھادىن (بۇ نوچى يىگىت دۇكاننى ئېچىشنى ئېغىر كۈرۈپ ئۈزىچە ساتماقچى بولغان .)نەچچە ئون مىڭ يۈۋەن يۇقىرى باھادا ئالاي دىسىمۇ بەرمىگەن ئەمەلىيەتتە دۇكاننى ئالسا مېلى بىلەن قۇشۇلۇپ 25مىڭ 30ئەتراپىدا مال بار بۇلۇپ نەرقى شۇنچىلىك بىراق دۇكاننى قىز تەرەپ مەيلى سەن بىلەن سەتلىشىپ يۈرگىچە ئىككى بالامنى باقمىسام بولمىغاندىكىن مەنلا ئالاي دىسە 55مىڭغىمۇ بەرمىدى ئۇنىڭغا ساتماشمىش .-بۇ نىمە دىگەن ناھەقچىلىك .
مەنچە بولسا ئەيدىز قىزنى ئۇنىڭ زۇمىگەر ئاكىسىنى ھايانى بىلمەيدىغان يىڭىتنىڭ ئاتا ئانىسىنى كۈيدۈرۈپ ئۈلتۈرسىمۇ ھەممىمىزنىڭ دەردىي چىقمايدۇ .ئۇ ھەممىگە زومىگەرلىك قىلىپ خەقنى جاق تويغۇزغان ھەممىمىز قارا يىتىم قىزغا ئىچ ئاغرىتىمىز . يەنە نۇرغۇن ئىشلارنى سىلەرگە دىگۈم كەلدى .بۇ يازمىنى باشقىلارغا تەشۋىق قىلىڭلار ئىبرەت ئالسۇن بۇلۇپمۇ ئەيدىزنىڭ ئىشىنى بەكرەك تەشۋىق قىلىڭلار .
قېنى تۈۋەندە ئۇيلىغىنىڭلارنى دەپ بېقىڭلار .
ئالدىراپ خەتلىرىم ئىملادا مەسىلە بار ھالەتتە يوللىنىپ قالدى توغرا چۈشىنىڭلار . بۇ يازمىنى ئاخىرىدا   Misranim2 تەھرىرلىگەن. ۋاقتى  2013-9-12 05:38 PM
