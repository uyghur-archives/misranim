مانا كۆزنى يۇمۇپ ئاشقىچە مىنىڭ ئاشۇ مىھرى ئىسسىق كىندىك قىنىم تۈكۈلگەن ئەزىز يۇرتۇم شىنجاڭدىن ئايرىلىپ  جاھاننىڭ ئۇ چىىتىدىكى مەن ئۈچۈن ناتۇنۇش بولغان بۇ زىمىندا ياشاۋاتقىنىمغا ئاز كەم بىر يىلچە ۋاقىت بوپتۇ، گەرچە بۇ يۇرۇتتا ئۆزۈمنى  تەنھا ھىساپىلىمىساممۇ  يۇرتۇمنى سىغىنىش تۇيغۇسى يەنىلا قەلبىمدە  مەۋىج ئۇرۇپ  تۇرماقتا،مەن تۇرۋاتقان بۇ جاي بولسا نامىنى ئاڭلىغانلىكى كىشىنىڭ ھەۋىسىنى كەلتۈردىغان ئاسىيادىكى داڭلىق ساياھەت شەھەرلىرىنىڭ بىرى سانىلدىغان دۈبەي شەھرى، ئويلاپ باقسام مەن بۇ كىشىنىڭ ھەۋىسى كەلگۈدەك گۈزەل ماكاندا بىر يىلچە ياشاپتىمەن،بۇ كۈنلەردە مىھرىبان ئاپام ۋە سەبىي ئۇكىلىرىم ماڭا ھەمراھ بۇلۇپ تۇرمۇش كەچۈرگەچكە  ناتۇنۇش ئەلنىڭ ماڭا ئاتا قىلغان مۇساپىرلىق  تۇيغۇسىنى ئانچە ھىس قىلىپ كەتمىدىم، خۇداغا شۈكۈر  يەنىلا يۇرتۇمدىكىدەك كىيىملىرىم پۈتۈن قورسىقىم توق يۇرىۋاتىمەن، ئەمما يات ئەل دىگەن يەنىلا يات ئەلكەن يۇرتتىكى چاغلىرىمدا چەتئەل دىسە قانداق يەرلەردۇر قانچىلىك  گۈزەلدۇر دەپ ھەۋەسلىنىپ كىتەتتىم ھازىر بىلسەم ئۇ چاغلاردا مەندە پايدا بولغان ھىسيات پەقەت قىزقىش تۇيغۇسىكەن ،توغرا ئىنساندا قىزقىش تۇيغۇسىنىڭ بۇلىشى نۇرمال ئەھۋال لىكىن بىر نۇقتىنى ئەمدى چۈشۈنۈپ يەتتىمكى نۇرغۇن ۋاقىتلاردا بىزدىكى قىزقىش ۋە ئىنتىلش تۇيغۇسى رىيانلىق بىلەن بىرىكمەيدىكەن  سىز پەقەت رىيانلقنى بىشىڭىزدىن كەچۈرگەندىن كىيىن سىز ئويلاپ يەتكەن ئىشلار ۋە قىزقىشلىرىڭىزنىڭ سىز ئەمەلىيىتىڭىزدە كۆرگەن بىشىڭىزدىن كەچۈرۋاتقان رىيانلىق بىلەن بىردەك ياكى بىردەك ئەمەسلىكىنى ھەقىقى تۈردەچۈشۈنۈپ يىتەلەيدىكەنسىز،ئۆزۈمنى ئىلىپ ئىيىتسام بۇ قىسقىغىنە بىر يىل يات ئەلدە تۇرمۇش كەچۈرۈش جەريانىدا شۇنى ھىس قىلىپ يەتتىمكى كونىلارنىڭ‹‹ئۆزگى يۇرتتا سۇلتان بولغىچە ئۆز يۇرتۇڭدا ئۇلتان بول›› دىگەن سۆزى بىكار ئىيتىلمىغانكەن،ئىلگىرى مەن بۇ ئەلگە بىر نەچچە قىتىم كىلىپ ساياھەتتە بولغان ھازىر بىلسەم ساياھەت بىلەن تۇرمۇش كەچۈرۈش ئىككىسى تۈپتىن ئايرىۋىتىلگەن ئىككى مەسلىكەن ،ساياھەتتە سىز پەقەت بۇ گۈزەل ساياھەت مەنزىلگاھىنىڭ گۈزەللىكىدىن، پاكىز مۇھىتىدىن ھۇزۇرلىنىپ  ئۆزگىچە ساياھەت ئورۇنلىرىنى ئىكىسكۇرسىيە قىلىپ تۇرمۇشنىڭ سىزگە ئىلىپ كەلگەن ھەر تۈرلۈك بىسىمىدىن يىراقلىشىپ ئازىراق بولسىمۇ كۆڭلىڭىزنى ئاچالايدىكەنسىز، ئەمما مۇشۇ يەردە تۇرۇپ كۈن ئالماقچى بولغان ۋاقتىڭىزدا تەبىيلا تۇغۇلۇپ ئۆسكەن يۇرتىڭىزنى ئەسلەپ قالىدىكەنسىز،سىز بۇ يەردە دۇچ كىلىدىغان مەسىلە ۋە سىناقلار بىلسىڭىز يۇرتىڭىزدا تۇرمۇش كەچۈرگەندىن كۆپ مۇرەككەپ بۇلىدىكەن،دەسلەپكى كەلگەن چاغلىرىمدا راستىمنى ئىيىتسام ئاپام ۋە ئۇكىلىرىمنىڭ ھەمرالىقىدا بۇ نۇختىنى تازا چۈشۈنۈپ يەتمەپتىكەنمەن سەۋەبىي يىڭى كەلگەن ۋاقىتلىرىم بولغاچقىمىكىن بۇ يەردىكى ھەممىلا نەرسە مەن ئۈچۈن يىڭى سانالغاچقا قىزقىش تۇيغۇسى ئىچىدە يۇرىتنى سىغىنىش تۇيغۇسى مەندە ئۇنچىۋالا كۈچلۈك بولماپتكەن، ئەمما ۋاقىتنىڭ ئۈتىشى بىلەن بۇ دۇنياغا داڭلىق ساياھەت شەھرىنىڭ مەن بارمىغان ساياھەت مەنزىلگاھلىرى ئاساسىي جەھەتتىن قالمىدى،قىزقىشلىرىمغىمۇ جاۋاپ تىپىلدى، مەن ئەمدى بۇ ئەلنىڭ تۇرمۇشىغا كۈنمىسەم بولمايتى تاماق،ئۆرۈپ-ئادەت،ھاۋا كىلمات ھەممىسى ئاساسەن ئۆزگەرگەن بىر يىڭىچە ياشاش مۇھىتى كۆز ئالدىمدا نامايەن بولدى ،ئەمما بۇنىڭغا كۈنمەي ئامالىم يوق مىنىڭ بۇ خىل مۇيىتقا كۈنىشىم ماڭا نىسپەتەن بىرشەرىت ئىدى،تاماق جەھەتتىن ئاپام بولغاچقا ئانچە قىينالمىدىم، يۇرتنىڭ تائاملىرىنى ئاپام ئىتىپ بىرىپ تۇردى ئەمما ئۇكىلىرىم مۇشۇ يەردە ئۇزۇن تۇرغانلىق سەۋەپلىىك مۇشۇ يەرنىڭ يەرىلىك تاماقلىرىغا كۈنۈپ كەتكەچكە بەزى چاغلاردا بىر مىنى دەپ پولو لاغمەن دىگەنلەرنى ئىتىشنىڭ ھاجىتى يوقتە! بۇ يەرلىكلەرنىڭ يەرىلىك تائاملىرى ئاساسىي جەھەتتىن گۈرۈچ ۋە گۆشنى ئاساس قىلىپ ئىتىلدىكەن،دىمىسىمۇ يىلدا بىرەر قىتىم يامغۇر ياغمايدىغان  5،6-ئاي مەزگىللىرى ھاۋا ئىسسىپ 50گىرادۇسقا يىقىنلىشىدىغان بۇ ئەلدە ئەگەر گۆش ئاساسىي ئىستىمال قىلىنمىسا ئىسسىقتا ئۆرە تۇرۇپ ياشىغىلى بۇلامدۇ دەيسز! ئاستا-ئاستا بۇ يەرنىڭ تاماقلىرىغىمۇ كۈنۈپ كەتتىم لىكىن كۆڭلۈمدە يەنىلا ئۆز يۇرتۇمنىڭ تائاملىرىغا تەڭ قىلغىن كەلمەيدۇ پەقەت مۇھىت مىنى ئۆزگەرتتى خالاس،ئاندىن بۇ يەردە ماڭا قۇيۇلغان يەنە بىر سۇئال ھاۋا كىلماتقا قانداق كۈنۈش بولدى مەن تۇرۋاتقان بۇ ماكاندا سىز پەقەت 2پەسىلنىلا ئۇچۇرتالايسىز بىرى ياز，بىرى باھار
كۈز ھەم قىش بۇ يەردە يوق، مەن دەپ ئۆتكەن باھار پەسلىمۇ يامغۇرسىز بۇلىدۇ پەقەت ھاۋا كىلمات سەل ئىللىپ كەچلىرى سىرىتقا چىقىپ قالسىڭىز ئۈستىڭىزگە نىپىز پۇپايكا ئارتىۋالىسىز ،مەن بۇ يەرگە قادەم باسقان بىر يىل جەريانىدا پەقەت بىرلا قىتىم يامغۇرنى كۆرەلىدىم شۇندىلا ئۆز يۇرتۇمنىڭ تۆت پەسلىنى ھەقىقىي سىغىندىم ،ھەممىلا ئۆيدە ھاۋا تەڭشىگۈچ، بۇ يەردە كۈندىزى سىرىتتا ئادەم ئۇچراتقىلى بولمايدۇ  خىزمەت قىلىش ئىھتىياجى بولمىسا بۇ يەرنىڭ كىشلىرى ئاساسەن ئىسسىق ئۈتۈپ قىلىشتىن ساقلىنىش ئۈچۈن سىرىتقا چىقمايدۇ  ئەمما كىچىلىرى كۇچىلارغا سودا -سارايلارغا قارايدىغان بولسىڭىز ئادەملەرنىڭ ئىنتايىن كۆپلىكىنى كۆرىۋالالايسىز مانا بۇ مەن ياشاۋاتقان مەن ئۈچۈن يات سانىلىدىغان ئەلنىڭ ئاددىي تۇرمۇش ئادىتىي ،مەن بۇلارنى كۆرۈپ چۈشۈنۈپ يەتكەندىن كىيىن  يات ئەلنىڭ مۇھىتىغا كۈنمەكنىڭ قانچىلىك قىيىنلىغىنى بىلدىم ۋاقىت ھەممىنى ئۆزگەرتىدۇ دەپ قۇيىدىكەنمىز ئەمما شۇ ئۆزگىرىش جەريانى باشتىن كەچۈرمەك  ھەقىقەتەن ئاسان ئەمەسكەن،  لىكىن بۇ ۋاقىتمۇ ئۆزگەرتەلمەيدىغان بىر نەرسە باركەن ئۇ بولسىمۇ تىل مەسلىسىكەن، سىز ھەر قانداق بىر يات ئەلگە بارسىڭىز شۇ يەردە ياشاۋاتقان مىللەتنىڭ تىلىنى بىلمىسىڭىز گاچىدىن پەرىقىڭىز قالمايدىكەن ئۇ مىللەتنىڭ مەدىنىيتى شۇنداقلا ئۆز -ئارا كىشلەر بىلەن پىكىر ئالماشتۇرىشىڭىزغا سىز ئويلاپ باقمىغان توسالغۇلارنى ئىلىپ كىلىدىكەن مەن بۇ يەردە تۇرۇپ يۇلۇققان ئەڭ چوڭ توسالغۇ دەل تىل توسالغۇسى بولدى، ھىلىمۇ ياخشى ئاپام يىنىمدا بۇلۇپ قاپتۇ بولمىسا مەن راستلار بۇ يەردە گاچىنىڭ ئۆزى بۇلاتتىم سىز ساياھەتكە چىقسىڭىز ساياھەت يىتەكچىسى بەلكىم سىزنىڭ تىل جەھەتتىكى يىتەرسىزلىكىڭىزنى چاندۇرمايدۇ ئەمما سىز بىر يات ئەلگە ساياھەت ئۈچۈن ئەمەس بەلكى تۇرمۇش كەچۈرۈشكە كەلگەن ۋاقتىڭىزدا تىل سىزنى ھەقىقەتەن قىيناپ قويىدىكەن ،ئويلاپ باقسام بىزنىڭ ئانا تىل مائارىپ دەرىسلىكىدە ئنگىلىستىلى يوق ئەمەس ئەمما سىستىمىلاشمىغان ئاساسىي جەھەتتىن ئىغىز تىلى بۇلۇپمۇ باشلانغۇچ دەرىسلىكىدە شۇنداق، ئەمما بىز خەنزۇ تىلى سىنىپىنى پۈتتەگەن بالىلارغا قارايدىغان بولساق  ئۇلار باشلانغۇچ باسقۇچىدىلا ئىنگىلىسچە گىراماتىكىنى پىششىق ئىگەللەپ ھەر قانداق ئىنگىلىسچە خەتنى كۆرسە مەنىسىنى چۈشەنمىسىمۇ ئوقۇيالايدىغان سەۋىيەنى ھازىرلىغان بۇلىدۇ، ئەمما بىزدىچۇ؟ھىلىمۇ ياخشى ئۆزۈم قىلچىلىك بولسىمۇ ئۈگۈنۈپ قۇيۇپتىكەنمەن شۇنىڭ زەرىچىلىك بولسىمۇ پايدىسى تەڭدى،ئۇنىڭغا قۇشۇلۇپ ئائىلىدە ئۇكىلىرىم ئىنگىلىستىلىدا سۆزلىگەچكە قۇلاققا سىڭىپ قالىدىكەن لىكىن ئاساسىمنىڭ يوقلىقى تۈپەيلى ھازىرغىچە كىچىك سۆزلەردىن چىقالىغىنىم يوق،ئۆزۈمنىڭ ئانا -تىلدا  پىكىر ئالماشتۇرىدىغان ئادىمىم پەقەت بىر ئاپام ،كىم دىگەن بولغىتى ئاشۇ ‹‹تىل قەلىپنىڭ ئەينىكى ›› دىگەن بىر ئىغىز سۆزنى ھەققەتەن ھەق ئىيتىلغان سۆزكەن، كۈنلەرنىڭ ئۈتىشى بىلەن بۇ يات ئەلنىڭ ئىجىر -بۈجىرىگىمۇ كۈنۈپ قاپتىمەن ، ئەمما توساتتىنلا قەلبىم دولقۇنلىرىدا بۇ يۇرتنى ئۆز دۆلىتىم  ئۆز يۇرتۇم  بىلەن سىلىشتۇرۇش پىسخىكىسى مەندە شەكىللىنىشكە باشلىدى بۇ قانداقتۇر مىنىڭ تۇغۇلۇپ ئۆسكەن ماكانىمنى كەمسىتكىنىم ياكى تۈۋەن كۆرگىنىم ئەمەس بەلكى مەندە پەيدا بولغان نۇرمال پىسخىك ھالەت ،مەن ئالدى بىلەن بۇ يەردە ياشاش جەريانىدا مۇھىتقا ئالاھىدە دىققەت قىلىپ قالدىم  بۇ يەرنىڭ يىشىللىق بىلەن پۈركۈنۈش نىسپىتى يۇرتۇمنىڭكىدىن خىلىلا ئىشىپ چۈشىدىكەن، يول ياقىلىرىدىكى ھەر خىل دەرەخلەر قەدەمدە بىر ئۇچرايدىغان چۆپلۈك ، چوپلۈكتە ئارام كۈنلىرى بىر ئائىلە كىشلىرى بالىلارنى ئىچىقىپ ئوينىتىدۇ  بۇ خىل مەنزىنى كۆرگەن كىشىنىڭ كۆڭلى سۈيۈنمەي قالسۈنمۇ؟ تىخى چۆپنىڭ پاكىزلىقىچۇ ياۋىلىرى قىرقىۋىتىلگەن، رەتلىك  شۇنچىلىك پاكىز  سىز ئايىغىڭىزنى سىلىۋىتىپ ئاق پايپاقتا ماڭسىڭىزمۇ  پايپىكىڭىز يەنىلا پاكىز تۇرىدۇ ،ئەكىسچە يۇرتۇمنى ئۆيلىسام ئىچىم سەل ئاچچىق بۇلىدۇ يىشىللىققا قاپلىنىش نىسپىتى بۇرۇنقىدىن كۆپ ئۆستۈرۈلگەن بولسىمۇ  يۇرتۇمدا پاكىز بىرەر چىملىقنى  تاپماقمۇ ئادەمگە سەل تەس توختاپ قالىدۇ ،قارىساق چۆپلەرنىڭ ياۋىلىرى ئۈسۈپ كەتكەن ھىچكىمنىڭ ئۇنىڭ بىلەن كارى يوق ،بەزى بىر ساپاسىز كىشلىرىمىز ئۇ يەردە ئولتۇرۇپ يىگەن ئىشكەنلىرىنى بىراقلا تۈكىۋاتقان مىنىڭ تىلغا ئالغىنىم پەقەت جەمىيىتىمىزدىكى بىر قىسىم كىشلەرنىڭ قىلمىشلىرى،مەن ئويلايمەن مىنىڭ يۇرتۇم شىنجاڭدەك گۈزەل يەر جاھاندا يوق مەسىلە مۇھىت
