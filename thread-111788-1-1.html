مۇھەببەت بەزىدە سىنى خوشاللىقتا كۈلدۈرۈپ ساڭا بەخت بەرگەن بولسا،بەزىدە سىنى ئازاپلاپ ،سىنىڭ  قەلبىڭنى ئۆلتۈردۇ،ئويلايسەن مەن ئۇنى ھەقىقىي ياخشى كۆرسەملا بولدى،ئۇ كىيىن مىنى ياخشى كۆرۈپ قالىدۇ دەپ ئۇنىڭ ئۈچۈن بارلىقىڭنى ئاتىۋىتشكە تەييار تۇرسەن ئۇنىڭ ئۈچۈن كۆپ قۇربانلىق بىرىسەن ،بىراق ئۇنىڭ سىنى ئالداۋاتقانلىقى سەندىن باشقا قىزلار بىلەن شىرىن-شىكەر سۆزلەرنى ئىيتىشىپ،مۇڭدىشىپ،ساڭا بەرگەن شۇ ۋەدىلەرنى باشقىلارغىمۇ بىرىۋاتقانلىقىنى بىلمەيسەن ،ھەرقىتىم ئۇنى كۆرسەڭ قەلبىڭ ئالەمچە شادلىققا تولىدۇ،ئۆزۈڭنى بۇ دۇنيادىكى ئەڭ بەخىتلىك مەلىكىلەردەك ھېس-تۇيغۇلاردا بولسەن،لىكىن مۇھەببەتنىڭ خوشاللىقى شىرىنلىكىلا بولماي ئۇنىڭ يەنە ئازاپ-ئۇقۇبەتلىرى بار،بەلكىم بىر كۈنلەردە ئۇنىڭ ئۈندىدارىدا ياكى چچى بۇشلۇقىدا باشقا قىزلار بىلەن شۇنچە شىرىن-شىكەر سۆزلەرنى يىزىشقانلىقىنى كۆرۈپ يۈرىكىڭ ئازاپ ئىچىدە ئۆرتىنىدۇ،كۆز ئالدىڭ قاراڭغىلىشىدۇ،شۇ مىنۇتلاردا سەن ھەرگىزمۇ ئۇنىڭ باشقا قىزلار بىلەن پاراڭلاشقىنىغا،ساڭا ساداقەتمەن بولمىغانلىقىغا ئەمەس،بەلكى ئۇنىڭغا بەرگەن مۇھەببىتىڭنىڭ تىخى يىتەرلىك بولمىغانلىقى،ئۇنىڭ قەلبىنى تۇلۇق چۈشۈنۈپ بولالمىغانلىقىڭ ئۈچۈن ئازاپلىنسەن ھەم ۋازكەچكۈڭ كىلىدۇ،خۇددى سەن ۋاز كەچسەڭ ئۇ بەختىنى تاپالايدىغاندەك سەن ئارتۇقچىدەك ،سىنىڭ ئۇنىڭ ئۈچۈن قىلغانلىرىڭ تىخى تۇلۇق ئەمەستەك ،يۈرەكلىرىڭ پۇچىلىنىدۇ،ئىچىڭدە كاشكى رەببىم ياراتقان ھەقىقىي ھالال جۈپتىلەردىن بولالىغان بولسام ،بىرلىكتە تىرىشىپ،ساداقەتمەن بۇلۇپ قەلبىمىزنى شادلىققا تولدۇرۇپ ئىككلا دۇنيا رەببىمىز ئەڭ سۆيگەن،ياخشى كۆرگەن ئەڭ بەختىيار بىر چۈپتىلەردىن بولالىغان بولساقچۇ دەپ يىغلايسەن،يەنە سەۋىر قىلىپ ئاخىرقى نەتىجىسىنى كۆرگىڭ كىلىدۇ،ئۇنى خاتا چۈشۈنۈپ قىلىشتىن،ئوتتۇراڭلاردىكى مۇناسىۋەتكە تەسىر يىتىشىدىن قورقۇسەن،توغرا مۇھەببەت مەڭگۈ چۈشىنىپ بولغىسىز بىرخىل تۇيغۇ،مۇھەببەت ئىككى ئادەمنىڭ تىرىشچانلىقى ،سەمىمىيلىكى،ساداقەتمەنلىكى بىلەن تەرەققىي قىلىدۇ،ئەبەدىيلىك بولىدۇ،دۇنيادا ھىچ نەرسە مۇكەممەل بولمايدۇ،نۇرغۇن كۈلپەتلەر ،تۈرلۈك سىناقلار بىزنى تاۋلايدۇ،بىزنىڭ بىر-بىرمىز ئالدىدىكى ھەقىقىي ماھىيتىمىزنى ئاشكارلاپ بىرىدۇ،بىر قىز ھەممىگە بەرداشلىق بىرىدۇھەم چىديدۇ،بىراق ئۆزى چوڭ كۆرگەن ئادىمىنىڭ ئۆزىنى ئالدىشىدەك بۇ خىل خاتالىقنى كەچۈرەلمەيدۇ،ھەم چۈشىنەلمەيدۇ.

      بىر قىز ماڭا نىمىلەرنى ئاتا قىلالايدۇ دەمسەن ؟ بىر قىز ساڭا سەمىمىيلىكنى،ئىشەنچنى ،بىخەتەرلىك تۇيغۇسىنى،ساداقەتنى،بەخىتنى بىرەلەيدۇ،ئۆزىنىڭ قەلىب گۈلشەنىدىن ساڭا بىر ئورۇن بىرەلەيدۇ،خۇددى يىنىپ تۇرغان بىر شامدەك بىر ئۆمۈر سەن ئۈچۈن ئىرىپ تۈگەشكە رازى بولىدۇ، خىزمىتىڭدە ساڭا مەسلىھەتچى،تۇرمۇشىڭدا ھالال چۈپتىڭ،جاپا-مۇشەققەتلەردە ھەمراھىڭ،يالغۇزچىلىقىڭدا سىردىشىڭ بولالايدۇ،رەببىڭ ئەگەر ئۇنى ساڭا نىسىپ قىلغان بولسا ئۇ سىنى گوياكى ئۆزىنىڭ تىغىدەك چوڭ كۆرۈپ ساڭا بىر ئۆمۈر ساداقەتمەن بولۇپ ئىككىنىچى بىر ئىنسانغا قارىمايدۇ،رەببىڭ ساڭا بىرىدۇ،بەرگەندىمۇ باشتا ئەڭ ياخشىسىنى،ئەڭ مۇنەۋۋەرسىنى بىرىدۇ،قىنى مەن بەرگەن بۇ بەخىتنى چىڭ تۇتالامدۇ،قەدىرلىيەلەمدۇ دەپ،ئەگەر قەدىرلىيەلمەيكەتكۈزۈپ قۇيساڭ ،ئۇنى سەندىن تاتىۋىلىپ ھەقىقىي قەدەرلەيدىغان كىشىگە بىرىدۇ،كىيىن ساڭا ھەرگىز ئاۋالقى بەخىتنى  ئامەتنى بىرىشى ناتايىن،

   قىنى سەن دەپ باققىنە سەن ئۇ قىزغا نىمىلەرنى بىرەلەيسەن؟
