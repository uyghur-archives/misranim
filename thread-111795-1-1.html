ئۇيغۇر خەلقى قائىدە -يوسۇننى تەلىم -تەربىيەنىڭ مۇھىم مەزمۇنى دەپ قارايدۇ .ئۇلار قائىدە يوسۇنلۇق ئادەمنى تەربىيە كۆرگەن ئۇقۇمۇشلۇق ئادەم دەپ قارىسا ، قائىدە يوسۇن بىلمەيدىغان ئادەمنى تەربىيە كۆرمىگەن قارا قۇرساق ئادەم دەپ قارايدۇ  يۇرتلاردا قائىدە -يوسۇن پىشىۋالىرىنى يۇرتنىڭ چوڭى قىلىپ بەلگىلەيدۇ ، ئۇ مۆتىۋەر جامائەتنىڭ ھەر خىل پائالىيەتلىرىگە باشچىلىق قىلىدۇ ، كىشىلەر ئارىسىدا يۈز بەرگەن توقۇنۇشلارنى قائىدە -يوسۇن بىلەن ھەل قىلىدۇ ، كىشىلەر ئارىسىدا قائىدە يوسۇنا بوي سۇنۇش بەخىتلىك تۇمۇشنىڭ ، ئىشلىرىنىڭ قۇتلۇق بولىشىنىڭ بىر شەرتى دەپ قارالاچقا ، ھەربىر ئىشنىڭ قائىدىدىن چەتنەپ كېتىشىگە يول قويۇلمايدۇ ، ھازىرقى ئەھۋالىمىزدىن قارىاندا بىزدە قائىدە -يوسۇندىن چەتنەپ كېتىش مەسىلىسى سەل گەۋدىلىنىپ قالماقتا ، بۇنىڭا نۇرغۇن ئوبېكتىپ ۋە سوبىكتىپ سەۋەپلەر بولسىمۇ ، نېگىزىدىن ئالغاندا بىزنىڭ قائىدە -يوسۇنلارنى ئۇنتۇپ قېلىشىمىز تۈپەيلىدىن ۋە ئۇنى زەنجىرسىمان ھالدا بىر بىرگە چېتىپ ماڭالمايۋاتقانلىقىمىزدىن بولماقتا دەپ قارايمەن . مانا شۇ زۆرۈريەت تۈپەيلىدىن مەن مەرھۇم زىيالىمىز يارمۇھەممەت تاھىر تۇغلۇق جاناپلىرىنىڭ << قائىدە يوسۇنلىرىمىز >> ناملىق كىتاۋىنى ئوقۇشنى بۇرچۇم دەپ قاراپ ئوقۇشقا باشلىدىم ، ئوقۇش جەريانىدا مەرھۇمنىڭ بىز ئۇنتۇشقا باشلىغان قائىدە يوسۇنلىرىمىزنى شۇنچىلىك تەبىئي شەرھىلەپ ئۆتكەنلىكىدىن بەكمۇ مەمنۇن بولدۇم ، شۇنداقلا بۇ كىتاپنى قېرىنداشلىرىمىزنىڭ دەستۇر سۈپىتىدە ئائىلىسىدە ساقلاپ قويىشىنى ھەم ئۇنى ھەر دائىم ئوقۇپ تۇرىشىنى تەۋسىيە قىلىشنى لايىق تاپتىم .كىتاپنى ئاپتۇرنىڭ ھەرگىزمۇ پۇل تېپىشنى ئارزۇ قىلىپ يازمىغانلىقى ھەر بىر قۇردىن مانا مەن دەپ جۇلالىنىپ تۇرۇپتۇ ، بەكمۇ يەڭگىل ، بەكمۇ ئەپچىل ئۇسۇل ئارقىلىق بىز چوقۇم ئەمەل قىلىشقا تېگىشلىك گۆھەردەك قائىدە يوسۇنلىرىمىزنى شەرھىلەپ بېرىپتۇ . بىزنىڭ ساپالىق ئادەم بولىشىمىزغا يول كۆرسۈتۈپتۇ ، ھەممە ئادەمنىڭ قىزىقىشى ھەر خىل بولىشى سەۋەپلىك بۇ كىتاپقا قىزىققۇچىلارنىڭ تۈرلۈك پىكىردە بولىشى نورمال ئىش ، ئەمما سىز بىر ئۇيغۇر ئىكەنسىز چوقۇم بۇ كىتاپنى كۆرۈڭ كۆرۈڭلا ئەمەس ئويلىنىڭ ، ئەنە شۇ چاغدا ئۆزىڭىزنىڭ قانداق مىللەتنىڭ ئەۋلادى ئىكەنلىكىنى ھېس قىلالايسىز ، باھاسى 48سوم ئىكەن بۇ پۇلنى بەلكىم چىقىرالايسىز
