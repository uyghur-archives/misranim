ئەسسالامۇئەلەيكۇم ھۆرمەتلىك قېرىنداشلار،ھەممىڭلارغااللەنىڭ سالامى بولسۇن.
كۈنلەرنىڭ كەينىدىن كۈنلەرئۆتۈپ ھەش-پەش دىگۈچەمانامۇبارەك قۇربان ھېيىتقىمۇنەچچەكۈنلاقالدى،ھەممىڭىلارنىڭ قۇربان ھېيىتنى خاتىرجەملىك ئىچىدەئۆتكۈزۈشكەنىسىپ قىلىشىنى للەدىن تىلەيمەن.ئەمدى مەقسەتكەكەلسەم سىلەرگەئاددىيغىنابىرھۈنەريەنى ئائىلىدەئاسان ئۇسۇلداھېيىتلىق باقالى پىشۇرۇش ئۇسۇلىنى تونۇشتۇرماقچى.
كىتەرلىك ماتىرياللار:تۇخۇم ئۈچ تال،شىكەربىرپىيالە،ھەسەل يېرىم پىيالە،سۈت بىريېرىم پىيالە(خالتىلىق سۈت بولسابىرخالتا)سۇمېيي بىرپىيالە،قۇرۇق ئۈزۈم ئازراق بىرپىيالە،ياڭاق مېغىزى ئازراق بىرپىيالە،ئۇن بىرپىيالە،يىمەكلىككەئىشلىتىدىغان سودابىرقوشۇق،يەنى مىقدارى 7~8گىرام كەلگۈدەك،فۇدافىن سودابىلەن تەڭ مىقداردا،بۇنەرسىنىڭ ئۇيغۇرچەئاتىلىشنى بىلمىگەچكەخەنچەئاتىلىشنى يازدىم(كۆپچىلىكتىن كەچۈرۈم سورايمەن)كۆپتۇرۈش رولىنى ئوينايدىغان سۈت پاراشوكىدەكرەك پۇراقلىق نەرسە،كۈنجۈتتىن ئازراق.
تەييارلاش ئۇسۇلى:ئالدى بىلەن،ئۆيدەئىشلىتىدىغان كىچىك داسقابىرپىيالەسۇمېينى قۇيىمىزئاندىن سودافۇدانى سېلىپ ئارىلاشتۇرىۋالغاندىن كېيىن تۇخۇم،شىكەر،ھەسەل،سۈتلەرنى قوشىمىز.بۇلارنى قوشۇپ قوچۇپ ئارىلاشتۇرغاندىن كېيىن ئۈستىگەئۇننى تۆكۈپ ئۇزاقراق قوچۇپ ئارىلاشتۇرىمىز،ئادەتتەقويۇق-سۇيۇقلۇقى قويۇقراق ئۇماچتەك بولسابولىدۇ،بۇنىڭدىن قويۇق-سۇيۇق بوپ قالغۇدەك بولسائۇننىڭ مىقدرىنى ئەھۋالغاقاراپ تەڭشەشكەبولىدۇ.تاكى خۇرۇچلارنىڭ ھەممىسى تەكشى ئارىلىشىپ بولغانغاقەدەرقوچۇپ ئارىلاشتۇرىمىز.بۇمەشغۇلارتلارنى تاماملاپ بولغاندىن كېيىن ياڭاق مېغىزىنى قۇرۇق ئۈزۈمنىڭ چوڭلۇقىدەك چوڭلۇقتائۇۋىتىپ قۇرۇق ئۈزۈم ياڭاقلارنى سېلىپ ئارىلاشتۇرىمىز،ئاندىن دۇخوپكېنىڭ تەخسىنى مايلاپ ئۈستىگەدۇخوپكاقەغىزىنى سالغان دىن كېيىن بۇقەغەزبولمىساپاكىزەقېلىنراق قەغەزئىشلەتسىمۇبولىدۇ.قەغەزنىڭ ئۈستىنى يەنەبىرقىتىم مايلىۋىتىپ تەييارلىۋالغان ماتىريالنى تۆكۈپ ئۈستىگەكۈنجۈتتىن ئازراق چېچىۋىتپ دۇخوپكېنىڭ تىمپىراتۇرىسىنى 155~160ئەتراپىداتەڭشەپ 45مىنۇت ئەراپىداپىشۇرساق بولىدۇ.مەن بۇيەردەكىچىك بىرتەخسىلىك دۇخوپكىلارنى ئاساس قىلىپ تونۇشتۇردۇم،ئىككى تەخسىلىك دۇخوپكىداجىقراق پىشۇرماقچى بولغان ماتىرياللارنى ئۈستىدەتونۇشتۇرغاننىڭ ئىككى ھەسسىسى بويىچەتەڭشىسەك بولىدۇ.چۈشىنەلمىگەن يەرلەربولسابۇ1329473672چ چ نى قاچىلاپ سورىساڭلارسىلەرئۈچۈن قولۇم كۆكسۈمدە.باشقۇرغۇچى قېرىنداشلارنىڭ بۇيازمامنىڭ ئەھمېيتى بار،تەسىتقلاشقابولىدۇدەپ قارىساتەسىتىقلاپ قويۇشىنى ئۆتۈنىمەن.
