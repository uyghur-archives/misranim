ئەسسالامۇ ئەلەيكۇم قىرىنداشلار ،مۇسۇلمانلارنىڭ قۇربان ھىتىغا ئازقالدى،رەججەپ ئيىنىڭ مۇشۇ10كۇنىدە
رۇزا تۇتۇش ئىنتايىن ياخشىدۇر ،1كۇنلۇك رۇزىغا ئاللاھ 1يىل رۇزا تۇتقاننىڭ سارابىنى بىرىشكە ۋەدە قىلغان
قىرىنداشلار .ئاللاھ ھەممىمنزگە ئىمان ،ئنساپ ، ھەقنى ئاڭلايدىغان قۇلاق ،كۇرىدىغان كۇز  ئاتاقىلسۇ ن ئامىن
مۇسۇلمانلارنىڭ قۇربان ھيتىغا مۇبارەك ،،،


نەبىيشەت رەزىيەللاھۇ ئەنھۇدىن، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم مۇنداق دىگەن: ھەقىقەتەن بىز سىلەرنى قۇربانلىقنىڭ گۆشىنى ئۈچ كۈندىن ئوشۇق يىيىشىڭلاردىن توسقان ئىدۇق، بۇ سىلەرگە كەڭرى بولسۇن ئۈچۈن ئىدى، ئەمدى ئالاھتائالا سىلەرگە كەڭرى قىلىپ بەردى، ھەممىڭلاردا بار بولدى، ئەمدى سىلەر خالىغانچە يەڭلار، ئېلىپ قويۇپ ساقلاپمۇ يەڭلار، ئۇنىڭدىن سەدىقە – ئېھسان قىلىپ،اللھ تىن ئەجرى تەلەپ قىلىڭلار، ئاگاھ بولۇڭلاركى، بۇ كۈنلەر يەپ ئىچىشنىڭ كۈنلىرىدۇر، ئاللاھنى ئەسلەپ، زىكرى ئېيتىش، تەگبىر ئېيتىش كۈنلىرىدۇر. ئەبۇ داۋۇد توپلىغان، سەھىھ.

ئەبۇ سەئىد،ئىبنى ئەكۋەئ،قەتادە رەزىيەللاھۇ ئەنھۇمدىن،رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم مۇنداق دىگەن: سىلەردىن بىرەركىشى قۇربانلىق قىلسا، ئۈچ كۈندىن كىيىن، ئۇنىڭ ئۆيىدە قۇربانلىقىدىن بىرەر نەرسە بارھالدا تاڭ ئاتقۇزمىسۇن، ( يەنى باشقىلارغا بەرسۇن).كېلەر يىلى بولغاندا، كىشىلەر؛ ئى اللھ نىڭ ئەلچىسى! بۇ يىلمۇ ئۆتكەن يىلى قىلغاندەك قىلامدۇق ؟دىدى، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم؛ مانا ئەمدى قۇربانلىقنىڭ گۆشلىرىنى يەڭلار، باشقىلارغا تاماق قىلىپ بىرىڭلار، ساقلاش ئۈچۈن ئېلىپمۇ قويۇڭلار، شۇ يىلى كىشىلەرگە مۇشەققەت،قىيىنچىلىق،ئاچارچىلىق بولغان ئىدى، مەن ئۇلارغا ياردەم قىلىشىڭلارنى مەقسەت قىلغان ئىدىم، دىدى. ئەھمەد،بۇخارى،مۇسلىم، ھاكىم توپلىغان، سەھىھ. (بۇرەيدە رەزىيەللاھۇ ئەنھۇدىن، رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم مۇنداق دىگەن: مەن سىلەرنى ئۈچ نەرسىدىن توسقان ئىدىم، مانا ئەمدى ئۇ نەرسىگە بۇيرۇيمەن، مەن سىلەرنى قەبرىلەرنى زىيارەت قىلىشدىن توسقان ئىدىم، مانا ئەمدى قەبرىلەرنى زىيارەت قىلساڭلار بولىدۇ. ئۇنىڭدا ۋەز – نەسىھەت، ئىبرەت بار، مەن سىلەرنى تېرىدىن باشقا تۇلۇن قاچىلاردا ئىچمەڭلار دەپ توسقان ئىدىم، مانا ئەمدى ھەر قانداق تۇلۇن، قاچا بولسا، ئۇنىڭدا نەرسە ئىچسەڭلار بولىدۇ، لېكىن، مەست قىلغۇچىنى ئىچمەڭلار، ئۇ ھاراقتۇر. مەن سىلەرنى قۇربانلىقنىڭ گۆشلىرىنى ئۈچ كۈندىن ئارتۇق ساقلاپ يىيىشىڭلارنى توسقان ئىدىم، ئەمدى ساقلاپ يەۋىرىڭلار، ئۇنىڭ بىلەن سەپەرلىرىڭلاردىمۇ پايدىلىنىڭلار، سەدىقە قىلىڭلار. ئەبۇ داۋۇد، مۇسلىم توپلىغان، سەھىھ.

ئىبنى ئۆمەر رەزىيەللاھۇ ئەنھۇ ، تۇگىسىنى چۇكىلاندۇرۇپ، ياتقۇزۇپ تۇرۇپ بۇغۇزلاۋاتقان بىر كىشىنىڭ ئۇستىگە كىلىپ قېلىپ؛ ھەي! سەن ئۇنى مۇھەممەد سەللەللاھۇ ئەلەيھى ۋەسەللەمنىڭ سۇننىتى بۇيىنچە، ئۆرە تۇرغۇزۇپ قۇيۇپ، پۇتلىرىنى باغلاپ بۇغۇزلا دىدى. بۇخارى،مۇسلىم بىرلىككە كەلگەن

جابىر رەزىيەللاھۇ ئەنھۇ مۇنداق دەيدۇ؛ رەسۇلۇللاھ سەللەللاھۇ ئەلەيھى ۋەسەللەم ھەج قىلغاندا ، ئاياللىرى ھەققىدىن، كالا بۇغۇزلاپ قۇربانلىق قىلغان ئىدى. مۇسلىم توپلىغان

مەن بىر ھەدىستە، قۇربان ھېيتقا ئون كۈن قالغاندىن باشلاپ ، ھېيت نامىزى ئوقۇلۇپ بولغانغا قەدەر ،ئەرلەرنىڭ  ساقىلىنى ئالسا بولمايدىغانلىغىنى ، ھېيت نامىزىدىن كىيىن ئاندىن ئالسا بولىدىغانلىغى توغرىسىدا ئوقۇغان ئىدىم، شۇ ھەدىسنى يوللاي دىسەم تاپالمىدىم، بىلىدىغانلار بولسا، يوللاپ  قويسا بوپتىكەن، بىلمىگەنلەر بىلىۋېلىپ، بىر سۈننەت بولسىمۇ قولدىن بەرمەي ئەمەل قىلسا دىمەكچىدىم. يەنە بىر ئىش، قۇربانلىق قىلغان گۆشنى تەڭ ئۈچكە بۆلۈپ، بىر قىسمىنى چوقۇم قۇربانلىق قىلالمىغان مىسكىن، غېرىپلارغا بېرىش شەرتكەن، بۇنىڭغا  كۆپۈنچە كىشى ئەھمىيەت بەرمەيدىكەن، شۇڭا ئۆزىنى مۇسۇلمان چاغلايدىغان قېرىنداشلىرىمىزنىڭ قۇربان ھېيتنى ، گۆش يىيىش بايرىمى  قىلىۋالماي، ئىسلام قانۇنى بويىچە،  ئىسلام ئەھكاملىرىغا بويسۇنغان ھالدا ئۆتكۈزىشىنى تەۋسىيە قىلىمەن..... بۇ يازمىنى ئاخىرىدا   asma0996 تەھرىرلىگەن. ۋاقتى  2013-10-5 12:01 AM
