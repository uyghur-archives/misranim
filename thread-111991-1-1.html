                                                 قارا قوساق قاسساپ
                                                                      سۆز ئوينى
  
      قۇمباغ قەلئەسىنىڭ قېرىلار قوروسىدىكى قېدىمدىنلا قاسساپلىق قىلىۋاتقان قابىلىيەتلىك قادىر قاسساپنى قاسساپلىقتىن قالدۇرۇپ،قوتانلاردىكى قوي-قوتازلارنىڭ قۇرىمىغان قىغلىرىنى قاپلارغا قاچىلاپ قىشلاقلاردىكى قوغۇنلارغا قۇيىدىغان،قارا قوساق قاسىم قوتاننى قورونىڭ قاسساپلىقىغا قويدى.قورودىكى قاراۋۇللارمۇ قوللاپ قۈۋەتلىدى.<قاسىم >قوتان لىق قۇرۇق قالپاقنى قالدۇرۇپ،قاسىم <قاسساپ> لىق قاشقىسىنى قاڭشىرغا قامەتلىك قىلىپ قادىدى.
       قاسىم قوتان قاسساپلىقنى،قادىر قاسساپتەك قاملاشتۇرالمىدى.قېنى،قاسساپلارنىڭ قاسساپلىق قىلغان قەدەملىرىگە قارايلى! قېرىنداشلار.
    قادىر قاسساپ قاسسپلىقنى؛قاسساپلىقنىڭ قىبىلنامىسىدىكى قىممەتلىك قائىدىلەرنى قۇرمۇ-قۇر قۇبۇل قىلىپ،قوينى قوتازغا،قېرىننى قۇيرۇققا قوشماي،قائىدىلىك قىلغاچقا قورودىكىلەر قوللىغان.
      قاسىم قاسساپ قاسساپلىقنىڭ قىممەتلىك قەدملىرىدە قائىدىسىز قىلىقلارنى قايتا-قايتا قىلغاچقا،قورودىكى قېرىنداشلار قاسىم قاسساپتىن قاتلاممۇ-قاتلام قاتتىق قاخشىدى.قورودىكىلەر قاسىم قاسساپنىڭ قائىدىسىز،قانۇنسىز قىلىقلىرىغا قايتا-قايتا قاراپ،قاسىم
قاسساپنى قاسساپلىقتىن قالدۇرۇپ،قېرىلار قوروسىدىن قەدىمقى قۇرۇق قوتانغا قوغلاشنى،قانۇن
نىڭ قاراۋۇللىرى قاسىم قاسساپنىڭ قاش قاپاقلىرىغا،قېرىنداشلىرىغا قۇرغان قەدەم-قەدەمدىكى قىلتاقلىرىغا قانۇنى قارار قىلىپ،قەلئەدىكى قارا قوساقلارنى قامايدىغان قاراڭغۇ قاماققا قاماشنى،قوللىرىنى قاتتىق قارا قۇلۇپتا قۇلۇپلاشنى  قارار قىلدى.
       قاسىم قاسساپنىڭ قاسساپلىق قىلغاندىكى قائىدىسىز قىلىقلىرى قالتىس قىزىقارلىق،قەبى قورقۇنۇشلۇق،قاراڭغۇ قوتاندەك قالايمىقان.
       قاسىم قاسساپ قانداق قىلىپ قاسساپلىقنى قائىدىلىك قىلماي،قائىدىسىز قىلىقلارنى قىلىپ قالدى؟
       قاسىم قاسساپنىڭ قەدىمدىنلا قارا قوساق،قابىلىيەتسىزلىكى،قائىدىلىك قاسساپلىق قىلىشىنى قاتتىق قاماق قىلىۋالغان.
     قاسىم قاسساپنىڭ قاسساپلىق قەدەملىرىگە قارىسىڭىز،قورودىكى قىلىقلىرى؛- قورودىكى قېرى-قېرانلارنى،قەۋمى-قېرىنداشلارنى قايىل قىلىپلا قالماي،قەلبى قۇلۇپلانغان قارا- قوساقلارنىمۇ قايىل قىلالايدۇ.
     قېرىنداشلار،قاسىم قاسسپنىڭ قائىدىسىز قىلغان قىلىقلىرىنىڭ قەدەملىرىگە قەلىپ-قارچۇقىمىزدا قارىساق قارشىلاشمايلا قايىل قىلالايدۇ؛-قېنى قاراڭلار!
     قاسىم قاسساپنىڭ قالاق قەلبىنى؛-قەلەم قاشلىق قىزلارنىڭ قاش-قاپاقلىرىدەك قىمىرلايدىغان.قەبرىدىكىلەرنىمۇ قىزىقتۇرالايدىغان،قەرزدارنى قوغلاپ قورتقىتالايدىغان قىزىل قەغەز،قىرىق قېتىم قاتلىسىمۇ قورۇلمايدىغان،قىزىل قەغەز؛-قىممەتلىك قېپ-قىزىل قەغەز،
قېلىن قىزىل قەغەز،قاسىم قاسساپنىڭ قۇرۇپ قالغان قەلبىنى،-قىزىل قەغەزگە قىزىتىپ،قوزغۇتۇپ
قالايمىقانلاشتۇرۋەتكەن.
      قاسىم قاسساپ،قىرىق قېتىم قاتلانسىمۇ قورولۇپ قېلىشتىن قورقمايدىغان،قەبرىگە قايتقىچە
قېرىمايدىغان،قۇرۇلمىسى قاتمۇ-قات،قامىتى قاردا قالسىمۇ قارايمايدىغان،قارىماققا قەلبى قېپ-
قىزىل قۇرۇق قەغەزنىڭ قىزىقتۇرشىدا قەدەممۇ-قەدەم قايمۇقۇپ قېلىپ،قاسساپلىق قاينىمىدا قېرىنداشلىرى قوللىمايدىغان قاملاشمىغان قىلىقلارنى قىلىپ قالغان.
       قاسىم قاسساپنىڭ قاملاشمىغان قىلىقلىرىنى قولىمىزغا قۇشتەك قوندۇرۇپ،قىياپىتىنى قايتىدىن قۇراشتۇرۇپ قىبلىگە قىرىق قېتىم قاراپ قۇبۇل قىلساق قەلبىمىزنى قايىل قىلالايدۇ.
      قاسىم قاسساپ قانداق قىلدى؟-قوتاندىكى قارا قاڭشالىق قېرى قوتازدىن-قوي قوزىلارغىچە؛ قونداقتىكى قارا قاغا-قۇزغۇندىن-قورقۇنچاق قۇچقاچقىچە؛قۇملۇقتىكى قىلتىرىقتەك قۇرۇتتىن-قاپ-قارا قوڭغازغىچە قىلچىلىكمۇ قالدۇرماي قاراڭغۇ قازناققا قالايمىقان قاچىلاپ؛
قوينىڭ قۇيرىقىدا قوتازنى قۇراشتۇرۇپ،قۇرۇت-قوڭغازنى قوي-قوتازنىڭ قوسىقىغا قاچىلاپ،قاغا
-قۇزغۇنلارنى قېرى قوتازلار كەينىدىكى قوروق قالپاققا قۇيۇپ قويغان قاڭسىق قايماقتا قاينىتىپ،
قۇرۇتنىڭ قىلتىرقىدا قۇراشتۇرۇپ،قۇچقاچنىڭ قانىتىغا قاچىلانغان قاتتىق قەندىن قىرىق قوشۇق
قۇيۇپ قەغەز قاچىلارغا قاچىلاپ،قاچىنىڭ قاڭشىرىغا قوينىڭ قۇيرىقىنى قىزىل قەغەزگە قوشوپ قاداپ،قاراڭغۇدا قارىغۇ  قارىسمۇ قۇبۇل قىلالايدىغان قىلىپ قۇراشتۇرۇپ قورودىكى قېرىداشلىرىغا قەدەممۇ-قەدەم قۇبۇل قىلدۇرغان.
      قورودىكى قېرىنداشلار.قاسىم قاسساپ قۇراشتۇرغان قاچىلاردىكى قورىمىلارغا قارسا،قورۇمىلار قاملاشمىغان،قەدىمقى قازاندا قايناتمىغان،قىغدەك قاڭسىق، قاچىنىڭ قاڭشىردا قوينىڭ قۇيرىقى،قاتلاملىرىدا قۇرۇت-قوڭغازلارنىڭ قوشولمىسى،قەتئى قاملاشماپتۇ.
     قاسىم قاسساپنىڭ قىلمىشلىرغا قارتا قورودىكى قېرىنداشلار قالتىس قىممەتلىك، قىزىقارلىق  قەلئەدىكىلەرنى قايىل قىلارلىق قارارنى قىلدى.
       قاسىم قاسساپنى قاسساپلىقتىن قالدۇرۇش،قورودىن قوتانغا قوغلاش،قانۇننىڭ قارارلىرىنى قۇبۇل قىلدۇرۇپ قاراڭغۇ قاماققا قىيامەت- قايىمغىچىە قاماش.
        قورودىكىلەر،قاسىم قاسساپنىڭ قىلمىشلىرىغا قانۇن قاراۋۇللىرىنىڭ قانداق قارايدىغانلىقىغا،قورودىكىلەرنىڭ قارارىغا قانداق قارايدىغانلىقىغا قىزىقارلىق قىياپەتتە قارىماقتا.
       قاسىم قاسساپنىڭ قىلمىشلىرىغا قارتا قانۇنىمىز قانداق قارار قىلار؟
                                    مەنبە؛ ئۆزەمنىڭ ئىجادىيتى.
