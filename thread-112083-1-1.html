                          ئاسان ۋە تەس  
                              سۆز ئوينى                                      

                      دۇنيادا بەزى ئىشلار ئاسان،--بەزى ئىشلار تەس.
                 بەزىلەرگە  ئاسان،--بەزىلەرگە تەس.

                گەپنى دىمەك ئاسان،--چۈشەنمەك تەس.
                دوست بولماق ئاسان،--ئىشەنمەك تەس.

                ئۇرۇشماق ئاسان،--ئەپلەشمەك تەس،
                كۆرۈشمەك ئاسان،--گەپلەشمەك تەس.

                كۈلمەك ئاسان،--يىغلىماق تەس،
                تۇغۇلماق ئاسان،--ياشىماق تەس.

                 مۇھەببەتلەشمەك ئاسان،--توي قىلماق تەس،
                ئەر-ئايال بولماق ئاسان،--ئاتا-ئانا بولماق تەس.

                پۇلنى خەجلىمەك ئاسان،--تاپماق تەس.
                ناننى يىمەك ئاسان،--ياپماق تەس.

                ئۇخلىماق ئاسان،--ئويغانماق تەس.
                 توختىماق ئاسان،--قوزغالماق تەس.

                ئوغرىنى قوغلىماق ئاسان،--تۇتماق تەس،
                قىمارنى ئوينىماق ئاسان،--ئۇتماق تەس.

                تويغا بارماق ئاسان،--كەلمەك تەس،
                سوغاتنى ئالماق ئاسان،--بەرمەك تەس.

                كىسەل بولماق ئاسان،--ساقايماق تەس،
                قارايماق ئاسان،--ئاقارماق تەس.

                  كىسەلنى بايقىماق ئاسان،--داۋالىماق تەس.
                 ئەخلەتنى تاشلىماق ئاسان،--ئادالىماق تەس.

                 ئوقۇماق ئاسان،--دىپلوم ئالماق تەس.
                 ئىشلىمەك ئاسان،--خىزمەت تاپماق تەس.

                 ئىشىكنى ياپماق ئاسان،--ئاچماق تەس.
                 مالنى ئالماق ئاسان،--ساتماق تەس.

                 ئىشەك بولماق ئاسان،--دۇلدۇل بولماق تەس.
                 قاغا بولماق ئاسان،--بۇلبۇل بولماق تەس.

                كادىر بولماق ئاسان،--دېھقان بولماق تەس.
                زالىم بولماق ئاسان،--ئالىم بولماق تەس.

                 تۇرماق ئاسان،--ماڭماق تەس.
                سۇ ئىچمەك ئاسان،--قانماق تەس.

                 ئوت ياقماق ئاسان،--ئۆچۈرمەك تەس.
                  ئىمتىھان بەرمەك ئاسان،--كۆچۈرمەك تەس.

                  ئەسەرنى ئوقۇماق ئاسان،--يازماق تەس.
                 نىفىتنى ئىشلەتمەك ئاسان،--قازماق تەس.

                ماشىنا ئالماق ئاسان،--ھەيدىمەك تەس.
               چاپاننى سالماق ئاسان،--كەيمەك تەس.

                         مەنبە؛ئۆزەمنىڭ ئىجادىيتى.

دەۋاقىلماق ئاسان ،ئاقمىقى تەس،
نىيەت قىلىش ئاسان ،ئەمەل قىلماق تەس،
  
شەيتان بولماق ئاسان،مۇسۇلمان بولماق تەس،
ئازدۇرۇش ئاسان ،توغرا يول تىپىش تەس،

ھارامنى تاپماق ئاسان،ھالالنى تىپىش تەس،
گالنى بۇلغىماق ئاسان،پاكىز تۇتۇش تەس،

بالا بولماق ئاسان،ئاتا-ئانا بولماق تەس،
بالا تۇغۇلمىقى ئاسان ،تەربىلىمەك تەس،
مىنى مۇسۇلمان تۇغۇلۇپ ئۇيغۇرلۇقۇمنى تۇنۇتقان ئاللاھقا كۇرمىڭ شۇككۇرلەر بولسۇن
