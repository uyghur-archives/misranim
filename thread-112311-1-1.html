ئىنسان دېگەن ئازغۇچى بولغاچقىمىكىن بۇ خىلدىكى سەۋەنلىكنى نورغۇنلىرىمىز ئۆتكۈزگەن،ئۆتكۈزۈۋاتىمىز ھەم داۋاملىق ئۆتكۈزمەكچى بولۇۋاتىمىز،تېمىنىڭ پوسكاللىسىغا كېلىشتىن بۇرۇن بىر نەچچە مىسال ئالاي،قانچە يىللىقىنى ئېنىق بىلمەيمەن ئاز دېگەندە ئون يىلدىن ئاشتى ئابدۇللا ئابدۇرېھىمنىڭ بىر VCDپىلاستىنكىسىدا بىر شوخ ناخشا بولىدىغان،كەچلىكى تازا ماشىنا كۆپ ۋاقتىدا يولنىڭ ئوتتۇرا سىزىقىدا مېڭىپ ناخشا ئېيتقان كۆرۈنۈش،ئىككى يېنىدا ماشىنىلار غۇيۇلداپ ئۆتۈۋاتقان، تېخى نەچچە قېتىم قايتىلانغاندەك ئېسىمدە،تۇنجى قېتىم بۇ كۆرۈنۈشنى كۆرگىنىمدە يۈرىكىم جىغ قىلىپ قالغان،مەن شۇنچە ياخشى كۆرىدىغان بۇ چولپان ئەجەپ قىلدا دەپ،مۇشۇ خاراكتېرىدىكى يۇقىرى سۈرئەتلىك يوللاردا ۋە شۇنىڭغا ئوخشاش پىيادىلەر يول يۈرۈشكە بولمايدىغان يەرلەردە موڭلۇق قەدەملەر بىلەن ناخشا ئېيتىپ ماڭغان كۆرۈنۈشلەر بىزنىڭ نۇرغۇن ناخشىچىلىرىمىزنىڭ MTVلىرىدا ئۇچراپ تۇرىدۇ.مەن بۇنى بىر مەسئۇلىيەتسىزلىك دەپ قارايمەن،داڭلىق شەخسلەرنىڭ كەڭ خەلق ئاممىسىغا ئۈلگە بولۇش مەجبۇرىيىتى بار،ئەتراپىمىزدىكى نۇرغۇن پىشىپ يېتىلمىگەن غۇنچىلار،نوتىلار بىلىپ بىلمەي ئۇلارنى دورايدۇ،تەسىرىگە ئۇچرايدۇ،بەك ھاياجانلىنىپ كەتكەندە ياكى كۆڭلى غەش بوپ قالغاندا ئابدۇللا ئاكام ماڭغان،پالانچى ھەدەم چىقىپتىكەن دەپ ، مەنمۇ مېڭىپ باقاي،چىقىپ باقاي دەپ يولنىڭ ئوتتۇرىسىدا ،تىز سۈرئەتلىك  يوللارغا كۆۋرۈكلەردە ماڭمايدۇ دېگىلى بولمايدۇ.ئابدۇللا ئابدۇرېھىمنىڭ ناخشىلىرىنى يىگىرمە نەچچە يىلدىن بېرى ياقتۇرۇپ ئاڭلاپ كېلىۋاتىمەن،ھەم بەك ھۆرمەت قىلىمەن،بۇ ئىشنى بىر مىسال قىلىپلا سۆزلەپ قويدۇم،مەنمۇ قاتتىق قوللىغۇچىلىرىغا كىرىمەن ،تىللاپ چۈشمەڭلار يەنە.
پوسكاللىسىغا كەلسەك مۇشۇ چوڭ بازارنىڭ ئالدىدىكى ئازادلىق جەنۇبى يول تۆت كوچا بويىدا قىزىل چىراغدا توختىغان ماشىنىلارغا ماشىنىلار ئارىلىقىدا مېڭىپ يۈرۈپ قەغەز ساتىدىغان كىچىك بالىلار بار،ھەر قېتىم ئۆتسەم بىر ئىككىسىنى ئۇچرىتاتتىم،قەغەز ئالامسىز دەپ كەلسە ئالغان يېرىممۇ بار،كىچىك تۇرۇپ ئۆز كۈچىگە تايىنىپ ھالال پۇل تېپىشنى بىلىپتۇ،ئاتا-ئانىسىنىڭ ئېغىرىنى يېنىكلىتىشنى بىلىپتۇ دەپ سۆيۈنۈپ كەتكەنتىم،بۈگۈن ئالدىمدا كىتىۋاتقانلاردىن بىرسى نېمە بولدىكىن تاڭ ،باشقا ماشىنا بىلەن يول تالاشتىمۇ تاس قالدى بىر بالىنى قىستاپ سوقۇپ قويغىلى،ھېلىمۇ ياخشى ئۇ بالا چاققانكەن شاپپىدە سەكرەپ يولنىڭ قىرىغا چىقىۋالدى،يۈرۈكۈم جىغغىدە قىلىپ قالدى،ئۇنىڭ ئۈستىگە بۈگۈن جىقلا بالا يولغا چىقىۋاپتۇ،دەرھاللا نەچچە ۋاقتىدىن بېرى بۇ ئىشنى ئەقىل بىلەن ئەمەس ھېسسىيات بىلەن تەپەككۈر قىلغىنىمدىن خىجىل بوپ قالدىم،ئەمدى بۇلاردىن ھەرگىز قەغەز ئالمايمەن دېدىم.بۇلارنىڭ ئۆز بىخەتەرلىكىنى ئويلاشماي يولغا چىقىپ سودا قىلغىنىنى قوللىغىنىمىز ،ئۇلارنى خەتەرلىك مۇھىتتا داۋاملىق تۇرىشىغا تۈرتكە بولغىنىمىز بولىدىكەن،يېنىمغا كەلگىنىگە ئەمدى سىلەرنىڭ قەغىزىڭلارنى ئالمايمەن دىدىم،نىمىشقا دىگەنتى، راست گەپ قىلە سىنىڭ مۇشۇ يەردە ساڭا ئوخشاش قەغەز ساتىدىغان بىرە ئاغىنەڭ يارىلىنىپ باقمىدىما دېسەم چىرايى ئۆزگۈرۈپ كەتتى،بىر خىل خىجىللىق تەمتىرەش ئىپادىسى بىلەن سوئالىمنى جاۋاپسىز قويدى،يانفونۇمنى چىقىرىپ رەسىمگە تارتسام يەنە بىرسى ۋاي رەسىمگە تارتىۋاتىدۇ دەپ بالىلارنى يىغىپ كەلدى،بىزنى تارتىپ نېمە قىلىسىز دەيدۇ،بۇيەردە بۇنداق قەغەز ساتساڭلار بەك خەتەرلىككەن تورغا چىقىرىۋىتىمەن،دېسەم بەزىسى ۋاي تورغا چىقىدىكەنمىز دەپ پوكس چىقىرىپ بەردى،يەنە بەزىلىرى يۈزىنى ئېتىۋالدى،بەزىلىرى قېچىپ كەتتى.ئون يەتتىنچى ئوتتورىنىڭ يېنىدىكى تار كوچىدا تېلەمچىمۇ ماشىنىلار ئارىسىدا تېلەمچىلىق قېلىۋېتىپتۇ،بىر بوۋايمۇ ماشىنىلار ئارىسىدا كىتاپ سېتىۋېتىپتۇ.مەنچە ئۇلار نېمە قىلسۇن ئۇلارغا ماسلاشقىنىمىز خەتەرلىك مۇھىتتا قىلىۋاتقان ئىشىنى قوللىغىنىمىز بولىدىكەن ئەمەسمۇ،تورداشلارنىڭ قانداق ئويلىشىدىن قەتئىينەزەر مەن ئەمدى سودىسىنى قىلمايمەن.بالىلا كىچىك بىخەتەرلىك ئېڭى پىشىپ يېتىلمىدى، بىز ئۇلارنىڭ بىخەتەرلىكىنى ئويلىمىساق بولمايدۇ.ماشىنا چاقچاق قىلىدىغان نەرسە ئەمەس،ھايات تېخىمۇ شۇنداق!
