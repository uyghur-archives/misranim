قارا ئۆڭكۈرنىڭ تۇرغۇن يۇلتۇزنى يۇتىشى


    ئاسترونۇملار يېقىندا قارا ئۆڭكۈرنىڭ تۇرغۇن يۇلتۇزنى يۇتۇپ كەتكەن كۆرۈنۈشىنى، ئالەمدىكى ئەڭ سىرلىق، كىشىنى تاڭ  قالدۇردىغان مەنزىرىنى سۆرەتكە ئېلىۋالدى. سۆرەتتىكى قارا ئۆڭكۈر يالماۋۇزغا ئوخشاپ كېتىدۇ،  ئۇنىڭغا يېقىنلاشقان تۇرغۇن يۇلتۇز قىسقىغىنا ۋاقىت ئىچىدە پارچىلىنىپ ، نۇر چىقىرىدىغان زەررىچىلەرگە ئايلىنىپ غايىپ بولدى. ئېيتىشلارغا قارىغاندا قارا ئۆڭكۈر تارتىش كۈچى غايەت زور بولغان ئاسمان جىسىمى بولۇپ ، نۇرمۇ ئۇنىڭدىن قېچىپ قۇتۇلالمايدىكەن.  ئۇ ئالەمدىكى تېگى يوق ئۆڭكۈر ، ئۇنىڭغا ئۇچرىغان ھەرقانداق نەرسە قېچىپ قۇتۇلالمايدۇ. قارا ئۆڭكۈردىن نۇرمۇ قېچىپ قۇتۇلالمىغاچقا، ئالىملار قارا ئۆڭكۈر بىۋاستە كۆزۈتۈشكە ئامالسىز.  بۇ سۆرەت بولسا ئاسترونوملارنى قارا ئۆڭكۈر توغۇرلۇق بىۋاستە يىپ ئۇچى بىلەن تەمىنلىدى.




خەۋەرلەرگە قارىغاندا ، سۆرەتتىكى قارا ئۆڭكۈر يەر شارىدىن 4 مىليارد نۇر يىلى يىراقلىقتا ئىكەن.  كۆپ قىسىم يۇلتۇزلار سېستىمىلىرى ئىچىدە  زور ماسسىلىق قارا ئۆڭكۈر بولىدۇ. بۇ قارا ئۆڭكۈرلەرنىڭ ماسىسى تەخمىنەن قۇياش ماسىسىنىڭ بىر مىليون ھەسىسىدىن 10 مىليارد ھەسىسىگىچە بولىدۇ. ئېيتىشلارغا قارىغاندا قارا ئۆڭكۈر ھەر 100مىليون يىلدا ئاران بىر تۇرغۇن يۇلتۇزنى يۇتۇپ كېتىدىكەن.



ماتىرىيال رەسىم: ئەجدىھا يۇلتۇز تۈركۈمىدىكى قارا ئۆڭكۈر بىر تۇرغۇن يۇلتۇزنى يۇتماقتا. ئامرىكا ئالەم قاتناش ئىدارىسى گامما نۇرلۇق كۆزىتىش ئەسۋابى بىلەن ئەجدىھا يۇلتۇز تۈركۈمىنى كۆزىتىۋاتقاندا، ئۇنىڭ ئىچكى قىسمىدا سىرلىق ئېنىرگىيە مەنبەسىنى بايقىغان. يەر شارى بىلەن بولغان ئارىلىقى 4 مىليارد نۇر يىلى بولۇپ ، ئالىملار ئۇنىڭغا Sw 1644+57 دەپ نومۇر قويغان. بۇ سىرلىق ئېنىرگىيە قارا ئۆڭكۈرنىڭ مەركىزىدىن كەلگەن بولۇپ، بۇ ھادىسە ئالىملارنى ناھايىتى قىزىقتۇرغان. كېيىن بۇ قارا ئۆڭكۈرنىڭ بىر تۇرغۇن يۇلتۇزنى يۇتىۋاتقانلىقى ئىسپاتلانغان.  بۇ بىر ناھايىتى كەم ئۇچرايدىغان ئالەم ھادىسىسى، بولۇپ 100 مىليون يىلدا ئاران بىر قېتىم ئۇچرىشى مۇمكىن.
