دۆلىتىمىز يىقىنقى يىللاردىن بۇيان مائارىپ سىلىنمىسىنى يىلدىن يىلغا كۆپەيىتۋاتىدۇ. بولۇپمۇ دۆلەت جەنۇبى شىنجاڭدىكى ئىككى ۋىلايەت،بىر ئوبلاسىىتا قوش كەچۈرۈم سىياسىتىنى يولغا قويغاندىن كىيىن جەنۇبى شىنجاڭدىكى نامرات رايۇنلارنىڭ مائارىپى، جۈملىدىن ئازسانلىق مىللەتلەر مائارىپى ناھايتى زور تەرەققىيات پۇرسىتىگە ئىرىشتى. دۆلەتنىڭ بۇ سىياسەتلەرنى يولغا قويۇشتىكى مەقسىتى ئىنىقكى ئاز سانلىق مىللەت رايونلىرىنىڭ مائارىپىنى تەرەققى قىلدۇرۇش،خەلىقنىڭ يۈكىنى يىنىكلىتىشتىن ئبارەت. ئەلۋەتتە بۇ ئەۋزەل سىياسەتلەرنىڭ نەپ ئىلىش ئوبىكتى مۇشۇ رايۇنلاردىكى خەلق ۋە ئۇلارنىڭ پەرزەنىتلىرى. لىكىن ئەمىليەتتىچۇ؟ ئەمىليەتتە نەپنى زادى كىملەر ئىلۋاتىدۇ؟ تۆۋەندىكى بىرقانچە مىسال بىزنى بۇ ھەقتە ئازراق بولسىمۇ چۈشەنچىگە ئىگە قىلسا ئەجەپ ئەمەس.
    1. بىر نەچچە يىلنىڭ ئالدىدا ناھىيىمىزدە چەت -يىراق تاغلىق رايۇنلاردىكى دىھقان-چارۋىچىلارنىڭ پەرزەنىتلىرىگە تىخىمۇ ياخشى ئوقۇش شارا ئىتى يارتىپ بىرىش ئۈچۈن ، يىڭىدىن بىر تۈركۈم مەكتەپلە سىلىندى ياكى كىڭەيتىپ سىلىندى. بۇمەكتەپلەرگە يىڭىدىن ئوقۇتۇش بىناسى،ياتاق بىناسى،ئاشخانا ،مۇنچا قاتارلىقلار سىلىندى.جۈملىدىن بىزنىڭ مەكتەپكىمۇ سىلىندى. لىكىن بۇلار سىلىنىپ تىخى بەش يىل بولماستىلا چاتاق چىققىلى تۇردى. 8بال يەر تەۋرەشكە بەرداشلىق بىرىدۇ دەپ سالغان بىنانىڭ تاملىرى تىخى يەر تەۋرىمەي تۇرۇپلا چاك كىتىشكە باشلىدى.  ئەۋرەز يوللىرى بىريىل بولا بولمايلا ئشلىتىلىشتىن توختىدى. پاريوللىرىمۇ ئومۇمىيۈزلۈك قايتا قوراشتۇرلۇش ئالدىدا.

2. دۆلەت ئىككى كەچۈرۈم، بىر تولۇقلىما سىياسىتىنى يولغا قويغاندىن كىيىن، ناھىيلىك مائارىپ ئىدارىسى مالىيە مەركىزىدىن بىرنى قۇرۇپ  مەكتەپلەرنىڭ ھىساۋاتىنى بىر تۇتاش باشقۇردى. مەكتەپلەرگە كىتەرلىك ھەرخىل ماتىرياللارنى ئىلىش ئۈچۈن مەخسۇس ماگزىنلار بىكىتىپ بىرىلدى. مەكتەپلەر كىتەرلىك ماتىرياللارنى بەلگىلەنگەن ماگىزىنلاردىن ئمىزا قويۇپ بىرىپ ئىلىۋالىدىغان، ماگزىنلار پۇلىنى مالىيە مەركىزىدىن ئلىۋالىدىغان بىر ئەندىزە شەكىللەندى. نەتىجىدە نەرسىلەرنى بازار باھاسىدىن نەتچچە ھەسسە قىممەت ئالىدىغان ئەھۋال شەكىللەندى. مەسىلەن؛باشقا ماگىزىنلاردا 30يۈئەنگە ساتىدىغان سۈپىتى ناچار توپنى 120يۈئەنگە، ئۈچ يۈەنگە ئالىدىغان بوياقنى يەتتە -سەككىز يۈئەنگە..............دىگەندەك ۋاھاكازا.......
3. مەكتەپ ئاشخانىسىغا كىتەرلىك سەي-كۆكتات، گۆش،ياغنى مەخسۇس بەلگۈلەنگەن سودىگەرلەر ئىلىپ چىقىدۇ. ئۇلارنىڭ بىكىتكەن باھاسىمۇ بازار باھاسىدىن نەچچە ھەسسە يۇقۇرى.
  بىر كۈنى مەكتەپ مۇدىرى بىر قانچىمىزنى كۆكتاتلارنىڭ بازار باھاسىنى بىلىپ كىلىشكە ئەۋەتتى. بازارغا بىرىپ بارلىق كۆكتاتلارنىڭ بىر قۇر سۈرۇشتۈردۇق. باھا پەرقى بىزنى چۆچۈتتى. مەسلەن؛ پارچە باھاسى بازاردا 1.20يۈەنلىك يىسىۋىلەك 5.00يۈئەن. بىر كىلوسى 1.50يۈئەنلىك كەرەپشە 8.00يۈەن. كىلوسى ئىىكى يۈئەنلىك پەمىدۇر 8.00يۈئەن........... شۇنىڭ بىلەن مەكتەپ مۇدىرى ھىلىقى خوجايىنغا سەن بەك قىممەت بىرىدىكەنسەن. ئەرزانراق بەرمىسەڭ ئالمايمىز دىۋىدى،ھىلىقى خوجايىن دەرھال بىرىگە تىلفۇن قىلىپ مۇدىرغا بەردى.
  ئاخىردا يەنە شۇ خوجايىن بۇرۇنقى باھا بويىچە ئاچىقىۋىرىدىغان بولدى. مۇدىر بولسا ئىدارىنىڭ پارتكۇم سېكرىتارى تىلفۇن قىلىپتۇ ،مەندە نىمە ئامال دەپ ساڭگىلاپ ئولتۇردى. قىزىق يىرى كۆكتات ئاچىقىدىغان بۇ خوجايىن بىرقانچە ئاي ئاۋال مەكتەپ مەيدانىنى رىمۇنىت قىلىپ يۈرىتتى. ئەقىللىق خوجايىن شۇ جەرياندا ئوقۇغۇچىلارنىڭ تاماق پۇلىغا يوشۇرۇنغا غايەت زور پايدىنى بىلىپ قاپتۇ. بۇ يەردە قىستۇرۇپ قويۇشقا تىگىشلىك بىر نۇقتا بار-- دۆلىتىمىزنىڭ قانۇنىدا مىللەتلەرنىڭ ئۆرۈپ -ئادىتىگە ھۆرمەت قىلىش دەپ بار. بۇيەردە قوي گۆشىنىڭ باھاسى ھەققىدە توختالمىدۇق..........
  دىمەك ، شۇنداق بولغاچقا ئوقۇغۇچىلارنىڭ تامىقىنىڭ سۈپىتىمۇ تەبى ھالدا<<ياخشى>>بولىدۇ ئەلۋەتتە....
  يۇقارقى بىرقاتار مەلۇماتلاردىن دۆلەتنىڭ بۇ مەبلەغلىرىنىڭ ئاران تۆتتىن بىرىنىڭ تىگىشلىك جايىغا بارالايدىغانلىقنى بىلمەك تەس ئەمەس.قالغان تۆتتىن ئۈچ قىسىم پايدىنىڭ كىمگە قالىدىغانلىقى، بۇيەردىكى ئۆمۈچۈك تورىدەك دىگىلى بولمايدىغان مۇناسىۋەتلەرمۇ كۆپچىلىككە سىر بولمىسا كىرەك.
