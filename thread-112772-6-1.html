سىز يەنە نورۇز بايرىمىنىمۇ تىلغا ئېلىپ ئۈلگۈرۈپسىز بىلىپ قويۇڭ ئۇ بىزنىڭ بايرىمىمىز ئەمەس بەلكى پاراسىلارنىڭ بايرىمى . ئەبو داۋود ، نىسائى ئەنەس رەزىيەللاھودىن رىۋايەت قىلغان ھەدىستە پەيغەمبەر ئەلەيھىسسالام مەدىنىگە كەلگەندە ئۇلار يىلدا ئىككى قېتىم بايرام قىلاتتى پەيغەمبەر ئەلەيھىسسالام بۇ ئىككى كۈندە بايرام قىلىشنىڭ سەۋەبىنى سورىغاندا ئۇلار بىز جاھالەت دەۋىردىمۇ بۇ ئىككى كۈندە بايرام قىلاتتۇق دىيىشكەن پەيغەمبەر ئەلەيھىسسالام ئۇلارغا ئاللا سىلەرگە بۇ ئىككى كۈندىنمۇ ياخشىراق بولغان ئىككى كۈننى ئالماشتۇرۇپ بەردى ئۇ ئىككى كۈن روزا ھېيت ۋە قۇربان ھېيت كۈنىدۇر دېگەن . ئەلبانى بۇ ھەدىسنى سەھىھ دېگەن .
يەنە پەيغەمبەر ئەلەيھىسسالام يات دىنسىز مىللەتكە ئوخشىۋالىدىكەن ئۇ كىشى شۇ مىللەتتىن ھىساپلىنىدۇ دېگەن . ئەبو داۋود رىۋايىتى.
قالدىسىنى مۇشۇ ھەدىسلەرگە باغلاپ ئۆزىڭىز تەھلىل قىلىۋالارسىز .
ھەراس سىز دىنىمىزدىكى مەزھەبلەرنى سۆكۈپ ئارام بەرمەيدىغاندەك تۇرىسىز ئۆزىڭىزنىڭ يەنە بىر مەزھەبنى ئوتتۇرغا ئەچىقىۋاتقانلىقڭىزنى تۇيمىدىڭىزمۇ؟ سىزنىڭ يەنە تەكرارلاپ دېسەم دىنى مەلۇماتىڭىز بەكلا توۋەنكەن پەيغەمبەر ئەلەيھىسسالامنىڭ ‹‹ ...ھەربىر ئەر ئايالغا بىلىم ئۈگۈنۈش پەرىزدۇر ›› دېگەن ھەدىسىنىڭ روھى بويىچە بوش ۋاقىتلىرىڭىزدا ئانچە -مۇنچە دىنى كىتاپلارنى كۆرۈپ قويۇڭ بولمىسا......
سىز يەنە ھەيكەل توغۇرسىدىمۇ پىكرىڭىزنى غىل-پال ئوتتۇرغا قويۇپ ئۆتۈپسىز . ئەبو داۋود، تىرمىزى ۋە نەسائىيلار رىۋايەت قىلغان ھەدىستە ‹‹ جىبرىئىل پەيغەمبەر ئەلەيھىسسالامنىڭ ئۆينىڭ دەرۋازىسى ئالدىدا بىر ھەيكەل بولغانلىقى ئۈچۈن پەيغەمبەر ئەلەيھىسسالامنىڭ ئۆيىگە كىرمىگەن ئىككىنجى كۈنىمۇ ئۆيگە كىرمەي پەيغەمبەر ئەلەيھىسسالامغا ‹‹ھەيكەلنىڭ بېشىنى كەستۈرۋەتكىن دەرەخكە ئوخشاپ قالسۇن ›› دېگەن .
بىر قىسىم ئالىملىرىمىز ھەرقانداق جانلىقنىڭ كەم بولسا ياشىيالمايدىغان بىر ئەزاسى ھەيكىلىدە كەم بولسا بۇنداق  پۇچۇق ھەيكەل ھالالدۇر دېگەن........

ئالىي مەكتەپتىكى بىرەر قىز سىزنى بەك قۇيۇۋەتكەنمۇ بىر ئۆمۈر سىز شۇلارنى غاجاپ ئۆتەمسىز
نېمە غاجاپ يۈرسىڭىز قوساق تويامدىكەن.
شۇ ئالى دەرىجىلىك پىكاپ بىلەن قىزلارنى يولدىن چىقىردىغانلارمۇ سىلىگە ئوخشاش ئەركەك توردا تەربىيە قىلىپ ئەمەليەتتە شۇنداق قىلدىغان ھەجەپ بەك بىلىدىكەنلا ئالى مەكتەپ قىزلىرىنى كەسىپلىرى دەللالمۇ نېمە...؟
شۇنداق سودا قىلىپ كۆنۈپ قالغانمۇ يا....
شۇ  قىزلارنى يولدىن چىقىرۋاتقانلار كۆپۈنچىسى ئازراق پۇل تاپسا خوتۇننى 4 قىلىمىز ساۋاب تاپىمىز دەيدىغان
سىلىدەك جا موللا ياكى دىننىڭ دۈشمەنلىرى...
خىجىل بولماي قىزلارنى غاجاپ جان باقىدىكەنلا ئەجىبا سىلىدەك بىر ئەركەكنىڭ شۇلارنى ياخشى يولغا باشلىيالمىغانلىقلىرىدىن ئەيمەنمەيدىغان ئوخشىمامسىز...
ياكى غاجاپ بەرسە ساۋاب تاپامدىكەن...
سۆيۈشنى بىلگەنلەر سۆيۈملۈك ئەزەل.

ئىلىم ئۈگەنمەك ھەر بىر مۇسۇلمانغا پەرزدۇر .
ئىلىمنى ئەخلاقسىز نائەھلىگە ئۈگەتمەك توڭگۇزنىڭ بوينىغا گۆھەر ، ئۈنچە ۋە ئالتۇن ئاسقانغا ئوخشايدۇ ... «تولىستوي» جوڭگۇ مۇسۇلمانلىرى _ 58بەت .

تولىستوي گەرچە «ئىقرارنامە»سىدە '' مەن ئۆتكۆزمىگەن جىنايەت قالمىغان '' دېگەن بولسىمۇ ئۇ بەرىبىر مەشھۇر ئەدىب. مۇھىمى ئۇنىڭ ئىقرارىدا ئاجايىپ سەمىمىيەت ،پۇشايمان ھەققە قايتىدىغان تەۋبە - ئىستىغپار تۇيغۇسى ئىنتايىن كۈچلىك . مۇشۇ سەۋەپتىن ئۇ ھەقنى تاپقان بولسا كېرەك . شۇڭا ئۇ« ئىقرارنامە»ىسىدە ‹‹ئاللاھنىڭ مەۋجۇتلۇقىغا ئىشەنچىمنى يوقاتقان ۋاقىتلاردىكى ھاياتىمنى ھايات دىمەيمەن ، مەن پەقەت ئۇنى ھىس قىلغان ۋە ئىزدىگەن چاغلىرىمدىلا ، ھەقىقىي ھاياتتا ياشايمەن . سەن نىمىنى ئىزدەۋاتىسەن ؟ دېگەن بىر سادا ھەمىشە قەلبىم ئاسمىنىدا ياڭراپ تۇرىدۇ ، ئۇنىڭسىز ياشىغىلى بولمايدۇ ، ئاللاھنى تونۇش بىلەن ھايات ياشاش ئەملىيەتتە بىر ، ئاللاھ ھاياتتۇر » دەپ يازغان . يەنى ئۇ مۇنداق دەپ يازغان ؛ ‹‹ ھىديەت يولىنى تاپقانلار بىھۇدە مۇنازىرلەرگە بىرىلمىسە ئىدى ، بۇ يولدىن ئاداشمىغان بولاتتى ››
ئاداشماسلىقىم ئۈچۈن ، شۇنداقلا مۇسۇلمانلىق پەزىلىتىم ئۈچۈن گەپ- سۆز ۋە پىكىرلەردە ئەخلاقلىق بولۇپ ، باشقىلارغا ياخشى تەسىرات قالدۇرۇپ ئىسىل خىسلەتلەرنى نامەيەن قىلاي .

ئىنساننىڭ تىلى ئىنتايىن  تاتلىق ، شۇنداقلا ئىنتايىن زەھەردۇر !  

تىل دىگەن بىر ئادەمنىڭ ئەخلاق- پەزىلىتى ، مەدىنىيەت ساپاسنىڭ ئەكس ئىتىشىدۇر . باشقىلار بىلەن بولغان ئالاقىدە ، گەپ - سۆزدە ئەدەپلىك ، سىلىق- سىپايە ، گۈزەل سۆز ئىبارىلەر بولغاندا ، شۇ ئادەمنىڭ ئىسىل ئەخلاق - پەزىلىتىنىڭ نامايەندىسىنى كۆرۈپلا قالماي ،يەنى ئۇ ئىچكى ساپاسنىڭ ئىنتايىن پاكىز ئىكەنلىكىنى بايقىغىلى بولىدۇ .
ئەجداتلىرمىز «تىلى بۇزۇقنىڭ دىلى بۇزۇق » دەپ توغرا ئىيتقان . تىل ھەم دىل بۇزۇلسا ئىنسان ئەخلاق كاتوگىرىيىسىدىن پۈتۈنلەي چىكەتكەن بولىدۇ . پەيغەمبىرمىز مۇھەممەد ئەلەيھىسالام ئەقىدە ، ئىبادەت قاتارلىق ئىسلامنىڭ ئاساسى گەۋدىسىگە ئەخلاقنى چەمبەرچاس باغلاپ ، ئەخلاقى يوق ئادەمنىڭ ھەرقانداق ئىشنى قىلىشتىن يانمايدىغانلىقىنى ، ئەخلاقتىن ئايرىلغان ئەقىدە ، ئىبادەت ، ئىلىم ۋە بىلىملەرنىڭ قىممىتى ۋە رولى بولمايدىغانلىقىنى تەكىتلەپ مۇنداق دىگەن : « ئەگەر ھايا قىلمىساڭ ، خالىغىنىڭنى قىل » ( بۇخارى رىۋايەت قىلغان )
دېمەك بۇ ھەدىستە ئەخلاقسىز ئىنساننىڭ ھەرقانداق  شەرمەندە، پەستىۋەت  ئىشلارنى قىلىشتىن يانمايدىغانلىقى ئىلگىرى سۈرۈلگەن .
پەيغەمبەر ئەلەيھىسالامنىڭ 1400 يىل ئىلگىرى قىلغان سۆزلىرى ،ئىلىم - پەن ئالىملىرى تەرپىدىن ئىسپاتلىنىپ ، ئاشۇ ئالىملار ئىسلامغا كىرىپ ، ئىسلام دىنىنىڭ ھەق بىر دىن ئىكەنلىكىنى تونۇپ يىتىۋاتىدۇ . خوش ئەمدى بىزچۇ ؟شۇنداق مۇنبەردە ئىسلام توغرلۇق تېما يوللانسا يەكى ھەدىس بايان قىلىنسا ، بىر - بىرىمىزنى يىڭىش ئۈچۈن ئىتاپاقسىزلىق تىرىپ ، مەنمەنچىلىك قىلىپ ، بىلەرمەنلىك قىلىپ ، بىر - بىرىمىزگە ئازاردىن باشقىنى بەرمەيۋاتىمىز .
مەن ئاخىردا ئەھمەد يەسەۋى ھېكمەتلىرىدىن بۇنى قالدۇراي :

سۇننەت ئەرمىش ، كاپىرغىمۇ بەرمە ئازار ،
كۆڭلى قاتتىق دىل ئازاردىن خۇدا بىزار .

سۈكۈت قىلىشنى بىل ،ئەممە رەزىللەرنىڭ ئالدىدا ئەمەس !
