تۇنجى قېتىملىق ئىلقۇت «يېڭى خوجايىن» مۇكاپاتلىق تۈر ئېچىش مۇسابىقىسىنىڭ بىرىنچى باسقۇچلۇق مۇسابىقە ئەھۋالىدىن خۇلاسە



ئىلقۇت كارخانىچىلار تورىنىڭ 2013-يىلى 11-ئاينىڭ 4-كۈنىدىكى خەۋىرى، ئىلقۇت ئىگىلىك قانىلىدىكى ھەر قايسى جاپاكەش باشقۇرغۇچىلارنىڭ جاپالىق تىرىشىشى، يېڭىلىق يارىتىشى ۋە كوللېكتىۋىزملىق روھىنىڭ تۈرتكىسىدە، تۇنجى قېتىملىق ئىلقۇت «يېڭى خوجايىن» مۇكاپاتلىق تۈر ئېچىش مۇسابىقىسى ئىلقۇت ئىگىلىك قانىلىدا ئورۇنلاشتۇرۇلۇپ، مۇسابىقىنىڭ دەسلەپكى باسقۇچى مۇشۇ ئاينىڭ 3-كۈنى (يەكشەنبە) كەچ ئۈرۈمچى ۋاقتى سائەت 7:00 دە ئىلقۇت ئىگىلىك ۋاي ۋاي قانىلىدا تۆت سائەتكە يېقىن داۋاملىشىپ، ناھايىتى غەلىبىلىك ئاياقلاشتى.
بۇ قېتىملىق مۇسابىقىگە ئالاھىدە تەكلىپلىك باھالىغۇچىلار، مۇسابىقىگە قاتناشقۇچى ماھىرلار، قانال باشقۇرغۇچىلىرى شۇنىڭدەك سىرتتىن قاتناشقان باشقا ئەزالار بولۇپ 100 گە يېقىن ئادەم قاتناشتى.
مۇسابىقە باشتىن ئاخىرى ناھايىتى تەرتىپلىك، جانلىق ئېلىپ بېرىلىپ، مۇسابىقىگە قاتناشقان ماھىرلارنىڭ دادىللىقى، رىياسەتچىلەرنىڭ جانلىقلىقى، باھالىغۇچىلارنىڭ كۈچلۈك مەسئۇلىيەتچانلىقى شۇنىڭدەك مۇسابىقىگە سىرتتىن قاتناشقان قېرىنداشلارنىڭ ئاكتىپچانلىقى نەتىجىسىدە، بۇ قېتىملىق مۇسابىقىنىڭ تۇنجى باسقۇچى ئەسلىدىكى ئويلىغاندىنمۇ نەچچە ھەسسە ياخشى ئېلىپ بېرىلىپ، پۈتكۈل مۇسابىقىگە قاتناشقۇچىلارنى ناھايىتى ھاياجانغا سالدى.
بۇ قېتىملىق مۇسابىقىنىڭ تۇنجى قېتىملىق مۇسابىقە بولۇشى، مۇسابىقە ئورۇنلاشتۇرغۇچىلار ۋە پىلانلىغۇچىلارنىڭ تەجرىبىسىنىڭ يېتەرلىك بولماسلىقى شۇنىڭدەك تەشۋىقات ئىشلىرىنىڭ دېگەندەك ياخشى بولۇپ بولالماسلىقىدەك بىر تۈرلۈك سەۋەبلەر تۈپەيلىدىن، مۇسابىقىگە تىزىملاتقان ماھىر 14 نەپەر، ئەمما مۇسابىقىگە ئەمەلىي قاتناشقان ماھىر 11 نەپەر، قالغان ئۈچ نەپەر ماھىر تورغا ۋاقتىدا چىقالمىغانلىق ياكى سىرتلاردا بولۇپ قېلىش سەۋەبلىك مۇسابىقىگە كېلىپ بولالمىغاچقا مۇسابىقىگە قاتنىشىش سالاھىتى ئېلىپ تاشلاندى.
11 نەپەر ماھىرنىڭ كەسكىن رىقابەتلىشىشى نەتىجىسىدە ئەڭ ئاخىرى 9 نەپەر ماھىر ئىككىنچى باسقۇچلۇق مۇسابىقىگە كىرىش سالاھىيىتىگە ئېرىشتى. (ئىككىنچى باسقۇچلۇق مۇسابىقىگە ئائىت ئۇچۇرلار داۋاملىق ئىلقۇت كارخانىچىلار مۇنبىرىنىڭ «يېڭى خوجايىن» سەھىپىسىدە ئېلان قىلىنىدۇ، دىققىتىڭىز مۇنبەردە بولغاي!)

مۇسابىقىگە ئالاھىدە تەكلىپ قىلىنغان باھالىغۇچىلار:
1.         خوتەن ناھىيەلىك مەدەنىيەت-تەنتەربىيە ئىدارىسىدىن ئېزىز مەمتىمىن
2.         يەكەن ناھىيەلىك يېزا ئىگىلىك تەرەققىيات بانكىسىدىن تۇرغۇنجان
3.         قەشقەر ۋىلايەتلىك ئايرودرومىدىن ئوسمانجان
4.         شىنجاڭ ئۇنىۋېرسىتېتى فىلىلوگىيە ئىنىستىتۇتىنىڭ ياش دوكتۇرى سەمەتجان مەمتىمىن
5.         قەشقەر ۋىلايەتلىك مائارىپ ئىنىستىتۇتىدىن ئىمىنجان
6.         فىرانسىيەدە ئىقتىساد پەنلىرى بويىچە ماگىستىر ئاسپىرانتلىقتا ئوقۇپ بىلىم ئاشۇرغان، شىنجاڭ يۇنىل ئوپال خەلقئارا سودا چەكلىك شىركىتىنىڭ باش دىرېكتورى، بىلكان ئېلېكتىرون پەن-تېخنىكا شىركىتىنىڭ ستېىش بۆلۈمىنىڭ نازارەتچىسى ئىلھامجان
7.         خوتەن ۋىلايەتلىك قاتناش تىرانسىپورت ئىدارىسىدىن ئەكبەر ئوسمان
8.         ئىلقۇت كارخانىچىلار تورىنىڭ قۇرغۇچىسى، ئۈرۈمچى پەرىشتە تور پەن-تېخنىكا چەكلىك شىركىتىنىڭ قۇرغۇچىلىرىنىڭ بىرى، تەڭرىتاغ ئۇيغۇرچە يانفون گېزىتى بۆلۈمىدىن ياسىنجان مۇھەممەدروزى ئىلقۇت

مۇسابىقە رىياسەتچىلەرنىڭ جانلىق رىياسەتچىلىكى ئاستىدا، پۈتكۈل مۇسابىقىگە قاتناشقۇچىلار (ماھىرلار، باھالىغۇچىلار ۋە سىرتتىن قاتناشقۇچىلار) جۇشقۇنلۇق بىلەن مۇسابىقە مەيدانىغا كىرىپ ئورۇنلىشىپ، پۈتكۈل مۇسابىقە مەيدانى ناھايىتى ھاياجانلىق كەيپىياتقا كىردى.
ئەڭ ئالدى بىلەن رىياسەتچىلەر مۇسابىقىنىڭ رەسمىي باشلانغانلىقىنى جاكارلاپ، تەكلىپلىك باھالىغۇچىلارنى تونۇشتۇرۇپ چىققاندىن كېيىن، ئىلقۇت ئىگىلىك قانىلىنىڭ باشقۇرغۇچىسى ئادۋۇكات (رەيھانگۈل تۇرغۇن، جۇڭگو سىياسىي قانۇن ئۇنىۋېرسىتېتىنىڭ قانۇنشۇناسلىق ئاسپىرانتى) ئۆزىنىڭ بۇ قېتىملىق مۇسابىقىگە ئاتاپ يازغان ئىلقۇت مارشىنى ناھايىتى جانلىق دېكلىماتسىيە قىلىپ ئۆتۈپ، پۈتكۈل مۇسابىقە كەيپىياتىنى ناھايىتى جانلاندۇرىۋەتتى.
ئارقىدىن ماھىرلار تەلەپ بويىچە مۇسابىقىگە چۈشكەن بولۇپ، مۇسابىقە ئالدى بىلەن ئۆزىنى قىسقىچە تونۇشتۇرۇپ، ئاندىن پائالىيەت ئورۇنلاشتۇرغۇچىلىرى ۋە پىلانلىغۇچىلىرى تەرىپىدىن ئالدىن تەييارلاپ قويۇلغان 3 تۈرلۈك سوئالدىن چەك تارتىپ جاۋاب بېرىدۇ (چەك تارتىش ئۇسۇلى: سۇئاللار ماھىرلارنىڭ ئېرادىسىنى سىناش، ھەمكارلىق ئىقتىدارىنى سىناش، خەتەرنى تەھلىل قىلىش ۋە كونترول قىلىش ئىقتىدارىنى سىناشتىن ئىبارەت 3 تۈرگە بۆلۈنگەن بولۇپ، ھەر بىر تۈر ئىچىدە 14 تىن جەمئىي 42 سۇئال تەييارلانغان. ماھىر ئايرىم-ئايرىم ھالدا ھەر بىر تۈردىكى 1 دىن باشلاپ 14 كىچە بولغان سانلار ئىچىدىن خالىغان بىر ساننى تاللايدۇ، ئاندىن ئالاھىدە تەكلىپلىك ئەزا سابا (گۈزەلنۇر پايتەخت پىداگوگىكا ئۇنېۋىرستېتى ئاممىۋىي ئىشلارنى باشقۇرۇش كەسپى 4 –يىللىق ئوقۇغۇچىسى) شۇ سانغا ماس سوئالنى ئوقۇپ ئۆتىدۇ، ئاندىن ماھىر نەق مەيداندا سوئاللارغا جاۋاب بېرىدۇ.)
سوئاللار ئالدىن تەييارلانغان، ئەمما ئاشكارىلانمىغان، شۇڭا ماھىرلار پەقەت شۇ ۋاقىتنىڭ ئۆزىدىلا ئاندىن سوئالنىڭ مەزمۇنىدىن خەۋەردار قىلىنغان. ھەر بىر ماھىر ئايرىم-ئايرىم ھالدا ئۈچ سوئال تارتىدۇ ۋە شۇ ئۈچ سوئالغا جاۋاب بېرىدۇ. سوئاللار ناھايىتى جانلىق، ئەتراپلىق تەييارلانغان بولۇپ، باھالىغۇچىلار ھەر بىر ماھىرنىڭ نەق مەيداندىكى ئىپادىسىگە ئاساسەن ماھىرلارغا ئۆزىنىڭ تەكلىپ-پىكىرلىرى ۋە ساقلانغان مەسىلىلەر ھەققىدە مەسلىھەت بېرىپ، ئومۇمىي نومۇرنى قويۇپ بولۇپ، نومۇر خاتىرىلىگۈچىلەرگە تەمىنلەپ بەرگەندىن كېيىن، نومۇر خاتىرىلىگۈچىلەر نومۇرنى ئىستاتىستىكىلاپ، جەملەپ ئوتتۇرىچە نومۇرنى ھېسابلاپ چىقىرىپ ئاخىرقى نەتىجە قىلىنغان. خاتالىق نىسبىتىنى ئەڭ تۆۋەن چەككە چۈشۈرۈش ئۈچۈن، نومۇر خاتىرىلەشكە مەخسۇس ئىككى ئادەم ئورۇنلاشتۇرۇلغان ۋە بۇ ئىككى ئادەم ئۆزئارا مەخپىي تۇتۇلۇپ، ئۆزئارا ئالاقىلىشىشىنىڭ ئالدى ئېلىندى. ماھىرلار تاللىماقچى بولغان سۇئاللارنىڭ تەكرارلىنىپ قېلىشىنىڭ ئالدىنى ئېلىش ئۈچۈن ئالدىنقى ماھىردىن سورۇلۇپ بولغان سۇئاللارنىڭ رەت تەرتىۋىنى تاللاشقا بولىدىغان سۇئاللار قاتارىدىن ۋاقتى-ۋاقتى چىقىرىپ ماھىرلارغا ئەسكەرتىپ ئۆتۈلدى.
باھالىغۇچىلارنىڭ باھالاش ئۆلچىمى ئايرىم ئايرىم ھالدا تىلى راۋان بولۇش (10 نومۇر)، لوگىكىلىق بولۇش (10 نومۇر)، ئىرادىسى (20 نومۇر)، كوللېكتىۋىزملىق روھى (20 نومۇر)، ئەتراپلىقلىقى، ئەمەلىيلىكى، خەتەرنى تىزگىنلەش ئىقتىدارى (20 نومۇر)، سوئالغا جاۋاب بېرىش ئەھۋالى (20 نومۇر) بويىچە، جەمئىي نومۇر 100 نومۇر بويىچە باھالاپ نومۇر قويۇپ چىقىدۇ.

باھالىغۇچىلارنىڭ كونكرېت باھالاش ئۆلچىمى تۆۋەندىكىچە:
تىلى راۋان بولۇش (جەمئىي نومۇر: 10 نومۇر):
تىلى راۋان، ئېنىق، چۈشەنمىگەن سۆز جۈملىلەر يوق، سۆزلەش سۈرئىتى نورمال. (8~10 نومۇر)
سۆز جۈملىلەر جانلىق، ئىنتايىن ئاز چۈشىنىكسىز ئىبارىلەر بار. (5~8 نومۇر)
ئاساسىي مەزمۇننى چۈشەنگىلى بولىدۇ، تىلى جانلىق، بەزى گەپلەر ئوچۇق ئەمەس، تىل سۈرئىتى تېز ياكى ئاستا. (3~5 نومۇر)
گەپلەرنى ئاڭلاپ چۈشىنىش قىيىن، جانلىق ئەمەس. (0~3 نومۇر)
        
لوگىكىلىق بولۇش (جەمئىي نومۇر: 10 نومۇر):
سەۋەب نەتىجە مۇناسىۋىتى ئېنىق، قاتلاملارغا بۆلۈپ سۆزلىگەن، سوئاللارغا بەرگەن جاۋابلىرىدا مۇھىم بىلەن ئادەتتىكى ئايرىلغان، مۇھىم نۇقتىلارنى ئاساس قىلىپ سۆزلىگەن، قايىل قىلىش كۈچى بار (8~10 نومۇر)
سەۋەب نەتىجە ئايرىلغان، قاتلاملارغا بۆلۈنگەن لېكىن تازا ئېنىق ئەمەس، سوئالنىڭ جاۋابلىرىدا ئاساس نۇقتىنى ئاكتىپ ھالدا سۆزلىمىگەن، مۇھىم بىلەن ئەمەس ئىشنى ئېنىق ئايرىمىغان. (5~8 نومۇر)
سەۋەب نەتىجە مۇناسىۋىتى تازا ئېنىق ئايرىلمىغان، قاتلاملارغا بۆلمىگەن، مۇھىم نۇقتىنى بىلمىگەن، مۇھىم ئىش بىلەن ئەمەسنى پەرق ئېتەلمىگەن. (3~5 نومۇر)
سەۋەب نەتىجە مۇناسىۋىتى يوق، قاتلاملارغا بۆلمىگەن، سوئالغا بەرگەن جاۋابى مۇۋاپىق ئەمەس، ئەقىل بىلەن ئەمەس ھېسىيات بويىچە جاۋاب بەرگەن. (0~3 نومۇر)

ئىرادىسىنى باھالاش (جەمئىي نومۇر: 20 نومۇر):
ئىرادىسى ئىنتايىن كۈچلۈك، ئىرادىگە مۇناسىۋەتلىك سوئاللارغا ياخشى جاۋاب بەرگەن. ئىككىلەنمىگەن، پىلانلىرىدا چېكىنىش يولى يوق، ئەڭ يامان ئەھۋالدا يەنىلا ئۆزىگە تايىنىدۇ. (15~20 نومۇر)
ئىرادىسى كۈچلۈك، جاۋاب ياخشى بولغان، ئەمما مۇھىم مەسىلىلەردە ئازراق ئىككىلىنىش بار، ئۆزىگە ئازراق يول ئېلىپ قويغان، ئەڭ يامان ئەھۋالدا ئۆزىنىڭ ھازىرقى خىزمىتىگە ۋە ئاتا ئانىسىغا يۆلىنىدۇ. (10~15 نومۇر)
ئۆزىگە ئىشەنچىسى ۋە ئىرادىسى بار، جاۋاب ياخشى، ئەمما مۇسابىقىدە خېلى مەسىلىلەردە ئىككىلىنىش بار، ئىختىيارى، ھەم ئۆزىگە چېكىنىش يولى قالدۇرۇپ قويغان. (5~10 نومۇر)
ئىشەنچىسى يوق، تەلەي سىناپ باقماقچى. (0~5 نومۇر)

كوللېكتىۋىزملىق روھىنى باھالاش (جەمئىي نومۇر: 20 نومۇر):
كوللېكتىپ تۇيغۇسى كۈچلۈك، كوللېكتىپ بىلەن خىزمەت قىلغان تەجرىبىسى بار، ھەمكارلىققا ئائىت سوئاللارغا ناھايىتى ياخشى جاۋاب بەرگەن، ئالاقە ئىقتىدارى كۈچلۈك، رەھبەرلىك، تەشكىللەش ئىقتىدارى كۈچلۈك، (15~20 نومۇر)
ھەمكارلىققا ئائىت سوئاللارغا ياخشى جاۋاب بەرگەن، ئالاقە قىلىشتا مەسىلە يوق، كوللېكتىپ تۇيغۇسى بار. (10~15 نومۇر)
ھەمكارلىققا ئائىت سوئاللارغا لاياقەتلىك جاۋاب بەرگەن، ئالاقە قىلىشتا سەل مەسىلە بار، لېكىن باشقىلار بىلەن ياخشى چىقىشىپ ئۆتەلەيدۇ. (5~10 نومۇر)
ئالاقە قىلىشتا ئېغىر مەسىلە بار، كوللېكتىپ تۇيغۇسى يوق. (0~5 نومۇر)

ئەتراپلىقلىقى، ئەمەلىيلىكى، خەتەرنى تىزگىنلەش ئىقتىدارى قاتارلىقلارنى باھالاش (جەمئىي نومۇر: 20 نومۇر):
ئۆزىنىڭ كېيىن ئۇچرايدىغان قىيىنچىلىقلىرىنى مۆلچەرلىيەلەيدۇ، خەتەرنى بايقاش ۋە خەتەرنى تىزگىنلەش ئىقتىدارى كۈچلۈك، مۇناسىۋەتلىك سوئاللارغا ياخشى جاۋاب بەرگەن. (15~20 نومۇر)
سوئاللارغا بەرگەن جاۋاب ياخشى، خەتەرنى ھېس قىلالايدۇ ئەمما ماس كەلگۈدەك تەدبىرلىرى يوق، خەتەرنى تىزگىنلەش ئىقتىدارى ئاجىز. (10~15 نومۇر)
سوئاللارغا بەرگەن جاۋاب لاياقەتلىك، ئومۇمى جەھەتتىن خەتەر بارلىقىنى بىلىدۇ ھەم مەغلۇب بولۇشقىمۇ كۆڭلىدە سان بار، ئەمما نېمە خەتەرلىكىنى ئىنچىكە بىلمەيدۇ. (5~10 نومۇر)
خەتەر قارىشى يوق، ئۆزىگە ھەددىدىن زىيادە ئىشىنىدۇ، مەغلۇب بولمايمەن دەپ قارايدۇ. (0~5 نومۇر)
        
سوئالغا جاۋاب بېرىش ئەھۋالىنى باھالاش (جەمئىي نومۇر: 20 نومۇر):
ھەممە سوئاللارغا قاراتمىلىقى كۈچلۈك جاۋاب بەرگەن، تېمىدىن چەتنىمىگەن، ئىنكاسى تېز. (15~20 نومۇر)
بەزى سوئاللارغا ئىككىلىنىپ قالغان، بەزى سوئاللارغا جاۋاب بېرەلمىگەن، بەزى سوئاللاردا تېمىدىن چەتنەپ كەتكەن، ئەمما ئومۇمى جەھەتتىن ياخشى. (10~15 نومۇر)
سوئاللارغا بەرگەن جاۋابى لاياقەتلىك، ئىنكاس تېز ئەمەس. (5~10 نومۇر)
بەزى سوئاللارنى ئويلاپمۇ باقمىغان، تەييارلىق پۇختا ئەمەس. (0~5 نومۇر)

ئاخىرقى سۆز
بۇ قېتىملىق مۇسابىقە گەرچە تۇنجى سان مۇسابىقىنىڭ تۇنجى باسقۇچى بولغان بولسىمۇ، يۇقىرىدا تىلغا ئېلىپ ئۆتۈلگەندەك ھەر خىل يېتەرسىزلىكلەر ساقلانغان بولسىمۇ، ئەمما مۇسابىقە ھەقىقەتەن جانلىق، ئوڭۇشلۇق، تەرتىپلىك ئېلىپ بېرىلىپ، پۈتكۈل قاتناشقۇچىلارنى ئالاھىدە خۇرسەن قىلدى.
ئاخىرىدا مۇسابىقىگە نەق مەيداندىن قاتناشقانلار مۇسابىقە ھەققىدە ئۆزىنىڭ قەلب سۆزلىرى، مۇسابىقىدىن كېيىنكى تەسىراتىنى سۆزلەپ، مۇسابىقىنىڭ مۇۋەپپەقىيەتلىك ئېلىپ بېرىلغانلىقىنى يەنە بىر قېتىم مۇئەييەنلەشتۈردى.
ئاخىرىدا باھالىغۇچىلار ئايرىم-ئايرىم ھالدا ئۆز يۈرەك سۆزلىرىنى بايان قىلىپ، بۇ قېتىملىق پائالىيەتنىڭ تولىمۇ ئوڭۇشلۇق، تەرتىپلىك، نەتىجىلىك ئېلىپ بېرىلغانلىقىنى، پىلانلىغۇچىلار ۋە قانال باشقۇرغۇچىلىرىنىڭ ئەمگىكىنىڭ تولىمۇ ئەھمىيەتلىك ئەمگەك بولغانلىقىنى تولۇق مۇئەييەنلەشتۈردى.
نەق مەيداندا قاتناشقان باشقا قېرىنداشلارمۇ ناھايىتى ھاياجانلىق ئىلكىدە ئۆزىنىڭ قەلپ سۆزلىرىنى، مۇسابىقىدىن ئالغان تەسىراتىنى سۆزلەپ مۇسابىقىنىڭ نەتىجىسىنى يەنە بىر قېتىم مۇئەييەنلەشتۈردى. ئاخىرىدا شىركەت ۋەكىلىدىن خوتەن يانغىن مەدەنىيەت تارقىتىش چەكلىك شىركىتىنىڭ قۇرغۇچىلىرى سۆز قىلىپ، بۇ قېتىملىق مۇسابىقىنىڭ تولىمۇ ئۇتۇقلۇق بولغانلىقىنى، بۇنىڭدىن كېيىن بۇ خىل مۇسابىقىلەرنىڭ داۋاملىق ئېلىپ بېرىلىشىنى ئۈمىد قىلىدىغانلىقىنى ھەمدە مەبلەغ ۋە باشقا جەھەتلەردە قوللايدىغانلىقىنى بىلدۈرۈپ ئۆتتى.
ئەڭ ئاخىرىدا ھەممەيلەننى ئالاھىدە خۇش قىلىدىغان خۇش خەۋەر كېلىپ، شىنجاڭ فەتىھ كىيىم-كېچەكلىرى سودا چەكلىك شىركىتىنىڭ ئىككىنچى سان ئىلقۇت «يېڭى خوجايىن» مۇكاپاتلىق تۈر ئېچىش مۇسابىقىسىگە كېتەرلىك مەبلەغ بىلەن تەمىنلەيدىغانلىقىنى بىلدۈرۈپ، پۈتكۈل قانال باشقۇرغۇچىلىرى ۋە باشقا قېرىنداشلارنى ئالاھىدە خۇش قىلىۋەتتى بۇنىڭ بىلەن بۇ قېتىملىق مۇسابىقىنى ئورۇنلاشتۇرغان، پىلانلىغان، باشتىن-ئاخىرى جاپالىق ئەجىر سىڭدۈرگەن باشقۇرغۇچىلار ئۆزلىرىنىڭ ئەمگىكىنىڭ بىكارغا كەتمىگەنلىكىنى ئېيتىشىپ ناھايىتى خوشال كەيپىياتقا چۆمدى.
