                                                 ئۈستىلەنگەن قەلىب يارىسى-پوۋېسىت  
               
                                                                 (سەرسانى)                                               


              ئىنسان قەلبى سىرلارغا تولغان ، ئۆز نۆۋىتىدە كىشىلىك ھايات ئادەمگە ئۇخلاپ چۈشىدىمۇ ئويلاپ باقمىغان قىسمەتلەرنى ئاتا قىلىدىكەن .ھەم شۇ يولدا ھەر كويلارغا سالىدىكەن ، داپنى چېلىپ بىرىپ بىزنى ئۇسۇلغا سالىدىكەن .
              قىسقىغىنە ھاياتىمدا بىر ئادەمگە بۇنچىلىك ئىشىنىپ ، بۇنچىلىك ئەقىدە قىلىپ ، بەخىت ۋە خۇشاللىقلارغا ئىرىشىمەن ، ئۆزۈمنى يوقۇتۇپ قويغۇدەك دەرىجىدە دەرت-ئازابلارغا دۇچار بولىمەن دەپ ئويلاپ باققان ئەمەسمەن . لىكىن بۇ ھەقىقەت ، ئاقىۋەت مەن جېنىمدىنمۇ ئەزىز كۆرگەن ، شۇنىڭ ئۈچۈن مىنى تالاي نەرسىلەردىن ۋاز كەچتۈرگەن قىز ۋاپاسىزلىق قىلدىغۇ ، باشقا بىرىنىڭ كەينىدىن كەتتىغۇ ...بىر ئوغۇل بالا ئۈچۈن بۇ قانچىلىك ئەلەم ؟ ئەرلىك غورۇرىغا قىلىنغان ھاقارەت ، چىن سۆيگۈسىگە بولغان ئاسىلىق . ئەقىدىسىنى خازان قىلىش ، ئارزۇ - ئارمانلىرىنى يوققا چىقىرىش ...   مانا بۇ مىنىڭ قىسمىتىم. سۇ بىلەن يۇيۇپ چىقىرۋەتكىلى بولمايدىغان، ئۆچۈرگۈچ بىلەن سۈرتۈپ تازلىغىلى بولمايدىغان قەلىب دېغى . ئويلانماقتىمەن ، تېگى يوق خىياللار بۈگۈنمۇ مىنى ئۆز قوينىغا ئالماقتا ، يۈرەك قەپىسىمنى ئەسىر قىلىپ ئازاپ دېڭىزلىرىغا تاشلىماقتا.
  --مەردان ، تۇر ئورنىڭدىن ، چىقىپ بىر نەرسە يەيلى ئاداش . دوستلىرىمنىڭ سالالىرى بىلەن تاماققا چىقتۇق ، بۇمۇ مىنىڭ دوستلىرىمنىڭ مەن شەمسىيەدىن ئايرىلىپ كەتكەندىن بېرى مىنىڭ تېخىمۇ چۈشكۈنلىشىپ ئۆزۈمنى يوقۇتۇپ قويماسلىقىم ئۈچۈن كۆرسىتىۋاتقان تىرىشچانلىقى ئىدى . نىمىلا بولمىسۇن ئىچ ئىچىمدىن ئۇلاردىن رازى ئىدىم .
بىر دەسسەپ ئىككى دەسسەپ بىز دائىم بىللە تاماق يەيدىغان ئاشخانا ئالدىغا كەلدۇق ، ئىشىكتىن كىرەي دەپ تۇرشىمغا ئىچىدىن چىققان رەيھان مىنى چۈچىتىۋەتتى .                                    
--ھەي مەردانمۇ سىز ، كۆرۈنمەي كەتتىڭىزغۇ ، ئەھۋالىڭىزچۇ ، ياشى بۇلۇپ قالغانسىز ؟               
--- اللاھقا شۈكىرى ، ئۆزىڭىزچۇ ؟ قايتۇرۇپ سورىدۇم مەن .                                            
-- مەن دىگەن كىم ؟ بۇنىمۇ سوراش كىتەمدۇ ، ياخشى شۇ . ھە توغرا بىر نەرسە بار ئىدى سىز گە بىرىدىغان . رەيھان شۇنداق دىگىنىچە سومكىسىدىن بىر كونۋىرتنى ئېلىپ ماڭا ئۇزاتتى .
--نىمە ئۇ ؟ دىدىم مەن ھەيران قېلىپ ، چۈنكى رەيھان قانداق گەپ بولسا خەت يېزىپ يۇرمەي ئۇدۇللا دەيدىغان قىز ئىدى .                                                                                    
-- كۆرۈپ باقمامسىز ، مەن نەدىن بىلەي بىراۋ نىمە يازىدى بۇنىڭغا . رەيھان ھەييارلارچە كۈلۈپ قويۇپ مېڭىپ كەتتى .                                                                                          
           رەيھان مەن  بىلەن يېقىن ئۆتىدىغان دوستلىرىمدىن ئىدى ، شۇنداقلا شەمىسىيەنىڭ دوستى بۇلۇپ بۇرۇنلاردا شەمسىيە ئىككىمىز ئۈچۈن نۇرغۇنلىغان ياخشىلىقلارنى قىلغان . رەيھان سىنىپىمىزنىڭ ئەڭ شوخ قىزى ، ھەر قانداق چاغدا ھېچچىكىمگە گەپ بەر مەيدىغان ئۆزى ئاق كۆڭۈل ، ھەققانىيەت قارشى تولىمۇ كۈچلۈك قىز .                                                                                   
                خىياللىرىمنىى يىغىشتۇرۇپ  دوستلىرىمنىڭ قېشىغا كىردىم . مىنىڭ ئامىراقلىقىمنى بىلگەچكە پولا بۇيرۇتۇپ قويۇپتىكەن كۈلكە چاقچاقلار بىلەن تاماقنى يەپ ياتاققا قايتتۇق ، شەنبە كۈنى بولغاچقا باشقا ئىشىمىزمۇ يوق ئىدى. كارۋىتىمغا چىقىپ يېتىشقا تەمشىلۋىدىم يانچۇقۇمدىكى كونۋىرت ئىسىمگە كەلدى . ئېچىپ ھەيران قالدىم، ھەيران قالغىنىم ھەرگىزمۇ سۆيگۈ خېتى بولغانلىقى ئۈچۈن ئەمەس بەلكى مەن ئەزەلدىن ئۇنداق ئويلاردا بولمايدۇ دەپ قاراپ كەلگەن قىزدىن كەلگەن سۆيگۈ خېتى ئىدى .
                                                          سالام سىزگە                                             

                 ھۆرمەتلىك مەردان ، ياخشىمۇسىز ؟ تىنىڭىز سالامەتمۇ ؟ ھەيرانمەن مەردان ، شەمسىيەنىڭ قىلغانلىرىغا ھەيرانمەن . بەلكىم ئادەم ئىرىشىپ بولغاندىن كىيىن قەدىرلەشنى بىلمىسە كىرەك ، ۋە ياكى يېڭىلىق ئىزدىگۈسى ، ئوخشىمىغان پۇراقلارنى پۇرىغۇسى كەلسە كېرەك . بولدىلا مەردان ، بۇ گەپلەرنى قىلىپ قەلىپ يارىڭىزغا تۇز سەپكەندەك ئىش قىلماي . ئەسلى گەپكە كەلسەم مەن سىزنى سۆيىمەن مەردان . بۇ چاقچاق ئەمەس ، بەلكى ھەقىقەت. بەلكىم شەمسىيە سىزگە ۋاپاسىزلىق قىلمىغان ، سىزنى بۇنچە ئازاپلارغا دۇچار قىلمىغان ، ئەقىدىڭىزنى بۇنچىلىك خاراپ قىلمىغىنىدا ئىدى مەنمۇ ھەم بۇندىن بۇرۇنقى ئىككى يىلنى قانداق ئۆتكۈزگەن بولسام يەنە تەرىقىدە ، سىزنىڭ بەختىڭىزنى تىلەپ قولۇمدىن كەلگەن ئىشلاردا سىزگە ھەمدەم بۇلۇپ ئاددى دوستىڭىز بۇلۇپ سىزگە بولغان چوڭقۇر سۆيگۈمگە نوختىلارنى سېلىپ ئۆتۈپ كەتكەن بۇلار ئىدىم . تولىمۇ ئەپسۇس مەردان ، مەن ئەمدى چىدىمىدىم ، سىزنىڭ بۇنداق ئازاپ تارتىپ يۈرشىڭىزنى خالىمايمەن......

               روشەنگۈلنىڭ خېتى شۇ تەرىقىدە مىنى قاچاندىن بېرى ياخشى كۆرۈپ قالغانلىقى ، بۇ ئىككى يىلنى قانداق ئۆتكۈزگەنلىكى ، مەن خالىساملا شەمسىيەنىڭ ماڭا قالدۇرغان يۈرەك يارىللىرىنى داۋالاش ئۈچۈن تەييارلىقى قاتارلىقلار بايان قىلىنغان ئىدى . يۈرۈكۈم يەنە بىر قېتىم ئېچىشتى . ھېلىھەم مىنى چىن دىلىدىن سۆيۈپ مىنىڭ ئوتۇمدا پۇچىلنىۋاتقانلار ئازمۇ ؟ ئۇلارغا كىرەكلىك بولغان چىن مۇھاببەت ، سۆيگۈلەرنى بىرەلمەي مىنىڭ يۈرۈكۈم ئاز ئېچىشتىمۇ ؟ بىر ساپ قەلىبنىڭ يەنە بىر قەلىبكە ئىنتىلىشى ، سۆيگۈ تەلەپ قىلىشى گۇناھ ئەمەستە...ئۇلارمۇ مەن ئەينى چاغلاردا شەمسىيەگە ئاشىق بولغاندەق ماڭا ئاشىق بولغاندۇ ، ئۇلارمۇ سانسىز ئۇيقىسىز كىچىلەرنى مىنى ئويلاش ، ئىسمىمنى زىكىر قىلىش بىلەن ئۆتكۈزگەندۇ ...بىراق مەندە نە  ئامال ؟ سۆيمەي تۇرۇپ بىراۋغا يالغاندىن مۇھەببەت بەرگىلى بولمىسا ، يەنە كىلىپ قايسى بىرىگە بىرىپ بولالايمەن ؟  خىياللىرىم يەنە داۋاملاشقانمۇ بۇلار ئىدى دىلشاتنىڭ چاقىرىشى مىنى ئويغاتتى .                                   
--نىمە دىدىڭ ئاداش ؟ تازا ئاڭقىرالمىدىم ... ئۇنىڭ نىمە دىگىنىنى راستلا ئاڭلىمىغان ئىدىم .                                                                                      
--سىنى رەيھان چاقىرىۋاتىدۇ دەيمەن ، بىنانىڭ ئالدىدا قالدى . تىزراق چۈشسۇن دەيدۇ               
رەيھاننىڭ مىنى نىمە ئۈچۈن ئىزدەيدىغىنىنى بىلەتتىم ، شۇڭا رۇشەنگۈلنىڭ خېتىنى ئېلىپ پەسكە چۈشتۈم .                                                                                               
-- چاقىرىپتىكەنسىز ... سۇئال نەزىرى بىلەن ئۇنىڭغا قارىدىم .                                       
-- ھەئە . مەسىلىڭىزنى ئۆزىڭىز تاپشۇرامسىز باشقا گېپىڭىزمۇ بارمۇ ؟ ئۇنىڭ چاقچىقىدىن كۈلۈپ كەتتىم ۋە خەتنى ئۇنىڭغا سۇندۇم . رەيھان تىزلا كۆز يۈگۈرتۈپ چىقىپلا كونا ئادىتى بىلەن قايناشقا باشلىدى.                                                                                                
-- نىمە دىگەن دۆت قىزلار بۇ ، قىلىدىغانغا ئىش تاپالماي قاپتۇ زادى . ھەي ... نىمانچە ئاخماق ، ئوتەپ تېشىپ يەنە سىزگە يېپىشقىنىنى . بۇ قىزلارغا سىزدىن باشقا ئوغۇل بالا يوقمىدۇ ؟ نىمانچە قىلىدۇ دەيمەن ، ھەي ... رەيھان قاخشاپ بۇلۇپ ماڭا ئالىيىپ قاراپ قويدى . شەمسىيەدىن ئايرىلغان بىر نەچچە ئايدىن بېرى كەيپىياتىم ياخشى بولمىغاچقىمۇ بۇرۇنقى شوخلۇقلىرىمدىن ، گەپتانلىقلىرىمدىن ئەسەرمۇ قالمىغان ئىدى .                                                            
-- نىمە ؟ گاچا بۇلۇپ قالدىڭىزما ؟ سالام خەتتىن ئۆزىڭىزنى يوقۇتۇپ قويۇۋاتامسىز يە ؟ بولدى بولدى ،يۈرۈڭ دوغ ئىچىپ كىرىمىز . مەن مىھمان قىلىمەن بۈگۈن . رەيھانغا ياق دىيەلمەيتىم ، ئۇنىڭ مىجەزىنى بىلگەچكە ياق دەپ قۇتۇلالمايدىغىنىمنى بىلىمەن . بىللە كىتىۋاتىمىز ، رەيھاننىڭ نىمىلەرنى دەۋاتقىنىنى بىلمەيمەن . پۇتۇن دىققىتىم بىلەن بۇ بىگۇناھ قىزنى نىمە دەپ ياندۇرارمەن دىگەننى ئويلاۋاتىمەن . تۇيۇقسىز دۈمبەمگە يىگەن بىر مۇشتىن چۇچۇپ ئىسىمگە كەلدىم . مەكتەپ ئالدىدىكى سوغۇق ئىچىملىك دۇككىنى ئادىغا كىلىپ قاپتىمىز . بۇلۇڭدىكى ئورۇندۇققا بېرىپ ئولتۇردۇق .                                                                                             
-- ئىسىت ئىسىت ئاشۇ ئەخمەق قىزلارغا ......   رەيھاننىڭ ئاغزى يەنە ئېچىلدى .                 
-- بولدى قىلىڭ رەيھان ، ئۇلاردا گۇناھ يوق «كۆڭۈل دىگەن پادىشاھ ، ئىشەككىمۇ تارتىشا» دەپ ئاڭلىمىغانمۇ ؟ بولدى قىلىڭ ھە، بولمىسا ئۆزىڭىزمۇ شۇ كۈنگە قېلىپ زارلاپ يۈرۈپ قالماڭ يەنە . توختاپ قالسۇن ئۈچۈن ئۇنىڭغا چاقچاق قىلدىم .                                                         
-- نىمە؟ ئۆتەپ بېرىڭە ساراڭ . مەن ئۇنداق ئەخمەقلەردىن ئەمەس .ئادەم تاپالماي سىزگە كۆيۈپ قالامدىم ؟                                                                                                
-- ئۇنداق دىمەڭ دەيمەن ، راستىنلا شۇنداق بۇلۇپ قالمىسۇن ھى ھى ھى  ئۇنى تازا بىر تىرىكتۈرەي دىدىم .                                                                                                  
-- ئەگەر شۇنداق بولسىچۇ مەردان، مەن ئىسمىمنى يۆتكىۋىتىمەن ، مىنىڭ ئىسمىم رەيھان بولىدىكەن بۇ مۇمكىن ئەمەس .                                                                                    
-- راستما ؟ ۋاختى كەلگەندە ئىسىمنى ئۆزۈم سىزگە تاللاپ بەرمىسەم ھىساپ ئەمەس ھا ھا ھا .                                                                                       
-- نېرى بېرىڭ ، مۇمكىن ئەمەس . تۇلا ئۇخلىماي چۈش كۆرمەڭ .                                    
بارغانسىرى قېنىم قىززىپ چاقچاقنى يىغىۋالالماي قالدىم.                                                
-- باغلىشامدۇق ؟ ئۇتتۇرۋەتسىڭىزچۇ ؟ دىدىم ئۇنىڭغا مەنسىتمەسلىك نەزىرى بىلەن .                                                            
-- دىدىمغۇ مۇمكىن بولمايدۇ دەپ . ئەگەر شۇنداق بۇلۇپ قالسا ئىسمىمنى راستلا يۆتكىۋىتىمەن . ئويۇن قىززىدى ، شۇنداق قىلىپ توختاممۇ تۈزۈشتۇق . ماڭا بىرىلگەن ۋاقىت ئۈچ ئاي بولدى . ئۈچ ئاي ئىچىدە رەيھان ماڭا كۆيۈپ قالسا ئىسمىنى يۆتكىۋىتىدىغان . ئەگەر كۆيۈپ قالمىسا مەن ئۇتتۇرغان بۇلۇپ رەيھاننى يۈدۈپ مەكتەپنى بىر ئايلاندۇردىغان بۇلۇپ توختام تۈزۈشتۇق . قىززىق ، ئۆز-ئارا ئمزا قويۇپ بارمىقىمىزنىمۇ باستۇق . توختام پۈتۈنلەي رەيھاننىڭ ئويلاپ تاپقان نەرسىسى بۇلۇپ ئۇنىڭ ھازىر ئويلايدىغىنى پەقەت ماڭا ھاپاچ بۇلۇپ مىنىڭ قىززىق ئويىنىمنى بىر كۆرمەكچى ئىدى . شۇنداق كىلىشىپ ئۆز ياتاقلىرمىزغا قايتتۇق .  


         قىززىق شەمسىيە توغۇرلۇق ئازاپلار ، رۇشەنگۈلنىڭ خېتى توغۇرلۇق باش قېتىنچىلىقلار ئۇنتۇلدى . بىر ھىسابتا ماڭا قىلغۇدەك ئىش چىققان ئىدى . چۈنكى ئەزەلدىن دەرىسلەر مىنىڭ بېشىمنى ئاغرىتالمايتى . قانچە ئويناپ يۈرسەممۇ نەتىجەم يىللىقنىڭ ئالدىدا ماڭاتتى . توپ تۈرلىرى بولسا دوختۇرنىڭ «توپ ئوينىماڭ ، ئۆزىڭىزنى ئاسىرىمىسىڭىز بولمايدۇ» دىگەن گىپى بىلەن مىنىڭ تۇرمۇشۇمدىن خوشلاشقىنىغا بىر نەچچە يىل بولغان ئىدى . كارۋىتىمغا چىقىپ رەيھاننى قانداق ئۇتىۋېلىش تۇغرىسىدا باش قاتۇرىشقا باشلىدىم . ئەلۋەتتە ، رەيھان ئادەتتىكى قىز ئەمەس . ئۇنىڭ مۇستەقىللىق قارىشى كۈچلۈپ نىمە دېسە شۇنداق قىلىدىغان قىزلاردىن ئەمەس ، يەنە كىلىپ ئۇ مەن بىلەن خۇددى ئوغۇل بالىلاردەك يېقىن ئۆتكەن شۇنداقلا مەن بىلەن دادىل مۇنازىرلىشەلەيدىغان بىردىنبىر قىز ئىدى . قانداق قىلىش كىرەك ؟  توغرا، ھازىردىن باشلاش كىرەك . كاللام يىشۇنىڭ بېشىدەك يورىدى دە قولۇمغا تېلفۇننى ئالدىم .  رەيھان قىز بارمۇ سىز ؟ دىسەم دىمىسەم ياتاققا كىرىپلا مىنى سېغىنغىلى تۇردىڭىزغۇ دەيمەن ؟ بوپتۇ سېغىنىۋىرىڭ . مەن يىراق يەردە بولمىغاندىكىن ...    ئۇچۇرنى يوللاپ ئۇزۇن بولمايلا رەيھاننىڭ ئۇچۇرى كەلدى .     نىمە دەيسىز ؟ ئېلىشىپ قالدىڭىزما ؟ تۆۋە جىمۇ ...اللا خالىسا  ئوڭشۇلۇپ قالارسىز ھە ...  ئۇچۇرنى كۆرۇپ ئىچىۋاتقان سۈيۈمنىمۇ پۇرقۇپ سالدىم . ھەي شەيتان پەقەت گەپ بەرمەيسەندە ئادەمگە.                                                                  بۈگۈن قايتا قېيىشمىدىم ، ئۇنى ھازىرلا بىزار قىلىپ قويۇپ ئۇتتۇرۋىتىشىمدىن ئەنسىردى. ئەتە قانداق قىلىشنى ئويلاپ يېتىپ ئۇخلاپ قاپتىمەن .                                                                                                                                                                                                         
         يېڭى بىر كۈن . ئەتتىگەندىلا رەيھانغا تۈنۈگۈنكىگە ئوخشاپ كىتىدىغان مەزمۇندا ئۇچۇر يوللاپ جىلە قىلىپ ئۈلگۈردۇم . ئاشۇنداق مەزمۇندا ئۇچۇر يېزىۋاتقىنىمدا شەمسىيەنى يەنە ئەسكە ئالدىم . ھەيرانمەن ، شۇنچە ئۇزۇن بولغان بولسىمۇ يۈرۈگۈمنىڭ بىر يەرلىرى يەنىلا ئېچىشىپ تۇراتتى . لىكىن ياخشى بولغىنى مەن بىلەن ئويۇن ئوينايدىغانغا ئادەم چىقتى . بەلكىم مۇشۇ ئويۇننى دەپ ئالدىراش بۇلۇپ كەتسەم ھەممە نەرسىلەرنى ئۇنتۇپ قالارمەن . شۇنداق بولشىنى ئۆزۈم تۇلىمۇ ئۆمۈد قىلىمەن . بولدىلا بۈگۈنكى ھەركىتىمنى باشلاي .                                                                        رەيھانغا تېلفۇن قىلدىم .
--ئەسسالامۇ ئەلەيكۇم ، بارمۇ سىز ؟ ئۇخلاپ ياتمىغانسىز ؟ تاماق يىدىڭىزمۇ ؟ سورۇدۇم رەيھاندىن . تىلفۇندىن ئۇنىڭ ئۇيقاچان ئاۋازى كەلدى .                                                               
-- ۋاي نىمە بولدى بۇ سەھەردە ؟ چۈشىڭىز بۇزۇلۇپ قالمىغاندۇ ؟                                       
-- ياقەي ، پەسكە چۈشسىڭىز بوپتىكەن . سىزگە دەيدىغان گەپ بار ئىدى .                             
-- ساقلاپ تۇرۇڭ ھە . قىزلار ياتاق بىناسى ئالدىدا ساقلاپ تۇردۇم . رەيھان بىنادىن چۇشۋاتقىنىدا     ئۇنىڭ چىرايىغا دىققەت قىلدىم ، بۇرۇن بايقىمىغان بولسام كىرەك رەيھانمۇ راۋۇرۇس چىرايلىق قىز ئىكەن .
-- نىمە دەيسىز ؟ ئەتتىگەندە ئادەمنى بىزار قىلىپ ... رەيھان كېلىۋاتقاچ سورىدى .                                                  
-- سىزنى تېخى تاماق يىمىگەن بولسا دەپ تاماق ئالغاچ كىرگەن ئىدىم .                             
--ۋاي ۋۇي ئاپپاق خوجام ، بۇگۈن كۈن قاياقتىن چىققاندۇ ؟ رەيھان ھەيرانلىق ۋە قىزىقىشتا سورىدى . -- قاياقتىن چىقاتتى ، ئەلۋەتتە مىنى سۆيىدىغان قىز تەرەپتىن چىقىدۇ دە . شۇنداق دىگەچ ئۇنىڭ ياتىقىنى ئىما قىلىپ قويدۇم . قىزلار ياتاق بىناسى شەرىق تەرەپكە توغرا كېلەتتى .
-- ئۆتەپ بېرىڭە ، نىمە تاماق ئالغاچ كىردىڭىز ماڭا ؟                                                     
-- سىز ئامراق نەرسىلەر شۇ .
--قاچاندىن بېرى بۇنچە ياخشى بۇلۇپ كەتتىڭىز ؟ راستىنلا ھەيران قېلىۋاتىمەن جىمۇ .
--مەن ئەزەلدىن ياخشىغۇ ، بولدى تولا گەپ قىلماي قايتىپ تامىقىڭىزنى يەڭ . سوۋۇپ قالىدۇ .  تاماقنى  ئۇنىڭغا بىرىپ ياتاققا قايتىپ كەلدىم .                        
كۈنلەر شۇ تەرىققىدە ئۆتتى ، كۈندە رەيھانغا بىر نەچچە پارچە ئۇچۇر يوللاپ ئۇنىڭ مىنى ياخشى كۆردىغىنىنى ئەسكەرتىپ تۇراتتىم . ۋاقىت چىقىرىپ ئۇ ياختۇردىغان نەرسىلەرنى ئەكىرىپ بىرەتتىم . بۇرۇنقى يېقىن دوسلىقىمىز سەۋەبىدىنمۇ ئىشقىلىپ ئۇمۇ ئارتۇق گەپ قىلىپ يۈرمەي قۇبۇل قىلاتتى ، مىنىڭ چاقچاقلىرىمغا چاقچاق بىلەن جاۋاپ قايتۇراتتى .                                                        
      شەمسىيە مەكتەپ ئالمىشىپ باشقا ۋىلايەتكە كەتتى ، يەنى بىز ئايرىلغان كۈننىڭ ئەتىسىدىن باشلاپ تا بۈگۈنگىچە ئۇنى كۆرۈپ باقمىدىم  . لېكىن بىزنىڭ ئايرىلشىمىزغا سەۋەپ بولغان شەمسىيەنىڭ يېڭى يىگىتى يەنىلا مۇشۇ شەھەردە ئىدى . نىمىلا بولمىسۇن بۇلارنى كۆپ ئويلاشنى خالىمايتىم . ھەر قېتىم ئۇ ئىسىمگە كەلسىلا نۇرغۇن ئازابلىق كەچمىشلەر كۆز ئالدىمدا قايتا نامايەن بۇلۇپ مىنى بىئارام قىلاتتى . لىكىن نىمىلا دىگەن بىلەن مىنىڭ بۇ ئەسلىملەردىن قۇتۇلۇپ چىقىشىم مۇمكىن ئەمەس . ھەر نىمە دىسىمۇ شەمسىيەگە مەن چىن دىلىمدىن ئاشىق بۇلۇپ كۆيگەن ، ھىچچىكىمگە بېغىشلاپ باقمىغان مۇھەببىتىم بىلەن ئۇنى سۆيگەن ئىدىم . ھەقىقى سۆيگەن بىرىنى ئۇنتۇش ئۇنچە ئاسان ئەمەس دە .

ھەر قېتىم سىنىپقا كىرسەم رۇشەنگۈلنىڭ مەنەلىك قاراشلىرىدىن يەرگە كىرىپ كەتكۈدەك بۇلۇپ كىتەتتىم . تەلمۈرۈپ بېقىشلارغا مەندە جاۋاب قايتۇرۇشقا ئامال يوق ئىدى. مەن نىمىمۇ قىلالايتىم ؟ مەن رۇشەنگۈلنىڭ مۇھەببىتى ئالدىدا ھىچ نەرسە قىلىپ بىرەلمەيتىم . ئۆزۈم قىلچە قىزىقمىغان قىزغا ئادەتتىكى دوسلۇقتىن باشقا يەنە نىمە بىرەلەيمەن ؟ نىمە بىرىشىم كىرەك ؟ سۆيگۈمۇ ؟ ياق ، ياق . بۇ مۇمكىن ئەمەس ، مەندە ھازىر سۆيگۈ نە قىلسۇن ؟ دوستلىرىم بىلەن بىرگە بۇلۇش ، رەيھان بىلەن قېيشىپ تىرىكتۈرۈش ...مىنىڭ قىلىپ يۈرگۈنۈم پەقەت ئۆزۈمنى باشقىلارغا خۇشال كۆرسۈتۈش ئۈچۈن ئويناۋاتقان ئويۇن ئەمەسمۇ ...مەندە نەدىمۇ بىراۋغا بەرگۈدەك سۆيگۈ بولسۇن ... نىمىلا دىگەن بىلەن رۇشەنگۈلگە قارىسام ئىچىم سىيرىلاتتى ، ئۇمۇ گۇناھسىز ، پاك مۇھەببەتنىڭ ئىگىسى بولغان بىچارە ئاشىق ئەمەسمۇ . لىكىن مەن پەقەت ھىسداشلىقلا قىلالايتىم ، تەسەللى بىرىشكىمۇ پېتىنالمىدىم . ھازىر پەيتى ئەمەس دەپ قارايتىم . ئازىراقلا دىققەت قىلمىسام ئۇنىڭ كۆڭلىگە بەكلا قاتتىق ئازار بىرىپ قۇيشۇم مۇمكىن ئىدى . چۈنكى  ھازىر يېنىمدا شەمسىيە يوق ، بۇرۇن بولسا مىنىڭ يېنىمدا باشقا بىرى بولغانلىقى ئۈچۈن سەبىر قىلىپ كەلگەن رۇشەنگۈل ئەمدى يەنە ئۇنداق قىلالمايتى . خەت مەزمۇنىدىن ئۇنىڭ قەتئى نېيەتكە كەلگەنلىكىنى بىلۋېلىش تەس ئەمەس ئىدى . مۇھەببەت ، ئاشىقلىق شۇنداق نەرسە بولسا كىرەك ، ئۆزۈڭ ئاشىق بولغان كىشى ھەرقانچە سەت بولسىمۇ ساڭا گۈل كۆرنىدىكەن ، قىلىقلىرى شۇنچە ئەسكى بولسىمۇ سەن ئۇنىڭدىن قۇسۇر تاپالمايدىكەنسەن . پەقەت كىچىككىنە ئىمكانىيەت بولسىلا شەخسىيتىڭ ئۈچۈن ، ئېنىق قىلىپ ئېيتقاندا يۈرۈكۈڭدە يېنىۋاتقان ئوت ئۈچۈن ، بەخىت بۇلقىغا چۈمۈلۈپ ۋىسال شەربەتلىرىدىن قانغىچە ئىچىش ئۈچۈن ھەممە نەرسىنى بىر ياققا قايرىپ قويۇپ تۇرۇپ تىرشىدىكەنسەن . بۇ جەرياندا ھەرقانداق قىيىنچىلىق بولسىمۇ كۆزۈڭگە كۆرۈنمەيدىكەن ، سەن سۆيگەن كىشىنىڭ يېنىدا ئۇ ئاللبۇرۇن ئۆزىگە لايىق قىلىپ بىراۋنى تاللىغان بولسا ئامال يوق ، بەخىت تىلەش ئەڭ ياخشى تاللاش . لىكىن ئۇنىڭ يېنىدا باشقا بىرى بولمىسىلا ساڭا ئېمكانىيەت بار دىگەنلىك ، شۇندىلا سەن بار كۈچۈڭ بىلەن ئۇنىڭغا ئىرىشىش ئۈچۈن كۆرەش قىلىدىكەنسەن . بۇلار مىنىڭ بېشىمدىن ئۆتكەن . بىر ۋاقىتلاردا مەنمۇ شەمسىيەگە ساراڭلارچە ئاشىق بولغان ، ئوتىدا كۆيۈپ پىشقان . ھېچچىكىمگە بىر ئېغىز ئارتۇق گەپ قىلمايدىغان ، تالاي يىگىتلەر قوغلۇشۇپ ۋىسالىغان يىتەلمىگەن ، ئاددىيسى بۇرنىغا يىگەن ، قىز بالىلاردىن باشقىلارغا تەبەسسۇم قىلىپ باقمىغان شەمسىيەنى مەنمۇ شۇنداق قوغلۇشۇپ ئۇنىڭ سۆيگۈسىگە ئېرىشكەن . ئەينى ۋاقىتلاردا شەمسىيە ئالتۇنغا بەرمىگۈدەك دەرىجىدە ئىسىل قىز ئىدى . شەرمى ھايالىق ، پەزىلەتلىك ، ئىمانلىق ، غورۇرلۇق . ئىشقىلىپ ئۇنىڭدا مەن بۇلۇشىنى ئارزۇ قىلىدىغان بارلىق ئەخلاق پەزىلەتلەر تېپىلاتتى . ئىككى يىل ئارلىشىپ ئۇنىڭ قولىنى تۇتۇپ باققان ئەمەسمەن ، گەپ سۆزلەر قىڭغىر كەتسىمۇ مەندىن ئاغرىنىپ قېيداپ يۈرەتتى . شۇنداق بولغاچقىمۇ باشقا ئوغۇللارنىڭ ئۇنىڭغا يېقىنلىشىشى مۇمكىن ئەمەس ئىدى . بىراق بەزىدە ئادەم خاتا باھا بىرىپ قويدىغان چاغلارمۇ بولىدىكەن . ئۆزۈمدىنمۇ بەكرەك ئىشەنچ قىلغان شەمسىيە ئاخىرى باشقا بىرى بىلەن تېپىشىۋالدى . مەن بولسام بۇ ئەلەمنى ئىچىمگە سىغدۇرالماي قىسقىغىنە ھاياتىمدا تۇنجى بۇلۇپ شەمسىيەنىڭ تەستىكىگە ئىككىنى سالدىم دە” ئايرىلىپ كىتەيلى” دەپ كەينىمگىمۇ قارىماي قايتىپ كىردىم . مانا شۇنىڭدىن باشلاپ شەمسىيەنى كۆرۇپ باقمىدىم . مەكتەپ ئالمىشىپ كەتكەندىن بېرى يېڭىچە بىرەر خەۋەر ئاڭلىمىدىم . ئاڭلاشنىمۇ خالىمايتىم ، مىنى شۇنچە ئازابلىغان ، سۆيگۈمگە ئاسىيلىق قىلغان تۇرسا . ئۆچ بۇلۇپ كەتمىگەن تەقدىردىمۇ ئۇنىڭدىن راھىم بەكلا قايتقان ئىدى . ئۇنى ئويلاشنىمۇ خالىمايتىم .

                بۇ بىر ئاي ۋاقىت دەرىس ئاڭلاشتىن باشقا رەيھان بىلەن قېيشىش بىلەن ئۆتۈپ كەتتى . كۆرۈنۈشتە خۇددى رەيھاننى قوغلىشۋاتقان ئادەمدەك ئۇنىڭ يېنىدىن كەتمەيتىم . ئۇمۇ ماڭا ماسلىشىپ بىرەتتى . بۇ بىر ئايلىق ئۇيۇن قۇيۇش ئارقىلىق فاكولتىتتىكى كۆپىنچە ساۋاقداشلارغا ئىككىمىز يۇرىۋاتقاندەك تۇيغۇ بىرىپ ئۈلگۈردۇق . رەيھان شەمسىيەنىڭ دوستى بۇلۇشقا لايىق قىز ئىدى . شەمسىيەدىكى كۆپ قىسىم پەزىلەتلەر رەيھاندىمۇ تېپىلاتتى . پەقەت ئوخشىمايدىغان يىرى رەيھان شوخ مىجەزلىك بۇلۇپ ئالدىراپ بىرەرىگە گەپ بەرمەيتى . ئازراق ئېشىپ كەتكۈدەك گەپ قىلىپ قالسام -- ناماز ئوقۇدىڭىزمۇ ؟ دەپ قۇيۇپلا ئارتۇق گەپ قىلماي ئۆز ئىشى بىلەن بۇلاتتى . ئەلۋەتتە مۇشۇ بىر ئېغىز سۆزىلا ئۇنىڭ ھەرقانداق سۆزىدىن ئېشىپ چۈشەتتى . شۇڭا ئامالسىز ئۆزۈمنى سورايتىم . شەيتاننىڭ كەينىگە كىرىپ باشلىغان بۇ ئويۇن بىر ئايدىن بېرى شۇنداق داۋاملىشىپ كىتىۋاتاتتى . ۋاز كېچەي دىسەم رەيھانغا ئۇتتۇرۇپ قويغۇم يوق ئىدى . شۇڭا داۋاملىق ئوينۇشۇم ، توختۇتۇپ قويماسلىقىم كىرەك .
                  روشەنگۈلنىڭ خېتىنى بىر نەچچە قېتىم قايتىلاپ كۆردۇم . باشقا مەخسەتتە ئەمەس ، ئۇنىڭ بۇنىداق خەت يازالايدىغانلىقىغا ، شۇنچىلىك يۈرەكلىك ئىكەنلىكىگە ھەيران بۇلاتتىم . بەلكىم ئاشىقلىق دىگەن ئاشۇنداق ھەممە نەرسىنى قايرىپ قويۇپ تۇرۇپ بەدەل تۆلەشكە مەجبۇر قىلسا كېرەك . بۇرۇنقى يۇۋاشلىقى ، كەم سۆزلىكى ، تارتىنچاقلىقى قاتارلىق ئىشقىلىپ مىنىڭ كاللامدا بۇ خەتنى رۇشەنگۈل ئۆز قولى بىلەن يازدى دىسە ئىشەنمىگۈدەك دەرجىدە تەسىر قالدۇرغان بىر قىز ئىدى . ھالا بۈگۈن شۇنچىلىك قەتئى نىيەتتە مىنىڭ لايىقىم بۇلۇشنى ئوتتۇرغا قويۇۋاتىدۇ . قىززىق ، بۇ بىر ئاي مەزگىلدە رەيھان ئىككىمىزنىڭ قىلىقلىرىدىن رۇشەنگۈلنىڭ ئەرۋايى ئۇچتى . بەلكىم كۆڭلىدە بىردىن ئايرىلىپ ئەمدىلەتىن ماڭا پۇرسەت كەلگەندە يەنە بىرسى چىقتى دەپ ئاچچىقى كىلىۋاتقاندۇ . نىمىلا دىگەن بىلەن رەيھان ئىككىمىزنىڭ مۇناسىۋىتى رۇشەنگۈل بىلەن مىنىڭ مۇناسىۋىتىمدىن ياخشى ئىدى.  
                    كەچلىك مۇزاكىرە ، ئارقا پارتا مىنىڭ ۋە مىنىڭ دوستلىرىمنىڭ دائىملىق ئورنى بۇلۇپ ، مەيلى قانچىلىك كېچىكمەيلى ھىچچىكىم بۇ يەردە ئولتۇرۋالمايتى . بىز يەنىلا شۇ ئورنىمىزغا كىلىپ ئولتۇردۇق . كىتابلىرىمنى قولۇمغا ئالاي دەپ تۇرۇشۇم يېنىمىزغا روشەنگۈل ئۈندى .
-- ئەسسالامۇ - ئەلەيكۇم . يۈزلىرى نۇمۇستىن ئوت ئالدى . بىر ئايدىن بېرى ئۇ ماڭا يۇز تۇرانە گەپ قلىپ باقمىغان ئىدى .
--ۋە ئەلەيكۇم ئەسسالام . قانداق ئەھۋالىڭىز ؟ بۇرۇنقىدەكلا مۇئامىلە قىلىۋاتاتتىم .
-- شۇكىرى... ئانچە ياخشى ئەمەس . روشەنگۈل تېخىمۇ قىزاردى . بۇنىڭ نىمە ئۈچۈنلىكىنى مەن تۇرماق يېنىمدىكىلەرنىڭ ھەممىسى بىلەتتى . ئۇنىڭ مەن بىلەن پاراڭلاشقۇسى بارلىقىنى سەزدى بولغاي دوستلىرىم بىر بىر لەپ تېلفۇننى باھانە قىلىپ يىراق كىتىشتى . مانا ھازىر ئارقا تەرەپتە پەقەت ئىككىمىزلا قالدۇق . ئالدىدىكىلەر بولسا بەزىلىرى تېلفۇن ئويناپ بەزىلىرى كېنو كۆرۈپ ، ئاز بىر قىسىملىرى كىتاپ كۆرىۋاتاتتى . ئومۇمەن بىزگە قارايدىغان ئادەم يوق ئىدى .
-- خەتنى كۆردىڭىزمۇ ؟ روشەنگۈل يەنە بىر قېتىم ئاناردەك قىزاردى .
-- ھەئە ، كۆردۇم . نىمە دىيىشىمنى بىلەلمەي توختاپ قالدىم .
-- كەچۈرۈڭ...... روشەنگۈل بېشىنى تۆۋەن قىلىۋالدى . ئۇنىڭ قول ياغلىق بىلەن كۆز يېشىنى سۈرتىۋاتقىنىنى بىلىپ قالدىم .
-- نىمە ، يىغلاۋاتامسىز ؟ نىمە بولدى سىزگە ؟ چوپ -چوڭ بىر قىز يىغلاپ يۈرسە يارىشامدۇ دەيمە . بولدى ، بولدى . يىغلىماڭ . كىچىك بالىلار بىزدىن ھۈنەر تالاشتىڭ دەپ ئۇرۇشۇپ يۈرمىسۇن يەنە . بولدى ھە . تەسەللى بەرگەچ يۈرۈگۈم يەنە بىر قېتىم ئېچىشتى . ئۇنىڭدا نىمە گۇناھ ...
-- شۇ...شۇ سىزنى بىئارام قىلىپ قويدۇممىكىن دەپ ... روشەنگۈلنىڭ كۆزلىردىن بەڭۋاش ياشلار يەنە سىرغىشقا باشلىدى . قوللىرىم بىلەن ئۇنىڭ يېشىنى سۈرتتۈم ، بۇ دەقىق مىنىڭ قەلبىمدىكى ئىسان دەل ئۆزۈمگە قاشى قوزغالغان ئىدى . توۋا ، ئادەم ئاجايىپ بولىدىكەن دىسە . ساڭا ۋاپاسىزلىق قىلغان بىرىنى ئۇنتىيالمايسەن ، ئەمما ئالدىڭدا تۇرغان ، ئۈچ يىلدىن بېرى سىنىڭ ئوتۇڭدا پۇچۇلنىپ كىلىۋاتقان بىرىگە قىلچە ھىسسىيات يوق ...
-- ئاخماق قىز . ئۇنداق ئەمەسقۇ . مەن تېخى سىزگە رەھمەت ئېيتسام بۇلىدۇ . قانچە ۋاقىتتىن بېرى ئۈن تىنسىز مىنى قوللاپ ، ياردەم بىرىپ كەپسىز ...يەنە نۇرغۇن ئۇنىڭغا تەسەللى بۇلغۇدەك سۆزلەرنى دىدىم . روشەنگۈل جىمجىت ئولتۇرۇپ ئاڭلىدى . “”ئۇنداقتا سىزنىڭ جاۋابىڭىزچۇ ؟ “دىگەن سۇئالىغا بولسا ، مىنىڭ تۇرمۇشۇم تېخى تولۇق ئەسلىگە كىلىپ بۇلالمىدى ، ھازىرچە جىمجىت يۈرۈپ ئۆزۈمنى ئوڭشىۋالاي . دەپ جاۋاب بەردىم . بۇ جاۋابىمدىن مىنى رەيھان بىلەن يۈرۈش ئىھتىماللىقى يوقكەن دەپ چۇشەندىمۇ ، ياكى ئۆزىنى ئوڭشىۋېلىپ مەن بىلەن يۈرىدىكەن دەپ چۇشەندىمۇ بىلمىدىم . ئىشقىلىپ ئاخىردا كۈلۈمسىرەپ تۇرۇپ “”ئۇنۇتماڭ ، مەن داۋاملىق كۈتىمەن. “ دەپ قويۇپ يۈرۈپ كەتتى . ئۇنىڭ خۇشال قايىتقىنىدىن كۆڭلۈم ئازراق بولسىمۇ ئەمىن تاپتى . لىكىن ئۇنى خاتا تۇيغۇغا سېلىپ قويدۇممۇ قانداق ، ئۆمۈدلەندۈرۈپ قويغاندىمەنمۇ ؟ نىمىلا بولمىسۇن بۈگۈنكى ئۆتكەلدىم ئۆتۈپ كەتتىم ، قالغىنىنى كىيىن كۆرەرمەن .


**************************************************************************************************

             رەيھان بىلەن بولغاب ئويۇن داۋاملاشتى . پۇرسەت بولسىلا ئۇنىڭ بىلەن قېيشىپ ، تىرىكتۈرۈپ دېگەندەك يۈردۇم . رەيھانمۇ ماڭا ماسلىشىپ كېتىۋاتاتتى . بىراق ھەر قېتىم تەسلىم بولسىڭىز سىزنى كەچۈرۈم قىلىمەن ، ئىسمىڭىزنى يۆتكەيمەن دەپ ئاۋارە بولمايسىز دىسەملا سەكرەپ كىتەتتى . يەنە شۇ بۇرۇنقى گەپلەر ، ۋاي ئۆتەپ تېشىپ سىزگىمۇ يەي ، ئادەم تېپىلماي قاپتۇمۇ يەي ، ئاسماندىكى غازنىڭ شورپىسىغا نان چىلاپ ئىچمەڭ يەي . ئىشقىلىپ ئاغزىدىن سىلىقراق گەپ چىقمايدۇ .بىراق كۆزلىرنىڭ بىر يەرلىرى ماڭا ئۇنىڭ يۈرەك سۆزلىرىنى ئاشكارلاپ قويۇۋاتقاندەك بىلىنەتتى . ماڭا قاراشلىرى ، چىرايىنىڭ قىزىرىپ كىتىشلىرىنى نۇرمال ئەھۋال دەپ قارىمايتىم . دېمەك مەن غەلبە قىلغىلى ئاز قالغان ئىدىم .


****************************************************************************************************

       چۈشلۈك تاماققا چىقىشقا خوش ياقماي ياتاقتا توردىكى يېڭى ئەسەرلەرنى كۆرۈشكە باشلىدىم . ھەيرانمەن ، جەمىيىتىمىزدە كۆمۈلۈپ قالغان شۇنچە كۆپ ياشلىرمىز باركەن . تولۇق ئوتتۇردىكى چاغلىرىمدا پەقەت ئاز بىر قىسىم كىشىلەرلا كىشىنى ئويغا سالالايدىغان ، تەسىرلەندۈرەلەيدىغان ئەسەرلەرنى يازالايدىغۇ دەپ قارايتىم ، لېكىن ھازىر بۇ قارىشىم تۈپتىن ئۆزگەردى ، ئەمەلىيەتتە بۇرۇنلاردا پەقەپ ئىچىدىكىنى چىقىرىشقا سەھنە كاملىق قىلىپتىكەن .
--مەردان ، مەردان ، تىز بول رەيھان ئۇ ...   سىرىتتىن ئوقتەك يۈگۈرۈپ كىرگەن ئادىل ئۆڭ سۆلى ئۆچكەن ھالدا ئۇدۇل ماڭا سوقۇلۇپ توختىدى ،
-- رەيھانغا نىمە بولدى ، ئالدىرماي دىمەمسەن ، ئاسمان ئۆرۈلۈپ چۈشمىگەندۇ ؟
-- چاقچىقىڭنى قوي ، رەيھان ئايلىنىپ چۈشۈپتۇ . بېشىمغا بىرسى قاتتىق ئۇرغاندەق بۇلۇپ كۆزلىرىم قاراڭغۇلاشتى . ئادىل قولۇمدىن سۆرەپ ماڭماقتا ، بىر ئازدىن كىيىن ئەسلىمگە كىلىپ ئادىلدىنمۇ تىز يۈگۈرۈپ مېڭىپ كىتىپتىمەن . ئويلاپ باقسام ئادىل ماڭا قەيەردە ئايلىنىپ كەتكىننى دېمەپتۇ .   
-- تىزراق بولساڭچۇ . ۋارقىردىم كەينىمدىن كېلىۋاتقان ئادىلغا قاراپ . قەيەردە ئايلىنىپ قاپتۇ ؟ ئۆزىچۇ ؟ ھازىر نەدە ؟
-- ئاڭلىسام مەكتەپ ئاشخانسىدا ئايلىنىپ كىتىپتۇ ، مەن سىنى خەۋەرلەندۇرەي دەپ تېلفۇن قىلسام ئالمىدىڭ ، شۇڭا يۈگۈرەپ چىققان ئىدىم . ھازىر قەيەردىكىن بىلمەيمەن .
مەكتەپ ئاشخانسى تەرەپكە يۈگۈردۇق ، بىنا ئالدىغا كىلىشىمىزگە بىر نەچچە قىزنىڭ رەيھاننى تەستە كۆتۈرۈپ چىقىۋاتقىنىنى كۆردۇق .
-- قويۇپ بىرىڭلار ، ئۆزۈم كۆتۈرىمەن . سىلەرگىمۇ رەھمەت . جىددىيلىشىپ شۇنچىلىك گەپنى ئاران دىدىم دە رەيھانغا ئېسىلدىم . فاكۇلتىتتىكى ئوقۇغۇچىلارنىڭ كۆپىنچىسى بىزنىڭ بىللە يۈرۈيدىغىنىمىزنى بىلەتتى ، شۇڭا ھېچچىكىم ئارتۇق ھېچنىمە دىمىدى . رەيھاننى كۆتۈرۈپ كىتىۋاتىمەن . ئادەم ھوشسىز بولسا بۇنچىلىك ئېغىرلىشىپ كىتىدىغانلىقىنى بىلمەيدىكەنمەن . مۇشۇ پەيىتتە قولۇمدىكى رەيھان بولماي باشقا نەرسە بولسىدى ئاللا بۇرۇن تاشلىۋەتكەن بۇلاتتىم . ئەتراپىمدا ئادىل بىر نەرسىلەرنى دەپ مەن بىلەن تەڭ كىتىۋاتىدۇ ، ئۇنىڭ زادى نىمە دەۋاتقىنى قۇلۇقۇمغا كىرمەيدۇ . كاللامدا رەيھانغا بىر ئىش بۇلۇپ قالمىسىدى ، زادى نىمىشقا ئايلىنىپ كىتىدۇ؟  دىگەندەك پەقەت رەيھانلا باش پىرسۇناژ بولغان خىياللار ئاللا بۇرۇن پۇتۇن كاللامنى ئەسىر قىلىپ بولغان ئىدى . تۇيۇقسىز ئادىل مىنى سىلكىپ توختاتتى ، قولۇمدا رەيھان پۇلاڭلاپ تاسلا قالدى چۈشۈپ كەتكىلى .
-- مەردان ، دوختۇرخانا خېلى يىراق ، تاكسى توسۇپ ئېلىپ بارايلى  دەۋاتىمەن ، ئاڭلاۋاتامسەن ؟
ئادىلنىڭ سۆزىدىن كېيىنلا تاكسى توسمىغىنىىمنى ، ھەم بۇنداق كۆتۈرۈپ بېرىپ بولالمايدىغىنىم ئىسىمگە كەلدى .
-- ماقۇل ئاداش ، سەن توسىغىن ، مەن ئالدىرماي ماڭغاچ تۇراي.  شۇنداق دىدىم دە يەنە ماڭدىم . تاكسى توسىغىلى بولمىسا ساقلاپ تۇرغىلى بولمايدۇ دە . قانچىلىك ۋقىت ئۆتتى بىلمىدىم . ئادىل تاكسىدىن چۈشۈپ يۈگۈرۈپ كىلىپ ماڭا ياردەملىشىپ رەيھاننى ماشىنىغا سالدۇق . مەن رەيھاننى قۇچۇقۇمغا ياتقۇزۇپ ئارقا تەرەپتە ئولتۇردۇم . رەيھانغا سەپ سالدىم ، ھەقىقەتەن تىپىك ئۇيغۇر قىز دە ، قاپقارا قاشلار ، ئاق ھەم سۈزۈك يۈز ، كىچىك ئېغىز . تەبى بۈدىرە كەلگەن چاچ .
-- رەيھان ، رەيھان ، گېپىمنى ئاڭلاتامسىز ؟ بىرەر يىرىڭىز ئاغرىۋاتامدۇ ؟  
ھوشىدىن كەتكىنىنى بىلسەممۇ گەپ قىلىپ بېقىۋاتىمەن ، بىر خىل بىئاراملىق قەلبىمنى چۇلغىۋالغان ئىدى  . رەيھاننىڭ ھازىرلا كۆزىنى ئېچىپ ماڭا بىرەر ئېغىز گەپ قايتۇرشىنى تولىمۇ ئارزۇ قىلاتتىم ، رەيھان گەپ قىلمىدى . ئەمما كۆزىدىن ئىككى تامچە ياش ئېقىپ چۈشتى . ھوشسىزلانغان ئادەمنىڭ گەپ ئاڭلىيالايدىغان ياكى ئاڭلىيالمايدىغىنىنى بىلمەيمەن ، ئەمما رەيھاننىڭ كۆزىدىن ياش ئاققىنى ئېنىق ، يەنە كىلىپ يۈزىنى تۇتۇپ ئولتۇرغان قولۇم ئۈستىگە ئېقىپ چۈشتى .
دوختۇرخانىغىمۇ يىتىپ كەلدۇق ، يەنە شۇ جىددىچىلىك بىلەن رەيھاننى كۆتۈرۈپ جىددى قۇتقۇزۇش بۆلىمىگە ئەكىردىم ، دوختۇرلار مىنىڭ چىقىپ كىتىشىمنى تەلەپ قىلدى ، ئامالسىزمەن . ئىشىك تۈۋىدە ساقلىدۇق .
-- نىمانچە تەرلەپ كەتتىڭ ؟ تومۇز ئىسسىق بولمىسا يە ؟ ئادىل چىلىق چىلىق ھۆل بۇلۇپ كەتكەن كۆينىكىمنى كۆرسەتتى . قاراپ باقسام راستىنلا سۇغا كىرىپ چىققاندەك بۇلۇپ كىتىپتىمەن . شۇندىلا رەيھاننى خېلى ئۇزۇن كۆتۈرۈپ مېڭىپتىمەن دەپ ئويلۇدۇم . ئالاھىزەل يېرىم سەئەتتىن كىيىن دوختۇر چىقتى ، ئۇنىڭ ئەھۋالى قانداق دەپ ئالدىغا يۈگۈردۈم.
--باشقا چوڭ ئىشمۇ يوق ، بۆرىكىگە تاش چۈشۈپتۈ ، ئاغرىق دەستىدىن ئايلىنىپ كەتكەن بولىشى مۇمكىن ، داۋالاشقا توغرا كىلىدۇ . قايسىڭلار بىمارنىڭ ئېگىسى ؟ مەن بىلەن بېرىپ پۇل تاپشۇرۇڭلار .
ئۇھ ، شۇكىرى ، ھەرنىمە بولسا باشقا كىسەل ئەمەسكەن . بۆرەككە تاش چۈشۈش دىگەن ھازىر چوڭ كىسەلمۇ ئەمەس ، مەنمۇ داۋالىنىپ باققان ئىدىم ، شۇڭا ئانچە بەك ئەنسىرەپ كەتمىدىم . دوختۇرغا ئەگىشىپ بېرىپ رەسمىيەتلەرنى بىجىرىپ رەيھاننى ياتاققا ئورۇنلاشتۇردۇق . كەينىمىزدىن يىتىپ كەلگەن بىرنەچچە قىزلارنى  “دوختۇر كۆپ ئادەم يېنىدا تۇرمىسۇن دىگەن ئىدى ، ھوشىغا كەلسە يوقلاپ كىلەرسىلە “دەپ يولغا سالدۇق ، رەيھاننىڭ دوستى بولسا “سىز نىمىنى بىلەتتىڭىز قىز بالا دىگەنگە قىز بالا قارىسا ياخشى “ دەپ قايتقىلى ئۇنىمىدى ، بىزمۇ قايتا زورلىمىدۇق . رەيھان ھوشىغا كەلگەندىن كېيىن ئادىلنىمۇ قايتۇرۋەتتىم . كەچلىك تاماقنى ئەكىرگىنىمدە رەيھان خېلىلا ياخشى بۇلۇپ قالغان بۇلۇپ ئىشىكتىن كىرىشىمگىلا
-- سىزگە يەتكۇچە ئېغىرىمنى ساپتىمەن ھە ، كەچۈرۈڭ .  رەيھاننىڭ سۆزى ئەمەس سۆزدىن كىيىنكى تەبەسسۇمى بەكىرەك تەسىر قىلدى .
-- نىمە دەپ يۈرۈيسىز . قىلىشقا تىگىشلىك ئىش . يات ئادەم بولمىساق يە .  ياخشىلنىپ قالغىنىنى كۆرۈپلا گەپتە چېقىۋالدىم . ئادەتتىكىگە ئوخشىمىغىنى رەيھان ھېچقانداق قارشى گەپ قىلمىدى ، تاتارغان يۈزلىرگە يۈگۈرگەن قىزىللىق ئادەتتىكىدىن بەكىرەك رۇشەن ئىپادىلنىپ چىقتى . خېجىل قىلىپ قويغىنىمنى بىلدىم دە ، تېمىنى بۇرۇدۇم .
-- قانداق ، يەنە ئاغرىۋاتامدۇ ؟ تاماق يەۋېلىڭ ، سوۋۇپ قالمىسۇن .
-- ھازىر ئاغرىمىدى . دوختۇر گەپ قىلمايدىغان بولسا مەكتەپكە قايتىپ كىتەتتىم ...
-- جاندىن تويغان ئوخشىمامسىز ؟ كىسەل بۇلۇپ مۇشۇ دەرجىگە يەتكۈچە بىرەر ئېغىز تىنىپ قويماپسىز ، يەنە تېخى كىتەتتىم دەيسىزغۇ ؟ باشقىلارنى ئويلاپ قويمىسىڭىزمۇ ئاتا - ئانىڭىزنى ئويلىشىڭىز كېرەققۇ .  يەنە بىر ئاز سۆزلىگەنمۇ بولار ئىدىم ئىشىكتىن رىزىۋانگۈل كىرىپ قالدى . -- سىنىپتا ئۇرۇشۇپ يۈرگىنىڭلار يەتمەي يەنە ئۇرشىۋاتامسىلە ر؟ مەردان ، سىزمۇ قاملاشماي قالدىڭىز ، كىسەل ئادەمگىمۇ ئاراملىق يوقمۇ ؟
-- بىراق رەيھان دوختۇرخانىدىن چىقىپ كېتىمەن دەيدۇ ... تېخىمۇ ئاچچىقىم كەلدى .
-- بولدى ، بولدى . تالاشماڭلار ھە . رەيھانمۇ چىقىپ كەتمەيدۇ ، سىزمۇ ئۆزىڭىزنى بېسىۋېلىڭ ھە . قېنى رەيھان تاماق يەۋال ئاداش . سىز چىقىپ ئازراق قايناقسۇ تېپڭپ كىرىڭ ، رەيھان ئىچىپ بەرسە پايدىلىق .  رىزۋاگۈل رەيھانغا تاماق يىگۈزۈپ قالدى ، مەن چىقىپ كەتتىم .  

         كتىۋاتىمەن ، خېياللىرىم يەنلا رەيھان ئۈستىدە ئات چاپتۇرماقتا ، نىمىشقا بۇنچە قىلىپ كەتتىم ، نىمىشقا شۇنچە ئەنسىردىم ، نىمىشقا ... سۇئاللار كاللامدا ئارقا ئارقىدىن تىزىلماقتا ، جاۋابى نىمە ؟ ئۇنى راستىنلا ياخشى كۆرۈپ قالدىممۇ ؟ شۇنداق بولىشى مۇمكىنمۇ ؟ شەمسىيەدىن ئايرىلغىنىمغا تېخى بىرەر يىل بولمىدىغۇ ، ئەجىبا تېخى بىر ئاي ئالدىدىلا ، ياق تېخى تۈنۈگۈنلا شەمسىيە ئىسىمگە كەلسە يۈرۈگۈم ئېچىشقان ئىدىغۇ ... بىر يۈرەككە ئىككى ئادەم سىغامدىغاندۇ ... ياق ، ياق ئۇنداق ئەمەس . ئويۇن ئوينايمەن دەپ ئۆزۈمگە ئىش تېپۋالسام بولمايدۇ دە ئاخىر . يەنە كېلىپ ئۇ شەمسىيەنىڭ جان جىگەر دوستى تۇرسا قانداقمۇ مۇمكىن بولسۇن .
ئۆزۇمنىڭ قاملاشمىغان خېياللىرىمدىن ئۇيالدىم . كىسەلخانىغا قايتىپ قايناقسۇ نى قويۇپ قويۇپلا خوشلۇشۇپ قايتىپ چىقتىم . ئىككى قىز بالا بار يەردە ماڭا نىمە بار . يەنە كېلىپ رەيھانغا قارىساملا يۈزلۈرۈم كۆيۈشكەندەك بۇلۇپ كەتتى . مەكتەپكە قايتىپ كېتىشكە كۆڭلۇم ئۇنىمىدى . دوختۇرخانا ئەتراپىدىكى مىھمانخانىدىن ياتاق ئېلىپ كىرىپ كەتتىم . ئۇخلاش ئالدىدا رىزۋانگۈلگە تېلفۇن قىلىپ قەيەردىلكىمنى بىرەر ئىش بولسا تېلفۇن قىلسىلا چىقىدىغانلىقىمنى ئېيتىپ قويدۇم . جىددىيلەشكەن نىرۋىللىرىمنى ئارام تاپقۇزاي دەپ يۇيۇندۇم . ھېچ پايدىسى بولىدىغاندەك ئەمەس . سائات ئىستىرىلكىسى سەھەر سائەت  بىرنى كۆرسەتكەن چاغدىمۇ يەنە شۇ ئىشلارنى ئويلاۋاتاتتىم . قاچانلاردا ئۇخلاپ قالغىنىمنى بىلمەيمەن . تۇيۇقسىز سايرىغان تېلفۇن ئويغىتىۋەتتى .
-- ئەسسالامۇ -ئەلەيكۇم . مەردان  ، ئورنىڭىز دىن تۇرغانسىز  ؟ رەيھاننىڭ ئاۋازى ئىدى ، يەنىلا بۇرۇنقىدەك شوخلۇق بىلەن سۆزلەۋاتاتتى .
-- ۋە ئەلەيكۇم - ئەسسالام . ھەئە تۇرغان . تېخىچە ئۇخلىدىم دىيىشكە يۈزۈم چىدىمىدى . بامداتنىمۇ يىگەن ئىدىم . قانداق ئازراق ياخشىلاندىڭىزمۇ ؟ يەنە ئاغرىۋاتامدۇ ؟
-- ھەئە ، كۆپ ياخشى ، ئاغرىمىدى ، يوقاپلا كەتتىڭىزغۇ؟ بىز ئاچ قالدۇق . كىچىك باللارچە ئەركىلەپ ئېيتقان سۆز باشقىچە تەسىر قىلدى بولغاي كۈلۈپ كەتتىم .
ئادەم دىگەن ئاجايىپ ھە ،بەزى ۋاقىتلاردا  كىچىككىنە ئىشلارمۇ ، بەزى كىشىلەرنىڭ بىرەر ئېغىز سۆزىمۇ كىشىنى شۇنچىلىك خۇشال قىلالايدۇ . بەزى ۋاقىتلاردا بولسا تەڭرى تېغىدەك ئالتۇن بەرسىمۇ خۇشال بولغىلى بولمايدۇ . بەلكىم بۇمۇ بىز ئىنسانلارنىڭ ياراتقۇچىدىن ئىرىشكەن ئەڭ چوڭ بايلىقىمىزنىڭ بىرى بولسا كېرەك .
-- بىردەم چىداپ تۇرۇڭلار ، مانا ھازىرلا تاماق ئېلىپ كىرمەن . اللا غا ئامانەت دىيىشنىمۇ ئۇنتۇپ تېلفۇننى قويدۇم دە “يۈز مىتىرلىق سۈرئەت “ بىلەن كىيىندىم . يۈگۈرۈپ چىقىپ ئۆي تائاملىرى بار بىر تاماقخانىدىن ئىككى كىشلىك ئۈگىرە ئالدىم دە كىسەلخانىغا ئۇچتۇم .
مەن كىرگەن ۋاقىتتا رىزۋالگۈل قايتىپ كېتىپتىكەن ياتاقتا پەقەت رەيھانلا كىتاب كۆرۈپ ياتاتتى .
-- ئەسسالامۇ - ئەلەيكۇم .
-- ۋە ئەلەيكۇم- ئەسسالام . ئاددىغىنە سالامدىن كىيىن رىزۋانگۈلنىڭ ئازراق ئىشى چىقېپ قېلىپ قايتىپ كەتكىنىنى ئۇختۇم . شۇنداق قىلىپ بىر - بىرمىزگە زورلاپ يۈرۈپ ئۈگىرىنى ئىچىپ بولدۇق .
-- قانچە ئۇزۇن بولدى ؟ گەپ قىلماپتىكەنسىزغۇ ؟ سوردۇم رەيھاندىن .
-- خېلى بولغان ، مەن سىزدىن ياخشى خەۋەر ئالىمەن دەپ باشقىلارغا ۋەدە بىرىپ قويۇپ ئۆزۈم كىسەل بۇلۇپ يېتىۋالسام قاملاشمايدۇ دە . رەيھاننىڭ دەرىزە سىرتىغا قاراپ تۇرۇپ قىلغان سۆزىدىن ھەيران قالدىم .
-- نىمە دىدىڭىز ؟ مىنىڭ ھالىمدىن خەۋەر ئالىسىز ؟ بىرسىگە ۋەدە بەرگەن ؟  ئۇنىڭ گەپلىرنى چۈشەنمەيلا قالغان ئىدىم .
-- ياق ياق ، نىمىدى ، شۇ ، شۇ ھەممە يەردە بىر يىرىم ئاغرىيتى دەپ يۈرمەيمەن دە . شۇنداق ئەمەسمۇ . رەيھان تىزلىك بىلەن گەپنى بۇرىۋەتتى .
-- سۇئالىمغا جاېاب بەرمىدىڭىز ، ئۇ كىمۇ ؟ يەنە سوراپ تۇرۋالدىم .
-- ھېچكىم ئەمەس . ئۆزۈم شۇ .  كۈلدى رەيھان . سىزنى ياخشى كۆرىمەن ئەمەسمۇ ، شۇڭا سىزدىن ياخشى خەۋەر ئېلىشىم كېرەك دە . ھەييارلىق بىلەن قىسىپ قويغان كۆزىدىن يەنە بىر قېتىم كۈلۈپ كەتتىم .
-- بوپتۇ ، سورىمىدىم . سىز ئارام ئالغاچ تۇرۇڭ . مەن چىقىپ دوختۇرنى ئىزدەپ باقاي .
قەلبىمدە بىر تۈگۈن پەيدا بولدى . رەيھان ئۇ كىمگە مەن دىن ياخشى خەۋەر ئېلىش توغۇرلۇق ۋەدە بىرىدۇ ؟ رۇشەنگۈلگىمۇ ؟ ياق ، مۇمكىن ئەمەس ، رەيھان رۇشەنگۈلنى ئانچە ياقتۇرۇپ كەتمەيدۇ . مىنىڭ ئاتا - ئانامغىمۇ ؟ ياق ، ياق . ئانام بىزنىڭ ئادەتتىكى دوستلىقىمىزنى ، مىنىڭ ئۇنىڭ بىلەن يۈرۈپ قالمايدىغانلىقىمنى بىلىدۇ . يەنە كىلىپ رەيھاننىڭ ئائىلىسى بىلەن بىزنىڭ ئائىلدىكى چوڭلار ئانچە كىلىشەلمەيدۇ . دېمەك بۇ پەرەزنىمۇ نەزەردىن ساقىت قىلىشقا بۇلىدۇ دېگەن گەپ . ئۇنداقتا زادى كىم ؟ شەمسىيە ؟ ياق ياق ، بۇ پەرەز ئالدىنقى ئىككى خىل پەرەزدىنمۇ ئەخمىقانىلىق . مۇمكىنچىلىكى نۆل . مەندىن كۆڭلى سوۋۇپ باشقا بىرىنى تاپقان ئادەم يەنە ماڭا كۆڭۈل بۆلۈپ ساراڭ ئەمەس دە . يەنە كىلىپ رەيھانمۇ شەمسىيەنىڭ ۋاپاسىزلىقىنى بىلىدۇ ، گەرچە ئوچۇق دەپ باقمىغان بولسىمۇ مىنىڭ يېنىمدا مىنى خۇشال قىلىمەن دەپ قىلغان ئىشلىرى ، گەپ سۆزلىرى ، چاقچاقلىرى ئۇنىڭ ماڭا ھېسداشلىق قىلىدىغىنىنى ،بىلدۈردۇ ئەلۋەتتە . شۇڭا رەيھانمۇ شەمسىيەدىن ئاغرىپ قالغان بۇلىشى مۇمكىن . بۇ بىر يىلغا يېقىن ۋاقىت ئىچىدە رەيھاننىڭ مىنىڭ ئالدىمدا شەمسىيەنىڭ گېپىنى قىلغىنىنى بىلمەيمەن . شۇڭا بۇمۇ مۇمكىن ئەمەس . بولدىلا ، كۆپ ئويلىماي ، ۋاقىتى كەلسە مەن سورىمىساممۇ ئېيتىپ بىرەر .

          ئۇنىڭدىن كېيىنكى بىر ھەپتە ئىچىدە ئالاھىدە ئىش يۈز بەرمىدى . مىنىڭ كۈنلۈرۈم مەكتەپ بىلەن دوختۇرخانا ئوتتۇرسىدا قاتىراش بىلەن ئۆتمەكتە . كەچلىكى يەنەلا شۇ مىھمانخانىدا يېتىۋاتاتتىم ، مەكتەپكە قايتىپ كەتسەم خاتىرجەم بۇلالمايتىم . رەيھان كۈندىن كۈنگە ياخشىلنىۋاتاتتى . لېكىن بۇگۈنكى بىر ئىش مىنى ئەنسىرتىپ قويدى . كىمدىن كەلدىكىن رەيھان بىر تېلفۇننى ئېلىپلا ئۆكسۈپ يىغلاپ كەتتى ، ئىشىك تۈۋىدە تۇرۇپ “خاتىرجەم بول ، مەندىنمۇ گۇمان قىلامسەن ؟ ماڭا تاپشۇر ، ئۆزۈڭ ياخشى داۋالان “ دېگىنىنى ئاڭلاپ قالدىم .  كىمىدۇ ؟ ئاتا - ئانىسىمىدۇ ؟ ياكى بىرەر تۇققىنىمىدۇ ؟ نىمانچە يىغلاپ كىتىدۇ ؟  رەيھان تېخىچە توختىمايۋاتاتتى ، ھازىر كىرسەم ئوڭايسىزلاندۇرۇپ قويشۇمدىن ئەنسىردىم . سەل تۇرۇپ كىرىشنى ئويلاپ سىرىتقا ماڭدىم .
-- مەردان ، كىرىۋىرىڭ . رەيھاننىڭ ئاۋازىدىن چۈچۈپ توختىدىم . قانداق بىلگەندۇ ؟ ئەمدى كىرمەي بولمايتى .
-- ئەسسالامۇ - ئەلەيكۇم . بۈگۈن قانداقراق . يىغلىغىنىغا دىققەت قىلمىغان بۇلۋالدىم .
-- ۋە ئەلەيكۇم ئەسسالام . شۇكىرى . ياخشىلا بۇلۇپ كەتتىم . ئەمدى چىقىپ كەتسەممىكىن دەيمەن مەردان . ھېلىمۇ سىلەرنى كۆپ ئاۋارە قىلدىم . ساپ ساق تۇرۇپ يېتىۋىرىشكە يۈزۈم چىدىمايۋاتىدۇ . رەيھاننىڭ كۆزلىردىن ئىككى تامچە ياش ئېقىپ چۈشتى . ئىچىم باشقىچە بىئارام بۇلۇپ كەتتى  .
-- نىمە دەپ يۈرۈيسىز ؟ بىز سىزگە مىننەت قىلدۇقما ؟ ياردەم قىلالىغىنىمىزدىن خۇشال بۇلۇپ يۈرۋاتساق ما گەپنى كۆرۈڭ . يەنە كېلىپ تېخى ساقىيىپ كىتەلمىدىڭىز . يەنە بىر نەچچە كۈن يېتىڭ . ئاندىن بىر گەپ بولسۇن ، ماقۇلما ؟
-- بۇنداق دەپ تۇرۋالسىڭىزمۇ مەيلى ، گېپىڭىزنى ئاڭلاي . لېكىن مەن تىزراق چىقىپ قىلمىسام بولمايدىغان ئىشلار بار .
-- نىمە ئىشلار ئۇ ؟ مەن قىلىشىپ بەرسەم بولمامدۇ ؟ سىز ياخشىراق ئارام ئالسىڭىز .
-- خاتىرجەم بۇلۇڭ ، سىزمۇ بىكار قالمايسىز . رەيھان بۇ سۆزلەرنى ئەستايىدىل دەۋاتاتتى .
-- نىمە ئىش ئىدى ؟ ئېيتىپ بەرسىڭىز ئىدىيە جەھەتتە تەييارلىق كۆرۇپ قويسام .


                                             ئۈستىلەنگەن قەلىب يارىسى (3)                           


-- سىزدىن ئۆتۈنەي دوختۇر ، شەمسىيەنى قۇتقۇزىۋېلىڭ . سىزدىن ئۆتۈنەي ...
-- بولدى قىلىڭ مەردان ، ئوپراتسىيەگە يەنە بىر نەچچە كۈن بار . سىزمۇ كەيپىياتىڭىزنى ياخشى تەڭشىمىسىڭىز بولمايدۇ . ياخشى ئارام ئېلىڭ ھە . مەسئۇل دوختۇر چىقىپ كەتتى . نېمىشقىكىن بىرسى شەمسىيەنى ھازىرلا تارتىپ كېتىدىغاندەك ئەنسىز ئىدىم . ئوپراتسىيەنىڭ بىرەر كۈن بولسىمۇ بۇرۇنراق قىلىنىشىنى ، ساقايغان شەمسىيەنى كۆرۈشنى شۇنچىلىق ئۆمۇد قىلاتتىم . لېكىن بۇ ئىشلار مەن ئالدىرغان بىلەن ھەل بولمايدىكەن . دوختۇرنىڭ دىيىشىچە مىنى تېخى ئومۇمىيۈزلۈك بىر قېتىم تەكشۇرمىسە بولمايدىكەن . ئۇلۇغ اللا  ... شەمسىيەنى ئۆز پاناھىڭدا ساقلىغىن .

                 ئەسلى ئىش مۇنداق ئىدى . رەيھان شۇ كۈنكى سۆزىدىن كىيىن  بىزنىڭ مەجبۇرىمىز بىلەن كىسەلخانىدا ئىككى كۈن ياتتى ،. دوختۇرخانىدىن چىققان كۈنى رەيھان مىنى بىز دائىم بىللە تاماق يەيدىغان تاماقخانىدا ساقلاپ تۇرۇشقا بۇيرۇپ ئۆزى مەكتەپكە قايىتتى ، بىر سائەتچە ساقلىغاندىن كىيىن بىر سومكىنى يۈدۈپ كىرىپ كەلدى . نىمە ئىش ئىكەنلىكىنى بىلمىدىم . سورىساممۇ ئېيتىپ بەرمىدى . تاماقنى ھاپىلا شاپىلا يىدۇق دە ئۇزۇن يوللۇق قاتناش بىكىتىگە كەلدۇق . مەن ھىچنەرسىنى بىلمەيمەن ، پەقەت رەيھان قىلىڭ دىگەننى قىلىۋاتاتتىم . ماشىنغا چىقىپ كىچىلەپ يول يۈرۈپ ئەتىسى سەھەردە مەن شەمسىيەنى يۆتكۈلۈپ كېتىپتۇ دەپ ئاڭلىغان ۋىلايەتكە يىتىپ كەلدۇق ، ئەمەلىيەتتە بۇ دەل شەمسىيەنىڭ يۇرتى ئىدى . بىراق كۆپ ئادەم ئۇنداقلىقىنى بىلمەيتى . ناشتىلىققا ئازراقلا بىر نەرسە يىدۇق دە يەنە يولغا چىقتۇق . بۇقېتىم ئالماشقىنىمىز دەل شەمسىيەنىڭ ئۆيى ئولتۇرۇشلۇق ناھىيگە باردىغان ماشىنا ئىدى . گۇمانلىرىم كۈچۈيىشكە باشلىدى . رەيھان يەنىلا ئۇن تىنسىز ئولتۇراتتى . چىرايىدىن ناھايىتى ئالدىراۋاتقانلىقى ، چوڭقۇر ئېچىنىش چىقىپ تۇراتتى ، يۈرۈگۈم تېخىمۇ تىز سوقۇپ مەنمۇ جىددىيلىشىشكە باشلىدىم . بىر سائەتچە ماڭغاندىن كىيىن مەلۇم ناھىيگە يىتىپ كەلدۇق . رەيھان تاكسى توسۇپ مەلۇم ئولتۇراق رايۇنىغا باردىغانلىقىمىزنى ئېيىتتى . ئۇھ ، بۇ شەمسىيەلەرنىڭ ئادىرىسى ئەمەس ئىدى . يۈرۈكۈم سەل ئىزىغا چۈشكەندەك قىلدى .
--رەيھان ، قەيەرگە كىتىۋاتىمىز ؟ دىسىڭىزچۇ ، ھېلىمۇ تۈنۈگۈندىن بېرى بىرەر ئېغىز گەپ قىلمىدىڭىز .
-- ھازىرلا بىلىسىز مەردان . ئالدىرماڭ . رەيھاننىڭ كۆزىدىن ياش چىقىپ كەتتى . نىمىنى خاتا قىلىپ قويغۇنىمنى بىلەلمىدىم . جىم ئوتۇرۇشتىن باشقا ئامال يوق . يىتىپ كەلدۇق . كونا بىر ئولتۇراق رايۇنى بۇلۇپ شەمسىيەنىڭ ئۆيى ئەمەسلىكىنى تېخىمۇ جەزىملەشتۈردۈم . ئىشىكنى چەكتۇق ، ئىچىدىن چىقىپ كەلگىنى دەل شەمسىيەنىڭ ئانىسى ئىدى ، يۈرۈگۈم قارت قىلىپ قالدى . سالاملاشقاندىن كىيىن رەيھان  ئۆزىنى پاتامخان ھەدەمگە ئاتتى ، ئىككەيلەن بىر ھازا يىغلاشقاندىن كىيىن بىزنى ئۆيگە باشلىدى ، تا ھازىرغىچە مەن ھېچنىمىدىن خەۋەرسىز ئىدىم .مىھمانخانا ئۆيگە كىرىشىم كۆزۈم يوتقان ئىچىدە ياتقان شەمسىيەگە چۈشتى ، بۇرۇنقى ئاناردەك مەزڭزى ئاللا قاياقلارغا يۈتكەن ، تاتارغان ، ئورۇقلاپ بىر تىرە بىر ئۇستىخان بۇلۇپ قالغان ، بۇرۇنقى قان تېمىپ تۇرىدىغان لەۋلىرى شورلاشقان بۇلۇپ كۆزلىرىنى ئىشىككە تىكىپ ياتاتتى ، بەلكىم بىزنىڭ كىلىشىمىزدىن خەۋەردار بولسا كېرەك . كۆز ئالدىم قاراڭغۇلاشتى ، تىزلىرىم ماغدۇرسىزلىنىپ يىقىلدىم . ..

             ئۇنىڭدىن كىيىن نىمە ئىشلار يۈز بەردى بىلمەيمەن . ھوشۇمغا كەلگەن ۋاقىتتا مەنمۇ شەمسىيەنىڭ يېنىدا يەنە بىر كۆرپە ئۈستىدە ياتاتتىم . چاچىراپ تۇردۇم دە شەمسىيەگە ئېسىلدىم . كۆزۈمگە نە شەمسىيەنىڭ ئانىسى نە رەيھان كۆرۈنمىدى .
-- بۇ نىمە ئىش شەمسىيە ، زادى نىمە ئىش ؟ سىز نىمە بولدىڭىز شەمسىيە ؟ نىمىشقا بۇنچە تاتىرىپ كەتتىڭىز شەمسىيە ؟ نىمىشقا بۇنچە ياداپ كەتتىڭىز شەمسىيە ؟ ئانارنىڭ دانىسىدەك چىرايىڭىز قېنى شەمسىيە ؟ نۇرلۇق كۆزلىرڭىز قېنى شەمسىيە ؟ گەپ قىلىڭ شەمسىيە ، زادى نىمە ئازابلارغا دۇچار بولدىڭىز ، شەمسىيە  ، گەپ قىلمامسىز شەمسىيە ... كۆزلىرىمدىن بەڭباش ياشلار تۆكۈلمەكتە ، ياق ، بۇ ياش ئەمەس ، يۈرۈگۈمدىن قانلار ئېقىۋاتمامدۇ ، يۈرۈگۈم پىچاق تىققاندەك ئاغرىۋاتمامدۇ . شەمسىيە بىرەر ئىغىز بولسىمۇ گەپ قىلىڭ بۇلامدۇ ، نىمە يىغلاپلا ئولتۇرامسىز ، گەپ قىلىڭ ، مەن ئاۋازىڭىزنى ئاڭلاي سەمسىيە ...
شەمسىيە ئۆكسۈپ ئۆكسۈپ يىغلىدى ، بىرەر ئېغىز گەپ قىلمىدى ، قۇچىغىمدا ماڭا يۆلۈنۈپ قانغۇچە يىغلىدى . تولا يىغلاپ مىنىڭمۇ ئۇنلىرىم پۈتۈپ كەتتى . رەيھان كېلىپ “بولدى قىلىڭ شەمسىيە بەرداشلىق بىرەلمەيدۇ “دەپ ئەسكەرتكەندىلا ئۇنى ئاۋايلاپ ياتقۇزۇپ قويدۇم . شەمسىيەنىڭ ئانىسى سىرىتقا چىقىپ كەتكەن بۇلۇپ ئۆيدە پەقەت ئۈچىمىز لا قالغان ئىدۇق .
-- ھەممىنى بىر بىرلەپ سۆزلەڭ رەيھان ، سىز ھەممىنى بىلىسىز ، مەندىن يۇشۇرۇپسىز . .. يەنە يىغا تۇتتى ،گەپ قىلالماي قالدىم . شەمسىيەنىڭ ھازىرقى تۇرقىنى كۆرسە تونىمايدىغان ئادەممۇ يىغلاپ سالاتتى .
-- ماقۇل مەردان . مەن ئېيتىپ بىرەي . ئەسلى ئىش مۇنداق . شەمسىيە ھېچقاچان سىزگە ئاسىيلىق قىلغىنى يوق ، ھەم سىزگە يۈز كېلەلمىگۈدەك ھېچ ئىش قىلغىنى يوق . سىلەر يۈرىۋاتقان مەزگىلىڭلاردا ھە دىسىلا شەمسىيەنىڭ بۇرنى قانايدىغانلىقىنى ئەسلىيەلەمسىز . دەل شۇ ئىش . شەمسىيە ئىزچىل سىزدىن يۇشۇرۇپ ئەڭ ئاخىرى سىز ۋە دوستلىرىڭىز مەلۇم شەھەرگە ئوينىغىلى كەتكەندە يىقىلدى . كىچىلەپ دوختۇرخانىغا ئېلىپ باردۇق ، تەكشۇرۇش نەتىجىسى ھەممىمىزنى چۈچۈتتى. شەمسىيە ئاق قان كىسىلىگە گىرىپتار بۇلۇپ قالغان ئىدى . ئارىمىزدا بىرىنچى بۇلۇپ ھوشىنى تاپقىنى شەمسىيە بولدى . ھەممىمىزگە يالۋۇرۇپ تۇرۇپ بۇ ئىشنى سىز ۋە باشقىلاردىن يۇشۇرۇن تۇتىشىمىزنى ئۆتۈندى ، بىزمۇ ماقۇل بولدۇق . سىز قايتىپ كەلگۈچە شەمسىيە ئىككىمىز “شەمسىيەنىڭ يېڭى يىگىتى” نى ئۇرۇنلاشتۇرۇپ ئۈلگۈردۇق . شەمسىيە سىزنى ئۇنىڭ كېسىلى سەۋەبلىك ئازابلاشنى خالىمىدى . بۇ كېسەلنىڭ داۋاسى يوقمۇ ئەمەس لېكىن ماس كېلىدىغان يىلىك تېپىلمىسىلا باشقا ئامال يوق بولغاچقا ئۇ سىزدىن ئايرىلىش قارارىنى چىقاردى . بىر ئاماللارنى قىلىپ سىزگە شەمسىيەنى ۋاپاسىزلىق قىلغان قىلىپ كۆرسەتتۇق . ئۆزىڭىز ئايرىلىش تەلپىنى ئوتتۇرغا قويدىڭىز، بۇ ئەلۋەتتە دەل بىز كۈتىۋاتقان نەتىجە ئىدى . ئەتىسى سەھەردىلا شەمسىيە يۈرتىغا قايىتتى . بىز داۋاملىق ئالاقىلشىپ تۇردۇق ، شەمسىيەنىڭ قەيەردە نىمە كۈنلەرنى كۆرگىنىنى  مەن ئېنىق  بىلىمەن مەردان . بىر يىلغا يېقىن ۋاقىت ئىچىدە شەمسىيە بارمىغان دوختۇرخانا قالمىدى ، خېمىيەلىك داۋالاشلارنى باشتىن ئۆتكۈزدى . لېكىن ئۇ سىزنى بىرەر دەقىقىمۇ ئۇنتۇپ قالغىنى يوق . ئىشەنمىسىڭىز مانا ئاڭلاڭ . رەيھان شەمسىيەنىڭ ياستۇقى يېنىغا قويۇلغان تېلفۇننى ئېلىپ بىر نەرسە قويدى ، بۇ مىنىڭ ئاۋازىم ئىدى . رەيھان يەنە سۆزلەشكە باشلىدى . شۇنداق مەردان ، بۇ دەل سىزنىڭ ئاۋازىڭىز ، ئىككىمىز بىللە تاماققا چىققان ۋاقىتلاردا سىز بىلەن بولغان پاراڭلىرمىزنى ئۈنگە ئېلىپ شەمسىيەگە يوللاپ بىرەتتىم . ئەلۋەتتە بۇمۇ شەمسىيەنىڭ سىزنىڭ ئاۋازىڭىزنى ئاڭلاش ئۈچۈن تاپقان ئامالى . مەردان ، مەن كىسەل بۇلۇپ قالغان ۋاقىتتا سىز ئاڭلاپ قالغان تېلفۇن دەل شەمسىيە بىلەن سۆزلەشكەن تېلفۇن ئىدى . سىزمۇ سورىمىدىڭىز ، مەنمۇ ئېيتىپ بەرمىدىم . شەمسىيەگە ماس كىلىدىغان يىلىك تېپىلمىدى .  شۇڭا شەمسىيەنىڭ ئارماندا كەتمەسلىكى ئۈچۈن شەمسىيەنىڭ ئۇنىمىغىنىغا ئۇنىماي سىزنى باشلاپ كەلدىم . مىنى كەچۈرۈڭلار .
كۆزلىرىمدىن ياشلار ئاقماقتا ، مەن زادى قانداق ئادەم ؟ مەن نىمە قىلىپ قويدۇم ؟ ئۆزۈم سۆيگەن ئادەمنى شۇنچە ئۇزۇن تاشلىۋەتتىممۇ ؟ ئۇنىڭ ئالدامچىلقىنى نىمىشقا بىلمىدىم ؟ مىنىڭ ئۇچۇن چەككەن ئازابلىرىنى نىمىشقا بىلمىدىم ؟ ئايرىلغان كۈنى ئۇنىڭ كۆزىگە تىكىلىپ قارىغان ئىدىمغۇ ، نىمىشقا ئۇنىڭ مىنى شۇنچىلىك سۆيىدىغانلىقىنى كۆرەلمىدىم ؟ كۆزلىرىم كور ئەمەس ئىدىغۇ ...كۆزلىرىم كور بولغان تەقدىردىمۇ يۈرۈگۈم سىزىشى كىرەكتىغۇ ...نىمىشقا سىزەلمىدىم ؟ ئۇنىڭ يالغان يىگىتى كۆزلىرىمنى كور ، يۈرىگىمنى سەزمەس قىلىۋىتەلىدىمۇ ؟؟؟
                 ئۆي ئىچى ئۈچىمىزنىڭ يىغىسى بىلەن تولغان ئىدى . شەمسىيەنىڭ ئانىسى كىرىپ كەلمىگەن بولسا بەلكىم توختىماسلىقىمىز مۇمكىن ئىدى . بالىسىنىڭ دەردىدە ئاران تۇرغان بىر ئانىنى تېخىمۇ ئازابلاشنى خالىمىدۇق . ياشلىرمىزنى سۈرتۈپ گەپنى بۇرىدۇق ، شەمسىيەنىڭ ئاۋازىمۇ سېغىنىپ قالغان ئىدىم ، ئەمدى قۇلىقىمىز ئۇنىڭدا بولدى . لېكىن ئاجىزلاپ كەتكەنلىكى ، ئۇنىڭ ئۈستىگە بىز بىلەن تەڭ يىغلاپ تېخىمۇ ماغدۇرسىزلىنىپ ناھايىتى بوش ئاۋازدا گەپ قىلىۋاتاتتى . مەندىن ئايرىلىپ يۇرتقا قايتقاندىن كىيىن بولغان ھەممە ئىشلارنى ئاڭلىدىم . كەچلىك تاماقتىن كىيىن ماڭا ئايرىم ھوجىرىغا ئورۇن راسلاپ بىرىشتى . سىرىتقا چىقىپ ياتاي دىسەم پاتەمخان ھەدەم يىغلاپ تۇرىۋالدى . شۇنداق قىلىپ شەمسىيەلەرنىڭ ئۆيىدە قونۇپ قالدۇق . كېچىچە ئۇخلىيالمىدىم ، ئەتتىگەنلىك تاماقتىن كىيىنلا “ ئازراق ئىشىم بار ئىدى “ دەپ دوختۇرخانىغا چاپتىم . ئەھۋالنى تونۇشتۇرغىنىمدىن كېيىن ئۇلار مىنى تەكشۇردى . بەزى نەتىجىلەر چۈشتىن كېيىن چىقىدىغان بولغاچقا دوختۇرخانىدىن قايتىپ چىقتىم . ئادىلغا تېلفۇن قىلىپ مەكتەپتىكى ئىشلارنى ئورۇنلاشتۇردۇم . بۇنداق بىكار ئايلىنىپ يۈرۈش ياخشى ئەمەس ئىدى ، يەنە كىلىپ كۆڭلۈمنىڭ ئارامى يوق ، شەمسىيەدىن ئەنسىرەپلا تۇرغاچقا شەمسىيەلەرنىڭ ئۆيىگە قايىتتىم . ئۆيگە كىرگەن ۋاختىمدا چۈشلۈك تاماقمۇ پىشىپ قالغان ئىكەن ، تاماق يىگەچ پاتەمخان ھەدەمدىن قانداق قىلىشنى ئويلىشىۋاتقىنىنى سورىدۇم . بىچارە ئانا شۇنچىلىك ئۆكسۈپ يىغلىدىكى سوراپ قالغىنىمدىن پۇشايمان قىلدىم . ئۇنىڭ باش چايقىشىدىن ئەمدى ئامال يوق دىگىنى چىقىپ تۇراتتى . ھەممىمىز شۇكلەپ قالدۇق ، تاماقتىن كىيىن رەيھاننى سىرىتقا تولىدىم .
-- رەيھان ، مەن دوختۇرخانىغا بېرىپ تەكشۈرتتۈم ، نەتىجىسى چۈشتىن كىيىن چىقىپ قېلىشى مۇمكىن ، بىر ئامال قىلىپ شەمسىيەنىڭ تەكشۈرتۈش قەغىزىنى ئېلىپ بەرسىڭىز ...
-- ماقۇل ، مەن ئامال قىلىپ باقاي . رەيھان كىرىپ كەتتى . تەكشۈرۈش نەتىجىسىنى بىلىشتىن بۇرۇن شەمسىيە ئۇقسا بەلكىم ئۇنىماسلىقىمۇ مۇمكىن ، مۇۋاپىق كەلمەي قالسا تېخىمۇ ئازاپلىنىدۇ دە . شۇڭا بىر ئامال قىلىپ ئۇختۇر ماي ئېلىش كىرەك ئىدى .
            رەيھان يامان قىز بىردەمدىلا ئىشنى پۈتتۈرۈپتۇ ، ئۇ بارلىق ماتىرياللارنى ماڭا بەردى ، كۆزلىردىن قانچىلىك ئۆمۈدنىڭ چاقناپ تۇرغىنىنى بەلكىم رەيھان ئۆزىمۇ بىلمىسە كېرەك .
ماتىرياللارنى سومكامغا سېلىپ يەنە دوختۇرخانىغا باردىم ، تەكشۈرۈش نەتىجىللىرى چىققان بۇلۇپ پەقەت سېلىشتۇرۇشلا قالغان ئىدى . دوختۇر بىر ھازا كۆرگەندىن كىيىن ،
-- مۇبارەك بولسۇن ، سىڭلىڭىزنى قۇتقۇزۇپ قالىدىغان بولدىڭىز . دىدى . مەن شەمسىيەنى سىڭلىم دىگەن ئىدىم . دوختۇر ئىشخانىسىدا كىچىك بالىدەك سەكىرەپ كەتتىم ، خۇشاللىقىم ئىچىمگە پاتمايۋاتاتتى . شۇكىرى ، ياراتقان ئىگەمگە مىڭ شۈكىرى . شەمسىيە ھايات قالىدىغان بولدى ، مىنىڭ مىھرىبانىم ...ھايات قالىدىغان بولدى ...
          تىز سۈرەتتە شەمسىيەلەرنىڭ ئۆيىگە قايتتىم
           ئىشىكتىن ئۇچقاندەك كىردىم دە شەمسىيەنىڭ قېشىغا كەلدىم ،
-- شەمسىيە ، يىلىك تېپىلىپتۇ ، ئاڭلىدىڭىزمۇ يىلىك تېپىلىپتۇ . سىزنىڭ تەييارلىقىڭىز پۈتسىلا ھەر ۋاقىت ئوپىراتسىيە قىلسا بولىدىكەن . يىلىك ئىئانە قىلغۇچىنىڭ مەن ئىكەنلىكىمنى ئېيتمىدىم .  
كۆزلەردىن ياشلار يەنە ئاقماقتا ، ئوخشىمايدىغىنى بۇ قېتىمقىسى خۇشاللىق يېشى ، تەكلىماكان قۇملىقىغا قاپسىلىپ قېلىپ ئۇسسۇزلىقتىن ئۆلەي دېگەن ئادەمگە بۇلاق ئۇچىرغاندەك بىر ئىش ...
ھەممىمىز خۇشال بولدۇق . شەمسىيەنىڭ دادىسىمۇ شۇ كۈنى قايتىپ كەلدى ، نىمە دەپ يۈز كۆرۈشىشىمنى بىلمەي تۇراتتىم مىنىڭ ئالدىغا ئۇدۇللا كىلىپ.
-- مەردانجان بالاممۇسىز ، ئۇزۇن بولغان گىپىڭىزنى ئاڭلىغىلى ، قېنى تارتىنماي ئولتۇرۇڭ بالام . “ بالام “ سۆزىدىن بىر تەرەپتىن قىزىرىپ يۈزلىرىم ئوت ئالسا يەنە بىر تەرەپتىن خۇشال بولغان ئىدىم . لېكىن مىنى قانداق بىلىدۇ ؟ بولدىلا بەلكىم رەيھاننىڭ كىسىلى ئېغىر مىنىڭ رەسىملىرىمنى كۆرۈپ ، ئاۋازلىرىمنى ئاڭلاپ تۇرغاچقا بىرەر قېتىم كۆرۈپ قالغان ، پاتەمخان ھەدەمدىن سوراپ بىلگەندۇ .
ھەممە ئىشنى قويۇپ تۇرۇپ دوختۇرخانىغا باردۇق . دوختۇرلار بۇ ئوپراتسىيەنى قىلالمايدىغانلىقىنى ، پەقەت تەكشۈرۈپ بىكىتىپ بىرەلەيدىغىنىنى ئېيىتتى . شەمسىيەنىڭ دادىسى چوڭ سودىگەر بۇلۇپ شەمسىيەنى داۋالىتىپ يۈرۈپ نۇرغۇن بىساتىدىن ئايرىلغان ئىكەن . شۇنداق بولسىمۇ ھەر جايدا تونۇشلىرى كۆپ بولغاچقا ئوپراتسىيەگە يىتەرلىك پۇلنى بىر كۈن ئىچىدىلا ھەل قىلىپ بىزنى ئېلىپ مەلۇم شەھەرگە كەلدۇق . كىلىپلا دوختۇرخانىغا ئۇرۇنلاشتۇق . تەكشۈرۈش نەتىجىللىرىنى مەسئۇل دوختۇرغا تاپشۇرۇپ كىسەلخانىغا قايىتتۇق .  بۈگۈن كېچە شەمسىيە باشقىچە سارغىيىپ كىتىپتۇ ، ئەتىسى سەھەر ئۇنى يوقلىغىلى كىرىپ شۇنچىلىك ئەنسىرەپ كەتتىم . ئەينەكلىك ئايرىلغان كىسەلخانىدا ماڭا كۈلۈمسىرەپ تۇرغان شەمسىيە بۇ  بىر كۈننىڭ ئىچىدە تېخىمۇ ياداپ كەتكەندەك كۆرىندى . ئۆزۈمنى ئاران بېسىۋېلىپ مەسئۇل دوختۇرنىڭ قېشىغا كىردىم ، ئەمدى ئۆزىمنى تۇتالمىدىم ...
-- سىزدىن ئۆتۈنەي دوختۇر ، شەمسىيەنى قۇتقۇزىۋېلىڭ . سىزدىن ئۆتۈنەي ...
-- بولدى قىلىڭ مەردان ، ئوپراتسىيەگە يەنە بىر نەچچە كۈن بار . سىزمۇ كەيپىياتىڭىزنى ياخشى تەڭشىمىسىڭىز بولمايدۇ . ياخشى ئارام ئېلىڭ ھە . مەسئۇل دوختۇر چىقىپ كىتىشىمنى ئىشارەت قىلدى ، دىققەت قىلماپتىمەن ئۇ بىرسى بىلەن بىر ئىش توغۇرلۇق پاراڭلىشىۋاتقان ئىكەن. ئېغىر قەدەملىرىمنى يۆتكەپ دوختۇر ئىخانسىدىن قايتىپ چىقتىم  . نىمىشقىكىن بىرسى شەمسىيەنى ھازىرلا تارتىپ كېتىدىغاندەك ئەنسىز ئىدىم . ئوپراتسىيەنىڭ بىرەر كۈن بولسىمۇ بۇرۇنراق قىلىنىشىنى ، ساقايغان شەمسىيەنى كۆرۈشنى شۇنچىلىق ئۆمۇد قىلاتتىم . لېكىن بۇ ئىشلار مەن ئالدىرغان بىلەن ھەل بولمايدىكەن . دوختۇرنىڭ دىيىشىچە مىنى تېخى ئومۇمىيۈزلۈك بىر قېتىم تەكشۇرمىسە بولمايدىكەن . ئۇلۇغ اللا  ... شەمسىيەنى ئۆز پاناھىڭدا ساقلىغىن .

شەمسىيە ، ئورنىڭىزدىن تۇرۇڭ شەمسىيە ، باشقىلار كەتتى ، يېنىڭىزدا مەنلا قالدىم شەمسىيە ، شەمسىيە ، ئاڭلىدىڭىزمۇ ؟ بىرەر ئېغىز گەپ قىلىڭ شەمسىيە . سىزدىن ئۆتۈنەي شەمسىيە ،مىنى مۇشۇنداقلا تاشلاپ كەتمەكچىمۇ سىز ، مىنى يالغۇز قويماقچىمۇ سىز ، شەمسىيە گەپ قىلىڭ ، مەندىن ئۆچىڭىزنى ئېلىۋاتامسىز ، يالغۇزلۇقنىڭ تەمىنى تېتىسۇن دىدىڭىزمۇ شەمسىيە ، مەن تارتقان ئازابنى تارتىپ باقسۇن دىدىڭىزمۇ شەمسىيە ، مىنى بۇ يالغۇزلۇققا چىدىيالايدۇ دىدىڭىزمۇ شەمسىيە ، بۇ دۇنيانىڭ ئازابلىرىنى ماڭىلا تاشلاپ كەتتىڭىزغۇ شەمسىيە ، ھەر ئىش بولسا تەڭ كۆرىمىز دىگەن ئىدىڭىزغۇ شەمسىيە ، ئەمدى ھەرگىز ئالداپ قويۇپ قاچمايمەن دىگەن ئىدىڭىزغۇ شەمسىيە ، يەنە قېچىپ كەتتىڭىزغۇ شەمسىيە ، مىنى يەنە ئالداپ قويۇپ يىراقلارغا كەتتىڭىزغۇ شەمسىيە ، كەلمەسلەرگە كەتتىڭىزغۇ شەمسىيە ، ئەمدى سىزنى  نەدىن تاپاي شەمسىيە ، ئەمدى دەردىمنى كىمگە تۆكەي شەمسىيە ، ئاتا - ئانام كېلىنىم قېنى دىسە نىمە دەپ جاۋاب بىرىمەن شەمسىيە ... سىز ماڭا ئېيتىپ بىرىڭ مەن زادى قانداق قىلىمەن ............سۆزلىرىم تۈگەيدىغاندەك ئەمەس ، قاراڭغۇ چۈشتى ، قەبىرە قېشىدا قانچىلىك ئولتۇرغىنىمنى بىلمەيمەن ، ئۈنلۈرۈم پۈتۈپ ئاۋازىم چىقماس بۇلۇپ كىتىپتۇ ، ئادىل كەلگەندە مەن تېخىچىلا خىيالەن سۆزلىشىۋاتاتتىم . ئادىل مىنى تارتىشتۇرۇپ ئېلىپ ماڭدى . ئۇنىڭ بىلەن قارشىلاشقۇدەك ماغدۇرۇم قالمىغان ئىدى .

                                             ئۈستىلەنگەن قەلىب يارىسى (4)  


                                         كۆزۈمدىكى مەنالىرىم سەن ئىدىڭ ،
                                         ئارزۇلاپ ھەم سۆيگەن يايرىم سەن ئىدىڭ.
                                          دەرتلىرىمگە دەرمانە ھەم سەن ئىدىڭ ،
                                          تاپالمىدىم ئىنتىزارىم سەن ئىدىڭ .

                                           يولگە قاراپ نىگارىمنى تاپالماي ،
                                           ساڭا بىر رەت ئەھۋالىمنى ئېيتالماي .
                                           يەنە كەلدىم قەبرەڭدىن ھەم تاپالماي ،
                                           شاتلىقىمنى ئېلىپ كەتكەن سەن ئىدىڭ .

                                            تاپالمىدىم نىگارىمنى ھەممىدە ،
                                            چۈنكى يايرىم ياتار غەمسىز قەبرىدە .
                                            كىممۇ يايرىم يىتەر ئەمدى سەن كەبى ،
                                             مەندەك غېرىپ يالغۇزنىڭ ھەم دەردىگە .

                              (تاشمۇھەممەت باتۇر ئىجىراسىدىكى سەن ئىدىڭ ناملىق ناخشىدىن ئېلىندى )
                       ھەقىقەتەن ئۆلگەننىڭ كەينىدىن ئۆلگىلى بولمايدىكەن ، “ئەلھۆكمۈلۈللا”  دىيىشتىن باشقا ئامالىمىز يوق ئىكەن . مىڭ قايغۇرساقمۇ بىكار ئىكەن . ھاياتلىرىمىز يەنىلا كۈنلىرمىزنى داۋاملاشتۇرۇشتىن قېچىپ قۇتۇلالمايدېكەنمىز . شەمسىيە كەتتى ، بۇ دۇنيانىڭ بارلىق خۇشاللىق ۋە ئازاب ئوقۇبەتلىرىدىن بىراقلا قۇتۇلدى. (ياتقان يىرى جەننەتتە بولسۇن .)  شەمسىيەنىڭ يىلىنى تۈگۈتۈپ يۇرتقا قايتىپ كەلگىنىمگە بىر ئايچە بولدى . بۈگۈن ئەتىگەن شەمسىيەنىڭ خېتى قولۇمغا چىقىپ قالدى . نىمىلا دېگەن بىلەن بۇ ئۇنىڭ ھاياتلىقىدا مەندىن كۈتكەنلىرى ئەمەسمىدى ، مىنىڭ ئۇ تاپىلىغان بويۇنچە قىلىپ روھىنى خۇشال قىلىشىمنى كۈتۈۋاتقاندۇ . بىر نەچچە قېتىملىق چۈشۈممۇ بۇنىڭغا بىشارەت بولسا كېرەك .
           رەيھان بىلەن بازار ئايلىنىۋاتقۇدەكمەن ، ماڭا ئاياغ سېتىۋالماقچى بۇلۇپ رەيھان تاللاپ بەرگەن ئاياقنى قىممەتكەن دەپ ئالغىلى ئۇنىمىغۇدەكمەن ، رەيھان مەن تاللاپ بەرگەننى ئالمىدىڭىز دەپ يىغلاۋاتقۇدەك ، ئوخشاش بىر چۈشنى ئىككى قېتىم كۆردۈم . بەلكىم شەمسىيە مىنىڭ رەيھاننى ئىزدىمىگىنىمدىن روھى قورۇنىۋاتقان بولسا كېرەك . ئىزدەي ، رەيھاننى ئىزدەپ باراي ، شەمسىيەنىڭ خېتىنى كۆرسىتەي ،.. شەمسىيەنىڭ روھى ئەمەن تاپسا ئەجەب ئەمەس .
بىر يىلدىن بېرى رەيھانغا تۇنجى قېتىم تېلفۇن قىلىشىم ئىدى. رەيھان تېلفۇننى ئالدى .
--ئەسسالامۇ- ئەلەيكۇم .مەردانمۇ سىز ،  قانداق ئەھۋالىڭىز ؟
-- ۋە ئەلەيكۇم ئەسسالام . ھەئە ، مەن مەردان ، شۇكىرى . ئۆزىڭىزچۇ ؟ سالامەتمۇ سىز؟ ئۇزۇندىن بېرى باردى كەلدى قىلمىغاچقا بىر خىل تارتىنىش ھەر ئىككىلمىزنى ئەسىرگە ئالغان ئىدى . ئەمدى نىمە دىيىشىم كېرەك ؟ گېپىمنى تاپالماي تۇراتتىم رەيھان ئۆزى گەپ قىلدى .
-- مۇنداق قىلساق قانداق ، مەن بۇگۈن دەم ئالغان ، سىزگىمۇ دەم ئېلىش بولغاندىكىن چۈشتە بىللە تاماق يىسەك ، ئۇزۇن بولدى سىزنى كۆرمىگىلى ...رەيھاننىڭ ئاخىرقى بىر جۈملە سۆزىدىن سەل خىجىل بولدۇم . دىمىسىمۇ ئۇنى بۇنداق تاشلىۋەتسەم بولمايتى ، تولىمۇ يۈزسىزلىك قىلغان بۇلۇپ قالاتتىم ، قانداقلا بولمىسۇن رەيھان شەمسىيەنىڭ بىردىنبىر جانجىگەر دوتى ئىدى ، بىزنىڭ نۇرغۇن ئىشلىرمىزغىمۇرەيھان يادەم قىلغان ئىدى .
-- ھە ، ماقۇل ، چۈشتە بىللە تاماق يەيلى . سىرىتتا يەمدۇق ياكى بىزنىڭ ئۆيگىلا كىلەمسىز؟ راستىنى ئېيتقاندا ئۆزۈمنىڭ سىرىتتا تاماق يىگۈم يوق ئىدى .
-- مىنىڭمۇ سىلەنىڭ ئۆيگە بىر بېرىپ كىلىش يولۇم بار ئىدى ، سىلەرنىڭ ئۆيگىلا باراي . ھە راست مەردان ، ئانىڭىزنى ئارتۇق ئاۋارە قىلىپ يۈرمەڭ ، بولسا مەن ئۆزۈم بېرىپ تاماق قىلاي . رەيھان بىزنىڭ ئۆيگە كۆنۈك ئىدى . گەرچە چوڭلار ئانچە كىلىشىپ كەتمىسىمۇ بىز ياشلارنىڭ بېرىش كىلىشىگە ئەزەلدىن قارشى چىقمايتى ، شۇنىڭ ئۈچۈن رەيھان بۇرۇنلاردىمۇ كىلىپ كىتىپ يۈرەتتى .
-- ماقۇل ئەمسە ، مەن ئۆيدە ساقلاي . ئاللاغا ئامانەت ، كەلگەندە كۆرۈشەرمىز . شۇنداق دەپ تېلفۇننى قۇيۋەتتىم . يۈرۈگۈم ئۆزۈمگە ئاڭلانغۇدەك سوقماقتا ، ئەمدى قانداق قىلىمەن ، ئۇ كەلسە نىمە دەيمەن ،.. خىيالىمنى بۆلۇپ ئانام كىرىپ كەلدى .
-- مەردانجان بالام ، چۈشلىكىڭىزگە نىمە قىلىپ بەرسەم بولار ، نىمىنى كۆڭلىڭىز تارتىۋاتىدۇ بالام ؟ ئانام ئەزەلدىنلا شۇنچىلىپ مىھرىبان ، كۆيۈمچان ئايال ئىدى ، بالا دىسە كىسەل بولسىمۇ تاماق ئىتىپ بېشىدا ئولتۇرۇپ يىگىنىنى كۆرۈپ ئاندىن خاتىرجەم بۇلاتتى .
-- بۈگۈن رەيھان بىزنىڭ ئۆيگە كەلمەكچى ئىدى ئانا . لەغمەن ئەتسىڭىز رەيھان ئامراق ئەمەسمۇ . ئانام كۆزۈمنىڭ ئىچىگە قاراپ قويۇپ قەۋەتلا خۇشال بۇلۇپ كەتتى ، شەمسىيە ئالەمدىن ئۆتكەندىن  بېرى ئۆزۈمنى كۆپ تاشلىۋەتتىم ، سىرىتلارغىمۇ خىزمەتنى دېمىسەم چىقمايتىم ، ئاتا - ئاناممۇ مىنىڭدىن ئەنسىرەپلا يۈرىۋاتاتتى . رەيھاننىڭ گېپىنى چىقارسام ئانامنىڭ باشقىچە خۇشال بولىشى بىكار ئەمەس ئىدى .
-- ماقۇل جېنىم بالام . مەن ھازىرلا چىقىپ تەييارلىق قىلاي . ئانام خۇشال چىقىپ كەتتى ، لېكىن مىنىڭ كۆزلىرىم يەنە ياشلار بىلەن ئۇچىرشىش قىلدى ... نىمىشقا شەمسىيەنى بىرەر قېتىم بولسىمۇ ئۆيگە ئېلىپ كىلىپ ئانام ئەتكەن تاماقلاردىن يىگۈزمىگەن بولغىيدىم ... شەمسىيەنىڭ مۇشۇ سەۋەبلىك بىرەر ئارمىنى قالغان بولارمىدى ، ئۆزى كىلىن بولۇپ كىرمەكچى بولغان ئائىلىنى بىرەر قېتىم بولسىمۇ كۆرۈپ باققۇسى كەلگەن بولغىيتى ھە ...
           ئۇلۇغ اللا ھەممىگە قادىر ، بىرىمەن دەپ بىرەلەيدۇ ، ئالىمەن دەپ ئالالايدۇ . سەبىر قىلماي ئامالىم يوق ... بېشىمدا بارىنى ئالدىرماي كۆرەي ...
بىرەر سائەتكە قالمايلا رەيھان يىتىپ كەلدى ، داستىخان سېلىپ چايدىن كىيىنلا “تاماققا ياردەملىشەي “دەپ ئۇنىماي تۇرىۋالدى . مەن ياتاق ئۆيۈمدە كىتاب كۆرۈش بىلەن مەشغۇل بولدۇم . تاماق پىشقاندا ھەممىمىز كۈلكە چاقچاقلار بىلەن تاماق يىدۇق . تاماقتىن كېيىن ئانام بىزنى “سىلەر ياشلار بىردەم پاراڭلىشىڭلار “ دەپ چىقىپ كەتتى . رەيھان ئىككىمىزلا قالدۇق ، نىمىسىنى دەي ، ھەر ئىككىلىمىز يەرگە قاراپ قىزىرىپ ئولتۇردۇق ، بۇنداق بولسا مىھمانغا بولغان ھۆرمەتسىزلىك بۇلۇپ قالاتتى ، مەن ئاۋال گەپ باشلىدىم .
-- سىزگە كۆرسىتىدىغان بىر نەرسە بار ئىدى رەيھان . شەمسىيە قالدۇرغان خەتنى ئۇنىڭغا سۇندۇم .
-- كۆرمىسەممۇ مەزمۇنىنى بىلىمەن مەردان ، مەندىمۇ بىر پارچە خەت بار ، بەلكىم ئىككىمىزگە ئوخشاش قىلىپ خەت قالدۇرغاندۇ . خاتالاشمىسام ئىككىمىزنى توي قىلىڭلار دىگەندۇ ؟ رەيھان تېخىمۇ قىزاردى .
-- مۇنداق دەڭ ، نىمىشقا بۇرۇنراق ماڭا دىمىدىڭىز ؟ سورۇدۇم ئەجەبلىنىپ .
-- قانداق دەيمەن مەردان ، سىزنى چۈشەنمىسەمكەن كاشكى . سىزگىمۇ ۋاقىت كېرەك ، ئۇنىڭ ئۈستىگە بۇ شەمسىيەنىڭ ئارزۇسى ، سىزنىڭ ئەمەس دە ...رەيھان مىنى جاۋاب بىرىشكە قىستاۋاتاتتى ، يەنى مىنىڭ زادى ئۇنىڭ بىلەن توي قىلىش نېيتىمنىڭ بار يوقلىقىنى بىلمەكچى ئىدى .
-- سىز بىلەن توي قىلىش نىيتىگە كەلدىم ، پەقەت سىز خالىسىڭىزلا ، ئىككىمىز بىر بىرمىزنى چۈشىنىمىز ،..
ئىشقىلىپ بىرقانچە سەۋەب تېپىپ بەردىم ، ئەمدى سەل قېلىنلاشقان بولسام كېرەك خېلى تارتىنماي سۆزلەۋاتاتتىم ،
-- لېكىن ھەر ئىككىلىمىزگە ۋاقىت كېرەك رەيھان .
-- مەنمۇ رازىمەن . رەيھان ئەمدى راستىنلا قىزىرىپ پىشقان ئانارغا ئوخشاش قىزاردى .-- سىز دېگەندەك ۋاقىت كېرەك . مەن ساقلىيالايمەن . ..
رەيھان قايتىپ كەتكەندىن كېيىن ئىزچىل ھالدا ئۇنى بىر ئامال قىلىپ بەخىتلىك قىلىشنى ئويلاپ كېلىۋاتىمەن .  شۇنداق قىلماي بولمايتى ، شەمسىيەگە ۋە ئۆزۈمگە يۈز كىلەلمەيتىم . ھەر قانداق ئىش يۈز بىرىشىدىن قەتئى نەزەر پەقەت اللا خالىسىلا ئۇنى بەخىتلىك قىلىش نېيتىگە كەلدىم .



                 تاك ، تاك ،تاك
                 ئىشخانامنىڭ ئىشىكى چىكىلىپ رۇشەنگۈل كىرىپ كەلدى ، رۈشەنگۈل ئاجايىپ ئۆزگەرگەن بۇلۇپ تاسلا قاپتىمەن تونۇماي قالغىلى .
-- ئەسسالامۇ - ئەلەيكۇم ، مەردان باشلىق . سالامەتمۇ سىز؟
-- ۋە ئەلەيكۇم ئەسسالام ، رۇشەنگۈل خىجىل قىلمىڭە ، دادامنىڭ شىركىتىدە باشلىق بولسام ماختانغۇدەك ئىش ئەمەسقۇ ، يە ئۆزۈم تىرشىپ ئىرىشكەن ئورۇن بولمىسا بۇ . سىزچۇ ، نىمىنداق ئۆزگۈرۈپ كەتتىڭز ؟  سىز شۇمۇ ؟  رۇشەنگۈلنىڭ ئۈستى بېشى پۈتۈنلەي ھېجابلانغان بۇلۇپ بۇرۇنقى رۇشەنگۈلنى ھازىرقى بىلەن سېلىشتۇرغىلى بولمايتى . باشلىقنىڭ قىزى بولغاچقىمۇ ئۆزى ياۋاش بولسىمۇ كىيىم دىگەننى بىر كىشلىك كىيىپ قويىدىغانلاردىن ئىدى . مانا ئەمدى بىراقلا ئۆزگۈرۈپ كىتىپتۇ .
-- گىپىڭىزنى قاراڭ ، مەن بۇنداق كىيىنسەم بولمامدىكەن ؟
-- ياق ياق ، ئۇنداق دىمەكچى ئەمەسمەن ، شۇنچىلىك يارىشىپتۇ ، اللا رەھمەت قىلسۇن . راستىنى ئېيتقاندا ئىچىمدە ئۇنىڭدىن سۆيۈنۈپ قالغان ئىدىم . بۇرۇنقى مىجەز خۇلقىغا بۇ كىيىملىرى قوشۇلسا تېخىمۇ ياخشى بۇلۇپ كىتەتتى .
-- نەق گەپكە كىلەيلى ، مەن چۈشلۈك تاماق ئورۇنلاشتۇراي ، بىللە تاماق يەيلى . شۇنداق دېدىم دە تېلفۇن قىلىپ ئورۇن زاكاس قىلدىم .
-- ھەئە ، ئۈچ كىشلىك . گېپىمنى تۈگۈتۈپ تېلفۇننى قۇيۇپ رۇشەنگۈلنىڭ ھەيرانلىق نەزىرى بىلەن ماڭا قاراپ تۇرغىنىنى كۆردۈم .
  -- نىمە ، رۇشەنگۈل ، قورۇنۇپ قېلىۋاتامسىز ، تارتىماي ئولتۇرۇڭ . نىمە ئىچىسىز ؟
-- ياق ، ھېچ ئىش يوق ، بىر نەرسە ئىچكىم يوق . رۇشەنەۈلنىڭ كەيپى ئۇچقان ئىدى . نىمە ئۈچۈنلىكىنى بىلمەك قىيىن ئەمەس ، ئۇنىڭ مەن بىلەنلا بىللە امق يىگۈسى بار ئىدى . مىنىڭ ئۈچ كىشلىك ئورۇن زاكاس قىلغىنىمنى كۆرۈپ يەنە بىرىنىڭ دەخلى قىلىدىغىنىنى بىلگەن ئىدى . ئەمەلىيەتتە بۇنى مەن قەستەن ئورۇنلاشتۇرىۋاتاتتىم ، شۇنداق قىلمىسام ئۇنىڭدىن قېچىپ قۇتۇلمىقىم تەس ئىدى . مەكتەپ پۈتتۈرگەندىن بېرى يۈز كۆرۈشۈپ باقمىغان بولساقمۇ دوستلىرىمدىن دائىملا مىنىڭ ئىز دىرىكىمنى قىلىپ تۇردىغانلىقىنى ئاڭلىغان ئىدىم . ئۇنىڭ مەندىن ۋازكەچمىگىنى ئېنىق ئىدى .
             ئىشخانامدىكى ئىشلارنى بىر قۇر بىر تەرەپ قىلغندىن كېيىن رۈشەنگۈلنى ئېلىپ تاماقخانىغا يىتىپ كەلدۇق ، پەرەزىمدىكىدەكلا رەيھان بىزدىن بۇرۇن كېلىپ بولغان بۇلۇپ بىزنى ساقلاپ ئولتۇرغان ئىكەن . ئىككىسى يالغاندىن بولسىمۇ ناھايىتى قىزغىن كۆرۈشتى . تاماق جەريانىدا رۇشەنگۈلنىڭ دادىسىنىڭ خىزمىتى  مۇشۇ شەھەرگە يۆتكۈلۈپ كەلگەنلىكىنى ، شۇڭا رۇشەنگۈللەرمۇ ئائىلە بويۇنچە كۆچۈپ كەلگەنلىكىنى بىلدۇق . ئۇنىڭدىن باشقا پاراڭلىرىمىز مەكتەپتىكى ئىشلار ئۈستىدە بولدى .  تۇيۇقسىز رۇشەنگۈل سوراپ قالدى .
-- مەردان ، سىز تېخى توي قىلمىغانسىز ؟ بىرەر خەۋەر ئاڭلىمىدۇق . بىر ھېساپتا  بۇمۇ ياخشى بولدى . رەيھان ئىككىمىزنىڭ توي قىلماقچى بولىۋاتقىنىمىزنى  دەپ قويۇشمۇ ياخشى ئىش ئىدى . شۇنداق بولغاندا رۇشەنگۈل مىنى  ئاۋارە قىلىپ يۈرمەيتى .
-- ياق تېخى ، بۇ يىل يىل ئاخىردا قىلماقچى . دىدىممەن كۈلۈپ تۇرۇپ .
-- ۋاي ۋۇي قايسى بەخىتلىك قىز ئۇ ؟ رەيھان سەن تونامسەن ئاداش ؟  رەيھان قىزىرىپ باشقا گەپ قىلمىدى .
-- رەيھان بىلەن قىلماقچى . شۇنداق دىدىم دە رەيھاننى ئىشارەت قىلىپ قويدۇم . رۇشەنگۈل چاندۇرماسلىققا شۇنچە تىرىشسىمۇ چىرايىدىن تولىمۇ بى ئارام بولىۋاتقىنى چىقىپ تۇراتتى . تاماق شۇنداق ئاخىر لاشتى . تاماقخانىدىن چىقىپ رۇشەنگۈل چۈتىن كېيىن يەنە ئىشى بارلىقىنى ئېيتىپ كىتىپ قالدى .
-- نىمە ئۇ گەپلەرنى دەپ يۈرۈيسىز ؟ رەيھان بىر ئاز خاپا بولغان ئىدى . كۆرمىدىڭىزمۇ تاتىرىپ بىر بولۇپ كەتتى بىچارە . ئۇنىڭغىمۇ ئاسان ئەمەس دە .
-- سىز شۇنداق دىگەن بىلەن ئەگەر بۈگۈن شۇنداق دېمىسەم بۇندىن كېيىن دائىم ئاۋارە قىلىپ تۇرىۋالسا قانداق قىلىمەن ؟ بىر شەھەردە بولغاندىكىن قاچان كىلىمەن دىسە كىلەلەيدۇ ... ئىشقىلىپ مىنىڭ ئۆزۈمگە ئارتۇق ئىش تېپىۋالغۇم يوق . ئەمدى ياخشى بولدى ، ئىككىمىزنىڭ توي قىلىدىغىنىمىزنى بىلدى . مەندىن ئۆزىنى تارتسا ئەجەپ ئەمەس ، يۈرۈڭ مەن سىزنى ئاپپىرىپ قويۇپ ئاندىن ئىشقا باراي .

              دوختۇرخانىدا ياتقىنىمغا ئون نەچچە كۈن بولدى ، بۇ قېتىم ھەقىقەتەن يىنىك يارلانمىغان ئىدىم . بىر پۇتۇم ، ئىككى قوۋرىغان سۇندى . بېشىم يېرىلدى . مىنى ئۇرغانلارنى ئون كۈن سولاپ جەرىمانە ئېلىپلا قويۇپ بەردى ، بىزمۇ ھېچ نەرسە دىمىدۇق ، دادامنىڭ “مىنىڭ ئوغلۇمغا چېقىلغىنىغا تويغۇزىمەن ...”  دېگىنىگە ئانام ئىككىمىز تەڭ نەسىھەت قىلىپ يۈرۈپ بۇ نىيتىدىن ياندۇردۇق . بەلكىم راستىلا سوقۇلۇپ كەتكەنگە شۇنچە ئاچچىقى كىلىپ ئۇرغاندۇ ، ياكى باشقىلاردىن ئالالمىغان ئۆچىنى مەندىن ئالغاندۇ . نىمىلا بولمىسۇن بولغان ئىش بولدى ، ھەرقانچە قىلساقمۇ ئورنىغا كەلمەيدۇ ، پەقەت ئۇلۇغ اا دىن تىزراق ساقىيىشىمنى تىلەيمەن خالاس .
رەيھان ئىككىمىزنىڭ تويىنى ئامالسىز كەينىگە سۈردۇق . مەن سەللەمازا ساقايماي تۇرۇپ توي قىلغىلى بولمايتى . رەيھانمۇ ھەر كۈنى يېنىمدا پاي پىتەك بۇلۇپ يۈرەتتى . تۈنۈگۈن رۇشەنگۈلمۇ كىلىپ يوقلاپ كەتتى ، رەيھان ئىككىمىزگە بەخىت تىلىدى ، تىزراق ساقىيىشىمنى ، توي ئوينىغىسى بارلىقىنى ئېيتىپ ھەممىمىزگە بىرەر قۇردىن چاقچاق قىلىپ خۇشال قىلدى . رەيھان ئىككىمىزمۇ خۇشال بولدۇق ، ئۇنىڭ كاللىسىدىن ئۆتكەندەك قىلاتتى .


****************************************************************************************************

               بۇرۇن بىلمىگەن بولسام كېرەك ، رەيھانغىمۇ بارا بارا ئىچەكىشىپ قاپتىمەن ، ئۇ يوق بولسا ئۇياق بۇياقتىن ئىز دەيدىغان دەرىجىگە بېرىپتىمەن . ئۇلۇغ اللا نىسىپ قىلسا كىلەر ئاي تويىمىزنى قىلساق بۇ ۋىسالغىمۇ يىتەرمىز ...
ھەر ئىككى ئائىلە توي تەييارلىقى بىلەن ئالدىراش ، توي كۈنىنى بىكىتىپ بولغاچقا باغاقلارنى تارقىتىشقا باشلىدۇق . رۈشەنگۈلگىمۇ بىر پارچە باغاق يېزىشنى ئۇنۇتمىدىم . تويغا پەقەت سەككىز كۈن ۋاقىت قالدى ، رەيھان بىلەن كېلىشىپ ئەتە توي خېتى ئېلىشقا بارىدىغان بولدۇق ، خۇشال ئىدىم . ئوچۇقىنى ئېيتقاندا رەيھاننى ياخشى كۆرۇپ قالغان ئىدىم ، بەلكىم بۇ ياخشى كۆرۈش بىز ئويۇن ئوينىغان ۋاقىتتىلا بىخ سۈرگەن بولىشى مۇمكىن . مەن شۇنى بىلىپ يەتتىم ، بىر كۆڭۇلدە ئىككى ئادەم مەۋجۇت بۇلۇپ تۇرالايدېكەن . يەنى مىنىڭ قەلبىمدە شەمسىيەگە بولغان مۇھەببىتىم يوقاپ كەتكىنى يوق ، رەيھاننىمۇ ياخشى كۆرىمەن .”لېيىغان بۇلاق”رومانىدىمۇ باش پىرسۇناژ ھېكىمەت گۈلزىبانى چىن دىلىدىن سۆيىدۇ ، ھەم ئۇنتۇپ كەتمىگەن شارائىتتا مەھبۇبە بىلەن توي قىلىدىغۇ .  بەلكىم بۇ دەل دېنىمىزدا كۆپ خوتۇنلۇق بۇلۇشقا قارشى چىقماسلىقىنىڭ سەۋەبى بولسا كېرەك ، ئىككىسىگە بولغان مۇھەببىتىم ئوخشىمايدۇ ، ئوخشىمايدىغان ھېسسىيات . بەلكىم بۇ خىل ھېسسىياتنى كونىلارنىڭ “ھەر تاماقنىڭ قورسىقى باشقا “ دېگەن سۆزى بىلەن ئىپادىلىسەك مۇۋاپىق بولار .
بۈگۈن ناھايىتى سەھەر تۇردۇم ، توي خېتى ئالغىلى بارىمەن ئەمەسمۇ ، خۇشالمەن . رەيھانغا تېلفۇن قىلدىم .
-- ئەسسالامۇ - ئەلەيكۇم ، سالامەت تۇرۇپسىزمۇ ؟
-- ۋە ئەلەيكۇم ئەسسالام ، شۇكىرى . نىمانچە بالدۇر تۇرىۋالدىڭىز ؟ سورىدى رەيھان .
-- كىچىكىپ كالماي دەپ بولمامدۇ . سائات سەككىزدە بىزنىڭ ئۆينىڭ ئۇدىلىدىكى بىكەتتە ساقلايمەن ، شۇ يەر يېقىن .
-- ماقۇل ، ھېلىگىچە كۈرۈشەرمىز . ئاللاغا ئامانات .  
كىيىملىرىمنى تۈزەشتۈرۈپ ناشتا قىلدىم ، ناشتىدىن كېيىن ئاتا - ئانامغا دەپ قويۇپ “بىسسىمىللا “دەپ ئۆيدىن چىقتىم .



                                           ئۈستىلەنگەن قەلىب يارىسى     (5)                              

                  بېكەت ئالدىدا خېلى ئۇزۇن ساقلىغان بولساممۇ رەيھان كەلمىدى . يەنە تېلفۇن قىلىشقا بىئەپ ئىدى . سائەت توققۇزدىن ئېشىپ كەتتى يەنىلا يوق . شۇنىڭ بىلەن ئەنسىرەشكە باشلىدىم . رەيھانلارنىڭ ئۆيىگە باردىم ، ئۆيدىن چىقىپ كەتكىلى خېلى بولغانلىقىنى ئېيىتتى . ھەرتەرەپتىن ئىزدەشكە باشلىدۇق . بىر كۈن ئىچىدە بارمىغان يىرىم قالمىدى . بارلىق دوستلىرىمنى ئىشقا سېلىپ ئىزدىدۇق ، ساقچى تەرەپمۇ نۇرغۇن ئادەمنى ئىشقا سېلىپ ئىزدىدى .

               ئىككى كۈنگىچە ھېچقانداق خەۋەرگە ئىرىشەلمىدىم ، ئىزدىمىگەن قوشنا شەھەر ناھيلەرمۇ قالمىدى. تىتىت بولۇۋاتىمەن ، لېكىن قولۇمدىن باشقىسى كەلمەيۋاتاتتى ، قانداق قىلىمەن ؟ رەيھان نەگىمۇ كەتكەندۇ ؟ ئۇلۇغ اللا مىنى يەنە بىر قېتىم ھىجىرانغا قويمىسا بۇلاتتىغۇ ...نىمە گۇناھلارنى قىلغان بولغىيتىم ؟ نىمىشقا ھەممە كەلگۈلۈك مىنىڭ بېشىمغىلا كىلىدىغاندۇ ؟ ......


                رەيھان يوقاپ كىتىپ ئۈچىنچى كۈنى ساقچى تەرەپتىن خەۋەر كەلدى ، بىزنىڭ بېرىپ جەزەملەشتۈرشىمىزنى سوراپ قىلغان تېلفۇن ئىدى . ساخچىخانىغا قاراپ ئۇچتۇم ، كۆڭلۈم بىر خىل ئەنسىز ...
ساقچىخانىغا يىتىپ كەلگىنىمدە رەيھاننىڭ ئاتا- ئانىسى ئىشخانىنى بېشىغا كىيىپ يىلىغىلى تۇرۇپتۇ ، نىمە ئىش ئىكەنلىكىنى ئاڭقىرالماي يۈگۈرۈپ كىردىم . كۆزلىرىم خاتالىشىۋاتامدۇ ...ياق ، خاتالاشمىدى ، قاراڭغۇلىشىۋاتىدۇ ، باشلىرىم قېيىۋاتىدۇ ، ئالدىمدا ياتقان رەيھانغۇ ... ئۇنىڭغا نىمە بولدى ... نىمىشقا كۆزلىرىنى ئاچمايدۇ ...نىمىشقا بىرەر ئېغىز گەپ قىلمايدۇ ... چاچلىرى نىمانچە قالايمىقانلىشىپ كەتكەن ، تارىۋالسا بولماسمىدى ...رەيھان مىنى تاشلاپ كەتتىڭىزمۇ ... شەمسىيەنى دورىمىسىڭىزمۇ بۇلاتتىغۇ ... شەمسىيە مىنى سىزگە تاپشۇرغان ئىدى ، سىز مىنى كىمگە تاپشۇردىڭىز ...شەمسىيە سىزنى ماڭا بەخىت بىرەلەيدۇ دىگەن ئىدىغۇ ... مىنىڭ بەختىم قېنى ... نىمىشقا ماڭا بەرمەيسىز ، نىمىشقا ھەممىڭلار مىنى تاشلاپ بۇ دۇنيادىن كېتىسىلەر ، مەن سىلەرنى شۇنچە بىزار قىلغانمۇ ، كىتىدىغان ئادەم مىنى بىللە ئېلىپ كەتسەڭلار بولماسمىدى ...ئىككى قېتىملىق زەربىگە مىنى بەرداشلىق بىرەلەيدۇ دەپ ئويلىغانسىزمۇ رەيھان ، مىنى يەنە ياشىيالايدۇ دەپ ئويلىغانسىزمۇ ، مىنى نېمىشقا ئويلاپ قويمايسىلەر ...


                  شۇ كۈنى يىقىلغانچە دوختۇرخانىدا بىر ھەپتىگىچە ھوشسىز ياتتىم . ھوشۇمغا كەلگەن ۋاختىمدا رەيھاننىڭ يەتتە نەزىرىمۇ ئۆتۈپ بولغان ئىكەن ،
ساقچى تەرەپنىڭ چۇشەندۈرشىچە رەيھان شۇ كۈنى مىنىڭ قېشىمغا كېلىش يولىدا تۇتقېن قىلىنغان بولىشى مۇمكىن ئىكەن ، جەسەت تەكشۇرۇش نەتىجىسىدىن قارىغاندا رەيھانغا ئېغىر دەرىجىدە باسقۇنچىلىق قىلىنغان ، بىر قانچە ئامما ئەھۋال مەلۇم قىلىپ كىيىملىرى يىرتىلغان بىر قىزنىڭ شەھرىمىزنى كېسىپ ئۆتىدىغان دەريا تەرەپكە كېتىۋاتقانلىقىنى كۆرگەنلىكىنى ئېيتقان . بىلىشكە بولۇدۇكى رەيھان بۇلغانغان بەدىنى بىلەن مىنىڭ قېشىمغا كېلىشكە ۋىجدانى يول قويمىغان ، ئاقىۋەت مۇشۇ ئۇسۇل بىلەن ئۆزىنىڭ ساداقىتىنى بىلدۇرۇپ بۇ دۇنيادىن خوشلاشقان .
              ئويلىغانسىرى ئىچىم ئېچىشىۋاتاتتى ... بارغانسىرى تۈگۈشۈپ كىتىۋاتىمەن ... بىر نەرسە يىگۈممۇ كەلمەيدۇ ، نىمىشقا بۇنداق ئىشلار مىنىڭ بېشىمغىلا كىلىدىغاندۇ ...قىسقىغىنە ھاياتىمدا مەن زادى نىمە گۈناھ قىلىپ ئۆتكەن بولغىيدىم ، مىنى ياخشى كۆرگەن مىنىڭ مۇھەببىتىمگە ئىرىشەلمىگەنلەرنىڭ ئاھى تۇتىۋاتامدۇ ، بۇنىڭدا مەن ئەيىپلىكمۇ ، نىمىشقا مىنى ئېلىپ كەتمەي مەن سۆيگەنلەرنى ئېلىپ كىتىدۇ، مىنىڭ پىشانەمگە ھىجىراندىن باشقىسى پۈتۈلمىگەنمۇ ....

          دوختۇرخانىدا شۇ ياتقانچە ئىككى ئاي ياتتىم ، زىيادە ئۆمۈدسىزلىنىش،روھى زەربە  كەلتۈرۈپ چىقارغان ئاجىزلىق مىنىمۇ ئۇ ئالەمگە يوللىغىلى ئاز قالغان ئىدى.  اللانىڭ شىپاسى دوختۇرلارنىڭ تىرىشچانلىقى بىلەن گۆر ئاغزىدىن قايتىپ كەلدىم . بۇ ئىككى ئايدا روشەنگۈل مىنى يېنىمدىن ئايرىلماي كۈتتى . شەھەر باشلىقىمۇ كېلىپ يوقلاپ كەتتى . روشەنگۈلنىڭ جاپاسىنىمۇ كۆرۈپ ياتتىم . مەن ئاجىزلاپ كەتكەن كۈنلەردە كىچە كىچىلەپ يىغلاپ چىقاتتى ،..  ئۆزۈم سۆيگەنلەردىن ئايرىلدىم ، مىنىڭ قايتا بىرىنى ياخشى كۆرشۈم مۇمكىن ئەمەس ، ئەمدى بۇ دۇنيادىن بويتاق ئۆتۇپ كېتىمەن ...

                 بىر يىل ئۆتۈپ كەتتى ، ھەر كۈنى روشەنگۈلنىڭ يالۋۇرۇشلىرىدىن تولىمۇ زىرىكتىم . قىز بالىغا نىمانچە بەرگەن بولغىيدى بۇ ياشنى ، كۆزىدىن يىپى ئۈزۈلگەن مارجاندەك تۆكىلىۋىردىكەن . ئاتا - ئانامنىڭ زورى بىلەن روشەنگۈل بىلەن توي قىلىدىغان بولدۇم . توي قىلماي دىسەم ئاتا - ئانامغا يۈز كىلەلمەيدىكەنمەن ، توي قىلاي دىسەم ئەمدى بىرسىنى ياخشى كۆرۈپ كۆڭلۈمدىن ئورۇن بىرىشىم مۇمكىن ئەمەس . شۇنداق بولغانلىقى ئۈچۈن بىر تەرەپتىن ئاتا ئانامنى رازى قىلىش ، يەنە بىر تەرەپتىن رۇشەنگۈلنى مۇرادىغا يەتكۈزۈشنى ئويلۇدۇم . ئۇمۇ مىنى دەپ كۆپ جاپا چەكتى ئەمەسمۇ ، ئۇنىڭغىمۇ ئاسان ئەمەس ، مىنى ساقلاپ قېرى قىز بۇلۇپ قالدى ھېلىمۇ .

                 توي باشلاندى ، تەبىرىك سۆلەر قۇلاق تۈۋىمدە ئۈزۈلمەستىن ياڭرىماقتا ، رۇشەنگۈل ئىككىمىزگە بەخىتلەرنى تىلەشمەكتە . ئاتا- ئانىلار تولىمۇ خۇشال ، بالىلىرى كۆڭۈلدىكىدەك لايىق بىلەن توي قىلغان تۇرسا ...داقا-دۇمباقلار ساداسى ئىچىدە توي ئاخىرلاشتى . رۇشەنگۈل بىلەن ئىككىمىزنى ھوجىرىمىزغا “سولىۋەتكەندىن”كېيىن قىقاس چوقانلار بىلەن قايتىشتى . مانا بۇ مىنىڭ توي كېچەم ، ئۈچ قېتىم تويغا تەييارلىق قىلىپ ئۈچۈنچى قېتىم ئۆزۈم قىلچىمۇ سۆيمەيدىغان بىرى بىلەن يەتكەن ۋىسالغۇ بۇ . قەلبىمدە رۇشەنگۈلگە بولغان ھىسدالىقتىن باشقا يەنە نىمە بار ، توغرا يەنە ئۇنى ھۆرمەت قىلىمەن .

        --مەردان ، مەن ئاخىرى سىزگە ئىرىشتىم ،سىزگە مەنسۈپ بولدۇم ، سىزنىڭ ئايالىڭىز بولدۇم ، مەن تولىمۇ بەختىلىك ...رۇشەنگۈل ئاستا پىچىرلىدى .
-- نىمە قۇلىقۇمنىڭ تۈۋىگە كىلىپ گەپ قىلىسىز ، مىنى ئويغىتىۋەتتىڭىز ، تاڭ ئاتمىغاندۇ ؟ ئۇيقۇچىلىقتا سورۇدۇم .
-- مەردان سىزگە دەيدىغان گەپلىرىم بار ئىدى . بىلمىدىم ئاڭلاشنى خالامسىزكىن ؟
-- قېنى دەڭ .دېدىم ئۇنىڭغا قاراپ .  قىزىقىپ قېلىۋاتاتتىم ، زادى نىمە ئىشتۇ ؟
-- ئۇنداقتا مىنى كەچۈرەلەمسىز ؟ كەچۈرىمەن دىسىڭىز ئېيتىپ بىرىمەن .
-- شۇنداق مۇھىم ئىشمىدى ، قېنى ئاۋال دەڭ . كەرڭ قورساق ئىكەنلىكىمنى بىلىسىزغۇ ؟
كەچۈرۈڭ مەردان ، سىزنى ئۆلگۈدەك ياخشى كۆرىمەن ، دەل شۇنداق بولغانلىقى ئۈچۈن بەزى بىر سىزنى رەنجىتىدىغان ئىشلارنى قىلىپ قويدۇم ، مۇھەببەت ئادەمنى شۇنچىلىك شەخسىيەتچىگە ئايلاندۇرۇپ تاشلايدىكەن ، بەزىدە ئۆزۈم ئۆزۈمنى تونىيالماي قالىمەن . مەردان ، سىز شەمسىيە بىلەن بەخىتلىك بولغان بولسىڭىز مەنمۇ ئۇنداق ئىشلارنى قىلىپ يۈرمىگەن ، ھازىرقىدەك مەڭگۈلۈك پۇشايماندا قالمىغان بۇلار ئىدىم .
-- مەڭگۈلۈك پۇشايمان ؟ زادى نىمە ئىش تىزراق دىمەمسىز . ئالدىرىدىم ، ئۇيقۇم پۈتۈنلەي ئېچىلىپ كەتتى .
-- شۇنداق ، مەڭگۈلۈك پۇشايمان ، مەردان ، شەمسىيە قازا قىلغاندىن كېيىن سىز بىر يىلغىچە يۇرتىڭىزغا بىكىنىۋېلىپ يۈردىڭىز ، قانچە قېتىم كېلىپ سىزنى پەقەت يىراقتى كۆرۈپلا قايتىپ كەتتىم ، كۆڭلۈمدە سىزگە ۋاقىت بىرىپ ئاندىن سۆيگۈ تەلەپ قىلاي دىدىم . لېكىن كىم بىلسۇن مەن ئاللا بۇرۇن كېچىكىپ قاپتىمەن ، سىز رەيھان بىلەن تويغا كېلىشىپ بولۇپسىز . قاتتىق ئازابلاندىم ، ئۆزۈمنى ئۇرۇپ يىغلىدىم ، بىراق بۇنداق قىلسام ئورنىغا كەلمەيدۇ دە . بىرسىگە پۇل بېرىپ رەيھانغا تەھدىت سالدىم ، ئۇ ئاڭلىمىدى ، ئادەم ياللاپ سىزنى ئۇرغۇزدۇم . لېكىن مەن پەقەت قورقۇتۇپ قويۇڭلار دېگەن ئىدىم ، ئاقىۋەت ... ئۇنداق قاتتىق ئۇرىۋىتىشىنى مەنمۇ بىلمەپتىمەن ، كەچۈرۈڭ .
              يۈرۈگۈمنى بىرەرمىڭ قۇرۇت يەۋاتقاندەك بۇلۇپ كەتتىم ، مىنىڭ شۇنچە ئۇزۇم دوختۇرخانىدا يېتىشىمغا قوينۇمدا ياتقان ئايالىم سەۋەبچى ؟ يەنە كېلىپ مەن شۇنچىلىك ياۋاش دەپ ئويلايدىغان روشەنگۈل ؟  ئىنكار قىلىشقا ئامالسىزمەن ، مانا ئۇ ئۆز ئاغزى بىلەن دەۋاتمامدۇ ، ئۇلۇغ اللا ...
-- تېخى تۈگىمىدى مەردان ، شۇ ئىشتىن كېيىن سىز دوختۇرخانىدا خېلى ئۇزۇن ياتتىڭز ، بىر ھېسابتا تولۇق غەلبە قىلمىغان بولساممۇ رەيھان ئىككىڭلارنىڭ تويىنى خېلىلا كەينىگە سۈردۈم . شۇندىلا ماڭا پۇرسەت بار ئەمەسمۇ . ئەمدى ئۇنىڭ مۈرەمگە قويۇۋالغان بېشىنى ئىتتىرۋىتىپ كارۋاتتا ئولتۇردۇم . ئاڭلاپ باقاي ئۇنىڭدىن كېيىن نىمە ئىشلارنى قىلدىكىن .
-- سىلەرنىڭ توي خەۋىرىڭلار كەلدى ، ماڭىمۇ بىر پارچا باغاق يېزىشنى ئۇنۇتمىدىڭىز . سىلەرنىڭ تويۇڭلار مەن ئۈچۈن ماتەم ئىدى ، شۇڭا باغاقنى كۆرۈپ خۇددى بىرى ماڭا مىنىڭ ئۆلۈش كۈنۈمنى خەۋەر قىلىپ كەلگەندەك بىلىندى ، نىمىسنى دەي ، قانچە يىللاپ ئوتىدا كۆيۈپ چەكمىگەن جاپا ، تارتمىغان ئازابىم قالمىغان بىرىنىڭ تويىغا قاراپ تۇرىمەنمۇ ؟ مەندە شۇنداق غەيرەت بارمۇ ؟ بۇ رىئاللىقنى قوبۇل قىلالامدىم ؟ ياق ، مەردان ، مەن قوبۇل قىلىشقا ئاجىزلىق قىلدىم . رۇشەننىڭ كۆزلىرىدىن غەزەپ ، ئۆچمەنلىك ئۇچقۇنلىرى چاقنىدى ،
-- شۇنىڭ ئۈچۈن رەيھاننى قەستلىدىڭمۇ بۇزۇق ؟ قاتتىق ئاچچىقىم كەلدى ، ۋارقىردىم .
-- ئۆزىڭىزنى بېسىۋېلىڭ مەردان ، مەن ھازىرلا ھەممىنى ئېيتىپ بىرىمەن .  شۇنداق مەن ئىككىڭلار توي خېتى ئالغىلى بارماقچى بولغىنىڭلارنى بىلگەن كۈن بىر كېچە ئۇخلىماي ئامال ئويلۇدۇم ، ئەتىسى سەھەردە مەن ياللىغان لۈكچەكلەر رەيھاننى تۇتقۇن قىلىپ شەھەر سىرتىدىكى بىر كونا ئامبارغا سولاپ باسقۇنچىلىق قىلدى ، بۇنى مەن ئورۇنلاشتۇرغان ئىدىم . خېيالىمدا شۇنداق بولسا رەيھان سىزنى ياخشى كۆرىدىغان بولغاندېكىن بۇ تويدىن ۋاز كېچىدۇ . بەك بولسا باشقا شەھەرگە بېرىپ يېڭى تۇرمۇشىنى باشلايدۇ دەپ ئويلۇدۇم . لېكىن رەيھاننىڭ ئۆزىنى دەرياغا تاشلاپ ئۆلىۋېلىشىنى ئويلىماپتىمەن . شۇ ئىشتىن كېيىن مەنمۇ قاتتىق پۇشايمان قىلدىم ...
