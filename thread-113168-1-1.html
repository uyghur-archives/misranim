بىر بالا كۆزىنى ئوپراتسىيە قىلدۇرغاندىن كېيىن كۆزىگە ئاپپئاق داكا تېڭىلغان ھالدا يورۇقلۇققا تەشنا بولۇپتۇ. بىر كۈنى، بۇ بالا دوختۇرخانىنىڭ ئارقا ھويلىسىغا چىقىپ ئولتۇرغىنىدا، بىر تال يوپۇرماق بېشىغا چۈشۈپتۇ. بالا يوپۇرماقنى ئېلىپ ئۆز ئۆزىگە:«بۇ ئاق تېرەكنىڭ يوپۇرمىقىدىمۇ ياكى...»دېيىشىگە،«شۇنداق...»دېگەن ئاۋاز ئاڭلىنىپتۇ. بالا بۇ كىشىگە كۆزىنىڭ كىچىكىدىن تارتىپ، ياخشى كۆرمەيدىغانلىقىنى ئېيتىپ بېرىپتۇ. بالا بۇ كىشىدىن:«دۇنيا گۈزەلمۇ ؟» دەپ سورىغىنىدا بۇ كىشى:«ئاسمان كۆپكۆك، بۇلۇتلار شۇنچىلىك ئاق،يىراقتىن تاغ ئۈستىدىكى قارىغايلارنىڭ قەد كۆتۈرۈپ تۇرغانلىقىنى كۆرىۋاتىمەن.ئۇدۇلىمىزدا سۈپسۈزۈك سۇلار ئويناقشىپ ئېقىۋاتىدۇ.»دەپ جاۋاب بېرىپتۇ.بالا بۇ كىشىدىن:«كۆزۈم كۆرەلەرمۇ؟»دەپ سورىغىنىدا بۇ كىشى:« چوقۇم كۆرەلەيدۇ، دوختۇرغا ياخشى ماسلىشىپ بەرسەڭلا، ھەممە ئىشلار ئوڭۇشلۇق بولىدۇ»دەپتۇ. بۇنى ئاڭلىغان بالىنىڭ قىەلبىدىكى ئەندىشە پەسىيىپتۇ،شۇنىڭدىن كېيىن، بۇ ئىككىيلەن دائىم بۇ جايدا مۇڭدىشىدىغان بوپتۇ.بىر مەزگىلدىن كېيىن، دوختۇرلار بالىنىڭ كۆزىدىكى داكىنى ئېلىۋېتىپتۇ. بالا ئاخرى يورۇقلۇقنى كۆرۈپتۇ. بالا يۈگرىگىنىچە دوختۇرخانىنىڭ ئارقا ھويلىسىغا _ھېلىقى كىشى بىلەن مۇڭداشقان جايغا بېرىپتۇ ۋە داڭ قېتىپ تۇرۇپلا قاپتۇ. ئەسلىدە ، بۇ جايدا ھېلىقى كىشى ئېيتقاندەك سۈزۈك سۇ، يېشىل دەرەخ ، تاغمۇ يوق بولۇپ، پەقەت پاسىل تام ۋە قېرى دەرەخلا بار ئىكەن . كۈزنىڭ سوغۇق شامىلىدا، دەرەخ ئاستىدا قارا كۆزەينەك تاقىۋالغان ياشانغان بىر بوۋاي ئولتۇرغان، يېنىغا ئەمالار ھاسسىسى قويۇلغان. بالا مۇشۇ چاغدىلا ئۆزى بىلەن مۇڭداشقان بۇ كىشىنىڭ ئەما ئىكەنلىكىنى بىلىپتۇ. شۇنىڭدىن كېيىن كىشىلەر دائىم دوختۇرخانا ھويلىسىدا بىر بالىنىڭ .ئەمە بوۋاينىڭ قولىنى تۇتقىنىچە ئايلىنىۋاتقانلىقىنى ئۇچرىتىدىغان بوپتۇ.بۇ مەزمۇنلار پۈتۈنلەي مىسرنىم مۇنبىرىدىن كۆچۈرۈلگەن @+ c1 z+ ?6 w, I) P. k
