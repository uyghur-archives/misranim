                                         نەسىرلەر
                                     1
      ئادەم بىر قۇشتۇر،بىلىمى قانات.ئالەمنى قوياشتەك ئايلىناي دىسەڭ ئەقىل-ئىدراكتىن چىقارغىن قوش-قانات.تەسەۋۋۇر قىلىپ يىراقلارغا سەييارى ئۇچالىساڭ مانا بۇ ئەڭ مەنىلىك- قەدىرلىك ھايات.
                                                            2
      مەن قېرىغاندا،ھاياتىمنىڭ ئۈچ باسقۇچىنى ئەسلەپ قالدىم،بالىلىق-يۇلتۇزغا،ياشلىق- قوياشقا،قېرىلىق-ئايغا ئوخشايدىكەن.ھەي ياشلىق!سەن نىمەدىگەن قۇدرەتلىكسەن.ھەي ياشلىق مەن سىنىڭ ئەتراپىڭنى ئايلىناي، ماڭا يوروقلۇق ئاتا قىلغىن!

                                                  3   
       ھايات خۇددى دولقۇن ياساپ ئېقىۋاتقان سۇ.مەن ئەۋج ئۇرۇپ ئېقشنى ياخشى كۆرۈمەن.
كۆزلىرىم بۇلاق،تومورلۇرۇم دەريا،زىمىننى بولسىكەن دەيمەن قولۋاق.

                                                 4
      سەن شاھلىق تەختىڭنى بەختىڭدىن ئىزدىسەڭ،مەن ھاياتلىق بەختىمنى ئەقلىمدىن ئىزدەيمەن.
                                                   5
     مەن تاققا ياماشتىم،قورام تاش يولومنى توسوپ تۇرۇپتۇ.قىسمىتىم ئوچوقلا قول شىلتاۋاتىدۇ،مەن يولومدىن چىكىنمەستىن بەلكىم قورام تاشنى پەللەمپەي قىلىپ داۋاملىق ئالغا،گۈزەل مەنزىلگە ئىلگىرلىدىم،قىسمىتىم بېشى ئىگىك ھالدا مەندىن ئۇزاپ قېلىۋەردى.
                                                    6
      مەن ساپ قان،تومورۇم گوياكى دەريا،يۈرىكىم بەئەينى ئوكيان،دەپنە قىلىنسام ئۆتمۈشكە،
چاچرايدۇ قەترەم ھەر يان.
                                                    7
       قەلپ قەسىرەم سىرلىق قەلئە،ئۆتكۈر قەلىمىم جاسارەتلىك خەيمان،بۇ سىرلىق قەلئەگە يوشۇرۇنغان بايلىقلارنى،يوشۇرۇن سىرلارنى ئېچىپ بىرەي دەيمەن.
                                                  8
       پۈتۈن زىمىن ئىز بىلەن تولۇپ كەتكچكە بەزىلەر ئىز قالدۇرۇشقا ئورۇن ئىزدەش بىلەن ئاۋارە،بەزىلەر قالدۇرۇلغان ئىزنى يوق قىلىش بىلەن ئاۋارە،بەزىلەرنىڭ باسقان ئىزىدىن گۈلشەن پەيدا بولسا،بەزىلەرنىڭ ئىزىدىن ئازغان،شۇمبۇيا پەيدا بولىدۇ.شۇڭا زىمىن تارىخنىڭ قىممەتلىك خەزىنىسى،ھەقىقى گۇۋاھچىسى ھەم سۆزلىگۈچىسىدۇر.
                                                 9
       چۈشۈم دېدى:مەن ئۆزۈمنى،ئۆزۇمگە ئايان بولغان قەلىبلەردىن ئىزلەيمەن.
        مەن دىدىم: مەن ئۆزۈمنى چۈشلىرىمدىن ئزلەيمەن.
        ئۇ دېدى:مەن ئۆزۈمنى ھەقىقى ئويغاق قەلىبلەردىن ئىزلەيمەن!
         -بارىكاللا! مەن ئۇنىڭغا بارىكاللا ئوقۇدۇم،رەھمەت ئېيتىم.
                                               10
         مەن چۈش كۆرۈۋاتىمەن،يازنىڭ تومۇز كۈنلىرى،جىسمىم ئوتقاش،بىر تال زەيتۇن ياپرىقى يۈزۈمنى سىپاپ ئۆتۈپ بېشىمغا سايە بولغىدەك.
        ئويغانسام قەدىرلىكىم يېنىمدا ئولتۇرۇپ قولىدىكى زەيتۇن ياپرىقى بىلەن مىنى تاماشا قىلىۋاتقان ئىكەن.
        مەن يەنە چۈش كۆرۈۋاتىمەن.مەن شىلدىرلاپ ئېقىۋاتقان زىلال سۇ،دولقۇنلىرىمنى ئارامدا قالدۇرۇپ ئۇزاپ كىتىۋاتىمەن.مەنزىلگاھىمغا يىتىپ بارغىنىمدا ئاخىرى تۇپراققا سىڭىپ كەتكۈدەكمەن،قەدىرلىكىم ھېلىقى زەيتۇن يوپۇرمىقىمنى مەن سىڭگەن تۇپراققا ئابىدە سۈپىتىدە تىكلەۋاتقۇدەك.
                                                 11
         -ئۆزۈڭنى دەقىقە ئىچىدەچاققان چاقماق ئىچىدىن ئىزلە.بۇ شىددەتلىك گۈلدۈرماما مىنىڭ ئاۋازىم.
     مەن چۆچۇپ ئويغانسام چۈشۈم شۇنداق دەپ قويۇپلا غايىپ بوپتۇ.
                                                 12
       بىر كۈنى چۈشۈم مىنى قاتتىق ھاقارەتلىدى.
       -سىنىڭ سەزگۈڭ تۇغۇلۇشىڭدىن بۇرۇنلا ئۆلگەن،چۈنكى سەن غەپلەت ئۇيقۇسىغا ئاشنا.
      شۇندىن كىيىن كىرپىكىمنى يا،كۆز نۇرۇمنى ئوق قىلىپ چۈشلىرىمنى يېقىن كەلتۈرمىدىم.

                                          مەنبە؛ ئۆزەمنىڭ ئىجادىيتى.

يېقىندىن بىرى ۋاسكىتبول ئويناش بىلەن بولوپ كىتىپ نەسىرى ئەسەرلەرنى يوللىيالماي قالدىم.
