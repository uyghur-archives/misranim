    ياۋرۇپادىكى بىر تەجىربىخانىدا ئېلىپ بېرىلغان بىر تەتقىقاتقا قارىغاندا زەمزەم سۈيىدىكى
    گۇڭگۈرت مىقتارى باشقا سۇلارنىڭ تەركىبىدىكى كۆڭگۈرتىن ئازكەن. يەنە بىر تەتقىقات نەتىىجىسىگە قارىغاندا زەمزەم سۈيى باشقا سۇلارغا قارىغاندا تېىخمۇ بەكرەك قۇۋۋەت بېرىدىغان مېنىرال ماددىلار باركەن. زەمزەم سۈيىنىڭ مەنبەسى تېخىچە تېپىلمىغان. بۇ سۇنىڭ نەدىن كەلگەنلىكى، سۇنىڭ مەنبەسىنى قېدىرىپ تەكشۈرۈش نەتىجىلىرىدىن قارىغاندا ھېچمۇ- ھېچ مەلۇم بولمىغان. زەمزەم قۇدىقىغا يېقىن بولغان رايۇنلاردا بىرەر قۇدۇقمۇ يوق. دېڭىز بولسا بۇ يەردىن 80 كىلومېتىر يېراقلىقتا. بۇ ئەھۋالدا زەمزەم سۈيىنىڭ دېڭىزدىن ۋە ياكى باشقا بىر يەردىن مەنبە ئېلىشى ئىمكانسىز. بۇ ھەقتە تەتقىقات ئېلىپ بارغانلار شۇنىڭغا ھەيران قالماقتىكى قانداق قىلىپ بۇ سۇ نەچچە مىڭ يىلدىن بېرىتۈگەپ قالماستىن ئاقماقتا. زەمزەم سۈيىنىڭ يەنە بىر مۆجىزىسى شۇكى، ئۇنى ئاچلىقىنى بېسىش ئۈچۈن ئىچسە ئاچلىقىنى كەتكۈزىدىغان، ئۇسۇزلۇقىنى كەتكۈزۈش ئۈچۈن ئىچسە ئۇسۇزلۇقىنى كەتكۈزىدىغانلىقىدۇر. پەقەتلە بىر يېرىم مېتىر چوڭقۇرلۇقتىن چىقىدىغان سۇ، ھەج مەۋسۈمى جەريانىدا نەچچە مىليون ئىنساننىڭ سۇ ئىھتىياجىنى پۈتۈنلەي قاندۇرۇشقا يېتىدۇ. دۇنيا ساغلىق تەشكىلاتىنىڭ دوكلاتلىرىغا قارىغاندا زەمزەم سۈيى دۇنيادىكى ئىچىلىدىغان سۇلار ئىچىدىكى ئەڭ ساغلام بولغان پاكىزە سۇدۇر. ئامېرىكىدا ئېلىپ بېرىلغان بىر تەتقىقاتتا بايان قىلىنىشىچە زەمزەم سۈيى ئىچىدە مىكروپ بولمىغان بىردىن- بىر سۇدۇر. يەنە بىر تەرەپتىن ئېچىلگەن زەمزەم سۈيى ھېچقاچان بەلدىن ئاستى تەرىپىگە چۈشمەيدىغان بولۇپ، ھەرگىزمۇ كىچىك تەرەت يولى ئارقىلىق چىقىرىلماستىن بەلكى بەدەندىن تەر يولى بىلەن چىقىرىلىدىغانلىقى ئىسپاتلانغان.
