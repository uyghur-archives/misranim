ـ ئەسسالامۇئەلەيكۇم ـ ئاسمان ـ زېمىن ئەبۇزەردىنمۇ سەمىمىي ئادەمنى كۆرۈپ باقىمىدى.  

ـ مۇھەممەد ئەلەيھىسسالام.


تاشقى جەھەتتىن مەككە بىلەن تۇتىشىپ تۇرىدىغان ”ۋەددانە“ جىلغىسىدا ”غىفار“ قەبىلىسى ياشايتتى.

”غىفارلار“ قۇرەيشلەرنىڭ تىجارىتى بىلەن مېڭىۋاتقان كارۋانلارنىڭ شامغا بېرىپ ـ كېلىشتە بەرگەن ئازغىنا نەرسىلەر بىلەن تۇرمۇشىنى قامدايتتى، ئەگەر كارۋانلار رازى قىلغۇدەك نەرسە بەرمىسە،ئۇلار كارۋانلارنىڭ يولىنى تۇساتتى.

ئەبۇزەر دەپ تەخەللۇسلانغان ”جۇندۇبۇبۇ ئىبنى جۇنادە“ بۇ قەبىلە پەرزەنتلىرىدىن بىرى ئىدى. ئۇ ئەقلىنىڭ ئۆتكۈرلىكى، يىراقنى كۆرەرلىكى بىلەن باشقىلاردىن ئالاھىدە پەرقلىنىپ تۇراتتى.

ئۇ قەۋمنىڭ، ئاللاھ تائالانى قويۇپ چوقۇنىۋاتقان، بۇتلىرىغا بەكمۇ ئىچى سىرىلاتتى. ئۇنىڭ ئىددىيىسى، كۆز قارىشى، ئەرەبلەرنىڭ بوزۇق ئېتىقادى ۋە ھاماقەتلىكىگە قارشى ئىدى.

ئۇ كىشىلەرنىڭ ئەقىل ۋە دىللىرىنى ئويغۇتىپ ئۇلارنى قاراڭغۇلۇقتىن يورۇقلۇققا چىقىرىدىغان پەيغەمبەرنىڭ ئاشكارىلىنىشىنى تۆت كۆزى بىلەن كۈتەتتى.

ئەبۇزەر ئۆز يېزىسىدىكى ۋاقىتىدا، مەككىدە ئاشكارا بولغان پەيغەمبەرنىڭ خەۋەرلىرىنى ئاڭلىدى ۋە ئىنىسى ئەنەسكە: سەن مەككىگە بېرىپ، ئۆزىنى پەيغەمبەر ۋە ئاسماندىن ماڭا ۋەھيى كېلىدۇ، دېگەن بۇ ئادەمنىڭ خەۋەرلىرىنى ئىگەللىگىن ۋە ئۇنىڭدىن ئاڭلىغانلىرىڭنى ماڭا يەتكۈزگىن، دېدى.

     ئەنەس مەككىگە بېرىپ، پەيغەمبەر ئەلەيھىسسالام بىلەن ئۇچراشتى، ئۇنىڭ سۆزلىرىدىن ئاڭلاپ قايتىپ كەلدى، ئەبۇزەر ئۇنى تەشنالىق بىلەن قارشى ئېلىپ يېڭى پەيغەمبەرنىڭ ئەھۋاللىرىنى ئىشتىياق بىلەن سورىدى.ئىنىسى: ئاللاھ بىلەن قەسەمكى! گۈزەل ئەخلاققا چاقىرىدىغان بىر ئادەمنى كۆردۈم. ئۇنىڭ سۆزىلىرى شېئىرغىمۇ ئوخشىمايدىغان، ناھايىتى بىر خىل جەلب قىلارلىق ئىكەن، دېدى.

ـ كىشىلەر ئۇ توغرىسىدا نېمە دەيدىكەن؟

ئۇنى: ساھىر، شائىر، كاھىن، دەيدىكەن.

ئاللاھ بىلەن قەسەمكى، مېنىڭ تەشنالىقىمنى قانائەتلەندۈرەلمىدىڭ، مەن بېرىپ ئۇنىڭ ئىشلىرىنى ئۆزەم كۆرۈپ كەلگۈچە ئائىلەمدىكىلەرنىڭ ھالىدىن خەۋەر ئېلىپ قويالامسەن؟

بولىدۇ، لېكىن سەن مەككىلىكلەردىن ئېھتىيات قىل.

ئەبۇزەر سەپەر تەييارلىقىنى پۇختا قىلىپ پەيغەمبەر بىلەن كۆرۈشۈش ،ئۆزى بىۋاستە ئەھۋال ئىگەللەش ئۈچۈن ئەتىسىلا يولغا چىقتى.

ئەبۇزەر مەككىگە كەلدى، ئۇ بۇ يەردىكىلەردىن يامانلىق يېتىپ قىلىشىدىن ئەنسىرەپ، ھۇشيار تۇردى.

قۇرەيشلەرنىڭ ئۆز ئىلاھلىرى ئۈچۈن مۇسۇلمانلارغا غەزەپلىنىشى ۋە ئۆزىنىڭ مۇھەممەدكە ئەگەشكەنلىكىنى ئېيتقان بارلىق كىشىلەرنى ئازابلىغانلىق خەۋىرىنى ئەبۇزەر ئاڭلىغان ئىدى.

شۇڭلاشقا ئۇ مۇھەممەد توغرۇلۇق بېرەر كىشىدىن سورىمىدى، چۈنكى سۇرالغۇچى ئۇنىڭ دوستىمۇ ياكى دۈشمىنىمۇ ئۇ بۇنى بىلمەيتتى.

كەچ كىرگەندە ئۇ مەسچىتكە بېرىپ ياتتى، ئۇ يەردىن ئەلى ئىبنى ئەبۇتالىب ئۆتۈپ، بۇ كىشىنىڭ بۇ يەرلىك ئەمەسلىكىنى بىلدى  ـ دە، ئۇنىڭغا:

ھەي ئادەم! بىزنىڭ قېشىمىزغا كەل، دېدى ئەبۇزەر ئۇنىڭ بىلەن ماڭدى ۋە ئۇنىڭ ئۆيىدە قونۇپ قالدى، ئەتىسى ئۇ ئوزۇق خالتىسىنى ۋە تولۇمنى ئېلىپ بىر ـ بىرىدىن گەپ سوراشمايلا، مەسىتچكە قايتتى. ئۇ ئىككىنچى كۈنىسىمۇ پەيغەمبەر ئەلەيھىسسالامنى تونۇماي ئۆتكەزدى، كەچتە مەسچىتكە بېرىپ ئۆز ئورنىدا ياتتى، ئەلى ئۇنىڭ قېشىدىن ئۆتۈپ: قۇنالغۇغا قايتىدىغان ۋاقىت بولمىدىمۇ؟ دېدى. ئەبۇزەر ئۇنىڭ بىلەن ئىككىنچى كېچىنىمۇ ئۆتكۈزى. ئۈچىنچى كېچىسى بولغاندا ئەلى:   ماڭا مەككىگە كېلىش سەۋەبىڭنى ئېيتىپ بەرمەمسەن؟ دېدى.

مەن تەلەپ قىلغان ئىشقا ياردەم بەرسەڭ ۋە شۇنداق قىلىدىغانلىقىڭغا ۋەدە بەسەڭ دەپ بېرەي، دېدى ئەبۇ زەر. ئەلى ۋەدە بەردى.

مەن يىراق جايدىن يېڭى پەيغەمبەر بىلەن كۆرۈشۈپ، ئۇنىڭ سۆزلىرىنى ئاڭلاپ بېقىش ئۈچۈن كەلدىم. دېگەندە ئەلىينىڭ چىرايى ئېچىلىپ: ئاللاھ بىلەن قەسەمكى، ئەلۋەتتە ئۇ ھەق پەيغەمبەر...دېدى. ۋەتاڭ يۇرىغاندا، مەن بىلەن ماڭ، ئەگەر سەندىن ئەنسىرگىدەك بىر ئەھۋال بولسا تەرەت قىلغان بولۇپ تۇرىۋالىمەن، ئەگەر ماڭسام مەن كىرگەن ئۆيگىچە ماڭا ئەگىشىپ ماڭغىن، دېدى.

ئەبۇزەر پەيغەمبەر ئەلەيھىسسالامنى كۆرۈشكە تەشنا بولۇپ، پۈتۈن كېچە ئورنىدا ياتالمىدى.

ئەتىگىنى ئۇلار ماڭدى، ئۇ ھېچنەرسىگە قارىماي، ئەلىنىڭ كەينىدىن ئەگەشتى، ئۇلار ئاخىرى پەيغەمبەر ئەلەيھىسسالام قېشىغا كىردى. ئەبۇزەر:

ئەسسالامۇ ئەلەيكۇم! ئى ئاللاھ نىڭ پەيغەمبەرى دېدى.

«ۋەئەلەيكۇم ئەسسالام ۋەرەھمەتۇللاھى ۋەبەرەكاتۇھ»دەپ جاۋاپ قايتۇردى پەيغەمبەر ئەلەيھىسالام .

ئەبۇزەر پەيغەمبەر ئەلەيھىسسالامغا تۇنجى قېتىم ئىسلامىي سالام قىلغۇچى بولدى، شۇنىڭدىن كېيىن شۇنداق سالاملىشىش مۇسۇلمانلار ئارىسىدا ئومۇملاشتى.

پەيغەمبەر ئەلەيھىسسالام ئەبۇزەرنىڭ قېشىغا كېلىپ، ئۇنى ئىسلامغا دەۋەت قىلىپ، قۇرئان تىلاۋەت قىلىپ بەردى. ئۇزاق ئۆتمەي ئۇ ھەق سۆزىنى ئىلان قىلىپ، يېڭى دىنغا كىردى، ئۇ ئىسلام كەلتۈرگەنلەرنىڭ تۆتىنچى ياكى بەشىنچىسى بولدى.

خەۋىرىنىڭ قالغىنىنى بايان قىلىشنى ئەبۇزەرگە قالدۇرايلى.

ئەبۇزەر مۇنداق دەيدۇ: شۇنىڭدىن كېيىن پەيغەمبەر ئەلەيھىسسالام بىلەن بىرگە مەككىدە تۇردۇم، ئۇ ماڭا ئىسلامنى ئۆگەتتى، قۇرئان ئوقۇتتى، ئاندىن ماڭا:

«ئىسلام كەلتۈرۈگىنىڭنى مەككىدە ھېچكىمگە دېمە، ئۇلارنىڭ سېنى ئۆلتۈرۈۋېتىشىدىن ئەنسىرەيمەن،»دېدى. مەن: ئى ئاللاھ نىڭ پەيغەمبەرى! مەن قۇرەيشلىكلەر ئارىسىدا ھەق سۆزنى ئاشكارا قىلمىغۇچە مەككىدىن كەتمەيمەن، دېدىم. پەيغەمبەر ئەلەيھىسسالام جىم تۇردى.

مەن قۇرەيشلىكلەر ئارىسىغا بېرىپ يۇقىرى ئاۋازىم بىلەن: ئى قۇرەيشلەر گۇرۇھى! مەن بىر ئاللاھ تائالادىن باشقا ئىلاھ يوق، مۇھەممەد ئاللاھ نىڭ ئەلچىسى دەپ گۇۋاھلىق بېرىمەن، دېدىم. ئۇلار سۆزۈمنى ئاڭلىشى بىلەن چۈچۈپ، ئورۇنلىرىدىن تۇرۇشتى ۋە: بۇ دىندىن چىققۇچىنى تۇتۇڭلار، دېيىشىپ يېنىمغا كېلىپ، ئۇرۇشقا باشلىدى. پەيغەمبەر ئەلەيھىسسالامنىڭ تاغىسى ئابباس ئىبنى ئابدۇلمۇتەللىب مېنى كۆرۈپ ئۇلاردىن قوغداش ئۈچۈن، مېنى قۇچاقىلىۋالدى ۋە ئۇلارغا: ھۇ، ھالاك بولغۇچىلار، سىلەر كارۋانلىرىڭلارنىڭ ئۆتۈش يولىدىكى غەففار قەبىلىسىنىڭ ئادىمىنى ئۆلتۈرەمسىلەر؟! دېدى. ئۇلار مېنى قويىۋەتتى.

مەن ھۇشۇمغا كېلىپ، پەيغەمبەر ئەلەيھىسسالامنىڭ قېشىغا باردىم، ئۇ مېنىڭ ھالىمنى كۆرۈپ: «سېنى ئىسلام كەلتۈرگەنلىكىڭنى ئاشكارا قىلىشىڭدىن ئاگاھلاندۇرمىغانمىدىم؟» دېدى.

ئى ئاللاھ نىڭ پەيغەمبەرى! مېنىڭ بىر ھاجىتىم بار ئىدى. ئۇنى ئادا قىلدىم.

«سەن قەۋمىڭگە بار، ئۇلارنى كۆرگەن ئاڭلىغانلىرىڭدىن خەۋەرلەندۈر. ئۇلارنى ئاللاھ تەرىپىگە چاقىرغىن، بەلكى ئاللاھ ئۇلارغا سەن ئارقىلىق مەنپەئەت بېرىشى، بۇنىڭ ئۈچۈن ساڭا ئەجىر بېرىشى مۇمكىن ئەگەر ساڭا مېنىڭ ئاشكارا بولغانلىقىم ئاڭلانسا، مېنىڭ قېشىمغا كەل» دېدى.

مەن قەۋمىمنىڭ قېشىغا باردىم، قېرىندىشىم ئەنەس يولۇقۇپ: نېمە ئىشلارنى قىلدىڭ؟ دەپ سورىدى.

مەن ئىسلام كەلتۈرۈپ، ئاللاھ ۋە پەيغەمبەرىگە ئىشەندىم، دېدىم. ئاللاھ ئۇنىڭ قەلبىنى ئىسلام ئۈچۈن ئېچىۋەتتى.ئۇ سېنىڭ دىنىڭدىن يۈز ئۆرىمەيمەن،مەنمۇ ئىمان ئېيتىپ مۇسۇلمان بولدۇم، دېدى. بىز ئانىمىزنىڭ قېشىغا كەلدۇق ۋە ئۇنى ئىسلامغا دەۋەت قىلدۇق، ئۇ: مەن دىنىڭلاردىن يۈز ئۆرۈيمەن، دەپ ئىسلام كەلتۈردى.

شۇ كۈندىن باشلاپ بۇ ئائىلە كىشىلىرى پۈتكۈل ۋۇجۇدى بىلەن غەفار قەبىلىسىنى ئىسلامغا چاقىرىشقا كىرىشتى،نۇرغۇن كىشىلەر ئىسلامغا كىردى،ئۇلار ئارىسىدا نامازلار ئوقۇلدى.ئۇ قەبىلىدىكى بەزى كىشىلەر ھەتتا پەيغەمبەر ئەلەيھىسالام مەدىنىگە كەلگىچە كۈتۈپ ئاندىن ئىمان ئېيتىمىز،دېيىشتى ـ دە پەيغەمبەر ئەلەيھىسالام مەدىنىگە كەلگەن ھامانلا ئىمان ئېيتىشتى . پەيغەمبەر ئەلەيھىسالام :«ئاللاھ غىفار قەبىلىسىنى مەغپىرەت قىلسۇن!ئۇلار ئىسلامغا كىردى،ئاللاھ ئۇلارنى سالامەت قىلسۇن!» دېدى.تاكى بەدىر،ئوھۇد،خەندەك غازاتلىرى بولۇپ ئۆتكىچە ئەبۇ زەر ئۆز قەبىلىسى بىلەن بىرگە ياشاپ شۇندىن كېيىن مەدىنىگە كېلىپ ئۆزىنى پەيغەمبەر ئەلەيھىسالام بىلەن ياشاشقا ئاتىدى،ئۇ پەيغەمبەر ئەلەيھىسالامدىن ئۇنىڭ ئۈچۈن خىزمەت قىلىشقا رۇخسەت بېرىشنى سورىغاندا پەيغەمبەر ئەلەيھىسالام ئۇنىڭغا قوشۇلدى. شۇندىن ئىتىۋارەن ئۇ پەيغەمبەر ئەلەيھىسالامغا خىزمەت قىلىشتەك نېمەتكە ئېرىشىپ بەخىتلىك بولدى.

پەيغەمبەر ئەلەيھىسالام ئۇنى دائىم باشقىلاردىن بەكرەك ئەتىۋارلىدى.ئۇنىڭغا قاچانلا ئۇچرىسا خۇش تەبەسۇم بىلەن قول ئېلىشىپ كۆرۈشەتتى. پەيغەمبەر ئەلەيھىسالام ۋاپات بولغاندىن كېيىن مەدىنە ـ مۇنەۋۋەر خوجايىنىدىن ئايرىلىپ  ئۇنىڭ سۆھبەتلىرىدىن مەھرۇم قالدى.ئەبۇ زەر مەدىنىدە پەيغەمبەر ئەلەيھىسالامسىز ياشاشقا تاقەت قىلالماي شامغا كۆچۈپ كېتىپ ئۇ يەردە تاكى ئابابەكرى(ئاللاھ ئۇنىڭدىن رازى بولسۇن! )نىڭ خەلىپىلىك زامانىسىغىچە تۇردى.ئوسماننىڭ (ئاللاھ ئۇنىڭدىن رازى بولسۇن! )خەلىپىلىك زامانىدا ئۇ دەمەشىققە كەلدى ۋە بۇ يەردىكى مۇسۇلمانلارنىڭ تۇرمۇشى ناھايتى بايلىشىپ، مۇسۇلمانلارنىڭ راھەت پاراغەتكە ناھايىتىمۇ بەك بېرىلىپ  ـ كەتكەنلىكىنى كۆردى.ئۆسمان ئۇنى مەدىنگە قايتىپ كېلىشكە تەكلىپ قىلدى.ئۇ مەدىنىگە كېلىپ بۇ يەردىمۇ كىشىلەرنىڭ مال دۇنياغا ھېرىسمەن بولۇپ كەتكەنلىكىنى كۆردى ـ دە بۇ ئەھۋالدىن ئۇنىڭ يۈرۈكى ناھايىتى بەك بىئارام بولدى.ئوسمان ئۇنى مەدىنىنىڭ يىراق بىر چېتىدىكى ”رەبزە“ دەپ ئاتىلىدىغان كىچىك بىر كەنتكە كىتىپ شۇ يەردە ياشاشقا روخسەت قىلدى.ئۇ بۇ يەرگە كېلىپ دۇنيانىڭ زىبۇ –زېننەتلىرىدىن يىراق بولغان ھالدا پەيغەمبەر ئەلەيھىسالام ۋە ئۇنىڭ ئىككى خەلىپىسىنىڭ يولىنى تۇتۇپ ياشىدى.

بىر كۈنى ئۇنىڭ ئۆيىگە بىرى مېھمان بولۇپ كېلىپ ئۇنىڭ ھېچقانداق بىساتى بولمىغان ئۆيىگە سەپسىلىپ قاراپ چىققاندىن كېيىن ئۇنىڭغا: ئى ئەبۇ زەر سېنىڭ تۇرمۇش لازىمەتلىكلىرىڭ قايەردە؟ دېدى.

ئەبۇ زەر:بىزنىڭ لازىمەتلىكلىرىمىز ئاۋۇ يەردىكى ئۆيدە (يەنى ئاخىرەتتىكى ئۆيدە)،بىز پايدىلىنىدىغان نەرسىلەرنىڭ ياخشىسىنى بىز  ئاشۇ ئۆيگە ئەۋەتىمىز،دېدى.

ئۇ مېھمان:لىكىن بۇ دۇنيادا ياشىشىڭلار ئۈچۈنمۇ سىلەرگە بىر نەرسىلەر كېرەكقۇ؟دېگەندە ئەبۇ زەر:ئۆي ئىگىسى اللھ بىزنى تاشلىۋەتمەيدۇ،دېدى.

شام پادىشاھى ئۇنىڭغا خىراجەت قىلسۇن،دەپ 300 دەرھەم ئەۋەتكەندە، ئەبۇ زەر شام پادىشاھى مەندىنمۇ بەتتەر ئاجىزراق بىرەرىنى تاپالىمىغانمىدۇ؟دەپ ئۇ پۇلنى قايتۇرۋېتىدۇ.
