                                          بولمىدى    

                                       ئالىمجان تۇرغۇن سەرسانى

                            ماڭسام دەۋىر كەتتىلەر ، يۈگۈرمىسەم بولمىدى ،
                            نىمە  بولسا   يېڭىلىق  ئۈلگۈرمىسەم  بولمىدى  .
                           “خىسلەت “ لەرنى قەلبىمدە ئۈندۈرمىسەم بولمىدى ،
                            بارچە ئىشقا  ۋىجداننى  كۆندۈرمىسەم بولمىدى .

                            ئۇزاپ  كەتتى  بىراۋلار   ئىلىم -  پەنگە  ئاتلىنىپ ،
                            كىيىم كىيدىم ئوخشۇتۇپ ، قالماي مەن دەپ ياتلىنىپ .
                            يۈرەك  ئېغىر  سوقتىلەر ،  ۋىجدان  كۆيدى قاقلىنىپ ،
                            شۇنىڭدىمۇ     دەۋىرگە     يىتىشمىسەم    بولمىدى .

                            سالام ئېيتسام “ ئەسسالام ...” قۇسۇر تاپتى بىرلىرى ،
                           “ كەچتە ياخشى چۈش تىلەڭ ...” دەپ ئۈگەتتى بىرلىرى .
                            تاپشۇراتتىم   “ ھەق “كە   مەن   ياقتۇرمىدى  بىرلىرى ،
                            سۆز   قىلىشنى   يېڭىدىن      ئۈگەنمىسەم     بولمىدى .

                            رومىلىنى    ئايلىمنىڭ   ئېلىپ     نېرى    تاشلىدىم ،
                            يارىشاتتى   بۇرۇتۇم    پاكىز   قىرىپ    تاشلىدىم .
                            “ يىتىشىش”    نى   دىمەككى   قىياپەتتە باشلىدىم ،
                           قارا   بوياپ    چېچىمنى   ، ياش يۈرمىسەم  بولمىدى .

                           پادىچىلار   ئىشتىنى   كىيدۈردۈم    مەن   قىزىمغا ،
                           قالمىسۇن  دەپ مودىدىن ،  قېتىلسۇن  دەپ ئىزىمغا .
                           بىراۋ كۆرسە  “ مەتۇ “ دەپ ، قار ياغمىسۇن يېزىمغا ،
                           باللىرىمغا   “ ئويناش “ نى   ئۈگەتمىسەم   بولمىدى .

                           ئايال زارلاپ ئېيتتىلەر “ دورامتۇق بىز خەقنى ؟ “ دەپ ،
                          “ قانداق   ئۆتسە   ئۆزىگە ، بىزگە نىمە كەپتۇ ؟ “ دەپ .
                          “پوقنى يىسە  بىراۋلار ، بىز ئەگىشىپ يەمتۇق ؟ “ دەپ ،
                          چۈشەندۈرۈپ    دەۋىرنى   ،   كۆندۈرمىسەم   بولمىدى .

                          ھەرخىل    تۈردە   مەرىكە   چىقتى    بۈگۈن   يېڭىدىن ،
                           ھەر   شەكىلدە   ئىشرەتكە    ، يول تېپىلدى  يېڭىدىن .
                          مەن ئالدىراش    شۇنىڭچۈن ،    قالماي دەۋەر سىپىدىن ،
                          چۆرۈپ    يېشىم   ، ئورنۇمنى    ، ئوينىمىسام  بولمىدى .

                           جان   تىكىپ  مەن   ئىشلىدىم ،  ئاشالمىدىم  قىرىنى ،
                           تونۇپ   يەتتىم   شۇندىلا   ، ئەمەلدارلىق    سىرىنى .
                           ئۇستاز   تۇتتۇم   ھەق تۆلەپ “مۇناسىۋەت پىرى “نى ،
                           باشلىقىمنىڭ   ئۆيىگە    كۆپ    بارمىسام   بولمىدى .

                           باشلىق بولۇپ ئۆستۈم مەن ، كارغا كېلىپ “ قەنت “ لىرىم ،
                           تولا    ئىگىپ    تالغانتى   ، تۈزلەندى   ھەم   بەللىرىم .
                           شۈلۈپ     يىدىم   ئامال    يوق ،  يىغلىسىمۇ  ئەللىرىم ،
                           شۇندا   ئارقا   ئىشىكتىن   چوڭ    ماڭمىسام    بولمىدى .

                                                     ***********

                          ھەممە    شۇنداق   ئەگەشتۇق ، خاسلىقىمىز  بولمىدى ،
                          ئارلاشتۇردۇق    بارچىنى   ، ساپلىقىمىز     بولمىدى .
                          كېسەل   تۈرى    كۆپەيدى    ، ساقلىقىمىز     بولمىدى ،
                          توختا !    سەنمۇ سەرسانى   ، سەبىر قىلغىن   ، بولمىدى .
