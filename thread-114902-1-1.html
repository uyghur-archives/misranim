پېقىر مۇنبەر بارلىققا كەلگەندىن تارتىپ،تېما يوللاشنى ئىزچىل داۋاملاشتۇرۇپ كەلدىم.بۇنىڭدا ئەلۋەتتە ماڭا شارائىت يارىتىپ بەرگەن ھەرقايسى مۇنبەرلەرنىڭ ۋە مىنى رىغبەتلەندۈرۈپ كەلگەن تورداشلارنىڭ تۆھپىسى ناھايىتى چوڭ.كۆپ كىشىلەر سىز كۈندە مۇنبەرگە نەرسە يېزىپ يۈرسىڭىز ئىقتىسادنى قانداق ھەل قىلىسىز دەپ سورايدۇ.ھەتتاكى كۈندە بىر نەرسە يېزىپ ئولتۇرغىنىمنى كۆرگەن تورخانا مۇلازىمىمۇ ئەتراپتىكى ۋاراڭ چۇرۇڭدىن بىخەۋەر كۇنۇپكا تاختىسىنى چوقۇپ ئولتۇرغان ماڭا قاراپ ياقىسىنى چىشلەپ كىتىدۇ.ئۇنداقتا مەن بۇ بىر نەچچە يىلنى قانداق ئۆتكۈزۈپ كەلدىم دىگەندە.ئەمەلىيەتتە مەن ھەر كۈنى 4سائەت يايما ئېچىپ،كۈندە 100يۈەن~~300يۈەنگىچە كىرىم قىلىپ ،شۇ ئىقتىساد بىلەن تۇرمۇشۇمنى ۋە يېزىقچىلىقىمنى داۋاملاشتۇرۇپ كەلدىم.مانا يېقىندا بالام ئىككى بولۇپ قېلىپ يايمىچىلىقتىن باشقا ئىش قىلىشنى نىيەت قىلغاچقا بۇ نەچچە يىل داۋامىدا يايمىچىلىق قىلىپ توپلىغان تەجىربىلىرىمنى سىلەرگە ئاشكارىلىۋەتمەكچى بولدۇم.
1.ئاۋۋال قەتئى نىيەتكە كىلىپ ،ئىناۋەت-پىناۋەت دىگەننى چۆرۈپ تاشلاپ،يايمىچىلىق قىلىشقا ئېنىق قارارغا كىلىسىز.ئەگەر يايمىدا مال سېتىۋاتقاندا بۇرۇنقى قىز دوستىڭىز كىلىپ قالسا،قىزىرىپ قالسىڭىز قەتئى بولمايدۇ.ئېسىڭىزدە بولسۇن سىز نومۇس قىلىپ يۈرگەن ئاشۇ شەخىسلەر بەرىبىر سىزگە بىر موچەن بەرمەيدۇ.
2.چىقىپ مال تاللايسىز. يايمىچىلىقنى باشلاشنىڭ ئەڭ ياخشى پەيتى ئېنىقكى قىش.چۈنكى قىشتا نۇرغۇن تاۋارلار چوقۇم كېرەك قىلىدىغان تاۋارغا ئايلىنىدۇ.مەسىلەن،پەلەي،ئۆتۈك،قونچىلىق ئاياق،پاختىلىق ئىشتان-چاپان،پوسما،خۇرۇم چاپان دىگەنلەرگە ئېھتىياج ئىنتايىن يۇقىرى بولىدۇ.توپچىلارنىڭ قېشىغا بېرىپ سۈپەتلىك قىشلىق مالدىن بىرنى تاللايسىز.ئىسىڭىزدە بولسۇن مېلىڭىز چوقۇم سۈپەتلىك بولسۇن.باھاسى بىر ئاز ئۆرە بولسىمۇ مەيلى.يەنە بىرگە ئالاھىدىلىكى بار مال سېتىۋېلىشىڭىز لازىم.چۈنكى سىز خېرىدارغا مالنى چۈشەندۈرگەندە باشقا مال بىلەن ھېچقانداق پەرقى بولمىسا،پەقەتلا باھا رىقابىتىگە كىرىپ قېلىپ،جاپاغا چۇشلۇق پۇل تاپالمايسىز.
3.تەققى تۇرقىڭىزنى ئوڭشايسىز.يامىغا چىقماقچى بولغان كۈنى يۇيۇنۇپ يىپ-يېڭى كىيىمدىن بىرنى كىيىپ،ئەتىر چېچىپ،چوقۇم ساتىراچخانىغا بېرىپ چاچ-ساقالنى ياسىتىسىز،ئاندىن پۇرىقى مىززىلىك،سۇس ئەتىردىن چاچىسىز.
4.ئىسىم-كارتۇچكىسى باسقۇزىسىز.ئىسىم كارتۇچكىنىڭ يايمىچىلىقتا ناھايىتى روشەن رولى بار.مەسىلەن يايمىدا كىيىمنى كىيىپ باققىلى بولمايدۇ.ئەگەر مال چوڭ كىچىك كىلىپ قالسا،يايمىنىڭ ئورنىنى بىر يەرگە مۇقىملاشتۇرۇش تەس بولغاچقا خېرىدار سىزنى تاپالماي قېلىشى سىزنى قاغىشى مۇمكىن.ئىسىم-كارتىڭىز بولسا سىزگە تېلىغۇن قىلىدۇ.سىزمۇ چوقۇم خوشاللىق بىلەن ئالماشتۇرىشىڭىز شەرت.
5.مالنى ئازراق ئېلىپ،بىر ئورۇننى تۇتىمىز.مالنى چوقۇم ئاز ئېلىڭ!!!!!!!چوقۇم ئېسىڭىزدە بولسۇن.مەن ئادەتتە ھىساپلاپ بېقىپ ،پۈتۈن پايدىسى 400يۈەنگە يەتكىچىلىكلا مال ئېلىپ چىقىمەن.مال ئاز بولسا ھەرقانداق بىر بۇلۇڭدا مالنى قويالايسىز،شەھەر قىياپىتىنى تۈزەشتىكى يولداشلار كەلسە مالنىڭ ئاغزىنى ئىتىپلا پاقىراپ تۇرىۋالسىڭىز ياكى كۆتۈرگەنچە ئەتراپتىكى ئاھالىلىكلەر قوروسىغا كىرىپ تۇرىۋالسىڭىز بولىدۇ.ئاچكۆزلۈك قىلىپ جىق مال ئېلىپ چىقىۋالسىڭىز كۆتۈرگىچە جاپا تارتىش،مۇۋاپىق ئورۇن تاپالماسلىق،
مالنى شەھەر قىياپىتى تۈزەشتىكىلەرگە تارتقۇزۇپ قويۇش،مالنى ئوغرى ئېلىش قاتارلىق دىشۋارچىلىقلارغا ئۇچرايسىز.
6.ھەممە مەسىلە ھەل بولۇپ يايمىغا چىقىپ بولدىڭىز.ئەمما ھەرگىز باشقىلارنى دوراپ 50كوي،50كوي !!دەپ توۋلىماڭ.بۇنداق قىلسىڭىز شاۋقۇن پەيدا قىلىپ باشقىلارنى نارازى قىلغاندىن سىرت،40كويلۇق ياكى 80كويلۇق مال ئالغۇسى بار خېرىدارلارنى قاچۇرۇپ قويىسىز.ئېسىڭىزدە بولسۇن ئالسۇن-ئالمىسۇن يايمىغا ئادەم ئولاشتى دىگەنلىك مال ساتتىڭىز دىگەنلىك.ئەمىسە توۋلىماي پاقىراپ تۇرسام بولغىدەكمۇ دەپ قالماڭ.سىزنىڭ خىزمىتىڭىز ھەرگىزمۇ توۋلاشتىن يىنىك ئەمەس.سىز مالنى قولىڭىزغا ئېلىپ قىشىڭىزدىن ئۆتكەنلەرگە مۇلايىملىق بىلەن مالنى تونۇشتۇرىسىز.مەسىلەن قېشىڭىزدىن پۇزۇر ياسانغان بىر قىز ئۆتۈپ قالدى دەيلى،سىز دەرھال جىلىتكىدىن بىر نى ئېلىپ:
-ماۋۇ مالنى كۆرەپ بېقىڭ،بەك سۈپەتلىك،نىپىز،ئىسسىق،ئىچىڭىزگە كىيىۋېلىپ توي تۆكۈنگە يارىشىملىق كىيىپ بارسىڭىزمۇ توڭلىمايسىز،-دەيسىز.ئەگەر ئۇ قارىماي ئۆتۈپ كەتسە،يەنى مالغا قىززىقمىسا ئۇنىڭ بۇ مالغا ئېھتىياجى يوقلىقىنى بىلدۈرىدۇ.سىز داۋاملىق ئۇنىڭغا مۇلازىمەت قىلماي دەرھال باشقا نىشان تاللايسىز.
ئەگەر ئۇ قىززىقىپ:
-نەچچە پۇل؟،-دەپ سوراپ قالسا،سىز ئاۋۋال باھانى دىمەي:
-ئاۋۋال مالنى كۆرۈڭ،ناچار مالنى مىڭ ئەرزان بەرسىمۇ ئالمايسىز،-دىگەندىن كىيىن مېلىڭىزنىڭ ئالاھىدىلىكىنى ئۇنىڭغا چۈشەندۈرىسىز.ھەرگىز ئاۋۋال باھانى دىمەيسىز.مال خېرىدارغا ياراپ قالسا ،ئۇ چوقۇم
سىزگە ئازراق نەپ بېرىش ھىسابىغا مالنى ئېلىپ قالىدۇ.ئەگەر ئالماي كەتسىمۇ يايمىنىڭ ئالدىدا بىردەم تۇرۇپ قالىدۇ.ئىنساننىڭ ماھىيىتى باشقىلارنىڭ قىززىققىنىغا ئەگىشىش بولغاچقا بىردەمدە ئادەم توپلىشىشى مۇمكىن.
يەنە تەكىتلەيمەن يايمىغا ئادەم توپلاشتى دىگەنلىك سودا قىلدىڭىز دىگەنلىك.
7.باھانى مۇقىم قىلىسىز.باھانى جىق قويۇۋالماي،بىرلا باھا قويۇپ چۈشۈرمەي ساتىسىز.چۈنكى باھا تالىشىش جەريانىدا ئوڭايلا ئىسسىق-سوغۇق گەپلەر بولۇپ قېلىشى مۇمكىن.ئۇنىڭ ئۈستىگە بۇ بىر پىسخىكا ئۇرۇشى بولغاچقا بەكلا زېھىن خورىتىدۇ ۋە ۋاقىت ئالىدۇ.سىز ۋاي ئازراقمۇ چۈشۈرۈپ بەرمەمسىز دەپ خاپا بولسا،
مالمۇ ماڭا  جىق پايدا قىلمايدۇ،100كوي دەپ سىزنى بىرمۇنچە ئاۋارە قىلىپ 80 كويغا بەرگەندىن،ئۇدۇللا 80 كوي دىسەم ياخشى ئەمەسمۇ دىسىڭىز ئۇلارنىڭ كاللىسىدىن ئۆتۈپ قالىدۇ.ئەلۋەتتە ئەگەر ئالغان مالنىڭ سانى 3تىن ئوشۇق بولسا 3تال مالدا 10 كوي قويۇۋەتسىڭىز بولىدۇ.ھەرگىز 15 كوي قويۇۋەتمەڭ.چۈنكى بۇنداق قىلسىڭىز باشقىلار پايدا قىلغاندىن كىيىن ماڭىمۇ بۇ بىر تال مالدا 5 كوي قويۇۋېتىڭ دەپ تۇرىۋالىدۇ.10 كوي قويىۋەتسىڭىز ماڭا 3.33333يۈەن قويىۋېتىڭ دىيەلمەيدۇ.
8.ھەرگىز بازانىڭ يىغىلىشىغا ساقلاپ تۇرماڭ.تىرىشچانلىق قىلدىم دەپ ھەرگىز ھەممە ئەدەم كەتكۈچە تۇرماڭ.چوقۇم بازاردا ئادەم ئازلىسىلا مالنى يىغىشتۇرۇپ قايتىپ كىتىپ ئەھمىيەتلىك ئىشلار بىلەن شۇغۇللىنىڭ.چۈنكى بازاردا كەچكىچە تۇرىۋالسىڭىز باشقىلار ئۆتمەس مال بولغاچقا مۇشۇنچىلىك قىلىپ كىتىپ بارىدۇ دەپ قېلىشى مۇمكىن.بازاردىكى ئۇچۇرنىڭ تارقىلىشىغا سەل قارىسىڭىزلا چوقۇم ئۇتتۇرىۋېتىسىز.
يەنە بىرى ۋاقىت ئالتۇندىن قىممەت 20،30كوي سودىنى دەپ ئادەم ئاز بازاردا 2.3 سائەتنى ئۆتكۈزىۋەتسىڭىز قەتئى ئۇتتۇرغان بولىسىز.
9.قوشنىلىرىڭىزغا مالنى بەك ئەرزان بەرمەڭ.يايمىچىلارمۇ نۆۋىتىدە تەغدىرداشلار.ئۇنىڭ ئۈستىگە بىز مۇسۇلمان بولغاچقا چوقۇم قوشنىلىرىمىز بىلەن چىقىشىپ ئۆتىشىمىز كېرەك.ھەم چىقىشىپ قالىمىز.بىرەر دۇكان قوشنىڭىز مالدىن بىرنى ئۆزى كەيگىلى ئەرزانراق بېرىشنى سورىسا،ئىككىلەنمەي سېتىۋاتقان مۇقىم باھادىن 5يۈەن چۈشۈرۈپ بېرىڭ.يەنە چۈشۈرۈڭ دىسە ماڭىمۇ جىق پايدا قىلمايدۇ دەپ تۇرىۋېلىڭ.ئىسىڭىزدە بولسۇن بەلكىم ئۇ قوشنىڭىزنى ئاباي تېخى سىزگە 10 يۈەن چۈشۈرۈپ بېرىڭ دىسە سىز ئۇنىمىغاندىن كىيىن ئەۋەتكەن خېرىدار ماڭا ئەرزانراق ئېلىپ بەرسىڭىز دەپ ئەۋەتكەن بولىشى مۇمكىن.
10.قاچىلاشقا ئەھمىيەت بېرىڭ.مەسىلەن سىز 30 يۈەندىن كىرگۈزگەن مال بولسا.2يۈەنلىك ئىسىلراق سولياۋ سومكا ئېلىشقا ئىككىلەنمەڭ.بۇ سومكا سىزنىڭ خېرىدارنىڭ مال ئالغانلىقى ئۈچۈن بەرگەن سوۋغىڭىز بولۇپ قالسۇن.ئېسىڭىزدە بولسۇن بەزىدە قاپاق پۇل ئەمەس،باراڭ پۇل بولۇپ قالىدۇ.

ئەڭ ئاخرىدا يايما ئاچقاندا ئۇچرايدىغان بەزىبىر ئاۋارىچىلىقلارنى ھەل قىلىش يولىنى دەپ بېرىمەن.
بىرىنچى خىل ئەھۋال.بەزىبىر خېرىدارلار خېرىدار توپلاشقان ۋاقىتتا،قەستەن مالدىن قۇسۇر چىقىرىشى مۇمكىن.
مەسىلەن بىز باشتا ھېلىقى پوزۇر قىزغا ساتماقچى بولغان جىلىتكىنى دەيلى.تازا كۆپ خېرىدار مال كۆرۈۋاتقاندا بىرى مالنى ئۆرۈپ-چۆرۈپ ئاغزىنى ئۆمچەيتىپ:
-بۇنىڭ ئىچىدىكىسى چوقۇم پاختا ئەمەس،-دەيدۇ.تازا قىززىق مال كۆرۈۋاتقان خېرىدارلارنىڭ ھەممىسى مال كۆرۈشنى توختىتىپ سىزگە قاراپ تۇرىدۇ.ئىسىڭىزدە بولسۇنكى ھەرگىز ئاچچىقلانماڭ.كۈلۈپ تۇرۇپ:
-ئەگەر پۇلنى تۆلىۋەتسىڭىز،جىلىتكىنى ھازىرلا كىسىپ بېرىمەن.پاختا چىقمىسا پۇلنى سىزگە ياندۇرۇپ بېرىمەن،پاختا چىقىپ قالسا سىزگە ھىساپ دەيسىز.
خاتىرجەم بولۇڭ،ئۇ پۇلنى سىزگە بېرىپ كىسە قېنى دىمەيدۇ.چۈنكى ھېچقانداق ئادەمنىڭ ئاۋۋال پۇل تاپشۇرۇپ ئوينايدىغان ئويۇن بىلەن خوشى يوق.
شۇنداق قىلىپ ئۇ قىزىرىپ تۇرۇپ ئۇققىلى بولمايدىغان بىر نەرسىلەرنى دەپ غودۇڭشىيدۇ.سىز:
-پۇل تاپماق تەس.شۇڭا مالنى ئوبدان تەكشۈرۈپ ئالغاننىڭ زىيىنى يوق.ئەسلى بۇرۇنلا بىرنى كىسىپ قوياي دىگەنىدىم بىكار بىر مالنى ئىسراپ قىلماي دىدىم،-دەپ ئۇنىڭ ئۆزەڭگىدىن چۈشۈرۈپ قويىسىز.چۈنكى خېرىدارلارنىڭ دېھقىتى مالدىن ئۇزاق بۇرىلىپ كەتسە بولمايدۇ.
ئىككىنچى خىل ئەھۋال.بەزىلەر يايمىغا كىلىپ مال كۆرۈپ سىزدىن بىر مالنى سورايدۇ.ئاندىن كىيىن يەنە بىر مالنى ئېلىپ ئۇنىڭغا سېلىشتۇرىدۇ.بۇنداق چاغدا قاتتىق ھوشيار بولىشىڭىز كېرەك.چۈنكى قولىغا ئىككى مالنى ئېلىۋالغان ئادەمنىڭ بىر مالنى قويۇپ قويۇپ يەنە بىر مالنى سىز دېھقەت قىلمىغان پۇرسەتتە كۆتۈرۈپ كىتىش ئېھتىمالى بەك چوڭ.بىر مال يۈتسە بىرمۇنچە ئەجرىڭىز بىكار بولدۇ.يەنە بىر تەكىتلەيمەن قولىغا ئىككى مال ئېلىۋالغان خېرىدارغا دېھقەت قىلىڭ.ئەگەر ئۇ راست دىگەندەك بىر تال مالنى كۆتۈرۈپ كەتسە دەرھال:
-ئاۋۇ مالنى خالتىغا سېلىپ بېرەيمۇ؟،-دەڭ.ھەرگىز قاتتىق تەگمەڭ چۈنكى ئۇنىڭ راستىنلا ئۇنتۇپ قېلىپ مالنى ئېلىپ كەتكەنلىك ئېھتىمالى بار.بۇنداق چاغدا ئۇ:
-ۋاي خۇدايىمەي.قەتئى خاپا بولماڭ،پەقەت دېھقەت قىلماپتىمەن،-دەيدۇ.سىز:
-بۇنداق ئىشلار بولۇپ قالىدۇ دەپ،-دەرھال مالنى خالتىغا سېلىپ ئۇنىڭغا ئۇزارتىپ زولاپ بىر سودا قىلىۋالىسىز.ئەگەر ئالمىسىمۇ مەيلى.ئەگر سىز ئوغرى پالان-پوكۇن دەپ گەپ تالاشسىڭىز بىكار ۋاقىتنى ئىسراپ قىلىسىز،بەلكىم بىكاردىن-بىكار بىرياخشى ئادەمگە تۆھمەت چاپلىغان بولۇپ قېلىشىڭىزمۇ مۇمكىن.
ئۈچىنچى خىل ئەھۋال.سىزنىڭ يايمىڭىز پىيادىلەر يولىدا بولسا،يولنىڭ بېقىنىدىكى دۇكاننىڭ خوجايىنلىرى سىز خېلى يىراقتا تۇرغان بولسىڭىزمۇ دۇكاننىڭ ئالدىنى توستۇڭ دەپ سىزنى ھەيدەيدۇ.(بۇ خىل ئەھۋال ئىنتايىن كۆپ ئۇچرايدۇ.)سىز بۇنداق ۋاقىتتا گۆش باشلىق قىلىپ كەتسىڭىز قەتئى بولمايدۇ.خەقلەرمۇ ئاچسا باشقا دۇكاندارلار گەپ قىلماپتۇ دەپ گەپ تالاشسىڭىزمۇ بولمايدۇ.سىز ئاۋۋال قىسقىلا قىلىپ:
-بو يولنى سىلە ئىجارىگە ئالمىغان بولغۇيتتىلە؟؟؟؟؟؟،-دەيسىز.گېپىڭىزنىڭ لى غا ياتىدىغانلىقىنى بىلگەندىن كىيىن ئۇ بىزدە بوشايدۇ.سىز پۇرسەتنى چىڭ تۇتۇپ:
-قاراڭ ئاكا،مەنمۇ شۇ جان باقاي دەپ چىقتىم.ئۇزاق ئاچمايمەن.مېلىممۇ جىق ئەمەس،بىردەم ئېچىۋالاي ئاكا،-دەپ بەكمۇ يالۋۇرماي سۆزلەپ بۇ تىجارەتنىڭ سىزنىڭ تۈپ مەنپەئەتىڭىزگە تاقىشىدىغانلىقىنى ئەسكەرتىپ قويىسىز.ئەگەر ئۇ ئىش كۆرگەن ئادەم بولسا بۇ گېپىڭىزنىڭ تىگىگە يىتىپ سىز بىلەن جىق تاقارىشمايدۇ.
سىزمۇ دەرھال مالنى ئالدىغا ئازراق (10،20سانتىمېتىر)سۈرۈپ ئۇنىڭغا يۈز بەرگەن بولىسىز.ئەگەر ئۇ سىزنىڭ تۈپ مەنپەئەتىڭىزگە دەخلى قىلماقچى بولۇپ يەنە ۋار-ۋۇر قىلىپ سىزنى بوزەك قىلماقچى بولسا،مالنىڭ ئاغزىنى يېپىۋېتىپ،چاپاننى ئۆرۈپ ئوڭ قولىڭىز بىلەن كەينى بېلىڭىزنى خۇددى پىچاق تۇتقاندەك تۇتۇپ:
-ئەڭ ياخشىسى مالغا چېقىلما.جاندىن تويمىغان بولسام بۇ يايمىدا نان ئىزدەپ يۈرمەيتتىم.،-دەپ بىر مۇش سالسا پىچاقنى تىقىدىغاندەكلا ئەلپازدا تۇرۇڭ.خەقنى جان باققۇزمايمەن دەيدىغان كىشىلەرنىڭ كۆپى قەغەز يولۋاس كىلىدۇ.ئۇ چوقۇم تېلىفۇننى كۆتۈرگىنىچە دۇكانغا كىرىپ ئۇنىڭ-بۇنىڭغا تېلىفۇن ئۇرۇشقا باشلايدۇ.
ئەمەلىيەتتە ھېچكىمنى ئەكىلەلمەسلىكى مۇمكىن.ئەكەلسىمۇ داۋلى سىز تەرەپتە،يول خەلىقنىڭ.سىز شۇخەلىق ئىچىدىكى بىر تال خەلىق.لىكىن زىيادە ئاۋارىچىلىقتىن قۇتۇلۇش ئۈچۈن قېلىنلىق بىلەن ئۇنىڭ دۇكىنىدىن خالىغانچە بىرەر نەرسە سېتىۋېلىپ ،مۇسۇلمان تۇرۇپ قاتتىق -قۇرۇق گەپ قىلىشىپ قالغانلىقىڭىز ئۈچۈن ئۇنىڭدىن ئەپۇ سورايسز.شۇنىڭ بىلەن كاۋاپمۇ-زىخمۇ كۆيمەيدۇ.
