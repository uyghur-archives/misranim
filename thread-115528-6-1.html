ئەلىمار نۇرمەمەتكە قانداق دىيىشنى بىلەلمىدى بولغاي بىر ھازا تۇرۋەتكەندىن كېيىن سۆز ئاچتى ،
‹‹ ئاداش ، نىمىشقا ئۇنداق سورىشىمدا سەۋەپ بار ، ئەمىليەتتە مەن جىن كۆردۇم ›› دېدى ئەلىمار بىر نۇقتقغا قاراپ تۇرۇپ ،ئەمدى چاي ئىچىپ ئولتاغان نۇرمەمەت بۇ گەپنى ئاڭلاپ ئۆپكىسگە كەتتىمۇ قانداق قاتتىق يوتىلۋاتقاندىن كېيىن گەپ قىلدى .
‹‹ نىمە ديدىغانسەن ئاداش ، جىننىڭ بالىقىغۇ ئىشنىمەن ، بىراق جىننى كۆدۈم دىسەڭ تازا ئىشەنگۈم كەلمەيۋاتىدۇ جۇمۇ ››
ساڭا چاقچاق قىلمىدىم ،چاقچاق  قىلغۇدەكمۇ ۋاقتىم يوق ، راس دەۋاتىمەن ، ئۇ باغچىدا يۈز بەرگەن ئىشلارنى بىر بىرلەپ سۆزلەپ بەردى ، نۇرمەمەت ھاڭۋاقىنچە تۇرۇپ قالدى ،ھەم ئۇنىڭ گېپىگە يا ئىشنىنى يا ئىشنەمەسلىكنى بىلەلمەي قالدى .
ئەلىمار نۇرمەمەتنىڭ چىراى ئىپادىسگە بىر قارىۋېتىپ مۇنداق دېدى ‹‹ ئىشەنمىسەڭ ئاغىنە ، يۈر باغچىغا  ھازىر بىللە كىرەيلى ، شۇ چاغدا ئىشىنسەن گېپىمگە ›› دەپ ئۇنى يولەشتۈرۈپ ئاشخاندىن ئېلىپ ماڭدى ،نۇرمەمەت ماشىنا ھايدىغاچ ئەلىماردىن ھېچقانداق گەپ سورىمىدى .
ماشىنا باغچنىڭ ئالدىغا كەپ توختىدى ،ئۇلار ماشىندىن چۈشۈپ باغچىغا كىردى دەل تۈنۈگۈن ئەلىمار ماڭغان يوللاردا ماڭدى ، نىمشقىكىن تۈنۈگۈن ئاڭلىغان ئاۋازلار بۇگۈن ئاڭلانمايتتى ، نۇرمەمەتنىڭ بۇنداق مېڭىپ ئىچى پۇشتى بولغاي ئەلىمارغا
‹‹ قويغىنە يالغان گەپ قىلماي مەن تېخىن راسمىكىن دەپ كىرسەم نەدىكى ئالۋاستى نەدىكى ئاۋازكەن ئۇ ، بۇ يەردە بىزدىن باشقا ئادەم بولمىسا يۈرە كىتەيلى ، بەڭ چارچاپ كەتكەن ئوخشايسەن بىر نەچچە كۈن ئارام ئېلۋاساڭ ئوڭشىلىپ قالىسەن يۈر كىتىمىز ›› نۇرمەمەت شۇندا دەپ غۇدۇرغىنچە كىتىپ قالدى ،
ئەمدى ئەلىمارنىڭ بىر نەرسىلەرنى دەپ چۈشەندۇرشىنىڭ ھاجىتى قالمىدى بولغاي نۇرمەمەتنىڭ ئارقىدىن مېڭىشغا توغرا كەلدى ، توساتتن يېلىقى ئاۋاز ئېچىنشلىق ئاۋاز ئاڭلاندى ، ئەلىمار نۇرمەمەتكە قارىدى نۇرمەمەتنىڭ تىترەۋاتقىنىدىن  بۇ ئاۋازنى ئۇنڭمۇ ئاڭلىنىنىنى جەزىمەلەشتۈردى .ئۇلار ماڭغان يەرلىردە توختاپ  ئەتراپىغا قارىدى دېگەندەڭ ھېچكىم يوق ،ئۇلار بۇ يەردىن تېزىراق ئايرىلىشن ئۈچۈن تېز تېز قەدەم تاشلاپ يۈرۈپ كەتتى ، تۇيۇقسىز نۇرمەمەت قاردا تېيلىپ يىقىلىپ كەتتى ،ئەلىمار ئۇنى يولەپ تۈزەشتۈرۋاتاتىتى بىرسىنىڭ ئۆزىگە قارۋاتقانلىقىنى سەزگەن ئەلىمار يېنىغا قارىدى ،ھەقىقەتەنمۇ ئۇنىڭ بوينىنى سىققان ئاپئاق كىيىملىك بىر قىز ئۇلارغا قاراپ تۇراتتى .
‹‹ھەي ھەي ، بولدى قىلە ، مەينەت يەرلىرى قالمغان ئوخشايدۇ ،بولدى يۈرە كىتەيلى ››دېدى نۇرمەمەت كىيىملىرنى قېقىشتۇرۇپ بىراق ئەلىماردىن سادا كەمىگىنىنى كۈرۈپ ئەلىمارغا قارىدى ۋە ئەلىمار قاراۋاتقان تەرەپكە قاراپ قورقۇنىدىن توۋلىۋېتىشكە تاسلا قالدى . ئۇ قىز ئۇلار تەرپكە قاراپ كىلۋاتاتتى ، ئۇ تېخىمۇ بەك يېىقىنلاشتى ، دەل مۇشۇ پەيىتتە ئەلىمار ئسىنى يىغدى بولغاي يېندىكى پىچاقنى ئېلىپ ئۇنىڭغا شىلتىدى ،   چوڭلار دېگەندەك جىن پىچاق ئەينەكلاردىن قورقۇدۇ دېگەن گەپ راسمۇ  قانداق ئۇلارغا يېقىنلاپ كەلگەن قىزمۇ يوقاپ كەتتى ، ئۇلار بىر بىرنى يولەپ بۇ يەردىن يۈگۈرگنچە چىقىپ كىتىشىتى .
