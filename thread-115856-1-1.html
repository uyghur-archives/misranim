                                    كارامەت
                              ‹‹مىكرۇ يىكايە››


يەتتەيىل يەرتاتلاپ،بىرئۆمۈريىغقان تۈگكىنمنى سەرپ قىلپ ساي سۈيىدىن ياندىما ئېرق ئالدم،
ھەريىلى قانچىلىك كۈچ،قانچىلىك پۇل كەتكىنى نامەلۇم،لېكىن ھەريىلى نەچچەيۈزمېتىر ياندىمائېرق
پۈتۈپ تۇردى.
_مانا،ئىككى يىل بولدى،ئاخىرئېرىق چېپىشتن توختىدم.قاراڭ،مۇشۇ قاپتالدىكى ئىككى يۈز ئەللىك مويەرمېنڭ،ماۋۇباغ،ئاۋۇ نەچچە مىڭ تۇپ تېرەكنى كۆردىڭزمۇ؟
شاپتۇل،ئالما،نەشپۈتلىرىمۇ مېۋىگە كىردى،خېرىدار دېگەننڭ ئايىغى ئۈزۈلمەيدۇ:ماشىنا،تىراكتۇر،
ھارۋىلاردا كېلىشىدۇ-دە،ئۆزلىرى ئۈزۈپ،ئۆزلىرى قاچىلاپ كېتىشىدۇ.مەن پۇل ئېلىشنىلا بىلىمەن،
بۇبىركارامەت،
_ھېرپ قالدىڭز-ھە،مۇخبر ئۇكام؟ئىككى يىلدىن كېيىن بېدىلىك،چۆپلۈك ئارىسىدا ئەمەس،سېمىنۇت
يولدا ماڭىسىز.باغنى ئايلاندۇرۇپ سېمىنۇت يول قىلىمەن.سىزدەك ساياھەتچىلەر ئۈچۈن ئۈستى ئوچۇق
‹‹تويوتا››دىن بىرنى ئېلىپمۇ قويالايمەن قاراڭ، ئاۋۇ ياش كۆچەتلەرگە:يەتتەمڭ ئىككى يۈز تۇپ،ھەرخىل سورتلۇق ئالما  ھقىقىي غولجا ئالمىسىنڭ يېڭى سورتلىرى...يەپ كۆرىڭە،تىلنى ياردۇ،بۇبىر كارامەت،
-ئېلىڭ ئۇكام، داستىخىنىمز ئاددىي بولۇپ قالدى.مېۋىلەرنڭ تەمى ياخشىكەن دەمسىزا؟شۇنداق تەملىك،قۇرت چۈشېدىغان،سېسىدىغان .ئشمۇ يوق.قارىڭا،بېجىرم، يېڭىلا رەسىدە بولغان قىزنڭ يۈزىدەك...بۇبىر كارامەت...
-پىلانم چوڭ،ئۇكام. ئىگىسى يەتكۈزسە ،ئاۋۇ يەرگە ئۈچ قەۋەتلىك مىھمانخانا سالىمەن، بىنانڭ چۆرىسى مېۋىلىك دەرەخ،ئالدى ئۈزۈم بارىڭى بولىدۇ،ساي سۈينى تۇربا بىلەن ئاچىقپ كۆلگە قۇيىمەن، كۆلنىڭ چۆرىسىنى مەجنۇنتاللىق قىلىپ ئوتتۇر سىغا فونتان ئورنىتىمەن. بېلقلار سەكرىشىدۇ، قۇشلار ،سالقىن شامال ئېسىل پۇراقلار بىلەن مېھمانلارنى ھوزۇرلاندۇرىدۇ،بۇبىر كارامەت.
توغرا،توغرا ئېتىسىز ئۇكام،مەن ئاتمىشىتىن ئاشىتىم،ئارام ئالدىغان،راھەت كۆرىدىغان چېغم مۇشۇ.
ئەمما لېكىن مەن ئارام ئالالمايمەن،ھوزۇر لىنىشنى بىلمەيمەن. راست دەيسىز،ھازىرغىچە سارجا،يىپەك دىگەننى كىيىقپ باقمىدم. كىشم چوقاي،يېتىشم ئاۋۇداق سۇپا،ئىچىشىم ئىككى ۋاخ
ئەتكەنچاي،يېيىشم بىر ۋاخ غىزا.ھەركۈنى كۈن چىققاندىن تارتپ قاراڭغۇ چۈشكىچە ئىشلەيمەن:
ئېرق چاپىمەن،سۇتۇتىمەن،قۇراي ئورىمەن،دەرەخ پۇتايمەن،يول ياسايمەن، يەر  تۈزلەيمەن...ۋاي-ۋۇي،ئىش دېگەن چاچتىن تولا،بۇبىر كارامەت،
-نىمانچە قىلىسىز  دەمسىزا؟ قىلماي قانداق قىلاي،بىكار يىتىشقا كۆنمىگەن- دە، بىز.ئازاب نېمە؟
بىكار چىلق،ئۇكام.دۇنيادا بىكار چىلقىتىن يامان ئازاب يوق.كەتمەن، ئورغاق،زەمبىل تۇتمىسام قوللۇرۇم سىرقىرايدۇ ئاغىرىدۇ،ھەرخىل كىسەل ماڭا يامىشىدۇ. قولۇمغا كەتمەن ياكى ئورغاق چىقسىلا تەنلىرم يايراپ،جانلىرىم ئويناپ كېتىدۇ،راھەتلىنىمەن،كۈلۈپ كىتىمەن،ناخشا ئېتىمەن...
بۇبىر كارامەت.
باللارنى دەمسىزا؟ئالتە ئوغۇل ،ئۈچ قىز-توققۇز بالام،ئون بىر نەۋرەم بار،تۆت ئوغلۇمنڭ ئۆيى ئايرم
ئىككى ئوغۇل،بىر قىزم بار،ئۆيدىكىلەرنىڭ چوڭى،ئابايا موتو بىلەن كىلپ پۇل ئىلپ كەتكىنى شۇ.
بىر قىز، بىر ئوغلۇم ئوقىۋاتىدۇ،ئەمدى بىلگەنسىز.مەن يالغۇز ئىشلەيمەن.موتومىنگەننى دەمسىز؟
نېمىسنى يۇشۇراي،ئۇبالام مۇشۇ باغنىڭ خوجايىنى،تولۇقسىزنىمۇ تۈگىتەلمىدى،لېكىن ئۇنىڭ ھالى
بەك چوڭ،نېمىلا  دېسەم:‹‹سەن بىلمەيسەن››دېشنىلا بىلىدۇ،راست،بىلمەيمەن،رېستوران،
كارائوك،دىسكۇ،،تانسا، ئولتۇرۇش-پولتۇرۇش دېگەنلەرنى بىلمەيدىغىنم راست.ئۇ،كۈنىگە ئىككى قېتم ناھىيە بازىرىغا ياكى شەھەرگە موتو بىلەن كىرپ چىقىدۇ.مەندىن بىسوراق مېۋە ساتىدۇ، خەقتىن پۇل ئالىدۇ،كىيىدۇ،يەيدۇ،ئوينايدۇ، ھاي دېسەڭلا قاچىدۇ، ئايلاپ-يىللاپ يوقاپ كەتكەن  چاغلىرىمۇ بولدى.ئانىسىنڭ يىغىسىنى دېمىسەم بۇ ئوغلۇمنى كۆرەرگە كۆزۈم يوق،ئۇكام.
بالامنىڭ يامان گىپىنى قىلپ نىمە قىلاي،بۇبالا بىر كارامەت،
-ئايالىمنى دەمسىزا؟ ئۇبىر ئالتۇن.دۇنيانىڭ نېمە راھىتىنى كۆردى،دەيىسىز؟غەمدىن باشقىنى بىلمەيدۇ.پۇل دىگەن بار،ئۇكام .مال-ۋارانمۇ يىتەرلىك ،لېكىن بۇبىچارە بۇلارنىڭ راھىتىنى كۆرمەيلا مۈكچىيىپ كەتتى توققۇز بالىنى زاكىسىدىن ئاجىرتپ قاتارغا قوشماق ئاسانمۇ؟مەن دېگەن ئېتىز ئادىمى،قازان بېشىغا قولۇمنى ئەگىرى قىلىپمۇ قويمايمەن.نوغۇچمۇ،سۇپرىمۇ،چۆمۈچمۇ شۇ-
بىچارە.تورغاي بىلەن تەڭ تۇردۇ،يېرم كىچىدە كاڭدا تۈگۈلۈپ ئۇخلاپ قالىدۇ.ئۇمىنڭ قاياشىم،
ھەمراھىم،قول قانتم، ئىككىمز باللار ئۈچۈن كۆيدۇق،كۈلدە پىشتۇق،كۆرمىگەننى كۆردۇق،ئاتا-ئانا
بولماق قەنچە تەس بولسىمۇ غىڭ قىلمىدۇق،بىر-بىريمىزگە ھال ئېتىشتۇق،لىكىن ئەلگە‹‹داد››
دەپ چاۋىمىزنى  چىتقا يايمىدۇق،بۇبىر كارامەت،
شۇنداق ئۇكام. بالىلار ئۈچۈن جاپا تارتىمز،قۇرۇلىمز،مۈكچىمز،ھەممىدىن ئايرىلىمز.بىز ئانا دۆگە
ئوخشايمىز ئۇكام.دۆ دېگەن نېمە دەمسىزا؟ئۆمۈچۈككە ئوخشايدىغان ھاشارەت.چاقسا جاننى ئالىدۇ
دەڭە.ئۇبىچارە تۇخۇم تۇغىدۇ،تۇخۇمنى دۈمبىسىدىكى كاۋاكقا سىلىپ شۇ يەردە بالا چىقىردۇ.
بالىلىرى تۇخۇمدىن چىقىپلا ئانىسىنڭ دۈمبىسىنى غاجايدۇ،ئۇلارنىڭ ئۇزقى ئانىسىنڭ دۈمبە گۆشى
ئانا بىچارە غاجاشقا چىدىماي تىنمسىز يۈگىرەيدۇ،تېپىرلايدۇ،بالىلىرى چوڭايغانسىرى غاجىلاش
كۈچىدۇ-دە،ئاخىرى بىچارە ئانا يىقىلىدۇ.بالىلىرى ئانىنى يەپ چوڭ بوېلىۋىلپ مۇستەققل ياشايدۇ.
ئانا ئۆز بەدىنىنى بالىلىرىغا ئۇزۇق قىلپ بولغاندىن كېين دۇنيا بىلەن خوشلىشىدۇ.
بىزمۇ بىر دۆ،ئۇكام بۇمۇ بىر كارامەت،
-مەن يىراقلارغا قارىدىم :تاغلارئۆز بالىلىرى -دەل-دەرەخ،گۈل-گىياھلار ئۈچۈن ئېرىمەكتە،ئۇنىڭ بەدىنىدىن دەريالار ئاقماقتا،يەرشارى ئىنسانلار ئۈچۈن دۆگە ئوخشاش ئۆز بەدىنىنى تۇتۇپ بەرمەكتە ،بىز ئۇنى تىنماستىن يېمەكتىمز، ئانىسنى يېمەيدىغان ھېچ نەرسە بولمىسا كېرەك،
بۇبىر قانۇنىيەت،ھەقىقي كارامەت،
