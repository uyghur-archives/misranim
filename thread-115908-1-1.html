توۋا تەقدىر قىسمەت دىگەن ئاجايىپ بۇلدىكەن ئادەمنى ئويلاپمۇ باقمىغان ئازاپلارغا دۇچار قىلدىكەن......
بۇرۇن ئەجەپ مۇھاببەت نىمە...؟ئازاپ نىمە ياخشى كۈرۈش نىمە بىلمەي يۈرۈپتىمەن....
مانا ئەمدىلىكتە  مىنىڭ سەبىلىكىم بالىلىقىم مۇشۇ بىر قېتىملىق ئۇچرىشىتىن كىين ئالاقاياقلارغىدۇر غايىپ بولغان. پەقەت قەلبىمدە قالغىنى بىر خىل سېغىنىش ئاچىق ئەسلىمە خالاس ياخشى كۈرۈپ تۇرۇپ يا ئېيتالمايمەن ئەمدىلىكتە ئېيىتقاندا بولسا قايتالمايمەن.......
دوستۇم سىز قەلبىڭىزدە ئويغانغان شۇ ئوتلۇق تۇنجى مۇھابىتىڭىزنى ئاشىقىڭىزغا ئېيتىماي يۇشۇرۇپ باققانمۇ....؟
ھىجراننىڭ ئۇزۇن كۇچىسدا ئۈزىڭىز يالغۇز ئۇزاق-ئۇزاق يۇللارنى بېسىپ مېڭىپ باققانمۇ...؟
قەلبىڭىز شۇنچە كۆپ ئازاپقا دۇچار بۇلۇپمۇ يەنە بىر يۈكسەك ئارمانلار ئۈچۈن توختاشنى ئويلىماي داۋملىق ئالغا ئنتىلىپ باققانمۇ....؟
ھەققى پاك مۇھاببەتىنڭ يۇلىدا مۇھاببەتىنىڭ دەھشەتلىك خەنجىرنى يۈركىڭىزگە قايتا-قايتا تىقىپ باققانمۇ....؟
ئۈزىڭىزنىڭ قەلبىدە ئويغانغان مۇھاببەتنىڭ قۇربانى بۇلۇپ باققانمۇ...؟
ئەگەر سىز ھەققى ئاشىق بولسڭىز قەلبىڭىزدىكى  شۇ مۇھەببەت ئۈچۈن بەدەل تۇلەشتىن قورقامسىز...؟ ئۇنداق بولسا بۇ يىگىت قىزنىڭ مۇھاببەت ھىكايىسىگە قۇلاق سېلىڭ ........
ئون ئىككىنجى ئاينىڭ باشلىرى بۇ شەھەرنىڭ  سوغۇق بىر كۈنلىرى ئىدى. قىش پەسلىنىڭ ئوتۇرلىرى بولغاچقا شۇنداق سوغاقمىكىن تاڭ ھېچ بىلىمدىم سوغۇق جاندىن ئۈتەتتى.
شۇنداق سوغوق كۈنلەرنىڭ بىردە بىر دوستۇمنىڭ ئالاھىدە ئۇرۇنلاشتۇرشى ۋە يېقىن دوسلىرىمنىڭ تەكلىپ قىلىشى بىلەن بۇ سورۇنغا مەنمۇ ئالاھىدە قەدەم تەشرىپ قىلدىم...
ئىككىمىزنىڭ مۇھاببىتىمىزنىڭ  تۇنجى قەدىمى مانا مۇشۇ يەردىن باشلانغان ئىدى.توۋا بۇمۇ تەقدىرنى ئۇرۇنلاشتۇرشى بولسا كېرەك .ئىككىمىزنىڭ تۇنجى ئۇچراشقان جايىمىز بولسا پەقەت  ئاددي بىر تاماقخانىدىن باشلانغان. بۇ كۈنى دوستۇمنىڭ بۇ سورۇننى دەپ نۇرغۇن ئىشلىرىمنى تاشلاپ قۇيۇپ كەلدىم....
چۈنكى يېقىن دوسلىرمىز بىلەن بىر يەرگە يىغلىپ ئولتۇرشىمىز بۇنچە ئاسان ئەمەستى. گەرچە ئالدىراش بولساقمۇ ھەممىز ئىشلىرمىزنى تاشلاپ سوغوقنى سوغۇق دىمەي كەلگەن ئىدۇق...!
مەن بۈگۈن باشقا كۈنلەرگە قارىغاندا رەتلىك كىينىپ ئالاھىدە تەيارلىق قىلدىم.مانا ئاخىرى دوستۇم ئايرىم ئېلىپ تەييارلاپ قويغان ئايىرىمخانىنڭ ئالدىغىمۇ كەلدىم يۈرەكلىرىم سېلىپ كىتىۋاتىدۇ....
چۈنكى دوستۇم بىر كۈن بۇرۇن ئۈزىنىڭ قىز دوستىنىڭمۇ بۇ سورۇنغا كىلدىغانلىقىنى ۋە يەنە ئۇ قىزنىڭ بىر قانچە يېقىن دوستىنىڭمۇ بىرگە كىلدغانلىقىنى ئېيتىقانتى...
شۇڭلاشقىمىكىن ئەزەلدىن مەندە كۈرلۈپ باقمىغان بىر خىل جىدديلىشىش قەلبىمىنى چىرمىۋالدى...
توۋا ئۈزۈممۇ بىلمەيمەن پۈتۈن قەلبىمدە بىرخىل جىدديلىشىش بىرخىل تەمتىرەش بار....ئاخىرى ئۈزۈمگە تەسەللي بىرپ كىيىملىرىمىنى  يەنە بىر قۇر تۈزەشتۇرگەندىن كىيىن غەيرەتكە كىلىپ بۇ ئايرىمغانىغا كىردىم.  ۋە يەنە چىرايلىقلا ئەسسالامۇ ئەلەيكۇم ......
-ۋەئەلەيكۇم ئەسسالام ياخشىمۇسىز....؟
-ياخشى ياخشى ئۇلار بىلەن بىر قۇر سالاملىشىپ چىققاندىن كىيىن ئۇلارغا قاراپ ئۈزۈمنىڭ سەل كىچىكىپ كەلگەنلىكىمنى بىلىپ خىجىل بۇلۇپ قالدىم...
كىچىكىپ قالدىم كەچۈرۈڭلار ئازراق ئاۋارەچىلىقلارغا يۇلۇقۇپ شۇڭا كىچىكىپ قالدىم..
-ۋاي  ياقاي ئۇنداق دىمەڭ بىزمۇ ئەمدى كەلدۇق تېخى ....
-ھە شۇنداق بولسىمۇ بۇندىن كىيىن ئۇنداق بوممايدۇ ئالدىڭلاردا خىجىلمەن.....
تۇيۇقسىز بۇلارنىڭ ئارسىدىكى كىچىك باللارغا خاس كىينۋالغان بىر قىزغا   كۈزۈم چۈشتى.
ئۇنىڭ  بىلەن بىلەن تەڭلا قارشىپ قالدىم كۈزلىرگە كۈزۈم ئۇچراشقاندا تىنىم جۇرىدە قىلىپ پۇت- قۇلىرىم بۇشۇپ كەتتى....
ئەسلا ھۇشۇمنى يۇقاتقان ھالدا  بىكار بىر ئۇرۇنغا كىلىپ ئۇلتۇردۇم نىمىش بولغىننى ئۈزۈممۇ بىلمەيمەن...
بۇ قىزنى مەن  بىر يەردە كۆرگەندەكلا قىلاتتىم...
خۇددي بۇ يەردە ئىككىمىز دىن باشقا  بىر ئىنسان يوقتەك بىر خىل سىزىمدە بولدۇم.
ھەممە ئۆز ئىشلىرى بىلەن ئۇلتۇراتتى...
لىكىن مەن بۇ يەرگە كىرشتىن بۇرۇنقى ھالتىمدىن بەكرەك جىدديلىشۋاتىمەن...
تېخىمۇ جىدديلەشكەن ئىدىم مەن ئامال قىلىپ بۇ جىدديچىلىكىنى بىلىندۈرمەسلىككە تىرشتىم،
لىكىن يەنىلا ئاجىزلىق قىلدىم چۈنكى مەن ئادەتتە قىزلار بىلەن بىر سۇرۇندا ئولتۇرمايىتىم  بۇ قېتىم دوستۇمنىڭ كۆڭلىنى دەپ بۇ سورۇنغا قاتناشقىنىمغىمۇ  پۇشايمەن قىلۋاتىمەن.... دوستۇم مىنى چۈشەنگەندىكىن سەزگەن بولسا كېرەك كۈزۈمگە قاراپ بىرنى كۈلۋىتىپلا گەپ باشلاپ كەتتى.
مەن نەقەدەر خوش بولدۇم  ئىچىمدە ئاھ ئاداش سەن بولساڭمۇ مىنى چۈشەندىڭ دەپ ئاللاھقا مىڭ شۈكىرلەرنى ئېيتىپ كەتتىم..
ئارمىزدىكى بىر قانچە مىنۇتلۇق جىمجىتلىقتىن كىين ئۆزئارا پاراڭلاشىپ ئولتۇرۇپ دوستۇم مىنى ۋە دوسلىرىمنى تۇنۇستۇرۇشقا باشلىدى،
- بۇ ياق  ئابدۇۋەلى دەپ بىر دوستۇمنى تۇنۇشتۇردى ..
دوستۇممۇ شوخ بولغاچقىمىكىن مەن ئابدۇۋەلى سىلەر بىلەن تۇنۇشقىنىمدىن خۇشالمەن دەپ ئولتۇردى..
دوستۇم كەينىدىنلا بۇ دوستۇمنىڭ ئىسمى نىجات دەپ تۇنۇشتۇردى...
ئۇ دوستۇم سەل كەم سۆز بولغاچقا  ئادىلا قىلىپ ئورنىدىن تۇرۇپ كۈلۈپ قۇيۇپلا ئولتۇردى.
ئەمدى بۇ قېتىم نۆۋەت ماڭا كەلگەن ئىدى.
-دوستۇم بۇ دوستۇم  بولسا دەپ بىر دەم كۈلۈپ تۇرۋىتىپ ئىسمى دەپ ماڭا قارىدى كۈزىگە قارىسام بىر ئۇيۇن چىقرىش مەخستى باردەك .دىمىسىمۇ بۇ دوستۇم بەك شوخ ھەم ئاقكۈڭۈل ئىدى....
مەن يەنە قىېيىن ئەھۋلغا چۈشۈپ قامماسلىق ئۈچۈن ئۇ دەپ بولغۇچە ھە مىنىڭ  ئىسمىم ئەلى دەپلا ئورنۇمدىن تۇرۇپ كەتتىم. يۈزلىرىم قىزرىپ كەتتى . مەن ئۈزۈمنىڭ قىلغان بۇ ئىشىدىن سەل خىجىل بولدۇم سورۇندا بىر كۈلۈشمەك چىقتى...
دوستۇم چاندۇرمايلا چاتاق يوختۇ دەپ سۇراپ قالدى...
- ۋاي چاتاق يوق جۇما چاتاق بار يەردە بىزمۇ يوق سىلەر بىلەن تۇنۇشقىنىمدىن ئىنتايىن خۇرسەنمەن داۋاملىق ئۇچرىشىشقا نىسىپ بولسۇن دەپ ئولتۇردۇم...
ئۇلارنىڭ ئارسىدىكى مەن بۇ سورۇنغا كىرىپلا كۈزۈم چۈشكەن قىز ماڭا قاراپ مىيقىدا بىرنى كۈلۈپ قويدى. مەنمۇ كۈلۈپ قۇيۇپ  دوستۇمنىڭ ئاغزىغا قاراپ ئولتۇردىم.
دوستۇممۇ كەينىدىنلا تۇرۇپ ۋاي ۋۇي مۇشۇ كاللامنى قانداق قىلسام بۇلا دەپلا  ئۈزىنى تۇنۇشتۇرۇپ كەتتى. دىمىسىۇ ئۇ بىزنى تۇنۇشتۇرىمەن دەپ ئۈزىنى تۇنۇشتۇرمىغان ئىدى...
- مەن چاقچاق قىپلا ۋاي ئاداش سەنمۇ تۇنۇشمامتىڭ بۇلار بىلەن ئەمدى مۇھاببەتلەشمەيدىغانسىلە دەپلا چاقچاق قىلدىم .يەنە بىر كۈلۈشىپ كەتتۇق .
-دوستۇممۇ تەمكىنلىك بىلەن مىنڭ ئىسمىم ئابباس دەپ ئۈزنى تۇنۇشتۇردى.
ۋە يەنە قىز دوستىنى تۇنۇشتۇردى بۇ مىنىڭ قىز دوستۇم ئسمى ئادىلە دەپ تۇنۇشتۇردى.
-ئۇ قىزمۇ  بىز بىلەن بىر قۇر سالاملاشىپ چىقتى، ئادىلە دىگەن بۇ قىز بەكلا چوڭ سۈپەت بولغاچقا كيىۋالغان كىيىملرىمۇ چوڭ سۈپەت ئۈزىگە ياشراتۇرۇپ كىيۋالغانتى.
ئۇ ئارتۇق گەپ قىممايلا  ئۈزىنىڭ دوسلىرىنى ئاددىلا تۇنۇشتۇردى.
-بۇ مىنىڭ دوستۇم ئىسمى نۇرئامىنەم دەپ سېمىزرەك كەلگەن بىر قىزنى كۆرسەتتى .
-ئۇقىز سەل شوخراق ئىدى . ئورنىدىن تۇرۇپ مىنىڭ ئىسىمىم نۇرئامىنەم تۇنۇشقىنىمدىن خۇشالمەن . سىلەر بىلەن ياخشى دوسلاردىن بۇلۇپ ئۈتشىمىزگە تىلەكداشمەن دەپ ئولتۇردى.
-ئادىلە ئارىقىدىنلا جىمغوراق كەلگەن چوڭ سۈپەت بىر قىزنى كۆرسۇتۇپ بۇ قىزنىڭ ئىسىمى پاتىمە دىدى.
-ئۇ ھېچقانداق گەپ قىمماي ئورنىدىن تۇرۇپلا ئولتۇردى.
-ئادىلە دوسلىرىنى  بىر بىرلەپ تۇنۇشتۇرۇپ چىقىتى.ئەڭ ئاخىردا ھېلقى ئۇماق قىزنى تۇنۇشتۇردى بۇ دوستۇمنىڭ ئىسىمى گۈلنىگار دەپلا تۇرشىغا ئۇ قىز  ئىرىنچەكلىك بىلەن ئورنۇڭدىن تۇرۇپ  بىز بىلەن سالاملىشىپ چىقتى.
- گۈلنىگار دىگەن بۇ قىز ئۈزى كىچىك بولغاچقىمىكىن كىچىك باللاردەكلا كىينۋالغان بۇلۇپ بېشىغا پۇسما قىزىل چاپان  كىيۋالغان بۇلۇپ بەكلا ياراشقانتى. تۇنۇشتۇرۇپ بولغاندىن كىيىن  بۇيرۇتقان تاماق   سوغاق سەي قاتارلىقلارمۇ چىقتى بىز بۇ تاماقلارنى يىگەچ  پاراڭلىشىپ ئۇلتۇردۇق .
ئارىدىن سۇرۇن كەيپىياتىنى سەل كۈتۈرمەكچى بۇلۇپ ناخشا-مۇزىكا ئاڭلىغاچ ئۇلتۇرۇپ  ئۇسۇل ئوينىماقچى بۇلۇپ ھېلىقى گۈلنىگار دىگەن قىز تۇرۇپلا ئۇسۇلدىگەننى باشلاپ كەتتى . ئۇ ئۇسۇلغا شۇ قەدەر ئۇستا ئىدىكى كەسپى ئۇسۇلچىلاردىن قېلىشمىغۇدەك ئوينىۋەتتى . ھەممىمىز ئاغزىمىزنى ئېچىپ قاراپلا قالدۇق . بۇ چاغدا كەيپىياتىمىز ئىنتايىن ياخشى ئىدى . مەن بۇ ئۇرۇمچى دىگەن شەھەرگە كەلگەندىن بۇيان بۇنچىلىك خوش بۇلۇپ باقماپتىكەنمەن . ئۇسۇل ئويناۋاتقان گۈلنىگار دىگەن بۇ قىز دىن كۈزۈمنى ئۈزمەيلى ئولتۇردۇم بۇ قىزمۇ بىر قانچە ناخشىغا  ئارقا-ئاراقىدىن ئوينىۋەتتى مەن شۇ ئەسنادا ئۇ قىزغا چاقچاق قىلىپ ئايرىمخانىنىڭ بىر چەتكىرەك تىزىپ قويغان گۈلدىن بىر دەستە ئەكىلىپ ئۇنىغا تۇتتۇم ، ھەممە كۈلشىپ كەتتى بۇ قىز ئۈزىمۇ كۈلۈپ كەتكەتتى . ئۇمۇ ھېرىپ قالدى بولغاي جايىغا كىلىپ ئولتۇردى،
سۇرۇن شۇ تەرىقىدە ئۈچ-تۆت  سائەت داۋمللاشتى بۈگۈن ھەمىمىز ئىنتايىن خۇشال ئىدۇق....
- ئادىلە  بىز ئەمدى كىتەيلى ۋاقىتمۇ بىر يەرگە بېرىپتۇ دىگۈچە ئولتۇرغان ئىدق.
خۇشاللىقىمزدا ۋاقىتنىڭ بۇنچە تىز ئۆتكىنىمۇ بىلمەيلا قاپتىمز...

  گەرچە قەستەن ئاستا ماڭساممۇ بۇ مەكتەپكىمۇ بىر دەمدىلا كىلىپ قالدۇق . خوشلىشىش ئالدىدا گۈلنگارنىڭ  ئاشۇ ئوتلۇق كۈزلىرنى يەنە بىر قېتىم  كۈرۋېلىش مەخسىتىدە ئونىڭغا ئوغۇرلۇقچە قارىدىم . ئۇ قىزمۇ قاراپ  قول پۇلاڭلىتىپ مەكتپىگە كىرىپ كەتتى . خيالىم بىرلا ئۇنى يەنە قايتا كۈرەلەرمەنمۇ دىگەن خىيالدىلا ئىدى... چۈنكى ئەمدى ئۇ قىزنى يەنە كۈرەلىشىمگە ئۈزۈممۇ ئىشەنمەيتىم . بولدىلا جىق ئويلىماي دەرتنى بەرگەن خۇدايىم داۋاسىنىمۇ بىردۇ دەپ ئۈزۈمگە تەسەللى بىرىپ شۇ خىياللار بىلەن  ياتاققا قايتىش ئۈچۈن ئابباس بىلەن ياتاققا قاراپ يول ئالدىم...
