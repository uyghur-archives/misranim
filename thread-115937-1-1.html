ئەسسالامۇ ئەلەيكۇم مىسرانىم مۇنبىرىمىزنىڭ ئەزىز ئىشتىراكچىللىر ھەممىڭلارغا كەمىنىدىن ئوتلۇق سالام، ئاللاھ ھەممىڭلارنى ئۆز پاناھىدا ساقلىسۇن( ئامىن).

تارىخ ئۇ ئۆتمۈشكە ئايلاندى، شۇنداق، لېكىن ئۇنىڭدا تالاي ئەجدادلارنىڭ شانلىق ئىزلىرى قالغان ، شۇ ئەجدادلارنىڭ قان تەرى ئارقىلىق بىزنىڭ بۈگۈنكى بەختىيار كۈنلىرىمىز بارلىققا كەلگەن، ئۇلارنى ئەسلەش، ئۇلارنى كېيىنكى ئەۋلاتلارغا تۇنۇتۇش ۋە ئىش-پائاليەتلىرىنى خاتىرە قىلىپ ساقلاپ، كەلگۈسى ئۈچۈن ئۆرنەك قىلىپ ساقلاش بىزنىڭ باش تارتىپ بولالمايدىغان مەسئۇليىتىمىز ئەلۋەتتە..

ئۇچۇر دەۋرىنىڭ تەرەققىياتى تولىمۇ تېز بولۋاتقان بۇ دەۋىردە، تۈرلۈك ئېلىكتورۇن مەھسۇلاتلىرى بىزنىڭ تۇرمىشىمىزغا قۇلايلىقلارنى ئېلىپ كەلدى، جۈملىدىن رەسىملەرنى خاتىرە قىلىپ ساقلىيالايدىغان بولدۇق، بۇرۇنقىلار نەچچە سائەتلەپ سىزىپ چىققان بولسا، بىز ھازىر بىر نەچچە مىنىۇت ئېچىدە تارتىپ چىقىرالايمىز...بۇ ئەجدادلارنىڭ سىماسىنى ساقلاپ قېلىشتا تولىمۇ چوڭ رول ئوينايدۇ..

يېقىنقى بىر نەچچە يىلدىن بۇيان، تور دۇنياسىدا ئۇيغۇر مەشھۇرلىرىنىڭ رەسىملىرى ئېقىپ يۈردى، بۇلارنىڭ ئىچىدە تېپىش ناھايىتى قېيىن بولغان، قىممەتلىك رەسىملەرمۇ بار ئەلۋەتتە.

مەن بۇرۇندىن قەدىمىي نەرسىلەرنى يىغىپ ساقلاشنى ياخشى كۈرەتتىم، كومپىيۇتېر بىلەن تۇنۇشقاندىن كېيىنمۇ، قانداق ياخشى نەرسىلەر ئۇچرىسا يىغىپ ساقلىۋالاتتىم، شۇنداق قىلىپ ئۇيغۇر مەشھۇرلىرى توغۇرلۇق نۇرغۇن رەسىملەرنى توپلىۋالدىم، قارىسام مۇنبىرىمىزدە بۇ مۇكەممەل بولغان  رەسىم توپلىمى يوقكەن، شۇڭا بىر نەچچە ئايدىن بۇيان كومپىيۇتېردىكى رەسىملەرنى بىر يەرگە يىغىپ، توپلاپ تۈرگە ئايرىپ چىقتىم، ھەمدە سېلەرنىڭ كۈرۈپ بېقىشىڭلار ئۈچۈن تېما قىلىپ يوللاش قارارىغا كەلدىم، قېنى ھۇزۇرلانغايسىلەر!
