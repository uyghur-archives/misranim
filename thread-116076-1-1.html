ئەسسالامۇ ئەلەيكۇم! شىنجاڭ تىلۋىزىيە ئىستانسىدىكى خادىملار،ياخشىمۇسىلەر!
ئىلگىرى بىرقىسىم تورداشلارنىڭ شىنجاڭ تىلۋىزىيە ئىستانسىغا تەكلىپ يازغانلىرىنى  كۆرۈپ نايىتى خۇرسەن بولدۇم،ئەمما شىنجاڭ تىلۋىزىيە ئىستانسىدىن سادا چىقمىغانلىقىغا قاراپ ھەم ئەپسۇسلاندىم. گەرچە بۇ تەكلىپكىمۇ يەنىلا جاۋاپ چىقمايدىغانلىقىنى،«ئات ئايلىخانغا،يول سارىخانغا»كىتىۋىرىدىغانلىقىنى كۆڭلۈم تۇيۇپ تۇرسىمۇ، مىنىڭ ۋېجدانىم مىنى ئارامىدا قويمىدى.مىنىڭ  شىنجاڭ تىلۋىزىيە ئىستانسىغا بىرىدىغان تەكلىپىم پەقەتلا بىرتۈرلۈك:
«كىنو-فىلىملەرنىڭ سەل-پەل ياخشىراقلىرىنى قويۇپ بەرسەڭلار»
ھەققانىي گەپ قىلغاندا ھازىر ئۇيغۇر تىلۋىزىيەسىدە بىرىلىۋاتقان تەرجىمە فىلىملەرنىڭ ئۇيغۇر خەلقىگە چىقانداق پايدىسى ياكى قىممىتى يوق بولۇپلا قالماي،مۇتلەق كۆپ قىسىم فىلىملەر ئۇيغۇر ئىدولوگىيىسىگە،مىللى مەدەنىيەتكە، خەلقىمىزنىڭ ئۆرپ-ئادىتىگە، ئومۇمى خەلق ئەخلاقىغا زىيانلىق. ئۇيغۇرچىدا بىرىلىۋاتقان ئازساندىكى تەرجىمە فىلىملىرىنى ھىساپقا ئالمىغاندا مۇتلەق كۆپ ساندىكىلىرى ھاياسىزلىقنى، ئىتقادسىزلىقنى،ئەنئەنىگە قارشى چىقىشنى  تەشەببۇس قىلىۋاتىدۇ.
بىز بۇ فىلىملەرنى كۆرمەيلى دېسەك، ئىشتىن چۈشكەندىن كىيىنكى ئاشۇ بوش ۋاقىتتا بىز ھاردۇقىمىزنى چىقىرىشىمىز كىرەك ئىدىغۇ! بۇنى كۆرەيلى دېسەك،نەچچە مىڭ سوملۇم تىلۋىزور ئەمدە دىلبۇزۇرغا ئايلانماقتىغۇ!
ئەمەلىيەتتە، خەنزۇچە قاناللاردا خىلى-خىلى ياخشى فىلىملەرنى كۆرسىتىۋاتىدۇ، چەتئەلنىڭ (مەسلەن تۈركىيەنىڭ) نۇرغۇن ياخشى فىلىملىرى پىلاستىنكىلارغا ئايلاندۇرلۇپ خەلقىمىزنىڭ ياقتۇرۇپ كۆرىشىگە ئىرىشىۋاتىدۇ. ياخشى فىلىملەر يوقمىكى دېسە،ئەمەلىيەتتە  بار، لىكىن نېمە ئۈچۈن تىلۋىزىيە ئىستانسىلىرى بۇلارنى تەرجىمە قىلىپ بەرمەي، نەدىكى نامى چىقمىغان،ئەھمىيىتى يوق فىلىملەرنى تەرجىمە قىلىپ، ھەم خەلقنى ھەم ئۆزىنى ئاۋارە قىلىپ يۈرىيدۇ. بەزىدە بىچارە ئاۋاز ئارتىسلىرىمىزغا شۇنچە ئىچىم ئاغرىپ قالىدۇ: شۇنچە لاۋزا،مەنىسىز فىلىملەرگە ئاۋاز بىرىۋاتقىنىدا، مۇشۇنىڭ ئورنىغا ئاۋۇنى تەرجىمە قىلساق بولمامدۇ،دەيدېغان ئۇيغۇردىن بىرەرسىمۇ چىقمىغانمىدۇ؟
تەرجىمە فىلىم بۆلۈمىدىكى قىرىنداشلىرىم، سىز تەرجىمە قىلغان ئاشۇ فىلىمنى ئون مىليۇنغا يىقىن ئۇيغۇر،نەچچە يۇزمىڭ قازاق ۋە باشقا ئۇيغۇر تىلىنى چۈشىنىدىغان قىرغىز، تاجىك قاتارلىق قىرىنداشلىرىمىز كۆرىشى مۇمكىن، سىز پەقەت پائاش ئۈچۈنلا ئىشىلىمەيسىز،ئېسىڭىزدە بولسۇن تىلىمىزغا تەرجىمە قىلىنغان فىلىملەر ئومۇمى خەلقنىڭ ساپاسىغا، جامەئەت پىكىرى يۆنىلىشىگە تەسىر كۆرسىتىدۇ.
سىلەردىن شۇنى ئۆتۈنىمىزكى،جۇڭگۇ ئىچى ۋە سىرتىدا مۇكاپاتقا ئىرىشكەن،سەرخىل فىلىملەرنى
تەرجىمە قىلغان بولساڭلار،ھېچ بولمىغاندا خەلقىمىزنىڭ ئىددىسىگە زىيانكەشلىك قىلىدىغان فىلىملەرنى تەرجىمە قىلىپ ئۆزۈڭلارنى ئۇپراتمىساڭلار!
مەدەنىيەتلىك ئائىلە بىلەن مەدەنىيەتسىز ئائىلىنىڭ پەرقى-ئۆيىدە كىتاب جازىسىنىڭ بار يوقلىقىدا!
