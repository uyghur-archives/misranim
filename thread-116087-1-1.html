مۇشۇ كۆكتات ۋە ھەر خىل ئۈندۈرمىلەرنى بازاردىن ئالغاندا بەزىدە سەل غەيرى پۇراق بولسىلا ئالغۇڭىز كەلمەيدۇ ياكى ئالغان تەقدىردىمۇ تاشلىۋىتىسىز، چۇنكى ئۈندۈرمىنى ئىشلەپچىقىرىدىغان بەزى قارا نىيەتلەر ئۈندۈرمە سۈيىگە كىچىك تەرىتىنى ئىلەشتۈرۈپ، يەتمىسە ئۈسۈشنى تىزلىتىدىغان قانونسىز خىمىيلىك دورىلارنى ئىشلىتىپ ئىستىمالچىلارنى ئالدايدىغان ئىشلار بار.
ماۋۇ تاقىۋەتكەن زاۋۇتلارنىڭ رەسمىيتى كەملىكىدىن ئەمەس بەلكى سۈپىتى ناچار ئۈندۈرمە چىقارغانلىقتىن تاقىۋىتىپتۇ؛http://www.boosj.com/1423298.html#cmt
  ئۆزىمىز ئۈندۈرمە ئىشلەپ بازارغا تەمىنلىسەكمۇ بىر تەرەپتىن كىرىم قىلساق، بىر تەرەپتىن قېرىنداشلېرىمىزنى پاكىز ئىستىمال قىلىش ئىمكانىيتىگە ئىگە قىلالايمىز.
ئادەتتىكى ئۆيلەردە قىلساقمۇ بۇلىدۇ، دىئامىتىرى 40-50 سانتىلىق توربىنى ئۇتتۇردىن يېرىپ ئۇقۇر تەييارلاپ، ئۆينى ئايلاندۇرۇپ نەچچە قەۋەت تىزىپ، ھەر بىر قەۋەتنىڭ باش-ئاخىرىنى ئايرىپ، باش تەرپىدىن ئازراق سۇ چۈشۈپ تۇرىدىغان قىلىپ قۇيىمىز،ئاخىرقى ئورۇننىڭ مۇۋاپىق ئىگىزلىك ئورنىدىن تۈشۈك ئىچىپ نەيچە ئارقىلىق ئاستىنقى قەۋىتىنىڭ باش تەرپىگە سۈ چۈشۈپ كىتىدىغان قىلىپ قۇيىمىز. بالدۇر تەييارلىق قىلىپ كۆپلەپ چىقىرىدىغانلار ئالدى بىلەن ئىككى مىتىر كەڭلىكتە 20دىن100 مىتىرغىچە ئىككى مىتىر چوڭقۇرلۇقتا قېزىپ، سۇلياۋ يۇپۇق بىلەن نەچچە قەۋەت ئارلىق قالدۇرۇپ يىپىپ، ئىچىگە يەنە ئىككى يانغا ئۇزۇن ھەم نەچچە قەۋەت ئۇقۇرنى ئورۇنلاشتۇرۇپ سۇ ئۈستى قەۋەتلەردىن بىرمۇ-بىر ئاستى قەۋەتلەرگە چۈشىدىغان قىلىپ تەييارلايمىز، ئۇتتۇردا 70 سانتىلار ئادەم ماڭىدىغان يول قالسۇن. ئۇقۇرغا ماش، سېرىق پۇرچاقلارنى سىلىپ سۇ تولدۇرغاندىن كىيىن، ئۈستىنكى باش تەرەپتىن ئازراق سۇ داۋاملىق سىرغىپ چۈشۈپ تۇرسا سۇ ئالماشتۇرۇش ئەمەلگە ئاشىدۇ، بۇنىڭدا ئۈندۈرمە پۇراپ قالمايدۇ. ئۇتتۇرلۇق تەييار بولغان يېرىنى بازارغا بېرىۋىتىپ، ئورنىغا يەنە ماش-پۇرچاق سېپ قويساق كۈندە بازارغا ئۈلگۈرىدۇ(سۇنىڭ باش-ئايىقى ۋە ئورۇننىڭ يۇقىرى تۈۋەنلىكى سەۋەبىدىن تەييار بۇلۇش ۋاىتى پەرىقلىنىپ قالىدۇ). دىققەت قىلىدىغىنىمىز ئۆي ياكى ئشلەپچىقىرىش مۇھىتى پاكىز بولسۇن، ھەم تىمپىراتورا 15 تىن 25 گىرادوسقىچە ئارلىقتا بولسۇن، بەك ئىسسىق بولسا ئاسان بۇزۇلۇپ قالىدۇ، بەك سۇغۇق بولسا بەك ئۇزاق ۋاقىتتا ئۈندۈرمە بۇلىدۇ. ئەگەر ئازراق غەيرى پۇراق بولغاندا ئايلىنىدىغان سۇنىڭ مىقتارىنى كۈپەيتىپ قويساق بۇزۇلمايدۇ. ئاستى قەۋەتكە چۈشكەن سۇ، بىكار بولغان سۇنى بىلىق قاچىسىغا سالىدىغان كىچىك سۇ پومپىسى بىلەن قەرەللىك ياكى سۇ ئورنىغا ئاساسەن كونتۇروللىنىدىغان قىلىپ سىرىتقا چىقىرۋىتىدىغان قىلىپ قويسا بولىدۇ.
بەك قىممىتى بولمىسىمۇ ئاز دەسمايە بىلەن ئىش باشلاشتا بۇمۇ بۇپ قالىدۇ. ئازوت، دورا ياكى سۈيدۈك ئىشلەتمەي تەبئى يۇسوندا قىلغان ئۈندۈرمىگە ئالدى بىلەن شۇ يەرلەردىكى خەنزولار ئۈگىنىپ كىتىشى تۇرغان گەپ. چوڭ قىلسىڭىز پارچە ساتقىلى ئىمكانىيتىڭىز بولمىسا سەي دۇكانلېرىغا توپ تەمىنلەڭ. ئىقتىسادىڭىز يار بەرمەي كىچىكرەك قىلسىڭىز پارچە سىتىڭ. ئىش قىلىپ يىمەكلىك سودىسى زىيان قىلىشى بەك ئاز ئۇچرايدۇ...
مەنمۇ قىپ باقماقچى بۇلۇپ، ئاز قىلسام كۆپ پايدا يوق، كۆپ قىلسام يەپ كىتىدىغان ئادەم يوق دەپ قىلمىغان، بۇنىڭدا ئىقىپ چۈشىدىغان سۇنىڭ تىمپىراتورىسىنى 25 گىرادوس ئەتراپىدا تەڭشىسە باشقا ئىسسىتىش ئەسلىھەسى ئورناتمىسىمۇ بۇلىدۇ، ھەم بۇنداق قىلغاندا ئۈندۈرمە ئىتتىك تەييار بۇلىدۇ. تېخنىكا كىتىدىغان ئىشلارغا دېھقانلېرىمىزنىڭ ئىسىلۋىلىشىنى تەلەپ قىلغىلى بولمىغاندىكىن مۇشۇنداق يوللارنى كۆرسىتىپ قويساق بىرە مىڭ ئادەم ئۆزىگە ئىش تاپالايدۇ دەپ ئويلىدىم.
