خېلى بۇرۇنلا بىر قىسىم تورداشلار بۇنى سورىغان، شۇڭا تورداشلارنىڭ تەلۋى بىلەن يازدىم. تۇلۇقسىزنىڭ خىمىيە دەرىسلىكىدە مىتان گازى تۇغرىلىق تەپسىلى تۇختالغان بۇلۇپ، بۇ يەردە قىسقىچە پاتقاق گازى كۆلچىگى ياساش تۇغرىلىقلا تۇختىلاي.
ئادەتتە 3 كىلو قاتتىق ئۇتۇننىڭ چىقىرىدىغان ئىسسىقلىقى بىر كىلو كۈمۈرگە تەڭ بۇلىدىغىنىنى بىلىمىز، بۇ خىل ئۇسولدا پايدىلانغاندا بولسا بىر كىلو ئۇغوتتىن تەخمىنەن 2 كىلو كۈمۈرنىڭ ئىسسىقلىقىغا ئېرىشىپلا قالماي يەنە قالدۇقى يۇقىرى سۈپەتلىك ئۇغوتقا ئايلىنىدۇ، تەركىبى ئۇخشىمىغان ماتىريالغا ئاساسەن ئۇخشاش بولمىسىمۇ ئەمما سۇغا سىڭدۈرۈلگەن ئازوتنىڭ خىمىيۋى ئۇغوتتىكى ئازوتتىن چوڭ پەرقى ،يەرنى قاتتىقلاشتۇرۋەتمەيدۇ، ھەم باشقا تەركىپلەرنىڭ ئۈسۈملۈككە پايدىسى ئىنتايىن يۇقىرى. تەركىبلېرىنى ئىسىمدە تۇتىۋالماپتىمەن، ‹‹ئۈسۈملۈكنىڭ ئاجايىپ شىپالىق دورىسى›› دەپ تۇرايلى.

بۇرۇنلاردا خىش ۋە سىمونىتتىن كۆلچەك ياساپ ، ئىچ تەرپىگە شام(موم) نى يالىتىپ سۇ سىڭمەسلەشتۈرىدۇ، يەنى كۆلچەك تىمىنى ئوت بىلەن قىزدۇرۇپ(ماشنىلارنى قىزىتىشتا ئىشلىتىدىغان پۈركىگۈچلۈك قىزىتقۇچ بىلەن)، مومنى تامغا سۈركىسە سىمونىت سۇۋاقنىڭ يۈزىگە شام ئېرىتمىسى سىڭىپ، بىر قەۋەت سۇ ئۈتكۈزمەس قەۋەت پەيدا بۇلىدۇ. ئاندىن ئۈستى قىسمىغا ھىم ئىتىلىدىغان قاپقاق ئورنىتىمىز، نەيچە ئارقىلىق ئىشلىتىدىغان ئەسۋاپلارغا ئىچىدىكى گازنى يەتكۈزىدىغان لىنىيە تەييارلاپ،  ئاندىن ئىچىگە %70 سۇ تولدۇرۇپ، دەرەخ يۇپۇرماقلېرى، ئوت-چۆپلەرنى ياكى قوي-كالا ئۇغوتلېرىنى سىلىپ بىر مەزگىل(15كۈندىن 1 ئايغىچە) تۇرغۇزسا مىتان گازى ھاسىل بۇلىدۇ. بىز بۇ يەردە يەنە زىيادە گازنى بىسىملىق قىلىپ ساقلاش ۋە كەڭلىكى 8 مىتىرلىق قىلىنراق سۇلياۋنى ئۇتتۇردىن يارمايلا ئىككى بىشىنى پۇختا ،ھىم قىلىپ يىپىشتۇرۇپ، نەيچىنى ئۇلاپ قويساق، شۇ خالتا ئىچىگە كىرىپ، ساقلىنىدۇ، ئەتراپىنى ياخشى تۇسۇش، بىخەتەرلىككە دىققەت قىلىش لازىم）يەنە مۇشۇ ئاددى سۇلياۋنىڭ ئىچى-تىشىنى باشقا نەرسە بىلەن قاپلاپ پاتقاق گازى كۆلچىكى ياسىسىمۇ بۇلىدۇ).
قىش-ياز پەسىللېرىدە مىتانغا ئايلىنىش ئۈنىمى ئۇخشاش بولمايدۇ، بۇ بولدۇرىدىغان خىمىرنى ئىسسىق يەرگە قويسىڭىز تىز بولغىنىدەك ئىش.
بىز لاھىيەلەشتە ئېھتىياجىمىزغا كىتەرلىك كۆلچەكنى قىش پەسلىدىكى ۋاقىتنى ئۆلچەم قىلىپ گاز ئاز ھاسىل بولغاندا يەنىلا ئېھتىياجىمىزنى قامدايدىغان قىلىپ بىكىتىمىز، نەچچە يىل بۇرۇن بۇنىڭ فورمۇلالېرىنى بىر زاۋۇت ماڭا ئەۋەتكەن، قاتتىق دېسكىنى نەچچىنى ئالماشتۇرۇپ كەتكەچ ساقلىۋالغىنىممۇ ھازىر يوق، ناۋادا بىر ماتور بىلەن ھەم توك چىقىرىپ، ھەم ماتورنىڭ ئاۋازىنى ئۈتكۈزمەي ئىسسىقلىقىدىن پايدىلىنىدىغان قۇرۇلما تەييارلىۋالساق،بىر ۋاقىتتىلا ئىسسىنىشنىمۇ ئەمەلگە ئاشۇرىمىز دىسەكمۇ ئىككى يۈز كۇۋادىرات كۆلچەك تەييارلىساق يىتىدۇ. كۆلچەك چوڭقۇرلىقى 3 مىتىر بۇلىدۇ، ماتىريال تاشلىغاندا بىر قىتىمدا بەك كۆپ تاشلىۋەتمەي، دەسلەپتە بىر مىتىر قىلىنلىقتا تاشلاپ، ئۈستىدىن بىسىپ سۇغا چۈكتۈرۈپ تۇرىمىز، ھەم بىسىپ تۇرىدىغان تاختاينى ھەر كۈندە ئىككى-ئۈچ قىتىم سىلكىنىپ تۇرىدىغان قىلىپ قويساق، ماتىريال ئارلىقلېرىدا گاز قاپلىشىۋالغان بولسا ۋاقتىدا بوشلۇققا لەيلەپ چىقىدۇ، گاز قاپلىشىۋىلىپ سۇدىن ئايرىلغان ماتىريالمۇ قايتا سۇ بىلەن ئۇچرىشىپ، داۋاملىق گاز ھاسىل بۇلۇپ تۇرىشى كاپالەتلەندۈرىلىدۇ.
نەچچە يىللاردىنلا بىر خىل يۇپۇق يوللۇق يۇمشاق خالتىلىق گاز كۆلچەكلېرى ئۇيغۇر سودىگەرلېرىمىز تەرپىدىن تورلاردا ئېلان قىلىنىپ سىتىلىۋاتقان ئىدى. ئېھتىياجىمىزغا ماس كەلسە شۇنى سىتىۋىلىپ ئىشلەتسەكمۇ بۇلىدۇ.
