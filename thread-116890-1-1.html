                                         بۇ دۇنيا

                  شۇنداق قىزىق ئىكەن بۇ دۇنيا ،
                  ھايات تۇرۇپ ئۆلگەنلەرمۇ بار .
                  بۇ جاھاندىن كەتسىمۇ بىراق ،
                  ھايات ياشاپ يۈرگەنلەرمۇ بار.

                 يۆلىمەسەستىن يىقىلغانلارنى ،
                 زاڭلىق قىلىپ كۈلگەنلەرمۇ بار .
                 چاپىنىدا تەرلەپ ئۆزگىنىڭ ،
                 بەخت پەيزىن سۈرگەنلەرمۇ بار.

                ھاجىتىدىن چىقماي مۇھتاجنىڭ ،
                ئاڭا قاپاق تۈرگەنلەرمۇ بار .
                 پۇل - ھوقۇقى بولغانلارغىلا ،
                 دائىم كۆڭۈل بۆلگەنلەرمۇ بار .
               
                 دوست تۇتماستىن مالۇ - دۇنيانى،
                 يوقسۇللارغا بەرگەنلەرمۇ بار .
                 قان - تەر تۆكمەي تۆھپىكارمەن دەپ ،
                 مەيدىسىنى كەرگەنلەرمۇ بار .

                ئېتىزىغا رەڭدار ھاياتنىڭ ،
                گۈل - چېچەكلەر تەرگەنلەرمۇ بار .
                خازان قىلىپ گۈلزار پەسلىنى ،
                ئۇرۇق چاچماي كەلگەنلەرمۇ بار .

بۇدۇنيا

ھەممىگە  باراۋەر  ئەمەس  بۇ  دۇنيا،

تەقدىرگە قوشۇلسۇن توغرابىر تەدبىر.

ئىزلىگىن ئىنتىلگىن ھاياتتىن قىممەت،

بىر ئۆلۈم بىر كۆرۈم  دېمە  بەرىبىر.

ھەممىگە  باراۋەر  ئەمەس  بۇ  دۇنيا،

ھاياتلىق سەپرىدە كۆيىدۇ يۈرەك .

ئەل بىزگە بولغۇسى ئادىل تارازا ،

غېزى غاز ئۆردىكى بولۇر سەم ئۆردەك.

ھەممىگە  باراۋەر  ئەمەس  بۇ  دۇنيا،

بەزىلەر گۆش يېسە بەزى يەيدۇ مۇش.

بەزىلەر بولسا ھەم ئۆيدىكى كەپتەر ،

بەزىلەر بولىدۇ كۆكتە ئۇچقان قۇش.

ھەممىگە  باراۋەر  ئەمەس  بۇ  دۇنيا،

كۈلىمىز يىغلايمىز ئاخىر كېتىمىز.

دۇنيادىن ئىزلەيمىز ئىنسانلىق قىممەت،

بىز كەتسەك قالىدۇ  نەۋرە ئەۋرىمىز.

ھەممىگە  باراۋەر  ئەمەس  بۇ  دۇنيا،

يوللىرى مۇرەككەپ گويا تىك قىيا .

كېتىمىز ھەممىمىز بالدۇر ۋە كېيىن،

ئەس قاتماس ئۇنىڭغا تۈگمەس مال-دۇنيا.

ھەممىگە  باراۋەر  ئەمەس  بۇ  دۇنيا،

بىزدىكى بەش بارماق بولمىغاندەك تەڭ.

بىر نىشان بىر غايە بولسا  قەدەمدە،

بۇ دۇنيا شۇندىلا گۈزەل بولۇر ئەڭ.

ھەممىگە  باراۋەر  ئەمەس  بۇ  دۇنيا،

بىر ئاكام  دېگەنتى تالاش  دۇنيانى .

تالىشاي دېگەندە  بىر ئاكام دېدى ،

كىملەردىن قالمىغان تالاشما ئۇنى.

ھەممىگە  باراۋەر  ئەمەس  بۇ  دۇنيا،

تەقدىرۋە تەدبىرى ئۆزنىڭ قولىدا.

يىغلامدۇ كۈلەمدۇ بەلگىلەر ئۆزى ،

ھاياتقا باسقان شۇ ھەربىر يولىدا.

ھەممىگە  باراۋەر  ئەمەس  بۇ  دۇنيا،

يوللىرى مۇرەككەپ ئەگرى ھەم توقاي.

گاھىدا تالىشىپ گاھى تالاشماي ،

ئەجدادلار ئىزىدا بىر مېڭىپ باقاي.

ھەممىگە  باراۋەر  ئەمەس  بۇ  دۇنيا،

مۆجىزە سىرلىرى كۆپتۇر كارامەت.

گەر توغرا بولمىسا باسقان قەدەملەر ،

بۇدۇنيا  قىلمايدۇ  ساڭا  ساخاۋەت .

ھەممىگە  باراۋەر  ئەمەس  بۇ  دۇنيا،

توغرىمۇ ئېيتىلغان كاككۇك سۆزلىرى.

مۇھىمى ئۇيغۇرغا ھەر بىر ئىنسانغا،

ئاللاھدىن تىلەيمەن بەخىت كۈنلىرى.

باغراش كاككۇكى
سۆيۈلۈشنى كۈتۈپ يۈرۈپ ، كۈتۈشلەرنى سۈيۈپ قالدىم . (تۈرۈك ھېكمەتلىرى)

بۇ دۇنيا
گۆھەرقىز ئىبرايىم
كىسەل بولساڭ تاتلىق بۇ دۇنيا،
يىتىم بولساڭ ياشلىق بۇ دۇنيا.
سىنى سىنار ھەر بىر قەدەمدە ،
بىلىمدارغا ئامراق بۇ دۇنيا.
   
ئازاپ چەكسەڭ ئاچچىق بۇ دۇنيا،
كۈلۈپ باقساڭ تاتلىق بۇ دۇنيا.
تولغان چەكسىز قىسمەت سىرىغا،
ئىنسان ئۈچۈن سىرلىق بۇ دۇنيا.
      
ۋىجدانلىققا رەزىل بۇ دۇنيا،
ئادالەتكە ئادىل  بۇ دۇنيا.
يولى داغدام ئوچۇق توساقسىز،
بايلار ئۈچۈن گۈزەل بۇ دۇنيا.
     
ئۆلۈك ئۈچۈن ئۆلگەن بۇ دۇنيا،
تىرىك ئۈچۈن كۈلگەن بۇ دۇنيا.
ھاياتلىقنى سۆيگەن كىشىگە،
گۈزەللىكنكە تولغان بۇ دۇنيا.
      
كۆز يىشىغا تويغان بۇ دۇنيا،
غەم قايغۇغا چۆمگەن بۇ دۇنيا.
ئاجىز كەلسەڭ قىستار ئۆلۈمگە،
جاندىن تويساڭ ئۇزۇن بۇ دۇنيا.
         
بولساڭ بىخىل تاردۇر بۇ دۇنيا،
قولى كەڭگە ياردۇر بۇ دۇنيا.
سىناق ئىلىپ ھەر بىر قەدەمدە،
سىنى ئىلغاپ سىنار بۇ دۇنيا.
      
نەزەر سالساڭ شاخمات بۇ دۇنيا،
يىڭىلگەنگە پەريات بۇ دۇنيا.
ئاجىز ئۈچۈن بولۇپ رەھىمسىز،
كۈچلۈكلەرگە قانات بۇ دۇنيا.
        
ئىنسانلارغا پانا بۇ دۇنيا،
غەزەپلەنساڭ مىنا بۇ دۇنيا.
گۇۋاچىسى سانسىز تىلسىمنىڭ،
سۆزلىيەلمەس گاچا بۇ دۇنيا.
      
يىتىملەرگە ئازاپ بۇ دۇنيا،
