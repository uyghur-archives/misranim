ئۈرۈمچى كوچىلىرىدا بولۇپمۇ مىللىيلار كۆپرەك ئولتۇراقلاشقان رايۇنلاردا 30 ياشلار چامىسىدىكى بۇ يىگىت تۇرۇپ -تۇرۇپ پەيدا بولۇپ قالاتتى .
تەقى -تۇرقى كېلىشكەن، سىرىتقى قىياپىتىگە تولىمۇ ئەھمىيەت بېرىدىغان ،بويى ئېگىزرەك ، ئۆزى سەل ئۇرۇق ، قاڭشارلىق چېچى بۈدۈرگە مايىل بۇغداي ئۆڭ كەلگەن ، چىرايىدىن ئاق كۆڭۈل ۋە سەمىمىيلىكى چىقىپ تۇرىدىغان بۇ يىگىتنى تونىغان بىلگەنلەر ھايات لېيفىڭ دەپ ئاتىشاتتى .
دېمىسىمۇ بۇ يىگىت ياخشى ئىش قىلغاندا كىملىك ۋە ئادىرسىنى ئەسلا ئېيتمايتى . ئۇ كىچىكىدىنلا باشقىلارغا ياردەم بېرىشنى ئۆزىنىڭ بۇرچى دەپ بىلگەن بولۇپ ئاجىز ، يىتىم ، بىچارە ، يوقسۇل ، مۇھتاج ۋە قىيىنچىلىقتا قالغانلارغا قولىدىن كېلىشىچە ياردەم قىلاتتى . دىلى يۇمشاق ئاقكۆڭۈللىكىدىن دەرتمەنلەرنىڭ دەردى ئۈچۈن ئازاپلىناتتى ، كۆز يېشى ئۈچۈن كۆز ياش قىلاتتى . يىغىپ كەلگەندە بىلگەنلەرگە نىسبەتەن پەيغەمبەر ئەلەيسھىسسالامنىڭ پەزىلەتلىك ساھابىلىرىنى ئەسلىتىپ قوياتتى .
ئۇ بۈگۈن بىر ئىش ئۈچۈن سەنشىخاڭزىغا كېلىپ بىر تونىشى بىلەن ئۇچىرشىپ قېلىپ پاراڭلىشىپ تۇرغان ئىدى تۇيۇقسىز كۆزى 70-75 ياشلار چامىسىدىكى بىر ياشانغان ئايالغان چۈشۈپ قالدى ، موماينىڭ بېشىغا كۈن چۈشكەنلىكى چىرايىدىن مانا مەن دەپ چىقىپ تۇراتتى .
يىگىت موماينى بىر ھازا كۈزەتكەندىن كېيىن قيىىنچىلقتا قالغان بولىشى مۇمكىن دەپ جەزىملەشتۇردى-دە، تونىشىدىن ئايرىلىپ دەرھال موماينىڭ يېنىغا بېرىپ سالام-سائەتتىن كېيىن ھال -ئەھۋالىنى سورىدى ،موماي يىگىتنى يات كۆرمەي بېشىغا كەلگەن قىيىنچىلىقلارنى قىسقا ئەمما چۈشنىشلىك قىلىپ سۆزلەپ بەردى .
ئەسلى موماي قەشقەر پەيزىۋاتتىن بولۇپ يولدىشى بالدۇرلا باقى دۇنياغا بېرىۋالغانلىقتىن ئۈچ قىزى بار لېكىن ئوغلى بولمىغان بۇ موماي يولدىشى تۈگەپ كەتكەن نەچچە ئون يىل مابەينىدە دەردىنى ئىچىگە يۈتۈپ كېلىۋاتقان بولسىمۇ < ئۆلگەننىڭ ئۈستىگە تەپمەك > دېگەندەك يولدىشى تەرىپىدىن خېتى بېرىلگەن ، نارىسىدە باللىرىنى ئېلىپ ئانىسىنىڭ ئۆيىدە تۇل ئولتۇرۇۋاتقان كىچىك قىزى سەل غەلىتىرەك بىر كېسەلگە گىرىپتار بولۇپ قېلىپ كېسىلى ئېغىرلاپ كەتكەنلىكتىن داۋالتىتىشقا ئامالسىز قالغان ئانا ئائىلىسىدىكى قىيىنچىلىقلار ئاز كەلگەندەك بۇ بەخىتسىزلىككە دۇچ كېلىپ ئېغىر دەرىجىدە ئىقتىسادقا مۇھتاج بولغانلىقتىن ئۆز يۇرتىدا ياردەمگە ئېرىشەلمەي بىرەر چىقىش يولى تېپىلىپ قالار دېگەن ئۈمۈدتە ئۈرۈمچىگە كەلگەنىكەن .
موماينىڭ سۆزلىرىدىن سەمىمىيلىك چىقىپ تۇراتتى يىگىت چوڭقۇر بىر ئۇھ تاتتى .ئارقىدىنلا ئىككىلەنمەستىن يانچۇقىدىن 1000 سوم پۇلنى چىقىرىپ :
-چوڭ ئانا خاتىرجەم بولۇپ ئاللا سىزنى ئۈمۈتسىزلەندۇرمەيدۇ، شەھرىمىزدىمۇ ئاقكۆڭۈل ساخاۋەتچى كىشلەر ئىنتايىن كۆپ ئۇلار سىزنى ياردەمسىز قويمايدۇ ، بۈگۈن مەن يېنىمغا كۆپ پۇل سېلىۋالماپتىمەن ئاز بولسىمۇ بۇ پۇلنى ئالغاچ تۇرۇڭ قالغىنىغا مەسچىتتىكى جامائەتتىن ياردەم سورايلى -دېدى ۋە موماينىڭ ئۇنىمىغىنىغا قارىماي پۇلنى قولىغا تۇتقۇزۇپ مەسچىتكە باشلاپ ئەكەلدى ھەمدە مەسجىد ھەيئەتلىرى بىلەن كۆرۈشۈپ موماينىڭ قىيىنچىلىقىنى يىگىت ئۆزى جاكارلاپ جامائەتتىن ئىئانە توپلايدىغانغا كېلىشتى ......
يىگىت ئىنتايىن خوش چۈنكى ئۇ بۈگۈن يەنە بىر ئاجىزەگە ياردەم قىلىپ قەلىب يارىسىغا ئاز بولسىمۇ ھەرھالدا مەلھەم بولالىغان ئىدى. ئەمما يىگىتنىڭ خوشاللىقىنىڭ ئۇزۇنغا بارمايدىغانلىقىنى بۇگۈن ئاخشام ئۇنى ئەزرائىلنىڭ كۈتۈۋالىدىغانلىقىنى ھېچكىم خىيالىغا كەلتۈرمىگەنىدى.....
كېچە سائەت ئىككىدىن ئاشقان چاغ ئەتراپنى جىمجىتلىق قاپلىغان بولۇپ كۈندۈزدىكى شاۋقۇن-سۈرەنلەردىن ئەسەرمۇ يوق ئىدى. يىگىت كۈندىكى ئادىتى بويىچە تاكسىسىنى ھەيدەپ كىراكەشلىك قىلىش ئۈچۈن ئۈرۈمچىنىڭ تار كوچىلىرىنىڭ بىرىدە كېتىۋاتاتتى .
تۇيۇقسىز ئۇنىڭ دىققىتىنى كوچىدا تارتىشماق ئويناۋاتقان بىرقانچە كىشلەرنىڭ قىيا-چىيا ، ۋاراڭ -چۇرۇڭلىرى بىردىنلا ئۆزىگە تاتتى . يىگىت ئىختىيارسىز ماشىنىسىنى توختىتىپ نېمىش بولغانلىقىنى بىلىپ بېقىش ئۈچۈن كۈزۈتۈپ ماشىننىڭ ئىچىدە ئولتۇردى .
كوچا چىراقلىرىنىڭ يورۇقىدا ئۇلارنىڭ ھەركىتى ئېنىق كۆرۈنۈپ تۇراتتى ، بىر ئايال بىر قىزنىڭ قولىدىن تۇتۇپ بۇياققا تاتسا يەنە بىر توپ ئوغۇللار قىزنىڭ قولىدىن تۇتۇپ ئۇياققا تارتاتتى ، قىز ۋە ئايال ئۈنىنى بولىشىغا قويىۋىتىپ باركۈچى بىلەن قارىشلىق كۆرسىتىۋاتاتتى ،بىركەمدە ئوغۇللار قىزنى ئايالنىڭ قولىدىن يۇلۇپ ئېلىپ ماڭماقچى بولۇشتى قىز ماڭغىلى ئۇنىماي سۆرىلىۋىلىۋاتاتتى .
ئەسلىدە بىر ئانا يېرىم كېچىدە قۇلىقىغا سۈۋەرەك كىرىپ كەتكەن ئەمدىلا 17 ياشقا كىرگەن قىزىنى دوختۇرغا ئەكىلىپ كۆرسىتىپ قايتاشىدا رىستۇراندىكى تويدىن يانغان بۇ ھاراقكەش لۈكچەكلەرگە يولۇقۇپ قالغانىدى .
ئانىنىڭ <قۇتقۇزۇڭلار !!! بىزگە ياردەم بېرىڭلار!!!<> دېگەن نالىلىرى ئەتراپنى بىر ئالغانىدى .
ئىشنىڭ تېگىگە يەتكەن يىگىت دەرھال ماشىنىدىن چۈشۈپ ۋۇ ھايۋانلار دەپ يۈگۈرۈپ بارغىنىچە بىرسىنىڭ تۇمشىقىغا كېلىشتۇرۇپ بىرنى سالدى ، لۈكچەكلەر 7،8 بالا بولغاچقا بىردەمدىلا يىگىتنىڭ ئاغزى بۇرنىنى قان قىلىۋەتتى ،ئاندىن يىگىتكە قاراپ بۇ ئىشقا ئارلاشماي چىرايلىقچە يولۇڭغا كېتىۋال بولمىسا ئانا - ماناڭنى كۆرسىتىپ قويىمىز دەپ تەھدىت سېلىشقا باشلىدى ، ئەمما يىگىت كەتمىدى ئۇ يەنىلا ۋۇ ھايۋانلار قىزنى قويۇپ بېرىش دەۋاتاتتى ،لۈكچەكلەر بۇ <سالجا>دىن ئەدىبىنى تېخىمۇ بەكەرەك بېرىپ قويمىغۇچە قۇتۇلالمايدىغانلىقىغا كۆزى يېتىپ ئۇنىڭغا قاراپ يوپۇرۇلۇپ كەلدى ، يىگىتمۇ بوش كەلمىدى ئۇمۇ ئۇلارغا قاراپ ئېتىلدى ....تاياق زەربىسىدىن يىگىت ھالسىراپ ماغدۇرسىزلىنىپ يېتىپ قالغانىدى شۇنداقتىمۇ قىزنى سۆرەپ ئېلىپ ماڭماقچى بولغان بىرسىنىڭ پۇتىغا جاھىللارچە ئېسىلىۋالدى ، بۇنى كۆرگەن  لۈكچەكلەر تېخىمۇ غالجىرلاشتى ئۇلار يىگىتنىڭ بېشىغا ئارقا- ئارقىدىن تېپىپ دەسسەشكە باشلىدى لېكىن بۇ قېتىم يىگىت باركۈچى بىلەن بىر ئاماللارنى قىلىپ قىزنىڭ پۇتىغا ئېسىلۋالدى، مەسچىلىكتە ئۇلارنىڭ ئىچىدىكى بىرتەلۋە ياندىكى پىچىقىنى چىقىرىپ يىگىتنىڭ بىرقانچە يېرىگە تىقىۋەتتى ، يىگىت قانسىرىدى بارا-بارا قوللىرى بوشاپ بارماقلىرى قىزنىڭ پۇتىدىن ئاجراپ چۈشتى......
يىگىتنىڭ مىيىت نامىزى ئادەم دېڭىزىغا ئايلانغان بولۇپ كېلىۋاتقانلارنىڭ ئايىقى ئۈزۈلمەيتى......
كونا قائىدە بويىچە يىگىتنىڭ ئاتا-ئانىسى جامائەتتىن :
-جامائەت باھا بەرگەن بولساڭلار بالىمىز ھايات چېغىدا قانداقراق بالا ئىدى؟-دەپ سورىدى،جامائەت بەس- بەستە :
-بالاڭلار ئاقكۆڭۈل ، سەمىمى ، باشقىلارغا ياردەم بېرىشنى خوشاللىق دەپ بىلىدىغان ئوبدان بالا بولۇپ بەئەينى <لېيفىڭ> غىلا ئوخشايتى -دەپ ماختاپ كېتىشتى ، ئەمما ئىمام ئاخونۇمنىڭ بەرگەن جاۋابى ئۆزگىچە بولۇپ ھەممىنى ھەيران قالدۇردى .
ئىمام يىگىتنى <لېيفىڭ> غا ئوخشىتىپ ئەمەس بەلكى ھەزرىتى ئەلىگە سېلىشتۇرۇپ ئۇ يىگىت بىزنىڭ ئىچىمىزدىن چىققان ئىنسانپەرۋەر، باتۇر ھەزرىتى ئەلىرىمىز ئىدى دەپ تەرىپلىگەنىدى .
ئىمامنىڭ كۆزىدىن ئاققان ئىككى تال ياش تامچىسى مەڭزىگە سېرىلىپ چۈشتى .......ئارقىدىن جامائەت توغرا ئۇ بىزنىڭ ئىچىمىزدىن چىققان باتۇر ئوغلان ئەلىرىمىز ئىدى دەپ توۋلاپ كېتىشتى .
