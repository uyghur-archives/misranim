                                                                  1
بايلىق ئۆمۈرنى خاتىرجەم ئۆتكۈزۈش ئۈچۈندۇر، ئەمما ئۆمۈر مال توپلاش ئۈچۈن ئەمەس . بىر ئاقىلدىن(كىشىلەر) سوراشتى : بەخىتلىك كىم  ۋە بەخىتسىز كىم؟  ئاقىل دېدى : بەخىتلىك شۇ كىشىكى، يەپ ئىچتى ھەم تېرىدى . بەخىتسىز شۇ كىشىكى ، ئۆلدى، ھەممە نېمىسى قېپقالدى.  
                                تۆھپىسىز ئۆلگەن بېخىلنىڭ نامىزىنى ئوقۇما،
                               مال يىغىشقا كەتتى ئۆمرى،ئۆزىمۇ نەپ كۆرمىدى.
                                                              2
مۇسا ئەلەيھىسسالام قارۇنغا <1> نەسىھەت قىلدى: ئاللاھ ساڭا ئىھسان قىلغاندەك ، سەنمۇ كىشىلەرگە ئىھسان قىل!  <2> ئۇ قۇلاق سالمىدى،ئۇنىڭ ئاقىۋىتىنى سەن ئاڭلىغان  !
                                   كىم ياخشى بىر ئىش قىلمىسا دىنار-دىرھەمدىن، <3>
                                  ئاخىر بېشىنى بەرگۈسىدۇر دىنار-دىرھەمگە .
                                 گەر  بەھرە ئالاي دىسەڭ  ھەر ئىككى دۇنيادىن ،
                                 ئاللاھ ساڭا قىلدى كەرەم، قىل سەنمۇ ئادەمگە  .
ئەرەب ئېيتىدۇ: ياخشىلىق قىل ، ئەمما مىننەت قىلما،چۈنكى، پايدىسى ئۆزۈڭگە  بولىدۇ!  يەنى سېخىيلىق قىل ، ئەمما مىننەتدارلىق تەلەپ قىلما، چۈنكى پايدىسى ئۆزۈڭگە قايتىدۇ !
                                كەرەم دەرىخى سۈرسە يىلتىز قايان ،
                               ئۆسەر شاخ ۋە تاجى پەلەكتىن بەلەن.
                              ئۇنىڭ مېۋىسىنى ئەگەر يەي دىسەڭ ،
                              تېگىن ھەرىلەپ كەسمە مىننەت بىلەن.
                              شۈكۈر قىلغىن خۇداغا، ياخشىلىققا ئېرىشتىڭ
                              سېنى نىئمەت-بەرىكەتتىن قالدۇرمىدى بىر چەتتە.
                             قىلچە مىننەت قىلمىغىن، خىزمەت قىلساڭ سۇلتانغا،
                             مىننەتدار بول ئۇنىڭدىن تۇتقاچ سېنى خىزمەتتە.
                                                                  3
    ئىككى خىل كىشىنىڭ جاپا تارتقىنى بىكار ، تىرىشقىنى پايدىسىز: بىرى مال يىغدى ، يېمىدى؛ يەنە بىرى ئۆگەندى ، ئەمەل قىلمىدى.  
                             ئىلىمنى قانچىلىك ئۆگەنسەڭ، ئۆگەن ،
                             مۇبادا ئىشلەتمىسەڭ نادانسەن.
                            ئەمەسسەن نە ئالىم، نە بىر دانىشمەن،
                            تۆت پۇتلۇق ئۇلاغسەن، كىتاب يۈكلەنگەن .
                           ئۇ بىر مېڭىسىزكى ، ھېستىن بىخەۋەر ،
                           بىلمەس ئۈستىدىكى ئوتۇنمۇ ، دەپتەر.
                                                                     4
ئىلىم دىننى كۈچلەندۈرۈش ئۈچۈندۇر، ھەرگىز مال دۇنيا تېپىش ئۈچۈن ئەمەس.
                         كىمكى پەرز ، ئىلمۇ تەقۋالىقنى سېتىپ پۇل قىلار ،
                        بىر دۆۋە خاماننى توپلاپ ئوت يېقىپلا كۈل قىلار.
                                                                      5
پەرز قىلمايدىغان ئالىم_ مەشئەل كۆتۈرۈۋالغان قارىغۇ ، ئۇ باشقىلارغا يول باشلايدۇ يۇ ، ئەمما ئۆزى توغرا يولدا ماڭمايدۇ.  
                             كىمكى بىپايدا ئۆمرىن ئۆتكۈزدى  ،
                            نەرسە ئالماستىن  پۇلىنى  بۇزدى .
                                                                6
مەملىكەت ئۆز ئالىملىرى بىلەن ھۆسىن تاپىدۇ، دىن پەرزكار كىشىلەر بىلەن كامالەتكە ئېرىشىدۇ. ئاقىللارنىڭ پادىشاھقا يېقىنلىشىشىقا مۇھتاج بولۇشىدىن كۆرە ،  پادىشاھلار دانىشمەنلەرنىڭ سۆھبىتىگە كۆپرەك مۇھتاج بولىدۇ.
                                    قۇلاق سال، بىر نەسىھەت ساڭا ئەي شاھ ،
                                   جاھاندا مۇندىن ئارتۇق يوق نەسىھەت .
                                  ئەمەل بەرمە ئاقىللاردىن بۆلەككە   ،
                                  ئەگەر ئاقىل ئىش ئەمەلدىن بولسىمۇ چەت .
