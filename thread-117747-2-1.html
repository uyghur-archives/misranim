ئەسسالاممۇ ئەلەيكۇم ؟ بارلىق مۇنبەرداشلار ، مۇنبىرىمىزدىكى جاپاكەش باشقۇرغۇچىلار، تىرىشچان قولى گۈل قەلبى ئاتەش ئىجتاھاتلىق قەلەم ساھىبىلىرى ياخشىمۇسىزلەر ؟
مۇنبىرىمىز قۇرۇلۇپ  يەتتە يىل ئىچىدە سىلەرگە ئوخشاش مۇنبىرىمىزنى يېقىندىن قوللاپ ، ئانا تىلىمىز ۋە ئەدەبىيات سەنئەتنى ، مەدىنىيەت ، ئۆرىپ ئادىتىمىزنى  تەرەققىي قىلدۇرۇشقا توختىماي كۈچ چىقىرۋاتقان مەۋھۇم تور دۇنياسىدىكى كەڭ تورداش دوسلىرىمنىڭ قوللىشى بىلەن  ئۇيۈۇر تور بەتچىلىكىدىكى سەركە بولغان بىر قۇشۇنغا ئايلاندى .
مۇنبىرىمىزنى گۈزەل بىر باققا ئوخشاتساق ،ئىلگىرى بىز شۇ باغدىكى يېڭى تىكىلگەن كۆچەتلەر ئىدۇق . بىكەت مەسئۇللىرى ۋە باشقۇرغۇچىلىرىمىز ، قەلەم ئۇستازلىرىمىز بىز ئۈچۈن باغۋەن بۇلۇپ بىزنى بۈگۈنكىدەك مىۋە بىرىپ كىشىنى ئۆزىنىڭ نىممەتلىرى بىلەن شاتلاندۇرالايدىغان مىۋىلىك دەرەخ قىلىپ تەربىيلىدى . مۇنبىرىمىزنى سىرىتتىن كۆزىتىپ تۇرۋاتقانلار بولسا بىزنىڭ مۇشۇ كىشىگە ئەقلىي ۋە مەنىۋى جەھەتتىن ئۇزۇق بىرەلەيدىغان نىمەتلىك بېغىمىزدىن شىرنە يىققۇچى ھەسەل ھەرىسىدەك مۇنبىرىمىزدىكى سەرخىل تىمىلاردىن قانغۇچە ھۇزۇر ئالماقتا .ۋە تىمىنىڭ ئىسىللىقى سەۋەپلىك ئەزا سانى بارا - بارا كۆپەيمەكتە . مۇنبىرىمىزنىڭ بۇنىڭدىن كىينكى تەرەققىياتىغا زور ئۇتۇق تىلەيمەن ، ئابابەكرى ئاكاشنىڭمۇ يولىنىڭ ئوچۇق ، ئىشلىرىنىڭ ئۇتۇقلۇق بولىشىغا چىن كۆڭلۈمدىن تىلەكداشمەن .
ئويغىنىپ كەتتى جاھان ، مەغرىپى مەشرىق تامان ،
مەن تىخى سۈت ئۇيقىدا ، چۈش كۆرۈپ ياتارىمەن .
