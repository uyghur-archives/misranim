ئۇ شۇنچە سەبىي قىز ئىدى ، مۇھەببەتنى – نەپرەتنى بىلمەيتتى ، شۇنچە ئاقكۆڭۈل ، كۆڭۈلچەك ئىدىكى ، چۈمۈلىگىمۇ ئازار بەرمەيتتى ، ھەر قانداق ئىش قىلىشتىن بۇرۇن باشقىلارنى ئويلايتتى ، ئۆزى قانچىلىك جاپا تارتسىمۇ باشقىلارنى جاپاغا قويۇشنى خالىمايتتى ، ئۆزىنىڭ كۆڭلى ئازار يىسىمۇ باشقىلارنىڭ كۆڭلىگە ئازار بېرىشنى خالىمايتتى . ھەتتا دۈشمىنى بولسىمۇ ئۇ ئۆز كۆكىرىكى بىلەن دۈشمىنى قارنىغا تىقىلماقچى بولغان خەنجەرنى توسۇۋېلىشنى خالايتتى .
ئۇ مۇھەببەتلەشتى ، تولۇقنىڭ ئۈچۈنجى يىللىقىدا ، ئۆز رازىلىقى بىلەن بولمىسىمۇ بىراق بۇ مۇھەببەت ئۇنى ھەققىي ساداقەتكە باشلاپ كىرەتتى . شۇنداقتىمۇ ئۇنىڭ بۇ مۇھەببەتىنى قۇبۇل قىلغۇسى يوق ئىدى ، بىراق بىراۋلارنىڭ ئۆزى سەۋەبلىك كەلگۈسىنى ، كىيىنلىكىنى تۈگەشتۈرۋېلىشنى خالىمايتتى . بۇ ئۇنىڭ ساددىلىقى ، ئاقكۆڭۈل ، كۆڭۈلچەكلىكىدىن ئىدى . بۇ مۇھەببەتنى ئۇ شەرتسىز قۇبۇل قىلدى ، كۆندى ، كىيىنچە ئۆزىمۇ بىلمىگەن ھالدا قارشى تەرەپنى ياخشى كۆردى . بىراق بۇ ياخشى كۆرگەن ئادىمى دىلشات ئەمەس ئىدى .
ئالىي مەكتەپتىكى ساۋاقدىشى ، يېققىن دوستى دىلشات ئۇنىڭ ئەڭ يېققىن ئادىمى ئىدى ، بەزىدە ئۇ ئۇنىڭدىن ئايرىلىپ قېلىشنى كۆڭلىگە كەلتۈرۈپمۇ باقاتتى ، بىراق يۈركىنىڭ بىر يەرلىرى ئۇنىڭدىن توختىماي سۇئال سوراپ گاڭگىرىتىپ قويۇۋاتاتتى . ئۇنىڭدىن ئايرىلىپ ياشاشقا ئۇ كۆنۈشى ياكى ئۇنى كىيىنكى ئۆمۈرۈدىكى يۈلەكچىسى قىلىپ تاللىشى كېرەك ئىدى . شۇنداق بولغاندىلا ئۇ زىخنىمۇ كاۋاپنىمۇ كۆيدۈرمەيتتى . بىراق ئۆزى ياخشى كۆرگەن يىگىت ئۇنى بىر يىل ساقلىغان ، ئۇنىڭغا نۇرغۇن شىرىن ئەسلىمىلەرنى قالدۇرغان تۇرسا ئۇ قانداقمۇ بۇ يۈزسىلىكنى قىلالىسۇن ! ؟
ئەقىدە شۇلارنى ئويلاپ كۆزلىرىنى خاتىرىگە تىكتى ، ئۇنىڭ كۆز ئالدىدا دىلشاتنىڭ سىماسى يەنە بىر مەرتە سايە تاشلىغىنىدا ئۇ ئۇلۇغ كىچىك تىنىپ قويدى دە خاتىرىنى يانغا قويۇپ ئۆي تورىسىغا قارغىنىچە قولىدىكى چاچ بوغقۇچنى ئويناشقا باشلىدى ، شۇ قېتىمقى پويىز سەپىرى ئۇنىڭ ئېسىدىن مەڭگۈ كۆتۈرۈلمىسە كېرەك .
ئەڭ ئاخىرقى تەتىلدە ئۇلار بىرگە قايتتى ، ئەقىدە پويىزغا چىقىپلا ئۇيقۇغا كەتتى ، دىلشات بولسا ئورننى ئۇنىڭ ئۇخلىشى ئۈچۈن بوشتىپ بەردى . ئۇ ئەقىدەنىڭ ئۇخلىغىنىغا برشىدا قاراپ تۇراتتى ، پات-پات ئۇنىڭ بېشىنى سەل كۆتۈرۈپ قوياتتى ، چاچلىرىنى تۈزەشتۈرۈپ قوياتتى ، پويىزدا پەقەت ئىككىلا ئۇيغۇر باردەك قىلاتتى ، شۇڭلاشقىمۇ دىلشات قورنۇپ يۈرمىدى ، ئەقىدە ئۆز يۈزىنىڭ شۇنچە يەڭگىل يەلپۈنگىننى ھېس قىلغىنىدا بېشىنى كۆتۈردى . دىلشات ئۇنىڭ بېشىدا ئۇنى يەلپۈپ ئۆرە تۇراتتى .
- قانچلىك ئۇخلۇدۇم .
- بەش سائەتچە بولدى ، قورسقۇڭ ئاچتىمۇ ؟ - دىلشات شۇنداق دىگىنىچە قولىدا سۇداننى كۆتۈرۈپ قايناق سۇ ئېلىشقا ماڭدى ، ئەقىدەمۇ پۇرسەتتىن پايدىلىنىپ سۇخانىغا بېرىپ كېلىشنى كۆڭلىگە پۈكۈپ ئورنىدىن تۇردى ۋە چاچلىرىنىڭ چۇۋلۇپ كەتكىننى ھېس قىلىپ ئورنىدىلا ئولتۇرۇپ ياغلىقىنى چەگدى .
- نەگە ؟
ئەقىدە قىزارغىنىچە يەرگە قارىدى ، ئۇ نىمە دەپ جاۋاپ بېرىشنى بىلەلمەي قولىدىكى تازلىق قەغىزىنى يۇشۇرۇشقا ئۇرۇندى .
- مەن بىللە باراي ، قارسام بازارچىلار باردەك قىلىدۇ .
دىلشات ئەقىدەنىڭ جاۋابىنى كۈتمەيلا ئالدىغا چۈشۈپ ماڭدى ، ئەقىدە بولسا ياۋاشلىقچە ئۇنىڭ كەينىدىن ماڭدى ، بىراق ئۇنىڭ قىزارغان يۈزلىرىدىن خىجىل بولغىنى ياكى خوش بولغىننى بىلگىلى بولمايدىغان بىر خىل نۇر ئەكىس ئېتىپ تۇراتتى . دىگەندەك يېشى ئۇلاردىن پەرىقلەنمەيدىغان بىر نەچچە ئۇيغۇر بالا ۋاگوننىڭ يەنە بىر بېشىدا تاماككىلىرىنى چىشلىگىنىچە كۈلۈشۈپ تۇراتتى .
- سەنمۇ ئۇخلىۋالامسەن ؟ - قولىدىكى تەييار چۆپ خالتىسىنى ئەخلەت ساندۇقىغا  تاشلىغاچ ئەقىدە دىلشاتتىن شۇنداق دەپ سورىدى .
- بولدى ، مېنىڭ ئۇيقۇم كەلمىدى ، ئۇيقۇم كەلسە سېنى تۇرغۇزۋېتىپ ئۇخلايمەن خاتىرجەم بولە !
ئەقىدە باشقا سۇئالنى سورىماي يەنە ئۇخلىماقچى بولۇپ ياتتى ، بىراق ئۇنىڭ ئىچىنى بىر نەرسە تاتلاۋاتقاندەك ، ئىچىگە بىر نىمە كىرۋالغاندەك دىلغۇللۇق ، تىتتىلىق ئىچىدە ئۇخلىيالمىدى ، توغرىسىنى دىگەندە ئۇ دىلشاتنىڭ يەنە شۇنداق ئۆزى ئۈچۈن قانچلىك بەدەل تۆلىشىنى ، قانچلىك جاپالارنى چېكىشىنى بىلمەيتتى ، ئۇنىڭ بىلىدىغىنى دىلشاتنىڭ شۇ قىلغانلىرىغا ئۆزىنى ماس كەلمەيدىغاندەك ھېس قىلغىنى ئىدى . بىراق ئۇ بۇنى قانداقمۇ دىلشاتقا يۈز تۇرانە دىيەلىسۇن ! ؟ دىلشات ئۆزى شۇنچلىك يەڭگىل ، ئوچۇق ، خۇشچاخچاق كۆرۈنگىنى بىلەن بەزى يەرلەردە شۇنچە چىڭ ئىدى ، بولۇپمۇ ئىچىدىكىنى بىراۋغا قەتتىيلە تۇيدۇرۇشنى خالىمايتتى ، ئۆزىنىڭ ئاجىزلىقىنى ئۆزى ياخشى كۆرگەن كىشىلىرى ئالدىدا كۆرسىتىشنى خالىمايتتى . ئىشنى قىلىپ ئۇ ئەقىدەنىڭ كۆڭلىدە نوقسانسىز ، ئېۋەنسىز ئادەم ئىدى ، شۇڭلاشقىمۇ ئۆزىنى دائىم ئۇنىڭغا ماس كەلمەيدىغاندەك ، نىمىشقىمۇ دىلشاتنىڭ باشقىچە بىر كىشى بولىشىنى ئويلاپ قالاتتى .
شۇنداق ، ئەقىدەنىڭ بىلىشىچە دىلشات ئۈچ خىل كىشىنى ياقتۇرمايتتى . ‹‹ رەھمەت ›› دىگەن سۆزىنى ئېيتىسلا ئۇ قىلغان ئىشىنى بەزىدە بۇزۋېتەتتى ، بەزىدە ئالايغىنىچە قىلىۋاتقان ئىششىنى يېرىم يولدا تاشلاپ قوياتتى ، چۈنكى ئۇنىڭ نەزىرىدە ئۆزى قىلىشقا تېگىشلىك ئىشنى قىلغان ۋاقىتتا باشقىلارنىڭ رەھمەت ئېيتىمىسمۇ بولدىغىننى ، پەقەت قىلماسقا تېگىشلىك ئىشنى قىلغاندا باشقىلارنىڭ رەھمەت ئېيتىشى ھەمدە رەھمىتىنى قۇبۇل قىلسا بولدىغننى بىلەتتى ، ئۇ تونۇشلىرىنىڭ رەھمەت دىگەن سۆزنى ئىشلىتىشىگە قارشى ئىدى .
‹‹كەچۈر›› بۇ سۆزنى ئۇ ئەزەلدىن ئىشلەتمەيتتى ، چۈنكى كەچۈر سۆزىنى ئېيتقان ئادەم يا شۇ ئىشنى قىلماسلىقى قىلغان ئىكەن تۈزۈشى كېرەك ، كەچۈر نىمىگە ئاقىدۇ ، ئادەمنى زاڭلىق قىلغانلىق بولماي نىمە ؟ قىلغۇلۇقنى مەيلى بىلىپ قىلسۇن مەيلى بىلمەي قىلسۇن بۇ سۆزنى ئۇ پەقەت كۆزگە ئىلماسلىق دەپ چۈشۈنەتتى .
‹‹ ئۆتۈنەي ›› ئۇ بۇ سۆزنى ئاڭلىغاندا كۈلۈۋېتەتتى ، چۈنكى ئادىمىزاتنىڭ ئادىمىزاتتىن ئۆتۈنگىدەك زادى شۇ ئادىمىزاتنىڭ نىمىسى ئارتۇق ، ئوخشاش ئادەملەر تۇرۇپ نىمىشقا باشقىلاردىن ئۆتۈنىدۇ ؟ مەيلى كۆڭلۈ بولسۇن ، مەيلى مال دۇنيا ..... دۇنيادا پەقەت ئۇنىڭ تىزلىنىپ تۇرۇپ تىلەيدىغان ، ئۆتۈنىدىغان بىرلا پەقەت ۋە پەقەت بىرلا ئىگىسى بار ، شۇ ئىگىسى ئۇنىڭغا ھەممىنى بېرىدۇ ، يەنە ھەممىنى تارتىۋالىدۇ !
ئەقىدە دىلشاتنىڭ خۇي پەيلىنى بىلگەچكە بەزىدە دىققەت قىلماي قىلغان سۆزلىرىگە دىلشات پەقەت كۈلۈپلا ئۆتكۈزۋېتەتتى ، بىراق ئۇ يەنە شۇنداق پەيىتلەرنى تېپىپ ئۇنى قايتۇرۋېتەتتى . شۇڭلاشقىمۇ ئەقىدە دىلشاتقا ئەڭ يېققىن ئادىمىگە مۇئامىلە قىلغاندەك مۇئامىلە قىلاتتى .
ئەقىدە بېشىنى كۆتۈردى ، دىلشات باشقا ياققا قاراپ خىيال تۇراتتى ، ئەقىدە ئورنىدا ئولتۇرغاندىن كىيىن
- سەنمۇ ئولتۇرە !
- ئۇخلاۋەرگىن بولدى .
- ئۇيقۇم كەلمىدى ، ئولتۇرە .
دىلشات ئورنىدا ئولتۇرغاندىن كىيىن يەنە باشقا ياقلارغا قاراپ تۇردى ، ئەقىدە بولسا قولىدىكى دىلشاتنىڭ چاپىننى پۇراپ
- پۇراپ باقە ؟ نىمە پۇراپ قاپتۇ ؟
دىلشات قولىغا چاپاننى ئېلىپ پۇرىدى . ئۇ شۇنچە كۈچۈپ پۇرىدى .
- ھېچنىمە پۇراپ قالماپتىغۇ ؟ بايا قولۇڭنى يۇيۋەتكەنمىدىڭ يا ؟
- ساراڭ – ئەقىدە شۇنداق دىگىنىچە قىزاردى – قايتا پۇراپ باقە ؟
- شۇ ھېچنىمە پۇرمايدىغۇ !
- بوپتۇ ئەمسە !
ئەقىدە دىلشاتنىڭ چاپىننى قولىغا ئالغىنىچە يەنە بېشىغا قويۇپ ۋاگون تۆمۈرۈگە يۆلۈنۈپ كۈلۈپ تۇردى ، دىلشات خۇددى سىناقتىن ئۆتەلمىگەن ئوقۇغۇچىدەك بېشىنى پەس قىلىپ كۈلۈپ تۇراتتى .
- ھەي ، نىمە پۇرايدۇ ؟ دىگىنە ؟
- مەن پۇرايمەن .
ئەقىدە شۇنچە بوش ئاۋازدا دىيىلگەن بۇ سۆز دىلشاتقا خۇددى ۋارقىراپ تۇرۇپ ئېيتىلغاندەك ئاڭلاندى . ئۇ چۈش كۆرۋاتقاندەك ئىدى ، ئەقىدە ئۇنىڭ ئۆزىنى ياخشى كۆرىدىغىننى سېزىپ قالدىمىكىنە ؟ دىلشات شۇلارنى ئويلۇغۇنىچە يەنە جىمىپ كەتتى . ئۇ ئاستالا تىلفۇننى قولىغا ئالدى ۋە قولاق تىڭشىغۇچنى تىققىنىچە ناخشا ئاڭلىدى .
- ئەكەل مەنمۇ ئاڭلاي – ئەقىدە شۇنداق دىگىنىچە قۇلاق تىڭشىغۇچىنى قۇلقىغا سالدى . قۇلاق تىڭشىغۇچتىن يەڭگىل ئۆزبىكچە ناخشا چىقىۋاتاتتى .
- سەنمۇ ئۆزبىكچە ناخشىغا ئامراقكەنسەن ھە ؟
- ھەئە ، ئامراق ، بۇرۇنمۇ بەك ئاڭلايتتىم .
ئۇلار شۇ تەرىقىدە ناخشا ئاڭلاۋاتاتتى . بەلكىم ئەقىدەدىلا ئەمەس دىلشاتتىمۇ بۇ پويىزنى مەڭگۈ توختىمىسىكەن ، ئاخىرقى ۋاگونغا بارمىسىكەن دىگەن ئۈمۈت پەيدە بولغان بولسا كېرەك . بىراق بۇنداق داۋاملىشىدىغان ئىش نەدە بار ، ھەر قانداق ئىش باشلانغان ئىكەن ئاخىرلىشىدۇ دە !
- ھەي ، مەكتەپ ئاشخانىسىدىكى ھېلقى سورالغان سۇئالغا نىمە دەپ جاۋپ بەرگىنىم يادىڭدا بارمۇ ؟
دىلشات شۇ زاماتلا مەكتەپ ئاشخانىسىدا قىزلارنىڭ بىر-بىردىن ‹‹ ئەگەردە ئالدىڭدا شۇنداق ئىككى كونۇپكا بولسا ، بىرسىدە ‹‹ھاياتىنى قايتا باشلاش ›› يەنە بىرسىدە ‹‹ داۋاملاشتۇرۇش ›› دەپ بولغان بولسا سەن قايسىسىنى تاللايتىڭ؟ ››دىگەن سۇئالغا جاۋاپ بېرىپ ئويناۋاتقىنى يادىغا كەلدى ، شۇ ۋاقىتتا ئەقىدە ‹‹ ھاياتىنى قايتا باشلاش ›› دىگەن كونۇپكىنى باسىدىغانلىقىنى دىگەن ئىدى .
- ھەئە ، يادىمدا ، قايتا باشلاشنى باسىمەن دىگەن ئىدىڭ
- نىمىشقىلىقىنى بىلەمسەن ؟
دىلشات بېشىنى ياق دىگەن مەنىدە لىڭىشتتى . بىراق ئۇ كۆڭلىدە بىلەتتى . ئۆزىگە شۇنچە ئىشىنەتتى ، بىراق بۇ ساددا ، سەبىي قىزنىڭ ئالدىدا ئۆزىنى ئاشكارلاشنى خالىمايتتى ، بەلكىم ئاللىقاچان ئاشكارلانغاندۇ ، پەقەت تەن ئالغۇسى يوق ئىدى ، ئۇ پەقەتلا مەكتەپنى پۈتتۈرسە ھەممە ئىشلار ئاخىرلىشىدۇ دەپ ئويلايتتى .
- مېنىڭ بۇنداق ئۈچ بۇرجەك مۇھەببەت ھالىتىدە ياشغۇم يوق . سەنچۇ قايسىنى باسىتىڭ .
- داۋاملىق داۋاملاشتۇرۇش دىگەننى باسىتىم – دىلشات قىلچە ئويلانمايلا شۇنداق دەپ جاۋاپ بەردى .  
- نىمىشقا ؟
- چۈنكى ، مەن قايتا داۋاملاشتۇرغان ھاياتىمدا سېنى ئۇچۇرتالماي قېلىشتىن ئەنسىرەيمەن !
ئۇلار جىمىپ كەتتى ، توغرا ، مۇشۇنداق ئاشكارلانغان دوسلۇق ياكى مۇھەببەت بولسۇن ، ئۇ ئاشكارلانغان ۋاقىتتا شۇنچىلىك تەبئىي ، شۇنچىلىك تۇيۇقسىز بولىدۇ ، ئادەمنى شۇنچە جىمخور ، شۇنچە مۇلايىم قىلىۋېتىدۇ ، ياكى ئەسەبىيلەشتۈرىدۇ .
ئەقىدە چېچىنى بوغۇش ئۈچۈن يېشىۋەتكەن چاچ بوغقۇچىنى قولىغا كەيدۈردى ، بىراق ھېچ ئەپلىشەلمەيۋاتاتتى ، بىر كۈن قويۇۋېتىلگەن بۇ چاچلار مەقەمگە كېلىدىغاندەك ئەمەس ئىدى . دىلشات گەپ قىلمايلا ئەقىدەنىڭ قولىدىن چاچ بوغقۇچنى ئالدى
- سەن چاچلىرىڭنى تۈزەشتۈر ، مەن بوغقۇچنى بىلەن بوغۇپ قوياي .
ئەقىدە قارشلىق قىلمىدى ، چۈنكى دىلشات ئۇ ئەڭ ئىشىندىغان ئادەمنىڭ بىرى ئىدى .
- ۋاي ئۈزلۈپ كەتتى – دىلشات شۇنداق دېدى دە ئورۇندۇقنىڭ ئاستىغا قارىدى – قەيەرگە چۈشتىكىنە ؟
- بولدى ئىزدىمە ، ئۈزلۈپ كەتكەن بولسا ئىشلەتكىلى بولمايدۇ ئەمدى . مەندە يەنە باشقىسى بار – ئەقىدە شۇنداق دەپ دىلشاتقا يەنە بىر چاچ بوغقۇچنى سومكىدىن ئېلىپ بەردى .
ئىشنى قىلىپ بۇ ئەڭ ئاخىرقى سەپەر ئەقىدەگە ئەڭ ئۇنتۇلغۇسىز ئەسلىمىلەرنى قالدۇرغان ئىدى ، دىلشات ئىككى كېچە ئۇخلىماي ئۇنىڭ بېشىدا تۇردى ، پۇتلىرى ئىشىشىپ كەتكەندە پۇتلىرىنى تۇتۇپ قويدى ، پۇت بارماقلىرىنى ئۇۋلىدى . قىززىق – قىززىق يۇمۇرلار بىلەن كۈلدۈردى ، بىراق ئاخىرقى بېكەتكە ئاز قالغانسىرى دىلشاتنىڭ ھەر بىر ئېيتقان سۆزى ئەقىدەگە گاھ مىق ، گاھ زىخ بوپ تۇيۇلماقتا ئىدى . دىلشات شۇنداق گەپلەرنى دەپ بولۇپلا خۇددى ھېچ ئىش بولمىغاندەك ‹‹ تاتلىق ئۇنتۇلىدۇ ، ئاچچىق ئۇنتۇلمايدۇ ›› دەپلا قوياتتى ، بىراق ئىچىنىڭ قانچلىك دەرىجىدە ئىچىششقىننى ، يۈرگىنىڭ قانچلىك دەرىجىدە ئاغرىغىننى ئەقىدە مەڭگۈ ھېس قىلالمايتتى . ھېس قىلالمايتى ئەمەس بەلكى ھېس قىلىشقا ئۇنىڭ قۇربى يەتمەيتتى . چۈنكى بۇ ئاچچق شۇنداق ئاچچىق ئىدىكى تېتىغان كىشى ئۆمۈرلۈككە ئاچچىق تەمنى سېزەلمەيتتى .
ئەقىدە خىيالدىن بېشىنى كۆتۈرگەندە ئۆي ئىچىنى ئاي نۇرى غۇۋا يۇرتۇپ تۇراتتى . ئۇ ئورنىدىن تۇرۇپ چىراقنى ياقتى ، خاتىرىنى قولغا قايتىدىن ئالدى .


داۋامى داۋاملىق ئىنكاس شەكلىدە يوللىنىدۇ !
ھېكايىگە بولغان ھېرىسمەنلىكىم سەۋەبىدىنمىكىن ، بۇ ھېكايىنى تومتاقلا ياكى ئاددىيلا ئاخىرلاشتۇرۇشقا كۆڭلۈم ئۇنىمايۋاتىدۇ ، تورداشلارنىڭ دىگىنىدەك ئەسلى بىراقلا يوللاپ ئۆزۈممۇ قۇتۇلۇپ سىلەرنىمۇ ھېكايە كۈتۈش تەقەززاسىدىن مەھرۇم قىلسام بولاتتى ، بىراق دىگىنىمدەك بۇ ھېكايىنى بۇنچە تومتاق ئاياقلاشتۇرۇشقا راستىنلا كۆڭلۈم ئۇنىمايۋاتىدۇ ، سىلەرنىڭمۇ توغرا چۈشۈنىشڭلارنى ئۈمۈد قىلىمەن ، ھېكايىگە قارىتا قانداق تەكلىپ پىكىرلەر بولسا چىن دىلىمدىن قۇبۇل قىلىمەن ، ھەمدە تەكلىپ - پىكىر بېرىشىڭلارنى چىن دىلىمدىن ئۈمۈد قىلىمەن !

بۇ دۇنيادا بەدەل تۆلەيدىغىنى ئەخمەقلەر .............. ھوزۇرلىنىدىغىنى ئالدامچىلار !
