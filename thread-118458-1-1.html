  ئەسسالامۇ ئەلەيكۇم ھۆرمەتلىك (گەرچە ئىككى ياشقا توشمىغان بولساڭمۇ ،سىنى ھۆرمەتلىك دەپ ئاتىشىمغا روخسەت قىل)قىزىم خەدىچە! شۇ تاپتا تاتلىق پۇشۇلداپ ئۇخلاشلىرىڭدىن مەس بۇلۇپ ئولتۇرۇپ بۇ يازمىنى يېزىۋاتىمەن.پەرزەنت مۇھەببىتىنىڭ قانچىلىك ئولۇغ ،قانچىلىك تاتلىق بۇلىدىغانلىقىنى تەسۋىرلەشكە ئاجىزمەن.بەلكىم مىنىڭ سەن ئۇخلاپ ئويغانغۇچە كۆرگۈم كىلىپ چىدىماسلىقىم،داداڭنىڭ ھەرقىتىم سىنى ئەركىلەتكەنلىرىدە كۆزىگە ياش، ئاغزىغا سېرىق سۇ يېغىلىپ قىلىشلىرى بۇ نۇقتىنى ئانچە مۇنچە يورۇتۇپ بېرەلىسە ئەجەپ ئەمەس.باشقىلار توغرا ئېيتىدۇ...سەن ھەقىقەتەن داداڭنىڭ قىرىق يىشىدا كۆرگەن بالىسى!
   سەن قانداق شارائىتتا توغۇلدۇڭ؟ سۆزلىرىمگە قۇلاق سال
   قىزىم خەدىچە،سىنىڭ ئاپاڭ يەنى مەن زامانىمىزدىكى ئەڭ ئىلغار ھىسابلىنىدىغان ئالى مەكتەپنى پۈتتۈرگەن ،ئەمما باشقىلارنىڭ گىپىچە بىلىمىمنى ئۆزۈم خالاپ قارا ھىجاپ ئاستىغا بەخشەندە قىلىۋىتىپ،ھاياتىمنى مەنىسىز ئۆتكۈزىۋاتقان بىچارە بىر ئائىلە ئايالى.گەرچە كىچىكىمدىن باشلاپلا پۈتۈنلەي ئىسلامى بولمىسىمۇ ئىشقىلىپ دىندىن خەۋەردار بىر ئائىلىدە چوڭ بولغان بولساممۇ ،لىكىن دىنغا بولغان ھەقىقى قىزىقىشىمنى ئالى مەكتەپنىڭ ئاخىرقى يىللىرىدىن باشلاندى دىسەم توغرا بۇلاتتى،شۇ خىل قىزىقىش ،ئىستەكنىڭ تۈرتكىسىدە ئالى مەكتەپنى پۈتتۈرۈپلا ئورىنىپ ئالدىم .لىكىن مىنىڭ بۇ خىل تاللىشىم ئەڭ ئالدى بىلەن بەش ۋاق نامازنى قازا قىلماي ئوقۇيدىغان دادام بىلەن ئاپامنىڭ قاتتىق قارشىلىقىغا ئۇچرىدى.ئۇلارنىڭ قارىشىچە مەن نامىزىمنى ئوقۇساملا ئورىنىپ يۈرمىسەممۇ ھېچ ئىش بولماسمىش...ئەمما مەنمۇ تاللىشىمدىن ھېچ ۋاز كەچمىدىم،ئۇلار ئاخىرى نىمە قىلساڭ قىل دىگەن يەرگە كەلدى ،ئەمدى تۇرمۇش ئىشىمنى ئويلىنىشىم كېرەك ئىدى ،بۇ جەھەتتە كەسكىن تاللاشقا دۇچ كەلدىم نۇقۇل ھالدا بەش ۋاق ناماز ئوقۇيدىغان ئادەمنىلا تاللىسام بولمايتتى،چۈنكى بۇنىڭ مەنىسىنى چۈشىنىدىغانلار زادى قانچىلىك؟ ئۆزۈمدەك ئالى مەكتەپنى پۈتتۈرگەنلەرنى تاللىسام يەنە بولمايتتى ئۇلارمۇ ھەم مىنى تاللىمايتتى چۇنكى ئۇلارنىڭ ئىتىقادىنى مەن ياقتۇرمىسام مىنىڭ مۇقىم خىزمىتىمنىڭ يوقلىقىنى ئۇلار ياقتۇرمايتتى.شۇنداق كۈنلەرنىڭ بىرىدە جانابى ئاللاھ ماڭا داداڭنى يەتكۈزدى .داداڭ قانداقتۇ ئالى مەلۇماتلىق ئەمەس ، ئېسىلزادە بايمۇ ئەمەس، كاتتا ھوقوقدارمۇ ئەمەس ۋە ياكى كېلىشكەن سالاپەتلىكمۇ ئەمەس ،ياش پەرقىمزمۇ خېلى چوڭ ئىدى .لىكىن بۇلار كۆزۈمگە كۆرۈنمىدى مىنىڭ كۆرگىنىم پەقەت ئاللاھ دەپ سوقۇۋاتقان بىر قىزىل يۈرەك ئىدى.قىلچە ئىككىلەنمەي رازىلىق بەردىم ،توي كۈنى ھازىر كاتتا خىزمەت ئورۇنلىرىغا ئېرىشكەن دوست بۇرادەرلىرىم كونا ئادەت سۆزى بويىچە كۈيئوغۇل نىمە ئىش قىلىدۇ دەپ سوراشتى،مەن ناھايتى ئورغۇلۇق قىلىپ ئاللاھ بۇيرىغاننى قىلىدۇ،چەكلىگەننى قىلمايدۇ دەپ جاۋاپ بەردىم.ئۇلار مىيىقىدا كۈلۈشتى بۇ كۈلكىلەرگەمەسخىرە ، ئېچىنىش،كەمسىتىش يەنە قانداقتۇ مەنىلەر يوشۇرۇنغان ئىدى،ئاللاھىم ئۆزى ئۈچۈن بەدەل تۆلىگەنلەرنى ھەرگىز قۇرۇق قالدۇرمايدىكەن مانا ھازىر شۇلار قەھرىتان سوغۇق ،تومۇز ئىسسىقلاردا بىر ئاي ئىشلەپ تاپقان پۇلىنى سىنىڭ داداڭ بىر كۈندە تاپىدۇ،شۇلار 20-30 يىل قەرز قىلىپ ئالىدىغان ئۆيىنى سىنىڭ داداڭ بىر قولدا پۇل تۆلەپ ئالىدۇ،شۇلار نەچچە رەت سىلاپ ئاندىن باھا كۆرسەتكۈچۈگە قاراپ يەنە ئاندىن پۇل قاپچۇقىغا قاراپ ئۇھسىنىپ قايتىپ كىتىشكەن كىيىملەرنى ئاپاڭ خالىسا كۇندە بىرنىمۇ ئالالايدۇ ،ئەڭ مۇھىمى ئاپاڭ خاتىرجەم ئۆيىدە پۈتۈن زېھنى بىلەن ئاللاھقا ئىبادەت قىلالايدۇ مانا بۇ ئالاھنىڭ ماڭا بىرقەدەم يېقىنلىشىپ باق ،ساڭا مىڭ قەدەم يىقىنلاشمىغىنىمنى كۆر دىگەن ۋەدىسى....
   تويدىن كىيىن داداڭنىڭ پىكىر تىرەنلىكى ،دىننى چۈشەنچىلىرىنىڭ موللىقى،ئىتىقادى مەسلىلەردىكى مۇستەھكەملىكى ئالدىدا ئۆزۈمنى تولىمۇ يىتەرسىز ھېس قىلدىم.بىر جەھەتتىن كۆڭلۈم يېرىم بولسا يەنە بىر جەھەتتىن خوشمۇ بولدۇم چۈنكى ماڭا ياخشى بىر ئۆگىنىش سورىنى ھازىرلانغان ئىدى.شۇنداق كۈنلەرنىڭ بىرىدە سەن بىزنىڭ ھاياتىمىزدا پەيدا بولدۇڭ،شۇ كۈندىن باشلاپ ماڭا ياق بىزگە ئالەم باشقىچە ....
   ساڭا ھامىلدار بولغان كۇندىن باشلاپ داداڭنىڭ ماڭا بولغان تەلىپى يەنىمۇ يوقۇرلاشقا باشلىدى ،بۇلۇپمۇ يىمەك ئىچمەك جەھەتتە.ھېلىمۇ ئىسىمدەبىر كۈنى كەچتە تويۇقسىزلا كاۋاپ يىگۈم كىلىپ زادىلا بولالماي قالدىم ۋە بۇ ئىشنى داداڭغا ئېيتتىم لىكىن داداڭنىڭ بەرگەن جاۋابى بۇ يەردە ئىشەنچىلىك كاۋاپ يوق دىگەندىن ئىبارەت بولدى.شۇ كىچە پەقەت ئۇخلىيالمىدىم ،تۇنجى قىتىم داداڭدىن ئاغرىنىش ھېس قىلدىم چۈنكى كوچىدا كاۋاپ سىتىپ يۈرگەن ئويغۇر بالىلارنى كۆپ كۆرگەن ئىدىم ،داداڭ خۇددى ئىچىمدىكىنى بىلگەندەكلا دىدىكى
ئۇيغۇر بولسىلا مۇسۇلمان بۇلامدۇ؟ مەن يەنە بىر قىتىم ھەيران قالدىم چۇنكى بۇ مەن ئويلاپ باقمىغان يەنە بىر مەسىلە ئىدى.
   توققۇز ئاي ھامىلدارلىق پەسلىدە قانچە قىتىم خەتمە قۇرئان قىلىپ بولغانلىقىم ئىسىمدە يوق،مىنىڭچە سەن چۇقۇم ئاڭلىدىڭ، ئاللاھتىن ئامان ئىسەن تۇغۇلۇشىڭنى تىلەيتتىم، ئىمان -ئىتىقادلىق تۇغۇلۇشىڭنى تىلەيتتىم،ماڭا ئاسان توغۇت ئاتاقىلىشىنى تىلەيتتىم چۇنكى داداڭ سەن بۇيۇمدا قالغان كۈندىن باشلاپلا قەتئى  دوختۇرخانىغا ئاپارمايدىغانلىقىنى جاكارلىغان ئىدى.دەسلەپ سەل ئەنسىرىدىم ، كىيىن ئۆزۈمنىڭ ئىمانىمنىڭ ئاجىزلىقىدىن خىجىللىق ھېس قىلدىم،چۇنكى ئاللا ھېچبىر ئايىتىدە ئاياللار دوختۇرخانىدا بوشانسۇن دىمىگەن ،ئىسا پەيغەمبەرنىڭ ئانىسى بۇۋىمەريەم ئانىمىزمۇ بىر تۈپ دەرەخ يىنىدا يالغۇز بوشانغان ...مۇشۇلارنى ئويلاپ  كۆپ يىنىكلەپ قالغاندەك بولدۇم.ھەممە ئىشنى ئاللاھقا تاپشۇرۇش نىمە دىگەن راھەت...
   شۇنداق قىلىپ ئاي-كۈنۈم توشۇپ ئالدى بىلەن ئاللاھنىڭ رەھمىتىدە ئاندىن بىر مۇسۇلمان توغۇت ئانىسىنىڭ ياردىمىدە سەن بۇ دۇنياغا كۆز ئاچتىڭ.قىز ئىكەن دىگەن گەپنى ئاڭلاپ ئېختىيارسىز كۆزۈمگە ياش كەلدى چۈنكى جاندىن تويغۇزىدىغان ئاچچىق تولغاق سىنىڭ بىشىڭدىمۇ بار ئەمەسمۇ ؟  داداڭنىڭ خۇشاللىقى ئالەمچە ئىدى.داداڭنىڭ تىلىكى مۇنداق < ئاللاھ ماڭا ئون پەرزەنت ئاتا قىلسا ھەممىسى قىز بولسا>.
    چەكسىز ئۈمۈد ئارزۇلار بىلەن سىنى بىقىپ تەربىيلەۋاتىمىز،ئەمدىكى ئارزۇيۇم سىنى بىر ياراملىق پەرزەنت قىلىپ يىتىشتۈرۈپ چىقىش. بىزنىڭ نەزىرىمىزدىكى ياراملىق دىگەن سۆزنىڭ مەنىسى مۇنداق:شۇ نۇرلۇق دىللىرىڭ ئاللاھنى تونىسا، شۇ چۈچۈك تىللىرىڭ ئاللاھ دەپ زىكرى قىلسا،شۇ ئۆتكۈر كۆزلىرىڭ ئاللاھنىڭ مۆجىزىلىرىنى كۆرەلىسە،شۇ سەزگۈر قۇلاقلىرىڭ ئاللاھنىڭ كالامىنى ئاڭلىسا ،شۇ چەبدەس پۇتلىرىڭ ئاللاھ ئۈچۈن بىسىلسا ئەڭ مۇھىمى ئۆزۈڭنىڭ ھەر ۋاقىت ئاللاھنىڭ نازارىتى ئاستىدا ئىكەنلىكىڭنى ھېس قىلالىساڭ .......
