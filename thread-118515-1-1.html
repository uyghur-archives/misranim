    "شىنجاڭ بانكىسى"نىڭ مۇكەممەل ئۇيغۇرچە تور بانكا مۇلازىمىتىنىڭ بولىشى توغرىسىدا ھۆكۈمەتكە تەلەپ


تورداشلارغا مەلۇم بولغىنىدەك "شىنجاڭ بانكىسى" قۇرۇش تەستىقلىنىپ،ۋىۋىسكا ئېسىش خىزمىتى بۇ يىل تاماملىنىدىكەن.
خەلق تورى خەۋىرى-شىنجاڭ بانكىسى قۇرۇش رەسمىي تەستىقلاندى
http://uyghur.people.com.cn/156325/15384715.html
شىنجاڭ كوئىنلۇن تورى خەۋىرى-شىنجاڭ بانكىسى ئاپتونوم رايونىمىزنىڭ تەرەققىياتىغا «قانات» بولىدۇ
http://uyghur.xjkunlun.gov.cn/10060/10180/00018/10000/2014/2097702.htm
ئۇنداقتا بۇ يەردە بىر تەكلىپنى ئالدىن بىرىشنى ئويلاشتىم،چۈنكى ئىش پۈتۈپ بولغاندا دىگەن بىلەن قۇلاققا خۇش ياقمايدۇ،رولىمۇ ھازىرقىدەك چوڭ بولمايدۇ دەپ قارايمەن.
"شىنجاڭ بانكىسى"نى قۇرۇشنىڭ ئاساسى مەقسىتى شىنجاڭنىڭ ئىقتىسادى تەرەققىيەتىغا تېرەكلىك كۈچ قوشۇش. شۇڭا بۇ ئىقتىسادى تەرەققىياتنى ئىشقا ئاشۇرۇشنى بۇ جەرياندىكى كەم بولسا بولمايدىغان تايانىچ كۈچلەرنىڭ بىرى، شىنجاڭ نوپۇسىنىڭ ئازكەم يىرىمىنى تەشكىل قىلىدىغان ئۇيغۇرلارنىڭ تەرەققىياتىدىن ئايرىپ قارىغىلى بولمايدۇ.
ئەمما كۆز سالساق ھازىر شىنجاڭدابانكا ساھاسىدە بىرەر بانكىنىڭ توربانكا مۇلازىمىتى ئۇيغۇر تىلىدا تەمىنلەنگىنى يوق،بىزدە بۇ راسلا جىددى ھەك قىلىشقا تىگىشلىك بوشلۇق.بىز ئاپتونۇم رايۇنىمىزنىڭ قۇرۇلغۇسى"شىنجاڭ بانكىسى"نىڭ پۈتۈنلەي،مۇكەممەل ئۇيغۇرچە توربانكا تورىنىڭ بولىشىنى،خەنزۇچىنى بىلمەيدىغان كىشىلەرنىڭمۇ بىمالال قالدۇق سوممىسىنى سۈرۈشتۈرۈش،ھىسابات ئاغدۇرۇش قاتارلىق بانكا مۇلازىمىتىدىن تولۇق100% ئۇيغۇرچە تور مۇھىتىدا بەھرىمان بولىشىنى سورايمىز(ھىسابات يۆتكەشتە كارتا ئىگىسى نامىغا كەلسەك،بولسا خەنزۇچە بىلمەيدىغان كىشىلەرگە قولايلىق يارىتىلسا،زادى بولمىسا ئىسىم ۋاقتىنچە خەنزۇچە بولسىمۇمەيلى،چۈنكى ئىسىمنى كومپىيۇتىردا كۆچۈرۈپ چاپلىغىلى بولىدۇ ئەمەسمۇ)،ئەمما مۇشۇنىڭدىن باشقىسىنىڭ ھەممىسى،ئەڭ ئەقەللىسى كۆرۈنمە يۈزى 100%ئۇيغۇرچە بولسا.
بۇ توربانكىسى 100%تولۇق ئۇيغۇرچە بولسا ئىلىپ كىلىدىغان پايدىلار:

1.بىز ئۇيغۇرلارنىڭ تور سودا كەسپىمىزمۇ گۈللىنىۋاتىدۇ،لىكىن بۇيەردىكى ئەڭ چوڭ مەسىلە 支付宝，财付通دىگەندەك بىخەتەر پۇل تۆلەش قوراللىرىغا ھىسابات ئاغدۇرۇشتا،ئۇيغۇرچە تور بانكىسىنىڭ بولماسلىقى بىر تەبىئىي توساق بولۇپ قالدى. شۇڭا ئۇيغۇر سودا تورلىرى باشقا جەھەتتىن مۇكەممەل بولسىمۇ مۇشۇ ئۈچىنجى تەرەپ،يەنى تور سودىسىدىكى ئەڭ مۇھىم ئۆتكەلنىڭ كەمتۈكلۈكى تۈپەيلىدىن تەرەققى قىلىشقا  يۈزلىنەلمىدى. بەلكى تور سودىسى تەرەققى قىلسا ئىشقا ئورۇنلىشىشقا تەسىرى بار دىيىش مۇمكىن،لىكىن بۇ قاراش تاۋباۋ، دوڭجىڭ قاتارلىق تور بەتلەرنىڭ تەرەققىياتى ۋە ئىقتىسادى ئۈنۈمى بىلەن ئاغدۇرۇلدى. ئۇيغۇرلاردىكى مۇكەممەلرەك بولغان توردىكى سودا تور بەتلىرى:       http://www.tawarim.com/      http://www.alnur.com.cn/     www.xjtawar.com    http://www.xjsoda.com   مۇشۇلارنىڭ ھەممىسى مۇشۇ مەسىلىگە دۇچ كەلدى،يەنى ئۇيغۇرچە تور بانكا مۇلازىمىتى تەمىنلەيدىغان بانكا يوق.

2. تور سودىسى گۈللەنسە رايۇنىمىزنىڭ تىز يەتكۈزۈش كەسپىنىڭ  تەرەققى قىلىشىغا مۇھىم تۈرۈتكىلىك رول ئوينايدۇ،نۆۋەتتە رايۇنىمىزدا تاۋار ئالماشتۇرۇشىنىڭ ئاستا،سودا رىقابىتىنىڭ تۆۋەن بولىشىدىكى ئەڭ مۇھىم سەۋەپلەرنىڭ بىرى كۆلەملەشكەن تىز يەتكۈزۈش كەسپىنىڭ بولماسلىقى،بولغانلىرىنىڭ چوڭ شەھەرلەرگىلا مۇلازىمەت قىلىشى ۋە باھاسىنىڭ يۇقىرى بولىشى. ئەگەر تور سودىسى كۆلەملەشسە تىز يەتكۈزۈش كەسپى يىڭى تەرەققىيات پۇرسىتىگەئىرىش تىخىمۇ كۆپ تىز يەتكۈزۈش شىركەتلىرى مەيدانغا كىلىپ،رىقابەت كەسكىنلىشىپتو شۇش تەننەرخى تۆۋەنلەيدۇ. تىز يەتكۈزۈش كەسپىنىڭ تەرەققىياتى رايۇن تەرەققىياتىغا تۆھپىسى زور،بىر رايۇن تەرەققى قىلىشى ئۈچۈن يول بولىشى كىرەك،ھازىرقى زاماندا يول پەقەت كۆز بىلەن كۆرگىلى بولىدىغان فىزىكىلىق جىسىملا بولماستىن بەلكى مەۋھۇم بولغان تىز يەتكۈزۈش كەسپمۇ بەك مۇھىم يولدۇر. يىقىنلاردا  دىيىلگەن بىر سۆز خىلى ئەتراپلىق ئىيتىلغاندۇر،يەنى:  رايۇننىڭ  ئىقتىسادى  تەرەققىياتىغا  شۇ رايۇننىڭ  تىز يەتكۈزۈش كەسپىنىڭ  تەرەققىياتى بىلەن باھا بەرگىلى بولىدۇ.شۇنداق، تاۋار  ئالماشتۇرۇش قانچە تىز بولسا،ئىقتىسادى تەرەققىياتمۇ  شۇنچە تىز بولىدۇ.

3. تىز يەتكۈزۈش  كەسپى تەرەققى قىلسا،يەككە ئىگىلىك زور دەرىجىدە يۈكسىلىدۇ.ھەرقانداق كىشى ئۆزىنىڭ ئىشلەپچىقارغان مەھسۇلاتىنى مىقدارىنىڭ ئاز كۆپلىكىگە قارىماي ساتالايدۇ.ھەتتا غۇلجىلىق دىھقان ئۆزىنىڭ قىچاسىنى خوتەندىكى زاۋۇتقا،خوتەندىكى گىلەمچى ئايال گىلىمىنى تۇرپان ھەتتا شېنجېندىكى ئۇيغۇرغا، قەشقەرلىك ھۆنەرۋەن سىپتا بويۇملىرىنى شىنجاڭنىڭ يەنە بىر  يىرىگە ساتالايدۇ. بۇنداق سودا ھەم ئىشلەپچىقىرىش تەننەرخنى تۆۋەنلەتسە،ھەم ئەشيا تىجەيدۇ."دۇنيادا ئەخلەت يوق،پەقەت ئورنى خاتا قويۇپ قويۇلغان نەرسە بار"دىگەندەك،ئاقسۇدىكى بىرسىگە كىرەكسىز نەرسە،خوتەندىكى بىرسىگە كىرەك بولىشى مۇمكىن،خوتەندىكى  بىرسىگە  كىرەكسىز  نەرسە  ئاقسۇدىكى  بىرسىگە كىرەك بولىشى مۇمكىن، "بايقىساڭ توپىمۇ ئالتۇن".  كىشىگە ئازراق بىمەنە تۇيۇلسىمۇ ،بۇ يەردە دىيىش ئانچە  ياخشى بولمىسىمۇ بىر ئىش ئىسىمدە:   بىزنىڭ يېزىنىڭ ئوتتۇرا مەكتىپىنىڭ خالاسىنىڭ ئورىكى گەندىگە تولۇپ كىتىپ تارتىدىغاندا پۇل بىرەي دەپمۇ تارتىدىغانغا ئادەم يوق،تۇيۇقسىز ئىلان بىرىشنىڭ ئورنىدا مەكتەپتىكى بىر ئوقۇتقۇچى ئوقۇغۇچىلارغا يىغىن ئىچىپ بوز يەر ئۆزلەشتۈرىدىغان ئاتا-ئانا، ئۇرۇق تۇققانلىرى بارلار ئۇلارغا ئىيتىپ قويۇڭلار،مەكتەپ خالاسىدىكى "ئوغۇت" ھەقسىز سېتىلىدۇ دەپ ئۆتكەنتى، شۇ گەپ دىيىلىپ نەچچە كۈن ئۆتمەيلا بۇز يەر ئاچىدىغان بىر نەچچەيلەن كىلىپ تالىشىپ دىگۈدەك يۇقىرى بىسىملىق سۇ تارتىش پومپىسى بىلەن ھەممىنى تارتىپ بىكالاپ ئىلىپ كەتكەن. مەكتەپ چىقىمسىز بىكارغىلا بۇ باش قىتىقچىلىقىنى يوقاتقانغا خۇش،دىھقان بىكارغا بىرىلگەن نەچچە توننا ئوغۇتقا خۇش.دىمەككىم ئۇچۇر ،ئالماشتۇرۇش بولسا ئەشيا تىجەش بولىدۇ،مۇھىت بۇلغىنىش ئاز بولىدۇ.ئالماشتۇرۇش بولسا ھەممە نەرسە قىممىتىنى تاپالايدۇ.يەنە شۇگەپ "دۇنيادا ئەخلەت يوق،پەقەت ئورنى خاتا قويۇپ قويۇلغان نەرسە بار"،تور سودىسى بىلەن تىز يەتكۈزۈش كەسپىنىڭ تەرەققى قىلىشىغا ئەگىشىپ يىنىمىزدىكى "ئەخلەت"لەرمۇ پۇلغا ئايلىنالىشى،ئەشيا تىجىلىشى، بۇ ئارقىلىق مۇھىت بۇلغىنىش تۆۋەنلىشى مۇمكىن.


4. ئۇيغۇرلار ھۆكۈمەتنىڭ ئاپتونومىيە قانۇنىنىڭ ئەمەلىيلەشكەنلىكىدىن سۆيىنىدۇ،مىللەتلەر ئىناق ئۆتۈشتە بۇنى ئۇيغۇرلار ئۆزلىرىگە قىلىنغان ھۆرمەت دەپ بىلىدۇ،ۋە پارتىيە ھۆكۈمەتنىڭ ئەمەلىي ھەركىتىدىن سۆيۈنىدۇ، دۆلەتنى تونۇش، سۆيۈش تۇيغۇسى ئاشىدۇ. مىللى تىل-يېزىق توغرىسىدا باتناشلار ئازراق بولسىمۇ پەسىيىدۇ.جۈملىدىن بۇ ئىشتىن رازى بولمايدىغان ئۇيغۇر يوق.خەلق رازى بولدىغان،ئىقتىسادى تەرەققىياتقا پايدىسى بار ئىشنى "خەلق ئۈچۈن خىزمەت قىلىش"نى ئۆزىگە قىبلىنامە قىلغان ھۆكىمىتىمىز چوقۇم قىلىشقا ئالدىرايدۇ دەپ قارايمىز،ۋە ئۈمۈد كۈتىمىز.ئەمەلگە ئاشمىسا خەلقتەئاغرىنىش بولىدۇ،ئەمەلگە ئاشسا خەلق رازى بولىدۇ.


5.بۇ قۇرۇلغىسى"شىنجاڭ بانكىسى" ئۈچۈن شىنجاڭدا خەلق ئارىسىدا مۇكەممەل ئاساس تىكلىشى،ۋە رىقابەتچىلىرى بىلەن بولغان كۈچ سىنىشىشتا ئۇتۇپ چىقىشىدا مۇھىم رول ئوينايدۇ. ئۇيغۇرچە تور بانكا تورىنىڭ بولىشى بىلەن" شىنجاڭ بانكىسى"مۇتلەق كۆپ قىسىم ئۇيغۇر خەلقىنى ئۆزىگە رام قىلالايدۇ،شۇنداقتا ئۇيغۇر خەلقى بىلەن پۇل-مۇئامىلە  قىلىدىغان تىخىمۇ كۆپ خەنزۇلارمۇ "شىنجاڭ بانكىسى"نىڭ مۇئامىلىدارى بولىدۇ. چۈنكى ئۇلار بىلەن مۇئامىلە قىلىدىغان  ئۇيغۇرلار ئاساسەن"شىنجاڭ بانكىسى"نىڭ  مۇئامىلىدار  ئەمەسمۇ . بۇنداق بولغاندا "شىنجاڭ بانكىسى"نىڭ تەرەققىياتى تىز بولۇپلا قالماي، پۇل مەبلىغىنىڭ شىنجاڭ ئىچىدە ئايلىنىشى ۋە بانكا  ساقلانما  مەبلىغىنىڭ  شىنجاڭدا  مۇقىم ئارتىشى ۋە مۇقىم ئاساسى بولىشىنى ئىشقا ئاشۇرغىلى بولىدۇ.

شۇڭا  ئالدى بىلەن ئۇيغۇر خەلقى نامىدىن،ئاندىن يۇرت سۆيەر شىنجاڭ خەلقى نامىدىن، شىنجاڭنىڭ ئىقتىسادى تەرەققىياتى ئۈچۈن "شىنجاڭ بانكىسى"نىڭ مۇكەممەل ئۇيغۇرچە تور بانكا مۇلازىمىتىنىڭ بولىشىنى سورايمىز. ھەم ئاپتونومىيە قانۇنىنى چىقىش قىلىپ تۇرۇپ،ئۇيغۇرچە مۇكەممەل تور بانكا  مۇلازىمىتىنىڭ بانكىنىڭ قۇرۇلۇش جەريانىدىكى مۇھىم كۈن تەرتىپلەرنىڭ بىرى بولۇپ،بانكىمىز بىلەن بىرلىكتە روياپقا چىقىشىنى تەلەپ قىلىمىز.
