مەن ھەقىقەتەن ھەيران قالدىم، ئۇ مۈشۈك قۇيرىقىنى ئاسمانغا قىلىپ قۇسىقىنى سۆرەپ ماڭىدۇ،  خوشنامدىن، نىمە بوپتۇ دەپ سورۇسام،  ئۇ ماغا چۈشۈپتۇ دەيدۇ، كىيىن بىر ئەركەك مۈشۈك تېپىپ ئەكەلدى،  چىشى مۈشۈك ئەركىگىنىڭ ئىگىكىگە كىرىپ،  ئىشقىلىپ ئۇ بەكلا ناز قىلدى، مەن خوشنامدىن ئەركەك مۈشۈكمۇ ماغا چۈشەمدۇ؟ دىسەم، كۈلۈپ قويۇپ ئۇنى ئۇقمايمەن، ئەمما چىشىسىنىڭ ھەركىتىدىن بىلۋاغىلى بولىدۇ، دىدى

مەن بىلۋالدىمكى چىشى ھايۋانلار ئەۋلات قالدۇرماقچى بولسا ئەڭ كۈچلۈك ئەركەك ھايۋانلارنى چاقىردىكەن
2 يىلدىن كىيىن  ئامرىكا
ۋېلىسىپىتلىك بىر قىزساۋاقدېشىمنىڭ تۇغۇلغان كۈن مۇراسىمىغا كىتىپ بارىمەن، مەن بۇ يىل 19 ياش، ئۇيغۇر ئامرىكىدا تۇغۇلۇپ ئامرىكىدا چوڭ بولدۇم، دادام ئۇيغۇر، ئانام تۈرۈك، ئۆيدە كۈپۈنچە ئۇيغۇرچە پاراڭلىشىمىز،  تاغامنىڭ دەپ بىرىشىچە ئانام دادامغا كۈيۈپ قېلىپ ئۇيغۇرچە ئۈگەن ئىكەن،قەقىقەتەن مىنىڭ دادام بەكلا قىززىق ئادەم، يۇرتىنى سېغىنىدىيۇ ئەمما بارمايدۇ، دادام ئۈزۈنىڭ ئاتا -ئانىسىنى ئۇقمايدىكەن، دادام ئىسىنى ئۇققاندا ئۇ بىر پاراۋانلىق ئورنىدىكەن، دادام چىننىڭ شىنجاڭ غۇلجىدا چوڭ بوپتىكەن، دادامنىڭ دىيىشىچە ئۇيەر تەبىيكى بىر جەننەتمىش،  دادامغا بىرە يىلى بېرىپ كىلەيلى دىسەم،  ئۇيەردە تۇققانلىرىمىز يوق، ئۇجايغا بارغىچە تۈركىيە ئىستانبولغا بار، تاغاڭلا بار، ھامماڭلار بار، دەيدۇ، جىق گەپ قىلسام دادام مەندىن قاچىدۇ، مەن يەنىلا دادامنىڭ يۇرتىغا بېرىشنى بەكلا ئارزۇ قىلمەن،  ھەي ئامرىكا   بۇجاي ھەرخىل مەدەنىيەتلەر توپلانغان جاي، بىزنىڭ ئويدىمۇ خىلمۇخىل مەدەنىيەتلەر مەۋجۇت،   مىنىڭ ئاچام تىپىك ئامرىكىلىق، مەن ئۇنى ياقتۇرمايمەن، ئۇنىڭغا تەربىيە قىلغۇم كىلىدۇ، ئەمما مىنىڭ تەربىيەم 100- پىرسەنت ئۈنۈمسىز، مەن ئۇنداق قىمما، دەپ ئۇ ئىشنى شەرئىلىسەم ئاچام ئىككى ھەسسى ئاشۇرۇپ قىلىدۇ، دادامدىن ئازراق قورقۇدۇ خالاس، بوممىسا ئۇ ... بولدىلا .... دادام بىرە ئايچە بولدى، بەش ۋاق ناماز ئۇقۇيدىغان بولۋالدى،  بۇرۇن جۈممە ۋەياكى ئەتىگەندىلام ناماز ئۇقۇيتى، يېشى ئەللىككە يەتكەندە، تۇرۇپلا نامازخانا بولۇپ كەتتى، يا خۇش بولشۇمنى ياكى خاپا بولشۇمنى بىلمىدىم، دادام نامازخان بولۇپ،  يا ئاچامنى سەل پاكىز كىيىنىشكە بۇيرۇمايدۇ، مەن ئۈزۈمنى ئۈزۈم باشقۇرۇپ يۈرىمەن،  ئۈزۈمنى باشقۇرمىسام، دادام ماڭىمۇ تازا تەلەپ قويمامدىكىن دەيمەن، دۇنيادا ھەممە ئاياللارنىڭ كاسىسىنى كۆرسەممۇ ئاچامنىڭ، ئانامنىڭ كاسىسىنى كۆرگۈم يوقتى،  لىكىن ئاچام داۋامەت كاسىسىنى غادايتىپ يۈرۈيدۇ، گەپ قىلسام، ماڭە ماڭقا، تۈنۈگۈن ماڭقاڭنى تاتامماي يۈرەتتىڭ، ماڭقاڭنى تاتىشنى ئۈگۈتۈپ قويسام ماڭا تەربىيە قىلغۇدەك بولدىڭما؟ دەيدۇ ئىككىمىز ئۈچۈنجى دۇنيا ئۇرىشىنى باشلايمىز، ئانام دادام، بىرى تۈرۈكچە بىرى ئۇيغۇرچە بىزنى ئاجرىتىشىدۇ، ئاچام شۇ ھامان ئىڭگىلىزچە مىنى تىللايدۇ، ئۇلار ئىڭگىلىزچە سۆزلىسە ئاچام  باشقا تىلدا تىللايدۇ،  نىمە ئۈچۈنكى باشقا بالىلار بىلەن ئۇرۇشمايمەن، ئاچام بىلەنلا شۇنداق ئۇرىشىمەن.    بۇ دىگەن ئامرىكا  ئەمدى باشقىلار مۇسۇلمانلارغا چاندۇرمىسىمۇ سەل بۈلەكچە قارايدۇ، شۇڭا، گەرچە مىنىڭ ئاتا -ئانام بىز مۇسۇلمان دەپ بەك ئىتقاتتا چىڭ تۇرمىسىمۇ مەن مۇسۇلمان سانىلىپ،  ساۋاقداشلىرىم مىنى بۇ مۇسۇلمان دەپ يۈرۈدۇ، شۇڭا بۇيەردىكى قارا تەنلىك، ئاق سېرىق تەنلىك مۇسۇلمان بالىلار بىلەن بىللە يۈرۈپ( ئۇلارنىڭ ئاتا -ئانىلىرى ئىتقاددا بەك چىڭ)  ئۇلارنىڭ تەسىرىگە ئۇچرۇدۇم،   شۇڭا تار ئىشتانلىق قىزلارنى ياقتۇرمايمەن،  نەزىرىمدە قىزلارنىڭ ئۇ يەرلىرىنى ئېرىلا كۈرۈشكە لايىق.    بىر كۈنى بىزنىڭ بىر ئايال مۇئەللىمىمىز بىزگە،  ئۈزۈڭلارنىڭ ئارزۇ -ئارمىنىڭلارنى،  قەلبىڭلار دىكى دۇنيارىڭلارنى تەۋرىرلەپ بىردىن، ماقالە يېزىڭلار دەپ يېرىم سائەت ۋاقىت بىرىپ، يېزىپ بولغانلارنى ئۇقۇغۇلى سالدى،  بىر ساۋاقدېشىم دەيدۇ،  يوللار، تۈرۈكلەر، ئۆيلەر ھەممىسى شاكلاتتىن ياسالسا، كىتىپ بېرىپ يۇنۇپلا يىيەلىسەك .... يەنە بىر نىگىر ساۋاقدېشىم،  مەن تىرەمنىڭ ئاق بولۇشۇنى ھەممىدىن بەك ئارزۇ قىلىمەن دەيدۇ.... يەنە بىر قىز ،   ئۈزىنىڭ مەلىكىلردەك بولۇشلۇرىنى چوڭ كىيىم ئالماشتۇردىغان ئۈيى بولىشىنى،  كىتاپ ئۇقۇماقچى بولسا 10 قەۋەتلىك كىتاپخانلار  بولىشىنى .... يەنە بىرسى مەن جوبىستەك بولىمەن...   بىلگەيتىس كە ئوخشاش ياشاشنى ياقتۇرمەن دىگەنلارمۇ بار، ئۇچالىسامكەن ..... جىق قىز دوستۇم بولسىكەن،   چوڭ بولساق خىزمەت قىلمىساقمۇ ماشىنىمىز، پۇلىمىز، ئۈيىمىز بولسا .... ئىشقىلىپ بۇ خالىغانچە خىيال قىلىشقا بولىدىغان قىزقارلىق بىر ئىش ئىدى، نۆۋەت ماڭا كەلدى،  مەن، دۇنيادىكى ھەممە قىز ئاياللار تار ئىشتان كەيمىگەن،چىرايلىق يوپكا كەيگەن،  كىندىكىنى، كۆكسىنى ئۇچۇق قويمىغان،ئەرلەرنىڭ ئالدىدا ئۆزلىرىنى بازارغا ساممىغان،  ئەرلەر قەددى قامەتلىك، ئىلكىدىكىلەرنى ئوبدان يولغا باشلىيالايدىغان بىر جەمىيەت بولىشىنى ئارزۇ قىلىمەن دىدىم، سىنىپتىكى بالىلار كۈلۈشۈپ كەتتى، نەچچە ئوغۇل بالا مىنى،  ئۇ قىزلارغا قىززىقمايدۇ،  ئۇنۇڭ ھېلىقى بىرنىمىسى يوق، ياكى كىچىك، دەپ ئوراڭگوتاننى دوراپ پارتىدا ماڭدى، مىنى ئۈچۈنجى دۇنيا ئەللىرىنىڭ تىپىك مىسالى دىدى، مەن چاپراپ ئۇنىڭغا ئېتىلدىم بالىلار بىزنى ئاجرىتىشتى، راستىنى دىسەم سىنىپتىكى ھەممە ئۇغۇل بالا ماڭا ھەسەت قىلشىدۇ،  ئۇ ھەسەت كۈچىدىن مىنى تىللىدى، باشقا مەخسەت يوق،  مەن بەكلا كىلىشكەن،  سىنىپتىكى ماڭا خەت يازمىغان قىزلار ساناقلىقلا، ئىلخەت ئارقىلىق خەت يازىدىغانلارمۇ كۆپ،  ماڭا ئۈزىنىڭ ھەرەڭ   سەرەڭ رەسىملىرىنى يوللايدۇ،  19 ياشتا بەزىدە ئىختىيارسىز ئىرادەمگە خىلاپ شۇلارنىڭ شۇ رەسىملىرىنى كۈرۈپمۇ سالدىم،  ئارسىدا ئۈزىنىڭ يالىڭاچ رەسىمىنى يوللىغانلارمۇ بار،   كىيىن ئۈزۈمنى قاتتىق تۇتۋېلىپ ئىلخېتىمنى ئاچمىدىم 3 ئاي بولدى،،،،   شۇنىڭ بىلەن مىنى ئۇقۇتقۇچۇم ئىشخانىسىگە ئېچىقىپ،  قارا بالام،  سەن مۇسۇلمان، ئۇلار ھەرخىل دىندا،  مەكتەپتە دىننىي تۇقۇنۇشى قىلساڭلار بوللمايدۇ،  سىنىڭ دۇنيارىىڭ بەكلا گۈزەلكەن، سەنمۇ ئۆز دۇنيارىڭدا ياشا ۋەرگىن، ئۇلارمۇ ئۆز دۇنيارىدا ياشا ۋەرسۇن،   ئاقىبەت سەنمۇ ئۆز نىسبەڭگە ئۇلارمۇ ئۆز نىسىبىسىگە ئېرشىدۇ،  سىنىڭ ئەخلاق قارشىڭ كىيىم - كىچەككە ئارلىشىدىكەن، ئۇلارنىڭ ئارلاشمايدۇ،  شۇڭا ئۇلارنىمۇ چۈشنۈپ قوي دىدى،  مەنمۇ ماقۇل دىدىم،  بۇ ئۇقۇتقۇچىمىز سېرىق چاچ،  دايىم سېرىق رەڭلىك تىزىدىن ئۇرۇن يوپپا كىيدىغان بىر ئوتتۇرا ياش ئايال،  ئۇ مىنىڭ بېشىمنى سىلاپ تۇرۇپ، چىقىپ ئۈگۈنۈشۈڭنى قىل دىدى،  بەكلا خۇرسەن بولدۇم، ئۇمىنىڭ نەزرەمدىكى ئەڭ ياخشى مۇسۇلمان ئەمەس ئادەم، ھەم شۇلارنىڭ ئەڭ ھۆرمەتلىكرەكى   .ھەي مەن نىمىلەرنى دەپ كەتتىم،   بۈگۈن بىر قىز ساۋاقدېشىم تىلفۇن قىپتۇ،  بۈگۈن مىنىڭ تۇغۇلغان كۈنۈم،  ھەممە بالىلار كىلىدۇ، سەنمۇ كەل دەپ،  مەن ساۋاقداشلىرىمنىڭ تۇغۇلغان كۈنى تازا چوڭ بىلىپمۇ كەتمەيمەن،  ئەستىمۇ ساقلىمايمەن،  بۈگۈن بارمىسام،  ئۇغۇل بالىلارنىڭ ھەممىسى مىنى زاڭلىق قىلىدۇ،  شۇڭا كىتىپ بارىمەن،  مانا بۇ قىزنىڭ ئۈيىگە كەلدىم،  ھىچ بىر تۇغۇلغان كۈن ئۆتكۈزۈلۋاتقاندەك ئەمەس،  قوڭغۇراقنى باستىم،   سارا ئىشىكنى ئاچتى ئۈيىدە ھىچ ئادەم كۈرۈنمەيدۇ،  ئىچىدە بالىلار كۈرۈنمەيدىغۇ دىسەم ھازىر كىلىدۇ، دىدى، كىرىپ كىروسلودا ئولتۇدۇم،  ئۇ قىزغىن قارشى ئالدى،  ماڭا ئۆيلىرىنى كۆرسەتتى،  ھۇجرىسىنى كۈرۈپ بېقىشىمنى دىدى،  كىرسەم ھۇجرىسىدا توپچىلارنىڭ رەسىمى،  ( مۇسكۇللۇق) ھەرخىل كىنو چولپانلىرىنىڭ رەسىملىرى تامغا چاپلانغان،  ئاچامنىڭ ھۇجرىسى بۇنىڭدىن ياخشىكەن، ئۇ مىنى بىردەم ئولتۇرۇپ تۇرۇڭ دەپ چىقىپ كەتتى، ئۇنىڭ نەرسە كىرەكلىرىنى كۆردۈم،  بىريەلەدىن سەسكىنسەم،  بىريەلەدىن قىززىقۋاتىمەن،   قەلبىمنىڭ بىريەرلىرى مىنى چىكىتىشكە ئۈندى ۋاتىدۇ،  ئەمما بەدىنىم ھىچ مىدىرلايدىغاندەك ئەمەس،  ئېشىك ئېچىلدى،  بىر غېرىچ يوپكا كەتكەن تور ئاسما كۆينەكلىك سارا  (ئىسمى )ماڭا قاراپ كەلدى،  ئۇ ماغا چۈشكەن مۈشۈكنىڭ ئۈزى ئىدى،  ماڭا ئېتىلدى، مىنى سۈيۋاتىدۇ .. مىنىڭ كاللام قۇپ قۇرۇق .......  ئو مەن ھازىر بۈلەك ئادەم،  ئۇ مەندىن تەلەپ قىلۋاتقان ئىشنى مەن ھازىر ئۇنىڭدىن 10 ھەسسە كۈچلۈك ئىرادىدە تەلەپ قىلۋاتىمەن ... ئۇ موزوينىڭ ئۈزى بولدى، مەنمۇ كۆڭلىكىمنى سالدىم،  ئىشتاننى سامماق بولدۇم،  تاسمامنى بۇشۇتۋاتىمەن .... ئەمما بۇشۇمۇۋاتىدۇ،  نىمىشتۇ بۇ،  قۇلپىسى ئېچىلماۋاتىدۇ،  ئۇمۇ تاسمامغا ھەپلىشىپ ئاچاممىدى،  قېرىشقاندەك سىرىتماممۇ يەنجىلىپ ئېچىلمىدى،  مەن ئوڭايسىز لاندىم،  كاللام سەل ئەسلگە كىلۋاتىدۇ،  ئاللاھ    ..... يا ئاللاھ  مەن سىنىڭ مىنى كۈردىغانلىقىڭنى،  سىنىڭ مەۋقۇتلۇقۇڭنى ئۇنتۇپ تىمەنغۇ؟ چاقماق تىزلىكىدە يۈسۈپنىڭ قىسسىسى ئىسىمگە كەلدى، توغرا يۈسۈف نىمۇ زۈلەيخا زىناغا چاقىرغان، قۇرئاندا دىلىشىچە ئاللاھ،يۈسۈپكە ئۇچۇق ئالامەتلەرنى كۆرسەتمىگەن بولسا يۈسۈفمۇ زۈلەيخاغا مايىل بولۇپ قالغان بولاتتىكەن تۇق .... ئەجەبا بۇ ( تاسمامنىڭ قۇلپىسىنىڭ ئېچىلمىغانلىقى)  ماڭا ئاللاڭنىڭ كۆرسەتكەن روشەن ئالامەتلىرىدىنمۇ؟؟ كاللام قېيىۋاتىدۇ،  ئالدىمدا بىر تۇغما ئايال،  يا ئاللاھ دىدىمدە كۆڭلىكىمنى ئېلىپ ئۇنىڭ ئۈيىدىن قاچتىم،  ئۇ ئىشىكلەرنىڭ ئىلىپ قۇيۇپتىكەن،  ئىچىدىن ئىلغان بولغاچ قۇلۇمدىلا تاتىپ ئېچىپ قاچتىم،   كۆڭلىكىمنىڭ كىيىپلا ۋېلىسپىتلىرى مىنىپ ئۇدۇل كەلگەن ئەڭ يىراققا كەتتىم،  ئېغىزىمغا پەقەت،  ئاللاھىم مەن ئېزىپتىمەن، مىنى كەچۈر دىيىشتىن باشقىسى كەممەيۋەتىدۇ،  قوللىرىم تىترەپ ھىلىلا بىرە ماشىنىنىڭ تىگىغا كىرىپ كەتمەي دەپ،  ئادەم جىق بىر بىكەتكە بېرىپ ئورۇندۇقتا ئولتۇرۇپ ئويلاندىم، ئى ئاللاھ مىنى كەچۈر، سەن يامان كۆرگەن بىر ئىشنى قىلدىم،  ساڭا ھەم ھەمدۇ سانالار ئېيتىمەن ئاللاھىم، سەن ماڭا يۈسۈفكە ئاتا قىلغاندەك رەھمەت ئاتا قىلدىڭ..... ئۇنىڭغا تەۋبە قىلغاچ ھەمدۇ سانا ئېيتىۋاتىمەن، مەن نىمىشقا مۇشۇ كۈنگە قالدىم؟(  ھەرقېتىم بىرە ئىش ئەپلەشمىسە ئۈزۈمدىن سۇرايدىغان سۇئالىم، جاۋاپ ئىزدەۋاتىمەن ).... توغرا بىلەتتىم،  غىلپال ئۇنىڭ ماڭا قىلتاق قۇرغىنىنى بىلەتتىم ،ئەمما ئۈزۈمنى ئالداپ، ياشلىق ئۇچقۇنىدا ئۇچۇپ ئۇنىڭ يېنىنى پاناھ تارتىپتىمەن، ياشلىق ...
ياشلىق مۇشۇنداق ئازغۇنلۇقلارغا تولۇپتۇ، ئاللاھ مىنى كەچۈر   ،   تۇيۇقسىز ئىچىمدىن غەۋغالار كۈتۈرۈلدى    ئىچىمدىكى ئىككى خىل روھنى سۇقاشقا سالدىم

___ماڭەۋا ساراڭ،  ئۇقىزنىڭ چوڭ بولسا قىزلىق پەردىسى يىرتىلار، ياسىنىڭ قىزلىق پەردەڭ بولمىسا؟ ئەخمەق
___ياق ياق،    توغرا مىنىڭ قىزلىق پەردەم يوق،  ئەمما ئىتقادىم بار،  ئاللاھىم بار،  ئۇ ماڭا تاپشۇرغان ئامانەت بار،  ئۈزۈمگە سۇئال تاشلىدىم، سىنىڭ داداڭنىڭ ئاناڭدىن بۈلەك ئاياللاربىلەن يوللاردا يۈرگىنىنى كۆرگۈڭ بارمۇ؟

__داداڭنىڭ ئاشنىسى بولغان بولغىنى خالامسەن ياق، داداڭنىڭ كىچىكىدە سەن قىلماقچى بولغان ئىشنى قىلىپ قويغان بولشنى خالامسەن؟
___  ياق ،مەن بىلەن ئۇنىڭ نىمە مۇناسىۋىتى؟؟  
___سەنمۇ كەلگۈسىدە دادا بولىسەن!!!!  بالاڭ بولدۇ، ئۇمۇ سەن داداڭنىڭ ئۇ ئىشنى قىلغان بولىشىنى خالىمىغاندەك بالاڭمۇ سىنىڭ قىلىشىڭنى خالىمايدۇ
قەلبىمدىكى ئىتقادسىز روھ ئەمەللىي مىسالدا يېڭىلدى  مەنمۇ يەڭگىللەپ قالدىم   راستىنى دىسەم ماڭا نىمە بولدى؟؟؟  
چوڭ بوپتىمەن،  بالاغەتكە يىتىپ خۇتۇن ئالسام بولغىدەك بوپتىمەن،  خۇتۇنلا ئالايمۇيە؟؟  ......   ئويلىسام ھازىر بولمىغۇدەك نىمىشقا دىگەندە، مەن پۇل تاپاممايمەن، ئەمىسە نىمىش قىلاي؟   پۇل تاپاي، ئىككى ئادەمنى باققىدە بىر ئىشنىڭ يولىنى قىلاي... يەنە خىياللىرىم قوزغالدى،     ھازىر 3 تىلنى پىششىق 1 تىلنى يېرىمدىن ئارتۇق بىلىمەن، شىنجاڭغا بېرىپ مىنىڭ تىلىمنى تۇلۇق چۈشنىدىغان ئادەملەرنىڭ ئارىسىدا ياشىسام
...... بىرە شىركەتكە تەرجىماللىق قىلساممۇ خېلى پۇل تاپامە .....
نىمىش بۇ، قۇسۇقۇم ئاغراۋاتىدۇ، ۋايجان قىستاپ لا كەتتۇغۇ؟  بايام ياقىم ئويلىنىش بىلەنلە بولۇپ،  بۇنى بەكلا قىسۋىتىپتىمەن،  ئوبىرنى تاپمىسام چاتاق،  تاسمامنى قانداق قىلارمەن؟   تاسمامغا قاراپ باقتىم، كىچىك ماگىنىت تىكلىنىپ قاپتۇ،  ئاپلا دەيمەن ھازىر ئۇنى ئوڭشىغىلى بوممايدۇ،  بايام تاسمام ماڭا پاكلىق ئەكەگەن، ئەمدى .....
