مەركىزىي خەلىق رادىيۇ ئىستانىسى ئۇيغۇرچە ئاڭلىتىشى بۆلومدىكى بۇرچ مەسئولىيەت تۇيغۇسى كۈچلۈك يىڭىلىق يارتىشى روھىغا باي مۇھەررىر ،دىكتۇر رىياسەتچىلەرنىڭ كۆڭۇل قويۇپ پىلانلىشى ،جاپالىق خىزمەت قىلىشى نەتىجىسىدە بىر قاتار رەڭدار ،نادىر رادىيۇ پىرگىراممىلىرى رۇياپقا چىقتى  رادىيۇ ئاڭلىغۇچىلارنىڭ كۆپ قاتلاملىق مەنىۋى ئېھتىياجنى  قاندۇرپ كەلمەكتە ،بولۇپمۇ ئەدىبىي ئاڭلتىشى <،پىرۇگراممىسى ئۇيغۇر ئەدەبىياتىنىڭ ھەقىقي مۇنبەت توپرىقىغا ھەم خوش پۇراقلىق گۈلىستانىغا ئايلىنىپ رادىيۇ ئاڭلىغۇچىلارنىڭ دىققەت -ئېتىبارى ھۆرمەت قايىللىغىنى ۋە تەبرىك -ئالقىشىغا سازاۋەر بولماقتا. بۇ<< ئەدەبىي ئاڭلىتىش>> پىرۇگراممسىدا ئاڭلىتىلغان بىر بىرىدىن ئېسىل رومانلارنى ئاڭلىغىىنىمىزدا مەركىزىي خەلىق رادىيۇ ئىستانسى ئۇيغۇرچە ئاڭلىتىشى بولومىنىڭ ئۇيغۇر ئەدبىياتىنى گوللەندۇرۇش يولىدا ئىتايىن مول ۋەئەھمىيەتلىك ئەجىر سىڭدۇرۋاتقانغىلىغنى قايىللىق ۋە ئىپتىخارلىق ئىلكىدە چوڭقۇر ھىس قىلمىز .
يىقىندا مەركىزي خەلىق رادىيۇ ئىستانسى ئۇيغۇرچە ئاڭلىتىشى بولومىنىڭ مۇھەررى ،دىكتۇر -رىياسەتچىسى ئابدۇرۇسۇل ئابدىرىھىمنىڭ غۇلجا شەھەرلىك 2-ئوتتۇرا مەكتەپتە قۇرۇلغان <<.زۇردۇن سابىر خاتىرە ئۆيى>>گە ئالاھىدە سۇۋغا سۈپىتىدە ئەۋەتكەن يازغۇچى زۇردۇن سابىرنىڭ <<ئانا يۇرت >>ناملىق رۇماننىڭ رادىيۇدا ئاڭلتىلىغان ئاۋازلىق نۇسخىسىنى تاپشۇرۇپ ئېلىپ ناھايىتى خوش بولدۇم .گەرچە مەن يازغۇچى زۇردۇن سابىرنىڭ 2000-يىلى 11-ئايدا نەشىر قىلىنغان <<ئانا يۇرت >>،روماننى توت قىتىم ئوقۇغان ،2010-يىلى 12-ئايدا مەركىي خەلىق رادىيۇ ئىستانسى ئۇيغۇرچە ئاڭلىتىشى  بولومى .ئەدەبىي ئاڭلىتىشى <،پىرگىراممىسىدا <<ئانا يۇرت >>روماننى ئاڭلىغان بولساممۇ ،بۇ قىتىم يەنە ئاڭلىدىم ،<<ئانا يۇرت>>روماننى ھەر قىتىم ئوقۇغىنىمدا ياكى ئاڭلىغىنىمدا بۇ رۇماننىڭ يۈكسەك تىل ماھارىتى ۋە ئۆزىگە خاس تىل ئوسلوبىدىن ئۆزگىچە ئېسىتېتىك زوق ئېلىپ چوڭقۇر ھېسلارغا چۆكىمەن .بۇنى ئەلۋەتتە ئانا يۇرت رۇماننى  ئەدەبىي ئاڭلىتىشىقا تەييارلىغۇچ پىرۇگىرامما نازارەتچىسى ،مەسئول مۇھەررىر ،تېخنىك ،دىكتۇر رىياسەتچىلەرنىڭ ئىجادكارلىق روھىدىن ئايرىپ قارىغىلى بولمايدۇ .
مەن يازغۇچى زوردۇن سابىرنىڭ <<ئانا يۇرت >>،ناملىق روماننىڭ ئەدبىي ئاڭلىتىشى جەريانىغا مۇناسىۋەتلىك ئۇچۇرلارنى ئوقۇرمەنلەرگە ۋە ئىخلاسمەن ئاڭلىغۇچىلارغا بىلدۇرۇش مەقىستىدە 2013-يىلى 9-ئاينىڭ 7-كۇنى شىنجاڭ سەنئەت ئىنىستۇتى رادىيۇ -تېلېۋزىيە دىكتۇر -رىياسەتچىلىك كەسپىنىڭ ئوقۇتقۇچىسى ،سىنارسىت،مەركىزىي خەلىق رادىيۇ ئىستانسى ئۇيغۇرچە ئاڭلىتىشى بۆلومى <<ئەدەبىي ئاڭلىتىشى >>پىرگىرامىسىنىڭ ھەمكارلاشقۇچىسى ،تىخنىك پولات تۇرسۇن مۇئەللىم بىلەن بۇ ھەقتە سوھبەتلەشتىم ،پولات تۇرسۇن مۇئەللىم يازغۇچى زۇردۇن سابىرنىڭ <<ئانا يۇرت>>ناملىق رۇماننىڭ ئاۋازلىق نۇسخىسىنىڭ تەيارلىنشى جەرياننى ئەسلەپ مۇنداق دىدى :
-2010-يىلى 10-ئايدا مەركىزي خەلىق رادىيۇ ئىستانىسى ئۇيغۇرچە ئاڭلىتىشى بولىمدىكى مۇناسىۋەتلىك خادىملار <<ئانا يۇرت>>،ناملىق ئۈچ توملوق  روماننى <<.ئەدەبىي ئاڭلىتىشى >>پىرۇگىراممسسىدا ئاڭلىتىشىنى پىلانلاپ .ئاۋازلىق نۇسىخسىنى تەيارلاش ئىشىىنى ماڭا تاپشۇردى ،مەن ئىلگىرى بۇ روماننى ياقتۇرۇپ ئوقۇغان ،بۇ رومان بۇگۇنكى زامان ئۇيغۇر ئەدبىياتىدا كاتتا شوھرەت قازانغان سالمىقى ئېغىر ،باشقا رومانلاردىن رۇشەن پەرىقلىنىدىغان ئېسىل ئەسەر ،مەن بۇ روماننى ئاۋازلىق قىلىپ ئەدبىي ئاڭلتىىشقا تەيارلاش ئچۇن كۆپ باش قاتۇردۇم ،ئىشىنى ئالدى بىلەن <<ئانا يۇرت>>،روماننى قايتا ئوقۇشتىن باشلىدىم روماننى ئوقۇغانسىرى كاللامغا كىرۋالغان <ئوقۇغۇچىلىرىم بۇ رۇماننىڭ ئاۋازلىق ئوقۇش ھۆددىسىدىن چىقالارمۇ؟.>دەيدىغان ئەندىشىلىك ئوي كۆچىيىپ كەتتى ،ئاخرىدا بىر قاتار ئويلىنشى ،ئىزدىنشىلەردىن كىيىن <<كۆز قورقاق ،قول باتور دىگەندەك  << ئانا يۇرت>>روماننى ئاۋازلىق ئوقۇش خىزمىتىگە رەسمىي كىرىشىپ كەتتىم .فاكولىتېتمىزدىكى سالىھجان رۇزى،دىلنىگار ھەسەن قاتارلىق ئوقۇغۇچىلىرىمنى چاقىرىپ دەسلەپ سالىھجانغا روماننىڭ بايان قىسىمىنى ئوقۇتۇپ ئۇنگە ئالدىم ،بۇنى قايتا -قايتا ئاڭلاپ بايان بىلەن دىيالوگنىڭ ئاساسەن ئوخشاپ قالغاندەك ئاجىزلىقىنى ھېس قىلىدىم ،ئوقۇغۇچۇم دىلنىگارغا بايان قىسىمىنى ئوقۇتۇپ ،سالىھجانغا بايان قىسىمىنى ئوقۇتۇپ ئۇنگە ئالدىم ،بۇنى قايتا-قايتا ئاڭلاپ دىيالوگ قىسىمىنىڭ ئاۋاز تۈسىنىڭ سەل ئاجىز بولوپ قالغانلىغىنى ھىس قىلدىم ،ئاخر دىيالوگ قىسىمىغا يەنە ئوقۇغۇچۇم ئەرافات پولاتنى قاتناشتۇردۇم .ئوقۇغۇچىلارنىڭ بۇ ئەسەرنى ئوقۇغاندىكى  سادىر قىلغان سەۋەنلىك خاتالىقلارنى ئەستايىدىل تۇزەتتىم.نەتىجىدە ئوقۇغۇچىلارنىڭ ئاۋاز روشەن ئۆزگىرىشى پەيدا بولدى ،دەسلەپ بۇ رۇماننى 38مىنۇت ئوقۇتتۇم،ئونۇمى ناھيىتى ياخشى بولدى ،بۇنىڭغا ئىككى مىنۇتلوق باشلاش مۇزىكىسى ۋە ئاخرلاشتۇرۇش مۇزكىسىنى كىرىشتۇرگەندە نەق 40مىنۇتلوق ئاۋازلىق نۇسخا تەيارلاندى ،ئادەتتە باشقا رومانلارنى كۇنىگە سەككىز سائەتتە ئۈچ ئاڭلىتىشى يەنى 120مىنۇتلوق ئاۋازلىق نۇسخا تەيارلاتتۇق <ئانا يورت>>روماننى ئوقۇغاندا كۇنىگە سەككىز سائەتتە بىر ئاڭلىتىشى يەنى 40مىنۇتلوق ئاۋازلىق نۇسخا تەيارلىدۇق .چۈنكى بۇ رومان ئادەتتىكە رومان ئەمەس ،شۇڭا بىر بۇ روماننى تېز ئوقۇشنى ئەمەس،بەلكى ئالدىماي سۇپەتلىك ،لاياقەتلىك ئوقۇشقا تىرىشىتۇق بۇ <ئانا يورت>>روماننى ئوقۇپ ئاۋازلىق نۇسخا تەيارلاشقا بىر يېرىم ئاي ۋاقىت كەتتى ،مۇزىكا كىرىشتۇرگۇچى مۇئەللىم بىر ئاۋازلىق نۇسخىغا ئىككى ھەپتە ئەتراپىدا مۇزىكا كىرىشتۇردى ،شۇنىڭ بىلەن بۇ روماننىڭ ئاۋازلىق نۇسىىسى ئىككى ئاي ئىچىدە تەيارلاندى ،كىيىن بۇ رومان <<ئەدىبىي ئاڭلىتىشى >>پىرگىراممىسىدا ئاڭلىتىلغاندىن كىيىن جەمىيەتتە ناھايىتى ياخشى باھا ،ئىنكاسلار بولدى ،بىزھازىرغىچە 32روماننىڭ ئاۋازلىق نۇسخسىنى تەيارلاپ ئەدەبىي ئاڭلىتىشتا ئاڭلاتتۇق ،بىز ھازىر <<ئەنجانباي >><كوكتىن چۇشكەن پەرىشـتە><كىرۇراندىكى ئىزچىلار>>......قاتارلىق سەككىز روماننىڭ ئاۋازلىق نۇسىخسىنى جىددىي ئىشلەۋاتىمىز .
