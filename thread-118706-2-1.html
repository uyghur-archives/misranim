بۇلتۇر تازا سۇغۇق كۈنلەر ئىدى.مەن ئىرىم بىلەن پۇسكام ناھىيە نىفتكە ئاشغانىدا ئېشلەيمىز دەپ چىققان ئىدۇق.مەن ئۇنىڭغا مۇشۇ قاتارلىق ئېتكى كەڭىرى قەشقەردە ئېىش چىقىمىغانىدەك ياقا شەھەرگە چىقامدۇق،دەپ ئۆزەمچە ئۇنى ئەيىپلەپ ئۇرشۇپ قالدىم.ئادەم بۇلۇپمۇ ئايال كىشى ئەر كىشگەبويسۇنماي ئامال يوق ئىكەن.شۇنداق قىلىپ يولغا چىقدۇق.ئۇيەرگە چىققاندىن كىيىن نىفتنىڭ شۇنچىلىك ئازادا ئىكەنلىكى ئۆستەڭدىكى سوپسۈك ئىقىن سۇلىرى ئادەمنىڭ زۇقىنى كەلدۇرەتتى.بىز ئايلىنىپ كىلىپ ھىلقى ئېشلەيدىغان ئاشغانىغا كىرىپ قالدۇق،ئاشغانا خوجايىنى بىزنى تولىمۇ قارشى ئېلىپ كۈتىۋالدى،بىركۈن ئۆي ئىزدەپ ئورۇنلىشۋىلىپ ئەتىسى ئىشقا چىقىدۇق.بۇ دۇكاندا تاڭجاڭلىق قىلدىغان،ئاي پەرى ئېسملىك بىر قىز بار ئېكەن،لېكىن بۇ قىز ئۆزى چاققان لېكىن ئېستل جەھەتتە زادىلا مەسلە بار دەك قىلاتتى.ئۇنىڭ شالاقلىقى تۇرقىدىن چىقىپلا تۇراتتى.بىز بىلەن ئۇ خنىم ھەپتە ئون كۈن چىرايلىق ئۆتتى.لېكىن نىمىشكىكىن تاڭ، ئۇماڭا ئۆزىچلا سېڭايان قارايدىغان بولۇپ قاپدۇ.ئۇنڭغا خىلى دىققەت قىلىپ قاراپ باقسام ماڭا تۇتقان پوزسيەسى سۇغۇقلىشۋاتقاندەك تۇراتتى.ئىرىمغا ئۇخنىم ھە دىسە سۇينى چىقىرىپ بىر-بىرىگە بىكار تۇرماي كۆز قىسدىغان بىقىنلىرىنى چىمدايدىغان بولۇپ قالدى.ئۇ ئىرىمغا ھەي، توختاجىم، ماڭا تېلفۇنىڭىزنى بىرىپ تۇرىڭە مەن بىردەم ناخشا تىڭشاي دەيتى.ئىرىم،توختاجىم،ئىغىزىدىن چىقىپ بولغىچە ئۇنىڭغا قىزغىنلىق بىلەن بىرەتتى.تىلفۇن ئۇنىڭ قۇلىدىن چۈشمەيتى.ئىرىمنىڭ،مەن ئايالى بولغان ئادەم سورىسام ئىشڭنى قىل ياق دەپ ئۇنىمايتى.ئۆز-ئۆزەمچە ئادەم يوق يەرگە كىلىپ يىغلاپ-يغلاپ دەردىمنى ئالاتتىم،ھەتتە چۈشلۇق تاماقدىمۇ ئۇ ئالاھىدە ھىلىقى خىنىمغا بىر قوللۇق تاماق ئىتىپ بىرىىپ ئۆزى ئالدىغا ئەكىلىپ بىر چەينەك چاينى تەملەپ ئۇنىڭ كۆزىنىڭ ئىچكىرلىرىگە ناھايتى سھىرلىك قارايتى.منىڭ بۇچاغدا ئاچىغىم كېلىپ ئۆزەمنى قويدىغانغا يەر تاپالماي قالاتتم،قىنى قانداق قىلىدۇ دەپ ئۇنىڭغا،ھەي قەدىرلىكىم ئىرىم، توختاجىم،ماڭىمۇ تاماق قىلىپ بىرىىڭە دەپ ئايىرىم ئۆيگە چاقىرىپ دەيتم،مەن ئۇنى شۇنچە ئاۋايلىغان بولساممۇ لېكىن ئۇمىنى قىلچە ئاۋايلىماي ئادەم بار يەردىلا قولۇڭ بارمۇ،تاماق دىېگەن ئۇستامنىڭ بىرىپ دىىمەمسەن!دەپ پەشتامباللىرىنى سلكىپ قىقىپ قۇيۇپ ئۆرۇلۇپ كىتەتتى.بۇ چاغدا بولسا ھىلىقى خىنىم ئۆزىچىلا تىخىتنمۇ بەك يىقنچىلىق قىلاتتى.مەن ناھايتى ئوڭايسزلىنپ ئۇنۈم ئىچىمگە چۇشۈپ كىتەتتى،بىز ئىشدىن چۈشەيلى دەپ تۇرغاندا ھىلقى خىنممۇ ئۆزىنى ئۆزىچىلا سۇۋايتى.مەن يەنىلا كۆرمەسگە سىلىپ توغرا چۇشەنگەن بولۇپ،ھە،دەپ قۇيۇپلا بىللە ماڭاتتۇق.لېكىن تەردىمنى ئۆزۇم بىلەتتم،ئۆيگە كىرىگەندىن كىيىن ئارام ئالدىغان ۋاقىت بولۇپ قالدى،گىچە سائەت بىر بولغان ئەتراپدا ئۆزىچە تېلفۇن كەلدى،ئۇ قاراپ قۇيۇپ بىسۋەتتى،يەنە تېلفۇن كەلدى،ئۇيەنىلا بىسۋەتتى.بۇ تېلفۇن مۇشۇ ئەسنادا يەتتە قىتىم ئۇرۇلدى،لېكىن ئىرىىم بۇ تېلفۇ ئالغىنى ئۇنىمىدى.مەن ئۇنىڭغا خۇددى ھىچنىمىنى بىلمەيدىغان قىياپەتگە كىرۋىلىپ توختاجىم،تېلفۇننى ئىلىڭە بەك ئۇرۇپ كەتتى،زۇرۇل گىپى بار ئوخشايدۇ،دىدىم ،كىم دۇبۇ؟دەپ قويدى سەكىكىزنىچى قىتم دا بولسا تېلفۇننى ئۇرماي سارىتپ قويدى.بۇ چاغدا توختاجىم،خۇتۇن سىرتقا چىقىپ كىرەي،دەپ بانا قىلدى.بۇ بانا بۇنىڭغا بەك كەلدىغۇ ھەمدى،بۇ تەرەت كەلدى دىگەندىن كىيىن ئىشتانغا چىقىرىىڭ دىگىلى بولمايدۇ،دە،ئۇ كىرگىچە مەن چىقاي دىسەم ئاسمان يىراق،كىرىپ كىتەي دىسەم يەر قاتتق بولۇپ كەتتى.ئۇ بىر ھازا ئۆتگەندىن كىين قايتىپ كىردى،ۋاي كىمىكىن دەپ سورىۋىدىم،ئۇنېمە دىېدى دىسەڭلار،منىڭ يىزىدىكى ئاغىنەم ئىكەن،بولدى خۇتۇن ئارام ئالايلى،چۇ،دىىدى.مەن ئۇنىڭ كىپىگە ئېشەنگەن بولۇپ ئارام ئالدۇق.ئەتىسى تاڭ ئاتتى،يەنىلا دۇكانغا قاراپ يول ئالدۇق،ھەر قايسمىز ئۆزىمىزنىڭ ئىشنى تاپدۇق،سائەت ئونلەر بولغان ئەتراپىدا ئۇنىڭغا يەنە تېلفۇن كەلدى،ئۇتېلفۇننى كۆرۇۋىتى،ئۆزىنڭ  كۆرۇشكشى يوق گەدەنگىچە ئېلشى بار بولغان بىر ئاغىنسى پۇلنى سۇيلەپ تېلفۇن قىلىپتۇ،شۇ ئارلىقدا مەن ئۇنىڭغا تېلفۇننى ئەكېلىڭە بىردەم ناخشا ئاڭلاپ تۇراي،قاندىغى تېلفۇن قىلسا ئەدىپنى راسا بىرەي دىدىم،ئۇتەلپىمگە قۇشۇلدى.تېلفۇننى ماڭا بىرىۋىدى ئۆز ئىش ئورنۇمغا ئىلىپ چىكەتتم.ئۇ نىڭغا يەنە تېلفۇن كەلدى،بۇ تېلفۇننى بىر ئالايچۇ دەپ بۇ تىلفۇننى ئالدىم.تۆۋە خۇدايىم،بۇ تېلفۇننى ئۇرغۇچى يەنىلا بىر ئايال كىشى ئىكەن،ئۇتېلفۇننى ئىلىپ تەپ تارتمايلا،توختاجىم،بارمىدۇ؟دەيدۇ،تىخى.مەن چاندۇر مايلا ئۆزلىرى كىم بولدىلا،ئاكام قازان بىشدا ئىدى،دىېدىم.ئۇ سىزنىڭ كەلگۈسى خاناچىڭىز دىدى.ئاكىڭىزنى چاقىرڭە دىىۋىدى،نېمە گەپلىرى بار ئىدى،مەن يەتگۈزۇپ قۇياي دىدىم.ئۇ ياق،بولدى،ئۆزىگىلا دەي،دىېدى.مەن ئاندىن ئۆزەمنى ئاشكارىلاپ،قايلىسا خان سىلى كىمۇ؟ئېسملىرى نېمە،ئۆيلىرى نەدە،كىمنىڭ بالىسى؟سلىگە ئېگە بولدىغان ئادەم يوقمۇ ئۆيلىرىدە،ئېرىم سىلىگە نېمىلەرنى دىىدى،خۇلاسەڭلار نەگە بىرىپ توختدى؟دىىگەندەك سۇئاللارنى سورۇدۇم.ئۇ مەندىن سلە كىمۇ دىۋىدى،مەن ئۇنىڭ ئايالىي بولىمەن،بىز توي قىلغىلى ئۈچ يىل بولدى،ئىككى بالام بار،بۇلارنى سىلىگە دىمېگەنمۇ؟باشقىلارنىڭ ئۆيىنى بۇزۇش شۇنداق شەرەپلىك ئىش ئوخشمامدۇ؟دىىدىم.ئۇ خنىم ئەركىشى دىېگەن ئەجەپ بىر بولدىگەن،ھە،ئۇماڭا مەن تىخى ئۆيلەنمىگەن،منىڭ سلىگە كۆزۇم چۇشۋاتىدۇ،سلى بەئەينى بىر بىرپارچە ئاينىڭ ئۆزىگەنلا دەپ ئۆزىنى مەن دىېگەن ھۈنەرۋەن دەپ داڭلاپ بىرىۋىدى،دىىدى،مەن ئۇنىڭ بىلەن يەنە بىر دەم پاراڭلىشپ زادى ئۇنىڭ مەنبەسى ،مەخستى نېمە؟ ھەممنى سورۇدۇم. ئۇبىر-بىرلەپ جاۋاپ بەردى، منىڭ ئىسمىم:ئاينۇر،ئۆيۈم يىزىدا، ئۆيۈمدە،مەندىن باشقا ئانام ۋە بىر ئاكام بار، توختاجىم،ماڭا توي قىلش تەكلىۋىنى قويغان ئىدى، مەن ئۇنىڭغا قۇشۇلغان دىىدى، مەن ئۇنىڭغا ئىلمي يۇسۇندا قانداق قىلىمىز ئەمسە؟ ئىككى بالىنى چۇ؟دىىدىم.ئۇ بولدى،منىڭمۇ ئۆيۈمگە مەھەلىدىن ئەلچى كىرىپدىگەن،دەپ دىىدىيۈ،خىجىل بولغاندەك تۇراتتى.ئۇنىڭ خىجالەتچىلىكتەدەۋاتقان گەپلىرى خۇددى ماڭا ساختاپەزلىكتەك بىلنپ كەتتى. شۇئارلىقىدا ئېرىم،نېمە بولۇپ چىرىپ قالدى،تاڭ ھە،دەپ سۆزلىشۋاتقانلىقىمنى كۆرۇپ دەرھال يۇگۈرۇپ كەلدى.تېلقۇن ئېكرانىغا قاراپ تەستىكەمگە چاڭڭدە بىرنى ئۇرۇۋىدى،بەكمۇ ئۆكسۈپ يىغلاپ كەتتىم.ئۇرشاي دىسەم ئۇدېگەن ئامېۋى سورۇن بەك سەت تۇرغۇدەك،بوبدۇ دەپ ئۆتكۈزۋەتتدىم.لېكىن ئۇ تېلفۇن يەنە ئالتەقىتم كەلدى،ھە، دىسلا ئۇ سىرتقا يۇگۈرەيدۇ.ياكى تىنچ بىر يەرنى تىپپ بەھۇزۇر ئولتۇرۇپ ،ھە،خان قانداق مىجەزىلىرى؟ساق سالامەتمۇ، يا،بىز دىگەن داۋاملىق شاش،ھەرقاچان ۋە دىىمىز،ۋەدە،بولىدۇ، دىىدى. لېكىن قارشى تەرەپ ، مەن بىلەن پاراڭلاشقان گەپلەرنى دىگەنمۇ، قانداق،تېلفۇندا خۇشلۈشۇپ بولۇپلا، منى ئامىۋى سورۇن ئىچىگە تارتىپ بەك ئۇرۇپ كەتتى. ئادەم بار يوق دىمدى ئۇرۇپمۇ، ۋەھشى ئۇردى.سەن بۇگۈنلا كەت جىمۇ، دەپ تەھىتىد سالغىلى تۇردى.نىمشقىقىن دىسەم،ئەسلىدە ھىلىقى ئاشنسى، ئاينۇر،يىزىدىن پوسكامغا يولغا چىقىپتىكەن.ھەي خۇدايىم،ئاشۇ كۆڭلۈمنىڭ ساددىلىقى،ئۆزۇمنىڭ كۆيۈمچانلىقىم،ساداقەتمەنلىقىم،ئۆزۇمنىڭ بىشغا چىقتى.شۇنداق قىلىپ قەستەن، ئىرىمنىڭ چىشغا تەكدىم.تىللاپ ھەيدەشلىرىم بىلەن ئەتىلا قەشقەرگە قاراپ يولغا چىقماقچى بولدۇق،ئەتىسى ئەتىگەندە بىكەتكە بىرىپ قەشقەرنىڭ بىلىتدىن ئىككى كىشلىك ئىلىپ ماشنىغا چىققان ئىدۇق،تۇيۇقسىز ئۇنىڭ تېلفۇنى سايراپ كەتتى.ئۇ تېلفۇنىنى ئالدى،دە،بىرسائەتدەك سۆزلەشدى،ئۇنىڭ سۆزىدىن،ھە ماقۇل،خاتىرجەم بۇلۇڭە،گىپىم-گەپ،منىڭ ئانام مىنى ئوغۇل
بالادەپ تۇغقان،سىز ئاڭلىمىغان بولشڭىز مۇمكىن،ئوغۇل باللارنىڭ ۋەدىسى ۋەدە،قىز باللارنىڭ ۋەدىسى يەردە،دەپ ھىجايتى،مەن بۇ تىېلفۇننى چوقۇم ئاشۇ پاسكىنىدىن كەلگەنمىكىن دەپ گۇمانلاندىم.ئاخىرى گۇمانىم توغراچىقتى.ئۇ ماڭا قاراپ تۇرۇپ خۇتۇن مىنى راستنلا چۈشەن مىنڭ ئازراق ئېشىم چىقىپ قالدى ،سەن كىتىۋەركىن مەن شۇ ئىشمنى تۇگىتىپ بۇلۇپلا كەينىڭدىن بارىمەن دىدى .مەن ياق بولمايدۇ،مىنمۇ ئويلاپ قويشڭىز كىرەك ،دە،سىز قىلىپ قالسىڭىز،مەنمۇ سىزبىلەن بىرگە قالىمەن دىىدىم ،ئۇ،توختماي سەندىن خۇشبۇلاي دىگەندەك گەپلەرنى قىلاتتى .مەن كۆزۇمگە لىققىنە ياش ئالغان ھالدا ماقۇل دىىدىم ، بىزچىققان ماشىنا ئەمدى قۇزغىلاي دەپ تۇرشغا،شۇپۇر ئۇستام توختاپ تۇرسلا مەن چۈشۋالاي دىىگىنچە ماشىندىن چۈشۈپ قالدى.ماشنا يولغاچىقتى ،مەن خۇددى مۇھەببتمدىن ئايرىلىپ قالغاندەك ئارقامغا قاراپ ئۆكسۈپ-ئۆكسۈپ يىغلايتىم.ماشنىنىڭ ئىچىدىكى ئادەملەر مىنى مازاق قىلىۋاتامدۇ،ھەي تاڭ،كىچىك بالىلارمۇھەتتا چۇڭلارمۇ،ھەريان بۇلۇپ قارىشاتتى.مەن شۇئانلا يىشىمنى ئەيتىۋىتىپ ھىچنىمە بولمىغاندەك ئالدىمغا قاراپ ئولدۇردۇم
