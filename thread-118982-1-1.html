    يېقىندىن بۇيان ئۈندىدار ۋە باشقا تورۋاستىلىرىدە تۈركىيە سەنئەتكارلىرىنىڭ ئۈرۈمچىدە ئويۇ ن قويۇش مەسلىسىدە بىر بىرىنى ئەيىلەش خاراكتېرىنى ئالغان تېمىلار ئەۋىج ئالغىلى تۇردى . بۇ تېمىلارنىڭ ئوتتۇرىغا چىقىشى بىزدە شۇنداق ھەرخىل ئىدىيە ۋە ھەرخىل كۆزقاراشلارنىڭ مەۋجۇتلىقىنى چۈشەندۈرىدۇ ھەم بۇنىڭ ھېچقانداق ھەيران قالارلىق بىرئىشمۇ ئەمەسلىگىنى بىلىشىمىز كېرەك . بىر مىللەت كىشىلىرى قېلىپتا قۇيغان كېسەك بولمىغاندەك ھەرخىل كۆز قاراشلار بولىدۇ بۇ نورمال ئەھۋال . ئەمما مېنى ھەيرانقالدۇرغېنى ئۇ يازمىلاردا باشقىلارغا نەسىھەت قىلىش ، باشقىلارغا دىنى تەربىيەبېرىش ئۈچۈن يازغان كىشىنىڭ ئىنتايىن مەينەت بىر تىل ئۇسلۇبى ئىشلەتكەنلىگى بولدى. يازمىنىڭ بېشىدىلا تۈركىيە ناخشىچىسى ئىسمائىلنى تىللاش بىلەن باشلانغان بولۇپ ئاخىرغىچە تۈرك سەنئەتكارلىرىنى تىللاش قاغاش ھەم قاخشاش بىلەن تولغان . ئۇيازمىدا ئېيتىلىشىچە تۈرك سەنئەتكارلىرى ئۈرۈمچىگە كېلىپ يىگىتلەردىن غورۇرىنى قىزلاردىن نومۇسنى ئېلىپ كېتىپتۇمىش . نىمە دېگەن كۈلكىلىك مەنتىقە بۇ ! غورۇردېگەن شۇنداق ئاسان ئېلىپ كەتكىلى بولىدىغان نەرسە بولسا ئۇ ئادەملەردە ئەسلىدىلاغورۇر يوقلىقىنى چۈشەندۈرۈپ بېرىدۇ . نومۇسمۇ شۇنداق ئەرزان باھالىق نەرسە بولسا ئۇ قىزلاردا نومۇس ئەسلىدىلا يوقلىغىنى چۈشۈندۈرىدۇ . قارىماققا دەبدەبىلىك ئەمما ئىچى قۇرۇق بۇ يازمىلاردا پۈتۈنلەي قارىسىغا ھۆكۈم چىقىرىش ، مۇناسىۋەتسىز ئىشلارنى قەستەن بىربىرىگە باغلاپ يەكۈن چىقىرىش خاتالىقى ئېغىر بولغان . بۇيازمىلارنىڭ تىل ئۇسلۇبىىدىن  يازغان كىشىنىڭ كۆپ ئوقۇمىغان ، نورمال ئەخلاقى قائىدىلەردىنمۇ خەۋىرى يوقلىغى چىقىپ تۇرىدۇ . چۈنكى يازما ئىگىسى ئۆزى ئىستانبۇلدا ياشاپ تۇرۇپ باشقىلارنىڭ كىشىلىك ھوقۇقىغا ھۇجۇم قىلىدىغان گەپلەرنى ئىشلىتىشنىڭ قانۇنغا خىلاپلىغىنى بىلمەمدۇ ؟ياكى بىلسىمۇ ئۇيغۇرلار ئارىسىدا تارقاتسا ھېچقانداق چەكلىمىگە ئۇچرىمايدۇ دەپ قارامدۇ ؟ شۇ يازمىلارنى ئۆز نامىدا تۈركىيەدىكى ھەرقانداق بىر تاراتقۇدا ئېلان قىلسا بۇنىڭ ئاقىۋىتىنىڭ قانداقلىغىنى بىلىسە كېرەك دەپ ئويلايمەن . چۈنكى باشقىلارنىڭ كىشىلىك ئىززىتىگە زىت سۆزلەرنى ئىشلىتىش ، باشقىلارنىڭ شەرىپىگەقارشى يازمىلارنى يوللاش ھەرقانداق قانۇندا جۈملىدىن تۈركىيە قانۇنىدا قاتتىق چەكلەنگەنلىكى ھەرقانداق بىر ئەقىل ئىگىسىگە مەلۇم . بۇ گەپلەرنى تۈركىيەدىكى ھەرقانداق ئاشكارا سورۇندا ئېيتقان بولسا قانۇنى جازاغا ئۇچرىشى ئېنىق ئىدى . شۇڭابۇ نۇقتىدا ئۇ يازمىنى يازغان ھەم تارقاتقان كىشىلىرىمىز ئۆزىنى دەڭسەپ بېقىشى كېرەك ئىدى .

     ئۇ يازمىلاردا تۈركىيە ناخشىچىلىرىنىڭ ئېرقىنى د ئېن ئەي سىنى تەكشۈرۈپ باققاندەك ئۇلارنى تۈرك ئەمەس يەھۇى دېگەندەك بەتناملارنى چاپلاپ ئۈلگۈرۈپتۇ . يەنە تېخى تۈركىيەدىكى خەلقنى نەچچە تۈرگە بۆلۈپ ئەردوغاننى قوللايدىغانلارنى مۇسۇلمان باشقىلارنى كاپىرغا چىقىرىپتۇ ... بۇ گەپلەرنى بەلكىم باش مېنىستىر تايىپ ئەردوغان ئاڭلىسىمۇ شۇ گەپلەرنى قىلغان نادانلارنىڭ يۈزىگەتۈكۈرگەن بولاتتى . چۈنكى تۈركىيە كۆپ پارتىيەلىك تۈزۈمدىكى دېموكراتىك سايلام ئارقىلق ھۆكۈمەت ۋۇجۇتقا كېلىدىغان دۆلەت . ئۇ يەردىكى كىشىلەرنىڭ ئەردوغان ھۆكىمىتىنى قوللاش قوللىماسلىقى ئۇلارنىڭ تۈرك كىملىگى ۋە دىنى ئېتىقادى بىلەن قىلچە مۇناسىۋەرسىز بولۇپ بۇ پەقەت ئۇلارنىڭ سىياسى ئىدىيەسىدىن ئىبارەت . مەيلى جۇمھۇرىيەت خەلق پارتىيەسىنى قوللايدىغان سوتسىيال دېموكراتلار بولسۇن ياكى ئەردوغاننى قوللايدىغان  مۇھاپىزىكاردېموكراتلار بولسۇن ھەممىسى ئوخشاشلا تۈركىيە دۆلىتىنىڭ پۇقرالىرى ھېساپلىنىدۇ . بەلكىم ئۇ يازمىنى يازغان كىشىلەر بېسىم ئاستىدا بىرسىنى قوللاپ ئۆگىنىپ قالغان بولغاچقا          تۈركىيەنىڭ دېموكراتىك سايلام سىياسىتىنى قوبۇل قىلالمىغان بولىشى مۇمكىن...

    بىلىشىمىز كېرەككى تۈركىيەجۇمھۇريىتىدە ياشاۋاتقان ھەم شۇ دۆلەتگە باج تۆلەۋاتقان ھەرقانداق ئادەم تۈركىيەدەتۈرك پۇقراسى دەپ ئاتىلىدۇ . بەلكىم شۇ ئىستانبۇدا ياشاۋاتقان بىچارە ئۇيغۇرلارمۇشۇيەرنىڭ نوپۇسىنى ئېلىش ئۈچۈن شۇ يەردىكى مەمۇرى تارماقلارنىڭ مىللىتىنىسۈرۈشتۈرمەي پالاقشىپ يۈرۈپ مىڭ تەسلىكتە شۇيەردە تۇرغۇدەك سالاھىيەتگە ئېرىشكەنلىكىنىئۇنتۇپ قالغان بولسا كېرەك . ئەگەر شۇيازمىدا ئېيتىلغان سۆزلەر بىلەن تۈركلەرئۇلارنى  ھاقارەت قىلسا ئۇلارنى مىللىتىڭتۈرك ئەمەس دەپ پەس كۆرسە ئۇلار قانداق ھېسياتتا بولغان بولاتتى ؟! مەن ئىسمائىليېكنىڭ مىللىتى كۇرتمۇ ، تۈركمۇ بۇنىڭغا قىزىقمايمەن ئەمما كىشىلەرنىڭ تۈرك ياكىئەمەسلىگىنى شۇ كىشىگە باھا بېرىشنىڭ ئۆلچىمى قىلىشنىڭ نەقەدەر نادانلىق ئىكەنلىگىگە ئىشىنىمەن .

    ئەسلى مەسىلە سەنئەت ۋە ئۇنىڭغاقانداق مۇئامىلە قىلىش ئۈستىدە بولسا ھەرگىزمۇ باشقىلارنىڭ كىشىلىكىگە ۋە ئۇنىڭمىللىتى ئېرقىغا ھۇجۇم قىلىشنىڭ ھاجىتى يوق ئىدى . ئۈرۈمچىگە تۈرك ناخشىچىلا كېلىپئويۇن قويامدۇ ياكى باشقا مىللەتلەر كېلامدۇ كۆرىدىغان ئادەملەر چاغلىقلا !ھەرگىزمۇ پۈتۈن ئۇيغۇر ئىشىنى تاشلاپ چەتتىن كەلگەن ناخشىچىلارنىڭ ئارقىسىدىنكەتكىنى يوق ! ئەگەر سەنئەتنى ياخشى كۆرىدىغان ۋە تۈرك ناخشىلىرىنى ياختۇردىغانكىشىلەر تۈرك سەنەتچىلەرنىڭ كېچىلىگىگە بارسا بۇنى توسۇشقا ھېچكىمنىڭ ھەققى يوق . خۇددى كېچىلىككە بېرىشنىخالىمايدىغانلارنىڭ ۋە سەنئەتنى ياخشى كۆرمەيدىغانلارنىڭ بارماسلىق ھوقۇقىبولغاندەك ياخشى كۆرىدىغانلارنىڭمۇ تاللاش ھوقۇقى بولىدۇ . بۇنى ھەرقانداق بىرنورمال ئەقلى ئىقتىدارغا ئىگە ئادەم بىلەلەيدۇ . بۇ 21-ئەسىردە ياشاۋاتقانئادەملەر بىلىشكە تېگىشلىك ئاددى ئۇقۇم .  لېكىن باشقىلارنىڭ شەخسىيىتىگە ھۇجۇم قىلىش ۋەتۈركىيە سەنئەتكارلىرىنى ھاقاراتلەش بىلەن ئۇيغۇرنى سىگانلارغا ئوخشىتىپ ئاغزىغاكەلگەننى جۆيلىۋاتقان شۇ نادان كىشىلەر ئويلاپ باقمامدۇكى ئۇلار باشقىلارنىڭ ئېتىقادىغا، قىزىقىشىغا ئارلىشىش ۋە باشقىلارغا بېسىم قىلىشنىڭ زىيىنى تارتىپ باقمىدىمۇ ؟ بىزمۇسۇلمانلارنىڭ پەيغەمبىرى  دىنى تەربىيەقىلىشنىڭ سىلىق ھەم ئەدەپ قائىدە ئىچىدە بولۇش ئىكەنلىگىنى ھەدىسلەر ۋە ئۆز ھاياتىئارقىلىق كۆرسىتىپ بەرگەن ئىدىغۇ ! بۈگۈنكى دەۋىردىكى دىنى ئالىملارمۇمۇنازىرىلىشى ۋە باشقىلارنى دىنغا دەۋەت قىلىشنىڭ باشقىلارنى شەخسىيىتىگە ھۇجۇمقىلماسلىق شەرتى ئاستىدا بولىدىغانلىقىنى تەكىتلەيدۇ . ئۇنداقتا بۇ يازمىلارنىڭئاپتورى دىننى ئاغزىغا ئېلىۋالغان تۇرۇپ يەنە نىمە دەپ شۇ دىن چەكلىگەن مەينەتئىبارىلەرنى ئىشلىتىپ باشقىلارنىڭ شەخسىيىتىگە ھۇجۇم قىلىدۇ ! بۈگۈنكى كۈندە سىزبىلەن بىزنىڭ  باشقىلارنىڭ قىزىقىشىغا ياكىسەنئىتىگە ئارلاشقۇدەك ئۇلارنى كاپىر ياكى يەھۇدىغا چىقىرىۋەتكۈدەك قانچلىك بىرسالاھىيىتىمىز ۋە ھوقۇقىمىز بار ؟! باشقىلارنىڭ سەنئەت كېچىلىگى ئۆتكۈزۈش ، نېمىنىياخشى كۆرۈش كۆرمەسلىك  ئەركىنلىگگە ھۆرمەتقىلىشنىڭ بۈگۈنكى دۇنيادىكى ئەقلى بىر تەلەپ ئىكەنلىگىنى ئۇلار بىلمەمدۇ ؟! ئۇنداقمەينەت يازمىلارنى ‹‹ ئۇيغۇر بولسىڭىز تارقىتىڭ ! مۇسۇلمان بولسىڭىز تارقىتىڭ ! ››دېگەندەك تەھدىت ئارلاش چاقىرىقلار بىلەن تارقىتىپ نېمىگە ئېرىشمەكچى ، كىمنىتەسىرلەندۈرمەكچى ياكى كىمنى كىمگە سوقۇشقا سالماقچى !!!

   شۇ مەينەت جۈملىلەر بىلەن تولغانيازمىلارنى تارقىتىپ يۈرىۋاتقان نادان ياشلار ئويلاپ باقمامدۇ ؟! شۇ يازمىدا نەقىل قىلىپ يېزىلغان تۈركىيە باشمېنىستىرى ئەردوغان شۇ دۆلەتتىكى ھەرخىل كۆز قاراش ۋە ھەرخىل دىندىكى كىشىلەرنىڭھەممىسىنى تۈركىيە خەلقى دەپ قارايدۇ . ئۇلارنىڭ ھەممىسىنى قۇچاقلايدۇ ! قايسىپارتىيەنى قوللاش قوللىماسلىق تۈركىيە خەلقىنىڭ دېموكراتىك ھەق ھوقۇقى بۇنىڭغاھېچكىم ئارلىشالمايدۇ . شۇڭا ئەردوغان يالغۇز قوللىغۇچىلىىنىڭلا ئەمەس باشقاپارتىيەگە ئاۋاز بەرگەنلەرنىڭمۇ ئوخشاشلا باش مېنىستىرى ئىكەنلىگىنى تەكىتلەيدۇ .  ئۇلارمۇ سىياسى كۆز قارىشى ئوخشاش بولسۇنبولمىسۇن تۈركىيە جۇمھۇرىيىتىنى ۋەتىنىم دەپ قارايدۇ ئۆزلىرىنىڭ تۈرك كىملىگىبىلەن پەخىرلىنىدۇ . بۇيەردىكى مەسىلە قانداقتۇر ئۇلارنىڭ دىنى ئېتىقادى ياكىمۇسۇلمانلىق دەرىجىسى ياكى ئېرقى بىلەن قىلچە مۇناسىۋەتسىز .
