ئەسسالامۇ ئەلەيكۇم تورداشلار ، مەنكى توردىشىڭلار گۈزەل يۈرتۈمدىن سىزلەرگە ئوتلۇق سالام...

  ھەممىمىز بىلدۇق،  شىنجاڭ خەۋەرلىرىدىن، نوپۇزلۇق دولىتىمىزدىكى داڭلىق توربەتلەردىن، شىنجاڭ ئۇيغۇر ئاپتۇنۇم رايۇنىمىزنىڭ خوتەن  شەھرى كىرىيە ناھىيىسى قاتارلىق يىزا قىشلاقلىرىدا ، ئىغىر دەرىجىدە يەرتەۋرەش ئاپىتى يۈزبەرگەنلىكىنى بىلىپ ، كوڭلىمىز شۇنداق يىرىم بولغاندۇ... ئەلۋەتتە ھەممىمىزنىڭ كوڭلى يىرىم بولدى، ھىسداشلىق قىلدى.]دەپ ئويلايمەن.
بىر كىشى قىيىنچىلققا يولۇققاندا،   ھەممىمىز تەڭ ياردەمدە بولۇپ ، قىينچىلقىنى يىڭىشىغا ياردەمدە بولۇش بىزنىڭمۇ ئىسىل ئەخلاقىمىزنىڭ ناماياندىسى  دەپ بىلىمىزغۇ دەيمەن.

مۇشۇ يەرتەۋرەش خەۋىرىنى ، بىز تورداشلار تولىمۇ كىچىكىپ ئۇقۇپتۇق....
شۇ يەرتەۋرەش تىمىدا،  بەزەن ئىنكاسچىلار، << تىما يوللىغاندا،  ئەۋەتىلگەن، كوماندىر جەڭچىلەرنى  سانىنى بەكلا ئاشۇرۋىتىپتۇ دىگەن مۇلاھىزەبولىپتىيۇ، قانداق قىلغاندا ياردەم قولمىزنى سۇنالايمىز دىگەن قارش شەكىللىنەلمەي قاپتۇ...

كەمىنە شۇ<< قانداق قىلغاندا ياردەم قولىمىزنى سۇنالايمىز>> دىگەن مەخسەتتە ئىنكاس يوللىغان ئىدىم...
تورداشلارنىڭ قوللىشىغا، ھەم قوللاپ قۇۋەتلىشىگە مۇيەسسەر بولدۇم...
پۇتۇن خوتەن شەھرى، ھەم كىرىيە ناھىيسىدىكى ھەرمىللەت خەلقى نامىدىن تەشەككۇرۇمنى بىلدۇرىمەن..

ئاتالمىش كورەلمەسلىك، جىدەل ماجرا..... ئۇيغۇر تورچىلقىنىڭ سەركىسى بولغان مىسرانىم مۇنبىرىگە باشقۇرغۇچى تەكلىپ قىلىنىدۇ  دىگەندىن باشلاپ، تاكى تۇركىيلىك سەنئەتكار  شىنجاڭ  ئۇيغۇر ئاپتۇنۇم رايۇنىمىزنىڭ مەركىزى بولغان ئۇرۇمچىدە سەنئەت  كىچىلىكى ئوتكۈزگەنلىكى توغرىسىدىكى خەۋەر بىلەن   جىدەل ماجرا ، سەلپەل بىسىقاي دىگەندە، بۇ كوڭلىمىز يىرىم بولىدىغان يەرتەۋرەش ئاپىتى كوڭلىمىزنى يىرىم قىلدى.. ئەندىشىگە سىلىپ قويدى... شۇنىڭ بىلەن تورداشلىرىمىزنىڭ دىققىتى مۇشۇ يەرتەۋرەش تىمىسىغا يوتكەلدى .. قاندق قىلغاندا ياردەم قوللىمىزنى سۇنالايمىز دىگەن پوزىتسىي، ئىللىقلىق يەتكۇزۇش مىھرى مۇھەببىتىمىزيەتكۇزۇش ئۇچۇن، كوڭلىمىزدىكى  ئۇيغۇر تورچىلىقىمىزنىڭ، سەركىسى  قاتارىدىن بولغان مىسرانىم مۇنبىرىنىڭ بىزگە باشلامچى بولۇپ، بىزگە يول كورسۇتۇپ، ئاش قىيىنچىلقتا قالغان ھەرمىللەت خەلقىمىزگە ياردەم قولىمىزنى سۇنۇشنىڭ باشلامچى بولىشنى لايىق تىپىپ، ئۇشبۇ ئىنكاسنى يازغان ئىدىم...

ئاللاھقا مىڭلاپ شۇكرى!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
  تۇنۇگۇنكى كىچىدە ئۇخلىماي ، ئويلىنىپ يازغان ئىنكاسىمنى ، مىسرانىم باشلىق باشچىلقىدىكى باشقۇرغۇچىلارنىڭ، پىكرىمىنى مۇۋاپىق كورۇپ ، قوللايدىغانلىقىنى بىلدۇرۇپ، قايسى تىمىنى ئاچساق،قايسى تىمىنىڭ قايسى بىتىنى ئاچساق....  مۇبارەك تىلەكلىرىنى ، ھەم بۇ يەرتەۋرەش ئىشىغا ئالاھىدە كۆڭۇل بولگەنلىكىنى بىلىپ.. بەكلا خۇش بولدۇم...

ئالاھىدە ، چىن يۇرۇكۇمدىن رەھمىتىمنى بىلدۇرىمەن...

مۇشۇ تورداشلارغا دەيدىغان  يۇرەك سۆزۇم بارئىدى دەۋالاي... خاپا بولىشماي تىڭشاپ قۇيۇشالا..

مۇشۇ يىقىنياغى،  بەزەنبىر تورداشلارنىڭ ، پىتىنى پاسات تىرىش، قۇتراتقۇلۇق قىلىش، ئىناقلىقىنى بۇزۇش،  مۇنبەردە، ۋەياكى مۇنبەرئارقا سەپ قىسقا ئۇچۇر يوللايدىغان كۆزنىكىدە، بىربىرسىنى، سەن پالانىتىڭ،  ھازىر نامىڭنى ئوزگەرتىۋاپسەن..
باشقىلارنى تىللىسام نىما كارىڭ....
  نىمىشكە ئىنكاسىمنى نەقىل ئالىسەن ھىلىقى دەپ سەت گەپلەر بىلەن  تىللىايدىغان، بىربىرسىنى كورەلمەيدىغان غەيۋەتخۇر كىشلىرىمىزئارقى ئارقىدىن مەيدانغا كىلىشكە باشلىدى..
مۇنبەر تىنىچلىق بۇزۇلۇشقا باشلىدى..

شۇنىڭ بىلەن مىسرانىم باشلىق باشقۇرغۇچى تاللاش ئىشىنى ئوتۇرىغا قويۇپ، باشتا دوسلارنىڭ كورسەتمىسى  ، ھەم باشقۇرغۇچى  ئارزۇسى بولغۇچىلارنىڭ كوڭلىنى يەرگە قويماي،ئىشىنىپ تاللىغان ئىدى...
بەزەن تورداشلار.. بۇنىڭغىمۇ قانائەت قىلماي... شۇ تاللاشقا رازى بولماي.. يەنە، شاللايدىغىنى شاللاپ، كىرگۇزدىغىنى  كىرگۇزۇپ.. ئاخىرى 20 نەچچە باشقۇرغۇچىنى بىكىتتى..
ياخشى تاللىنىش بولدى دەپ خۇش بولغان ئىدۇق...
ئەپسۇس، بۇنىڭغىمۇقانا ئەت قىلماي.. كونا خاماننى سورۇشقا، سىسىق توپا چىقىرىشقا ئۇلگۇردى....
بەزەن باشقۇرغۇچىلارنىڭ كوڭلىگە ئازار بىرىپ، خۇشاللىق ئىزدىدى..
بۇمۇ ئوتۇپ كەتتى!
ئوتتۇرلقىتا، ئۇيغۇر تىل يزىقىمىز ئۇچۇن كۈچ چىقىرايلى دىگەن مەندە، ۋىندۇۋىۇس شىركىتىنىڭ يۇمشاق دىتالىغا ئۇيغۇر تىلىنى كىرگۇزۇش  تەكلىۋى، ھەم تورداشلارنىڭ قزغىنلىق بىلەن تاشلىغان بىلىتى بىلەن 5-ئورۇندىن بىراقلا 149مىڭ نەچچە يۇز بىلەت تاشلاش سانى بىلەن بىرىنجى بولدۇق...
ھەجەپ بىز ئومكەنمىز، ئۇرۇشماي، تىللاشمىساقلا.. باشقىلار ھايران قالغۇدەك نەتىجىگە ئىرىشەلەيمىزكەنغۇ دەپ  خوشاللىقنىڭ ئۇستىگىلا، ،،،،،،،،،،،،،

  تۇركىيەدىن داڭلىق سەنئەتكار، ئۇرۇمچىدە  كىچىلىك ئوتكۇزدى..  بىلەت باھاسىنىڭ ھەقىقەتەن ، ئاددىي پۇقرانى چۇچۇتكۇدەك.. 880سوم دىن،،،120 سومغىچە بولغانلىقى، بىلەتنى تالىشىپ ستۋالغانلىقى توغرىسىدىكى ئۇچۇرغا ئىگە بولغانلىقى، بەزەنبىر تورداشلىرىمىزنڭ جۇدۇمىنى ئورلەتتى...
ئۇلار دەيدۇ<< بىز شۇنچىۋالا بايمۇ.. بىزنىڭ قىيىنچىقىتا قالغان قىرىنداشلىرىمىز ھەقىقەتەن كوپ تۇرسا، شۇ ئالغان بىلەتنىڭ پۇلىنى شۇ كىشلىرىمىزگە بەرسە بولمامدۇ.. بىزقاچىنغىچە ناخشا خۇمار خەلق دىگەن مۇبارەك قالپاقنى كىيىۋەرگۇچە.... بىلىم خۇمار خەلىق دىگەن ئاتاققا ئىرىشسەك، بىز ئىرىشەلمىسەكمۇ كەلگۇسىدىكى ، كىلەچىكىمىزنىڭ يۇلتۇزى بولغان باللىرىمىزئىرىشسە، شۇ ئىرىشىش ئۇچۇن، ئۇل سالساق، مائارىپقا، مائارىپىمىزغا، يوقسىزلىرىمىزغا كۆڭۈل بولسەك بولمامدۇ...
ھوكۇمىتىمىزنىڭ ئەڭ توۋەن تۇرمۇش سوممىسىغا تايىنىۋىلىپ، ئويدە بىكىنىۋىلىپ ئولتۇرغۇچە، ئوزىمىز چىقىش يوللىمىزنى تىپىپ،  ساختا نامراتلىق قالپىقىمىزنى چورۇپ تاشلاپ.. بىيىش يولىدا بولساق بولمامدۇ دىگەنلارمۇ بولدى..

بەزىلەر، بۇ  شەخسىنىڭ ئوزىنىڭ ئىشى، ياردەم قىلامدۇ ، قىلمامدۇئوزىنىڭ تاپقن پۇلى، بىرىپ كورسە نىمە بولۇپتۇ دىگەنلەرمۇ بولدى..
شۇنىڭ بىلەن.. ئىككى گورۇھ شەكلىىنىپ، جىدەل مەجرا بولغىلى تۇردى..
بۇلارنى قانداق قىلساق ئىناق قىلغۇلى بولىدۇ دىگەننى ئويلاپ، بىرقارارغاكىلەلمەي ئولتۇرغۇنىمىزدا، بۇ كوڭلىمىزنى يىرىم قىلىپ، يىغا كەلتۇرىدىغان خەۋەرمۇ يىتىپ كەلدى..

دوسلىرىم  سىز ۋە بىزنىڭ ياشىشىمىز قىسقە مۇددەتلىك ھايات .. بۇ جاھاندا مەڭگۇ پۇت تىرەپ تۇرەلمايمىز..
بۇ زىمىنغا يالىڭاچ كەلدۇقمۇ، ھامان  بىر كۈنى ئىككى مىتىر داكا بىلەن كىتىمىز...
ھەممە ئىشنىڭ سورىقى بولدۇ..
سىز.. ئوزىڭىزچە، مىنى ھىچ كىم تونىمىسا، مەن بۇ  تاختا ئارقىلىق  خەت  بىسىپ تىللىۋالسام، ئىتىۋالسام بولىدىغۇ..مىنىڭ ياغىنىمنى كىم بىلەتتى دىگىىڭىزبىلەن.. شۇ قىلغاننىيىتىڭىز.. شۇ 10 بارمىقىڭىزبىلەن يازغىنىڭىز بىلەن شۇ قىلغان ناچار نىيىتىڭىزنڭ سورىقى بولىدۇ..

بىز  بىر ئىنسان بالىسىدۇرمىز...
باشقىلارنى ئازار بىرىش بىزگە ياراشمايدۇ..
  بىلدىڭلارغۇ...شۇ بالايى ئاپەتلەرنى،  ئئاپەتلەرنىڭ دەستىدىن ، قىينىلىپ كىتىۋاتقانلارنى، ئاغرىق ئازابى تارتىۋاتقانلارنى، بىلىپ تۇرىمىز، ئۇلارغا ماددىي جەھەتتىن داۋا تىپىلغىنى بىلەن، شۇ كۆزئالدىغا كەلگەن، كورگەن ئويلەر ئورۇلگەن، تاش خىشلارنىڭ ئارىسىدا، مىنۇت سىكۇنىتنىڭ ئىچىدە  قىلىپ، جان تالىشىپ، كۆزئالدىدا ئولۇپ كەتكەن... قۇتقۇزۇشقىمۇ قۇربىي يەتمەيدىغان ئاجىز ئىنسان ئىكەنلىكى بىلگىنىدە، شۇ، يۇرەك  ئازابىنىڭ داۋاسىغا داۋا تىپىلامدۇ..

  باشقىلارنى تىللاش ھاقارەتلەش بىلەن خۇمار بولۇپ قالغان دوستۇم.( راسۇلىلاھ مۇنداق دىگەن..
ساڭا زۇلۇم قىلغان ھاقارەتلىگەن ،  دۇشمىنىڭ ھىسابلايدىغان ئىنسان بالىسىنى دوستۇم دە...
چۇنكى، سىنىڭ كەڭ قۇرساقلىقى، شۇ دۇشمىنىڭ ھىسانلايدىغان دۇشمىنىڭ، ئوزىنىڭ اتالىقىنى بىلىپ يىتكەندە.. ھامانەم يىنىڭغا كىلىدۇ.. شۇڭا، ئۇنى دوستۇم دىگىن.. تەربىيە قىلغىن دىگەن..
دىمەك.. كەمنىنە ئوچۇق يۇرۇقلۇق بىلەن. كوڭلۇمدىكىنى دىدىم..
ئۇمۇد قىلىدىغىنىم.. بر بىرسىمىزنىڭ ئاچقان يەرىگە بىرىپ، ئىناق ئوتسەك ئومۇتتە..

  خەير ئامان بولۇڭلار.. قىيبچىلىق بار كىشلىرىمىزگە ياردەم قىلىشنى ئۇنۇتماڭلار!
