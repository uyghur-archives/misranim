ئەسسالامۇ ئەلەيكۇم ھۆرمەتلىك قېرىنداشلار .ئۇزۇن بولدى سىلەر بىلەن يۈز كۆرۈشمىگىلى .سەۋەبىنى سىلەر بىلىسىلەر ئىنشائاللاھ .
ئاللاھقا بولغان ھەمدۇسانالار ۋە ھۆرمەتلىك پەيغەمبىرىمىز مۇھەممەد سەللەللەھۇ ئەلەيھىي ۋەسسەلەمگە سالاملار ۋە ئۇنىڭ ئائىلە تاۋاباتلىرىغا دۇئالاردىن كېيىن ‹ۋەتەننىڭ ئەبەدىي ئەمىنلىكى ۋە ئۇزۇن ۋاقىت مۇقىملىقى ھەممىدىن ئەلا › دىگەن  سۆز بىلەن سۆزۈمنى باشلاي .
شۇنداق قىلىپ جېكسون ئۆلۈپ كەتتى .ئۇ توغرىسىدا ئاز بولمىغا مۇنازىرىلەر تالاش تارتىشلار بولدى .نۇرغۇن ئادەم ئۇنىڭ مۇسۇلمان ئىكەنلىكىگە ئىشىنىدۇ .ياكى ياخشى ئارزۇدا بولىۋاتىدۇ .جۈملىدىن مەنمۇ شۇنىڭ ئىچىدە .ئۇنۇڭ پەردە ئارقىسىدىكىلەر بىلەن قىلغان ئىدىيىۋى كۆرىشى بەلكىم راست بولىشى مۇمكىن .مەيلى راس بولسۇن ،يالغان بولسۇن ۋە ياكى جېكسون مۇسۇلمان بولسۇن بولمىسۇن ،جېكسوننى ئۇلار يەھۇدىيچە قىممەت قارىشى، ئوچۇغىنى ئېلىپ ئېيتقاندا شەيتانلىق تۇرمۇشقا كۆنۈش قارىشىنى ئىنسانلارغا ،جۈملىدىن ئالدى بىلەن خىرىستىيانلىقنى ،ئىسلاملىقنى ۋە ياكى باشقا دىنلىرىدىن ئازغان ياشلارنى تېخىمۇ پاتقاققا پاتۇرۋېتىش ،خۇداسىزلىققا ،ھاياسىزلىققا ،ئەدەپسىز -ئەخلاقسىز ۋە جىنايەتكە مۇپتىلا قىلىش،ئىنسانلىقنى يوقىتىپ ھايۋانى تۇرمۇشقا قىزىقتۇرۇش ،ئىنسانى ئەخلاققا ،ئىنسانلار ئىتىپاقلىقىغا بۇزغۇنچىلىق قىلىش مەقسىتىدە جېكسوننى تاللىغانلىقى ھەقىقەت .يەنە بىر قېتىم دەيمەنكى ئۇلارنىڭ مەقسىتى ئالدى بىلەن تەقۋالارنى ئازدۇرۇش ئەمەس بەلكى ئازغان ياشلارنى تېخىمۇ ئازدۇرۇش ،شەيتاننى ،شەيتانىۋى نەپسىنى چىرايلىق كۆرسىتىش .
جېكسوننىڭ نۇرغۇن ناخشىلىرىدا ھاياسىزلىق ،جىنسىيەت ھۆكۈم سۈرىدۇ .بەزىلىرى ھەتتا ھەركەت بىلەن تەسۋىرلىنىدۇ .جېكسون مودا بولغان بۇ دەۋىردە دۇنيادا مىتال موزىكا ،دېسكو موزىكا ئەڭ بازار تاپقان دەۋىر بولدى .جېكسوننى دېسكو ۋە مىتال موزىكىلىرىنىڭ يول ئاچقۇچىسى دىسەك ئارتۇق كەتمەس .شۇڭا ئامىرىكا پىرىزىنتى ئۇنىڭغا شەرەپ ئوردىنى بەردى .ئۇنىڭ ئامىرىكىچە ئوچىغىنى ئېيتقاندا يەھۇدىيچە ياشاش قارىشىنى ئىنسانلارغا تارقىتىش رولى ھەقىقەتەن زور .يەھۇدىيچە قىممەت قارىشى دىگەن نىمە ؟ئۇنى سىز يەھۇدىيلارنىڭ تەلمۇد دىگەن كىتابىنى ئوقىسىڭىز بىلىسىز .ماددىي مەئىشەتنى ئاللاھتىن ئۈستۈن قويۇش ۋە خۇداسىزلىق ئۇلارنىڭ ئىنسانلارغا سىڭدۈرمەكچى بولغان قىممەت قارىشى .
سىلەر بىلەمسىلەركىن ھەر يىلى ئۆتكۈزىلىدىغان روك ،مېتال موزىكا بايرىمىدا كوللىكتىپ بۇزۇقچىلىق قىلىش،قالايمىقان بۇزۇقچىلىق قىلىش ..............تەك ئىنسان ئەقلى كۆتىرۋالغۇسىز ئەخلاقسىزلىقلارنىڭ خەۋىرى كېلىۋاتىدۇ .مانا بۇ پەردە ئارقىسىدىكىلەرنىڭ ئاخىرقى مەقسىتى .موزىكىنىڭ ئىنساندىكى شەيتانلىقنى قوزغاش رولى ھەتتا بەزىدە ھاراقتىن قېلىشمايدۇ .ھاراقنىڭ تەسىرى بىردەملىك بولسا موزىكىنىڭ تەسىرى ئۇزۇن مۇددەتلىك بولىدۇ .جەميەتتىكى نۇرغۇن موزىكىغا ھېرىسمەن بالىلارنى بولۇپمۇ قىزلارنى ،جىنسىيەت كۆزىڭىز بىلەن كۈزىتىپ باقسىڭىز بىر ھەقىقەتنى بايقايسىز .بۇ ھەقتە ئارتۇق گەپ قىلىش بىھاجەت .لېكىن مەن موزىكىنى بىراقلا رەت قىلمايمەن .لېكىن مۇشۇ دىنى سەۋىيەرىم بىلەن قانداق موزىكىنى ئاڭلاشقا بولىدىغانلىقىنى بىلەلمىدىم .بولۇپمۇ ھازىر ئېيتىلىۋاتقان نۇرغۇن موزىكىلار ئەخلاقنى چىرىتىش ۋە بىخوتلۇق ئېلىپ كېلىدىكەن .بىر قىسىم ئابلاجان ئاۋۇتسىمان ،ئالتىشەسىمان ناخشىلاردا ئەخلاق ،جەسورلۇق مەزمون قىلىنسىمۇ لېكىن يەنىلا شەيتانىۋى مۇھەببەت ،شەيتانىۋى قىلىق ھەركەت ھۆكۈم سۇرۇپ دېسكو ئۇسلۇبىنى ئىنسانلارغا سىلىق ئۇسول بىلەن ياخشى كۆرسىتىش مەقسىتى يوشۇرۇنغان .بولۇپمۇ ئابلاجان ئاۋۇت موزىكىلىرىدا .ئابلاجان ئاۋۇت موزىكىلىرىنىڭ بالىلارنىڭ ئۆسۈپ يىتىلىشىگە بولغان سەلبىي تەسىرىدىن ئەنسىرەپ تۇرغۇچىنىڭ بىرىمەن .
گەپ ئۇزاپ كەتتى .جېكسوننى مۇسۇلمان دىدۇق .ئۇنى ئاقلىدۇق ،ئۇنىڭغا چاپا ياپتۇق .لېكىن نىمىلا دىگەن بىلەن ئۇ بىر ئۆمۈر يەھۇدىيلارغا پۇل تېپىپ بىرىش ،شەيتانىۋى كۈچلەرنىڭ مەقسىتىنى ئەمەلگە ئاشۇرۇش ئۈچۈن ئىشلىدى .ئۇ مۇشۇ جەرياندا نۇرغۇن سەلبىي قالدۇقلارنى قالدۇرۇپ كەتتى .ئەمدى بىز داۋاملىق ئۇنى ئاقلىساق ئۇنىڭ ياشلارغا بولغان  سەلبىي تەسىرىنى ئادىللىق بىلەن ماس قەسەمدە ئوتتۇرغا قويمىساق ،جېكسونغا بولغان چوقۇنۇش ،جېكسوننى دوراش خائىشىنى پەيدا قىلىۋېلىشىمىز مۇمكىن .يەنى پەرىشتە سۈپەت جېكسوننى ئەمەس بەلكى شەيتانلار لاھىلەپ بەرگەن شەيتان سۈپەت جېكسوننى ياخشى كۆرۈپ قېلىشىمىز مۇمكىن .
مەن دائىم چچ بوشلۇقلىىرىدا ،فېنبېيسىمان بلوگ تور بەتلىرىدە بەتنىڭ ئۈستىگە ئاللاھ ،مۇھەممەد دىگەن رەسىمنى بىكەت ئىچىگە جېكسوننىڭ چاترىقىنى تۇتۇپ دېسكو ئويناۋاتقان كۆرۈنۈش بار مەنزىرىلەرنى كۆرۈپ قالىمەن .بۇنداق ھادىسە ئوتتۇرا مەكتەپ بالىلىرىدا بەك ئومومىي يۈزلۈك ھادىسىكەن .
ھەقىقى مۇسۇلمان بولغان موزىكانتلار بار .ئۇلار ئاشكارە ھالدا ئۆزىنىڭ ئۆتۈپ كەتكەن بىھۇدە ھاياتىدىن ئۆكۈنۈش سۆزلىرىنى قىلىدۇ .ئۇلار ھەقىقەتەن پۇشايمان يىدى ،ھەقىقەتەن ئۆكۈندى .ئۆزلىرىنىڭ بۇرۇنلاردا ئېيتىۋەتكەن قىلىقسىز ناخشىلرىنى ياشلارنىڭ ئاڭلىماسلىقىنى دائىم تەۋسىيە قىلىدى .لېكىن جېكسون ئاشكارە ھالدا بۇنى تەۋسىيە قىلالمىدى .ئاللاھ ئۇنىڭغا بۇنداق پۇرسەت بەرمىدى .لېكىن بىزنىڭ ياشلىرىمىزدا جېكسون مۇسۇلمان بوپتۇ دىگەن خەۋەرنى ئاڭلىغاندىن كېيىن ئۇنىڭ ناخشىرىنى ئاڭلاش خائىشى ئەكسىنچە كۆپەيگىلى تۇردى .جېكسوننىڭ مۇسۇلمان بولۇش مەقسىتى بىلەن بىزنىڭ ئەمەلىي ھەركىتىمىز تەتۇرۇ تاناسىپ تۈزۈپ قالدى .
شۇنىسى ئېنىقكى مەيلى شەرئىي ھۆكۈم ۋە پاتىۋالاردىن ئىزدىسىڭىز جېكسوننىڭ ئۇسلۇبىدەك دېسكو ۋە دېسكوسىمان موزىكىلار .مىتال موزىكىلار بىردەك ھارام دائىرىسىگە كىرىدۇ .گەرچە بەزىلىرىنىڭ تېكىستلىرىدە شەيتانلىق كۈيلەنمىسىمۇ لېكىن ئۇنىڭ ئۇسلۇبى ئاشۇ شەيتانىۋى موزىكا ئۇسلۇبىنى يوشۇرۇن تەستىق قىلىدۇ ۋە سىزنىڭ سەزگۈڭىزنى تۇيغۇزماي باشقىلىرىنى ئاڭلاشقا مايىل قىلىدۇ .موزىكىنىڭ ئىنسان ئەقلىنى ۋە ئەخلاقىنى تۇيغۇزماي بۇزۇش رولىنى ھەقىقەتەن سەل چاغلىغىلى بولمايدۇ .بۇ ھەقتە ماۋۇ تېمىنى تەۋسىيە قىلىمەن
http://www.shewqiy.com/thread-1304-1-1.html
دىمەك جېكسونغا مەبلەغنى يەھۇدىي ئىقسادىي گۇروھىي سالغان،ئۇلار جېكسوننى تەربىلىگەنكەن ئۇلاردا بىر رەزىل مەقسەت پۈكۈلگەن بولىدۇ .شۇڭا كەڭ قېرىنداشلىرىمدىن سورايدىغىنىم جېكسوننىڭ مۇسۇلمان ياكى ئەمەسلىكى توغرىسىدىكى مۇنازىرىلەر سىزنى شەيتان سۈپەتتىكى جېكسوننى ۋە ئۇنىڭ موزىكىلىرىنى ياخشى كۆرۈشكە ئېلىپ بارمىسۇن .بەلكى ئاللاھقا يېقىلىشىشنى ئىستىگەن ،ئۆتكەن ئۆمىرىگە ئېچىنغان ۋە ئازغۇن ئادىمىي شەيتانلارغا لەنەت تىلىگەن پەرىشتە سۈپەت جېكسوننىڭ رۇھىدىن ئۆگۈنۈپ ،مۇسۇلمان سۈپەتتە ياشاشقا ئىنتىلدۇرسۇن .ئىبرەت ۋە ئۆگىنىش ،ئەمەلىيتىمىمىزگە تەدبىغلاش ئۆز ئىپادىسىنى تاپسۇن .
