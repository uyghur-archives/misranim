رەھمەتلىككە ئىچىم ئاغرىيدۇ جۇمۇ ... سەھنىدە ئۇ كىشىلەرنى سەۋدا قىلىۋېتىدىغان گېگانىت چولپان سۈپىتىدە تۇرغان بىلەن ، ئۇنىڭ شەخسى تۇرمۇشى تولىمۇ ئوڭۇشسۇزلۇق ئىچىدە ئۆتكەن .ئۇنىڭ ئەڭ تونۇلغان ۋاقىتلىرىدا تاپقان پۇللىرىغا ئەڭ گۈزەل جايلاردىن ئېسىل ئۆيلەرنى سېلىپ ، ئاۋۇ ئۆزىنى تاتىلاپ يىغلاپ گۈپ-گۈپ يىقىلىۋاتقان گۈزەللەردىن ئەڭ گۈزەللىرىنى تۇتۇپ ھاۋايدا بىرنى ،پارىژدا بىرنى ئويناپ،كاتتا سورۇنلاردا يەپ-ئىچىپ يۈرسىمۇ ھەرقانداق بىر ئادەمدىن ئاشۇرۇپ قىلالىشى مۇمكىن ئىدى...لېكىن ئۇ ئۆزىنىڭ غېمىنى يىمىدى ، كوچىلاردىكى ئاچ-يالىڭاچ سەرگەردانلارنىڭ غېمىنى يىدى ،ئېرقى كەمسىتىشكە ، ئادالەتسىزلىككە ئۇچرىغان قەۋمىنىڭ غېمىنى يېدى . زۇلۇمغا ئۇچرىغان ، ئېگىسىز-قالغانلارنىڭ ئاھۇ زارىنى ناخشا قىلىپ توۋلىدى . شۇ چاغدىكى ‹ئۇلار بىزگە كۆڭۈل بۆلمەيدۇ›دېگەن ناخشىسى ھازىرقى بىزنىڭ چۈشەنچىمىزدىكى ‹سىياسى ناخشا›ئىدى. ئۇ قورقماي ئوقۇدى ،مۇكاپاتىغا شۇ ناخشىسى قاتارلىق ئالبۇملىرى بىر مەزگىل چەكلەندى .ناخشا يۈرىكىگە تېگىپ كەتكەن ئاقسۆڭەكلەر ئۇنى قەستلەشكە ئۇرۇندى.ئۇنىڭغا ھەرخىل تۆھمەتلەرنى چاپلاپ ئۇجۇقتۇرۇشنىڭ كويىدا يۈردى...لېكىن ئۇ بوشاشمىدى ، ھەقىقەت ئۈچۈن كۈرەش قىلىشنى توختاتمىدى، ساخاۋەت ئىشلىرىنى توختاتمىدى .ئۇ سەھنىگە چىقمىغان 12يىل ئىچىدە پۇللىرىنى ئىئانە قىلىپ تۈگىتىپ ، قەرزگە بوغۇلۇپ ، ئۆزىنىڭ بەزى ماددى بۇيۇملىرىنى سېتىپ پۇلىنى ئىشلەتكەن...ئۇنىڭ نۇرغۇن پۇللىرى نەگە كەتكەن؟ ساخاۋەت ئىشلىرىغا ، ھەقسىز كېسەل داۋالاش ئورۇنلىرىنى قۇرۇشقا ، بالىلار ھەقسىز ئويۇن ئوينايدىغان باغچىلارنى قۇرۇشقا سەرىپ قىلدى ...12يىلدىن كېيىن ئۆتكۈزۈلمەكچى بولغان كېچىلىكى ئۇنىڭ ئۇزۇن يىللىق ئىسسىق-سوغۇقلىرىنىڭ تەجىربىسى ، ھايات مۇساپىسىنىڭ جەۋھىرى ، دۇنيا تاماشا ساھەسىدىكى غايەت زور كېچىلىك بولماقچى ئىدى،لېكىن كېچىلىككە3 كۈن قالغاندا جان ئۈزدى...ھەقىقەت ئۇ ئۆلگەندىن كېيىن ئاشكارلاندى ، ئۇ ھايات ۋاقتىدا ئۇنىڭغا تۆھمەت چاپ]لىغان كىشىلەر يىغلاپ تۇرۇپ خەلق ئالەم ئالدىدا ھەقىقى ئەھۋاللارنى ئاشكارلىدى...بىچارە رەھمەتلىك باشقىلارنى ئويلاپ ياشىغان بىر ئۆمرىدە تۈزۈكرەك بىر راھەت كۆرمەي ئۆتۈپ ، ئەمدى ئۆلگەندىن كىيىنمۇ بەزىلەرنىڭ ھاقارىتىگە ئۇچراپ تۇرۇپتۇ ...
بىر مۇسۇلمان ئۆلگەن كىشىنىڭ يامان گېپىنى قىلمايدۇ ، چۈنكى ئۇنىڭ ھاياتلىقىدا قىلغان ياخشى-يامان ئەمەللىرىگە قۇدرەتلىك ئېگىمىز ھۆكۈم چىقىرىپ بولغان بولىدۇ...
ھاي...ھاي....باشقىلارغا باھا بېرىش ئوڭاي ھە...ئۆزىمىزچۇ؟ مەن ئۆزۈمنى ئۇنىڭ ئورنىغا قويۇپ ئويلىنىپ باقتىم ، مەن ئۇنىڭدەك ئۇتۇق ئېگىسى بولغان بولسام ، قوڭنى چىڭ قىسىپ تاپقان پۇلنى يىرتىپ خەجلەپ جاھاننىڭ راھىتىنى سۈرۈپ ياشار ئىدىم بەلكىم، تۆت بۇلۇڭغا ئىككى تال قوتا مەسچىت سەپ قويۇپ ، ھەرەمگە ماڭدىم دەپ قويۇپ رىمدا ئويناپ يۈرسەم ھەممىسىنىڭ قەلبىدە ئورنۇم بولىۋېرەر ئىدى...مەنلا ئەمەس كۆپىنچىمىز شۇنداق بولۇپ كېتىشىمىز مۇمكىن ، لېكىن ئۇ ئۇنداق قىلمىدى ... ئۇنى باشقۇرىدىغان ، ئۇنى چۈشەپ تۇرىدىغان ئادەملەر بولمىسىمۇ ئۇ ئۆزى بىلىپ ئىنسانىيەت ئۈچۈن قولىدىن كېلىدىغىنىنى قىلدى ...............

بۇرادەر، خاپا بولمىسىڭىزكەن، گەپ قىلغاندا ئانىلارنى  ...

ماڭا مايكول جىكسۇننىڭ ھەرقانداق ئىشى كىرەك ئەمەس .ئۇ نىمە ئىش قىلدى ؟ يامان ئىش قىلدىمۇ ياخشى ئىش قىلدىمۇ مەن بىلەن مۇناسىۋەتسىز .چۇنكى ماڭا پايدىسى يوق ھەم ئۆلگەن ئادەم ئۆزىنى گۇۋاھلىق بىرەلمەيدۇ .بۇ تىمىنى پەقەت كۆرۇپ قويساملا بولدى .
  
  سىزدىن سوراپ باقسام ،مۇشۇ تىمىنى يوللىغان ئادەم نىمە مەقسەتتە بۇنى يازغاندۇ ؟ كاللىسى ئىششىپ قىلىپ يازغانمىدۇ ياكى ئوقۇغان ئادەم ئۇچۇن پايدىسى تىگىپ قالا دەپ يازغانمىدۇ ؟ مەيلى ئۇنىڭدىكى ئىشلار راست بولسۇن ياكى يالغان بولسۇن ،بىزگە قانچىلىك تەسىر كۆرسىتەلەيدۇ ؟

  ھەرھالدا ئانىڭىزنىڭ سىزنى ئوغانبالا دەپ تۇغقانلىقىنى بىلىدىكەنسىز .ئۇنداق بولغاندىن ئانىڭىزنىڭ 9 ئاي قوساق كۆتەگەن ھەققى ھۆرمىتى ئۇچۇن بولسىمۇ ، باشقىلارنىڭ ئەمگىكىگە سەن پەن دىمەڭ .ئۇ يا سىزنىڭ چۆمىچىڭىڭىزنى ئۆرۇپ ، چۆچۆرىڭىزنى خام ساناتقۇزغان يىرى يوقتۇ .
