ئەسسالامۇئەلەيكۇم قېرىنداشلار !      مىنىڭ بۇ تېمىنى يوللىشىمدىكى مەقسەت ئوغۇز،ھون،تۈرك،ئۇيغۇر لارنىڭ ئارسدىكى مۇناسىۋەتنى ئايدىڭلاشتۇرۇش . خوش ئۇنداقتا ئوغۇزلار كىم¿ نىمە ئۈچۈن ئوغۇزلار دەپ ئاتىلدۇ ¿ھەممىمىزگە مەلۇم ئوغۇز دىگەن گەپ يىڭى تۇغۇلغان بوۋاقنىڭ ئانىسنى تۇنجى ،بىرىنچى قېتىم ئەمگەن سۈتىنى ئوغۇز سۈتى دەيمىز .ئوغۇزخان دەل شۇ تۇغۇلۇپ ئوغۇز سۈتنى بىر ئىمىپ ئىككى ئەممەي ئورنىدىن دەس تۇرغانلىقى ئۈچۈن ئوغۇزخان دەپ ئاتالدى ،يىڭى تۇغۇلغان بوۋاقنىڭ ئوغۇز سۈتنى ئىمىپلا ئورنىدىن تۇرۇپ كىتىشى ئادەمنىڭ ئەقلىگە سىغمايدىغان بىر ئىش ئەمما قەدىمكى كىشلەردە ئۆز ئاتامانلىرىنى ،پادىشاھلىرىنى ئۇلۇغلايدىغان،بەھەيۋەت ،قۇدرەتلىك بىلىدىغان ئادەتلەر بولغان ،ھەرقانداق بىر قەۋم ئۆز سەردارلىرىنى ئۇلۇغلايدىغان مىساللار كۆپ .بۇنداق قىلىشتىكى مەقسەت ئۆز قەۋمىنىڭ روھىنى ،جاسارىتنى ئۇرغۇتۇپ ھاياتقا بولغان ماسلىشىش ئىقدىدارىنى يۇقىرى كۆتۈرۈپ ئۈستۈنلۈك تالىشىش ،ياخشى يىيىش ،ياخشى كۈنگە ئىرىشىش ئۈچۈن . شۇ زامانلاردا ياخشى كۈن ئۇرۇش ئارقىلىق قولغا كەلگەن .ئۇرۇشتا يېڭىش ئۈچۈن بىر ئۇلۇغ ،قۇدرەتلىك رەھبەر بۇلۇشى كېرەك .ئەجدادلىرىمىز مۇشۇ ئۇلۇغ رەھبەرنى ئوغۇز خان دەپ بىلىپ رەھبىرىمىز ئانىسنى بىر ئىمىپلا ئورنىدىن دەس تۇرغان 7يېشىدا 20ياشلىق ئادەمدەك كۈچ ۇۋەتكە تولغان بىزنى ھىچقانداق كۈچ يىڭەلمەيدۇ ،چۈنكى بىزنىڭ شۇنداق كۈچلۈك رەھبىرىمىز بار دەپ بىلىپ ئۆزلىرىنىىڭ روھىنى ،جاسارىتنى ئۇرغۇتۇپ زور شان -شەرەپلەرگە ئىرىشكەن .ھەم پادىشاھىنى ئەزىز بىلىپ ئوغۇزخان ئۇرۇق ئەۋلادلارنى ،قەۋمىنى ،ئوغۇزلار دەپ ئاتىغان .ئوغۇزخاننىڭ 6ئوغلى ھەربىر ئوغلىنىڭ 4تىن ئوغلى بۇلۇپ جەمئىي 24نەۋرسى بولغان بۇ نەۋرلىرىنىڭ ئۇرۇق ئەۋلادلىرى بىردىن ئوغۇز قە ىلىسنى تەشكىللەپ 24تۈركمەن قەبىلىسى ياكى 24ئوغۇز قەبىلىسى دەپ ئاتالغان .بۇ توغرىسىدا مەھمۇت بوۋىمىز ۋەئوسمانلى تارىخچىللىرى خەۋەر بىرىۋاتىدۇ .ھەمدەئۆزلىرىنى ئوغۇزخان غا باغلاپ چۈشەندۈرمەكتە .ئەمدى بىزئۇيغۇرلار بىلەن بۇ ئوغۇز قەبىللىرى ئوتتۇرسىدا قانداق مۇناسىۋەت بار دىگەندە بىز ئۇيغۇرلار تارىختا 9ئوغۇز نامىدا تىلغا ئېلىندۇق دىمەك ىز 9ئوغۇزلار دەل شۇ 24ئوغۇز (تۈركمەن )قەبىلىسدىكى 9ئوغۇز قەبىلىسى قالغان 15قەبىلە بولسا ھازىرقى تۈرك ،تۈركمەن ،ئازەربەيجان ،بۇلغار ،ھونگىر ،پەچەنەك قاتارلىق ئوغۇز قەبىللىرىدۇر .ئەمدى ھون سۆزى بىلەن ئوغۇز سۆزىنىڭ قانداق مۇناسىۋىتى بار دىگەندە خەنزۇچە تارىخنامىلەردە 9ئوغۇزلار 胡人(خۇرېن)دەپ ئاتالغاي يەنە تەنرى (تەڭرى)دىگەن سۆز 天人(تىيەنرېن)،قەيەردە ،نەدە دىگەن سۆز (哪的，哪里)دەپ ئاتىلدۇ بۇ يەردىكى نەدە ،تەنرى دىگەن سۆز ھازىرغىچە ياۋرۇپا ئاسىيا چىەرىسدىكى تۈركىيە دە قوللىنىلماقتا .قېنى ئەقلىمىز بىلەن بىر تەلەپپۇز قىلايلى تەنرى ،تەڭرىم ،تىيەنرېن  (天人)نەدە نادې (哪的)قانداق ئاھاڭى ،تەلەپپۇزى ئوخشامدىكەن ،ئەمدى خۇرېن (胡人)نى ئىشتىىك تەلەپۇز قىلايلى خۇرن ،خۇن ،خون ،ھون…قانداق خۇرېن (胡人)ھونغا ئۆزگىرىپ كەتتىمۇ ¿قەدىمدە خەنلەر بىلەن خوشنا ئولتۇرغان بەزىدە ياخشى ئۈتۈپبەزىدە سوقۇشۇپ ئارلىشىپ تىلدىكى بەزى ئاتالغۇلار ئارلىشىپ كەتمەيدۇ ،سىڭىشىپ كەتمەيدۇ دەپ كىم كىسىپ ئېيتالايدۇ .دىمەك ھون دىگەن نامنى خەنلەر بىزگە قۇيۇپ قويغان ،ھازىرقى ۋېيزۇ ،ۋېيۋۇر ،خۇيگۇ ،خۇيخې دىگەندەك .ئەمدى ئۇيغۇر دىگەن گەپكە كەلسەك 9ئوغۇزلار شەرقى تۈرىكچە 9ئۇيغۇر غەربى تۈركچە 9ئوغۇز دىيىلدۇ بۇنى يەنە چۈشەندۈرۈشنىڭ ھاجىتى بارمۇ ¿ئۇيغۇر ،ئوغۇز بىر گەپ خۇددى تۈركچىدە كۆزنى ،گۆز   توغرىنى ،،دوگرا           مەن نى   ببەن دەپ ئاتىغاندەك بىر ئىش ،ياكى ئۇيۇشقان ،بىرلەشكەن ئوغۇزلار ،ئۇيۇشئوغۇز ئۇيۇشغۇز ئۇيغۇز ئۇيغۇر دەپ بارابارا ئىتتىك تەلەپپۇز قىلىنىپ ئوڭاي تەلەپپۇز قىلىندىغان ئۇيغۇر سۆزىگە ئۆزگىرىپ كەتكەن بۇنداق مىساللار ھەم ئۆزگىرىپ كەتكەن سۆزلەر تىلىمىزدا نۇرغۇن .خوش ئۇداقتا تۈرك سۆزى بىلەن ئوغۇز سۆزىنىڭ مۇناسىۋىتىگە كەلسەك تۈرك تۈركمەنلەر 24قەبىلە بۇلۇپ ئۇلار ئوغۇزخاننىڭ 24نەۋرىىسىنىڭ ئىسمىنى ئۆزىگە ئۇرۇق قەبىلە نامى قىلىپ بىر پۈتۈن ئوغۇز ئەۋلادلىرى 24ئوغۇز (تۈركمەن)قەبىلىسگە ئايرىلغان ،بۇنى مەھمۇد قەشقىرى ۋە ئوسمانلى تارىخچىللىرى ئۆز ئەسەرلىرىدە ئېنىق ئېيتقان .يەنە بىر جەھەتتىن تۈرك دىگەن سۆز ،سۈرلۈك ،قەپىقى تۈرۈك،بەھەيۋەت ،قامەتلىك ،كۈچلۈك دىگەن مەنىلەردە .مەسلەنسۈرتوقاي قىلدى ،تۈرلۈپ كەلدى ،قاپاقلىرى تۈرۈلۈپ كىتىپتۇ ،چىرايى ئېچىلمايدىغان تۈرۈك تۇردىغان ،ئادەم قارىسا ئەيمىندىغان ،قۇدرەتلىك ،بەستلىك دىگەن مەنىلەرنى بۈلدۈردىغۇ ،بۇھەقتە خەن تارىخنامىلەردە ھونلار ،سۈرلۈك ،تۈرۈك ،بەستلىك،قاڭشارلىق ،كۆزلىرى چوڭ ،ئورا كۆز ،ئادەمنىڭ قوققىسى كەلگىدەك ھەيۋەت ،ئۇسقانلىق كىلىدۇ دەپ خاتىرلەنگەن .دىمەك بىز كىمكەممىز …بۇنى تەتقىق قىلىش كەتمەيدۇ ،ئەجدادلىرىنى بىلمەيدىغانلەر ئە مدى بىلدىڭلارمۇ ¿ بىزدىگەن ئوغۇزلار شۇ !ھە راست سەلجۇقلارمۇ 24ئوغۇز قەبىلىسنىڭ بىرى
