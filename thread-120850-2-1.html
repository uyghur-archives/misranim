يوللىغان ۋاقتى 2014-3-15 01:30:12 PM |ھەممە قەۋەتنى كۆرۈش |
بۇرۇن تاكسىلارنىڭ ئارىسىدىن قارا ماشىنىلارنى تاللاپ توساپ كوياتتۇق، مانا ئەمدى قارا ماشىنىلارنىڭ ئارىسىدىن تاكسىنى ئاران-ئاران توسىيالايدىغان بولدۇق.  ئەتىگەندە ئىشقا ئالدىرىغاندا، تاكسى بولامدۇ، قارا ماشىنا بولامدۇ، توختىغىنىغا خۇش بولۇپ چىقىۋېرىمەن، باھانى ئىشقىلىپ بەك ئېشىپ كەتمىسىلا تالاشماي بېرىمەن. نەچچە قېتىم كەينى كوزۇپكۇلۇق ماشىنىغىمۇ چىقىۋەرگەنمەن.  ھەي شۇ خىزمەت بولمامدۇ؟  ئۆيۈم خىزمەت ئورنۇمدىن خېلىلا يىراق، ئەمما...ھەي بىچارە ئاپام دائىم مېنى:
-سەن ئەتىگەن 6 دە ئورنۇڭدىن تۇرساڭمۇ، يېرىم كېچە ئۈچتە تۇرساڭمۇ ئوخشاش كېچىكىسەن!-دەيدۇ.                                  قارىغاندا شۇنداق ئوخشايمەن، تاكسى، قارا ماشىنىغا چىقىپمۇ كېچىكىپ قالىمەن، بۇنى مەندىنلا كورسىڭىز بولمايدۇ، نەچچە يىل بۇرۇن سالغان تاشقى ئايلانما يولى مانا ئەمدى بىردەمدىلا خەلىققە بەخىت ياغدۇرىمىز دەپ، چېقىپ ئوڭشىماقچى بولىۋاتىدۇ، بىزگە بەخىت بولاتتىمۇ بۇ؟ ئاپتوبۇس ماڭغىلى بولمايدىغان يولنى ياساپ، ئۇ چېقىپ تۆھپە كۆرسىتىپ يۈرۈپ.      كېلىن قىز نېمە قىلىۋاتىدۇ؟دېسە، سۆكۈپ تىكىۋاتىدۇ، دېگەندەك ئىشلا.  شۇنداق قىلىپمۇ بۈگۈن يول بويدا تۇرۋېلىپ، بوينۇمنى غازنىڭ بوينىدەك سوزۇپ ئۇياق-بۇياقلارغا ئىلتىجا قىلىۋاتقىنىمدا، ئاسماندىن چۈشكەندەكلا ئالدىمدا بىر سېرىق رەڭلىك تاكسى توختىدى. ‹‹خۇدا بەرسە، پەيغەمبىرىمىز غىڭ قىلالماپتۇ››دېگەندەك، ماڭا ھەسەت ھەۋەس ئارىلاش قاراشلىرىغا پەرۋا قىلماي، تاكسىنىڭ ئارقا ئىشىكىنى ئاچتىم.                                                                                                     -ئەسسالامۇ ئەلەيكۇم!                                                                                                                               تاكسىغا چىقىپ ئەمدىلا ئىشىكنى يېپىشىمغا قۇلىقىمغا ئۇرۇلغان بۇ يېقىملىق سالامدىن مەڭدەپ ئىتىىك بېشىمنى كۆتۈرۈپ ئۇدۇلغا قارىدىم. 30ياشلار چامىسىدىكى بۇ ئۇيغۇر يىگىت ماڭا قاراپ كۈلۈپ تۇراتتى، ‹‹يا ئاللاھ ، ماۋۇ مەركىزى شەھەردە ھېلىغۇ تاكسىكەن، بومبا ساقاللىقلارنىڭ قارا ماشىنىسىغا چىقسىمۇ بۇنداق سالامنىڭ ئورنىغا ‹‹15كوي بېرىسىز جۇمۇ›› دېگەن گەپ چىقىدىغان بۇلۇپ كەتكەن ئىدى، ئەمما ئەتىگەندىلا قانداق تاكسىغا چېقىپ قالدىم مەن››                   -ۋەئەلەيكۇم ئەسسالام، خاپا بولماي شىنجاڭ ئۇنۋېرستېتىغا ئاپىرىپ قويامسىز؟-دېدىم مەنمۇ دەرھاللا ئۇيغۇر قىزى تۈسۈم بىلەن مۇلايىمغىنە. ‹‹باشقا كۈنلىرى قارا ماشىنا شوپۇرلىرى بىلەن باھا توغۇرلۇق قىزىرىشىپ قوپاللىشىپ كەتكەن ئەلپازىمنى كورگەن بولسىڭىز، بۈگۈنكى مەن بىلەن سېلىشتۇرۇپ ياندۇرۇپ سالاتتىڭىز، ھېچ بولمىغاندا ئۆزىڭىزنى نەچچىنى كاچاتلىۋىتەتتىڭز.   ئەلۋتتىدە، كىمگە ياخشى گەپ ياراشمايدۇ دەيسىز، ھەممە ئادەم ياخشى، سىلىق گەپ ئالدىدا ئەسلى تەبىيتىمىزگە يانىمىزغۇ؟››    دېدى تاكسى شوپۇرىمۇ.                                                                                                                 بىزنىڭ مەھەللىدىن 500 مىتىر ماڭغاندىن كېيىنلا بىرىنچى بۆلۈملۈك ماشىنا توسۇلۇش ھادىسىسى يۈز بېرىدۇ. مەن كۇنۈپ كەتكەچكىمۇ، ئىشقىلىپ ئۆزۈمنى سوراپلا ئۇزۇنراق ئىشتىن بىرنى قىلىمەن. شۇڭا بۈگۇنمۇ ئېلكىتاپنى ئېچىپ تۇرشۇمغا، يېنىمىزدا ئوخشاش بىر تاكسىدىن،يەنى شۇنداق سېرىق تاكسىدىن بىرسى كېلىپ توختىدى، بۇنداق يېرىمى سېرىق تاكسىلار بىز تەرەپلەردە ئاز بولغاچقا، كۆزگە ئالاھىدە چېلىقىپ تۇراتتى. ئەمما تېخىمۇ قىزىقارلىق يېرى مەن چېققان تاكسىنىڭ شوپۇرى بىلەن بۇ تاكسىنىڭ بېشىغا سېرىققا مايىل گاز شارپىنى يۆگەپ قارا كۆزەينەكنى ياراشتۇرۇپ تاقىۋالغان ئۇيغۇر قىز ئىككىسى پاراڭلىشىۋاتاتتى:                                                                                                                بەك ئەتىگەن چىقىپ كېتىپسىز بۈگۈن؟ مەڭ مانى!                                                                                                      ياندىكى تاكسىنىڭ قىز شوپۇرى شۇنداق دەپلا بۇ يېگىتكە ئەپچىل تاماق قاچىسىنى سۇنۇپ بەردى.                                -تۈنۈگۈن بەك چارچاپ قاپسىز قارىسام، شۇڭا ئويغاتمايلا چىقىپ كەتكەنىدىم،- يىگىتمۇ شۇنداق دېدى.                            ئاڭغىچە ماشىنىلار لۆمۈلدەپ مېڭىشقا باشلىدى، ئۇلار چىرايلىق خوشلاشقاندىن كېيىن بىز ياندىكى ئايلانما يولغا، ئۇ قىز ئۇدۇل كېتىپ قالدى.   تاشقى يولغا چىققاندىن كېيىن، كۈن نۇرى كۆزىنى چېقىپ كەتكەن بولسا كېرەك، شوپۇر يىگىت ئالدىدىكى تاختىنى چۈشۈرۈپ قويدى،كۆزۈمگە بىر پارچە رەسىم كۆرۈندى. رەسىمدە ئىككى سېرىق تاكسى، بىر جۈپ ئاق كېيىملىك قىز-ئوغۇل، يەنى بايىقى ياغلىقلىق قىز بىلەن بۇ شوپۇر يىگىت كەڭ كەتكەن دالىدا كۈلۈپ تۇراتتى. شۇندىلا بۇ ئككەيلەننىڭ ئەر-ئايال ئىكەنلىكىنى بىلىپ، سورىدىم:                                                                                                   -ئەسلىدە بايىقى قىز ئايالىڭىز ئىكەندە؟ مەن ئەزەلدىن تاكسى ھەيدەيدىغان ئەر-ئاياللارنى ئۇچۇرىتىپ باقماپتىكەنمەن، بۇلۇپمۇ ئايالىڭىزدەك چىرايلىق، ئىمانلىق ئايال تاكسى شوپۇرلىرىنى! سەل ھاياجانلانغان بولسام كېرەك، ئالدىغا ئېڭشىپ تۇرۇپ كۈچەپ سۆزلىدىم.                                                                                                                    -ھاھاھا....شۈكرى بەك بەخىتلىك ياشاۋاتىمىز. بۇ ھېكايىنى سۆزلەپ بېرەيمۇ سزگە، قارسام ئوقۇغان قىزدەك قىلىپ تۇرىسىز!                                                                                                                                                          كۈلۈپ قويدۇم، يىگىتنىڭ تەلەپپۇزدىن بۇ ئىككىسىنىڭ مۇھەببەت ھېكايىسى ئادەتتىكىدەك ئەمەس ئىدى.                          -ماقۇل، بۇ ماشىنا توسۇقچىلىقىدا سۆزلەپ بەرسىڭىزمۇ بولغۇدەك.  ماشىنىلا ئايلانما يولدىمۇ شۇنداق ئاستا مېڭىۋاتاتتى. ‹‹كۆرۈكنىڭ ئۈستى بۇنداق، ئاستى قانچىلىكتۇ؟››                                                                                                 مەن بۇ يىل 33 ياشقا كىردىم، ئۆزۈم سانجلىق.  سانجى ئېنىستىتۇتىنىڭ ئىقتىساد كەسپىنى پۈتتۈرۈپ، بىر شېركەتتە مالىيە ئىشلىرىنى ئىشلىدىم. 8 يىل ئىشلەپ، چارچاپتىمەن، شۇنىڭ بىلەن ئۇنى-بۇنى قىلىپ، تاكسىدىن بىرنى سېتىۋالدىم  كۈندۈزى ھەيدەپ، كېچىلىككە ئادەم ياللىدىم، گەرچە بۇ تاكساۋاتلىق قىلىش ئەركىن ئىش بولسىمۇ، ئەمما تۇرمۇشىڭىز مۇشۇنچىلىك يەرگىلا قامىلىپ قالدىكەن دەڭە،- دەپ ئىككى قولى بىلەن ماشىنىنىڭ كورسۈتۈپ قويۇپ،- گېپىنى يەنە داۋاملاشتۇردى.- بېشىدا ئانچە دىققەت قىلماپتىمەن، كېيىنچە يېشىمنىڭ چوڭىيشىغا ئەگىشىپ ئەتىراپىمدىكى جۈپ-جۈپلەرگە قاراپ ھەۋىسىم كېلىدىغان بۇلۇپ قالدى،دېمەيسىز. ئاشىق-مەشۇق كۈنى دېگەن كۈننى ئويلاپمۇ قويمايتىم،  شۇ يىلى ئاشىق-مەشۇق بايرىمىدا بىر جۈپ قىز-ئوغۇلنى شەھەر سىرتىدىكى قار تېيلىش مەيدانىغا ئەچىقىپ كويغۇچە،  ئۇ ئككىسىنىڭ بىر-بىرىگە بولغان مېھىرى-مۇھەبىتىنى، ئامراقلىقىنى، كۆيۈنۈشىنى كورۈپ كۆڭلۈم شۇنداق يېرىم بۇلۇپ قالدى.    2010-يىلى10-ئاينىڭ 10-كۈنى تۇغۇلغان كۈنۈم ئىدى، بىر كۈن دەم ئېلىپ ئۆيۈمدە يېقىن بىر ئاغېينەمنى ساقلىدىم، ئەمما خېلى ساخلىساممۇ كەلمىدى، بىر ۋاقىتتا تېلفۇنۇمغا ناتۇنۇش نۇمۇر كەلدى، ئالسام دوختۇرخاندىن كەپتۇ، سىز پالانچىنى دەيدۇ،ھەئە دەپلا دوختۇرخنىغا چاپتىم،ئەسلىدە ئۇ ئاغىنەم ئەتىگەندە ئۆيدىن چىققىچە تېيلىپ كېتىپ يىقىلىپتۇ،شۇ يەردىلا يۈرەك كېسىلى قوزغىلىپ قاپتۇ،ئۇنى دوختۇرخانىغا ئېلىپ بېرىپ يانچۇقىنى ئاقتۇرسا مېنىڭ ئىسىم كارتۇچكام چىققاندىن كېيىن ماڭا تېلفۇن قىپتۇ، ئۇ ئاغىينەممۇ ماڭا ئوخشاش بويتاق دەڭە، شۇڭا بىرەر ئىش چىقىپ قالسا،چوڭلارنى چۆچۈتمەيلى دەپ بىر-بىرىمىزنىڭ ئسىم كارتوچكىمىزنى پورتېمالدىن ئايرىمايمىز، مۇشۇنداق ئىش بۇلۇپ قالساجددى چاقىرتىدىغانغا! بۇ ئىش بىلەن ئادەمگە ھەمراھ بولمىسا زادى بولمايدىكەن، بۇلۇپمۇ بىر ھەقىيقى ئەر بۇلۇشتىن ئىلگىرى ئۇنىڭ چوقۇم ئۆزىنىڭ ئائىلىسى بولۇشى كېرەك ئىكەن، شۇندىلا مۇكەممەل ئەر بۇلىدىكەن دەڭە!         شۇنداق قىلىپ مەن ئىككى يىل ئىچىدە چوقۇم بويتاقلىقتىن قۇتۇلىمەن دەپ ۋەدە قىلدىم، بىر ئىش قىلسىام بەك كىرشىپ قالىمەن، شۇڭا بۇ ئىشقىمۇ ئالاھىدە مۇئامىلە تۇتۇپ كەتتىم، ئەتراپىمدىكىلەر ھەرخىل قىزلارنى تۇنۇشتۇرۇشقا باشلىدى، ئۇلارنىڭ بەزىلىرى بىلەن مىجەزىمىز ئەپ كەلمىدى، بەزىلىرى بىلەن تۇرمۇش سەۋىيىمىز توغرا كەلمىدى دېگەندەك. شۇنىڭ بىلەن مەن رەسمىي بولغان يوللار بىلەن مېڭىپ باقاي دەپ، لايىق تۇنۇشتۇرۇش ئورۇنلىرىغا تىزىملاتتىم، ئۇلار يا ئاجىرىشپ كەتكەنلەرنى يا بولمىسا ئاجىرىشپ كېتىپمۇ بۇرۇنقى يولدىشى بىلەن ئىشى پاكىز تۈگمىگەنلەرنى تۇنۇشتۇردى.ھەتتا يولدىشى تۈگەپ كېتىپ، يەنىلا قېينئانىسىنىڭ ئۆيىدە تۇرۇپ ساداقەتمەنلىك كۆرسىتىۋاتقانلارنى تۇنۇشتۇردى. ئەمدى سىڭلىم، قاراڭ مېنىڭ تۇرمۇشۇم ھىندىستان كىنولىردەك بولمىغاندىن كېيىن، ئامال يوق ئۇنداق مۇرەككەپ ئىشلارنى قىلغۇم كەلمىدى.                                                                                                                 2011-يىلى 1- ئاي ۋاقىتتا، بىر چەتئەللىكنى ئايدۇرۇمغا ئېلىپ باردىم،ئۇ كىشى ماشىنىدىن چۈشتى، كەينىدىن مەنمۇ چۈشۈپ، ئارقا ساندۇقتىكى يۈكىنى ئېلىپ بەردىم، دەل شۇچاغدا، ياندا مېنىڭكىگە ئوخشاش تاكسى ماشىنا شوپۇرى بىر قىزنىڭمۇ ماشىنا ئارقا ساندۇقىدىن يۈك ئېلىۋاتقانلىقىنى كۆرۈپ قالدىم، ئۇنىڭ قىزىل رەڭلىك چاپىنى، ئۇزۇن چاچلىرىنى ئۆرۈپ بېشىغا مۇنداقلا ئارتىۋالغان شارپىسى كۆزۈمگە ئوت بولۇپ كۆرۈندى، ئۇنىڭغا قاراپ كۈلۈپ سالدىم، ئۇمۇ ئەدەپ يۈزسىدىن ماڭا كۈلۈپ قويدى-دە، ماشىنىسىنى ھەيدەپ كېتىپ قالدى. ئاپلا، دەپ ئارقىدىن قاراپلا قالدىم.                        شۇ چاغدا، ئۇ قىزنى توي قىلغانمىدۇ، مۇھىمى مېنىڭ توي قىلمىغانلىقىمنى، مۇشۇ تاپتا بىر لايىق ئىزدەۋاتقانلىقىمنى شۇ قىز بىلەمدىغاندۇ، دەپ ئويلاپ پۇچۇلىنىپ كەتتىم،دەڭە! خەقنىڭ پىشانىسىگە يېزىپ قويمايدىكەن، ئەلۋەتتە مېنىڭمۇ پىشانەمگە يېزىقلىق بولمىغاندىن كېيىن، خەق قانداق بىلەتتى.                                                                                شۇ كۈنى كەچتە، شۇ قىز بىلەن ئۇچۇرىشىپ قالغان مەنزىرىنى ئويلاپلا ياتتىم، ئۇيان ئويلاپ، بۇيان ئويلاپ كاللامغا تۇتۇقسىزلا مۇنداق خىيال كېلىپ قالدى. ئەگەر مەن تاكسىرىمغا شۇنداق بىر لايىق ئىزدەش ئېلانى چاپلىۋالسام، قانداق بۇلار دەپ ئويلۇدۇم.                                                                                                                                           ھاھاھا....شۇنداقمۇ قىلىپ يۈرمىگەنسىز ھە ئاكا؟-دېدىم مەن تۆمۈر رېشاتكىنڭ كەينىدىكى مەھبۇستەك تىرەپ تۇرۇپ كۈلۈپ كېتىپ.                                                                                                                                                 -ۋاي ئۇكام شۇ يىلى چاغاندىن كېيىنلا مەن شۇنداق ئېلاندىن بىرنى چاپلىۋالدىم. ماشىنام باشقا تاكسىلاردىن ئانچە پەرق قىلمىسىمۇ، پاكىز، پارقىراپ تۇرىدۇ، ئۇنىڭدىن باشقا رەڭلىك لايىق ئىزدەش ئېلانىم چاپلاقلىق: ئەسقەر، تاكسى شوپۇرى ۋە ئىگىسى، ئالىي مەكتەپ مەلۇماتلىق، سەمىمىي، ھېسىياتقا باي يىگىت، سىز بىلەن بىرلىكتە بىر ئائىلە قۇرۇشنى خالايدۇ، سىزنى مەڭگۈ قەدىرلەيدۇ...شۇنداق ئويى بارلار،*****نومۇرغا تېلىفون قىلىڭلار.  مۇنداق دېسەم باشقچە ئويلاپ قالماڭ، مەن ئادەتتىكى تاكسى شوپۇرلا ئەمەس، تاكسىنىڭ ئىگىسىمۇ بولغاندىن كېيىن، ماڭا تېلفون قېلىپ ئالاقە ئورنىتىشنى خالايدىغانلار ئىنتايىن كۆپىيىپ كەتتى، بۇ باشقا ئۇسۇل بىلەن لايىق تاللىغاندىن كۆپ راس ئىش ئىكەن دەڭە. ئۆزۈم كۆرۈشۈپ باققان كىشىلا203 بوپتۇ. ئەمما كۆڭلۈمگە ياققۇدەك بىرى يوق، كۆڭلۈمگە ھېلىقى قىز بەك ئورنىۋالدىمۇ بىلمەي يۈرگەن كۈنلەر، 5-ئاينىڭ بىر كۈنى غۇلجىدىن بىر نۇمۇر كېلىپ قالدى. تېلفوننى ئالسام، ئىنتايىن تاتلىق بىر ئاۋاز،  ئەسسالاممۇ ئەلەيكۇم، دەپ سالام قىلدى. كۆڭلۈم بىردىنلا سۆيۈنۈپ كەتتى. ئۇ قىزمۇ مەندەك ئۆزىگە لايىق تاپالماي يۈرگەن تاكسى شوپۇرى ئىكەن، شۇنىڭ بىلەن ئۇمۇ مەندەك ئېلان چاپلاشنى ئويلۇشۇپتۇ، سىزمۇ چاپلاڭ' ئاينۇر دېدىم، ھە راست ئۇ قىزنىڭ ئىسمى ئاينۇر ئىدى.                                                                                                                  ئۇ قىزمۇ ماڭا ئوخشاش ئادەتتە شوخ، ئوچۇق-يورۇق بولسىمۇ، مۇھەببەت ئىشىكىگە كەلگەندە پاسىپىراق، تارتىنچاق ئىكەن. شۇڭا ئىككىمىز ئىككى شەھەردە ئىكەنلىكىمىزگە باقماي گېپىمىز تۇگمەيتتى، تېلفونىمىزنىڭ باتاريىسى قالماي ئۆزلىكىدىن ئېتىلىپ قالغۇچە سۆزلىشەتتۇق. ئۇ قىزمۇ ماڭا ئوخشاش كۈندىكى نەچچە يۈز كوي پۇلنىڭ مەرىدىن كېچەلمەي، ئۆزنىڭ تاكسىسىنى ئۆزى ھەيدەپ كېچىسى باشقىلارغا ھەيدىتىپ، مۇھەببەتلىشىدىغانغا ۋاقتى يوق چوڭ بۇلۇپ كېتىپتىكەن. شۇڭا مەن ئۇ قىزغا دائىم، ئەنسىرمەڭ، ئىگىمىز ھەممىمىزنى جۈپ ياراتقان، ھېچكىم يالغۇز ئەمەس دەپ تەسەللى قىلاتتىم.   ئۇ قىزغا يەنە ئۆزەمنىڭ بەزى ئاماللىرىنىمۇ ئۆگتىپ قويدۇم، مەسلەن، ماشىنىغا ئېلان چاپلانغان ئىكەن، بىز ماشىنىنى چوقۇم ئۆزىمىزگە لايىق ئادەملەر بار جايلارغا ئاپىرشىمىز كېرەك ئىكەن، مەسلەن مەكتەپلەرگە،         شېركەتلەر بار يەرگە ئاپارساق، بىردەمدىلا شۇنىڭغا لايىق ئادەملەردىن تېلفون كېلىدىكەن. ئۆزىمىز مۇنداق ئەستايىدىل ئويلىۋېلىپ، تېلىفون قىلدىغانلار ئادەم ئەخماق قىلىدىغان كىچىك قىزلار بولسا ئاچچىقىڭىز كېلىپ قالدىكەن دەڭە!            ئۈرۈمچىدە ماشىنا توسۇلۇش ئەمەس، ماشىنا توختىتىش مەيدانى بۇلۇپ كېتىدۇ ئەتىگگندە، مېڭىپ بارغان بولساممۇ، خىلى يەرگە باراتتىم دەپ ئويلاپ كېتىدىغان مەن، ئەمدى ماۋۇ شوپۇر ئاكاشنىڭ ھېكايىسىگە قىزقىپ، ئاپلا، ھېكايە تۇگىگىچە باردىغان يېرىمگە بېرىپ قالسام، قانداق قىلارمەن دەپ ئويلاپ كېتىۋاتاتتىم.                                                                         شۇنداق قىلىپ مەن بىلەن ئۇ قىز ئىككىمىز ئىككى شەھەردە ئىككى ئېلاننى چاپلاپ، مۇشۇنداق قىزىل، يېشىل چىراغ ئارا مېڭىپ كەتتۇق، بىر ياقتىن خىزمەت قىلىمىز، بىر ياقتىن ھەمراھىمىزنى ئىزدەيمىز، بەزىدە چارچاپ قالسام، ماشىنىنى چەترەك يەرگە تارتىپ، چىملېقلاردا ئاسمانغا قاراپ يېتىپ كېتىمەن.                                                                                                                           شۇ يىلى9-ئايدا ئاينۇر ئۈرۈمچىگە ھەدىسىنى يوقلاش ئۈچۈن كېلىدىغان بولدى، ئېلان چاپلاپ يېرىم يىلغىچە ئىككىمىز تېخىچە بىرەر كۆڭۈلدىكىدەك لايىق تاپالمىغاچقا، بۇ قېتىم پۇرسەت بولسا كۆرۈشۈپ بۇ مەسىلىنى ھەل قىلماقچى بولدۇق، ئەمما ئۆكتەبىرلىك مېنىڭ جددى بىر ئىشىم چىقىپ قېلىپ مەن غۇلجىغا كېتىپ قالدىم، كۆرگەنسىز، ماۋۇ تەقدىرنى! بوپتۇلا كېيىنچە كورىشەمىز يەنە، دەپ ئۇلاپ بىر ھەپتىدىن كېيىن ئۈرۈمچىگە قايتىپ كېلىشىمدە، سايرامدا ماشىنامنىڭ چاقىنى تاش كېسىۋىتىپ نەچچە سائەت ھايال بۇلۇپ قالدىم، ئەمدى ماشىنىنى سۈرتۈپ ماڭاي دىيىشىمگە، مەن ھېلىقى چاغدا ئايدورۇمدا ئۇچۇراتقان قىز، تاكسىدىن چۈشىۋاتمامدۇ؟ ئوڭۇممۇ چۈشۈممۇ دەپ قالدىم. توۋا، يەنە شۇنداق قىزىل چاپان كىيىۋالغان ئۇ قىزنى بېشىدىكى شارپىسى كۆزەينىكى بىلەن قوشۇپلا تۇنىۋالدىم.                                                     ئەمما قىزلار ئەزەلدەن ئاكتىپلىق بىلەن گەپ قىلىپ باقمىغاچقا، قانداق گەپ ئېچىشىمنى بىلەلمەي، ئاران كەلگەن پۇرسەتنى قولدىن بېرىۋەتمەي دەپ ئاينۇرغا تېلفون قىلدىم. ئۇ كىرا ئېلىپ شىخوغا كەتكەن، بۈگۈن ماڭدىغان كۈنى ئىدى. ئۇنۇڭ نۇمۇرىنى بېسىپ بۇلۇپ، ئۇ قىزنىڭ ماشىنىسىغا شۇنداق قارىسام ھېلىقى ئېلان چاپلاقلىق تۇرمامدۇ، يۈرۈگۈم قېپىدىن چىقىپ كېتەي دېدى.
