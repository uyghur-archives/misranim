ئەسسالامۇئەلەيكۇم مىسرانىمدىكى قىرىنداشلار  ،ئۇلۇغ ئاللا ھەمىمىزگە نۇسرەت ئاتا قىلسۇن.مەن مىسرانىمغا ئەزا بولۇپ كىرىگەندىن كىيىن گەرچە دىدارلاشمىساممۇ  نۇرغۇنلىغان پىكىرى ئۆتكۈر قېرىنداشلار بىلەن تونۇشتۇم ھەم يىقىنلا دوسىت بولۇپ قالدىم،مەن بۇنىڭدىن بەك خوش.ئىشقىلىپ مۇنبەرگە كىرسەملا شۇنداق مەنىلىك ھېس قىلىمەن .مەقستكە كەلسەم،يىقىندىن بېرى تورداشلا زەن قويغان بولسا كېرەك يۇقۇرقى تېمىدا ئېيتىلغاندەك ،تېمىغا چۈشكەن ئىنكاس توغۇرلۇق بىر بىرسىمىزنىڭ كۆڭلىگە ئازار بىرىپ قويىۋاتىمىز،مىنىڭ بۇن ئىنكاسنى يىزىشىمدىكى سەۋەپمۇ مۇشۇ ئىشلارنى توغرا ئۇسسۇل بىلەن بىر تەرەپ قىلىشنى ئويلىشىپ بېقىش.مۇنبەردە جەمى 131800نەپەر ئادەم بار .بۇ ئەزالارنىڭ ھەر خىل خاراكتىردىكى ئادەملەرنىڭ بولىدىغانلىقىنى ھەمىمىزنىڭ بىلىشى زۆرۈر دەپ ئويلايمەن،يەنى ھەر قانداق بىر تېمىغا سوغۇققانلىق بىلەن مۇئامىلە قىلىشنى ئويلىشىپ بېقىش زۆرۈر.شۇ تېما بىزگە ياقسۇن ياقمىسۇن تېما ئىگىسىنىڭ مەقسىتىنى  ۋە كىشلەرگە بەرگەن تەسىرىنى   چوقۇم بىلىشىمىز كېرەك دەپ قارايمەن. تېمىغا چۈشكەن ئىنكاسلارنىمۇ ئوقۇش جەريانىدا شۇ ئىنكاس يازغۇچىنىڭ مەقسىدىنى ئوبدان چۈشىنىپ ،ئەگەر سەل چەكتىن ئىشىپ كەتكەن بولسا ئۇدۇللا ھەممە كۆرىدىغان ئىنكاس تاختىسىغا   يازماي بەلكى شۇ كىشىنىڭ ئۇچۇر ساندۇقىغا دەلىللە ئارقىلىق شۇ كىشىنىڭ خاتالىقىنى كۆرسىتىپ بېرەلەيدىغان ياخشى سۆز ئىبارىلەرنى ئىشلەتسەك چۇقۇم ئۇ قېرىندىشىمىزمۇ بىزدىن خۇرسەن بولىشى مۇمكىن.ئەگەر ئوچۇق ئاشكارىلا تەنبىھ بەرسەك شۇ كىشى چۇقۇم سىزنىڭ چىرايىڭىزدىكى تەبەسسۇمنى كۆرەلمەيدۇ،چۈنكى تور بىر مەۋھۇم دۇنيا،شۇنىڭ بىلەن شۇ  كىشى نۇرغۇنلىغان ئەزالارنىڭ ئالدىدا ئاڭلىغان بۇ تەنبىھنى خاتا چۈشىنىۋىلىشى ئارقىلىق مۇشۇنداق كۆڭۈلسىزلىكلەر تۇغۇلىشى مۇمكىن. مۇنبەردىكى ھەمىمىز قېرىنداشمىز ،شۇڭا ھەمىمىز ئۆتكۇر پىكىرلىرىمىز بىلەن بىر بىرسىمىزگە بىلمىگەننى بىلدۈرۈش  ھەمىمىزنىڭ ئورتاق مەجبۇرىيىتى... شۇڭا كەڭ تورداشلارنىڭ بۇ تەۋسىيەگە زەن قويۇپ بېقىشىنى تۆۋەنچىلىك بىلەن سورايمەن.

مەن مۇنبەرگە ئۇيغۇرۇمغا ۋە دىنىمغا موناسىۋەتلىك تىمىلارنى كۆرۈش ئۈچۈن چىقىمەن .مەن بۇ مۇنبەرگە يىڭىدىن ئەزا بولدۇم، بۇندىن بۇرۇن سەيياھ سۈپىتىدە زىيارەت قىلاتتىم. بىر قىسىم ئسمى ئۇلۇق تورداشلار ئىسمىغا چۇشلۇق ئىش قىلماي  تېمىدىن بوران چىقىرۋاتقانلىقىنى كۆرۈپ قورسىقىم كۆپسە، بەزى  ئىددىيسى ئوچۇق قىرىنداشلارنىڭ پىكىرلىرىدىن خوش بولدۇم. ۋە شۇلار بىلەن پىكىرلىشەي دەپ ئەزا بولدۇم. بەزى دىنى تېمىلاردىن قەستەن قۇسۇر تېپىپ تېمىنى تاقاتقۇزغانلار سىلەرگە دەيدىغىنىم ئۆزۈڭلارنى بىسىۋىلىڭلار كەسپى مۇنبەرگە يوللىسىمۇ بولىدۇ، لىكىن شۇ كىشىلەر سىلەرنىڭ ھالىڭلارنى كۆرۈپ مۇشۇ تىمىنى كۆرسە گۇناھىدىن  يېنىپ قالا دىگەن مەقسەتتە يوللايدۇ! شۇڭا قۇسۇر تاپمىساڭلار!! بۇ مۇنبەر ئۇنۋېرسال مۇنبەر بولغانلىقى ئۈچۈن .تېمىلارنىڭ سەرخىل بولغىنى ياخشى!! باشقۇرغۇچىلار سىلەرگە تەكلىپ قۇيۇن چىقىپ ئۇرۇش تالاش بولىۋاتقان  تېمىنى يۇيۇپ تاشلىماي، قۇلۇپلاپ قويساڭلار لازىملار ئىزدەپ كۆرسە، ئەجرى سىلەرگىمۇ بولاتتى. كەچۈرۈڭلار كۆپ سۆزلەپ قويدۇم.

كەچۈرۈڭ، خېنىم. سىز مېنى خاتا چۈشىنىپ قالغان ئوخشايسىز! ھەممىمىز ئوخشاشلا بىر قەۋم بولغانلىقىمىز ئۈچۈنلا، سىزنىڭ ئويلاشمايلا يازغان ساددا ئىنكاسلىرىڭىزنى ئوقۇپ، ئۆزۈمگە خاس بولغان غورۇر تۈپەيلىدىن سىزگە قىسقا ئۇچۇر ئارقىلىق رەددىيە بىلدۈردۈم. ھەرگىزمۇ سىز ئېيتقاندەك قىسقۇ ئۇچۇر ئارقىلىق قىزلارغا ‹‹ھاقارەت›› قىلمىدىم، باشقىلارغا ‹‹ئەخمىقانىلەرچە باھا بېرىش›› ئۈچۈن، باشقىلارنىڭ ‹‹رەددىيە›› سىگە ‹‹ئۇنى، بۇنى ››دەپ ئۆزۈمنىڭ ‹‹گېپىنى يورغىلىتىش›› ئۈچۈن مۈنبەرگە چىقمىدىم!
