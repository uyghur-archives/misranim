ئەسسالامۇ ئەلەيكۇم قىرىنداشلار ياخشىمۇ سىلەر بىز قىرىندىشىمىز بىلتاپنىڭ " توردا قەلەم ھەققى بەرگىلى بولماسمۇ؟ https://uyghur-archive.com/misranim/thread-121095-1-1.html  " دىگەن تىمىسىنى كورۇپ ناھايىتى مەمنۇن بولدۇق ھەمدە 99% قىرىندىشىمىز بۇ تەكلىپنى قوللاپ قۇۋۋەتلەيدىغانلىقىنى بىلدۇردى
لىكىن تىمىغا چۇشكەن ئىنكاسلاردىن بۇ ئىشنىڭ تىخىنىكىلىق ۋە ئىقتىسادى قىيىنچىلىقلىرى ۋە ئەسەر ئوغۇرلىرنىڭ پايدىلنىپ كىتىشى مومكىنلىكى قاتارلىق ئىنكاسلار بىزنى ھەقىقەتەنمۇ ئىككىلەندۇرۇپ قويىدۇ مەن مۇشۇ مەزمۇنلارغا قارىتا ئەڭ موھىم دەپ قارىغان ئىقتىسادى مەسىلىگە نىسبەتەن يەنە بىر تەكلىپ بىرىپ بىقىشنى ئويلاپ قالدىم .
ھازىرقى مۇنبەر توربەتچىلىكىدە ھازىرچە مىسرانىمنىڭ ئەزا سانى ۋە ئەسەر قوبۇل قىلىشى ئالدىنقى قاتاردا تۇرىدۇ شۇڭا مەنمۇ بۇ ئىشنى مىسرانىم توربىتىنىڭ ئالدىن ئويلىنىشىغا سۇنىمەن .
1 . ئالدى بىلەن بىلتاپ قىرىندىشىمىزنىڭ تەكلىپىنى ئەمەلگە ئاشۇرۇش ئۇچۇن   مىسرانىم مۇنداق بىر تىخنىكىلىق مەسلىنى يولغا قويۇشقا تىرشىپ باقسا يەنى بۇ مۇنبەرگە ئەزا بولغاندىن تارتىپ ھەر كۇنى مۇنبەرگە تىزىملىتىپ ھەر بىر كىرىپ-چىقىشتا ئەزالارنىڭ بانكا ھىساباتدىن بىر يۇەن ھەق ئىلىنسا ۋە بۇ ھەقنى
بىلتاپ قىرىندىشىمىزنىڭ تەكلىپىدىكى تىخنىكىلىق مەسلىنى ھەل قىلىش تەتقىقاتغا ئىشلەتسە ۋە شۇنداقلا بۇ
مەسىلە ھەل بولغاندىن كىيىنمۇ بۇ تۇزۇم داۋاملىشىپ چۇشكەن پۇلنىڭ 90%تى ئەسەر مۇكاپاتىغا ئىشلىتىلسە
قالغان 10%نى ، بىر مۇنبەر خەيىر-ساخاۋەت ۋەقپىسى (فونىد) قۇرلۇپ ، مۇنبەرگە يوللىنىۋاتقان كىسەللەرگە
ياردەم بىرىش توغرىسىدىكى ئەسەرلەردىكى شەخىسلەرگە ياردەم قىلىنسا ، بۇنىڭدا پۇتۇن ئىقتىسادى ئىغىرچىلىق مىسرانىمغلا يۇكلەنمەستىن بۇ مۇنبەرنى زىيارەت قىلغۇچىلارمۇ مۇشۇ ئۇسۇل ئارقىلىق
ئىقتىسادى جەھەتتىن قوللاپ قۇۋۋەتلىسە .     
2 . بىلتاپ قىرىندىشىمىزنىڭ تەكلىپىگە ئاساسەن مۇنبەردىكى ئەسەرلەرنى ئەزالارنىڭ باھالىشغا مۇنبەر پۇلىنىڭ
ئورنىغا ھەقىقى پۇل ئارقىلىق باھالاش ئەمەلگە ئاشۇرۇلسا بۇنىڭ ئەڭ توۋەن سوممىسى 0.1 مو بولسا
مۇشۇ ئارقىلىق ئەسەر ئىگسىنى ۋە ياخشى ئىنكاسلارنى رىغبەتلەندۇرۇش ئەمەلگە ئاشۇرۇلسا .
3 . بەزى قىرىنداشلار ئۇ تىمىغا بۇ ئىشنىڭ ۋۇجۇدقا كىلىشى مومكىن بولمايدىغانلىقى توغرىسىدا ئىنكاس يىزىپتۇ بۇنىڭغا مەن "مۇۋاپىقيەت دەستۇرى " دىگەن كىتابنىڭ ئاپتۇرىنىڭ بىر سوزىنى نەقىل كەلتۇرەي " تەڭرى
سىنىڭ قولۇڭدىن كەلمەيدىغان ئىشنى سىنىڭ خىيالىڭغا كەلتۇرمەيدۇ ، سىنىڭ ئوزۇڭنىڭ خىيالىنى ئەمەلگە
ئاشۇرۇش ئىقتىدارىڭدىن گۇمانلانغىنىڭ ، تەڭرىگە قىلىنغان ھاقارەتتۇر ، چۇنكى تەڭرى ئوزى ئىنتايىن قۇدرەتلىك ۋە ئىقتىدارلىقتۇر ، شۇڭا ئۇمۇ ئىقتىدارسىز ئادەملەرنى ياراتمايدۇ ، ئىنسان تەڭرى ئوزىگە بەرگەن
ئەڭ ئۇلۇغ نىئمەتنىڭ تىخى 5%تىنىمۇ ئىشلەتمىدى ئۇ بولسىمۇ سىنىڭ مىڭەڭ ، ئادەمنىڭ مىڭىسىگە دۇنيادىكى
ئەڭ چوڭ  ئىسكەندىرىيە  كۇتۇبخانسىدا ساقلانغان 3 مىليۇن دانە كىتابدىكى ھەممە ئۇچۇرلار سىغىدۇ ، ئەگەر
سەن مەندىن تەڭرىنىڭ ئەڭ ئۇلۇغ موجىزىسى نىمە دەپ سورساڭ مەن ساڭا ھەرگىزمۇ ئاسماندىكى ھەيران قالارلىق يۇلتۇزلار ھەركىتى دەپ جاۋاب بەرمەيمەن ، مىنىڭ جاۋابىم تەڭرىنىڭ ئەڭ ئۇلۇغ موجىزىسى بىزنىڭ مىڭىمىز دەپ جاۋاب بىرىمەن ، چۇنكى ئادەمنىڭ مىڭىسىدىكى ھۇجەيرىلەرنى تىزىپ چىقساق پۇتۇن يەرشارىنى
بىر ئايلاندۇرۇشقا يىتىدۇ ، سەن ئاخىرەتتە تەڭرىگە بىرىنجى بولۇپ مۇشۇ ئۇلۇغ موجىزىنى نىمىگە ئىشلەتكىنىڭدىن سورلىسەن ."
ئەمدى بىز ئۇنىڭ دەۋاتقىنىنى ئىسپاتلايمىز دىسەك گەپ ئۇزراپ كىتىدۇ ، ئەڭ ئاددىيسى بىزنىڭ قولىمىزدىكى
يانفۇن ، ئاسماندىكى ئۇچقۇ ۋە مەن ھازىر ئالدىدا ئولتۇرۇپ تىما يىزىۋاتقان مۇشۇ كومپىيۇتىر ئەشۇ خام-
خىياللارنىڭ مۇۋاپپىقىيتى .
3 . ئەمدى ئەسەر ئوغۇرلىرنىڭ پايدىلنىپ كىتىشگە كەلسەك بىزنىڭ مۇنداق بىر ماقال- تەمسىللىرمىز بار
" باشقا كەلگەندە باتۇر " "جاندىن كەچمىگىچە جانانغا يەتكىلى بولماس " تۇرىكلەردە " ئاقىۋىتىنى ئويلىغان
قەھرىمان بولالماس " " ئولۇمنى كوز ئالغان ھەممە ئىشنى قىلالايدۇ " . ئۇ ئوغۇرلارنىڭ ئىشىنى يۇقۇردىكى
تىخنىكىلىق ئىشلار ھەل بولۇپ بولغاندىن كىيىن ئويلاشساقمۇ كىچىكمەيمىز ، ۋاقتى كەلگەندە ئۇنىڭغىمۇ ئامال
تىپىلىدۇ .
