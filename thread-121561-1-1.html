  چىڭخۇا ئۇنىۋىرسىتىتىدىكى بىر قىزنىڭ  ئوتتۇرا تىخنىكۇمدا ئوقۇۋاتقان بىر   ئوغۇل بىلەن بولغان كىيۇكىيۇ پاراڭ خاتىرىسى  



بىر ئادەم ئۈچۈن ئەڭ قورقۇنۇشلۇق ئىش ، تەرەققىياتتا بىزنىڭ ئالدىمىزدا مىڭىۋاتقان بەزى كىشلەرنىڭ بىزدىنمۇبەكرەك تىرىشىشى ،ئەگرى توقاي كەتكەن ھايات دولقۇنلىرى بىلەن توختاۋسىز كۈرەشقىلىشدۇر .مۇشۇنداق تۇرۇغلۇق بىزنىڭ تىرىشماسلىقمىزغا يەنە نىمە باھانىمىز بار .


قىز :بار كەنسىز ھە


ئوغۇل :با ،بىز ،كۆرۈشمگىلى ناھايتىمۇ بەك ئۇزاق بوپتۇ ،ھەي چىڭخۇائۇنىۋىرسىتىت ئوقۇغۇچىسى ،يىقىندىن بۇيان نىمە ئىشلار بىلەن ئالدىراش يۈرىۋاتىسىز؟


قىز :باشقا .ئىشمۇ يوق،شۇ ئۈگۈنۈش ۋە خىزمەتتىن باش كۆتۈرۈپ بولالماي،ئەمما ئىشقىلىپ ناھايتىمۇ بەك ئەھمىيەتلىك ئۆتىۋاتىدۇ بىزنىڭ ۋاقىتلىرىمىز

ئوغۇل :ھە مۇنداق دەڭ ،چى چى دىن ئىزدەپ تاپقىلىمۇ بولمايدۇ سىزنى ،نەچچەقىتىم  خەت يازسام خەتمۇ قايتۇرۇپقويمىدىڭىز ؟بەكلا ئالدىراشقۇ سىز ،چى چى تور بىتىڭىزنى لاھىلەپمۇ قويماپسىز،ھەتتا بىر نەچچە پارچە رەسىممۇ چىقىرىپ قويماپسىز ھە ؟


قىز :مەن ئادەتتە چى چى غا چىقىپ پاراڭلىشىپ ئولتۇرمايمەن .ئۇنداق قىلىشھازىرقى بىزنىڭ ياشلىق باھارىمىزنى ،قىممەتلىك ۋاقتىمىزنى ئىسراپ قىلغانلىق ئەمەسمۇ؟ھەر كۈنى چى چى ئىچىپ قاراپ ئولتۇرۇش ماڭا بەكمۇ مەنىسىز تۇيۇلىدۇ ،يەنە نۇرغۇنۋاقىت سەرپ قىلىپ چى چى تور بەتلىرىنى لاھىيلەپ ئولتۇرۇشمۇ بەزىبىر  بىكار تەلەپلەرنىڭ ئىشىدۇر .يەنىلا ئۆگىنىشتىنسىرتقى ۋاقىتلاردا رەسىم سىزىشنى ئۆگۈنۈش ،خەت مەشىقى قىلىش قاتارلىقلارغايەتمەيدۇ .


ئوغۇل :ھى  ھى ھى ....... ھازىر سىز نىمە ئىشقىلىۋاتىسىز ،سىزمۇ چى چى پاراڭلىشىپ ئولتۇرمىدىڭزىمۇ ؟


قىز :ئانچە مۇنچە ئىچىپ قويۇمەن شۇ ،ناھايتىمۇبەك ئاز ،بۈگۈن تۇيۇقسىز سىزنىڭ چى چى تور بىتىڭىزدىن بىر نىمىلەرنى كۆرۈپ قالدىم ،ئۇلارنى كۆرۈپ سىزگە ئازراق گەپقىلغۇم كىلىپ قالدى .


ئوغۇل :يازغانلىرىم قانداق ياخشى بوپتىمۇ ؟راسىتىنلا ئادەمنى تەسىرلەندۈرىدۇ ھە ،ھى ھى ھى ........


قىز :نىمىسى تەسىرلىككەن ،سىز قارىغاندامۇھەببەتكە تەشنا بولۇپ يۈرىۋىتىپسز،مانا بۇ مەۋھۇم تور دۇنياسىدىن قىز دوسىتتىپىپ مۇھەببەتلىشىشىنى ئويلاۋىتىپسىز ،مەن سىزگە دەي ،ھەقىقى قىز دوسىت تاپاي،دەپ ئويلىسىڭىز ،بۇ مەۋھۇم تور دۇنياسىدىن مۇھەببەت تاپالمايسىز ،بىر كۈن بۇمەۋھۇم تور دۇنياسىدىن بىر قىز دوسىت تاپتىمۇ دەيلى ،سىز تاپقان ئۇ قىزنىڭ ئەخلاقىپەزىلىتى ياخشى بولىشى ناتايىن ،چۈنكى ئەخلاقى پەزىلىتى ناچار قىزلار بۇ ئەمەلي تۇرمۇشتىنمۇھەببەتكە ئىرىشەلمەي بۇ مەۋھۇم تور دۇنياسىغا مۇھەببەت ئىز دەپ كىرگەنلەردۇر،ئەجىبا مۇشۇنداق قزلار سىزنىڭ مۇھەببەتلىشىشىڭىزگە ئەرزىمدا ؟


ئوغۇل :مەن بۇ سىزنىڭ بۇ كۆز قاراشلىرىڭزىغائانچە قوشۇلۇپمۇ كەتمەيمەن .


قىز :ھى ھى   ھى  ........ئەقىلگە ئىشارەت ،نادانغا جۇۋالدۇز ،ھەراسىت سىز بۇ يىل نەچچە ياشقا كىردىڭىز ؟


ئوغۇل :نىمە ،سوراپ قالدىڭىزغۇ ،سىزنىڭ بىرەرمەقسىتىڭز بارمۇ ؟


قىز :مۇنداقلا سوراپ قويدۇم .


ئوغۇل :24 ياش يىلان يىللىق


قىز:ھە ،ئۇنداقتا سىز چوڭ ئادەم قاتارىغا كىرىپ بوپسىز دە


ئوغۇل :قۇرۇق گەپ


قىز :سىز تىخى كىچىك گۆدەك بالا ئىكەنسىز تىخى ،ئەمدىبولدىمۇ ؟


ئوغۇل :سىز نەچچە ياشقا كىردىڭىز ؟


قىز :مەن قوي يىللىق ،سىز دىن ئىككى ياش كىچىك،ئەمما مەن پاراڭلىشىش جەريانىدا شۇنى ھىس قىلدىمكى گەرچە سىزنىڭ يىشىڭىز مەندىنچوڭ بولغىنى بىلەن ئەمما تۇرمۇشتا ئانچە پىششىپ يىتىلمەپسىز ،مەندىن نەچچە ياشكىچىكتەك ھىس قىلدىم سىزنى ،سىز تىلفۇندا تورغا چىقتىڭزمۇ ؟


ئوغۇل: ھە شۇنداق


قىز :قايسى ،ماركىدىكى تىلفۇن ؟


ئوغۇل :موتۇرۇلا


قىز :سۈرەتلىك كۆرۈشكىلى بولامدۇ ؟


ئوغۇل :مىنىڭ تىلفۇنىمنىڭ سىستىمىسى بەكلا تۆۋەن ،سۈرەت يوللىغىلى بولمايدۇ .


قىز :نىمىشقا ياخشىراقىغا ئالماشتۇرمىدىڭىز ؟


ئوغۇل :پۇل يوق ئەمەسمۇ ؟بەكلا كەمبەغەل ،سىزنىڭ مۇشۇنداق سورىغىنىڭىزدىنقارىغاندا سىزنىڭ ئىشلىتىۋاتقىنىڭىز ئالي ماركىدىكى تىلفۇن ئوخشىمامدۇ


قىز :مىنىڭ ئىشلىتىۋاتقىنىم ئالما ماركىدىى تىلفۇن ،ھازىر ئاليى مەكتەپئوقۇغۇچىلىرى ئاساسەن مۇشۇ ماركىدىكى تىلفۇننى ئىشلىتىۋاتىدۇ ،يىقىندا تىلفۇنۇمنىئالما بەشكە ئالماشتۇرۇشنى ئويلىشىۋاتىمەن .


ئوغۇل :ئۇنداقتا سىز پۇلدار ئائىلىدىن ئوخشىمامسىز ؟


قىز :مەن بۇ تىلفۇننى ئۆزۈم دەرسىتىن سىرتقى ۋاقىتلاردا ئىشلەپ تاپقان پۇلغائالغان .بۇ قىتىمقى قەلەپ ھەققىمنى ئالساملا يىڭىسىغا ئالماشتۇرىمەن .


ئوغۇل :ئۇنداقتا سىز دەرستىن سىرتقى ۋاقىتلاردا نىمە خىزمەت قىلىسىز ؟


قىز :مەن شۇ بىيجىڭدىكى بىر گىزىت باسمىخانىغا ياردەملىشىمەن ،بەك قىيىنمۇئەمەس ،كومپىيوتىردا شۇ كۈنلۈك خەۋەر مەزمۇنى بىلەن رەسىملەرنى بىرلەشتۈرۈپلاھىيلەپ بەرسەملا بولىدۇ ،ئەمما ئۇلارنىڭ بىرىدىغان مۇئاشى بەك ئازكەن ،بۇنىڭدىنياخشىرق خىزمەت تاپايمىكىن دەپ ئويلىشىۋاتىمەن .


ئوغۇل :ھە     مۇنداق دەڭ


قىز :سىزچۇ ،قانداق پۇل تىپىۋاتىسىز ؟


ئوغۇل :مەن پەقەتلا ئاتا ئانام ئەۋەتكەن پۇلنىلا خەجلاۋاتىمەن ،سىرتقا چىقىپئىشلەپ باقمىدىم .


قىز :ھەي سىز مۇشۇ ياشقا كىرگەن تۇرۇغلۇق يەنە ئاتا ئانىڭزىغايۆلىنىۋالدىڭزىما


ئوغۇل :دىگىنىڭىزغۇ شۇ نىمە ئامال دەيسىز ؟


قىز :سىزنىڭ ئاتا ئانىڭىز بەكلا باي كىشلەردىن ئوخشىمامدۇ ؟


ئوغۇل :ياقەي ،مىنىڭ ئۆيۈم يىزىدا ،بەكلا كەمبەغەلرەك  كىشلەردىن بىز


قىز :ئۇنداقتا سىز ئاتا ئانىڭزىنى بەكلا قىيناپ قويۇپسىز ،ئاتا ئانىڭىزغائىچىڭىز ئاغىرمىدىما ؟


ئوغۇل :ئاغىرماي قالامدۇ ؟


قىز:ئۇنداقتا دەرسىتىن سىرتقى ۋاقىتلاردا چىقىپ خىزمەت ئىزدىسىڭىز بولمامدۇ،ھازىرقى ئالىي مەكتەپ ئوقۇغۇچىلىرىدىن يەنە كىم بار دەيسىز ،دەرسىتىن سىرتقىۋاقىتلاردا ئىشلىمەيدىغان ،مىنىڭ بەزى ساۋاغداشلىرىم گىزىت سىتىۋاتىدۇ ،بەزىساۋاقداشلىرىم تازلىق ئىشچىسى بولۇپ يول تازلاۋاتىدۇ ،كۆپىنچىسى چوپچوڭ ئوغۇلبالىلار تىخى


ئوغۇل :ھە    شۇنداقمۇ


قىز :تەشۋىقات ۋاراقلىرىنى تارقىتىدىغان خىزمەت تاپسىڭزىمۇ بولىدۇ يا ،گەرچەتاپقان پۇل ئاز بولسىمۇ ئۆزىڭىزنىڭ تۇرمۇش خىراجىتىڭىزگە يىتىدۇ ،ئاتا ئانىڭزىنىئازراق قىينايسىز


ئوغۇل :مەن چوقۇم تىرىشىمەن


قىز :ئىغىزدىلا دەپ قويغان ھىساپ ئەمەس ئەمىليەتتە كۆرسەتسە ھىساپ


ئوغۇل :ئەمىليەتتە مەن شۇنداق قىلىشنىمۇ ئويلاشقان ،ئەمما مۇۋاپىقراقىنىتاپالمىدىم .


قىز :خى   خى   خى.......مۇۋاپىقراقى ،سىزنىڭ چوڭ شىركەتلەرگەكىرىپ قەھۋە ئىچىپ ئولتۇرۇپ خىزمەت قىلغۇڭىز بارمتتى يە ؟شۇنداق بولغان تەغدىردىمۇشۇنىڭغا چۇشلۇق قابىليەت بارمۇ ؟ھازىر ئۆزىمىزنى چىنىقتۇرىدىغان ياخىش ۋاقىت .ئاددىبولسىمۇ خىزمەت تىتىپ قىلمىسىڭىز قانداق بولىدۇ ؟


ئوغۇل ؛ھە ............


قىز :ئەر كىشى دىگەن كەسىپتە ئۇتۇق قازىنىشنى ئەڭ ئالا بىلىشى كىرەك ،مۇشۇنداقئەرلەر ئاندىن قىزلارنى  ئۆزىگە جەلىپقىلالايدۇ ،سىزنىڭ جەمىئەتتە بىر كىشلىك ئورنىڭىز ،مەلۇم مىقداردا ئىقتىسادىڭىز،مۇقىم ئولتۇراق ئۆيىڭىز بولغان ۋاقىتتا مۇھەببەت تەبىلا سىزنى ئىزدەپ كىلىدۇ . ھازىربەزى ئوغۇللار تالاي كەتكەن مۇھەببەتنامىلەرنى يىزىپ كىتىدىكەن .قىز رەت قىلسائەرلىك غورورىنى ئۇنتۇپ يىغلاپ كىتىدىكەن .مۇشۇنداق قىلىش ئەرزىمدۇ؟ مۇنداقئوغۇللار ئازراقمۇ غەيرەت جاسارىتى يوق ئوغۇللاردۇ ر .بىر قىز ئۈچۈن ئۆزىنى شۇنچەقىيناش كىتەمدۇ ،بىر كۈنى شۇ ئوغۇلنىڭ قىزلانى جەلپ قىلىش ئىقتىدارى بولسۇن قىنىشۇ چاغدا قىز ئوغۇلنى ئىزدەپ كىلىدۇ .


ئوغۇل :شۇنداقمۇ ؟


قىز :خى     خى     خى ........مەن سىزگە نۇرغۇن ئىشلانى سۆزلىسەم سىز ھىچنىمە بولمىغاندەكلاجاۋاپ قايتۇرىسىزغۇ ؟سىزدكى ئەركەكلەردە بولىدىغان غەيرەت جاسارەت نەگە كەتتى ؟


ئوغۇل :غەيرەت جاسارەت مەندە بارغۇ


قىز :مەن ئۇنداق ھىس قىلمىدىمغۇ تاڭ ،پاراڭلاشمايلا بولدى ،مەن يەنەگىزىتخانىغا بىرىپ ئىشلىمىسەم بولمايدۇ .مەن چى چى دىن چۈشۈپ كەتتىم خوش


ئوغۇل :خوش




   ئۇلۇغ ئاللا ئىگەم قىزلىرىمىزغا پاكلىق ،ئەخلاقى پەزىلەت ،يىگىتلىرىمىزگەئۇرغۇپ تۇرغان غەيرەت جاسارەت ئاتا قىلغاي .
