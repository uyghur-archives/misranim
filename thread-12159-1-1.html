ئەسسالامۇ ئەلەيكۇم مىسرانىم تور ئىشتىراكچىلىرى،مۇنبىرىمىزنىڭ ئەزىز كۆرۈرمەنلىرى،
مۇنبىرىمىز ساغلاملىق،مۇقىملىق، ئىتتىپاقلىقنى بايراق قىلىپ كۆڭۈل ئېچىشنى ئاساس قىلغان ئۇنۋېرسال مۇنبەر!
مۇنبىرىمىز ئىچكى قىسمى قايتىدىن تەڭشىلىپ ،ئەزالارغا نىسبەتەن نازارەت ۋە باشقۇرۇشنى كۈچەيتىپ ،مۇنبەر بىخەتەرلىكىنى قايتىدىن چىڭىتىپ ،ئېچىۋىتىلدى، بىر قىسىم سىياسىي خەۋەر ياكى ئويدۇرما ئىش،بۇرمىلانغان شېئىر نەسىرلەر تازلاندى، قالايمىقان باش سۈرەت چىقىرۋالغان، قالايمىقان ئىمزا قوللىنىۋالغانلارنىڭ پۈتكۈل ئارخبى يۇيۋىتىلدى .مۇنبەر مۇھىيتى قايتىدىن پاكىزلاندى، ئەگەر بىزنىڭ تەكشۈرشىمىزدىن ئۆتمىگەن ئەزالار بايقىلىپلا قالسا ۋاقتىدا ئۆچۈرلۈپ مېڭىلىدۇ! شۇڭىلاشقا ئىمزا ياكى باش سۈرەتكە ئالاھىدە سۈرەت ھەم گەپلەرنى يېزىۋالغانلار ۋاقتىدا ئۆزگەرتىۋىلىشىڭلارنى تەۋسىيە قىلىمىز!




مۇنبەر بىخەتەرلىكىمىز ئالاھىدە چىڭىتىلدى، خەۋەر ياكى زىل سەھىپىلەرگە يوللغانغان تىمىلارغا 3 كۈندىن ئاشقاندىن كىيىن ئىنكاس قايتۇرۇش چەكلەندى. يۇمۇر قوشاق ،ياكى كۆڭۈل ئېچىش خاراكتىرلىق سەھىپىلەرنىڭ ئىنكاس يېزىش ۋاقتى 20~30 كۈنگىچە بولدى،ئۇقتۇرۇش ياكى مۇنبەردىكى ئەزالارغا مۇناسىۋەتلىك تىمىلارنى پەقەتلا مۇنبەر ئەزاسىلا كۆرۈش ھوقوقى ئورنىتىلدى! شۇڭا داۋامى يوق،يېزىلۋاتىدۇ دېگەندەك ئەسەرلەرنى يوللاپ يېرىم يولدا قالماسلىقىنى ئۈمۈد قىلىمىز! 

بىخەتەرلىكنى كۆزدە تۇتۇپ ،ئالىي ئەزالارنىڭمۇ تېمىسى تەستىق بىلەن چىقىدىغان قىلىندى




بۇنىڭدىن كىين تەرجىمە خەۋەر ،ياكى قىسقا فىلىم ،ئېتوت كۈلدۈرگە،ھېكايە رومانلار ،مەزمۇنىڭنىڭ قانچىلىك بولشىدىن قەتئىي نەزەر تەستىقلىنىدۇ، بىر فىلىمنىمۇ تەستىقلىساق بولىدۇ! بىراق سىياسىيلىقى كۈچلۈك،ياكى مىللىتىمىز ھەم قېرىنداش مىللەتلەرنى كەمسىتىدىغان ھەرقانداق تىما ئىنكاس يوللىغان ئاپتورلارنىڭ IP سى ساقلىنىپ ،بارلىق ئارخىبى مۇنبەردىن پاكىز يۇيۇپ تازلىنىدۇ! ياكى چەكلەش ئېلىپ بېرىلىدۇ..يەنە بوي سۇنمىغانلارنىڭ IP سى چەكلىنىپ مۇنبىرىمىزنى زىيارەت قىلىشتىن توختىتىمىز، ئۇنىڭغىمۇ بوي سۇنماي داۋاملىق مۇنبەر مۇھىيتىنى قالايمىقان قىلغانلارنى مۇناسىۋەتلىك ئورۇنلارغا تاپشۇرۇپ بېرىمىز!



مىسرانىم نامى بىلەن ھەرقانداق يىغىلىش ياكى چچ توپ قۇرۇش قاتارلىق پائالىيەتلەر بىلەن شۇغۇللىنىشىنى قاتتىق چەكلەيمىز، خەيرى ساخاۋەت ئىشلىرىمۇ مىسرانىم تورىنىڭ رۇخسىتىنى ئالغاندىن كىيىن داۋاملاشتۇرۇلىدۇ!بۇيرۇققا بوي سۇنمىغانلارنىڭ قانۇنىي جاۋابكارلىقى قاتتىق سۈرۈشتە قىلىنىدۇ!



جىدەل چىقارغان، قەستەن شەخىس زىتىغا تېگىدىغان تېما ئىنكاس يازغۇچىلارنىڭ ماتىرياللىرى بىردەك پاكىزلىنىدۇ.مۇنبىرىمىزگە كەسپىي بىلىم، ھېكايە رومان ،تەرجىمە ئەسەر قاتارلىقلارنى يوللاپ تۇرغانلارغا ئالاھىدە ئىمتىيازلار بېرىلىدۇ،ھەم ئۇنۋاق قوشۇپ بېرىلىدۇ، دائىم باشقا توربەتلەردىن تىما ئوغۇرلاپ ئېلىپ كەلگەن تورداشلارغا ئەسكەرتىش بېرىلىدۇ، يەنە ئاڭلىمىسا ئارخىبى پۈتۈنلەي ئۆچۈرۋىتىلىدۇ!مەنبەسى ئەسكەرتىلمىگەن ھەرقانداق تىما ئۆچۈرۋىتىلدۇ،خاتىرەمدىن ،ساقلانمامدىن دېگەندەك تىمىلارمۇ شۇنىڭ ئىچىدە! گەپ ئېغىزدىن چىقماستا گەپنى ئادەم باشقۇرسا،چىقىپ بولغاندىن كىيىن گەپ ئادەمنى باشقۇرىدۇ. .ئەقىل كۆزىمىزنى يوغان ئېچىپ ھۆكۈمەت سىياسىتىنىڭ ئەۋزەللىكلىرىدىن تولۇق پايدىلنىپ ،مۇنبىرىمىزنى بىلىملىكلەر سورۇنى قىلىپ قۇرۇپ چىقىش ئۈچۈن تېرىشىشىمىز كېرەك!




باشقۇرغۇچىلارغا ئەسكەرتىلدىغان بارلىق ئىشلار مۇۋاپىق ئۇسۇللار ئارقىلىق ئەسكەرتىلىپ بولدى! ئەزالارنىڭ پائال ماسلىشىشنى ئۈمۈد قىلىمىز!



سىياسىي ئېڭى پىشىپ يېتىلمىگەن، مۇنبەردە دەرىجە قوغلىشىپ ،قىز ئىزدەپ كىرىدىغان،ئاغزىغا كەلگىننى جۆيلىيدىغان ئەقىلدە كىچىك <مۇنبەر بالا ئەزالىرىمىز> نى ئۇچراتقان ھامان مۇنبىرىمىزدىن پاك پاكىز ئۆچۈرۈپ تاشلايمىز! باشقا ئالاھىدە ئەھۋال بولسا مۇنبەر قىسقا ئۇچۇرى ئارقىلىق بارلىق تورداشلارغا ئەسكەرتىش بېرىمىز!




ئاخىردا تورداشلاردىن ئۆتىندىغىنىمىز ،مۇنبىرىمىزگە يامان غەرەز بىلەن كىرگەن ھەرقانداق ئەزا، يوللانغان ھەرقانداق تىما ھەم ئىنكاسلارنى كۆرگەن ھەر بىر <مىسرانىم سۆيەر> ئەزالىرىمىز بىزگە پائال ماسلىشىپ شۇ ئەزا ھەم تىمىلارنىڭ ۋاقتىدا تازلىنىشى ئۈچۈن ئالاھىدە كۈچ چىقىرىشىنى ئۈمۈد قىلىمىز! شۇنداقلا مېھرى ئىللىق ،ئاتەش قەلىب ئائىلىمىزنى كۆز قارچۇقىمىزنى ئاسرىغاندەك ئاسرىشىڭلارنى يەنە بىر قېتىم سەمىمىي ئۈمۈد قىلىمىز!



ئەسكەرتىش: تىما ۋە ئىنكاسلىرى ئۆچۈرۈلگەن تورداشلار بىزدىن سەۋەبسىزلا تىزلا ئاغرىنمىسا، ئاۋال سەۋەبنى ئۆزىدىن ئىزدەپ ئاندىن باشقۇرغۇچى قوشۇنىدىن مەسىلە تاپسا بولىدۇ..



ئەگەر مۇنبىرىمىزدىن كۆڭلى سۇ ئىچمىگەن، ياكى يۇقارقى تەلەپلەردىن نارازى تورداشلىرىمىزنىڭ زىيارەتنى توختىشىنى ،ئۆزى سەۋەبلىك 8000 كىشىلىك چوڭ ئائىلىنىڭ تىنىچلىقىنى بۇزماسلىقىنى ئۈمۈد قىلىمىز!
