ئابابەكرى قېرىندىشىم ،بۇ ساياھەت خاتىرىڭىز مېنى بۇ مۇنبەرگە تارتىپ كىردى . ناھايىتى ساپ ،چىن ھېسياتىڭىزنى ئىپادىلەۋاتىسىز ،سەمىمىيىتىڭزگە ئاپىرىن . كۈنلىرىڭىزنىڭ يېڭى مەنالارغا تولىشىنى ،بىزلەرنى ھەر ۋاقىت يېڭى ئۇچۇرلاردىن خەۋەردار قىلىپ تۇرىشىڭىزنى ئۈمىد قىلىمەن ،   ئامېرىكىدىكى ئۇيغۇر قېرىنداشلار بىلەن ئۇچىرىشالىغان بولسىڭىز بەكلا ياخشى بولاتتى ،«ۋاشىنگىتۇندا بېشىمدىن ئۆتكەنلەر » نىڭ ئاپتۇرى ئابدۇراھمان ئوغلى بىلەن كۆرۈشەلىگەن بولسىڭىز ، يەنە نۇرغۇن قېرىنداشلىرىمىز بىلەن كۆرۈشەلىگەن بولسىڭىز نىمە دېگەن ياخشى بولاتتى -ھە!
شۇ يىراق ئەللەردە ياشاۋاتقان بەخىت ئىچىدىكى بەخىتسىز قېرىنداشلىرىمىزغا  ئۆز ۋەتىنىدە ياشاپ تۇرۇپمۇ بەخىتنىڭ نىمىلىگىنى ھېس قىلالمايۋاتقان قېرىنداشلارنىڭ چوڭقۇر سالىمىنى يەتكۈزۈپ قويارسىز ،كىچىككىنە تەكلىپىم ، باشقا تورداشلار دېگەن بولسىمۇ دېگۈم كەلدى ،بولسا مىللىي خاسلىقىمىزنىڭ نامايەندىسى بولغان دوپپىمىز ۋە كانىۋاي كۆينىكىمىزنى ۋەتەنگە قايتىپ كەلگۈچە بولسىمۇ ئۇچىڭىزدىن ئايرىمىغان بولسىڭىز .
