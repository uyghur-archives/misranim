                                       كۈتەرمەن سېنى....
                                 __ئەلى تۇرسۇن سىرلىق....



                           ئەزىز دەپ بىلگەنتىىم سېنى جېنىمدىن،
                           كەتمەكچى بوپسەن ئەپسۇس يېنىمدىن.
                             كەتسەڭمۇ گەرچە يىراق مېنىڭدىن،
                            ئىزلەرمەن سېنى،مەڭگۈ سۆيگۈنۈم.
                          
                          يۈرەككە ئوت سېلىپ كەتسەڭمۇ يېراق،
                           ئازاپنى باخاشلاپ چەكسەممۇ پىراق.
                           سەن ئۈچۈن كۈيلىرىم ياڭرايدۇ بىراق،
                           كۈيلەرمەن سېنى،مەڭگۈ سۆيگۈنۈم.

                           ئۆتمەكتە كۈنلىرىم سۆيگۈڭنى كۈتۈپ،
                             تەنھالىق ئىچرە دەردىمنى يۈتۈپ.
                           ئۆتسىمۇ تالاي ئاي،يىللار ھەم ئۆتۈپ،
                           ئەسلەرمەن سېنى،مەڭگۈ سۆيگۈنۈم.

                           سۆيمەكچى بولساڭلا قەلبىڭدىن جانىم،
                             ئاتايمەن مەن ساڭا ھاياتىم بارىم.
                           چىن-سۆيگۈ بەرسەڭلا،پەقەتلا يارىم،
                            كۈتەرمەن سېنى،مەڭگۈ سۆيگۈنۈم.

                           شامۇ-سەھەر يادىمدا سۆيگۈنىمسەن،
                           جاندىن ھەم ئەزىز دەپ بىلگىنىمسەن.
                           قەلىبىم كۆكىدە نۇرلۇق يۇلتۇزۇمسەن،
                            سۆيەرمەن سېنى.مەڭگۈ سۆيگۈنۈم.
               
            ئاپتورى : ئۆزەم 2014-يىلى نۇرۇزنىڭ 23-كۈنى....مۇنبەردىكى بىرسىگە.....!

                       سۆيگۈندىن بۇرۇن ئايىرىلىغان  دوستۇم خاتىرسىگە......   
   
                                               گۈلۈم يوق .....
                                  ـــــ ئەلى تۇرسۇن < سىرلىق >


                                      باھار ئەلچىسى قارلىغاچ،
                                     بولدى كەلمەس گۈلۈم يوق.
                                      تاڭ ساباسى قۇشلار ھەم،
                                       ناۋا قىلماس گۈلۈم يوق.


                                     سۆيگۈم  ئۈچۈن پىنھاندا،
                                       نالە قىلدىم گۈلۈم يوق.
                                       كۆرمەس كىشى نالەمنى،
                                      دەتتە ئۆلدۈم گۈلۈم يوق.


                                      ۋىسال ئۈچۈن ھەرجايدا،
                                     ئازاپ چەكتىم گۈلۈم يوق.
                                      كۈتۈپ ئۇنى سۆيگۈنى،
                                    خاراپ بولدۇم گۈلۈم يوق.


                                      ئۆتۈپ كەتتى كۈنلىرىم،
                                       يوللىرىمدا گۈلۈم يوق.
                                       يالغۇز ئۆتەر تۈنلىرىم،
                                       يانلىرىمدا گۈلۈم يوق.


                                      ئەلچە باركەن ئاڭلىدىم،
                                     يىغلار بولدۇم گۈلۈم يوق.
                                    توي بۇلغىدەك يار بىلسەم،
                                     ئەمدى مەڭگۈ گۈلۈم يوق.


                       ئاپتورى: ئۆزەم 2014-يىلى نۇرۇزنىڭ 19-كۈنى



                                                بەختىڭ ئۈچۈن....!
                                               ئەلى تۇرسۇن سىرلىق

                                         نەپرەتلەنمە مېنىڭدىن گۈلۈم،
                                    بەختىڭ ئۈچۈن سۆيگۈمدىن كەچتىم.
                                      چىن-يۈرەكتىن سۆيگەنتىم سېنى،
                                        قىيالماستىن يېنىڭدىن كەتتىم.

                                         تەقدىرىمگە يېزىلساڭ ئەگەر،
                                          بىر كۈنلەردە ۋىسال تاپارمىز.
                                         پاك-سۆيگۈنىڭ گۈلزارلىقىدا،
                                          تا ئەبەدكە چېچەك ئاچارمىز.

                                      سۆيگۈ ئۈچۈن ئۆرتەنسە يۈرەك،
                                      پاك قەلبىڭدە پەقەت مەن ئۈچۈن.
                                        يېغلىمىغىن بولدى ئەزىزىم،
                                      ئاشىق دىلىم بىرلا سەن ئۈچۈن.

                                     ئۆتمەي تۇرۇپ يىللارمۇ تېخى،
                                     پاك-سۆيگۈمىز بولدىغۇ تامان.
                                        ئۇنتالمايمەن ھاياتتا سېنى،
                                      يۈرەكىمدە مەڭگۈ سەن ھامان.

                      ئاپتورى : ئۆزەم 2014-يىلى نورۇزنىڭ 27-كۈنى....

                            مۇھاببەتلك شېئرلىرىمغا  خاتىمە......!

  مىنىڭچە بولغاندا مۇھەببەت توغۇرلۇق يىزىلغان شېئىرلار سان جەھەتتە ھەغدادىغا يەتتى ، ئەمما سۈپەت ۋە تەسىر كۈچى جەھەتتە تىخى خىلىلا ئاز .
  بىزدە ھازىر تۆت قۇر مۇئەييەن مەنا بىلدۈرەلەيدىغان ، قاپىيەلىك جۈملىلەرنى شىئىر دەپ بىلىش ئاساسىي ئۇقۇمغا ئايلاندى ، ئەمما ھەقىقىي يۈرەكنىڭ تۇمۇرىغا بارىدىغان ، ئابى چەشمە تامچىلىرىدەك ، ئوقۇغان ئادەم مىنى يىزىپتۇ ، مىنىڭ كەچمىشىمكەن دەپ بىلىدىغان شىئىرلار تىخى خىلى ئاز . ھەممىسى مۇھەببەتتىن كىيىنكى قايغۇرۇش ، ئۆچمەنلىك ، ۋە ياكى مۇھەببەتكە بولغان غايىبانە قىزىقىش .
  شېئىر ئابدۇرىھىم ئۆتكۈر ئىيتقاندەك بىر بولاققا ئوخشاش ، دەرتمەنلەرگە داۋا ، ئاجىزلارغا جاسارەت ، مىسكىنلەرگە تەسەللى بىرىشى كىرەك . چۈنكى ئۇ ئەدەبىياتنىڭ ئەڭ يارقىن ، ئەڭ ئىخچام ، ئەڭ پاساھەتلىك بىر تۈرى ، شۇنداقلا ئەدەبىياتنىڭ جىنى دىسەكمۇ مىنىڭچە ئارتۇق كەتمەس . بىر نەچچە بەت ماقالە بىلەن ئەكىس ئەتتۈرەلمىگەن مەزمۇننى بىر ئىككى كوبلىت شېئىر بىلەن ئەكىس ئەتتۈرۈشكە ، شۇنداقلا تىخىمۇ روشەن ، يارقىن ئىپادىلەشكە تامامەن بولىدۇ ، چۈنكى شىئىر مەيلى ئۇسلۇپ ۋە ياكى ھەجىم جەھەتتىن بولسۇن ئوقۇغان كىشىگە ھوزۇر بىرەلەيدۇ . شۇڭا بەزى شىئىرلار مەنسىنىڭ چوڭقۇرلۇقى ، ئىپادىلەش كۈچىنىڭ يارقىنلىقى بىلەن ئاھاڭغا ئىلىنىپ ناخشا بولىدۇ ، ھەم شۇنداقلا ياخشى ئاھاڭغا چۈشۈپ خەلق قەلبىدىن ئورۇن ئالالىسا خەلق ناخشىسىمۇ بولالايدۇ .
  مىنىڭ تەكلىپىم : بولسا مۇھەببەتلىك شىئىرلار ئەمدى ساننى جەھەتتىن ئەمەس بەلكى سۈپەت جەھەتتىن ، ئىپادىلەش ، مەزمۇن ، قاپىيە ، تۇراق جەھەتتىن تىخىمۇ مۇكەمملىشىپ ، كەچكىچە ئادەمگە ئۈمىدسىزلىك ، ۋە ياكى دەرت ئەلەمنى ھىس قىلدۇرماي ، ئۈمىدۋارلىقنى ، خوشاللىقنى بەخش ئەتسە ، شۇنداقلا كەچكىچە ئاشىق مەشۇق ھىكايىلىرى ئەمەس بەلكى باشقا تەلىم تەربىيە ۋە ياكى رىغبەت ، مەدەت خاراكتىردىكى بىر يۈرۈش يىڭى مەزمۇندىكى شېئىرلار كۆپلەپ يىزىلسا . يەنە بىرسى ۋەتەن ، ئاتا-ئانا ، مۇھەببەت ، ئىلىم قاتارلىق ئۆلمەس تېمىلار تىخىمۇ جۇلالىنىپ شېئىرىيەتكە ھەقىقىي ھۆسىن قوشسا بەلكىم يىقىن كەلگۈسىدە سۈپەت ئۆلچىمىگە تامامەن يىتىپ ، ئۇيغۇر ئەدەبىياتى يەنە بىر بالداق يوقىرى كۆتۈرۈلگەن بولاتتى ...
