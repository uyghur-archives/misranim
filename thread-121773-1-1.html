  مەن دەردىمنى نە دوستۇمغا دېيەلمەيمەن،نە دۈشمىنىمگە.چۈنكى دوستۇمنىڭ كۆڭلى يېرىم بولىدۇ،دۈشمىنىم خۇرسەن بولىدۇ،ھالىمنى ئەڭ ياخشى رەببىم ئاللاھ بىلىدۇ.شۇنداق بىر ئادەم بولىمەنكى سەن تونۇيالمىغۇدەك،ئاللاھ مەندىن رازى بولغۇدەك،كىشىلەرنىڭ بەزىلىرى غەيۋەتكە چۈشۈپ كەتسە بەزىلىرى سۆيۈنۈپ كەتكۈدەك....!

  بەخت نەدە بىلەمسىز؟كۈلگەن كۆزلەردە،تاتلىق سۆزلەردە،بەختلىك يۈزلەردە،سۆيگەن لەۋلەردە،بەزەن كۈندۈزلەردە،بەزەن كېچىلەردە،ئەمما ئەڭ گۈزىلى ئىناق-ئىتتىپاق ئۆيلەردە،سۆيگۈ تولغان يۈرەكلەردە،پەرزەزەنتىڭنىڭ كۈلكىسىدە،ئاتا-ئانىنىڭ دۇئاسىدا،ئاللاھقا تەلپۈنگەن قەلىپتە.........!   


بەزىلەرگە پەش قويدۇم يەنە نېمىلەرنى قىلاركىن دەپ،بەزىلەرگە چېكىت قويدۇم مۇشۇ يەرگىچە بولسۇن دەپ،بەزىلەرگە سوراق قويدۇم ئەسلىدىنلا شۇنداقمىدىڭ دەپ،بەزىلەرگە ئۈندەش قويدۇم ھوشىنى تېپىۋالسۇن دەپ......!

بىر قېتىم كۆڭۈل ئاغىرىتماق كەئبىنى ئاستىن-ئۈستۈن قىلغاندىنمۇ ياماندۇر،چۈنكى كەئبىنى ھەزرىتى ئىبىراھىم قۇرغان،كۆڭۈلنى ئاللاھ ياراتقاندۇر......!

  ئاللاھىم،ماڭا مەن ئۆزگەرتەلمەيدىغان ئىشلارنى قۇبۇل قىلالىغۇدەك ئىشەنچ،مەن ئۆزگەرتەلەيدىغان ئىشلارنى ئۆزگەرتكۈدەك جاسارەت،بۇ ئىككىسىنىڭ ئوتتۇرىسىدىكى پەرىقنى بىلەلىگۈدەك ئەقىل بەر......!

ھاياتىمدا شۇنداق گۈزەل ئارزۇ-ئارمانلارنى نىيەت قىلىپ قوينىڭ قوزىسىدەك ياشىغۇم بار ئىدى ئەمما ھاياتىمدا،مەن ياشاۋاتقان دىيارىمدا تۈلكىدەك ھىيلىگەر ئادملەر سەۋەبىدىن يولۋاستەك ياشاشقا مەجبۇر بولىۋاتىمەن....!

ئەي ناماز،نەقەدەر بۈيۈك نىئمەتسەن قىيام سەندە،قىرائەت سەندە،رۇكۇ سەندە،سەجدە سەندە،ئىشىق سەندە،ۋىسال سەندە،ھۇزۇر سەندە،قەلبىم سەنسىز خۇش بولماس،گۇناھلىرىم سەنسىز يۇيۇلماس......يا رەببىم ئاللاھ...!

  بىر ئادەمنىڭ كۆڭلىنى ئۇتماق تولىمۇ تەس،ئەتراپىڭدىكىلەرنىڭ،ساڭا كۆيگەنلەرنىڭ،ساڭا كۆيۈنىدىغانلارنىڭ يۈرۈكىنى ئەزمىگىن،كۆڭلىنى رەنجىتمىگىن.كونىلار(ئالتۇن بېرىپمۇ ئالغىلى بولماس قالغان كۆڭۈلنى)دەپ توغرا ئېيتقان ئەمەسمۇ؟؟!ئۇلارنىڭ بېشىغا قىسىدىغان گۈل بولغىن،يارا قەلبىنى ساقايتىدىغان مەلھەم دورا بولغىنكى زېدە قەلىپلەرنى تاتىلايدىغان زالىم بولما.....!

  بىز ئاجايىپ دۇنيادا ياشاۋاتىمىز،كەمبەغەللەر بىر بۇردا نان ئۈچۈن نەچچە چاقىرىملاپ يول مېڭىشقا مەجبۇر بولسا،بايلار يەنەشۇنچىلىك يول مېڭىپ ئارتۇق يەۋالغان تامىقىنى سىڭدۈرۈشكە مەجبۇر.....!

بەزىدە سۇ بولۇش لازىم،خۇددى سۇ كەبى سۆزسىز ۋە تىنىچ؛بەزىدە كەلكۈن بولۇش لازىم،خۇددى كەلكۈن كەبى ئەسەبى ۋە قىزىققان؛بەزىدە شام ئوتى بولۇش لازىم،خۇددى شام ئوتى كەبى سەبىر بىلەن كۆيۈپ تۈگەيدىغان؛بەزىدە ۋولقان بولۇش لازىمكى،ئۇنىڭ ئالدىغا كەلگەن ھەر نەرسىنى تېزلا يوقۇتىدىغان؛كىمگە سۇ،كىمگە كەلكۈن،كىمگە شام،كىمگە ۋولقان بول ئەمما ئاللاھتىن باشقا ھېچكىمگگ قۇل بولما.....!

  دوستۇم:
كۆزىڭىز ئىككى بولغاندىن كېيىن ھەممە كىشىلەرگە تەكشى قاراشنى ئۆگۈنىۋېلىڭ،قۇلىقىڭىز ئىككى تەرەپتە بولغاندىن كېيىن بىر تەرەپلىمە سۆزلەرگىلا قۇلاق سېلىپ يۈرمەڭ،بۇرنىڭىزنىڭ تۆشۈكى ئىككى بولغاندىن كېيىن ھەرگىز باشقىلار بىلەن بىر بۇرۇندا نەپەس ئالماڭ،تىلىڭىز بىر بولغاندىن كېيىن ھەرگىز ئىككى خىل گەپ قىلماڭ،يۈرىكىڭىز بىر بولسىمۇ ئۇنىڭ دالانچىسى ئىككى بولغاندىن كېيىن ھەممە ئىشتا ئۆزىڭىزنىلا ئەمەس باشقىلارنىمۇ ئويلاڭ.....!

  كۆڭۈل ھەقىقەتەنمۇ نازۇك بىر نەرسە،سەن مەيلى چاقچاق قىل مەيلى راستى دېگىن سەن ئۈچۈن پۈتۈن كۆڭلىنى ئاتاپ پەقەت سەن ئۈچۈنلا خىزمەت قىلىۋاتقان شۇ كۆڭۈلنى يەرگە ئۇرمىغىن،ئېسىڭدە بولسۇن:
قەلبىمنىڭ ئىشىكىنى چەكسەڭ مەيلى بىراق ئېھتىيات قىل،غۇرۇرۇمنىڭ دېزىسىنى چېكىپ سالما.....!

  مۇھاببەت بەدەل بىلەن بىلەن گۈزەل ئەمەس قەدىرلەش بىلەن گۈزەل!!!...سېنى سۆيگەننىڭ قەدرىنى بىل،قەدرىنى قىل!ئۆز ھاياتىڭنىمۇ ھەم ياخشى قەدىرلە،چۈنكى بۈگۈن پاتقان قۇياش ئەتىسى چىققىنى بىلەن بۈگۈن قەبرىگە  پاتقان باش ئەتىسى چىقمايدۇ....!

  ئۈچ ئادەمنىڭ كۆز يېشىدىن ھەزەر ئەيلەڭ:
ئانىڭىزنىڭ كۆز يېشىدىن،يېتىمنىڭ كۆز يېشىدىن،ئايالىڭىزنىڭ كۆز يېشىدىن!!!چۈنكى بۇلار كۆزى بىلەن ئەمەس يۈرىكى بىلەن يىغلايدۇ،ياش ئەمەس قان يىغلايدۇ.....!!!

  ھەقىقى دوستۇڭ سەن بار يەردە ساڭا ئاللاھنىڭ گېپىنى قىلىدۇ،سەن يوق يەردە سېنىڭ گېپىڭنى ئاللاھقا قىلىدۇ،ھەقىقى دوستلۇق قول بىلەن كۆزگە ئوخشايدۇ،قول يارىلانسا كۆز ياش تۆكىدۇ،كۆز يىغلىسا قول سۈرتىدۇ.....!

  ھەممە ئادەمنى رازى قىلىمەن دەپ ئالدىراپ يۈرۈپ ئاخىرى كارغا كەلمەس ئادەمگە ئايلىنىپ قالساڭ كىشىلەر سېنى كۆزگە ئىلمايدۇكى،ھەرگىزمۇ بىز ئۈچۈن تۆلىگەن بەدەل دەپ ئويلىمايدۇ......!

بىر قېتىم دىلىڭنى رەنجىتكەن ئادەمگە قايتىدىن ئۈمىد بىلەن قارىما،چۈنكى ئۇنداقلار سېنى بۈگۈنمۇ،ئەتىمۇ،ھەتتا بىر ئۆمۈر ئۈمىدسىز قىلىۋېتىدۇ.ئەڭ مۇھىم بولغىنى سېنىڭ تۇرمۇشقا بولغان ئىشەنچەڭنى ھەم ئۈمۈدىڭنى خۇددى قاقشال دەرەخكە ئوخشاش قۇرۇتىۋېتىدۇ.......!

  ئايال كىشىنىڭ قەلبى بازارغا ئوخشايدۇ،كىرگەن كىشى چىقمايدۇ،ئەر كىشىنىڭ قەلبى بازارغا ئوخشايدۇ،كىرگەن چىققاننىڭ سانى يوق.....!

  يۈرەك يىغلايدۇ نېمىشقا دەپ،ئەقىل كۈلىدۇ خوپ بولدى دەپ؛چۈنكى ئادەم قانچىكى كۆڭۈلچەك بولسا زىممىسىدىكى يۈك شۇنچە ئېغىر بولىدۇ،كۆڭۈل دېگەن ئاجايىپ نەرسە ھە.....بىر ئېغىز سۆز بىلەن ئاسمانغا چىقىپ خان بولىدىغان.....بىر ئېغىز سۆز بىلەن يەرگە چۈشۈپ لاي بولىدىغان......شۇڭا بەزى گەپلەر كۆڭۈلدە قالغىنى ياخشى،بەزى ئازاپلار ئۈن-تنسىز ئۇنتۇلۇپ كەتكىنى ياخشى،ھاياتلىق مانا مۇشۇنداق.....
  بەزى ئىشلارنى تولا ئويلىسىڭىز بېشىڭىز ئاغىرىيدۇ،ئويلاپ تېگىگە يەتسىڭىز يۈرىكىڭىز ئاغىرىيدۇ،سىزنىڭ توغرا بولغاندا ھېچكىمنىڭ ئېسىدە قالمايدۇ،بىراق خاتالاشسىڭىز ھېچكىم ئۇنتۇپ قالمايدۇ،ھايات يولى ئىككىلىنىشكە ئەمەس بەلكى كەسكىنلىككە مۇھتاج،چۈنكى:
تەۋرىنىپ تۇرغان روھتا مۇكەممەللىك بولمايدۇ......!
