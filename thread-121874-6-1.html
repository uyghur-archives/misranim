     ئەسسالامۇ ئالەيكۇم تور دۇنياسىدىكى مىللى مارىپىمىزغا كوڭۇل بولەۋاتقان ئەزىزلەر ،  سىلەرنىڭ مۇشۇ توربەتتە مۇشۇ تىمىغا ئىنكاس يازغىنىڭلار سىلەرنىڭ قانچىلىك مارىپىمىزغا كوڭۇل بولەۋاتقانلىغىڭلارنى بىلدۇرىدۇ .  دىمىسىمۇ دولىتىمىزنىڭ ھازىرقى مارىپى . كونا ئۇسۇلدىن   ھالقىپ كىتەلمەتىدۇ . ھالقىپ يىڭىلىققا كوچۇشمۇ بىر جەرياننى بىشىدىن چوقۇم ئوتكۇزىشى كىرەك ،چىچىنى تورۇسقا باغلاپ قويۇپ ئۇگۇنۇش كوڭزى مىڭزىدىن قالغان ئەنئەنە بولغاچقا بۇنى تەلتوكوس يۇقۇتۇش ئۇنچە ئاسان ئىش ئەمەس . ئامىرىكا . كاپىتالىستىك تەرەقى قىلغان دولەت بولغاچقا ،ئۇلارنىڭ مائارىپ تۇزۇلمىسى بىلەن بىزنىڭ مائارىپ تۇزۇلمىمىز ئوتتۇرسىدا پەرىق ئىنتايىن كوپ .بىز تەرەققى قىلىۋاتقان دولەتتە ياشاۋاتىمىز ئامىرىكا تەرەققى قىلىپ بولغان . ئوزىنى تىخىمۇ مۇكەممەللەشتۇرۇۋاتقان . يۇكسىلىۋاتقان دولەت . ئۇلارنىڭ  گ د پ سى 1 % ئاشسىمۇ . بىزنىڭ 20 % ئاشقىنىمىزدىن يەنىلا يۇقۇرى تۇرىدۇ ،ئۇلارنىڭ ساپا مارىپىغا يىلدا سالىدىغان مەبلىغى  ،ھەربىر ئوقۇغۇچىغا 20 مىڭ دوللار ئەتراپىدا بولاۋاتىدۇ . بىزنىڭچۇ ،بىز تىخى  خەتەرلىك بىنالارنى ئوزگەرتىپ بولالمىدۇق . ساپا مارىپى تەلىۋى بويىچە دەرىس ئوتەڭلەر دەپ ئوقۇتقۇچىلارغا تەلەپنى قاتتىق قويىمىز ، دەرىسلىك پىراگىراممىلارنى ساپا مارىپى تەلىۋى بويىچە تۇزۇپمۇ قويدۇق .ئەپسۇس شۇ ساپا مارىپى تەلىۋى بويىچە دەرىس سوزلىيەلەيدىغان  ئوقۇتقۇچىلار قۇشۇنىنى تەربىلىمىدۇق . ساپا مارىپى بويىچە سىلىنمىنى ئاشۇرمىدۇق .شۇنىڭ بىلەن ، قانداق ئاقىۋەت كىلىپ چىقتى .ئوقۇغۇچىلاردا دەرىسكە قىزىقماسلىق ئوقۇي دىسە ئوقۇيدىغان يەرنى تاپالماسلىق .ئوقۇتقۇچى دەرىسنى بىلگىنىچە سوزلەۋىرىش . تاپشۇرۇقلار ھەددىدىن زىيادە كوپ بولۇش . داشۇ ئىمتانلىرى يەنىلا شۇ بۇرۇنقى سەنەمدە . يادلاشنى ئاساس قىلىش .   ھازىر  دولىتىمىز مارىپ سىلىنمىسىنى يىلدىن يىلغا ئاشۇرىۋاتىدۇ .خەتەتلىك قۇرلۇشلارنى ئوزگەرتىش ،  يىزا مارىپ سىلىنمىسىنى شەھەر بىلەن ئوخشاش قىلىش ، يىزا  قىشلاقلاردىكى ئوقۇغۇچىلارنى شەھەرلىك ئوقۇغۇچىلارغا ئوخشاش ئازادە ئىسساق مۇھىت بىلەن تەمىن ئىتىش . ساپا مارىپى ئۇچۇن مەبلەغ سىلىش قاتارلىقلار .ئىمتىھان مارىپىنى تەدرىجى ئەمەلدىن قالدۇرۇش . ئوقۇغۇچىلارنىڭ يۇقۇرى ئورلەپ ئوقۇش نىسبىتىنى ئاشۇرۇش ئۇچۇن بىر يىل ئىچىدە كوپ قىتىم ئالى مەكتەپكە ئىمتىھان بىرىشنى ئىشقا ئاشۇرۇش قاتارلىقلار . بىز ئۇيغۇرلار . قوش تىل مارىپىغا دۇچ كەلدۇق . .. ئەمىلىيەتتە مارىپتا ئارقىدا قىلىشىمىز . يەنى ئۇيغۇر ئوقۇغۇچىلارنىڭ ئالى مەكتەپ ئىمتىھانىدا خەنزۇ ئوقۇغۇچىلاردەك يۇقۇرى نۇمۇر ئالالماسلىغىنى . تىلىمىز بىلەن مۇناسىۋەتلىك بولۇشى مۇمكىن دەپ قارىدۇق .  ئەمىلىيەتتە ئۇنداق ئەمەس . بەلكى بىزدە  ئوقۇتقۇچىلار قوشۇنى سەرخىللاشمىدى . ئوقۇتقۇچىلار  مەمۇرى ئورگانلارغا يۇتكىلىپ تۇرسا . يىڭى داشۇ پۇتۇرگەن ياش ئوقۇتقۇچىلار ئورنىغا سەپلىنىپ تۇرسا .بەلكىم ئۇيغۇر ئوقۇغۇچىلارنىڭ ساپاسى تىخىمۇ ئوسكەن . خەنزۇ ئوقۇغۇچىلاردەك يۇقۇرى نۇمۇرغا ئىگە بولغان بولاتتى دەپ قارايمەن .  بىزنىڭ نەتىجىمىزنىڭ توۋەنلىشىدىكى يەنە بىر سەۋەپ . بىزگە بىرىۋاتقان ئىتىۋار   سىياسەتتۇر . ماتىماتىكىدىكى  توۋەن چەك . مىنكوخەنلەرگە قوشۇپ بىرىدىغان 50 نۇمۇر قاتارلىقلار .  ئەينى ۋاقىتتا ئامىرىكىدىكى نىگىرلار . مۇشۇنداق ئىتىبار سىياسەتلەرگە  قارشى نامايىش قىپ تىكەن ،شۇڭا ھازىر ئامىرىكىدىكى . يۇقۇرى قاتلامدىكى باشلىقلارنىڭ كوپ قىسمى نىگىرلاردۇر . بىز تىرىشىمىز . مەدىنىيەتلىك ساپالىق 21- ئەسىرگە لايىق ئۇيغۇر بولۇش ئۇچۇن باشقا مىللەتلەرنىڭ ئارقىسىدا قالماسلىغىمىز ئۇچۇن ئۇيغۇر مارىپى ئۇچۇن پۇتۇن كۇچىمىز بىلەن تىرىشچانلىك كورسىتىشىمىزگە توغرا كىلىدۇ
