بولسا ماۋۇ تېمىنى تەڭ ھەممە يەردە تەشۋىق قىلىپ نۇمۇسسىزلارنى سازايى قىلمايلىمۇ؟
ئۆتكەندە ئوكيان تورىدا قايتا قايتا ئاگاھلاندۇرۇش بەرسىمۇ ئەلكۈيى تورى ئوكياننىڭ خەۋەرلىرىنى ئۆزىنىڭكى دەپ ئېلان قىلىۋەردى، ئەلكۈيى مىكرو بلوگىدا بۇ توغۇرلۇق گەپ قىلسام ‹خادىملىرىمىزنىڭ خاتالىقى› دەپ تۇرىۋالدى، خادىملىرىغا گېپى ئۆتمەيدىغان قانداق ‹شىركەت›كەن ئۇ؟
بۇلارنى سۈرۈشتە قىلسا ئەخلاققا قانۇنغا چېتىلىدۇ، بەكرەك سۈرۈشتە قىلسا ئىماننىڭ يوقلىقىغا بېرىپ تاقىلىدۇ، ئەسەر يازغان تەرجىمە قىلغان ئاپتور تەرجىمانلارغا بىر پىيالە چاي قۇيۇپ بەرگەندىكىدەك يوغانچىلىق قىلىشلىرى بار تېخى.
يەنە كېلىپ كۆپىنچە بېكەت قۇرغۇچىلار، سالون قۇرغۇچىلار تېخى تولۇقسىزنىمۇ پۈتۈرمەي جەمىيەتكە چىقىپ كەتكەن لۈكچەك ساياق شۇملار، ئۇلاردا نە ساپا، نە ئەخلاق.  ئۆتكەندە ئاق بۈركۈت خەتلىك فىلىم تورىنىڭ ھەقلىق فىلىملىرىنى ئۆزىچىلا بېكەتلىرىگە يوللىغىلى تۇرۇپتۇ، چىرايلىق گەپ قىلساق ئاڭلىمىدى، شۇنىڭ بىلەن خاككىر باللا ئارقا بېتىگە كىرىپ ۋاللەي ئويناپ قويىۋىدى، ئاكا دېگىنىچە ھازىرغىچە بىرەر فىلىمنى ئوغۇرلاشقا پېتىنالماي يۈرىدۇ، شۇنىڭ بىلەن دەپ قالدىم، قانۇن ئەخلاقنى قوغدىيالمايۋاتقان جەمىيەتتە بەزىدە ناتوغرا ئۇسۇللارنىمۇ ئاز تولا قوللىنىپ قويۇش زۆرۈركەن.

توردا پۇل تاپامسىز؟   مۇڭ كۈيلەر ۋە كىنولار  خىسلەت مۇنبىرى

مەنمۇ ئۈچ يىل بۇرۇن كۈچۈرمىكەشلەرگە رەددىيە بېرىپ بىر يازما يازغان ئىدىم. https://uyghur-archive.com/misranim/thread-7436-1-1.html
بۇ يەرگە چاپلاپ قۇياي.

         ھەر قانداق بىر ئەسەر مەيلى ئۇ شېئىر ياكى ئەدەبىياتنىڭ باشقا ژانىرى بولسۇن ئاپتۇرنىڭ جاپالىق مېھنىتى ، ئىزدىنىشى ، تەپەككۇر قىلىشى ، قان تەرى بەدىلى بىلەن ۋۇجۇتقا چىقىدۇ . باشقىلارنىڭ ئەجىر تۆھپىسىنى ئۆزىنىڭ قىلىۋېلىش ئەڭ چوڭ نۇمۇسسىزلىق ۋە پەسكەشلىكتىن باشقا نەرسە ئەمەس .
         بۈگۈن كېچە بۇلىشىغا قارىماي مۇنبەردىكى تېمىلارنى كۈرۈۋېتىپ ئىنتىزار دىگەن بىرسى يوللىغان  << ياشايمەن دۇنيادا سېنى دەپ ئانا >> دىگەن شېئىرغا كۆزۈم چۈشتى . قىزىقىش بىلەن شېئىرنى ئوقۇپ چىقتىم . ئوقۇۋېتىپ ، مۇھەممەتجان راشىدىن ئاكا مۇنبەرگە ئەزا بولغان ئوخشايدۇ ، ئەمدى ئەركىن پىكىر ئالماشتۇرالايدىغان بوپتىمىز دەپ خوش بۇلۇپ كېتىپتىمەن . كىم بىلسۇن مەنبە دىگەن يەردىكى << ئۆز قەلىمىم >> دىگەن خەتنى كۆرۈپ كۈزۈمگە ئىشەنمەيلا قالدىم ، قايسى قۇيۇندا قالغان ،  كاللىسىغا كېسەك چۈشۈپ كەتكەننىڭ قىلىقىدۇ بۇ ؟ دەپ ئاچچىقىمنى باسالماي تاس قاپتىمەن كومپىيوتىرنى چېقىۋەتكىلى .
        بۇ شېئىرنى سەنەۋەر تۇرسۇن 90 - يىللاردا ناخشا قىلىپ ئېيتىپ خەلقنىڭ زور ئالقىشىغا ئېرىشكەن ، مۇشۇنداق زور ئالقىشقا ئېرىشكەن ناخشىنىڭ تېكىستىنىڭ ، ئاپتۇرىنىڭ كىشىنىڭ قەلبىدىن چىقىپ كېتىشى مۈمكىنمۇ؟ مەنمۇ ئۆز ھەۋىسىم بىلەن ناخشا تۇنجى قېتىم تېلۋۇزۇردا بەرگەن چاغدىلا ئۇئالغۇغا كۈچۈرىۋېلىپ ئېيتىشنى ئۈگەنگەن تۇرسام ، قانداقمۇ ئېسىمدىن چىقىپ كەتكۈدەك ؟
        2005 - يىلى 8 - ئاينىڭ 31 - كۈنى ئىدارىدىن بىزنى غۇلجىغا ساياھەتكە ئېلىپ بارغاندا سورۇنىمىزغا ئاتايىن مۇھەممەتجان ئاكا بىلەن ھېسام ئاكىنى تەكلىپ قىلغان ۋە قىزغىن موڭداشقان ئىدۇق . شۇ سورۇندا بىزنىڭ بىر خىزمەتچىمىز مۇشۇ ناخشىنى ئېيتىپ مۇھەممەتجان ئاكا بىلەن ھېسام ئاكىنىڭ تەبرىك سۆزىگە ئرىشكەن ، مەن ھېسام ئاكا بىلەن بىر نەچچە قەدەھ سوقۇشتۇرغان ، مۇھەممەتجان ئاكادىن ئەدەبىيات توغرىلىق تەلىم ئالغان ئىدىم . شۇ كۈنلەر مەڭگۇ ئېسىمدىن چىقمايدۇ .
         مانا قۇلۇمدا مۇھەممەتجان راشىدىن ئاكىنىڭ 2006 - 2 - ئايدا نەشىر قىلىنغان << ياخشىلارغا ناخشىلار تولا >> دىگەن شېئىرلار توپلىمى ، ( بۇ توپلامغا شائىر يازغان ، باشقىلار ناىشا قىلىپ ئېيتقان شېئىرلىرى كىرگۈزۈلگەن ) توپلامنىڭ 4 - 5 بېتىگە چىقىرىلغان 6 كۇبلىتلىق << قانداق قىلارمەن >> دىگەن شېئىرىنى كۈرۈپ ئولتۇرۇپتىمەن . خۇددى شېئىرلارنىڭ ئىچىدىن مۇلايىملىغى چىقىپ تۇرغان بىر چىراي << قايسى ئوغرى ، مۆرتەد سېنىڭ ماۋزويىڭنى ئۆزگەرتىپ ، قۇيرۇقلىرىڭنى قىرقىۋەتكەندۇ ؟>> دەپ ئۆتكۈر نەزىرى بىلەن تىكىلگەندەك بۇلۇپ بەتتام تائام يەۋالغاندەك شۈركىنىپ كەتتىم .  
         ئىنتىزار تېمىنىڭ مەنبە دىگەن يېرىگە << ئۆز قەلىمىم >> دەپ ، يازغۇچە قوللىرى تىترىمىگەنمىدۇ، ئىشەكنىڭ تېرىسىدەك قېلىن بۇلۇپ كەتكەن يۈزى قىزارمىغانمىدۇ؟ دەيمەن .
        ئىنتىزار سىزدىن سوراپ باقاي ! سىز قانچە ياشقا كىردىڭىز ؟ تېخى 30 ياشقا بارمىغانسىز ؟  مۇشۇ شېئىر دۇنياغا كەلگەن چاغدا بەلكىم سىز قەيەرلەردىدۇ ئېغى ئۇچۇق ئىشتان كىيىپ ، ماڭقىڭىزنى سىڭايان تارتىپ ئويناپ يۈرگەن بولغىيتىڭىز؟ ( ئىشەنمىسىڭىز كۆزىڭىزنى لاخشىگىردا ئىككى تەرەپكە يوغان كېرىپ ، توپلامدىكى شېئىرنىڭ ئارقىسىدىكى يېزىلغان 1989 - يىل . 25 - مارت دىگەن ۋاقىتقا قاراڭ >>  تېخى بۇ شېئىرغا قارىتا تورداشلارنىڭ ئىنكاسىغا ئۇيالماي ، قىزارماي تۇرۇپ << بۈگۈن مەن 4 شېئىر يوللىدىم ، ھەممىسى خاتىرەمدىن ئېلىنغان ، مەن پەقەت ئۆز شېئىرلىرىمنىلا خاتىرىگە يازاتتىم ، ھەم ئاتاغلىق شائىرلارنىڭكىنى كۈچۈرىدىغان يۈرەك يوق >> ئۆزىڭىزچە چۈشەندۈرۈپ كېتىپسىز ، ( بۇ گېپىڭىزنى ئاڭلاپ ، نېرۋىسىنىڭ بىرەر ۋېنتىسى كاممىكىن دەپ ئويلاپ قالدىم ) . سىز(( ئەلنىڭ كۆزى ئەللىك )) دىگەن ھىكمەتنى ئاڭلىمىغانمىتىڭىز ؟ ئەجەبا خاتىرىگە كۈچۈرۈۋالسىڭىزلا سىزنىڭ بوپ قالامدىكەن ؟  بىڭسىڭىز بولسا نەجىپ مەھبۇزنىڭ << ئۈچ كوچا تىرىلوگىيىسى >> دىگەن ئەسىرىنى كۈچۈرۈپلا ئىككىنچى قېتىم نوبىل مۇكاپاتى ئېلىۋالمامسىز ؟ مۇشۇ قىلىقىڭىزنى ئەسەر ئاپتۇرى كۆرسە نىمە دەر ؟ بۇ قىلىقىڭىز ئەسەرنىڭ ئەسلى ئاپتۇرىغا ، شۇنداقلا ئەدەبىياتىمىزغا قىلىنغان ئەڭ چوڭ ھاقارەت .
       سىزگە يەنە شۇنىمۇ ئەسكەرتىپ قوياي : مۇنبەرگە نۇرغۇن ئادەم كىرىدۇ ، بۇ قىلىقىڭىزنى بىرسى بايقىمىسا يەنە بىرسى چوقۇم بايقايدۇ . ئەڭ ياخشىسى بۇنداق كەمپۈت بەرسە ئالدىنىدىغان كىچىك بالىنىڭ ئويۇنىنى ئاستا يىغىشتۇرۇۋېلىڭ . بۇنداق <<<<<<<<<< نادىر >>>>>>>>>> ئەسىرىڭىزنىڭ ئابرۇيىنى مۇنبەردە پاخاللاشتۇرغىچە ئەڭ ياخشىسى <<<<<< داڭلىق >>>>>> مەتبۇئاتلارغا ئەۋەتىڭ . سىز يوللىغان باشقا شېئىرلارنىمۇ كۆردۈم ، بۇ شېئىرلارنىمۇ خۇددى قەيەرلەردە كۆرگەندەكلا قىلىمەن ، لېكىن قۇلۇمدا پاكىت بولمىغاچقا بۇ ھەقتە ئېغىز ئاچماي تۇراي ، تېما يازالمىسىڭىز ئىنكاس يېزىپ باشقىلارنىڭ تېمىسىدىن ھۇزۇرلىنىڭ ، << يېزىقچىلىق بىلىملىرى >>  دىگەن كىتاپنى ئەستايىدىل كۈرۈپ تۇرمۇشنى كۈزۈتۈڭ ، تەپەككۇر قىلىڭ . شۇنداق قىلالىسىڭىزلا كۈنلەرنىڭ بىرىدە ياخشى ئەسەرلەرنى يېزىپ چىقالىشىڭىز مۈمكىن . قىلىدىغانغا ئىش تاپالماي قالغان بولسىڭىز بېرىپ گاگا ئويناڭ ، ۋاللەي ئويناڭ !!!  ئېسىڭىزدە بولسۇن ! (( مېھنەتسىز تائام ھەر ۋاقىت ئېغىزىڭىزنى كۆيدۈرىدۇ )) .
        ھۆرمەتلىك تورداشلار ھەممىمىز بۇنداق كۈچۈرمىكەشلىكتىن ئىبرەت ئالايلى !!
