يېقىنقى يىللاردىن بۇيان كىشىلەرنىڭ تۇرمۇش سەۋىيەسىنىڭ ئېشىشىغا ئەگىشىپ، ماشىنا ھەيدىغانلارنىڭ سانى كۈنسايىن كۆپەيدى. ھەر كۈنى خىزمەتكە ماشىنا ھەيدەپ بېرىپ – كېلىش ياكى دائىم ماشىنا ھەيدەش ئەرلەر تۇرمۇشىنىڭ مۇھىم تەركىبىي قىسمىغا ئايلاندى. لېكىن، كۆپ ساندىكى ئەرلەر دائىم ماشىنا ھەيدىگەنلىكى تۈپەيلىدىن ئۆزلىرىدە مەلۇم بىر خىل ئەرلەر كېسىلىنىڭ مىسقال – مىسقالدىن پەيدا بولۇۋاتقانلىقىنى خىيالىغا كەلتۈرمەيدۇ. بۇ دەل تۇغماسلىق كېسىلىدۇر. ئەڭ يېڭى تەكشۈرۈش نەتىجىسىدىن مەلۇم بولۇشىچە، ماشىنا ھەيدەيدىغان ئەرلەر ئاسانلا تۇغماسلىق كېسىلىگە گىرىپتار بولىدىكەن. تەتقىقاتچىلارنىڭ بايقىشىچە، ماشىنا ھەيدىگەندە ئۇرۇقدان خالتىسىنىڭ تېمپېراتۇرىسى ئۆرلەيدىكەن، بۇ ھارارەت ئىسپېرمىغا تەسىر كۆرسىتىدىكەن.

مۇتەخەسسىسلەرنىڭ تەھلىل قىلىشىچە، ئەرلەرنىڭ كۆپىيىش ئەزالىرى مۇھىتنىڭ تەسىرىگە ناھايىتى ئاسان ئۇچرايدىكەن. بولۇپمۇ ئەرلەر ئۇزاق ۋاقىت ئولتۇرغان، ئىسسىق مۇھىتتا خىزمەت قىلغاندا ئۇرۇقدان خالتىسىنىڭ تېمپېراتۇرىسى ئۆرلەيدىكەن، بۇ يۇقىرى ھارارەت ئىسپېرمىغا تەسىر كۆرسىتىدىكەن، بۇنىڭ بىلەن ئىسپېرما مىقدارى ئازىيىپ، غەيرىي شەكىللىك ئۇرۇقنىڭ نىسبىتى كۆپىيىپ، تۇغماسلىق كېلىپ چىقىدىكەن. مۇتلەق كۆپ ساندىكى ئەرلەرنىڭ تۇغماسلىقىنى مۇھىت كەلتۈرۈپ چىقىرىدىكەن، ئەمما قارىماققا بۇ ئانچە بايقالمايدىكەن، پەقەت ئىسپېرمىنى تەكشۈرگەندىلا بۇ خىل ئەھۋالنى بىلگىلى بولمايدىكەن.

   ئەرلەردىكى ئۇزاق ماشىنا ھەيدەشتىن كېلىپ چىقىدىغان تۇغماسلىقنى تۆۋەندىكى بىرقانچە ئامىل كەلتۈرۈپ چىقىرىدۇ.

1. يۇقىرى تېمپېراتۇرا. ماتور زور مىقداردا ئىسسىقلىق چىقىرىدۇ. شوپۇر ئۇزاق رول تۇتۇپ ئولتۇرغاندا جىنسىي ئەزا، ئۇرۇقدان خالتىسى ۋېناسىدىكى قايتما ئېقىم توسالغۇغا ئۇچراپ، جىنسىي ئەزا ۋە ئۇرۇقداننىڭ تېمپېراتۇرىسى ئۆرلەپ كېتىدۇ.

2. ئۇرۇقدان ۋە قوشۇمچە ئۇرۇقدانغا ئوكسىگېن يېتىشمەسلىك، كېرەكسىز نەرسىلەر مېتابولىزمى راۋان بولماسلىق. ئۇزاق ئولتۇرغاندا ئۇرۇقدان، قوشۇمچە ئۇرۇقدان ۋە ئۇرۇقدان بېغى ئۇزۇن ۋاقىت بېسىلىپ، جىنسىي ئەزانىڭ قان ئايلىنىشى توسالغۇغا ئۇچراپ، ماددا ئالمىشىشتىن پەيدا بولغان كېرەكسىز نەرسىلەر يىغىلىپ قالىدۇ، بۇنىڭ بىلەن ئىسپېرمىنىڭ ساپلىقى ۋە ساقلىنىشىغا زىيان يېتىدۇ.  

3. جىددىي كەيپىيات. شوپۇر ھەرۋاقىت زېھنىنى يۈكسەك دەرىجىدە مەركەزلەشتۈرگەن ھالەتتە تۇرىدۇ. بۇ ئىچكى ئاجراتما ھورمونىنىڭ ئاجرىتىلىشى ۋە سەزگۈرلۈكىنى ئۆزگەرتىپ، ئىسپېرمىنىڭ ساپلىقى ۋە ھەرىكەتچانلىقىغا تەسىر كۆرسىتىدۇ                                     .

4. بېنزىن خۇرۇچى. تەتقىقاتلاردىن ئىسپاتلىنىشىچە، بېنزىن خۇرۇچلىرىدىن مېتىل تېرتىئارى بۇتىل ئېفىر (MTBE) ئۇرۇقنىڭ غەيرىي شەكىلدە بولۇپ قېلىش نىسبىتىنى ئاشۇرۇۋېتىدىكەن.

5.ماشىنا شاۋقۇنى. شاۋقۇن ئادەمنىڭ ئىچكى ئاجراتمىلىرىنى قالايمىقانلاشتۇرۇپ، مەنى ۋە ئىسپېرمىنىڭ نورمالسىزلىقىنى كەلتۈرۈپ چىقىرىدۇ.

6. ماشىنا قۇيرۇق گازى. كىشى ماشىنا قۇيرۇق گازى بىلەن ئۇزۇن ۋاقىت ئۇچراشقاندا جىنسىي ئەزانىڭ ساغلاملىقىغا تەسىر يېتىدۇ  .

قانداق قىلغاندا ماشىنا ھەيدەشتىن كېلىپ چىقىدىغان تۇغماسلىقنىڭ ئالدىنى ئالغىلى بولىدۇ؟

1. ماشىنا تېمپېراتۇرىسىنى كونترول قىلىش كېرەك. ماشىنا ھەيدىگەندە تېمپېراتۇرىنى تەڭشەش، يازدا ھاۋا تەڭشىگۈچ ئېچىپ مۇۋاپىق تېمپېراتۇرىنى ساقلاش كېرەك. ماشىنا ھەيدەش ۋاقتى بەك ئۇزۇن بولسا ئارام ئېلىش ۋە ماشىنىدىن چۈشۈپ ھەرىكەت قىلىشقا دىققەت قىلىش، بىر سائەتتىن ئارتۇق ماشىنا ھەيدىگەندە چوقۇم ماشىنىدىن چۈشۈپ ئون مىنۇت ھەرىكەت قىلىپ، ئۇرۇقداننىڭ ئۇزۇن ۋاقىت يۇقىرى تېمپېراتۇرىلىق ھالەتتە تۇرۇشىنىڭ ئالدىنى ئېلىش كېرەك.

2. پاختا رەختتىن تىكىلگەن ئازادە ئىشتان كىيىشكە دىققەت قىلىش كېرەك. تار ئىشتان، پادىچىلار ئىشتىنى قاتارلىق چىڭ كېلىدىغان ئىشتانلار ئەرلەر كۆپىيىش ئەزالىرىنىڭ تېمپېراتۇرىسىنى ناھايىتى تېز ئۆرلىتىپ، ئۇرۇقدان خالتىسىنىڭ قان ئايلىنىشىنى توسالغۇغا ئۇچرىتىدۇ. شۇنىڭ ئۈچۈن ئەرلەر سۈپەتلىك، ھاۋا ئۆتۈشىدىغان ئىشتان كىيىپ، كۆپىيىش ئەزالىرىنىڭ بېسىلىشىنىڭ ئالدىنى ئېلىشى لازىم.

3. ئېلېكترونلۇق مەھسۇلاتلار بىلەن ئۇچرىشىشنىڭ ئالدىنى ئېلىش كېرەك. يانفون، كومپيۇتېر، مىكرو دولقۇنلۇق ئوچاق، چاچ قۇرۇتقۇچ، پىرىنتېر قاتارلىق ئېلېكتىرونلۇق مەھسۇلاتلارنى ئىشلىتىش ۋە مەشغۇلات قىلىش داۋامىدا ئۇلاردىن دولقۇن ئۇزۇنلۇقى ۋە چاستوتىسى ئوخشاش بولمىغان ئېلېكتىر ماگىنىت دولقۇنى تارقىلىدۇ. ئۇلارنىمۇۋاپىق ئىشلەتمىگەندە ئەرلەرنىڭ كۆپىيىش ئىقتىدارىغا ئوخشاش بولمىغان دەرىجىدە تەسىر يېتىدۇ. شۇنىڭ ئۈچۈن خاتىرە كومپيۇتېر ئىشلەتكەندە تىزغا قويۇۋالماسلىق، يانفوننى ئىشتان يانچۇقىغا سالماسلىق ياكى بەلگە قىستۇرۇۋالماسلىق كېرەك.

4. ياخشى، تەرتىپلىك تاماق ئادىتىنى يېتىلدۈرۈش كېرەك. يېشىل كۆكتاتلارنى، بېلىق، ئۇيىستېر قۇلۇلىسى، جىگەر، دادۇر، يىرىك گۈرۈچ قاتارلىق تەركىبىدە سىنىك، سېلېن مول يېمەكلىكلەرنى كۆپ يىيىش، تېز تاماقلارنى ئاز يېيىش كېرەك.  

مۇتەخەسسىسلەر شۇنى سەمىڭىزگە سالىدۇكى، ئايالىڭىز ھامىلىدار بولۇشتىن ئۈچ ئاي بۇرۇن ئامال بار ماشىنا ھەيدىمەڭ، ماشىنا ھەيدىگەندە پادىچىلار ئىشتىنى كىيمەستىن، بەلكى ئازادە، راھەت ئىشتان كىيىڭ، ئورۇندۇق سېلىنچىسىغا سالقىن، ھاۋا ئۆتۇشىدىغاننى تاللاڭ. شۇنىڭ بىلەن بىللە ئىشخانىدا ئولتۇرۇپ خىزمەت قىلىدىغان بولسىڭىز ئەڭ ياخشىسى ياغاچ ئورۇندۇق تاللاڭ، يېرىم سائەتتىن بىر سائەتكىچە بولغان ئارىلىقتا ئورنىڭىزدىن تۇرۇپ شامالداڭ.
