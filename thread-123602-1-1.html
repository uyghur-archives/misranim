   كورلا شەھەر باشلىقىغا ئوچۇق خەت


    ئەسسالامۇ ئەلەيكۇم ھۆرمەتلىك شەھەر باشلىقى،يېقىندىن بۇيان ئۆز يېزامدىكى بىر قىسىم خەلق مەنپەتىگە چېتىلدىغان مۇھىم مەۋقەلەر توغىرسىدا كۆپ ئويلىنىپ كەتتىم.بۇ ھەقتە كەنتتە ئېچىلغان ئاممىدىن پىكىر-تەلەپ ئالدىغان جەدۋەلگە نەچچە ئاينىڭ ئالدىدا ئۆز پىكىر-تەكلىپىمنى يازغان ئىدىم.لېكىن،5-ئايلار كىرىشى بىلەن پۈتكۈل يېزىمىزدىكى ھەممە دېھقانلارنىڭ تىرىكچىلىك قىلىشى ئۈچۈن ئەڭ مۇھىم بولغان سۇ مەسىلسى كاللىمىزنى قاتۇردى.بۇ ھەقتە بازارلىق ھۆكۈمەتنىڭ رەھبەرلىرى بىلەن كۆرۈشۈپ بېقىش ئويىغىمۇ كەلگەن.ئەمما،بۈگۈن ئەتىگەندە گۇرۇپ باشلىقىنىڭ ياڭراتقۇدا ئەرتىدىن باشلاپ ھەر قايسى توتەن،مەيدان بويىنچە باغقا قۇدۇق سۈيى ئاچىسىلەر،ئېرىق سۈيى كەلمەيدۇ...دېگەن ئۇقتۇرىشىنى ئاڭلىغاندىن كېيىن شۇڭا بۇ ئۇچۇق خەتنى يېزىپ ئولتۇرۇپتىمەن.ھەر ھەپتىدە بىر قېتىم بايىنغولىن رادىئۇ ئىستانىسىدا بېردىغان ‹‹ئاۋام ساداسى››پىرۇگراممىسىغا مۇشۇ ھەقتە تەكلىپىمنى سۇناي دېسەممۇ ئەمما ۋاقىت چەكلىمىسى تۈپەيلىدىن تېلفۇن ئۇرسام ئۇلانمىدى.شۇڭا،يېزىمىزدا ساقلىنۋاتقان بىر قىسىم مەسىللەر ھەققىدەتور ئارقىلىقلا دېيىش مەقسىتىگە كەلدىم.
      كورلا شەھىرىمىزنىڭ مەملىكەت بويىنچە پاكىزە شەھەر ئىكەنلىكىنى پۈتۈن مەملىكەت خەلقى بىلىدۇ،نەچچە ئون يىللاردىن بۇيان شەھىرىمىزدىكى دېھقان ئاممىسى پارتىيە-ھۆكۈمەت بىلەن بىردەكلىكنى ساقلاپ،ئىناق سوتسىيالىستىك يېزا قۇرۇلىشى،ھاللىق سەۋىيەگە يېتىشى ئۈچۈن پارتىيە-ھۆكۈمەتنىڭ يوليۇرۇقى بويىنچە تىرىشىپ كەلمەكتە.شەھىرىمىزدىكى يېقىنقى يىللاردىن بۇيان بولغان ئالەمشۇمۇل ئۆزگىرىشلەر،تەرەقىياتلارنى پارتىيە-ھۆكۈمەتنىڭ كۈچ چىقىرىپ قوللىشىدىن ئايرىپ قايرىغىلى بولمايدۇ.شۇ سەۋەبلىك،كورلا دېھقانلىرى ھازىرمۇ نەشپۈتنى ئاساس،كېپەز-چارۋىچىلىقنى قۇشۇمچە كەسىپ قىلىپ ھاللىق سەۋىيەگە يېتىش ئۈچۈن ئۈزلۈكسىز تىرىشۋاتىدۇ.ئەمدى مەن ئۆز يېزامدا ساقلىنۋاتقان بەزى مەسىللەرنى ئوتتۇرغا قويماقچى،شەھەر باشلىقىنىڭ بىزنىڭ تەكلىپ-پىكرىمىزگە قۇلاق سېلىپ بىزگە قانائەتلىنەرلىك جاۋاب بېرىشىنى سەمىمىي ئۈمىد قىلىمەن.
  1.يېزىمىزدىكى سۇ مەسىلسى
     سۇ مەسىلسى-پۈتكۈل شىنجاڭ ئۇيغۇر ئاپتۇنۇم رايۇنىدا كۆرۈلۋاتقان قىيىنچىلىقلارنىڭ بىرى.بۇ ھەقتە پارتىيە-ھۆكۈمەت خەلقنىڭ جانىجان مەنپەتىنى ئويلاپ تېمىتىپ سوغۇرۇش،پۈركۈپ سوغۇرۇش،يەر ئاستى سۇ تۇربىسى ئارقىلىق سوغۇرۇش،سۇ سىڭمەس ئۆستەڭ قۇرۇلىشى قىلىپ سوغۇرۇش قاتارلىق ئىشلارغا زور ئىقتىساد چىقىرىپ سۇ مەسىلسىنى مۇۋاپىق يۇسۇندا ھەل قىلىپ كەلدى.ئەمما بۇ يىل بىزنىڭ يېزىدا سۇ قىس بولۇش ئەھۋالى ئېغىرلاپ كەتتى،كورلا شەھىرىدىكى نەشپۈت بىلەن تىرىكچىلىك قىلىپ كېلىۋاتقان يېزىلار ئىچىدە سايدۆڭ باغۋەنچىلىك مەيدانى،توبرىچى يېزىسى،قارا يۇلغۇن يېزىسى،شاڭخۇ بازىرى،چارباغ يېزىسى...قاتارلىق يېزىلار بار.ئەمما بۇ يىلنىڭ بېشىدا بىزنىڭ شاڭخۇ يېزىدا نۇرغۇن دېھقانلار ئۆستەڭ سۈيى ئاچالماسلىق،قۇدۇق سۈيى ئاچاي دېسىمۇ سۇنىڭ مەنبەسى ئىنتايىن ئاز بولغانلىقى ئۈچۈن باغلارغا سۇ ئاچالمايۋاتىدۇ.ئەرتىدىن باشلاپ قۇدۇق سۈيى ئاچساق بىر كەنتتە ئوتتۇرا ھېساب بىلەن 1000مودىن يەر بار دەپ ھېسابلىساق،ئىككى-ئۈچ قۇدۇقنىڭ سۈيى بىلەن ئۇسساپ قاغجىراپ كەتكەن يەرلەرنى سۇ بىلەن قاندۇرۇپ بولۇش ھەرگىز مۇمكىن ئەمەس.ئاڭلىسام بۇ يىلقى سۇ ئېچىش پىلانىدا ئالدىنقى قېتىم ئېچىلغان سۇ پۇلىنى ھەممە كىشى تاپشۇرۇپ بولغاندىن كېيىن ئاندىن ئۆستەڭ سۈيى ئالالايمىز دېگەن ئىشلارنى ئاڭلىدۇق.بۇ مەسىلىدە دېھقانلارنىڭ ھەممىسى سۇ پۇلىنى تۆلەپ بولالمايدۇ.پەقەت كۈز ۋاقتى ھوسۇل يىغقان ۋاقىتتا ئاندىن سۇ پۇلىنى تۇلۇق تاپشۇرالايدۇ.شۇڭا،بۇ ھەقتە شەھەرلىك پارتكۇم-خەلق ھۆكۈمىتىنىڭ بىزلەرگە مۇۋاپىق يۇسۇندا ئېتىبار قىلىشىنى سەمىمىي ئۈمىد قىلىمەن.
شاڭخۇ بازىرىدىكى دېھقان-چارۋىچىلارنىڭ مۇتلەق كۆپى باغۋەنچىلىك ئارقىلىق كىرىم قىلىپ ئۆز تۇرمۇشىنى قامدايدۇ.ھازىر باغلارغا سۇ ئاچمىغىلى 40كۈندىن ئاشتى.شۇڭا،شەھەرلىك پارتكۇم-خەلق ھۆكۈمىتىنىڭ يېزىمىزىكى دېھقانلارنىڭ جانىجان مەنپەتىگە چېتىلدىغان،ھاللىق سەۋىيەگە يېتىشى ئۈچۈن مۇھىم بولغان ئۆستەڭ سۈيىنى ھەل قىپ بېرىشىنى ئۈمىد قىلىمەن.
    2.يېزىمىزغا يېڭىدىن سېلىنغان سودا بازىرى ھەققىدە
  شاڭخۇنىڭ يېزا بازىرىنىڭ 28-پولككا كېتىدىغان تەرەپتە نەچچە يىل ئالدىدا شاڭخۇ بازىرى 1-گۇرۇپتىكى ئۈچ-تۆت دېھقانننىڭ 20مو كۆلەمدىكى يېرىنى ئېلىپ مال-چارۋا بازىرى قىلىمىز دەپ ئالغان ئىدى.لېكىن،شاڭخۇ كەنت مالىيەسى تۇلۇق ئىقتىساد ھەل قىلالماي نەچچە يىل ئاق تاشلىنىپ قالغان.ئالدىنقى يىلى شاڭخۇ كەنت پارتىيە ياچىكىسى-ئاھالىلەر كومىتىتى بۇ يەرگە سودا بازىرى قىلىش قۇرۇلىشىنى باشلىدى.ئاڭلىسام بۇ يەرگە كېتىدىغان ئىقتىساد ئۈچۈن ئۈچ يېرىم مىليۇن مەبلەغ كېتىدىكەن.ھازىر بىر مىليۇنغا يېقىن ئىقتىسادنى شۇ يەرگە چىقىرىپ بوپتۇ.بۇ قۇرۇلۇشنى پۈتتۈرۈش ئۈچۈن يەنە ئىككى مىليۇنغا يېقىن مەبلەغ كېتىدىكەن.مېنىڭ بۇ ھەقتە سۆزلىشىمدىكى ئاساسىي سەۋەب شۇكى،يېزىمىز كورلا شەھىرىگە ئىنتايىن يېقىن.ئون بەش كېلۇمېتىر كېلىدۇ.تۇرمۇشقا كېرەكلىك زۆرۈر نەرسىلەرنى شۇ يېزىدا تۇرۇۋاتقان ھەر قانداق دېھقان شۇ كۈنىلا شەھەرگە بېرىپ سېتىۋالالايدۇ.ھەممە ئائىلىدە ماشىنا بار.قاتناشمۇ ئىنتايىن قولايلىق.شۇنداق تۇرۇقلۇق نېمە ئۈچۈن كەنت پارتىيە ياچىكىسى-ئاھالىلەر كومىتى كەنت مالىيەسىگە توپلانغان ئىقتىسادنى قالايمىقان ئىشلىتىدۇ.سالمىسىمۇ بولدىغان سودا بازىرىغا نېمە ئۈچۈن كۈچەپ كەتتى.ئەسلى كەنت مالىيەسىدە يىغىلغان پۇللارنى ھەربىر گۇرۇپقا قۇدۇق قېزىپ سۇ مەسىلسىنى ھەل قىلىش،سۇ ئامبىرى ياساپ باغلارغا دېھقانچىلىق دورىسى چاچىدىغانغا قولايلىق سۇ مەنبەسى بىلەن تەمىنلەش،چارۋىچىلىق كەسپىنى تەرەقىي قىلدۇرۇش ئۈچۈن رىغبەتلەندۈرۈش ...قاتارلىق مۇھىم ئىشلارغا قاراتسا بولاتتى.ئۇنىڭ ئۈستىگە شاڭخۇ كەنت 1-گۇرۇپتا نۇپۇس سانى باشقا گۇرۇپقا قارىغاندا زىيادە كۆپ،ئۆيلۈك مەسىلسىمۇ كاللىنى قاتۇرۋاتقان ئەڭ مۇھىم ئىشلارنىڭ بىرى.شۇ يەرگە شۇ گۇرۇپتىكى ياشلارغا ئۆيلۈك يەر كېسىپ بەرسىمۇ خەلقنىڭ كۆڭلۈدىكىدەك ئىش بولاتتى.بۇ ئىشلاردا شاڭخۇ1-گۇرۇپتىكى نۇرغۇن بالىلار نارازى.شۇڭا،بۇ ھەقتە بىزگە قانائەتلىنەرلىك جاۋاب لازىم.
   3.شاڭخۇ 5-گۇرۇپتىكى يول-كۆۋرۈك مەسىلسى
    شاڭخۇ بازىرى شاڭخۇ 5-گۇرۇپ ئىسلاھات-ئىشىكنى ئېچىۋىتىش يولغا قۇيۇلغاندىن كېيىن يېڭىدىن قۇرۇلغان مەھەللىنىڭ بىرى.شاڭخۇ 5-گۇرۇپ قۇرۇلۇش ئالدىدا يېزا رەھبەرلىرى ھەربىر گۇرۇپتىن ئەمگەكچان،تېز روناق تاپالايدىغان،نەزەر-دائىرسى كەڭ بىر قىسىم ئاۋانگىرات ياشلارنى شۇ مەھەللىگە توپلاپ بىر گۇرۇپنى مەيدانغا كەلتۈرگەن.نەچچە ئون يىلدىن بۇيان نۇرغۇن پىشقەدەم يەرلىك كادىرلار شۇ مەھەللىنىڭ تەرەقىياتى ئۈچۈن تىنىمسىز ئىشلەپ ئىناق سوتسىيالىستىك يېزا-قۇرۇلىشى ئۈچۈن نۇرغۇن خىزمەتلەرنى قىلىپ كەلدى.يېقىنقى يىلاردىن بۇيان شۇ مەھەللىنىڭ يولى ئىنتايىن ناچار ھالەتكە چۈشۈپ قالدى.كۆۋرۈكمۇ ئىنتايىن بۇرۇن سېلىنغان بولۇپ ھازىر خەتەرلىك ھالەتتە.يامغۇر يېغىپ قالسا يوللار ئىنتايىن ئېگىز-پەس بولۇپ قاتناش ئىنتايىن قولايسىز.باشقا تۆت گۇرۇپنىڭ يول،كۆۋرۈك قۇرۇلىشى ئاساسەن يېڭىلىنىپ بولغان بولۇپ پەقەت مۇشۇ مەھەللە تاكى ھازىرغا كەلگۈچە مۇشۇنداق ھالەتتە تۇرماقتا.شۇڭا،شەھەرلىك پارتكۇم-خەلق ھۆكۈمىتىنىڭ شۇ مەھەللە ئىچىدىكى يوللارنى ئاسفالىت قىلىپ بېرىشىنى،خەتەر ئىچىدە قالغان كۆۋرۈكنى يېڭىلاپ بېرىشىنى سەمىمىي ئۈمىد قىلىمەن.
يۇقۇرقى بايانلار پۈتكۈل مەھەللىدىكى كىشىلەر ئويلىنۋاتقان مۇھىم مەسىلەرنىڭ بىرى.توردا جامائەت پىكرى توپلاش،خەلقنىڭ ئاھۇ-زارىنى ئېيتىش جامائەت پىكرى نازارەتچىلىكىگە كىرىدۇ.بۇ بىر تەرەپتىن يۇقۇرى دەرىجىلىك ھۆكۈمەت ئاپراتلىرىغا ئۇچۇر-ماتىرياللارنى تەمىنلەپ،خەلق بىلەن ھۆكۈمەت ئوتتۇرسىدا ھەل بولۇنماي كېلىۋاتقان بەزى مەسىللەرنىڭ تېزدىن ھەل بولۇشى ئۈچۈن پايدىلىق شارائىت بىلەن تەمىن ئېتىدۇ.شۇڭا،مەن ئوتتۇرغا قويغان بۇ مەسىللەر ھەققىدە ئالاقىدار ئورۇندىكى رەھبەرلەرنىڭ قانائەتلىنەرلىك جاۋاب بېرىشىنى ھەمدە ھەل قىپ بېرىشىنى سەمىمىي ئۈمىد قىلىمەن.
كامالىي ئېتىرام بىلەن:شاڭخۇ بازىرى شاڭخۇ 3-كەنتتە ئولتۇرۇشلۇق دېھقان.

ئەسكەرتىش:گەرچە ئۆزۈم مىسرانىم تورىنىڭ مۇنبەر باشلىقى بولساممۇ،ئاۋامنىڭ ئەڭ مۇھىم جانىجان مەنپەتىگە چېتىلدىغان بۇ تېمىنى ئەڭ ئاۋال شىنجاڭ ئۇيغۇر ئاپتۇنۇم رايۇنلۇق ئاخبارات ئىشخانىسى باشقۇرىشىدىكى تەڭرىتاغ تورىغا يوللىدىم.كەينىدىن مىسرانىم مۇنبىرىگە يوللىدىم.تەڭرىتاغ تورى ھۆكۈمەت باشقۇرىشىدىكى تور بەت،مەنچە  تەڭرىتاغ تورىدىكى باشقۇرغۇچىلار بۇ تېمىنى تەستىقلاپ جامائەت بىلەن يۈز كۆرۈشتۈرەلەيدىغانغا ئىشەنچىم كامىل.
