   تويىمىزغا ئاز قالغانلىقى ئۈچۈن بۈگۈن ئىشتىن چۈشۈپ تاماقمۇ يىمەي يېڭى ئالغان ئۆيىمىزگە كەلدىم،مېڭىشتىن ئىلگىرى ئوغۇل دوستۇم يەنى بولغۇسى ئېرىم ئەنۋەرگە تېلفۇن قىلسام بۈگۈن ئىدارىدىكىلەرنىڭ يىغىلىشى باركەن بالدۇرراق قايىتسام بارىمەن دېگەچكە ئۆزۈم كېىلىۋەرگەن ئىدىم،ئۆي ئىچىنى يېڭىلا زىننەتلەپ بولغان بولغاچقا ئىشىك-دېرىزىدىن تارتىپ ھەممە يەرنى چاڭ توزان قاپلاپ كەتكەن،شۇڭا بىرمۇ بىر ئەستايىدىللىق بىلەن تازلىماي بولمايىتتى،ئامال يوق ئۆزۈم يالغۇز ئۆي تازلاشقا كىرىشىپ كەتتىم،ئالاھەزەل 3-4سائەتلەردىن كىيىن بەللىرىم ئۇيۇشۇپ ئاغرىشقا باشلىغاندىلا ئاللىقاچان قاراڭغۇ چۈشۈپ كەتكىننى بىلدىم ۋە دېرىزە ئالدىغا كەلدىم،ئاپلا سىرىتتا قاتتىق يامغۇر يېغىپ كېتىپتۇ ئەمەسمۇ،يا كۈنلۈك ئېلىۋالماپتىمەن،قانداقمۇ قىلارمەن،شۇ خىياللار بىلەن ئەنۋەرگە تېلفۇن قىلدىم،لېكىن ئۇ شۇنداق تەستە تېلفۇننى ئالدى ۋە پەس ئاۋازدا:
-ھەنېمە ئىش>>دەپ سورىدى.ئۇنىڭ بۇ ئۇدۇللا دىيلگەن گېپىدىن سەل ئاچچىم كەلدى لېكىن چاندۇرمىدىم:
-سىرىتتا بەك يامغۇر يېغىپ كېتىپتۇ،ئىشىڭىز تۈگكەن بولسا ئالغىلى كېلەمدىكىن دېگەنتىم
-ھە،،يەنە بىرەر سائەتكىچە بېرىپ بولىمەن،سىز ئۆيدە ئولتۇرۇپ تۇرۇڭ،
-بولىدۇ،ئەمسە سىزنى ئۆيدە ساقلاي دېدىم ۋە تېلفۇننى قويىۋېتىپ ئىشلىرىمنى داۋملاشتۇرۇپ كەتتىم،ئارىدىن 1سائەتتىن ئارتۇق ۋاقىت ئۆتكەندە ئاپام تېلفۇن قىلدى،ئاپامغا ئەنۋەرنىڭ ئاپىرىپ قويىدىغانلىقىنى ئېيتىپ خاتىرجەم قىلغاندىن كىيىن سائەتكە قارىدىم،سائەت ئاللىقاچان 9دىن ئاشقان ئىدى،توۋۋا ئۇ بىر سائەتكىچە بارىمەن دېگەن ،لېكىن ئىككى سائەت بولاي دېدى،نېمىشقا كېلىپ بولالمايدىغاندۇ؟بىرەر ئىش بولمىغاندۇ،شۇ خىياللار بىلەن ئۇنىڭغا تېلفۇن قىدىم،لېكىن ئۇنىڭ تېلفۇنى ئېتىك ئىدى،كۆڭلۈمگە يەتمىگەن خىياللار يېتىپ ئەنسىرەشكە باشلىدىم،يەنە بىر تۇرۇپ بەلكىم تېلفۇننىڭ توكى تۈگگەندۇ،يولدا كېلىۋاتقاندۇ ،يەنە بىر ئاز ساقلاپ باقاي دەپ ئويلىدىم ۋە تازلىق ئۆيىگە كىرىپ كىيىملىرىمنى ئالماشتۇرۇپ قولۇمنى يۇيۇپ ئۇنى ساقلاپ ئولتۇردۇم،لېكىن ئۇنىڭ قارىسىمۇ كۆرۈنمەيىتتى،يامغۇر بولسا تېخىمۇ كۈچىيىپ يېغىپ كېتىۋاتاتتى،قانداق قىلارمەن ئەمدى ،بۇ ئۆيدە يا قونۇپ قالغىلى بولمىسا،ئۇنىڭغا تېلفۇن قىلدىم يەنىلا سادا يوق،ئامالسىز ئۆيدىن چىقتىم،يامغۇر خۇددى چېلەكتە قۇيغاندەك يېغىپ ،ئۇنىڭغا قوشۇلۇپ شاما ل چىقىپ چاقماق چېقىپ كۆڭلۈمنى تېخىمۇ ئەنسىزلىككە تولدۇرغان ئىدى،يولدا بىرمۇ ئادەم يوق،ئالدىمغا ئازراق مېڭىپ بىر دالدا يەردە ماشىنا ساقلىدىم،لېكىن بىرمۇ ماشىنا توختىمايىتتى،ئارىدىن 5مىنۇتلار ئۆتۈپ ئالدىمدىن شىددەت بىلەن ئۆتۈپ كەتكەن قارا رەڭلىك دالا ماشىنىسى كەينىگە قايتىپ يېنىمدا توختىدى،قورىققىنىمدىن كەينىمگە نەچچە قەدەم داجىدىم،لېكىن شوپۇر ماشىندىنمۇ چۈشمەي تۇرۇپ دەرىزىنى چۈشۈردى ۋە مۇلايىملىق بىلەن:
-قىزچاق ،ماشىنا ساقلاۋاتامسىز،دەپ سورىدى
-ھە ،ھە شۇنداق دېدىم مەن دۇدۇقلاپ تۇرۇپ
-ئەمسە ماشىنىغاچىقىڭە ئالغاچ كېتەي
-ياق بولدى،كېتىۋېرىڭ،رەھمەت سىزگە
-ئۇنداق دېمەڭ بۇنچە كەچ بۇ يەردە يالغۇز تۇرسىڭىز خەتەرلىك،مەنمۇ يامان ئادەملەردىن ئەمەس،مېنىڭمۇ بىر سىڭلىم بار،نىيىتىم ئۇنچە يامان ئەمەس،خاتا چۈشەنمەڭ
ئۇنىڭ سۆزلىرىدىن كۆڭلۈم سەل جايىغا چۈشكەندەك بولدىيۇ،لېكىن يەنىلا سەل ئەنسىزلىك بىلەن جايىمدا تۇرىۋەردىم
تېز بولىڭە ،ئەمدى چىقمىسىڭىز كېتىمەن جۇمۇ،قالىسىز مۇشۇ يەردە
ئامال يوق سەل ئەنسىزچىلىك ئىچىدە ئۇنىڭ ماشىنىسىغا چىقتىم،ماشىنا ئىچى تولىمۇ راھەت ،پاكىز ئىدى،شوپۇر يىگىت قارىماققا ياش جەھەتتە بىزدىن ئانچە پەرىقلىنىپ كەتمەيىتتى،
-ھە راست ،ئۆيىڭىز قايسى تەرەپتە؟
-ياۋرۇپا گۈللىكى 1-قاراردا
-شۇنداقمۇ قوشنىكەنمىز ئەمسە،مېنىڭ ئۆيۈم 2-قاراردا،دېدى ئۇ كۈلۈپ تۇرۇپ،
ھە شۇنداقمۇ،،،دېدىم مەن ئارتۇق سۆزلىمەي.
خۇددى مەن سىزنى يەۋېتىدىغاندەك،ياكى ئاپىرىپ سېتىۋېتىدىغاندەك ئەجەپمۇ تەستە چىقتىڭىز ھە ماشىنىغا؟
-ئۇنداق ئەمەس،كەچ كىرىپ قالدى شۇڭا،،،،
-ھا ھا ھا ،خاتىرجەم بولۇڭ،سزدەك سەت قىزلارنى ھېچكىم ئەپ قاچمايدۇ،پايدىسى يوق ،دېدى ئۇ كۆزىنى قىسىپ قويۇپ،ئۇنىڭ گېپىدىن سەل خاتىرجەم بولغان بولساممۇ ،سەت دېگىنىگە  ئاچچىقىم كەلدى،بوپتۇلا،نېمىلا دېسە دېسۇن،ئۆيۈمگە ساق-سالامەت بېرىۋالساملا بولمىدىمۇ،يول بويى ئۇنىڭ سۆزلىرىگە ھە ياق دەپلا جاۋاپ بەردىم،ئاخىرى ئىشىك ئالدىغا كېلىۋالدىم،ماشىندىن چۈشىدىغان چاغدا ئۇنىڭغا پۇل تەڭلىدىم ،لېكىن ئۇ مەن كىرا قىلمايمەن دەپ پەقەت ئالغىلى ئۇنىمىدى،مەنمۇ جاھىللىق قىلىپ تۇرىۋالدىم،ئۇ ئاخىرى قولۇمدىكى 100يۈەنلىك پۇلنى ئېلىپ پارچىلاپ قايتۇردى،ئۇ قايتۇرغان پۇلنى سانىمايلا پۇرلاپ يانچۇقۇمغا سالدىم ۋە رەھمەت دەپلا ئۆيگە قاراپ يۈرۈپ كەتتىم،ئۆيگە كىرسەم ئاپام مەندىن ئەنسىرەپ ئۇخلىماي ئولتۇرۇپتۇ،مېنى كۆرۈپ خاتىرجەم بولۇپ ياتاق ئۆيىگە قاراپ ماڭدى،مەنمۇ ياتاق ئۆيۈمگە كىرىپلا سومكامنى قويۇپ ئەنۋەرگە تېلفۇن قىلدىم،لېكىن نەتىجە ئوخشاش يەنىلا ئېتىك،ئەمدى ئۇنىڭدىن رەسمى ئەنسىرەشكەباشلىدىم،لېكىن ئۇنى نەدىن ئىزدەيمەن؟توغرا بىر ئىشخانىدا ئىشلەيدىغان خىزمەتدىشى ئالىمغا تېلفۇن قىلاي،كاللامغا كەلگەن ئەقىلدىن خوش بولۇپ ئۇنىڭ نومۇرىن باستىم
-ۋەي ،كىمۇ؟
ئالىمنىڭ ئۇيقۇ چىراي ئاۋازىدىن يۈرىگىم قارىتلا قىلىپ قالدى،
-ياخشىمۇ سىز ئالىم،مەن دىلناز،
-ھە دىلنازمۇ سىز،،ياخشىمۇسىز،ئۇ قايتىدىن ئەھۋال سورىدى.
-ياخشى ،سىزنى كەچ بولغاندا ئاۋارە قىلدىم،ئەنۋەريېنىڭدا بارمىدۇ؟
-ئەنۋەر؟ئۇ ئىشتىن چۈشۈپلا ئالدىراپ چىقىپ كەتكەن، مېنىڭ يېنىمغا كېلىمەن دېگەنمۇ؟
-ياق،،سىلەر بۈگۈن ئىدارىدىكىلەر يىغىلىش ئۆتكۈزمىگەنمۇ؟
-ۋاي،ياق،نەدىكى يىغىلىشكەن ئۇ،ئەنۋەر سىزگە شۇنداق دېگەنمۇ؟
شۇ چاغدىلا ئەنۋەرنىڭ يالغان ئېيىتقىنىنى بىلدىم،لېكىن دوستىنىڭ ئالدىدا سەت تۇرمىسۇن دەپ چاندۇرمىدىم
-ئۇنداق دېمىگەن،تېلفۇنى ئېتىك،شۇڭا ئۆزەم پەرەز قىلىپ سىزگە تېلفۇن قىلغان،قارىغاندا تېلفۇننىڭ توكى تۈگكەن ئوخشايدۇ،بولدى سىزمۇ ياخشى ئارام ئېلىڭ،ئاۋارە قىلدىم،
-ھېچقىسى يوق،بەك ئەنسىرەپ كەتمەڭ،ھە،
-بولىدۇ،،خوش
تېلفۇننى قويىۋېتىپ لاسسىدە بولۇپ كارۋاتتا ئولتۇرۇپ قالدىم،كاللامدا پەقەت بىرلا سوئال،ئۇ نېمىشقا مېنى ئالدايدۇ؟نېمىشقا تېلفۇننى ئېتىۋالىدۇ؟ئىرماش چىرماش تەھلىللەر مېنى قايمۇقتۇردى،بولدىلا ئەتە ئۆزىدىن سوراپ باقاي قېنى نېمە دەيدۇ،ئورنۇمدىن تۇرۇپ چاپىنىمنى سالدىم،يانچۇقۇمدىكى بايا قايتۇرغان پۇللارنى رەتلەپ پورتىمالىمغا سالاي دەپ قارىسام،مەن ئۇنىڭغا بەرگەن 100يۈەنلىك پۇلنى پەقەت ماڭا پارچە پۇللارغا ئالماشتۇرۇپلا بېرىپتۇ،50يۈەنلىكتىن بىر،20يۈەنلىكتىن ئىككى،ئون يۈەنلىكتىن بىر،،،ئەجەپ كاللىسى ئىشلەيدىغان شوپۇركىنا بۇ،،،ھەقىقەتەن ياخشى نىيەتلىككەن،شۇلارنى ئويلاپ ئۇنىڭغا بىر خىل ھۆرمىتىم ئاشتى.
  ئۇخلاش ئۈچۈن كارۋاتقا چىقتىم،ئەنۋەرگە يەنە تېلفۇن قىلدىم،.يەنىلا ئېتىك،ھەي نېمە ئىشلاردۇ بۇ،سەل ھېرىپ قالغاچقىمۇ تېزلا ئۇخلاپ قاپتىمەن،ئەتىسى سەھەر ئورنۇمدىن تۇرۇپلا يەنە تېلفۇننى قولۇمغا ئالدىم ،ئەپسۇس،،،،،
ئىشقا كېلىپ بىېرەر سائەتلەردىن كىيىن،ئالىمغا تېلفۇن قىلىپ بۈگۈن ئەنۋەرنىڭ ئىشقا كەلمىگەنلىكىنى بىلدىم ۋە ئەنسىزلىكىم تېخىمۇ ئېشىپ،ئۇنىڭ ئۆيىگە تېلفۇن قىلدىم:
-ئەسسالامۇ ئەلەيكۇم ھەدە!
-ۋەئەلەيكۇم ئەسسالام ،دىلناز قىزىممۇ سىز؟ياخشى تۇردىڭىزمۇ ؟
-ياخشى ،ئۆزەڭلارچۇ،ياخشى تۇرۋاتامسىلەر،سالامەتلىكىڭىز ياخشىدۇ؟
-ياخشى،خۇداغا شۈكرى بالام،.تېخى سىزگە تېلفۇن قىلاي دەپ تۇرغان،ئەنۋەر ئاخشام ئۆيگە قايىتمىغان،دادىسى ئىككىمىز ئەنسىرەپ كېچىچە ئۇخلىمۇدۇق،سىزنى خەۋىرنى بىلەمدىكىن دەپ تېلفۇن قىلايلى دەپ تۇرغان،
بېشىمدىن بىر قاپاق سۇ قۇيغاندەك ئەندىكىپ كەتتىم،دېمەك ئۇ ئاخشام ئۆيگە قايىتماپتۇ،ئەمسە ئۇ ھازىر نەدە،ئۇنىڭ ئاپىسىغا خەۋىرنى ئالساملا تېلفۇن قىلىمەن دەپ قويۇپ،تېلفۇننى قويىۋەتتىم،چۈشكىچە ئۇنىڭدىن سادا يوق،،سائەت 12يېرىمدىن ئاشقاندا تېلفۇن قىلسام تېلفۇنى ئۇلاندى،لېكىن ئۇ ھېچ ئىش بولمىغاندەك تېلفۇننى ئالدى:
-ھە دىلنازخان ،تاماق يىدىڭىزمۇ؟دەپ سورىدى ئۇ ،ئۇنىڭدىن بىر گەپ ئېلىپ بېقىش ئۈچۈن،مەنمۇ چاندۇرماي سورىدىم
-ئاخشام ئەجەپ كەلمىدىڭىزغۇ،سىزنى ساقلاپ بولالماي ،ئۆزۈم قايتىپ كەتتىم،
-ھە ،كەچۈرۈڭ جېنىم،ئاخشام بىزنىڭ ئىشخانىدىكى ئالىم مەس بولۇپ قاپتىكەن،شۇنى ئۆيىگە ئاپىرۋېتىپ،قارىسام تېلفۇننىڭ توكى تۈگەپتۇ،ئۆيگە بارسام كېتىپ بوپسز،ئەتىگەندە ئاپامنىڭ مىجەزى يوق،دوختۇرغا ئاپىرىپ شۇ ھازىر بىكار بولدۇم،
ئۇنىڭ قىلچە دۇدۇقلىماي تۇرۇپ ئېيىتقان قېپقىزىل يالغانچىلىقىدىن بەكلا ئازاپلاندىم،كۆزلىرىمدىن ياش ئەگىدى،
-بوپتۇ ئەمسە تاماقنى ۋاقتىدا يەڭ ،دەپلا تېلفۇننى قويىۋەتتىم،لېكىن قەلبىم بىر خىل ئازاپ ئىلكىدە پۇچۇلانماقتا ئىدى.
