ئۇ نېمىشقا مېنى ئالدايدۇ،نېمىشقا كېچىچە ئۆيگە قايىتمايدۇ،ئاخشام كېچە نەدە قونغاندۇ؟شۇنداق ئايىقى يوق خىياللار بىلەن كۈننى كەچ قىلدىم،ئىشتىن چۈشەي دەپ تۇرسام ئۇ تېلفۇن قىلىپ ئۆزىنىڭ مېنى تۆۋەندە ساقلاۋاتقانلىقىنى ئېيىتتى،ئېرىنچەكلىك بىلەن تۆۋەنگە چۈشتۈم،ئۇنىڭ چىرايلىرى بىر كۈننىڭ ئىچىدىلا قارىداپ،مەينەتچىلىك يېغىپ كەتكەندەك ھېس قىلىپ ئۇنىڭدىن بىر ئاز يىرگەندىم،ئىككىمىز گەپ سۆزسىزلا شەھەر سىرتىدىكى بىر ئاشخانىدا كۆڭۈلسىز غىزالاندۇق،ئۇنىڭدىن ئاخشام قەيەرگە بارغانلىقىنى سۈرۈشتۈرۈپ ،تازا بىر ئۇرۇشماقچى بولدۇم،لېكىن ئۇ ماڭا راست گەپ قىلامدۇ،ئۇنىڭ ئۈستىگە ئەھۋالنى بىلسەم نېمە پايدىسى،بىز توينىمۇ بېكىتىپ بولغان تۇرساق،يا تويدىن ۋاز كېچىپ يېنىپ كەتكىلى بولمىسا،خەيىر،مۇشۇ بىرقېتىم ئۆتۈپ كەتسۇن،يەنە مۇشۇنداق ئىش تەكرارلانسا،ئاندىن يۈزىگە سالاي دەپ ئويلىدىم،
-ئەجەپ جىمىپ كېتىپسىزغۇ بۈگۈن؟››ئۇ شۇنداق دەپ ئاستا قولۇمنى تۇتتى،لېكىن نېمىشقىدۇر ئۇنىڭ قوللىرى پاكىز ئەمەستەك بىر خىل سەسكىنىش تۇيغۇسى بىلەن،تېزلا قولۇمنى تارتىۋالدىم،ئۇ نېمە بولدى دېگەندەك كۆزلىرىنى يوغان ئېچىپ ماڭا قارىدى.
-سەل مىجەزىم يوق،بېشىم ئاغىرىپ تۇرىدۇ،قايتىپ كېتەيلى››دېدىم
-ھە بوپتۇ ئەمسە،ئەسلى كىنو كۆرۈپ كىلەيلى دېمەكچىتىم ،باشقا بىرەر كۈنى بارامىز،
شۇنداق كۆڭۈلسىز غىزادىن كىيىن قايتىش يولىغا چىقتۇق،ئۇنىڭ تېلفۇنى ھېلىدىن ھېلىغا توختىماي سايراپ كەتتى،كىم بۇ دەپ شۇنچە سۈرۈشتۈرگۈم كەلسىمۇ پەقەتلا سورىمىدىم،ئۇ ئاخىرى گەپ سۆزسىز تېلفۇننى ئېتىۋەتتى،شۇ تاپتا شۇنىڭغا ئېنىق ھۆكۈم قىلالايمەنكى ئۇنىڭ بۇ تېلفۇن بىلەن چوقۇم بىر ئىشى بار،ئىچىمنى بىر نەرسە غاجىغاندەك شۇنداق بىئارام بولۇپ كېتىۋاتىمەن،ئۆزۈمگە قانداق تەسەللىي بېرىشنىمۇ بىلمەيۋاتاتتىم،ماشىنا ئىشىك ئالدىغا كېلىپ توختىغاندىلا،خىياللىرىمدىن ئويغىنىپ چۈشمەكچى بولدۇم،لېكىن ئۇ قولۇمنى چىڭ تۇتۇپ تۇرۇپ،كۆزلىرىمگە تىكىلدى،
-كەچۈرۈڭ،ئاپپىغىم،ئاخشام سىزنى يالغۇز قويدۇم،جاپا چەكتىڭىز،ئەمدى ئۇ ئۆينى تازلىماڭ،مەخسۇس ئۆي تازلايدىغانلارنى تەكلىپ قىلايلى،ياخشى ئارام ئېلىڭ،ئەتە كۆرىشەيلى،
ئۇنىڭ گەپلىرىنى ئاڭلاپ تېخىمۇ بىزارلىق ھېس قىلدىم ۋە قولىنى سىلكىۋېتىپلا ماشىندىن چۈشۈپ كەتتىم،
ئۆيگە كىرىپ ھېچ كۆزلىرىمگە ئۇيقۇ كەلمەيىتتى،ئۇنىڭ بىلەن بولغان ئارلىقىم يىراقلىشىپ كېتىۋاتقاندەك،ئۇنىڭدىن رايىم قايىتقاندەك ھېس قىلىشقا باشلىدىم،دېمەك ئۇنىڭ شۇ نامەلۇم ئىشلىرىنى مەن كەچۈردۈم،لېكىن قەلبىم كەچۈرەلمەيۋاتاتتى،،،
  كۈنلەر شۇ تەرىقىدە ئۆتىۋەردى،ئۇ ئىشلارمۇ خېلىلا ئېسىمدىن كۆتۈرۈلۈپ قالدى،،تويىمىزغا 15 كۈنلا ۋاقىت قالغان بولغاچ شۇنداق ئالدىراش بولۇپ كېتىۋاتاتتىم،بۈگۈن شەنبە،ئەتىگەندە ئەنۋەرگە ئۆينىڭ پەردىلىرىنى ئېسىپ،بىرەر قۇر تازلىۋېتىپ كېلەيلى دەپ تېلفۇن قىلدىم،لېكىن ئۇ‹‹ئۆيدە چوڭلارغا ياردەملىشپ ئەكىلىدىغان نەرسىلەر بار ئىدى،مەن چۈشتىن كىيىنرەك بىكار بولىمەن ››دېدى،ئۇنىڭغا قاراپ تۇرسام ۋاقىت بەك ئۆتۈپ كەتمىسۇن دەپ،ئۆزۈملا كېلىۋەردىم،ئۆيدىكى ئۇششاق چۈششەك نەرسىلەرنى رەتلىدىم،ئاشخانا ،تازلىق ئۆيلىرىنى دېزىنفېكسىيىلەش سۇيۇقلىقىدا قايتا سۈرتۈپ تازلىدىم،تۇيۇقسىز تېلفۇنىم جىرىڭلىدى،يۈگۈرۈپ بېرىپ تېلفۇننى ئالدىم،تېلفۇن قىلغىنى مېنىڭ ئەڭ يېقىن جان جىگەر دوستۇم ساباھەت ئىدى،
-ۋەي ،ساباھەت،ياخشىمۇ؟
-ياخشى ئەمەس مەن،نەدە ئۆزۈڭ؟
-ۋاي نېمە بولدۇڭ دوستۇم،مەن يېڭى ئۆيدە ئۆي تازلىغىلى كەلگەن،
-ۋاي ئەخمەق جېنىم،سەن ئاشۇنداق جاپا تارتىپ ئۆي تازلاپ يۈرىۋەر،ئەنۋەر بايۋەتچى قىزلارنى قولتۇقلاپ كوچىدا يۈرسۇن ھە؟
ئۇنىڭ گېپىنى ئاڭلاپ جايىمدا ئولتۇرۇپلا قالدىم
-نېمە دەۋاتىسەن ساباھەت،ئېنىقراق گەپ قىلە؟
-ئەسلى ساڭا دېمەي دېگەنتىم،لېكىن پەقەت كاللامدىن ئۆتمىدى،بايا قارامايدىن نەۋرە ئاكام كەپتىكەن،شۇنىڭ بىلەن تاماق يىگىلى ھەرەمباغقا بارغان،چىقىۋېتىپ غىل پاللائەنۋەر كۆزۈمگە چېلىقىپ قالدى،يېنىدا بىر ئالۋاستى بار،شۇمۇ ئەمەسمۇ دەپ قاراپ تۇرسام دەل ئەنۋەرنىڭ ماشىنىسىغا چىقىپ كېتىپ قالدى،شۇڭا ساڭا تېلفۇن قىلىشىم،بىرەر نەۋرە ئاچا سىڭىللىرى بولۇپ قالسا خاتا چۈشەنمەي دەپ،ۋەي،دىلناز،دىلناز،،بامۇ سەن ئاداش؟
كۆزلىرىمگە كەلگەن ياشلارنى زورىغا يۈتۈپ ساباھەتكە جاۋاپ قايتۇردۇم،
-بىلمىدىم ئاداش،بەلكىم تۇغقانلىرىنىڭ قىزلىرىمىكىن،مەن تېلفۇن قىلىپ باقاي،دەپلا تېلفۇننى ئۈزۈۋېتىپ ئەنۋەرگە تېلفۇن قىلدىم،ئۇ تېزلا تېلفۇننى ئالدى،
-نەدە سىز،،،زەرىد بىلەن سورۇدۇم
-مەن چوڭلار بىلەن بىللە،سىرىتتا،ئىشىم تېخى تۈگىمىدى،ئىشىمنى تۈگىتىپ بارىمەن
-شۇنداقمۇ،بوپتۇ ئەمسە،،
تېلفۇننى قويىۋېتىپ ئۆزۈمنى كىرىسلۇغا تاشلاپ يىغلىۋەتتىم، ئۇ نېمىشقا ماڭا بۇنداق مۇئامىلە قىلىدۇ؟زادى نېمىشقا؟قوللىرىم دېزىنفېكسىيىلەش سۇيۇقلىقىدا يېرىلىپ ،ئېچىشىپ ئاغىرىپ كېتىۋاتىدۇ،لېكىن ئۇچۇ،ئۇ سىرىتتا كۆڭۈل خوشى قىلىپ يۈرۈپتۇ،ئەگەر ساباھىەتنىڭ كۆرگىنى توغرا بولسا ئۇ قىز زادى كىم؟مەن ھازىر تويغا ئاز قالدى دەپ ئەمەس،ئۇنىڭ بىلەن توي قىلسام بولامدۇ دېگەننى ئويلىششىم كېرەك ئىدى،،قىلىۋاتقان ئىشىمنى تاشلاپ سومكامنى ئېلىپ سىرىتقا چىقتىم،ۋە ساباھەتنىڭ يېنىغا يۈرۈپ كەتتىم،
ساباھەتنىڭ يېنىغا كېلىپ ئۇنىڭغا سوئال سوراقلارنى ياغدۇردۇم،ساباھەت ئېرىنمەي جاۋاپ بەردى،ساباھەتنىڭ سۆزلىرىدىن قارىغاندا ،قىزنىڭ ئەنۋەرنى قولتۇقلاپ ماڭغىنىغا قارىغاندا ئىككىنىڭ مۇناسىۋىتى ئادەتتىكىچە مۇناسىۋەت ئەمەس ئىدى،
-قانداق قىلىمەن ئاداش،تويغا ئاز قالغان تۇرسا دېدىم ۋە ئىلگىرىكى بولغان ئىشلارنىمۇ سۆزلەپ بەردىم
-ۋاي ئەخمەق قىز،تويغا ئاز قالسا نېمە بوپتۇ،ئۇنىڭ بۇ ئىشلىرىنى تويدىن كىيىن بىلگەن بولساڭ قانداق قىلاتتىڭ،بىز چوقۇم بۇنى سۈرۈشتۈرىشىمىز كېرەك،سەن خاتىرجەم بول،ئىشلار راستىنلا بىز ئويلىغاندەك بولسا ،چوقۇم ئاشكارلىنىدۇ،
ئۇ شۇنداق دەپ كۆڭلۈمگە تەسەللىي بەردى،ئۆيگە قايتىپ كېلىىپ كەچكىچە ئۇنىڭ تېلفۇننى ساقلىغان بولساممۇ،تېلفۇن قىلمىدى،،ئۇنىڭ قىلغانلىرى ماڭا بەكلا ئېغىر كېلىۋاتاتتى،بەلكىم تېلفۇن قىلىپ يەنە بىرەر باھانە سەۋەپ تاپقان بولسا،مەن ئەخمەق يەنىلا ئۇنى كەچۈرەر بولغىيتىم.توۋۋا ئۇ مۇشۇنداقلا ئۆزگىرىپ كېتەرمۇ؟چېكى يوق خىياللار بىلەن ئۇخلاپ قاپتىمەن،كېچە چۈشۈمدە ئەنۋەرنى كۆردۈم،ئىككىمىز بىر باغدا قول تۇتۇشۇپ خوشال خورام ئويناپ يۈرۈپتۇق،ئۇ ماڭا شۇنداق چىرايلىق بىرتال ئەتتىرگۈل ئۈزۈپ سوۋغا قىپتۇ،لېكىن مەن گۈلنى قولۇمغا ئېلىپ ئەمدىلا پۇراي دىيىشىمگە شامال چىقىپ گۈل بەرگىلىرىنى ئۇچۇرتۇپ كېتىپتۇ،ئۇنىڭغىچە شۇنداق كۆرۈمسىز،بەدبەشىرە بىر ئايال ئەنۋەرنىڭ يېنىدا پەيدا بوپتۇ،مەن قورىققىنىمدىن ئەنۋەرنىڭ بىلىكىدىن چىڭ قۇچاقلىۋاپتىمەن،لېىكىن ئۇ ھېلىقى كۆرۈمسز ئايال بىلەن مڭىپلاكۆزدىن غايىپ بولۇپتۇ،چۆچۈپ ئويغىنىپ سائەتكە قارىسام كېچە 1دىن ئېشىپتۇ،ئۈستى بېشىم چىلىق چىلىق تەرگە چۆمۈپ كېتىپتۇ،نېمە چۈش كۆردۈم مەن،ئەجىبا ئۇنىڭغا بىرەر ئىش بولمىغاندۇ،،قولۇمغا تېلفۇننى ئېلىپ ئۇنىڭ نومۇرىنى باستىم ،لېكىن تېلفۇنى ئېتىك ئىدى،ئورنۇمدىن تۇرۇپ بىر ئىستاكان قايناق سۇ ئىچىپ قايتا كىرىپ ياتتىم.
دۇنيادا ھەممە ئازاپقا چىدايمەن،لېكىن ئاتا ئانامنىڭ كۆزىدىن ئاققان بىر تامچە ياشقا ھەرگىز چىدىمايمەن
