ئۇ نېمىشقا مېنى ئالدايدۇ،نېمىشقا كېچىچە ئۆيگە قايىتم ...

بۈگۈن شۇنداق ماغدۇرسىزلىنىپ ئورنۇمدىن تۇرۇپتىمەن،بېشىم ئاغىرىپ شۇنداق مىجەزىم يوق،بۈگۈن گەرچە دەم ئېلىش بولسىمۇ ئىدارىدە نۆۋەتچىلىك ئورۇنلاشتۇرۇپ قويغاچقا بارمىسام بولمايىتتى،ئامال يوق ئېغىر قەدەملىرىم بىلەن ئىشخانىغا كەلدىم،كاللامدا يەنە تۈنۈگۈنكى خىيال،يەنە شۇ سوئال،،بولدىلا ئويلىماي دەپ ئەمدىلا كومپىيوتۇرنى ئېچىپ تۇرسام،تېلفۇنۇمغا رەڭلىك ئۇچۇر كەلدى،ئاھ خۇدا كۆزلىرىمگە ئىشەنمەي قېلىۋاتاتتىم،مەن نېمىنى كۆردۈم،رەسىم3پارچە بولۇپ،ئىككىسى ئەنۋەرنىڭ يېرىم يالىڭاچ ياتقان ھالىتى،يەنە بىرسى ئۇنىڭ بىلەن يەنە بىر قىزنىڭ بىللە چۈشكەن رەسىمى ئىدى،ئەۋەتكەن نومۇرىغا قارىدىم،ناتۇنۇش نومۇر ئىدى،كۆزلىرىمدىن بۇلاقتەك ئېتىلىپ چىققان كۆز ياشلىرىم تېلفۇن ئېكراننى ھۆل قىلىۋەتتى،بۇ زادى نېمە ئىش،كۆزلىرىمگە پەقەتلا ئىشەنمەي قېلىۋاتاتتىم،كەينىدىن يەنە بىر ئۇچۇر كەلدى،ئۇچۇر مەزمۇنى بولسا،ھەقىقىي ئەھۋالنى بىلمەكچى بولسىڭىز بۇ يەرگە كېلىڭ دەپ يېزىلغان بولۇپ،ئاستىدا بىر مېھمانخانىنىڭ ئادىرسى،ياتاق نومۇرى يېزىلغان ئىدى،كۆز ئالدىم قاراڭغۇلىشىپ نېمە قىلارىمنى بىلەلمەيلا قالدىم،توغرا ساباھەت!ئېسىمنى يىغىپ ساباھەتكە تېلفۇن قىلدىم،ۋە بولغان ئىشلارنى دەپ يىغلاپ كەتتىم.
-جىنىم دوستۇم،ئۆزەڭنى تۇتىۋال،ھازىر يىغلايدىغان چاغ ئەمەس بۇ سەن ئۈچۈن پولاتتەك پاكىت،سەن ھازىرلا ماشىنا توسۇپ ئۇ يەرگە بار،مەنمۇ ھازىرلا ئۆيدىن چىقىىمەن،ئىككىمىز شۇ يەردە كۆرىشەيلى،تېز بول،بولمىسا ئۇلار كېتىپ قالىدۇ،››دەپ يول كۆرسەتتى،ساباھەتنىڭ گېپى بويىچە تېزلا ئۇ مېھمانخانىنىڭ ئالدىغا كەلدىم،مەن كېلىپ 5مىنۇتقا قالماي ئۇمۇ يېتىپ كەلدى،لېكىن مەن ئىچىگە كىرىشكە جۈرئەت قىلالمايۋاتاتتىم،كىرسەملا ھەممە ئىش تۈگەيىتتى،لېكىن كىرمىسەم تېخىمۇ بولمايىتتى،بۇ ھاياسىز ئاشىقىمنىڭ ھايۋانلارچە قىلىقلىرىنى ئۆز كۆزۈم بىلەن كۆرمىسەم ئىشەنمەيىتتىم،بىز ئۇچۇردىكى ئادرىس بويىچە ياتاق ئالدىغا كەلدۇق،يۈرىگىم ئۆزەمگە ئاڭلانغۇدەك ئەنسىز سوقۇپ كېتىۋاتاتتى،ساباھەت بېرىپلا ئىشىك قوڭغۇرقىنى باستى،ياتاق ئىچىدىن ياش بىر قىزنىڭ كىمۇ دېگەن ئاۋازى ئاڭلاندى،ساباھەتمۇ قىلچە چاندۇرماي،مەن كۈتكۈچى بىر نەرسە ئەكىرگەن،دەپ جاۋاپ بەردى،نەچچە سىكىنۇتلاردىن كىيىن ياش بىر قىز ئۇخلاش كىيمى بىلەنلا چىقىپ ئىشىكنى ئاچتى،كۆز ئالدىمدىكى بۇ قىز 25-26ياشلار چامىسىدىكى،چاچلىرى خۇددى سۈپۈرگىدەك تۈز،كۆزلىرى سەل كۆك،تېرىسى ئاپئاق،بويلىرى ئېگىز تولىمۇ ئۇرۇق،بىر قاراشقا باشقا بىر مىللەتنىڭ قىزلىرىدەك تەسىر بېرىدىغان ئىدى،قىز بىزنى كۆرۈپ ئانچە ھەيران قالمىدى ۋە مېيىقىدا كۈلۈپ قويدى،ساباھەت قىزنى ئىتتىرۋېتىپ قولۇمدىن سۆرەپلا ئىچىگە كىردى،دىلنازخان ئەمدى ئىشەنمىسە بولمايىتتى،تويغا پۈتۈشكەن ئاشىقى دەل كۆز ئالدىدا ،باشقا يات قىزلار بىلەن خالىي جايلاردا ئىدى،ئەنۋەر بىزنى كۆرۈپ چاچىراپ ئورنىدىن تۇرۇپ كەتتى،نېمە دەپ چۈشەندۈرىشىنىمۇ بىلمەي ئالدىمدا تۇرۇپ قالدى
-ۋۇ ئىپلاس،ئىسىت ساڭا قىلغان ئەقىدەم،ئاللاھ ئالدىدا خار بولغايسەن،،شۇنداق دېدىم ۋە يۈزىگە بىر تەستەك سېلىپلا كەينىمگە ئۆرۈلۈپ چىقىپ كەتمەكچى بولدۇم،
-مېنى كەچۈرۈڭ ،دىلناز،مەن ئېزىپتىمەن،مۇشۇ بىر قېتىم،ئۇ قوللۇرۇمغا ئېسىلىپ ئۆتۈنۈپ يىغلايىتتى،لېكىن مەن ئۇنداق ئىپلاسلارنى كەچۈرۈشكە ئامالسىز ئىدىم،چۈنكى ئاللاھلا ھەقىقىي كەچۈرەلەيدۇ،ئۇنى مەن ئەمەس ئاللا كەچۈرسۇن،كىيىنكى ھېچقانداق گەپلەر خىيالىمغا كىرمىدى،بىر يەرلەردە ساباھەت ھېلىقى قىزنى تىللاپ سوقۇشۇپ كېتىۋاتىدۇ،يەنە بىر يەرلەردە ئەنۋەر بىر نېملەرنى دەپ مەندىن ئۆتۈنىۋاتىدۇ،لېكىن ھېچقايسىسى قۇلۇقۇمغا كىرمىدى،ئوقتەك يۈگۈرۈپ سىرىتقا يېنىپ چىقتىم،كۆزلىرىمگە ئەمدىلىكتە ياشمۇ كەلمەيۋاتاتتى،كەينىمدىن يۈگۈرۈپ چىققان ساباھەت مېنى چىڭ قۇچاقلاپ تۇرۇپ :
-جېنىم دىلناز،يىغلىۋال،بۇنداق قىلما،بۇنداق ئىچىڭگە ئالساڭ تۈگىشىپ كېتىسەن،يىغلىۋال ئاداش،تازا بىر يىغلىساڭ ئوڭشىلىپ قالىسەن،
-ياق ئاداش،بايا كۆرگەنلىرىم ھەممىسى يالغان شۇنداقمۇ؟مەن خاتا كۆرۈپ قالدىم،ئۇ ئەنۋەر ئەمەس،ئەنۋەر ھەرگىز ماڭا بۇنداق يۈزسىزلىك قىلمايدۇ،،بىز خاتا كۆرۈپ قاپتۇق ساباھەت،چوقۇم خاتا كۆرۈپ قاپتۇق،
-ياق ئاداش رېئاللىققا تەر بەر،،بۇنداق قىلما،،،،ئۇ شۇنداق دەپلا ماڭا ئېسىلىپ يىغلاپ كەتتى بۇ يازمىنى ئاخىرىدا   dilnaz911 تەھرىرلىگەن. ۋاقتى  2014-5-15 06:01 PM
