ئەسسالامۇ ئەلەيكوم كەڭ مىسرانىم ئىشتىراكچىلىرى ھازىرقى ۋاقتىڭلار خەيرىلىك بولسۇن . پىقىردىن ھەممىڭلەرگە ئوتلۇق سالام. جاھاننىڭ ئۇچقاندەك تەرەققى قىلشى ، پەن تىخنىكىنىڭ راۋاجلىنشى بىلەن ئادەمنىڭ بىلىش تەرەققىياتىمۇ ئانچە مۇنچە ئۆسۇپ قالدىكەن. ئادەمنىڭ بىلگەن نەرسىسى قانچە كۆپ بولغانسىرى ئۆزىنىڭ بىلمەيدىغان نەرسىسىنىڭمۇ خىلى كۆپ ئىكەنلىكى ئاشكارلىندىكەن . ئەمدى مەن شۇ كەم بىلمىمنى كۆپچىلىك بىلەن ئورتاقلىشىش ،بىلمىگەنلەرگە ئازراق بولسىمۇ بىلدۇرۇش نىيتىدە ئىكران ئالدىدا مۇنۇ قۇرلارنى يىزىپ ئولتۇرۇپتىمەن . تىمىنى دۇنيانى زادى كىم كونتىرول قىلۋاتىدۇ دەپ باشلىدىم . ئەمدى سەل دۇنيانىڭ ئىشلىردىن خەۋرى بار قىرىنداشلار ئاللىقاچان بۇ سۇئالغا جاۋاپ تىپىپ بولدى . باشقا كۆپ ساندىكى قىرىنداشلىرىم بۇ سۇئالغا جاۋاپ ئىزدەۋاتقان بولشى ياكى ئازراق بولسىمۇ ئويلاپ ئامىرىكا ،ئەنگىليە دەپ پەرەز قىلۋاتقان بولشى مۇمكىن. ئەمدى سىلەردىن سۇئال سوراپ باقاي .قايسىڭلار قولۇڭلارغا ئامىرىكىنىڭ دوللىرنى ئىلىپ سىنچىلاپ قاراپ باققان . بۇنى سوراپ نىمە قىلدىغاندۇ دەپ ھەيران قالماڭلار تىمىدىكى سۇئالغا شۇ ئامرىكا دوللىرنىڭ كەينىدىكى سۇرەت جاۋاپ تاپالايدۇ. مانا ھازىر قولۇمدا بىر ئامىرىكا دوللىرى تۇرۇپتۇ . سىلەرگە بۇ پۇلنىڭ كەينىدىكى سۇرەتنى تەسۋىرلەپ بىرەي .سۇرەتتە بىر ياسىلۋاتقان ئەلىھرام لىكىن ئۇنىڭ پىرامىدا ئۇچى تىخى پۇتمىگەن . ئەلىھرامنىڭ ئۇستىدە بىر كۆز بار. بۇ نىمنى چۇشەندۇردۇ دەپ گاڭگىراتامسىز؟؟ئەمدى مەن بىر بىرلەپ چۇشەندۇرەي .كەچرۇڭلار يەنە سۇئال قويۇپ چۇشەندۇرۇش بىردىغان بوپ قالدىم .بىر كۆزنىڭ قانداق مەنە بىردىغانلىقىنى بىلەمسىز؟ ماسون تەشكىلاتى توغرىسدا قانچىلىك مەلۇماتتا؟؟ھەببەللى بىر كۆز يەھۇدى دىنى ۋە خىرىستىئان دىندا تەڭرىنىڭ سىزنى كۆرۇپ تۇرۋاتقانلىقىنى ئىپادىلەيدۇ . بىر كۆز دەل ماسون تەشكىلاتىنىڭ بەلگىسى.ماسون تەشكىلاتى دۇنيادىكى ئەڭ چوڭ ئەڭ سىرلىق ئەڭ مەخپى تەشكىلات .ئىنگىلىسچە Freemasonryدەپ ئاتىلدۇ. ئۇيغۇرچىغا تەرجىمە قىلساق ئەركىن تاشچىلار تەشكىلاتى. بۇ تەشكىلاتنىڭ تارىخنى 18-ئەسىردىن باشلاپ سۇرۇشتۇرۇشكە توغرا كىلدۇ . ئۇلار تاش تۇپراق ئارقىلىق دۇنيانى ئۆزگەرتكىلى بولدىغانلىقىغا ئىشەنگەچكە ئىسمىنى تاشچىلار دەپ قويغان .شۇڭا بايرىقىدىمۇ سىزغۇچ بىلەن سىركول چۇشرىلگەن . ئۇلار نەزىريە جەھەتتىن دىنسىز تەشكىلات بولسىمۇ ، شەيتان دىننى يەنى يەھۇدى دىننى تەرغىپ قىلدۇ . بۇ دىندىن باشقا ھەممە دىنلارنى يوقتىشقا كۇچ چىقىردۇ . ماسون تتەشكىلاتنىڭ 6يۇزمىڭ ئەزاسى بولۇپ،100مىڭى ئەنگىليەدە،400مىڭى ئامرىكىدا70مىڭى فىرانسىيەدە.تارخىتا سوۋىت ئىتپاقى ، ئىتالىيە ياپۇنيە قاتارلىق دۆلەتلەر بۇ تەشكىلاتنى چەكلىگەن. ھازىرمۇ كۆپ قىسىم مۇسۇلمان دۆلەتلەر دە بۇ تەشكىلاتنى چەكلەش دولقۇنى بار.1962-يىلى جوڭگو چوڭ قۇرۇقلىقىدىن چىكىنىپ شىياڭگاڭ تەيۋەنلەردە تەرەققى قىلغان. ئۇلارنىڭ پىلانى دۇنيانىڭ يىڭى تەرتىپىنى ئورنۇتۇش، يەھۇدى دىندىن باشقا ھەممە دىنلارنى تازلاش ، ھەممىدىن رەزىلى دۇنيا نوپۇسىنى 500مىليونغا چۇشۇرۇش . ئۇلارنىڭ قارشىچە يەرشارىدا 500 مىليوندىن باشقا نوپۇس ئەخلەت نوپۇس . شۇڭا ئۇلار بۇ پىلاننى نوپۇسى ئەڭ كۆپ دۆلەتلەرگە قاراتتى . كاللىڭىزغا ئەڭ ئاۋال قايسى دۆلەت كەلدى . ئەلۋەتتە بىزنىڭ جوڭگو. ئۇلار بۇ ئويۇننى ناھىيتى ئەپچىل ناھىيتى ئۇستىلىق بىلەن ئورۇنلىدى 2002-يىلدىكى سارىس دەل بۇ تەشكىلاتنىڭ جوڭگوغا تاشلىغان سىناق بومبىسى. 11-سىنتەبىر ۋەقەسى، ئوتتۇرا شەرىق ئۇرشى .ۋەھاكازالار .داڭلىق ناخشىچى  مايكول جىكسون بۇ تەشكىلاتقا ئاشكارە قارشى چىققانلارنىڭ بىرسى بولۇپ ، ئۇ بىر قەھرىمان ئەمدى ئۇنىڭ سىرلىق ئۆلمىنى ئويلاڭ .بۇ تەشكىلاتنىڭ ئەزالىرى ئادەتتىكى ئادەملەر ئەمەس بولۇپ ،ئامرىكا ،ئەنگىليە ،ئىسرالىيە قاتارلىق زومىگەر دۆلەتلەرنىڭ يۇقرى دەرجىلىك باشلىقلىرى.داڭلىق مىليادىرلار(بىلگايتىس دەل بۇ تەشكىلاتنىڭ غوللۇق ئەزاسى) . ئۇلار ئىنتايىن مەخپى ھەركەت قىلدۇ .ئەڭ يۇقىردىكى ئوننەچچە ئادەملار يۇقرى دەرجىلىك يىغىن ئىچىپ باشقىلارغا بۇيرۇق قىلدۇ. ئەمدى ھىلقى بىر دوللارنىڭ كەينىدىكى رەسىمگە كەلسەك ، ئامىرىكا ھوكۇمىتىڭ پۇل بىسىش، تارقىتىش ھوقۇقى يوق .ئامرىكا دوللىرنى بىسىشقا ئامىرىكا پۇل بىسىش ئورگننى بىۋاستە ھوقۇقلوق. شۇڭا بۇ ئورگاندا ماسون تەشكىلاتنىڭ قولى بار دىگەن گەپ. ھىلقى تىخى پۇتمىگەن پىرامىدانىڭ نىمنى ئىپادىلەيدىنى تىخى سىر.ماسون تەشكىلاتى ئىسلام دىنىنى پۇتۇن كۇچى بىلەن يوقۇتۇشقا كۇچ چىقىردىغان بولۇپ مۇنۇ ۋاستىنى ئىشىلتىدۇ، ئۇلار ئاۋال ساختا مۆجىزلەرنى ئويدۇرۇپ چىقىردۇ ،ئاندىن ئۇنى كەڭ تور مىدىيالارغا تارقىتىدۇ،ئاندىن بىر مەزگىلدىن كىيىن ئۇنىڭ ساختىلقىنى ئاشكارلايدۇ .بۇ قىلمىش نۇرغۇنلىغان مۇسۇلمان قىرىنداشلىرمىزنى گاڭگىرتىپ قويماقتا ،شۇڭا كەڭ قىرىنداشلارنىڭ بەزى <مۆجىزلەر>گە ئالدىنىپ قالماسلىقىنى سورايمىز . بىلەمسىز دۇنيادىكى تۇنجى سىرىق فىلىم يەھۇدى ،ماسون تەشكىلاتنىڭ رەزىل ئوينى .ئۇلار بۇ ئويۇننى ئاۋال ئىسلام دۇنياسىغا كىرگۇزىشكە ئالدىراپ كەتمىدى ،ئاۋال ياۋروپا ئامىرىكىلاردا تەرەققى قىلدۇرۇپ ،ئوڭايلا ئەرەپ دۇنياسىغا كىرگۇزۋەتتى. بۇ نىجىسنىڭ ھازىر قانچىلىك تارالغانلىقىغا بىر نىمە دىيىش تەس . ھازىر دۇنيادا بىكاردىن نەچچە ئونمىڭلىغان بىگۇناھ نۇپۇس ئۇرۇش ۋىروسلارنىڭ كاساپىتىدىن ئۆلۇپ كەتمەكتە. بۇلاردا ماسون تەشكىلاتنىڭ بىۋاستە قولى بولمىسىمۇ ،سىياسى نەيرەڭدە رول ئوينايدۇ .ئەمدى خۇلاسە كالام ،ئويلاڭ قىرىندىشىم مانا بۇ دۇنيا سىياسىسىنىڭ بىر كىچىك كارتىنىسى . بىز بۇ چوڭ شاخمات تاقتىسنىڭ يىيلىشنى كۇتۇپ تۇرغان پىچكىسى، سىز بۇ گەپكە سەل قارماڭ .نەچچە يىل ئىلگىركى جوڭگودىكى ئەڭ چوڭ يىمەكلىكتىن زەھەرلىنىش دىلوسى سەيلۇ سۇت پاراشوكىدىكى زەھەرلىك ماددا گۇڭگۇرۇت مىقدارنىڭ پەۋقۇلادە يۇقرى بولشى بۇمۇ بىر پاجىئە . ئەمدى بۇنىڭ كىم بىلەن مۇناسىۋەتلىك ئىكەنلىكىنى ئىيتىپ ئولتاماي. بىلمىمنىڭ كەمچىلكى ، يىزىقچىلق قابىليىتىمنىڭ تۆۋەنلىكىدىن تىمامدا ئازدۇر كۆپتۇر خاتالىق بار بولشى مۇمكىن .كەڭ تورداشلارنىڭ ئىجابى ھالدا تۇزتۇپ چۇشنىپ چۇشەنچەڭلارنى ،كۆز قارشىڭلارنى ئوتتۇرغا قويشۇڭلارنى ئۇمىد قىلمەن.

يېرىم يىل بۇرۇن شىنجاڭدىكى شىركەت ئېچىپ يۈرگەن بىر نەچچە چەتئەللىك بىلەن پاراڭلىشىش جەريانىدا  ماسون دىگەن تەشكىلات بامۇ زادى  دىسەم  بار دىگەن....

ۋاي ھەممىڭلار ماسوننى بىلىدىكەنسىلەرمۇنىمە.ماسون تەشكىلاتىنى مەن يەنە نۇرچىلار تەشكىلاتى يەنى نۇر تەشكىلاتى دەپمۇ ئۇقۇغاندەك قىلغان بىر چاغلاردا.بىراق قىزىقماپتىكەنمەن.دىمەك دۆلىتىمىزدىكى 2003- يىللىرى باشلانغان سارىس قانداقتۇر يەيدىغان تائىپىدىكىلەرنىڭ مايمۇننى يەپ تاپقان كىسىلى ئەمەسكەندە.ئۇلارنىڭ ياۋرۇپا سان ئانگىلو پىلانى دەپ بىر پىلانى بارلىقى قەيىت قىلىنغان سابىق مۇنبەردە.ئىسىمدە ھازىر بەك قالماپتۇ.ئۇلار يەنى شۇ تىرورچى تەشكىلات ماسونلارنىڭ ئىنسانىيەت نوپۇسىنى بۇ دەرىجىدە تازلىۋىتىش پىلانى بەك رەزىل ئىكەن.بەش يۈز مىليۇن نوپۇس دىگەن پەقەت شىمالى ئامېرىكا ۋە قىسمەن ياۋرۇپا دۆلىتى ۋە يەھۇدى قىساسچىلىرىدىن باشقىلارنى بىردەك قىرىشنى كۆرسىتىدۇ.يەھۇدىيلارنىڭ مىنىڭچە ھەممىسىلا بۇ تەشكىلاتقا ئەزا ياكى قوللايدىغانلىرى ئاز.كۆپىنچىسى بىرلىكتە تەرەققى قىلغان يەرشارىدا ياشاشنى ئۆمۈد قىلىدۇ.يامانلىق ئويلىغۇچىلارنىڭ ھامان يولى ھامان راۋان بولمايدۇ.ماسونلارنى روسلارمۇ چەكلىگەن بولسا ئوبدان بوپتۇ.ئامېرىكىدا ماسونلارنىڭ ئادەملىرى بولسىمۇ ئۇلار يۈگۈرۈپلا بىر ئىش تەۋرىتەلمەيدۇ.خەلىق بىسىمى يامان يەر ئۇ.قورال ياراغ كەڭ تارالغان.ماسونلار نىشانىنى نوپۇس كۆپ دۆلەتلەرگە قاراتسا بۇ نىشان دۆلىتىمىز يەنە ھىندىستان قاتارلىق دۆلەتلەر بولىدۇ. قارا نىيەتنىڭ نىيىتى ئەمەلگە ئاشماس.ھەممىنى بىلگۈچى ئاللاھ.ئامېرىكا دوللىرىدىكى بىر كۆز يىنىدا بىز خۇداغا ئىشىنىمىز دىگەن گەپ بار دىگەندەك بىر مەزمۇننى بىر تىمىدىمۇ بىر يەردە ئوقۇغان ئىشقىلىپ.بۇنىڭغا چەتئەلدىكى دوللار خەجلەۋاتقانلارمۇ پىكىر قىلسا ئەلۋەتتە خالايدىغانلىرى سۆزلەپ بەرسە تىخىمۇ ياخشى بولاتتىكەن.مەن بۇ تەشكىلاتنى تارقىلىپ كەتتى دەپ ئويلايمەن.پۇلنىڭ ھەر بىر بەلگىسىگە دىققەت قىلغىلى تۇرساق خەلىق پۇلىدىمۇ رۇبكىدىمۇ لىرادىمۇ يىندىمۇ ۋوندىمۇ باخىتتىمۇ بار گۇمان قىلىپ باققۇدەك رەسىملەر.بىر چاغدىمۇ بىرى مۇشۇ دوللاردىكى كۆز تۇغۇرلۇق نىر نەرسە يىزىپتىكەن بىرەيلەن مۇشۇ خەلىق پۇلىدىمۇ ئىسرائىلنىڭ بەلگىسى بار دىققەت قىلسا بىلىندۇ دىدى.قىزىقچىلىقتا ئىيتتىمۇ بىلمىدىم.بۇنى ئاڭلاپ ياندىن پۇل چىقىرىپ قارايمەن قارايمەن ئىسرائىلنىڭ نە بەلگىسى بولغاي.شۇنىڭ بىلەن بۇنداق تىمىلارغا قىزىقماسلا بولدۇم.
