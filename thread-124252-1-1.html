                            __ ئەلى تۇرسۇن  سىرلىق

                          گۈلۈم كەتسەڭ يېراق-يېراققا،
                         ئەسلەيدىغان يۈرەك مەندە بار.
                        ئۆلسەممۇ گەر سەن ئۈچۈن بىراق،
                         كەچمەيدىغان يۈرەك مەندە بار.

                          يار يوقتىمۇ پەقەت يار ئۈچۈن،
                          يۈرەكتىكى ئاشىق دىل ئۈچۈن.
                          سۆيگۈ كۈتۈپ غېرىپ تۈنلەردە،
                          ئۆرتەيدىغان يۈرەك مەندە بار.

                            بىر كېتىپلا يېراققا گۈلۈم،
                          كەلمىسەڭمۇ مەڭگۈگە جېنىم.
                           دىدارىڭنى كۈتۈپ ئۈمۈدتتە،
                          ساقلايدىغان يۈرەك مەندە بار.

                             ئۇچۇرۇڭ يوق ئۆتكەن كۈنۈمدە،
                          يۈرەك يىغلار يالغۇز تۈنۈمدە.
                         سۆيگۈم ئۈچۈن كۈندۈز تۈنۈمدە،
                          يىغلايدىغان يۈرەك مەندە بار.

                          سۆيگۈدىكى بۇ دەت پىراققا،
                          كەتسەڭمۇ گەر يېراق-يېراققا.
                          يار ۋەدەڭنى ئەسلەپ ئەبەدكە،
                          چىدايدىغان يۈرەك مەندە بار.

               ئاپتورى : ئۆزەم 2014-يىلى نورۇزنىڭ 27-كۈنى...

                                   نەدە سەن.....
                              ئەلى تۇرسۇن سىرلىق

                           كۆرەلمىدىم بويۇڭنى جانان،
                        خەۋەر بەرگىن ئېيىت نەدە سەن.
                        كۈتۈشلەردىن چارچىدى يۈرەك،
                        تاقەت تامان قايىت نەدە سەن؟
                        
                      چىن-سۆيگۈمنى ئالساڭ قۇلۇڭغا،
                        زار قىلمىغىن مېنى يۇلۇڭغا.
                      كىرگەن بولسام مەن ھەم ئويۇڭغا،
                       ساقلاتمىغىن قايىت نەدە سەن ؟
                     
                         خەۋەر كۈتۈپ يولدا كۆزلىرىم،
                          ئۇيقىدىنمۇ بىدار كۆزلىرىم .
                         يەتكەن بولسا ساڭا سۆزلىرىم،
                        زار قىلمىغىن قايىت نەدە سەن ؟
                        
                   ئاپتورى : ئۆزەم 2014-يىلى 17-ماي

                                 ياشاپ باقاي....
                           _ ئەلى تۇرسۇن سىرلىق
               
                         كەتمەكچىمەن سەندىن يېراققا،
                          تاشلاپ سېنى بۇ دەت پىراققا.
                          چىداپ باققىن سەنمۇ ئازاپقا،
                          ياشاپ باقاي مەنمۇ جانانسىز.
                       
                         بەرمىگەن شۇ سۆيگۈڭنى كۈتۈپ،
                        كېتىپتۇ تالاي ئاي،يىللارمۇ ئۆتۈپ.
                           ماڭا ئەجەل كەلگۈچە يېتىپ،
                          ياشاپ باقاي مەنمۇ نىگارسىز.

                            مۇھاببەتتىن كەلمىدى ۋاپا،
                        سۆيگۈ ئىزلەپ چەكتىم كۆپ جاپا.
                          كۆڭۈللەر بولدى ئورۇنسىز خاپا،
                           ياشاپ باقاي مەنمۇ سېنىڭسىز.
                          
                               ئاتا-ئانام يېنىغا بېرىپ،
                            مېھىر-ۋاپا گۈللىرىنى تېرىپ.
                             ئۇلارمۇ ھەم كەتكۈچە قېرىپ،
                             ياشاپ باقاي دۇئاسى ئېلىپ.

                         ئاپتورى : ئۆزەم 2014-يىلى 17-ماي...

                                كۆيدۈم دەپتىمەن....
                             __ ئەلى تۇرسۇن سىرلىق

                           كۆيەلمەسكە كۆيدۈم دەپتىمەن،
                           سۆيەلمەسكە سۆيدۈم دەپتىمەن.
                           ھەرچاغ ۋەدە بولغاچقا تىلدا !
                           ئايرىلماسقا يۈردۈم دەپتىمەن .

                           شۇنچە ۋاقتىن بىلمەي كەپتىمەن،
                           چىن-سۆيگۈسى بەردى دەپتىمەن.
                           مەن چۈشەنمەي يارنىڭ كۆڭلىنى،
                           چىن-سۆيگۈمنى بېرىپ كەپتىمەن.

                            ئەمدى بىلسەم كۆڭلىنى بۈگۈن،
                            مەن ئويۇنچۇق قونچاق بوپتىمەن.
                             لەيلايىڭمەن دېگەچكە تىلدا !
                           مەن بىلمەستىن مەجنۇن بوپتىمەن .

                             بوپتۇ يارىم يالغان سۆيسەڭمۇ،
                           مەن بىلمەستىن سۆيۈپ كەپتىمەن.
                            يالغان سۆيگۈڭ قوينىدا ئەجەپ !
                            شېرىن خىيال سۈرۈپ كەپتىمەن ؟
