                               1
                      ۋاقىتنى ئىسراپ قىلماڭ

پەيغەمبىرىڭىز ئائىشەگە مۇنداق دېگەن: «ناۋادا گۇناھ قىلىپ سالساڭ، اللەتىن ئەپۇ تىلىگىن، ئۇنىڭغا تەۋبە قىلغىن، چۈنكى، بەندە ئۆز گۇناھىنى تونۇپ تەۋبە قىلسا، اللە ئۇنىڭ تەۋبىسىنى قوبۇل قىلىدۇ.»
تەسەۋۋۇر قىلىپ بېقىڭ، ناۋادا سىز تاپماقچى بولغان بارلىق مال-دۇنياغا ئىگە بولدىڭىز، بارلىق ئارزۇلىرىڭىزغا يەتتىڭىز، پۇتۇن ئۈمۈتلىرىڭىز ئەمەلگە ئاشتى، ئارقىدىن سىز يەنە مۇشۇ نەرسىلەردىن ئايرىلىپ قالدىڭىز، سىز يىغلايسىز، نەپرەتلىنىسىز، ئازاپلىنىسىز. يوقاتقانلىرىڭىز ئۈچۈن ھەسرەت چىكىىپ ئۆزىڭىزنىڭ بارمىقىنى چىشلەيسىز. ئۇنداقتا، سىزنىڭ ھاياتىڭىز ھەتتا سىز ئويلاپ يەتمىگەن بىر ۋاقىتتا تېنىڭىزدىن سۇغۇرۇلۇپ چىقىپ كېتىدۇ. سىز بۇ چاغدا قانداق قىلىسىز؟
   سىزنىڭ ھاياتىڭىز قىممەت، بىباھادۇر. ئۇنى ماددا بىلەن ئۆلچىگىلى بولمايدۇ. ھاياتىڭىزدىكى ھەربىر مىنۇت ئۆتۈپلا كەتسە مەڭگۈ قايتىپ كەلمەيدۇ. بۇ مىنۇت-سىكۇنتلار سىزنىڭ بۇ دۇنيادىكى بايلىقىڭىزدۇر. ئۇلاردىن پايدىلىنىڭ. سىز ئۇلارغا تايىنىپ جەننەتتىكى سىز ئارزۇ قىلغان ھەرقانداق ئېسىل نەرسىگە ئېرىشەلەيسىز. سىز قانداقمۇ ئۆمۈرنى ئىسراپ قىلىپ، ھەقىقىي تەۋبە قىلماي تۇرالايدىغانسىز؟!
   

«بەخىتكە تۇتاشقان بىر يول بار، ئۇ بولسا چامىڭىز يەتمەيدىغان ئىشلار ئۈچۈن ئەنسىرمەسلىك.»

ـــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
« ئۇلارغا قارشى ئاللاھ ساڭا كۇپايىدۇر.» (137 :2)

                            2
             بەخىتنى پۇلغا سېتىۋالغىلى بولمايدۇ

   نۇرغۇن كىشىلەر ياشلىقىنى ۋە ساقلىقىنى سەرپ قىلىپ مال توپلايدۇ. ئاندىن قالغان ئۆمۈرىدە، بەخىتنى سېتىۋېلىش ئۈچۈن بۇ ماللارنى خەجلەيدۇ. ئەمما، ئۇلارنىڭ ئېرىشىدىغىنى پەقەتلا بەخىتسىزلىك بولىدۇ ياكى ئۇلار ياشلىقىغا قايتماقچى بولىدۇ ئەمما قېرىلىق ھامان ئۇلارنى قوغلاپ يېتىشىۋېلىپ، ئالدىغا چىقىپ كېتىدۇ. ئۇلار ساغلاملىقىنى ئەسلىگە كەلتۈرمەكچى بولىدۇ، ئەمما داۋالاپ ساقايتقىلى بولمايدىغان كېسەللىك ئۇلارنى مەغلۇپ قىلىپ تاشلايدۇ.
   مىسىرلىق بىر داڭلىق ئارتىسنىڭ دەپ بېرىشىچە: ئۇ بىر ئۆمۈر ئىزدىگەن نەرسە پەقەتلا مال-دۇنيا ئىكەن. ئۇ پۇل بولسىلا دۇنيادىكى ئەڭ خۇشال، ئۇزۇن ئۆمۈر كۆرىدىغان ئادەمگە ئايلانغىلى بولىدۇ دەپ قارايدىكەن. مۇبادا پۇللا بولسا بارلىق ئارزۇلارنى ئەمەلگە ئاشۇرغىلى بولىدۇ دىگەنگە قەتئىي ئىشىنىدىكەن. ئۇ ئارزۇ قىلغان ھەممە نەرسە ئۇنىڭ قولىغا چۈشۈپتۇ. يىگىرمە يىلدىن كىيىن ، اللە ئۇنىڭغا ئۇ ئارزۇ قىلغاندىنمۇ زور مال-دۇنيا ئاتا قېلىپتۇ. ئەمما، ئۇنىڭ ساقلىقىنى،ياشلىق باھارىنى،شېرىن چۈشلىرىنى ئېلىۋاپتۇ. ئېيتىلىشىچە، ئۇ يىغلاپ تۇرۇپ مۇنداق دىگەن: «كاشكى اللە تىن مال-دۇنيا تىلىمىسەمچۇ، كاشكى اللە تىن مېنى بىر ئۆمۈر نامراتلىق ئىچىدە، دۈملەپ پۇشۇرۇلغان پۇرچاق يەپ، بېلەت پۇلىدىن قېچىپ تىر امۋاينىڭ شوتىسىغا ئېسىلىۋېلىپ تۇرمۇش كەچۈرۈشنى تىلىسەمچۇ...»
   بۇ ئارتىس ساقلىقنىڭ قىممىتىنى پەقەت ئۇنىڭدىن ئايرىلغاندىلا بىلدى. ئۇ مىسىردىكى ئەڭ باي سەنئەتكارغا ئايلانغۇچە، مال-دۇنيانىڭ ھەممە نەرسىنى سېتىۋېلىشقا قۇربىتى يەتمەيدىغانلىقىنى بىلمەيتتى. ئۆلۈم كەلگەندە، ئۇ بارچە مال-دۇنيانى خەجلەپمۇ، ئۆز ئۆمۈرىنى بىركۈنلۈك ئۇزارتالمايدىغانلىقىنى تونۇپ يەتتى.
   
«ئۆمۈرىڭىزنىڭ تەڭ يېرىمىنى بەس-مۇنازىرە بىلەن ئۆتكۈزىۋەتمەڭ.»
  
ـــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
«سەۋر ۋە ناماز ئارقىلىق ياردەم تىلەڭلار» (45 :2)

                                  3
         ئالدىراڭغۇلۇق ۋە تەنتەكلىك بەخىتسىزلىكنىڭ يېقىلغۇسى

سەۋرچانلىق ھاماقەت نەپسىنى يېڭەلەيدىغان بىر جەڭچى. تەمكىنلىك  ئۆزىنى تۇتىۋېلىش، ئالدىراڭغۇلۇق قىلماسلىق، يەتتە ئۆلچەپ بىر كېسىشتىن دېرەك بېرىدۇ. بۇ ئىككى ئالاھىدىلىك غەم-ئەندىشىنى يېڭەلەيدۇ. كىمدە ئۇلار كەم بولسا ، شۇ كۆپلىگەن گۈزەل ئەخلاقلارنى يوقىتىدۇ. نۇرغۇنلىغان دەرت-ئەلەمگە مۇپتىلا بولىدۇ. سەۋرچان ئادەم قەبىھلىك ۋە بەخىتسىزلىكتىن ساقلىنالايدۇ. ئەمما، ھاماقەت، ئالدىراڭغۇ ئادەم ئۆزىنىڭ ئاچچىقىنى كونترۇل قىلالمايدۇ، بەخىتسىزلىكنى كۆپەيتىدۇ، ھەتتا ئەنسىزلىكنىڭ مەنبەسىنى يېىتىلدۈرىدۇ. ئېغىر-بېسىق ئادەم دېگۈدەك پۇشايمان قىلمايدۇ ياكى ئالدىن پەرەز قىلغىلى بولمايدىغان ئىشنى قىلمايدۇ. ئەمما، ئەخمەق ۋە تەنتەك كىشى پۇشايمان بىلەن غەم-ئەندىشىنىڭ يامان ئاقىۋىتىگە كۆنۈپ كەتكەن بولىدۇ. شۇنىڭدەك، بىر ئادەم  تىنىچ، خاتىرجەم روھى ھالەت بىلەن ئۆزىگە ۋە باشقىلارغا مۇئامىلە قىلسا، ئۇ چوقۇم مۇۋەپپىقىيەتكە ئېرىشىدۇ.
   بىز ئىشىنىدىغان ئىسلام دىنى بىزنى چىقىشقاقلىق، سەۋرچانلىق ۋە تەمكىنلىككە چاقىرىدۇ. پەيغەمبەر ئەلەيھىسسالام مۇنداق دەيدۇ: «ھەرقانداق بىر نەرسىدە مۇلايىملىق بولىدىكەن ئۇ گۈزەللىككە ئېرىشىدۇ. ھەرقانداق بىر نەرسىدىن مۇلايىملىق تارتىۋېلىنىدىكەن ئۇ سەتلىشىپ كېتىدۇ.»
   
«بىز دائىم بەخىتلىك چاغلىرىمىزنى بىھۇدە ئىشلارغا سەرپ قىلىمىز.»

ـــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
« ئۇ سىلەرگە دىندا ھېچقانداق مۈشكۈللۈكنى قىلمىدى.» (78 :22)

          بايلىق يىغىش ئويۇنىنىڭ ئاخىرى چىقمايدۇ

بىفېر برۇك مۇنداق دىگەن ئىكەن: «مەن نۇرغۇن بايلىق توپلاپتىمەن، ئەمما،  تەجىربىلىرىمدىن شۇنى تونۇپ يەتتىمكى، ئاشۇ بايلىق توپلاش ئويۇنىنى ئويناۋېرىش ئىنتايىن خەتەرلىك ئىكەن، ئۇنىڭ ئاخىرى يوق ئىكەن، يەنە ئۆمۈرۈمنى ۋە بەختىمنى يەپ كېتىدىكەن. شۇڭلاشقا، مەن كەسىپ ئالماشتۇردۇم. ئۆزۈم ياخشى كۆرگەن نەشىرىيات ساھەسىدە بىر خىزمەت تاپتىم. بۇ خىزمەت ماڭا كۆپ پۇل تېپىپ بەرمەيدۇ، بىراق ماڭا شاتلىق ۋە ئىجتىمائىي مۇلازىمەت پۇرسىتى ئېلىپ كېلىدۇ. مەن يېتەرلىك بايلىق توپلاپ بولغان ھەر بىر كىشىگە شۇنداق تەكلىپ بېرىمەنكى: سىز بۇ پۇل-مال ئويۇنىنى توختۇتۇپ، بالدۇرراق پىنسىيەگە چىقىڭ، شۇندىلا، ئېرىشكەن مۇۋەپپىقىيەتلىرىڭىزدىن بەھرىمەن بولالايسىز، ياخشى كۆرگەن خىزمەتنى باشلىيالايسىز، جەمئيەت ئۈچۈن خىزمەت قىلىپ، كىشىلىك تۇرمۇشتىن ھوزۇر ئالالايسىز.»
   زور بايلىققا ئىگە بۇ شەخىس ۋارىسلىرىغا كۆپ بايلىق مىراس قالدۇرۇش ياكى قالدۇرالماسلىققا كۆڭۈل بۆلمەيدۇ. چۈنكى ئۇ شۇنى بىلىدۇكى، مۇبادا ئۇلارنىڭ كۆپ پۇلى بولماي، كاللىسى ۋە پوزىتسىيىسىگە تايىنىپ، بۇ جەمئىيەتكە كىرسە تېخىمۇ ياخشى ئادەم بولالايدۇ. ئىشلىمەي تاپقان مال-دۇنيا ھامان ئامەت ئەمەس ئاپەت، بەخىت ئەمەس مۇسىبەت بولىدۇ. چۈنكى، كىشىلەر ئۆزلىرىنىڭ كەيىپ-ساپا خاھىشلىرىنى قاندۇرالىغاندا، ئۇلارنىڭ ئوي-خىيالى ئەرزىمەس، پارچە-پۇرات ئىشلارنى قوغلىشىدۇ. نەتىجىدە، ئۇلارنىڭ ياشلىق باھارى ئاستا-ئاستا يوقۇلۇپ، تاكى ئۆلۈپ تۈگەيدۇ.
   
«ھاياتتا مۇمكىن بولمايدىغان ئىشنىڭ يوقلىقىغا قەتئىي ئىشىنىڭ.»
  
ـــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
«بىز ئېيتتۇق: ئى ئوت! ئىبراھىمغا سالقىن ۋە سالامەت بولغىن.» (69 :21)

                بىكارچىلىقتىن رەزىللىك تۇغۇلىدۇ

   بىكارچىلىقتىن رەزىللىك تۇغۇلىدۇ، ھالاكەتنىڭ يىلتىزى شەكىللنىدۇ، مۇبادا ھاياتلارنىڭ بۇرچى ئىشلەش بولسا، ئۇنداقتا بىكارچىلار ئۆلۈكلەردۇر.
   ئەگەر بۇ دۇنيا يېقىن كەلگۈسىدىكى ئاخىرەتنىڭ تەييارلىق مەزگىلى بولسا، ئۇنداقتا، لايغەزەللىكنىڭ قىلچە قىممىتى، ھوسۇلى يوق، پەقەت زىيىنى ۋە ھالاكىتىلا باردۇر.
   پەيغەمبەر ئەلەيھىسسالام تۈمەن مىڭلىغان كىشىلەرنىڭ ئۆزلىرىگە بېرىلگەن ساقلىق ۋە بوش ۋاقىتنى تونۇپ يەتمىگەنلىكىنى ئەسكەرتكەنىدى: «ئىككى خىل نېئمەت كۆپلىگەن كىشىلەر تەرىپىدىن سەل قارالدى، ئۇلار ساغلاملىق ۋە بوش ۋاقىتتۇر.»
   قانچىلىغان بېجىرىم كىشىلەر قىلچە نىشانسىز تۇرمۇش كەچۈرىدۇ، ئۇلارنىڭ نە ئالدىرىغۇدەك ئىشى، نە كۆرەش قىلىپ پۈتۈن كۈچى بىلەن ئورۇنلىغۇدەك بۇرچى يوق.
   بۇ ئىنسانىيەتنىڭ يارىتىلىشىدىكى مەقسەتمۇ؟ ھەرگىز ئەمەس. اللە ئېيتىدۇ: «ئەجىبا سىلەر بىزنى ئۆزۈڭلارنى بىھۇدە ياراتتى ۋە بىز تەرەپكە قايتۇرۇلمايمىز، دەپ ئويلامسىلەر؟ ھەق پادىشاھ - ئالاھ ئۈستۈندۇر، ئۇنىڭدىن باشقا ھېچ ئىلاھ يوقتۇر، (ئۇ) ئۇلۇغ ئەرشىنىڭ ئىگىسىدۇر.» (116-115 :23)
   ھاياتلىق ھەق بىلەن يارىتىلدى، ئاسمان-زېمىندىكى بارچە شەيئىلەرمۇ شۇنداق، ئىنسان مۇشۇ ھەقنى چۈشۈنۈشى ۋە شۇ بۇيىچە ياشىشى كېرەك. ئەمما، ئۇ تار نەپسىنىڭ كۈلەڭگىسى ئىچىدە ياشاپ، ئۆز ئىدىيىسى تەرىپىدىن كۆزى ئېتىلىپ قالسا، ئۇنىڭ ھازىرى ۋە كەلگۈسى ئۈچۈن قىلغان تاللىشى نېمىدىگەن ئېچىنىشلىق-ھە!
   
«مۇۋەپپىقىيەتنىڭ سۈرىتىنى دائىم ئېسىڭىزدە تۇتۇڭ، ئۇ كاللىڭىزدا ئوچۇق بولسۇن.»

ــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
« كىمكى ئاللاھتىن قورقىدىكەن ئاللاھ ئۇنىڭغا بىر چىقىش يولى بېرىدۇ، ھەمدە ئۇ ئويلىمىغان يەردىن ئۇنىڭغا رىزىق بېرىدۇ.» (3-2 :65)

                                6
            غەزەپ، غەۋغا، ھارغىنلىق بولمىغان بىر ئائىلە

ئۇ ئاتىسىغا يىغلاپ تۇرۇپ ئېيتتى: « ئاتا، ئاخشام دېگەن گەپلىرىم تۈپەيلىدن مەن بىلەن ئېرىم ئوتتۇرسىدا بەزى مەسىلىلەر چىقتى. مەن ئۇنىڭ شۇنچىلىك  ئاچچىقلىنىپ كەتكەنلىكىنى كۆرگەن چېغىمدا، قىلمىشىمدىن پۇشايمان قىلدىم. مەن ئۇنىڭدىن ئەپۇ سورىدىم، ئەمما، ئۇ مەن بىلەن سۆزلەشكىلى ئۇنىماي، چىرايىنى ئۆرىۋالدى. مەن ئۇدا ھەپلىشىپ ئۇنى ئاچچىقىدىن ياندۇردۇم، ئۇ ئاخىر كۈلدى ۋە مەندىن رازى بولدى. مەن پەرۋەردىگارىمنىڭ ئېرىمنى ئاچچىقلاتقان ئاشۇ ئىشلىرىم تۈپەيلى مەندىن ھېساب ئېلىشىدىن قورقىمەن.»
ئاتىسى ئۇنىڭغا دېدى: « قىزىم، جېنىم ئىلكىدە بولغان ئاللاھ بىلەن قەسەمكى، مۇبادا سىز ئېرىڭىز سىزدىن رازى بولۇشتىن ئىلگىرى ئۆلۈپ كەتكەن بولسىڭىز، مەن سىزدىن رازى بولمايتتىم. سىز ئېرىنى قاخشاتقان ئايالنىڭ «تەۋرات» «ئىنجىل» «زەبۇر» ۋە «قۇرئان» دا لەنەت قىلىنغانلىقىنى بىلمەمسىز؟ ئۆلۈمنىڭ سەكراتى ئۇنى قىينايدۇ. ئۇنىڭ قەبرىسى تارىيىدۇ. ئېرىنى سۈيۈندۈرىدىغان ئايالغا خۇش بىشارەت بولسۇن!»
سالىھە ئايال ئېرىنىڭ خۇرسەنلىكىنى ئارزۇلايدۇ. شۇڭلاشقا، ئۇ ئەر-خوتۇنلۇق ھاياتتا ئېرىنى خاپا قىلىدىغان ھەرقانداق ئىشنى قىلمايدۇ.

«مەغلۇبىيەتنىڭ خاتىىرىسىنى كاللىڭىزدىن سۈپۈرۈپ تاشلاڭ.»

ــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
«ئىمان يوق ئادەمدە خاتىرجەملىك يوق.»
                                 
                               7
             ئىپپەت ۋە ھايا ھۆسنىڭىزگە ھۆسىن قوشىدۇ

سىز پەيغەمبىرىمىزنىڭ ئايالى ئۇممۇ سەلىمەنىڭ ھېكايىسىنى ئاڭلىغانمۇ؟ ئۇ پەيغەمبەر ئەلەيھىسسالامنىڭ «كىمكى تەكەببۇرلۇق بىلەن ئىشتاننى ئۇزۇن سۆرەپ كېيسە، قىيامەت كۈنى ئاللاھ ئۇنىڭغا قارىمايدۇ.» دېگەنلىكىنى ئاڭلىغاندا، «ئاياللارنىڭ كۆڭلىكىچۇ؟» دەيدۇ. پەيغەمبەر ئەلەيھىسسالام «ئۇلار بىر غېرىچ ئۇزۇن كېيسۇن.» دەيدۇ. ئۇ «ئەمما ئۇلارنىڭ پاچىقى ئېچىلىپ قالىدۇ.» دەيدۇ. پەيغەمبەر ئەلەيھىسسالام «بىر گەز ئۇزارتسۇن، ئەمما يەنىمۇ ئۇزارتىشقا بولمايدۇ.» دەيدۇ.
  ئۇممۇ سەلىمە نەقەدەر ياخشى بىر ئۆلگە-ھە! ئۇ ھاكاۋۇر، مەنمەنچى ئايال ئەمەس ئىدى. مۇسۇلمان ئاياللار ئاددى-ساددا، پاكىزە، ئېسىل بولۇشى كېرەك. ئۇلارنىڭ پاچىقى ئېچىلىپ قالسا بولمايدۇ. ئۇلارنىڭ كۆڭلىكى ئەرلەر ئارقىسىدىن ئۇلارنى كۆرەلمەيدىغان دەرىجىدە ئۇزۇن، يەرگە تېگىپ تۇرىدىغان بولۇشى كېرەك. ئەمما، بۈگۈنكى مۇسۇلمان ئاياللار (ئاللاھ رەھىم قىلغانلاردىن سېرت) ئىىمكانىيەتنىڭ يار بېرىشىچە، كۆڭلىكىنى قىسقارتىپ، ئاشۇ يالىڭاچ، شاللاق، كاپىر  ئاياللارنى دورىماقتا. ئۇلار بۇخىل يالىڭاچلىق ۋە شاللاقلىق ئۈچۈن مىڭلارچە باھانە ئىزدەۋاتىدۇ. «ئاللاھقا تايانماستىن كۈچ ۋە چارە تاپقىلى بولمايدۇ.» ئۇلارنىڭ ئەرلىرىمۇ پەقەت نامدىلا ئەر بولۇپ، ئاياللىرىنىڭ قېشىدا ماڭىدۇيۇ، ئەمما، ئۇلارنىڭ ھاياسىزلىقىغا پەرۋا قىلمايدۇ.

«تەننىڭ ئەمىنلىكى ئاز يەپ-ئاز ئېچىىشتىن ، روھنىڭ ئەمىنلىكى ئاز گۇناھ قىلىشتىن،  قەلبنىڭ ئەمىنلىكى ئاز ئەندىشە قىلىشتىن،  تىلنىڭ ئەمىنلىكى ئاز سۆزلەشتىن كېلىدۇ.»

ـــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
«سەۋرچان كىشى ئەڭ ياخشى نەتىجىگە ئېرىشىدۇ.»

                            8
              ئاللاھ ئايرىلغانلارنى ئۇچراشتۇرىدۇ

يىگىرمە نەچچە يىلغا سۇزۇلغان جۇدالىقتىن كېيىن، ئاللاھ بىر ئانا بىلەن ئۇنىڭ يىگىرمە بەش ياشلىق قىىزىنىڭ قايتا ئۇچرىشىشىنى تەقدىر قىلدى. بۇ ۋەقە ئۇنىڭ يىگىرمە بەش ياشلىق قىزى شېرىن ئاي ئۆتكۈزۈۋاتقان سەۋدە تېغىدىكى تەتىل كەنتىدە يۈزبەرگەن. بۇ ئانا ئېرىدىن ئاجراشقاندىن كېيىن قايتا توي قىلغان. يىڭى ئېرىنىڭ ئەھۋالى سەۋەبىدن، ئۇلار ئۆزۈلمەي بىر شەھەردىن يەنە بىر شەھەرگە كۆچۈپ يۈرگەن. بۇ ئۇنى ئالدىنقى  ئېرى قارايدىغان قىزىنى كۆرۈشتىن`تۇسۇپ قويغان.
  بىر گۈزەل ياز كۈنى، ئەبھادىكى سەۋدە تېغىدا، بۇ قىز تەتىل كەنتىدە ئۇنىڭ بىلەن ئۇچرىشىپ قالىدۇ. ئۇلار سۆھبەت باشلايدۇ. بىراق بىر –بىرىنى تۇنۇمايدۇ. پاراڭ ئەسناسىدا بۇ ئانا قىزنىڭ بىر بارمىقىنىڭ يوقلۇقىنى بايقاپ، ئۇنىڭدىن ئانىسىنىڭ ئەھۋالىنى سۇرايدۇ. شۇنىڭ بىلەن قىز ئۆز ھېكايىسىنى سۆزلەپ بىرىدۇ. بۇ ئانا ئۆزىنىڭ دەل ئايرىلىپ كەتكىلى يىگىرمە نەچچە يىل بولغان قىزى بىلەن يۈز تۇرانە پاراڭلىشىۋاتقانلىقىنى تۇنۇپ يىتىدۇ. ئۇ قىزىنى باغرىغا باسىدۇ. تۇلۇپ تاشقان مېھرى بىلەن ئۇنىڭ مەڭزىگە سۆيىدۇ ۋە ئۆزىنىڭ شۇنچە يىللار مابەينىدە ئۇنى نەقەدەر سېغىنغانلىقىنى ئېيتىۇ.

  «بەخت ھەققىدىكى ئوي-خىيال ئالدى-كەينىگە قاراب بېقىشنى كەلتۈرۈپ چىقىرىدۇ،بۇنىڭ ئۆزى بەخت تۇيغۇسىنى بۇزىدۇ.»

ـــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
«ئۇلار گۇيا ياقۇت ۋە مارجانغا ئوخشايدۇ.» (58 :55)  

                                  9
                 زامان ۋە ماكاننى توشقۇزغان بىر كەلىمە
   
    مۇسا ئەلەيھىسسالام: «ئى ئاللاھ، ساڭا دۇئا قىلىشىم، سەن بىلەن ئۈلپەتلىشىشىم ئۈچۈن ماڭا بىر دۇڭا ئۆگىتىپ قويساڭ.» دەيدۇ. ئاللاھ : «ئى مۇسا، ئاللاھتىن باشقا ئىلاھ يوق، دېگىن.» دەيدۇ. مۇسا ئەلەيھىسسالام: «ھەممە ئادەم ‹ئاللاھتىن باشقا ئىلاھ يوق› دەيدۇغۇ؟» دېگەندە، ئاللاھ مۇنداق دەيدۇ:« ئى مۇسا، مۇبادا يەتتە ئاسمان ۋە كەڭ زېمىن تارازىنىڭ بىر پەللىسىگە، ‹ئاللاھتىن باشقا ئىلاھ يوق› يەنە بىر پەللىسىگە قۇيۇلسا،بۇ بىر كەلىمە سۆز ئېغىر كەلگەن بولاتتى.»
«ئاللاھتىن باشقا ئىلاھ يوق» دېگەن سۆز يامانلىقنىڭ قاپقاراڭغۇ زۇلمىتىنى  قوغلاپ، ھىدايەتنىڭ يۇپيۇرۇق نۇرىنى ئېلىپ كېلىدۇ. ئەمما، بۇ كەلىمەنى ئېيتقان كىشى ئوخشىمىغان دەرىجىدىكى نۇرغا ئىگە بولىدۇ. بۇ نۇرلارنىڭ كۈچلۈك-ئاجىزلىقىنى ئاللاھتىن باشقا ھېچكىم بىلمەيدۇ. بەزىلەرگە نىسبەتەن، بۇ سۆزنىڭ ئۆزى ئۇلارنىڭ قەلبىدە قۇياشقا ئوخشايدۇ، بەزىلەرگە نىسبەتەن پارلاق يۇلتۇزغا ئوخشايدۇ،بەزىلەرگە نىسبەتەن ئېلىكتىر چىراقنىڭ نۇرىغا، يەنە بەزىلەرگە نىسبەتەن خىرە چىراق نۇرىغا ئوخشايدۇ. بۇ سۆزنىڭ يۇرۇقلۇقى ئاشقانسېرى گۇمان ۋە شەھۋەتنى قوغلاش كۈچىمۇ شۇنچە ئاشىدۇ.
  
«مۇئمىننىڭ بەختى ئاللاھنى سۆيۈشتە، ئاللاھ سۆيگۈسى ئۇلارغا تېخىمۇ  چوڭقۇر بەخت تۇيغۇسى ئېلىپ كېلىدۇ. ساداقەتمەن مۇئمىن ئۇنى چۈشۈنەلەيدۇ ۋە ئۇنى ھەرقانداق نەرسىگە تېگىشىشنى خالىمايدۇ.»

ـــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
« ئايال مال- دۇنيا ۋە ئۈنچە- مەرۋايىتتىن قىممەتلىك.»

                                10
                      جەننەتكە تەشنا قەلبلەر

سىز ئاشۇ پاك-دىيانەتلىك ئايالنىڭ ھېكايىسىنى ئاڭلىغانمۇ؟ ئۇنىڭ ئېرى ئىككى بالىنى قالدۇرۇپ ئالەمدىن ئۆتتى. بالىلار چوڭ بولغاندىن كېيىن، ئۇ ئۇلارغا ئۆگەتكەن تۇنجى نەرسە ئىبادەت، ئاللاھقا ئىتائەت قىلىش ۋە تەھەججۇد نامىزى بولدى.
   ئۇ ئۇغۇللىرىغا ئېيتتى: « ئائىلىمىزدە كېچىنىڭ ھەر بىر دەقىقىسى بىرەيلەننىڭ ناماز ئۇقۇشى ۋە ئاللاھنى ئەسلىشى ئىچىدە ئۆتىشى كېرەك.» ئۇلار ئېيتتى : « ئانا، سىز نىمە دېمەكچى؟» ئۇ ئېيتتى: « بىز كېچىنى ئۈچكە بۆلىمىز،ئىچىڭلاردىن بىرى دەسلەپكى ئۈچتىن بىرىدە ناماز ئۇقۇيدۇ، يەنە بىرى ئىككىنچى ئۈچتىن بىرىدە ناماز ئۇقۇيدۇ، مەن ئەڭ ئاخىرقى ئۈچتىن بىرىدە ناماز ئوقۇيمەن، ئاندىن سىلەرنى بامداتقا ئويغۇتىمەن.»    ئۇلار ئېيتتى : « ئانا، بىز رىئايە قىلىمىز.»
ئانىسى ئالەمدىن ئۆتكەندىن كېيىنمۇ، بۇ ئىككى ئۇغۇل تەھەججۇد نامىزىنى تاشلىمىدى. چۈنكى، ئۇلارنىڭ قەلبى ئىمان ۋە نامازغا قارىتا قىزغىن مۇھەببەتكە تولغان ئىدى. ئۇلارنىڭ ھاياتىدىكى ئەڭ شېرىن دەقىقىلەر دەل كېچىدىكى ناماز ئۆتەۋاتقان چاغ ئىدى. شۇڭلاشقا، ئۇلار بىر كېچە ۋاقىتنى تەڭ بۆلىشەتتى. ئۇلارنىڭ ئىچىدىكى بىرەيلەن تازا ھالسىز لىنىپ كەتكەندە،  يەنە بىرەيلەن تۇرۇپ، پۈتۈن كېچە ناماز ئۇقۇپ چىقاتتى.

   
«ئەتراپىمىزدىكى ھاياتلىق ۋە ئۇنىڭ بارچە گۈزەللىكى ۋە ئېسىللىكى دەل بەخت- سائادەتنىڭ تەكلىپنامىسىدۇر.»
                                      

ــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ

« مۇبادا سىلەر ئاللاھنىڭ نىئمەتلىرىنى سانىساڭلار، سىلەر ساناپ بولالمايسىلەر.» (34 :14)

                                        1
                               تەقدىرگە ئىشىنىڭ
   
قۇدرەتلىك، شانۇ-شەۋكەتلىك ئاللاھ ئېيتىدۇ:
   « يەر يۈزىدىكى بارلىق ھادىسىلەر ۋە ئۆزۈڭلار ئۇچرىغان مۇسىبەتلەر، بىز ئۇلارنى يارىتىشتىن بۇرۇن لەۋھۇلمەھپۇزدا خاتىرىلەپ قويۇلغان. بۇ ئاللاھقا تۇلىمۇ ئاساندۇر. (بۇنداق بولىشى) سىلەرنىڭ ئۆزۈڭلار  يۇقاتقانغا قايغۇرۇپ، ئۇ سىلەرگە ئاتا قىلغىنى ئۈچۈن مەغررۇرلىنىپ كېتىشتىن ساقلىنىشىڭلار ئۈچۈندۇر. ئاللاھ ھەقىقەتەن بارچە ھاكاۋۇر، ماختانچاقنى ياقتۇرمايدۇ.»    (23-22 :57)
« سىلەرگە جىھاد پەرز قىلىندى، ھالبۇكى سىلەر ئۇنى ياقتۇرمايسىلەر. سىلەر بىرەر نەرسىنى ياقتۇرماسلىقىڭلار مۇمكىن، ئەمما ئۇ سىلەر ئۈچۈن پايدىلىقتۇر. سىلەر بىرەر نەرسىنى ياقتۇرۇشۇڭلار مۇمكىن، ئەمما ئۇ سىلەر ئۈچۈن زىيانلىقتۇر. (سىلەرگە نېمىنىڭ پايدىلىق ئىكەنلىكىنى ئاللاھ بىلىدۇ، سىلەر بىلمەيسىلەر.»  (216  :2 )
تەقدىرگە ئىشىنىش مۇسىبەت كەلگەندە، مۇھىم رول ئوينايدۇ. ئۇ كۆڭۈل خاتىرجەملىكى ئېلىپ كېلىدۇ. بۇلۇپمۇ بىر ئادەم ئاللاھنىڭ بەندىلىرىگە مىھرىبانلىقىنى، ئۇلارغا ئاسانلىقنى خالايدىغانلىقىنى تۇلۇق چۈشۈنۈپ يەتكەندە، تېخىمۇ شۇنداق بولىدۇ. ئاللاھ ھىكمەت بىلەن ئىش قىلغۇچىدۇر. ئۇ بەندىلىرى ئۈچۈن ئاخىرەتتىكى ساۋاپ-مۇكاپاتنى ساقلاپ قۇيىدۇ. ئۇ سەۋر قىلغۇچىلارغا زور ئەجر بېرىدۇ. مۇبادا بىز بۇنى مىزان قىلىپ ئويلانساق ۋە ھەركەت قىلساق، مۇسىبەت ئېلىپ كەلگەن ئازاب- ئۇقۇبەت بەخت ۋە خۇشاللىققا ئايلىنىدۇ. ئەمما، بۇ ھەممە ئادەم قىلالايدىغان ئىش ئەمەس. ئۇنداقتا، بىز قانداق قىلغاندا ئاپەتنىڭ تەسىرىنى كېچىكلىتىپ، زىيىنىنى ئازايتالايمىز؟
1- تېخىمۇ چوڭ مۇسىبەتنى ۋە ئەڭ ئۇسال ئاقىۋەتنى تەسەۋۋۇر قىلىڭ.
2- تېخىمۇ چوڭ مۇسىبەتكە ئۇچرىغان كىشىلەرنى ئويلاڭ.
3- كۆپلىگەن كىشىلەر مەھرۇم، ئەمما سىز نائىل بولغان نىئمەتلەرگە قاراڭ.  
4- مۇسىبەتكە ئەگىشىپ كېلىدىغان چۈشكۈنلۈككە تەسلىم بولماڭ.
                                                                     
« ھەقىقەتەن ھەر بىر مۈشكۈللۈك بىلەن بىر ئاسانلىق بار. ھەقىقەتەن ھەر بىر مۈشكۈللۈك بىلەن بىر ئاسانلىق بار.»    (6-5 :94)
                                                                     
« باشقىلارغا خۇشاللىق يەتكۈزۈشنىڭ ئەڭ تىز ئۇسۇلى چىن كۆڭۈلدىن چىققان سەمىمى   تەبەسسۇم.»

ــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ

                     ئوتتۇرھاللىق ھەممىدىن ئەلا  

مۇستافا مۇنداق دەيدۇ:
مەن ئۆزۈمنى بەختلىك ھېس قىلىمەن، چۈنكى مەن ئوتتۇرھال ئادەم، ئوتتۇرھال كېرىمىم، ئوتتۇرھال سالامەتلىكىم، ئوتتۇرھال تۇرمۇش سەۋىيەم بار. ھەر بىر نەرسىدىن ئازراقتىن بار. بۇ مېنىڭ نۇرغۇن ھەركەتلەندۈرگۈچ كۆچۈم بارلىقىدىن دېرەك بېرىدۇ. ھەركەتلەندۈرگۈچ كۈچ دېمەك - ھايات دېمەكتۇر. قەلبىمىزدىكى ھەركەتلەندۈرگۈچ كۈچ ھاياتنىڭ ھەقىقى ھارارىتى بولۇپ، بەختنى ئۆلچەيدىغان ئۆلچەمدۇر...
    مەن ئاللاھتىن ئوقۇرمەنگە بىر ئوتتۇرھال تۇرمۇش ئاتا قىلىشىنى، ھەر نەرسىدىن   
ئازراقتىن بېرىشىنى تىلەيمەن. ئۇلۇغ ئاللاھ بىلەن قەسەمكى، بۇ بىر ياخشى تىلەكتۇر.
ئانام پەلسەپىنى چۈشەنمەيدۇ. بىراق ئۇ بىر ئاددى قەلبكە ئىگە. شۇڭا، ئوقۇمايمۇ بۇ سۆزلەرنىڭ مەنىسىنى چۈشۈنەلەيدۇ. ئۇ ئادەتتە ئۇنى «شۈكۈر-قانائەت» دەپ ئاتايدۇ، قانائەت قىلغۇچى دائىم خۇشال يۈرۈيدۇ، ھەر نەرسىدىن ئازراققا ھەمدە مول روھى ئېنىرگىيەگە ئىگە بولىدۇ.

    « ساختا كۈلكە ساختا ياخشىلىقنىڭ رۇشەن ئىپادىسى.»

ــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
« مېنىڭ خۇرسەنلىكىم نامازدا.»  ( ھەدىس شەرىف )

                  چۈشكۈن ئادەم تىنجىق كەيپىيات يارىتىدۇ

   بىر ئادەمنىڭ پوزىتسىيەسى ھەمراھىنىڭ كەيپىياتى ۋە پوزىتسىيەسىگە تەسىر قىلىدۇ. مۇبادا بىر ئادەم (دوست، جۈپ، خىزمەتداش) مۇلايىم، خۇشخۇي، ئۈمىدۋار بولسا ئۇ مۇشۇ خىسلەتلىرىنى ئۆز ھەمرالىرىغا تارقىتىدۇ. ئەمما، ئۇ مۇبادا قاخشايدىغان، دەرتمەن بولسا ئۇ ھاياتقا خۇشاللىق بىلەن باقالمايدۇ، دائىم ئەنسىرەيدۇ، ئۈمدسىزلىنىدۇ. ئۇ بۇ خىل ئادەمنى روھسىزلاندۇرىدىغان ئەنسىز كەيپىياتنى ئەتراپىدىكىلەرگە تارقىتىدۇ، خۇددى كېسەل تارقاتقانغا ئوخشاش.
پەقەت ئادەملا بۇخىل تەسىرنى پەيدا قىلىدۇ ئەمەس، كىتاب، رادىئو، تېلېۋىزو رلارمۇ ئوخشاش. بەزىسى جۇشقۇن، بەزىسى چۈشكۈن، بەزىسى ئەنسىزلىك قوزغايدۇ، بەزىسى خاتىرجەملىك بېرىدۇ. بولۇپمۇ كىتاب تۆت پەسىلگە ئوخشايدۇ، بەزىسى باھارغا، بەزىسى كۈزگە ئوخشايدۇ. مۇبادا بىر ئادەم ئۈمدۋار، ھاياتنى كۈيلەيدىغان، كۆرەشكە، مۇۋەپپىقىيەتكە، ئۆزىگە ئىشىنىشكە چاقىرىدىغان كىتابلارنى تاللاشقا مۇۋەپپەق بولسا ئۇ ياخشى نەتىجىگە ئېرىشىدۇ. ئەمما، مۇبادا ئۇ غەم-قايغۇ، شەك-شۈبھە قوزغايدىغان قىممەت قاراشلارنى، ئىنسان ھاياتى ۋە ئىسان تەبىئىتىگە قارىتا چۈشكۈن پوزىتسىيە تۇتىدىغان كىتابلارنى تاللىسا ،ئۇ بۇخىل ئىدىيەلەر تەرىپىدىن يۇقۇملىنىپ قالىدۇ، خۇددى جۇزام كېسىلى ساق ئادەمگە يۇققاندەك. بۇخىل كىتابلار ئۇنىڭ ھاياتىنى ۋەيران قىلىدۇ.

« بەختكە تۇتاشقان يول سىزنىڭ كۆز ئالدىڭىزدا، بىلىم-مەرىپەت، سالىھ ئەمەل، گۈزەل ئەخلاق بىلەن ئۇنى ئىزدەڭ. بارچە ئىشلاردا ئوتتۇرھاللىقنى ساقلاڭ، خۇشال بولىسىز.»

ــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
« ئاللاھ ئۇلارغا تەمكىنلىك چۈشۈرۈپ بەردى.»  (18 :48)

                      ئاغرىنىش ۋە نارازىلىقتىن ساقلىنىڭ

بىر دانىشمەن بۇۋاي مۇنداق دەيدۇ:
« مەن يىگىرمە-ئوتتۇز ياشلىرىمدا ھەمىشە غۇدۇراپ-رەنجىپلا يۈرەتتىم، گەرچە مەن ناھايىتى بەختلىك ئۆتكەن بولساممۇ ؛ چۈنكى مەن ئۇ چاغلاردا ئۆزۈمنىڭ بەختلىك ئىكەنلىكىمنى بىلمەيتتىم. ھازىر مەن ئاتمىش نەچچە ياشقا كىردىم. مەن ئۆزۈمنىڭ يىگىرمە-ئوتتۇز ياشلاردىكى ۋاقتىمدا شۇ قەدەر بەختلىك ئىكەنلىكىمنى تۇلۇق چۈشەندىم. لىكىن بۇ خىل تۇيغۇ تۇلىمۇ كېچىكىپ كەلدى. مېنىڭ ئەسلىمىدىن باشقا ھېچنىمەم يوق. ئەسلىمىنىڭ ھەممىسى ئەپسۇسلىنىش. مۇبادا مەن ئەينى ۋاقىتتا چۈشەنگەن بولسام تولىمۇ خۇشال-خۇرام ياشىغان بۇلار ئىدىم. مېنىڭ ياشلىقىمدىكى ئەڭ گۈزەل مەزگىللەردە رەنجىيدىغانغا ھەرقانداق سەۋەب بولماسلىقى كېرەك ئىكەن. ئەپسۇس، ھازىر مەن پەقەت يېشىمنىڭ چوڭىيىشىغا ئەگىشىپ، ئاستا-ئاستا قېرىپ كېتىۋاتقانلىقىمغا قاراپلا تۇرالايمەن.»
سۆيۈملۈك ئوقۇرمەن سىزگە شۇنداق دېمەكچىمەن:
يا بەختىڭىزنى ھېس قىلىپ، قىزغىنلىق بىلەن ئۇنىڭدىن بەھرىمەن بولۇڭ ياكى ئۇنى مەنسىتمەي، ھەممە جايدىن بەخت ئىزدەڭ. ئۇنداقتا، سىز ئاغرىنىش ۋە نارازىلىقىنىڭ كونتروللىقىغا چۈشۈپ قالىسىز، ھازىر ئۆتمۈشكە ئايلانغاندا بۇنىڭ ئۈچۈن ياش تۆكىسىز، ئۆزىڭىزنىڭ شۇ چاغدا تولىمۇ بەختلىك ئۆتكەنلىكىڭىزنى، ئەمما ئۆزىڭىزنىڭ قىلچىمۇ سەزمىگەنلىكىڭىزنى تۇنۇپ يېتىسىز. ئەمدىكى سىزدە رەڭگى ئۆچكەن ئەسلىمىدىن ئۆزگە ھېچنىمە قالمىغان.

«بىر ئايال ئۆينى جەننەتكە ئايلاندۇرالايدۇ، شۇنداقلا دوزاخقىمۇ ئايلاندۇرالايدۇ.»

ـــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
«ئاللاھ ئۇلاردىن رازى بولىدۇ، ئۇلارمۇ ئاللاھتىن  مەمنۇن بولىدۇ.» (119 :5)

                                       5
                نۇرغۇنلىغان مەسىلىلەر كىچىك ئىشلاردىن تۇغۇلىدۇ

ئەپسۇسلىنارلىقى شۇكى، ئەرزىمەس ئىشلار ھەمىشە مىڭلىغان-ئونمىڭلىغان كىشىلەرنىڭ ھۆكۈم قىلىش ئىقتىدارىدىن ئايرىلىپ قېلىشىنى، ئائىلىلەرنىڭ ۋەيران بولۇشىنى، دوستلۇقنىڭ بەربات بولۇشى ۋە كىشىلەرنىڭ ھەيرانە-چۈشكۈن بولۇشنى كەلتۈرۈپ چىقىرىدۇ. دالې گارنېك پۇچەك ئىشلارنىڭ كەينىگە كىرىشنىڭ  ئاقىۋىتىنى مۇنداق شەرھىلەيدۇ:
« نىكاھتىكى كىچىك ئىشلار ئەر-خوتۇن ئىككى تەرەپنىڭ ئەقلىنى يۇقۇتۇشىنى كەلتۈرۈپ چىقىرىدۇ، شۇنداقلا دۇنيادىكى يۈرەك كېسىلىنىڭ يېرىمى شۇلار تەرىپىدىن كېلىپ چىققان.»
بۇ ئەمەلىيەت مۇتەخەسىسلەر تەرىپىدىن ئىسپاتلانغان، تۆت مىڭدىن كۆپرەك ئاجرىشىش دەۋاسىنى سوت قىلىپ بىر تەرەپ قىلغاندىن كېيىن، چىكاگولىق سوتچى جوزىف ساباس مۇنداق دەيدۇ: « سىز ھەمىشە بىر قىسىم ئەرزىمەس ئىشلارنىڭ نىكاھ مەغلۇبىيىتىنى كەلتۈرۈپ چىقارغانلىقىنى بايقاسىز.»
فرانك ھوگېن ــ نيۇيوركتىكى بىر تەپتىش ئەمەلدارى مۇنداق دەيدۇ: « جىنايى ئىشلار سوتىدىكى دىلولارنىڭ يېرىمى كىچىك ئىشلاردىن كېلىپ چىىققان. مەسىلەن: ئائىلە ئەزالىرى ئارىسىدىكى جەڭگى-جىدەل، غەرەزسىز چىشىغا تېگىپ قۇيۇش، ئادەمنىڭ كۆڭلىگە كېلىدىغان بىر ئېغىز گەپ ياكى ھاقارەت قىلىدىغان بىر قول ئىشارىتى... قاتارلىقلار. مانا بۇ كىچىك ئىشلار ئادەم ئۆلتۈش ۋە باشقا جىنايەتلەرنى كەلتۈرۈپ چىقىرىدۇ.
بىزنىڭ ئارىمىزدىكى ئىنتايىن ئاز ساندىكى كىشىلەرنىڭ تەبىئىتى ھەقىقەتەن رەھىمسىز ۋە قەبىھتۇر، ئەمما، بىزنىڭ ئابرويىمىز ۋە ئىززىتىمىزگە زىيان سالىدىغان ئاشۇ ئىشلار دۇنيادىكى ئىجتىمائى مەسىلىلەرنىڭ يېرىمىنىڭ سەۋەبىگە ئايلاندى.

« ئەڭ چوڭ ئامەت دەل روھىي گۈزەللىككە ئىگە بولۇشتۇر.»

ــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
« ئۆز ۋاقتىدا سىلەر پەرۋەردىگارىڭلاردىن ياردەم تىلىدىڭلار، ئۇ سىلەرگە ياردەم بەردى.»  (9  :8)

                                 تىلىڭىزنى يىغىڭ

تارىخچىلارنىڭ خاتىرىلىشىچە: خالىد بىن يەزىد بىر كۈنى ئۇمەييە جەمەتىنىڭ ئەشەددى دۈشمىنى ئابدۇللاھ بىن زۇبەيرگە «بېخىل» دەپ، تىل تەگكۈزىدۇ. خالىدنىڭ ئايالى رەملە ئابدۇللاھ نىڭ سىڭلىسى بۇلۇپ، ئۇمۇ ئەينى چاغدا شۇيەردە ئولتۇراتتى. ئەمما ئۇ جىمجىت تۇرىدۇ، بىر ئېغىزمۇ گەپ قىلمايدۇ. خالىد ئۇنىڭغا « نىمىشقا گەپ قىلمايسەن، مېنىڭ گېپىمگە قوشۇلدۇڭمۇ؟ ياكى ئەسلا گەپ قىلىشنى خالىمامسەن؟» دېگەندە، ئۇ مۇنداق دەيدۇ: « ھېچقايسىسى ئەمەس، ئەمما  ئاياللار ئەرلەرنىڭ ئىشلىرىغا ئارلاشقىلى يارىتىلمىغان. بەلكى گۈلگە ئوخشاش، ھوزۇرلانغىلى ۋە قۇچاقلانغىلى يارىتىلغان. نىمە ئۈچۈن بىز سىلەرنىڭ ئىشلىرىڭلارغا ئارلاشقۇدەكمىز؟» خالىد بۇ گەپلەردىن ھاياجانلىنىپ ئۇنىڭ پىشانىسىگە سۆيۈپ قويىدۇ.
ئاللاھنىڭ ئەلچىسى تەكىتلەپ، ئەر-ئايال ئوتتۇرسىدىكى سىرلارنى تارقىتىشتىن توسقان. ئەھمەدنىڭ رىۋايەت قىلىشىچە: ئەسما بىنتى يەزىد ئۆزىنىڭ پەيغەمبەر ئەلەيھىسسالامنڭ ھۇزۇرىدا كۆپلىگەن ئەر-ئاياللار بىلەن بىللە ئولتۇرغانلىقىنى ئېيتىپ مۇنداق دىگەن، پەيغەمبەر ئەلەيھسسالام: «بەلكىم ئۆزىنىڭ ئايالى بىلەن نېمە ئش قىلغانلىقىنى دەيدىغان ئەرلەر ۋە ئۆزىنىڭ ئېرى بىلەن نېمە ئىش قىلغانلىقىنى دەيدىغان ئاياللار بولىشى مۇمكىن»، دېدى. كۆپچىلىىك جاۋاپسىز جىمىپ كەتتى. مەن  دېدىم: «ئى ئاللاھنىڭ ئەلچىسى ، ئۇلار ھەقىقەتەن ئاشۇنداق قىلىۋاتىدۇ». پەيغەمبەر ئەلەيھىسسالام ئېيتتى: «ئۇنداق قىلماڭلار، ئاشۇ بىر ئەركەك شەيتان كوچىدا يەنە بىر چىشى شەيتان بىلەن ئۇچرىشىپ، كۆپچىلىكنىڭ ئالدىدا جۈپلەشكىنىگە ئوخشايدۇ.»
بەزى تەپسىرچىلەر : « ياخشى ئاياللار ئىتائەت قىلغۇچىلاردۇر، ئاللاھنىڭ پاناھلىقىدا غەيىبنى ساقلىغۇچىدۇر.» (34 :4) دېگەن  ئايەتتىكى « غەيىبنى ساقلاش » نىڭ مەنىسى : ئايال كىشىنىڭ ئېرى بىلەن ئۆزى ئوتتۇرسىدىكى مەخپىيەتلىكنى ساقلىشىنى كۆرسىتىدۇ، دېگەن.

« ئۇچرىغان ئاۋارىچىلىقلارنى ساناپ ئولتۇرماي، ئاللاھ سىزگە ئاتا قىغان نىئمەتلەرنى ساناڭ.»

ــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
« ھايات قىسقا، غەم-قايغۇ سەۋەبلىك ئۇنى تېخىمۇ قىسقارتىۋەتمەڭ.»

                                   7
                    ناماز ئارقىلىق غەم-قايغۇنى قوغلاڭ

دەسلەپكى مەزگىلدىكى مۇسۇلمان ئاياللار نامازنىڭ بىر ئادەم بىلەن ئۇنىڭ پەرۋەردىگارى ئوتتۇرسىدىكى مۇناسىۋەت ئىكەنلىكىنى، كۆڭۈل قۇيۇپ، ئىخلاس بىلەن ناماز ئوقۇغان كىشىلەرنىڭ مۇۋەپپەقىيەت قازىنىدىغانلىقىنى بىلەتتى.
« مۇئمىنلەر ھەقىقەتەن مۇۋەپپەقىيەت قازاندى. ئۇلار نامازلىرىدا ئەيمىنىپ تۇرغۇچىلاردۇر.» (2-1  :23) ئۇلار دائىم كېچىلىرى ئورنىدىن تۇرۇپ،كەمتەرلىك ۋە ئىتائەتمەنلىك بىلەن ئاللاھقا ئىبادەت قىلىدۇ. ئۇلار ئاخىرەت ئۈچۈن ئەڭ ياخشى تەييارلىق قىلىشنىڭ ۋە ئىسلامنى جاھان ئەھلىگە تارقىتىشنىڭ ئەڭ ياخشى ئۇسۇلىنىڭ ناماز ئىكەنلىكىنى بىلىدۇ. ناماز كىشىنىڭ ئىرادىسىنى چىڭىتىدۇ، كىشىنى مەسىلىلەرگە يۈزلىنىپ، ھەل قىلالايدىغان قىلىدۇ. ئۇلار يەنە تەھەججۇد نامىزىنىڭ بىر ئىنساننىڭ ئۇلۇغ ئاللاھقا يېقىنلىشىشتىكى ئەڭ ياخشى شەكلى ئىكەنلىكىنى بىلىدۇ. ئاللاھ تۇنجى دەۋەتچى ( مۇھەممەد ئەلەيھىسسالامنى دىمەكچى ) گە مۇنداق خىتاب قىلىدۇ: « كېچىنىڭ بىر قىسمىدا ساڭا تېگىشلىك بولغان قۇشۇمچە ئىبادەتلەرنى ئۆتىگىن، پەرۋەردىگارىڭ سېنى مەدھىيەگە لايىق بىر ئۇرۇنغا كۆتۈرگۈسى.» (79 :17)
ئۇ تەھەججۇد نامىزى ئۆتىگۈچىلەرنى مەدھىيەلەيدۇ: « ئۇلار كېچىلەردە پەقەت ئازغىنا ئۇخلايدۇ.»  (17 :51)
ئەنەسنىڭ رىۋايەت قىلىشىچە، پەيغەمبەر ئەلەيھىسسالام مەسچىتكە كىرىپ، بىر ئارغامچىنىڭ ئىككى تۈۋرۈك ئارىسىغا تارتىپ قۇيۇلغانلىقىنى كۆرۈپ سۇرايدۇ: « بۇ نىمە؟» ئۇلار: « بۇ زەينەپنىڭ، ئۇ ھېرىپ قالسا ئۇنىڭغا ئېسىلىدۇ.» دېگەندە، پەيغەمبەر ئەلەيھىسسام: « ئۇنى يېشىۋېتىڭلار، ئاراڭلاردا كىم روھلۇق بولسا ناماز ئوقۇسۇن، ناۋادا ھېرىپ قالسا ئارام ئالسۇن.» دەيدۇ.
ئىمانلىق ئاياللار ھەمىشە ئۆزلىرىدىن كېچىلىرى ناماز ئۆتەشنى تەلەپ قىلاتتى ۋە بۇنىڭ بىلەن ئاللاھنىڭ رازىلىقىنى تىلەيتتى. ئەمما، پەيغەمبەر ئۇلارنى تاقىتى يەتمەيدىغان ئىشنى قىلماسلىققا بۇيرىدى. ئەڭ ياخشى ئىبادەت ئاز بولسىمۇ ئۆزۈلمەي داۋاملاشقىنىدۇر. بىز بىلىمىز، بۈگۈنكى ئاياللارنىڭ ۋاقتى دۇنيانىڭ پارچە-پۇرات ئىشلىرى بىلەن تۇشۇپ كەتكەن. ئەمما، ئۇلار ھېچبولمىغاندا شەيتانغا تاقابىل تۇرۇش ئۈچۈن، ئوتتۇرا كېچىدە ئىككى رەكئەت ناماز ئوقۇسا بولاتتى. ئەڭ ياخشى ئىش ئوتتۇرھال بولغىنىدۇر، خۇددى ئاللاھنىڭ ئەلچىسى ئېيتقان كەبى: « چەكتىن ئاشۇرۋەتكۈچى (ئاشقۇن) ھالاك بولدى.»

« مۇبادا سىز سادىق بولسىڭىز، ئاللاھقا تەۋەككۇل قىلىڭ. مۇبادا سىز پۇشايمان قىلغان بولسىڭىز، ئەتە ئۈچۈن خۇشال بولۇڭ.»

ــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
« سەۋر قۇتۇلۇشنىڭ ئاچقۇچى.»

                                  8
              مۇۋەپپەقىيەت قازانغان بىر ئايالنىڭ نەسىھەتلىرى

ئاڭلىق بىر ئانا تەبەسسۇمى ۋە كۆز ياشلىرى بىلەن ئۆز قىزىغا مۇنداق نەسىھەت قىلغان:
قىزىم، سەن يىڭى بىر تۇرمۇشنى، ئاتا-ئاناڭ، قېرىنداش-ھەمشىرىلىرىڭ بولمىغان بىر تۇرمۇشنى باشلايسەن. بۇ يىڭى تۇرمۇشتا سەن ئېرىڭنىڭ ھەمرايىغا ئايلىنىسەن. ئۇ ھەرقانداق ئادەم ( ھەتتا سېنىڭ ئۆزۈڭ ) بىلەن سەندىن ئورتاق بەھرىمەن بولۇشنى خالىمايدۇ.
ئېرىڭ ئۈچۈن ئايال ۋە ئانا بول، ئۇ سېنىڭ ئۆزىنىڭ ھاياتىدىكى ھەممە ئىكەنلىكىڭنى ھېس قىلسۇن. ئېسىڭدە تۇت، ھەرقانداق ئەر كىشى ــ بىر چوڭ بالا، ھەرقانداق شىرىن سۆز ئۇنى خۇرسەن قىلالايدۇ. ئۇ سېنى ئېلىپلا سېنىڭ ئۇنى ئۆزىنىڭ ئائىلىسىدىن ئايرىۋەتكەنلىكىڭنى ھېس قىلمىسۇن، چۈنكى، ئۇمۇ ئوخشاش ھېسسىياتتا. ئۇمۇ ئۆز ئاتا-ئاناسىنىڭ ئۆيىدىن ئايرىلدى، سەن سەۋەبلىك ئۆز ئائىلىسىدىن ئايرىلدى، ئەمما، ئۇنىڭ بىلەن سېنىڭ پەرقىڭ پەقەتلا ئەر بىلەن ئايالنىڭ پەرقى خالاس. ئايال كىشى دائىم ئۆز ئائىلىسىدىكىلەر ۋە ئۆزى تۇغۇلۇپ ئۆسكەن جاينىڭ غېمىنى يەيدۇ، ئەمما، ئۇ يىڭى تۇرمۇشقا ماسلىشىشى، ئۆزىنىڭ ئېرى، بالىلىرىنىڭ ئاتىسى بولغان بۇ كىشى بىلەن بىرگە ياشاشقا ماسلىشىشى كېرەك، چۈنكى بۇ ئۇنىڭ يېڭى دۇنياسى.
قىزىم، بۇ سېنىڭ ھازىرىڭ ۋە كەلگۈسىڭ. بۇ سەن ۋە سېنىڭ ئېرىڭ قۇرماقچى بولغان ئائىلە. مەن ساڭا ئاتا-ئاناڭنى ئونتۇپ كەت، دېمەكچى ئەمەس، چۈنكى ئۇلار سېنى مەڭگۈ ئۇنتۇمايدۇ. بىر ئانا ئۆزىنىڭ يۈرەك پارىسىنى قانداقمۇ ئۇنتۇپ كېتەلىسۇن؟ ئەمما، مەن سەندىن ئېرىڭنى ياخشى كۆرۈشۈڭنى، ئۇنىڭ بىلەن خۇشال ئۆتۈشۈڭنى تىلەيمەن.

«سەۋرنى ئاسىيەدىن، ۋاپانى خەدىچەدىن، ساداقەتنى ئائىشەدىن، چىدامچانلىقنى فاتىمەدىن ئۆگېنىڭ.»  

ــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
« ئەرلىرىدىن ئىلگىرى ئۇلارغا ھېچ ئىنسان ۋە جىن يېقىنچىلىق قىلمىغان.»  (74  :55)

                                   9
                      تەسەللى پەقەت ئاللاھتىن كېلىدۇ

ئۇلۇغ ئاللاھ مۇئمىنلەر تەسەللى ۋە خۇشاللىق تاپىدىغان بۇلاقتۇر، چۇقۇنغۇچىلارنىڭ مەھبۇبىدۇر. ئاللاھنىڭ دەرگاھىدىن تەسەللىگە ئېرىشكەن كىشىلەر خۇشال ھالدا تۇرمۇشتىن بەھرىمەن بولىدۇ. ئۇلارنىڭ قەلبى خاتىرجەملىك ۋە ئاللاھقا بولغان سۆيگۈگە تولغان بولىدۇ. ئۇلار دائىم ئاللاھنىڭ گۈزەل سۈپەتلىرىنى ئويلايدۇ. ئاللاھنىڭ گۈزەل ئىسىملىرى ــ ناھايىتى شەپقەتلىك، ناھايىتى مىھرىبان، مەدھىيەگە لايىق، ئەپۇچان، پاك، ئىللىق، ئېھسان قىلغۇچى، ھۆرمەتلىك... نى ياد ئېتىدۇ. ئۇلار ياراتقۇچىنىڭ دەرگاھىدىن تەسەللى ئىزدەيدۇ. ئۇلۇغ ئاللاھنى قىزغىن سۆيىدۇ. ھەممىنى بىلىپ تۇرغۇچى ئاللاھقا شۈكۈر قىلىدۇ ۋە يېقىنلىشىدۇ. ئاللاھ بەندىلىرىگە يېقىندۇر. بۇ خىل تۇيغۇ تەسەللى ئېلىپ كېلىدۇ. كىشىلەرنى ئاللاھنىڭ ھىمايىسىدە بەختلىك، خۇشاللىق تۇيغۇسىغا ئىگە قىلىدۇ.
« مېنىڭ بەندىلىرىم سەندىن مەن توغرۇلۇق سورىسا، (ئۇلارغا ئېيتىپ قويغىنكى)، مەن ھەقىقەتەن ئۇلارغا يېقىندۇرمەن. ماڭا دوڭا قىلسا، مەن دۇئا قىلغۇچىنىڭ دۇئاسىنى ئىجابەت قىلىمەن،ئۇلار توغرا يول تېپىش ئۈچۈن مېنىڭ دەۋېتىمنى قۇبۇل قىلسۇن ۋە ماڭا ئىمان ئېيتسۇن.»  (186  :2)
ئەمما، ئاللاھنىڭ دەرگاھىدىن تەسەللى تېپىش ئۈچۈن تىرىشىش ۋە كۆرەش قىلىشقا توغرا كېلىدۇ. ئۇ بولسا ئىتائەت ۋە مۇھەببەتنىڭ مېۋىسىدۇر. ئاللاھقا ئىتائەت قىلىدىغان، ئاللاھنىڭ بۇيرىقىغا رىئايە قىلىدىغان، ئاللاھنىڭ توسقانلىرىدىن يىراق تۇرىدىغان، ئاللاھنى چىن سۆيىدىغان ھەرقانداق كىشى خۇشاللىق ھېس قىلىدۇ ۋە ئاللاھقا يېقىلىشىشتىن تەسەللى تاپىدۇ.

« ھەقىقى گۈزەللىك گۈزەل ئىش-ھەركەت ۋە ئوي-پىكىردىن كېلىدۇ.»

ــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــــ
« مەن سىلەرنىڭ ئاياللارغا ياخشى مۇئامىلە قىلىشىڭلارنى تەۋسىيە قىلىمەن.»  ( ھەدىس )

                                   10
                      ئەسمانىڭ ئىككى خىل تۇرمۇشى

ئەسما مۇشەققەتلىق تۇرمۇشتا سەۋر قىلىش، ئېرىگە ئىتائەت قىلىش ۋە ئۇنىڭ رازىلىقىغا ئېرىشىشنىڭ ياخشى نەمۇنىسىنى تىكلىدى.
سەھىھ ھەدىستە خاتىرىلىنىشىچە، ئۇ مۇنداق دىگەن: زۇبەير مېنى ئالغان ۋاقىتتا ئۇنىڭ مەن قارايدىغان ۋە باقىدىغان ئېتىدىن باشقا ھېچنىمىسى يوق ئىدى. مەن سۇ ئەكىلەتتىم، نان ياقاتتىم، يەنە سۇ تۇشۇيدىغان تۈگە ئۈچۈن خورما ئۇرۇقلىرىنى تۇشۇيتتىم. بىر كۈنى مەن پەيغەمبەر ئەلەيھىسسالام زۇبەيرگە بۆلۈپ بەرگەن يەردىن ئۇرۇق تۇشۇۋاتقاندا، پەيغەمبەر ئەلەيھىسسالام باشلىق بىر توپ كىشىلەرگە يۇلۇقۇپ قالدىم. پەيغەمبەر ئەلەيھىسسالام مېنى چاقىردى ۋە كەينىگە مېنىۋالسۇن دەپ، تۆگىسىنى چۆكتۈردى. ئەمما مەن ئوڭايسىزلىنىپ قالدىم ھەمدە زۇبەيرنىڭ كۈنلەمچىلىكىنى ئېسىمگە ئالدىم. شۇنىڭ بىلەن پەيغەمبەر ئەلەيھىسسالام كېتىپ قالدى. مەن زۇبەيرنىڭ قېشىغا قايتىپ كەلگەندە، ئۇنىڭغا ئىشنىڭ جەريانىنى سۆزلەپ بەردىم. ئۇ « ئاللاھ بىلەن قەسەمكى، ماڭا نىسبەتەن، سېنىڭ ئۇرۇقنى يۈدۈپ كەلگەنلىكىڭ ئۇنىڭ كەينىگە مىنىشىڭدىن بەكرەك ئېغىر كەلدى.» دېدى. كېيىن، ئەبۇ بەكرى ماڭا بىر خىزمەتچى ئەۋەتىپ بەردى، مەن ئاتقا قارىمىساممۇ بولىدىغان بولدى، ئۆزۈمنى گۇيا ئازات قىلىنغان قۇلدەك ھېس قىلىپ قالدىم.
مۇشۇ سەۋرلەرنىڭ ھەممىسىدىن كېيىن، ئاللاھ ئۇنىڭغا ۋە ئۇنىڭ ئېرىگە ئىلتىپات كۆرسەتتى. ئەمما ئۇ باي بولغانلىقتىن ئۆزىنىڭ مىزانىنى يۇقىتىپ قويمىدى. ئۇ ئىنتايىن سېخى ئىدى، ئەزەلدىن ئالتۇن-كۆمۈش يىغمايتتى. بىر قېتىم ئۇ كېسەل بولۇپ يېتىپ قالدى، ياخشىلانغاندىن كېيىن بارلىق قۇللارنى ئازات قىلىۋەتتى. ئۇ ئۆزىنىڭ قىزلىرىغا ۋە ئۆيىدىكىلەرگە «سەدىقە قىلىڭلار، خەجلەڭلار، ئېشىپ قالغۇچە ساقلىماڭلار.» دەيتتى.

« ھايات مۇئمىنلەرگە نىسبەتەن گۈزەلدۇر، تەقۋادارلار ئاخىرەتنى قىزغىن سۆيىدۇ، ئۇلارلا ھەقىقى بەختلىك كىشىلەردۇر.»
