ھاياتتا شۇنداق كۈنلەر بېشىڭغا كېلىدۇكى شۇنچە كەڭرى جاھان سەن ئۈچۈن تار بىلىندىغان ،بارچە كىشلەر ئۇزاقراق ياشاشنى ئۈمىد قىلسا ،سەن بۇنداق ياشىغۇچە ئۆلۈپ كەتسەمچۇ دەپ ئۆلۈمنى ئارزۇ قىلىدىغان ！！！！
ئەلى ئاپتۇبۇستا ئىككى كۈن يۈرگەندىن كېيىن ئاخىرى ئانا يۇرتىغا قايتىپ كەلدى.
ھەي نىمىلا دىگەن بىلەن ئانا يۇرت مېھرى باشقىچىدە ،يېزىنىڭ دەل-دەرەخلىرى،گۈل-گىياھلىرى ھەتتاكى تۇپا-چاڭلىرىمۇ ئەلىگە شۇنداق گۈزەل شۇنداق ھۇزۇر بىلىنمەكتە ئىدى.
ئەلى ئۆيىگە كەلگەندە كەچ كىرىپ قالغانىدى. ئۇ ھويلىسىغا كىرگەندە ئاپىسى پاتىمىخان قويلارغا يەم سېلۋاتاتتى، ئۇ ئەلىنى كۆرۈش بىلەنلا تۇرۇپلا قالدى ئەلى ئاپىسىنى كۆرۈپ ئاپىسنىڭ باغرىغا ئۆزىنى ئاتتى،سىزنى بەك سېغىندىم ئاپا سىزشۇنداق كۆرگۈم كەلدى ،سىزنى ساق سالامەت كۆرۈشكە نىسىپ قىلغان اللاھقا شۈكۈرلەر بولسۇن.
ئەلىنىڭ ئاپىسى بولسا ئەمدىلەتتىن ئەسلىگە كەلگەن ئىدى،ئۇ ئەلىگە تويماي قارايتتى جېنىم ئوغلۇم مېنىڭ يۈرەك پارەم،مېنىڭ كۆز قارچۇغۇم سېنى قېشىمغا ساق-سالامەت يەتكۈزگەن رەببىمگە چەكسىز ھەمدۇ-سانا ۋە شۈكۈرلەر بولسۇن دەيتتى .
نىمىشقا كېلىشتىن بۇرۇن خەۋەر بېرىۋەتمىدىڭ ئوغلۇم ساڭا تاماق قىلىپ تۇرماسمىدىم دېدى
ئۇ ئەلىگە . سىزنى تۇيۇقسىز خۇشال قىلۋېتەي دېدىم ئاپا دېدى ئەلى.
سېنىڭ ساقلىغىڭ مېنىڭ ئەڭ چوڭ خوشاللىقىم ئوغلۇم مۇشۇ كۈنلەرگە ئېرىشتۈرگەن رەببىمگە شۈكۈرلەر بولسۇن ،ئەپسۇس داداڭ رەھمىتى بۇ كۈنلەرنى كۆرەلىگەن بولسىدى دەپ كۆزلىرىگە ياش ئالدى .
ئەلىنىڭ كۆزلىرىمۇ نەمدەلدى.
ۋاي ما ئەقلىمنى ئوغلۇم كەلسە گەپ سېتىپ ئولتۇرغىنىمنى كۆرۈڭ ،مەن ساڭا ھاردۇق ئېشى ئېتىپ بېرەي بالام دېدى پاتىمىخان.
بولدى ئاپا ئاۋارە بولمىسىڭىزچۇ سىزنى بەك سېغىندىم بىردەم پاراڭلىشىپ ئولتۇرايلىچۇ ئاپا دېدى ئەلى . نەچچە ۋاقىتتىن ياقى سىرتنىڭ تامىقىنى يەپ ئۆينىڭ تامىقىنىمۇ سېغىنىپ قالغانسەن بالام بىردەمدىلا تەييار قىلىپ بولىمەن دېدى پاتىمىخان،ئۇنداقتا مەن سىزگە ياردەملىشىپ كۆكتات ئاقلىشىپ بېرەي دېدى ئەلى。 بولدى بولدى بالام ئولتۇرۇپ ئارامىڭنى ئالغاچ ساۋاقداشلىرىڭغا ساق سالامەت كەلگنلىكىڭنى بىلدۈرۈپ تېلفۇن ئۇرۋەتكىن بالام ئۇلارنىڭمۇ كۆڭلى خاتىرجەم بولسۇن دېدى پاتىمخان.
ھە ماقۇل رەھمەت ئاپا ئەلى شۇنداق دېگىنىچە بەرنا سوۋغا قىلغان يانفۇننى ئېلىپ تېفۇن قىلۋىدى تېلفۇنى ئىتىك چىقتى ، بەلكىم تېلفۇننى تېخى ئاچمىغان ئوخشايدۇ دەپ ئويلىدى ئۇ كۆڭلىدە ،ئاندىن ئۇنىڭغا قىسقا ئۇچۇر يېزىشقا باشلىدى ：ئەسسالامۇ ئەلەيكۇم مەلىكەم قانداق ئەھۋالىڭىز ياخشى تۇرۋاتامسىز ئۆزۈم اللاھنىڭ ئىلتىپاتى بىلەن ساق سالامەت يېتىپ كەلدىم،سىزنى شۇنداق سېغىندىم مەلىكەم يانفۇنىڭىزنى تېزرەك ئېچىپ ماڭا ئۇر قايتۇرۇشنى ئۇنتۇپ قالماڭ سىزنى مەڭگۈ سۆيىمەن مەلىكەم ، ئەلى شۇنداق دەپ ئۇچۇرنى ئاخىرلاشتۇردى . ئاندىن سەمىگە ۋە باشقا ياتاقداشلىرىغا تېلفۇن قىلىپ ئۇلارنى خاتىرجەم قىلىپ قويدى . شۇ ئارىدا   ئاپىسى ھورى چىقىپ تۇرغان ئۈگرىنى ئېلىپ كىردى：ئاددى قىلىپ ھاردۇق ئېشى قىلدىم بالام ھاردۇقىڭنى چىقىرىپ قالسا ئەجەپ ئەمەس دېدى.
رەھمەت ئاپا ئاللاھ ئىگەم سىزنى سالامەت قىلسۇن ، مۇشۇ ئەتكەن تاماقلىرڭىزنى بەك سېغىندىم، ئۇ شۇنداق دەپ تامىقىنى يېيىشكە باشلىدى .
تاماقتىن كېيىن ئۇلار نەچچە ۋاقىتتىن ياقى يىغلىپ قالغان كۆڭۈل سۆزلىرى سېغىنىشلىرى بىلەن قىزغىن پاراڭغا چۈشتى ، بۇ چاغدا ۋاقىتمۇ بىر يەرگە بېرىپ قالغان ئىدى . نامىزىڭنى ئوقۇپ ئارام ئالغىن ئوغلۇم قالغان گەپلەرنى كېيىنچە دىيىشىۋالارمىز دېدى پاتىمىخان . ئەلى ئورنىدىن تۇرۇپ تاھارەتكە سۇ ئېلىپ تاھارەت ئالغىلى چىقتى ،تاھارەت ئېلىپ بولۇپ بەرناغا تېلفۇن قىلۋىدى يەنە ئىتىك چىقتى ، ئەلى نامىزىنى ئۆتەپ بولۇپ تۇرىشىغا ئاپىسى كىرىپ كەلدىدە ،ئورنۇڭنى يان ھۇجرىغا سېلىپ قويدۇم بالام بالدۇرراق ئارام ئالغىن دېدى .
سىزنى بەك سېغىندىم ئاپا بۈگۈن ئاخشام سىزنىڭ يېنىڭىزدا قونۇپ قالايچۇ دېدى ئەلى ئاپىسىنىڭ بوينىغا ئېسىلىپ. ھەي شۇمتەك يەنە ئۆزگەرمەپسەندە تېخى قاچانمۇ چوڭ بولارسەن دېدى پاتىمىخان كۈلۈپ . مەن مەڭگۈ چوڭ بولمايمەن مەن مەڭگۈ سىزنىڭ بالىڭىز دېدى ئەلىمۇ كۈلۈپ . ماقۇل ئەمىسە شۇنداق بولسۇن ئورنۇڭنى قايتا سېلىپ قۇياي پاتىمىخان شۇنداق دەپ چىقىپ كەتتى . ئەلى بەرناغا يەنە تېلفۇن قىلۋىدى يەنە ئىتىك چىقتى ، ئۇ بىئارامسىزلىنىپ قالدى لېكىن ئاپىسنى ئەنسىرەتمەسلىك ئۈچۈن چاندۇرمىدى.
ئەتىسى ئەتىگەن ئەلى ئورنىدىن تۇرۇپ تاھارەت ئېلىپ نامىزىنى ئۆتەپ بولۇپ تۇرىشىغا تېلفۇن سايرىدى ،ئەلىنىڭ يۈرىكى تۇيغاندەك بولۇپ دەرھال تېلفۇنغا قارىۋىدى تېلفۇن ئېكرانىدا《مەلىكەم》دىگەن ئىسىم چىقىپ تۇراتتى ،ئەلى خۇشاللىقىدىن تېلفۇننى ئېلىشنى ئۇنتۇپ قالغىلى تاس قالدى ۋە تېخىچە سايراۋاتقان يانفۇنغا قاراپ ئېسىنى يىغىپ تېلفۇننى ئالدى.
ئەسسالامۇ ئەلەيكۇم مەلىكەم قانداق ئەھۋالىڭىز بەكمۇ ئەنسىرەتتىڭىز مېنى سىزنى بەك سېغىندىم دېدى ئەلى . ۋەئەلەيكۇم ئەسسالام اللاھقا شۈكرى سىزچۇ قانداق ئەھۋالىڭىز موزايچاق ،سىزنى بەكرەك سېغىنسۇن دەپ تېلفۇن قىلمىدىم تازا قانداق بوپتىمۇ ،بەرنا شۇنداق دېگىنىچە تاتلىق كۈلدى. مېنى ئۇنداق سېغىندۇرماڭ مەلىكەم سىزگە پەقەت چىدىمايمەن ئەگەر بۈگۈنمۇ تېل قىلمىغان بولسىڭىز نىمە بولۇپ كېتەتتىمكىن تاڭ دېدى ئەلى. راستما شۇنداق بەك سېغىندىڭىزما ？مەن بەك ئوماق بولغاچقا سېغىنىپ كېتىپسىز ھە دېدى بەرنا  . راست مەلىكەم سىزگە پەقەتلا  چىدىمايمەن دېدى ئەلى . چاقچاق قىپ قويدۇم موزايچاق مەنمۇ سىزنى شۇنداق سېغىندىم ئۆيگە كىلىشىمگىلا تەرەپ تەرەپتىن ئۇرۇق تۇققانلىنىڭ ئۆيىگە چاقىرىپ تېلفۇن قىلالمىدىم ،بۈگۈن ئەتگەن تېلفۇنۇمنى ئېچىپلا سىزدىن كەلگەن ئۇچۇرنى كۆرۈپ بەك خوش بولدۇم، شۇ ھامانلا سىزگە تېلفۇن قىلدىم بوپتىمۇ مەن سىزنى بەك سېغىندىم دېدى بەرنا . ئۇلار شۇ تەرىقىدە قىزغىن پاراڭلىشۋاتقاندا ،ئەلىنى ئاپىسى چاقىرىپ قالدى .
ئاپام چاقىرۋاتىدۇ مەلىكەم چىقىپ چىقاي تېلفۇنىڭىزنى ئېتىپ قويماڭ مەن سىزگە ھېلىغىچە تېلفۇن قايتۇراي ماقۇلمۇ  ？ دېدى ئەلى .
ماقۇل ئەمسە تېلفۇنىڭىزنى كۈتىمەن اللاھقا ئامانەت موزايچاق دېدى بەرنا تاتلىق كۈلۈپ . اللاھقا ئامانەت مەلىكەم ئەلى شۇنداق دېگىنىچە ئاپىسىنىڭ قېشىغا چىقىپ كەتتى 。
