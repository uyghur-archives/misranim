بالىسى بارلارنىڭ بەختىنى بۇزماڭ(سانائىل ھەرىپتىن تۈزۈلگەن سۆز ئويۇنى)

بىتىمىزدىكى«بالىسى بارنىڭ بەختىنى بۇزماڭ»بابىدىن بەخىتنى بىجىرىم بىلەلمىگەنلەر بىلىۋالسۇنكى؛بىلسىڭىز –بەخت،بىسەرەمجانلاشماي،بىر-بىرى بىلەن بوغۇشماي، باشقىسىغا بۇيۇندىماستىن،بالايى-بەتتەرسىز بولغان بۇلسا بۇبەختلىك بولغانلىقتۇر.بىلىپ-بىلمەي بۇرۇنقى بەختىڭىزنى بىلەلمەي،باشقا بىرسى بىلەن بىرگە بولۇپ،بەختىلىك بولالمىغاندەك بىلىنگەندەك بولۇپ،باشقىلارنىڭ بىربەلەن بەختىنى بۇزۇپ،بەزەنبىر بولمىغۇر بىمەنىلىكلەرنى باشلىغان بولسىڭىز..بولدى بەس...بۇنداق بەخت،بەخت بولمايدۇ.

بەخت بىلەن بەخىتسىزلىككە بىر بىشارەت بەرمىسەم بولمىغۇدەك:

بىزنىڭ  بايئېرىقتا بىرباغۋەن بالا بولىدىغان.بۇباغۋەن بالابايئېىرىق باشلانغۇچتىكى باھارگۈل بىلەن بۇرۇندىنلا بىرگە بولۇشقا بىر-بىرى بىلەن باغلاشقان.باغۋەن بالىنىڭ بىساتى بولمىسىمۇ باھارگۈل باشلانغۇچ بازىسىدىكى بىڭسى بىلەن باغۋەن بالىنى بوزەكلىمەي،بارلىقىنى باغۋەن بالىغا بىرىپ بىرگە بولغان.بۇلار بىر-بىرى بىلەن بەكمۇ بەلەن بىرىكىپ،بىر-بىرىنى بوزەكلىمەي،بەك باياشات بولمىسىمۇ بالىلىق بولۇپ بەختلىك بولغان.بىراق بايئېرىقتا باشقىلارنىڭ بالىسى باقىدىغان،باشقىلارنىڭ بىدىسىنى باغلاپ بەش بالىسىنى بېقىۋاتقان بۇمەريەم بۇلارنىڭ بەختىنى بىراقلا بۇزۇۋەتتى.بەختىنى بۇزغانلىقىنى بىزباشتىنراق باشلىمىساق بىلەلمەيسىز.بۇمەريەم بىلەن باغۋەن بالا بۇرۇن  بالىلىقىنى باشلانغۇچتا بىرگە باشلىغان.باغۋەن بالا بارا-بارا بۇمەريەمنىڭ بەرنالىشىۋاتقان بويلىرىدىن بىھوشلىنىپ،بۇمەريەم بىلەن بىرگەبولغۇسى بارلىقىنى بىلدۈرگەندە،بۇمەريەم،باغۋەن بالىنى- «بىجىرىم بالا بولمىدى،بوشاڭ بالا بولدى»غا باھانىلاپ،باشقا بىر بىلىملىك باشلىقنىڭ بايلىقىغا بىرىلىپ،باغۋەن بالىنى بىچارىلاشتۇرۇپ،باشلىق بىلەن بىرگە بولوۋالغان.باشلىق بولسا باشتابۇمەريەمنىڭ بىجىرىم بوي-بەستىگە بىرىلىپ،بەخت بەرگەن بولسا بارا-بارا بۇمەريەمدىن بىزىرىشكە باشلىدى.باشلىق بىردىنلا باشقا بىربۆلۈمنىڭ بىرىنجى باشلىقى بولۇۋېلىپلابۇمەريەمنىڭ بۇقۇلداشلىرىغا باقماي بۇمەريەمگە بىرىدىغاننى بىرىۋىتىپ،باشقا بىرسى بىرسى بىلەن بىرگە بولۇۋالدى.باشلىق بالا بۇمەريەمدىن بەزگەندە بۇمەريەم بەش بالىلىق بولۇپ بولغاچقا،باشلىقتىن باشقا بىرەر باشپاناھى بولمىغاچ،باشقىلارنىڭ بىدىلىرىنى باغلىدى،باشقىلارنىڭ بالىسىنى باقىدىغان بالا باققۇچى بولدى.باغۋەن بالا بولسا باغۋەنچىلىك بىلىملىرىنى بىجىرىم بىلىپ،بايئېرىقتا بىرباغۋەنچىلىك بازىسىنى بەرپالاپ،بازىغاباشلىق بولدى.بولۇپمۇ باھارگۈل بىلەن بىرگە بولۇپ بىر-بىرىدىن بەلەن،بۇدۇرۇق بالىلاربىلەن بەختلىك بولدى.بۇنىڭغا بەرداشلىق بىرەلمىگەن بۇمەريەم بىرەرباھانىلەربىلەن باھارگۈلگە بەزى-بىر بىشارەتلەرنى بىرىپ،بىكاردىنلا باھارگۈلگە«بۇزۇق»،«بىشەم»لىكتە باھالاپ،باھارگۈلنى بوزەكلىدى.باھارگۈل بىلىملىك بولغاچقا بۇمەريەمدەك بۇنداق بىلىمسىزبالاباققۇچى بىلەن بوغۇشمىدى.بۇنداق بولۇۋەرگەنسىرى بۇمەريەم بارا-باراباتۇرلىشىپ،باھارگۈلنىڭ بەختىنى بۇزۇپ،باغۋەن بالا بىلەن بىرگە بولماقچى بولدى.بىكاربولسۇن ،بولمىسۇن باغۋەنچىلىك بازىسىغابېرىۋالدى.باغۋەن بالىمۇ بۇمەريەمنىڭ بىلجىرلاشلىرىدىن،بۇدۇشقاقتەك بىرىكىپ،بىقىنداشلىرىدىن، بۇرۇنقى بەخىتسىزلىكىدىن باتناپ،بىلىپ-بىلمەي بۇمەريەمگە باشقىدىن بىرىلىشكە باشلىدى.بىرىلگەنسىرى بۇزۇلدى.بۇلارنىڭ بەتقىلىقلىرى باغۋەنچلىك بازىسىدا،بايئېرىقنىڭ بۇلۇڭ-بۇلۇڭلىرىدا بىربولدى.بۇنىڭدىن بەكمۇبىسەرەمجانلاشقان باھارگۈل بۇنى بىلمەس بولۇۋالسىمۇ باغۋەن بالىنىڭ بىكاردىنلا  بوزەكلىشىدىن،بۇرۇنقى بەلەنلىكى بۇرمىلىنىپ،باھارگۈلدىن بىزار بولۇشلىرىدىن،باغۋەن بالىغا،بۇمەريەمگە بۇزولماسلىققا،بۇدۇرۇق بالىرىنى بىرگە بېقىپ،باغۋەنچىلىك بازىسىنى بىرلىكتە باشقۇرۇشنى بىلدۈرگەندە،باغۋەن بالا بىردىنلا بىشەملىشپ،باھارگۈلنى بىگىزبارماقىلىرى بىلەن بىقىنداپ بېسىقتۇردى.بولۇپمۇ بىربازاردا باھارگۈل بالىلىرىغا بايرامغا بېغىشلاپ،بىربۆلۈم بىزەك،باشقا بويۇملارنى  بىربوغچىغا باغلاپ،بوخچىنى بىلىكىگە بىجىرىم بىكىتىپ،بازاردىن بايئېرىققا بارماقچى بولۇۋاتقاندا بىردىنلا بۇمەريەم بىدە باسقان بۇقىسىنى بىربۇلۇڭغا باغلىۋىتىپ،باھارگۈلنى بازاردىلا باسماقلىدى.باھارگۈلنڭ بويۇنلىرىدىن بوغۇشقا باشلىدى.باھارگۈلمۇ بوشاشمىدى.بۇلار بىر-بىرى بىلەن بەسلەشكەندەك باتۇرلۇق بىلەن بىر-بىرىنى بوغۇشقاباشلىدى.بازارچىلارمۇ بۇلارنى بېسىقتۇرالمىدى.بۇنداق بىنۇمۇسلۇقنى،بۇنداقلارنىڭ بۇغۇشقىنىنىمۇ بىلمىگەنكەنمەن.بۇغۇشا-بوغۇشا بۇمەريەم بىردىنلا بىلىدىكى بىلەنگەن بەكىسى بىلەن باھارگۈلنىڭ بىلىكىنى بىسلاپ بۇزۇۋەتتى.باھارگۈلمۇ بېشىغا بىكىتىۋالغان بىگىزنى بۇمەريەمنىڭ بېقىنىغا،بىلىگە بولۇشىچە بېسىۋەتتى.بۇمەريەم بولالماي بېلىقتەك بىربۇيانغا،بىربۇيانغا بېغىرلاشقا باشلىدى.بارا-بارا بۇلارنىڭ بۇغۇشۇشلىرى بازارچىلارنىڭ بىرىكىشى بىلەن بسېقتى.بۇنداق بىمەنە،بىشەملىكتىن بەكمۇ بىزاربولغان باھارگۈل؛بايئېرىقتىكى باغۋەنچىلىك بازىسىغا بېرىپلا باغۋەن بالا بىلەن بىرگە بولمايدىغانلىقىنى، بالىلىرىنىمۇ بىرىدىغانلىقىنى، بۇمەريەمنىڭ بىڭسى بولسا بەش بالىسىغا بىرلەشتۈرۈپ،بالىلىرىنىمۇ بېقىپ بېقىشنى باغۋەن بالىغا بىراقلا بىلدۈردى.بۇمەريەمگە بۇزۇلغان باغۋەن بالىمۇ باھارگۈلنىڭ بىلدۈرشى بىلەن،بىردىنلا باشقىچە بولۇپ،باھارگۈلگە باغۋەنچىلىك بازىسى بىلەن بالىدىن باشقىنى بىرىپ،باھارگۈلدىن بوشىنىپ،بۇمەريەم بىلەن بىرگە بولدى.بۇلارمۇ باشتا بىر-بىرىگە بۇرۇن بىرەلمىگەن بەختىنى بەرگەن بولسىمۇ بارا-بارا باغۋەن بالىنىڭ بالىلىرى بىلەن بۇمەريەمنىڭ بالىلىرىنىڭ بىرلىشەلمەسلىكىدە،بۇلارنىڭ بەختى بۇزۇلۇشقا باشلىدى.بۇمەريەم باغۋەن بالىنىڭ بايلىقىنىلا بىلەتتى. بۇمەريەممۇباغۋەن بالا بەرسە بەلەنگە،بەرمىسە بىخلغا بۇرمىلايتى. باغۋەن بالا بارا-بارابۇنداق بۇرۇقتۇمچىلىقتىن بەكمۇ بىزاربولدى.بەرىبىر بۇرۇن بىرگە بولغانلار بىلەن بىرگە بولغاندەك بولمايدۇ-دە.باھارگۈلغۇ بۇنداق بىمەنلىكىلەرنى بىلمەيتى.باغۋەن بالا بەرسۇن،بەرمىسۇن بۇمەريەمدەك بۇنداق بىمەنە بىلجىرلىمايتى.باغۋەن بالا باھارگۈلنىڭ بەكمۇ بەلەنلىكىنى بىلىپ،باشلانغۇچقا بېرىپ ،بۇمەريەمدىن بىزاربولغانلىقىنى،باھارگۈل بىلەن باشقىدىن بىرگە بولۇپ،بالىلىرىنى بىرگە باقىدىغانلىقىنى بىلدۈرگەندە،باھارگۈل باغۋەن بالىنىڭ بۇرۇنقى بەتقىلىقلىرىنى بىلگەچكە،بىردىنلا باغۋەن بالا بىلەن بىرلەشمەيدىغانلقىنى،بەك بولمىسا بالىلىرىنى بەرسە باغۋەن بالىسىزمۇ بىجىرىم باقىدىغانلىقىنى بىر-بىرلەپ بىلدۈردى.باغۋەن بالابۇنىڭدىن بەكمۇ بوشىشىپ،بىتاپ بولدى.باھارگۈل بولسا بالىلىرىنى بېقىپ،باغۋەن بالىدەك بۇنداق بىۋاپا بىلەن بىرگە بولمىدى.

بايانىمدىن بىلىۋالغانسىلەر..بەلكى بەختىنى بىلەلمەيۋاتقان،بىلسىمۇ بىلىپ-بىلمەي باشقىلارغا بويۇنداپ،بۇزۇلۇشقا باشلاۋاتقانلار باغۋەن بالىدەك بىچارىلىكتە بەربات بولماسلىقنى بىكىتمەكچى بولساڭلار..بەختىڭلارنىڭ بىربۆلىكى بولغان بولماقتەك بالىلىرىڭلارنى بىچارىلاشتۇرماي،بالاڭلارنى بېقىڭلار...بەختىڭلارنى بۇزغان باشقىلار بۇمەريەمدەك بولىدۇ.بالىلىرىڭلارنى بەلەن باقمايدۇ.
