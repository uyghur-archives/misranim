مەن تېمىنىڭ يىرىمىغىچە ئوقۇپلا بۇ ياقنىڭ غەرىپنىڭ مەدىنىيتىگە قىززىقۇچىلاردىن ئىكەنلىكىنى ھىس قىلىۋاتىمەن،سىلەرنىڭ نەزىرىڭلاردىكى مەدىنىيەت پەقەت ئىشەكتەك ھاڭراش بىلەن مايمۇندەك غىلجىڭلاشلار بولىدىغان بولسا، مەن شۇنداق مەدىنىيەتسىز، ئاڭسىز ،خۇراپى ،گۇپپاڭچى بولوپ ياشاشقا،"ئۇيغۇرنىڭ سەنئەت ئارقىلىق تۇنۇلىشىغا قارشى تۇرىدىغان مۇناپىق" دىگەن نامنى ئىلىپ ياشاشقا (ئاللاھ نامى بىلەن قەسەم بولسۇن) مەن ئاللاھ رازىلىقى ئۈچۈن مەن رازى!!!!ئەمما توغرا گەپ !قايسى داڭدار دۆلەتنىڭ داڭلىق بولوشىدا سەنئەت مۇھىم ئورۇننى ئىگەللەپتىكەن ھە؟مەن بىر ھەدىس ئىلىپ مىسال كەلتۈرەي،«ھالال ئىنىق،ھاراممۇ ئىنىق،ئەمما ئارىسدا شۈبىھىلىك نەرسىلەر بار»،«مۇشۇ شۈبھىلىك نەرسىلەردىن ئۆزىنى تارتقاندىلا ئاندىن ھەقىقى مۆئمىن بولالايدۇ،ھارام نەرسىلەر بولسا ئاللاھنىڭ چارگاھى ھىسابلىنىدۇ،شۈبھىلىك نەرسىلەرنى تەرىك ئەتمىگۈچى بولسا پادىچىغا ئوخشايدۇ،ئەگەرپادىچى پادىسىنى ئۇنىڭدىن  تاتمىسا پادا ھەر قاچان چارىگاھقا كىرىپ كىتىشى مۈمكىن».مانا،يەنە ئىيتىلدۇكى«پەيغەمبىرىمىزبىر رەسسامنىڭ ئالدىغا كىلىپ سىنىڭ بۇ قىلغانلىرىڭ ھارام،سەن سىزغان جانلىقلارغا قىيامەتتە جان كىرگۈز دىيىلىدۇ،سەن قىلالمايسەن" دىدى،رەسسام تىتىرەپ رەڭگى ئۆچۈپ كەتتى،ئۇنىڭ ھالىنى كۆرگەن پەيغەمبىرىمىز :ھالىڭغا ۋاي،سەن جانسىز نەرسىلەرنىڭ سۈرىتىنى سىزىپ پۇل تىپىپ تۇرمۇشڭنى قامدىساڭ بولىدۇ»دىگەن،مانا شۇ جەھەتتىن ئىلىپ ئېيتقاندا پەقەت رەسساملىق شۇ جەھەتتىن ھالال سانىلىدۇ،ھەيكەلتاراشلىق  بولسا بۇتلارنىڭ پەيدا بولىشىغا سەۋەپ بولىدىغان ئىش،ئەگەر ھەيكەلتاراش مۇسۇلمان بولسا چۇقۇم ئۆيىدە ئۆزى ياسىغان ھەيكەللەرنى ئۆيىدە ساقلىمايدۇ.شۇنداق ھالەتتە ئۆزىنى مۇسۇلمانمەن دەپ ئۆز ئۆيىدە ساقلىمىغان ئاددم ئۇنى باشقىلارغا راۋا كۆرىسە ئۇ قانداق گۇناھ؟بىلىپ تۇرۇپ قىلغان گۇناھ بىلمەستىن قىلغان گۇناھقا قارىغاندا ئىغىرراق.مانا بۇ بىزنىڭ بەزىلەرنىڭ نەزىرىدە «دىننى دەستەك قىلىپ تۇرۇپ،مىللىتىمىزنىڭ مەدىنىيتىگە ھۇجۇم قىلماقچى»بولالىشىمىزدىكى ئاساسلىرىمىز،ئەمما ئۇيغۇرنىڭ مەدەنىيتى، دۇنياغا ئۇنى تۇنۇتىدىغان ئىسىل مەدىنىيتى ئۇنىڭ دىننى ئىتىقادىدىن تاشقىرى بولمىغان ئىسىل ئۆرۈپ ئادەتلىرى.بەزىلەر بۇ يەردىكى ھۇجۇم قىلىش نىشانى ئىچىگە يەنە نەۋرۇز بايراملىرىمىزنىمۇ تىلغا ئىلىپ ئۈلگۈرۈپتۇ،بۇ يەردە سەنئەت تىلغا ئىلىنىۋاتىدۇ،ئەمما نەۋرۇز توغۇرلۇق ئىشقىلىپ باشقىلار قانداقكىن ئۇنى مەن سەنئەت،ناخشا مۇزىكا قاتارلىقلار بىلەن بىر ئورۇنغا قويۇپ تەھلىل قىلىشنى خالىمايمەن،چۈنكى ئۇنى بۇلارنىڭ قاتارىغا قويۇۋالساق، بۇ يەردە يەنە ئۇيغۇر مەدىنىيتى ھەم ئۆرۈپ ئادىتىگە تاقىشىدىغان "ئۆلۈم يىتىم، نەزىر چىراق"،"كىيىم كىچەك" قاتارلىق نۇرغۇن مىللى ئەنئەنلىرىمىزنى سۆزكۈپ چىقىشقا توغرا كىلىدۇ،شۇڭا بىز مۇشۇ نەق تېما ئۈستىدە توختالساق ئۇيغۇر سەنئەت ئارقىلىق دۇنياغا تونۇلغان بولسا،رەجەپ ئەردوغانمۇ مۇسۇلمانلىق يولىنى ھاكىمىيەت يولىدىن ئەلا بىلمىگەن بولاتتى.ئۇيغۇرمۇ شۇنچە ئۇزۇن يىللىق سەنئەت تارىخى بىلەن ئالدىنقى قاتاردىكى مەدىنىيەتلىك مىللەتلەر قاتارىغا كىرگەن بولاتتى،ئەمما ساندا بار ساپادا يوق ناخشا مۇزىكىللىرىمىز بىلەن ئەمەس ئۇنىڭدىن باشقا نۇرغۇن ئىسىل مەدىنىيتىمىز،يىزىقچلىقىمىز شۇنداقلا ئادىمىيلىكىمىز قاتارلىق جەھەتلەردە بىز ئۆز مىللى كىملىكىمىزنى تۇنۇيمىز ھەم تۇنۇتىمىز. بىز ئۆزىمىزنى يۈكسەلدۈرۈشنى ئويلىماستىن غەرىپ مەدىنىتىنى مەدھىيلەپ ئۆتىدىغان،ئاغزىمىز ئىچىلسىلا ئامرىكىنىڭ ئادەملىرى مۇنداق مەدىنىيەتلىك،گىرمانىيە كىشلىرى مۇنداق ئەخلاقلىق،ئەنگىلىيە خەلىقى مۇنداق جەسۇر دەپ شۆلگەيلىرىمىزنى ئىقىتىپ خەقنى ماختاپ ئۆتىمىز،مىللىتىمىز دەپ قويسا بارلىق يۇرىت كىشلىرنىڭ ئەيىپىنى بىر بىرلەپ ئىىچىپ بىرىدىغان،ئۇ دۇنيا دەپ سۆزلىسە«كىم ئۇ دۇنياغا بىرىپ كەپتىكەن،ئاللاھنى كۆزىڭىز بىلەن  كۆردىڭىزمۇ؟ئۇ دىگەن پەقەت قەلپتىكى ئىتىقاد خالاس»دەپ بارلىكى ئىلىمنىڭ ھەممىسى قۇرئان،ھەممىسى ھەدىستە ئۆز نۇرىنى دۇنياغا تارقاتقان.شۇڭا بىز قورىقمايمىز، باشقا مەدىنىيەتلەرنى ئىنكار قىلمىدۇق،"ئاتالمىش ناخشا مۇزىكا بىزنىڭ روھىمىزنىڭ ئارامى
