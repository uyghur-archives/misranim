مەن بىر چاغدا مەلۇم بىر ماتىريالدىن ، قورساقتىكى بالىغا قوشاق ۋە خەلق چۆچەكلىرىنى ئوقۇپ بەرسە تۇغۇلغاندىن كېيىن ئۇنى ناھايىتى ئاسانلا يادلىۋالالايدىغانلىقىنى ، ھەتتا بىر ھامىلدار ئانىنىڭ ھامىلىگە مەلۇم بىر ئەسەرنى ئوقۇپ بەرگىنىدە بالا تۇغۇلۇپ تىلى چىققاندىن كېيىن ھېلىقى ئەسەرنى يادلاپ بەرگەنلىكىنى ئوقۇغان ئىدىم .

‹‹ سۇ ھەممىنى بىلىدۇ ›› ناملىق سىن ماتىريالىدىن بىز جانسىز دەپ قارايدىغان سۇنىڭمۇ خۇددى ئىنساندەك ئىنكاس قايتۇرالايدىغانلىقىنى كۆرۈپ يىغلىۋەتكەن ئىدىم ، ئۇنىڭغا قارىغاندا ، ھامىلە دېگەن ئاللاھ ئاپىرىدە قىلغان ھاياتلىق ، ئۇ گەر چە ئانا قورسىقىدا بولسىمۇ ئانىنىڭ ھەركىتىگە ، گەپ - سۆزىگە ئىنكاس قايتۇرالىشى ئەلۋەتتە مۈمكىن بولىدىغان ئىش- تە ، بۈگۈن ئەركىن ئاكامنىڭ بۇ تېمىسىنى كۆرۈپ بۇرۇن ئوقۇغان شۇ يازمىنى شۇ ھامانلا يادىمغا ئالدىم .

ئەپسۇس ، بىزنىڭ ئارىمىزدىكى بەزى  ئاتا - ئانىلار تولىمۇ مەسئۇلىيەتسىز ، بالىنى تېپىپ قويۇپ تەربىيىلەشكە كۆڭۈل بۆلمەيدۇ . بىر دوستۇمنى يوقلاپ بارسام دوستۇم بىر خوشنىسىنىڭ گېپىنى قىلىپ بەش پەرزەنتلىك بولغانلىقىنى بىرىنىمۇ ئوقۇتمىغانلىقىنى ئېيتىۋىدى ، ئىچىم شۇنداق ئاچچىق بولدى .
پەن - تېخنىكا ئۇچقاندەك  تەرەققى قىلىۋاتقان مۇشۇ دەۋردە مىللىتىمىز ئىچىدىكى بەزى كىشىلەر نادانلىق قىلىپ بالىلارنى ئوقۇتمايۋاتىدۇ ، ھەپتە ئارىلىقلىرىدا كوچا - كويلاردا لاغايلاپ يۈرگەن ئوقۇش يېشىدىكى بالىلارنى كۆرسەم يىغا تۇتۇپ كېتىدۇ ، ئاتا - ئانىسىنى تېپىپ گەپ قىلغۇم كېلىدۇ ، لېكىن بەزى كىشىلەرنىڭ ئىدىيىسى زامانننىڭ ئېقىمىغا تەتۈر بولۇپ قاتماللىشىپ كەتتى ، قاراپ تۇرۇپ مىللىتىمىزگە مەسئۇلىيەتسىزلىك قىلىۋاتىدۇ .
بۇ ئەھۋاللار تېزرەك ئوڭشالغان بولسا ، ئوقۇشسىز قېلىۋاتقان ئاشۇ پەرزەنتلىرىمىز مەكتەپ قوينىغا قايتىپ كەلسە ، سۆيۈملۈك بالىرىلىرىمىزنىڭ مەكتەپتە ، ئائىلىلەردە كىتاپ ئوقۇۋاتقان جاراڭلىق ئاۋازىنى ئاڭلاپ تۇرساق ، قانداق ياخشى بولار ئىدى - ھە؟!

ئەركىن ئاكىمىزنىڭ ئالدىراشلىقتىن ۋاقىت چىقىرىپ بىز ئۈچۈن قىلىۋاتقان مىھنەتسىز ئەجىرى يىراق بولمىغان يىقىن كەلگۈسىدە ئۈنۈمىنى كۆرسەتكۈسى.سالامەت بولۇڭ ئاكا.



بۇرۇن كىچىك ۋاقىتلىرىمدا يەنى ئىغى ئوچۇق ئىشتان كىيىپ تايتاڭلاپ يۈرگەندە ئاپامدىن ھىكايە ئاڭلاپ ئۇخلاشقا كۆنگەن ئىدىم.ئۇ چاغدا مەن تۇغۇلغان تاغلىق رايوندا كىتاپ بەك ئازمۇ ئىشقىلىپ....قارا قەلەم باركەنتۇق.بارماق بىلەن تۇتقۇدەك يىرى قالمىغۇچە جېجېلاپ بىر نىمىلەرنى قىلاتتىم.شۇنداق بولغاچقىمىكىن مەكتەپ يىشىغا تۇشقاندا(ئۇ چاغدا يەتتە سەككىز يىشىمىزدا بىرىنجى سىنىپقا كىرەتتۇق.يەسلى يوق)خەتنە قىلدۇرۇپ ئەتىسىلا مەكتەپكە بىرۋالغان ئالچاڭشىپ.ئىشقىلىپ ئوقۇشقا قىزىقاتتىم.ئۇ چاغدا ئاتا ئانىللىرىمىزنىڭ باللىرى چوڭ مەكتەپكە كىرگەنلىرىگە:كىيىن بالاڭغا كىتاپ ئۇقۇپ چۆچەك ئىيتىپ بەرسەڭ بالاڭ ئەقىللىق بولىدۇ دىيىشكىنىنى ئاڭلايتىم.ھەي.....كىيىن ئۇ چاغدىكى ئائىلە شارائىتىمىز نامېرات.كىتاب دەپتەر ئوقۇش قوراللىرىدىن تارتىپ كىتاپ سومكىلارغىچە ھەممىنى سىتىۋالىدىغان گەپ.شارائىت نۇرغۇن ئائىلىنى قىيناپ قوياتتى.بىراق يەنە كۆپىنچىسى قەرىز ئىلىپ دىگۈدەك يۇقۇرىلاپ ئوقۇپ نەتىنجە قازاندى.مەن ۋە مەندەكلەر ھۈنەر بىلەن ئىش ئوقەتتە ھۆكۈمەتكە تايانماي تىرىشىپ ئىشلەپ ئائىلە قۇرالىدۇق.كۈنىمىز خۇداغا شۈكىرى.يەنە بەزىلىرى شارائىتى بار باللار ئوقۇماي ئاتا ئانىسىنى قاقشىتىپ يۈرىيدۇ.
ھازىر شارائىت بۇرۇنقىدىن ياخشى.ئوقۇش ماتىرياللىرى سەرخىل.بۇرۇنقىدەك سائەت تۇققۇز ئوندىن كىيىن توك تۇختايدىغان ئىش يوق.بىراق بەزى باللارغا نىمە بولدىكىن ئۇقۇشقا قىزغىنلىق يوق.ئاتا ئانىنىڭ زورى بىلەن مەجبۇر بارىدۇ مەكتەپكە.ئاتا ئانىللىرى باللىرىغا كىتاب ئوقۇشنى ئەمەس ئائىلىدە بىر موچەننى ئىككى قىلىشنىڭ سودىسىنى تۈزەپ يۈرگەچكە باللاردىمۇ پۇلدىن ئىبارەت بۇ كۈچكە قىزىققانلىقمۇ بىر مەسىلە.پۇلمۇ موھىم.لىكىن كىچىكىدىن بالىنى كىتاپقا قىزىقتۇرالىساق شۇ كۈچ ئاخىردا بالىنىڭ كىيىنكى ئىستىقبالىنى بەلگىلەپ قويىدىكەن.

ئاڭلىسام ئامېرىكىدا باللار 18 گە كىرگەندىن كىيىن ئاتا ئانىسىدىن ئايرىلىدىكەن.يەنى مۇستەققىل تۇرمۇشنى باشلايدىكەن.ئۇ  يەردە مەكتەپتە ئوقۇش ۋاقتى بۇ يەردىكىدەك چۈشتىن بۇرۇن تۆت سائەت چۈشتىن كىيىن ئۈچ سائەت ئوقۇماي چۈشكىچە ئوقۇيدىكەن.چۈشتىن كىيىن قويۋىتىدىكەن.شۇنداق بولغاچقا مۇستەققىل بولغان بۇ بالا مەكتىپىدە چۈشكىچە ئوقۇپ چۈشتىن كىيىن بىرەر يەردە ۋاقىتلىق ئىشلەش بولامدۇ ئىشلارنى قىلىپ پۇل تاپىدىكەن.ياكى مەكتەپ ئوقۇغۇچىلارغا شارائىت ھازىرلاپ بىرىدىكەن.شۇ تاپقان پۇلى بىلەن ئۆينىڭ ئىجارىسى،تۇرمۇش خىراجىتى ئوقۇش پۇلىسىنى تاپىدىكەن.ئامېرىكىدا يەنە ئۇقۇتقۇچى ئوقۇغۇچىلارغا بۇ يەردىكىدەك تازا بىر نەچچە سائەت مۈكچەيمىسە تۈگىمىگۈدەك تاپشۇرۇق بەرمەيدىكەن.ئىشقىلىپ شۇنداقراق ئاڭلىدىم.بۇ يەردە ھەممە جاپاغا ئاتا ئانا بىچارە چاپمىسا ئىش پۈتمىگەن.ھىلى مەكتەپتىن قىچىپ كىتىپ ئۆيگە ۋاقتىدا كەلمەيۋاتقان.ھىلى بىر يەرلەردە بىر كىملەر بىلەن ئۇرۇشۇپ ساقچى چاقىرتىپ يۈرگەن.ھىلى پۇل دىسە ئاتىسى بىرىپ بولالمىسا ئۆيدە پۇللۇق نەرسىنى ئەرزان سىتىۋىتىپ يوقىتىۋاتقان.بىچارە ئاتا يەنە شۇ سىتىۋىتىلگەن نەرسىسىنى ئالغان ئىز پۇلىغا ئۆيگە ئەكەلگەن.ئۇراي دىسە بالا كىچىك،ئەقلىنى بىر كۈنى تاپار دەپ قاتتىق بولالمىغان.مەكتەپ پۈتكەندە ئىش ئىزدىمىگەن.يوغانغا كۈچ يەتمىگەن.كىچىكنى ياراتمىغان.يىشى ئوتتۇزدىن ئاشسىمۇ ئاتا ئانىسىنىڭ بىقىنىدىن چىقالماي يۈرگەن.بۇ بىچارە ئاتا ئانا ئاقارغان چاچ ساقىلىنى سۆرەپ قىزغا ئەلچىلىكتە سوراپ يۈرگەن.نىمە قەرزى بولغاچقا شۇنداق كۆتۈردىكىن بالىسىنىڭ دەردىنى.

ئامېرىكىدا باللىرى چوڭ ئائىلىدىن مۇستەققىللىق جاكارلىغاندىن كىيىن تۇرمۇش ئايرىم.ئىشلەپ ئوقۇش جەريانىدا كۆپ ئەمىلىى بىلىمگە ئىرىشىدىكەن.شۇنداق ئۇقۇش ئىشلەپ يەنە ئۇقۇپ يۈرگەن ئامېركىلىقلار شوڭا دۇنياغا خوجا.بۇ يەردە بالام دەپ بالىسىنىڭ پوق سۈدۈكىدىن تارتىپ تازلاپ چوڭ قىلغان ئاتا ئانا ئۇقۇشتىن تارتىپ خىزمەت ئىزدىشىگىچە ھەم بالىسى قۇڭالتاق پۇل تاپماي خۇتۇن ئىلىپ بەر دىيىشلىرىگىچە كۆتۈرۈشلۈك.
بۆرىمۇ بالىسى ئانچە مۇنچە بىر كىچىك ھايۋاننى ئوۋلىيالىغۇدەك ھىچ بولمىسا كۈچى يەتمىگۈدەك بولسا قاچالىغۇدەكلا بولسا ئانىسى بالىسىنى چىشلەپ ئۇۋىسىدىن يىراق يەرگە تاشلىۋىتىدىكەن.بالىسى شۇ يەردە قىقىلىپ سۇقۇلۇپ ئاچ تۇق قىلىپ سوغۇق ئىسسىقتا يۈرۈپ ئاخىرى ياراملىق بۆرە بولىدىكەن.بىزنىڭ بۇ يەرگە گەرچە ئامېرىكا مائارىپ ئەندىزىسىنى كۆچۈرۈپ ئەكىلىش مۇمكىن بولمىسىمۇ ئەمما بالىنى كىچىكىدىن مۇستەققىل خارەكتىردە چوڭ قىلىشقا بەل باغلاش ھەم شۇنىڭغا تىرىشچانلىق كۆرسىتىش ھەقىقەتەن كىلىدۇ قىرىنداشلار.بالىمىز پەقەتلا كىچىك بالا بولسۇن ئەمما چوڭ بولغاندا ئىغىرنى سالىدىغان بالا بولمىسۇن.بالام ھۇرۇن دەپ ئۇھسىنىشتىن ساقلىنىش ئۈچۈن تەربىيەدە ئۇسۇلنى ئۆزگەرتىپ قىرىغاندا بالام تىرىشچان دەپ پەخىرلىنىدىغان بولايلى.

چوچەكلىرى بىلەن قاتارغا قاتتى  شۇڭا شۇ دەۋىردە ياشىغان ئاتىللىرىمىز  . تۇرمۇشنىڭ  ئىسسىق سوغىقىنى يەتكىچە تارتقان بولسىمۇ 'يامان بولىدۇ ؛دىگەن ئاتام بومبىسىدىن كۇچلۇك ئىبارىلەر بىلەن پەرزەنتلىرىنى تەربىلەپ قاتارغا قاتتى . شۇ دەۋىرنىڭ ئادەتتىكى پۇقرالىرىدىن تارتىپ ئولىما ئاخۇنلارغىچە ئىسلامدا 'ئوتتۇر يولنى؛ تاللاپ .تۇرمۇشتا 'شۇكرى' نى دوس تۇتۇپ ،ئەتراپىدىكى ھەر مىللەت ئاممىسى بىلەن ئىناق ئىتىپاق ياشاپ قاتتىق كۇنلەرنى مەنىلىك ئوتكۇزۇپ ،مۇشۇ زامانغا ئۇلاشتى .80- يىللاردا ئوزىنى ھورلىككە چىققاندەك  ھىس قىلغان ئاتىللىرىمىز ،پۇتۇن دىققىتىنى بالا تەربىلەش بىلەن تۇرمۇشنى ياخشىلاشقا سەرىپ قىلىپ خىلى ھاللاندى . ئەپسۇس  90-يىللارغا كەلگەندە ئوزى چوڭ بولىۋالغان باللىرىمىز تەدرىجى ئىسلامدىكى ئوتۇرا يۇلنى تاشلاپ ئاشقۇن ئىدىيەنى قۇبۇل قىلىشقا باشلىدى . شۇنداق قىلىپ ھازىرقى دەۋىرنى رۇياتقا چىقاردى .خوش مەدىنىيىتىدىن ياتلىشىپ . ئوزىدىن نەچچە مىڭ كىلومىتىر يىراقتىكى ئولسە قىنى قوشۇلمايدىغان ،قالاق ئەرەپ ،ئاۋغاننىڭ مەدىنىيىتىنى ئۇلۇغلاپ مىللەتتىن يىراقلىشىشقا مىللەتنى پارچىلاشقا يۇزلەندى ، زامان ماكان
