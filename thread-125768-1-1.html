ئەسسالامۇ ئەلەيكۇم دەسلار ئادەم ھاياتىدا نۇرغۇنلىغان ئىسسىق سوغۇقلارنى بىشىدىن،ئۆتكۈزىدۇ بۇلارنىڭ ئىچىدە يىتىمچىلىك،مۇھەببەت،بايلىق،يوقسۇزلۇق،دىگەندەك كۆپلىگەن ئامللاربار بۇلارنىڭ ئىچىدە ئەڭ نازۇك بىرى باربولۇپ ئۇ بولسىمۇ مۇھەببەت،مانا مىنىڭ يازماقچى بولغان ئەسىرىممۇ دەل ئاشۇ مۇھاببەتنىڭ ئىگىز پەس داۋدانلىرىنى باشتىن ئۆتكۈزگەن بىر،مىسكىن يۈرەكنىڭ شاتلىق ۋە قاۇغۇلىرى.....

بىزنىڭ مەھەللىدە بىرنەچچە باللار باربولۇپ بۇلارنىڭ ئىچىدە مەن ئوتتۇرانجىسى ئىدىم،بىزنىڭ مەھەللىمىز تولىمۇ ئاۋات ئادەملىرى مىھمان دوس،مەنزىرلىك تولىمۇ چىراقلىق ئىدى كەچ ۋاختلىرى بولسا مەھەللىمىزدىكى بالىلار بىر بولۇپ ئەتىراپنى بىشىمىزغا كىيەتتۇقتە،ئىشىك ئالدىغا شامالدىغىلى چىققان چوڭلار بولسا ۋوي پالانى(تاز) بوش ۋاقىرە ئادەمنىڭ بىشىنى ئاغىرتمايدىسە يەنە بىرلىرى ھەي بالام پۇستانى گۈل قىزىم ماياققا ئۆتىڭە،يىلى بىكا يىغلايسىز دىگەندەك بىرنەچچە ئىغىز ۋاقىراپ قويۇپلا يەنە ئوزلىرىنىڭ پاراڭلىرى ئۆستىدە مۇنازىرلىشەتتى،بىزبولساق چوڭلار توسقاندا ئانچە مۇنچە توختاپ قويۇپلا يەنە ئۆزىمىزنىڭ ئوينىنى ئويناۋىرەتتۇق،مانامۇشۇ ئىللىق مىھرى ئىسسىق بىزنى ئۆزقوينىدا ئەركىلىتىپ بىقىپ چوڭ قىلغان مەھەللىمىزدە يەنە مەن بىلەن بىرياشتا زورەم ئىسىملىك بۇقىزمۇ مەھەللىمىزنىڭ بىردىن بىر غۇنچىسى بولۇپ چوڭ بولىۋاتاتتى ،كۈنلەرنىڭ ئۆتىشى بىلەن مانا ھەش-پەش دىگۈچە مەنمۇ مەكتەپ يىشىغا بىرىپ قالدىم،ئەتىرەت باشلىقى ۋە مەھەللە كادىرلىرىنىڭ،يىتەكلىشىدە بىزمۇ مەكتەپكە باردۇق،مەن بىلەن تەڭ ياشتىكى ۋە چوڭ كىچىك ئاساسى مەكتەپ يىشىدىكى باللىلارنىڭ ھەممىسى دىگىدەك مەكتەپكە قوبۇل قىلىندى،مانا شۇنىڭ بىلەن بۇ گۈزەل مەھەللىدىكى ۋاراڭ چۇرۇڭلارمۇ بىرمەھەل پەس كويغا چۈشۈپ قالدى،مەن چۈشلۈك دەم ئىلىشتا ئۆيگە قايتقىنىمدا بولسا ئەتىراپتىكى چوڭلارنىڭ بىرسى ۋوي تاز مەكتەپتىن قايتتىڭما دىسە بەزىلىرى،ھە بالام دەم ئىلىش قىلغان ئوخشىمامسەن جۈرە ئۆيگە كىرىپ ئۇسۇزلۇق ئىچىۋال دەيتتى،شۇنىڭ بىلەن بىرگە (ھەسەن)كاسكىمۇ ئەنچە مۇنچە چاقچاق قىلىپ ۋوي مەكتەپ مايمۇنى تۇرۋاخۇن ئاكامنىڭ كىچىك تىزى كەلدىڭما،سەن تاز يوقزە قۇلاقمۇ خىلى تىنجىپ قالدىجۇما ھى ھى ھى ھى چاقچاق ھە بالام،ماڭە ئاۋۇ ئۈجمىدىن ئىلىپ بىراي دەيتتى،مەن رەخمەت ئاكا ئۆيگە قايتمىسام بولمايدۇ چۈشلۈك تاپشۇرقۇم بار دىگەن پىتىم كىتىپ قالاتتىم،ھەسەنكام بولسا كۈلۈپ قويۇپ ئاچچىق موخۇركىسىنى گىزىتنىڭ پارچىسىغا ئوراپ ئوتتۇتاشتۇرۇپ ئىغىر ئىغىر شورغىنىچە تۇرۇپ قالاتتى،دادامنىڭ ئىيىتىپ بىرىشچە بۇ ھەسەن كاسكى ھەسەنكام باشقا يۇرتلۇق بولۇپ ئازاتلىقتىن كىيىن،ئوتۇن باسقان تۆت ئاتلىق ھارۋا بىلەن مۇشۇ مەھەللىگە پەيدابولغانچە مۇشۇيەردە تۇرۇپ قالغان ئىكەن ئۇنىڭ ئەسلى ئىسمى ھەسەن يىلىم بولۇپ،تولا گەپ قىلىپ يۈرگەچكە كىشىلەر ئۇنى ھەسەن كاسكى دەپ ئاتىشاتتى،ئۇ ئۆزى ئۇنداق كاسكى بولغانبىلەن مەھەللىمىزدە كىشىلەرنىڭ سۈيىنى توشۇپ مىلىنى بىقىپ،دىگەندەك ئىشلارنى قىلىپ مەھەللىمىزنىڭ بىر ئەزاسى بولۇپقالغان ئىدى،بەزىدا دادامبىلەن ھىسەنكام ئۇزىندىن ئۇزۇڭغا مۇڭدۇشۇپ ئولتۇرۇپ كىتەتتى،ئۇلارنىڭ ۋوي تۇراپ ئاخۇن ئاڭلىدىلىما ما يىقىندا ئەمىركىدا،«ئامېركا دىمەكچى»بىرنەرسە پەيدابولۇپتۇ ئۇنىڭغائادەم چىقىرىپ ماڭامىش،دىسە دادام ياقىسىنى چىشلىگىنىچە ياۋ  توۋا مۇشۇ چەتەللىك دىگەننى ئاشۇ،ھە ھەسىناخۇن سىلىمۇ بىلىلىغۇ جاڭجىشىمۇ ئاشۇنداق بى نەرسە بىلەن،يىلىقى تايۋان دەمدۇ بىريەرگە قىچىپ كىتىپتىغۇ«تەيۋەن»دىمەكچى:
دىگەندەك مۇشۇنداق پاراڭلارنى سىلىشىپ ئولتۇراتتى،مەن بولسام بۇلارنىڭ پاراڭلىرىنى گەرچە چۈشەنمىسەممۇ بىرە خوشنىمىزنىڭ بالىسى سىرتقا چىققۇچە ئاڭلاپ ئولتۇرۇپ كەتسەم،بەزىدە مىھرىبان سۆيۈملىق دادامنىڭ يوتىسىغا بىشىمنى قويغۇنۇمچە مىلىق مىلىق ئۇيقۇغا غەرىق بولاتتىم،شىرىن ئۇيقۇدا ئۇخلاۋاتقىنىمدا دادام تولىمۇ يىقىملىق ئاۋازدا بالام ئاندۇسوپۇر دىدىدە يۈزلۈرۈمنى سىلىغاچ مەن كۆزۈمنى مىڭبىر تەستە ئىچىپ دادامغا قاردىم،دادام تۇرە بالام ئاۋۇ ئادىشىڭ ئوينىغىلى چىقىپتۇ سىنى چاقىرۋاتىدۇدىدى،مەن چالا ئۇيقۇلۇق چاپاق كۆزلىرىمنى ئىچىپ قارغىنىمدا مىنىڭ جانجىگەر دوستۇم مۇھەممەت ئىلى مىنى ساقلاپ قاراپ تۇراتتى،مەن ئۇنى كۆرۈپلا ئورنۇمدىن چاچچاپ تۇردۇم ھەمدە دادا ئۇماق دادا مەن ئويناپكىلەيچۇ دىدم دادامنىڭ بوينىغا ئىسىلىپ دادام بولىدۇ باغىن بالام بىرىپ ئوينا لىكىن يىراققا كەتمە جۇمۇ دىدى مەندىن ئەنسىرگەندەك قىلىپ مەن ماقۇل دىگىنىمچە يۈگۈرۈپ ماڭدىم،دادام ۋوي تاز قوسقۇڭنى تويغۇزىۋال راس دىدى مەن بولدى قوسقۇم توق دىگنمچە دوستۇمنىڭ قولىنى تۇتۇپ يۈگۈرۈپ كىتىپ قالدىم بىز ئاززىراق ماڭغاندىن كىيىن كونا ئادىتىمىز بويچە مەھەللىمىزدىكى قىرى ئۈجمىنىڭ تۈۋىگە يىغىلدۇق بۇباھار ئايلىرىنىڭ خىلى ئىللىپ قالغان  كۈنى بولغاچقا بىز ھەممىمىز ئۇچىمىزدىكى ئىغىر قولايسىز قىشلىق كىيىملەرنى سىلىپ تاشلاپ يەڭگىل ئەركىن ئازادە،ئويناش ئۈچۈن تەمشەلدۇق  ،شۇنىڭ بىلەن بىزنىڭ ھەربى دۈشمەن ئويىنىمىز باشلىنىش ئالدىدا تۇراتتى مەن يەنىلا باش قوماندان سالاھىيىتىم بىلەن ئۆزۈم ياسىۋالغان ياغاچ مىلتىقنى دولامغائىسىپ يەنە قوناقنىڭ شىخدا چىرايلىق ھەم راۋرۇس ئوخشۇتۇپ ياسىۋالغان،يان تاپانچامنى بىلىمغا قىستۇرۇپ گىدىيىپ قاراپ تۇراتتىم،ئالدىمدابولسا قولئاستىمدىكى بالا ئەسكەرلەر تىزىلىپ  مىنىڭ بۇرۇق بىرىشمنى ساقلاپ تۇرماقتا ئىدى بۇلارنىڭ ئىچىدە ئۇچىسىغا ھاۋارەڭ كۆينەك بىشىغا  كاتەكچە گۈللۈك ياغلىقنى چىرايلىق ياراشتۇرۇپ ئارتىۋالغان زورەممۇ،بارئىدى ئۇ بىلىكىگە ئۆزىمىز قىزىل رەڭدە ئاق خەسىگە دوختۇر بەلگىسى سىزىپ ياسىۋالغان يەڭ بەلگىسىنى تاقاپ ياغاچتىن يارالغان ئۇكۇل ۋە باشقا داۋالاش سايمانلىرىنى پوپكىسىغا سىلىپ قاراپ تۇراتتى  ئۇ قىسمىمىزدىكى بىردىن بىر ئايال ئەسكەر بولۇپ ياردارلارنى داۋالاشقا مەسئۇل ئىدى،مەن ئەسكەرلەرنى بىرقۇر كۆزدىن كۆچۈرگەندىن كىيىن قانداق جەڭ قىلىش توغۇرسىدا بۇيرۇق بەرگەچ جەڭ بولىدىغان ئورۇننى بەلگىلەپ بەردىم،بىزنىڭ جەڭ مەيدانىمىز بولسا،مەھەللىمىزدىكى خىش زاۋۇتى بولۇپ بىزنىڭ چۈشمەنلىرمىز بىزبىلەن قارشى مەھەللىدىكى بالىلار ئىدى،بىز 14 بالا ئىككىگە بۆلۈنۈپ بىرتەرەپ دۈشمەن يەنە بىرتەرەپ ھەربى بولۇشاتتۇق،شۇنداق قىلىپ بۇ بىرمەيدان جەڭ باشلىنىش ئالدىدا تۇراتتى،مەن بىلىمدىكى تاپانچمنى چىقاردىمدە ئاسمانغا قارتىم ئىغىزىمدا پاڭ قىلىپلا بىرپاي ئوق ئاتتىم ھەمدە ۋۇررا دىگىنىمچە يۈگۈردۈم شۇنداق قىلىپ بۇ شىددەتلىك جەڭ باشلاندى تاتاتاتات ،چۇ چۇ چۇ،پاڭپاڭپاڭ،بوڭ،دىگەندەك ئاۋازلارنى چىقىرشىپ دۈشمەنلەرگە بەربە بىرۋاتاتتۇق ئاردىن {ئا} دىگەن ئاۋاز بىلەن بىرى يىقىلدى مەن چىئۇ چىئۇ قىلىپ ئىككى پاي ئوق ئىتىۋاتقاندىن كىيىن،دوختۇر،دوختۇر تىز تىز بۇل خەتەرلىك دەپ تولىدىم شىددەت بىلەن ئىتىلىۋاتقان ماياقچە چوڭلۇقتىكى چالما ئوقلار ئارسىدىن دوختۇرقىز زورەم كىلىپ كەلدىدە پوپكىسىنى ئىچىپ ياردارنى داۋالاشقا باشلىدى،مەن ئىككى ئەسكەرنى ياردارنى دەرھال يۆتكەشنى بۇيرىغاندىن كىيىن دولامدىكى ياغاچ پىلىمۇتۇمنى قولۇمغا ئىلىپ چاچچاراپ چىقىپ دۈشمەنلەرگە زەربە بەردىم تاتاتاتاتاتات دىگەن ئاۋاز بىلەن كەينى كەينىدىن ئا    ئا   ئا دىگىنچە ئۈچ دۈشمەن يىقىلدى، قىرىشقاندەك مۇشۇنداق چاغدا قوقۇمنىڭ تۈگەپ كەتكەنلىكىنى بىلگەن دۈشمەنلەردىن بىرى خاخا ئۇنىڭ ئوقى تۈگىدى،ئۇنى تۇتايلى دىگىنچە يۈگۈرۈپ كەلدى مەن بىلىمدىكى ياغاچ پىچىقىمى چىقىرىپ ئۇلاربىلەن ئىلىشتىم،ئاردىن پاڭ قىلغان ئاۋازبىلەن يەرگە  يىقىلدىمدە كەينىمگە قارسام دۈشمەنلەرنىڭ باش قوماندانى ماڭاقاراپ تۇراتتى ئۇ ماڭا قاراپ تەسلىم بول بولمىسا ئىتىۋىتىمەن دىدىدە قولىدىكى سۈپۈرگىنىڭ سىپىدە ياسىۋالغان مىلتىقىنى ماڭا تەڭلەپ تۇرۇپ،مەن ئاتە قىنى مەن تەسلىم بولمايمەن دىيىشىمگىلا ئۇ پاڭ قىلىپلا يەنە بىرپاي ئوق ئاتتى بۇ ئوق مىنىڭ پۇتۇمغا تەگدى،ئاردىن چىئۇ قىلغان ئاۋازبىلەن بۇ باشقۇماندانمۇ يىقىلدىدە ئورنىدىن تۇرالمىدى،بۇ ئوقنى بىزنىڭ سويۈملۇك دوختۇر قىزىمىز زورەم ئاتقان بولۇپ ئۇنىڭ تۇنجى قىتىم قورال ئىتىشى ئىدى، ئۇ يۈگۈرۈپ كىلىپ باشلىق يارلىنىپسىز دىدىدە يىنىدىكى داۋالاش سايمانلىرىنى چىقىرىپ پۇتۇمنى تاڭنى ھەمدە ناننىڭ ئۇۋۇقىدا ياسىۋالغان دورسىدىن ئىككىتالنى ماڭا ئىچۆۈزدى،كەچنىڭ كىرىشى بىلەن بىزنىڭ بۈگۈنكى بۇ جىڭىمىز مۇشۇيەرگىچە بولدى ھەمدە ھەممىمىز توپىلىرمىزنى قىقىشىپ،ئۆيلەرمىزگە قايتىدىغان بولدۇق.
مانا مۇشۇنداق سەبى بالىلىق چاغلىرىم ئاياغلىشىپ بىشىمدا ئەقىل ئىدىراك قۇشلىرى ئەگىپ يۈرگەن كۈنلەرنىڭ بىردە،بىز،بۇ بىزنى ئۆزقوينىدا يىتەكلەپ كىلىۋاتقان مەكتىۋىمىزدىن ئايرىلىپ،ئولۇقسىزغا كۆچىدىغان بولدۇق،بۇيىڭى مەكتەپ مۇھىتى بىزنى ئانچە قىزىىقتۇرالمىغىنى بىلەن يىڭى ئوقۇتقۇچىلارنىڭ ئىللىقلىقى،ۋە ئۇلانىڭ مىھرىبانلىقى ئاستىدابىزمۇ بۇمەكتەپكە كۆنۇپ قالدۇق،مىنىڭ ئاياتتىكى مۇھەببەت كەچۈرمىشلىرىممۇ مۇشۇ مەكتەپتىن باشلاندى،بىزنىڭ سىنىپتا كوناساۋاغ داشلىرىمدىن باشقا يەنە بىرنەچچە يىڭى باللارمۇ بارئىدى،شۇڭا بىز مۇھەللىمگە يالئۇرۇپ يۈرۈپ كونا ساۋاغداشلارنىڭ بىرپارتىدا ئولتۇرشىمىزغا روخسەت قىلىشنى قولغا كەلتۈردۇق،مەن ئوتتۇردىكى پارتىدا ئولتۇردىغان بولۇپ مەن بىلەن بىرگە يەنە ئىككى ساۋاغدىشىم بىللە ئىدۇق،بۇلارنىڭ بىرسى بولسا ئاشۇ مەھەللىمىزنىڭ گۈلى زورەم ئىدى،مەن ئوڭياندا ئۇ بولسا سولياندا ئولتۇردىغان بولۇپ ئوتتۇرمىزدا ئەكبەرجان ئىسىملىك بىر ساۋاغدىشىمىز ئولتۇراتتى،بىزئۈچىمىز ئانچە مۇنچە قارشىپ قويغاچ يەنە ئۆزىمىزنىڭ دەرسى بىلەن بولاتتۇق،مەن قىززىقىپ تاپشۇرۇق يىزىۋاتاتتىم تۇيۇقسىز {ئابدۇسوپۇر}ئۇنىڭ مىنى تۇنجى قىتىم چاقىرشى ئىدى ئۇنىڭ بۇ يىقىملىق سىپايە تەبى ئاۋازى ھىلىھەم ئىسىمدە،مەن بىشىمنى كۆتۈرۈپ قاردىمدە قەلەمنى ئىككى بارمىقىم بىلەن چىڭ قىسىپ تۇرۇپ ئۇماڭا تولىمۇ ئىللىق چىرايبىلەن قاراپ تۇرۇپ
---- ئۆچۈرگۈچىڭىزنى  بىرىپتۇرڭەدىدى،مەن كۈلۈمسىگىنىمچە ئۇنڭغا ئۆچۈرگۈچنى سۇندۇم،ئۇنىڭ يۈزىگەقارغان ۋاختىمدا چىرايىم ئوت بولۇپ يىنىپ كەتكەندەكلا بولۇپ كىتىۋاتاتتىم،چۈكى مەھەللىمىزدىكى ھەرقانداق بىرە قىزماڭاگەپ قىلسا ياكى چاقىرسا خىجىللىقنىڭ دەدىدىن يەرگە كىرىپ كەتكۈدەك بولاتتىم،ئۇئۆچۈرگۈچنى ئىشلىتىپ بولۇپلا ماڭاقايتۇردى
----رەھمەتسىزگە دىدىدە ماڭا ئۆچۈرگۈچنى سۇنغاچ،مەن ئالدىمدىكى دەپتاردىن بىشىمنى كۆتۈرمەيلا بولدى سىز ئىشلىتىڭ مەندە يەنە باردىدىم،ئۇمۇ گەپ قىلمايلا ئوقۇشقا كىرشىپ كەتتى،مانابۇ كۈن مەن ئۈچۈن ئۇنتۇلغىسىز بىركۈن بولدى.
