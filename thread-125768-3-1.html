قانچىلىك ياتتىم بىلمەيمەن بىرچاغدا كۆزۈمنى ئاچسان دوختۇركارۋىتىدا يىتىپتىمەن،مىھرىبان دادامبولسا بىشىمدا قاراپ تۇرۇپتۇ زورەم ۋە بىرنەچچە ساۋاقداشلارمۇ بارئىدى،دادامغا دوختۇرنىڭ دىيىشچە مەندىن قان كۆپ چىقىپكەتكەنلىكى ئۈچۈن ۋۇشۇمدىن كىتىپتىمەن،
-سەسسالامۇ ئەلەيكۇم:بۇ بىزنىڭ ئوقۇتقۇچىمىز پاشاگۈل مۇئەللىم ئىدى،مەن مۇئەللىمنىڭ ئاۋازىنى ئاڭلىغانىدن كىيىن ۋەئەلەيكۇم ئەسسالام دىدىم ھەمدە ئورنۇمدىن تۇرۇشقا تىرىشتىميۇ لىكىن تۇرالمىدىم بىشىم قىيىپ كۆزلىرىم ئىلىشىپ كىتىۋاتاتتى مۇئەللىم قىشىمغا كىلىپ:
-قانداقراق تۇرسەن دىدى ھەم قولىدىكى نەرسىلەرنى بىشىمدا قويۇپ قويغاچ
-خىلى  ياخشى سىلىنى ئەۋارە قىلىپ قويۇپتىمەن دىدىم خىجىل بولۇپ،
-يچقىسى يوق مەن مەكتەپ رەھبەرلىكىنىبىلەن كۆرشۈپ قويدۇم سەن خاتىرجەپ داۋالانغىن،ساڭا نەچچە كۈنلۈك روخسەت قىلىپ قويدۇم شۇڭا دەرىستىن ئەنسىرمەي ياخشى داۋالان جۇمۇقوزام دىدى كۈلۈپ تۇرۇپ،ئادەتتە مۇئەللىم مىنى خىلى چوڭ كۆرەتتى شۇڭا بەزىدە چاقچاق قىلىپ قۇزام مايمۇنۇم دىگەندەك ئەكىلەتمە سۆزلەربىلەن چاقچاق قىلىپ قوياتتى،مەن مۇئەللىمنىڭ گەپلىرنى ئاڭلىغانىدىن كىيىن پۇتۇمنىڭ كىسىلىپ كەتكەن لىكىدىن تولۇمۇ خوش بولدۇم چۈنكى زورەم كۈندە دىگۈدەك كىلىدۇ بۇيەردە بىزنىڭ سۆزىمىزگە ئارلىشىۋالىدىغان ساۋاق داشلارمۇ يوق شۇڭا بىزمۇ ئەركىن پاراڭلىشالايمىز دىگەنلەرنى ئويلاپ ئىچ ئىچىمدىن خوش بولىۋاتاتتىم تۇيۇقسىز،
-بالام دىدى دادام .
خوش دادا دىدمدە دادام تەرەپكە قاراپ.
-مىنىڭ بىرئاز ئىشىم بارئىدى بىرىپ قىلىپ ۋىلىپ كىلەي قىشىڭدا زورەمگۈل قىزىممۇ بولغاندىكىن يالغۇز قالمايسەن:دىدى
مەن بۇلارنى ئاڭلاپ تىخىمۇ خوش بولۇپ كاتتىم چۈنكى دادام چىقىپ كەتسە دىمەك زورەم بىلەن ئىككىمىزلاقالىمىز  خوشاللىق ئۈستىگە  بولىۋاتقان خوش خوشاللىقىمدىن پۇتۇمنىڭ ئاغىرقىنى سەزمەيلا كىتىپتىمەن،تۇيۇقسىز ۋاي......ھەجەپ دىدىمدە گىپىمنىڭ ئاخىرنى يۇتۇۋەتتىم،
-نىمەبولدىڭىز:سوردى زورەم يىچ يىچنىمە بولمۇدۇم شۇ پۇتۇم بىنەرسىگەئۇرلۇپ كەتكەنتى.....
-ھە مۇنداقدەڭ بىشىڭىزنى ئىگىزراق قىلىپ يىتىڭ دىدى،مەن قەستەنگە شۇ مىدىرلىيالمايۋاتىمەن دىدىممەن ئۇنىڭ ياردەم قىلىشىنى قەستەن قولغا كەلتۈرۈش ئۈچۈن،
-ۋاي ئۆلەك ئۇغۇل بالىمۇ سىزدەك جىنى يوق بولامدۇ ھى... ھى... ھى...مەن جىنىمدە بىريىرىم ئاغىرسىمۇ سىزدەك ۋايسىمايمەن لىكىن  بەكلا لەقۋاكەنسىز خى خى خى:ئۇنىڭ مانا مۇشۇ تاتلىق كۈلكىلىرنى ئاڭلىغانسىرى بۇ دوختۇرخانىدىن زادى چىقىپ كەتكۈپ كەلمەي ۋاتاتتى،ماڭاخۇددى بۇيەردىن چىقىپ كەتسەملا ئۇنى قايتا كۆرەلمەيدىغاندەك ئۇنىڭدىن يىراقلاپ كىتىدىغابدەكلا قىلاتتىم،ئۆزەمچە بۇيەردىن چىقىپ كەتمەسلىكنىڭ ئامالىنى ئىزدەيتتىم ئويلا-ئويلائاخرى پۇتۇمنى پات ساقايتماسلىق قارارىغا كەلدىمدە زورەمگە قاراپ دىدىم،ھەي مىنى يۈيۈلەپ ئاچىقىڭە،
نەگە؟نەگە ئاچىقىمەن؟
شۇدىيىشتىن خىجىل بولىدىگەنمەن ئاچىقسىڭىزلا بولاتتى دىدىم ئۇنىڭ ئاڭا ياردەم قىلىشىنى تۆت كۆزۈم بىلەن كۈتۈپ،
-توۋا شۇ دىمەمسىز نەگەئاچىقىمەن ھە بولىڭە دەڭە.
شۇ ئىشۇ يەرگە يىلىقى....
-ئاللا ئاللا ئەدەپە دەڭە ھى ھى بۇرۇنراق ئوچوق دىسىڭىز بولمامدۇ،سىزنىڭ مۇشۇ تارتىنچاقلىقىڭىززە ئابدۇسوپۇر،مۇشۇنداق قىلسىڭىز كىيىن بىرە قىزبالا بىلەن تونۇيشۇپ تويمۇ قىلالمايسىز تازا ۋۇ لەقۋا خى... خى... خى...:دىدىدە ئۇماققىنە سادداكۈلۈپ ھەمدە مىنى يۈلىگەچ،نىمە دىدىڭىزئەسكى سىزتۇرسىڭىز نىمە قىلىمەن باشقا قىزلاربىلەن تونۇيشۇپ،دىيىشىمگىلا ماڭاشۇنداق ئىللىق چىرايلىق بىلەن قاردىكى ئۇنىڭ بۈگۈنكى بۇقاراشلىرنى تىلىم بىلەن تەسۋىرلەپ بىرىشكە تولىمۇ ئاجىز كىلىمەن،ئۇمانا مۇۇشنداق قارىغاچ
-نىمە؟نىمە دىدىڭىزئابدۇسوپۇر
يىچنىمە يىچنىمە دىمىدىم دىدىم كۆزۈمنى باشقا ياققا يۈگۈرتكەچ،ئۇيەنە شۇنداق سىلىق سىپايلىق بىلەن
-ماڭاقارىڭا ئابدۇسوپۇر ماڭاقاراپ دەڭ
مەنشۇ شۇ،ئۇنىڭغىچە بىزئوبۇرنىنىڭ ئالدىغا كىلىپمۇ قالدۇق،مەن ئوبۇرنىغا كىرىپ پۇتۇمدىكى تىنىڭنى يىشىۋىتىپ دورا يۇقىلىرنى پاكىز يۇيۋىتىپ پۇتۇمنى قەستەنگە قانىتىۋالدىم،توۋا ئادەمدىگەن غەلىتىلا بىرنەرسە ئىكەنمىز،بەزى ئىشلار ئۈچۈن جىنىمىزنى شۇنداق ئايايدىگەنمىز،يەنە بەزى ئىشلار ئۈچۈن جىنىمىزنى قىيىنايدىگەنمىز،مەن شۇنداق پۇتۇمدىن قان ئاققۇزۇپ دىڭگوسلاپ چىقىشىمغىلا ئۇ مىنى كۆرۈپ
-ئاللا ئاللا نىمەبولدىڭىز پۇتىڭىز يەنە قاناپتىغۇ تىڭىقنى نىمەدەپ ئىلىۋەتتىڭىز:دىدى مىنىي يۈلەپ ماڭغاچ،ياق مەن ئىلىۋەتمىدىم شۇ كىرىپ بولغۇچە يىشىلىپ كەتتىشۇ دىدىم خۇددى يالغان سۆزلەۋاتقىنىمنى ئۇ بىلىپ قالغاندەك بىرخىل خىقىلچىلىق ھىس قىلىپ ئۇ ماڭابىر قارىۋەتكەندىنكىيىن
-ھە ھەجەپ تىڭىقكەن ھە ئۇ بىكالايىشىلىپ كىتىدىغان ھى...ھى...قارسام تىڭىق سىگە ئىچى ئاغىرىپ ئۆزى يىشىلىپ كەتكەندەك قىلىدۇ:دىدىدە مەنىلىك قىلىپ،مەمۇ ئارتۇق گەپقىل مىدىم
