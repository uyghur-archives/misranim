ئەسسالامۇ -ئەلەيكۇم تورداشلار ،بۇ ئەسلىدە بىر سىمۋوللۇق تىما ئىكەن ،بۇ تىمىنى يوللىغاندىن كىيىن بەزى تورداشلار بۇ تىمىنىڭ ئەسلى ئىپادىلىمەكچى بولغان مەنىسىنى ئاساسى جەھەتتىن چۈشۈنۈپ بوپتۇ ئەمما بەزى تورداشلار  يۈزەكى مەنىسىنى چۈشۈنۈپ ئەخلەت تىما قاراۋاتىدۇ ،مەن بۇ تىما ھەققىدە چۈشۈنۈش ئىقتىدارىمنىڭ يىتىشچە ئۆزۈمنىڭ بايانىمنى ئوتتۇرغا قويۇپ باقاي .
بۇ ئەترەت ئەسلى جەمىتىمىزدىكى بىر شركەتكە ۋەكىللىك قىلىدىكەن ،تاڭسىڭ بولسا شۇ شىركەتنىڭ باش دىرىكتۇرلۇق ۋەزىپىسىنى ئۆتەيدۇ ،ئەلۋەتتە مۇشۇنىڭدىن بىلىشكە بولىدۇكى تاڭسىڭ ئىشتىن بوشىتىلمايدۇ دىگەن گەپ ،ئەگەر تاڭسىڭ ئىشتىن بوشىتىلسا بۇ شىركەت تەرەققى قىلماقچى بولغان يۆنىلىشىنى بىلەلمەي يولدىن ئىزىپ كىتىدۇ ،گەپنىڭ ئوچۇڭراقىنى قىلىپ ئىيتقاندا بۇ شىركەت ۋەيران بولىدۇ دىگەن .
ئەمدى سۇن ۋۇكۇڭغا كەلسەك ئۇ تاڭسىڭنىڭ ئەڭ چوڭ شاگىرتى ،سۇن ۋۇكۇڭنىڭ غەرپكى ساياھەت فىلىمىدىكى ئوبرازى دەل شىركەتتىكى بىلىملىك  ،ئىقتىدارلىق تىخنىك شۇنداقلا شىركەتنىڭ باش مەسلىھەتچى ئوبرازىغا ۋەكىللىك قىلىدۇ ،نۇرغۇن ئىشلاردا سۇن ۋۇكۇڭنىڭ يول كۆرسىتىشى ،كارامەتلىرىنى ئىشقا سىلىشى بىلەن بۇ ئەترەت داۋاملىق ئالغا ئىلگىرلەۋاتىدۇ .بۇ دەل شىركەتنىڭ باش مەسلىھەتچىسى يۇقىرى ئىقتىدارلىق ،ساپالىق بولسا بۇ شىركەتنىڭ ھەر كەسكىن رىقابەتلەردىن قورقۇپ قالماي تىخمۇ ئىزدىنىپ تىرىشىپ باشقا شىركەتلەرگە جەڭ ئىلان قىلالىغانلىقىغا ئوخشايدۇ ،ھەر قانداق بىر شىركەتتە شۇ سۇن  ۋۇكۇڭغا ئوخشاش ئىقتىدارلىق باش مەسىلەتچى كەم بولسا بولمايدۇ ،شۇڭا سۇن ۋۇكۇڭمۇ ئىشتىن بوشىتىلمايدۇ .
ئەمدى جۇباجىغا كەلسەك ئۇنىڭ خاراكتى بەكمۇ ئوچوق يورۇق ،ھەر خىل تەنقىدلارغا ئەتراپلىق مۇئامىلە قىلالايدۇ ،ئىغىر بىسىملارغا بەرداشلىق بىرەلەيدۇ ،ئۇ باشقىلاردىن پەرقلىنىدىغان ئالاھىدىلىكى شۇكى ئۇ باشقىلار بىلەن ئالاقە قىلىشقا بەكمۇ ماھىر ،تۇرقى سەت بولغان بىلەن ئاسانلا باشقىلارنىڭ دىققەت ئىتىبارىنى تارتالايدۇ ،باشقىلارنى ئۆزىگە رام قىلالايدۇ ،مۇشۇ ئالاھىدىلىك جەھەتتە باشقىلاردىن ئالاھىدە پەرقىلىنىپ تۇرىدۇ .ئەلۋەتتە ھەر قانداق بىر شىركەتتە مۇشۇ جۇباجىنىڭ ئوبرازىغا ئوخشاش بىر ئەزا چوقۇم لازىم ،مۇشۇنداق خاركتىردىكى ئەزا ھەر ۋاقىت شوخ ئوچوق يورۇقلىقى بىلەن باشقا ئىشچىلارنىڭ كەيپىياتىنى تەڭشىيەلەيدۇ ،خىزمەتكە بولغان جۇشقۇنلىقىنى ئاشۇرالايدۇ ،باشقىلارغا يىڭى بىر خىزمەت مۇھىتى يارىتىپ بىرەلەيدۇ دىگەن گەپ .
ئەمدى شا راھىپقا كەلسەك ئۇ تولىمۇ جاپاكەش ،يول بويى شۇ ئىغىر ئەپكەشنى كۆتۈرىدۇ ،ئەمما ھىچبىر ئۇنىڭ مەن چارچىدىم ،مەن ھاردىم ئەمدى كۆتۈرمەيمەن ،دەپ ئىغىر يۈكنى باشقىلارغا تاشلاپ قويغىننى كۆرۈپ باقمىدۇق ،ئۇ شۇنچە ئىغىر ئەپكەشنى كۆتۈرۈپمۇ مەيلى قانچىلىك ھىرىپ چارچىشىدىن قەتئىنەزەر بەل قويۋەتكىنى يوق ،ئۇ ئۆز خىزمىتىگە بەكمۇ مەسئۇليەتچانلىق بىلەن مۇئامىلە قىلىدۇ .ئەلۋەتتە بىر شىركەتتىمۇ مۇشۇ شا راھپقا ئوخشاش ئاقكۆڭۈل ،ئىرادىسى كۈچلۈك ،مەسئۇليەتچان   كىشلەرگە تولىمۇ ئىھتىياجلىق ،مۇشۇ ئوبرازدىكى كىشلەرلا ئاندىن شىركەتنىڭ ئىغىر يۈكلىرىنى كۆتۈرۈپ شۇ شىركەتنىڭ تەرەققىياتى ئۈچۈن تىخمۇ كۆپ كۈچ چىقىرالايدۇ .
ئەمدى ئاق ئەجدەرھا ئاتقا كەلسەك ئۇ خوجايىننىڭ ئەڭ يىقىن ياردەمچىسى ،ئۇ ھەر ۋاقىت خوجاينىغا ئەڭ يىقىن تۇرۇپ خوجايىننىڭ ئىغىر يۈكىنى كۆتىرىدۇ ،ھالقىلىق پەيىتتە خوجايىننى قۇتقۇزۇپ قالالايدۇ .شۇڭلاشقا شىركەتتىمۇ خوجايىنىغا يىقىن تۇرىدىغان مۇشۇ ئاق ئەجدەرھا ئات ئوبرازىدىكى كىشلەر كەم بولسا بولمايدۇ،شۇڭلاشقا بۇ ئاق ئەجدەرھا ئاتمۇ ئىشتىن بوشىتلمايدۇ .
تاڭ تەيزۇڭ ئاخىردا  ئەزالارنىڭ مۇشۇ  بىر -بىرىگە ئوخشىمايدىغان ،باشقىلاردىن پەرقلىنىپ تۇرىدىغان ئالاھىدىلىكىنى كۆزدە تۇتۇپ قارارىدىن يىنىپ ئەترەتتىكى ھىچكىمنى  ئىشتىن بوشاتماپتۇ .



ئىزاھلاش :ئەگەر سىز بىر ئورۇندا پۇت دەسسەپ تۇرالايمەن ،باشقىلاردىن كۆپرەك قوللاشقا ،تەرەققى قىلىش پۇرسىتىگە ئىرىشىمەن دەيدىكەنسىز مۇشۇ ئەترەتتىكى ئەزالارغا ئوخشاش باشقىلاردىن پەرقىلىنىپ تۇرىدىغان ئالاھىدىلىكىڭىز بولىشى كىرەك ،باشقىلار سىزنىڭ ئورنىڭىزنى باسالمايدىغان ھالەتكە يەتكىنىڭىزدە ئاندىن  مەيلى سىز قەيەردە خىزمەت قىلماڭ يۇقىرنىڭ قوللىشىغا  ئىرىشەلەيسىز ،تىخمۇ زور تەرەققى قىلىش پۇرسىتىگە ئىرىشەلەيسىز ،ھەرگىزمۇ ئىشتىن بوشىتىلىش خەۋپىگە دۇچ كەلمەيسىز .جەمىئەتتە تىخمۇ چىڭراق پۇت دەسسەپ تۇرالايسىز .

مېنىڭىچە  بۇ كوللكتىپتىن مەلۇم بىر ئەزانى ئىشتىن قالدۇرسا بۇ كوللكىتىپ قالايمىقان بولۇپ كىتىدۇ.
1.تاڭسىڭىنى بوشاتسا بۇ ۋەزىپە ئورۇنلانمايدۇ.
2.ئالۋاكسورانى بوشاتسا جىددى ۋاقىتا تەدىبر بەلگلەيدىغان ئادەم يوق بولۇپ قالىدۇ. ھەمدە بۇ كوللكىتىپنىڭ ئارقا تىرىكى.
3.سۈن ۋۇكۇڭغا كەلسەك ئۇنى بوشاتسا ئەرشىنى يەنە قالايمىقان قىلىدۇ. ئۇ بولمىسا بۇ كوللكىتىپ دۈشمەنلىرىنىڭ ھۇجۇمغا تاقابىل تۇرالمايدۇ. شۇڭا ئۇنى بوشاتسا بولمايدۇ.
سۈن ۋۇكۇڭ يەنە ئەرشىتىڭى ئەمەلدار ئارىسىدا يۇرۇپ ئۆزلىرىنىڭ ئىشلىرى ئۈچۈن يول ئاچمىسا بولمايدۇ. ئۇنىڭ رولىمۇ مۇھىم.
