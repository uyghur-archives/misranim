                                    (رەڭلىك دۇنيا)

مۇشۇ ئىنسانلار  تەبىئەتتىكى ھايۋاناتلار بىلەن بىۋاستە پاراڭلىشىلالىغان بولسا،ئىنسانلار ئۇلارنىڭ تىلنى تولۇق چۈشەنگەن بولسا،جىق ئىشلار ئاسان بولۇپ كىتەتتى…… مەسىلەن:

1.گۈزەل يۇرت غۇلجىنىڭ  ئىشچان،جاپاكەش يېزا ئاياللىرى سەھەردە كالا ساغقىلى چىقتى،
سىيىر：          -ھۇي ئايشىمىدە،يەنە كەلدىڭىزمۇ،خويمارا،مۇشۇنداق ئەتتىگەندە يوقلاپ كەلسىڭىز خويمۇ ئوبدان ھە، ۋۇييەي ئاستا،ئاغرىتىۋەتتىڭىز，مەن دەيمەن ئايشىمدە، مۇشۇ سۈتنى سېغىپ ئېلىپ،ئاندىن يەنە قاينىتىپ،ئاندىن سىنچاي دەملەپ،ئۆيىڭىزدىكىلەر ئەتكەن چاي قىلىپ ئاۋارە بولىدىكەنسىز،ئۇنىڭدىن كۆرە ھە،ماڭا ماۋۇ كىچىدە تۇزلۇقراق قىلىپ قارا چاي دەملەپ ئىچكۈزىۋېتېپ ، ئەتىگەندە ئەتكەن چاي سېغىۋالسىڭىز بولمامدۇ ھۇ ھۇ ھۇ ھۇ، ۋۇي ئاستا،،،    توشتىمۇ چىلەك؟

2-.ئەزىزانە قەشقەرنىڭ يېزىسىدا……سەھەر بولغاندا          خوراز چىللىدى：          -قوپاڭلا！！！قوپاڭلا خانلار،
قوپاڭلا جانلا،تاڭ ئاتتىغۇ……غوجام بەك تۇراممايمىزغۇ……يۆلەپ قويايما؟！ماڭ قايلاڭلا～～ئاخشام ياتمايسىلە لىڭشىپ،سەدە قوپمايسىلە غىڭشىپ ……شۇنداما～～بولاڭلا ،چىيىڭلىنى ئىچىڭلا……مەن بېرىپ پۈۋلەپ بېرەيمۇيا؟

3-قەدىمىي يۇرت كۇچادا،بىر دېھقان ئىشەك ھارۋىسىنى ھەيدەپ،تۇققىنىنىڭ ئۆيىدىن چىقتى……          يولدا ئىشەك سۆز باشلاپ قالدى：          -ماڭ قارىسىلا،بۇ تۇققانلىرىدا باغۇ……زادى مېھىر يوق جۇمۇسىلا……پۇل بېرىپ تۇ دەپ شۇنچە يەلۋەسىلە قاراپمۇ قويمىدى جۇمۇسىلا……          دېھقان ئىشەكنى قامچىلىدى          -ئۇرسىلا،كۈچلىرى ماڭىلا يەتكەندىكىن～ھە ،مەن يامان گەپ قىممىدىم～،سىلىگە ، ياخشى بوسۇن دىدىم～،بوممىسا سىلى بىي مىنىڭ تۇققىنىممۇ ئەمەس،،،،،

4-ئۈزۈم ماكانى تۇرپاننىڭ ھويلىلىرىدا بىر كۈنلۈك تۇرۇش باشلاندى……          ھويلىدا بېقىلىۋاتقان ئۆچكە پاراڭ باشلىدى：          -ئۈكەەەەەەم～ئايناقمۇ چۈپەيلىشىپ كەتمەڭلارا، يەم خەشەكنى قورقماي سەپ باقمامسەن ھە،ئۈزەڭ يەلەمىسەڭ يە،ھەي پوخۋاشلار،گەپ قىۋاتىمەن سەنلەرگە،ئايلا ،سەن بولساڭمۇ قاراپ قويسىنا ماڭا،،،،،،،

5-قوغۇن ماكانى قۇمۇلنىڭ يېزىلىرىدا بىر ئادەم قويلىرىنى قوتانغا ھەيدەپ كىردى،قوتاننىڭ كىرىش تەرىپىدە قويلار سۆز باشلىدى：          -قېنى،ئۆزلەدىن كەلسۇن          -قېنى ئۆزلەدىن كەلسۇن          -ئاۋۇ ئىككىڭلار،مېڭىپ باقسىلاكى……          -ئارقامدىكى لەر،سۈزمىسەڭلا بولۇرمۇ مىنى……          -ماۋۇ پادىچى ئەجەپ ئۇرەلەۋەردىغۇ چىۋىقنى،ئىلدام ئىلدام ماڭغايسىڭىزلەر......

6-خوتەن شەھەر ئىچىدە، ناھايىتى شەپشەك ئاتتىن بىرى باغلاقتا تۇرماي قىچىپ كەتتى؛ بۇنىڭغا يىتىشەلمىگەن ساقچى يولداشلار ئاپشاركىنى قويۋەتتى،،ئاپشاركا قوغلاۋىتىپ،،ئاتقا： ھاي غوجىما،  ھۆددەس چاپىلىكەنيا ئەينا？ئۆزلىمۇغو  شۇنداق ئۇز، بوي بەستلىمۇ ھۆددەس قاملاشقان يىگىتكەنلا، توختاپ تۇسا سويايتقان گەپ بەيدى سىلىدىن،،،، ئان ماڭسىلىمۇ بوتتو،،،،،
ئات چىپىۋىتىم بۇ ئىتقا مەنسىتمىگەندەك قاراپ قويۇپ  مۇنۇ ناخشىنى باشلىدى،،،

ساماۋەيىم بولسىدى،
چايلا قايناپ تۇسىدى،
بىى چىنە چاينى ئىچكۈچە،
ساقچى قالاپ تۇسىدى،

ساماۋەينىڭ سازى مەن،
چۆلدە ئۇچقان غازى مەن،
ئىتلا بىلەن جەڭ قىلىپ،
ئۆلۈپكەتسەم  يازى مەن،،،،

بۇنى ئاڭلىغان ئاپشاركا مۇشائىردە يىڭىلىپ قاتتىق نۇمۇستا قىلىپ ، جان جەھلى بىلەن ئاتنى قوغلاپ كەتتىلەر،،
ھازىرغىچە تۇتامماغلىق،،،

ئارقا سەپتە تەشۋىقات خىزمىتىنى قىلىۋاتقان فىرما خوجايىنى مۇنۇ ئىلاننى تاملارغا چاپلىدىلەر،
ھۆيمەتلىك ئەھلى سۇنئى ئەل جامائەت؛
بۆگۈن ناھايىتى خەتەيلىك بى تىيوچى ئات   باغلاقتا تۇماي قەچكەتتى، تۆت پۇتىنىڭ ئاستىدا يۇشۇرۇن تىغلىق ئەسۋاپ با،  ناھايىتى قوقۇنۇشلۇق،بۇ ئاتنى تۇتۇپ بەگەنلە گە،،، ئوتقاشتەك جەينىمازدىن بى پاچە، ئىككى مىتى يەخ، يەنە تېخى ئىككى مىتى يەلىك  سوغات بىيتمىز،،،،،

ئۈرۈمچىدە،  چ چ دا قىزكەلتۈرۈپ ئويناۋاتقان  70～80 موللەكلەر بۇخەۋەرنى كۆرۈپ-
-تۆھپىنى باشقىلاردىن قىزغىنىپ،  ھاپىلا شاپىلا سەللىلىرىنى يۆگىشىپ تونلىرىنى كىيىشىپ مەدرىسكە كىرىپ كەلدىلەر،،،
كامىراغا تازىم بەجا كەلتۈرىۋەتكەندىن كىيىن ،سۆزىنى مۇنداق باشلىدىلەر،،،،،،،،،

نەئۇزى بىللەھى مىنەش،،،،، ئاپ،،،
ئەئۇزۇ بىللەھى كىڭەش،،، ئاپ،، ،،
تەرەت ئەپ كىشەي ئاخۇنۇم، سىلىدىن كەسۇن چا～
ok
ئەئۇزۇ بىللەھى مىنەششەيتانىر رەجىم،   بىسمىللەھىر رەھمانىر رەھىم،
ئەلىف لا～م !    مى～م،      ئاتلا نىڭ ھەممىسى جى～م،،،،،،،،،،،

ھە ،، ھۆرمەتلىك جامائەت يىقىندىن بىرى شەھىرىمىزدە،،،  ئىتلانىڭ چىچىشى ،  ئاتلا نىڭ قېچىشى ئاۋۇپقالدى؛ شۇ مۇناسىۋەت بىلەن بۇيەرگە ھەممەيلەن جەم بولۇپ ئۇلارغا نارازىلىقىمىزنى ئىپادىلەپ ئولتۇرۇپتىمىز،
يۇقارقى ئايەتنىڭ روھىغا ئاساسەن  ئىتلا نىڭ چىچىشى نۇرمال ئەھۋال بولسىمۇ، ئەمما؛ ئاتلا  نىڭ باغلاقتىن قېچىشى شەرئەن دۇرۇس بولمايدۇ !  بىكالىق پاخال يەپ  شۈكرى قىلىشنى بىلمىگەن، ئىشەك چىلىكمۇ ئەقلى يوق  ئاتلا، ئۆزى خالىغانچە چاپراقشىپ شەھرىمىزنىڭ مۇھىتىغا ئىغىر دەرىجىدە بۇزغۇنچىلىق قىلدى، شۇڭا،،،،،“”
،،،(يىنىدىكى بىرەيلەن پەس ئاۋازدا)،،، -ئاخۇنۇم تىزرەك بوسا،  تىلپۇنلىرى گۈەنجىكەن ،ھىلىقى چوكان ماڭا توختىماي شىنشى يوللاپكەتتى،،دوستىمۇ باكەن تېخى،،،،،،،،،،،
شوڭا،،، شۇڭا،،،،ئاللاھۇ  ئەپكەر .

تىلى قىزىقارلىق،يۇمۇرلۇق چىقىپتۇ،ھەر  قايسى يۇتلارنىڭ يەرلىك شېۋىلىرىنى جايىدا كەلتۈرۈپسىز....


بىراقزە  ‹‹    ئۈرۈمچىدە،  چ چ دا قىزكەلتۈرۈپ ئويناۋاتقان  70～80 موللەكلەر بۇخەۋەرنى كۆرۈپ-
-تۆھپىنى باشقىلاردىن قىزغىنىپ،  ھاپىلا شاپىلا سەللىلىرىنى يۆگىشىپ تونلىرىنى كىيىشىپ مەدرىسكە كىرىپ كەلدىلەر،،،
كامىراغا تازىم بەجا كەلتۈرىۋەتكەندىن كىيىن ،سۆزىنى مۇنداق باشلىدىلەر،،،،،،،،،

نەئۇزى بىللەھى مىنەش،،،،، ئاپ،،،
ئەئۇزۇ بىللەھى كىڭەش،،، ئاپ،،
تەرەت ئەپ كىشەي ئاخۇنۇم، سىلىدىن كەسۇن چا～

ئەئۇزۇ بىللەھى مىنەششەيتانىر رەجىم،   بىسمىللەھىر رەھمانىر رەھىم،
ئەلىف لا～م !    مى～م،      ئاتلا نىڭ ھەممىسى جى～م،،،،،،،،،،،

ھە ،، ھۆرمەتلىك جامائەت يىقىندىن بىرى شەھىرىمىزدە،،،  ئىتلانىڭ چىچىشى ،  ئاتلا نىڭ قېچىشى ئاۋۇپقالدى؛ شۇ مۇناسىۋەت بىلەن بۇيەرگە ھەممەيلەن جەم بولۇپ ئۇلارغا نارازىلىقىمىزنى ئىپادىلەپ ئولتۇرۇپتىمىز،
يۇقارقى ئايەتنىڭ روھىغا ئاساسەن  ئىتلا نىڭ چىچىشى نۇرمال ئەھۋال بولسىمۇ، ئەمما؛ ئاتلا  نىڭ باغلاقتىن قېچىشى شەرئەن دۇرۇس بولمايدۇ !  بىكالىق پاخال يەپ  شۈكرى قىلىشنى بىلمىگەن، ئىشەك چىلىكمۇ ئەقلى يوق  ئاتلا، ئۆزى خالىغانچە چاپراقشىپ شەھرىمىزنىڭ مۇھىتىغا ئىغىر دەرىجىدە بۇزغۇنچىلىق قىلدى، شۇڭا،،،،،
،،،(يىنىدىكى بىرەيلەن پەس ئاۋازدا)،،، -ئاخۇنۇم تىزرەك بوسا،  تىلپۇنلىرى گۈەنجىكەن ،ھىلىقى چوكان ماڭا توختىماي شىنشى يوللاپكەتتى،،دوستىمۇ باكەن تېخى،،،،،،،،،،،
شوڭا،،، شۇڭا،،،،ئاللاھۇ  ئەپكەر .››
