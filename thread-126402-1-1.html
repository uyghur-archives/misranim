                                               ۋىسالسز سۆيگۈ

                                       بىر مىنىڭلا ئىدىڭغۇ گۈلۈم
                                       زىبا ئۆسكەن سۆيگۈ بېغىمدا
                                       خوشلۇقۇمدا يايرايتتى دىلىم
                                      گۈلۈم شىرىن كۈلگەن چېغىڭدا

                                     بولساممۇ گەر ئاشىق دىۋانە
                                     شۇنچە شىرىن ئىدى كۈنلرىم
                                     تاشلدېڭ يار قالدىم يىگانە
                                     ئۇيقۇسىزدۇر ھەر بىر تۈنلرىم

                                    تۇيماي قالدىم كەلگەندە ئەجەل
                                    يۈرۈكۈمگە سانجلىدى بىر ئوق
                                    لەيلى-مەجنۇن ئىدۇق ئەسلىدە
                                    لەيلىگۈلۈم ئەمدى مەڭگۈ يوق

                                   چىن سۆيگۈممۇ قالدى ۋىسالسز
                                   سۇئاللىرىم ھېچبىر  جاۋاپسىز
                                   بىلمەپتىمەن ياركەن ۋاپاسىز
                                  بۇ قىسمەتكە  قالدىم ئامالسىز

سۈيەلمىدىڭ ؛ ئەكرەم

                                             سۈيەلمىدىڭ

                                بىر سىنىڭ دەپ ۋەدىلەر بىرىپ
                               ئۈتىمەن دۇنيادىن سىنىلا سۈيۈپ
                              قېنى شۇ ۋەدىلەر سأيەلمىدىڭغۇ؟
                           سۆيدىڭغۇ ئۆزگىنى مەندىن يۈز ئۆرۈپ

                           سېغىنىپ مىنىڭدەپ ھېچ تۈنەلمىدىڭ
                           بەلكى بۇ ئازاپقا سەن كۈنەلمىدىڭ
                          تەلمۇرگەن شۇنچىلىك چىن ئەقىدەڭگە
                         ۋەلىكىن كۈتكەننى  سەن بىرەلمىدىڭ

                         مەڭگۈلۈك سۆيگۈگە بەردىڭ خاتىمە
                         قالدۇرۇپ قەلبىمدە ئاچچىق ئەسلىمە
                        خارلاندى چىن سۆيگۈم سىنىڭ ئالدېڭدا
                         ئىزىلدى يۈرۈكۈم بولدى مېڭ پارە

                        خەيىر-خوش ۋاپاسز ئەمدى كىتىمەن
                        ئازاپقا ئۆلمەيمەن كۈلۈپ ئۈتىمەن
                       سۆيسەممۇ شۇنچىلىك سەن سأيەلمىدىڭ
                       ئەمدى مەن ئۆزگىدىن بەخىت كۈتىمەن
