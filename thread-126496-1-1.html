                                             تۇنجى سۆيگۈنۈم

                                   مۇھەببەت ئۇتىدا تۇنجى كۆيدۈرگەن،
                                   سەۋدا قىلىپ ئۇتىدا مىنى سۆيدۈرگەن،
                                   مۇھەببەت تەمىنى ماڭا بىلدۈرگەن،
                                   قىزىل گۈلۈم تۇنجى سۆيگۈنۈم.


                                 پاك-سەبى قەلبىمگە ئۇتىنى سالغان،
                                 ئاشىق قىپ ئۇزىگە كۆڭلۈمنى ئالغان،
                                 مۇھەببەت يۇلىدا يولدىشىم بولغان،
                                 تىلتۇمارىم تۇنجى سۆيگۈنۈم.


                               قايغۇلۇق كۈنلەردە تەسەللى بولغان،
                               شاتلانسام مەن بىلەن شاتلىققا تولغان،
                              قەلبىمدە سىز گۇيا كۈچلۈك بېر ۋولقان،
                              گۈل-زىبايىم تۇنجى سۆيگۈنۈم.


                             دۇنيادا بارلىقىم بېر سىز بىلسىڭىز،
                             كۈنىلرىم شاتلانماس ھەرگىزمۇ سىز سىز،
                             ھاياتىم بىتىدە سىز شۇنچە ئەزىز،
                             قەدىردانىم تۇنجى سۆيگۈنۈم.


                            شاتلىنىڭ بېر ئۆمۈر ئەلەم يەتمىسۇن،
                            غۇبارسىز قەلبىڭىز دەرتتە قالمىسۇن،
                            تىلەيمەن ئاللاھىم سىزنى ساقلىسۇن،
                           يۈرەك باغرىم تۇنجى سۆيگۈنۈم.
