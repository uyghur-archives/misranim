مەسلىگە ئىلمى مۇئامىلە قىلساق قانداق دوسىتلار ، بۇ تىما ھەقىقەتەن ئەمەلى ئەھۋالىنى چىقىش قىلىپ يىزىلىپتۇ ، جەنۇبى شىنجاڭ جۈملىدىن قەشقەرنىڭ مەلۈم بىر يىزىلىرىنى تىپ قىلىپ كۆرسىتىپتۇ ، جەبۇنى شىنجاڭنىڭ باشقا يىزىلىرىدا يۇقارقىدەك ئەھۋال يوق دەپ كىم دىيەلەيدۇ (ئەجەبا بۇ كىشى يىزا رەھبىرى كادىرى بولمىسىلا) ؟؟؟؟ توغرا مەسىلە ھەقىقەتەن تۈزۈلمىدە ، يۇقۇرىنىڭ سىياسىتى ياخشى ، تۆۋەندە سەۋەب نىمىكىن ئىشقىلىپ قالايمىقان بولۇپ كىتىدۇ ، ھەممە كىشى كىچە كۈندۈز توختىماي تىرشىۋاتقاندەك ، نۇرغۇن بەدەل تۆلەۋاتقاندەك لىكىن يەنىلا ئاتام ئىيىتقان بايقى !  كۈندىن كۈنگە ياخشى بولۇپ كىتەر ، يۇقىرى دەرجىلىك رەھبەرلەرنىڭ ئىچىدىمۇ ھەققانى ئىشلەيدىغان ، سەمىمى ، ئەستايىدىل رەھبەرلەر چىقىپ قالار !!

راسىتىمنى ئېيىتسام مەنمۇ بىر ئاساسى قاتلام يېزا كادىرى. بەزىدە ئويلاپ قالىمەن نىمىشقىمۇ مەمۇرلۇق ئىمتىھانى بەرگەن بولغىيىدىم دەپ . مەن ئىمتىھان بەرگەندە مەمۇرلۇقنى چوڭ بىلىپ كىتىپمۇ ئەمەس ۋەياكى باشلىق بۇلارمەنمىكىن دەپمۇ ئەمەس پەقەت بىر كىشىلىك خىزمەت تېپىش ئۈچۈنلا ئىمتىھانغا قاتناشقاندىم. چۈنكى خىزمەت تېپىش ھەرقانداق يېڭى ئوقۇش پۈتتۈرگەن ئالى مەكتەپ ئوقۇغۇچىلىرىنىڭ ئەڭ جىددى ئەڭ شىرىن ئارزۇسى. ئىمتىھاندىن ئۆتۈپ شۇنچىلىك خۇشال بولغانتىم ،، ئەپسۇس مىڭ ئەپسۇس.............  ھەرقانچە شۈكرى قىلاي،ئۈمىدۋار بولاي دىگەن بىلەنمۇ شارائىت ئادەمگە تەسىر قىلىدىكەن، كۈنلەپ ئىشخانىدا ئولتۇرساق بولىدۇكى رۇخىسەت سوراپ ئالمىقىم بەك تەس،بىر كۈنمۇ ئارام يوق(بولۇپمۇ يېقىنقى مەزگىلدىن بۇيان). ئادەم دىگەننىڭ ئانچە-مۇنچە شەخسىي ئىشلىرى بولمامدۇ،دەم ئالغۇسى كەلمەمدۇ دەيمەن.....
ھازىر بىر قىسىم كىشىلەرنىڭ نەزىرىدە يېزا ئاساسى قاتلام كادىرلىرىنىڭ ئوبرازى تازا ياخشى ئەمەس،ئۇلارنى كۈن ئۆتكۈزىدۇ،دېھقانلارنى چۈشەنمەيدۇ،باشلىققا خۇشامەت قىلىدۇ دەيدۇ.ھەتتا مەنمۇ بۇرۇن شۇنداق ئويلايىتتىم،ئەمما ئۆزۈم قول تىقىپ ئىشلەپ بىلدىمكى،ئىشەنگەنلەر ئۈچۈن %90تىن ئارتۇق يېزا كادىرلىرى ھەقىقەتنمۇ دېھقانلارنىڭ مەنپەئىتىنى ئويلايدۇ.ھەتتا بەزىبىر ئېتىبار سىياسەتلەردىن دېھقانلارنى خەۋەرسىز قالمىسۇن ۋە ياكى كەنىت كادىرلىرى يەتكۈزمەي قالمىسۇن دەپ ئائىلىلەرگىچە بېرىپ تەشۋىق قىلىدۇ........
ھەقىقىي ئىشلەيدۇ.
كۈنلەر ئۆتۈۋاتىدۇ كۈتۈۋاتىمەن، ئۆتكەن كۈنلەر كۈتكەندەك ئەمەس، كۈتكەن كۈنلەر كەلگەندەك ئەمەس.......

خىزمەتنىڭ ھەممىسى جاپالىق ، ئەمما شۇ ئىش كۈتۈپ تۇرغان  ...

توغرا، خىزمەتنىڭ ھەممىسى جاپالىق. بىراق كۆڭلىڭىز ئازادە بولسا، قىلغان خىزمەتلىرىڭىزنى ئەتراپىڭىزدىكىلەر ئازراق بولسىمۇ مۇئەييەشتۈرسە، ياخشى ئىشلىگەن بىلەن ياخشى ئىشىمىگەننىڭ ئازراق بولسىمۇ پەرقى بولسا، خىزمىتىڭىز ھەرقانچە جاپالىق بولسىمۇ سۈيۈنۈپ تۇرۇپ ئىشلەيسىز، ئەكسىچە، ياخشى ئىشلىگەنلەر ئاكتىپ دەپ كەمسىتىلسە، يىغىنلارغا دەل ۋاقتىدا ئۈلگۈرۈپ كەلسىڭىز، ۋاقتىدا كەلمىگەنلەر سىزنى باشلىققا ياخشىچاق بولماقچى دەپ  دىمىقىنى قېقىپ، كەينىڭىزدىن گەپ توشۇپ يۈرسە ھەرقانچە ئاددى خىزمەتتىنمۇ رايىڭىز قايتقاندەك بولۇپ قالىدۇ.
ئاشۇ ئىش ئىزدەۋاتقان قېرنداشلىرمىز ئەلۋەتتە جەمىئەتتە بىر كىشىلىك ئورنىنى تېپىش ئۈچۈن ئىزدىنىشنى خالايدۇ، ھەرقانچە جاپالىق بولسىمۇ ئاشۇ تۆمۈر تاۋاققا ئېرىشىپ بولغاندىن كېيىن بېشىغا نېمە كەلسە، شۇنى كۆرمەكچى بولىدۇ، بىراق ئىنەكنىڭ بېشىغا كەلگەن ئىش موزاينىڭمۇ بېشىغا كەلگەندەك، ئاشۇ مىڭ تەسلىكتە تۆمۈر تاۋاققا ئېرىشىپ، ھەر قانداق جاپالىق مۇھىتقا كۆنۈشكە‹رازى›بولغان قېرىندىشىمىزمۇ ئادالەتسىزلىككە ئۇچرىسا بىزگە ئوخشاش دەردىنى ئىچىگە پاتۇرالمايدىغانلارنىڭ قوشۇنىغا تەبىئىيلا ئەزا بولۇپ قالىدۇ. شۇڭا تېمىمىزدا ئوتتۇرىغا قويغاندەك  يېزىلىرىمىزدا ئوتتۇرا قاتلام رەھبىرى كادىرلار قوشۇنىنى سەرخىلاشتۇرۇشقا بەكمۇ ئىھتىياجىمىز بار
ھەربىر كۈنىنى خۇشال-خۇرام، مەنىلىك ئۆتكۈزەلىگەنلەر مۇۋەپپىقىيەت قازانغۇچىلاردۇر!
