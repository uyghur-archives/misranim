دوختۇرخانىمىزنىڭ كەسپىي تەرەققىيات جەريانىدىكى ئىختىساسلىق خادىملارغابولغان ئېھتىياجىنى قاندۇرۇش، كەڭ ئىش كۈتۈپ تۇرغان ئىختىساسلىق خادىملارنى ئۆزقابىلىيىتىنى نامايەن قىلىدىغان كەڭ سەھنە بىلەن تەمىنلەش، ھۆكۈمەتنىڭ ئىشقائورۇنلاشتۇرۇش بېسىمىنى يېنىكلىتىش مەقسىتىدە دوختۇرخانىمىز پۈتۈن شىنجاڭ تەۋەسىدىن كەڭ كۆلەمدە تېببىي خادىم قوبۇل قىلىشنى قارارقىلدۇق.


بىزنى تاللىسىڭىز ئاپتونوم رايونىمىز بويىچە ئەڭ يۇقىرى تەمىنات ۋە ئەڭ ئەۋزەل كەسپىي مۇھىتتىن بەھرىمەن بولىسىز!

قوبۇل قىلىنىدىغان كەسىپلەر:

كىلىنىكا دوختۇرى: سانى، جىنسى، نوپۇس تەۋەلىكى چەكلەنمەيدۇ.

شەرتى: ئوقۇش تارىخى مەخسۇس كۇرۇس (داجۈەن) تىن يۇقىرى ئوقۇش تارىخىغا ئىگەبولغانلار قوبۇل قىلىنىدۇ.

كەسپىي سالاھىيەت كىنىشكىسىنى ھازىرلاپ بولغان دوختۇرلارغا ئوقۇش تارىخى چەكلىمىسى قويۇلمايدۇ.

سېسترا: سانى، جىنسى، نوپۇس تەۋەلىكى چەكلەنمەيدۇ.

شەرتى: كەسپىي سالاھىيەت كىنىشكىسىنى ئېلىپ بولغانلار قوبۇل قىلىنىدۇ.

بىز تەمىنلەيدىغان شارائىت:

تاماق: دوختۇرخانىمىزدا مەخسۇس خىزمەتچىلەر ئاشخانىسى بار، تاماق باھاسى بازار باھاسىدىن %50 تۆۋەن، سۈپىتى بازاردىن كۆپ ياخشى.

ياتاق: سىرتتىن كەلگەن خادىملار ئۈچۈن ھەقسىز ياتاق ئورۇنلاشتۇرۇلىدۇ.

سۇغۇرتا: ئىشسىزلىق سۇغۇرتىسى، داۋالىنىش سۇغۇرتىسى، ياشانغاندا كۈتۈنۈش سۇغۇرتىسى قاتارلىق ئۈچ خىل سۇغۇرتا بار.

مائاش: ئاساسى مائاش (ئىش ئورنى مائاش، ئوقۇش تارىخى مائاشى، سالاھىيەت مائاش، ئۇنۋان مائاش، ئوقۇتۇش مائاش، تېببىي ئەخلاق، تېببىي ئىستىل تولۇقلىمىسى)،ۋەزىپە مائاش، ئۈنۈم مائاش.

تولۇقلىما مائاش: قاتناش تولۇقلىمىسى، تازىلىق تولۇقلىمىسى، كىتاب كۆرۈش تولۇقلىمىسى، تېلېفون ھەققى تولۇقلىمىسى.

تەربىيىلەش: ئىشتىن بۇرۇنقى تەربىيىلەش، ئىش ئورنى ئالماشتۇرۇش تەربىيىلەش،مەخسۇس كەسىپ بويىچە تەربىيىلەش، مۇتەخەسسىسلەر مەخسۇس تەربىيىلەش، باشقۇرۇش ئىلمى بويىچە تەربىيىلەش، ئىلىم ئالماشتۇرۇش، زىيارەت ئۆگىنىشى، چەتئەللەردە تەكشۈرۈش،ئالماشتۇرۇشتا بولۇش


تىزىمغا ئالدۇرۇش ئورنى: لوقمان دوختۇرخانىسى ئادەم كۈچى بايلىقى بۆلۈمى

دوختۇرخانىمىزنىڭ ئومۇمىي ئەھۋالى

لوقمان دوختۇرخانىسى 1981-يىلى قۇرۇلغان بولۇپ 30 نەچچە يىللىق تارىخى تەرەققىيات جەريانىنى باشتىن كەچۈرگەن، داۋالاش، ئالدىنى ئېلىش، ساقلىقنى ساقلاش،ئوقۇتۇش، تەتقىقات، ساغلاملىقنى ئەسلىگە كەلتۈرۈش بىرگەۋدە قىلىنغان، خوتەن ۋىلايىتى بويىچە چوڭ تىپتىكى ئۇنىۋېرسال ئىككىنچى دەرىجىلىك پۇقراۋى دوختۇرخانابولۇپ ھېسابلىنىدۇ.

نۆۋەتتە دوختۇرخانىمىزدا 450 دىن ئارتۇق ئىشچى-خىزمەتچى بار، ئومۇمىي كۆلىمى 16200 كۋادرات مېتىردىن ئارتۇق، تەستىقلانغان كېسەل كارۋىتى 290 دانە،ئەمەلىي كارۋات 350 دىن ئارتۇق. يىللىق ئامبۇلاتورىيە كېسەل كۆرۈش مىقدارى 115000ئادەم قېتىمدىن ئارتۇق، بالنىستتىن داۋالىنىپ چىقىپ كېتىدىغان كېسەل 18196 ئادەمقېتىمدىن ئارتۇق، يىللىق ئوپېراتسىيە مىقدارى 4000 ئادەم قېتىمدىن ئارتۇق بولۇپ،نۆۋەتتە دوختۇرخانىمىزدا 14 دانە كېلىنىكىلىق داۋالاش بۆلۈم، 13 دانە قوشۇمچەتەكشۈرۈش بۆلۈم، 8 دانە فۇنكتسىلىك بۆلۈم تەسىس قىلىنغان، س ت ، خېسى، رەڭلىك بچاۋ، پۈتۈن ئاپتوماتىك بىئولوگىيىلىك تەكشۈرۈش ئۈسكۈنىسى، قورساق بوشلۇقى ئەينىكى قاتارلىق ئىلغار سايمانلاردىن سىرت چوڭ تىپىتكى ئۇنىۋېرسال دوختۇرخانىلاردابولۇشقا تېگىشلىك بارلىق سايمانلار تولۇق بولۇپ، ھازىرقى زامان ئىلغار باشقۇرۇش ئەندىزىسى قوللىنىلغان.

دوختۇرخانىمىز شىنجاڭ ئۇيغۇر ئاپتونوم رايونلۇق خەلق دوختۇرخانىسى ۋەشىنجاڭ تېببىي ئۇنىۋېرسىتېتى قارمىقىدىكى 1-دوختۇرخانا بىلەن تېخنىكا، تەربىيىلەش،ھەمكارلىق، باشقۇرۇش قاتارلىق جەھەتلەردە ئومۇميۈزلۈك ھەمكارلىق ئورناتقان،ئۇنىڭدىن باشقا دوختۇرخانىمىز ھازىر جۇڭگو دوختۇرخانا ئىلمىي جەمئىيىتى پۇقراۋى دوختۇرخانىلارنى باشقۇرۇش تارماق كومىتېتىنىڭ دائىمىي ئەزاسى، شىنجاڭ ئۇيغۇرئاپتونوم رايونلۇق سودا-سانائەتچىلەر بىرلەشمىسى تېببىي دورىگەرلەر جەمئىيىتىنىڭ دائىمى مۇئاۋىن جەمئىيەت باشلىقى، شىنجاڭ دوختۇرخانا ئىلمىي جەمئىيىتى پۇقراۋى دوختۇرخانىلارنى باشقۇرۇش مەخسۇس كومىتېتىنىڭ دائىمىي مۇئاۋىن جەمئىيەت باشلىقى قاتارلىق شەرەپلەرگە نائىل بولغان.

دوختۇرخانىمىز بىمارلارنىڭ داۋالىنىش شارائىتىنى تېخىمۇ ياخشىلاش، كېسەل كۆرسىتىش قىيىن بولۇش مەسىلىسىنى ئۈنۈملۈك ھەل قىلىش ئۈچۈن 2012-يىلى داۋالاش، تېزقۇتقۇزۇش، ياشانغانلارنى كۈتۈش، ساغلاملىقنى ئەسلىگە كەلتۈرۈش، يەسلى، ئائىلىلىك بىرگەۋدە قىلىنغان چوڭ تىپتىكى "لوقمان ساغلاملىق شەھەرچىسى" قۇرۇلۇشىغائىلتىماس قىلىش ئارقىلىق، تەستىقلانغاندىن كېيىن 2013-يىلى 6-ئايدا رەسمىي ئىش باشلىدۇق، ئومۇمىي قۇرۇلۇش كۆلىمى 87500 كۋادرات مېتىر، پىلاندىكى كارۋات سانى1000 كارۋات ئەتراپىدا، ئومۇمىي سېلىنىدىغان مەبلەغ تەخمىنەن 390 مىليون يۈەندىن ئارتۇق، ھازىر بۇ تۈرنىڭ ئۇنىۋېرسال كېسەلخانا بىناسىنىڭ 8 – قەۋىتىگە چىقىپ بولدى، 2015-يىلى يىل ئاخىرىدا پۈتۈپ ئىشقا كىرىشىش پىلانلىنىپ بولدى.

ئىش ھەققىنى جىق بىرىمىز دەپسىز يۇ، لىكىن يۇقىرى ئۆلچەم بىلەن تۆۋەن ئۆلچەمنى بولسىمۇ دىمەپسىز.  سىستىرالارنىڭ مائاشى مانچە پۇلدىن تۆۋەن بولمايدۇ، مانچە پۇلدىن يۇقىرى بولمايدۇ. دوختۇرلارنىڭ ئۇنۋان، ئوقۇش تارىخى، خىزمەت تارىخىغا ئاساسەن مانچە پۇلدىن تۆۋەن بولمايدۇ دىگەندەك جىڭ باسىدىغان گەپنى دىگەن بولسىڭىز ۋىلايەت ئاتلاپمۇ ئىشلىگىلى باراتتى. چۈنكى ھازىر  دۆلەتنىڭ رەسمىيلەشتۈرۈش ئىمتاھانى چىقمىغىلى ئۇزۇن بولغاچقا ھۆكۈمەت دوختۇرخانىلىرىدا ۋاقىتلىق ئىشلەۋاتقان سىستىرالار ئىنتايىن كۆپ،  لىكىن بېسىم ئېغىر. تەمىناتى ياخشى بولسىلا پۇقراۋىي دوختۇرخانىغا بېرىپ ئىشلەشنى چوقۇم ئويلىشىدۇ.  
دوختۇرغا كەلسەك: مەخسۇس كۇرىس ئوقۇغانلاردىن خىزمەت تاپالماي يۈرگەنلەر ئاز ساندا. لىكىن ئوتتۇرا تېخنىكوم سەۋيىسىدىكى خىزمەت كۈتۈپ تۇرغانلار، ۋاقىتلىق ئىشلەۋاتقانلار ئىنتايىن كۆپ  (مىنىڭ ئۆز كۆزۈم بىلەن كۆرگىنىم شۇ، مەن ئىشلەۋاتقان دوختۇرخانىدىمۇ ۋاقىتلىق ئىشلەۋاتقان دوختۇر سىستىرالار رەسمىيلەشكەنلەرنىڭ 2 ھەسسىسىدىن كۆپ)
مىنىڭچە بولغاندا قالپاق(دېپلوم)مۇ مۇھىم، لىكىن ئەمەلىي سەۋىيە قالپاقتىن بەكرەك مۇھىم.
خوتەندە، قەشقەردە كېلىنكا كەسپىدە داۋالاشتا ئوقۇغانلارنىڭ كۆپ قىسمى ئوتتۇرا تېخنىكوم قالپىقى بار.  
ئاجىزلارنىڭ ھەتتا قانداق چۈش كۆرۈشىمۇ باشقىلارنىڭ ئىنژېنىرلىقىدا لاھىيەلەنمەكتە چىڭغىز ئايتىماتوف ‹ئەسىردىن ھالقىغان بىر كۈن›

ئىش ھەققىنى جىق بىرىمىز دەپسىز يۇ، لىكىن يۇقىرى ئۆلچەم ...

ئەسسالامۇ ئەلەيكۇم قېرىندىشىم، مەن بۇ دوختۇرخانىنىڭ دوختۇرخانا باشلىقى ياردەمچىسى بولىمەن، ئۇنىڭدىن باشقا ئادەم كۈچى بايلىقى بۆلۈمىگە مەسئۇل.
دوختۇرخانىمىزنىڭ سىستېمىلاشقان، مۇنتىزىملاشقان مائاش-تەمىنات تۈزۈلمىسى بار، ھەربىر ئادەم ئۆزىنىڭ ئوقۇش تارىخى، كەسپىي سالاھىيىتى، ئۇنۋانى ۋە ئىقتىدارىغا ئاساسەن ئاساسىي مائاش، ئۈنۈم مائاش ۋە باشقا ھەرخىل مائاشلاردىن بەھرىمەن بولىدۇ.
ئوقۇش تارىخى، ۋە كەسپىي سالاھىيىتىنىڭ بار-يوقلۇقىغا دەرىجىسىگە ئاساسەن بېكىتىلگەن ئۆلچەممۇ ئوخشاش ئەمەس، شۇڭا بۇ ھەقتە تېخىمۇ تەپسىلىي تەپسىلاتقا ئىگە بولماقچى بولغانلار مەن بىلەن بىۋاسىتە ئالاقىلاشساڭلا مەن چۈشىنىشلىك قىلىپ ئېيتىپ بېرىمەن.
جۇڭگۇدا ساختىلىق يوق، ياكى ئۆتۈش ئىمكانىيىتى ئىنتايىن قىيىن بولغان ئىمتىھانلارنىڭ يەنە بىرسى دەل تېببىي خادىملارنىڭ كەسپىي سالاھىيەت ئىمتىھانى، ئۇنىڭ ئۈستىگە كەسپىي سالاھىيەت كىنىشكىسى بولمىسا قانۇنسىز كەسىپ قىلغانلىق بولىدۇ، دوختۇرخانىمىز ھازىر ئىككىنچى دەرىجىلىك ئۇنىۋېرسال دوختۇرخانا (يەنى ناھىيە دەرىجىلىكتىن يۇقىرى)، شۇڭلاشقا دوختۇرخانىمىزدا ئىشلىمەكچى بولغان ھەربىر تېببىي خادىمنىڭ كەسپىي سالاھىيىتىگە قويۇلىدىغان تەلەپ ئۆلچەممۇ يۇقىرى ۋە قاتتىق. ھازىر ئوتتۇرا تېخنىكومنى پۈتتۈرگەن خادىملارغا بولغان ئېھتىياج توشۇپ بولدى، شۇڭا مەخسۇس كۇرستىن يۇقىرى ئوقۇش تارىخىغا ئىگە بولغانلارنى قوبۇل قىلىمىز، كەسپىي سالاھىيەت كىنىشكىسىنى ھازىرلاپ بولغانلار بولسا ئوقۇش تارىخى شەرتى قويۇلمايدۇ.
پۈتۈن ئۇيغۇر ئاپتونوم رايونى تەۋەسىدىن قوبۇل قىلىمىز.
