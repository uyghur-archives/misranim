6- باپ

سۈمەييەنىڭ چوڭ ھاممىسىنىڭ مەھەللىسىدە تەقرىبەن 24~25  ياشلار چامىسىدا كېلىدىغان موتسىكىلىت رېمونتچىلىقى بىلەن شۇغۇللىنىدىغان بىر يىگىت بار ئىدى، بۇ بالىنىڭ دۇكىنى يول ئۈستىدىلا بولغاچقا، سۈمەييە بىر قانچە قېتىم بازارغا بېرىپ كېلىش جەريانىدا ،ئۇ يىگىت سۈمەييەنى كۆرگەن ھەم ياقتۇرۇپ قالغان ئىدى، بۇ ۋاقىتلاردا سۈمەييەگە باشقا جايدىنمۇ ئەلچى كەلگەن بولۇپ، ئەگەر بۇ يىگىت تىزراق تۇتۇش قىلمىسا، سۈمەييەدىن قۇرۇق قېلىشى مۇمكىن ئىدى، ئۇ يىگىت ئاۋال ئائىلىسىدىكىلەرنى ماقۇل كەلتۈردى، كېيىن سۈمەييەلەرنىڭ كىچىك ھاممىسىنىڭ ئۆيىگە ئەلچىلىككە كىردى، ئۇنىڭ سۈمەييەنىڭ دادىسىنىڭ ئۆيىگە ئەلچىلىككە كىرمەستىن، بەلكى، سۈمەييەنىڭ كىچىك ھاممىسىنىڭ ئۆيىگە ئەلچىلىككە كىرىشنىڭ سەۋەبى شۇكى، بۇ ۋاقىتلاردا سۈمەييە شۇ كىچىك ھاممىسىنىڭ ئۆيىدە تۇرۇۋاتاتتى، يەنە بىر تەرەپتىن بۇ يىگىتنىڭ سۈمەييەنىڭ ھاممىسىنىڭ ئېرى بىلەن خېلى ئوبدان تونۇشلىقى باتتى. شۇنداق قىلىپ ئەلچىلەر كىردى، ئىككى تەرەپ ئۆز ئارا كېلىشىپمۇ قالدى، ئەمدىكى مەسىلە سۈمەييەنىڭ رازىلىقىدا قالغان ئىدى، سۈمەييە بۇ تويغا رازىلىق بەردى، يىگىت بۇنى ئاڭلاپ شۇنداق خوش بۇلۇپ كەتتى، نېمىشقا خوش بولمىسۇن؟ چۈنكى، يىگىت ئەمدى كۆڭلىدىكى ئادىمى بىلەن توي قىلالايدىغان تۇرسا! ئىككىلى تەرەپ كېلىشىش ئارقىلىق تويلۇق مەھرى ھەققىمۇ بېكىتىلدى ۋە توينىڭ ۋاقتىنى بېكىتكەندىن كېيىن ئۇلار توي تەييارلىقىغا چۈشۈپ كېتىشتى. بۇ يەردە شۇنىمۇ تەكىتلەپ ئۆتىمىزكى، بۇ تويغا سۈمەييەنىڭ دادىسىدىن بەكراق، سۈمەييەنىڭ كىچىك ھاممىسى كۈچىدى، چۈنكى ئۇ ئايال ئاپىسىنىڭ ۋەسىيتىنى ئۇنۇتمىغان ئىدى، ئەلۋەتتە سۈمەييەنىڭ بەختىنى كۆرۈش ئۇ ئايال ئۈچۈن،  ئۆز بالىلىرىنىڭ بەختىنى كۆرگەنلىك بىلەن باراۋەر ئىدى، لېكىن بۇ ئايال ھەم ئۆز نۆۋىتىدە ئاز خاپىلىقلارنى تارتمىدى،《بەزىلەر》نىڭ ئويىچە بولغاندا، بۇ ئايال سۈمەييەنىڭ مەھرى ھەققىنى تاپشۇرۇپ بېرىشكە《تېگىشلىك》ئادەملەرگە تاپشۇرۇپ بېرىپ، ئۆزى بۇ ئىشلارغا بەك ئارلىشىپ كەتمەسلىكى كېرەك ئىدى، بۇ يەردىكى《بەزىلەرنىڭ》كىملىكىنى ئوقۇرمەنلەر پەرەز قىلىپ بولدىغۇ دەيمەن، لېكىن سۈمەييەنىڭ ھاممىسى ئۇنداق قىلمىدى، بۇ ئايالغا ئايان، ئەگەر بۇ توي ئىشىغا بۇ ئايال قول تىقمىسا، بۇ توينىڭ ئاخىرىنىڭ سۈمەييەنىڭ يىغىسى بىلەن ئاخىرلىشىشى تۇرغانلا گەپ، ئەلۋەتتە، تويى بولغان قىزلارنىڭ يىغلىشى تەبىي ئەھۋال، لېكىن بۇ يىغا ئوخشىمايتى، بۇ يىغا غېرىپلىق يىغىسى بولاتتى، سۈمەييەنىڭ ھاممىسى بۇ بوشلۇقنى تولدۇرۇشقا تىرىشتى، سۈمەييەگە ئاپىسىنىڭ يوقلىقىنى بىلدۈرمەسلىككە تىرىشتى، مەيلى قانچىلىك خاپىلىق تارتسۇن، مەھرى ھەققىنى ھېچكىمگە بەرمەي، سۈمەييەنى بازارغا ئۆزى ئەچىقىپ، سۈمەييەنىڭ كۆڭلى خالىغان نەرسىلەرنى ئېلىپ بەردى، ئۆزى يېنىدىن ئارتۇق چىقىم قىلسا قىلدىكىن، ھەرگىزمۇ سۈمەييەنىڭ ھەققىگە تاجاۋۇز قىلمىدى، سۈمەييەمۇ مۇشۇنداق بىر ھاممىسىنىڭ بولغىنىدىن ئىنتايىن خوشال ئىدى.
ئەپسۇسلەر بولسۇنكى، سۈمەييەنىڭ ھاممىسىنى ئارقىمۇ-ئارقا كەلگەن خاپىلىقلار ھالسىراتتى، بىر سەبىر قىلدى، ئىككى سەبىر قىلدى، لېكىن بۇ ئايالنىڭ بېشىدىمۇ ئۆزىگە چۇشلۇق تۇرمۇش بېسىمى بولغاچقا ئاخىرى سەبىر قاچىسى تاشتى، شۇڭا پۇلنى سۈمەييەنىڭ دادىسى بىلەن ئۆگەي ئاپىسىغا تاشلاپ بىرىپ، ئۆزى بۇ توي ئىشلىرىغا ئەمدى ئارلاشماسلىق قارارىغا كەلدى، بىچارە سۈمەييە نىمە قىلارىنى بىلمەي تەڭقىسلىقتا تۇراتتى، ئۇ چېتىگە قارىسا دادىسى، بۇ چېتىگە قارىسا ھاممىسى. ئەمەلىيەتتە، سۈمەييەنىڭ ھاممىسىنىڭمۇ سۈمەييەنى بۇنداق قىيىن ئەھۋالدا قويغۇسى يوق، لېكىن بۇ ئايال توينىڭ ئىشلىرى ئۈچۈن ئۇيان-بۇيانغا يۈگىرەپ، ھاردىم-تالدىم دىمەي بۇ تويغا كۈچەۋاتقاندا، ئارقىمۇ-ئارقا كېلىۋەرگەن خاپىلىقلار ئۇنى بەكلا چارچاتقان ئىدى، شۇ سەۋەپتىن بۇ ئايالمۇ شۇ قارارنى چىقىرىشقا مەجبۇر بولدى. ئېنىقكى، ئىشنىڭ بۇ تەرەپكە يۈزلىنىپ قېلىشى ياخشىلىقتىن دېرەك بەرمەيتتى، ئەگەر يەنە مۇشۇنداق كېتىۋەرسە، بۇ توينىڭ بۇزىلىشى تۇرغانلا گەپ، ھەي.....

7-باپ   يىتىم قىزنىڭ بەختى

سۈمەييەنىڭ كىچىك ھاممىسى ئەمدى بۇ توي ئىشىغا ئارلاشمايمەن دەپ تۇرۇۋالغاچقا، توي بۇزۇلۇش گىردابىغا بېرىپ قالغان ئىدى، بۇ ۋاقىتتا سۈمەييەنىڭ باشقا ھاممىلىرى ئارىغا چۈشتى ۋە:《سىڭلىم،غۇلاچ تۈگەپ غېرىچى قالغاندا بۇنداق دەپمۇ تۇرۇۋالمىغىن، ھېچ بولمىسا شۇ يىتىم سۈمەييەنى بولسىمۇ ئويلاپ قوي》دېيىشتى، سۈمەييەنىڭ كىچىك ھاممىسىنىڭمۇ يۈرىكى گۆشتىن پۈتۈلگەندە! ئاچىلىرى شۇنداق دەپ تۇرىۋاتسا، ئۇ نېمىمۇ دىيەلىسۇن؟شۇنداق قىلىپ، ئىشلار بىرەر قۇر ئىزىغا چۈشتى، توي كۈنىمۇ يېقىنلاپ كېلىۋاتاتتى،ئۇلار بۇ تويغا ھەددىدىن زىيادە ھەشەم قىلىپمۇ كەتمىدى، بەك غورىگىلمۇ قىلىپ قويمىدى، ئىشقىلىپ ھەممە تەييارلىقلارنى نورمال ھالەتتە ئېلىپ باردى...
ھەممە خوشاللىققا چۆمگەن، ئۇششاش-بالىلارنىڭ قىقاس-چوقانلىرى ھەممە يەرنى بىر ئالغان قۇتلۇق توي كۈنىمۇ يىتىپ كەلدى، نىكاھ ئوقۇيدىغانلارمۇ كېلىپ بولۇشقانىدى، نىكاھ ئوقۇيدىغان موللام ئاۋال نىكاھ خۇتبىسىنى ئوقۇغاندىن كېيىن، سۈمەييەگە قاراپ:سىز پالانچى ئوغلى پالانچىنى ئۆزىڭىزگە يولداشلىققا قوبۇل قىلدىڭىزمۇ؟ دېگەن ۋاقىتتا،كىچىكىدىنلا ئەدەپ-ئەخلاق، شەرمى-ھايانى ئۆزىنىڭ لىباسى قىلىپ چوڭ بولغان بۇ قىز خىجىللىقتىن قىزىرىپ-تاتىرىپ يۈرۈپ، قوبۇل قىلدىم دېگەن بىر ئېغىز گەپنى شۇنداق تەستە دېدى. نىكاھ سورۇنى ئاخىرلىشىپ، مېھمانلار ئۇزىغاندىن كېيىن، ئۆيدىكى چوڭلار ئەمدى تويغا كېلىدىغان مېھمانلارنىڭ تەييارلىقىنى باشلىۋەتتى...
چوڭلار ئالدىراش مېھمان كۈتىۋاتاتتى، كىچىك بالىلار بولسا قىقاس-چوقانلىرى بىلەن ئەتراپنى بىر ئالغانىدى، ياش قىزلار بولسا سۈمەييەنىڭ قېشىغا كىرىپ، ئۇنىڭ بىلەن چاقچاقلاشقاچ، ئۇنىڭ كام-كۇتىلىرىنى تولۇقلاش بىلەن ئاۋارە ئىدى، ھاراق-شاراپتىن خالى بولغان بۇ توي شۇنداق كۆڭۈللۈك ھالەتتە ئېلىپ بېرىلىۋاتاتتى، سىرىتتىن يىگىت كەلدى! دېگەن ئاۋازنىڭ چىقىشى بىلەن تەڭ قىزلار بار ئۆيدىكى ۋاراڭ-چۇرۇڭ سەل بېسىقتى، يىگىت تەرەپتىكىلەر ئۆيگە كىرىپ كەلدى، سۈمەييە خىجىللىقتىن يەردىن ئۈستۈن قارىماي ئولتۇراتتى، يىگىت كىلىپ سۈمەييەنىڭ يېنىغا جايلاشقاندىن كېيىن، يىگىت تەرەپتىن كەلگەنلەرگە قايتىدىن داستىخان ھازىرلىنىپ، نازۇ-نېمەتلەر تىزىلدى، قىزنى كۆچۈرىدىغان ۋاقىتتا، چوڭلار بۇ ئىككى ياشنىڭ بەختى ئۈچۈن دۇئاغا قول كۆتۈردى، ھەممەيلەن دۇئا قىلىشىۋاتقان بۇ دەقىقىدە، كىملەردۇر بىرلىرىنىڭ مىشىلداپ يىغلىغان ئاۋازىنى ھېسابقا ئالمىغاندا، ئۆي ئىچى تىمتاسلىققا چۆمگەنىدى، دۇئامۇ ئاياغلاشتى، سۈمەييە دادىسى بىلەن ئۆگەي ئاپىسىنىڭ ۋە ھاممىلىرىنىڭ ئالدىغا كېلىپ شۇنداق يىغلاپ كەتتى، بەلكىم ئۇنىڭ بۇ قەدىردانلىرىدىن ئايرىلىدىغانلىقى ئۈچۈن كۆڭلى يېرىم بولىۋاتقان بولسا كېرەك، سۈمەييەنىڭ ئۆگەي ئاپىسىمۇ ئۈن سېلىپ يىغلاپ كەتتى، نىمىلا دىمىگەن بىلەن ياخشى بولسىمۇ يامان بولسىمۇ بىر نەچچە يىللار ئانا-بالا بولۇشتى ئەمەسمۇ؟ سۈمەييەمۇ ئەينى چاغدا ئاران ئىككى ياشلىق گۆدەك قىز ئىدى، مانا ئەمدى ئۇچۇرۇم بولۇش ئالدىدا تۇرغان قۇش بالىسى كەبى، باشقىلارنىڭ ئىلكىگە ئۆتۈش ئالدىدا تۇراتتى، ھەممە بۇ ئىككىسىگە بەخت تىلەشتى، شۇنداق ھاياجانلىق تۇيغۇلار ئىلكىدە بۇ ئىككى ياش قەلبىگە نۇرغۇنلىغان گۈزەل-ئارزۇ ئارمانلارنى پۈكۈشتى، ۋاپا شارابى بىلەن سۇغۇرۇلغان ئىللىق بىر ئائىلە قۇرۇشنى نىيەت قىلدى، بۈگۈن بۇ ئىككىسى بۇندىن كېيىنكى ھايات مۇساپىسىنى بىللە داۋاملاشتۇرۇش ئۈچۈن قوللىرىنى مەھكەم تۇتۇشقانىدى، بۇلار باشقا ياشلاردەك مۇھەببەتلىشىپ توي قىلمىدى، ئۇلار نىكاھلىنىپ بىر-بىرىگە ھالال بولۇشقاندىن كېيىن، بىر-بىرىگە بولغان ھەقىقى، پاك مۇھەببىتىنى ئەنە شۇ ۋاقىتتا ئىزھار قىلىشسىمۇ كېچىكمەيدىغانلىقىنى بىلىپ يېتىشكەنىدى، ھەقىقەتەن بەخت ئۇلارغا قۇچاق ئاچتى، قىز كۆچۈرۈلدى، سۈمەييە نۇرغۇنلىغان ئاچۋىق-تاتلىق ئەسلىمىلەرگە تولغان بۇ مەھەللىنى ئارقىدا قالدۇرۇپ، بۇندىن كېيىنكى يېڭى ھاياتى باشلىنىدىغان يېڭى《دۇنياسى》تامان يۈرۈپ كەتتى....



                     خاتىمە

بىر يىلدىن كېيىن


ئۇچقان قۇش ئاسماندا ئۇچىۋەرمەيدۇ، شۇ قۇشنىڭمۇ ھامان پەسكە قونىدىغان ۋاقتى بولىدۇ ۋە پەستىكى قۇشنىڭمۇ كۆككە قاراپ ئۇچىدىغان ۋاقىت بولىدۇ، شۇنىڭدەك، مەن بەختسىز دەيدىغان ئادەمنىڭمۇ كۈلۈپ تۇرۇپ:مەن بۈگۈن ئۆزەمنى شۇنداق بەختلىك ھېس قىلدىم دەيدىغان ۋاقىتلىرى بولىدۇ ۋە بىر چاغلاردا ئۆزىنى ئاجايىپ قالتىس چاغلىغان، ئۆزىگە كەلگەن بەختتىن تېلىقىپ كۈلگەن ئادەملەرنىڭمۇ، بېشىغا كەلگەن قىسمەتكە قاراپ قان يۈتۈپ يىغلايدىغان ۋاقىتلىرى بولىدۇ. كىشىلىك تۇرمۇش ئەنە شۇنداق رەڭدارلىققا تولغان، مانا سۈمەييەمۇ تۈنۈگۈنكى سەبىي قىزدىن بۈگۈن بىر ئانا بولۇش ئالدىدا تۇراتتى، ئۆزىنىڭ ئانا بولىدىغانلىقىنى ئويلىسا، تەسۋىرلىگۈسىز بىر لەززەت ئۇنى خۇشال قىلاتتى، ئەلۋەتتە سۈمەييەنىڭ ئېرىنىڭ خوشاللىقىمۇ بۇنىڭدىن قېلىشمايتتى، مەن دادا بۇلىدىغان بولدۇم دەپ كىچىك بالىدەك سەكرەپ كەتكەن ئادەم باشقا بىرى بولماستىن، دەل سۈمەييەنىڭ ئېرى ئىدى، سۈمەييە كەينىگە قارىدى، ئۆزىنىڭ 19 يىللىق ھاياتىدىكى ئۆتمۈشىنى كۆردى، ئەمدى ئۇ بۇ ئۆتمۈشلەرنى كەينىگە قالدۇرۇپ قويۇپ، ئالدىغا قاراپ دادىل قەدەم ئالدى، تۇرمۇشىدا قانداقلا قىيىنچىلىقلارغا ئۇچرىمىسۇن، چوقۇم سەبىرچان بولۇپ، بالىسىنى يىتىم قىلماسلىقنى كۆڭلىگە پۈكتى، ئۆزىنىڭ ئانا بولىدىغانلىقىدىن ئىپتىخارلاندى، مۇھىمى ئۇ ھاياتىدىكى ئەڭ چوڭ بەختكە يەنى ئىسلامغا كىرىش شەرىپىگە مۇيەسسەر بولالىدى، مانا بۇ يىتىم قىزنىڭ بەختى ئىدى.......


تامام...




ئەسەر ۋەقەلىكىگە توقۇلما ئارلاشتۇرماي، پەقەت ھەقىقى بولغان ۋەقەلىكنىلا يازغاچقا،  ئەسەرنىڭ بەزى جايلىرى كەمتۈك بولۇپ قالغان بولىشى مۇمكىن، ئوقۇرمەنلەرنىڭ كۆرسىتىپ بېرىشىنى سورايمەن.....
