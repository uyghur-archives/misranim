روزا ھېيىت توغرىسىدا

   مەزكۇر تىمىنى توردىن ئىزدە تېپىپ دوستلارنىڭ كۆرشىگە سۇندۇم،مەنچە تۆۋەندىكىلەر ھەربىرىمىز بىلىشكە تىگىشىلىك زۈرۈر بىلىملەر ئىكەن،ھەمدە پەرزەنتلەرگە سۆزلەپ بىرىشىمىز لازىملىقنى ھىس قىلىپ،مىسرانىم تورىدا تېخىمۇ كۆپ تورداشلارنىڭ پايدىلنىشى ئۈچۈن سۇندۇم.

روزا ھېيىت توغىسىدا:

روزا ھېيىت مۇسۇلمانلارنىڭ مۇھىم بىر ھېيىت ئايىمى بۇلۇپ ،قۇربان ھېيىت بىلەن ئىككىسى ئىسلام دىنىدىكى ئىككى چوڭ ھېيىت ھېسابلىنىدۇ.روزا ھېيىت ئەرەپچە [ئەيد فىتىر] دەپ ئاتىلىدۇ .بۇ ھېيىت مۇسۇلمانلار ھىجىرىيە شەۋۋال[-10 -ئاي]نىڭ 1 - كۈنى ئۆتكۈزىلىدۇ ۋە بۇ ئارقىلىق بىر ئاي روزىنىڭ مۇۋاپىقىيەتلىك تاماملانغانلىقىنى خوشال -خورام تەبرىكلىنىدۇ.روزا ھېيىت ھىجريىنىڭ 2 -يىلى يەنى مىلادىيا 623 -يىلى بېكىتىلگەن.ھېيىت رەسمىيىتى مۇنداق بۇلىدۇ ؛ [1] مۇسۇلمانلار بۇ كۈنى ناماز بامداتتىن كىيىن دەرھال ئازراق تائام يەپ ،رامىزان روزىسىنى مەمنۇنىيەتلىك تاماملاپ روزىدىن چىققانلىقى مۇناسىۋىتى بىلەن ئاللاھ تائالاغا بولغان تەشەككۈرىنى ئىپادىلەيدۇ[2] فىتىر سەدىقىسى [ئەرەپچەئاتىلىشى ،سەدىقە فىتىر،]بىرىدۇ.بۇ ۋاجىپ سەدىقە بۇلۇپ،مۇسۇلمانلار ئەر-ئايال،چوڭ-كىچىك دەپ ئايرىماي،ھەربىر ئائىلىدىكى جان سانى بۇيىچە ھېيىت نامىزىدىن ئاۋۋال ھېيىت خوشاللىقى ئۈچۈن بىرىدىغان بىرخىل سەدىقە. ھەرقايسى جايلارنىڭ ئەمىلى ئەھۋالى،ئۆلچەم بىرلىكى،مال باھالىرى ئوخشاش بولمىغاچقا،بۇرۇنقى زامانلاردا ھەرقايسى ئەللەردە شۇ ئەلنىڭ ئەمەلىي ئەھۋالىغا قاراپ ئاشلىقلارنى ئۆلچەم قىلىپ،تۈرلۈك ئۆلچەملەر بىېكىتىلگەن ئىدى.بۇ سەدىقە ھازىركۆپىنچە پۇلغا سۇندۇرلۇپ بېرىلىۋاتىدۇ،يەنى ھەر قايسى ئەللەردىكى،ھەر قايسى جايلاردىكى مۇسۇلمانلار ئۆز جايىدىن چىقىدىغان ئاساسلىق ئاشلىق [بوغداي،قۇناق،گۈرۈچ قاتارلىقلار]دىن ئىككى يېرىم كىلوگىرام [بەش جىڭ ]ئاشلىقنىڭ شۇ چاغدىكى،شۇ جايدىكى باھاسى بۇيىچە نەق پۇلغا سۇندۇرۇپ بىرىدىغان بولدى. فىتىر سەدىقىسى چوقۇم يوقسۇل،غېرىپ-غۇرۇۋالارغا بىرىلىشى كىرەك،بەزى جايلاردا مەسچىتكە تاپشۇرلىدۇ ،ئاندىن مەسچىت شەرىئەت بۇيىچە بېرىشكە تېگىشلىك كىشىلەرگە بىرىدۇ.فىتىر سەدىقىسى بىرىپ بولغاندىن كىيىن ھېيىت نامىزى ئوقۇلىدۇ.[3] ھېيىت نامىزى ئۆتەش.ھېيىت نامىزى چۈشتىن بۇرۇن ئوقىلىدۇ.مۇسۇلمانلار ھېيىت نامىزى ئۆتەشتىن بۇرۇن غۇسلى تاھارەت ئېلىپ،ساقال-بۇرۇتنى ياساپ،يىڭى ياكى پاكىز كىيىملىرىنى كىيىپ،قەلبىدە ۋە ئېغىزىدا ھەمدۇسانا ئوقۇپ شۇ جايدىكى ئەڭ چوڭ جامەگە ياكى ھېيىتگاھقا ھەيۋەتلىك ھېيىت نامىزىغا قاتنىشىدۇ،ھېيىت نامىزى شەرىئەتتە ۋاجىپ ناماز ھېسابلىنىدۇ،ناماز ئۆتەپ بولغاندىن كىيىن مۇسۇلمانلار ئۆز ئارا سالام ۋە قولبىرىشىپ .بىر -بىرىنىڭ ھېيىتىنى تەبرىكلىشىدۇ.ئاندىن قەبرىستانغا بېرىپ زىيارەت قىلىدۇ.

روزا ھېيت نامىزى بىسمىللاھىر رەھمانىر رەھىم ھېيتىڭلارغا مۇبارەك بولسۇن! مۆھتەرەم قېرىنداشلار! ھازىر ھىجرىيە 1429- يىلىنىڭ رامزان ئېي ئاخىرىلىشىش ئالدىدا تۇرىۋاتىدۇ، ئىنشائاللاھ بىر قانچە كۈندىن كېيىن روزا ھېيت يېتىپ كېلىدۇ. مەن بۇ مۇناسىۋەت بىلەن سىلەر قېرىنداشلىرىمنىڭ روزا ھېيتىڭلار تەبرىكلەيمىز ۋە بۇ ئۇلۇغ كۈنلەردە نېمىلەرنى قىلىشىمىز ۋە نېمىلەرنى قىلماسلىقىمىز تۇغرىسىدا بىرئاز مەلۇمات بەرمەكچىمەن. قېرىنداشلار! ھېيت نامىزىغا بېرىشتا، ھېيت نامىزى ئوقۇشتا، ھېيت نامىزى ئوقۇش جەريانىدا ۋە ھېيت نامىزى ئوقۇپ بولغاندىن كېيىن قىلىشىمىز كېرەك بولغان، قىلىساق ياخشى بولىدىغان ۋە قىلشىمىزغا رۇخسەت قىلىنغان ئىشلار تۆۋەندىكىچە: ھېيت نامىزىغا بېرىشتىن بۇرۇن قىلىنىدىغان ئىشلار 1- روزا ھېيت كۈنى ئەتە شەۋۋال ئېيىغا بىر دېگەن كۈنىنىڭ ئاخشىمى يېڭى ئاينى كۆرۈش بىلەن، ئەگەر ئاي كۆرۈنمىسە رامزان ئېيىنى 30 كۈن قىلىپ تاماملاش بىلەن بەلگىلىنىدۇ. بۇنىڭغا ئاساسەن رامزاننىڭ 29 - كۈنى كەچتە يېڭى ئاينى كۆرسەك ئەتىسى روزا تۇتۇشىمىز توغرا ئەمەس. ئەگەر رامزاننىڭ 29 - كۈنى كەچتە يېڭى ئاينى كۆرمىسەك، ئەتسى بىر كۈن روزا تۇتۇپ رامزاننى 30 كۈن قىلىپ تاماملايمىز ۋە بۇنىڭدىن كېيىنكى كۈنى روزا تۇتمايمىز. بۇ ئىش، بىرەر سەۋەب بىلەن ھېيت نامىزى ئوقۇش كېچىككەن بولسىمۇ مۇشۇنداق بولىدۇ. يەنى بىز ئاي كۆرگەن ئاخشامنىڭ ئەتىسى ياكى رامزاننى 30 كۈن قىلىپ تاماملىغان كۈننىڭ ئەتىسى ھېيت نامىزى ئوقۇلمىسىمۇ روزا تۇتمايمىز. 2- ئۆزىمىز ئۈچۈن ۋە ئائىلىمىزنىڭ ھەر ئەزاسى ئۈچۈن (چوڭ- كىچىك، ئەر- ئايال پەرقى يوق) سەدىقە فىترە بېرىشمىز كېرەك. مىقدارى بىر كىشى ئۈچۈن بىر مىسكىننىڭ بىر كۈنلۈك يىمەك- ئىچمىكىگە يەتكۈدەك بىر نەرسىدۇر. سەدىقە فىترىنى ھېيت نامىزىغا چىقىشتىن بۇرۇن بېرىۋەتسەك ئەڭ تۇغرىسىنى قىلغان بولىمىز؛ ئەگەر بىرەر سەۋەب بىلەن كېچىكىپ قالغان بولسا، ھېيت نامىزىدىن كېين بولسىمۇ بېرىشىمىز لازىم. ھېيتتىن بىر - ئىككى كۈن بۇرۇن بەرسەكمۇ بولىدۇ. بۇنى بېرىش ئۈچۈن ئىقتىسادىي جەھەتتە مەلۇم بىر نىسابقا يېتىش شەرت ئەمەس، ھەتتا بۇنى بىر كۈنلۈك ئوزۇق- تۈلۈكى بار كىشىلەرمۇ بېرىشى كېرەك. 3- ھېيت نامىزىغا چىقىشتىن بۇرۇن بىر نەرسە يەپ- ئىچىشمىز ياخشى، ئەگەر مۇمكىن بولسا خورما يەپ بارساق تېخىمۇ ياخشىسىنى قىلغان بولىمىز. «ھېيت نامىزى ئوقۇلۇپ بولىغۇچە بىر نەرسە يېمەسلىك ياخشى» دېگەن سۆز قۇربان ھېيتتا كۈچگە ئىگە. روزا ھېيتتا بولسا، يەپ - ئىچىپ، ئاندىن ھېيت نامىزىغا بېرىش ياخشىدۇر. 4- ھېيت نامىزىغا غۇسۇل يەنى بوي تاھارەت ئېلىپ، تازىلىق قىلىپ، ئەڭ چىرايلىق ۋە ئەڭ پاكىز كىيىملەرنى كىيىپ بېرىش ياخشىدۇر. 5- ھېيت نامىزىغا كېتىۋېتىپ تۆۋەندىكى تەكبىرنى ئوقۇيمىز: ئاللاھۇ ئەكبەر، ئاللاھۇ ئەكبەر، لا ئىلاھە ئىللەلاھۇ، ئاللاھۇ ئەكبەر، ئاللاھۇ ئەكبەر ۋە لىللاھىلھەمد. تەرجىمىسى: ئاللاھ بۈيۈكتۇر، ئاللاھ بۈيۈكتۇر، ئاللاھدىن باشقا ھېچ ئىلاھ يوقتۇر، ئاللاھ بۈيۈكتۇر، ئاللاھ بۈيۈكتۇر، ئاللاھقا ھەمدۇسانالار بولسۇن، ھېيت نامىزى ئوقۇلىدىغان يەرگە ئەنە شۇ شەكىلدە بارىمىز. ھېيت نامىزى1 - ھۆكمى ھېيت نامىزى ۋاجىپ بولۇپ، جامائەت بىلەن ئوقۇلىدۇ. زۆرۈرەتتىن ھېيت نامىزى ئوقۇلىدىغان يەرگە ھازىر بولالمىغانلار، ئۆزلىرى يالغۇز ئوقۇمايدۇ ۋە قازاسىنىمۇ قىلمايدۇ، شۇنداقلا گۇناھكارمۇ بولمايدۇ. شۇنىڭ ئۈچۈن ئوقۇش ياكى تىجارەت ۋە ياكى خىزمەت مۇناسىۋىتى بىلەن ھېيت نامىزى ئوقۇلىدىغان يەرگە ھازىر بولالماسلىقىنى چاغلىغانلارنىڭ ئوقۇش يېرىنى ياكى تىجارەت ۋە خىزمەت قىلىش ئورنىنى تەرك ئېتىپ، ھېيت نامىزى ئوقۇش ئۈچۈن ئاتايىن باشقا شەھەرگە بېرىشىغا ھاجەت يوقتۇر. 2 - ۋاقتى ھېيت نامىزىنىڭ ۋاقتى، كۈن نەيزە بۇيى ئۆرلىگەندىن تارتىپ كۈن قايرىلغۇچە بولغان ئارلىقتىكى ۋاقىتتۇر. 3 - ئوقۇلۇش شەكلى 1 - روزا ھېيت نامىزى بولسۇن، قۇربان ھېيت نامىزى بولسۇن، ھېچبىر ھېيت نامىزىغا ئەزان - تەكبىر ئوقۇلمايدۇ. 2 - ئىمامنىڭ ياكى مەزىن (مۇئەززىن) نىڭ (ھازىر ناماز باشلىنىدۇ دېگەندەك) ئېلان قىلىشى بىلەن جامائەت ئورنىدىن تۇرۇپ سەپ بولىدۇ. 3 - بۇ ئەسنادا ھەممەيلەننىڭ كۆڭلىدە روزا ھېيت نامىزى ئوقۇش نىيىتى بولىدۇ. نىيەتنى ئاغزىدا سۆزلەش ياخشى ئەمەس، چۈنكى نىيەتنىڭ ئورنى قەلبتۇر. 4 - ئالدى بىلەن ئىمام «ئاللاھۇ ئەكبەر» دەپ نامازغا شۇرۇ قىلىدۇ (نامازغا باشلايدۇ)، جامائەت ئىمامغا ئەگىشىپ، ئىچىدە «ئاللاھۇ ئەكبەر» دەپ نامازغا شۇرۇ قىلىدۇ. 5 - ئىمام ۋە جامائەت ئىچىدە سانا (سۇبھانەكەللاھۇممە ۋەبىھەمدىكە ۋە تەبارەكەسمۇكە ۋە تەئالا جەددۇكە ۋە لائىلاھە غەيرۇكە) نى ئوقۇيدۇ. 6 - سانادىن كېيىن ئىمام ئۈچ قېتىم ئۈنلۈك «ئاللاھۇ ئەكبەر» دەپ تەكبىر ئېيتىدۇ. ئىمامنىڭ ھەر تەكبىر ئېيتىشىدا جامائەت ئىمامغا ئەگىشىپ ئىچىدە «ئاللاھۇ ئەكبەر» دەپ تەكبىر ئېيتىدۇ. 7 - ئاندىن ئىمام ئىچىدە «ئەئۇزۇ بىللاھى مىنەششەيتانىر رەجىم، بىسمىللاھىر رەھمانىر رەھىم» نى ئوقۇغاندىن كېيىن، فاتىھە سۈرىسىنى ۋە باشقا بىر سۈرىنى (بۇ سۈرىنىڭ ئەئلا سۈرىسى بولۇشى ياخشى) ئۈنلۈك ئوقۇيدۇ. جامائەت جىم تۇرۇپ ئاڭلايدۇ. 8 - ئاندىن ئىمام رۇكۇغا بارىدۇ، جامائەتمۇ ئىمامغا ئەگىشىپ رۇكۇغا بارىدۇ. ئاندىن ئىمام سەجدىگە بارىدۇ، جامائەتمۇ ئىمامغا ئەگىشىپ سەجدىگە بارىدۇ. ھېيت نامازلىرىنىڭ رۇكۇ ۋە سەجدىلىرىدە ئوقۇلىدىغان زىكىر - تەسبىھلەر باشقا نامازلاردىكىگە ئوخشاشتۇر. 9 - ئاندىن ئىمام ئىككىنچى رەكئەتنى ئوقۇش ئۈچۈن ئورنىدىن تۇرىدۇ، جامائەتمۇ ئىمامغا ئەگىشىپ ئورنىدىن تۇرۇشىدۇ. 10 - ئىمام ئورنىدىن تۇرۇپ ئىچىدە «بىسمىللاھ...» نى ئوقۇغاندىن كېيىن فاتىھە ۋە بىر سۈرىنى (بۇ سۈرىنىڭ غاشىيە سۈرىسى بولۇشى ياخشى) ئۈنلۈك ئوقۇيدۇ، جامائەت جىم تۇرۇپ ئاڭلايدۇ. 11 - ئىمام قىرائەتتىن كېيىن يەنى فاتىھەنى ۋە سۈرىسىنى ئوقۇپ بولغاندىن كېيىن ئۈچ قېتىم «ئاللاھۇ ئەكبەر» دەپ ئۈنلۈك تەكبىر ئېيتىدۇ، جامائەتمۇ ھەر تەكبىردە ئىمامغا ئەگىشىپ ئىچىدە «ئاللاھۇ ئەكبەر» دەپ تەكبىر ئېيتىدۇ. 12 - ئىمام ئاندىن قايتا «ئاللاھۇ ئەكبەر» دەپ تەكبىر ئېيتىپ رۇكۇغا بارىدۇ، جامائەت ئىمامغا ئەگىشىپ ئىچىدە «ئاللاھۇ ئەكبەر» دەپ رۇكۇغا بارىدۇ. 13 - ھېيت نامىزىنىڭ قالغان قىسمى ئادەتتىكى نامازغا ئوخشايدۇ، يەنى ئولتۇرۇش، ئەتتەھىيياتۇ... دۇرۇت ۋە دۇئالار باشقا نامازلاردا قانداق ئوقۇلسا، ھېيت نامىزىدىمۇ شۇنداق ئوقۇلۇپ، ناماز ئاخىرلاشتۇرۇلىدۇ. 14 - ئىمام نامازدىن سالام بەرگەندىن كېيىن جۈمە خۇتبىسىگە ئوخشاش ئىككى خۇتبە ئوقۇيدۇ، جامائەت جىم ئولتۇرۇپ ئاڭلايدۇ. ئىككىنچى خۇتبىنىڭ ئاخىرلىشىشى بىلەن ھېيت نامىزى ئاخىرلاشقان بولىدۇ. يىغىپ ئېيتقاندا ھېيت نامىزى ئىككى رەكئەت بولۇپ، ئۇنىڭ ئادەتتىكى نامازغا ئوخشىمايدىغان تەرەپلىرى بار. شۇنىڭ ئۈچۈن ئىمامنىڭ بەرگەن تەلىمىگە دىققەت بىلەن قۇلاق سېلىشىمىز ۋە دىققەت بىلەن ئوقۇشىمىز كېرەك. ھېيت نامىزى ئوقۇلۇپ بولغاندىن كېيىن قىلىنىدىغان ئىشلار 1 - ھېيت نامىزىدىن كېيىن بىر- بىرىمىزنىڭ ھېيتىنى مۇبارەكلەش ياخشى. مەسىلەن: جامەدىن ياكى ھېيت نامىزى ئوقۇلغان يەردىن چىقىپ، قېرىنداشلىرىمىز بىلەن قول ئېلىشىپ كۆرۈشۈپ «اللە قوبۇل قىلسۇن، ھېيتلىرىغا مۇبارەك بولسۇن» دېيىشىشىمىز ياخشى ئىشتۇر. 2 - ھېيت نامىزىدىن قايتىشتا، مۇمكىن بولسا ھېيت نامىزىغا بارغان يولىمىزدىن باشقا بىر يول بىلەن قايتىشىمىز ياخشىدۇر. ھېيت ئىچىدە قىلىنىدىغان ئىشلار 1 - ھېيت ئىچىدە بىر - بىرىمىز بىلەن ھېيتلىشىشىمىز، يەنى ئاتا - ئانىمىزدىن باشلاپ پۈتۈن ئۇرۇق - تۇغقانلىرىمىز ۋە باشقا مۇسۇلمان قېرىنداشلىرىمىز بىلەن بىر - بىرىمىزنىڭ ئۆيىنى زىيارەت قىلىشىشىمىز ناھايىتى ياخشى ئىشتۇر. 2 - ھېيت ئىچىدە نامازلارنى جامائەت بىلەن ئوقۇش، زىكىر- تەسبىھ ئېيتىش، قۇرئان تىلاۋەت قىلىش قاتارلىق ئىشلارنى ئۈزلۈكسىز داۋاملاشتۇرۇشىمىز لازىم. 3 - پېقىر - مىسكىنلەرگە خەير - سەدىقە قىلىش، يېتىم - يېسىرلارنىڭ بېشىنى سىلاش، غېرىب - غۇرۋالارغا خەير - ساخاۋەت قىلىش كېرەك. 4 - بۇ ئۇلۇغ كۈنلەردە پۈتۈن دۇنيادىكى مۇسۇلمان قېرىنداشلىرىمىز ئۈچۈن اللە تەئالاغا دۇئا قىلىشىمىز كېرەك. 5 - ھېيت مۇناسىۋىتى بىلەن بالا- چاقىلىرىمىزنىڭ يىمەك- ئىچمەك، كىيىم- كىچەكلىرىنى ئىسراپ قىلمىغان ئاساستا كەڭ- كەڭرى قىلىپ بېرىمىشىز ياخشىدۇر. 6 - ھېيتتا شەرىئەت رۇخسەت قىلغان (دىن- دىيانەتنى مەسخىرە قىلمايدىغان، ئەخلاقنىڭ بۇزۇلۇشىغا ۋە پىسق- پۇجۇرغا چاقىرمايدىغان، ئەر- ئاياللار ئارىلىشىپ ئوينالمايدىغان) ئويۇن- تاماشالارنى قىلشقا بولىدۇ. ھېيتتىن كېيىن قىلىنىدىغان ئىشلار 1 - رامزاندا تەۋبىگە باغلانغان بېلىمىز، ھېيت بىلەن بوشاپ كەتمەسلىكى لازىم. 2 - رامىزان بويىچە ئادەتلەنگەن تەقۋالىقىمىزنى ئەينەن داۋاملاشتۇرۇشىمىز لازىم. يەنى رامزاننىڭ چىقىپ كېتىشى بىلەن «تەرىتىمىز سۇنۇپ كەتمىسۇن!». 3 - ھېيتتىن كېيىن، ھېيت ئېيى چىقىپ كېتىپ بولغۇچە 6 كۈن نەپلە روزا تۇتۇشىمىز ياخشىدۇر. بۇ روزىنى ئارقىمۇئارقا تۇتۇشىمىز شەرت ئەمەس. 4 - بەرىكەت ۋە رەھمەت ئېيى بولغان رامزان چىقىپ كەتكەن بىلەن، رەھمەت ۋە بەرىكەتنىڭ ئەسلا تۈگىمەيدىغانلىقىنى ئۇتنۇماسلىقىمىز، شۇنداقلا ئۇنىڭغا ھەر دائىم لايىق بولۇپ تۇرۇشىمىز لازىم. خاتىمە قېرىنداشلار! ئىنساننىڭ ئۆمرى بەك قىسقا. بۇ قىسقا مۇددەتلىك ئۆمرىمىزنىڭ بىر پارچىسىنى رامزان ئېيىدا تائەت - ئىبادەتلەر بىلەن ئۆتكۈزگىىنمىزدەك، ئۆمرىمىزنىڭ قالغان قىسمىنىمۇ تائەت - ئىبادەتلەر بىلەن ئۆتكىزىشىمىز لازىم. چۈنكى بىز زامانغا ئەمەس زاماننىڭ ۋە ھەر نەرسىنىڭ پەرۋەردىگارى بولغان اللەغا ئىبادەت قىلىمىز. دىنىمىز بىزنى ھاياتىمىزنىڭ ئاخىرىغىچە توغرا يول ئۈستىدە مۇستەھكەم تۇرۇشقا بۇيرۇيدۇ. ھاياتىمىزنىڭ ھەر سېكۇنتى بىز ئۈچۈن بەك قىممەتلىكتۇر. ئالغان ھەر قەدىمىمىزنىڭ، ئاغزىمىزدىن چىققان ھەر بىر كەلىمىنىڭ ھېسابى بىزدىن ئېلىنىدۇ. بۇنى ئۇنتۇماسلىقىمىز لازىم. بۇ دۇنيادا ھايات تۇرغانلىقىمىز ئىنتايىن كاتتا پۇرسەت؛ تەن سالامەتلىكىمىز، قولىمىزدىكى ئىمكانلىرىمىز بەك چوڭ نېمەت، پۇرسەت قولىمىزدا بار چاغدا پۈتۈن ئىمكانلىرىمىز بىلەن اللە تەئالانىڭ رازىلىقىنى قولغا كەلتۈرۈشكە تىرىشىشىمىز لازىم. قىيامەت كۈنى ئاللاھ تەئالانىڭ ھۇزۇرىغا چىققان ۋاقتمىزدا «بىلمەپتىكەنمەن» ياكى «خەۋىرىم يوقكەن» دېيىش ئىمكانىمىزمۇ بولمايدۇ. چۈنكى ئاللاھ تەئالا قاچان قەيەردە قانداق ئىش قىلىشىمىزنى بۇيرۇش بىلەن بىرگە، ھاياتىمىز بويىچە نېمىلەردىن ساقلىنىشىمىزنىڭ لازىملىقىنى ئۆزىنىڭ كىتابىدا ئوچۇق بايان قىلغان. ئاللاھ تەئالانىڭ ئاخىرقى ئەلچىسى مۇھەممەد ئەلەيھىسسالام بولسا، ئۇ كىتابنى بىزگە يەتكۈزۈش بىلەن بىرگە ئۇ كىتابقا قانداق ئەمەل قىلىشىمىزنى ئۆگەتكەن، شۇنداقلا ۋىدالىشىش نۇتقىدا ساھابىلىرىغا، بىزگە ۋە قىيامەتكىچە دۇنياغا كېلىدىغان پۈتكۈل ئىنسانىيەتكە خىتاب قىلىپ «مەن سىلەرنىڭ ئاراڭلاردا ئاللاھنىڭ كىتابىنى قالدۇرۇپ قويدۇم، بۇنى چىڭ تۇتساڭلار ھەرگىزمۇ ئېزىپ كەتمەيسىلەر» دەپ سۆزىنى ئاخىرلاشتۇرغان. ئاخىرەتتە پۈتمەس - تۈگىمەس پۇشايمانغا قالماسلىقىمىز ئۈچۈن كۈندىلىك ۋاقتىمىزنى دىنىمىزنىڭ كۆرسەتمىسى بويىچە ئۆتكۈزۈشىمىز لازىم. چۈنكى بىزنىڭ ئۇ دۇنيادىكى ئەبەدىي ھاياتىمىزنىڭ راھەت - پاراغەت ئورنى بولغان جەننەتتە باشلىنىشى ۋە جەننەتتە داۋاملىشىشى، بۇ دۇنيادىكى چېغىمىزدا ئۇ كىتابقا بويسۇنۇشىمىز ۋە ئۇ پەيغەمبەرگە ئەگىشىشىمىزگە باغلىقتۇر.
