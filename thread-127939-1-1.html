  مەھەللىمىزدە بىرقىز ئىدىڭ .شەھىرىمىزدە بىر قىز ئىدىڭ،پۈتكۈل ئالەمدە بىرلا قىز ئىدىڭ ... مەن ساڭا ئاشىق ئىدىم.
   خۇدايا توۋا!... سەن ئەمدىلا ئون تۆت ياش ئىدىڭ،مەن بولسام يىگىرمە ياش.لىكىن مەن ساڭا چىندىن ئاشىق ئىدىم.
   ... خۇدايا توۋا!سەن تېخى كىچىككىنە تۇرساڭ-ھە؟! مەن قانداقىسىغا ...ئۆزەمنى گۇناھكار سانايمەن ۋە بۇ گۇناھكار خىيالىمدىن قورقۇپ قالىمەن،سىڭلىمدىن ئۇيىلىمەن (ئۇ سېنىڭ دوستۇڭ)،مەھەللە-كويدىن ،دوسىت-بۇرادەرلىرىمدىن ئۇيىلىمەن.ياق...ياق...زېمىندىن ،ئاسماندىن ئۇيىلىمەن،كۈندىن،ئايدىن،يۇلتۇزلاردىن ئۇيىلىمەن...
  ئۆزەمنى ،ئۆزەمنىڭ ۋۇجۇدىنى،خىياللىرىمنى،چۈشلىرىمنى ،شېرىن ئارمانلىرىمنى گۇناھلىق دەپ سانايمەن.قايسىبىر ئازابلىق گۇناھلىق سائەتتە سېنى كۆرۈپ قېلىۋېدىمكىن ،بۇنى زادىلا ئەسلىيەلمەيمەن،ئەسلەشنىمۇ خالىمايمەن.
  ... سېنى سېغىنىمەن،سېنى ئەسلەيمەن،سېنى ئويلايمەن،ساڭا تەلپۈنىمەن،ساڭا زارىقىمەن... لېكىن بۇنى ئاللادىن ئۆزگە ھېچكىم سەزمەيدۇ،ھەتتا ئۆزەڭمۇ ھەم...!
    ياق...ياق ...خاتالىشپتىمەن،مېنىڭ بۇ ئىزتىراپىمنى كۈن بىلىدۇ،چۈنكى مەن ئۇنىڭدىن گۇناھلىرىمىنى تىلىگەنمەن،سېنى كۆرۈپ قالغاندا ‹‹ ھۈپپىدە››قىزارغان يۈزۈمنى ئۇنىڭدىن يۇشۇرۇشقا ئۇرۇنغانمەن.
   ئاي بىلىدۇ،چۈنكى مەن ئاشۇ ئاسمان مەلىكىسىنى ساڭا ئوخشىتىپ ئۇنىڭغا تەلمۈرگەنمەن،يازنىڭ ئىللىق كېچىلىرىدە ،دەريا بويىدىكى چىمىلىقتا يالغۇز تېرەككە يۆلىنىپ تۇرۇپ«ئون تۆت كۈنلۈك ئايىمسەن...» دەپ تەنھا ناخشا ئېيتىمەن.
  يۇلتۇزلار بىلىدۇ،مەن سېنىڭ كۆزلىرىڭنى ،ياق ئاشۇ چاراقلاپ يانغان يۇلتۇزلارنىڭ ئەڭ نۇرلۇقى ۋە ئەڭ يارقىننى سېنىڭ كۆزلىرىڭگە ئوخشاتقانمەن...
  شاماللار بىلىدۇ،چۈنكى مەن ئۇلاردىن «پەرىشتەمنى كۆردۈڭلارمۇ؟«دەپ تالاي قېتىم سورىغانمەن...
  سۇلار بىلىدۇ،چۈنكى مېنىڭ كۆز ياشلىرىم ئۇلارغا قۇشۇلۇپ ئاققان ...
  زىمىن بىلىدۇ،چۈنكى مەن ئاشۇ پاك زىمىنغا ئەرزىمىنى ئېيتقان ،«ئانا،ئانا ،دەردىمگە يەتسەڭچۇ؟!»دەپ بوتلاقتەك زارلاپ ،ئوتلۇق مەڭزىمنى ئۇنىڭ باغرىغا ياققان ... زېمىن ئانا ،مېنىڭ قانلىق يۈرىكىمدىن ئېتىلىپ چىققان نىدايمدىن تىتىرگەن ...
                                                          2
  ئۇ چاغدا سەن ئون تۆت ياش ئىدىڭ،پەقەتلا ئون تۆت ياش .ۋىجىككىنە ئىدىڭ،جۈدەڭگۈ ئىدىڭ ،لەۋلىرىڭ قانسىز،چىرايىڭ قارىمۇتۇق ۋە نۇرسىز ،مەڭزىلىرىڭ رەڭسىز ،چاچلىرىڭ رەتسىز ئىدى.كىيگەنلىرىڭ«نېرى-بېرى»لا ئىدى،سۆزلىرىڭ شىرىسىز،قاراشلىرىڭ جەزبىسىز ئىدى.قوللىرىڭ كىچىككىنە ۋە ئاۋاق (ھەتتا بەزىدە پاكىزرەكمۇ يۇيۇلمىغان !)،پۇتلىرىڭ ئۇزۇن،لىكىن ئىنچىكە ۋە ئەتسىز ئىدى.كۆكرىكىڭ سىدام،ئارقا سىنىڭ ياغاق،يەلكىلىرىڭ تارغىنە،قاش-كىرپىكلىرىڭ شالاڭ ئىدى.
   لىكىن،مەن ساڭا ئاشىق ئىدىم.
   سەن گەپ قىلغاندا دۇدۇقلايتتىڭ،لىكىن گەپدان ئىدىڭ،خۇدا،جەننەت ،ھۆر-پەرى،قۇياش،شامال،يۇلتۇز،كۆك ئاسمان ،گۈزەل ۋادا،زىلال سۇ ،بۇلبۇلگويا ،كۆك دالا ،نۇزۇگۇم ... توغرىسىدا تۈگىمەس سوئاللارنى سورايتتىڭ ،سوئاللىرىڭ شۇقەدەر سەبىي ئىدىكى،بەزىدە سېنىڭ ئەخمىقانە سوئاللىرىڭ ۋە شىرىسىز تولا گەپلىرىڭ كىشىنى بىزار قىلىۋېتەتتى.
   دادامدىن ھېيقاتتىڭ،بەكمۇ ھۆرمەت قىلاتتىڭ ،لىكىن يەنە نېمىشقىدۇر ئۇنىڭدىن قورقاتتىڭ،ئاپامغا ئۆز ئاپاڭدەك ئەركىلەيتتىڭ.گۈلنى سۆيەتتىڭ،ناخشىغا ئامراق ئىدىڭ،لىكىن ناخشىنى ياخشى ئېيتالمايتتىڭ ،ئۇسۇل ساڭا ياراشمايتتى.ئەڭ كىچىك ئوغۇل ئۇكامنى(ئۇ تېخى بوۋاق ئىدى) گاھ «ساقام»دېسەڭ،گاھ«ھەي ئاداش!»دەيتتىڭ،مېنى گاھىدا تەپ تارتماي«ئايلىقچى!»دەيتتىڭ.يەنە... يەنە...يەنە... لىكىن مەن ساڭا ئاشىق ئىدىم!
                                                              3
  مەن ساڭا ئاشىق ئىدىم ...مەن سېنى ئىچىمدە «پەرىشتە!»،«ھۆر-غىلىمان»دەيتتىم.دىلىمدا ئايەت يادلىغاندەك سېنىڭ ئېتىڭنى «ھەي قىز!»دەپ يادلايتتىم.ھەر ئاخشىمى چۈشۈمدە كۆرىمەن دەپ،سېنى ئويلايتتىم.ئەڭ يېقىملىق ناخشامنى ساڭا باغلايتتىم،ئەڭ گۈزەل خىيالىمنى ساڭا باغلايتتىم،ئەڭ چىرايلىق ئارزۇلىرىمنى سەندىن توقۇيتتىم،ئادەتتە باشقا تېمىلاردا يازما يازسام،«ئىلھام»ئىزدىسەم ،«قىز» دېگەن تېما كاللامغا كېلىشى ھامانلا ،ئىلھام دولقۇنى مېنى باغاشلايتتى،بىر خىل ئىلاھىي ناخشىنىڭ شاۋقۇنى مېنى ئورايتتى.
    باشقا تېمىلاردا يازما يازسام قولۇمغا قەلەمنى ئېلىپلا يېزىپ كەتسەم ،«قىز»دىگەن تېمىدا بىرەر نەرسە،ھەتتا بىرەر قۇر بىر نەرسە يېزىش ئۈچۈنمۇ«بىسمىللاھ»دەپ قولۇمغا قەلەم ئالاتتىم ۋە «قەلىمىمگە قۇۋۋەت بەرگەيسەن !»دەپ ئاللادىن تىلەيتتىم ...
   «ھەي قىز !»«ھەي قىز !»مەن سېنى شۇنداق سۆيەتتىم ،مەن ساڭا شۇنداق ئاشىق ئىدىم.
                                                       4
  مەن ساڭا ئاشىق ئىدىم،لىكىن مەن ئاشۇ گۇناھلىق ئىزتىراپىمدىن قورقاتتىم،ئۇيىلاتتىم،ئۆزۈمدىن سەسكىنەتتىم،چۈنكى ،سەن تېخى ئەمدىلا ئون تۆت ياش ئىدىڭ.مەن بولسام يىگىرمە ياش.

   مەن ئازابلاندىم.مەن تولغاندىم،مەن پەرياد چەكتىم.
   «ئوغلۇم!-دېدى بوۋاي بېشىمنى سىلاپ تۇرۇپ،-سەن كىمگە ئاشىق ؟»دېدى.«قىزغا...»دېدىم.ئۇيەنە «سەن كىمگە ئاشىق؟»دېدى.«قىزغا...»دېدىم .ئۇيەنە «سەن كىمگە ئاشىق؟»دېدى.«قىزغا...»دېدىم .ئۇ «ياق!»دېدى.مەن يىغلىدىم،مەن پەرياد چەكتىم،مەن زارلىدىم...
   «قۇلاق سال،ئوغلۇم،-دېدى بوۋاي قولۇمنى تۇتۇپ باش-كۆزۈمنى سىلاپ تۇرۇپ ،-‹قىز! ›دەپ ئاتىما ،ئۇنىڭ ئېتى پاك،ئۇنىڭ مەنىسى پاك،ئۇنىڭ بارىلىق پاك،ئۇنىڭ يارالمىشى پاك.ئۇقىز ئەمەس،پاكنىڭ ماددىلىشى،پاكنىڭ ئادەمىيلىشىشى،پاكنىڭ ھاياتلىششى!»
  ئۇ زاھىرەن «قىز »بولسمۇ،باتىندىن پاك،...
  مەن سورىدىم:«پاك »دېگەن نېمە؟
   بوۋاي دېدى:
   «گۈزەللىكنىڭ ئېتى پاك،پەرىشتىنىڭ ئېتى پاك،ياخشلىقنىڭ ئېتى پاك،مېھنەتنىڭ ئېتى پاك،ئىززەتنىڭ ئېتى پاك،شەبنەمنىڭ ئېتى پاك،بىخنىڭ ئېتى پاك،ھاياتلقنىڭ ئەڭ ئىپتىدايى شەكلىنىڭ ئېتى پاك،بەخىتنىڭ ئېتى پاك،خاتىرجەملىك ۋە ئاسايىش ھاياتنىڭ ئېتى پاك،ساغلام روھ ۋە ساغلام تەننىڭ ئېتى پاك،گۈزەل ئارماننىڭ ئېتى پاك،پاكىز نىيەتنڭ ئېتى پاك ،مۇنەۋۋەر مىللەتنىڭ ئېتى پاك،قۇدرەتلىك دۆلەتنىڭ ئېتى پاك،ئىلغار ۋە مەدىنىي جەمىيەتنىڭ ئېتى پاك،شەرەپلىك جەمىئيەت پۇقرالىرىنىڭ ئېتى پاك...ئىنسانىيەتنىڭ ئەڭ نادىر ،ئەڭ سەرخىل ،ئەڭ مۇنەۋۋەر ئەۋلادلىرىنىڭ ئېتى پاك... سېنىڭ مۇھەببىتىڭنىڭمۇ ئېتى پاك!»
  ئەسلىدە سەن قىز شەكلىدىكى پاكلىققا ئاشىقسەن ،ئۇنى ئون تۆت ياش دېدىڭ،ئۇ ھەتتا ئاتمىش ياش بولۇشىمۇ مۇمكىن ۋەياكى ئون ئىككى ياش بۇلۇشىمۇ مۇمكىن،ئۇپەقەت پاكلا بولسا ھەرقاچان،ھەرزامان،ھەرماكاندا سۆيۈلىدۇ،سۆيۈلدى ۋە سۆيۈلسۇن!چۈنكى ،«قىز» «قىز»شەكلىدە يەر يۈزىگە ئەۋەتىلگەن پاكلىق ئەلچىسى!!!
  ئۇ ئۆزىنىڭ ئاشۇ پاكلىقى بىلەن يەريۈزىدىكى ھەممە ئىزگۈ ھەم بۈيۈك نەرسىلەرنى ھاياتنى، تارىخنى،دەۋىرنى،جەمىئيەتنى،دۆلەتنى،مىللەتنى،گۈزەللكنى،ياخشىلىقنى،تەرەققىيات ۋە گۈللىنىشنى،ساغلام تەن ۋە ساغلام روھنى ،پاكىز نىيەت ۋە گۈزەل ئەخلاقنى ،ئىللىق ۋە بەختىيار ئائىلىنى ،بالا-بوغۇنلارنىڭ بەخىت-ئىقبالىنى _ئىشقىلىپ يەر يۈزىدىكى ئارزۇ،غايە ۋە ئارمان مەنسىدىكى جىمىكى ئۇلۇغلۇقنى تەقدىر يارتىدۇ.بىرمىللەتنىڭ ،بىر دۆلەتنىڭ ،بىر جەمىئيەتنىڭ قىزلىرىنىڭ پاكلىق دەرىجىسىنىڭ قانچىلىك بۇلۇشى شۇ مىللەتنىڭ ،شۇ دۆلەتنىڭ ،شۇ جەمئيەتنىڭ تەقدىرىنى،ئىستىقبالىنى ۋە ساپاسىنى بەلگىلەيدۇ.قايسى بىر مىللەتنىڭ زەئىپلىكى ،قايسى بىر دۆلەتنىڭ ئاجىزلىقى ،قايسى بىر جەمئيەتنىڭ چۈشكۈنلىكى مەلۇم مەنىدىن ئالغاندا شۇ مىللەت ،شۇ دۆلەت ۋە شۇ جەمئيەت قىزلىرىنىڭ پاكلىق دەرىجىسىنىڭ تۆۋەنلىكىنڭ ئالامىتىدۇر.ئۇلارنىڭ گۈللىنىشىدە بولسا ،ئاشۇ قىزلارنىڭ تۆھپىسى مۆلچەرلىگىسىزدۇر.جەمئيەتنىڭ مۇقىم، ئائىلىنڭ ئىناق ۋە كۆڭۈللۈك ،پەرزەنتلەرنىڭ بەختىيار ۋە ئىستىقباللىق بۇلۇشى ۋە ياكى ئۇنىڭ ئەكسىچە بۇلۇشى ھەم ئاشۇ قىزلارنىڭ نامى _«پاكلىق »بىلەن چەمبەرچاس باغلانغان،ئۇنىڭ ئەكسىچە بولۇشىمۇ ھەم شۇنداق!
   مۇشۇ مەنىدىن ئالغاندا «قىز» مىللەت ئالدىدا،جەمئيەت ئالدىدا ،ئائىلە ئالدىدا ،تارىخ ئالدىدا ،دەۋىر ئالدىدا باش تارتىپ بولمايدىغان ئېغىر بۇرۇچ ۋە مەسئۇليەتنىڭ ئىگىسىدۇر!ھەتتا ھاياتلىقنىڭ ھالاكىتى ۋەياكى ئۇنىڭ ئۈزۈلمەي داۋام قىلىشنىمۇ ئاشۇ «قىز»دەپ ئاتالغانلار ئۇچلاپ تۇرالايدۇ!
   شۇڭا ،ئوغلۇم،سېنىڭ ئۇنىڭغا ئاشىق بولغىنىڭ گۇناھ ئەمەس،گۇناھلىقى ئۆزۈڭنىڭ قىزنى ئەمەس،پاكلىقنى سۆيۈۋۈتقانلىقڭنى،پاكلىقنى قەدىرلەۋاتقانلىقىڭنى ۋە ياكى قىزنى پاكلىق دەپ تۇنۇماي،ئۇنى پەقەت ماددىي ھالدىكى قىز دەپ تونۇپ ئۇنىڭغا ھاۋايى-ھەۋەس ۋەياكى ھەرقانداق باشقا غەيرىي نەزەردە تىكىلىپ قارىغىنىڭدۇر!
