نۇپۇس سانىنى مەن تىخى ئىشىپ كەتسە يەھۇدىيلارنىڭ نۇپۇسى بىلەن تەڭ چىقار دىسەم ئوتتتتتۇز ئالتتىمىليۇن دەيدا.شۇنداق بولغاندا ۋىكپىدىيەنىڭ سىتاتسىتكىسى بۇيىنچە نۇپۇسىمىز ئۆزبىكىستان دۆلىتىنىڭ نوپۇسىغا تەڭ كىلىپ قالامدۇ ھە.بىر چاغدا ئۆزبىكىستان نوپۇسىنى ئوتتۇز نەچچە مىليۇن كۆرگەندەك يادىمدا.
شۇنداق بولغاندا ئۇيغۇر نوپۇسىنىڭ ئاساسى گەۋدىسى شىنجاڭ بولماي قالار.دۆلىتىمىز نوپۇس سىتاسىتكىسىدا 2012- يىللىقمۇ ئىشقىلىپ،ئۇيغۇر نوپۇسى ئون بىر مىليونغا يىتەركەن.دۇنيادىكى نوپۇسى ئون ئۈچ مىليۇنلار ئەتراپىدا ئىكەن.ئەمدى ئۇنى بۇنى دىگەن بىلەن قىرىنداشلار،كۈندە قانچىلىك بوۋاق تۇغۇلىدۇ.قانچىلىك نوپۇس ھەر خىل سەۋەب بىلەن قازاغا ئۇچرايدۇ بۇنى بىرمۇ بىر ئىنىقلاش تەس بىر گەپ.نوپۇسنى ھەر قايسى ئىجتىمايى رايون ساقچىسى ئىنىقلايدۇ.بۇ بىر ناھىيە ھەر بىر شەھەر ھەر بىر ۋىلايەت ئوبلاسىتلارغا قەدەر ئىنىقلىنىدۇ.بۇ چىققان نەتىنجە ئاپتۇنۇم رايۇنلۇق نوپۇس خىزمىتى ئىشخانىسىغا،ياكى ھەر قايسى ئۆلكىلىك نوپۇس پىلانلىق تۇغۇت ئىشخانىللىرىغا يوللىنىدۇ.ئاپتۇنۇم رايون ياكى ئۆلكە ۋە مەركەزگە بىۋاستە قارايدىغان شەھەرلەر تەۋەيىدىكى نوپۇسنى پۈتۈن ئىنىقداپ بولدۇق دەپ قارىغان چاغدا ئاندىن ھەر قايسى ئۆلكە ئاپتۇنۇم رايۇنلار مەركەزگە دوكىلات بىرىش ئۇچۇرىنى ئەۋەتىدۇ.مەركەزنىڭ نوپۇس ئىشلىرىغا مەسئۇل باشلىقى دوكىلات ۋە سان سىپىرلارنى كۆرۈپ مەملىكەتلىك نوپۇس خىزمىتى يىغىنى ئاچىدۇ.بۇ يىغىنغا مەملىكەتكە قاراشلىق بارىلىق ئۆلكە ئاپتۇنۇم رايونلارنىڭ ۋەكىللىرى قاتنىشىدۇ.دوكىلات بىرىدۇ.ئۇزاق ئۆتمەي بۇ يىغىن ھەر قايسى تاراتقۇلارنىڭ ياردىمى بىلەن ئىلان قىلىنىدۇ.ئاندىن مەركەز ھەر قايسى ئۆلكە ئاپتۇنۇم رايۇنلارغا بىرىدىغان تولۇقلىما بولامدۇ ياكى شۇنچىلىك نوپۇسقا كىتەرلىك ماددىي بۇيۇم كىيىنچە ئىش ئورىنى ۋە باشقا سىياسەت تۈزۈيدۇ.دىمەك نۇپۇسلا ئەمەس مەدىنىيەت،تىببىي سەھىيە،قۇرلۇش،پۇل موئامىلە قاتارلىقلارنىڭ ھەممىسى مەركەزنىڭ ئەلگىكىدىن ئۆتىدۇ.بۇنى ئەلۋەتتە ھەر قايسى يىزا بازارلىق ئاساسى قاتلام ئىشچى خىزمەتچىلىرى ۋە شۇ خىزمەتكە مەسئۇل تارماقلار مەسئۇل بولىدىغان گەپ.
ئۆزىڭىز ياخشى كۆرگەن ئادەمنىڭ ئېللىتى خېسلەت بېلىنسە،ئۆزىڭىز يامان كۆرگەن ئادەمنىڭ خېسلىتى ئېللەت بېلىنىدۇ.

تورلاردا قاراڭ،بىرى كۆرمىدى دەپ بولسىمۇ ئىككى ئىغىز گ ...

ئالدى بىلەن ئىنكاسىڭىزغا رەھمەت.پىكىرلىرىڭىز ئۇرۇنلۇق. لىكىن مەن نۇقۇل توردىكى مۇشۇ ئەھۋالنى ئەمەس بەلكى پىكىر قىلىش يولىنى تەكىتلىمەكچى.«تۇخۇ بالدۇر پەيدا بولغانمۇ تۇخۇممۇ»دىگەندەك ئايىغى چىقمايدىغان نەتىجىسىنى چىقارسىمۇ بىزگە ھىچ پايدىسى يوق ئىشلارنى تەپەككۇر قىلغاننىڭ ئورنىغا ئىلمىي تەپەككۇر قىلىپ ئۆزىمىزنىڭ ھالىتىنى ئۆزگەرتىشنى بەكراق ئويلاشساق دىمەكچى.بىر پەلەكتە ھەقىقەتەن مىڭ خەمەك بولىدۇ.خىلمۇ خىل ئادەملەر بولىدۇ.بولسا مۇشۇ خىلمۇ خىل ئادەملەرنىڭ بىر ئورتاق ئۆگىنىش نىشانى بۇلۇشنى ئارزۇ قىلىپ يازدىم.چۈنكى ئۇيغۇرلار بۇ دەرىجە بىلىمگە تۇيۇنۇپ مۇشۇنداق ئىشلارنى تەپەككۇر قىلىدىغان باسقۇچقا يەتمىدى. تەپەككۇر قىلغاندا قىممىتى بار ئىشلارنى تەپەككۇر قىلساق.ئەھمىيەتسىز ئىشلاردىن ئۆزىمىزنى قاچۇرساق.نۇرغۇن كىشىللىرىمىز مۇشۇنداق قىلىۋاتۇىدۇ.بۇ ھەقتە مىسالمۇ قىلىپ بەردىم.مانا مۇشۇلاردىن ئۆگىنىپ بىر بالداق ئاتلىغانلار يەنە بىر بالداق ئاتلاشنى بىلسە نىمە دىگەن ياخشى ئىش ھە.قورسىقى تويمىغان ئادەم كىيىمنىڭ غېمىنى قىلمايدۇ.قورسىقى تۇيۇپ كىيىمى پۈتۈن بولمىغان ئادەم ئۆي شارئىتىنىڭ غېمىنى قىلمايدۇ.ئەمدى بۇ ئۈچ شارائىت ھازىرلانغان ئادەم تۇرمۇش سۈپىتىنى يۇقىرى كۆتىرىشنى ئويلايدۇ.بىلىم ئىگەللەشتىمۇ مۇشۇنداق ئاڭ بولسا نىمە دىگەن ياخشى. قىمىمىتى بار ئىشلارغا باش قاتۇرماي ئايىغى چىقمايدىغان ياكى ھىچقانداق ئەھمىيتى يوق ئىشلارنى تەپەككۇر قىلىش خۇددى كىيىمى يوق يالىڭاچ تۇرۇپ  داچىدا ئولتۇرۇشنى ئويلىغاندەك ئىش. تەپەككۇرنىڭ بولغىنى ھەقىقەتەن ياخشى ئىش.بولسا كىچىك ئىشلاردىمۇ تەپەككۇرنىڭ ئەھمىيتىنى قىممىتىنى ئويلاپ قىلساق بۇ نىڭ ئۈنۈمى ھەقىقەتەن چوڭ بولغان بولاتتى.بۇنى ئەنە ئاشۇ ئوغلان قىزلىرىمىز قىلالىدى. قىلالمىغانلار كىملىكىنى ۋاقتىدا تېپىۋىلىپ قىلىشنى ئۆگەنسە دىمەكچى.ئاخىرىدا يەنە بىر قتېىم ئىنكاسىڭىزغا رەھمەت.

مەنچىمۇ مۇشۇنداق ھەق سۆزلەرنى قىلىدىغان بەزى كاللىسىدە ئۆزىگە ياكى ئۆزلۈككەر يات نەرسىلەر توشۇپ كەتكەن .مېللەتسىزلىقنى ،غۇرۇرسىزلىقنى ،ئىتىقادنى خەلقئارالاشتۇرۇشنى تەكىتلەپ ئاممىۋى ھۇجۇم قىلىشقا ئامراق كىشىلەرگە زادى كىم ،قانداق خەق ئۇيغۇر ئىكەنلىكىنى بىلدۈرىدىغان ۋاقىت كەلدى .ئەگەر يەنىلا مۇشۇنداق يىرىم ئۆزىمىزنى تاشلاپ زەئىبلىشىپ ،يۆلىنىش سىزىمىمىزنى يوقىتىپ ،قارىغۇلارچە پىتنە پاسات ،بىدئەت ۋە چىكىدىن ئاشقان ئىدىيەلەر بىلەن جەمىئىتىمىزنى خەلقىمىزنىڭ ھۆرۈپ-ئادەت ۋە مەدىنىيەت-سەنئەتقاتارلىقلارنى ئۆلچەۋەرسەك بىزمۇ تۈگىشىمىز ھەم باشقىلارنىڭ قول سىلىشىغا باھانە -سەۋەب،شارائىت يارتىپ بىرىمىز (ھازىرقى بىرقىسىم ئىشلارغا ئوخشاش .ھازىر دەل بىزنىڭ كىيىنكى تەقدىرىمىزنىڭ قانداق بولىشىغا ئالدىن بىشارەت بولىۋاتقان بولىشى مۇمكىن).
       ئۆز مېللىتىنىڭ تەۋەلىكىنى ياكى مېللىتىنى يوقاتقان ھەرقانداق كىشى ياكى شۇ مېللەت ئەزاسى ئۆزىنىڭ ئىككىنجى سالاھىتى (بەزىلەرنىڭ دىنىي سالاھىتى)بىلەن سۆز قىلىش ،ياشاش ۋە ئىنسانىي ھوقۇق ،تەربىيلىنىش قاتارلىق ئەڭ ئاددى ئىنسان ئىرىشىشكە تىگىشلىك سالاھىيەتتتىن بەھىرلىنەلمەيدۇ .
      ھەرقانداق دۆلەتنىڭ ئەڭ ئاددى شەخسى ئۇچۇر تېزىملاش جەدىۋىلىدە
ئىسىم -فامىلە،جىنىس ،ياش،مېللەت دىگەن ئۆزلۈك ئۈچۈر بولىدۇ .مانا مۇشۇنىڭدىن بىلىۋىلىشكە بولىدۇكى .ئۆزلۈك ئىنتايىن مۇھىم .كىملىكىڭىز سىزنىڭ ياشاش ھوقۇقىڭىز ،ئەكىنلىكىڭىز !

دۇنيادىكى ئەڭ خۇشپۇراق ئەتېر-مىھنەتتىن چىققان تەرنىڭ پۇرىقىدۇر!
