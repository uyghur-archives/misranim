                                                              قوشاقلار

                                                 قارا يۇلغۇن كىسىپ قويدۇم ،
                                                 قۇرىسا قالارمەن دەپ .
                                                 ئونبەش ياشلىق يار تۇتتۇم ،
                                                 چوڭ بولسا ئالارمەن دەپ .

                                                 قۇچقاچ قوندى قامغاققا ،
                                                 قارلىغاچ قوندى يانتاققا .
                                                 چىرايلىقنى بىر سۆيسەم ،
                                                 ئۇيۇپ قالدى تامغاققا .

                                                  سۇنى سەپتىم سۇپىغا ،
                                                  ئىلىۋالدىم قۇتىغا .
                                                  يارىم كەتتى يۇرتىغا ،
                                                  مىنى قويدى ئوتىغا .

                                                  ئىشىك ئالدى قوغۇنلۇق ،
                                                  سويما چۇشتى پىلەككە .
                                                  سۆيۇپ قوياي دىمەيلا ،
                                                  تاش سالغانمۇ يۇرەككە .

                                                  سىم سىم يامغۇر تۆكۇلەر ،
                                                  كۆز يىشىدەك توختىماي .
                                                  مەن يارىمنى سىغىندىم ،
                                                  كىچە - كۇندۇز ئۇخلىماي .

                                                  ئالا ئات مىنىڭ بولسا ،
                                                  كەڭ دالادا چاپتۇرسام .
                                                  قارا كۆزنى كۆرگىچە ،
                                                  بۇ دۇنيادا ساق تۇرسام .

                                                  ئاق قۇچقاچ - سىرىق قۇچقاچ ،
                                                  باغدا قالدى ئەمەسمۇ .
                                                  بىزگە كۆيگەن شۇ يارلار ،
                                                  داغدا قالدى ئەمەسمۇ .

                                                  تومۇچۇقۇم بار بولسا ،
                                                  ئالقىنىمدا دان بەرسەم .
                                                  ۋاپادار يارىم بولسا ،
                                                  قۇچىغىدا جان بەرسەم .

                                                 مەن يارىمغا سۇ بەرسەم ،
                                                 ھەسەل  دەپ يۇتار .
                                                 ياشلىقتا سۆيگەن يارىنى ،
                                                 كىممۇ ئۇنىتار .

                                                 ئۆستەڭ بويىدا تۇرۇپ ،
                                                 سۇغا قانمىغان تىرەك .
                                                 يارنىڭ قىشىدا تۇرۇپ ،
                                                 يارغا قانمىغان يۇرەك .

                                                 قويلىرىڭ بارمۇ سىنىڭ ،
                                                 تاققا چىقىپ  قوي باققىلى .
                                                 يارى يوققا يار بولۇپ ،
                                                 بىر ئىرىقتا ئاققىلى .

                                                 چۇشۇمگە كىرىپ قاپلا ،
                                                 كۆڭۇللىرى تۇيدىمۇ ؟
                                                 بىزنىڭ يۇرەك بەك كۆيدى ،
                                                 سىلىنىڭ يۇرەك كۆيدىمۇ ؟

                                                 ئوسمىنى قاشقا قويۇڭلا ،
                                                 خىنىنى تىرناقىغا .
                                                 يامان خوتۇننى چىڭ تۇتۇڭلا ،
                                                چۇشۇرمىسۇن قىلتاقىغا .

                                                 ھويلاڭدىكى ئۇجمەڭنىڭ ،
                                                 بىر شىخى راۋاپ بولدى .
                                                 سەن يارىمنىڭ ئوتىدا ،
                                                 يۇرىكىم كاۋاپ بولدى .

                                                 بىغىڭغا كىرەر بولدۇم ،
                                                 ئالما بارمۇ ئانارمۇ .
                                                 تىۋىپتىن سوراپ بىقىڭ ،
                                                 كۆيۇككە دورا بارمۇ .

                                                 ئاق قۇشقاچ ھارامزەدە ،
                                                 جاڭگالدا بىنەم يەردە .
                                                 ئوينىغىلى قويمامسەن ،
                                                 كۆڭلۇم خالىغان يەردە .

                                                ئىلىخونىڭ يوللىرى ،
                                                 ئەگرى - بۇگرى توقايلىق  .
                                                 سۆيگەن يارىم ئانارگۇل ،
                                                 ماڭا بەردى قول ياغلىق .

                                                 ئىشىك ئالدىڭدىن  ئۆتسەم ،
                                                 ئىتىڭنى قويۇۋەتتىڭ .
                                                 نىمە قىلدىم يار ساڭا ،
                                                 گىپىڭنى ئۇزۇۋەتتىڭ .

                                                 باغ ئارا باغنىڭ ئوتى ،
                                                 ياندى يۇرەكنىڭ ئوتى .
                                                 كۆيگەن يۇرەككە شىپادۇر ،
                                                 يىراقتىكى يارنىڭ ئوتى .

                                                 يىنىپ تۇرغان چىراققا ،
                                                 پەرۋانە ئۇچۇپ كەلدى .
                                                 مىنى تاشلاپ كەتكەن يار ،
                                                 قىشىمغا كۇلۇپ كەلدى .

                                                 ئۇرۇكۇڭدىن تاتلىقكەن ،
                                                 ئىچىدىكى مىغىزى .
                                                 شىكەر تىتار ، قەن تىتار ،
                                                 سەن يارىمنىڭ ئىغىزى .

                                                 باققا بۇغداي تىرىسام ،
                                                 تىڭى بۇدۇشقاق بولغۇدەك .
                                                 يىنىدا پۇل بولمىسا ،
                                                 يارى ئۇرۇشقاق بولغۇدەك .

                                                 بىغىڭدىكى ئالماڭنىڭ ،
                                                 شىخى ئىگىلىپ قاپتۇ .
                                                 ساڭا نىمە دەرت كەلدى ،
                                                 چىرايىڭ سارغىيىپ قاپتۇ .
