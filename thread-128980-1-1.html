ئەسسالامۇ ئەلەيكۇم تورداش ،مېنىڭ ماقالە يېزىشتىن بۇرۇن تىلەيدىغىنىم تور بەتنىڭ ۋاراقلىرنى زىيان قىلىپ ھېچقانداق قىممىتى يوق نەرسىلەرنى  يېزىپ قېلىشتىن ساقلىنشتۇر.شۇڭلاشقا بۇ ئويلىرىمنى ماقالە قىلىپ يېزىشتىن بۇرۇن سىلەرنىڭ مېنى چۈشنشىڭلارنى ۋە تەنقىدىي پىكىر بېرشىڭلارنى سەمىمى ئۈمىد قىلىمەن.

بىزگە كىرىكى ساغلام مىللى روھىمىزدۇر.

ئاپتورى:پەرھات ئەمەت


   ئۇيغۇرلار ئۈچۈن ئەزقاتىدىغان نەرسنىڭ كۆپ خىل بولۇشى تەبئىي بۇنىڭغا گەپ كەتمەيدۇ.ئۆتمۇشنى ۋە يىراق كەلگۇسنى ئەمەس دەل ئۆزىمىز ياشاۋاتقان نۆۋەتتە رىئاللىقىمىزنى چىقىش قىلىپ ئۆزىمىزنىڭ تەرەققىياتى ھەققىدە باش قاتۇرۇش،ئويلىنىش ۋە ئويلىغانلىرىمىزنى كۆپچىلىك بىلەن ئورتاقلىشىش ھەربىر ئەقىل ئىگسىنىڭ بۇرچى.دېمسىمۇ مۇشۇ ئەسىردە بىز باشقىلارغا قارىغاندا ئەڭ كۆپ ئويلاندۇق، بۇنىڭمۇ پايدسى چوقۇم بولىدۇ. ئۆزىمىزگە نەزەر سالساق مەدەنىيەت كېسىشكەن ‹يىپەك يولى›  دەپ ئاتالغان بەلباغنىڭ ئوتتۇرسىدا تۇرغانلىقىمىز سەۋەبىدىن ‹دېرىزلىرى سىرتقا ئېچىۋىتىلىگەن›  ،تۆت تەرەپتىن ھەرخىل مەدەنىيەت يېڭلىقلىرى،ئامىللىرى قاپسىاپ كىرىپ تۇرىۋاتقان بىر دەۋرىردە تۇرۋاتىمىز.مەن تارىخىمىزغا ئائىت كىتابلارنى كۆرگىنمىدە بۇخىل ئۆز ئارا گىرەلشىپ   ئارلىشىپ كەتكەن كۆپ خىل مەدەنىيەتنى تارىختا ئۆتكەن ھادىسە دەپلا ئويلاپ كەلگەن ئىكەنمەن. لىكىن كىتابتىن نەزەرنى ھايات ياشاۋاتقان مۇشۇ رېئاللىقىمىزغا تاشلاپ تەپەككۇر قىلغان چېغىمدا تارىختىنمۇ يۇقىرى دەرىجدىكى مەدەنىيەتلەر گىرەلەشكەن زامان دەل ئۆزەم ياشاۋاتقان مۇشۇ ئەسىر ئىكەنلكىنى ھېس قىلماقتىمەن.جۇغراپىيەلىك ئورنىمىز ئۆزگەرمىگەن ئەھۋالدا  بۇ زىمىندا ئاپىردە بولغان چېغىمزدىن تارتىپ بۇخىل كۆپ قىرلىك تەقدىر قىسمەت بىزگە نېسپ بولغان ئىكەن. ئۇنداقتا بۇ بىز ئۈچۈن  ئامەتمۇ؟ياكى ئاپەت بولغانمۇ؟
   چۈنكى بەزىدە <قازانغا يولۇقساڭ قارىسى يۇقار> دېيلسە بەزىدە <ئالتۇن ياتىدۇ سايدا بايقىمساڭ نېمە پايدا؟>دېگەندەك ھېكمەتلەر  ئادەمنى پىكىر قلىشقا ئۇندەيدۇ .بەلكىم بۇنى تەڭرىنىڭ بىزگە بەرگەن ئىلتىپاتى دەپ چۈشەنسەك توغرا بولار، چۈنكى بىر خىللا مەدەنىيەت بىلەن چىگرالانغان رايونلارنىڭ تەرەققىياتى بىلەن كۆپ خىل مىللەت ۋە كۆپ خىل مەدەنىيەتلەر گىرەلەشكەن جايلارنىڭ پەرىقى چوڭ بولىدۇ. ئالدىنقىسىنىڭ ھاياتى كۈچى ،قارشلىقى كۈچى يوق بولىدۇ.ئەنئەنە دېگەننىڭمۇ ئانچە رولى ،ئەھمىيتى بولمايدۇ.
بىزنىڭ ئەقىللىق ئەجدادلىرىمىز دۇنيانىڭ يۈرىكىدەك  ياكى ئۇنىڭ كىندىكىدەك مۇشۇنداق بىر زىمىندا ياشاپ،شۇنچە كۆپ خىل تىل ۋە دىنلارنى ئۆزلەشتۇرۇپ تۇرۇپ،شۇنچە كۆپ مىللى قوشۇلۇشلارنى باشتىن كەچۇرۇپ تۇرۇپ،شۇنچە كۆپ يوقىتىشلارنى ئۆتكۈزۇپ  تۇرۇپ يەنىلا مۇشۇ زىمنىدا قەد كۆتۇرۇپ كەلدى. بەزىدە ئۆزلىرنىڭ سەلتەنەتلىرىنى ياراتقان بولسا گاھى تەكلىماكاندىكى قۇم بورانلاردەك ئۇچۇپ  يۈردى.شۇنداقتىمۇ ئۆزلىرىنىڭ كىملىكىنى يوقاتماي ھەرقانداق  چاغدامەۋجۇتلۇقىنى ساقلاپ كەلدى.
مۇشۇ ئەسىردە بىز ئۆزىمىز ھەققىدە تەرەققىياتىمىز ھەققىدە ھەممىدىن كۆپ ئويلاندۇق .  نۇرغۇن مەسىىلىلەرنى ئوتتۇرغا قويدۇق.ئۆزىمىزگە كېرەكلىك بولغان نەرسىلەرنى بىر بىرلەپ ئوتتۇرغا قويدۇق.مەيلى بىزبۇ ھەقتە قانچىلىك مۇنازىرە قىلايلى ياكى مۇلاھىزە قىلىپ  ياقا سىقشىپ كېتەيلى  ئۆزگىچە، يېڭى كۆز قاراشلارنى ئوتتۇرغا قويايلى ھەممسىلا ئەڭ كۈچلۈك ھەقىقەتنىڭ سىناق مەيدانى ___ ئەمەلىيەتنىڭ سىنقىدىن ئۆتۈشى كېرەك.ئەمەلىيەتچانلىقى بولمىغان ھەرقانداق نەرسە خام خىيالدىن ئىبارەت.
ھەقىقەتنى ئەمەلىيەتتىن ئىزدەش دىگەنلىك ئادەتتىكى بىر گەپ ئەمەس بەلكى ئۇ تولىمۇ تىرەن پىكىر بار كۆز قاراشتىن ئىبارەت.بۇنى ھەرقانداق مىللەت ئۆزنىڭ تەرەققىياتىدا ئۆلچەم قىلسا بولىدىغان ئاساس دەپ قارساقمۇ بولىدۇ.
تارىخنى ئىسمىزدە چىڭ تۇتۇشقا توغرا كېلىدۇ. ئۇ تەجرىبە ساۋاق ۋە ئەسلىمدىن ئىبارەت.بىزنىڭ ۋەزىپمىز ۋە بۇرچىمىز بولسا يىڭى دۇنياغا يەنى  ھازىرغا يۈزلىنىش،مەيلى ھالىمىز قانچلىك بولۇشىمىزدىن قەتئىنەزەر ئۆزىمىزنى ساقلاش ۋە تەرەققىياتتا دۇنياغا يۈزلىنىش بىزنىڭ ئاداققى نىشانمىز بولۇشى كېرەك.
يېڭى دەۋر ھەرقانداق مىللەتكە ھەممىنى تەييار تۇتقۇزۇپ قويمايدۇ.ئۇنىڭغا تىرشچانلىق،ئەقىل ۋە ئويغۇنۇش كېرەك.
مۇشۇلارنى ئويلىغان چېغىمىدا ئۆز مىللتىمىزنىڭ تەرەققىياتى زادى نېمگە مۇناسىۋەتلىك دېگەن بىرسوئالنى سورىغىم كەلدى. بۇنىڭ ئۈچۈن ئىككىنچى دۇنيا ئۇرۇشىدا ئېغىر دەرىجىدە مەغلۇپ بولغان گىرمانلار بىلەن ياپونلارنى تىلغا ئېلىپ ئۆتۈپ كېتشكە بولىدۇ. بۇلارنىڭ ھەر ئىككلسلا قايتا باش كۆتۇرۇپ كېتەلىشنى ھەرخىل سەۋەبلەردىن تەتقىق قىلىنسمۇ لىكىن بىر سەۋەب ھەممىدىن مۇھىىم يەنى ھەر ئىككىلا مىللەت ئۆتمۈشنى ئۇنتۇپ ھازىرنى ئۆزلىرى مەۋجۇت بۇلۇپ تۇرىۋاتقان رېئاللىقنى ئۆزگەرتىشكە كۈچىگەن.بۇخىل ئەمەلىيەتچان روھ ئۇلارغا كۈچ مەدەت بېرىپ دۆلەتنى قايتا ئەسلىگە كەلتۇرۇش، مىللەتنى دۇنيادا تەرققىياتتا ئالدىنىقى قاتارغا ئۆتۈپ  كېتىشكە يول ئاچتى. بۇنداق ساغلام مىللى روھ مىللەتنىڭ ئىشەنچە ۋە جاسارىتىنى ئۇرغۇتىدۇ،
ھەممە ئادەم  ئۆز خىزمتىنى ئەستايدىل بىجاندىل ئىشلەيدۇ،بۇنداق ئەھۋالدا تەرەققى قىلماسلىق مۇمكىنمۇ؟
ئۆز گېپمىزگە كەلسەك،بىز ئۇنى دېدىۇق بۇنى دېدۇق،ھېلى بۇنىڭغا ئېرشسەك ھېلى بۇنىڭغا قولىمىز يەتمەي قالدى،ھېلى  بىر يوللاردىن بىر ئىشلاردىن چەكلەنسەك ھېلى يەنە بىر  بوشلۇق ئورۇن  بىزنى ساقلاپ تۇرۇپ قالدى....بىرسىگە قايغۇرۇش،بىرسگە ئارمان قىلىش،بەزى مەسىلىلەرنى يۇشۇرۇش  ۋە پەرىق ئېتەلمەسلىك سەۋەبىدىن قايمۇقۇپ قېلىشلار بۇ ئەسىردە بىزدە مەۋجۇت ئىشلار بولدى.
رېئاللىقنى ھەقىقى ئۆزگەرتىشكە كېرەك بولدىغان نەرسە ئەمەلىيەتتە ئەقىل - پاراسەت،ھەرقانداق ۋاقىتتا پۇرسەتتىن پايدىلىندىغان ئەقىليەتچان روھتا. بۇ خىل روھ قېينچلىق ئالدىدا بوي ئەگمەيدىغان،قايغۇغا ۋە يوقىتشلارغا سەبىرى بېرەلەيدىغان كۈچلۈك ئېنىرگىيەدىن ئىبارەت.بۇ چوڭ جەھەتتە مىللى روھىمىزغا بېرىپ تاقىلىدۇ.  . كۆپچلىك كىشلەرنىڭ  توي مەشرەپلەردە ئەمەس بەلكى مىللەتنىڭ شەرەپكە ئېرشكەندىكى خوشاللىقىغاجۆر بولۇشى،نەزىر چىراقلاردا ئۆزىنىڭ قېرىنداشلىرى ئۈچۈن ئەمەس بەلكى مىللەتنىڭ قايغۇسغا ئورتاقلىشدىغان بىر خىل روھنىڭ بارلىقنىڭ ئۆزىلا مىللى روھنىڭ ئىپادىسى. ئۆز ئارا خەيرى ساخاۋەتلىك قىلدىغان  شەخس ۋە كوللكىتىپلارمۇ  مىللەت ئىچدىكى ئاكتىپ ئىجابى تەرەپلەرگە ياردەم بېرىدىغان فۇندى جەمئىيەتلەرنىڭ قۇرۇلۇشىمۇ بىرخىل مىللى روھنى ئەكس ئەتتۇرىدۇ.
مىللەتنىڭ تەرەققياتنى مىللى روھقا باغلاپ  چۈشەندۇرۇشتىكى سەۋەبمۇ مىللى روھ قانداقتۇر دوكتور ئاسپرانت،ئاكادىمك ،زىياللار ئۈستىگە ئالغان ياكى بىر قانچە كاپىتالى بار بايلار ئۈسىتگە ئالغان ئىش ئەمەس. بەلكى ئەڭ ئاددى ئاۋامدىن تارتىپ يۇقىرى مەدەنىيەت سەۋىيىرىمىزگە ۋەكىللىك قىلغۇچى كشلىرىمىزگە قەدەر  ئورتاق بولغان پىداكارانە ،ئۆزىنى بېغىشلاش،جاپا تارتىشلاردىن قورۇقمايدىغان خىزمەت قىلىش ،ئەمگەك قىلىش روھى دېگەنلىك. كۈچ ئۇلاپ يۈگۇرۇش شەكىلىدىكى تەرەققىياتىمىز ئىزچىل ھالدا ئۆز كۈچىمىزگە تاينىپ ئاستا ئاستا بارلىققا كەلمەكتە.
بىز ياراتقان قىممەتلىرىمىزنىڭ ئىچىدە ئۆزىمىزگە تەۋە ئامىللارنى ساقلاشقا تىرشساقلا، ھەمدە ئىجادىكارلىققا باي يېڭلىىق يارىتىشتىن ئىبارەت بىردىنبىر يولنى تاللاپ داغدام ماڭالىساقلا بۇ دەۋردە ئۇتۇپ چىققۇچىلار قاتارىدىن ئورۇن ئالالايمىز.
پەقەت ئېڭىمىزدا ھەرقايسىمىز ئۆزىمىز قىلىۋاتقان ئىشلاردا چوقۇم نەتىجە قازىنشنى ئويلىغاندا ۋە بۇ نەتىجلەر ئارقىلىق مىللىتىمزنىڭ تەرەققىياتىغا مىسقالچلىك بولسمۇ تۆھپە قوشىمەن دەيدىغان ئىرادە بولغاندا ۋە ھەرۋاقىت باش چۆكۈرۇپ ئىشلىگە چېغىمىزدا بىز ئۆزىمىزنى  مىللەتىنىڭ  مىللى روھىنىڭ ئىچىگە  ئېلىپ كىرەلەيمىز.
ئەگەر بىر ئوقۇغۇچى بولسىڭىز مەكتەپتە چوقۇم  تىرىشىپ ئۆگىنىپ كەلگۈسىدە دۆلەت ئىچى ۋە چەتئەللەرگىچە ئوقۇپ كېلىڭ، ھەمدە قايتىپ كېلىپ  مۇنەۋۋەر كەسپ ئەھلى بولۇڭ!
ئەگەر سىز بىر دوختۇر بولسىڭىز ئۇ چاغدا نامى تارالغان ئەڭ ئۇستا ۋىراچ بولۇڭ!
ئەگەر سىز قول ھۆنەرۋەن  بولسىڭىز ئاتا بۇۋىلاردىن قالغان بۇ تەۋەرۇك ھۆنەرنى يىڭى دەۋرگە ماسلاشتۇرۇپ بىز ئۇيغۇرلارنىڭ ئىجاد قىلش ۋە سەنئەت ئالاھىدىلىكمىزنى باشقىلارھەيران قالغۇدەك دەرىجىدە كۆرستىپ قويۇڭ!
مەيلى سىز كىم بولۇڭ ناۋادا ئۆزىڭىزنى مەن بىر شەخس پەقەت  ئۆزەمگىلا تەۋە دەپ قارىمسىڭىز ئۇنداقتا سىزدە كۈچلۈك بولغان مىللى روھ ئاپىردە بولغان بولىدۇ.ھەمدە ئىشلارنى ئەڭ ياخشى تەرتىپتە راۋاجلاندۇرغان جاسارەتكە ئىگە بولسىىز.ھەر قىېتىم سەھەر ئورنىڭىزدىن تۇرۇپ شەرققە قارىغان چېغىڭىزدا،قۇياشنىڭ ئاستا ئاستا كۆتۇرۇلگەنلىكىدەك مەنزىرنى كۆرگەن چېغىڭىزدا ئاشۇ قۇياش ماڭا مەنسۇپ مەن مىللىتىم ئۈچۈن  خىزمەت قىلدىغان يېڭى بىر كۈننى كۈتىۋالدىم دەپ خوشاللىنىڭ!
