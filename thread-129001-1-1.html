ئەسسالامۇ –ئەلەيكۇم تورداشلار ،ھەممىڭلارنىڭ مۇشۇ ۋاقىتى خەيرىلىك بولسۇن .مەن ئالدىنقى قىتىم بۇ مىسرانىم مۇنبىرىغا  تاللاش تىرىشىشقا قارىغاندا تىخمۇ مۇھىم  دىگەن كىتاپنىڭ تەرجىمىسنى يوللاپ كەڭ تورداشلارنىڭ قوللاپ قۇۋۋەتلىشى ۋە قىزغىن ئالقىشىغا ئىرىشتىم .مەن بۇنىڭدىن ياخشى   ئىلھام ئىلىپ  مىسرانىم مۇنبىرىگە تىخمۇ سەر خىل بولغان ئەسەرلەرنى يوللاش توغرىسىدا خىلى ئىزدىنىپ باقتىم .بىز ياشاۋاتقان كەڭ جەمىئەتكە نەزەر سالدىم .جەمىئەتنى ،كىشلەر ئوتتۇرسىدىكى مۇناسىۋەتنى ئىنچىكىلىك بىلەن كۈزەتتىم .
بىزدە ئادەم ئادەم بىلەنلا ئادەم دەيدىغان گەپ بار ،بىزنىڭ ياشىشىمىز ،جەمىئەتتىكى تەرەقىقياتىمىز كىشلەر ئوتتۇرسىدىكى كۈندىلىك ئالاقىدىن ،جەمىئەت مۇناسىۋىتىدىن ئايرىلالمايدۇ . مەن مۇشۇ نوقتىنى چىقىش قىلىپ،كەڭ تورداشلارنىڭ پايدىلنىشى ئۈچۈن  ئۆزۈم ياخشى كۆرۈپ ئوقۇغان كىشلىك مۇناسىۋەت چەمبىرىكى دىگەن بۇ كىتاپنىڭ ئالدىنقى ئىككى باب تەرجىمىسنى 2014 –يىللىق مىسرانىم مۇكاپاتلىق ئەسەرلەر پائالىتىگە يوللىدىم .كەڭ تورداشلارنىڭ بۇ ئەسەرنى تەنقىدى پىكىرلەرنى بەرگەن ئاساستا قوبۇل قىلىشنى ئۈمۈد قىلىمەن .

كىتاپ تۈرى :تەلىم تەربىيە خاراكتىرلىك

1.باب كىشلىك مۇناسىۋەت تورىنى ھاسىل قىلغان  كىشلەرنىڭ مۇۋاپىقىيىتى خۇددى پەلەمپەيدىن مىڭىپ چىققانغا ئوخشايدۇ ،كىشلىك مۇناسىۋەت تورى يوق كىشلەرنىڭ مۇۋاپىقىيىتى خۇددى پەلەمپەيدىن ئۆمىلەپ چىققانغا ئوخشايدۇ.

كىشلىك مۇناسىۋەتنى بىر كۆرۈنمەس ئالتۇن كانغا ئوخشىتىشقا بولىدۇ ،بۇ ئالتۇن كاندىن قىزىپ تۈگەتكىلى بولمايدىغان بايلىققا ئىرىشكىلى بولىدۇ .بايلار بۇ  نوقتىنى  ياخشى تونۇپ      يەتكەن ،شۇڭلاشقا باي كىشلەر تىخمۇ باي بولماقتا ،ئۇلارنىڭ بۇ نوقتىغا بولغان چۈشەنچىسى   تىخمۇ ئاشماقتا ،نامرات كىشلەر بۇ نوقتىنى چۈشۈنۈپ يىتەلمىگەن ،شۇڭلاشقا نامرات كىشلەر   تىخمۇ نامراتچىلىققا يۈزلەنمەكتە ،ئەگەر نامرات كىشلەر بىر ئۆمۈر بۇ نوقتىنى چۈشۈنۈپ يىتەلمىسە ،ئۇلارنىڭ تۇرمۇشى بىر ئۆمۈر نامراتچىلىقتا ئۆتىدۇ ،بىر ئەۋلاد نامراتچىلىقتا يۈرۈش بۇ بىر چوڭ مەسىلىمۇ ئەمەس ،ئەمما ئەۋلادمۇ ئەۋلاد نامراتچىلىقتا يۈرۈشى كىشىنى تولىمۇ ئويلاندۇرىدۇ .
پۇرسەت جۇغراپىلىك ئەۋزەللىككە يەتمەيدۇ ،جۇغراپىلىك ئەۋزەللىك خەلقنڭ بىرلىك ئىتپاقلىقىغا يەتمەيدۇ .
تورلاردا كەڭ تارقىلىپ يۈرگەن مۇنداق بىر گەپ بار : كىشلىك مۇناسىۋەت تورىنى شەكىللەندۈرگەن كىشلەرنىڭ مۇۋاپىقىيىتى خۇددى پەلەمپەيدىن مىڭىپ چىققانغا ئوخشايدۇ ،كىشلىك مۇناسىۋەت تورى يوق كىشلەرنىڭ مۇۋاپىقىيىتى خۇددى پەلەمپەيدىن ئۆمىلەپ چىققانغا ئوخشايدۇ.
ئۈچ پادىشاھلىق ھەققىدە قىسسە دىگەن كىتاپنى كۆرگەن بارلىق دوستلار بۇ بىر جۈملە سۆزنى   ئالاھىدە چۈشىنىپ يىتىشى مۇمكىن .ئۈچ پادىشاھلىق ھەققىدە قىسسەنىڭ باشلىنىشدا       مۇنداق بىر جۈملە سۆز تىلغا ئىلىنغان .پۇرسەت جۇغراپىلىك ئەۋزەللىككە يەتمەيدۇ ،جۇغراپىلىك ئەۋزەللىك خەلقنڭ بىرلىك ئىتپاقلىقىغا يەتمەيدۇ .گەرچە كۈچلۈك كىشلەردە پۇرسەت ۋە جۇغراپىلىك ئەۋزەللكىتىن ئىبارەت شەرت -شارائىت تولۇق پىششپ يىتىلگەن بولسىمۇ ئەمما ئىتپاقلىق  روھى بولمىسا ھىچقانداق ئىشنى ياخىشى ۋۇجۇدقا چىقىرالمايدۇ ،بارلىق تىرىشچانلىقلىرى بىكار بولىدۇ .بەزى بىر قىسىم ئاجىز كىشلەردە ئىتپاقلىق روھىلا بولىدىكەن ئەنە شۇ كۈچلۈك كىشلەر بىلەن تەڭ رىقابەتكە چۈشىدىغان پۇرسەتكە ئىگە بولالايدۇ ،دىمەك ئىتپاقلىق روھى ناھايتىمۇ مۇھىمدۇر .بۇ مىساللارنىڭ بىزنىڭ ھازىرقى ئەمەلىي تۇرمۇشىمىزغا بولغان تەسىرى ناھايتى كۈچلۈك بولماقتا .ئەگەر سىز پۇرسەتنى ياخشى قەدىرلىسىڭىز ،تەلىيىڭىز ياخشى بولسا ھەر خىل شەرت -شارائىتلاردىن ياخىشى پايدىلىنالىسىڭىز سىز ياراتقان ئىگىلىك جەمىئەتتە ئەڭ ياخىشى روناق تىپىۋاتقان ئىگىلىك بولۇپ چىقىشى مۇمكىن .ئەمما بۇ يەردە مىنىڭ دىمەكچى بولغىنىم يۇقىردا ئىيتىلغان ئامىللارنىڭ ھەممىسى ئىتپاقلىق روھىغا ھەرگىزمۇ يەتمەيدۇ ،ئىتپاقلىق روھى بىزنىڭ  بىر ئىشتا نەتىجە قازىنىشمىزدىكى ئەڭ ياخشى ياردەمچى بولالايدۇ .
قەدىمكى كىشلەرنىڭ مۇنداق بىر سۆزى بار :ئەگەر دۇنيانىڭ ھەمىمىلا يىرىدە دوستۇڭ بولسا ،ئۇنداقتا سەن دۇنيانى ئايلىنىپ ئىزىپ قىلىشتىن قورقمىساڭ بولىدۇ .بۇ سۆز ئىتپاقلىق روھىنىڭ بىزنىڭ ئەمەلي تۇرمۇشىمىزدىكى تەسىر  كۈچىنى  تىخمۇ كەڭ دائىردە  يورۇتۇپ بەردى .
جاھان كەزدى كىشلەرمۇ بۇ نوقتىنى ئاللىقاچان چۈشۈنۈپ يەتكەن بولىشى مۇمكىن ،شۇڭلاشقا ئۇلار ئادەتتە ئىتتىپاقلىق روھىنى يىتىلدۈرۈشكە ئالاھىدە ئەھمىيەت بىرىدۇ .ئۇلار كۆپلەپ دوست تۇتىدۇ ،كىشلىك مۇناسىۋەتنى بىر ئىشتا مۇۋاپىقىيەت قازىنىشنىڭ ئاساسى دەپ قارايدۇ .
ئەمما  ئەتراپىمىزدىكى بەزى كىشلەر ئىتپاقلىق روھىنى ئانچە مۇھىم بىلىپ كەتمەيدۇ ،ئۇلار بۇ ئىتپاقلىق روھىنى يىتىلدۈرۈشكە ،قوغداشقا ئەھمىيەت بەرمەيدۇ .ئۇلار سىرتقى دۇنيا بىلەن ئالاقە قىلىدىغان دىرىزىسىنى ئىتىۋەتكەن گەرچە بۇنداق كىشلەر دىرىزىنىڭ سىرتىدا  مۇۋاپىقىيەت قازانغان بولسىمۇ ئەمما بۇ دىرىزە سىرتقى دۇنيا بىلەن ئالاقە قىلىشتىكى زور بىر توسالغۇغا ئايلىنىپ قالغان .شۇڭا سىز ئىتپاقلىق روھىغا ،كىشلىك مۇناسىۋەتكە ھەرگىز سەل قارىماڭ ،بىلىشىڭىز     كىرەككى كىشلىك مۇناسۋەت سىزنىڭ ئەڭ زور بايلىقىڭىزدۇر .شۇڭا سىز بۇ بايلىقتىن ياخشى پايدىلنىڭ .ھەممىمىز كۆرۈپ تۇرۇپتىمىزكى ھازىر جەمىئەتتە كىشلىك مۇناسىۋەت ناھايتىمۇ مۇرەككەپ .خەلقنىڭ بىرلىك مەسىلىسى بىزنىڭ باش تارتىپ بولمايدىغان مۇھىم مەسىللەرنىڭ بىرى .خەلقنىڭ بىرلىكلىكى ،ئىتپاقلىقى  ياخشى بولغاندىلا ئاندىن ھەر قانداق قىيىنچىلىقنى يىڭىپ تاشلىغىلى ،ئىللىق قۇياشنىڭ نۇرىدىن بەھىرلەنگىلى بولىدۇ .
ئىجتىمائىي  ئىش تەقسىم قىلىشتەك بۇ دەۋىردە يەنىلا ھەمكارلىق بولغاندىلا ئاندىن نەتىجە قازانغىلى بولىدىغانلىقىنى تەشەببۇس قىلىدۇ .ھەر بىرەيلەن جەمئەتنىڭ بىر ئەزاسى ،پەقەتلا جەمىئەتكە چۆكۈپ كىرگەندە ،جەمىئەتنىڭ كۈچىگە تايانغاندىلا ئاندىن بىر ئىشنى ياخىشى ۋۇجۇدقا ئىلىپ چىققىلى بولىدۇ .سودىگەرلەرمۇ نۇرغۇن دوستلىرى بىلەن ھەمكارلاشقاندا ئاندىن سودىسى روناق تاپالايدۇ ،ئاندىن باي كىشلەرگە ئايلىنالايدۇ .بىلىشىمىز كىرەككى بىر تامچە سۇ ئاسانلا قۇرۇپ كىتىدۇ ،ئەمما بۇ بىر تامچە سۇنى بىز دەرياغا ئىلىپ چوڭ دەريادىكى سۇغا ئارلاشتۇرۋەتسەك ئۇنداقتا بۇ بىر بىر تامچە سۇنىڭ كۈچىمۇ ناھايتى زور بولىدۇ ،ھەرگىز قۇرۇپ كەتمەيدۇ .بىر ئادەم چوڭ ئىشلارنى ۋۇجۇدقا چىقىرىمەن دەيدىكەن چوقۇم ئۇ كوللىكتىپنىڭ كۈچىگە تايىنىششى كىرەك ،ئەگەر كوللىكتىپنىڭ كۈچى بولمايدىكەن بۇ ئادەمنىڭ ئارزۇسى ئەمەلگە ئاشمايدۇ .
مۇۋاپىقىيەت قازانغۇچىلار ئۆزىنىڭ مۇۋاپىقىيىتنى ئەزەلدىن ئۆزۈمنىڭ كۈچىگىلا تاياندىم دەپ قارىمايدۇ .
بىلگايتىسمۇ دائىم مۇنداق دەيدۇ ؛مىنىڭ مۇشۇنداق مۇۋاپىقىيەت قازىنىشىمدىكى سەۋەپ نۇرغۇن مۇۋاپىقىيەت قازانغۇچىلار مەن ئۈچۈن خىزمەت قىلغان ،مەن شۇڭلاشقا مۇۋاپىقىيەت قازىنالىدىم .نىيوتۇنمۇ ئۆزىنىڭ باشقىلارغا قارىغاندا تىخمۇ يىراقنى كۆرەلەيدىغانلىقىنى ئىتراپ قىلغان ،   چۈنكى  ئۇ ئالاھىدە ئىمتىيازغا ئىگە بولغان ،بۇ دەل ئۇنىڭ دوستلىرى ۋە كىشلىك            مۇناسىۋىتىدىن ئىبارەت .
بىر ئادەمنىڭ زور مۇۋاپىقىيەتلەرنى قولغا كەلتۈرىشىدىكى سەۋەپ دەل ئۇنىڭ ئارقىسىدىكى بىر    توپ كۈچنىڭ ئۇنى قوللىغانلىقىدا .سىز كۆڭۈل ئىچىش پىروگىراممىلىرىغا دىققەت قىلسىڭىز بۇ نوقتىنى تىخمۇ ئىنىق چۈشۈنۈپ يىتەلەيسىز .بىر چولپاننىڭ كەينىدە چوقۇم تەسىر كۈچى زور بولغان بىر شىركەت ئۇنى قوللاپ قۇۋۋەتلەپ تۇرىدۇ ،ئۇنىڭ نامىنى چىقىرىشقا تىرىشىدۇ .كىنو چولپانلىرىمۇ كوللىكتىپنىك كۈچىدىن،كوللىكتىپنىڭ ھەمكارلىقىدىن ئايرىلالمايدۇ .ناخشا چولپانلىرىمۇ ھەم شۇنداق .نۇرغۇن كىشلەر بۇ چولپان ئۈچۈن كەڭ تەشۋىقات ئىلىپ بارىدۇ ،جەمئەتنىڭ ئۇنىڭغا بولغان قاراتمىلىقىنى زورايتىدۇ .
ھازىرقى دەۋىر بولسا ھەمكارلىقىنى زور دەرىجىدە تەشەببۇس قىلىدىغان دەۋىر ،ھەمكارلق بولمىسا مۇۋاپىقىيەتتىن سۆز ئاچقىلى بولمايدۇ .
جاڭ رۈي مىن رەھبەرلىك قىلغان خەيئىر شىركىتى جوڭگولۇقلارنىڭ پەخرى .بۇ شىركەتمۇ كوللىكتىپنىڭ زور كۈچىگە تايىنىپ تەرەققى بۈگۈنكىدەك نەتىجىلەرگە ئىرىشتى .بۇ شىركەتنىڭ ھەر بىر تۈردىكى سودىسىنى ھەمكارلىقتىن ئايرىپ قارىغىلى بولمايدۇ .
بىر ئادەمنىڭ مۇۋاپىقىيىتى بولسۇن ياكى بىر شىركەتنىڭ مۇۋاپىقىيىتى بولسۇن چوقۇم ھەمكارلىقتىن ئايرىپ قارىغىلى بولمايدۇ ،ھەمكارلىق بولمايدىكەن مۇۋاپىقىيەتتىن سۆز ئاچقىلى بولمايدۇ .
خەيئىر شىركىتى بۇنىڭ تىپىك مىسالى .
پەقەت ھەمكارلاشقاندىلا ئاندىن ھەر بىر ئادەم ئۆزىنىڭ تالانتىنى ئەڭ زور دەرىجىدە نامايەن قىلالايدىغان چوڭ سەھنىگە ئىگە بولالايدۇ .شۇڭا دەرھال ھەمكارلىق ئىقتىدارىنى يىتىلدۈرۈڭ ،باشقىلار بىلەن ھەمكارلىشىپ تەڭلا مۇۋاپىقىيەت قازىنىدىغان پۇرسەتنى يارىتىشقا تىرىشىڭ .
ھازىرقى دەۋىردىكى مۇۋاپىقىيەت سىزنىڭ نىمىنى بىلگەنلىكىڭىز ياكى نىمىنى قىلغانلىقىڭىزدا ئەمەس بەلكى سىزنىڭ كىمنى تونۇشىڭىزدا ،كىشلىك مۇناسىۋەت چەمبىرىكىڭىزنىڭ قانداق بولىشىدا ئىپادىلىنىدۇ .
ھوللىۋودتا مۇنداق بىر جۈملە سۆز مودا بولغان :ھازىرقى مۇۋاپىقىيەت سىزنىڭ نىمە ئىش قىلغانلىقىڭىزدا ئەمەس بەلكى سىزنىڭ كىمنى تونۇغانلىقىڭىزدا .ئەمەليەتتە بىر ئادەمنىڭ مۇۋاپىقىيەت قازىنىشىدىكى ھالقىلىق مەسىلە يەنىلا كىشلىك مۇناسىۋەت .دوست دىگەن بۇ ئۇقۇم ھەممە كىشى سەل قاراشقا بولمايدىغان ئىنتايىن مۇھىم بىر ئۇقۇم .مەيلى سىز جەمىئەتتىكى چوڭ رەھبەر بولۇڭ ياكى ئاددى پۇقرا بولۇڭ دوستىڭىز كەمچىل بولسا بولمايدۇ .دوستىڭىز كۆپ بولسا سىز       باشقىلار تەرىپىدىن ئاسانلىقچە ئۇنتۇلۇپ قىلىنمايسسىز .
ئەمەلىيەتتە دوسىت دىگەن بۇ ئۇقۇم ئىنتايىن ئاددى ،ئەگەر سىزنىڭ ياخشى بىر دوستىڭىز بولسا بۇ دوستىڭىز بىلەن ئۆز ئارا ھەمكارلىشىش ،ئۆز ئارا ياردەم قىلىش ئارقىلىق سودا ئىشلىرىڭىزنى  راۋاجلاندۇرالايسىز ،سودا يولىڭىزنى كىڭەيتەلەيسىز ،ھەرخىل قىيىن ئۆتكەللەردىن ئوڭۇشلۇق    ئۆتۈپ كىتەلەيسىز ،كىشلىك ئالاقىنى تىخمۇ قويۇقلاشتۇرالايسىز ،تاسادىپى سودا ئىشلىرىدا مەغلۇپ بولسىڭىز سىزنىڭ دوستلىرىڭىز سىزگە تايانچ بولالايدۇ ،ئۇلارنىڭ ياردىمىدە قايتىدىن باش كۆتۈرۈپ سودا ئىشلرىڭىزنى داۋاملاشتۇرالايسىز ،شۇڭا سىز دوست تاللاشقا ،دوست تۇتۇشقا ئالاھىدە ئەھمىيەت بىرىڭ .
تاڭشۇ چۈەنىڭ بىر ئاددى پادىچى بالىدىن بىر مەشھۇر كارخانچىغا ئايلانغاندىكى ئىشلىرى بىز ھەممەيلەننى قاتتىق ئويلاندۇرماي قالمايدۇ ئەلۋەتتە .تاڭشۇ چۈەن ئەسلىدە خۇچىڭ كەنتىدىكى بىر ئاددى دىھقان ،تۇغما مىيىپ ،ئۇ 20 ياش ۋاقتىدا  پۇزۇر كىيىنگەن بىر يولۇچى ئۇنىڭ قىشىغا كىلىپ ئۇنىڭدىن يول سورايدۇ ،ھەم تاڭشۇ چۈەنگە دىققەت نەزىرى بىلەن ئۇنىڭغا بىر قانچە خىل قول ھۈنەرۋەنچىلىكىنى ئۆگىنىش تەكلىپىنى بىرىدۇ ،بۇنداق بولغاندا پادا باققاندىن كۆپرەك كىرىىمگە ئىگە بولغىلى بولىدىكەن .تاڭشۇ چۈەن خۇشاللىق بىلەن بۇ يولۇچىنىڭ تەكلىپنى قوبۇل قىلىدۇ ۋە ئاتا -ئانىسى بىلەن خوشلىشىپ خۇچىڭ كوچىسىغا كىلىپ بىر ساتىراچلىق دۇكىنى ئاچىدۇ ،ھەيران قالارلىق يىرى شۇ ئۆزىگە ھۈنەر ئۆگىنىش تەكلىپىنى بەرگەن ھىلىقى يولۇچىنى كۈندە دىگۈدەك ئىشىك ئالدىدىن بىر نەچچە قىتىم ئۆتكەنلىكىنى  كۆرىدىكەن ،تاڭشۇ چۈەن باشقىلاردىن سۈرۈشتۈرۈش ئارقىلىق ئۆزىگە تەكلىپ بەرگەن بۇ يولۇچىنىڭ كىملىكىنى ئۇقۇپتۇ .بۇ يولۇچى ئەسلىدە خۇچىڭ يىزىسىنىڭ يىزا باشلىقى ئىكەن .بۇ يىزا باشلىقى تاڭشۇ چۈەننىڭ ئىشلىرىن تولۇق خەۋەردار بولغاندىن كىيىن ھەر كۈنى دىگۈدەك ئۇنى يوقلاپ كەپتۇ .يىزا   باشلىقى تاڭشۇ چۈەنگە قاتتىق ئىشەنچ باشلاپتۇ ۋە ھەر دائىم ئۇنى ئىلھاملاندۇرۇپ تۇرۇپتۇ ،ۋە بۇ مىيىپنىڭ تىرىشچانلىقىغا قايىل بوپتۇ .
يىزا باشلقىنىڭ تاڭشۇ چۈنگە قىلغان ئىلھام سۆزلىرى ئۇنىڭ بىر ئۆمۈر ھاياتىغا تەسىر قىپتۇ .كىيىن بۇ دۇكاننى كىڭەيتىپ ئۆزگەرتىپتۇ،شۇنىڭدىن باشلاپ قۇرلۇش ماتىرياللىرى سودىسى بىلەن شۇغۇللىنىپتۇ ،سودىسى بەكمۇ تىز راۋاجلىنىپتۇ .بۇ ۋاقىتتا يىزا باشلىقى ئۇنىڭغا بىر شىركەت قۇرۇش تەكلىپىنى بىرىدۇ ،لىكىن تاڭشۇ چۈەننىڭ يىتەرلىك تەجىربىسى ۋە مەبلىغى بولمىغاچقا، يىزا باشلىقى شىركەت قۇرۇش جەھەتتە مول تەجىربىلىك كىشلەرنى تاڭشۇ چۈەنگە تونۇشتۇرۇپ كويىدۇ ،ھەم ئۇنىڭ بىلەن بىللە تەكشۈرۈشكە چىقىشقا تەيىنلەيدۇ ،يىزا باشلىقى ئۇنىڭ مەبلەغ مەسىلىسىنى ھەل قىلىش ئۈچۈن خۇچىڭ يىزىلىق ئامانەت قەرز كوپىراتىۋىدىن نۇرغۇن قەرز ھەل قىلىپ بىرىدۇ ،شۇنىڭ بىلەن تاڭشۇ چۈەننىڭ قۇرلۇش ماتىرياللىرى سودىسى بۇرۇنقىغا قارىغاندا تىخمۇ راۋاجلىنىدۇ .
كىيىن تاڭشۇ چۈەن شىركەت باشقۇرۇش جەھەتتە نۇرغۇن قىيىنچىلىقلارغا يولۇقىدۇ ،بۇ ۋاقىتتا يىزا باشلىقى يەنە قاراپ تۇرمايدۇ ،ئۇنىڭغا نۇرغۇن ياردەملەرنى قىلىدۇ ،تاڭشۇ چۈەننىڭ بۇ شىركىتى بارغانسىرى چوڭىيدۇ .ھازىر تاڭشۇ چۈەن خۇچىڭ ،جىجاڭ ،گۇيلىن قاتارلىق شەھەرلەردە شىركەت قۇرىدۇ ،تاڭشۇ چۈەننىڭ تۇرمۇشىدا ئالامشۇمۇل ئۆزگىرىش ھاسىل بولىدۇ .تاڭشۇ چۈەننىڭ بۇ مۇۋاپىقىيەتلەرنى قولغا كەلتۈرىشىدىكى ئاساسلىق سەۋەبى نەدە ؟بۇنى ئەستايىدىل تەھلىل قىلىدىغان بولساق  كۆرۈپ يىتەلەيمىزكى ئۇنىڭ مۇشۇ مۇۋاپىقىيەتلەرنى قولغا كەلتۈرىشىدىكى مۇھىم سەۋەپ دەل ئۇنىڭ بىر ياخشى يىزا باشلىقى بىلەن تونۇشقانلىقىدا .
تاڭشۇ چۈەن بۇ يىزا باشلىقى بىلەن تونۇشقاندىن كىيىن بىر مىيىپ نامرات پادىچى بالىدىن داڭلىق كارخانچىغا ئايلاندى ،ئىشلىردا ناھايتىمۇ چوڭ مۇۋاپىقىيەتلەرنى قازاندى ،كىشلىك ھايات يولىدا ئويلاپ يەتكۈسىز قىيىنچىلىقلارنى يىڭىپ تاشلىدى .ئۇنىڭ بۇ نەتىجىلىرىنى كىشلىك مۇناسىۋەت چەمبىرىكى ھاسىل قىلغان بىر مۆجىزە دىيىشكە بولىدۇ ،بەزى دوستنىڭ كۆپ بولىشى تەلەپ قىلىنمايدۇ ،سىز قىيىنچىلىققا يولۇقۇپ ياردەمگە مۇھتاج بولغان ھالقىلىق پەيتلەردە بىر دوستىڭىز سىزگە يادەم قىلىپ  قىيىنچىلىقىڭىزنى ھەل قىلىپ بەرسە بۇمۇ بىر خىل كىشلىك          مۇناسىۋەت چەمبرىكى پەيدا قىلغان بىر مۆجىزە ھىساپلىنىدۇ
ئەمما بەزىلەر بۇنداق دەپ قارىمايدۇ ،ئۆزىنى ناھايتىمۇ بەك ئىقتىدارلىق ھىساپلايدۇ ،باشقىلارغا تايىنىشنىڭ ،باشقىلار بىلەن ھەمكارلىشىشنىڭ زۆرۈريىتى يوق دەپ قارايدۇ  .مۇۋاپىقىيەت قازىنىشنىڭ مۇھىم ئامىللىرى بىلەن بىرلەشتۈرۈپ بۇ مەسىلىنى تەھلىل قىلغاندا ئۇلارنىڭ بۇ قارىشى   پۈتۈنلەي خاتا بولىدۇ ،بىز بىلىشىمىز كىرەككى ئۇچقاندەك تەرەققى قىلىۋاتقان بۇ جەمىئەت ھەمكارلىشىشقا مۇھتاج،بىر قانچىلىك ئىقتىدار يىتىلدۈرۈشتىن قەتئىنەزەر ئۇ ئادەمنىڭ ئىقتىدارى ناھايتىمۇ چەكلىك بولىدۇ .بىز پۈتۈن جەمىئەتنى بىر دىڭىز دىسەك بىز ھەر ئىنسان بۇ دىڭىزدىكى بىر تامچە ھىساپلىنىمىز .ئويلاپ باقايلى بۇ بىر تامچىنىڭ كۈچى قانچىلىك بولماقچىتتى ،ئەگەر بۇ تامچە كۈچۈمنى ھەقىقى تۈردە نامايەن قىلىمەن دەيدىكەن چوقۇم دىڭىزغا قوشۇلىشى كىرەك ،ئۇنداق بولمايدىكەن بۇ تامچە ئاسانلا قۇرۇپ كىتىدۇ .
بىز باشلانغۇچ مەكتەپنىڭ دەرسلىك كىتابىدا ئۆگەنگەن مۇنداق بىر ھىكايە ئىسىڭىزدىمۇ ؟بىر كۈنى يولۋاس تۈلكىنى تۇتۇۋاپتۇ ،ۋە تۈلكىنى چۈشلۈك تاماق قىلماقچى بوپتۇ .بىز ھەممىمىز بىلىمىز تۈلكە ئىنتايىن بىر ھىيلىگەر ھايۋان .بۇ ۋاقىتتا تۈلكە يولۋاسقا دەپتۇ :مىنى يىسەڭ سىنىڭ قورسۇقۇڭ يىرىم بولمايدۇ ،مەن سىنى نۇرغۇن ئوزوق بار يەرگە باشلاپ باراي ،ئۇ ۋاقىتتا سەن تىخمۇ كۆپ ئوزوقتىن بەھىرمان بولىسەن دەپتۇ .يولۋاس ماقۇل بوپتۇ ،شۇنىڭ بىلەن تۈلكە يولۋاسىقا يول باشلاپ مىڭىپتۇ ،ماڭغاندىمۇ ئىككى تەرەپكە ئىغاڭلاپ پوچى ھالەتتە مىڭىپتۇ ،بۇنى     بۇنى كۆرگەن باشقا ھايۋانلار قىچىپ كىتىپتۇ ، تۈلكە قاقاقلاپ كۈلۈپتۇ ،يولۋاسنىڭ ياردىمى بىلەن تۈلكە ئۆزىنى باشقا ھايۋانلارغا ناھايتىمۇ بەك ھەيۋەتلىك كۆرسىتىۋاپتۇ .كىچككىنە بىر تۈلكە يولۋاسىنى ئالدىيالاپتۇ ،بۇمۇ  تۈلكىنىڭ بىر ئىقتىدارى ھىساپلىنىدۇ ئەلۋەتتە.
مۇكەممەللەشكەن سودا ساھەسىمۇ مۇشۇ ئەقىللىق تۈلكىگە ئوخشاش ئۆزىدىن كۈچلۈك كىشلەر بىلەن مۇناسىۋەت قىلىشنى ،ئۇلار بىلەن ھەمكارلىشىشنى تەلەپ قىلىدۇ.بۇ ئارقىلىق كىشلەر ئۆزىدىكى چەكلىملىكنى بۆسۈپ تاشلىيالايدۇ .دۇنيادىكى نۇرغۇن ئەمەلگە ئاشمايدۇ دەپ قارالغان مۆجىزىلەر دەل مۇشۇنداق پەيدا بولغان .
ئەمەلىيەت شۇنى ئىسپاتلاپ تۇرىدىكى ،بىز چوقۇم مۇنداق بىر نوقتىنى ياخشى ئىسىمىزدە تۇتىشىمىز لازىم ،ئۇ بولسىمۇ ،سىزنىڭ مۇۋاپىقىيەت قازىنىشىڭىز سىزنىڭ نىمە بىلىشىڭىز ،نىمىنى قىلغانلىقىڭىز ئەمەس بەلكى سىزنىڭ كىمنى تونۇغانلىقىڭىزدا ئىپادىلنىدۇ .بىر ئەقىللىق سودىگەر ئەڭ ئاۋۋال ھەل قىلىدىغان مەسىلە بولسا كىشلىك مۇناسىۋەت مەسىلىىسىدۇر .
بۇ دۇنيادا تالانتلىق كىشلەر كەمچىل ئەمەس ،ئەمما ئورۇن چەكلىك .ئەگەر ئىككى ئورۇندۇق بولسا ،بىر ئورۇندۇقتا باشقا بىرسى ئولتۇرۇپ بولغان بولسا پەقەتلا بىرلا ئورۇنداق قالغان بولىدۇ ئەگەر سىز بۇ ئورۇندۇققا ئىرىشىش ئۈچۈن تىرىشمىسىڭىز باشقىلار بۇ ئورۇندۇقنى ئىگەللەپ       كىتىدۇ ،ئۇنداقتا يەنە سىزنىڭ ئورنىڭىز بارمۇ ؟ئورۇندۇق قالمىغاننىڭ ئاقىۋىتى قانداق بولار ؟
بۇ يەردىكى مەسىلەر دەل سودا ساھەسىگە ئوخشايدۇ ،ئەگەر سىز ئىگىلىك تىكلىمەكچى بولۇپ تىرىشمىسىڭىز ،باشقىلارنىڭ سىزنىڭ يولىڭىزنى ئىگەللىۋالىدۇ،سىز يولىڭىز بارغانسىرى ئازلايدۇ دىگەن گەپ .
شۇڭلاشقا سىز ياخشىراق ئويلىنىپ بىقىڭ ،سىز نۇرغۇن جاپا مۇشەققەتلەرنى تارتىپ ھىچقانداق  نەتىجىگە ئىرىشەلمىسىڭىز ،بىر ئىشلاردا ئارىسالدى بولۇپ ئاسانلقچە ھۆكۈم چىقىرالمىسىڭىز ئۇنداقتا سىز «ئۆتكۈر قىلىچ روھى»نى ئۈگۈنۈڭ .مۇشۇنداق قىلسىڭىز سىز ئىقتىدارىڭىزنى يۇقىرى كۆتۈرەلەيسىز ۋە نۇرغۇن سىزدىنمۇ كۈچلۈك كىشلەر بىلەن مۇناسىۋەت قىلالايسىز ،شۇنداقلا        كەسىپتە مول نەتىجىلەرنى قولغا كەلتۈرەلەيسىز .
باي ۋە نامرات كىشلەرنىڭ پەرقى پەقەت ئىككى خەت ئارقىلىق ئىپادىلنىدۇ .نۇرغۇن كىشلەر باي ئادەملەرنى كۆرسىلا ئىچىدە تىللاپلا يۈرىدۇ ،بۇ كۆپىلىگەن جوڭگولۇقلارنىڭ خاراكتىرى .بەزى بىر قىسىم كىشلەر باي ئادەملەرنىڭ باي بولۇش ئىقتىدارىغا بەكمۇ ھەسەت قىلىدۇ ،ئەمما ئكزىدىكى يوشۇرۇن ئىقتىدارنى قىزىپ چىقىش ئۈچۈن تىرىشىپ كۆرمەيدۇ .ئۇلار باي ئادەملەرنى تىللايدۇ ،ئەمما ئۆزىنىڭ باي بولالمىغانلىق سەۋەبى ئۈستىدە ئويلانمايدۇ .ئۇلار كەمبەغەللىك ھالىتىنى ئۆزگەرتىپ باي كىشلەر قاتارىدىن ئورۇن ئىلىشنى ناھايتىمۇ ئۈمۈد قىلىدۇ ،ئەمما باي كىشلەر نىمە باي ،نامرات كىشلەر نىمە ئۈچۈن نامرات ،بۇ مەسلىللەر ئۈستىدە باش چۆكۈرۈپ ئويلانمايدۇ .
بىز بىرلىكتە بۇ مەسىلىنى تەھلىل قىلىپ باقايلى ،باي كىشلەر نىمە ئۈچۈن باي ،نامرات       كىشلەر نىمە ئۈچۈن نامرات بولىدۇ ؟
باي كىشلەرنىڭ باي بولىشىدىكى ئاساسى سەۋەبى شۇكى ،باي كىشلەر ھەر دائىم باي بولۇش يوللىرى ئۈستىدە توختىماي ئىزدەنگەن ۋە تالانتىنى جارى قىلدۇرغان .نۇرغۇن ئەمەلي مىساللار شۇنى ئىسپاتلاپ تۇرىدىكى باي كىشلەرنى باي بولىشىدىكى سەۋەبى پەقەتلا ئۇلاردا تالانىتنىڭ بولغانلىقى ۋە تالانتىنى ھەقىقى تۈردە جارى قىلدۇرغانلىقىدا ئەمەس ،نامرات كىشلەرنىڭ نامرات بولىشىدىكى سەۋەبى ئۇلاردا تالانىت ۋە ئەقىل ئىدراكنىڭ كەمچىل بولغانلىقىدا ئەمەس .
ئۇنداقتا بۇنداق چوڭ پەرقنى كەلتۈرۈپ چىقارغان ئاساسلىق سەۋەب زادى نىمە؟
ھەممە ئادەمنىڭ بۇ سۇئالغا بەرگەن جاۋابى ئوخشاش بولمايدۇ . بۇنچىلىك چوڭ پەرقنى كەلتۈرۈپ چىقارغان ھالىقىلق مەسىلە دەل كىشلىك مۇناسىۋەتتۇر .باي كىشلەر باشقىلارغا قارىغاندا تىخمۇ كەڭ دائىرلىك بولغان كىشلىك مۇناسىۋەتتىن پايدىلنىپ كىشلەرنى ھەيران قالدۇرىدىغان بايلىققا ئىرىشەلەيدۇ ۋە مۇۋاپىقىيەتلەرنى قولغا كەلتۈرەلەيدۇ .نامرات كىشلەر ئادەتتە كىشلىك مۇناسىۋەتكە ئەھمىيەت بەرمىگەن ،شۇڭلاشقا ئۇلارنىڭ كىشلىك مۇناسىۋەت چەمبىرىكى ناھايتىمۇ تار كىشلىك مۇناسىۋەت چەمبىرىكىنىڭ تار بولىشى ئاخىردا ئۇلارنى نامرات ھالەتتە قالدۇرىدۇ .ھىچقانداق بايلىق ۋە مۇۋاپىقىيەتلەرگە ئىرىشەلمەيدۇ .
كىشلىك مۇناسىۋەت بىر كۆرۈنمەس ئالتۇن كانغا ئوخشايدۇ ئەگەر سىز بۇ ئالتۇن كاننى ياخشى قىزىپ چىقالىسىڭىز ،ئۇنداقتا سىز تەڭداشسىز بايلىققا ئىرىشەلەيسىز .باي كىشلەر بۇ نوقتىنى ياخشى تونۇپ يەتكەن بولغاچقا ئۇلار باي بولالىغان ،نامرات كىشلەر بۇ نوقتىنى ياخىشى تونۇپ يەتمىگەن بولغاچقا ئۇلارنىڭ ئەۋلادلىرى بىر ئۆمۈر نامراتلىق ھالەتتە ئۆتىدۇ .
ئەقىللىق سودىگەرلەر كىشلىك مۇناسىۋەتنىڭ مۇھىم دەرىجىسىنى ناھايتىمۇ ياخىشى بىلىدۇ .
كۆپچىلىك ھەممەيلەن مۇنداق بىر ئەمەليەتكە دىققەت قىلغان بولىشى مۇمكىن ،تارىختىن تارتىپ ھازىرغىچە كىشلىك مۇۇناسىۋەتنى  كىگەيتكەن ۋە ياخىشى ساقلاپ كەلگەن كىشلەر كەسىپتە ياخىشى نەتىجە قازىنالايدۇ،ئەمما كىشلىك مۇناسىۋەتنى كىڭەيتىشكە ئەھمىيەت بەرمىگەن ۋە ياكى ئۈزۈپ قويغان كىشلەر كەسىپتە چوقۇم مەغلۇپ بولىدۇ .
شۇڭلاشقا بىز ئادەتتە كىشلىك مۇناسىۋەتنى ساقلاشقا ۋە كىڭەيتىشكە ئەھمىيەت بىرىشىمىز لازىم .ئەگەر سىز كىشلىك مۇناسىۋەتنى ساقلاشقا ۋە كىڭەيتىشكە ئەھمىيەت بەرسىڭىز ،تونۇشىڭىز كۆپ بولسا ئىش بىجىرىش يوللىرىڭىز ناھايتىمۇ راۋان بولىدۇ .
بىز تۆۋەندە لىيۇۋ فىينىڭ ھىكايىسى ئارقىلىق بۇ پاكىتنى تىخمۇ چوڭقۇر دەرىجىدە چۈشۈنۈپ    يىتەيلى .
ئون يىلنىڭ ئالدىدا لىيۇۋ فىي خىنەن ئۆلكىسى بىر نامرات  ئائىلىدە تۇغۇلغان  نامراتلىقنىڭ ئازابىنى بوش تارتمىغان بالا ئىدى ،10 يىلدىن كىيىن مەبلىغى بىر مىليۇندىن ئاشىدىغان داڭلىق بىر كارخانچىغا ئايلىنىدىغانلىقىنى كىممۇ ئويلاپ يەتكەن .بەزىلەر ئۇنىڭدىن قانداق قىلىپ بۇ غايات زور مۇۋاپىقىيەتلەرنى قولغا كەلتۈرگەنلىكىنى سوراپتۇ .ئۇ ناھايتىمۇ ئەستايىدىللىق بىلەن سوراپتۇ .ئۇ ئەستايىدىللىق بىلەن جاۋاپ بىرىپ ئۆزىنىڭ بۇ مۇۋاپىقىيەتلىرى پۈتۈنلەي دوستلىرنىڭ ياردىمى بىلەن قولغا كەلگەنلىكىنى ،ئەگەر دوستلىرنىڭ ياردىمى بولمىسا بۇ مۇۋاپىقىيەتلەرنى    قولغا كەلتۈرەلمەيدىغانلىقىنى ،ئۆزىنىڭ ئەسلىدىكى نامراتلىق ھالىتىدىن قۇتۇلۇپ چىقالمايدىغانلىقنى ئىيتىپتۇ .ئەسلىدە ئۇ ئالي مەكتەپنى پۈتتۈرۈپ تۇرىشىغا ،دوستلىرى ئۇنىڭغا گۇاڭجۇدىكى بىر ئۈنچە-مەرۋايىت شىركىتىگە بىرىپ ياردەمچى مال ساتقۇچى بولۇش تەكلىۋىنى بىرىپتۇ .ئۇ ياردەمچى سودىگەر بولۇش جەريانىدا بىر توپ سودىگەرلەر بىلەن دوست بولۇپتۇ .ئۇ شاڭگاڭدىن كەلگەن چوڭ سودىگەرلەرنىڭ ياردىمىدە شاڭگاڭ سودىگەرلەر ئۇيۇشمىسىغا كىرىپتۇ .بۇ ئۇيۇشمىغا كىرگەندىن كىيىن نۇرغۇن مۇۋاپىقىيەت قازانغۇچىلار بىلەن دوست بوپتۇ.ئۇنىڭ نۇرغۇن دوست تۇتىشى بىلەن ئۇنىڭ سودا ساھەسىگە يۈرىشىگە بىر يول ئىچىلىپتۇ .ئۇ ھەر قىتىم سىرتقا چىققاندا ئسىم كارتىسىنى ئىلىۋىلىشنى ئۇنۇتمايدىكەن ۋە يولدا ئۇچراتقان دوستلىرغا بىردىن تارقىتىپ بىرىدىكەن .
ئۇ مول بولغان كىشلىك مۇناسىۋەت تورى بىلەن سودا ساھەسىگە يۈرۈش قىپتۇ ،دوستلىرنىڭ ياردىمى بىلەن ئۇنىڭ سودا ناھايتى تىزلا روناق تىپىپتۇ ،سودا ساھەسىدە غايەت زور مۇۋاپىقەيتلەرنى قولغا كەلتۈرۈپتۇ .مانا كۆرۈپ تۇرۇپتىمىز لىيۇۋ فىينىڭ بۇ ھىكايىسى كىشلىك مۇناسىۋەتنىڭ مۇھىملىقىنى ھەقىقى تۈردە ئىسپاتلاپ بەردى .بىز بىلىشمىز كىرەككى ھەر بىرەيلەننىڭ مۇۋاپىقىيتى بىز ياشىغان كوللكتىپ ۋە جەمىئەتنىڭ يارىدىمى بىلەن بارلىققا كىلىدۇ .بىز بىر ئىشتا مۇۋاپىقىيەت قازىنىمىز دەيدىكەنمىز چوقۇم كىشلىك  مۇناسۋەت تورىنى كىڭەيتىشىمىز لازىم .كۆرۈپ تۇرۇپتىمىزكى باي كىشلەر ۋە نامرات كىشلەرنىڭ ئاساسلىق پەرقى پەقەتلا ئىككى خەت ئۇ بولسىمۇ كىشلىك مۇناسىۋەتتىن ئىبارەت.ئەگەر سىز كىشلىك مۇناسىۋەتنى كىڭەيتىسىڭىز ،ياخشى داۋاملاشتۇرسىڭىز ئۇنداقتا سىز مۇۋاپىقىيەتكە بىر چوڭ قەدەم يىقىنلاشقان بولىسىز .
سىز ھازىرغىچە نىمىشقا نامرات ،نىمىشقا يىتەرلىك خەجلىگۈدەك پۇلىڭىز يوق ؟چۈنكى سىزدە دەل كىشلىك مۇناسىۋەت يوق ،كىشلك مۇناسىۋەتنىڭ يوقلىقى سىزنى كەمبەغەللىكنىڭ پاتقىقىغا تاشلاپ قويىدۇ .بىلىپ قويۇڭ ،كىشلىك مۇناسۋەت تەڭداشسىز بايلىق .
نۇرغۇن كىشلەر تەلي ياخشى كىشلەرگە ھەسەت قىلىدۇ ،ئۇنى پۇتلاپ يىقىتۋىتىشنى ئويلايدۇ ،ئۇ نىمىشقا بۇنداق قىلىدۇ ؟نىمە باشقىلارغا ھەۋەس قىلىدۇ يۇ ،ئەمما ئۆزى مۇۋاپىقىەت قازىنىشنىڭ يوللىرى ئۈستىدە تىرشىپ ئىزدەنمەيدۇ ؟بۇنداق كىشلەر ھەر دائىم ئۆزىنى ئالدايدۇ ،ھەر دائىم مىنىڭ تەليىم بەك كاج،مەيلى مەن قانداق قىلاي باشقىلارغا يىتەلمەيمەن دەپ قارايدۇ .ئەمەليەتتە بۇ كىشلەر راستىنلا شۇنداق ئادەممۇ ؟بۇ كىشلەرنىڭ قابىليىتى باشقا كىشلەرگە يەتمەسمۇ ؟ئۇلارنڭ تەلي راستىنلا كاجمۇ ؟ئەمەليەتتە ئۇنداق ئەمەس ،مۇۋاپىقىيەت قازىنىش ،بايلىق توپلاشنىڭ پۇرسىتى ھەممەيلەنگە ئوخشاش ،پەرق شۇكى سىز نىڭ بۇ پۇرسەتنى قانداق قەدىرلىشىڭىز ،بۇ پۇرسەتتىن قانداق پايدىلنىشڭىز .
مەن سىزگە بىر تەكلىپ بىرەي ،سىز تەلي ياخىشى دەپ قارىغان كىشلەرنىڭ مۇۋاپىقىيەت قازىنىش يوللىرىغا دىققەت قىلىپ بىقىڭ .سىز دىققەت قىلسىڭىز شۇنى بايقايسىزكى ، بۇ مۇۋاپىقىيەت قازانغۇچىلار كىشلىك مۇناسۋەتتىن پايدىلنىشقا ناھايتىمۇ ماھىر كىلىدۇ ،كىشلىك مۇناسىۋەتتىن ياخىشى پايدىلنالايدۇ .چۈنكى بۇنداق كىشلەرنىڭ مەيلى سىياسى جەھەتتە بولسۇن ،ياكى سودا ساھەسىدە بولسۇن ئىشقىلىپ جەمئەتنىڭ ھەر قايسى ساھە ،ھەر قايسى كەسىپلىرىدە  دوستلىرى بولىدۇ ،ئۇلارنىڭ ئىش ئۇلارنىڭ ئىش بىجىرمىكى ناھايتىمۇ ئاسان بولىدۇ .ئۇلار كىشلىك  مۇناسىۋەتنى بىر ئىشتا مۇۋاپىقىيەت قازىنىشنىڭ ئالدىنقى شەرتى قىلىدۇ .
ھازىرقىدەك ئۇچۇرلاشقان ،زامانىۋىلاشقان دەۋىردە كىشلىك مۇناسىۋەت ئۆزىڭىزنى جەمىئەتكە    تونۇتىشىڭىزدىكى بىرىنجى قول ماتىريال ھىساپلىنىدۇ.شۇڭلاشقا بۇ دەۋىردە كىشلىك            مۇناسىۋەتكە ھەرگىز سەل قاراشقا بولمايدۇ .مۇۋاپىقىيەت قازانغۇچىلار دەل مۇشۇ كىشلىك مۇناسىۋەتتىن ئىبارەت بۇ ماتىريالغا تايىنىپ ئۆزىنىڭ كىشلىك ھاياتى يولىنى يورۇتقان مۇشۇ كەسپىدە  مۇۋاپىقىيەت قازانغان كىشلەرنىڭ ئىش -ئىزلىرىغا نەزەر سالىدىغان شۇنى كۆرۈۋالالايمىزكى ،كىشلىك مۇناسىۋەت بىر زور بايلىق .شۇنداقلا ئۆزىڭىزنى باشقىلارغا تونۇتىشىڭىزدىكى قىممەتلىك ماتىريال ،سىزنىڭ كىشلىك مۇناسىۋىتىڭىز قانچە كۆپ بولسا ،قانچە يۇقىرى بولسا سىزنىڭ كەسىپتە مۇۋاپىقىيەت قازىنىشىڭىز شۇنچە ئاسان ،پۇل تىپىش يوللىرىڭىز شۇنچە كۆپ بولىدۇ .بۇ دىگەنلىرىمىز ھەممەيلەن ئىتراپ قىلىدىغان ئەمەليەت .
بىلىپ قويۇڭ ،ئۇچۇرلاشقان بۇ دەۋىردە كىشلىك مۇناسىۋەت بىر پۇرسەت ،بىر قىزىپ تۈگەتكۈسىز كان بايلىقى دىمەكتۇر .يۇقىرى مەرتىۋلىك كىشلەرنىڭ ئىش بىجىرىش ئۇسۇلى دەل بۇ كان بايلىقىنى ياخشى قىزىپ چىقىشتۇر .
بىزنىڭ ھازىرقى ئەمەليەت بىزگە شۇنى ئەسكەرتىدۇكى كىشلىك مۇناسىۋەت بولمىسا ھىچقانداق ئىشنى ياخشى بىر باشقا ئىلىپ چىققىلى بولمايدۇ .
گەرچە سىزدە يۇقىرى تىخنىكا بىلىم ۋە قابىليەت ،شۇنداقلا يتەرلىك مەبلەغ بولسىمۇ سىزنىڭ كىشلىك مۇناسىۋىتىڭىز كەمچىل بولىدىكەن كەسپىىڭىزدە مۇۋاپىقىيەت قازىنىشىڭىز ناھايتىمۇ تەسكە توختايدۇ .
ئەگەر سىز ھەقىقى بىر مۇۋاپىقىيەت قازانغۇچى بولىمەن دەيدىكەنسىز ئەڭ ئالدى بىلەن كىشلىك مۇناسىۋەت مەسىلىسىنى ھەل قىلىشىڭىز لازىم .ئەگەر سىز كىشلىك مۇناسىۋەتتىن ئىبارەت بۇ مەسىلىنى ھەل قىلالىسىڭز ئۇنداقتا كەسپىڭىزدە مۇۋاپىقىيەت قازىنىشىڭىز ئۈچۈن توختىماي ئۇل سالالايسىز .ئەگەر سىزنىڭ كىشلىك مۇناسىۋىتىڭىز ياخىشى بولىدىكەن باشقا ئىشلاردىن ئانچە غەم قىلىپ كەتمىسىڭىزمۇ بولىدۇ ،ئەگەر سىزدە تىخنىكا بولمىسا دوستلىرىڭىز سىزنى ياخشى تىخنىكا بىلەن تەمىنلەيدۇ ،ئەگەر سىزدە مەبلەغ كەمچىل بولسا دوستلىرىڭىز سىزنى يىتەرلىك مەبلەغ      بىلەن تەمىن ئىتىدۇ .سىز بۇ ئەۋزەل شارائىتلاردىن ياخشى پايدىلنىپ مۇۋاپىقىيەتنىڭ بىرىنجى قەدىمىنى باسالايسىز .
