ئەسسالامۇ -ئەلەيكۇم مېھرى ئاتاش ، باغرى ئىللىق ماكاندىكى بارلىق قېرىنداشلىرىم  . ھەممىڭلارغا مېھرىيە دىن  مۇنۇ بۈگۈنكى   قۇياشتەك  كۈلۈپ  سالام   .  
ئەمدى ئۇشبۇ تېمىنى يوللاشتىكى مەقسەتكە  كەلسەم ، شۇ مۇنبەرنىڭ كەيپىياتىنى كۆتۈرىش ئۈچۈن تاققا -تۇققا سۆزلەرنى قىلىپ ئولتۇرۇپتىمەن . بىر قىسىم ئەزالارنى تېمىغا قوشۇش بىلەن ،جەمىيەتتىكى بىر قىسىم ناچار خائىشلارنى سۆزلەپ ئۆتمەكچىمەن . ئەمما ھەرگىزمۇ ئىسمى چىققانلارغا ۋەكىللىك قىلمايدۇ  -ھە !





مۇشۇ بالىلارنى تەتىل قىلىپ قويۈۋەتكەندە ئۇياق -بۇياقلارغا بېرىپ ساياھەت قىلىپ كېلىش توغرۇلۇك پىلانلارنى تۈزۈيمىز . بىر كۈنى بەگدۆلەتخان  (باغ دۆلەتمۇيا ،كەچۈرۈڭ  )  مىسرانىم  ئەپەندىمگە دەپتىغۇ :
__ بۇ تەتىلدە مېنىڭ غەلىتە -غەلىتە جايلارغا بېرىپ چۆگۈلەپ كەلگۈم بار ، سىز چوقۇم مېنىڭ بۇ ئۈمۈدۈمنى يەردە قويماڭ ، بولمىسا قىيامەتتە سوراققا قالىسىز .
__ ماقۇل ، ئەمىسە سېنى لىئاۋيوڭدىكى غەلىتە دۆڭگە ئاپىراي . شۇ يەردە بالىلار بىلەن تازا مۈكۈشمەك ئويناپ كەل .
ــ ياق مېنىڭ تۈكىيەگە بارغۇم بار .
_ توۋۋا ، مۇشۇ كۈندىكى خوتۇنلارغا تۈركىيە جىنى چاپلاشتىمۇ نىمە ، كەچكىچە <تۈركىيە  ،تۈركىيـــــــــە > دەپ ئاغزىڭلاردىن چۈشۈرمەيسىلەر .
__ ھازىر دىگەن شۇ مودا . سىزگە دىسەم مۇنۇ مۇنبەردىكى يۇلتۇز  باغۇ ، تۇغۇلغان كۈنىدە تۈركىيەنىڭ زىبۇ -زىننەتلىرىنى تاقاپ  ئىسمىنى نېمە دەيدىكەن تېخى <ھۆررەم >دەمدىكەن ئىشقىلىپ ،  تۈركىيەنىڭ تېرا ئاسراش مايلىرىدا ھۆسۈن گۈزەللەشتۈرۈپ ، يەنە ئاشۇ تۈركىيەنىڭ  ھازىر تازا مودا بۇلىۋاتقان كىيىملىرىنى كىيىپ ، مۇنبەردىكى بىر نەچچەيلەننى تۈركىيە  تائاملىرى رېستۇرانىغا تەكلىپ قىلىپتى ،ۋاي رېستۇراندىكى ھەممە ئەرلەرنىڭ كۆزىدىن ئوت چىقىپ كەتتىمىكىن ،ئىشقىلىپ ھەممە كىشىنىڭ كۆزى بىزنىڭ جوزىدىلا دەڭە ! شۇڭا مېنىڭ ئاشۇ تۈركىيەگە بىر بېرىپ كەلگۈم بار زادى ؟؟! بىزنىڭ خانىم -قىزلىرىمىزنى ئۆزىگە شۇنچە جەلىپ قىلغان ئاشۇ كىيىم -كېچەك ، زىبۇ -زىننەت ، خىرىستال -پىرىستال ، قاچا -قۇچا ، قازان قۇمۇچ ، تاتلىق تۈرۈملىرىگە تازا بىر چىلىشىپ ، يانچۇقىڭىزنى تەتۈر ئۆرۈپ كەلمىسەم ھېساپ ئەمەس  !!!!
__ ھە شۇنداق قىلىپ ئەرلەرنىڭ كۆزىدىن ئوت چىقاردۇق دەڭلار ، ئىسسىق ئۆتۈشۈپ قالمىغاندۇ ؟
__ ۋاي نىمىسىنى دەيسىز ، بىرەيلەن باركەن بەزدەك قارايدىغان ، سىنجىلاپ قاراپ باقسام مۇنبەردىكى ئۇيغۇرۇم ئىكەن  ئەمەسمۇ ؟!
__   ئۇيغۇرۇمنى  كۆرۈپ بەلكىم ھەممىڭلار خوش بۇلۇپ كەتكەنسىلەر ؟ شۇ بانادا باشقىلار بىلەنمۇ تۇنۇشۋالغۇدەكمىز دىيىشىپ .
__ مەنغۇ شۇنداق ئويلۇغان ،ئەمما مېھرىيە  بىلەن  مۇسلىنە  قايتايلى دەپ تۇرۇۋالدى . يۇلتۇزنىڭ  تازا ئاچچىقى كەلدى ئۇ ئىككىسىگە !
__ نىمىشقا ئاچچىقى كېلىدىكەن ؟ سىلەر بەك يىقىن ئاچا -سېڭىل ،دوستلاردىن  ئىدىڭلارغۇ ؟
__ ۋېييـــــەي قويىڭە ئۇلارنى . شۇ ئالدىدا كۈلۈشۈپ ۋاي ئاداش، ھەدەش  دەپ قۇيىمىزغۇ ! ئەمىليەتتە ئۇلارنى كىم بىلمەيدۇ . قارىماققا ئەرلىرىگە سادىقتەك قىلغىنى بىلەن . كىملەرگە قاش ئېتىپ يۈرىيدۇ تېخى !
ــ توۋا سىلەر  ئاياللارنى پەقەت چۈشەنمىدىم .
__ ھــە؟ ئەمدى باشقىلارنى چۈشۈنىپ باققىڭىز كېلىۋاتامدۇ ؟ تۈنۈگۈن مۇنبەردىكى جىلۋە  ئۆيىمىزگە كەلسە ،كۆزىڭىزنى ئۈزەلمەي قاراپ كەتتىڭىز ، مېنى كۆرمىدىمىكىن دېمەڭ ، يەرنىڭ ئاستىدىكى يىلاننىڭ كۈشۈشىنى بىلەلەيدىغان ئادەم مەن !
__توۋا ....
ـــ ۋاي نىمىگە توۋا توۋا دەۋېرىسىز ، يالغانمۇ-يە ؟
ــ ھەي ئۇ دىگەن مېنىڭ سىڭلىمدەك قىز . يەنە كېلىپ سەن بىلەن ئاچا -سېڭىلدەك ئۆتىدىغان ئادەم . سەن  زادى كىمگە ئىشىنىسەن ؟ ھەممە ئادەمگە گۇمانى نەزەردە  قاراپلا يۈرگەن ؟؟
__ ھـــوي... بەز يىمەيدىغان مۆشۈك يوق . كىم بىلىدۇ !! ئەمىسە نىمىگە ھاڭۋېقىپ قاراپ كەتتىڭىز ؟؟
ــ مەن ئۇ سىڭلىمغا مەسلىكىم كىلىپ قارىدىم قانداق ؟
ـــ مانا - مانا دىمىدىممۇ ، ۋاي ئۆلەي ، ئەمدى مەن كۆزىڭىزگە سەت كۆرۈنۋېتىپتىمەن-دە ؟!
__ توۋا ...
ــ يەنە نىمىگە توۋا . جاۋاپ بېرەلمىگەندىكىن شۇ <توۋا> نى مالال قىلدىم دەڭە مۇنداق !
__ھەي خوتۇن سېنىڭ ئەقلىڭ بارمۇ -يوقمۇ ؟ مەن ئۇ سىڭلىمنىڭ ئەتلەستىن كېيىۋالغان يارىشىملىق كۆينىكىگە قاراپ ، بىر چاغلاردىكى سېنىڭمۇ ئاشۇ ئەتلەستىن كۆينەك كىيىپ ، چېچىڭنى ئۇششاق تال چېۋىقتەك ئۆرۈپ ، چىمەن دوپپاڭنى بېشىڭدىن ئايرىماي يۈرگەن چاغلىرىڭ ئېسىمگە كېلىپ قالدى .
ـــ ھە مۇنداق دەڭ ، مۇنۇ نالەمنىڭ   باش سۆرىتىدەك چاغلىرىمنى دەيسىزغۇ . لېكىن ئۇنداق كىيىنسەك ھازىر كىشىلەر بىزنى مـــەتـــــــــــــــــــــۇ  دىيىشىدىكەن ئەمەسمۇ ؟!
ـــ كىم شۇنداق دەيدىكەن ؟ ئۇنداق گەپنىمۇ يەنە شۇ سىلەر خوتۇنلار دىيىشسىلەر . شۇڭا سىلەرگە قاراپ توۋا دەيمەن -دە مەن ! بىز ئەرلەرنىڭ كۆزىمىزگە (مېنىڭچە بارلىق ئەرلەرنىڭ كۆزىگە شۇنداق كۆرۈندىغاندۇ) ، يەنىلا ئاشۇ ئۇزۇن سۇمبۇل قارا چېچىڭلار ، ئۇزۇن يارىشىملىق كۆينىكىڭلار  ئىسسىق كۆرۈنىدۇ .
ــ شۇنداق دەيدىكەنسىلەريۇ ،ئاشۇنداق مودا كىيىنۋالغان خانىم -قىزلارنى كۆرگەندە ، ھەتتا ئۇلارنىڭ قارىسى يۈتكىچە شالۋىقىڭلارنى  ئېقىتىپ قاراپ كېتىدىكەنسىلەر . شۇڭا ئۆتكەندە بازادا بىزنىڭ ئالدىمىزدىن ئۆتۈپ كېتىۋاتقان  مودا سەتەڭگە بوينىڭىزنى غازنىڭ بوينىدەك سۆزۈپ قاراشلىرىڭىزغا قاراپ <ياخشىراق قاراپ كۆزىڭىزنىڭ  پىتىنى ئۆلتۈرۋىلىڭ بوۋاي > دېگىنىم .
ــ ھەي خوتۇن ، مەن ئۇ مودا سەتەڭگە  سېنىڭچە ئاشۇ ئالۋاستى سۈپەت تۇرقىغا  مەسلىكىم كىلىپ قارىغاندىمەنما ؟ ـــ
__ ھەي بىر نىمە دىمەك تەس !!
ــ ھەي بولدىلا سەن بىلەن تالاشماي .  بۇ توغرۇلۇق مۇنبەردىكىلەر بىلەن مۇنازىرە قىلىشىپ باقايلى ، قېنى باشقىلار قانداق قارايدىكىن .
ــ بۇ ئەقىل جايىدا بولدى ، ئەقلىڭىزگە بىر مۇش بوۋاي . مۇشۇنداق يەرلىرىڭىزنى دىمىسەم ...
ـــ ھـــە قانداق قىلماقچى ؟
ـــ ياق -ياق ھېچنىمە قىلماقچىغۇ ئەمەس . شۇ شۇ سىزگە تېگىپ نىمە قىلاتتىم دىمەكچى .
ــ ھى ھى ھى ،گېپىڭ  راست بولسا مۇنبەردىكى باش سۆرىتىڭنى ئالماشتۇرۋەت ، ئۆزىمىزنىڭ ئۇيغۇر قىزلىرىنىڭ باش سۆرىتىگە .
ــ نىمىشقا ئەمدى ؟_ ئادەتتە ئۇيغۇرلىرىمىزغا خاس كىيىنىپ يۈرەلمىسەكمۇ ،مۇشۇنداق يەرلەردە بولسىمۇ ئۆزىمىزنىڭ مەدىنيىتىنى تەشۋىق قىلايلى خوتۇن .
ــ ۋاي -ۋۇي بۇ بىر ياسالمىلىققۇ !
__ نېمىسى ياسالمىلىقكەن ؟
__ بىلمىدىڭىزما ، مۇنۇ يۇلتۇزغا قارىڭە ، بىر چاغلاردا ئاشۇنداق ياسالمىلىق قىلىپ ، ئىككى كۈن ئۇرغۇي قىزلىرىدەك چىرايلىق باش سۆرەت ئىشلىتىپتى،  بانكىر بىلەن دائىم مۇنبەرگە <سۈف > دەپلا ئىنكاس يوللايدىغان بىر ئەزا بارغۇ ، ئىسمى نىمىتى ئەزىزانمۇ ئىشقىلىپ ، شۇلارنىڭ ئاغزىدىن چىققان گەپنى ئۇنتۇپ قالغان ئوخشايسىز .
شۇڭا يۇلتۇز  ھازىر بىر يېرىم كۈندە بىر قېتىم باش شەكلىنى ئۆزگەرتىدۇ .  مودا قىزلارنىڭ بېشىنى كېسىۋېلىپ تازا !
__ يا ئاللا يا ئاللا ... ئۇرغۇي ئەمەس ئۇيغۇر دىدىم باياملا ساڭا . يەنە كېلىپ ، باشقىلارنىڭ ئىشىغا بەك ئارلىشىپ نېمە قىلاتتىڭ . ئالدى بىلەن ئۆزىمىزدىن باشلايلى ماقۇلما ؟
__ ۋاي ماقۇل ، ئەتىلا دوپپا كىيىپ ئالدىڭىزدا پەيدا بۇلىمەن . يەنە كېلىپ مېھرىيە  دېگەن قۇسۇرچىغىمۇ دەپ قويمىسام ، ئەتلەس كۆينىكىىزنى كېيىۋېلىڭ دەپ .
__ يەنە مېھرىيەنى چىشلەپ تارتتىڭغۇ ؟ ھەي بولدىلا  قوساق ئېچىۋاتىدۇ خوتۇن ، ئوخشۇتۇپ بىر لەغمەن ئەتكىنە ، سېنىڭ لەغمىنىڭنى يىمىسەم تاماق يىگەندەك بولمايدىكەنمەن .
__ يۈرىكىڭىزدىن چىققاندۇ بوۋاي ھە ماۋۇ گەپ ؟
__ نېمە دەيدىغانسەن خوتۇن ، ئەلۋەتتە يۈرىكىمدىن چىقماي ، پايپاقنىڭ قونچىدىن چىقامتى ئەمدى ؟!
__ ھى ھى ھى ، ئەمىسە مەن تاماق ئەتكىلى كىرىپ كەتتىم . تېلفۇنىڭىزنى بەرسىڭىز بوپتىكەن ؟
__ نېمە قىلىسەن ، ئۆزەڭنىڭچۇ ؟
__ ئۆزەمنىڭمۇ بار ئەمما ناخشىلىرىنى تولا ئاڭلاپ كەتتىم . سىزنىڭكىدە يېڭىسى بولسا ئاڭلاي دەيمەنغۇ ، يەنە كېلىپ مەن تاماق ئەتكەن چاغدا ، تېلفۇنىڭىز قۇچقاچتەك ۋىچىرلاپ ، ئارامىڭىزنى بۇزمىسۇن دەيمىنا !
__ ۋاي بولدى ،تولا گەپ قىلماي تامىقىڭنى ئەتكىنە بىر
__ ئەمىسە سەيلەرنى سىز تازىلغاچ تۇرامسىز   ،مەن سىرتقا ماڭدىم ، بىردەمدىلا كىرىمەن .
__ نەگە ؟
__  سىرتتىن  تەييار قىلىپ قويغان لەغمەن پىلتىسىنى ئالغىلى چىقىمەن .
__ ئۆزەڭ خېمىر يۇغۇرۇپ ئەتسەڭ بولماسمىدى ؟!
__ ۋاي بىچارە قوللىرىمغا ئىچىڭىز ئاغرىسۇن ، خېمىر يۇغۇرسام قوللىرىم سەتلىشىپ كەتمەمدۇ !
__ توۋا ! ھازىرقى خوتۇنلارغا راستلا ھەيران قالمىدىم جۇمۇڭلار . ھە بوپتۇ تېزراق بۇلاسەن .

چۈشۈدە ئوغلى ‹دادا ، دادوي › دەپ چاقىرامىش . بۇنىڭدىن خوشاللىقىدا نېمە قىلىشىنى بىلەلمەي قالغانمىسرانىم ئەپەندىم ، ئاپىسى سەكىلىخان مومايغا تېلىفۇن قىلىپتۇمىش ؛
__ ئاپا ، ئاپــا ساڭا خوش خەۋەر ، بۈگۈن ئوغلۇم مېنى دادا دىدى  .
__ ۋاي بالام ، سەن ئەسلىدىغۇ ئۇغلۇڭنىڭ دادىسىغۇ بالام ؟
__ ۋاي جېنىم ئاپا  راست دەيسەن ، ئۇ مېنىڭ بالام ، مەن ئۇنىڭ دادىسى  .  ئەمما ، ئۇ ئىلگىرى مېنى دادا دىمەي ‹بابا › دەيدىغان . بۈگۈن كۈن قاياقتىن چىقتىكىن ، تۇيۇقسىزلا دادا دەپ چاقىرۋىدى ، ئۆزەمنى بىر قۇتان قوينىڭ ئىچىگە ئۆچكە كىرىپ قالغاندەكلا ھېس قىلىپ كەتتىم دىگىنە .
__ ۋاي جېنىم بالام ، ئەمدى بالاڭنىڭ ئەقلى تۆشۇپتۇ . كېچە -كۈندۈز قىلغان دۇئالىرىم زايە كەتمىگەن ئوخشايدۇ بالام (مىچىلدىغان ئاۋاز )
__ ۋاي -ۋۇي نېمىگە يىغلايسەن  ، خوشال بولايلى ماقىما ئاپا . كەچتە ئۆيگە كېلىپ ، ئوخشۇتۇپ قازان پۆرىسى ئېتىپ بەرسەڭ ، ھەممىمىز ئولتۇرۇپ بىر تەبرىكلىمەمدۇق . قانداق دىدىم ئاپا ؟
__ پۆرىنى مەن ئېتەمدىم بالام ، ئاۋۇ ئۆيىڭدىكى خوتۇننى قوچاقتەك ياساپ ، مۇزىيخانىغا ئاسارە -ئەتىقە قىلىپ ، تىزىپ قويغىلى ئالغانمىدىڭ .
__ ئۇنداق ئەمەس ئاپا ،  سەن ئەتكەن تاماقنى بەك سېغىنىپ كەتتىم .شۇڭــــــا ( ئۇلۇغ كىچىك تىنىش )
__ ماقــــۇل بالام  . مەن كەچتە بېرىپ ئىتىپ بىرەي .  مېنى ئالغىلى كېلىمەن دەپ ئاۋارىمۇ بولما ، ئۆزەم ئاپتۇۋۇزدىلا باراي . ماشىناڭ سۇدا ماڭمايدۇ ئەمەسمۇ ؟!
__ ئەمىسە ئۆزەڭلا كېلەمسەن ؟ ھە ماقۇل ئاپا ، سېنىڭ گېپىڭنى يېرىشقا ھەددىم ئەمەس .

مىسرانىم  ئەپەندىم تېلىفۇننى قۇيۇپ خوشاللىقىدا بىر سەكرەپتىكەن ، كىروسلودىن داققىدە يەرگە چۈشۈپ ، ئۇيغۇنىپ كېتىپتۇ .
بۇ ئەھۋالغا كۈلكىسى قىستاپ كەتكەن بەگدۆلەتخان  ، تالاغا چىقىپ راسا بىر كۈلۈپتىكەن ، مىسرانىم  ئەپەندىمنىڭ ئاچچىقى قىرىق گەز ئۆرلەپ  ، پەينەك بېشىدىكى ئۈچ -تۆت تال چېچى تىك تۇرۇپ كېتىپتۇ . ھەم ئايالىنى  تازا ئەدەپلەپ قويمەقچى بۇلۇپ يۈگۈرۈپ تالاغا چىقىپتۇ -يۇ ، خوتۇننىڭ يېنىغا كېلىشىگە ئال كۈچۈكلىرى تاققا قېچىپ كېتىپتۇ .........

ۋەئەلەيكۇم -ئەسسالام .
كەسپىم تىلشۇناسلىق ئەمەس ، مەن بۇ ساھەدىن بەكمۇ يىراقتا .
بۇ تېمىنى پەقەت ئۆزەم كومنىڭ ئالدىدا تاراقشىتىپ ئۇرۇپلا يوللاپ قويغان . ئىملا خاتلىقىدىن نېرى بۇلالمايمەن . ئۆزەمگە خاتا بىلىنگەن يەرلىرىنى شۇ يۈزەكى ھالدىلا تەھرىرلەپ قۇيۇمەن .
خىزمىتىم مالىيە بولغاچ ، يەنە كىلىپ بۇ ئانا تىل يىزىقىمنى خىزمەت ئورنىمدا ( مەن يالغۇزلا ئۇيغۇر ) ئىشلەتمىگەچ چالا ساۋات ئادەمگە ئايلىنىپ قالغىلى تۇردۇم . شۇ ۋەجىدىن مۇشۇ مۇنبەر ئارقىلىققا يېزىقىم بىلەن مۇنازىرىدە سىلەردەك قېرىنداشلىرىم بىلەن پىكىر ئالماشتۇرالايمەن . سىز دېگەن فورت دېگەن قانداق دېتال بولغۇيتتىكىن ، بۇنىمۇ بىلمەيدىكەنمەن .
ئەرزىمەس تېمىلىرىمغا ئىزچىل تۈردە پىكىر قاتناشتۇرۇپ كىلىۋاتىسىز . تېتىملىق ئىنكاسلىرىڭىز  ئۈزۈلۈپ قالمىسۇن .
