ۋەئەلەيكۇم ئەسسالام مۇسلىنە خانىم،بەكمۇ ياخشى تەكلىپنى بېرىپسىز .مەن بەك خۈش بولدۇم.
مەنمۇ يېزىپ باقاي؛
  ئەبەيدۇللا ئىبراھىمنىڭ؛(تۈندىكى چاقماق) رومانىدىن؛

     توغراق-تۈرلۈك-تۈمەن خىسلەتلەرنى ئۆز جىسمىدا جەم قىلغان ئاجايىپ خاسىيەتلىك دەرەخ.تەبىئەت دۇنياسىدىكى مەۋجۇداتلارنىڭ خىسلەتلەرىگە ئادالەت كۆزى بىلەن قارايدىغان،كۆڭلى-كۆكسى دېڭېزدەك كەڭ كىشىلەرنىڭ ھەرقاندىقى توغراقنىڭ بۇ خىسلەتلىرىگە قايىللىق بىلدۈرنەي قالمايدۇ.
`ئۇ ئىشچان،باتۇر،ئۈمۈتلۈك ئۇيغۇر خەلقىنىڭ يىشىل ئابىدىسىدەك،چۆل-جەزىرىلەردە قەيسىرانە قەد كۆتۈرۈپ تۇرىدۇ.ئۇنىڭ مەۋجۈتلىقىغا تارلىقى تۇتۇپ،ئۇنى تەبىئەت ئالىمىدىن يوق قىلىشقا ئاتايەن ئاپىرىدە بولغان،ياۋا توڭگۇزدەك چىقىراپ،قۇتراپ تۇرىدىغان بورانمۇ ئۇنى كۆمىۋېتەلمەيدۇ.ئۇ يەر چوڭقۇرلۇقىدىن ئوزۇق سۈمۈرۋالىدۇ-دە،مەۋجۇتلۇقىنى دۈشمىنىنىڭ كۆزىگە كۆرسىتىپ قويماقچى بولغاندەك،بارغانسىرى باراقسانلاپ،ئاسمان-پەلەك بوي سوزىدۇ.تۈۋىدە بوران باشلاپ كەلگەن قۇم دۆۋىلىرى بىر گەز ئۆرلىسە،ئۇمۇ كۆك قەرىگە قاراپ بىر گەز كۆتۈرلىدۇ.لەنىتى بىچچەرە قۇم پەستە قالدۇ.توغراق بولسا،ھامان ئىگىزدە تۇرىدۇ ۋە قېلىن يوپۇرماقلىرىنى ،شىلدىرلىتىپ،ئۆزىنى چاغلىمىغان قۇم ئۈستىدىن مەسخىرە قىلىپ كۈلىدۇ.
سۆيۈشنى بىلمەي تۇرۇپ،سۆيۈلۈشتىن ئۈمۇت كۈتمە!

مەنمۇ ئۆزۈم ئوقۇغان كىتابلار ئىچىدىن ماڭا تەسىر قىلغان جۈملىلەردىن يىزىپ قوياي :
1.بىرەر ئىشتا مۇۋاپىقيەت قازىنىشىڭىزدىن بۇرۇن ، ھەرگىزمۇ يۈز-ئابرۇيغا ھەددىدىن زىيادە ئەھمىيەت بىرىپ كەتمەڭ ؛
2.ئەدەپ-ئەخلاق دىگەن كىچىك-كىچىك ئىشلاردىمۇ قۇربانلىق بىرىش بەدىلىگە كىلىدۇ ؛
3.رىئاللىقتىكى زىددىيەتلەرنىڭ ئۇلۇغۋار غايىلەر بىلەن ھەل بولۇپ كىتىشى ناتايىن ؛
4.بۇ دۇنيادا ۋاقىت ھەممىدىن قىممەتلىك ، كۆز ھەممىدىن ئالدامچى ؛
5.سىزنىڭ تىنىڭىزدە يوشۇرنۇپ ياتقان بىر كۈچ بار ، ئۇ سىزنىڭ ئەمەلىيتىڭىز ئارقىلىق ئۇنى ئالەمگە ئاشكارىلىشىڭىزنى كۈتۈپ ياتماقتا ؛
6.ئۆتكەن كۈنلەر ئۆتۈپ كەتتى ، بالام ، ئۇ كۈنلىرىڭ ئەمدى كەلمەسكە كەتتى . سەن قەيسەر بولىشىڭ كىرەك ، ئەس ھۇشىڭنى يىغىشىڭ كىرەك . چۈنكى سەن يەنە بىر مەزگىل ئۇزاق مۇساپنى بىسىشىڭ كىرەك ، بۇ يولنى قانچىلىك بىسىشتىن قەتئىينەزەر، سەن ئاۋۋالقىدىنمۇ چىرايلىق مىڭىشىڭ كىرەك ؛ بالام ، ئۈمىدسىزلەنمە ، سەن ماڭىدىغان يول يەنىلا شۇنچە ئازادە ، سەن جۈرئەتلىك بولۇشۇڭ كىرەك ، جۈرئەتلىك بولۇشۇڭ كىرەك ، جۈرئەتلىك ھالدا بۇ مۇساپەڭنى مىڭىپ تۈگىتىشىڭ كىرەك . (بۇ بىر ئانىنىڭ راك كىسىلىگە گىرىپتار بولۇپ ئۆلۈم گىرداۋىغا بىرىپ قالغان 9 ياشلىق بالىسىنى رىغبەتلەندۈرگەندە ئىيتقان سۆزى . بالىسىغا خىمىيلىك داۋالاش توختىتىلىدۇ ، بالا ئاغرىق ئازابىغا چىدىماي ئاپسىدىن ئۆزىنىڭ تىنچ ئۆلگۈسى بارلىقىنى تەلەپ قىلىپ تۇرۇۋالغاندا ئانىسى شۇنداق رىغبەتلەندۈرىدۇ ، لىكىن ئاخىرى ، يەنىلا بالىسى ئۈچۈن تىنچ ئۆلۈش يولىنى تاللايدۇ  . ئالىي مەكتەپ ئىمتىھانىغا تەييارلىق قلىۋاتقىنىمدا بۇ گەپلەرنىڭ خۇددى ماڭا دەۋاتقاندەك ، مىنى قانچە رىغبەتلەندۈرگىنىنى بىلمەيمەن ، ئىھتىمال ھەممە ئانىلارنىڭ مۇھەببىتى ھەممە بالىلارنى رىغبەتلەندۈرەلىسە كىرەك )
7.بەزى ئادەملەرنىڭ دوستلىشىشى خۇددى ئىتنىڭ سۆڭەك ئىزدىشىگە ئوخشايدۇ ، پايدا چىقمىغاندا ئۇ بەرىبىر سىزنى تاشلاپ كىتىپ قالىدۇ .  
8.ئادەملەر دوستلىرى بىلەن باردى-كەلدىسى ئۇزاق بولغانسىرى ، دوستلار ئارا ھىسيات ئاشكارىلىشى قانچە كۆپ بولغانسىرى شۇنچە ئاسان قارشى تەرەپنىڭ كەمچىلىكىنى بايقىشىدۇ .
9.مۇكەممەل ،ساغلام بولغان پىسخىك ھالەتنىڭ مەلۇم دەرىجىدە راك كىسىلى ۋە يۈرەك-مىڭە كىسەللىرىنى داۋالاشتا پايدىسى بار.

(ھەممىسىنى خەنزۇچە كىتابلاردىن ساقلىۋالغان ، مەنبەسى قايسى كىتابلار يازماپتىمەن )
چىكىدىن ئاشقان ھەرقانداق ئىش ئەكىس تەسىر كۆرسىتىدۇ .

‹‹سالچىلار ››رومانىنىڭ داۋامى



سۇۋاقلىرى چۈشۈپ ئىچىدىكى ياغاچ -كېسەكلىرى كۆرۈنۈپ قالغان كونا تاملار،ئىشىكلىرى قاغجىراپ،قىيسىيىپ كەتكەن دېھقان ئۆيلىرى،ئۆينىڭ ئارقىسىدا دانلاپ يۈرگەن  توخۇلار،يوغان ئۈجمىنىڭ سايىسىدە ھورۇنلۇق بىلەن كۆشەپ ياتقان كالا،قىيقاس -چۇقان سېلىپ توپىلىق يولدا يالىڭاچ دېگۈدەك ئويناۋاتقان ئۇششاق بالىلار ،پاكار -پاكار تاملار ئارقىسىدىن كۆرۈنۈپ تۇرغان مېۋىلىك دەرەخلەر،يوپۇرماقلار ئارىسىدىن شوخ قىزلاردەك مارىشىپ تۇرغان ئالما -ئانارلار،مەھەللىنىڭ بىر چېتى ياكى قاپ ئوتتۇرىسىدا تىنىق ئاسماندىىك ئاپئاق بۇلۇتلارنى ئۆزىدە ئەكس ئەتتۈرۈپ تېپتىنچ سوزۇلۇپ ياتقان كۆل.....بۇلارنىڭ ھەممىسى سىزگە راھەتلىك بىر يېنىك تۇيغۇ بېغىشلايدۇ -دە ئىختىيارسىز قەدىمىڭىزنى توختۇتۇپ،يول ياقىسىدىكى قايسىبىر دېھقان قايسىبىر يىلى نېمە سەۋەبتىندۇر بۇ يەرگە ئەكىلىپ تاشلا قويۇپ  ئۇنتىغان،كېيىن كىشىلەر ئۇنىڭ ئۈستىدە ئولتۇرىۋېرىپ سىلىقلىشىپ كەتكەن ياغاچ ئۈستىدە بىردەم ئولتۇرۇپ ئارام ئالغۇڭىز كېلىپ قالىدۇ.شۇ چاغدا كۆڭلىكىنىڭ ئېتىكىنى ئەپچىللىك بىلەن يىغىپ، كۆلدىن سۇ ئېڭىشىپ سۇ ئېلىۋاتقان گۈزەل سەھرا قىزى ئىختىيارسىز دىققىتىڭىزنى تارتىدۇ.قىز چېلىكىنى سۇغا تولدۇرۇپ قەددىنى رۇسلىغاندا گويا كۆلنىڭ ئەتراپىدىكى شاخلىرىنى سۇ يۈزىگە ئېگىپ تۇرغان سۆگەتلەرمۇ قەددىنى رۇسلىغاندەك بولۇپ ،يوپۇرماقلار ئارىسىدا مۈگىدەپ ئولتۇرغان قۇشقاچلار ئىختىيارسىز چۇرۇقلىشىپ كېتىدۇ.قىزنىڭ چېلىكى ئەتراپىدا پەيدا بولغان دۈپدۈگىلەك دائىرە قارشى قىرغاققىچە كېڭىيىپ ،چېلەكتىن تۆكۈلگەن سۇ تامچىلىرىدىن سانسىز ھەسەن -ھۈسەنلەر پەيدا بولغاندەك بولىدۇ.قىز ئەپكىشىنى مۆرىسىگە ئېلىپ لەرزان تەۋرىنىپ مېڭىپ كەتكەندە نېمىشقىدۇر ئاۋازىڭىزنى بولىشىغا قويۇۋېتىپ ناخشا ئېيتقۇڭىز كېلىپ قالىدۇ....بۇ يەردە ھەممە نېمە شۇنچىلىك ئاددىي،شۇنچىلىك گۈزەل،مۇشۇنداق ئازادە يېزىلار تۇرغان يەردە،يەنە نېمىشقا ئادەملەرنىڭ شۇنچە دىمىقتا شەھەرلەرنى بىنا قىلىپ ،قىستاڭچىلىقتا تىقىلىپ ياشايدىغانلىقىنى ئويلاپ يېتەلمەيسىز.


داۋامى بار....
