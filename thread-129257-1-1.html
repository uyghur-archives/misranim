ئەسسالام تورداشلار كۆپچىلىك بىلەن بۇيەردە كۆرۈشكىنىمدىن ئىنتايىن خوشال بۇلىۋاتىمەن.مىسرانىم تور بىتىگە دائىم كىرىپ تۇرساممۇ ئەمما ۋاقىت مۇناسىۋىتى بىلەن ئەزا بولالمىغان ئىدىم.يىقىندا ئالدىراشچىلىق ئىچىدىن ۋاقىت چىقىرىپ ئەزا بۇلۇش شەرىپىگە مۇيەسسەر بولدۇم.بۇنىڭدىن كىيىن تورداشلار بىلەن كۆپ پىكىر ئالماشتۇرۇپ تۇرۇشنى ئۈمىد قىلىمەن.
يىقىندا ئوغلۇمغا ئاتاپ يازغان بىر پارچە يازمامنى ئۈندىدار دوستلۇق چەمبىرىكىدە ئېلان قىلغان ئىدىم ،نۇرغۇن دوستلارنىڭ قارشى ئىلىشىغا ئېرىشتى. ئانا بۇلۇش سۈپىتىم بىلەن بىر ياشقا كىرىش ئالدىدا تۇرىۋاتقان ئوغلۇمغا سوۋغا قىلىش ھەمدە ئاتا-ئانا بولغان ھەم بۇلۇش ئالدىدا تۇرىۋاتقان بارلىق قېرىنداشلىرىم بىلەن ئورتاق بەھىرلىنىش مەقسىتىدە كۈپچىلىككە سۇندۇم.

    مەن ئۈچۈن ئەڭ قەدىرلىك بولغان سۇبات ئوغلۇم سىنىڭ كۈندىن-كۈنگە ئۇماق چوڭ بۇلىۋاتقان ھالىتىڭنى كۆرۈپ خوشال بۇلۇپ تۇرغان مۇشۇ كۈنلەردە يۇرتىمىزدىكى توختىماي يۈتكەن بالىلارنىڭ خەۋەرلىرى شۇنداقلا پەلەستىندىكى سەبىي بالىلارنىڭ ھەم مۇسۇلمان قېرىنداشلىرىمىزنىڭ ھايۋاندىن بەتتەر يەھۇدىيلار تەرىپىدىن ئۆلتۈرۈلگەنلىكلىرى ئىچىمنى ئىزىپ لەختە-لەختە قىلىۋەتتى.ئىنسان دىگەن شۇنداق ئاجىز ئىكەن ،مۇسۇلمان قېرىنداشلىرىمىزنىڭ قىرىلىپ كىتىشلىرىگە،ئۇيغۇر بالىلىرىمىزنىڭ،كەلگۈسى ئۈمىدلىرىمىزنىڭ ھايۋان كەبى ئادەم بىدىكلىرى تەرىپىدىن ئالدىنىپ كىتىلىپ ھىچقانداق خەۋىرى يوق ئىز-دېرەكسىز يۇقىلىپ كەتكەنلىكىنى كۆرۈپ ئولتۇرۇشتىن باشقا ئامال قىلالمىدۇق.ئەمما ئوغلۇم بۇ دۇنيادىكى پۈتۈن ئىشنى ھەممىگە قادىر ئاللاھ كۆرۈپ تۇرىدۇ.مۈڭگۈزى بار قوي مۈڭگۈزى يوق قوينى ئۈسىۋالغان بولسا،قىيامەتتە مۈڭگۈزى يوق قوينىڭ ئىنتىقامىنى ئىلىپ بېرىدىغان ئاللاھ شۇ نارسىدە ،سەبىيلەرنىڭ ئىنتىقامىنى ئىلىپ بەرمەسمۇ؟
    ئوغلۇم ئاللاھنىڭ مۆجىزىلىرى قاتارىدىن بۇ دۇنياغا كۆزئىچىپ ئانا-بالا بولدۇق. بۇ دۇنيا بەزىدە رەھىمدىللىككە،مۇھەببەتكە تولغان. بەزىدە رەھىمسىزلىككە، ئىنسىاپسىزلىققا تولغان.ئەڭ مۇھىمى سىنىڭ ئۇنىڭغا قانداق مۇئامىلە تۇتىشىڭدا.
    ئوغلۇم سىنىڭ ئاپاڭ بۇلۇش سۈپىتىم بىلەن ئەلۋەتتە سەندىن ھەرخىل ئۈمىدلەرنى كۈتىمەن.سىنىڭ ئۆزەم ئارزۇ قىلغاندەك چوڭ بۇلىشىڭنى ئۈمىد قىلىمەن.بۇ ھەرقانداق ئانىدا بار بولغان ئورتاق ماھىيەت. مەن ساڭا ‹يۇقىرى ئۇنۋان ئىلىش ئۈچۈن ،ياخشى ئورۇنلاردا ئىشلەش ئۈچۈن ياخشى ئوقۇ›دىگەندەك بىر تەرەپلىمە قاراشلارنى تاڭمايمەن. بەلكى ساڭا پەقەت ئاتا-ئاناڭنى ياراتقان ھەممىگە قادىر ئاللاھنى ياخشى كۆر دىگەن تەلەپنى قۇيىمەن.چۈنكى سەن ئاللاھنى ياخشى كۆرسەڭ،ئاللاھ ياراتقان بەندىلەر قاتارىدىن چوڭ داداڭ، ئاناڭلارنى،داداڭ،ئاپاڭ،ئۇرۇق-تۇققان قېرىنداشلىرىڭنى ياخشى كۆرىسەن.ئەتراپىڭدىكىلەرنى ياخشى كۆرسەڭ ھاياتىڭ مېھىر-مۇھەببەتكە تولىدۇ.مېھىر-مۇھەببەت ئىچىدە چوڭ بولساڭ ئاق كۆڭۈل،باشقىلارغا ياردەم بېرىشنى خالايدىغان بۇلۇپ يىتىشىپ چىقالايسەن.ئاللاھ تائالا پەيغەمبىرىمىز مۇھەممەد سەللاھى ئەلەيھى ۋەسسالامغا بىرىنچى قىتىم ئوقۇ دىگەن ئايەتنى چۈشۈرۈپتىكەن.ئوغلۇم ئاللاھنى ياخشى كۆرۈپ ئاللاھ يولىدا ماڭساڭ تەبىئيلا ئىلىم تەھسىل قىلىشنىڭ قانچىلىك مۇھىم ئىكەنلىكىنى،بىلىم ئۆگىنىش ئۈچۈن ۋاقىتتىن پايدىلىنىشنىڭ نەقەدەر زۆرۆرلىكىنى ھىس قىلالايسەن..دىمەك ئوغلۇم ئاللاھنى ياخشى كۆرسەڭ بەخىت سىنى ئۆزلىكىدىن ئىزدەپ كىلىدۇ.
    ئانا بولغان 10ئاي جەريانىدا كۆپ ۋاقتىمنى ساڭا بەردىم.بۇنى كۆرگەن بەزىلەر‹بالا دىگەن بىر ئىككى كۈندىلا چوڭ بۇلۇپ قالمايدۇ،ئۆزەڭگىمۇ ۋاقىت بەر›دىدى.بىلىمەن ئۇلارنىڭ ماڭا كۈيۈنۈپ دىگەنلىكىنى،بىراق مىنىڭ ئۇلارغا شۇنداق دىگۈم بار:‹بالىنىڭ چوڭ بۇلىشى ئۈچۈن ناھايىتى ئۇزاق بىر جەريان ،ئۇزاق ۋاقىت كىتىدۇ.ئەمما بالىنىڭ بىر ياش ۋاقتى،ئىككى ياش ۋاقتى ....ھەم تۇنجى قىتىم ماڭغان ۋاقتى،تۇنجى قىتىم ئاپا دىگەن ۋاقتىدىن قانچە قىتىم بار؟پەقەت ۋە پەقەت بىرلا قىتىم.›شۇڭلاشقا ئوغلۇم مەن سىنىڭ ئاللاھ خالىسا ھەممە تۇنجىلىرىڭنى ئۆز كۆزۈم بىلەن كۆرگۈم بار.ھەم يەنە شۇنداق ئاپا بولغۇم باركى،يىقىلىپ چۈشكەن ۋاقىتلىرىڭدا بېرىپ يۈلەش ئەمەس بەلكى‹بالام،ئۆزەڭ تۇرۇپ كەت.سەن باتۇر ئوغۇل بالا نەدە يىقىلىپ چۈشسە شۇيەردە ئورنىدىن تۇرىدۇ.›دىيەلەيدىغان ‹‹باغرى تاش ئانا››بولالىشىمنى،ئۆزەڭ ياشاۋاتقان مۇھىتقا قىزىقىپ ئۇنى-بۇنى سورايدىغان بولغان ۋاقتىڭدا شۇ تۈگىمەس سۇئاللىرىڭنى جاۋاپسىز قالدۇرماي،زېرىكمەي تېرىكمەي قايىل قىلارلىق جاۋاپ بېرەلەيدىغان سەۋىر-تاقەتچان،بىلىملىك ئانا بولالىشىمنى،چەت دۆلەتلەرگەچىقىپ ئوقۇشنى تاللىغان ۋاقتىڭدا بىر مىنۇتمۇ ئايرىلشقا كۆزۈم قىيمايدىغان بالا دەپ سوقىۋاتقان يۈرىكىمنى قاتتىق قىلىپ پىشانەڭگە سۈيۈپ تۇرۇپ :‹بالام، ياخشى ئوقۇپ كىلىپ مىللەتكە ياراملىق ئادەم بولغىن ›دىيەلەيدىغان،پەرزەنتلىك بولغان ۋاقتىڭدا :‹ ئوغلۇم، پەرزەنتىڭنى مەن بېقىپ بېرەلمەيمەن،سەن ئۆز قۇلۇڭ بىلەن تەربىيلە.چۈنكى شۇنداق قىلساڭ ئاتا-بالىلىق مېھىر-مۇھەببىتىڭلار چوڭقۇرلىشىدۇ.سەنمۇ بالاڭغا قاراپ ئۆزەڭنىڭ مەسئۇلىيىتىڭنىڭ نەقەدەر ئىغىرئىكەنلىكىنى چۈشىنىسەن.خۇددى داداڭ بىلەن ئىككىمىز سەن ئارقىلىق ئاتا-ئانا بۇلىشنى ئۆگەنگەن ھەم ھىلىھەم ئۆگىنىۋاتقاندەك ،سەنمۇ ھەم قانداق ئاتا بۇلۇشنى پەرزەنتىڭ ئارقىلىق ئۆگىنىسەن›دىيەلەيدىغان ئانا بولالىشىمنى ئۈمىد قىلىمەن،تىرىشىمەن،ئاللاھدىن تىلەيمەن.
    ئوغلۇم،ياقا يۇرتلاردا ياشاۋاتىمىز.بىر كۈنلەردە ئازراق ئىش بىلگۈدەك بولغان ۋاقىتلىرىڭدا ئەتراپتىكى كىشىلەرگە قاراپ چىرايىمىزنىڭ،تىلىمىزنىڭ،دىنىمىزنىڭ،ئۆرۈپ-ئادىتىمىزنىڭ ئوخشىمايدىغانلىقىنى ھىس قىلغان چىغىڭدا شۇنداق سوراپ قىلىشىڭ مۇمكىن:‹ئاپا،بىزنىڭ قېرىندىشىمىز كىم؟ئۇلار نەدە؟›دەپ.مەن ساڭا شۇنداق جاۋاپ بېرىمەنكى :‹بالام، پۈتۈن ياپونىيەدىكى مۇسۇلمان ئۇيغۇرلار سىنىڭ قېرىندىشىڭ بۇلۇپلا قالماستىن بەلكى پۇتۇن دۇنيادىكى مۇسۇلمان،ئۇيغۇرلار سىنىڭ قېرىندىشىڭ.چۈنكى داداڭنىڭ،ئاپاڭنىڭ،سىنىڭ تۇمۇرۇڭدا ئېقىۋاتقىنى مۇسۇلماننىڭ قىنى، ئۇيغۇرنىڭ قىنى.›
    مۇشۇ يازمامنىڭ ئاخىرىدا ساڭىلا ئەمەس پۈتۈن دۇنياغا شۇنداق دىگۇم بار.بىزدە ئاللاھقا،دىنىمىزغا،مىللىتىمىزگە بولغان تۇلۇپ تاشقان مۇھەببەتلا بۇلىدىكەن،بىز مۇساپىر ئەمەس،بىز يالغۇز ئەمەس،بىز بىچارە ئەمەس،بىز ئاجىز ئەمەس،بەلكى بىز بۇدۇنيادىكى ئەڭ ئەڭ ئەڭ بەخىتىلىك كىشىلەر.چۇنكى بىزنىڭ ئاھۇ-زارىمىزنى ئاڭلاپ تۇرىدىغان ،بىزگە ھەرزامان مەدەت بېرىپ تۇرىدىغان يۆلەنچۈكىمىز،ھەممىگە قادىر ئاللاھىمىز بار!!!!!!!!
