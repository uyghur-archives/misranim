  ۋاھ ماۋۇ بىخلننىڭ مەرتلىگىنى كۆرۈڭ . بۈگۈن قۇياش قاياقتىن چىقتى سىزگە . ھەر قىتىم پەتىگە كەلسىڭىز پىىسىقلىق قىلىپ ئاران نەچچە تىيىن تاشلايدىغان ئادەم ھەجەپ لەڭپۇڭدىن بىر تەخسىمۇ دىمەي دابو قىلىپ ئەۋەتىپ بىرەيمۇ دەپ كەتكىننى . ماقۇل دىسەم ھەقاچان لەڭپۇڭچى ئۇستامغا ئۇستام  بىر يۈۋەنلىك پۇلغا چۈشلۈق قىلىپ بەك ئېسىل بولۇپ كەتمىسمۇ مەيلى ئىسمى لەڭپۇڭدىن بىرنى قىلىپ بەرسىلە ئىكىسپورىت قىلاتتىم دەپ ئۇششاق خالتىلىق گازىرنىڭ خالىتسىغا بىر خالتا لەڭپۇڭ ئەتكۈزۈپ  مال سالدىغان ئورۇنغا بىرىپ بىر يۈۋەنلىك لەڭپۇڭنى بەش يۈۋنلىك پوچتا خالتىسىغا سىلىپ بولۇپ توشۇش ھەققىنى چىقىم قىلىشقا چىدىماي مال ئەۋەتىدىغانلاردىن مالنى سىلۋەتسەم توشۇش ھەققىنى قارشى تەرەپ بەرسىمۇ بولامدۇ ؟ دەپ ئۇلارنىڭ ئالىدا خىجىللىق تارتىپ يۈرمەڭ بولدى شاتۇتگۈل . ياخشى كۆڭلىڭىز بەك تىگىل بولدى ماڭا . ئىشقىلىپ مۇشۇ گەپلىرىڭىزدىن قولىڭىزنىڭ ئوچۇقلىغى . ئاق كۆڭۈللىكىڭىز . مەرتلىكىڭىز چىقىپ تۇرۇپتۇ    نىمە دەيمەن سىزگە ئىشقىلىپ بەك ھاياجانلىپ كىتىپ گەپلىرىمنى تاپالماي قىلۋاتمە . ئاللاھ بەختىڭىزنى بەرسۇن . بەخىت قۇشلىرى قەلىبىڭزنىڭ قونداقلىرىغا قونۇپ خەندان ئۇرۇپ سايراپ كەتسۇن . قازىنىڭزىدىن سۇ ئورنىغا ماي ماي ئورنىغا  شورپا قايناپ تۇرسۇن ، تۇرخۇنىڭىزدىن  ئىس ئورنغا گاز چىقسۇن  ئاۋازىڭىز تەمبۇردەك ساز چىقسۇن . توپا ئالسسىڭىز توپا ئورنىغا ئالتۇن چىقسۇن . ياز پەسلدە سوغۇقنىڭ دەردىنى قىش پەسلىدە كۆمۇرنىڭ دەردىنى تارتىپ يۈرمەسسىز . يىنىڭزىدىن ئۆتكەن -كەچكەنلەرنىڭ مۇنبەردىكى كۆزى يامانلارنىڭ كۆزىدىن ئاللاھ يىراق قىلسۇن ئامىيىن ئاللاھۇ ئەكبەر !  سۈف...سۈف...سۈف

بەزى چاغلاردا مۇشۇ گەپ- سۆزلەردە بەك ئاشۇرۋېتىدىغان ئ ...



    يا يا بۇنداق ئويلىۋالماڭ مەن سىزنى يىمەك-ئىچمەك تازلىقىغا زىيادە دىققەت قىلغىنىڭىزغا قاراپ مەسلىكىم كىلىپ دەپ قويغان گەپ ئۇ .

    مەن سىزگە قارىغاندا كۆپ ئارقىدا ئىكەنمەن . مەكتەپتە ئوقۇيدىغان ۋاقىتمدا  سىنىپىمزىدا ياش پەرقى بىر-بىردىن ئانچە پەرىق قىلىپ كەتمەيدىغان قوشىكىزەك ئاكا-ئۇكا ئوقۇغۇچى بولدىغان ھەر ئىككىسىىى تاز بولۇپ  بىشىنىڭ كەينىدە تېخى كۆينەكىنىڭ ياقىسىغا تۇتاشقان يىرىدە بىر تۇتام چاچ  بار باشقا يەرلىرى  ساپلا يارا  بولۇپ مۇدۇر-مۇدۇر قىتىپ كەتكەن گادا بىشنى بىر ئالغان بولۇپ بەزى گادالار تولا قىچىشىپ تاتلاش جەرياندا شەلۋەرلەپ ياز كۈنلىرىدە  باشقىلارنى خىجىل قىلىشنى بىر خىل كەسىپلەشكەن ئەخلاق دەپ بىلدىغان قاشاق چىۋىنلەر تولا ئاۋارە قىلاتتى ،  بالىلار ئىغىزىدا چاندۇرماي ۋاي ساقدىىشم ، ئاداش دەپ قويغان بىلەن بىر پارتىدا ئولتۇرشقا . كوللىكتىپ سورۇنلاردا بىر يەردە ئولتۇرشتىن ئۆزىنى چاندۇرماي ئىلىپ قىچىشاتتى . باشقىلار ھىس قىلدىمۇ-قىلمىدىمۇ بىلمەيمەن  بىراق مەن  بۇ ئاكا-ئۇكا ئىككەيلەن ئىغىزىدا كۈلۈپ خۇشالدەك يۈرگەن بىلەن بىر خىل ئۆزىنى كەم سۇندۇرۇپ يۈرىيدىغانلىقىنى چىرايىدىن  بىلۋالاتتىم تە شۇنداق ئىچىم ئاغىرىپ قالدىغان . ئاخىرى ئىنىسى تۆۋەن يىللىقتىكى ئاق-قارىنى پەرىق ئىتىپ بولالمايدىغان بىر قىسىم ئىغىزى ئىتىك بالىلارنىڭ كەيمە تاز   . لاچىن تاز....دىگەندەك مەسخىرە قىلىشلىرىغا چىداپ تۇرالماي تولا سوقۇشۇپ . تىللىشىپ دىگەندەك ئاخىرى ئوقۇشتىن رايى ياندىمۇ يوقالدى . بىراق ئاكىسى ئىزچىل بىز بىلەن بىرگە ئوقۇغان . بىر قىتىملىق تەتىلدىن يىغىلىشتا يىڭى ئوقۇغۇچىلاردىن بىر قانچىنى سىنىپىمىزغا بۆلگەن سەۋەپلىك پارتا يىتىشمەي ھەر ئۈچ ئوقۇغۇچى بىر پارتىدا ئولتۇردىغان قىلىپ ئورۇنلاشتۇرۇش ئۇچۇن بالىلارنى ئۈچتىن قىلىپ بۆلگەندە ئۇ ئاداش بىلەن ھىچ بىرەسى بىرگە ئولتۇرشقا ئۇنماي قويدى .شۇنىڭ بىلەن ئۇ ئاداش بەك خىجىللىقتا قىلىپ يەرگە قارىۋىلىپ تۇرغانتى . مەن مۇئەللىمگە ئۇ ئاداش دەرىسكە ئۇستا مەن بىرگە ئولتۇراي ئۇ ماڭا كۆپ ياردەم قىلىدۇ دەپ تەلەپ قىلىپ بىرگە ئولتۇرغان ئىدىم . مۇئەللىممۇ ئىغىزىدە مىنى ماختتاپ قويغان بىلەن ئۇ ئاداشتىن ئۆزىنى خىلى ئىلىپ قاچاتتى شۇڭا دەرىس ۋاقتىدا ئوقۇغچىلارنىڭ ئىشلەۋاتقان تاپشۇرقىنى نازارەت قىلىش ئۈچۈن پارتىلارنى ئارىلاپ كىلۋىتىپ نەق شۇ تاز ئاداشنىڭ پارتىسىغا كەلگەندە ئاستا كەينىگە ياناتتى . بۇ مەن ئۈچۈن دەرەسلىكتىن سىرىت باشقا ژانىردىكى كىتاپلارنى ئوقۇۋىلىشىمغا بەك ياخشى پۇرسەت بىلەن تەمىن ئەتكەن ئىدى . قىش پەسىلدە سىنىپتا چاي قاينىتىپ ئىچىپ قوياتتۇق . كۆپ ئادەم بىرگە تاماقلانسا داغ چايمۇ ئەتكەن چايدەك بىلىنەمدۇ قانداق شۇنداق ھوزۇر قىلىپ ئىچىپ كىتىدىغان  . نەچچە قىتىم ئۇ ئاداش چاي ئىچىدىغان تەلەڭگىسىنى ئۇنتۇپ قىلىپ مەن چاي تەلەڭگەمدە  بىرگە چاي ئىچكەنتىم . كىيىن چوڭ بولدۇق جەمىيئەتكە چىقتۇق .  كۈندە بەش ۋاقىت نامازنى ئوقۇيالمىساممۇ بىر ۋاق نامازنى ئانامنىڭ قاتتىق ھەيدەكچىلىگى نازارتى ئاستىدا  ئەل-جامائەت بىلەن ئوقۇپ كەتتىم . بەزى كۈنلىرى نەزىر ۋە توي-تۆكۈنلەرگە چىللانغان جامائەت بىلەن بىرگە بىرىپ قالغاندا جامائەت ئىچىدە يۈز-كۆزىدە ۋە ياكى قوللىرىدا ھەر خىل ياللۇغ ، پىسە.  ئاق كىسەل ...دىگەندەك جاراھەتلەر بولۇپ قالغان كىشىلەردىن كۆپ سانلىق بىر پەتنۇسقا قول سىلىپ سىلىشتىن ئەنسرەپ  يىنىدا ئولتۇرشتىن ئۆزىن  تارتاتتى . مەن بەك زىيادە باشقىلارنىڭ كۆڭلىنى ئايايمەنمۇ قانداق مۇشۇلار خىجىللىق ھىس قىلىپ قالمىسۇن دەپ ئەتەي قىزغىنلىق بىلەن ئۇلار ياقتۇردىغان پاراڭ قىلىپ بىرگە ئولتۇرۇپ بىر پەتنۇستا بىرگە تاماق يەپ قوياتتىم . راستىنى دىسەم مەنمۇ شەكلىنىمەن بىراق خۇددى ھەر قانچە ئېسل بىزەلگەن ھاجەتخانىغا كىرىپ ھايال ئۆتمەي چىقساقمۇ رىفلىكىس نۇقتىسىدىن كۆنغپ قالغان ئادەتمۇ قانداق بىرنى تۈكۈرۋىتىپ ماڭغاندەك تاماقتىن پارىغ بولغاندىن كىيىن خالىي جاينى تىپىپ تۈكۈرۋالاتتىم .....ئادەتتە ئۆيگە مىھمان بولۇپ كەلگەنلەر ئىچىدىن بىرەسى تاماق كۆپ بولۇپ قالغان سەۋەپلىك ئازىراق ئاشۇرۇپ قويۇپ ئىشىپ قالغان تاماقنى يەپ بولۇشقا كۆزى يەتمەي داستىخانغا قايتۇرۋىتىشتىن   خىجىل بولۇپ قويالمغىان يەي  دىسە يەپ بولالماي شۇنىڭ بىلەن تاماقنى مۈشۈكتەك ئىمىپ  ئۆينىڭ توت تىمىغا ئوغىرى مۈشۈكتەك قاراپ قالغان مىھماننىڭ تامقىنى شۇ مىھاننىڭ ھۆرمىتى ئۈچۈن بولسىمۇ يەپ قوياتىم . بىراق  ياشقا ياش قىتىلغانسىرى بۇ ئىش ئۆزۈمگە بۇ ئىش بىر ئاز زىيادە قىلىنغان رەسمىيەتچىلىك ۋە تولىمۇ ئۆزۈمنى چۈشۈرۋالغانلىق بىلىنىپ  ئۇزۇن يىللىق بۇ ئادىتىمنى تاشلاپ بۇ ئىشقا چەكلىمە قويدۇم ، يەنى تامىقى ئىشىپ قالغان مىھمان پاكىز چىراي بولسا . مىجەز-خاراكتىرى ماڭا يىقىشلىق بىلىنسە ياش پارقى مەندىن پەرىقلىق چوڭ بولسا ئاندىن يەپ قويدىغان بولدۇم بولمىسا  بولدى خىجىل بولماي تاماقنى داستىخاندا قويۇۋىرىڭ دەل مۇشۇنداق ئىشىپ-تىشىپ قالغان تاماقلارنى مەخسۇس يىسۇن دەپ بەش-ئالتە ساغلىقنى بىقىپ قويغان شۇلارنىڭ رىسقى بۇ دەپ تەڭلىكتە قالغان مىھماننى خىجىلقتىن قۇتۇلدۇرۇپ قويدىغان بولدۇم . بىراق ئانام  بالامنىڭ مىجەزى غەلتە نورمال ئادەملەرنىڭ مىجەزىگە پەقەت ئوخشىمايدۇ قىلغان-ئەتكەن ئىشلىرى غەلىتىلا دەپ كىچىك ۋاقتىمدا مۇشۇ ياسىن سۈينى ۋە ھەر خىل دىمىدە قىلغان سۇنى ماڭا تولا ئۇچۇرۋاتىغان پەيدا قىلىدۇ دەپ . ئۇ دىمىدىگە تۈكۈرگەن كىشىلەرنىڭ چىرايىنى ئويلىسام ھازىر بۇھھھھھھھھھھھ   ( تىتىرەك باسقان چىراي ) ئەجەپ ئىچىپتىكەنمەن شۇ ۋاقتىدا  خۇددى كىشلەر نونىنى چوڭ بىلىپ ئىچكەندەك ئانام كۆرمەستە غۇرتۇلدىتىپ . ئۆزۈمچە بۇ دىمىدە پەيدا قىلىپ تومۇر-تومۇرلىرىمدا مىڭىپ تومۇرلىرىمدىكى بۇزۇق قانلارنى  . ياۋايىلىقنىڭ گىنلىرىنى ، ھەر خىل بەد قىلىق خىياللارنى خۇددى تازلىق ئىشچىسى يەر سۈپۈرگەندەك سۈپۈرۈپ تازلاپ قەلبىمنىڭ ئىچى پاكىزلىنىپ يوپيورۇق بولۇپ ئىماننىڭ نۇرلىرى چاقناپ تۇرغاندەك بىر سىزىم ئىچىدە تۇرۋاتقاندەك ھىس قىلاتتىم . بىراق ئىغىزمنىڭ شورلىغى . بىرە قىتىممۇ مۇنبەرداشلار خۇشال بولغۇدەك ياخشى گەپ چىقايدىغان تەرەپىم  مانا ، قىلچىمۇ ئۆزگەرمەپتۇ .  ماڭا دىمىدنىڭ خاسىيتى پايدا قىلماسلىقى ،بەلىكم ، دىمىدىنىڭ خاسىيتى دىمىدىگە تۈكۈرگەن كىشىلەرنىڭ تۈكۈركىدىكى غەلغەش  تەركىبىدىكى ھەر خىل مۇرەككەپ ئىنزىم ئىچىدە ،خېمىۋىيلىك رىئاكسىيەلىشىپ ئۆزىنىڭ خاسىيتەتلىك خۇسۇسسىتىنى يوقاتقان ئوخشايدۇ . بۈگۈن دىمىدە ھەققىدە ئىيتقان كىلاسىك گەپلىرىڭىز بىلەن قاپاقباش كاللام ئاپتاپپەرەستەك ئىچىلىپ كەتتى . بۇندىن كىيىن دىمىدە قىلىنماقچى بولغان سۇنى توكتا قاينىتىپ بولغاندىن كىيىن ئىچىگە چىرىشتىن ساقلىنىش دورىسى . نەچچە چىمدىم شىكەر ، نەچچە تىمىم لىمون ، بىزە تۇز ، مۇۋاپىق قار لەيلىسى سىلىپ ، ھەر ئەتتىگىنى  ئىچىش ئۈچۈن ئىغىزنى ئاچقاندا مىكىرۇپ كىرىپ بۇزلۇپ قىلىشنىڭ ئالدىنى ئىلىش ئۈچۈن ئىغىزنى ئەتىر قۇتىسىنىڭ ئىغىزىدەك توپچەكنى بارماق بىلەن باسسا پۈركۈپ چىقدىغان قىلىپ ئالەم ئۇچقۇچىللىرى ئۇسىغاندا  ئۇسسۇزلۇقنى ئىغىزغا پۈركۈپ كىرگۈزگەندەك ئىغىزمىغا پۈركۈپ ئىچىدىغان قىلىپ ، دىمە ئوقۇماقچى بولغان ئادەمنىڭ ئىغىز بوشلىقىنى ئاپتۇنۇم رايۇنلۇق ئىغىز بوشلىقى دوىتۇرخانىسىدا تەكشۈرتۈپ  ، لاياقەتلىك ھىچقانداق  يۇقۇملۇق كىسەللىك ئالامىتى يوق دەپ دىياگۇنۇز چىققاندا ئاندىن دىمىدە قۇتۇسنىڭ تىشىدىن سۈفلىتىپ بولۇپ تېخىمۇ بىخەتەر بولۇشى ئۈچۈن دىمەدە قۇتۇسىنى توك ئوچاقتا ئۈچ مىنۇت قاينىتۋىتىپ ئىچىشنى خىيال قىلۋاتىمەن .
سىزنى تىرىكتۇرۇپ قوياي دەپ يىزىپ قويدۇم جۇمۇ بۇ پاراڭنى خەپزە ماڭا دارتىملاپ يازغانمىدۇ دەپ ئويلاپ قالمىسڭىزكەن  ، خۇش -خەيرى بىردەم  كارتون كۆرەي .
