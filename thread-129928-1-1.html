                                  ئازغۇن يوللاردىكى قىسمەتلەر
                                             (پوۋېسىت )   
                          (مىسرانىم مۇكاپاتلىق ئەسەرلەر سەھىپىسىگە)  
                                                        
       ئەسسالامۇ ئەلەيكۇم ئەزىز مۇنبەر ئەھلى:
                                                كەمىنە ئەپيۇنگۈلدىن ھەممىڭلارغا ئوتلۇق سالام!
تالاي قىتىملىق ئىككىلىنىش ۋە ئارىسالدى بولۇشلاردىن كىيىن  مىسرانىم مۇنبىرىدە تور يۈزىدىكى تۇنجى ئىجادىيىتىمنى ھۇزۇرۇڭلارغا سۇنۇش ئالدىدا تۇرۇپتىمەن.شۇ تاپتا تۇيۇقسىزلا ئەجدادلىرىمىزنىڭ ‹‹يىڭى تۇغۇلغان موزاي يولۋاستىن قورۇقماپتۇ ››دىگەن تەمسىلى ئىسىمگە كەلدى. پىكىر -خىياللىرىمنىڭ يۈزەكى ۋە ئاددىيلىقىغا ، قەلەم قۇۋۋىتىمنىڭ گاللىقىغا،ئىجادىيەت جەھەتتە تېخى ئەمدىلا ئايىغى چىققان  گۆدەكتەك تەمتىلەپ يۈرگىنىمگە باقماستىن ئىجدىمائىي تۇرمۇشنىڭ ئەڭ نازۇك تەرەپلىرىدىن بىرى بولغان نىكاھ ، ئائىلە، ھايا ۋە ئىپپەت ، غورۇر - ۋىجدان ھەققىدىكى ماڭا تەسىر قىلغان بەزىبىر رىئال تۇرمۇش كارتىنىلىرىنى ئەدەبىياتنىڭ بەدىئىيلىك ۋاستىسىگە كۆتۈرۈپ قەلەمگە ئىلىپ چىقتىم . ئەسەر يۈزىدە قىسمەن سەۋەنلىكلەردىن ساقلىنالمىغان بولۇشۇم مۇمكىن . بۇنىڭ ئۈچۈن مۆھتىرەم قېرىنداشلىرىمدىن ئۆزرە سورايمەن .
ئەلۋەتتە، دۇنيادىكى ھەممە ئىشنىڭ تۇنجى قىتىملىقى بولغىنىدەك مىنىڭمۇ ھەم تور  يۈزىدىكى تۇنجى ئىجادىيىتىم بولغان مەزكۇر يازمامغا مۇنبەر يۈزىدىكى كەڭ تورداشلار ۋە ئۇستازلارنىڭ ئەگرىلىرىمنى تۈزلەپ ، خاتالىقلىرىمغا  تۈزۈتۈش بېرىپ يىتۈكلۈككە قاراپ يول ئىچىپ بېرىشتە قىممەتلىك تەكلىپ - پىكىرلىرىنى ئايىمايدىغانلىقىغا ئىشىنىمەن ۋە ئەزىز قېرىنداشلىرىمدىن تۆۋەنچىلىك بىلەن ئۆتۈنىمەن.
                                                        __مىھرىئاي مۇھەممەتتۇرسۇن(ئەپيۇنگۈل)  

                                                 ( 1 )        

ئىگىسى بەرسە ئايال كىشىنىڭ قارنىدىن ئارغامچا - توقۇناقتىن باشقىسى چۈشەرمىش......



                   زىۋىدە خۇددى بىرى سىلكىپ ئىتتىرىۋەتكەندەك مەيدىسىنى تۇتقىنىچە دىۋاندا ئولتۇرۇپ قالدى. تۇغۇلۇپ مۇشۇ كەمگىچە 53 قارا قىشنى ئۇزۇتۇپ تۇرمۇشنىڭ ھەممە قازانلىرىدا دىگۈدەك قايناپ چىققان ، مەن - مەن دىگەن ئوتتۇرا قولدەك ئەرلەرنى پىقىرىتىپ يۈرۈپ ئىتەككە چۈشكەن ياڭاقتەك ئوينىتىپ كەلگەن ، ھەممىلا ئىشقا چىۋەرلىكى گەپ - ئاغزىنىڭ جايىدىلىقى بىلەن  پۈتۈن قىلغان - ئەتكەنلىرىنى يىغۇشتۇرۇپ ھالا بۈگۈنكى كۈنگە كەلگەندە(زىۋىدە ھاجى خېنىم)دەپ شەھەرنىڭ يۈز - ئابرويلۇق باي خوتۇنلىرىنىڭ بىرى سۈپىتىدە توي - تۆكۈن ، مەرىكە- چايلاردا داستىخاننىڭ ئەڭ تۆرىگە باشلىنىدىغان ئەزىز مېھمانلار قاتارىدىن ئورۇن ئىلىپ بىر كىچىدىلا ھىچنىمە كۆرمىگەن ئاق بالىخان بولۇۋالغان بۇ شەرەنداز ئايالغا دۇنيالىقتىكى بىرتاللا قىزى ئايدىلبەرنىڭ تېخى توي قىلغىلى 3 ئاي بولار بولمايلا تارتىۋاتقان تۇرمۇش سورۇقچىلىقلىرى يۈرىكىنى ئېزىپ خېلىلا ھورىدىن چۈشۈردى.
