ئۇنىڭ ئۈستىگە تور بىكەتلەرنىڭ مۇراسىم كىيمى دىگەننى بىز تازا چۈشەنمەيدىغان تۇرساق،تىما يوللىغۇچى ئاۋارچىلىق ھىس قىلمىسا بۇنىڭ قانداق ئىش ئىكەنلىكىنى بىلدۈرۈپ قۇيسا دىگەن ئۆمۈدتتە مەن،ئۇنىڭ ئۈستىگە ئۇيغۇر تور بىكەتلىرىگە ياۋرۇپادىكى سىگان،ئارىيان،روسلارنىڭ مەدىنىيەت كىيىمىدەكلا چىقىپتۇ،ئۇيغۇرنىڭ مىللىي كىملىكىگە ئائىت ئەتلەس قاتارلىق رەخىتلىرى ۋە پاسۇن تۈزىلىشى،مەسلەن بۇرۇنقى ئاياللارنىڭ يەنى مىللىي پۇرىقى بار كىيىملەرنىڭ شەكلىنى چىقىرىپ قۇيۇڭ،بولمىسا بۇنداق ياۋرۇپا ئۇسلۇبىدىكى كىيىم كىچەك ئۇيغۇرغا ۋەكىللىك قىلمايدۇ.بۇ راس جۇمۇ.
