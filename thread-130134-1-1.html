ئەسسالامۇئەلەيكۇم !ئاللاھ ھەممىمىزنى سالامەت قىلسۇن!!!   مىسرانىم مۇنبىرىدىكى  تورداشلىرىم ، كەمىنەسىرداش تېۋىپ مۇنداق بىر بىلىمنى ئېلىپ كەلدىم قېنى ھەممىمىز ئورتاق بەھىرلىنەيلى، بۆرەك قۇۋۋەتلەش بۇيۇملىرىنىڭ كۆپىيشىگە ئەگىشىپ، بىر قىسىم دۇرىلارنىڭ نامىمۇ  ھەر خىل ئاتىلىپ كېلىۋاتىدۇ ، ھەمدە بۇنىڭ ئىشلىتىش ئۇسۇلىدىمۇ  بەزى ئىلمىي بولماي قالغان تەرەپلەردىن  ساقلىنىپ كىلىۋاتقان مەسىللەرمۇ مەۋجۈت بولغاچ  بۇنىمۇ بىلىپ قويسا زىينى يوققۇ دىگەن مەقسەتتە بۇ مۇنبەرگىمۇ ئېلىپ كەلدىم ، قېنى بۇ ئاددىيغىنە يازمامدىن بەھىرلەنگەيسىزلەر!!!  
  ‹‹ ئىمساك ››  بۇ ئەرەپچە ئاتالغۇ بۇلۇپ ،ئۇيغۇر تېبابەتتىكى مەنىسى ‹قورۇش،تۇتۇش ،ساقلاش ›دېگەن مەنىدە !بۇ دۇرىنىڭ ياسىلىش ئالاھىدىلىكى ھەر خىل بۇلۇپ ئەرلەردىكى مەنىي تۇتالماسلىق ، مەنىي بالدۇر كېتىشكە قارىتا ئىشلىتىدىغان نۇسخىلىرى ،ئەر-ئاياللاردىكى جىنسىي ھېسسىيات سۇسلاپ كېتىشكە قارىتا ئىشلىتىدىغان نۇسخىلىرى كىچىك بالىلاردىكى كېچىسى  سىيىپ قۇيۇشقا  ئىشلىتىدىغان نۇسخىلىرى ، ،ئاياللاردىكى ئاق خۇن كۆپ كېلىشكە ۋە ئاياللاردىكى ھامىلىنى ساقلاش ئۈچۈن ئىشلىتىدىغان نۇسخىلىرى قاتارلىق بىر قانچە تۈرلۇك نۇسخىلىرى بار! سىزنىڭ بۇ  يەردە سورىغىنىڭىز ئەرلەردىكى مەنىي بالدۇر كېتىش،مەنىي تۇتالماسلىققا قارىتا ئىشلىتىدىغان نۇسخىسى بۇلۇپ، بۇ دۇرىنىڭ تەركىبىدە قاننى جانلاندۇرىدىغان ،جىنسي ھەۋەسنى كۈچەيتىدىغان،مەنىي تۇتىدىغان،جىنسىي ھېسسىياتنى كۇچەيتىدىغان ...  دۇرىلار بار !ئەسلىدە بۇ  دۇرىلار جىنسىي جەھەتتىن بەكلا ئاجىزلاپ كەتكەن، ئۈزىنىڭ ھالال ئايالى بىلەن بولغان جىنسىي تۇرمۇشنىڭ ھۆددىسىدىن چىقالمىغان ئەرلەردە كۈرۈلگەن كېسەللەرگە بېرىلسە ئەڭ مۇاپىق ئىدى !ئەمما جەمىئەتتىكى بىر قىسىم تېۋىپ قېرىنداشلىرىمىز ئىمساك دۇرىلارنى سېتىشنى داۋاخانىسىنىڭ كىرىم مەنبەسىگە ۋە بىمارلارنى جەلىپ قىلىش قۇرالىغا ئايلاندۇرۋالدى ،ھەمدە بىر قىسىم ئىمساك نامىدىكى  چەت ئەللەرنىڭ خىمىيىلىك دۇرىلىرى ئارلاشتۇرۇلۇپ ياسالغان دۇرىلار ۋە ھورمۇن تۇردىكى دۇرىلار قۇشۇلغان ئىمساكلار  تىۋىپلارنىڭ دۇكانلىرىدىن ئۇرۇن ئالدى !بۇنىڭ بەزى سەۋەپلىرىمۇ بار چۈنكى بەزى كىشىلەر تەبىئي دۇرىلار قۇشۇلغان دۇرىلارنىڭ ئاستا تەسىر قىلىشىغا تاقەت  قىلالمىغاچقا شۇنداق ھالەت شەكىللىنىپ قالدى !لىكىن  ئىشەنچىلىك ئىمساكلارنىڭمۇ شۇنداق تېز تەسىر قىلىدىغىنى بارئاساسى ھالقا يەنىلا سىزنىڭ توغرا تاللىشىڭىزدا  !!!شۇڭا مۇشۇ تۈردىكى دۇرىلارنى  تاللىغاندا يۇقارقى ئەھۋاللارغىمۇ  سەل قارىماسلىقىڭلارنى ئۇمۇد قىلىمەن !
