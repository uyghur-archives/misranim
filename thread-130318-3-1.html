پۇتىننىڭ بۇرۇندىنلا قىرىمنى ئېلىۋېلىش ئارقىلىق قارا دېڭىزغا ئۈتۈش يولىنى راۋانلاشتۇرۇش خىيالى بار ئىدى، ئەپسۇسكى، قىرىم ئوكرانىيەنىڭ ئىدى، ئۇ ئاخىرى قىرىمنى يۇتۇۋەتتى، لېكىن، ئامېرىكالىقلار دەل پۇتىننىڭ ئاشۇ قىلىقىنى كۈتۈپ تاقىتى تاق بولغان ئىدى. چۈنكى، ئامېرىكا باشچىلىقىدىكى بۇ غەرب دۇنياسىنىڭ روسىيەنى يانباشقا ئېلىشىغا ئىسىل باناھ بولاتتى، ئامېرىكالىقلار روسىيەنىڭ ئۇنداق قىلىشىنى قارار قىلىشىدىن بۇرۇنلا خەۋەر تاپقان بولۇشى مۇمكىن، شۇڭا قىرىمدىن ئىبارەت بۇ قىلتىرىقى كۆپ بېلىقنى يۇتۇۋالسا گېلىغا سۈڭەك تۇرۇپ قالىدىغانلىقىنى پۇتىن ئەمەس، بەلكى، ئوباما ئۇبدان بىلەتتى. مەنچە ئامېرىكانىڭ مەقسىتى روسىيەنى يىڭىدىن دۈشمەن قىلىپ تىكلەش، بۇرۇنقى سۆۋىتنىڭ ئورنىغا قويۇپ ئۇنىڭ بىلەن باشقىدىن ئۈزەڭگە سوقۇشتۇرۇپ سوغۇق ئۇرۇش قىلىش. بەلكىم ئۇنىڭ ئاخىرقى مەقسىتى روسىيەنى قايتا بىر قېتىم پارچىلىۋېتىش بولۇشى مۇمكىن، شۇنداق بولغاندا ئامېرىكانىڭ دۇنيانىڭ غاجاكىسى بولۇشىغا پۇتلىكاشاڭ بولىدىغان ئەڭ زور بىر مىق سۇغۇرۇۋېتىلىدۇ. قىرىم ۋەقەسى، ئۇندىن كېيىنكى بىرقاتار ئىشلار، ھەتتا، نۆۋەتتىكى نېفىت باھاسىنىڭ تۈۋەنلىشلىرىنىڭ ھەممىسى يىڭى بىر سوغۇق ئۇرۇشنىڭ باشلىنىپ كەتكەنلىكىدىن بىشارەت بېرىدۇ. كەلگۈسى، ھەتتا، ئەتە قانداق ئىشلارنىڭ يۈز بېرىدىغانلىقىنى خۇدا بىلىدۇ، لېكىن، ئامىرىكا دىگەن بۇ دۆلەت ناھايتى قورقۇنچلۇق دۆلەتتۇر. ئۇ ئاللىكىملەرنىڭ ئەۋلىياسى، يەنە ئاللىكىملەرنىڭ نەزەرىدىكى قورقۇنچلۇق ئالۋاستىدۇر. بۇ قېتىملىق ئامېرىكا بىلەن روسىيە ئوتتۇرىسىدىكى ئۆزەڭگە سوقۇشتۇرۇشلاردا روسىيە غەربنىڭ ئىمبارگوسىدىن قوتۇلسىمۇ زىيان تارتىدۇ، ئەگەر قۇتۇلالمىسىمۇ يەنىلا زىيان تارتىدۇ. ئەكسىچە ئامىرىكانىڭ روسىيەگە قاراتقان ئىمبارگۇسى غەلبە قىلسىمۇ مەغلۇب بولسىمۇ پايدا ئالىدىغىنى يەنىلا ئامېرىكا بولۇپ قالىدۇ، ئىككى قوچقار ئۈسسۈشسە بۆرىگە ئويۇن دىگەندەك ئامىرىكا بىلەن روسىيەنىڭ جىدىلىدىن ھەقىقىي نەپ ئالىدىغىنى يەنىلا دۆلىتىمىزجوڭگۇ بولۇپ قېلىشى مۇمكىن، بۇ يەردە جوڭگۇنىڭ ئىناۋىتى بولۇپمۇ روسىيەنىڭ ئالدىدىكى ئىناۋىتى تېخىمۇ يۇقىرى كۈتۈرۈلۈپ قالىدۇ

پۇتىننى پۇتىن قىلغان نېفىت.سوۋېت ئىتتىپاقى مۇنقەرز بولۇشنىڭ ئالدى كەينىدە(80-يىللىرىدىن 90- يىللىرىنىڭ ئوتتۇرلىرىغىچە خەلقارا بازاردا نېفىت باھاسى خەلقارا نېفىت باھاسى ئىنتايىن تۆۋەن بولغانلىقتىن،تاشقى پىرۋورتى ۋە دۆلەت غەزىنىسى نېفىتكە تايىنىدىغان سوۋېت ئىتتىپاقى ۋە روسىيەگە ئىنتايىن ئېغىر بېسىم ئېلىپ كەلدى،ئىقتىساد چېكىندى،خەلق تۇرمۇشى قىيىنچىلىققا ئۇچراپ،ھەممىمىزگە مەلۇم بولغان سىياسىي بۆھرانلارنى كەلتۈرۈپ چىقاردى.پۇتىن تەخىتكە چىقىشى بىلەن ئوتتۇرا شەرق ۋەزىيىتى ۋە تېرورلۇققا قارشى ئۇرۇشنىڭ تەسىرىدە نېفىت باھاسى شىددەت بىلەن ئۆسۈپ،روسىيە ئىنتايىن زور نەپكە ئېرىشتى،ئىقتىساد ئىلمىدىن پەرۋايى پەلەك پۇقرالار بۇ نەتىجىلەرنى پۇتىننىڭ نەتىجىسى دەپ قاراپ ئۇنى قەھرىمانلىق ئورنىغا چىقىرىپ قويدى.ھازىر نېفىت باھاسى %30 چۈشە چۈشمەيلا،«قەھرىمان پۇتىن»تىپىرلاپ،باشقا كوزۇرى قالمىغانلىقتىن ئامالسىز ئۆزىنىڭ يادرو قوراللىرىنى كۆتۈرۈپ چىقىپ،باشقىلارغا پوپوزا قىلغىلى تۇردى.ئەمىلىيەتتە يادرو قوراللىرى  كارغا كەلمەس ئەخلەتتىن باشقا نەرسە ئەمەس.روسىيەگە 15 يىل ھۆكۈمرانلىق قىلىپمۇ بىر توغرا يولغا سالالمىغان پۇتىن ھەرگىزمۇ سىز ئويلىغاندەك ئىقتىدارلىق رەھبەر ئەمەس،پەقەت بىر قاتتىق قول رەھبەر،خۇددى كېمئېرسىن،ستالىن ۋە شاۋېسلار كەبى.ھەقىقى ئۆلمەس رەھبەر دەپ ۋاشىنگتوننى دەيمىز،8 يىل دۆلەت سوراپ،تخىتتىن ۋاز كەچكەن بولسىمۇ،لېكىن ئۇنىڭ تۇتقان يولى ھازىر ئامېرىكىلا ئەمەس پۈتۈن دۇنيانىڭ بەخىت -سائادەت يولىغا ئايلاندى.

پۇتىننى پۇتىن قىلغان نېفىت.سوۋېت ئىتتىپاقى مۇنقەرز ب ...

روسىيەگە 15 يىل ھۆكۈمرانلىق قىلىپمۇ بىر توغرا يولغا سالالمىغان پۇتىن ھەرگىزمۇ سىز ئويلىغاندەك ئىقتىدارلىق رەھبەر ئەمەس،
----------------------------------------------------------------

ماۋۇ گەپكە قۇرت قوڭغۇزلارمۇ قوڭىدا كۈلە....  پۇتىن ئىقتىدارلىقمۇ ياكى ئىقتىدارلىق ئەمەسمۇ ھازىرقى روسىيەنىڭ ئەھۋالى بۇنىىڭغا جاۋاپ بولالايدۇ. نىفىت بولغانغىلام دۆلەت گۈلىللۈنۈپ قۇدرەتلىك بوپ كىتىدىغان ئىش بولسا ھازىرقى لىۋىيە سۈرىيە دىگەن دۆلەتلەردە نىفىت ئېشىپ تېشىپ تۇرۇپتۇ. شۇلا نىمىشقا روسىيەدەك بولامممىدى؟ ئامىركىنىڭ بەخىت سائادىتى دۇنياغا ئەمەس ئۆزىنىڭ بۇيرۇقلىرىغا قۇيرۇق شىپپاڭشىتقان دۆلەتلەرگىلا سايە تاشلىدى! پەلەستىننىڭ مۇستەقىللىقنى بىز تىللاۋاتقان روسىيە قوللىدىيۇ ئامىركا قوللىماي تۇرۇۋالدى. بۇنى قانداق چۈشەندۈرىسىز؟ ئامىركىنىڭ ئاتالمىش دىموكراتىيەسىدىكى بارەۋەرلىك ئىنسانى ھوقۇق دىگەنلەر 3-دۇنيا ئەللىرىگە مەنسۇپ ئەمەسمىكەن؟ يەنە شۇ بىز تىللاۋاتقان روسىيە ئاچارچىلىقتا قالغان سۈرىيە خەلقىگە 20مىليون دوللارلىق ھەقسىز ئاشلىق ياردىمىنى نەقلەشتۈردى. ئەمما سۆزلەڭ دىسە6چەرەك سۆزلەيدىغان  ئاۋۇ قارماتاقچۇ؟ بىر تىيىنلىك ئەمەلى ھەرىكەت قىلماي ئەتىدىن كەچكىچە قۇرۇق گەپ سېتىپ دۇنيانىڭ كۆزىنى  ئىسرائىلىيەنىڭ پەلەستىنلىكلەرگە قىلىۋاتقان قىرغىنچىلىقىدىن سۈرىيە مەسىلىسىگە بۇرېۋېتىپلا يۈردى. بولدىلا سىزگە مىڭ چۈشەندۈرسەممۇ چۈشىنىدىغاندەك ئەمەسسىز شۇڭا ئەڭ ياخشىسى بىر بىرىمىزگە دىكتاتورا يۈرگۈزۈشمەيلى. سىزمۇ ئۆز پىكرىڭىزدە قېلىڭ مەنمۇ ئۆز پىكرىمدە قالاي. ئاللاھقا ئامانەت!
