فوتو كامىرا بىلەن قايتا چىچەكلىگەن ھايات


ئايتىللا قۇربان تەرجىمىسى :

ئۇ  ئالتە يېشىدا مانا مۇشۇنداق بىر قېتىم قاتتىق قىزىپ قالغانلىق سەۋەبى بىلەن ئاڭلىيالمايدىغان ھەم سۆزلىيەلمەيدىغان بولۇپ قالغان.چوڭ بولۇپ قورامىغا يەتكەندىن كىيىن بولسا لىمفا ئۆسمىسى خۇددى كانىدەك چاپلىشىپ ئۇنىڭغا ئارام بەرمىدى.ئۇ ئۆزىنىڭ نەچچە قېتىم ئۆزىگە ھاي بىرىپ ھاياتىنى داۋاملاشتۇرىۋاتقانلىقىنى بىلمەيتى.مۇنداقچە قىلىپ ئېيىتقاندا ئاشۇ ھەر بىر پارچە رەسىم ئۇنى داۋاملىق ياشاشقا ئۈندەيتى.«فوتوگىرافلىق مىنىڭ ئىككىنچى ھاياتىم ،ھەربىر پارچە رەسىم مەن ئۈچۈن ئىككىنچى مۇڭلۇق كۈي»دەيتى ئۇ.



ئاپارات زاتىۋورىنىڭ ئاۋازىنى ئاڭلىيالمايدىغان بۇ 33 ياشلىق ئەر جۈمەخۇن ياقۇپ.ئۇ قەشقەر ۋىلايىتىنىڭ مارالبېشى ناھىيىسدا تۇغۇلغان بولۇپ،ھازىر ئۈرۈمچى شەھىرى تىيانشان رايونى بەيگە مەيدانى شەرقى ئاھالىلەر كومىتىتىدا بىر رەسىمخانا ئاچقان بولۇپ،قارامتۇل يېشىل رەڭلىك جىلىتكە،بىر سومكا بۇلار ئۇنىڭ ئەڭ ياخشى جابدۇقلىرى ئىدى.پۈتۈن زىيارەت جەريانىدا ئۇ قەلەم ئارقىلىق مۇخبىر بىلەن پىكىر ئالماشتۇردى.ئۇ مۇنداق دەيتى:«تۇرمۇشۇمدىكى مەن ئاڭلىيالمىغان ھەربىر تىۋىش،مەن ئېيتالمىغان ھەربىر ئېغىز سۆز مەن تارتقان رەسىملەردە».

ئۇنىڭ قولىدىكى بۇ ئاپاراتنى دادىسى ئۇنىڭغا تېرىيدىغان پۈتۈن يېرىنى سېتىپ ئېلىپ بەرگەنىدى.ئالتە ياش ۋاقتىدىكى ئۇ  ئىشتىن كىيىن دادىسى ئۇنىڭغا داۋا ئىزدەپ بارمىغان يېرى ئىدىمىگەن دوختۇر تىۋىپ قالمىدى.تاكى ئۇ 12ياشقا كىرگۈچە دادىسى ئۇنى ساقايتىشتىن ئۈمىدىنى ئۈزمىگەنىدى.دادىسى ئۇنىڭ دائىم ئۆيدىكى رەسىملەرگە قاراپ خىيال سۈرىدىغىنى،رەسىمخانىلارنىڭ ئالدىدىن ئۆتسە تۇرىۋىلىپ ماڭغىنى ئۇنىمىغانلىقىنى ،ھەر قېتىم دادىسىغا بىر خىل يىلىنىش نەزىرىدە قاراۋاتقانلىقىنى كۆرۈپ ئۆز ئوغلىنىڭ بىرەر ھۈنەر كەسىپنى ئۆزىگە ھەمراھ قىلغاندىلا ئاندىن داۋاملىق مەۋجۇتلىقىنى ساقلىيالايدىغانلىقىنى ھىس قىلىپ يەتكەن ئاتا قەتئى نىيەتكە كىلىپ ،ئاشۇ بىر نەچچە مو يېرىنى سېتىپ 30يۈەن پۇلغا ئوغلىغا تۇنجى شاپىگىرافلىق ئاپارات ئېلىپ بەردى.بۇ چاغ1992-يىلى بولۇپ،جۈمەخۇن بىرىنجى قېتىم ئاپاراتنى قولىدا تۇتۇپ تۇراتتى.مانا شۇنىڭدىن باشلاپ ئۇ نىمىنى كۆرسە شۇنى رەسىمگە تارتىپ توختىمىدى.كەچتە ھويلىغا چىقىپ ئاي نۇرىدا رەسىمگە تارتاتتى.كىيىن ئۇ قەشقەر مىيىپلار مەكتىپىدە ئوقۇۋاتقان چېغىدا ساۋاقدېشىنى رەسىمگە تارتىپ بەش مو پۇل ئالغانىدى،لىكىن ئۇ رەسىمنى يۇدۇرۇپ چىقىرىش ئۈچۈن بىر يۈەن پۇل سەرپ قىلاتتى.«بىر پۇڭمۇ پۇل تاپالمىساممۇ،ئىشقىلىپ مىنى رەسىمگە تارتقۇزىدىغان ئادەم بولسىلا مەن خۇش بولاتتىم.مەن ئۇلارنىڭ رەسىملىرىدىن ئۇلارنىڭ خۇشاللىق خاپىلىقلىرىنى ھىس قىلاتتىم».
ئوقۇش پۈتكەندىن كىيىن ،ئۇ مارالبېشى ناھىيىلىك بىر رەسىمخانىدا ئىشلەيدۇ.ئۇ ھەركۈنى موتىسىكىلىتىنى مىنىپ چوڭ-كىچىك كوچىلارنى ئارىلاپ رەسىمگە تارتاتتى.كىينچە ئۇ يەردىكىلەر ئورۇق، قاپقارا بۈدۈر چاچ،ماشرەڭ جىلىتكە،سومكىسىنى ئېسىپ موتسىكىلىت بىلەن باشقىلارنى رەسىمگە تارتىپ قويىدىغان  بۇ يىگىتنى ھەممىسى بىلىپ كەتكەنىدى.

1999-يىلى ئۇ مېيىپلار مەكتىپىدە تونۇشقان مىھرىگۈل بىلەن تۇرمۇشلىق بولدى.ئۇ يەنىلا رەسىمخانىدا ئىشلەيتى،ئىشچان ئايال مىھرىگۈل بولسا سىرىتتا ئۇششاق ئىشلارنى قىلاتتى.ئۇلار توي قىلىپ ئىككى يىلدىن كىيىن جۈمەخۇن بوينىغا بىر نەرسىنىڭ چىقىپ قالغانلىقىنى بايقايدۇ،دوختۇرخانىدا تەكشۈرتكەندىن كىيىن بۇنىڭ لىمفا ئۆسمىسى ئىكەنلىكى ئېنىقلىنىدۇ.بۇ كىسەلنى داۋالىتىش ئۈچۈن ئەر-ئايال ئىككىسى ئۆيىدىكى بارلىق پۇلغا يارىغۇدەك بىساتىنىڭ ھەممىسنى سېتىۋىتىدۇ.لىكىن ئاشۇ ئاپاراتنى سېتىشقا كۆزى زادىلا قېيمايدۇ.
2003-يىلىغا كەلگەندە ئەر-ئايال ئىككىسى نەچچە مىڭ يۈەن پۇل بىلەن ئاپاراتىنى ئېلىپ مارالبېشىدىن ئۈرۈمچىگە داۋالىنىشقا كىلىدۇ.داۋالىنىش مەزگىلىدە جۈمەخۇن قۇرلۇش ئورۇنلىرىدا ئىشلەپ باقىدۇ،ئۇنىڭ بويۇن،بىلەك ،پاقالچاق ئىشقىلىپ لىمفا تومۇر بار جاينىڭ ھەممىسىگە ئۆسمە چىقىپ كىتىدۇ.ئۇنىڭ تىنى ئېغىر ئەمگەكنى كۆتۈرەلمەيتى.ئۇنىڭ بىردىنبىر ئۈمىدى چوڭ بازاردىكى رەسىمخانىدا ئىدى.ئۇ ئۆزىنىڭ داۋالىنىش مەزگىلىدە تارتقان رەسىملىرى بىلەن چوڭ بازاردىكى بىر رەسىمخانىدىن ياردەمچىلىك خىزمىتى تاپىدۇ.ئايالىنىڭ خىزمىتى مۇقىم ئەمەس بولۇپ،قەيەردە ئىش ھەققىنى كۆپرەك بەرسە شۇيەردە ئىشلەيتى.فوتۇگىراف ياردەمچىسىنىڭ بىر ئايلىق مۇئاشى 1200يۈەن بولۇپ ئۆي ئىجارىسىنى تاپشۇرۇش ۋە داۋالىنىشقا يەتمەيتى.ئۇ نەچچە قېتىم بۇنداق ھاياتنى ئاخىرلاشتۇرۇپ ئايالىنىڭ يۈكىنى يەڭگىلتىشنى ئويلىغان بولسىمۇ،بۇنى سەزگەن مىھرىبان ئايال ئاۋازسىز ئۇنىڭغا تەسەللى بىرەتتى.مىھرىبان مىھرىگۈل ئاپاراتنى ئۇنىڭ قولىغا تۇتقۇزۇپ رەسىمگە تارتىش كونۇپكىسىنى باسقان ۋاقتىڭدا ھەممىنى ئۇنتۇيسەن،سەن ئۈچۈن پەقەتلا خۇشاللىق قالىدۇ دەپ ئۇنىڭغا ئۈمىد بېغىشلاپ تۇراتتى.

مىھرىگۈل يىغقان ھەممە پۇلنى يولدىشىنى داۋالىتىشقا ئىشلىتەتتى.مانا مۇشۇنداق قىلىپ ئۇ ئاشۇ بىر قارا يوپكىسى بىلەن ئالتە يازنى چىقارغانىدى.مۇشۇ مەزگىللەردە جۈمەخۇن ھەتتا ساقال تىغىمۇ ئىشلىتىپ باقمىغانىدى.ھەر ئايدا 200يۈەن ئىجارە تاپشۇرىدىغان بۇ ئۆيدە ئۇلارنىڭ ھەممىدىن قىممەتلىك نەرسىسى مۇشۇ تاق سىفىرلىق ئاپاراتى ئىدى.«نەگىلا بارمىسۇن ئاپاراتىنى ئېلىۋالاتتى،سەي سېتىۋالغىلى چىقسىمۇ قولىدا ئاپارات.ئوغلىمىز بىلەن قىزىمىز تۇغۇلغاندىن كىيىن ئۇ ھاياتقا باشقچە نەزەر بىلەن قارايدىغان بولدى»دىدى مىھرىگۈل.جۈمەخۇننىڭ ئاپاراتقا بولغان مىھرى مىھرىگۈلگە بولغان مىھىردىنمۇ ئېشىپ چۈشەتتى.2010-يىلى تۆتىنچى قېتىملىق ئوپراتسىيەدىن كىيىن ئۇنىڭ بەدىنىدىكى لىمفا ئۆسمىلىرى ئاساسەن تۈگەپ ،ھازىر دورا بىلەن داۋالاشنى قوبۇل قلىۋاتاتتى.
ئېغىز شەكىلى ۋە قەلەم ئارقىلىق پىكىر ئالماشتۇرۇش
2009-يىلى جۈمەخۇننىڭ فوتوگىراف ھەۋەسكارى ئىكەنلىكىنى بىلگەن ئاھالىلەر كومتىتى بۇ بىر ئائىلە كىشلىرىگە ياردەم قىلىپ  ھەقسىز بىر ئېغىز ئۆي بەرگەن .مانا شۇنىڭدىن باشلاپ ئۇلانىڭ ئۆزىنىڭ رەسىمخانىسى بولۇپ،مىھرىگۈل بالىلارغا قاراشقا جۈمەخۇن بولسا بۇ ئائىلىنى قامداش ئۈچۈن تىجارىتىنى باشلىۋالدى.

بۇ 30كىۋادىرات مىتىر كىلىدىغان رەسىمخانىدا ئىككى چاقماق لامپا،بىر كومپىيۇتېر،بىر كۆپەيتىش ماشىنىسى ،بىر ئۈچ پۇتلۇق ئاپارات جازىسى بار بولۇپ .ھەركۈنى بۇ يەرگە كىلىپ رەسىمگە چۈشىدىغانلار بار ئىدى.سائەت بىرلەردىن ئاشقاندا ئىككى ئاپا تۆت بالىنى يىتلەپ بۇ يەرگە رەسىمگە چۈشكىلى كىرىپ كەلدى.بالىلار بەكمۇ كەپسىز بولۇپ،ئاپىلار بالىلارنى توسۇپمۇ توسۇپ بولالمايتى.بىر مىنۇتقا يەتمىگەن ۋاقىت ئىچىدە جۈمەخۇن چاقماق لامپىنى ياندۇرۇپ،ئۇچ پۇتلۇق جازىغا ئاپاراتنى مۇقىملاشتۇرۇپ تەييار بولدى.بۇنى كۆرگەن كەپسىز شاكىچىكلەرمۇ ئاپاراتنىڭ ئالدىغا كىلىپ تەييار بولدى.ئۇ بۇ ئوماق شاكىچىكلەرنى رەتكە تىزىپ كىيىملىرىنى تۈزەپ ،بارمىقى بىلەن شاكىچىكلەرنى جىم تۇرۇشقا ئىما قىلىپ كۆزىنى قىسىپ قويدى.ئوماقلارنىڭ  تاتلىق كۈلۈشى بىلەن تەڭلا جۈمەخۇنمۇ ئاپاراتنىڭ كۇنۇپكىسىنى باستى.
يەنە بىردەمدىن كىيىن بىر جۈپ ياش رەسىمخانىغا كىرىپ كەلدى.جۈمەخۇن قەغەزگە سىلەر بىللە رەسىمگە چۈشەمسىلەر دەپ يازدى.ئىككىيلەن تەڭلا باش لىڭشىتتى،قىز ئۆزىنىڭ ئۈستى بېشىنى ئىما قىلدى.جۈمەخۇن گىرىم بويۇملىرىنى قىزنىڭ ئالدىغا ئېلىپ كەلدى.يېرىم سائەتتىن كىيىن جۈمەخۇن قىزغا گىرىم قىلىپ تەييار قىلىپ ئۇلارنى بىرگە رەسىمگە تارتىپ قويدى.

ئۇ رەسىمخانىغا كىرگەن ھەربىر ئادەمنىڭ ئېغىز شەكىلىگە قاراپ ئۇلارنىڭ نىمە دىگەنلىكىنى پەرەز قىلاتتى.ئۇنىڭدىن كىيىن قەغەزگە يېزىپ ئۇلارغا كۆرسىتەتتى.تونۇش خېرىدارلار دۇكانغا كىرگەندە رەسىمگە چۈشۈپ بولۇپ ئارتۇق گەپ-سۆزمۇ قىلماي پۇلنى ئۈستەل ئۈستىگە قويۇپ قويۇپلا چىقىپ كىتەتتى.
جۈمەخۇن ئەر-ئايال ئىككىسى ھەممىدىن بەك ئىپتىخارلىنىدىغان ئىش شۇكى،گەرچە ئۇلار كەمچىلىك بولسىمۇ لىكىن ئۇلارنىڭ ئوماق پەرزەنتلىرى ساغلام چوڭ بولماقتا ئىدى.ئۇلار خۇددى قۇشقاچتەك ۋىچىرلاپ دادىسى ۋە ئاپىسىنى ئەگىپ،بۇ قەيسەر ئاتا-ئانىغا كۈچ-قۇۋەت ۋە ئۈمىد ئاتا قىلاتتى.ئادەتتە جۈمەخۇننىڭ قول تىلفۇنىنى ئەقىللىق ئوغلى ئارمان بىلەن چىچەن ئوماق ئارزۇ ئالاتتى.ئۇلار خېرىدارلارنىڭ سۆزىنى ئاڭلىغاندىن كىيىن بارماق تىلى ئارقىلىق دادىسىغا بىلدۈرەتتى.بۇ ئاكا-سىڭىللار ئۆز-ئارا ئاۋازلىق پاراڭلىشاتتى.دادىسى ۋە ئاپىسى بىلەن بولسا بارماق تىلى ئارقىلىق دىمەكچى بولغىنىنى ئىپادىلەيتى.ھازىر ئۇلارنىڭ ئۆيىدە ئىككى ئاپارات بار بولۇپ،جۈمەخۇن كىچىك ئاپاراتتا ئەققىللىق ئارمانغا رەسىمگە تارتىشنى ئۈگىتىۋاتاتتى.ئارمان دادىسىغا ئەگىشىپ رەسىم كىسىش،لايىھىلەش قاتارلىق ئاساسلىق مەشغۇلاتلارنى ئۈگنىۋاتقىلى ئۈچ يىل بولغانىدى.ئارمان بۇ يىل يازلىق تەتىلدە ساۋاقداشلىرىنى رەسىمگە تارتىپ ئىككى قىسقا سىن فىلىمىمۇ ئىشلەپ بولغانىدى.ئالتە ياشقا كىرگەن چىچەن ئارزۇگۈل بولسا:«مەن چوڭ بولغاندا گىرىمچىك بولىمەن،دادام رەسىمگە تارتىدۇ،مەن گىرىم قىلىپ بىرىمەن»دەيتى.جۈمەخۇن ئوماق قىزىنىڭ بارماق تىل بىلەن دىگەن سۆزىنى ئاڭلاپ ئارزۇگۈلنى مەھكەم باغرىغا بېسىپ رەسىمگە چۈشتى.ئۇ مۇنداق دەيتى:«تېخمۇ كۆپ پۇل تاپقاندىن كىيىن رەسىمخانىنىڭ كۆلىمىنى كىڭەيتىمەن.قىزىم گىرىمچىك بولىدۇ،ئوغلۇم رەسىمگە تارتىدۇ ھەم ئۇنىڭدىن كىيىنىكى تېخنىكىلىق خىزمەتلەرنى قىلىدۇ.»ھازىر ئۇ پۇل يىغىپ مىھرىبان مىھرىگۈلگە بىر تال تويلۇق ئۈزۈك تولۇقلاپ بەرمەكچى ئىكەنلىكىنى ئېيىتتى.بۇ قەيسەر ئەر ۋە كۆيۈمچان ئاتىغا  چىن كۆڭلۈمدىن سالامەتلىك  ۋە ئىشلىرىغا ئۇتۇق تىلەيمەن.ھايات مانا مۇشۇنداق.بىز ئۇنىڭغان قانداق قارىساق ئۇ بىزگە شۇنى ئېلىپ كىلىدۇ.بەلكىم مىھىرلىك بىر نەزەر،مۇھەببەتلىك بىر ئېغىز سۆز.... بۇ يازمىنى ئاخىرىدا   Misranim2 تەھرىرلىگەن. ۋاقتى  2014-11-4 02:51 AM
