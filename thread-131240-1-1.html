ئەسسالامۇ ئەلەيكۇم مۇنبەردىكى قىرىنداشلار ،ئۇزۇن بوپتۇ مەن مۇنبەر توغرىسىدا كۆڭۈل سۆزۈمنى يازمىغىلى،بۈگۈن مۇنبەر كۆرۈپ ئولتۇرۇپ مۇشۇ خىياللارغا كېلىپ قالدىم .
   مىسرانىم مۇنبىرى تەرەققى قىلىپ ئۇيغۇر تورچىلىق ساھەسىدىكى سەركە مۇنبەرلەرنىڭ بىرىگە ئايلاندى ،مۇشۇنداق ئۇنۋىرسال مۇنبەرگە ئايلىنىشىدا ئەلۋەتتە ئابەكرى مۇختەر مىسرانىم باشلىق بارلىق باشقۇرغۇچى ،ۋە ئەزالارنىڭ ئەجرى سىڭگەن ،ھازىر مۇنبەرگە قارايدىغان بولساق مىللىتىمىز ئىچىدىن چىققان ئوتيۈرەك زىيالىيلىرىمىزدىن تارتىپ تەكلىماكاننىڭ ئەڭ چىتىدىكى ئاددى بىر دېھقان بالىسىغىچە مۇنبەرنى زىيارەت قىلىپ <<ئىلىم زاكىتى>>نى بۇ داستۇرخانغا تاشلىسا ،نۇرغۇنلىرىمىز روھى ،مەنىۋى ئىھتىياجلىرىمىزنى قاندۇرۇپ كەلمەكتىمىز ،لېكىن مۇنبەر تەرەققى قىلغانچە ئەزالار كۆپەيگەنچە مۇنبەر يېڭى قۇرۇلغان ۋاقىتتىكى ،قىزغىنلىق دوستلۇق ،مېھرى -مۇھەببەت سۇسلاپ كەتكەندەك تېمىلاردا ھەدىسىلا بىر-بىرنىڭ قەستەن زىتىغا تىگىپ ئورۇنسىز جىدەل -ماجىرالار چىقىپ تۇرماقتا ،مەن بەزىدە مۇنبەر يېڭى قۇرۇلغان ۋاقىتىنى ئەسلەپ قالىمەن،ئۇچاغلاردا مۇنبەر ئىنتايىن كۆڭۈللۈك ئىدى ،ھازىرقىدەك ئېسىل ،نادىر تېمىلار ئانچە كۆپ بولمىسىمۇ مۇنبەر راستلا قىززىيتى ،مۇشائىرە ئىيتىشىش ،بىر-بىرىگە ئاتاپ يېزىلغان شئېرى چاقچاق ،لەتىپىلەر كۆپ ئىدى .مۇنبەرداشلار ئاكا-ئۇكا ئاچا-سىڭىلدەك ئىناق ئۆتەتتى .
ئۇۋاقىتتا مۇنبەردە نېمىلەر بار ئىدى ،تور قايتا ئىچىلىپ بىر مەزگىلدىن كېيىنغۇ دەيمەن ،تورداشلارنى مۇنبەرگە جەلب قىلىش ئۈچۈنمۇ ،دېھقانچىلىق مەيدانى ،چارۋىچىلىق فىرمىسى دېگەندەك قىزىقارلىق ئويۇن تۈرلىرى بارئىدى.مۇنبەرداشلار بۇ ئويۇندا بىرىنجى دەرىجىنى تالىشىشى ئۈچۈن خىلى قىزىقىپ ئوينايتتى ،بەزىدە ئويۇن نەتىجىسى مۇنبەردە قىزغىن پاراڭ تېمىسى بولاتتى ،يەنە مۇنبەر بانكىسى دەيدىغان بىر ئىقتىدار بار ئىدى گەرچە مۇنبەر پۇلىغا بىر چىش كولىغۇچ كەلمىسىمۇ بۇ ئارقىلىقمۇ نۇرغۇن يېڭى پاراڭلار قىززىق ئىشلار بولۇپ تۇراتتى ،مۇنبەردىكى مىليادىرلىرىمىز ئۇستۈنكى قەۋەتتە غادىيىپ تۇراتتى ،شۇ ۋاقىتتىكى داڭلىق مىليادىرلار ،بىرىنجى دەرىجىدىكى داڭلىق چولپانلار نېمە سەۋەپتىنكىنتاڭ ،بەزىلىرى چەكلىنىپتۇ ،بەزىلىرى كېرمەسلا بوپتۇ ،بەزىلىرى كېرسىمۇ پىكىر قاتناشتۇرمايدىكەن .مەن دېمەكچى بولغىنىم ئۇۋاقىتتا مۇنبەردىكى جىددى تالاش-تارتىش مۇنازىرىلەردىن كېيىن يەنە ئازراق بولسىمۇ مۇنبەر ئىچىدە كۆڭۈل ئىچىش شەكىللىرى كۆپ ئىدى دىمەكچى .
ھازىر قاراپ باقسام چاقچاق مۇشائىرە تېمىلىرىنى بىر مۇسلىنە ھەدەم كۆتۈرە ئىلىۋالغاندەك مۇنبەرگە پات-پات قىزغىنلىق ئىلىپ كېلىدىكەن ،مەنچە باشقا تورداشلارمۇ خۇشال بولغاندا ،ئىچى پۇشقاندا ،قىززىق -قىززىق گەپلەر بىلەن مۇنبەرنى قىززىتسا ،يۈرەك سۆزلىرىنى مۇنبەرگە تۆكۈپ ھەممەيلەن بىلەن بىرلىكتە ئورتاقلاشسا ،باشقا مۇنبەرداشلار بۇ خىل شەكىلدىكى يازمىلارنى ياقتۇرمىسا ئەخلەت ياكى يانا بىرنېمە دېمەي كۆزىنى يۇمۇۋىلىپ ئۆتكۈزىۋەتسە دېگەن ئۈمىدتىمەن .
باشقىلار يازغان تېما كۆڭلىڭىزگە يارىغان بولسا ئىنكاس يازمىسىڭىزمۇ ،ھىلىقى باھالاش دېگەن يەرنى بىسىپ مۇنبەر پۇلىغا ھىچنىمە كەلمىسىمۇ باھالاپ قويۇڭ ،مۇشۇ ئارقىلىق شۇ يازمىنى يازغان كىشى ئازغىنە بولسىمۇ ئىلھام مەدەتكە ،باشقىلارنىڭ ئىتىراپىغا ئىرىشىپ ئۆزىگە بولغان ئىشەنچىسى ئىشىپ قالسا ئەجەپ ئەمەس .
يەنە پات-پات مۇنبەرداشلارنى قىزىقتۇرالايدىغان تېمىلارنى يوللاپ تۇرساق ،مەسىلەن ئاۋۇ ئىسمىڭىزنى نېمىشقا بۇنداق قويدىڭىز ؟باش سۈرىتىڭىزنىڭ مەنىسى نېمە ؟مۇنبەردىكىلەرنى يوقلىما قىلىمىز ؟مۇنبەردىكىلەرنىڭ يىشىنى سىتاستىسكا قىلىمىز ؟مۇنبەردە ئاياللار كۆپمۇ ئەرلەرمۇ ؟.......قاتارلىق مۇشۇنىڭغا ئوخشاش ھەممەيلەن ئورتاق قىزىقىدىغان پىكىر قاتناشتۇرۇشنى خالايدىغان يۈرەك سۆزلىرىنى ئىيتالايدىغان تېمىلار كۆپرەك يوللانسا گەرچە بۇ خىلدىكى يازمىلارنىڭ ئەھمىيىتى ئانچە كۆپ بولمىسىمۇ لېكىن تورداشلارنىڭ بىر-بىرىنى چۈشىنىشى ،ئۇقۇشماسلىقلار ئازراق بولسىمۇ تۈگەيدۇ دەپ قارايمەن .
بولۇپمۇ قايسى بىر يىلى مۇنبەر ئەزالىرىنى باھالايمىز دەپ بىر تېما يوللانغان ئىدى ،تېمىغا چۈشكەن ئىنكاسلار ،تېمىنىڭ قىزىقتۇرىشى ھەقىقەتەن ياخشى بولغانىدى ،يىل ئاخىرى بوپ قالدى مېنىڭچە مۇشۇنداق تېمىدىن بىرسى يېزىلسا مۇنبەر خىلى قىززىيتى ،يىللىق باھالاش مەيلى مۇشۇ بىر يىل جەريانىدا قايسى جەھەتتىن بولسۇن كۆزگە كۆرۈنگەن ئەزالار باھالاپ چىقىلسا ،مەسىلەن بۇيىللىق ئەڭ ياخشى مۇنبەر باشقۇرغۇچىللىرى ،ئەڭ ياخشى تور يازغۇچىلىرى ،ئەڭ ئۇرۇشقاق ئەزالار ......دېگەندەك مۇشۇ دائىرىدە پىكىر قاتناشتۇرساق ئازراق بولسىمۇ دىمكوراتىيە ئىڭىمىزنى ئۆسسە ھەممەيلەن شۇ ئارقىلىق ئۆزىمىزنى بىر دەڭسەپ باقساق .
يەنە بىر مەسىلە شۇكى باشقۇرغۇچىلار مەسلىسى ،مەن بۇ مەسلىنى ئوتتۇرىغا قويۇشتىن بۇرۇن مەن بۇنبەرنىڭ سىتاستىسكا دەيدىغان بىر ئىقىتىدارىنىڭ ئەسلىگە كەلتۈرىشىنى ئۈمىد قىلىمەن ،بۇ ئىقتىدار بۇرۇن بار ئىدى،بۇ سىتاستىكا ئىقتىدارى ئارقىلىق بىز مۇنبەردىكى باشقۇرغۇچىلارنىڭ شۈكۈنى مۇنبەرگە كېرگەن كېرمىگەنلىكى ،كېرگەن بولسا قانداق مەشخۇلاتلارنى قىلغانلىقىدىن تارتىپ ئەزالارنىڭ يازما يوللاش ئەھۋالى ،نادىر تىمىلار قاتارى .......قاتارلىق نۇرغۇن مۇنبەرگە ئائىت ئىشلارنى بىلىپ تۇرغىلى بولاتتى ،لېكىن نېمە سەۋەبتىندۇر تاڭ بۇ ئىقتىدارمۇ يوقاپ كېتىپتۇ ،ھازىر قارايدىغان بولساق باشقۇرغۇچىلارنىڭ قانداق مەشخۇلات قىلغانلىقىنى ئەمەس ،مۇنبەردە نەچچە باشقۇرغۇچى باردۇ ؟ئۇلار كىملەر ؟دېگەننىمۇ ئىنىق بىلەلمەيمىز ،تېمىلارنىڭ ئۈن-تىنسىز غايىپ بولىشى ،بۇنىڭغا تورداشلارنىڭ پىكىرى كۈچلۈك .
ئەمەلىيەتتە بىزمۇ باشقۇرغۇچىلارنىڭ دەردىنى چۈشىنىمىز ،مۇنبەرنىڭ ئومۇمى كەيپىياتى ،چەكلىمىلەر تۈپەيلى ،بەزى تېمىلار ئۆچۈرۈلگەن بولشى مۇمكىن تېمىسى ئۆچۈرۈلگۈچى ئۆچۈرگەن باشقۇرغۇچىغا تىل سالغان بولشى،ياكى ئارىدا ئۇقۇشماسلىقلار بولغاچقا،ئۆچۈرگەن باشقۇرغۇچى ئەسكەتىش بىرىشتىن ئىھتىيات قىلغان بولىشى مۇمكىن ،باشقۇرغۇچى بولماق تەس ،لېكىن ئوچۇق-ئاشكارا بولۇش مۇھىم مەسىلە .
مۇنبەر توغرىسدا باشقۇرغۇچىلار توغرىسىدا نۇرغۇن تېما ،ئىنكاس پىكىرلەر يېزىپ ئۆتۈلدى ،لېكىن مەن بىرە باشقۇرغۇچىنىڭ ئۇتېمىلارغا قارىتا پىكىر قىلىپ يازغان تېمىسىنى كۆرمىدىم ،نېمە ئۈچۈن شۇنچە كۆپ تورداشلار تەلمۈرۈپ باشقۇرغۇچىلاردىن ئۇ مەسىلىگە چۈشۈنچە بىرىشىنى ساقلاۋاتسا ،بىرەر باشقۇرغۇچى ئوتتۇرىدىكى ئۇقۇشماسلىقنى تۈگىتىشكە تىرىشمايدۇ .نېمە ئۈچۈن تېمىلاردىكى بەزى خاتالىقلارنى كۆرسىتىپ بىرىلگەن ئىنكاسلارغا نەزىرىنى ئاغدۇرمايدۇ ؟بەزى تېمىلار مۇنازىرە جەريانىدا تالاش-تارتىش ئىغىز ئۇرۇشىغا ئايلىنىپ كەتسە باشقۇرغۇچىلار جىڭدە ئولتۇرغىدەك ئىنكاستىن بىرنى يېزىپ تېمىنى تۈز لىنىيەگە يىتەكلىمەي ئەكسىچە ھەدىسلا تېمىنى ئۆچۈرۈشكە كۈچەيدۇ ؟
ھەممە <<غەيۋەت>>نى مەنلا قىلىۋالماي باشقىلارمۇ ئانچە-مۇنچە قىلىپ باقسۇن ،ئىشنىقىلىپ مۇنبەر دېگەن رىئال دۇنيانىڭ كىچىكلىتىلگەن كارتىنىسى بىز بۇيەرگە خۇشاللىق ئىزدەپ ، (بەزىلەرنى ھىسابقائالمىغاندا )
بىلمىگەننى بىلىلى دەپ رىئال دۇنيادا تارتىۋاتقان بىسىملىرىمىزنى ئازراق بولسىمۇ يەڭگىللىتەيلى دەپ كېرىمىز ،شۇڭا بىز بۇ مەۋھۇم دۇنيادىن ئازراق بولسىمۇ خۇشاللىق تاپالىساق ،بىلمىگەننى بىلىۋالساق ،باشقا يەردە ھېس قىلالمىغان دىمكوراتىيە ،ئەركىن -ئازادىلىكنى ھېس قىلالىساق بۇ بىزنىڭ ئەڭ ئارزۇيىمىز بولسا كېرەك .

مۇنبەردە جەڭگە جىدەل قىلىش، مىسرانىمدىلا ئەمەس باشقا مۇنازىرە مۇنبەرلىرىدىمۇ بار ھازىر.
ئىلگىرى مەنمۇ مىسرانىمغا بەزى تېمىلارنى يوللاپ قوياتتىم، بىر شېئىرىم تۈپەيلى مۇنبەرداشلارنى تاس قالغان ئىككىگە ئايرىۋەتكىلى

ئەسسالامۇ ئەلەيكۇم!
ئەلزامان ئەپەندى:
غەيۋەت قىلىشقا شۇنچىلىك ئۆچ ئىدىم. بىراق بۇ يازمىڭىزدىكى باشقۇرغۇچىلار توغرۇلۇق قىلغان ‹‹غەيۋىتىڭىز››گە جۈر بولماسلىققا باھانە تاپالمىدىم.

قارىغاندا كۆپلىگەن تورداشلاردا باشقۇرغۇچىلارنى چۈشەنمەسلىك ئېغىردەك قىلىدۇ.  ئۆز تۇرمۇشىنىڭ تۈرلۈك ئىشلىرىدىن ئېشىنىپ، ئۇيغۇر تورچىللىقىنىڭ تەرەققىياتى، خەلقىمىزنىڭ مەنىۋىيىتىنى بېيىتىش ئۈچۈن خالىس خىزمەت قىلىۋاتقان باشقۇرغۇچىلاردىن ئاغرىنىشقا ھېچكىمنىڭ ھەققى يوق. چۈنكى مۇنبەر باشقۇرغۇچىلىرى نۇرغۇن كىشىلەر ئۆز ئۈستىگە ئېلىشنى خالىمايدىغان، خوشاللىقىدىن خاپىلىقى، ئاۋارىچىلىقى كۆپ بولغان بۇ خىزمەتنى يەنىلا ئۆز ئۈستىگە ئېلىپ كېلىۋاتىدۇ.   
   تور نازارەتچىلىك مىخاينىزىمىنىڭ كۈچەيتىلىشىگە ئەگىشىپ ھەر دەرىجىلىك ئورگانلارنىڭ مۇنبەر نازارەتچىلىك خىزمىتىمۇ ئىنتايىن چىڭ تۇتۇلۇپ ئىشلىنىۋاتىدۇ. بۇلارنىڭ ھەممىسى دۆلىتىمىزنىڭ مۇقىملىقى، خەلق ئاممىسىنىڭ بىخەتەرلىكى، تورداشلارنىڭ ئامانلىقى ئۈچۈن ئىشلىنىۋاتقان خىزمەتلەر.
  ئەتراپىمىزدا نېمە ئىشلار بولۇۋاتىدۇ، گېزىت-ژورنال، تېلۋىزورلاردا قانداق مەزمۇنلار تەشۋىق قىلىنىۋاتىدۇ؟ ھەممىمىز ئەتراپىمىزغا نەزەر-كۆزىمىزنى سېلىپ، گېزىت-ژورنال، تېلۋىزور كۆرۈپ تەشۋىق قىلىنىۋاتقان سىياسەتلىرىمىزدىن ۋاقتىدا خەۋەردار بولۇپ تۇرساقلا ساغلام بولغان يازما، ئىنكاسلىرىمىز بىلەن مۇنبىرىمىزنى تېخىمۇ تەرەققىي قىلدۇرالايمىز. بولمايدىكەن دۆلىتىمىزنىڭ مۇقىملىقى ئۈچۈن تەھدىت ئېلىپ كېلىدىغان بىر جۈملە سۆز، ياكى ئىنكاسنىڭ مۇنبەرگە چىقىشى مۇنبەر ۋە ئەزالارنىڭ بىخەتەرلىكى ئۈچۈن ئېغىر خەۋپ ئېلىپ كېلىدۇ. يۇقارقى قۇرلارنى ئوقۇغان ھەر بىر تورداشنىڭ تېمام نېمىشقا ئىز-دېرەكسىز يوقىلىپ كەتتى؟ نېمىشقا تەستىقلىنىپ نەچچە كۈندىن كېيىن ئۈچۈرۈلۈپ كېتىدۇ؟ دەپ زارلىنىشتىن ئىلگىرى سەۋەبنى بىزدىن ئەمەس بەلكى ئۆزىنىڭ يازمىسىدىن ۋە يازمىسىنىڭ سۈپىتىدىن(مۇنبەر مەشىق دەپتىرى ئەمەس) ئىزدەپ كۆرۈشىنى ئۈمىد قىلىمەن.

   مەندىن نۇرغۇنلىغان كىشىلەر شۇ بىرلا سۇئالنى سورايدۇ:
-باشقۇرغۇچىلىق قىلغىنىڭىزغا قانچىلىك ھەق بېرىدۇ؟
كۈلۈپ تۇرۇپ جاۋاب بېرىمەن:
- مەن بۇ خىزمەتنى خالىس قىلىمەن. مائاش بەرمەيدۇ.
ئۇلار مەسخىرە قىلىشىدۇ، سۆزلىرىمگە ئەسلا ئىشىنىشمەيدۇ. ھەممە ئىش پۇل بىلەن ئۆلچىنىدىغان مۇشۇ جەمىيەتتىمۇ ھەر كۈنى نەچچە سائەتلەپ كومپىيوتېر ئالدىدا ئولتۇرۇپ، ئۆزى تونۇمايدىغان بىراۋلارنىڭ ئەسىرى ئۈچۈن نۇرغۇن ۋاقتىنى ئىسراپ قىلىپ، بىكىرىغا ئىشلەيدىغان‹‹ساراڭ›› باشقۇرغۇچىلارنىڭ مەۋجۇت ئىكەنلىكىگە ھاڭ-تاڭ قېلىشىدۇ.
  ئۇنداقلاردىن رەنجىپمۇ ئولتۇرمايمەن، پەقەت كۆلۈپلا قويىمەن خالاس. خەلقىمىز ئۈچۈن ئازغىنە بولسىمۇ خىزمەت قىلىش پۇرسىتىگە ئېرىشىشنىڭ، باشقىلارغا كىچىككىنە بولسىمۇ مەنپەئەت يەتكۈزۈش ئارقىلىق ئۆزىگە خوشاللىق تېپىشىنىڭ بىر ئىنساننىڭ ئادىمىيلىك، مەسئۇليەتچانلىق تۇيغۇسىنىڭ ئويغانغانلىقىدىن دېرەك بېرىدىغانلىقىنى چۈشۈنۈشكە قادىر ئەمەس كىشىلەرگە ئېغىز ئۇپرىتىشنىڭ نېمە ھاجىتى؟
  بەزىلەر نەسىھەت قىلىشىدۇ:
-بۇنداق تۈكى بار ئىشنى قىلما! بىرەر يازمىدىن چاتاق چىقسىلا بىكار ئاۋارىچىلىققا قالىسەن.
  ئۇلارنىڭ ماڭا كۆيۈنگىنىدىن، بىخەتەرلىكىمگە كۆڭۈل بۆلگىنىدىن كۆڭلۈم تولىمۇ سۈيۈنۈپ كېتىدۇ. يۇشۇرۇپ ئولتۇرمايمەن. گايى كېچىلىرى شۇنداق ‹‹ قار بسىپ›› قالىدۇكى چۈچۈپ ئۆندەرەپ قۇپۇپ كېتىمەن. مۇنبەردىكى تېما ۋە ئىنكاسلارنى كۆرۈپ بىر تەرەپ قىلىپ بولغاندىن كېيىنلا، چۈشۈمدىكى مۇنبەر تېمىلىرى تۈپەيلى يۈز بەرگەن قورقۇنۇچلۇق كۆرۈنۈشلەر كاللامدىن چىقىپ كېتىدۇ.
  ئۆز- ئۆزۈمگە دائىم مەدەت بېرىمەن:
دادىل بول، كەسكىن بول، مەسئۇلىيەتچان بول! بىر-ئىككى يازما ئۈچۈن مۇنبەرنى، ئۆزۈڭنى قۇربان قىلىۋەتمە! سەندىن بۈگۈن رەنجىگەنلەر ئەتە سەندىن خوشال بولىدۇ. چۈنكى زىممەڭدە دۆلەتنىڭ مۇقىملىقى، مۇنبەرنىڭ بىخەتەرلىكى، يۈز مىڭدىن ئارتۇق ئەزانىڭ ئامانلىقىنى قوغداشتىن ئىبارەت مۇقەددەس، ئېغىر ۋەزىپە بار.  

ئارمانسىز كېتەتتىم ئۇشبۇ دۇنيادىن،
خەلقىمگە بىر بوردا نان بېرەلىسەم،
يىغلىسا تەڭ يىغلاپ، كۈلسە تەڭ كۈلۈپ،
