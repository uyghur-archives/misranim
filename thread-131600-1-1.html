بىلىشىمچە«بالىم»ماركىسىنى تونىمايدىغانلارمۇ خېلى باركەن ، ئالدى بىلەن بالىم شىركىتىنى تونۇشتۇرۋېتەي:

خوتەن بالىم خەلىقئارا سودا چەكلىك مەسئۇلىيەت شىركىتىنڭ قىسقىچە چۈشەندۈرلىشى

خوتەن بالىم خەلىقئارا سودا چەكلىك مەسئۇلىيەت شىركىتى 2010-يىىلى 10-ئايدا قارقاش ناھىيە بالىم دىھقانلار كەسپى ھەمكارلىق كوپراتىپى ئاساسىدا تەرەققى قىلىپ،  2012-يىلى 6-ئايدا ( خوتەن بالىم كەسپى تېىخنىكا تەربىيلەش مەكتىپى) نى رەسمى قۇرۇپ چىقتى. شىركىتىمىزنىڭ ئەنگە ئالدۇرغان مەبلىغى 10 مىلىيۇن بۇلۇپ، بۈگۈنگە كەلگەندە، ئۇمۇمى مۈلىكى 20 مىليۇن 500 مىڭ يۈەندىن ئاشىدىغان خەلىقئارالىق شىركەتكە تەرەققى قىلدى. شىركىتىمىز كىيىم-كىچەك تەتقىق قىلىش، لاھىيلەش، ئشلەپچىقىرىش، سېتىشنى بىر گەۋدە قىلغان كىيىم-كىچەك ئىشلەپچىقىرىش كارخانىسى بولۇپ ئەنگە  ئالدۇرغان  ماركىسىى«بالىم». شىركىتىمىز كىيىم-كىچەك لاھيىلەش، ئىشلەپچىقىرىشتىن باشقا يەنە ئەتلەس تۇقۇش ۋە سېتىش، قۇلدا تۇقۇلغان قوي يۇڭى گىلىمى تۇقۇش ۋە سېتىش، خوتەن قاشتېشى ۋە خوتەن چوڭ چىلىنى، خوتەن قەغەز ياڭىقىى قاتارلىق ئالاھىدە مەھسۇلاتلارنى سېتىش تىجارىتى بىلەن شۇغۇللىنىدۇ. شىركىتىمىزنڭ تىجارەت تۈرلىرى 12 خىل بۇلۇپ، ئاساسلىقى كىيىم –كىچەك ئىشلەپ چىقىرىش، گىلەم تۇقۇش، ئەتلەس تۇقۇپ سىتىش، ئۇنىڭدىن باشقا رەڭلىك چىنە خىش پىششىقلاپ ئىشلەش ۋە سېتىشتىن باشقا، رەخلەرنى پارچە سىتىش، قۇرۇق مىۋە-چىېۋە پىششىقلاپ ئىشلەش ۋە سىتىش، قاشتىشى سېتىش، ئائىلە مۇلازىمىتى، ئاۋئاتسىيە يۈك تۇشۇش تىرانىسپورتى ۋە ئايرىپلان بىلىتى سېتىش بىلەن شۇغۇللىنىدۇ. ئۇنىڭدىن باشقا شىركىتىمىزنىڭ خوتەن قىزىل يۇلتۇز سودا ساريى 1-نۇمۇرلۇق مەخسۇس كىيىم- كىچەك سىتىش دۇكىنى، 2-نۇمۇرلۇق ئاۋئاتسىيە يۇك تۇشۇش تىرانىسپورتى ۋە ئايروپىلان بىلتىنى مەخسۇس سېتىش دۇكىنى، خوتەن شەھرى جەمە يولىغا جايلاشقان ئايروپىلان بىلتىى سېتىش دۇكىنى، خوتەن شەھرى چوڭ بازار رەخىت سىتىش دۇكىنى ، قارقاش ئاۋئاتسىيە يۇك تۇشۇش تىرانىسپورتى ۋە ئايروپىلان بىلتىنى مەخسۇس سېتىش دۇكىنىىدىن ئىبارەت 6 زەنجىرسىمان دۇكىنى بار. بىرلا ۋاقىتتا قارقاش ناھىيە زاۋا بازىرنىڭ مەركىزىگە ئەڭ چوڭ كۆلەمدە كىيىم –كىچەك سىتىش،قولدا تۇقۇلغان  قوي يۇڭى گىلىمى تۇقۇش، ئەتلەس تۇقۇش، رەڭلىك چىنە خىش پىششىقلاپ ئىشلەش ۋە سىتىش ئۇنىڭدىن باشقا پىششىقلاپ ئىشلەش ئۇرۇنلىرى تەسىىس قىلىندى. شىركىتىمىز قۇرۇلغان قېسقىغىنا ۋاقىت ئىچىدە تىجارەت دائىرسى پۈتۈن شىنجاڭ ۋە ئىچكىرى ئۆلكىلەردىكى تەرەققى قلغان شەھەرلەرنى قاپلىدى. كۆپلىگەن شەھەر يىزالاردىكى ئەمگەك كۈچلىرنىڭ ئىشسقا ئۇرۇنلىشىش مەسىلىسىنى ھەل قىلدى. شىركىتىمىز تەرەققى قىلغاندىن بۇيانقى شىركىتىمىزنڭ بىردىن بىر ئەۋزەللىكى  ئائىلىسىدە قىيىنچىلىقى بار،  قىز– ئۇغۇللارنى ئىشسقا ئۇرۇنلاشتۇرۇشنى مەخسەت قىلغان ھەم توختىماستىن ئۈزىنىڭ يۇشۇرۇن بىلىم كۈچىنى قىزىپ، ئىشلەپ چىقىرىش يوللىرىنى ئىچىپ، كۆپخىل تىجارەت تۈرلىرنى يولغا قۇيۇپ، ئۈزىنڭ ئقتىدارىنى تەرەققى قىلدۇرۇپ ۋە كۈچەيتىپ، يىزىلاردىكى ئەمگەك كۈچلىرنى ئىلگىرى سۈرۈپ، ئىش ئورنىدىن قالغان ئىشچى–خىزمەتچىلەرنى ۋە (پىلانلىق تۇققان) لارنى ئىشقا ئۇرۇنلاشتۇرۇش، ئىقتىساسلىقلارنى كۆپەيتىش، ھەرخىل كەسىپلەرنى تەرەققى قىلدۇرۇشقا كاپالەتلىك قىلىش. نۆۋەتتە قارقاش ناھىيە زاۋا بازىرى لەڭگە كەنتى دۆلەت يولى يىنىدىكى لاھىيلەش، تەربىيلەش، ئىشقا ئۇرۇنلاشتۇرۇش، سېتىش بىر گەۋدە قىلىنغان كەسپى تېخنىكا تەربىيلەش ئىشقا ئۇرۇنلاشتۇرۇش بازىسى، يەنى (قارقاش ناھىيە بالىم كەسپى تېىخنىكا تەربىيلەش مەكتىپى) نى تەسىس قىلدى. شىركەت ئىگەللىگەن ئورنى 48 مو بۇلۇپ، كىيىم –كىچەك پىچىش ۋە لاھىيلەش ئۈسكۈنلىرى بىر گەۋدە قىلىنغان، تېكىش ماشىنىسىدىن 165، گىلەم جازىسى 50 دانە، ئەتلەس تۇقۇش ماشىنىسى 25 بۇلۇپ، يەنى چوڭ تىپتىكى گۈل كەشتىلەش ماشىنىسىدىن بىر يۇرۇش سەپلەنگەن. ھەر خىل تۈردە كىيىم –كىچەك پىچىش، تەربىيلەش، لاھىيلەش ۋە تىكىش، گىلەم تۇقۇش، ئەتلەس تۇقۇش، ئۇيغۇرلارنىڭ گۈللۈك دوپپىسىنى تۇقۇش، ھەرخىل گۈل كەشتىلەش قاتارلىق كەسىپلەردە تەربىيلەيمىز. يىللىق تەربىيلەش ئادەم  سانى 2000 ئادەمدىن ئاشىدۇ. تەربىيلەشتە 80%تىن يۇقىرى كۇرسانىتنى خىزمەت ئورنى بىلەن تەمىنلەشتىن باشقا، ھەر ئايدا  ئايلىق مائاشتىن بەھىرمان قىلىدۇ. مائاشى 2000 مىڭ يۈەن ئەتراپىدا چىقىدۇ. تەربىيلەپ ئىشقا ئۇرۇنلاشتۇرۇش، كىيم-كىچەك تىكىش بازىمىزدا نەزىرىيەنى ئەمىلىيەتكە بىرلەشتۈرۈپ ئۆگۈتىشتىن باشقا، ئۇقۇتۇشنى قىلىپلاشتۇرۇپ باشقۇرۇش، دەرىسلەرنى مۇۋاپىق تەسىس قىلىپ، كۇرسانىتلارنڭ ئەمەلىي قول سىلىپ ئىشلەش ئىقتىدارنى يۇقىرى كۆتۈرۈش، كۇرسانىتلار تەربىيلىنىشنى باشىتىن كەچۇرگەندىن كېيىن، ئىشقا ئۇرۇنلاشتۇرۇلىدۇ. 2014-يىلى ھەر تۈردىن 32 سىنىپ ئىچىلىپ، تەربىيلەنگەن كۇرسانىتلارنىڭ سانى 1732 نەپەرگە يىتىپ، لايەقەتلىك بۇلۇش نىسپىتى 90%تىن يۇقىرى بولدى. يېزىلاردىكى ئارتۇقچە ئەمگەك كۈچى ۋە ئىقتىدارىنى يۇقىرى كۆتۇرۇپ، ئەمگەك كۈچلىرنىڭ ئىشقا ئۇرۇنلىشىشىنى ئىلگىرى سۈرۈش، دىھقانلارنڭ كىرىمىنى ئىلگىرى سۈرۈپ يۇقىرى كۆتۈرۈشنى مەخسەت قىلغان.نۆۋەتتە قاراقاش ناھىيە بالىم كەسپى تېخىنىكا تەربىيلەش مەكتىپى ۋە قاراقاش ناھىيە كەسپى تېخنىكا تۇلۇق ئوتتۇرا مەكتىپى بىلەن ھەمكارلىشىپ،  قاراقاش ناھىيە بالىم كەسپى تېخىنىكا تەربىيلەش مەكتىپىنڭ تارمىقىنى،  زاۋا بازىرىغا قاراقاش ناھىيە بالىم كەسپى تېخىكا  تۇلۇق ئوتتۇرا تەربىيلەش مەكتىپنى تەسىس قىلدى. قاراقاش ناھىيە كەسپى تېخىكا تۇلۇق ئوتتۇرا تەربىيلەش مەكتىپى، "بالىم خەلىقئارا سودا چەكلىك مەسئۇلىيەت شىركىتىنڭ باش لىدىرى ئۇبۇلھەسەن  ئۇبۇلقاسىم"ھازىر قاراقاش ناھىيە كەسپى تېخىنىكا تۇلۇق ئوتتۇرا مەكتىپىنڭ تەكلىپلىك ئوقۇتقۇچىسى بولۇش بىلەن بىرگە، شۇ مەكتەپنىڭ مۇئاۋىن مەكتەپ مۇدىرلىق ۋەزىپىسىنى ئۆتەۋاتىدۇ. مەكتەپنڭ تېخنىكا ئىقتىدارىنى يۇقىرى كۆتۈرۈش ئۈچۈن، قاراقاش ناھىيە بالىم كەسپى تېخىكا تەربىيلەش مەكتىپى ئىچكىردىكى ئىلغار تېخنىكا ۋە تالانىت ئىگلىرىدىن 11 نەپەر كىيىم –كىچەك لاھىيلىگۈچىنى يۇقىرى  مائاش (300000 يۈەن ) بىلەن مەكتىپىمىزدە ئۇقۇتۇش خىزمىتى ۋە ئۇقۇتۇشنى باشقۇرۇش، تېخنىكىغا يىتەكچىلىك قىلىش  كەسپگە تەكلىپ بىلەن ئىشلەۋاتىدۇ. شىركىتىمىز كارخانىنڭ مائارىپ ۋە ئۇقۇتۇش سەۋىيسىنى يۇقىرى كۆتۈرۈش ئۈچۈن گۇۋاڭدوڭ كىيىم- كىچەك لاھىيلەش مەكتىپى بىلەن كىلىشىم تۈزۈپ، ھەريىلى ئەلا نەتىجە بىلەن ئۇقۇش پۈتتۈرگەن ئۇقۇغۇچىلارنى شىركىتىمىزگە يەتكۈزۈپ بىرىدۇ. يېقىنقى يىللاردىن بۇيان، قاراقاش ناھىيە بالىم كەسپى تېخىنىكا تەربىيلەش مەكتىپىنڭ ئۇقۇتقۇچىلار سىپى، ئىلغار باشقۇرۇش ئۇسۇلى ۋە مول تەجىربىلەرنى يەكۈنلەپ، كۇرسانىتلارنڭ ھەر جەھەتتىن ئۇقۇتۇش ۋە ئۈگۈنۈش سۇپىتىنى ياخشىلاش ۋە يۇقىرى كۈتۈرۈش. ئۇقۇتۇشتا ئاساسلىقى نەزىرىيەنى ئەمىلىيەتكە بىرلەشتۈرۈشنى ئاساس قىلىشتىن باشقا ئۇقۇتۇشنى باشقۇرۇشنى ئۆلچەم قىلغان بۇلۇپ، دەرىسلەرنى ئۇرۇنلاشتۇرلىشى مۇۋاپىق، كۇرسانىتلارنىڭ ئەمەلى قول سىلىپ ئىشلەش ئىقتىدارنى يۇقىرى كۆتۈرۈپ، كۇرسانىتلار تەربىيلەشتىن ئۆتكەندە، ئىشقا ئۇرۇنلاشتۇرۇش مۇلازىمىتىدە چىڭ تۇرۇش، دۆلەتنىڭ ئىشقا ئۇرۇنلاشتۇرۇش ۋە ئىشقا ئۇرۇنلىشىش پىرىنسىپىنى يەكۈنلەپ چىقىش، ھۆكۈمەتنىڭ تەربىيلەش تۈرلىرى ۋە ئۇسۇللىرىنى ئۈستىگە ئىلىپ ،ئەمگەكچىلەرنىڭ ساپا مۇلازىمىتىنى يۇقىرى كۆتۈرۈپ، ھەرقايسى ئۇرۇنلارنىڭ ئىقتىسادى تەرەققىياتى ئۈچۈن بىر كىشلىك تۆھپە قۇشۇشنى مەخسەت قىلغان.شىركىتىمىز “كىيىم – كىچەك، تۇقۇمىچىلىق سانائىتىنى ئىلگىرى سۈرۈپ يۈزمىڭلىغان ئادەملەرنىڭ ئىشقا ئۇرۇنلاشتۇرۇش” چاقىرقىغا ئاكىتىپ ئاۋاز قۇشۇپ، باشلامچىلىق بىلەن بۇ تۈرنىڭ ھەقىقى ئەمەلگە ئىشىشى ئۈچۈن جان دىل بىلەن خىزمەت قىلىدۇ. بۇ جەھەتلەردە شىركىتىمىزنىڭ تەرەققىياتى پارتىيە ھۈكۈمەتنىڭ ياردىمىدىن ئايرىلىپ قالغىنى يوق . (1). قارقاش ناھىيلىك پارتىكوم خەلىق ھۈكۈمىتى شىركىتىمىزگە قەدەممۇ قەدەم ياردەم بىرىۋاتىدۇ. يەنى قارقاش ناھىيلىك پارتىكوم خەلىق ھۈكۈمىتى ، شىركەت ئىشلىتىۋاتقان ئۇرۇننىڭ ئىجارە ھەققىنى تۈۋەنلىتىپ ، ئەسلىدىكى زاۋا بازىرى، بازارلىق خەلىق ھۈكۈمىتىنڭ ھويلىسىنى شىركىتىمىزگە ئەڭ ئەرزان باھادا 40 يىللىق ئىجارىگە بەردى. بۇ ئارقىلىق مائارىپ، ئۇقۇ-ئۇقۇتۇش، تەربىيلەش ئورنى“قاراقاش ناھىيە بالىم كەسپى تېخىكا تۇلۇق ئوتتۇرا مەكتىپى” نىڭ ئورنىنى، مەخسۇس  ئۇقۇتۇش ئىلىپ بىرىش ۋە ھەر تەرەپلەردىن يىتەكچىلىك قىىلىپ كەسپى تېخىنىكا جەھەتتىكى قىيىنچىلىقىمىزدا  ماسلىشىپ كىلىۋاتىدۇ. (2). قاراقاش ناھىيە بالىم كەسپى تېخىنىكا تەربىيلەش تۇلۇق ئوتتۇرا مەكتىپى يەنىمۇ ئىلگىرلىگەن ئاساستا، دىھقانلارنىڭ ئەمگەك كۇچىنى راۋاجلاندۇرۇش، "پىلانلىق تۇققان "لارنى تېخنىكا جەھەتتىن تەربىيلەپ، تەربىيلەشتە لاياقەتلىك بولغانلارنى شىركىتىمىزدە ئۇزۇن مۇددەتلىك ئەمگەك توختامى تۇزۈپ، ئىشقا ئۇرۇنلاشتۇرلىدۇ. (3). شىركەتنڭ "2014يىللىق تىز ئەمىللىشىدىغان يۈز مڭلىغان ئادەمنڭ ئىشقا ئۇرۇنلىشىشىغا كاپالەتلىك قىلىش" تۈر مەبلەغ پۇلدىن 8مىلىيۇن يۈەن ياردەم قىلدى. ھەمدە شىركىتىمى زنڭ 2 مىلىيۇن يۈەن كىپىلك قەرزى بىلەن تەمىنلىدى. (4).شىركىتىمىزگە ناھىيلىك  ئىگىلىك تىكلەش مەكتىپى،  قارقاش ناھىيە ئوتتۇرا دەرىجىلىك  كەسپى تېخنىكا مەكتىپىنڭ كەسپى بازىسىدىن 4000 كىۋادىرات مىتىرلىق كەسپى تەجىربە بىناسى بىلەن تەمىنلىدى. مەكتەپ بىلەن كارخانىنى بىرلەشتۈرۈپ، مەكتەپ قۇرۇشنى ئۈلگە قىلىپ، ئقتىسادنى راۋاجلاندۇرۇش ئۈچۈن ئالدىنقى 3يىللىق ئىجارە ھەققىنى كەچۈرۈم قىلىش. ئۆي زىمىن خىراجەتلىردىن سۇ، توك، گاز پۇللىرنى يىغىش. 3 يىلدىن كىيىن شۇ ۋاقىتتىكى بازار باھاسىنڭ ئەڭ تۈۋەن ئىجارە ھەققى بۇيىچە ھەق تۈلەش شەكلى بىلەن ئىجارىگە بەردى.پارتىيە ۋە ھۈكۈمەتنڭ مۇشۇنداق ياردىمى ئاستىدا،  كارخانىمىزدىكى ئىشچى –خىزمەتچىلەرنڭ ياتاقلىرى ئۆزگەرتىلدى. تۇرالغۇسى بار بولدى. تاماق مەسىلىسىمۇ، تۇرمۇش مەسىلىسىمۇ ھەل بولۇپ، نۇرمال ئىشلەپچىقىرىش  شارائىتى ھازىرلاپ بىرىشتىن باشقا، شىركىتىمىزگە بوستان كۆل ئچىۋىتىلگەن سانائەت رايۇنىدن زاۋۇت بىناسى قۇرۇلۇش  قىلىش ئۇچۇن 30 مۇ يەر بىلەن تەمىنلىدى. (5).ناھىيلىك ئەمگەك كۈچى ئىجتىمائى كاپالەت ئىدارىسى ھەرخىل تۈردىكى ھەقسىز ئىلان تەشۋىقات ئۇسۇللىرى ئارقىلىق شىركىتىمىزنى ئىشقا ئۇرۇنلاشتۇرۇش مۇلازىمىتى بىلەن تەمىنلىدى.
نۆۋەتتە، خوتەن بالىم خەلىقئارا سودا چەكلىك مەسئۇلىيەت شىركىتىمىز "2013-يىللق جۇڭگۇ –ئاسىيا ياۋروپا كۆرگەزمىسىدە ئىلغار كارخانا " بۇلۇپ باھالاندى. شىركەت باش لىدىرى ئۇبۇلھەسەن ئۇبۇلقاسىم 2014- يىلى شىنجاڭ ئۇيغۇر ئاپتۇنۇم رايونلوق كىيىم –كىچەك ئۇيۇشمىسىنڭ مۇئاۋىن باشلىقى، "دۆلەت ئىقتىسادى تەرەققىيات  مەركىزىنڭ قول ھۈنەرۋەنچىلىك كەسپىنڭ يۇقرى دەرىجىلىك تەتقىقاتچىسى" بۇلۇپ باھالاندى. بىرلا ۋاقىتتا، "2012-يىللىق جۇڭگۇدىكى ئىسلاھاتقا يىڭىلىك ياراتقۇچى ئىلغار شەخىس"،"خوتەن ۋىلايەتلىك بىرىنجى نۆۋەتلىك ئىگلىك تىكلەش مۇسابىقىسى ۋە ئاپتۇنۇم رايۇنلوق 2 نۆۋەتلىك ئىگلىك تىكلەش مۇسابىقىسىدە 2-دەرجىلىك مۇكاپات " قا ئىرىشكەن، "2012-يىللق جۇڭگۇ ئىسلاھاتىدىكى يىڭىلىق يارتىپ تۆھپە قوشقان كارخانا"، " 2012 يىللىق ئىناق جۇڭگۇ بەرپا قىلىشتىكى ئىلغار شەخىس"، "ئالى مەكتەپ ئۇقۇغۇچىلىرنىڭ ئىگلىك تىكلەش ئۈلگىىسى"،"2014-يىللىق شىنجاڭدىكى 10 چوڭ داڭلىق ماركىلىق كارخانا ” نىڭ بىرى ،"ۋىلايەت دەرىىجىلىك ئاۋانگارت ئىشچىلار ئۇيۇشمىسى" دىگەن شەرەپلىك نامغا ئىرىشكەن ، "ناھىيە دەرىجىلىك ئەلا كارخانا" دىگەن نامغا ، "2014-يىللىق شىنجاڭنى ئىچىۋىتىش، قۇرۇش تەقدىرنامىسى"،"شىنجاڭنى ئىچىۋىتىش، قۇرۇش مىدالى"،" ئاپتۇنۇم رايۇنلۇق ئىشچىلار ئاۋانگارتى" دىگەن شەرەپلىك ناملارغا ئىرىشكەن بۇلۇپ، شىركىتىمىز ئۈزىنى تەرەققى قىلدۇرۇش بىلەن بىللە، جامائەت پاراۋانلىق ئىشلىرىغا ئاكىىپ قاتنىشىپ، جەمىيئەتكە جاۋاپ قايتۇردى. شىركەت زاۋا بازىرى لەڭگەر كەنتى خەلقىنىڭ تۇربا سۈيى ئىچىش قىيىن بۇلۇش مەسىلىسىنى ھەل قىلىش ئۈچۈن، ھەر يىلى شىركەت مالىيەسىدىن 12000 يۈەن ئاجىرتىپ، كەڭ خەلىق ئاممىسىنڭ سۇ ئىچىش مەسىلىسىنى ھەل قىلدى. ھەمدە لەڭگەر كەنتى باشلانغۇچ مەكتىپىگە 4300 يۈەن قىممىتىدىكى مەكتەپ فورمىسىنى ئىئانە قىلدى . ھەر يىلى بايرام كەلگەندە ، ئائىلىسىدە قىيىنچىلىقى بار خەلىققە ئۇن. ماي، گۈرۈچ قاتارلىق تۇرمۇش بۇيۇملىرنى تارقىتىپ بىرىپ، نامرات خەلىقتىن ھال سوردى.ھەر يىلى يىتىم –يىسىر پاراۋانلىقى مەكتىپىنڭ ئۇقۇغۇچىلىرى  ئۈچۈن 25000 يۈەن قىمىتىدىكى ئۈگۈنۈش ۋە تۇرمۇش بۇيۇملىرى ئىئانە قىلدى. بۇيىل كىرىيەدە يەر تەۋرىگەندە، ئاپەت يۈز بەرگەن رايۇنلارغا 170000 يۈەن قىممىتىدىكى تۇرمۇش بۇيۇملىرنى ياردەم قىلدى.دۆلەتنڭ خەلىقنى بىيىتىش سىياسىتى ئاستىدا، خوتەننىڭ ئقتىسادى تەرەققى قىلىپ ، خوتەن بالىم خەلىقئارا سودا چەكلىك مەسئۇلىيەت شىركىتى تەرەققىياتى ئەڭ تىز بولغان خۇسۇسى كارخانىلارنڭ بىرى بۇلۇپ، ئۈزىگە خاس ئەۋزەللىك بىلەن ۋىلايىتىمىزنڭ ئىقتىسادى تەرەققىياتى ۋە جەميئەتنڭ مۇقىملىقى ئۈچۈن بىر كىشلىك تۆھپىسىنى قۇشۇپ كىلىۋاتىدۇ.
