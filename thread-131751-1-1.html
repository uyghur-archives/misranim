ئەسسالامۇ ئەلەيكۇم مۇنبەر ئەھلى ھەممىڭلارنىڭ مۇشۇ ۋاقتى خەيرىلىك بولسۇن. كەمىنە  ARMANI123 مۇنبەرگە ئەزا بۇلۇپ كىرگىلى ئانچە ئۇزاق بولمىدى. گەرچە بۇرۇنلام كىرىشنى ئويلىغان بولساممۇ، ئالدىراش تۇرمۇش رىتىمىنىڭ تىز بۇلۇشى ،شۇنداقلا ئۇنۋىرسال كەسپى بىلىمىمنىڭ چەكلىك بۇلۇشى تۈپەيلى سەل ئەنسىرگەن ئىدىم، ئاخىرى غەيرەتكە كىلىپ مۇنبەردىكى، قەلەم كەش شائىر، ئەدىبلەر ھەر ساھەدە يىتىشىپ چىقىۋاتقان دەۋىرمىز ياشلىرى بىلەن ھەم سۆھپەت بۇلۇش شەرىپىگە ئېرىشتىم.

ئۆزەمگە كەلسەك پۇتبولنىڭ ئوت مەستانىسى پۇتبولغابەك قىزىقىمەن مەستانىچىلىك ھاياتىمغىمۇ20يىلچە بۇلۇپ قاپتۇ. دۇنيا لوڭقىسى، ياۋرۇپا لوڭقىسى، ياكى ياۋرۇپا ئۇزغۇنلۇق، باشقا تۈردىكى ھەرتۈلۈك مۇسابىقىلەرنى ئۇزۇن يىل كۈرۈپتىمەن. مۇنبەر گىمۇ تېما يوللاش خىيالىم بولسىمۇ، قايسى مەزمۇندا يوللاشنى بىلەممەي. يەنىلام ئۆزقىزىقىشىم بۇيىچە شىنجاڭ پۇتبولچىلىقغا ئائىت ئەسلىمىلەرنى ئېلىپ كەلدىم ئەيىپكە بۇيرىمىغايسىلە.

مەمىمىزگەمەلۇم 2012-يىلى شىنجاڭ تاشئالما كوماندىسى قۇرۇلۇپ   ب  دەرىجىلىك مۇسابىقىگە قاتنىشىپ پۈتۈن مەملىكەت تەۋەسىدە كوماندىمىزنىڭ نامى چىققان.( سايىپخان مەيدان تاماشىبىن سانى كۆپ بۇلۇش پۇتبول قىزغىنلىقى جەھەتتە) شۇچاغدىكى ھاياجانلىق مىنۇتلارنى ئەسلىسەم كۆزۈمگە ھازىرمۇ  ئىسسىق ياش كىلىدۇ. مەن ئۈچۈن شۇمۇسابىقىنى نەق مەيدانىدن تۇرۇپ قانغۇچە ھۇررا توۋلاپ مۇسابقەكۆرگەن چاغلارھەرقانداق بىرمۇسابقە كۆرگەن، چاغدىكى ھېسىياتقا تامامەن ئوخشىمايتى. شۇكۈنلەرنى پەقەت ئۇنۇتمايمەن . شۇنداق كۈنلەر يەنەكىلەرمۇ دەپ ئەسلەپمۇ قالىمەن. بۈگۈن يەنە تورئارلاۋېتىپ سابىق شىنجاڭ تاشئالما كوماندىسىغا مۇناسىۋەتلىك فىلىملەرنى كۆرۈپ چۇڭقۇر ئوي خىيالغا چۆكتۈم تەسىرلەندىم . ئاللاھ خالىسا بۇيىل ھەرمباغ كۇلۇبى قۇرۇلۇپ  ب دەرىجىلىك مۇسابىقىگە قاتنىشىپ قالسا بىز دەك پۇتبول ھەۋەسكارلىرى شۇنداقلام پۈتكۈل شىنجاڭ خەلقى ئۈچۈن ئەڭ چوڭ خوشاللىق ئىش بولغۇسى .مەن ئىشىنىمەكى دىيارىمىزنىڭ پۇتبول روھى ، مىللىتىمىزنىڭ پۇتبولغا بولغان ئۆتتەك قىزىقىشى مەڭگۈ توختاپ قالمايدۇ. ئەنە شۇكۈنلەر ھامىنى بىركۈنى يەنە كىلىدۇ.

قېنى ئەمىسە شۇ گۈزەل ھاياجانلىق مىنۇتلارنى بىرگە ئەسلەيلى،









ئىزاھات سابىق شنجاڭ تاشئالما كوماندىسى، 2012-يىللىق جۇڭگۇ  ب  دەرجىلىكلەر مۇسابىقىسىگە قاتناشقان بىرلەشمە مۇسابىقىگە 13كوماندا قاتناشقان بۇلۇپ شىنجاڭ تاشئالما كوماندىسى بىرلەشمىدە5-بولغان بىرلەشمە مۇسابىقە ئاياقلاشقاندىن كىيىن شىجاڭ كوماندىسىنى قوللاپ كەلگەن خەيتاڭ گوروھى چىكىنپ چىققانلقىتىن باش-پاناسىز قالغان شىجاڭ كوماندىسى مەملىكەتلىك نەنھەركەت مۇسابىقىسى، پۇتبولچىلار جەمىيىتى لوڭقىسى مۇسابىقىسىگە قاتنىشىپ بولغاندىن كىيىن تارقىلىپ كەتكەن.
