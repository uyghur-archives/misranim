ئەتتىگەن ئىشخانىغا شۇنچە ياخشى كەيپىياتتا كىرگەن ئېدىم ئەمما كەلگەن بىر تېلىفۇن بىلەنلا كەيپىياتىم لاي سۇغىلا ئوخشاپ قالدى .
مۇشۇ بىز ئاياللارنىڭ نۇرغۇن ياخشى تەرەپلىرىمىزدىن نۇرغۇن ئەرلەر سۆيۈنسىمۇ ، ئەمما بەزى بىر تەرەپلىرىمىز ئۇلارنىڭمۇ دېتىغا ياقماي قېلىۋاتىدۇ . بۇ ئىشلارنىڭ مەنبەسى نېمە دېگەندە يەنىلا شۇ بىزدىكى << مودا >> ئېقىمىغا ئايىقىمىز بىلەنلا سەكرىگەنلىكىمىز . مۇشۇ چاي دېسە ، قېنى دەپ بېقىڭلار، مۇنبەردىكى خانىم -قىزلار قايسىڭلارنىڭ ئەرلىرى قوش قوللاپ قوللايدۇ ؟ مېنىڭ يولدۇشۇممۇ بۇنىڭغا قارشى . شۇڭا ئىلگىرى قوللىمىسىمۇ ، بىر تاتلىق سۆزلەر بىلەن ئۇنى ئىۋاشقا كەلتۈرۈپ ئايدا بىر ياكى ئىككى چاي ئوينايتتىم . ئەمما بالىلىق بۇلۇپ مەسئۇليىتىمىز ئېغىرلىغاندا  شۇنى ھېس قىلىپ قالدىمكى بۇ بىر ۋاقىت ، زېھىن ئىسراپچىلىقى ئىكەن . شۇڭا ئويناۋاتقان چايلىرىمدىن قول ئۈزگىلىمۇ خېلى يىللار بۇلۇپ قالغان . ئەمما بۈگۈنكى بۇ تېلىفۇن مېنى تولىمۇ ھەيران قالدۇرۇپ قويدى ؛
ـــــ ئەسسالامۇ -ئەلەيكۇم ، ياخشىمۇ سىز تۇرسۇنئاي  ، ئۇزۇن بوپتۇ ئالاقىمۇ قىلىشمىغىلى ؛
ـــــ ۋەئەلەيكۇم - ئەسسالام . زەينۇرمۇ سىز ؟ ياخشى تۇردىڭىزمۇ ؟ شۇنى دەڭە ، خېلى ئۇزۇن بوپتۇ بىز كۆرۈشۈپ باقمىغىلى ، مەھەللىدىن كۆچۈپ كەتكەندىن بېرى سىزنى تۈزۈكرەك ئۇچۇرتۇپ باقماپتىمەن .
ــــ ۋاي شۇنى دەڭە ! سىز شۇ باللىرىم دەپلا ماۋۇ خۇشنىلار چېيىدىنمۇ چىقىۋالدىڭىز . چاي ئوينىمىسىڭىز مۇشۇنداق كۆرۈشەلمەي قالىدىغان ئىشكەن . ئەمدى باللىرىڭىزمۇ چوڭ بۇلۇپ قالدى ، كېلەر ئايدىن باشلاپ <نوركا >چېيىنى باشلايلى دېيىشكەن . شۇڭا سىزنىمۇ ئوينامدىكىن دەپ تېلىفۇن قىلىشىم ئېدى .
ــــ ھە مۇنداق دەڭ ، ئوينىسامغۇ ياخشى بۇلاتتى ، ئەمما يەنە شۇ باللىرىمنى تاشلاپ قۇيالمايمەن ،   چوڭ قىزىم تۇلۇقسىزغا چىقتى ، كىچىكلىرى بىرىنچى سىنىپقا كىردى . كۈندە شۇ بالىلارنى مەكتەپكە ئاپىرىپ ،ئەكىلىپ ، خىزمەت قىلىپ بوش ۋاقتىمىزمۇ چىقمايدۇ . ھازىرقى بالىلارنىڭ تاپشۇرىقىنىڭ جىقلىقىنى بىر دېمەڭ .
ــــ دېمەك ئوينىمايمەن دېمەكچىمۇ خوشنام ؟
ــــ ھەئە ، ئوينىيالمايمەن .  توغرا چۈشەنگەن بولسىڭىز .
ــــ ۋاي خۇدا  ، ئالدىنقى قېتىملىق <خىرىستال > ، < ئالتۇن > چايلىرىمىزغىمۇ شۇنداق باھانە بىلەن قاتناشمىغان ئېدىڭىز ، بۇداقىسىغىمۇ قاتناشمايمەن دېسىڭىز ، ئەمدى سىزنى خوشنىدارچىلىقى يوقكەن دەپ قالىمىز . يا بىزنى ياراتمامسىز بىلمىدۇق ؟!
ــــ ۋاي ھەرگىز ئۇنداق ئويلىماڭ ، مېنىڭمۇ سىلەر بىلەن بىرە كۈن مۇڭدۇشۇپ ئولتۇرۇپ كەلگىم كىلىدۇ . ئەمما شۇ سىزمۇ بىلىسىز ، يولدۇشۇم سىرتتا ئىشلىگەچ كۈندە ئۆيگە كىلىپ بۇلالمايدۇ . شۇڭا ئۈچ بالىنىڭ ھاجىتىدىن چىقىپ بولغۇچە كۈنلەرنىڭ قانداق ئۆتۈپ كىتىۋاتقىنىنىمۇ سەزمەي قېلىۋاتىمەن . راستلا سەت بولدى زەينۇر ، كېيىنكى چايلىرىڭلارغا قوشۇلۋالارمەن .
ــــ ۋاي ئۇنداق دېمەي ئوينىڭە ، بۇداقىسى باشقا چايلارغا ئوخشۇمايدۇ ، نوركىنىڭ چىشىنى چاقىدىغان چاي بۇ ، مۇشۇنداق چاي ئويناپ كىيىۋالمىساق ئۇ نوركىنى ، ئاۋۇ ئەرلىرىمىزنىڭ ئېلىپ بېرىشىغا قاراپ ئولتۇرساق ، ئۆمۈر بويى كىيەلمەيمىز . ئوتكەنكى ئالتۇن چېيىمىزدىمۇ ، چايدىكى ئونىمىز ئالتۇنلۇق بۇلۇپ قالدۇق . مۇشۇنداق باھانە سەۋەپ بىلەن كۆزىمىزنى يۇمۇپلا ئالمىساق ، باشقىلارنىڭكىگە قاراپ ھەۋەس قىلىپلا ئولتۇرامدۇق خوشنام !  ئەمدى ئۇنىمىسڭىز خوشنىلارنىڭ ھەممىسى سىزدىن راسلا رەنجىپ قالىدۇ ، بۇ يىل ئوغلۇنى ياتقۇزىدىكەن دەپ ئاڭلىدىم سىزنى ، ناۋادا بۇداقى چايدىن چۈشۈپ قالسىڭىز ، ئوغلۇڭىزنىڭ خەتمە تويىدا كىم سىزگە ساھىپخان بۇلۇپ ياردەملىشىپ بېرىدۇ . بۇلارنىمۇ ئوبدان ئويلىشىپ قۇيۇڭ .
   خوشنامنىڭ بۇ سۆزلىرىنى ئاڭلاپ ئىچىمدە توۋا دېدىم ، ئەمدى بىزنىڭ خوشنىدارچىلىقىمىزمۇ مۇشۇنداق تۈگۈمەس  چاي ئارقىلىقلا ئۆز خاسلىقىنى تاپامدىغاندۇ ؟ ئىلگىرىكى خوشنىلار ئارىسىدىكى مۇناسىۋەتتە  ، پەقەت ئىسسىق دەملەنگەن بىر چەينەك چاي بىلەنمۇ قېرىنداشتىنمۇ يىقىن بۇلىشىپ كېتەتتۇققۇ ؟! ئەجىبا بۈگۈنكى كۈندە مۇشۇ چاينى ئوينىمىسا ، ھېچ ئىش باشقا چىقماسمۇ ؟ ! ...
نۇرغۇن سۇئال كاللامدا پەيدا بۇلىشقا باشلىدى ، خوشنام زەينۇر يەنە بىر مۇنچە بىر نىمىلەرنى دەپ چۈشەندۈرۈپ ھاردى بولغاي ؛
ــــ ماقۇل ئەمىسە ، ئوبدانراق ئويلىشىپ باقارسىز ، مەن يەنە تېلىفۇن قىلىمەن ، ئەڭ ياخشىسى بۇدا سەپتىن چۈشۈپ قالماڭ ، بولمىسا چوقۇم پۇشايمان قىلىپ قالىسىز ،- دېدى .
مەن ئوھسىرىغان ھالدا تېلىفۇننى قويدۇم . . .

مېھرىيە يازمىسى .
