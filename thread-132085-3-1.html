    گۈلشەن، قەشقەرلىك، قارامتۇل  كەلگەن ،زىلۋا بويلۇق، شەھلا كۆزلۈك قىز.ئۇنىڭ ئاتا-ئانىسى دېھقان، كىچىكىدىن باشلاپ غورىگۈل تۇرمۇش كەچۈرۈپ چوڭ بولغان، ئۇمۇ باشقا قىزلاردەك چىرايلىق كىيىملەرنى كىيىشنى، ئۆزىنى ياساپ يۈرۈشنى ياخشى كۆرىدۇ، لېكىن ئائىلە ئىقتىسادى بۇنىڭغا يار بەرمەيدۇ، گەرچە ئۇلارنىڭ ئائىلىسى نامرات بولسىمۇ، ئاتا-ئانىسى ئۇنى بىر تال قىزىم دەپ ئەتىۋارلاپ چوڭ قىلدى، ئۆيدە ئۇنىڭ دېگىنى ھېساب ئىدى، ئۇنىڭ ئۈستىگە نەتىجىسى ياخشى بولغاچقا، ئاتا-ئانىسى ئۇنى ھېچ ئىش قىلغۇزماي ئۆگىنىش قىلىشقىلا بۇيرۇيتى، گەرچە ئۇ دېھقان ئائىلىسىدىن كېلىپ چىقسىمۇ، ئۇ تېخى بىرەر قېتىم جاپانىڭ تەمىنى تېتىپ باققىنى يوق.پەقەت ئۇچىسىدىكى كىيىملەرنىڭ باشقا قىزلاردىن پەرقلىنىپ تۇرغىنىنى ھېسابقا ئالمىغاندا، ھېچكىم ئۇنى دېھقان قىزلىرىغا ئوخشاتمايدۇ، ئۇمۇ ئۆزىنىڭ چىرايىغا بەك ئىشىنىدۇ، ئەسلى مەن دېھقان ئائىلىسىدە خاتا تۇغۇلۇپ قالغان، كەلگۈسۈمنى چوقۇم ئۆزگەرتىشىم كېرەك، چوقۇم پۇلدار بىرسى بىلەن توي قىلىپ تۇرمۇشتىن ھۇزۇرلىنىپ ياشايمەن دەپ ئويلايدۇ.ئاتا-ئانىسى كۆپ ئوقۇمىغاچقا، ئۇنىڭغا قانداق تەربىيە قىلىشنى بىلمەيدۇ، ئۇنىڭ كۆڭلىدە نېمە ئويلاۋاتقىنىنىمۇ بىلمەيدۇ، پەقەت: قىزىم، ئۆزۈڭگە چىڭ بولغىن، قىز بالا دېگەن ئۆزىگە چىڭ بولۇشى كېرەك، بىزنىڭ يۈزىمىزنى يەرگە قارىتىدىغان ئىشلارنى قىلما دەپ ئەسكەرتىپ تۇرىدۇ، لېكىن بۇ گەپلەر ئۇنىڭغا خۇددى قىزىم قولىڭنى يۇيۇپ تاماق يە دېگەنگە ئوخشاشلا سۆزلەر.ئۆزىنىڭ تەقدىرىنى ئۆزگەرتىشكە ئىرادە قىلغان گۈلشەن بېلىنى چىڭ باغلاپ ئۆگەندى، تىرىشقانغا تۇشلۇق ئاخىرى ئۇنىڭغا ئالىي مەكتەپتىن چاقىرىق قەغىزى كەلدى. بىر پۈتۈن تەتىلنى دوستلىرى بىلەن ئويناپ ئۆتكۈزۈپ، ئالىي مەكتەپكە بېرىش تەييارلىقلىرىنى قىلدى.ئاتا-ئانىسى ئۆيىدىكى كالىلىرىنى سېتىپ، ئۇنىڭ ئوقۇش پۇلىنى جەملەپ كۆزلىرى نەمدەلگەن ھالدا ئۇنى ئۇزاتتى.گۈلشەننىڭ كۆزلىرىدىمۇ ھەم ياش، لېكىن ھاياجانلىنىش ھېسياتى كۆڭلى يېرىم بولۇش ھېسياتىنى بېسىپ چۈشەتتى.ئۇنىڭ قەلبى چوڭ شەھەرگە بولغان ئىنتىلىش، گۈزەل مۇھىتقا بولغان تەلپۈنۈش، كەلگۈسىگە بولغان ئارزۇ-ئارمانلىرى بىلەن تولغان ئىدى.ئاپتوبۇس قوزغالدىن كېيىن ئۇ كۆزلىرى ياش قولىنى پۇلاڭلىتىپ قالغان ئاتا-ئانىسىغا خوش دەپ ئىشارەت قىلدى دە، تېلېفونىنى ئېلىپ دوستلار چەمبىرىگە ئۇچۇر چىقاردى-خوش،ئانا يۇرت، ئۈرۈمچى مېنى ساقلىغىن.4-نومۇرلۇق قىزىمىزنىڭ ئىسمى دىلدار    دىلدار، غۇلجىلىق، چاچلىرى ئۇزۇن، سېرىق، زىلۋا بوي، ئورا كۆز قىز، ئۇ ھەقىقەتەنمۇ چىرايلىق، جەلپكار، ئاتا-ئانىسىنىڭ ئىككىلىسى ئوقۇتقۇچى، ئۇلارنىڭ ئائىلە شارائىتى ياخشى، لېكىن ئاتا-ئانىسى ئۇنىڭغا ئۇنداق قىممەت كىيىملەرنى ئېلىپ بەرمەيدۇ، ئاپىسى كىچىكىدىن ئۇنىڭغا:قىزىم، كىيىم دېگەننى پاكىز يۇيۇپ، رەتلىك كىيسىڭىزلا بولىدۇ، قانچە كىيىم ئالسىڭىزمۇ ئۇ ھامىنى بىر رەخت، پۇلىڭىزنى ئۇنىڭغا ئىسراپ قىلماي قورسىقىڭىزغا ياخشى يەڭ دەپ جېكىلەيتى.شۇڭلاشقىمۇ ئۇ كىيىم كىيىشكە ئالاھىدە ئېتىبار بېرىپ كەتمەيدۇ، پەقەت كىيىملىرىنى پاكىز يۇيۇپ، دەزمال سېلىپ، رەتلىك كىيىنىپ مەكتەپكە ماڭىدۇ.دادىسى ئۇنىڭ ھەرقېتىملىق تۇغۇلغان كۈنىدە ياخشى كىتاپلارنى سوۋغا قىلىدۇ، شۇڭا ئۇ كىچىكىدىن باشلاپ كىتاپ ئوقۇشقا ھېرىسمەن، نەزەر دائىرىسى كەڭ، تەپەككۇرى كۈچلۈك چوڭ بولدى، ئۇ ھەر قېتىم كىتاپلارنى ئوقۇپ ئالغان تەسىراتىنى دادىسىغا سۆزلەپ بەرگىنىدە دادىسى ئۇنىڭدىن مەمنۇن بولۇپ مېنىڭ قىزىم بەك ئەقىللىق دەپ ماختاپ قويىدۇ.دادىسى ئۇنى يەنە بىللە بازارغا ئېلىپ چىقىپ ،دۇكانلارنى ئارىلاپ، پىۋا بوتۇلكىلىرىنى يىغىپ سېتىشقا بۇيرىدى، ئۇ بىركۈن جاپا تارتىپ پىۋا بوتۇلكىلىرىنى سېتىپ پۇلنىڭ نەقەدەر تەستە تېپىلىدىغانلىقىنى بىلدىدە، بۇندىن كېيىن پۇلنى ھەرگىز بۇزۇپ چاچمايمەن دەپ ئۆز-ئۆزىگە قەسەم بەردى.شۇنىڭدىن كېيىن ئۇ ئاتا-ئانىسىدىن پارچە خىراجەت سوراپ باقمىدى، چۈنكى ئۇ ئاتا-ئانىسىنىڭ ھەركۈنى مەكتەپكە بېرىپ، چاڭ-توزان يەپ، ناھايىتى جاپا تارتىپ تاپىدىغانلىقىنى بىلەتتى.ئاپىسى مەكتىپىدىكى بىر قىزنىڭ ھامىلىدار بولۇپ قالغانلىقىنى بىلىپ، قىزىغىمۇ تەربىيە قىلىشنى زۆرۈر بىلىپ، ئۇنى يېنىغا چاقىردى.
-قىزىم، بىر قىزبالا ئۈچۈن ھەممىدىن مۇھىمى پەزىلەت، پەزىلەتسىز قىزنىڭ ئېتىبارى بولمايدۇ، مەن سىزنىڭ ئەقىللىق قىز ئىكەنلىكىڭىزگە ئىشىنىمەن، سىزنىڭ ئىپپىتىڭىز سىزنىڭ يۈزىڭىز، مېنىڭ قىزىم بولسىڭىز ئۆزىڭىزنى قەدىرلەپ، بېشىڭىزنى تىك تۇتۇپ ياشاڭ بولامدۇ قىزىم؟مەن سىزنى خەقلەرنىڭ ئوغۇللىرى ئالدىدا بېشىڭىزنى ساڭگىلىتىپ قېلىشىڭىزنى خالىمايمەن، چۈنكى مەنمۇ سىزنى ئوغۇل بالىدەك جىگەرلىك، غورۇرلۇق چوڭ قىلدىم دەپ قارايمەن، شۇنداق قىلالامسىز؟دىلدار يېشىنى سۈرتۈپ، ئاپىسىغا جىددىي تەرىزدە تىكىلدى:
- خاتىرجەم بولغىن، ئاپا، مەن نۇرغۇنلىغان كىتاپلارنى كۆردۈم، نۇرغۇنلىغان قىزلارنىڭ تەقدىرىنى كۆردۈم، بىلىپ تۇرۇپ ئۆزۈم ئورەككە چۈشىدىغان ئىشنى قىلمايمەن، پۈتۈن ۋۇجۇدۇم بىلەن ساڭا قەسەم بېرەيكى، سېنىڭ ۋە ئۆزۈمنىڭ ئىززەت-ھۆرمىتى ئۈچۈن، غورۇرۇمنى ھاياتىم بىلەن قوغدايمەن.بۇ ئاپىسى ئېرىشمەكچى بولغان جاۋاپ ئىدى، ئۇ قىزىنى قۇچاقلاپ چوڭقۇر مېھرى بىلەن پېشانىسىگە بىرنى سۆيدى.
دىلدارنىڭ نەتىجىسى ئەلا بولغاچقا، تەبىئىيلا 1-تۈركۈمدىن ئارزۇ قىلغان مەكتىپىدىن چاقىرىق قەغىزى كەلدى، ئۇ كىيىملىرىنى يىغىشتۇرۇپ، كىتاپلىرىنى ئېلىپ يېڭى بىر ھاياتقا قەدەم قويۇشقا تەييارلىنىپ ماڭدى.ئۇمۇ ئۆزىنى  قانداق تەقدىرنىڭ كۈتۈپ تۇرىۋاتقانلىقىنى بىلمەيتى.5-نومۇرلۇق قىزىمىزنىڭ ئىسمى دىلرۇز.
   دىلرۇز، ئۈرۈمچىدىن، يۈزلىرى ئاق ، سۈزۈك، ئوماققىنە زىنىقى بار سۆيۈملۈك قىز.ئۇنىڭ ئاتا-ئانىسى كىچىكىدىن ئاجرىشىپ كەتكەچكە، مومىسىنىڭ يېنىدا چوڭ بولدى.ئۇ مېھرى-مۇھەببەتكە تەشنا قىز، ھەرقېتىم ئاتا-ئانىغا مۇناسىۋەتلىك فىلىملەرنى كۆرسە كۆڭلى بۇزۇلۇپ يىغلايدۇ.ئۇ ئەدەبلىك چوڭ بولدى، مومىسىغا ياردەملىشىپ ئۆي-ئىشلىرىنى قىلىشىپ بېرىدۇ، گەرچە مومىسى يالغۇز بولسىمۇ، لېكىن بوۋىسى ئەينى چاغلاردا سودا قىلىپ، ناھايىتى كۆپ پۇل تېپىپ، مومىسىغا مىراس قالدۇرغان.شۇڭا مومىسى بۇ بىر تال ئەتىۋارلىق قىزىنى كۆز قارىچۇغىنى ئاسرىغاندەك ئاسراپ چوڭ قىلدى.ئۇ ئاتا-ئانىسىنىڭ ئۆيلىرىگە بېرىشنى ياقتۇرمايدۇ، چۈنكى ئۆگەي دادىسى بىلەن ئۆگەي ئاپىسى ئۇنىڭغا سوغۇق مۇئامىلە قىلىشنىلا بىلىدۇ.ئۇ ھەممە نەرسىنى مومىسىنىڭ چۆچەكلىرىدەك گۈزەل دەپ ئويلايدۇ.ئۇ تولۇق ئۇتتۇرا مەكتەپنىڭ3-يىللىقىدا مۇھەببەتلىشىشنى باشلىدى، ئۇ بالىمۇ بۇنىڭغا شۇنداق كۆيىنىدۇ، ئۇ تۇنجى قېتىم مۇھەببەتنىڭ ئىللىقلىقىنى ھېس قىلدى.ئۇلار بىللە بىر مەكتەپكە ئۆتۈش ئۈچۈن، بىللە تەكرار قىلىپ ئالىي مەكتەپكە بىللە قەدەم قويدى، ئۇ پۈتۈن بىر تەتىلنى ئوغۇل دوستى بىلەن بىللە مۇھەببەتنىڭ قوينىدا ئۆتكۈزدى.مانە تەتىلمۇ توشۇپ مەكتەپكە بېرىش ۋاقتى كەلدى، ئۇ كىيىملىرىنى ماسلاشتۇرۇپ كىيىپ، پەستە ساقلاپ تۇرغان ئوغۇل دوستىنىڭ قولىنى چىڭ تۇتۇپ، مەكتەپ دەرۋازىسىغا قەدەم قويدى.
  « شىنجاڭ ئۇنىۋېرسىتېتى» بەش قىز ئىچىدە ئۇيۇل تاشتىكى بۇ خەتلەرنى قايتا-قايتا تەكرارلىدىدە، نۇرغۇن مۇرەككەپ ھېسلار بىلەن تەڭلا مەكتەپ بوسۇغىسىغا قەدەم ئالدى، قانداق تەقدىرلەرنىڭ ئۇلارنى چەمبەرچاس باغلاپ تۇرىدىغىنىنى ئۇلار قانداقمۇ ئويلاپ يەتسۇن؟! مەكتەپ تولىمۇ گۈزەل ئىدى، دىلرۇزدىن باشقىلار مۇ مەكتەپكە كېلىپ باقمىغاچقا، مەكتەپنىڭ گۈزەل يېشىللىققا تولغان مەنزىرىسىدىن تويماي ھۇزۇرلىناتتى، كۆڭلى ئېيتىپ تۈگەتكۈسىز ھاياجان ۋە شادلىققا چۆمەتتى، كەلگۈسىگە بولغان پۈتمەس-تۈگىمەس شېرىن خىياللارنى سۈرۈپ، ئاخىرى ياتىقىغا يېتىپ كەلدى.

ھېكايە ناھايىتى ياخشى باشلىنىپتۇ... ئاخىرىغا قىزىقىپ قالدىم.
ئەمما، ئويلىغانلىرىمنى دەپ باققۇم كەپقالدى.
1- قىز ئاقسۇلۇق ماھىرە، ئاتا- ئانا بىرسى ساقچى، بىرسى مەمۇر، شۇنچە بىلىملىك كىشلەر. ئەمما قىزى بىلەن بىللە ئولتۇرۇپ قىزىل ھاراق ئىچكەننى ئاز دەپ، رومانتىك فىلىملەرنى بىللە كۆرىدۇ. دېمەك بۇ يەردە ئاتا- ئانا بىلەن قىز ئوتتۇرىسدا پەردىشەپ يوقكەن.  ئالى مەلۇماتلىقلار مۇشۇنداق بولسا، ئادەتتىكى ئائىلىللەر قانداق بولا..... بەزى تور يازمىلىرىدا ئوقۇغان خىزمەتچىلەرنىڭ ئەخلاق كۆز قارىشى چەتكە قېقلدىغان مەزمۇنلار كۆپ ئۇچرايدىكەن. يەنى، ئالى مەكتەپلەردە ئوقۇغانلار ۋە كادىرلارنىڭ ئەخلاق ئۆلچىمى ئوقۇمىغانلارغا يەتمەيدىغاندەك كۆز قاراش مەۋجۇت ئىكەن،
2- قىز گۈلقىز، بۇ قىزنىڭ ئاپىسى قىزغا ئىپپەت توغرىسىدا گەپ ئېچىپ باقمىغان، دېگەن يەر دېققىتىمنى تارىتتى...
ئەلۋەتتە، ھەر ئائىلىنىڭ پەرزەنىت تەربىيلەش ئۇسۇلى ئوخشاشمايدۇ. بۇ نوقتىغا قارىتا مەن ئۆزۈمنىڭ ئائىلىسىنى كۆز ئالدىمغا كەلتۈرۈپ باقتىم. بىز 4قىز ئىدۇق. ئارىمىزدا ئوغۇل قېرىندىشىمىز يوق. ئېسىمنى بىلسەم دادام ئاچچىقى تولىمۇ يامان ئەمما كۆيۈمچان ئادەم ئىدى. كىچىكىمدىن ئانام ماڭا ئىپپەت توغرلىق گەپ ئېچپ باقىمىغان. ئىپپەت توغرىلىق گەپ ئېچىش تۈگۈل ئائىلىمىزدە ھەر بىر گەپ- سۆز، ئىش- ھەرىكەت ئىنتايىن پەردىشەپ بىلەن بولاتتى. ماڭا ئەلچىلەر كەلگەن كۈنگىچە قەدەر ئۆيدە ھەتتا بىزنىڭ ئۆيدىكى قىزلارنىڭ ياتلق بولدىغانلىقىغا دائىر گەپلەرمۇ دېيىلمەيتى. پەقەت ياخشى ئوقۇش، ياخشى ئۆگىنىشلا تەكىتلىنەتتى. ئۇندىن باشقا، باشقىلارغا ئەدەپ بىلەن مۇئامىلە قىلىش تەلەپ قىلناتتى. ئەمما ئاشۇنداق پەردىشەپ، ھايا ئىچىدە ئۆسۈپ يىتىلىش جەريانىدا شەكىللەنگەن قىزلىق غورۇر ۋە ھىس- تۇيغۇم قىزلارنىڭ ئىپپەت كۆز قارىشى، قەدىر قىممەت كۆز- قارىشى، ھايا، نۇمۇسنىڭ نەقەدەر قىممەتلىك ئىكەنلىكى نامايان قىلىپ تۇراتتى. ئۆيىمىزدە ئاۋال مەن كېيىن مەندىن كېيىنكى سڭلىم توي قىلدى. ھەر ئىككىمىز قىزلىق شەرمى ھايانىڭ تەڭداشسىز كۈچ قۇدىرىتى بىلەن ھەقىقى بەخىتكە ئېرىشتۇق. بۇ يىل ئالى مەكتەپنى پۈتتۈرۈپ مەن بىلەن بىر ئورۇندا ئىشلەۋاتقان ئائىلىمىزنىڭ 3-قىزىنى خىزمەتداشلىرىمىزنىڭ ھەممىسى قايللىق نەزىرى بىلەن ماختىشىدۇ، ھەقىقەتەن يۈزى تۆۋەن ئەقىللىق قىز دەپ، ئالى مەكتەپتە ئوقۇۋاتقان ئەڭ كىچىكىنىڭ سەبىيلىكىگە گەپ كەتمەيدۇ. ئەمما، بەزى قىزلار ئاممۋى سورۇنلاردا قىزلار ئۈچۈن ئىپپەت دېگەن ئانداق مۇھىم، ھايا نۇمۇس دېگەن مۇنداق، ئانام ماڭا مۇندق دەيتى، ئۇنداق تەربىيلەيتى.... دېگەن گەپلەرنى ھېچبىر تارتىنماي دەپ ئولتۇرغىنىنى ئاڭلىسام ھازىرقى چېغىمدىمۇ سەل بىئارام بولىمەن. مەن ئويلايمەن. قىزلارغا بېرىلدىغان تەربىيە ئېغىزدا، ھەممىنى سۆزلەپ تۇرۇپ قىلش ئەمەس. بەلكى شەرمى ھايا، پەردىشەپكە تۇيۇنغان ئائىلە مۇھىتىدا يۇشۇرۇن روھىيىتىگە سىڭدۈرگەن ئىپپەت قارىشى ئەڭ مۇھىممكىن دەپ ئويلايمەن.
ئەلۋەتتە، ھەر كىمنىڭ ئويلىغنى ئوخشاش بولمايدۇ. ھەر كاللىدا ھەر خىيال دەپتىكەن. تورداشلارنىڭ باشقىچە كۆز قارىشى بولۇشى مۇمكىن. بۇ پەقەت ئۆزۈمنىڭ ئوي- پىكىرى.
ئاخىرىغا شۇنچىلىك قىزىقىپ قالدىم.....

ھېكايە ناھايىتى ياخشى باشلىنىپتۇ... ئاخىرىغا قىزىقىپ ق ...

ناھايىتى ئورۇنلۇق دەپسىز، لېكىن بىر كىشىنىڭ خاراكتېرىگە تەسىر كۆرسىتىدىغان ئامىللار پەقەت ئائىلە مۇھىتى بولۇپلا قالماستىن، يەنە مەكتەپ مۇھىتى، ئىجتىمائىي ئالاقە، مەنىۋى مۇھىت، مۇشۇلارنىڭ ھەممىسى بىر بالىنىڭ خاراكتېرىنىڭ يېتىلىشىگە تۈرتكە بولىدۇ.پەقەت ھەر تەرەپتىن ئەڭ ساغلام مۇھىت يارىتىپ بەرگەندىلا ئۇ ئېسىل پەزىلەتلىك چوڭ بولالايدۇ، خاراكتېرى پۈتۈنلەي يېتىلىپ بولغاندىلا ئاندىن ئۆزى مۇئەييەن ھالدا شەيئىلەرگە، ئىشلارغا قارار چىقىرالايدۇ.مانا بۇ قارار ئۇلارنىڭ تەقدىرىنى بەلگىلەيدۇ.لېكىن ئەسەردە بۇ بىر نەچچە تەرەپتىن كەلگەن تەسىرنى تولۇق تەكىتلەپ ئۆتۈش ئۇزۇن ۋاقىت، ھەمدە بەلگىلىك ھەجىم تەلەپ قىلغىنى ئۈچۈن، بۇلارنىڭ بىر-بىرىسىدىن بولغان پەرقىنى گەۋدىلىك ئىپادىلەپ بېرىش ئۈچۈن شۇنداق ئېلىندى.ياخشى پىكىرلىرىڭىزنى ئۈزۈلۈپ قالماسلىقىنى ئۈمىد قىلىمەن.
