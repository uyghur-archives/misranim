بۇ يەردە شۇنى بىلىش كىرەككى ، بىزگە ئېقتىدارلىقلار كىرەكمۇ ياكى قالپىقى بار ساپالىقلار كىرەكمۇ؟
ئاۋال قولدىن ئېش كىلدىغانلار ئۇستىدە توختۇلۇپ باقايلى،قاراپ باقساق بىز ئۇيغۇرلاردا قولدىن ئېش كىلدىغان تېخنىكلار شۇنچە كۆپ موزدۇز ناۋاي سەتىراچ، لەڭپۇچى... ۋاھاكازالار  سىزنىڭ ئويىڭىز بويچە قولدىن ئېش كىلدىغانلار موھېم بولغان جەميەتتە نىمىشقا ئۇلار داڭلىق كارخانچى بوپ كېتەلمىدى؟ ئۇلاردا تېخنىكا بارغۇ؟  
قولدىن ئېش كىلدىغانلار كىرەك بولسا ئەمسە ئېشەكنىڭ كۈچى بار بۇرمۇنچە پۇلغا تىراكتۇر سىتىۋالماي ئېشەكنى ئېشلەتسىڭىز بولىۋىردۇ،چۈنكى كۈچى بار تىراكتۇرنىڭ قالپىقى چوڭ، ئېشەكتەك سېتىۋىلىپلا ئۈنۈم ھاسىل قىلغىلى بولمىغان بىلەن ئۇنى ئۈگەنگەندە ئورۇنلۇق ئېشلەتكەندە ئۈنىمى چوڭ بولىدۇ،
سىز بىز سىتۇدىنتلارنى كۆپىنچىسى قالپاققا تاينىىپ يۈردۇ قولدىن ئېش كەلمەيدۇ دەپسىز،شىركەتنى چۈشىنەمدۇق؟ قانداق ئادەم ئېشلىتىش دىگەننى چۈشۈنەمدۇق؟
مەن ئۆز كۆز قارشىمنى دەپ باقاي،شېركەت دىگەن توختاۋسىز يىڭىلىق يارتىپ تەرەققى قىلىدۇ ۋە ئېقتىسدى ئۈنۈم يارتىدۇ،تېخنىكىسى بار ئادەمدىن بىرنى ئېشلەتسىىڭز ئۇ مەلۇم بىر تۈرنى سىزگە ياخشى ئېقتىسادى ئۈنۈم ھاسىل قىلىپ بىردۇ،بىراق باشقا تۈر ئاچقاندا ئۇ كىشىگە ئۈگتۈش ئۈچۈن قانچىلەك ۋاقىت ۋە ئېقتىساد سەرىپ قىلىدۇ؟ ئالى مەكتەپتە ئوقۇغان ئادەمگىچۇ؟ ياخشى باشقۇرغۇچى بۇرۇننىڭ ئۇچىنى كۆرمەسلىكى كىرەك، يېراقنى كۆزلىگەندە ئاندىن تىز روناق تاپىدۇ تىز كۆزگە كۆرنىدۇ،قايسى بىر ئاتۇشلۇق باينىڭ بىر سۆھبىتى ئېسىمگە كېلىپ قالدى،ئۇنىڭ چەتئەلدىمۇ ئاشخانسى باركەن ئۈرۈمچىدىمۇ ھەم شۇنداق،ئۇنىڭدىن ئوقۇغان بىلەن ئوقۇمغاننىڭ پەرقى نەدە دىسە ئۇ ئادەم مەن پەرقىنى بىلمەيمەن بىراق مەن ئويلامەن ئەگەر مەن ئالى مەكتەپتە ئوقۇغان بولسام بۇنچىلەك نەتىجە قازنىشىمغا كېتىدىغان ۋاقىت چوقۇم نەچچە ھەسسە ئاز بولاتتى دىدى. بۇنىڭدىن نىمىنى چۈشىنىشكە بولىدۇ؟ رادىل ئابلا ئەپەندىمۇ ئوقۇمغان بىراق ئۇ كىيىن ئوقۇشنىڭ موھېملىقىنى بىلىپ ئاسپىرانتلىق ئوقۇشىنى تاماملىدى،ئەگەر ئۇنۇڭغا تېخنىكا موھېم بولغان بولسا ئاسپىرانتلىق قالپىقىنى ئالماي يىمەكلىك ياساشنىڭ يېڭى بولغان تېخنىكىسنى ئۈگەنگەن بولاتتىغۇ؟
ئەمدى ساپا توغرىستا توختىلاي مەن ئوقۇش پۇتكۈزۈپ بىر مەزگىل شىركەتلەردە ئېشلىدىم،مەن ئېشلىگەن كۆپۈنچە شىركەتلەردە سىز دىگەندەك تېخنىكىسى بار قالپىقى يوق خېزمەتچىلەر شۇنداق ئوبدان ئېشلەيدىكەن بىراق ھەيران قالارلىقى بىرمۇ ئالى مەكتەپ ئوقۇغان ئادەم ئېشلىمىەيدىكەن سەۋەبى مۇئاشتىمۇ؟ ياق مۇئاشى 3000غا يېقىن ،مەن ئېستىپا بىرىپ يېنىپ چىقشىمنىڭ سەۋەبى بەلكىم ئوقۇغانلارنىڭ سەۋەبى بولشىى مومكىندۇ، مەن خېزمەت موھېتىنى ياقتۇرمۇدۇم چۈنكى ئاغزىنى بۇزۇپ تېللاش بەس سېلىش باشقۇرغىچىلار بىلەن ئۇرشۇش خۇددى نۇرمال ھەۋالدەك ئۈزۈلمەيدىكەن قارسا بۇرمۇنچە ئادەم ياراتقان خېزمەت ئۈنىمى تايىنلىق،بۇنداق شىرتكەتلەردە توختىماي ئادام ئالمىشىپ تۇردىكەن.قېلچىمۇ مەسۇلىيەت تۇيغىسى ،ھەمكارلىق تۇيغۇسى يوق، شۇ سەۋەبتىنمۇ بۇ شىركەتنىڭ تەرەققىياتى خۇددى قۇلۇلىدەك ئاستا، سىزچە بىز سېتۇدىنتلار شۇنداق خېزمەت مۇھېتىدا كۆنەمدۇق ھەم كۆنشىمىز كىرەكمۇ‹؟ مانا بۇ ساپا نىڭ شىركەتكە كۆرسىتىدىغان تەسىرى،
ئەمدى قالپاققا كېلەيلى،ئاڭلىسام خارۋارد ئەركىسى قۇدرەت ياقۇپ ئېگلىك تىكلىگەندە ئەڭ ئاۋال ئاۋدى ماركىلىق ماشىنا سېتىۋاپتۇ،بۇ نىمە ئۈچۈن دەل قالپاققار تاينىىش ئەمەسمۇ؟ يەنە قاراپ باقساق ھەرقايسى تەربىيلەش كۇرۇسلىرى ئوقۇتقۇچى قوشۇنىنى بىرىنجى بولۇپ شۇ ئوقۇتقۇچىلارنىڭ قالپىقىنى پەش قېلىدۇ،بۇ قالپاقتىن پايدىلنىش ئەمەسمۇ؟  مەھسۇلات ئېلانلىرغا قارايدىغان بولساق چولپانلارنى دەستەك قېلىپ ئۆز مەسۇلىيتىنى داڭلايدۇ بۇ قالپاققا تاينىش ئەمەسمۇ؟ مانا بۇلار قالپاقنىڭ موھېملىقىنى ئوبدان چۈشەندۈرەلەيدۇ.
ئەمدى سېتۇدىنتلارنىڭ قولدىن ئېش كەلمەيدۇ دىيشىپتۇق،مەەن ئاۋال ئۆزۈمنىڭ كېچچىكىنە تارىخىمنى سۆزلەي بۇ قېتىمقى مەمۇرلۇق ئېمتىھاندا يېزىقچىدا بېرىنجى بولغان ئادەمدىن 10 نۇمۇر ئاز ئېلىپ يېزىقچە نەتىجىدە 2-بولۇپ يۇز تۇرانە ئېمتىھانغا كىردىم بىرىنجى بولغىنى خەنزۇ يولداشكەن،خەنزۇچە ئېغىز تىلدا جاۋاپ بىردىغان يۈز تۇرانە ئېمتىھاندىن شۇ ئانا تىلدىن ئېمتىھان بىردىغان خەنزۇدىن 15نۇمۇر يۇقرى ئېلىپ بىرنجى بولۇپ مەن ئۆتتۈم ، بۇ قولدىن ئېش كىلىش ئەمەسمۇ بۇنداق مېساللار بەكلا كۆپ ئويلاپ باقساڭلار بولدۇ.
.ئاخىرقى سۆزۈم بۇنداق قالپاق موھېم ئەمەس بىزگە تېخنىكا موھىم دەيدىغان ئويلاردا بولۇپ بۇرمۇنچە ئادەمنىڭ ئوقۇماي يۈرشىگە سەۋەپچى بولمايلى،ئوقۇغاننىڭ ھەممىسى ياخشى بولشى ناتايىن بىراق ياخشىلار كۆپ چىقىدىغان سىتۇدىنتلارنى بۇنچە ئەيىبلىمەيلى.

ئەسسالامۇ ئەلەيكۇم! توردىشىم ۋەخپە:
بۇ ئوبزورىڭىزنىڭ مېنىڭ ‹‹سىتودىنتلاردىكى ئىقتىدارسىزلىق ۋە دېپلوم›› ماۋزولۇق ماقالەمگە قارىتا يېزىلغانلىقىنى بىلىپ ئۆز قۇلۇم بىلەن تەستىقلاپ بۇ تۇنجى ئىنكاسنى يېزىۋاتىمەن.
لېكىن سىزنىڭ مېنىڭ دېمەكچى بولغان تۈپ مەقسىدىمنى چۈشەنمىگەنلىكىڭىزدىن تولىمۇ ئەپسۇسلاندىم. مەن ئۇ ماقالەمدە ھېچقاچان بالىلارنى ئوقۇماسلىققا ئۈندىگىنىم يوق، پەقەت سىتودىنتلىرىمىزنىڭ ئوقۇش ھاياتىنى پەقەت دېپلوم(سىز تەكىتلىگەن قالپاق›› ئېلىشقىلا قاراتماي تېخىمۇ كۆپ ئونۋېرسال بىلىملەرگە ئېگە بولۇش ئۈچۈن تىرىشىشقا، ھازىرقىدەك كەسكىن رىقابەت دەۋرىدە بىر كىشىلىك ئورنىنى تېپىش ئۈچۈن ئىزدىنىشكە دەۋەت قىلدىم.

بىز نېمىشقا ئۆزىمىزنى ئېتىراپ قىلالمايمىز؟ نېمىشقا ئەتراپىمىزدىكى رىئاللىققا كۆز يۇمۇۋالىمىز؟.
ئەلۋەتتە، سىزنىڭ مەمۇرىيلىقتىن ئوتكىنىڭىزنى يالغۇز قالپىقىڭىز(دېپلومىڭىز)نىڭ بارلىقى ئۈچۈنلا بولدى دەپ قارامسىز؟ بۇنداق ئويلىسىڭىز توغرا بولمايدۇ. دېپلوم پەقەت سىزنىڭ مەمۇرىيلىق ئىمتىھانىغا قاتنىشىشىڭىزغا ۋاستە بولدى خالاس، نەزەرىيە ۋە يۈز تۇرانە ئىمتىھاندىن ئۆتەلىشىڭىزنىڭ ھەممىسى سىزنىڭ تىرىشچانلىقىڭىزنىڭ نەتىجىسى. سىزغۇ تىرىشىپ بۇ ئۆتكەلدىن ئۆتەلەپسىز، ئەمما ئەتراپىمىزدىكى نۇرغۇن بالىلارنىڭ سىز ئۆتكەن ئۆتكەلدىن ئۆتەلمەي، جەمىيەتكە ۋە ئاتا-ئانىسىغا يۈك بولۇپ ياشاۋاتقانلىقىنى نېمىشقا نەزەردىن ساقىت قىلىسىز؟ ئەگەر شۇ بالىلارمۇ ئالىي مەكتەپ ھاياتىنى قەدىرلەپ سىزگە ئوخشاش ئونۋېرسال بىلىمگە ئېگە بولغان بولسا ئۇلارمۇ سىز ئېرىشكەن تۈمۈر تاۋاققا ئىگە بولغان بولاتتى. چۈنكى مەمۇرىيلىققا كىرىش ئىمتىھانىنىنىڭ ئونۋېرسال بىلىم تەلەپ قىلىدىغانلىقىنى سىز مەندىن ياخشىراق بىلىسىز.

خارۋارد ئەركىسى قۇدرەت ياقۇپ ھەققىدە گەپ بولۇپتۇ، بۇ توغرۇلۇق ئارتۇق گەپ قىلغۇم يوق. سىز پەقەت قۇدرەت ياقۇپ قانداق قىلىپ خارۋاردقا كىرەلىدى؟ دېگەن مەسىلە ھەققىدە ئويلانسىڭىزلا جاۋاب ئۆزلۈكىدىن چىقىدۇ. چۈنكى ئۇ ئالىي مەكتەپتە ئوقۇش جەريانىدا ئوقۇغان كەسپىدىن سىرت ئېنگىلىز تىلى ئۈگەنمىگەن بولسا ۋە توختىماي ئىزدەنمىگەن بولسا بۈگۈنكىدەك ئۇتۇققا ئېرىشەلمىگەن بولاتتى. بۇ ھەقتە ئۇنىڭ ماتېريالىنى تەپسىلى كۆرۈپ چىقىڭ!

  ئەلۋەتتە، ھازىر قالپاقنى چوڭ بىلىش ھەممە يەردە ئەۋج ئېلىپ كەتتى.  مەيلى ئىشىنىڭ، ئىشەنمەڭ، بۇ خىل قاراش ھامان كېيىنكى رىقابەت دەۋرىدە ئۆز ئەھمىيىتىنى يوقىتىدۇ.

شۇڭا ئاخىرىدا تەكىتلەيدىغىنىم: مەن قانداقتۇر بالىلارنى ئوقۇشتىن ۋاز كېچىشكە دەۋەت قىلماقچى ئەمەس، بەلكى  بالىلارنىڭ چوقۇم ئوقۇشىنى، ئوقۇغاندىمۇ دېپلوم ئېلىش ئۈچۈنلا ئوقۇماي ۋاقىتنى قەدىرلەپ، ئونۋېرسال ساپاغا ئېگە بولۇش ئۈچۈن تېخىمۇ تىرىشىشى كېرەكلىكىنى تەكىتلىمەكچى. شۇنىڭ ئۈچۈن ھەر قانداق بىر ماقالە ئۈچۈن قەلەم تەۋرەتكەندە پەقەت ئۆزىڭىزنىڭ قارىشى ۋە كەچۈرمىشىڭىز ئارقىلىقلا دۇنيانى چۈشىنىشنى ئاساس قىلماي، رىئاللىقنى ئەتراپلىق كۈزەتكەندىن كېيىن قەلەم تەۋرىتىشىڭىزنى سەمىمىي ئۈمىد قىلىمەن.

ئەسسالامۇئەلەيكۇم ھۇكۇمران ئالىيلىرى .
ئوخشىمىغان قاراشلارنىڭ مەيدانغا چىقىشى نورمال ئەھۋال ، چىقماسلىقى بىنورمال ھىساپلىنىدۇ. كوڭۇلگە بەك ئېلىپ كەتمىگۇلۇك ، مەزكۇر تىمىنىڭ ئاپتۇرىمۇ بىرقىسىم قاراشلىرىنى ئوتتۇرىغا قۇيۇپتۇ ، مەنمۇ ئەلۋەتتە سىزنىڭ قارىشىڭىزنى ئەتراپلىق دەپ قارىدىم .
بىز كوپچىلىكنىڭ نەزىرىدە ھەقىقى ئىقتىدار ناھايىتى مۇھىم ، ئەمما ھازىرقى جەمىيىتىمىزدىكى قۇيۇق بولغان ‹‹كېنىشكە ئاتمۇسفىراسى ›› سەۋەبىدىن ھەقىقى ئىقتىدارنى دىپلۇمدەك مۇھىم ئەمەس دەيدىغان قاراشنىڭ مەيدانغا كىلىشىنى بەكمۇ ئەيىپلەپ كەتكىلى بولمايدۇ ، بىزدە نۇرغۇن ئەمما يەنىلا سان جەھەتتىن يىتەرسىز بولغان ئالى مەكتەپنى پۇتتۇرگەن ئۇقۇغۇچىلار ياكى ئۇقۇۋاتقانلار بار ، بىز ھەممىمىز بۇساننىڭ تېخىمۇ كوپىيىشىنى ئۇمىد قىلىمىز ئەلۋەتتە .
جەميىتىمىزگە تارىختىكى ھەرقانداق ۋاقىتقا سېلىشتۇرغاندىمۇ ئالى مەكتەپ ئۇقۇغۇچىلىرى تېخىمۇ بەك لازىم ، بۇ مۇھىملىق- بىر تۇركۇم ئىقتىدارسىز ئەمما يۇقۇرى دىپلۇمغا ئىگە كىشىلەرتۇپىنىڭ ۋۇجۇتقا كىلىشىگە تورتكە بۇلۇشتىن دىرەك بەرمەيدۇ .
مۇھىمى ھەقىقى ئىقتىدار ۋە ئونۋېرسال ساپالىق ئالى مەكتەپ ئۇقۇغۇچىلىرىنىڭ كوپلەپ ۋۇجۇتقا كىلىشى ۋە جەميىتىمىزنىڭ ئۇلارنىڭ ئىقتىدار ۋە ساپاسىنى جارى قىلدۇرۇشىغا قولايلىق پۇرسەتلەرنى بىرىشىدۇر.
گىپىمنىڭ بېشىدا چاقچاق قىلىپ قويدۇم ئېغىر ئالمىغايسىز ، ئاخىرىدا داۋاملىق تىرىشىدىغانلىقىڭىزغا قېرىنداشلارچە ئىشىنىدىغانلىغىمنى بىلدۇرىمەن.
سىزنىڭ مېھنەتسىز ئەمگەكلىرىڭىزگە كوپ رەھمەت.
