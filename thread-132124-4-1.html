ئۇكام نىمە بولغانسەن ئۇكام ، شانىياز مەتنىيازنىڭ تىتىرەپ جۇغۇلداپ كەتكەنلىكىنى ،كۆزىنىڭمۇ ئېقى توللاپ كەتكەنلىكىنى كۆرۇپ ئۇنىڭ يۇزىنى ئۆزىگە قاراتتى .ئەپسۇس مەتنىيازنىڭ ئاغزىدىن پونتاندەك  ئېتىلىپ چىققان قۇسۇقتىن شانىيازپەقەت يۇزىنىڭ يېرىمىنىلا ئەپقاچالىدى . -ھۆ ، تۇفى تۇفى  شانىياز شۇنداق بىر سەسكەنگەن تەبئى تۇيغۇنىڭ كۇچى بىلەن مەتنىيازنى تاشلاپ ئۆزىنى سۇخانىغا ئاتتى ، توختىماي يۇزىنى يۇيىتتى ، ئاغىزىنى چايقايتتى . بىر ھازادىن كىيىن شانىياز سەل ئۆزىگە كىلۋېلىپ ھوي ئۇ قانداقراق دەپ ئويلىدى ۋە ئالدىراش ساراي ئۆيگە چىقتى ، مەتنىياز قۇسۇدىغاننى قۇسۇپ بولۇپ يەنە  ئۇيقۇغا كەتكەن ئىدى . تەكەخاننىڭ مېھمانلىق يوتقان كۆرپىلىرى ، مېھمانخانا ئۆيگە بىر كەلگەن خوتەننىڭ ئېكىسپورت  گىلىمى، ياستۇق دىگەنلەرقۇسۇققا مىلىنىپ رەسۋايى ئەزىم بولغان ھالدا تۇراتتى  ، ئۆيدىن  ھاراق قۇسۇقىنىڭ تىن تارتقۇسىز بەتبۇي پۇرۇقى قاپلاپ كەتكەنلىكىنى ھىس قىلغان شانىياز دەرھال بېرىپ دەرىزىنى ئاچتى . -تەكەخان ، ھوي تەكەخان بۇيانغا چىقىڭلار ، -ھە نىم بولدى نىم بولدى .دىگىنىچە چىقىپ كەلدى تەكەخان باياتىن يۇز بەرگەن فۇنتان ۋەقەسىنى خۇددى ئۆز كۆزى بىلەن كۆرگەندەك بىر ياغىلىقنى بۇرنىغا چەگكەن پېتى چىقىپ كەلدى ، ۋاي بېشىم ۋاي بېشىم ۋاي يوتقان گىلەم ھەممسى تۇگەپتىغۇ ، -ئاز گەپ قىلە ، مانى نىمە قىلساڭ قىلىپ تازلا ، ھە راست ئۆينى ساپلاشتۇرىغۇچ دەيدىغان بىر نىمىنى ئەپكەلدىم داۋاتاتتىڭغۇ .-ئۆزىڭىز بۇ خيىمىيىلىك زەھەرلىك ئاندىغەي مۇندىغەي تاشلا دىگەنغۇ تاشلىۋەتتىم شۇ ، - ماۋۇ گىپىمنى بەك تىز ئاڭلاپ كىتىپسىنا ، توغرا ئەمەسقۇ ، سەن يېڭى ئالغان نەرسىنى تاشلىۋېتىدىغان خوتۇنما ماڭ تىز ئەپكەل-ۋاي سىز زەھەرلىك بولسىمۇ مەيلىما ، -زەھەرلىك بولسىمۇ مەيلى ماۋۇ  قۇسۇقنىڭ پۇرىقىدىن ياخشىراق دىگىنە بۇنىڭغا چىدىغىلى بولمىدى .-شۇ ھاراق سېسىق بولمىسا كىم ئىچىدۇ بۇنى ، -ماڭا تولا شىتىللە بولماي چاپسان ئەكەل .ھاۋا ساپلاشتۇرغۇچنى ئەپلىسىن پۇراقلىق ھاۋا ساپلاشتۇرغۇچىنى چاچقاننىڭياغى ئۆي ئۆيگە ئوخشىغاندەك بولدى . ئاڭىغچە تەكەخان غوتۇلداپ يۇرۇپ يوتقاننىڭ ياستۇقنىڭ  تېشىنى سۆكتى ۋە گىلەمنى قىرىپ ئادالاپ لۇڭگىنى ھۆل قىلىپ تازلاپ پاكىزلىدى . شانىياز ھاۋا ساپلاشتۇرغۇچنى شۇنچە چاچقان بولسىمۇ ئۆيدىن قۇسۇققا ئوخشىمايدىغان بىر خىل ئوبۇرنىنىڭ پۇرقىدەك بىر پۇراقنى زادىلا يوقۇتالمىدى

پۇفۋە  زادى نىمە پۇراق بۇ ، تەكەخان ئاۋۇ ئوبۇرنىنىڭ يېنىدىكى ئەۋرەز ئېغزىنى توساپ قويغانما  ،-توساپ قويغان  ، تۇيۇقسىز مەتنىياز يوتقاننى پۇتى بىلەن تىپىپ ئېچىۋەتتى . ئەسلىدە بۇ پۇراق يوتقاننىڭ ئىچىدىن كىلۋاتقاندەك قىلاتتى .-ئىم ، بۇ گۇي باياتىن قوساقتىكىنىڭ يېرىمىنى ئاغزىدىن يېرىمىنى ئارقا تەرەپتىن بىراقلا چىقىرىپ ئارام تاپقانكەندە . ھالىڭغا ۋاي گۇينىڭ بالسى . يېرىلغۇدەك يەپ ئۇستىگە ھاراقنى بولىشغا ئىشسەڭ موشۇنداق بولىدۇ ، ۋاييەي بۇرۇننى چېچىۋەتتىغۇ بۇ گۇي ئىپلاس پۇراپ . ئىچى سۇرۇپ كىتىپتۇدە بۇ گۇينى ، مەيلى دەپ ئويلىدى شانىياز ئۆز ئۆزىگە ھەر ھالدا بۇنىڭ ھازىر ئىچى پاكىزلىنىپ قالدى .ھاراقكەشلەرنىڭ خەتەرلىك ۋاقتى ئۆتۇپ كەتتى دەمدۇق بۇخىل ئالدى كەينىدىن يۇرۇشۇپ كەتكەن ئەھۋالاتلارنى .ئەمدى بۇنى بۇنى قانداق قىلماق كېرەك ، -تەكەخان ھوي تەكەخان ، جانىم خوتۇن  بۇ ئوقەتنى ماڭا يەردەملىشىپ ئوبۇرنىغا ئەكىرشىپ بەرسەڭ قانداق ، مەسەلەن مەن بۇنىڭ ئىككى قولتۇقنىڭ ئاستىدىن قۇچاقلاپ كۆتۇرىمەن ،سەن ئاۋۇ ئىككى پۇتتىن كۆتۇرشىپ بەرسەڭلا بولىدۇ . تەكەخان بېشىنى چايقاپ  قويدى ، شۇنداق قىلمايمۇ ئامال يوقتەك تۇراتتى .ئۇلار ھە دىسە ھە دەپ ئاخىرى مەتنىيازنى ئوبۇرنىغا ئەكىرىپ تامغا يۆلەپ ئولتۇرغۇزۇپ قويدى ، ئۇ تېخىچىلا بارلىق ئىشلاردىن بىخەۋەر ئۇيقۇسىنى داۋام قىلۋاتاتتى .
   شانىياز ئىپلاس پۇراپ تۇرغان ئۇكىسغا بىر نەچچە دەقىقە قاراپ ئارسالدى بولغاندىن كىيىن تەكەخاننى چاقاردى .-تەكەخان ھوي تەكەخان ، ئاۋۇ ئىدارىدىن تارقاتقان ئاق پەلەيلەرىن بىر جۇپنى تېپىپ كەلگىنە ،نىمە قىلاي دەيسىز ، -نىمە قىلاتتىم مونۇ بايۋەكچىنى پاكىزلەپ قويمايمەنمۇ ، ئۇ ئوبۇرنىدا تۇرىۋەرسە بولمايدۇدە . شانىياز بىر جۇپ پەلەينى كىيىپ  ،ئوبۇرنىنى مەھكەم تاقاپ ،مەتنىيازنىڭ بۇلغانغان ئىچ كىيىملىرىنى سالدۇرۇپ مۇنچىنىڭ سىقما بېشىدىن چىققان بېسىملىق سۇ نى پۇركۇتۇپ مەتنىيازنى ئۇزۇل كىسىل پاكىزلەشكە باشلىدى .قولىدىكى بىر جۇپ ئاق پەلەيگە قاراپ ئۆزىنى جەسەت يۇغۇچىلارغا ئوخشۇتۇپ خۇدۇكسىرەپ قويغان شانىياز يەنىلا ئۆزىنى تۇتۇۋېلىپ مەتنىيازنى رەسمى غۇسىل قىلدۇردى ، ھەر قېتىم مەتنىيازنىڭ يۇزىگە ۋاجىلدىتىپ سۇ چاچقاندا مەتنىياز سەسكىنىپ  بېشىنى سىلكەپ قوياتتى ، ئەمما كۆزىنى ئاچالمايتتى .

-تەكەخان ھوي تەكەخان ، مېنىڭ ئاۋۇ ئىچ كىيىملىرىمدىن بىر يۇرۇش ئەپ چىققىنە - سىزنىڭ ئىچ كىيىملىرىڭىزنىما ، قانداق بولار - ماڭ تولا گەپ قىلماي ئىچ كىيىملىرىمىنى ئېلىپ چىق ،  بۇ ئەمدى بۇيەردە موشۇنداق تۇرامدۇ  . تەكەخان ئۇن چىقارماي ئىچكىرى ھوجۇرغا كىرىپ كىتىپ بىر پەستىن كىيىن يېنىپ چىقتى . ئوبۇرنىنىڭ ئىشىكىنى ئاستا چەكتى .شانىياز ئوبۇرنىنىڭ ئىشكىنى ئازراق قىيا ئېچىپلا تەكەخان سۇنغان كىيىملەرنى ئېلۋالدى ،ھەمدە بىر بىرلەپ بايۋەتچىگە كەيگۇزگىلى تۇردى . بايۋەتچە بولسا بىر خىلدا پۇشۇلداپ ئۇخلاۋاتاتتى . شانىياز ئۆزىنىڭ ئىچ كىيىملىرىنى ئۇكىسىنىڭ بەدىنىدە كۆرۇپ سۇس كۇلۇمسىرەپ قويدى ، چۇنكى ئىچ مايكىمۇ ، ئىچ ئىشتانمۇ نەق ئون سانتىدەك قىسقا كىلىپ قالغان ئىدى . ۋاي كاساپەت ! -ھوي تەكەخان ، تەكەخان ، مەۋەگە كىلىپ ماڭا ياردەملەشكىن ،  بۇ بايۋەتچىنى يەنە جايىغا ئاپىرىپ قويايلى . تەكەخان پەس ئاۋازدا غۇدۇڭشىغىنىچە يىتىپ كەلدى،ئۇلار  مەتنىيازنى بۇرۇنقىدەك كۆتۇرۇپ ساراي ئۆيدىكى ئورنىغا ياتقۇزدى . -سەن ئاۋۇ مەتنىيازنىڭ ئىچ مايكىسىنى يۇيۇپ قويغىن ، ئاۋۇ ئىشتانلىرى بۇلغۇلنىپ كىتىپتۇ  يۇيغۇڭ كەلمىسە مەيلى سولياۋغا قاچىلاپ ئشىكنىڭ سىرتىغا قويۇپ قويغىن تاشلىۋېتەيلى  ھازىرقى كىلىنلەردىن ئۇنداق جىق ئىشنى تاما قىلغىلى بولمايدۇ ، - بولدىلا تاشلىۋەتمەيلى يۇيۇپلا ۋېتەي پارغا يېيىپ قويسا بىردەمدە قۇرۇيدۇ . ئاۋۇ خاشو پەلەينى كىيىۋالسام بولىدۇغۇ . -مەيلى ئەمىسە ئەمما خاتۇن مەن سېنى بۇنى يۇيۇشقا زورلىمىدىم ھە ، ئەتە ئۆگۇن قىلغىنىڭغا مىننەت قىلىپ كوچا كويدا سۆزلەپ يۇرۇيدىغان ئىش بولغاندىن تاشلاپلىۋەتكەن تۇزۇك . ياش بالا تۇرسا غۇرۇرى كۇچلۇك بۇ ئاقنانچىنىڭ  ئەگەر يۇيۇپ پارغا يېيپ قويساڭغۇ يېرىم سائەتتە قۇرۇيدۇ شاپپىدە چاندۇرماي كەيگۇزۇپ قويىمىزدە مەتنىيازنى بىر خىجىلچىلىقتىن ساقلاپ قالىمىز . شانىياز سۆزلەۋاتقاندا تەكەخان مەتنىيازنىڭ ئىچى كىيىملىرىنى يۇيۇپ بولاي دەپ قالغان ئىدى .كىيىملەرمۇ يۇيۇلۇپ پارغا يېيىپ قويۇلدى . تەكەخان كىيىملەرنى يۇيۇپ بولۇپ شانىيازغا قاراپ ئۇنىڭ ئۇسساپ چاي خالاپ تۇرغانلىقىنى بىلىۋالدىدە ئۇن تۇنسىز ئاشخانىغا كىرىپ چاي دەملەشكە تۇتۇندى .ۋە تۇيۇقسز شانىيازغا قاراپ -ئاۋۇ توڭلاتقۇدا ئەمەتجاننىڭ سۇتى بار ، قايناتسام ئۇكۇڭىزغا ئىچكۇزەمدۇق يە ، سۇتمۇ ھاراقنىڭ تەسىرىنى كىسەمدىكىن دەيمەن .  ھەئە شۇنداق قىلايلى شۇنداق قىلايلى ، ياق  توختا ئۇ ھازىر سۇتنى ئىچىپ ئىسىگە كەپ قالسا ،مەن ئىچى كىيىمىنى ئالماشتۇرسام ، ئىسىگە كىلىپ قېلىپ ئاكام نىمىگە ئشىتنىمنى سېرىپ يۇرۇيدىغاندۇ دەپ ئويلاپ قالسىچۇ ، بولدى.-ھەي ئەدەم ماۋۇ سۇتنى قايناتقىچىمۇ بىر مۇنچە ۋاقىت كىتىدۇ قىززىق سۇتنى سۇۋىتىش ئۇچۇنمۇ بىر مۇنچە ۋاقىت كىتىدۇ ، ئاڭىغىچە كىيىملىرىمۇ قۇرۇپ كىتىدۇ ، -ماقۇل ئەمسە شۇنداق قىلايلى . يېرىم سائەت ئۆتتى . شانىياز  يېيىپ قويغان كىيىملەرنى ئۆرۇپ چۆرۇپ قۇرۇتتى ،تەكەخان بولسا سۇتنى قايناتتى . شۇنداق قىلىپ شانىياز مەتنىياز ئۇيغانماستا ئۇنىڭ ئىچ كىيىملىرىنى ئالماشتۇرۇپ كەيگۇزۇپ قويدى .سوۋۇپ ئىلمان بولغان سۇتنى ئىچكەن شانىيازنىڭ مەڭزىگە ئاستا ئاستا قىزىللىق يۇگۇرۇپ ئۇيقۇسى ۋەخورىكىمۇ نورمال ئادەملەرنىڭىگە ئوخشاش ھالغا كەپ قالدى .
    بۇ ئىچ كىيىملەرنىچۇ دىدى باياتىن مەتنىيازغا كەيگۇزگەن كىيملەرنى كۆرسىتىپ ، -بۇلارنى تاشلىۋەتكىن . -ۋاي كىيىۋەرمەمسىز ئۇكىڭىز  يېرىم سائەت كىيىپ قويغانغا نىمە بولاتتى . -تاشلىۋەت دىگەندىن كىيىن تاشلىۋەت ئاكىسىنىڭكىنى ئۇكۇىسى كەيگەن بار ، نەدە ئۇكىسىدىن ئاشقاننى ئاكىسى قايتا كەيگەن بار .-ۋاي نىم بولاتتى ، تەكەخان قەستەن قېرىشىپ شانىيازنىڭ ئاغزىنى تاتىلاۋاتاتتى . -تاشلىۋەت ، ئۇ بۇگۇن ئاخشام مېنىڭ ئىچ كىيىممنى كىيىشكە مەجبور ، ئەمما مەن ئۇنىڭدىن قالغاننى كىيشىكە مەجبور ئەمەس ، چۇشەندىڭما . ئەي رەببىم  كۆڭلىم خالىمىغان ئىشلارغا مەجبور بولۇپ قېلىشتىن ساڭا سېغىنىپ پاناھ تىلەيمەن . -مەن سىزنى موشۇنداق دەيدىغانلىقىڭىزنى ئالدىن مۆلچەرلەپ سۇزۇلۇپ قالغان كونا ئىچ كىيىملىرىڭىزنى ئەپ چىققان ئىدىم  .شانىياز خوتۇن كىشى دىگەنزە دىگىنىچە بېشىنى چايقاپ كۇلۇپ قويدى . بۇ يازمىنى ئاخىرىدا   tohotur تەھرىرلىگەن. ۋاقتى  2014-12-12 01:10 PM

تەكەخان ئىشكنىڭ ئۇستىدىكى تام سائەتكە قاراپ قويۇپ -بىردەم ئۇخلىۋېلىڭ دادىسى بامداتقا يەنە بىر سائەت بارئىكەن .بۇ گەپ شانىيازغا خېلى خوش ياققان بولسىمۇ ئەمما بىر سائەتتىن كىيىن تۇرالىشىغا پەقەت ئىشەنچى يوق ئىدى .بۇرۇندىن شۇنداق تەكەخان  قاچان دىسە شۇ چاغدا تۇرالايدۇ ،ئەمما شانىيازنىڭ ئۇيقۇسى ئېغىر . -بولدىلا مەن تاھارەت ئېلىۋېلىپ ئاندىن قۇرئان ئوقۇيمەن ، سەن ئۇخلىساڭ ئازراق ئۇخلىۋال . شانىياز تاھارەتنى ئېلىپ بولۇپ بالىخانا ئۆيگە كېرىپ قۇرئان ئوقۇماقچى بولۇپ يەنە بىر قاراپ باقاي بۇ گۇيغا دەپ مەتنىيازنىڭ يېنىغا چىقىتى مەتنىياز بىغەم ئۇخلاۋاتقان بولۇپ پىشانىسىدە مونچاقتەك تەرتامچىلىرى تۇراتتى .شانىياز مەتنىيازنىڭ پىشانىسىنى سۇرتىۋېتىپ تۇيۇقسىزلا قۇرئاننى مەتنىيازنىڭ يېنىدا ئوقۇشنى قارار قىلدىدە بالىخانىدىن ئەپلىك كىتاپ ئوقۇيدىغان قومۇش شىرەسىنى ئېلىپ چىقىپ مەتنىيازنىڭ بېشىدا رەسمى چازا قۇرۇپ ئولتۇرۇپ قۇرئان ئوقۇشقا باشلىدى. تەكەخان ئىچكىرى ھوجۇرىدا شانىيازنىڭ قۇرئان ئوقۇشىنى تىڭشاپ ياتتى .شانىياز بولسا سۇرە ياسىننى قىرائەت قىلۋاتاتتى ، تەكەخاننىڭ كاللسىغا بىر ئىش جىككىدە كەچتىدە چېچراپ ئورنىدىن تۇرۇپ ساراي ئۆيگە كىردى . ھوي سىز مەتنىيازنىڭ بېشىدا سۇرە ياسىننى ئوقۇسىڭىز بولامدۇ . ئۇكىڭىز ئۇيغۇنۇپ قالسا بېشىدا سۇرە ياسىن ئوقۇلغانلىقىنى بىلسە مەن سەكراتتا ئوخشايمەن دەپ ئويلاپ قالمامدۇ ، قاتتىق قورقۇپ كەتمەمدۇ . شانىياز تەكەخانغا خاتىرجەم قاراپ قويۇپ -توغرا دىدىڭ مەنمۇ ئۇنى ئازراق قورۇقسۇن داۋاتىمەن . ئۇ قۇرئاننى مەندىنمۇ كۆپ يات ئالغان ، سۇرە ياسىننىڭ بېىشدا تۇرۇپ ئوقۇلۇشىنىڭ مەنىسىنى بۇ جەڭۋا ئوبدان بىلىدۇ ، مەنمۇ ئۇنى قورۇقسۇن دەۋاتىمەن . قورقۇش دىگەن ئۇنچە بەك يامان ئىش ئەمەس تەكەخان . ھازىرقى باللاردا ئەڭ كەمچىل بولىۋاتقان ئىشلارنىڭ بىرسى دەل قورقۇش . -شۇنداقمۇ دەيلى بىراق ئۇدۇم قىلساڭ ئۇلارسەن دىگەن گەپمۇ بار بىزدە ، - ئاللاھ بەندىلەرنىڭ نىيىتىنى ياخشى بىلگۇچىدۇر ، مېنىڭ نىمە ئۇچۇن ئۇكامنى قورقۇتماقچى بولغانلىقىمنى ئاللاھ ئوبدان بىلىدۇ ، ماڭ ئەمدى گەپنى ئاز قىلىپ ئۇخلا ، ھىيلەھى شەرھى دىگەننى داداڭدىن تۆت بەش ۋاراق جىقراق ئوقۇغانمەن  ، -ئوۋھوش مېنىڭ نىمە كارىم .
  شانىياز سۇرە ياسىىننى قايتا قايتا ئوقۇۋەردى ،ئەمما كۇتكىنىدەك مەتنىياز چۆچۇپ ئويغانمىدى ، كۈتكەن قورقۇشمۇ يۇز بەرمىدى ،ئەمما شانىياز مەتنىيازنىڭ سىڭايان قىلىپ ياتقان ھالىتىدىكى كۆزىدىن سىرغىپ چىققان ياشلارنى كۆرەلمىدى . -ئۇكام ئۇكام مەتنىياز تۇر ئورنۇڭدىن ، نامازنىڭ ۋاقتى بولدى . -ھە ،ئاكا ھە مەن سىلەرنىڭ ئۆيگە قانداق كەپ قالغاندىمەن ، ئاكا بولدى سەن ئۆتىۋەرگىن ، مەن مەن ئاخشام ..... مەن يەنە قىرىق كۇندىن كىيىن ناماز ئوقۇسام بولامدىكىن . -بولدى قىل كىسپۇرۇش ، ئاكاڭنىڭ ئالدىدا پەتۋا سۆزلەپ يۇرمەي تىز قوپۇپ تاھارەت ئال ،ئاندىن ئەزەن ئوقۇ ، مەتنىياز ئامالسىز ئورنىدىن تۇردى ۋە تاھارەت ئېلىپ غاراڭ غۇرۇڭ بىر ئاۋازدا ئەزان ئېيتتى .ئاكىسى ئۇنىڭ سۇننەتنى ئۆتىۋېلىشىنى كۇتۇپ تۇردى ئاندىن ئاكسىى ئۇنىڭغا تەگبىر ئېيتىشقا بۇيرىدى . تەگبىر مۇ ئوقۇلدى، شانىياز ئىمام بولدى .ئۇ ئۇششاق سۇرىلەرنى بىر باشتىن قىرائەت قىلىپ چىقتى ، ئاكا ئۇكا ئىككەيلەن ئاۋۋال رۇكو قىلدى  ئاندىن سەجدىگە باش قويدى.  (تۇگۇدى )
