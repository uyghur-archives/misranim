       بۈگۈن كەچ سائەت بېيجىڭ ۋاقتى 7:35 ئۆتكەندە (2014يىل 12ئاي7كۈنى ،يەكشەنبە)ۋاسكىتبۇل بىرلەشمىسى 16-ئايلانما مۇسابىقىسى شىنجاڭ كۇماندىسى بىلەن شەندۇڭ كۇماندىسى ئوتتۇرىسىدا شەندۇڭ كۇماندىسى مەيدانىدا باشلاندى.
      بۇ مەيدان مۇسابىقىدە شىنجاڭ كۇماندىسى بۇرۇنقىدەكلا بۇراچ ،قىرانبەگ،لىيۇۋېي،جوۋچى،ۋە شىرئەلىجاننى بىرىنجى بۇلۇپ  مەيدانغا چىقارغان ئىدى،(ئالدىنقى مەيداندا شىرئەلىجان يارلانغانلىق سەۋەپلىك مەيدانغا چىقمىغان ئىدى،)ئەزىمەتلىرىمىز باشىلىنىشى بىلەنلا ئۈستۈنلىكنى ئىگەللەپ مۇسابىقىنى كونتىرول قىلغان ئىدى، ھەمدە ئۆزىنىڭ ئىقتىدارنى تۇلۇق جارى قىلدۇرۇپ باشتىن -ئاخىر نۇمۇر پەرقىنى ساقلاپ كەلدى،بۇ مەيدان مۇسابىقىدە سۇۋېينىڭ ئىپادىسى ئالاھىدە ياخشى بولدى،سۇۋېي شەندۇڭ ئۆلكىسى رېجاۋ شەھىرىلىك بۇلۇپ، بۈگۈن ئۆزىنىڭ ئانا يۇرتۇغا قايتىپ مۇسابىقە ئوينىغانتى،ئۇ ھاياجان ھەم يۇقىرى كەيپىيات بىلەن ئويناپ، ئۆزىنىڭ ئەڭ ياخشى تاكتىكا ۋە 14نۇمۇر بىلەن ئۆزىنىڭ ئەڭ ياخشى ئىپادىسىنى كۆرسەتتى ، بۈگۈنكى مۇسابىقىدە باشقا ئەزالارمۇ قىلچىلىك ئۆزىنى ياقا يۇرۇتتا تۇرغاندەك ھىس قىلمىدى، مەيداندا توختىماستىن ‹شىنجاڭ غەيرەت قىل، قىرانبەگ غەيرەت قىل، شىرئەلى غەيرەت قىل،ئۇچار يولۋاس ئەزىمەتلىرى غەيرەت قىل ›دىگەن رىغىبەتلەر ئۈزۈلمىدى،بۇ ئەزىمەتلىرىمىزگە تېخىمۇ ھەركەتلەندۈرگۈچ كۈچ بولدى، تاماشىبىنلار ئارىسىدا قېرغىز قېرىنداشلىرىمىز ئۆزىنىڭ پەخىرلىك مىللىچە بۆكلىرىنى كىيىپ ئەزىمەتلىرىمىزگە باشتىن-ئاخىر مەدەت تىلىدى، بىز شىنجاڭ خەلىقىنىڭ ئەزىمەتلىرىمىزگە بولغان قوللاش ۋە ئىشەنچىمىزنى بىلدۈردى.
          ئەزىمەتلىرىمىزنىڭ تىرىشچانلىقى بىلەن مۇسابىقە ئاخىر 112-88 بۇلۇپ، چوڭ نۇمۇر پەرىقى بىلەن ئاخىرلاشتى، پۈتۈن مەيدان مۇسابىقىدە، بۇراچ 23نۇمۇر، تېرېفىر22نۇمۇر، جوۋچى 19نۇمۇر، سۇۋېي14نۇمۇر، قىرانبەگ9نۇمۇر، شىرئەلىجان9نۇمۇر،يۈچاڭدۇڭ4نۇمۇر، لىيۇۋېي3نۇمۇر ئالدى، شىنجاڭ كۇماندىسى ھازىر جەمىئى 16ئايلانما مۇسابىقىدە 11قېتىم ئۇتۇپ،5قېتىم ئۇتۇپ ،ئۇتۇش نىسبىتى68.8%بۇلۇپ، بىرلەشمە بۇيىچە 5-ئۇرۇندا تۇرماقتا، 17ئايلانمىدا (2014يىل12ئاي10كۈنى 7؛35)ئۆزى بىلەن تەڭ ئۇرۇندا تۇرۇپ كىلىۋاتقان چىڭداۋ كۇماندىسى بىلەن چىڭداۋ كۇماندىسى مەيدانىدا ئوينايدۇ ، ئەزىمەتلىرىمىزگە تۇلۇق ئىشەنچىمىز بار، ھەم زور ئۇتۇقلارنى تىلەيمىز.
