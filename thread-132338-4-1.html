  چوڭ بولغانسېرى ئاپامنىڭ مەن ئۈچۈن تارتقان جاپالىرى كۆزۈمگە كۆرۈنۈشكە باشلىدى، ئۇنىڭ يۈزىدىكى كۈنسېرى كۆپىيىۋاتقان قورۇقلار، ياشاڭغۇراپ تۇرىدىغان كۆزلىرى، مۈكچەيگەن گەۋدىسىگە قاراپ ئىچىم ئىختىيارسىز ئېچىشىپ كېتەتتى.شۇڭا مەن تولۇق ئوتتۇرىدىن باشلاپ تىرىشىپ ئوقۇدۇم، تەقدىرىمنى ئۆزگەرتىش، ئاپامنى ئېسىل تۇرمۇشقا ئېرىشتۈرۈش ئۈچۈن كېچىنى كۈندۈزگە ئۇلاپ تەكرار قىلدىم، كىچىكىمدىن باشلاپ ئەقىللىق بولغاچقا ھەم تىرىشىشىم ئاستىدا ئالىي مەكتەپ ئىمتىھانىنى ئوڭۇشلۇق بېرىپ يېزا ئىگىلىك ئۇنىۋېرسىتېتىغا ئۆتتۈم.ئاپام ئۇزۇندىن بېرى يىغىپ قويغان پۇللىرىنى ۋە تويلۇق زىبۇ-زىننەتلىرىنى سېتىپ، كۆزلىرىگە ياش ئالغان ھالدا مېنى ئۇزۇتۇپ قالدى.
   ئۇنىۋېرسىتېت ئۇ ئارزۇ-ئارمانغا، مۇھەببەتكە تولغان جاي ئىكەن.ياتاقداشلىرىم بىلەنلا تېزلا چىقىشىپ كەتتىم، ئۇلارنىڭ ئائىلە شارائىتى ئاساسەن ياخشى ئىكەن، ئىككىسى باشلىقنىڭ بالىلىرى ئىكەن، قالغانلىرى سودىگەرلەرنىڭ، ئەڭ بولمىغاندا خىزمەتچىلەرنىڭ بالىلىرى ئىكەن.لېكىن ئۇلار مېنى كەمسىتمەيتى، سورۇنلارغا بارساق بىرەر قېتىم ماڭا پۇل چىقارغۇزمايتى، ئۇلارنىڭ ئاغىنىدارچىلىقىدىن، ئوغۇلبالاچىلىقىدىن سۆيۈنسەممۇ، لېكىن ئەرلىك غورۇرۇم ھەرقېتىم ئۇلارنىڭ پۇلىنى  خەجلەشكە يول قويمايتى، ئۆيىمىزمىڭ ئەھۋالى ماڭا بەش قولدەك ئايان بولغاچقا ئۆزۈم ئىشلەپ پۇل تېپىشنى قارار قىلدىم.
ئالىي مەكتەپ ئوقۇغۇچىلىرىغا پۇل تېپىشمۇ تەسكە توختايدىكەن، بىر يۇرتدىشىمنىڭ ياردىمىدە ئاخىرى جۇڭشەندىن تېلېفون سېتىش خىزمىتىنى تاپتىم، نىمكەش تېلېفونلارنى ئېلىپ يېڭىلاپ قىممەت باھادا سېتىش ئارقىلىق ئوتتۇرىدىكى پايدىغا ئېرىشەتتۇق، بۇ خىزمەت بىلەن خېلى پۇل تېپىپ قالدىم، دەرس يوق ۋاقىتلاردا، ياكى قوشۇمچە پەن دەرسلىرى ئۆتۈلىدىغان كۈنلىرى تېلېفون بازىرىغا يۈگۈرۈپ سودا قىلىشقا ئالدىرايتىم، بۇنىڭ بىلەن بىرەر يېڭى قۇر كىيىم ئېلىپ، ئاغىنىلىرىم بىلەن ئولتۇرغان سورۇننىڭ پۇلىنى تۆلىۋېتەلەيتىم.شۇنداق تېلېفونچىلىق قىلىپ يۈرگەن ۋاقىتلىرىمدا تۇيۇقسىزلا بىر باينىڭ قىزى بىلەن تونۇشۇپ قالدىم.  بۈگۈن شەنبە، دەم ئېلىش بولغاچقا، مەن دۇكانغا قاراپ تۇرۇشقا ماقۇل بولۇپ، ئەتىگەندىلا كېلىپ دۇكاننى تازىلاپ، دېرىزىنى ئېچىۋەتتىم.بۈگۈن ھاۋا بەك ئىللىق ئىدى، ۋۇجۇدۇمنى سىرتتىن كەلگەن ئىللىق ئېقىم ئېرىتىۋەتتى، ئاللاھنىڭ تەبىئەتنى بۇنچە گۈزەل ياراتقىنىغا مەپتۇن بولۇپ يىراقلارغا نەزەر سېلىپ تۇرسام، ئېگىز پاشنىلىق ئاياقنىڭ يەرگە كۈچەپ ئۇرۇلۇشى دىققىتىمنى بۇزدى، ئازراق نارازى بولغان ھالدا كەينىمگە بۇرۇلۇپ قارىدىميۇ، شۇ قارىغان بىلەن يۈرىكىمنىڭ بىر پارچىسىغا ئوت ئۇتاشتى.ئالدىمدا20نەچچە ياشلار ئەتراپىدىكى، ئېگىز، ئورۇق، تېرىسى سۈزۈك ھەم ئاق، كۆزلىرى بۇلاقتەك، قاملاشتۇرۇپ سىزغان قاشلىرى ھىلال ئاينى ئەسلىتىدىغان بىر زىلۋا بەدەن، نازۇك چىمەن تۇراتتى.كىيىۋالغان سۇس يېشىل، ئۇزۇن كۆينىكى ئۇنىڭ بەدىنىگە چىپپىدە كەلگەن بولۇپ، زىۋا بەدىنى تولۇق نامايەن قىلىپ تۇراتتى، ئۇ كۆزەينىكىنى قولىغا ئېلىپ، ماڭا سالام قىلدى، مەن شۇھان ئېسىمنى تاپتىم.
-تېلېفون ئالاي دېگەنتىم.ئۇنى ئاۋازىمۇ ئۆزىگە ئوخشاش چىرايلىق ئىدى.
-ھە، قانداق ماركىنى ئالىسىز؟
-ئالما ماركىلىقنى، يېڭى چىققان نۇسخىسىنى، ئۇنىڭ پۇلدار ئائىلىنىڭ قىزى ئىكەنلىكىنى جەزىملەشتۈردۈم، دېگەندەك ئۇنىڭ يېنىدا بىر كىشى پەيدا بولدى. ئۇ 50ياشلار چامىسىدىكى سالاپەتلىك، قورساق سالغان ئادەم بولۇپ قاراشتىنلا بىرەر باشلىقلىقى چىقىپ تۇراتتى.-دادا، بۇ يەردە ئالما تېلېفوننىڭ يېڭى نۇسخىسى باركەن، مەن تېخى ئۇنىڭغا بار دېگەن گەپنى قىلمىغانىدىم.دادىسى ئۇنى ئەركىلىتىپ قويدى.
-ۋۇتان، ۋۇتان قىزىم، بار بولسىلا سىزگە ئېلىپ بېرىمەن، نېم دېسىڭىز شۇنى ئېلىپ بېرىمەن.ئۇ قىز مېنىڭ ئالدىمدا دادىسىنىڭ ئۆزىنى ئەركىلىتىشىنى سەل بىئەپ كۆردى بولغاي، خىجىل بولۇپ ئىتتىك يۈزىنى تاتتىۋالدى.
-دادا، بۇ ئۆيىمىز ئەمەس، دادىسى قاقاقلاپ كۈلۈپ كەتتى.
-ھاھا، توغرا قىزىم، ھە ئۇكام، ئەكىلە تېلېفونىڭنى، مەن پۇكەيدىن ئالما تېلېفوننىڭ يېڭى نۇسخىسىنى ئېلىپ ئۇنىڭغا كۆرسەتتىم.
-ھە، سۈپىتىدە مەسىلە يوقتەك قىلىدۇ، نەچچە باھاسى؟
-5800يۈەن
-سەل قىممەت دېدىڭ،5500يۈەن بولسۇن، مېنىڭ ئاغىنەممۇ ئالدى مۇشۇ باھادا يېقىندا.ئۇنىڭ دېگەن باھاسى مۇۋاپىق بولغاچقا مەن تالاشمايلا ئۇ دېگەن باھادا تېلېفوننى بەردىم.
-مەيلى ئاكا، رازى بولۇڭ.
-ھەببەللى، كەسكىن بالىكەنسەن، ياقتۇردۇم. ئۇ ۋاققىچە ئۇنىڭغا تېلېفون كېلىپ قېلىپ قىزىغا ئۆزىنىڭ قايتىپ تۇرۇشىنى، خىزمەتدىشىنىڭ جىددىي چاقىرغانلىقىنى، كەچتە ئېسىل سوۋغاتلارنى ئېلىپ كېلىپ تولۇقلاپ بېرىدىغىنىنى ئېيتىپ كېتىپ قالدى.ئۇ قىز بىلەن مەنلا قالغانىدىم، ئۇنىڭغا تېلېفوننى قۇتىسىغا قاچىلاپ بېرەي دەپ بولغۇچە ئۇ ئېغىز ئاچتى:
-ئەپ دېتاللىرىنى قاچىلاپ بېرىڭە، ئۇنىڭ ئۇيغۇرگە گەپ قىلغىنىدىن ھەرھالدا خۇشال بولدۇم.
-ھە ماقۇل، چاتاق يوق.
مەن ئەپ دېتال چۈرۈشۈشكە كىرىشىپ كەتتىم، لېكىن كۆزۈمنىڭ قۇيرۇغىدا قارىسام ئۇمۇ ماڭا ئىزچىل تىكىلىپ تۇرۇپتۇ.ھەرھالدا يېزى بالىسى بولساممۇ شەھەرلىكلەرگە خاس سالاپىتىم ۋە چىرايىم بارئىدى، ياتاقتىكى بالىلارمۇ ئاداش، سەن خېلى قىزلارنى كۆيدۈرۈپ قويىسەن جۇمۇ، شۇ بويۇڭ، چىرايىڭ بىلەن دەپ دولامغا ئۇرۇپ قوياتتى.مەن بۇرۇن بۇنىڭغا ئانچە ئېتىبار بېرىپ كەتمىسەممۇ، شۇ تاپتا تۇنجى بولۇپ ئاللاھنىڭ ماڭا بۇ چىراينى بۇ بوينى بەرگىنىگە چىن كۆڭلۈمدىن رەھمەت ئېيتتىم.
-ئۈندىدار قاچىلىدىڭىزمۇ؟
-ھەئە، ئۇنى قاچىلىماي بولمايدۇدە.
-شۇ، كىم باركىن ھازىر ئۈندىدار ئوينىمايدىغان، سىزمۇ ئوينايدىغانسىز؟ئۇنىڭ بۇ گېپىنى يا مەسخىرە، يا تەنە ئىكەنلىكىنى بىلەلمىدىم، لېكىن سەل ئاچچىقىم كەلگەنىدى.
- ئەمدى ئادەم بولغاندىن كېيىن ئوينايمەيدە، ئوقۇغۇچىنىڭ ئۈندىدار ئوينىمايدىغىنى يوققۇ.ئۇنىڭغا قەستەن ئوقۇغۇچى ئىكەنلىكىمنى بىلدۈرۈپ قويدۇم.ئۇنىڭ چىرايىدىن ھەيران قالغانلىق ئالامىتى چىقىپ تۇراتتى.
-ھە، سىز قايسى مەكتەپتە ئوقۇيسىز؟
-يېزا ئىگىلىك ئۇنىۋېرسىتېتىدا، ئىشتىن سىرتقى ۋاقىتلاردا قوشۇمچە تېلېفونچىلىق قىلىمەن.
-ھە، تىرىشچان بالىكەنسىز،مەن شىنجاڭ ئۇنىۋېرسىتېتىدا ئوقۇيمەن.تونۇشۇپ قالايلى، ئىسمىم پەرى، ئۇنىڭ ئوچۇق -يورۇق مىجەزى ماڭا يېقىپ قالدى.-مېنىڭ ئىسمىم دىلشات، قانداق تونۇشىمىز، ئۈندىدارىڭىزنى قاچىلىۋالسام بولامدۇ؟پۇرسەتنى چىڭ تۇتۇشنى ئويلىدىم، ئۇمۇ شۇنداق دېيىشىمنى كۈتۈپ تۇرغاندەك تېزلا جاۋاب بەردى.
-ئەلۋەتتە بولىدۇ، ئۇ ئۈندىدار نومۇرىنى ئېيتىپ بەردى. مەن ئۇنى يولغا سېلىپ قويۇپ خىيالغا پاتتىم.راستىمنى ئېيتسام ئۆزۈمنىڭ بۇنداق ئىتتىك بىر قىز بالىغا مەيلىمنى بېرىشنى خىيالىمغىمۇ كەلتۈرۈپ باقماپتىمەنكەن، ئۇنىڭ شۇ سىرلىق بېقىشلىرى، مىيىغىدا كۈلۈشلىرى، كۈلسە پەيدا بولىدىغان شۇ زىناقلىرى، ئوچۇق-يورۇق مىجەزى پۈتۈن ئوي-خىيالىمنى ئەسرا قىلدى، لېكىن كۆڭلۈمنىڭ ئاللىقانداقتۇر جايلىرىدا بىر تەشۋىش يوشۇرۇنۇپ تۇرغان بولۇپ، ئۇ بولسىمۇ ئىككىمىزنىڭ ئائىلە شارائىتى ئوتتۇرىسىدىكى پەرق ئىدى، بۇنى ئويلىغىنىمدا بىرخىل ئۈمىدسىزلەندىم دە سائەتكە قاراپ 5بولۇپ كەتكىنى كۆرۈپ دۇكاننى تاقاپ ئىشتىن چۈشتۈم، بۇ مۇشۇ ۋاقىتقىچە تېخى مېنى قاتمىغانىدى، بەلكىم مېنى ئالداپ قويغان بولسا كېرەك، بەلكىم مېنى ياراتمىغاندۇ، كۆڭلۈم شۇنداق يېرىم بولدى، ئىچىم پۇشقان كەيپىياتىمنى تەڭشەش ئۈچۈن تورخانىغا كىرىپ ئويۇن ئويناشقا باشلىدىم.بىردەمدە تېلېفونۇمغا ئۇچۇر كەلدى، خۇشياقماسلىق بىلەن ئۇچۇرنى ئاچتىم، بۇ ئۈندىدار ئۇچۇرى بولۇپ، ئۇنىڭ مېنى قاتقانلىق ئۇقتۇرۇشى ئىكەن.خۇشاللىقىمدىن توردىن چۈشۈپ ھېساباتنى ئۈزۈۋېتىپ، ياتاققا چاپتىم.

كېچە-كېچىلەپ ئۇنىڭ بىلەن ئۈندىداردا پاراڭلىشىش، نەچچە قېتىم كۆرۈشۈش بىلەن ئۇنى خېلى چۈشىنىپ قالدىم، ئۇنىڭغا بولغان مۇھەببىتىم بارغانسېرى ئۇلغىيىشقا باشلىدى.ئۇ پۇلدار ئائىلىنىڭ قىزى بولغان بىلەن پەقەتلا كىبىرى يوق ئىدى، گەرچە ئەركە مىجەزى بولسىمۇ لېكىن گەپ چۈشەندۈرسە ئاڭلايتى، ئۇنىڭ شۇئەركىلىگەن چاغدىكى يېقىملىق ئاۋازى، تاتلىق گەپلىرى  يۈرۈگۈمنى تامامەن سۇغۇرۇپ ئالدى، ئۇنىڭغا نەچچە قېتىم ئۆزۈمنىڭ يالغان سۆزلىگىنىمنى بوينۇمغا ئېلىپ، ھەقىقىي ئائىلە ئەـھۋالىمنى ئېيتىشنى  ئويلىغان بولساممۇ، لېكىن جۈرئەت قىلالمىدىم، چۈنكى ئۇنىڭدىن ئايرىلىپ قېلىشتىن شۇنچىلىك قورقىمەن.………………
تۆت يىل مانا شۇنداق شېرىن مۇھەببەت ئىلكىدە ئاخىرلاشتى، تۆت يىل ئىچىدە ئاپامنى بىر تېيىن خەجلەتمىدىم،ئوقۇش پۇلۇم ۋە تۇرمۇش خىراجىتىمنىڭ ھەممىسىنى ئۆزۈم تۆلىدىم، پەرى گەرچە باينىڭ قىزى بولسىمۇ، ئۇنىڭغا بىر پۇڭمۇ خەجلەتكۈزمىدىم، ئۇنىڭ پۇلىنى خەجلەشكە ئەرلىك غورۇرۇم يول قويمايتى.مەكتەپ پۈتتۈرىدىغان ياقا يۇرتلۇق ھەرقانداق بىر ئوقۇغۇچى ئىككى مەسىلىگە دۇچ كېلىدۇ، ئۇ بولسىمۇ يۇرتىغا قايتىش ۋە ياكى ئۈرۈمچىدە قېلىش.ئەگەر ئاپامنى ئويلىسام يۇرتۇمغا قايتىپ، خىزمەت تېپىپ، شۇيەردە توي قىلىپ، ئاپامنىڭ ھالىدىن خەۋەر ئالغۇم بار، پەرىنى ئويلىسام ئۈرۈمچىدە قېلىپ ئىگىلىك تىكلەپ پۇل تېپىپ مەيدەمنى تىك كۆتۈرۈپ ئۇنىڭ ئائىلىسىگە ئەلچى كىرگۈزۈپ تويۇمنى قىلغۇم بار،نۇرغۇن ئىكلىنىشلەردىن كېيىن مەن كېيىنكىسىنى تاللىدىم.ئىگىلىك تىكلەش يولى ئاسانغا توختىمىدى، 60مىڭ يۈەن ئۆسۈمسىز قەرزگە ئىلتىماس قىلىپ، بىر نەچچە بالىلار بىلەن كومپيۇتېر شىركىتى ئېچىۋالدۇق، باشتا زىيان تارتقان بولساقمۇ، كېيىن كىرگەن بالىنىڭ مەبلىغى بىلەن خېلى تىرىلىپ قالدۇق، كېيىنكى كۈنلىرىمىزدە سودىمىز خېلى ياخشىلىنىپ كەتتى.نەچچە ۋاقىتتىن بېرى ئاپام نۇرغۇن قېتىم تېلېفون ئۇرغان بولسىمۇ، ئۇنىڭ بىلەن تۈزۈك پاراڭلىشىپ باقمىدىم، بەش قېتىم ئۇرسا بىر قېتىم ئېلىپ ئەھۋاللىشىپ قوياتتىم، ئۇنىڭ دەيدىغان گەپلىرى قۇلۇغۇمغا ياد بولۇپ كەتتى، بالام، توي قىلغىن، سېنى سېغىندىم، يۇرتقا قايتىپ كەلگىن، ئاپامنىڭ بۇ گەپلىرى ماڭا بارغانسېرى تېتىمايدىغان بولۇپ قالدى، ئاپامنىڭ مېنى چۈشەنمەيدىغانلىقىدىن، چوڭ شەھەردە قېلىپ تەرەققىي قىلىشىمنى خالىمايدىغانلىقىدىن، مېنىڭ ئارزۇ-ئارمانلىرىمنى قوللىمىغايدىغانلىقىدىن ئىچىم پۇشاتتى.شۇڭا كېيىن ئۇنىڭ تېلېفونىنى ئالماسلا بولدۇم، كېيىن ئۇقسام ئاپام ماڭا تېلېفون قىلىپ مەن ئالماي كۆزى
ياخشى كۆرمەس بولۇپ قاپتۇ.پەرى بىلەن بولغان مۇناسىۋىتىم ئىنتايىن تۇراقلىق ئىدىم، ئۇ ماڭا تەگسە بەختلىك بولىدىغانلىقىغا ئىشىنەتتى، شۇڭا ئۇ قورقماي مېنى ئۆيىگە باشلاپ باردى.مەن ئۇنىڭ ئاتا-ئانىسى بىلەن كۆرۈشۈش ئۈچۈن ئاتايىتەن تۈركىيەنىڭ كاستيۇم بۇرۇلكىسىنى سېتىۋالدىم، ئايىغىمنى مايلاپ پارقىرىتىپ، قۇۋۋەت يېمەكلىكلەر، مېۋە-چېۋىلەرنى ئېلىپ ئۇنىڭ ئۆيى ئالدىغا بېرىپ تېلېفون قىلدىم.ئۇ چۈشۈپ مېنىڭ تۇرقۇمغا بىر سەپ سېلىۋەتكەندىن كېيىن، قولى بىلەن Ok دەپ ئىشارەت قىلدى.ئۇنىڭ ئوماق ھەرىكىتىدىن، ئۇنى قۇچاغلىغىم كەلگەن بولسىمۇ، ئۇنىڭ ئۆيىنىڭ ئالدىدا ئىكەنلىكىمنى تۇيۇپ ئۆزۈمنى بېسىۋېلىپ ئۇنىڭ ئارقىسىدىن ماڭدىم.ئىشىكنى ئۇنىڭ ئاپىسى ئاچتى.
-كېلىڭ، بالام، كېلىڭ.ئۇنىڭ ئاپىسىىڭ چېكىسىدىكى قورۇقلار ساناقلىقلا بولۇپ، ئاپامنىڭ پېشانىسىدىكى يول-يول قورۇقلارغا ئوخشىمايتى، ئۇنىڭ ئاپىسىنىڭ يۈزلىرى پارقىراپ تۇرىدىغان بولۇپ، مېنىڭ ئاپامنىڭ خۇنۇك، توپا باسقان يۈزلىرىگە ئوخشىمايتى، قىسقىسى ئۇنىڭ ئاپىسى بىلەن مېنىڭ ئاپامنى بىر يەرگە قويۇپ سېلىشتۇرسا، ئاپام ئۇنىڭدىن20ياش قېرى كۆرىنەتتى، بۇنى ئويلاپ كۆڭلۈم تۇيۇقسىز يېرىم بولۇپ قالدى، بۇ ئاپامدىن نومۇس قىلغىنىممۇ ياكى ئىچ ئاغرىتقىنىممۇ راستىمنى ئېيتسام ئۆزۈممۇ دەپ بېرەلمەيتىم.ئۇنىڭ دادىسى مېنى بۇرۇن بىر قېتىم كۆرگەن بولغاچقا تېزلا چىقىشىپ پاراڭلىشىپ كەتتۇق، ئۆزۈمچە ئۇلارغا خېلى ياراپ قالغاندەك تۇيۇلدۇم.گەپنىڭ ئارىلىقىدا پەرىنىڭ ئاپىسى تولۇمدىن توقماق چىققاندەكلا بىر سوئالنى سورىدى：
-ئاپىڭىزچۇ، ئاپىڭىز يۇرتۇڭلاردىماـ.بۇ سوئالغا قانداق جاۋاپ بېرىشنى بىلمىدىم، ئەگەر ھەئە دېسەم ئۇلارغا ئاپامنى، شۇ ئۆسۈپ چوڭ بولغان نامرات ئائىلەمنى كۆرسىتىشىم كېرەك، يالغان سۆزلىگىنىمنى ئېتىراپ قىلىشىم كۋرەك، ئۇ چاغدا ھەممە نەرسە چۈشكە ئايلىنىپ قالىدۇ، شۇڭا ئۇنىڭغا ۋىجدانىمغا مۇخالىپ جاۋابتىن بىرنى بەردىم.
-بۇلتۇر تۈگەپ كەتتى، ھازىر ھېچقانداق يېقىن تۇققانلىرىم يوق، ئاۋازىم تىترەك ئارىلاش خىرىلداپ چىقاتتى، پەرى لاپپىدە ماڭا قارىدى يۇ كۆزۈمدىكى پىلدىرلاپ تۇرغان ياش ئۇچقۇنلىرىنى كۆرۈپ ئارتۇق سوئال سورىمىدى، ئۇنىڭ ئاپىسىمۇ يېنىك ئۇھ تارتىپ قويدى.راستىمنى ئېيتسام نېمىگە يىغلىغىنىم ئۆزۈمگە بەش قولدەك ئايان ئىدى، مەن بىر ۋاپاسىز، ساپمۇ ساق ئانامنى ئۆلۈپ كەتكەنگە چىقىرىۋەتتىم، مەن بىر تۇزكۇر،ھازىرقى تۇرمۇشنى دەپ مېنى مۇشۇ كۈنگە ئۇلاشتۇرغان ئانامنى ئۇنۇتتۇم.ئۇنىڭ ئۆيىدىن چىقىپ پەرى مېنى قۇچاغلىۋالدى.
-ئاپىڭىزنىڭ تۈگەپ كەتكىنىنى نېمىشكە ماڭا دېمىدىڭىز، ئۇ ئازاپلارنىڭ ھەممىسىنى ئۆزىڭىز يالغۇز تارتتىڭىزمۇ، بۇنىڭدىن كېيىن نېمە گەپ بولسا ماڭا ئېيتىڭ بولامدۇ.ئۇنىڭ مۈرىسىنى يېنىك ئۇرۇپ قويدۇم، بىرەر گەپنى ئارتۇق قىلىشقا ماجالىم يوق ئىدى.
شۇنداق قىلىپ تويىمىزنىمۇ قىلىۋالدۇق، ئانام تويىمىزغىمۇ قاتنىشالمىدى، ئانام تويۇمدا بەختىمنى تىلىيەلمىدى، يالغانچىلىقىمنىڭ چېنىپ قالماسلىقى ئۈچۈن
20دەك كىشىنى تويۇمدا تۇققانلىرىم بولۇپ بېرىشكە چاقىردىم،ئۆزۈمنىڭ نېمىلەرنى قىلىپ يۈرگەنلىكىمنى ئويلاشقىمۇ ۋاقتىم يوق، كاللامدا پەقەت تويۇمنى تېزراق ئاخىرلاشتۇرۇش، ئايالىمغا ئالدىرىماي چۈشەندۈرۈپ، ئۇنىڭ بىلەن بىللە ئانامنى يوقلاپ كېلىش، ئانامنىڭ دۇئاسىنى ئېلىش، ئايالىم ئۇنىسا ئۇنى قايتۇرۇپ كېلىش، ئۇنىمىسا شۇيەردىكى ساناتورىيەگە ئورۇنلاشتۇرۇش پىلانىم بار ئىدى، لېكىن ئادەمنىڭ پىلانى تەقدىرنىڭ چاقپەلىكى ئالدىدا ھېچنىمىگە ئەرزىمەيدىكەن، ئايالىمغا ھەممە ئىشلارنى ئېيتىشنى ئويلاپ يۈرگىنىمدە ھەممە ئىش ئۆزلىكىدىن ئاشكارىلىنىپ كەتتى. بۇ يازمىنى ئاخىرىدا   Girl تەھرىرلىگەن. ۋاقتى  2014-12-21 11:21 AM

ئانە بىزنىڭ توپىلىق يوللىرىمىز، ئانە مەن ئۆسۈپ چوڭ بولغان تۇپراق، ئاپتوبۇسنىڭ دېرىزىسى سىرتىدىن مەن ئۆسۈپ چوڭ بولغان مەنزىرىلەرگە توختىماي سەپ سالاتتىم، بۈگۈن كەلمىگەن بولسام ئۆزۈمنىڭ بۇيەرنى قانچىلىك سېغىنغىنىمنى تۇيماسكەنمەن، ئىچىمنى ئاچچىق بىر يىغا قاپلىدى، مەن قاچاندىن باشلاپ ئۆزگەردىم، ئانامنى ئۇنۇتتۇم، چوڭ بولغان يېرىمنى ئۇنۇتتۇم، يالغان سۆزلەشكە باشلىدىم ئېيتىشكە تىلىم كۆيىدۇ ، شۇ تاپتا ئاپامنىڭ ئايىغى ئاستىغا يىقىلىپ ئۇنىڭدىن ئەپۇ سوراشتىن باشقىنى ئويلىغۇم كەلمەيتى.

چوڭ ئاپام ئالدىمغا چىقىپ، مەن بىلەن كۆرۈشكەندىن كېيىن، مېنى ئاپامنىڭ ياتىقىغا باشلاپ ئەكىرىپ ئۆزى چىقىپ كەتتى، ئۇنىڭ قوپال مۇئامىلىسىدىن ماڭا نارازى بولغانلىقى چىقىپ تۇراتتى.
ئاپامنىڭ كارىۋىتى يېنىغا كەلدىم، ئۇ ئۇخلاۋاتقان ئىدى، ئۇ تېخىمۇ قېرىپ كېتىپتۇ، چاچلىرى پۈتۈنلەي ئاقىرىپ كېتىپتۇ، كۆزلىرى ئولتۇرۇشۇپ، قوللىرى قېتىپ كېتىپتۇ، ئۇنىڭ بۇ ھالىتىگە قاراپ ياشلىرىمنى توختىتالمىدىم، ئۇنىڭ قولىغا چۈشكەن تامچە ياشلىرىم ئۇنى ئويغىتىۋەتتى.

-ھەئە،ئاپا، مېنى سېغىندىڭمۇ
-ئەلۋەتتە بالام،بەك سېغىمدىم، قولۇڭنى بەر يۈزلىرىمگە سۈرتىۋالاي، قولۇمنى ئۇنىڭغا ئۇزاتتىم، ئۇنىڭ مېنى تىللىشىنى،ئۇرىشىنى ئۈمىد قىلاتتىم، لېكىن ئۇ پەقەت قوللىرىمنى سۆيۈش، يۈزىگە يېقىشتىن باشقا ھېچ نەرسە دېمىدى.ئاخىرى ئۆزۈم چىدىيالمىدىم.

-دۇنيادا بالىسىنى كەچۈرمەيدىغان ئانا يوق بالام، مەن سېنى ئاللىقاچان كەچۈردۈم بالام، مەن ئۈچۈن سەن بەختلىك ياشىساڭ بولدى بالام، سەن كەتكىن، يالغانچىلىقىڭ چېنىپ قالمىسۇن، مەن ئەمدى مېيىپ بولدۇم، ساڭا دەخلى بولۇپ قالىمەن.
