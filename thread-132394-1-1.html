بۇ كىتاپنىڭ نامىنى كۆپ ئاڭلىغان لىكىن ئوقۇشقا ئەمدى مۇيەسسەر بۇلۇپتىمەن، بۇنى مەن ئۈرۈمچى شىنخۇا كىتابخانىسىدىن قانۇنلۇق سىتىۋالغان، شۇڭا سىلەرگە قىسقىچە مەزمۇنىنى دەپ بەرمەكچىمەن، چۈنكى بۇ مىنىڭ ھىس قىلشىمچە تولىمۇ ئۆزگىچە .مەن ئۆمرىمدە تۇنجى قىتىم قۇربان بولغان باش قەھرىمان ئۈچۈن ئىچىنمىغان بىر كىتاپ.

1936    يىللار ئەتراپىدا مىسىر ئەنگىلىيەنىڭ مۇستەھكەمچىلىكىدە بۇلۇپ،ئەنگىلىيە ساقچىلىرى ئەلېھرام كوچىلىرىدا ئۆزلىرى خالىغانچە ئەسكىلىك بىلەن بەخىرامان يۈرىشەتتى.

   ئىبراھىم ئەزەلدىن ئۆزىنىڭ بىر قەھرىمان بۇلۇپ قىلىشىنى خىيالىغىمۇ كەلتۈرۈپ باقمىغان ئىدى.چۈنكى ئۇ بىر ئادەتتىكى 17ياشلىق ئوتتۇرا مەكتەپ ئوقۇغۇچىسى ئىدى. ئۇنىڭ سىياسى بىلەنمۇ ھەپىلىشىپ كەتمەيتتى،سىياسى مەسللىلەر ئۈستىدە باش قاتۇرۇشنى خالىمايتتى ، بىراق ئۇنىڭ كۈچلۈك ۋەتەنپەرۋەرلىك ھىسياتى ئۇنى كۆپچىلىك بىللە بۇلۇشقا ، ساقچىلارغا قارشى تۇرۇش پىلانى تۈزۈپ ئۇلار ئۈستىدىن غەلبە قىلىشقا ئۈندەيتتى.ئۇ تولىمۇ ئېغىر بىسىق كۆپ گەپ قىلمايدىغان، ئىگىز بويلۇق بەستلىك بالا ئىدى،ئۇ مەن دەپ ھىچقاچان ئوتتۇرىغىمۇ چىقمايتتى،ساۋاقداشلارنى باشلىمايتتى،شوئارلارنى توۋلىمايتتى،ياكى كىشىنى ھاياجانغا سالىدىغان قىزغۇن نۇتۇقلارنىمۇ سۆزلىمەيتتى. بىراق ئۇنىڭ ئاندا ساندا دەپ قويغان شۇ ئورۇنلۇق گەپلىرى ۋە ئۇنىڭ شۇ ئاقىلانە تەدبىرلىرى ئۇنى بىلىپ بىلمەي كۆپچىلىككە باش قىلىپ  قوياتتى. كۆپچىلىك بىلىپ بىلمەي ئۇنىڭ ئەتراپىغا يىغىلىشاتتى.ئۇ ياشلىقىمدا ئادەتتىكى بىر ھەققانى ئادەم بۇلۇپ تۇرمۇش كەچۈرسەملا بولدى دەپ ئويلايتتى.

  ئۇ ئەنگىلىيە ساقچىلىرىنى كۆرسە يۈز-ئابرويى قاتتىق دەپسەندە بولغاندەك ئۇلارغا قاتتىق نەپرەتلىنەتتى.ئۇ ساۋاقداشلىرىغا شۇنداق ئامراق ئىدى، ئۇ نەچچە قىتىم شۇلار ئۈچۈن ئەنگىلىيەلىكلەرنىڭ ئوقى ئاستىدا ئۈلگىلى قىل قالغان ئىدى. بىر كۈنى ئۇ بۇنداق كىتىۋېرىشنى تولىمۇ ئەخمىقانىلىك دەپ قاراپ بىرەر ئىش قىلىش نىيىتىگە كىلىدۇ. ئۇ ئەڭ ئىشەنچلىك ئادەم ئىزدىدى، ئاخىرى تاپتى، ئۇلار قاتتىق پىلان تۈزدى، ۋە تۇنجى قىتىملىق ھەركەتنى ئېلىپ باردى، ئۇلار بىر ئەنگىلىيىلىك ساقچىنى ئۆلۈرىۋەتتى،ئۇنىڭدىن كىيىن ئۇلار تەشكىلاتىنى كېڭەيتتى، بۇ تەشكىلاتتا ئەڭ ئىشەنچلىك ئادەملەرلا بار ئىدى.ئۇلارنىڭ ھەركىتىمۇ كۆپەيدى، بىراق ئىبرايىم پىلاننى ناھايىتى پۇختا تۈزگەچكە ھەر قىتىم ئوڭۇشلۇق ھالدا قىچىپ كېتەتتى. بۇ ئەنگىلىيە ھۆكىمىتىنى ئاستى ئۈستى قىلىۋەتكەن ئىدى.بىراق مىسىر ۋەزىيىتى يەنە شۇنداق بىر ئىزدا ئىدى. زادى نىمىشقا؟ ئىبرايىم غەزەپلەندى، ئۇ ئەزەلدىن مىللىتىنى سۆيەتتى، مىللەتكە خائىنلىق قىلىۋاتقان ئاشۇ ئىتلارنى كۆرسە غەزەپلىنىپ كىتەتتى،ئۇ ئاخىرى مىسىرنىڭ دەل مۇشۇنداقلارنى يوقىتىشقا تەشنا ئىكەنلىكىنى ھىس قىلىپ يەتتى ۋە پىلان تۈزۈشكە پاشلىدى.ئۇلارنىڭ نىشانى ئەنگىلىيە غالچىسى ئابدۇرىھىم پاشانى يوقىتىش ئىدى. بىراق بۇقىتىم ئۇلارنىڭ ماشىنىسىدىن كاشال چىقىپ قىلىپ دەل ھىلىقى غالچا يىقىلغاندا ئۇ قاچالماي قالدى، ئەنگىلىيە ساقچىلىرى ئۇنى تۇتىۋالدى، كوچىدىكى مىسىرلىقلار بۇ غالچىنىڭ جەسسىتىگە نەزەر گۈرزىنىمۇ سالمايتتى، خوپ بوپتۇ ۋۇ ئىت! دەپ ئۆتۈپ كەتتى ، بىراق ئىبرايىمغا باشقىچە بىر ھۆرمەت بىلەن قارايتتى، ئىبرايىم بىر كىچىدىلا مىسىر خەلقىنىڭ قەلبىدىكى قەھرىمانغا ئايلاندى.ئۇنىڭ خەۋىرى گىزىتكە بىسىلدى،ئۇنى مىسىردا تونىمىغان ئادەم قالمىدى. ئۇنى تۈرمىگە ئاپىرىشىپ قويدى.ئىبرايىم بۇ يەردىن قاچماقچى بولدى ، ئۇ قەستەن ئاغرىق قىياپىتىدە تۇرىۋېلىپ دوختۇرخانىغا ئېلىپ بېرىلدى، مىسىز خەلىقى ئۇنىڭغا ئالاھىدە كۆڭۈل بۆلەتتى ، شۇڭا ئەنگىلىيە ھۆكىمىتى مىسىرنىڭ ئۇنى قىينىدى دەپ توپىلاڭ كۆتىرىشىدىن قورقۇپلا ئۇنى دوختۇرخانىغا ئېلىپ كەلگەنىدى، كىسەلخانىنىڭ ئىشىكىدە ئىككى ساقچى كۈزەتچىلىك قىلاتتى.ئۇدۇلدىكى ياتاقنى ساقچى باشلىقى ئىشخانا قىلدى. بۇ دەل رامىزان مەزگىلى بۇلۇپ ، نۇرغۇن كىشىلەر ئىبرايىمغا نەرسە كېرەكلەرنى ئېۋەتىپ تۇراتتى، نۇ نەرسىلەرنىڭ ھەممىدىنى ئۇ قاراۋۇللار ۋە تۈرمە باشلىقىغا بېرەتتى. ئىبرايىم ئۇلارغا ئۆچ ئەمەستى، ئۇلارنىڭ ھەممىدى مىسىرلىق بۇلۇپ جان بىقىش ئۈچۈن ئۆز ۋەزىپىسىنى ئادا قىلىۋاتاتتى خالاس.ئىبرايىم ئۇلارنىڭ ھۆرمىتىگە ئىشەنچىسىگە ئېرىشتى، ئۇلار ئۇنى تولىمۇ ھۆرمەتلەيتتى،پات-پات تۈرمە باشلىقىنىڭ ئىشخانىسىغا كىرىپ پاراڭلىشىپ تۇراتتى، پاتپات  دوختۇرخانىنىمۇ ئايلىنىپ قويدى، بارا-بارا ئايلىنىش ۋاقتىنى ئۇزارتتى، شارائىت پىشىپ يىتىلدى. ھەممە ئۇنى قايتىپ كىلىدۇ دەپ ھىچكىم ئۇنىڭغا دىققەت قىلمىدى. ئۇ بىر كۈنى دوختۇرخانىنىڭ ئىشىكىدىن چىقتى، ۋە بىر تاكسىنى توستى.ۋە تولىمۇ يىراق بىر يەردە چۈشۈپ قالدى. تاكسى شوپۇرىمۇ ۋە ھىلىقى دوختۇرخانا ئىشىكىگە قارايدىغان قاراۋۇلمۇ ئۇنى ھىچبىر تونىمايدىغاندەك، ئۇنىڭ چىرايىغا بەكرەك قاراپ قويدى-شۇ، ئۇلارنىڭ بۇنىڭ بىلەن نىمە مۇناسىۋىتى...

شۇنىڭ بىلەن ئۇ قىچىپ كەتتى، بىراق زادى نەگە بېرىش كېرەك،بىرنەچچە سائەتتىن كىيىن پۈتۈن مىسىر جىددى ھالەتكە ئۆتىدۇ.بىراق ئۇ ئاللىقاچان بۇنىڭ تەييارلىقىنىمۇ قىلىپ قويغان ئىدى.بۇ ساۋاقدىشى مۇھىيدنىڭ ئۆيى ئىدى.مۇھىيد ھەر قىتىملىق ئىمتىھاندا بىرىنجى بولىدىغان،سىياسەتتىن يىراق،نامايىش دىگەندەك ئىشلارغا ئارلاشمايدىغان،سەل يۈرەكزادىرەك، ئۆزىمۇ كۆزەينەك تاقايدىغان ۋىجىك بالا ئىدى. ھىچكىم ساراڭ بۇلۇپ قالغان تەقدىردىمۇ ئىبرايىمنىڭ ئۇنىڭدەك بىرىنى ئىزدەيدىغانلىقىنى خىيالىغا كەلتۈرۈپمۇ قويمايتتى شۇنىڭ بىلەن ئۇ مۇھىيدنىڭ ئۆيىگە باردى.مۇھىيد ئىشىكنى ئىچىپ خۇددى ئالۋاستىنى كۆرگەندەك قورقۇپ كەتتى. ئىبرايىم پەقەت ئىككى ئۈچ كۈن يۇشۇرۇنۇپ تۇرسىلا بولىدىغانلىقىنى ، ئەگەر ئائىلىسىدىكىلەر قوشۇلمىسا شۇ ھامان كىتىدىغانلىقىنى ئېيتتى، مۇھىيدنىڭ دادىسى قاتتىق روھى توسالغۇنى يىڭىپ ئۇنىڭ قىلىپ قىلىشىغا قوشولدى،مۇھىيدنىڭ ئىككى سىڭلىسى بۇلۇپ جەمىېي بەش جان بىللە تۇراتتى، ئۇنىڭ چوڭ سىڭلىسى سەمىيە ئاپىسى ۋە مۇھىيد ۋە دادىىىسى تۆتەيلەن كۈن بويى دەككە دۈكىدىلا يۈردى، بىراق ئۇنىڭ كىچىك سىڭلىسى ناۋال بولسا بىر خىل ھاياجان ئىلكىدە ئىدى، چۈنكى ئۇلارنىڭ ئۆيىدە بىر ۋەتەن قەھرىمانى تۇراتتى. ئۇ ئۆزىنى خۇددى ۋەتەن ئۈچۈن بىر ياخشى ئىش قىلىۋاتقان جەڭچىدەك ھىس قىلاتتى، ئۇ ئىبرايىمنى ھۆرمەتلەيتتى، بۇ باتۇر قىز بارا بارا ئىبرايىمنىڭ قەلبىگە بۈسۈپ كىرىشكە باشلىدى. ئىبرايىم بۇرۇن ئەزەلدىن ئاپىسىدىن باشقا ئاياللارغا دىققەت قىلىپ باقمىغان ،بۇ ئىبرايىمنىڭ قىزلارغا ئۆچلىكى ئەمەس بەلكى قىزلار ئۇنىڭغا ئەزەلدىن تەسىز قىلىپ باقمىغان ۋە قەلبىدە مۇھىم ئورۇنمۇ تۇتۇپ باقمىغان ئىدى. مىسىردا ئىبرايىمنى تۇتۇپ بەرگۈچىلەرگە بەش مىڭ فونىدسىرلىڭ بېرىدىغانلىق توغۇرلۇق بۇيرۇق چۈشۈرۈلدى، بۇ ئەينى ۋاقىتتا 50مىليوندەك كۈچ قۇدرەتكە ئىگە ئىدى. بىراق مۇھىيد ئائىلىسىدىكىلەر قۇلىقىنى يۇپۇرۇپ يۈرىۋەردى. بىراق ئەگەر ئىبرايىمنىڭ بۇ ئائىلىدىكىلەر بىلەن مۇناسىۋىتى بىلىنىپلا قالىدىغان بولسا بولدى بۇ ئائىلە تۈگەشتى دىگەن گەپ بولاتتى.سىياسى جىنايەتچىلەرگە ھەرگىزمۇ كەڭچىلىق قىلىش مۇمكىن بولمايتتى. ئىبرايىم ناۋل ئارقىلىق يىقىن دوستلىرى بىلەن كۆرىشىپ بۇ دۆلەتتىن كەتمەكچى بولدى ، ئۇنىڭغا بۇنىڭدىن باشقا يول يوق ئىدى.ئۇ ناۋالنى ئوتتۇردا مىڭىپ بېرىشكە تاللايدۇ. ناۋال ناھايىتى ھاياجان بىلەن ماقۇل بولىدۇ، بۇ ۋەتەنپەرۋەر باتۇر قىز پەقەت بىر ئېغىز گەپنى ئىبرايىمنىڭ دوستىغا شۇنداقلا يەتكۈزۈپ قۇيۇش ۋەزىپىسىنى شۇنچىلىك ئەزىز بىر قىممەتلىك ۋەزىپە قاتارىدا كۆرۈپ ماڭىدۇ. ئۇ ئۇنىڭ ئىنقىلاپچى دوستىنى تاپىدۇ، بىراق بۇ ئۇنىڭ ئويلىغىنىدەك ئۇنداق ئىبرايىمدەك بەستىلىك بالا ئەمەس ئىدى، بىر ئىسىلزادە ئائىلىسىدىكى ئورۇقراق كەلگەن بىر بايۋەتچە ئىدى. ناۋال مۇنداقلارمۇ ئىنقىلاپچى بولامدۇ دەپ ھەيران قالدى،

توغرا ئەسەرنىڭ مىنى بەكمۇ جەلىپ قىلىدىغان يېرى مانا مۇشۇ! بۇ يەردە جەڭ مەيدانى يوق ، لىكىن ئۇنىڭدىنمۇ ۋەھشى سىرلىق سىياسى سەھنە بار، بۇ يەردە باتۇر چىنىققان جەڭچى يوق، بىر مىللى غورورى ئۈچۈن كۆرەش قلىۋاتقان باي ۋە نامرات دەپ ئايرىلمايدىغان ئوقۇغۇچىلاردى تەركىپ تاپقان ئىنقىلاپچىلار بار. بۇ يەردە  يۈزەكى نامايىش قىلىپ ئالدىغا چىقىدىغانلار يوق ، بىراق سۈكۈت ئىچىدە بىراق زور غەيرەت بىلەن ۋەتەنگە تۆھپە قۇشۇۋاتقانلار بار.ئىنقىلابى قۇربانلارنى ھەممە بىلىدۇ، ئۇلار تارىخ بىتىگە يىزىلىدۇ، ئۇلارنى ھەممە كۈيلەيدۇ، بىراق بۇلار مەڭگۈ شۇنداق ئۈن تىنسىز ئۆتۈپ كىتىدۇ!
