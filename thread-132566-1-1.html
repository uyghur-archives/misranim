ئەسسالامۇ ئەلەيكۇم  مىسرانىم مۇنبىرىدىكى بارلىق قىرىنداشلار قانداق ئەھۋالىڭلار .
      ھازىر بەزىبىر ئۇيغۇرچە توربەتلەردە شەھۋانى ئىلانلار ئوچۇق ئاشكارە يامراپ كىتىپتۇ ، ئۇنداق توربەتلەرگە ئۇقۇشماي كىرىپ قالسىڭىز، ئەتراپىڭىزدا بىرەسى بولۇپ قالسا نىمە قىلىشىڭىزنى بىلمەي بەكلا ئوڭايسىزلىنىپ كىتىدىكەنسىز ھەم ئالدىراپ بىرە توربەتكە يۈرەكلىك كىرىشكىمۇ ئەيمىنىپ قالدىىكەنسىز .  ئوڭايسىزلىنىشقۇ كىچىك ئىش ،لىكىن ئۇنداق توبەتلەرنىڭ  تەسىرى  بەك  چوڭ بۇنى ھەممىمىز بىلىمىز  . خۇددى بىز بىلگەندەك شۇنداق ئىلانلارنى پەخىرلىنىپ تۇرۇپ توربەتلىرىگە قىستۇرۋاغان نۇمۇسسىزلا ئۇنۇڭ تەسىرىنى تىخىمۇ بىلىدۇ . بۇنداق ئىلانلارنى باشقۇردىغان ئورۇن يوقمۇ يا بوسىمۇ كارى يوقمۇ بىلمىدىم . بۇنداق ئىلانلارغا باشقىلا قانداق قارايدۇ پىكىرىڭلانى بىرىپ بىرىپ باقساڭلا ،چۇنكى كۆپىنچىمىزنىڭ تورغا چىىقالىغۇدەك بالا - چاقىمىز با ر، ئۇنداق توربەتلەرنىڭ كۆپىنچىسىدە كىچىك بالىلا ئامراق كارتۇن فىلىم ،ئويۇن قاتارلىقلارمۇ بولغاچقا ئۇلارنىڭ شۇنداق توربەتلەرگە كىرىپ قىلىشىدىن ساقلىنالماي قاىدىكەنمىز .

بۇ دۇنيانىڭ بەختى ياخشى ئەمەس ئۇ دۇنيالىقنىڭ بەختى ئۇچۇن خۇداغا قاتتىق يالۋۇرۇپ سېغىنغىن ئەي ئىنسان!
