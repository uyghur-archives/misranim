جەمىيەتنىڭ تەرەققى قىلىشى بىلەن ئۇچۇر ۋاستىلىرىمۇ كۈندىن كۈنگە تەتەققى قىلىۋاتىدۇ ،مۇنداراق قىلىپ ئېيىتقاندا تەرەققىيات ئۇچۇر ۋاستىلىرىدىن ئايرىلالمايدۇ ،بۇ ئۇچۇر ۋاستىلىرى ئىچىدە ئەڭ قولايلىق ،ئەڭ تارقىلىشچانلىقى تىز بولىۋاتقىنى تور ،ھازىر دۇنيادىكى ھەرقانداق بىر ئۇچۇر ۋاستىسى تىزلىك جەھەتتە تورغا يىتىدىغىنى يوق .بىز ئەتراپىمىزغا قارايدىغان بولساقمۇ ،ھازىر تېلۋېزور ،رادىيول ياكى باشقا ئۇچۇر ۋاستىسى ياشتا چوڭلارنىڭ ياشلارنىڭ بولسا يەنە شۇ تور دۇنياسىدا ئۈزۈپ يۈرىۋاتقىنىنى كۈرۈمىز ،ئەپسۇس مىڭلارچە ئەپسۇس ما ۋېيشىن دىگەن بىرنىمە چىقىپ دىيارىمىزدىكى ئەمدىلاتىن پەيدا بولىۋاتقان توربەت قۇرغۇچىلارنى پات پاراق قىلىۋەتتى .
بۇ ئىشلارمۇ ئاستا ئاستا بېسىقتى، چۈنكى بۇ ۋېيشىن دىگەن نەرسىگە تېما لازىم بولاتتى ،بۇ تېمىنى ساقلاشقا يەنە توربەتلەر لازىم ئىدى ،ئىزدەشكىمۇ يەنە شۇ توربەت،بۇ توربەتلەرنىڭ زىيارەت سانى يەنە ئاشقىلى تۇردى زىيارەتچىلەرمۇ سەل ئاشتى ،لېكىن زىيارەتچىلەردە ئۆزگىرىش بولىۋاتاتتى بۇ زىيارەتچىلەر بۇرۇنقىغا زادىلا ئوخشىمايتى ،بۇ زىيارەتچىلەر مۇنبەر ئارىلايتى تېما كۈرەتتى سەللا بولىدىكەن دىگىنىنى بىر ئامال بىلەن كۈچۈرۈپ ئۆز سالۇنلىرىدا ئۆز نامىدە ئېلان قىلىشاتتى، لېكىن بىرە توربەت بىرە سالۇندا يېڭىلىق دىگەن نەرسە مەۋجۇت ئەمەس ھەممىسى كونا ئۇيەر بۇيەردىن تېرىپ توشاپ قۇراشتۇرۇپ قويغان تىجىمەللەرنىڭ يىرتىق كوناپالىزىغىلا ئوخشايتى  ،ھەي جاھان بۇزۇلدى ما توربەت بۇزۇلدى ،توربەت قۇرغۇچىلارمۇ بۇزۇلىۋاتاتتى بۇزۇلغاندىمۇ ئادەمنىڭ ئىچى ئېلىشقۇدەك بۇزىلىۋاتاتتى ،بەزەندە بىزنىڭ توربەت لىرىمىزگە توربەت قۇرغىچىلارغا نىم بولىۋاتقاندۇ ؟بۇلار قاچانغىچە بەرداشلىق بېرەر ؟دەپ ئويلاپمۇ قالىمەن . ھازىر بۇ توربەتلەرنى ئارقىدىن يۇشۇرۇن يۈلەپ تۇرىۋاتقىنى بىرتوپ مىللى كارخانىلار ئەگەردە بۇ يۇشۇرۇن كۈچ بولمايدىكەن بۇ توربەتلەرنىڭ تەغدىرىدىن سۆز ئېچىش مۈمكۈن ئەمەس (بۇلار بۇنى بەلكىم ئېتىراپ قىلماس) لېكىن بۇ  كارخانىلار ئۇلارغا مۇھتاج ئۇلارمۇ بۇ كارخانىلارغا مۇھتاج ئىككىسى بىر يىپقا باغلانغان لېكىن ما توربەتچىلەر بەك  ھۇرۇن كارخانىچىلاردىن چوڭ تامادا لېكىن كارخانىچىلارنىڭ تەلىۋىدىن خېلى يىراقتا ،كارخانىچىلار توربەتچىلەردىن نارازى ،توربەتچىلەر كارخانىچىلاردىن نارازى ،بىرسى تاراتقۇنىڭ ھەققىگە چۈشلۈك ھەق ئالالمىدۇق دەيدۇ ،كارخانىچىلار بولسا بۇ بىرتوپ غىلاپ بەندىلىرىدىن نارازى ،مانا ئوتتۇرىدىكى بۇ يىپلار ئۈزۈلۈش باسقۇچىدا ،بۇ يىپ ئۈزۈلسە كىم زىيان تارتىدۇ ؟ ئەلۋەتتە تاراتقۇلار زىيان تارتىدۇ ئاۋۇ كارخانىچىلار يەنە بىر يول تېپىشى تەس ئەمەس لېكىن .............ھەي  سىز بەلكىم بۇ تاراتقۇلار نەدىن كەتكۈزۈپ قويىۋاتقاندۇ دەپ ئويلىغان بولىشىڭىز مۈمكۈن ،سىز بۇلارنىڭ شىركەتلەرئاچقان يىللىق خۇلاسە ياكى ھەرخىل ناملاردا ئۆتكۈزىۋاتقان يىغىنلىرىنى زىيارەت قىلىپ يوللىغان تېمىلىرىنى بىرە قېتىم كۆرسىڭىز بۇ نۇقتىنى ئېنىق ھىس قىلىسىز ،بۇلار كارخانىچىلارنى بەس بەستە زىيارەت قىلىشىدۇ تېما يوق . تېمىنى بىرەسى ئۇيەر بۇيەردىن قۇراشتۇرۇپ يوللىغاندىن كېيىن ھەممىسى شۇنى كۈچۈرۈپ ئۆز توربەتلىرىنىڭ نامىنى قۇيۇپ تارقىتىشىپ  كارخانىچىلارغا خىزمىتىدىن دوكلات قىلىشىدۇ .بىرەسى بولسىمۇ بۇنى ئازىراق قىسقارتىپ ياكى قۇشۇپ دىگىدەك بىرنەرسە قىلىشنى ئويلاپ قويۇشمايدۇ ،تېمىنى كۆزگىنىڭىزدە ياكۈلۈشنى ياكى يىغلاشنى بىلمەيلا قالىسىز،ئۆز ۋاقتىدىكى نەچچە ئونلاپ تاراتقۇ مۇخبىرلىرىنىڭ داغ-دۇغىلىق ھالدا (گوياكى كىنولاردىكىدەك)مىكراپۇننى تەڭلەپ سىزنى زىيارەت قىلغاندا سورىغان سوئاللاردىن بىرسىمۇ يوق. ھەممىسى كونا بىرنىمىلەر شۇمۇ قوراش تۇرۇلغان ،ھازىر دىيارىمىزدا شەخىسلەر نامىدىكى توربەت 200 ئاشىدىكەن ھۆكىمەتنىڭ چەكلىمىسى بولمايدىغان بولسا بۇ سان نەچچە قاتلىنىشى ئېنىق ھەرخىل سەۋەپتىن بۇ توربەتلەرنىڭ تەرەققىياتى ناھايتى ئاستا ياكى ئىستىقبالى يوق دىيىشكىمۇ بولىدۇ ما قېلىپ قالغان ئازغىنە تاراتقۇ قۇرغۇچىلىرى  بولسىمۇ بېشىنى بىر سىلكىۋېتىپ  مۇشۇ كارخانىلارنى زىيارەت قىلغاندا بولسىمۇ يېڭىلىق ئۆزگىچىلىك ھەققىدە بىر ئىزدىنىپ باقسا ،بۇ ھىچقانداق چەكلىمىگە ئۇچرىماي ئەكسىچە كارخانىچىلارنىڭ ئالقىشىغا ئېرىشسە تەبىئىلا كارخانىچىلار تاراتقۇلارنى قوللاش سالمىقىنى ئاشۇرغان بولاتتى .
بۇ مەزمۇنلار مېنىڭ بىرتەرەپلىمە كۆز قارىشىم مۇشۇ مىللى تىل يېزىقتىكى تاراتقۇلارنىڭ كەلگۈسى تەرەققىياتىغا ئازىراق ياردىمى بولاپ قالسا ھەجەپ ئەمەس توغرا تەھلىل قىلارسىلەر
