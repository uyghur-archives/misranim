بۇ يازما قارىغاندا رېژسور ئۆزى يازغان تىمىدەك قىلىدۇ ،مىنى رېژسورلۇققا بىكىتتى دەپ يېزىپتۇ  . ھەقىقەتەن بىر فىلىمنىڭ ئىشلىنىشىدە رېژسور ۋە باشقا خادىملار كۆپ جاپا تارتىدۇ  . رېژسورنىڭ ئۆزىنىڭ يېقىن ئاغىنىسىنى تىلغا ئېلىپ ، جاپانى ئاز  تارتمىغان تۇرسۇنبەگ ياسىن قېرىندىشىمىزنى تىلغا ئېلىپمۇ قويماسلىقى ئۇنى  ئوقۇغۇچى دەپ كۆزگە ئىلمىغىنىمۇ ياكى باشقا قانداق مەقسەت بار ؟  ئەسلىدىكى تەرجىمە سىنارىستنى بۇنچە كۆپتۈرۈپ ماختاپ ، ھەقىقى ئەجىرر قىلغۇچىنىڭ تىلغا ئېلىنىپمۇ قويۇلمىغىنىدىن ئۆكۈندۈم .  بىز ئۇيغۇرلاردا جاپا چەككەنلەرنىڭ زادى ھۆرمىتى بولمامدۇ ؟ يەنە ئەسلى سىنارىستنىڭ تەرجىمە سىنارىيسىنىمۇ ئۆزگەرتكەن دەپ ئەسكەرتىپتۇ بىر قېرىندىشىمىز . رېژسور سىز بۇ تىمىنى بىر مىھنەتكەش قېرىندىشىمىزنى مۇشۇنداقلا چەتكە قايرىپ قويۇپ تورمۇ تور ، سالۇنمۇ سالۇن يوللاتقۇزۇشىڭىزنى پەقەت چۈشەنمىدىم ، چۈشەنچە بەرگەن بولسىڭىز ، رەھمەت .

بىز كۆرۋاتقان  مۇشۇنداق تەربەيىۋى ئەھمىيەتكە ئىگە مەزمۇنى ياخشى ئىسىل فىلىملىرىمىزگە  رول ئىلىۋاتقان ئىختىدارلىق ئارتىسلىرىمىزنىڭ تىرىشچانلىقىدىن باشقا   شۇ فىلىمنىڭ ۋۇجۇتقا چىقىشى ئۈچۈن  قان تەرىنى سىڭدۇرگەن پەردە ئارقىسىدىكى جاپاكەش كەسىپ ئىگىلىرىگمۇ رەھمەت ئىيتساق بولدۇ جىمۇ  .مەنمۇ بۇ فىلىمنى تۇلۇق كۆرىۋاتىمەن ھەقىقەتەنمۇ تۇرمۇشقا يىقىن مەزمۇنى ئەھمىيەتلىك ياخشى فىلىم دەپ قارايمەن  شەخسەن بۇ فىلىمدىكى رول ئىلىۋاتقان  ئەڭ كىچىك ئارتىسنى تولمۇ ياختۇردۇم[ئىلفىرات ئەخمەت]  فىلىم دەسلەپ قۇيۇلغان كۈنلەردە ئۇنىڭ منىڭ كىچىك ۋاختىمدىكى رەسمىمگە   بەك ئوخشايدىغانلىقىنى  ھىس قىلدىم  ھى ھى ھى ھى ھى ھى فىلىمدىكى بارلىق جاپاكەشلەرنىڭ ئىشلىرىغا ئۇتۇق تۇرمۇشىغا بەخىت تىلەيمەن
