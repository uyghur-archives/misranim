نىيە ناھىيىسىنىڭ ناھىيە بولۇپ قۇرۇلۇشى جەريانىدا كۆرگەن ئاڭلىغانلىرىم

سادىق مۇھەممەد

مەن 1943-يىلنىڭ ئاخىرى كېرىيە ناھىيىسىدىن نىيىغا كەلدىم.بۇ چاغدا نىيىدا ساقچى ئىدارىسى قۇرۇلغىلى تېخى ئۇزۇن بولمىغان بولۇپ،زىياۋۇدۇن دىگەن كىشى ئىدارە باشلىقى ئىكەن.ساقچى ئىدارىسى 11-ئايدىن باشلاپ ئىش بىجىرىپ تۇرغان بولسىمۇ،لېكىن نىيىنىڭ رەسمى نىيە بولۇپ قۇرۇلغانلىقى تېخى ئېلان قىلىنمىغانىدى،ئەينى ۋاقىتتىكى رايون باشلىقى سوپىبەگ دىگەن كشى ئىدى.بۇ ئادەم8-ئاينىڭ23-كۈنى مىنى كېرىيە ناھىيىسىدىن ئۆزى بىلەن بىللە ئېلىپ كەلدى.كىيىن رايونلۇق ھۆكۈمەتتە كاتىپ بولۇپ ئىشلىدىم.شۇ چاغلاردا رايونلۇق ھۆكۈمەتتە بىر نەپەر رايون باشلىقى،بىر نەپەر كاتىپ ،بىر نەپەر دورغا بىگى ۋە بىرنەپەر جارا خىزمەتچى بار ئىدۇق.

شۇ يىلى 11-ئايدا جۇڭخۇامىنگونىڭ 32 يىللىقىنى خاتىرىلەشكە بىرلەشتۈرۈپ ،نىيە ناھىيىلىك ھۆكۈمەت(مۇئاۋىن ناھىيە دەرىجىسىگە تەڭ مەمۇرى ئاپپارات-ت)نىڭ قۇرۇلغانلىقى ئېلان قىلىنىپ،ناھىيەقۇرۇشقا تەييارلىق قىلىش-باشقۇرۇش ئىدارىسى دەپ ئاتالدى.مەنمۇ شۇ ۋاقىتتا كاتىبات بۆلۈم ئەزالىقىغا قۇبۇل قىلىندىم.ئۇ چاغدا ھۆكۈمەت باشلىقى ئۆزى بىلەن بىللە ئېلىپ كەلگەن بىر نەپەر تەرجىماندىن باشقا،نىيىدا خەنزۇتىلىنى بىلىدىغان ئادەم بولمىغاچقا كېرىيە ناھىيىسىنىڭ ئويتوغراق رايونىغا قاراشلىق يېسيۇلغۇن كەنتىدىكى خەنزۇتىلىنى بىلىدىغان ئىسمايىل ئاخۇن دىگەن كىشى بالا چاقىلىرى بىلەن نىيىغاكۆچۈرۈپ كىلىنىپ،خەۋەرلىشىش بۆلۈمىگە ئورۇنلاشتۇرۇلغان.

ئۇ چاغلاردا نىيىنىڭ يەركۆلىمىنىڭ قانچىلىك ئىكەنلىكىنى ھېچكىم بىلمەيتى.سوپىبەگ كىچىك ۋاقتىدا دادىسى غولجىدىن بىر تاتار مۇئەللىمنى تەكلىپ قىلىپ،ئۇنى ماتىماتىكا،جۇغراپىيە،تارىخ قاتارلىق پەنلەردە ئۇقۇتۇپ تەربىيىلىگەن بولغاچقا،ئۇ شۇ چاغدىكى سەۋىيىلىك زىيالى ھېسابلىناتتى.شۇ يىلى ئۇ نىيىنىڭ ئاددى خەرىتىسىنى تۇرغۇزۇپ چىقتى.خەرىتىدەنىيىنىڭ غەربى چىگرىسى ئەۋراز(نىيە بىلەن كېرىيە ئارىسىدىكى جاي ئىسمى)نىڭ ئېقىن ئورنىغا،جەنۇبى كوئىنلون تېغىغا،شەرىق تەرىپى ئەندىرە دەرياسىنىڭ بويىغىچە،شىمال تەرىپى كەشمىكۆل كەنتىنىڭ ئاياغ تەرىپى،ياتوڭغۇز ۋە ئىمام جەئفىر سادىق مازىرىنىڭ ئەتراپىدىكى بىپايان قۇملۇقلارغىچە توغرا كىلەتتى.بۇنىڭ بىلەن كىشىلەرنىڭ ئېڭىدانىيە ناھىيىسىنىڭ جۇغراپىيىلىك  ئورنى ھەققىدە دەسلەپكى چۈشەنچە ھاسىل بولدى.

شۇنىڭ بىلەن نىيىنىڭ خەنزۇچەئىسمىنى بىكىتىشكە توغرا كەلدى.بىر بۆلۈك كىشىلەر نىيىنىڭ خەنزۇچە نامى يەنىلا نىيا بولىۋەرسۇن دىگەن پىكىرنى ئوتتۇرىغا قويدى.بىراق،نىيا باشقۇرۇش ئىدارىسىنىڭ باشلىق گاۋجۈنتىيەن بۇ پىكىرگە قوشۇلماي مىن فېڭ دىگەن ئىسىمنى قويۇشتا چىڭ تۇردى ۋە بۇئىسىمنى يۇقىرىغا يوللاپ تەستىقلاتقاندىن كىيىن،نىيە ناھىيىسىنىڭ خەنزۇچە نامى مىن فېڭ دەپ بىكىتىلدى.1944-يىلى ھەر بىر كىشىگە پۇقرالىق كىنىشكىسى(ھازىرقى كىملىك ئورنىدا ئىشلىتىلەتتى)تارقىتىلدى.شۇ يىلى 4-ئايغا كەلگەندە نىيەرەسمى ناھىيە دەرىجىسىگە كۆتۈرۈلۈپ نىيە ناھىيىسى دەپ ئاتالدى.ھەر قايسى يېزىلارنىڭ ئىسمىمۇ ئۇيغۇرچە خەنزۇچە يېزىلىدىغان بولدى.
