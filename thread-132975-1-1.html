                                                   بويتاق ئاكام
                                              ئاپتۇرى: ئىليار كۆرەش

                                     بويتاق ئاكام  توي قىلاتتىم  دېدى بىزگە،
                                     كىرمەستىنلا  تېخى  ئۆيگە بۇسۇغا  ئاتلاپ.
                                    چۈشەنمەستىن تۇرۇپ قالدۇق تەڭلا خەپشۈك،
                                     تەسىرلىك  بۇ  ھېكايىسىنى  كەتتى  باشلاپ .

                                     دېدى  بىزگە : ئۆتۈپ  كەتتى  ياشلىقىم ،
                                     خىزمەت سەۋەب ئويلىيالمىدىم باشقىنى .
                                     تاللاپ  ئېلىش  بەكلا  زۆرۆر  بىزلەرگە ،
                                     ساختىسى كۆپ خەقتىن كىلىپ ئاشقىنى.

                                     تۇرۇپ  قالدىم  بازار  كاسات  نە  ئامال ،
                                     ئوپمۇ-ئوخشاش ھەقىقىسىگە ھەممىسى .
                                     تەڭ  ئۇرۇندا سېتىلماقتا  بىر - بىرلەپ ،
                                      قاماشتۇرار  بېشى  يوغان  سەللىسى .

                                     يالتايدىم مەن  تاپالمىدىم  بىرنىمۇ ،
                                      ياقتۇردىغان  ئۆزۈمنىڭكى  خىلىنى .
                                     ئېچىنىمەن ئەركەك تۇرۇپ ئۆزۈمدىن ،
                                     دىيەلمىدىم  ماۋۇ  ئاپامنىڭ  كېلىنى .

                                     دېدى ئاپام ئىچى ئاغرىپ ئاكامغا،
                                      يۇقىتىمەن  بۇ  ئازاپنىڭ  ئىزىنى .
                                     ئىلىپ بىرىمەن سېنى باشلاپ ھازىرلا،
                                     قوشنىمىزنىڭ خەلچەمخان دەپ قىزىنى.

                                     بۇيىغا  يەتكەن  پاكىزلا بىر قىزىم ئۇ ،
                                     يىراق تۇردى خەقنىڭكى يات كۆزىدىن .
                                     يۈزى تۈۋەن مۇلايىمدۇر شۇنچە بەك ،
                                     چىقىپ  تۇرار  ئەخلاقلىقى  سۆزىدىن .

                                     پىشقان  ئۆزى  تاماققىمۇ  قولى  گۈل ،
                                     ھەممە  ئىشتا  ھەقىقەتەن  يارايدۇ .
                                     بۇنى  ئاڭلاپ  ئەلچىلەرمۇ  تۇرالماي ،
                                     كىتەلمەستىن  ئىشىك تۈۋىدە مارايدۇ.

                                    يۈرەلمەيمەن بۇنداق تۇرساڭ جان ئوغلۇم،
                                    ئارزۇيۇمدۇر  كېلىن  قىلىش  مەن  ئۇنى  .
                                    قۇشۇپ  قويسام  بېشىڭلارنى  سىلەرنىڭ ،
                                    ئېچىلاتتى  بەختىڭ  سېنىڭ  شۇ  كۈنى .

                                    سەكرەپ كەتتى چۈشەندۈرۈپ ئاكاممۇ ،
                                    كۆڭلىڭىزمۇ  تېگىل  بولدى  ئاپا  دەپ .
                                    ئويلىدىڭىز ھەممىسى شۇ مەن ئۈچۈن ،
                                    دەيمەن سىزگە قۇشۇلمايمەن بىرلا گەپ.

                                    ئۇنداق  ئەمەس  چۈشەنمەيلا  ئانىجان ،
                                     بۈلۈۋەتتىڭىز  ئالدىراپلا  گىپىمنى .
                                     قۇلاق سېلىڭ مۇقەددىمە پەقەت بۇ ،
                                     ئۆرۈمەكتىمەن شانلىق تارىخ بىتىمنى.

                                    پۇرسەت كىلىپ ئۇچراتتىم مەن تۇيۇقسىز،
                                     مەلىكەمدۇر  قەلبىمدىكى  ئۇ  بەرنا .
                                    ھەيران قالدىم مودا يۈرەر ئاجايىپ ،
                                    نازلىنىشى شۇنچە تاتلىق ئۇ ھەتتا .

                                    جاننى ئالار كۆرگىنىمدە ھەر قېتىم ،
                                    خانېشلاردەك مېڭىپ كەتسە كوچىدا.
                                    تېلغاپ ئۆتەر يۈرىكىمنى ئۇ لەرزان ،
                                    مېڭىپ قالدىم ئاشىقلارنىڭ يولىدا .

                                    تارتماقتىمەن بۇ ئازاپنىڭ دەردىنى ،
                                    بىرەلمىدىم  بەرداشلىقنى  ئۇزۇنغا .
                                   يىتەلمەيدۇ بەسلەشسە ئەگەر خەلچەمخان،
                                    تەڭ  بۇلالماس  يېرىممۇ  ئۇنىڭغا .

                                   ئايمۇ خىجىل بۇلار ئۇنى كۆرگەندە ،
                                   جىسمىغا لايىق ئېسمىمىۇ ئايجامال .
                                   مېنىڭ ئۈچۈن بىرىڭ ئۆيىگە ئەلچىگە،
                                   ئېرىشىشتە قىلىڭ تېزدىن بىر ئامال .

                                  خىلىم ئەمەس خەلچەمخانلار پەقەتلا ،
                                  تۇرار دائىم ياغلىقى بېشىدا ھامان .
                                   يىتەلمەپتۇ  ھەقىقەتەن  مودىغا ،
                                  قارشى ئالماس كونىلىقنى بۇ زامان.

                                  قىلالمىدۇق قارشى پىكىر ئاكامغا ،
                                  ئاپام باردى ئالدىراپلا ئەلچىگە .
                                  ھەل بۇلۇپتۇ ئاكامنىڭكى ئارزۇسى،
                                  دىيشىپتۇ  توي  كۈنىنى  ئەتىگە .

                                 ئۆتكۈزۋالدۇق توينى شۇنداق كۆڭۈللۈك،
                                 كۈنۈپ  قالدى  ئۆيىمىزگە  ئۇ  قىزمۇ .
                                 يېرتىپ تاشلاپ نىقابىنى شۇ ھامان ،
                                 ئاشكارلاندى بارا - بارا مۆڭگۈزىمۇ .

                                 ھاكاۋۇرلۇق قۇيۇپ يۈرەر تەمەننا ،
                                 پەرداز قىلار توختىماستىن ئۆزىنى .
                                كۆزگە ئىلماس چوڭلارنى ئۇ ھېچقاچان،
                                 ساۋاتسىزدۇر ئۆي ئىشىدا يوق ئىپى .

                                 ئايلىنىپتۇ  ئەخلەتخانىغا  ئۆيىمىز ،
                                 قېززىپ كەتتى چاشقانلارنىڭ بەزمىسى.
                                جەڭلەر دەھشەت توختىماستىن قاباھەت،
                                بولدى تارمار چاشقانلارنىڭ ھەممىسى .

                                ئىشغال قىلىپ ئىستىكامنى ئاخىرى ،
                                لەپىلدىدى  غەلبە  تۇغى  قۇلۇمدا .
                                قاراپ قالدىم ھەيرانلىقتا ئىختىيار ،
                                ئولتۇرۇپتۇ خۇتۇن خىيالىي بۇلۇڭدا.

                               كۆرگىنمىز  شۇ  قوللىرىدا  تېلفۇن ،
                              بەكلا خۇشال تېپۋالغاندەك بىر گۆھەر.
                              كۆردى بىزنى قۇيۇپ تېلفۇن شۇ زامان،
                              دېدى بىزگە : كېلىشڭلاردىن بىخەۋەر.

                              قەتئىي كەلمەس ئاشپەزلىكمۇ قۇلىدىن ،
                               قىلالمايدۇ  تاماقنىمۇ  ئوخشىتىپ  .
                              ئايان بۇلار بىر قۇشۇقتا ھەممىسى ،
                              يىگەن كىشىنىڭ ئىشتىھاسىنى توختىتپ.

                              بىكار تۇرماي توختىماستىن ياسىنىپ ،
                              ئۇچىسىدا  يىرىم  يالىڭاچ  يوپكىسى .
                             ماختاپ كىتەر ئاكاممۇ تەڭ جۆر بۇلۇپ،
                              ئەكىلىپ بىرەر تۈگەپ كەتسە ئوسمىسى.

                              پەخىرلىندۇ  ئاكام  چىقسا  سىرتقا ،
                              ئايالىنىمۇ بىللە سۆرەپ تەڭ ئېلىپ .
                              ئوخشىدىغۇ بۈگۈن ئىشلار ئاجايىپ ،
                              كۆرۈمەكتە ئەپتى - بەشىرە ئېچىلىپ.

                              جىرىڭلىدى  تۇيۇقسىزلا بىر  تېلفۇن ،
                              ئايالىمنىڭ سومكىسىنىڭ ئىچىدىن .
                             ئىلىپ شۇئان تۇرۇپ قالدىم مەن ھەيران،
                             كەلگەن تېلفۇن نۇمۇرىدىكى ئېسىمىدىن.

                             ئېلىشىمغا قارشى تەرەپتىن بىر ئاۋاز ،
                             ئاھ ئۇمىقىم! سېغىندىمغۇ بارمۇ سەن.
                             توي قىلساڭمۇ مېنىڭ ئۈچۈن بەرىبىر ،
                             يەنە ئوخشاش بۇرۇنقىدەك سۆيىمەن .

                             ئۇنۇتمايمەن دېگەن سۆزۈڭ ئېسىمدە ،
                            ھەرگىز سېنى ئېرىمگە ئوخشاش ئالدىماي.
                             ئۇچىغا  چىققان  ھاماقەتتۇر  ئۇ  ئېرىم ،
                             شۇنداقتىمۇ بىللە يۈرىمىز چاندۇرماي .

                             تىللىدىم مەن باسالمىدىم دەر-غەزەپ ،
                             نائەھلىمۇ زۇۋان سۈرمەي شۈمشىيىپ .
                             ئۈزۋەتتى  غايىپ  بۇلۇپ  دېرەكسىز ،
                             ئازاپلاندىم بېشىم پەسكە مۈكچىيىپ .

                             تىللاپ كەتتۇق ئاكامنى تەڭ ھەممىمىز،
                             تېقىۋېتەيمۇ  پەرق  ئەتمەس  كۆزۈڭگە .
                             جاۋاب بەرگىن نېچۈن بۇنداق سەۋەبى ،
                             زادى  يارىشار  بويتاقلىقىڭ  ئۆزۈڭگە .

                             قىلدى ئاكام ئايىماستىن قاتتىق قول ،
                              بىرۋەتتى   ئايالىنىڭ   خېتىنى  .
                             شۇندىن بۇيان بەردى قەسەم ئۆزىگە،
                             قىلمايدىغان توينىڭ ئەمدى گىپىنى .

                                       مەنبە : ئۆز قەلىمم

راستنىى دېسەم بۇ شېئىرنى يازغاندا ھەقىقەتەن قېينالدىم،بۇلۇپمۇ شېئىرنى قانداق باشلاپ قانداق ئاخىرلاشتۇرۇش قانداق يازسام ھەجۋى يۇمۇرىستىك ھەم ناچار ئىدىيەلەرنى ئۆتكۈر ھالدا قامچىلىغان ئېسمى-جىسمىغا لايىق ساتىرا بۇلىدۇ دېگەنلەر  توغرىسىدا كۆپ باش قاتۇرردۇم. ئاز يازاي دېسەم مەزمۇن ۋە ئىدىيەلەر  تۇلۇق بولماي مۇجمەل تۇتۇق بۇلۇپ قىلشتىن ئەنسىردىم شۇنىڭ بىلەن ئۆزۈمنىڭ ئەركىن مەيدانىمدا تۇرۇپ ھەر بىر ھالقىلارغا دىققەت قىلغان ئاساستا مەزمۇنى چۇڭقۇر تۇلۇق مەناسى بار ئۇشبۇ ئەسەرنى يىزىپ چىقتىم.  بۇ شېئىرنى يىزىش ئۈچۈن ساق بىركۈن ۋاقىت كەتتى يېزىپ بۇلۇپ قارىسام تالا قاپقاراڭغۇ بۇلۇپ كىتىپتىكەن، شېئىرنى ئەتىگەندە يىزىشقا باشلاپتىكەنمەن ۋاقىتىنىڭ قانداق ئۆتۈپ كەتكەنلىكىنى تۇيماي قاپتىمەن. قەلەمنى توختىتىپ  ئاستا ياستۇققا بېشىمنى قويغىنىمنى بىلمەن كەينىدىنلا  شۇزامان ئۇيقۇغا كىتىپتىمەن! قېزىقىشى بار نەرسىگە  ھەرقانداق جاپالارمۇ كۆزىگە كۆرۈنمەيدۇ ئادەمنىڭ....
