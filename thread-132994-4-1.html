سىز مۇشۇ دېتالنى ياسىغان ئابدۇراخمانجان بىلەن مۇشۇ دىتال توغرىسىدا سۆزلىشىپ باقتىڭىزمۇ ؟ سىز ئىلىم - پەنگە ھۆرمەت قىلىپ مۇشۇ بالچىلىك نەتىجە قازنالدىڭىزمۇ ؟ مەنمۇ كومپىيوتېر كەسپىدە تۆت يىل ئوقۇدۇم.گەرچە بۇ سېستىمىنى بەك كۆپتۈرۋەتتىمۇ دەيلى ئەمما مۇشۇنداق سېستىمىنىڭ ياسىغىنى توقۇلما ئەمەستۇ ؟ گەرچە بۇ سېستىما ھەممە سۆزنى تەرجىمە قىلاممىسىمۇ دەسلەپ ياسىغاندا سىناق باسقۇچتا بۇ يىگتى ئۆزنىڭ ئىددىسىيگە ئاسەسەن ئاز سانلىق جۈمىلىنى تەرجىمە قىلدۇرۇپ مۇشۇ دىتال ئۈچۈن ئاساس سېلىشقا ئىشىك ئاچقاندۇ ؟ سىز نىمىشقا بۇ يىگىت بىلەن خېلى ئوبدان تونشىسىز نىمىشقا ئۆزىدىن مۇشۇ دىتال توغرىسىدا پىكرىلىشىپ باقمايسىز ؟ سىز خەخنىڭ كومپىيوتېرغا سېستىما قاچلىياممايدىكەن دەپ كىچىكككىنە كەمچىلكىنى تۇتۋىلىپ باشقىلارنىڭ قىزقىشىغا سۇ سېپىسىز ؟ نىمىشقا توردا تارقىلىپ يۈرگەن كۆپتۈرمىلەرنى زادى مۇشۇ دىتالنى ياسىغان ئابدۇراخمانجان تارقاتتىمۇ ئەمەسمۇ تەپسىلى سۈرۈشتۈرمەي ئاتايتىن بىر كۈن ۋاقتىڭىزنى سەرىپ قىلىپ باشقىلارنىڭ ئەجىرنى يوق قىلىسىز ؟ مەنمۇ كومپىيوتېرغا سېستىما قاچلاپ رايۇنغا بۆلۈشنى تازا ئۇقمايمەن ئەممە مەن تور بەت،مۇنبەر بېتى ،ئۇنۋىرسال بەت قاتارلىق سېستىمىلارنى مۇستەقىل تۈزەلەيمەن،ھەم ئوقۇت پۈتتۈردىغان ۋاقىتتا سىنىپكى ئالتە بالىنىڭ ئوقۇش پۈتتۈرش لايھھەسىنى مەن بىر قوللۇق پۈتتۈردۇم. ھەم ئۆزۈمىنىڭ ئوقۇش پۈتتۈرۈش لايھەسىدە خەنزۇچە ئىسمىنى ئۇيغۇرچە ئىسمغا ئاپتوماتىك تەرجىمە قىلدىغان سېستىمىنى ياساپ باشقىلارنى ھاڭ تاڭ قالدۇردۇم ھەم ئوقۇتقۇچۇم 500 يۈەن مۇكاپات بەردى.مەن مۇ بۇ سېستىمىغا ئاران بىر ھەپتە ۋاقىت سەرىپ قىلدىم ھېچقانداق ساندان ئىشلەتمىدىم مەيلى ئۇيغۇرلارنىڭ ئىسمى بولامدۇ خەنزۇلارنىڭ ئسىمى بولامدۇ ھەممە ئسىمبنىڭ خەنزۇچىسىنى ئۇيغۇرچىغا ئاپتوماتىك تەرجىمە قىلىدۇ. دۇنيادا بىر مىليون ئىسىم بولسا ھەممىنى ئۇيغۇرچىغا تەرجىمە قىلالايدۇ.ھېچقانداق ساندان يوق.مەنمۇ نۇرغۇن ئەھۋالنى ئويلىشىىپ تەرجىمىنى توغرا تەرجىمە قىلىش نىسىبىتى  80 % ئەتراپىدا چىقتى دەپ ئويلۇدۇم.قانداقلا بولمىسۇن باشقىلارنىڭ قىزغىنلىقىغا ھەرگىز سۇ سەپمەيلى .بۇبالنىڭ بىلدىغىنى سىز ھەم بىزندىن ھەرگزى تۆۋەن ئەمەس.دۇنيادا قىلغىنى بولمايدغان ئىش يوق. بىر ئەسىر بۇرۇن بەلكىم ئېغرىلى 10 توننا كېلدىغان تۆمۈر ئاسماندا ئۇچىدۇ ،دۇنيانىڭ بۇ چېتىدىكى ئادەم بىلەن ئۇ چېدىدا تۇرۇپ پاراڭلاشقىلى بولىدۇ دىسە بەلكىم مۇمكىن ئەمەس ئىشەنمەيمىز ،بەلكىم ئۇيغۇر تىلى كۆمپىيوتېرغا كىرمىگەن ۋاقىتا بەلكىم ئۇيغۇر تىلىنى كۆمپىيوتېرغا تونۇتقىلى بولمايدۇ .ئۇيغۇرچىنى ئوڭدىن سولغا يازىدى ھەم بىر ھەرىپنىڭ تۆت خىل شەكلىنىڭ ئورنىڭ ئوخشماسلىقىغا ئاساسەن ئوخشاش بولمايدۇ دەپ ئويلىشىمىز مۇمكىن. شۇڭا تىرىشسىلامۇمكىن بولمايدىغان  ئىش يوق ئۇيغۇرلانىڭ بىلىم سەھيەسى ھەرگىز بوش ئەمەس ئۆتكۈر ئەركىنمۇ جۇگگۇ بويچە 1 مىليون ئادەم قاتناشقان مۇسابىقىدە چىمپىيون بوپ بىزنى شۇنداق خوش قىلىدۇ.مانا بىز ئۇيغۇرلارنىڭ ئەقىل پاراستى ۋە تەسەۋۇرغا ئىنتايىن باي ئىكەنلىىنى چۈشەندۇردۇ.
