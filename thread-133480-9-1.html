ئەسسالامۇ ئەلەيكۇم مەن مۇشۇ يىقىندىن بۇيان ئەلى قۇربان لىكسىيەلىرى ۋە زىۋىدە ئابدۇكىرىم ئوبزۇر ماقالىسىغا قارىتا چوڭقۇر ئويلىنىپ چىقتىم ، ئەلى قۇربان مۇئەللىم بولسا غەرىپ پىسخولوگىيسىنىڭ ۋە يەھۇدى ئالىم پەياسوپلارنىڭ تەزكىرىسىنىڭ تەسىرىگە ئۇچىرغان ئىلغار ئىدىيدىكى پىسخىكا مەسلىھەتچىسى ، ناتىق، يىتۈك ياش پىسخولۇگ، زىۋىدە ئابدۇكىرىم ھەر جەھەتتىن ئەتراپلىق بىلىم ئىگەللىگەن،تۇنجى ئۇيغۇر ئايال پىسخىكا  دوكتۇرى. كۆپچىلىك ھەممىڭلار كۆردىڭلار نەچچە ۋاقىتتىن بېرى ئەلى قۇرباننىڭ بىيجىڭدىكى لىكسىيەسىنى كۆرگەنلەر ئەلى مۇئەللىمنىڭ بىلىمىگە قايىل بولۇشقانئىدى، ئارقىدىنلا زىۋىدە ئابدۇكىرىم مۇئەللىمنىڭ ئوبزورىنى كۆرگەندىن كىيىن بۇ ئوبزورغىمۇ ئالقىشياڭرىتىپ بەس بەستە ئىنكاسقا قارىغۇلارچە ئەگىشىش دولقۇنى كۆتۈرۈلدى، مۇنبەرلەردە ئىنكاسلار پاخاللىشىپ ھەق بىلەن ناھەق ئارلىشىپ تورداشلارنىڭ كاللىسى بۇرمىلىنىپ، ئاز ئازدىن تاللىۋىلىش خائىشى بويچە ئاخىرىدا مەن مۇھىممۇ يۈسۈپ خاس ھاجىپمۇ دىگەن بىر ئاددىي ئىيتىلغان سۆزگە يىغىنچاقلاندى، ئەمىليەتتە  بۇ سۆزنى دىيىش ئۇنچە تەس بولمىسىمۇ، بۇ سۆزنىڭ مەسئۇلىيتىدىن قاچماق ھەقىقەتەن تەس ئىش. مەن بۇ سۆزنى خاتا دىمەكچى ئەمەس، چۈنكى بۇ شۇ ۋاقىتتا شارائىت ماكاننىڭ چەكلىمىسى ئىچىدە ئېيتىلغان دەپ قارايمەن، ئوقۇغۇچىلارنىڭ ئالدىدا بۇ سۆزنى ئېيتىش بەلكىم ئۇلارنىڭ روھىنى قېزىپ چىقىرىشقا ياردەم بېرىشى مۇمكىن، ياكى كونىلارنىڭ مىڭ شھىتتىن بىر  تىرىك ئەلا دەيدىغان پەلسەپىلىك ئەمما دەبدەبىسى يوق بۇ سۆزگە ماس كەلگەن بولىشى مۇمكىن، مەنچە بولغاندا زىۋىدە مۇئەللىم خېلى ئەتراپلىق تەھلىل قىلغان بولسىمۇ، بىراق 6 قىسىم لىكسىيەنى تولۇق كۆرۈپ بولۇپ تەنقىد يازغان بولسا تېخىمۇ ياخشى  تولۇق بولغان پاكىتلارغا مۇيەسسەر بولاتتى بەلكىم .
