مىنىڭ تور مۇنبەرلىرى بىلەن ئۇچىراشقىنىمغا ئەمدى ئىككى يىل بولغاچقا بۇ قىتىمىقى ئەلى مۇئەللىمنىڭ بىيجىڭدا سوزلىگەن لىكسىيسى مۇناسىۋىتى بىلەن كۈتىرلگەن زوركۈلەمدىكى ، كوپ تورداشلار ئىشتىراك قىلغان ئىلمى، ھەم پىكىرى چوڭقۇر ئىلىپ بىرىلغان بۇنداق مۇنازىر لەر نى  بىلمەيدىكەنمەن. بۇ قىتىمقى مۇنازىرە گەرچە ئەلى مۇئەللىم، زىۋىيدە خانىم،ۋە تورداشلار ئوتتۇرسىدا ئىلىپ بىرىلغان بولسىمۇ بۇ جەريانىدا مەن پىسخىكا ئىلمى توغىرسىدا خىلى كوپ بىلىملەرگە ئىرىشتىم. مەن بۇنىڭغا ناھايىتىخۇشال. بۇلۇپمۇ بۇ جەريانىدا نۇرغۇن تورداشلارنىڭ ئەلى مۇئەللىم، زىۋىيدە خانىم  ۋە بەزى تورداشلار ئارسىدىكى تالاش-تارتىشلارنى قايىل قىلارلىق ئىلمى پاكىتلار ئارقىلىقئەتىراپلىق مۇنازىرە ئىلىپ بارىدىغان ئىلىمى مۇنازىرە روھى مىنى تەسىرلەندۈردى.

ئەمما مۇنازىرە نوقتىسىنى تالاش-تارتىش قىلىدىغان ئەمەلى مەسىللەرگەقاراتماي ھەدىسىلا ئەلى مۇئەللىم، زىۋىيدە خانىملارغا ۋە ئوخشىمغان كوز قاراشتىكى تورداشلارغا قارىتدىغان، كىشىلەرنىڭ شەخسىيتىگە دەخلى – تەرىز قىلىدىغان يازما ۋە ئىنكاسلارغا مەن يىقىندىن دىققەت قىلدىم.شۇڭا تورمۇنبەرلىرىدىكى ئىلمى مۇنازىرەروھى ئۈستىدە ئويلىنىپ قالدىم.

يىقىندا مەن مىللىتمىزدە كوزگە كۈرۈنگەن بىر مەرىپەتپەرۋەركىشىنىڭ  ئاۋامنىڭ  قايىللىقىنى قوزغىغان بىر ئۇنتۇلماس ئىشى توغىرسىدا ئۇنىڭغا مىننەتدارلىقىمنى بىلدۈرۈپ بىر پارچە خەت يازغان ۋە بۇخەتنى بىرەر مۇنازىرە مۇھىتى ئىلمىراق مۇنبەرگە بىرىشنى ئويلىغان ئىدىم.كىيىن سەل ئىككىلىنىپ قالدىم. نەتىجىدە بۇنى يىشى مەندىن كىچىك بولسىمۇ بىلىمى مەندىن يۇقىرى، خەنزۇچە، ئىنگىلىز تىللىرنىمۇ پىششق بىلدىغان ، تورلاردىمۇ قەلىمى خىلى تۇنۇشلۇق بولغان ئۇقۇتقۇچى ئىنىمىزغا بۇ خەتنى كورسەتتىم ۋە بۇ ئىش توغىرسىدامەسلىھەت سالدىم. ئۇ مىنىڭ مەقسىدىمنى ئۇققاندىن كىيىن خەتنىڭ مەزمۇنىدىن مەمنۇن بولغانلىقىنى بىلدۈردى.ئارقىدىن ئۇ؛

ئاكا،بۇخەت ناھايىتى ياخشى يىزىلىپتۇ، بۇنداق مەدىھيە سىوزلىرىنى ئەشۇنداق كىشىلىرىمىز ئەڭ ياخشىسى ھايات ۋاقتىدا ئاڭلىغىنى ياخشى،ئەمما بۇ خەتنى توردائىلان قىلمىغىنىمىزتۈزۈك، مەتبۇئات ئۇرۇنلىرىغا بىرەيلى دىدى. مەن سۇئال نەزىرىدە ئۇنىڭغا قارىدىم، ئۇ ؛ ئەسلىدە بۇنى توردا ئىلان قىلساق بۇ مەرىپەت ئىگىسىنىڭ ياخشى ئىشلىرىدىن تىىخمۇ نۇرغۇن كىشىلەرنى خەۋرلەندۈرگىلى بۇلاتتى.ئۇ ھايات ۋاقتىدىلا نۇرغۇنلىغان مەدىھىيە سوزلىرىنى ئوز قۇلىقى بىلەن ئاڭلاشقا مۇيەسسەربولغان بۇلاتتى، چۈنكى بۇ كىشىنى مىللتىمىزدە ناھايىتى نۇرغۇن كىشىلەر چىن دىلىدىن ھورمەتلەيدۇ. شۇڭا توردىمۇ نۇرغۇن ياخشى باھالارغا ئىرشىدىغانلىقى ئىنىق. لىكىن بىزنىڭ تور مۇھىتىمىزدا ئىلمى بولمىغان بىرقىسىم تورداشلىرىمىز بار، نۇرغۇن ئىلمى باھالار بىرىلگىنى بىلەن يەنە بىرنەچچە ئىلمى بولمىغان ، كىشىلەرنىڭ شەخسىيتىگە ھۇجۇم قىلىدىغان گەپلەر چىقسا ئۇ كىشىنىڭ كوڭلى ئازار يەپ قالىدۇ. نەتىجىدە بۇخەتنىڭ ئۈنۈمى بىز كۈتكەندەك بولماي قالىدۇ. دىدى. ئويلاپ باقسام بۇ گەپلەرنىڭ ئاساسى بار ئىدى.

چۈنكى مەنمۇ مۇشۇ ئىككى يىلدىن بىرى توردا نۇرغۇن تەسىرلىك ئىشلارنى، ئوز بىلىمىنى خەلقىدىن ئايىمايدىغان، مەۋقەسى ئىنىق، ئاق-قارىنى پەرىق ئىتەلە يدىغان،ھەققانىيەتنى ياقىلايدىغان، پىكىرى ئوتكۈر، ئىسىل  تورداشلارنى كوردۈم ，ئۇلارنى بىلدىم. شۇنىڭ بىلەن بىرۋاقىتتا توردىكى ئىلىم مۇنازىرلەرگە يۈزەكى قارايدىغان،مىنى باشقىلار بىلەلمەيدىغان تۇرسا دەپ مەسئۇليەتسىزلىك بىلەن ئىلمى ئاساسى بولمىغان ، كىشىلەرنىڭ روھىيتىگە ئازار بىرىدىغان ، بىزگە روھى بايراق بۇلىۋاتقان،بىز مۇھىم بىلىدىغان بەزى كىشىلىرىمىز ھاقارەتلىنىدىغان  ئىشلارنى ئۇچىراتتىم. ھەتتا مەن تۇنجى ئەزابولغان مەلۇم بىر تور مۇنبرىدە ئۇستاز شائىر مۇھەممەتجان راشىدىن ئاكاھاقارەتلەنگەن بىر تېمىنى كۈرۈپ شۇ چاغدا مۇنبەردە ئىنكاس يىزىپ بۇ مۇنبەردىن دەرھال چىكىنىگەنلىكىم توغىرىسىدا بايانات ئىلان قىلغان ئىدىم. كىيىن   مۇشۇلارنى ئويلاپ مەنمۇئۇنىڭ بۇ خەتنى تورغا بەرمەسلىك تەكلىۋىگە قۇشۇلدۇم. ئەمما بۇ ئىش ئارقىلىق تورمۇنازىرە مۇنبەرلىرىنى كوكلىتىدىغان زېمىن ئىلمى مۇنازىرە روھى ئىكەنلىكىنى ،ئۇيغۇر توردۇنياسىنىڭ  تىخىمۇ گۈللەپ-ياشنىشىنىڭ يۇرۇق نوقتىسى يەنىلا ئىلمى تور مۇھىتى ئىكەنلىكىنى يەنە بىر قىتىم چوڭقۇر ھىس قىلدىم.

يىقىنىقى يىللاردىن بۇيان ئۇيغۇر تور دۇنياسىنىڭ تەرەققىياتىنىڭ تىزلىششگەئەگىشىپ ئۇنىڭ كىشىلەرنىڭ روھى دۇنياسىغا، تۇرمۇشىغا، مەدەنىيەت ساپاسىغا، پىكىرقىلىش سەۋيەسى، ئۇسۇللىرى  قاتارلىقلارغابولغان تۇنۇشى بارغانسىرى ئوستى. ئەمما ئازساندىكى تورداشلارنىڭ توردا يازما ،ئىنكاسلارنى يىزىۋاتقاندىكى مەسئۇ لىيەت تۇيغۇسى يەنىلا ئاجىز بۇلىۋاتىدۇ. بەزىلەربۇ مۇنبەرگە نەچچە مىڭلىغان كىشىلەرنىڭ ، چالا ساۋات كىشىلەردىن باشلاپ مۇتەخەسس، ئالىملارغىچە، دولەت ئىچىدىن چەتئەللەرگىچە، ئوز مىللىتىمىزدىن قىرىنداشمىللەتلەرگىچە كىرىدىغا نلىقىنى ئويلىمايۋاتىدۇ. بەزى تورداشلار ، مىللىتىمىزدەئەزەلدىن  ئادىمىيلىك مىزانلىرىغا زىت كىلىدىغان بىر مەسىلىگە دىققەت قىلمايۋاتىدۇ. بۇ بولسىمۇ توردا كىشىلەرنىڭ شەخسىيتىگە ، ئىززەت-ھورمىتىگە دەخلى –تەرىز يەتكۈزۈش قىلمىشلىرىدىن ئىبارەت. بۇقىلمىش دولىتىمىزنىڭ قانۇنلىرىدىلا ئەمەس چەتئەل قانۇنلىرىدىمۇ چەكلەنگەن، بىزنىڭ دىنى ئىتقادىمىزغىمۇ ، ئادىمىيلىك مىزانلىرىغىمۇ پۈتۈنلەي خىلاپ . راست گەپنى دىگەندە بۇ ھەقىقى ئىلىم ئەھىلىلىرىنىڭ قىلىدىغان ئىشى ئەمەس. بۇنداق قىلغانلارئىلىم ئەھلى ھىساپلانمايدۇ ياكى تىخى ئۇ شەرتىنى ھازىرلىمىغان بۇلىدۇ. شۇڭا مىنىڭچە ئۈزىمىز بىلمەيدىغان مەسىللەرگە، ئادەملەرگە ئىھتىياتچانلىق بىلەنمۇ ئامىلە قىلشىمىز كىرەك.

مۇشۇنداق ئىللەتلەر تۈپەيلىدىن-مەلۇم شەخسلەر تەرپىدىن ئوتتۇرىغا قۇيۇلغان،جەمئىيەت كۈڭۈل بۈلۈۋاتقان مۇھىم بىر تېما مۇنبەرگە چىقىپ شۇ مەسىلە توغىرسىدىكى مۇنازىرە تىخى  چوڭقۇرلاشماي تۇرۇپلا شۇنى ئوتتۇرىغا قويغا شەخسىلەر مۇنازىرە ئۇبىكتىغا ئايلىنىپ كىتدىغان ئەھۋاللار ھىلىھەم بار. ھەتتا ئۇلارنىڭ تەق-تۇرقى، ئۇرۇق- سىمىزلىكى، كىيىم- كىچىكى،گىرىملىرىنىڭ  قىنىق-سۇسلىقى، ھەتتا كوزئەينەك تاقاشقىچە ھەممىنى مىكروسكوپ ئارقىلىق كۈزىتىدىغان، نەتىجىدە بىر ئۇبدان مۇنازىرە قىلىنىۋاتقان تېما مۇشۇنداق تۈتەك پەردە كەينىگە غەرىق بۇلىدىغان ئەھۋاللار يۈز بىرىپ تۇرىۋاتىدۇ.

ئەگەر بىز مۇشۇنداق مەسىللىلەرنى تىزراق تۈگۈتۈپ ئىلىمى تور مۇھىتىنى يىتىلدۈرەلىگەن بولساق تىخىمۇ كوپ ئىختىساسلىق كىشىلەرنى، ھەرقايسى سەپلەردە ئۇتۇق قازىنىۋاتقان ئالىملىرىمىز، مۇھەممەت سالھ داموللا ھاجىمغا ئوخشاش پىشىۋالىرىمىزنى،  بىزنىڭ روھىمىزنى يىتەكلىيەلەيدىغان ئابدۇقادىرجالالىدىن ، يالقۇن رۇزىلارغا ئوخشاش باشقا پىشقان زىيالىرىمىزنىڭ ، تۇرمۇش، خىزمەت مۇھىتىمىزغا يىقىن تۇرىدىغان رەئىس، ۋالى ، ھاكىملارنىڭ ، نۇرغۇن كەسىپ ئەھلىلىرىنىڭ تور مۇنبەرلىرىگە ئاشكارا كىرىشىنى ئىشقا ئاشۇرالىغان، ئۇلار بىلەن بولغان ئارىلىقىمىزنىڭ بارغانسىرى يىقىنلاشقانلىقىنى تىزرەك كورگەن بۇلاتتۇق. ئۇنداقتا مۇنبەرلەردە چۇقۇم تىخىمۇ بىرخىل جانلىنىش ۋەزىيتى بارلىققا كەلگەن بۇلاتتى.

بۇ قىتىمقى مۇنازىرە مەسىلىسىگە كەلسەك، مىنىڭچە ئەلى مۇئەللىم، زىۋىيدەخانىملار بىزنىڭ پەخىرلىك ئۇغۇل- قىزلىرىمىز، پىسخىكا ساھەسىنىڭ يول باشچىلىرى،ئۇلارنىڭ توھپىسى بار، ئۇلار ئوزلىرىنىڭ يۇلىنى تىپىپ ماڭالايدۇ. ئۇلارغا ئىشىنىشىمىز كىرەك. ئەمما ئۇلارنى قارا تاغلىق، ئاق تاغلىق قىلىپ قۇيساق ، ئەڭ مۇھىمى ئۇلارنىڭ شەخسىيتىگە تىگىدىغان سوزلەرنى ئىشلەتسەك بۇ ھەرگىز ئىلىم ئەھلىلىرىنىڭ ئىشى ئەمەس. مۇشۇنىڭغا ئەگىشىپ مۇنازىرىگە قاتناشقۇچىلارنىمۇ ئىككى تەرەپكە ئايرىش، ۋە ئادەمگە چۇقۇنۇشنىڭ ئۈزىمۇ بىمەنىلىك. بۇلارنى ھەرگىزمۇ ئىلمى بولغان تور مۇھىتى دىگىلى بولمايدۇ.

مىنىڭچە ھازىرقى ئەلى مۇئەللىمنىڭ بۇ مۇنازىرىگە تۇتقان پۇزۇتسيىسى،تورداشلار نىڭ يۈرەك تىۋىشىدىن قارىغاندا ئەمدى مۇنازىرلششنىڭ ئەھمىيتى يوقتەك قىلىدۇ. ئەڭ ياخشىسى بۇ مۇنازىرىنىڭ نەتىجىسىنىڭ قانداق بۇلىشىنى ۋاقىتتىن ئىبارەت ئەڭ ئادىل سوتچىنىڭ باھا بىرىشىگە قويساق بولغىدەك !چۈنكى ۋاقىت-ئەڭ توغراباھالىغۇچى،بۇ شۇنداقلا ئەڭ ئاقىلانە چارە. قىنى كۈرەرمىز، يەنە قانچە يىلدائۇلارنىڭ  قايسى بىرىگە ۋاقىتتىن ئىبارەت بۇ ئادىل سوتچى  مەخمۇت قەشقەرى، يۈسۈپ خاس ھاجىپلارنىڭكىدەك مىڭ يىللارغىچە چۈشۈپ كەتمەيدىغان تاج كىيگۈزۈپ قۇيار! ياكى
