بىلگىجاننىڭ بۇ ماقالىسىمۇ توغرىدەك قىلىدۇ...بىراق مەن يەنىلا بى تەرەپلىكنى ساقلايمەن.

چۈنكى،بىر مەرھۇم كىشىمىز بار ئىدى،95 يىشىدا قەشقەردىكى بىر مەسچىدنىڭ ئىمامىنىڭ 18 ياشلىق قىزىنى ئىلىپ بىر ئوغۇل بىر قىز بالىلىق بولۇپ ئۈچ يىل تۆت يىلدىن كىيىن مەزى بىزى سەۋەبى بىلەن ئالەمدىن ئۆتتى.ئۇ كىشىنىڭ ئالدىنقى ئايالىدىن بەش ئوغۇل،ئۈچ قىزى بار ئىدى.ھەممىسى نۇرمال،ھەممىسىنىڭلا ئۈچتىن تۆتتىن باللىرى بار.

مۇشۇ بالىلىق بولۇشقا كەلگەندە كونىلار دەپ قۇيىدىكەن،يىلىكى بوشنىڭ بالىسى بولمايدۇ دەپ.

يەنە بىرى مۇشۇ ئەرلەر ياش چوڭايغانچە جىنسى ئىقتىدارىدا چىكىنىش بولىدۇ دەيدىكەن،مىنىڭچە بۇنى ھازىرقى ھاراق تاماكا قاتارلىق كەيىپ قىلىدىغان ئىستىمال بۇيۇمى بىلەنمۇ سەۋەبلىك بولىشى مۇمكىن.ئازاتلىقتىن ئىلگىركى بۇلغانمىغان ھاۋا،تەبئىي شارائىتتىكى ئاتا بوۋىللىرىمىز شۇنداق بەستىلىك،يوغان كىشىلەركەنتۇق .
ئۆزىڭىز ياخشى كۆرگەن ئادەمنىڭ ئېللىتى خېسلەت بېلىنسە،ئۆزىڭىز يامان كۆرگەن ئادەمنىڭ خېسلىتى ئېللەت بېلىنىدۇ.

ياشلىقىدا ئالغان ئايالىدىن كىيىن يا ئايالى تۈگەپ كەتكەن يا ئايرىلىپ كەتكەن بولسىمۇ كىيىنكى ئايالىدىنمۇ كۆپ بالىلىق بولغان دادا دىمەتلىك كىشىلىرىمىز كۆپ.شۇنىڭغا قارىغاندا مۇشۇ يىمەك ئىچمەك ۋە ھەركەت كۈن نۇرى قاتارلىقلارغا تۇيۇنۇپ ياشاۋاتقان خەلىقىمىزنىڭ ياشانغاندىمۇ جىنسى تۇرمۇشتىن پۈتۈنلەي مەھرۇم بولىشى ناتايىن بىر ئىش،ئەمدى كىسىپ ئىيتىۋەتكىلى بولمايدۇ،بىزلەر 80-90-يىللاردىن كىيىن تۇغۇلغان ئەۋلاتلار ئاتا بوۋىمىزنىڭ يىشىغا شۇلاردەك زەبەردەس بارالامدۇق ياكى كىسەل كارۋىتىدا بىر ئۆمۈر يىغىققان پۇلىمىز بىلەن دوختۇرخانىدا داۋالىنىپ يىتەمدۇق،بالدۇرلا جىنسى تۇرمۇشتىن قىلىپ تۇرمۇشنى تەرىك ئىتىپ يۈرەمدۇق...ياكى مەنبەسى ئىنىق بولمىغان كىسەل ۋە مەنبەسى نا ئىنىق مۇھىتتا يوقاپ كىتەمدۇق...خۇداغا ئامانەت.

مۇشۇ ياشانغاندا تاپقان بالىنىڭ ئەقلى قابىلىيتىگە كەلسەك،بىرىنجىدىن شارائىت مەسىلىسى ئەڭ مۇھىم.ئىككىنجىدىن ئاتا ئانىنىڭ بالىغا قانچىلىك مېھىر بىرەلىشى ئەڭ مۇھىم.قالغىنىغا بالا يىتىشىپ كىتەلەيدۇ.مەن شۇنداق قارايمەن.
ئۆزىڭىز ياخشى كۆرگەن ئادەمنىڭ ئېللىتى خېسلەت بېلىنسە،ئۆزىڭىز يامان كۆرگەن ئادەمنىڭ خېسلىتى ئېللەت بېلىنىدۇ.

توۋا... رىسىق دىگەن  ...

بۇ يالغان ئەمەس.ئۇ باللارغا چوڭى ئوغۇل،كىچىكى قىز،چىرايى بەك ئوخشايدۇ بۇ باللارنىڭ بىر بىرىگە.بۇ باللار چوڭى 19 ياش،كىچىكى قىز 16 ياشقا كىردى.بۇ باللار ئىلگىركى ئۆگەي ئاپسىنىڭ 8 باللىرىغا تاغا بولدى دىگەن گەپ.بۇ باللار بىلەن بىرەر ئۇچىرشىپ تۇرىمىز.ئەقىللىقلىقىغا،ئەدەپلىكلىكىگە ئادەمنىڭ مەسىتلىكى كىلىدۇ.
ئۆزىڭىز ياخشى كۆرگەن ئادەمنىڭ ئېللىتى خېسلەت بېلىنسە،ئۆزىڭىز يامان كۆرگەن ئادەمنىڭ خېسلىتى ئېللەت بېلىنىدۇ.
