ئوقۇتقۇچىلارغا مەكتەپ فورمىسى سېتىش ھەققىدە مەخسۇس بەلگىلىمە بارمۇ ؟
تورداشلارغا ئايان بولسا كېرەك ، يېقىنقى يىللاردىن بۇيان جەميەت تەرەققىياتىنىڭ ئېھتىياجى ، دۆلىتىمىز ئۇنۋېرسال كۈچىنىڭ ئېشىشى ، ئىقتىسادنىڭ تەرەققى قىلىشى بىلەن جەميەتتىكى تۈرلىك كەسىپ ئورۇنلىردا  خىزمەت قىلىۋاتقان ،بولۇپمۇ فورما كىيىپ خىزمەت قىلىدىغان ئورۇنلاردىكى خادىملارنىڭ خىزمەت فورمىسى سۈپەت ، لاھيە -نۇسخىللىرى يىلدىن- يىلغا يېڭىلىنىپ ئۇلارنىڭ قەددى -قامىتىگە سالاپەت ،جەسۇرلۇق مۇجەسسەملىنىپ ، خىزمەتلىرگە ئاجايىپ قولايلىق بېغىشلىدى. خىزمەت فورمىسى كىيىپ خىزمەت قىلىدىغان ئورۇنلاردا چوقۇم فورما كىيىپ خىزمەت قىلمىسا نۇرغۇن ئاۋارچىلىقنىڭ كېلىپ چىقىشى ئېنىق ، قاتناش ساقچىسى فورمىسىز قاتناش باشقۇرالمايدۇ .شەھەر باشقۇرغۇچىلار فورما كىيىپ تەرتىپ ساقلىمىسا قانۇنسىز يايمىكەشلەر ئۇنىڭ گېپىنى ئاڭلىمايدۇ .ساناپ كەلسەك جەميەتنىڭ ھەرقايسى قاتلاملىردا خىزمەت فورمىسى كەيمەي شۇ خىل خىزمەتنى قىلغىلى بولمايدىغان ساھەلەر كۆپ ، بۇنى كۆپ ئادەم بىلىدۇ .ئەمدى بۇنىڭ ئەكسىچە خىزمەت فورمىسى كىيمەي خىزمەت قىلىدىغان ساھەلەرمۇ نۇرغۇن ، چۈنكى بۇ ساھە نورمال كىيىنسىلا كىشىلەرنىڭ قىلىۋاتقان ئىشىغا تەسىر قىلمايدىغان بولغاچقا ئېھتىياج شۇنى بەلگىلىگەن .
   ئەمدى گەپنىڭ نىگىزىگە كەلسەك مائارىپ ئۈچۈن فورما زۆرۇرمۇ يوق ؟ مۇشۇ تېمىدا تورداشلار بىلەن ئورتاقلىشىپ بېقىشنى ئويلاپ تۇنجى بۇ تېمىنى يوللاپ ئولتۇرۇپتىمەن .بۇ ئۆزەمنىڭ كۆرگەن ئاڭلىغانلىرغا ئاساسەن ئوتتۇرغا تاشلانغان تېما ، ئۆزەمنىڭ شەخسى قارىشىغا ۋەكىللىك قىلىدۇ ئەلۋەتتە .جايلاردىكى بىر قىسىم ھەر دەرجىلىك مەكتەپلەر ،بۇرۇندىن مەكتەپلەردە ئوقۇغۇچىلار مەكتەپ فورمىسى كىيىشنى تەشەببۇس قىلىپ ، ئوقۇغۇچىلارغا مەكتەپ تارقاتقاننى ئاڭلىغان ياكى كۆرگەن ، كىيگەن . بۇ ئەھۋال ھازىرغىچە ئومۇمىيۇزلىك ئەمەس بولۇپ ، بۇ سوقا -سۆھبەت تېمىسىغا خۇرۇج بولغىدەك چوڭ ئىشمۇ ئەمەس ئىدى . يېقىنقى يىللاردىن بۇيان بۇشامال ئۈزلىكسىز كۈچىيىپ يېزا -كەنىت باشلانغۇچ مەكتەپلىرگىچە كېڭەيدى !   بىر خىل ، رەتلىك ، چىرايلىق كىينىپ سىنىپلاردا ئولتۇرغان پەرزەنتلەنى كۆرسە ھەرقانداق ئادەمنى ھاياجان بېسىپ ئپتىخارلىق تۇيغۇسىنىڭ ئويغىنىشى ئېنىق .نۇرغۇن مەكتەپلەر مىللىي خاسلىققا ئىگە مەكتەپ ئوقۇغۇچىلار فورمىسى لاھيەلەپ بالىلارنىڭ ئۇچىسىغا چىقىرىپ ، بالىلارنىڭ شۇنداق روھلۇق ، جەڭگىۋار قىياپىتىنى ھەرقايسى تاراتقۇلاردا نامايەن قىلىپ ، كىشىلەرنىڭ كۆزىدىن ئوت چىقىرىپمۇ ئۇلگىرگەن ئىدى .ياخشى تەرپىنى ئويلىغاندا بۇ بىر كاتتا ئىش ! بۇ ئىشنىڭ پايدىسىنى ئەقلى جايىدا ھەرقانداق ئادەم ھىس قىلالايدۇ . ئەلۋەتتە مەكتەپ فورمىسىنى تىكىپ مەكتەپكە ساتقان ئادەم ئۇنى پۇلغا ساتىدۇ- دە . مەسىلە مۇشۇ يەردە !
   بىر قىسىم مەكتەپلەر بۇ مەسىلىنى مۇۋاپىق ھەل قىلدى . ئۇلار جەميەتنى ھەركەتلەندۇرۇپ ‹‹ ساخاۋەتچى›› لەرنى تەسىرلەندۈردى ، ئاتا ئانىلار بىر قىسمىنى ئۈستىگە ئالدى ، باشقا ئاماللارنى قىلدى . ھەرھالدا ئوقۇغۇچىلار رەتلىك كىينىپ مەكتەپكە كەلگەن بولدى ، پائالىيەتلەر جانلىق ، ئۇقۇغۇچىلارنىڭ مېڭىپ -تۇرۇشلىرى بۇرۇنقىدىن چەندان باشقىچە . بۇ قىياپەت رەھبەرلىكنىڭ مۇئەييەنلەشتۈرشىگە ئېرىشىپ ،مەكتەپ رەھبەرلىرىنىڭ خىزمىتى مۇشۇ ئىش تۈپەيلى ئالاھىدە گەۋدىلەنگەن بولدى .شۇنىڭ بىلەن ئۆزىنىڭ شارائىتى بىلەن ھىساپلاشماي بۇ ئىشقا قەدەم ئېلىپ ئوقۇتقۇچى، ئاتا -ئانا ۋە جەميەتكە ئىختىزادى بېسىم پەيدا قىلغانلارمۇ ئاز بولمىدى .بالا چوڭ بولىدۇ كىيىم كونىرايدۇ . بۇ ئىش ئىزچىللىقتىن قالغانمۇ بولدى . ئەمما ئوقۇغۇچىلاردىن كىيىمنىڭ پۇلىسىنى يىغىپ جايىغا كەلتۈرۈشمۇ ئاسان ئەمەس . ئەمدى نۆۋەت ئوقۇتقۇچىغا ، چۈنكى ئۇلارنىڭ مۇئاشى بار -دە . فورما شامىلى ھازىر ئوقۇتقۇچىلانى باسقىلى تۇردى . ئوقۇتقۇچىلارنىڭ خىزمەت فورمىسىنى بىرخىللاشتۇرۇش شامىلى ،دىيارىمىزدا ،ھەر دەرجىلىك مەكتەپلەرنى قاپلىدى .ھەر ئىشنىڭ نورمال بولغىنى تۈزۈك ئىدى ! بازاردا 30-40 يۇئەلىك كۆينەكلەر ئۇلارغا ھەسسىلەپ باھادا تېڭىلماقتا ، شۇنىڭ بىلەن بىر قىسىم مەكتەپنىڭ رەھبەرلىگى كىيىم سودىگىرىگە ئايلاندى . مەن بۇ يەردە مۇشۇنچىلىك كاتالىزاتورلاپ قويسام، مۇشۇ كۇندە ياشاۋاتقان ئوقۇتقۇچى تورداشلار كۆرگەنلىرنى ، ئۆز بېشىدىن كەچۈرگەنلىرنى چوقۇم ئىنكاسلىردا كەڭرى ئوتتۇردا قويدۇ ، مەن بۇنىڭغا ئىشىنىمە .
   ئەمدى سوراپ باقسام ؛ 1- دۆلەتنىڭ ،ھەر دەرجىلىك مەكتەپلەردە ئوقۇغۇچىلار ۋە ئوقۇتقۇچىلارنىڭ فورمىسىنى چوقۇم بىر خىللاشتۇرۇش ھەققىدە مەخسۇس بەلگىلىمىسى بارمۇ ؟ دۆلەتنى قانۇن بويىچە ئىدارە قىلىش تەكىتلىنىۋاتقان شارائىتتا ‹‹ ئوقۇتقۇچىلار قانۇنى›› دا بۇنىڭغا دائىر ماددا كىرگۈزۈلگەنمۇ ؟
2- بۇ ھەقتە بەلگىلىمە بار دېگەندىمۇ ئىختىسادنى كىم ئۇستىگە ئالىدۇ ؟ قانداق ئۈستىگە ئالىدۇ ؟ مەجبۇرلاش خاراكتىرنى ئالغانمۇ ؟
  3- مەخسۇس لاھىيەلەنگەن بىر ئۆلچەم بارمۇ ياكى مەكتەپ رەھبەرلىگىنىڭ خائىشىنى ئاساس قىلامدۇ ؟
  تورداشلارنىڭ مۇلاھيىزە قىلىپ بېقىشىنى سورايمەن .

مائارىپتا ئوقۇغۇچىلارغا فورما تارقىتىش ئىنتايىن زۆرۈر. ئاددىي تىلدا ئېيتساق، بۇ باي ۋە نامرات ئائىلىنىڭ بالىلىرىنىڭ باپباراۋەر ھالەتتە ئوقۇيدىغانلىقىنى ئەكس ئەتتۈرىدىغان بىر ۋاسىتە. ئەگەر شۇنداق قىلمىسا، باي ۋە نامرات ئائىلىنىڭ بالىلىرى ئارىسىدا كىيىمىنى كۆز-كۆز قىلىش ئەۋج ئالىدۇ.
ئەمما مەكتەپ فورمىسى توغرىسىدا ئىككى مەسىلە بەك ئېغىر. بىرى، مەكتەپ فورمىسىنىڭ سۈپىتى ئىنتايىن ناچار، ھەم كۆرۈنۈشىمۇ ئىنتايىن سەت. ( جەنۇب تەرەپتە قانداق فورما تارقىتىۋاتىدۇ ئۇنى بىلمەيمەن. مەن ئۈرۈمچىدىكى مەكتەپ فورمىلىرىغا قارىتا ئېيتتىم.) يەنە بىرى، باھاسى يۇقىرى، ئەمەلىيەتتە فورمىنىڭ سۈپىتى ئۇنچىۋالا پۇلغا يارىمايدۇ.
سۈپەت ناچار بولغانكىن، يا نورمال باھادا ساتقۇلۇق، يا ئالغان پۇلىغا لايىق سۈپىتىنى ئۆستۈرگۈلۈك.
بولسا شىنجاڭ رايونىدا مىللىي خاسلىق ئەكس ئەتكەن، سۈپەتلىك مەكتەپ فورمىلىرى تارقىتىلغان بولسا ئىنتايىن ياخشى بولاتتى. ياپونىيەنىڭ مەكتەپ فورمىسى ئىنتايىن يارىشىملىق. ماۋۇ جۇڭگونىڭ شۇنداق «جا» زادى.....
«يېشىل» دېسەم «ياق، بۇ قىزىل» دېدى، «قىزىل» دېسەم «ياق، بۇ كۆك» دېدى. ھەممىسى ئۆزىنىڭكىنىلا راست دېدى. شۇنىڭ بىلەن، خەق نېمىلا دېسە «ھە سېنىڭكى راست» دەپ قويۇپ، ھەق توغرىسىدا ئۆز ئالدىمغا ئىزدىنىدىغان بولدۇم.

بۇ تېمىنى يوللاپ بەك ياخشى قىپسىز.
ئون كويغا يارىمايدىغان سەت، سۈپەتسىز، لاھىيىلىنىشى شۇنداق ناچار(ئورۇق بالىلار ھەرقانچە كىچىكىنى كىيسىمۇ كىيىم بالىلارنى كىيىۋالغاندەك كۆرۈنىدۇ، سېمىز بالىلار سىرتمىسىنى ئېتەلمەي قالماسلىقى ئۈچۈن چوڭراقىنى كىيىدۇ. سىمىز بولغاچقا مەكتەپ فورمىسى شۇنداق ياراشقاندەك كۆرۈنىدۇ.)تاغاردەك كىيىملەرنى فورما ئېلىشمىساڭ سىنىپقا كىرگۈزمەيمەن دەپ بالىلارنى قورقۇتۇپ يۈرۈپ 150كويلارغا زورلاپ تاڭغاندىكىن (ھېچكىمنىڭ بۇنداق قىممەت باھادا بۇنچە ناچار نەرسىنى ئالغۇسى يوق. فورمىنى ئۆزگەرتىپتۇ، بۇرۇنقىغا قارىغاندا شۇنداق چىرايلىقكەن دەپ يالغان گەپ قىپ، فورمىنى ساتىدىغان چاغدىلا كۆرسىتىپ ئەتىگىچە پۇل ئەكىلىشمىسەڭ سىنىپقا كىرگۈزمەيمەن دەپ كېتىشلىرىچۇ تېخى. مەن  فورما تارقاتقاندا پۇل ئەكەلگىنىمگە قاتتىق پۇشايمان قىلىۋاتىمەن. بىر قانچە بالىلاردەك قېلىنلىق قىپ پۇل ئەكەلمەي ئولتۇرسام بوپتىكەنتۇق.)ئەلۋەتتە مەجبۇرلاش خاراكتىرىنى ئالىدۇدە.
بىر ئېشەكنىڭ كەينىگە كىرىۋالساڭ ئۆزىنى دۇبەينىڭ نەسىللىك ئېتى چاغلايدۇ

مەن ئىشلەۋاتقان مەكتەپتىمۇ بىر يىلدا بىر قىتىم فۇرما ئېلىپ بىرىدۇ. يەنى بارلىق ئىقدىسادنى مەكتەپ ئۈستىگە ئالىدۇ. 2014-يىلنىڭ ئۆزىدە ئايال ئوقۇتقۇچىلارغا يازلىق بىر قۇر كىيىم، ئەر-ئايال بارلىق ئوقۇتقۇچىلارغا قىشلىق فورما ئېلىپ بەردى. ئۇندىن باشقا سىنىپلارغا كىيىپ كىرىشكە ئەپلىك بولۇشى ئۈچۈن بارلىق ئەر -ئاياللارغا تەنھەرىكەت چاپىنىدىن بىرنى ، ئاياللارغا پاشنىسى پەس ئاياغ ئېلىپ بەردى. مەن مۇشۇ كەمگىچە  مەكتەپنىڭ بىزدىن پۇل يېغقىنىنى بىلمەيمەن. بولمىسا مەكتىۋىمىزدىمۇ 150دىن ئارتۇق ئوقۇتقۇچى بار. ئالدىمىزدا قىممىتى 600-700يۈەن ئارىلىقىدىكى بىر يۈرۈش مەكتەپ فورمىسى بۇيرۇتماقچى. مۇدىر ئالدىنقى ھەپتە ئوتتۇرا قاتلام رەھبەرلىرىنى يېغىپ ، بۇيرۇتماقچى بولغان كىيىمنىڭ رەڭگىنى ، سۈپىتىنى بىكىتتى. ئالدىمىزدا خىرىدار چاقىرىش ئارقىلىق زاكاس قىلىنىدىكەن.
ئوقۇغۇچىلار فورمىسىنى ئوقۇغۇچىلار ئۆزلىرى سىتىۋالىدۇ. مەيلى قايسىلا فورمىنى سېتىۋىلىش بولسۇن مەكتەپ ئىشچىلار ئۇيۇشمىسىنىڭ كونتىرۇللىقىدا بولىدۇ. يەنى ئالدى بىلەن يېغىن ئېچىلىپ ، مۇۋاپىق دەپ قارالسا ئاندىن سېتىۋالىدۇ. ئەگەر كۆپ سانلىق قارشى تۇرسا ھېلىغۇ مۇدىركەن ئۇنىڭدىن چوڭى گەپ قىلسىمۇ ئەمەلگە ئاشمايدۇ. -------------
يەنە بىر گەپ - مەكتەپتە مەكتەپ فورمىسى كىيىپ يۈرۈش، ئومۇمىي مەكتەپ قىياپىتى ،ئوقۇتقۇچى ، ئوقۇغۇچىلارنىڭ تېخىمۇ روھلۇق يۈرۈشىگە، خىزمەتلەرنىڭ ئۈنۈملۈك يۈرۈشىشىگە تولىمۇ پايدىلىق. بۇ ئەمەلىيەتتە مۇزاكىرە قىلىدىغانمۇ تېما ئەمەس.
