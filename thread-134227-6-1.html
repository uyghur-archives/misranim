بەزىدە مۇشۇ ھاۋا مېنىڭ كەيپىياتىمغا قاراپ ئۆزگىرەمدىكىن ياكى مېنىڭ كەيپىياتىم ھاۋاغا قاراپ ئۆزگىرەمدىكىن بىلمەيلا قالىمەن، مانا بۈگۈنمۇ كەيپىياتىمغا جۆر بولۇپ ئاسماننى بىردىنلا بۇلۇت قاپلاشقا باشلىدى، بۇنىڭ بىلەن كۆڭلۈم تېخىمۇ غەشلىككە تولدى، دېرىزىنى يەنە ئېچىۋەتتىم، دىمىقىمغا نەمخۇش ھاۋا ئۇرۇلۇشقا باشلىدى، ئاھ، تەبىئەت، تەبىئەتتىن ساپ نەرسە بولمىسا كېرەك، مەيلى شاھ، مەيلى گاداي بولسۇن، ھەممىسىنىڭ تەبىئەتتىن ھوزۇرلىنىش ھوقۇقى بار، بەزىدە ئالدىراشلىقىمىزدىن، ئەندىشە، غەم-قايغۇ بىلەن تەبىئەتكە كۆڭۈل قويۇپ زەن سالمايمىزكەن، ئەمىلىيەتتە تەبىئەتتىن ساپ نەرسە يوق، ھازىر ئادەملەرمۇ ساختىلاشتى، نەرسىلەرمۇ ساختىلاشتى، كۆڭۈللەرمۇ ساختىلاشتى، پەقەت تەبىئەتلا ساپ، لېكىن ئىنسانلارمۇ ئۇنىڭ ساپلىقىغا تەسىر يەتكۈزۋاتىدۇ، ئەگەر تەبىئەتمۇ ساختىلاشسا ئاسمان چوقۇم كۈندە يىغلايدۇ.مانا قاراڭ، ئۇ يىغلاشقا باشلىدى.تامچە-تامچە چۈشكەن يامغۇرلار كۆڭلۈمنى باشقىچە سۆيۈندۈردى، خۇددى يامغۇر بايامقى كۆڭۈلسىزلىكلەرنى يۇيۇپ تۈگىتىۋەتكەندەك يىنىككىنە ئۇھ تارتىۋېتىپ ئەينەككە قاراپ كۈلۈمسىرىدىم.قايتىدىن كىيىملىرىمنى تۈزۈشتۈرۈپ، قەددىمنى رۇستلاپ تاكسىرىمنى ھەيدەپ ماڭدىم.يىراقلا بىر يەردە ئىككى قىز تاكسى تورۇپ تۇراتتى، يامغۇر ئۇلارنىڭ ئۈستى-بېشىنى ھۆل قىلىۋەتكەن بولۇپ، ئىچىم ئاغرىپ سۈرئەتنى تېزلىتىپ ئۇلارنىڭ ئالدىدا توختىدىم.-ئەسسىلامۇ ئەلەيكۇم، ئاكا بىزنى خەلق دوختۇرخانىنىڭ ئالدىغا ئاپىرىپ قويۇڭە، يۈزى دوغىلاق كەلگەن ئوماق بىر قىز ئىللىق كۈلۈمسىرىدى.
-بولىدۇ. ئۇلار ئۈستى-بېشىنى تۈزەشتۈرۈپ، پاراڭغا چۈشۈپ كەتتى.
-ئاداش، بۈگۈن ساخاۋەت ئەترىتىمىز يوقلايدىغان كىشى نېم بولغان كىشىكەن بىلدىڭمۇ؟
-ھە، ئاداش، ئۇ ئادەم جىگەر ياللۇغىغا گىرىپتار بولغان كىشىكەن، بالىلىرىمۇ يوقكەن، ئايالى بۇرۇنلا ئۆلۈپ كېتىپتۇ، بەك بىچارە دېگىنە
-ۋاي خۇدايىم، ئاللاھ ئۇنىڭغا شىپالىق بەرسۇن، دادامغا بىرسىنى يوقلىغىلى بارىمەن دېسەم300يۈەن بەردى.
-ماڭىمۇ دادام300يۈەن بەردى.
-پۇلىمىز جىق بولغان بولسا جىقراق ياردەم قىلارمىزكەن.
-شۇنى دېگىنە، كىيىن مەكتەپ پۈتتۈرۈپ ياخشىراق خىزمەت قىلىپ قالساق چوقۇم تېخىمۇ كۆپ ئادەملەرگە ياردەم قىلايلى
-ماقۇل ئاداش.
بۇ گەپلەرنى ئاڭلاپ ۋۇجۇدۇم ئاجايىپ سۆيۈنۈپ كەتتى، ئۇلارغا قايتىدىن سەپ سالدىم، ئۇلار كىچىك پەرىشتىلەردەك بىر-بىرىگە كۈلۈمسىرەپ تۇراتتى، ئۇلارنىڭ شۇ سەمىمىي قەلبى چىرايىغا باشقىچە نۇر ئاتا قىلغان بولۇپ، بۇ نۇر مېنىڭمۇ قەلبىمنىمۇ يورۇتىۋەتتى. ماشىنىنى شۇنچىلىك سىلىق ھەيدىدىم، گەپدانلىقىم نەگە يوقالدىنكىن تاڭ، ئۇلارغا سۆز قىلالمىدىم، ياق سۆزلەشنى مۇۋاپىق كۆرمىدىم، پەقەت جىمجىت تۇرۇپ ئۇلار ئەكەلگەن مۇشۇ ساپلىقتىن ھوزۇر ئېلىشنى مۇۋاپىق كۆردۈم، ئۇلارنىڭ باشقا گەپلەرنى قىلماسلىقىنىمۇ ئۈمىد قىلدىم، ماڭا بەرگەن مۇشۇ ئوبرازىغا خۇنۇك يېتىپ قېلىشتىن، بۇ ساختىلىققا تولغان جەمئىيەتتە ئاران كۆرگەن بۇ ساپ قەلبلەرنىڭ ساختا چىقىپ قېلىشىدىن ئەنسىرىدىم، ئۇلارمۇ باشقا گەپ قىلىشمىدى، دوختۇرخانىنىڭ ئالدىغا كەلگەندە ئۇلار چۈشۈپ قالدى.
-نەچچە بېرىمىز ئاكا؟
-بولدى، ئۇكىلىرىم، ساخاۋىتىڭلارغا قىلغان ياردىمىم بولسۇن، مانا بۇ200يۈەننى ئالغاچ بېرىڭلار.ئۇلار كۆزلىرىگە ياش ئالدى.
-رەھمەت ئاكا، مەن بۇنى چوقۇم دوستلۇرۇمغا سۆزلەپ بېرىمەن، تېخىمۇ كۆپ كىشىلەرنى قېرىنداشلىرىمىزغا ياردەم قىلىشقا سەپەرۋەر قىلىمەن.
-بولىدۇ، ئۇكام. سىلەر بېرىڭلار، مەن ماڭاي.
-ماقۇل، ئاق يول بولسۇن، ئۇلار قوللىرىنى پۇلاڭلىتىپ قالدى.ئۇلار ئۈمىدسىزلەنگەن قەلبىمنى ئازراق بولسىمۇ ئۈمىدلەندۈرگەنىدى.ئۇلارغا قاراپ خۇش بولدۇم، ھەممە قىزلىرىمىز شۇنداق بولسا، مىللىتىمىزنىڭ كەلگۈسى چوقۇم پارلاق بولىدۇ، ئۇلارغا چىن كۆڭلۈمدىن بەخت تىلىدىم، كەلگۈسىدە ئوغلۇمغا شۇنداق ساپ قەلب ئىگىسىنى ئۇچراشتۇرۇشنى ئاللاھتىن تىلەپ يولۇمنى داۋاملاشتۇردۇم.ھاۋا ئېچىلىپ قالغانىدى.

چۈشتىن بۇرۇنقى كەيپىياتىمنى بۇزماسلىق ئۈچۈن، ئۇدۇل ئۆيگە قايتتىم.ئۆيدە ئايالىم ئىسسىق پولۇ ئېتىپ قويۇپتۇ، ئايالىمنى مېھىرلىك قۇچاغلاپ قويۇپ، ئوغلۇمنى قولۇمغا ئالدى.ئايالىم مېنىڭ تۇيۇقسىز قىلغان يىقىنچىلىقىمدىن ھەيران قالدى بولغاي قاراپلا قالدى، ئۇم چۈشتىن بۇرۇن قانداق ئادەملەرنى ئۇچراتقىنىمنى بىلمەيدۇ دە، بىلسە ئىدى، ئۇنىڭدىن قانچىلىك سۆيۈنگىنىمنى، شۇنچىلىك ئايالىمنىڭ بولغانلىقىدىن پەخرلەنگىنىمنى ھېس قىلاتتى. ئوغلۇمنى قولۇمغا ئېلىپ يېنىش-يېنىشلاپ سۆيۈپ كەتتىم، چوقۇم ئۇنى ياخشى تەربىيىلەپ ياراملىق ئەركەك قىلىشقا ئەھدى قىلدىم.پولۇ شۇنچىلىك يېيىشلىك ئىدى، ئايالىمغا رەھمەت ئېيتىپ، دۇئا قىلىپ شېرىن ئۇيقۇغا كەتتىم.چۈشۈمدە مەنمۇ قىز بالىلىق بولۇپتىمەن، ئۇ دادا دەپ ماڭا كۈلۈمسىرەپ ئېسىلىپتۇ، ئاپئاق كۆينەك كىيۋالغان بولۇپ خۇددى كىچىك پەرىزاتتەك يىراقتىن يۈگۈرۈپ كېلىۋاتاتتى.«ھۈ،ھۈ،ھۈ» شېرىن چۈشۈمنى ئوغلۇمنىڭ يىغىسى بۇزىۋەتتى.ئۇنى بىردەم بەزلەپ، يۈزلىرىمنى يۇيۇپ، يەنە يولغا چىقتىم.چۈشتىن كېيىن قانداق ئىشلارغا يولۇقىمەن، بۇ تېخى بىر سىر ئىدى، مۇشۇ خىزمىتىمنى بەك ياخشى كۆرىمەن، چۈنكى ئۇ ماڭا نۇرغۇن ھىكايىلەرنى سۆزلەپ بېرىدۇ، نۇرغۇن ئادەملەر بىلەن ئۇچراشتۇرىدۇ، نۇرغۇن ھېس-تۇيغۇلارغا باشلايدۇ، نۇرغۇن ھەقىقەتنى ھېس قىلدۇرىدۇ. بەزى كىشىلەر بار، شۇنداق يۈز-ئابرويلۇق خىزمەتلەرنى قىلىدۇ، لېكىن شۇنداق كۆڭۈلسىز، خىزمىتىدىن خۇشاللىق تاپالمايدۇ، ئەتىگەن ئىشقا بارسا، كەچنىڭ بولۇشىنى، دەم ئېلىشنىڭ بولۇشىنى كۈتۈپلا يۈرىدۇ،مەن بولسام قاچان ئىشقا چىقارمەن، كىملەرگە، قانداق يېڭى ئىشلارغا يولۇقارمەن دەپلا يۈرۈيمەن، شۇڭا خىزمىتىمدىن ئەزەلدىن ۋايساپ باقمىدىم، مەيلى جەمئىيەتتىكى ئورنۇم يۇقىرى ياكى تۆۋەن بولسۇن، ئۇ ماڭا خۇشاللىق ئېلىپ كېلەلىدى، شۇڭا خېلى كىشىلەرگە قارىغاندا ئۆزۈمنى بەختلىك ھېس قىلىمەن.شۇ خىياللار بىلەن مېڭىۋاتسام، بىر قىز-يىگىت تاكسى تورىدى، ھەممىسى ياش بالىلار بولۇپ، كىيىنىشلىرىدىن زامانغا لايىق «غەيرىي ئېقىم»باللىلىرى ئىكەنلىكى كۆرۈنۈپ تۇراتتى.قېنى بۇلارنىمۇ سەپ باققۇم كەلدى، تاكسىنى يېنىغا ئاپىرىپ توختاتتىم. بۇ يازمىنى ئاخىرىدا   Girl تەھرىرلىگەن. ۋاقتى  2015-1-27 12:17 PM

ئۇلار قولتۇقلىشىپ تاكسىغا چىقتى.
-ئاكا، بىزنى خەلقئارا مېھمانسارىيىغا ئاپىرىپ قويسىڭىز بوپتىكەن.ياپىرىم، ئۇلارغا يەنە بىر رەت سەپ سالدىم، تولۇق ئوتتۇرا مەكتەپ ئوقۇغۇچىلىرىدەكلا قىلاتتى، كۈپكۈندۈزدە يەنە كېلىپ ياتاققا قاراپ مېڭىپتۇ، ئۇلارغا قاراپ بېشىمنى چايقاپ قويدۇم.خەلقئارا مېھمانسارىيىغا ماڭغىنىغا قارىغاندا خېلى پۇللۇق ئائىلىلەرنىڭ بالىلىرىدەك قىلاتتى،ئېسىت، بالىلىرىغا پۇل بېرىشنى بىلىپ، ئادەم بولۇشنى ئۆگەتمىگەن ئاتا-ئانىلار ئوخشايدۇ، ئىچىم ئاچچىق بولۇپ قالدى، ئۇ قىزغا سەپ سالدىم، ئۇنىڭدا يا خىجىللىق يا ئوڭايسىزلىق ئالامەتلىرى يوق ئىدى. چۈشتە كۆرگەن چۈشۈمنى ئەسلىدىم، بەلكىم بۇ قىزمۇ كىچىك ۋاقتىدا شۇنداق سەبىي،ئوماق، پەرىشتىدەك بولغۇيتى ھە، بۇنىڭ بۇ ئەھۋالغا چۈشۈپ قېلىشغا كىمنىڭ سەۋەپچى ئىكەنلىكىنى بىلمىدىم، لېكىن كۆڭلۈم بەك يېرىم بولدى.لېكىن ئۇنىڭغا ئىچ ئاغرىتىش تۇيغۇمدىن ۋاز كەچتىم.چۈنكى ئەمدى بۇ ئىككى ھايۋان تاكسىنىڭ ئىچىدىلا ئاغزى-بۇرۇنلىرىنى يالىشىپ ئولتۇراتتى.بۇنى كۆرۈپ ئاران تەڭشەلگەن كەيپىياتىم نۆلگە چۈشۈپ قالدى.تاكسىنى دەرھال توختاتتىم.
-ئۇكام، ھەي، مېھمانخانىغا تېخى كەلمىدىڭ، بۇ سېنىڭ ياتىقىڭ ئەمەس.
-نېم كارىڭىز ئاكا، پۇل بەرگەندىن كېيىن تاكسىرىڭىزنى ھەيدەڭە، كىم سىزنى بىزگە قاراڭ دېدى بىزگە.قىزنىڭ ئاغزى ئېچىلىپ بىلجىرلاپلا كەتتى.ئۇنىڭ بۇ گەپلىرىنى ئاڭلاپ بايام ئۇنىڭغا ھېسداشلىق قىلىپ كەتكىنىمگە ئېچىندىم.
-قىزبالا كەنسىز، ئارتۇق گەپ قىلماي، ئۇيالسىڭىز بولمامدۇ، بۇنداق قىلىپ يۈرسىڭىز، ئاتا-ئانىڭىزغا يۈز كېلەلەمسىز، تېخى كىچىككەنسىز.
