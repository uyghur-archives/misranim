تېلېفۇنغىلا تۇغۇلغانمىتۇق(سانائىل ھەرىپلەردىن تۈزۈلگەن سۆز ئويۇنى)


توردىكى تۈرلۈك تىمىلارنى تەستىقلىغۇچى تەجىربىلىك تەھرىرلەر،تورغا تەستىقلانغان تىمىلارنى تەھقىقلاپ،تۈرلۈك تەكلىپلەرنى تاشلايدىغان تۈمەنلىگەن تورداشلىرىم:تۆۋەندىكى تىمامدا تەكىتلەيدىغىنىم تېلېفۇن توغرۇلۇق تەكلىپتۇر.
توغرا، تېلېفۇن  تەسەۋۋۇرىمىزدىكى تۈرلۈك تاتلىق تىلەكلىرىمىزنى، تىۋىشلىرىمىزنى،تۈرلۈك تاپشۇرمىللىرىمىزنى،توغرىسى، تىجارەت،تۈرلۈك تالاش-تارتىش،تاپا-تەنە،تۈرلۈك تەسىراتلىرىمىزنى، تۇنۇشلىرىمىزغا،تۇققانلىرىمىزغا،تۈرلۈك تەشكىلاتلارغا تىل تاۋۇشلىرىمىزغا تايىنىپ تەككۈزۈشكە تەييارلانغان،تولىمۇتېخنىكلىق تاۋاردۇر.تەكىتلەنگەندەك تۈرلۈك تاپشۇرمىلارنى تىلېفۇندا تەككۈزىمىز.تېلېفۇننى تەپسىلى تۇنۇشتۇرمىساممۇ تېلېفۇن تونۇشلۇق تۈر.
تەۋەيىمدىن تارتىپ،تۈرلۈك تەۋەلىكلەرگىچە توققۇزغا توشمىغانلاردىن تارتىپ تاكى توقسانغا تاقاشقانلارغىچە تېلېفۇن تۇتىدۇ.تېلېفۇن تۇتىدىغانلىرىمىز تولا.تېلېفۇن تويغۇ-تىۋىشلىرىمىزنى تەككۈزۈش تېخنىكىسىدىن تاشقىرى تۈرلۈك تۈرلەر تەڭشىلىپ،تېلېفۇننىڭ تۈرلۈك تېخنىكلىق تەلەپلىرى تولۇقلىنىۋاتىدۇ.تېلېفۇندىكى تۈرلۈك تېخنىكىلارغا تايانغىنىمىز توغرا.
تەئەججۇپلىنەرلىكى:تىلەكلىرىمىنى،تويغۇمىزنى تېلېفۇنغا تايىنىپ تەككۈزۈشتىن تاشقىرى،تېلېفۇن تۇتىدىغانلارنىڭ تەڭدىن تولىسى،تېلېفۇنغاتەشنالىشىپ،تېلېفۇنغا تىكىلىپ،تېلېفۇندىكى تۈرلۈك-تۈمەن تېخنىكىلىق تۈرلەرگە تاشلىنىپ،تۇرمۇشتىكى تۈرلۈك تەلەپلەرنى تاشلىۋىتىۋاتىدۇ.
تېخىمۇ تەپسىلىرەك توختالسام.تېلېفۇن تۇتىدىغانلارنىڭ تولىسى،تاكسىدا تۇرۇپمۇ تېلېفۇنغا تىكىلىۋاتقان،تاكسىدا تېلفۇنغا تىكىلىۋالسىڭىز تۇيۇقسىز تۈرلۈك تالاپەتلەرنى تارتماقسىز؟تېخى تولىسى تېلېۋىزورنى تەڭشۋېلىپ تۇرۇپمۇ،تېلېۋىزورغا تىكىلمەي،تېلېفۇنغا تىكىلىدىكەن.تېلېفۇندىكى تۈرلۈك تاماشالار تېلېۋىزوردىمۇ تەييارغۇ؟تېخى تۇرالغۇڭىزدا تەخسىگە تاماق تەييارلىنىپ تۇرسىمۇ،تاماققا تىلىڭىزنى تەككۈزمەي تۇرۇپ،تېلېفۇننى تۇتۇۋالسىز،تېلېفۇننىڭ تەشنالىقىدا تەخسىدىكى تاماقلىرىڭىزنىڭ تۇز تەمى تۈگەپ،تارشىدەك تاراڭشىغاندا تېتىشقا تاۋىڭىز تۈگەپ،تاماقنى تۆكىۋىتىسىز.توۋا تاماقنى تاشلىۋىتىش توغرىمۇ؟تېلېفۇن تاماقتىنمۇ تېتىملىقمۇ؟تېخى تۈندە تۆشەككە تاشلىنىپ تۇرۇپمۇ تەنلىرىمىزنى تىنىچلاندۇرماي تېلېفۇننى تىرىكشىتىدىكەنمىز.توۋا..توۋا...
تېلېفۇنغا تەشنالىشىپ، تۇرمۇشتىكى تاشلىۋىتىلىپ،تۈگۈمەيۋاتقان تەييارلاشقا،تۈگۈتۈشكە،تېپىشقا تىگىشلىك تۈرلەر تولا.تۇرمۇشتىغۇ تۈرلۈك تاپشۇرمىلارنىغۇ  تەييارلىماڭ.تىنىمىز تىك تۇرسىلا تەييارلىۋالارمىز.تەشۋىشلىكى:تېلېفۇندا تېلېفۇن تورىغا تەشنالارمۇ تولىكەن.تورغۇ تۇرمۇشتىكى تۈرلۈك تەلەپلىرىمىزنى توغرىلايدىغان تۈر.تەشۋىشلىكى،تېلېفۇن تۇتىدىغانلارغا تورلاشقاندا تەسەۋۋۇرى تۇيماي،تۈرلۈك تۇيۇق،تالاپەتلەرنى تارتىۋېلىۋاتىدۇ.تەكىتلىسەم،تۈرلۈك تۇترۇقسىز تىلەكلەر،توغرىلاشمىغان تەشەببۇسلارنى تويماي،تورنى تەكشۈرگۈچىلەر تەرپىدىن تۇتۇلۇپ،تۇرمىگىمۇ تاشلىنىۋاتىدۇ.
تەكىتلەيدىغىنىم:تېلېفۇنغىلا تەشنالاشماي،تۇرمۇشتىكى تۈرلۈك تۈمەن،تەييارلاشقا تىگىشلىك تاپشۇرۇقلارنى تەييارلاشقا تۇتۇنىڭلار...تۇرمۇشتا تېلېفۇنغا تايانغىنىمىزغۇ توغرا.تايانغاندىمۇ توغرا تەرتىپتە تايىنايلى!تېلېفۇنغىلا تايىنىپ،تۇرمۇشنى تاشلىۋەتمەيلى!تىنىمىز توختاۋسىز تىك تۇرۇۋەرمەيدىغۇ؟تېلېفۇنغىلا تۇغۇلغانمىتۇق-يە؟

   قاچانلا تېما يوللىماڭ تەپسىلى كۆڭۈل قويۇپ ئوقۇپ قالىمەن . ھەقىقەتەن تالانتىڭىز بار ئىكەن .  مەن ئا نەچچە ئاي بۇرۇن بىر تېمىڭىزنى كۆرۈپ مەشىق ئۈچۈن ئاز-تولا يېزىپ باقاي دەپ يىرىم كۈن ھەپىلىشىپ ئاران ئون قۇردەك بىر نىمە يىزىپتىمەن . قايتا ئوقۇسام خاتالىق شۇنداق ئىغىر دەيسىز   

   توشقانننىڭ تۈلكىنىڭ  ئىغىزىدىن ئەپچىللىك بىلەن قوتۇلىش جەريانىنى باش ھەرىپى قانداق ئىسىم بىلەن باشلانسا مەيلى بىر يىزىپ چىقىپ بىقىڭە بۇرادەر  . بۇ سىزنى خىجىللققا قويشۇش مەقسىىتىدە ئەمەس . مۇشۇ جەراياننى ئىپادىلەشتە ئىشلىتىدىغان ئەدەبىي سۆزلەرنىڭ ھەر بىرنى قايسى سۆزگە ئالماشتۇرۇپ ئىشلەتكەندە ماقالنىڭ مەركىزى ئىدىيسىنى بۇزماي ئىپادىلگىلى بولىدۇ بىر كۆرگۇم بار . سىزنى كەسىپ ئەسلى كۆرۈپ دەۋاتىمەن .

توخو   تۇيۇقسىز تۇلكىنىڭ تىكىلىپ تۇرغىنىدىن  توۋلىۋەتتى...
تۇلگە؛ توۋلىساڭمۇ  توۋلىمىساڭمۇ تۇتىۋۋىلىپ تازا تەمىڭنى تىتىمىسام ،
توخو  ؛تۇلكىنىڭ تۇرىقىدىن تۇرلۇك تالاپەتلەنىڭ تۇرغانلىقىنى  تۇرمۇشتىكى تەجىربىگە تاينىپ  تۇلكىگە تاقابىل تۇرۇشنى توغرا تاپتى
توختا؛تىپىشماق تىپىشايلى  تاپالمىسام تەلىۋەڭگە  تەييار تۇرىمەن.تۇلكىمۇ  تۇخونىڭ  تەلىۋىنى تۇغرا تاپتى .
تۇلكە ؛ توخو تۇخۇمنى  تۇغامدۇ تۇخۇم  توخونى تۇغامدۇ ؟
توخو ؛ توخو تۇخۇم تۇغىدۇ توخوسىز تۇخۇم تاپالمايسەن.
تۇلكە؛ توختا توخومسىز تۇغۇلغانمىتىڭ ؟
توخو ؛توغرا تۇخۇمسىز تۇغۇلمىغان توخومۇ توخوسىز تۇغۇلمىغان
تۇرلۇك  تالاش تارىشلار تۇپەيلى تۇلكە توختىدى .
توخو  ؛ تۇلكىدىن تۇخۇ تەملىكمۇ ؟ تۇلكە توغرا تەملىك .
توخو ؛ تىلفۇندىكى توخونى تەڭلەپ  تەملىكمىكەن  ؟ تۇلكە  
توخو ؛ تىپىشماق تاپالمىغىنىڭغا تىخى توخو  تۇتماقچىما   تۇفۇي
توخو  تۇلكىىنىڭ تىپىشمىقىنى تاپالمىغىنىدىن  تۇرمۇشتتىكى  تەجىربىىلەرگە  تايىنىپ  تۇلكىگە تاقابىل تۇرۇشنىڭ توغورلىقىدىن تۇلكىنىڭ تۇزىقىغا  تۇتۇلمىغىنىدىن تەئەججۇپلىنىپ تۇلپاردەك تىكىۋەتتى

تەسىرلىك تىما توقويالمىساممۇ  تىرىقچە تۆھپە تەيارلىدىم  توختاۋسىز تىرىشىپ تولىمۇ تەسىرلىك تىمىلارنى  تەيارلاشقا تىرىشىمەن
