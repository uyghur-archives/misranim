تۈلكىنىڭ توشقان تۇتۇشى(سانائىل ھەرىپلەردىن تۈزۈلگەن سۆز ئويۇنى)

تورداشلار تۈۋەندىكى تىمام «تۈلكىنىڭ توشقان تۇتۇش»ى توغرۇلۇق تىمىدۇر.

تەكىتلىمىسەممۇ تۇنۇشلۇق.تۈلكە تۇلىمۇ تەدان.تۈلكىنىڭ تەدانلىقىدىن تۈلكىنىڭ توزاق تۈرلىرىنىڭ تولىلىقىدىن تۈلكىدىن تۆۋەنرەك تۇرىدىغان تائىپىلەر تۇيمايلا تۈلىكىنىڭ توزىقىغا تاشلىنىپ تۈگۈشىدۇ.

تەۋەيىمىز تۆۋەنجاينىڭ تۆۋەن تەرپىدىكى تاشلىقنىڭ تەسكەي تەرىپى تۈپتىن تاغلىق.تاغلىقنىڭ تۆت تەرپىنى تىرەكلەر توراپ تۇرىدۇ.تاغلىقنىڭ تۆپىسىدە تۈلكىلەر توپلىشىپ تۇرىدۇ.تۈلكىنىڭ تولىلىقىغا تۇشلۇق،تۈلكىنىڭ تامىقىغا تەييار تۇرىدىغان توشقانلارمۇ تولا.تەڭرىم تۈلكىنىڭ تامىقىغا توشقاننىلا تەييارلىغانمىكىن تاڭ.تۈلكىلەر تاڭدىن تاكى تۈنتەڭگىچە توختىماي توشقان تۇتۇپ،توشقانلارنى تەشۋىشلەندۈرۈپ،توشقانلارنىمۇ تۈگەشتۇرۋاتىدۇ.تۈلكە توشقانلارنى تۇتۇۋەرگەنسىرى تاغلىقتا توشقانلارمۇ تۈگەپ،تاغلىقنىڭ تاقىرراق تەرەپتىكى توپىلىق تارچوقتا تۆت توشقاننىلا تۈلكىلەر تاپالمىغانلىقتىن تۇتالماي توشقانلار تىرىك تۇرۇۋاتاتتى.تەدان تۈلكىلەرتوشقانلارتۈگەپ،تاماق تەييارلىيالماي تەنلىرى تارتىشىپ،تاپتەك تاشلىنىپ تۈگۈشىۋاتاتتى.تۈلكىلەر توشقان تۇتمىسا تاغدا تۈلكىلەرنىڭ توشقاندىن تاشقىرى تۇتىدىغان تائىپىلەر تېپىلمايدۇ.تۈلكە تۇتسا توشقان،توخۇ تىپىدىكىلەرنىلا تۇتالايدۇ.توڭگۇزدەك تىمەن تائىپىلەرنىغۇ تۇتالمايدۇ.تۇتۇشقا تەييارلانسا توڭگۇز تۈلكىلەرنى تۈتۈپ،تىپىپ،تىرىسىنى تىتىپ،تۈلكىنى تۇغۇلغىنىغا تويغۇزۋىتىدۇ.تۈلكە تائىپىسىدىلەرغۇ تۈرقىدىنلا تۈلكىدىن تىمەن تىتىك تۇرغاچقا تۈلكىگە تۇتۇلمىغاندىن تاشقىرى  تېخى تۈلكىنى تۇتۇۋالىدۇ.تۈلكىلەر تاماقسىز تاماقلىرى تارتىشىۋەرگەچكە تاشلىنىپ تۇرۇۋېىرىشنى توغراتاپماي،تاماق تېپىشقا تەييارلاندى.تۈلكىلەر توختىماي تاشلىقنى،توپىلارنى تىلغاپمۇ تاماق تاپالمىدى.توختىماي تاغلىقنى تمىسقلاۋىرىپ،تۇيۇقسىز تىرىك تۇرۇۋاتقان تۆت توشقاننىڭ تۇرالغۇسىنى تۇيۇپ تېپىۋالدى.توشقانلار تىپىرلىغىنىچە توپلىقنى تىشىپ ،تۆپىسىگە تاياق تىزىپ،توپىدا تىندۇرۇپ تەييارلىۋالغان تارغىنە تۆشەكلىرىگە تىقىنىۋېلىشتى.تۈلكىلەر توپىنى تىماقداپ ،تاتلاپ توشقانلارنى تۇتۇشقا تەييارلاندى.تۈپىنىڭ تېگىدىكى توشقانلارمۇ تۈلكىدىن تەدانراقمىكىن-تاڭ،توشقانلار تۆشەك تەييارلىغاندا،توپراقنىڭ  تېگىنى تەشكەندە  تۆشەكلىرىنى تاغنىڭ تۆتلا تەرپىگە تۇتاشتۇرۇپ تەييارلىۋالغانكەن.تۈلكىلەر تۆشەكنىڭ تۆپىسىدىكى توپىلارنى تاتلاۋاتقاندا توشقانلار تەڭلا تەرەپ-تەرەپكە تارقىلۋېلىشتى.تۈلكىلەر توپىنى تاتلاپ تۆشەكنىڭ تېگىنى تاپقاندا توشقان تۇرماق،توشقاننىڭ تۈكىنىمۇ تاپالمىدى.تۈلكىلەر تەئەججۈپلەندى.توشقانلارنى تاپالمىغانسىرى تەلۋىلىشىپ، تىتىلداشتى.تۇيۇقسىز تۈلكلەر  تۇرغان تەرەپتىكى  تاشلىق تىرىقشىۋىدى تۈلكىلەر تەڭلا تاشقا تاشلاندى.تىمامنى تەتۈرسىگىرەك تولغىسام،تۈلكىلەر توشقانلارنىڭ تۆشەكلىرىنى تاتلاۋاتقاندا توشقانلار توپىدىكى تېخى تۇغۇلۇپ تىنى توشمىغان،تەجىربىسىز توشقانچاق،تۈلكىنىڭ تالاپىتىنى تويماستىن،تاغلىقنىڭ تۆۋەن،تارچۇق تەرەپلىرىگە تاشلانماي،تەمتىرەپ،تۆشەكنىڭ تۈۋىدىكى تاشلىققىلا تاشلىنىۋالغانلىقتىن،توشقانلارنىڭ توپىنى تاپالماي،توپىنى تىلغاۋاتقاندا تۈپىنىڭ تۈۋىدىكى تۈگمىدەك تاشلار تىرىقشىغاچقا تۈلكىلەر توشقانچاقنى تۇيۇۋالدى.تېخى تىنى تولمىغان توشقانچاقمۇ تەدانكەن.تۈلكىلەرنىڭ تاشلىنىشىدىن تەمتىرسىمۇ،تۇيۇقسىز تىتىكلىشىپ،تاشلىق تۈۋىدىكى  تۈگىشىشكىلا تەييارلىنىۋاتقان،تاراشماق تىرەكنىڭ تۈشۈكىگە تىقىنىۋالدى.تۈلكىلەر توشقاننى تۇتماققا تۇمشۇقلىرىنى تىرەكنىڭ تۆشۈكىگە تىقىشتى.تۈلكىلەرنىڭ توم تومشۇقلىرى تىرەكنىڭ تۈشۈكچىسىگە تاقىشىپ تۇرۇۋلىپ،توشقاننى تۇتالمىدى.تۈلكىلەر تۈن-تەڭگىچە تەرلەپ-تەپچىرەپ،تىرەك تۈشۈكنى تىتىپ،تىلغاشتى.تاتلا-تاتلا تار تۆشۈكنى تىتىپ تۆشۈككە تۇمشۇقلىرى تىقىلغۇدەك تەييارلاشتى.تۈلكىلەر تۇمشۇقلىرىنى تۈشۈككە تەخىرسىز تىقىشىۋىدى،تامشىتقان تىللىرى توشقانغا تەگمەستىن،تىللىرىغا تەمسىز،تاش-توپىلار تەگدى.توۋا توشقانچاق تۆشۈككە تىقىلۋالغانتىغۇ؟توشقانچاق تۇپراقنىڭ تېگىدىمىدۇ؟تۆۋەندە تىمامنى تىمامنىڭ تۆپىسىدىكى تۈلكىلەر تىرەك تۆشۈكىنى تاتلاۋاتقان تەرەپكىرەك تاشلىسام،تۈلكىلەر تۆشۈكنى تاتلاۋاتقاندا توشقانچاق تىتىرگىنچە تىرەككە تاقىشىۋالغاندا تۇيۇقسىز تىرەكمۇ تەڭلا تىترىدى.توشقانچاق تۇيۇقسىز تىرەكنى تاتلىۋىدى تۈگۈشىشكە تەييارلا تۇرغان تىرەكنىڭ تاسماقلىرى تۆۋەنگە تۆكۈلدى.توشقانچاق تاتلىغانسرى تاسماقلار تۆۋەنگە تاشلىنىپ،تىرەكتىن تۆپىسىگىچە تاسماقچىلاپ تۆشۈك تەييارلاندى،توشقانچاق تىرىشىپ-تامىشپ تىماقلىرىغا تايىنىپ تىرەكنىڭ تۆپىسىگە تاشلىنىۋالدى.تىرەكنىڭ تۆپىسىنى تىرەكنىڭ تاللىرى  تۇسۇۋالغاچقا تۈلكىلەر تىرەكنىڭ تۆۋىنىدە تۇرۇپ،توشقانچاقنى تويمىدى.توشقانچاقنى تۇتالمىغان تۈلكىلەر تۇرۇۋىرىپ تىتىلداپ،تاماقلىرىنى تامشىتقىنىچە تۇرالغۇسى تەرەپكە تىۋىندى.تۈلكىلەر  تارقاپ تامام تۈگىگەندە توشقانچاق تىرەكتىن-تىرەككە،تاشتىن-تاشقا تاقلاپ،توشقانلار توپىنى تېپىۋالدى.

تىماممۇ تۈگىدى.توشقاننى تۇتالمىغان تۈلكىلەر تاغدىن تارقاشتى.توشقانلار تۈلكىنىڭ تەھدىتىدىن تىتىرمەستىن تۇرالغۇسىدا تىپتىنىچ تۇرۇۋەردى.
