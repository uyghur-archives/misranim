ئادەتتىكى تىرانسىفورماتورلۇق توك مەنبەنىڭ ھەجىمى چوڭ، ئېغىر ، يۇقىرى-تۈۋەن بىسىملىق ئۆزگىرىشلېرىگە ماسلىشىشى ناچار بولغاچقا مۇشۇ 20يىللاردىن باشلاپلا دۆلىتىمىزدە تىلۋىزور قاتارلىق ئائىلە ئىلىكتىر ئۈسكىنىلېرىدە 70~260 ۋولىتقىچە دائىرىدە نورمال ئىشلىيەلەيدىغان يۇقىرى چاستوتىلىق توك مەنبەلەر بارلىققا كەلدى، تەرەققى قىلغان ئەللەردە بولسا 1960-يىللېرىدا ھازرقى ئۈنىمى خېلى يۇقىرى ھىساپلىنىدىغان توك مەنبەلەر ئىشلىتىلىپ كەلگەن.
  بۇ خىل توك مەنبەلەرنىڭ ئارتۇقچىلىقى ھەجىمى كىچىك، قۇۋۋىتى چوڭ، يىنىك، تەننەرخى تۈۋەن، يەنە توك بىسىمىنىڭ يۇقىرى تۈۋەن بىسىملىق ئۆزگىرىشكە ماسلىشىشچانلىقى يۇقىرى. تەۋرىنىش دىگەندە توك مەنبەسى مىخاينىكىلىق تەۋرەنمەيدۇ، توك مەنبەسىنى توك ئۈتىدىغان ئورام سىمغا ئۈتكۈزۈش ۋە ئۈزۈش ئۈزلۈكسىز داۋاملىشىدۇ، تەۋرىنىش چاستوتىسىمۇ نەچچە يۈز گىرتىستىن نەچچە يۈز مىڭ گىرتىسقىچە بۇلىدۇ.
  تۈۋەندىكىسى 6ۋولىت ياكى 12 ۋولىتلىق توك مەنبەدە كۈن نۇرلۇق لامپا يورىدىغان ئاددى ئىسخىما. باتارىيەنىڭ مۇسبەت قۇتۇبى تىرانسىفورماتورنىڭ ئىككى ئورامى ئۇتتۇردىكى نۇقتىغا ئۇلانغان، مەنفى قۇتۇبى كىرىستال ئۈچ قۇتۇپلۇق لامپىنىڭ تارقاتقۇچى قۇتۇبىغا ئۇلانغان، يەنە بىر قارشىلىق ئارقىلىق كىرىستال ئۈچ قۇتۇپلۇق لامپىنىڭ ئاساسى قۇتۇبىغا قوزغىتىش تۇكى تەمىنلەنگەن، سىم ئورامنىڭ ئۈستىنكى ئۇچى كىرىستال ئۈچ قۇتۇپلۇق لامپىنىڭ يىغقۇچى قۇتۇبىغا ئۇلانغان، ئاستىنقى ئۇچى بولسا كىچىك سىغىملىق قۇتۇپسىز كوندىساتور ئارقىلىق كىرىستال ئۈچ قۇتۇپلۇق  لامپىنىڭ ئاساسى قۇتۇبىغا ئۇلانغان.

   بۇ توك يۇلىدا ئۈز-چات توك ئۈتكۈزگەندە قارشىلىق ئارقىلىق ئاز مىقتاردىكى توك كىرىستال ئۈچ قۇتۇپلۇق لامپىنىڭ ئاساسى قۇتۇبىغا توك تەمىنلەش بىلەن بىرگە كوندىساتورمۇ زەرەتلىنىش بىلەن بىرگە ئاساس قۇتۇپنى توك بىلەن تەمىنلەيدۇ، توك مەنبەسىنىڭ مۇسبەت قۇتۇبىغا ئۇلانغان ئورامدىن ئۈتكەن توك  كىرىستال ئۈچ قۇتۇپلۇق لامپىنىڭ يىغقۇچى قۇتۇبى ۋە تارقاتقۇچى قۇتۇبىنىڭ توك ئۈتكىزىشى بىلەن ئورامدا ماگىنىت مەيدانى ھاسىل قىلىدۇ، ئاندىن ئاستىنقى ئورامدا مەنبە بىسىمىدىن يۇقىرى بولغان ئىندوكسىون توك پەيدا بۇلىدۇ، بۇ توك كوندىساتوردىكى زەرەتلەنگەن بىسىمنى تىخىمۇ ئاشۇرۇپ توك كۈچىنىڭ داۋاملىق ئىشىشىنى كەلتۈرۈپ چىقىرىدۇ، يۇقىرى چاستوتىلىق ئۈزەكنىڭ ئىندوكتىۋلىقى ۋە كوندىساتور سىغىمىنىڭ چوڭ-كىچىكلىكى تەۋرىنىش چاستوتىسىنىڭ يۇقىرى-تۈۋەنلىكىگە بىۋاستە مۇناسىۋەتلىك بۇلىدۇ. ئىندوكسىون توك بىسىمى ئاخىرقى نۇقتىغا يەتكەندە تۇراقلىق توك ئۈتىۋاتقاندەك ھالەت شەكىللىنىپ ئىندوكسىيلىنىش تۇختايدۇ،(توك ماگىنىت مەيدانى داۋاملىق ئاشمىغان ھالەتتە سىم ماەىنىت مەيدانىدا ھەركىتىنى تۇختاتقانغا ئۇخشايدۇ، بۇ چاغدا ئىندوكسىيلىك توكمۇ چىقمايدۇ) كىرىستال ئۈچ قۇتۇپلۇق لامپىنىڭ ئاساسى قۇتۇبىغا كىلىدىغان توك ئىقىمىمۇ كوندىساتورنىڭ زەرەتلىنىشتىن تۇختىشى سەۋەپلىك ئازىيدۇ، بۇنىڭ بىلەن ئىندوكسىون ئورامدىكى توك بىسىمىنىڭ تۈۋەنلىشىگە ئەگىشىپ كوندىساتورمۇ قارشىلىقتىن ۋە ئىندوكسىون ئورامدىن ھاسىل بولغان تۇيۇق يولدا زەرەتسىزىنىشكە باشلايدۇ، ئاخىرىدا ئىندوكسىون ئورامدا ھاسىل بولغان قارشى يۈنىلىشتىكى قۇيۇن توك(ئىندوكسىون ئورامدا ئادەتتە كىرىش تۇكىنىڭ يۈنىلىشى تەتۈر بولمىسىمۇ ئازراق مەنفى بىسىم ھاسىل بۇلىدۇ) تەسىرىدىن توك مەنبەسىدىنمۇ تۈۋەن بولغان ھالەتتە زەرەتسىزلىنىدۇ ھەم قارشىلىقنىڭ تەمىنلەيدىغان قوزغاتقۇچى تۇكىنىمۇ قىسقا پەيتلىك تۇختىتىپ قۇيىدۇ، ئاندىن يەنە قايتا قارشىلىق ۋە كوندىساتوردىن ئۈتكەن توكتا يەنە بىر دەۋىر توك ئۈتۈشكە باشلايدۇ.

  ئەلۋەتتە بۇ يەردىكى قارشىلىقنى كوندىساتورنىڭ ئورنىغا ئالماشتۇرۇپ قۇيۇپ، قارشىلىقنىڭ ئورنىنى بوش قويسىمۇ تەۋرىنىدۇ، ئۇنىڭدا تەۋرىنىش چاستوتىسىنى كىرىستال ئۈچ قۇتۇپلۇق لامپىنىڭ كۈچەيتىش ھەسسىلىگى ، ئۈزەكنىڭ ئىندوكتىۋلىقىدىن باشقا قارسىلىقنىڭ چوڭ-كىچىكلىكىگە مۇناسىۋەتلىك بۇلىدۇ. بۇنى سىناپ باقساڭلار بايقايدىغىنىڭلار ئۇخشاش قۇۋۋەتكە ئېرىشىش ئۈچۈن قارشىلىقتىلا تەۋرەتكەن توك يۇلىدا ھەم كۆپ توك ئىسراپ بۇلىدۇ ھەم كىرىستال ئۈچ قۇتۇپلۇق لامپا بەكرەك قىزىيدۇ. سەۋەبى كېرەكلىك ۋاقىتتىمۇ، كېرەك بولمىغان ۋاقىتتىمۇ قارشىلىقتىن ئۈتىدىغان توكنىڭ مۇقىم چوڭ بۇلىشى تەلەپ قىلىنىدۇ. كوندىساتور يادەمچى قىلىنغىنىدا بولسا ئىندوكسىون توك ئىشىشىغا ئەگىشىپ كوندىساتور ئارقىلىق كىرىستال ئۈچ قۇتۇپلۇق لامپىنىڭ ئاساسى قۇتۇبىغا بارىدىغان توكمۇ ئاشىدۇ، تۈۋەنلىگەندە بولسا كوندىساتوردىكى توك قارشىلىقتىن ئۈتۈپ زەرەتسىزلىنىدۇ.دىمەك بۇ جەرياندا ئىندوكتىپلىق يۇقىرى نۇقتىغا يىتىپ ئىككىلەمچى ئورامدا داۋاملىق ئىندوكسىيلىك توك ھاسىل بولمىغان ھالەتتىمۇ قارشىلىقتىن ئۈتىدىغان توك ئازايسىمۇ يەنىلا بىر قىسىم توكنىڭ داۋاملىق ئۈتۈپ تۇرىشى كاپالەتلىنىپ قالىدۇ.
* يۇقىرى بىسىملىق تەۋرەنگۈچلۈك توك مەنبەلەردە بولسا كىرىستال ئۈچ قۇتۇپلۇق لامپا يۇقىرى بىسىمغا بەرداشلىق بېرەلەيدىغان بولغاندىن باشقا كۈچەيتىشچانلىغى ۋە كىرىستال ئۈچ قۇتۇپلۇق لامپا ئاساسى قۇتۇبىنىڭ قوزغىتىش تۇكىنىڭ بىسىمى قاتارلىق تەرەپلەردە يەنىلا ئورتاقلىق بار بولغاچقا كىرىستال ئۈچ قۇتۇپلۇق لامپىنىڭ ئاساسى قۇتۇپ تۇكىنى قۇۋۋىتى سەل چوڭراق بولغان قارشىلىقتا تەمىنلەپ ئاندىن ئايرىم بىر ئىندوكتىپلىقنى قايتۇرغۇچى ئورام ئورىلىپ بىر ئۇچى كىرىستال ئۈچ قۇتۇپلۇق لامپىنىڭ تارقاتقۇچى قۇتۇبىغا ، يەنە بىر ئۇچى كوندىساتور ۋە قارشىلىق ئارقىلىق كىرىستال ئۈچ قۇتۇپلۇق لامپىنىڭ ئاساس قۇتۇبىغا ئۇلىنىدۇ.

  يۇقارقى ئىسخىمىدا 220ۋولىتلىق توك كۆۋرۈكسىمان رەتلەشتىن كىيىن كوندىساتوردا بىسىمى ئۆرلەپ 300ۋولىتقا يىتىدۇ، ئاندىن 680 كىلو ئوملۇق قارشىلىق بىلەن كىرىستال ئۈچ قۇتۇپلۇق لامپىنىڭ ئاساس ئاجىز قۇتۇبىنى قوزغىتىپ كىرىستال ئۈچ قۇتۇپلۇق لامپىنىڭ يىغقۇچى قۇتۇبى بىلەن تارقاتقۇچى قۇتۇبىدا ئازراق توك ئۈتكۈزۈش ھالىتىگە كىرگۈزىدۇ، شۇنىڭ بىلەن 220 ئورام ئورالغان يۇقىرى بىسىملىق ئورامدىن بۇ توك ئۈتكەندە 9 ئوراملىق يەنە بىر ئورام گۇرۇپپىسىدا ئىندوكسىيلىك توك ھاسىل بۇلىدۇ شۇنىڭ بىلەن ئىندوكسىيلەنگەن بىسىم كوندىساتور ۋە قارشىلىقتىن ئۈتۈپ كىرىستال ئۈچ قۇتۇپلۇق لامپىنىڭ ئاساس قۇتۇبىغا كىلىدىغان توك ئىقىمىنى تەدرىجى كۈچەيتىدۇ، شۇنىڭ بىلەن يىغقۇچى قۇتۇپ بىلەن تارقاتقۇچى قۇتۇپتىن ئۈتىدىغان توك ئىقىمىمۇ ئۈزلۈكسىز ئاشىدۇ، ئاخىرىدا يەنە شۇ ماگىنىت مەيدانىنىڭ تۇيۇنىشىدىن كىيىن ئىندوكسىون توك تۇختايدۇ . ئەمما بۇ توك يۇلىدا كوندىساتورنىڭ زەرەت قۇيۇپ بېرىش يۇلى ئۈزۋىتىلگەن بولغاچقا ئېغىر يۈكلۈكئىشلەشتە ئاسانلا تەۋرىنىشتىن تۇختاپ قالىدۇ ياكى توك كۈچى يىتەرلىك بولمايدۇ. شۇڭا ئەمىلى ئىشلىتىشتە بىر ئىككى قۇتۇپلۇق لامپا قۇشۇلۇپ كوندىساتورنىڭ ئورامدىن قارشى يۈنىلىشتە زەرەت قۇيۇپ بېرىلىشى كاپالەتلەندۈرۈلىدۇ.

   تۇرمۇشتا ئىشلىتىۋاتقان توك مەنبەلەر بولسا ئاساسەن 110~260ۋولىت بىسىمدا چىقىش بىسىمى مۇقىم بولغان مەنبەلەر. يەنى كىرىستال ئۈچ قۇتۇپلۇق لامپا توك ئۈتكۈزىشى ئۈزلۈكسىز ئىشىۋاتقان ئەھۋالدا چىقىش بىسىمى بىز بەلگىلىگەن 12 ۋولىتقا يەتكەندە داۋاملىق يۇقىرلاپ مىڭىشىنى چەكلىيەلەيدۇ.

  تۇرمۇشتا ئىشلىتىۋاتقان توك مەنبەلەرنىڭ تەلەپ قىلىدىغىنى يۈكلۈك بولسۇن ياكى يۈكسىز بولسۇن چىقىش بىسىمى قانچە مۇقىم بولسا شۇنچە ياخشى، شۇڭا قۇۋۋەت لامپىسىنىڭ بىسىمنى ئاشۇرىشى ياكى ئاشۇرماسلىقىنى پۈتۈنلەي چىقىش تۇكىدىن كونتۇرول قىلىمىز. چىقىش بىسىمى بىز بەلگىلىگەن بىسىمغا يەتكەندىن كىيىن ئازراقلا ئىشىپ قالسا بىسىم مۇقىملاشتۇرغۇچنىڭ سىگنال تۇكى يورۇقلۇق سەزگۈچى ۋە يوروغۇچى لامپىدىن تۈزۈلگەن تۆت پۇتلۇق قۇرۇلما pc817 ئارقىلىق چىقىش بىسىمىدىكى كونتۇرول سىگنالى قۇۋۋەت كۈچەيتكۈچ لامپىنىڭ ئاساسى قۇتۇپ تۇكىنى يەرلەشتۈرىدىغان لامپىنىڭ ئاساسى قۇتۇبىنى توك بىلەن تەمىنلەپ قۇۋۋەت لامپىسىنىڭ توكنى ئۈتكۈزۈش ياكى ئۈتكۈزمەسلىكىنى كونتۇرول قىلىدۇ. يەنە ئاساسى قۇۋۋەت لامپىسى 13003 نىڭ ئۈتكۈزىدىغان توك كۈچى ئىشىپ كەتسە تارقاتقۇچى قۇتۇبىدىكى يەرگە ئۇلانغان قارشىلىقنىڭ ئىككى بىشىدا توك بىسىمى ھاسىل بۇلىدۇ، بۇ بىسىم يەنە بىر لامپىنىڭ ئاساسى قۇتۇبىنى توك بىلەن تەمىنلىسە ئاساسى قۇۋۋەت لامپىسىنىڭ ئاساسى قۇتۇبىدىكى توك بىسىمىنى يەرلەستۈرۈپ، توك ئىقىمىنىڭ داۋاملىق ئىشىپ كىتىشىدىن ساقلىنالايدۇ. ئۇ يەردىكى قارشىلىق بىز لاھىيلىگەن توك مەنبەسىنىڭ قۇۋۋىتى بىلەن زىچ مۇناسىۋەتلىك، ھەم توك مەنبەسىنىڭ ئىشلىتىش ئۈمرى بىلەنمۇ زىچ مۇناسىۋەتلىك، بەك سەزگۈر قۇغدىنىشچان قىپ قويسا چىقىش توك بىسىمىنىڭ مۇقىملىقىغا تەسىرى چوڭ بۇپ كىتىدۇ، شۇڭا ئەمىلى ئەھۋالغا قاراپ زىيادە يۇقىرى توك ئۈتۈشتىن ساقلىنىدىغان قىلىپ تەڭشەيمىز.
بۇ چاغدا سىگنالنىڭ زىيادە قىسمى يەنى يۇقىرى قىسمى كىسىۋىتىلىدۇ.

   بۇ يەردىكى pc817 بولسا 0.1ۋولىت توكتا يانىدىغان لامپا ۋە يۇقىرى چىداش بىسىمى 35 ۋولىت بولغان يورۇقلۇق سەزگۈچى لامپىدىن تەركىپ تاپقان قۇرۇلما بۇلۇپ، سەزگۈرلىكى ۋە سۈرئىتى يۇقىرى، 50 مىللى ئامپىر توك ئىقىمىدا يورۇيدۇ، كونتۇرول قىلىش قۇۋۋىتى 70 مىللىۋات. يورۇقلۇق چىقىرىدىغان لامپا بىلەن يورۇقلۇق سەزگۈچى لامپا ئارلىقىدە توك ئۈتۈپ كىتىشتىن ساقلىنىش ئىقتىدارى 6000ۋولىتقا يىتىدۇ. شۇڭا 220 ۋولىت توكنىڭ ئىندوكسىيەلىك توك قىسمىغا ئتاراپ كىتىپ ئادەمنى توك سۇقىۋىتىشىدىن ساقلىنالايدۇ.
  ئۈنۈمى تىخىمۇ يۇقىرى، قۇراشتۇرلىشى ئاددى بۇلىشى ئۈچۈن ھازىرقى مەنبەلەردە ھەم تەۋرىنىش چاستوتىسى ھەم كونتۇرول قىلىنىدىغان تەۋرىنىش ۋە قوزغاتقۇچ ئىمپىلوس سىگنال چىقىرىدىغان توپلاشتۇرۇلغان توك يۇلى ئىشلىتىلىدۇ.

كۆپ ئىشلىتىدىغان ئىمپىلوسلۇق قوزغىتىش مەنبە توك يۇلى uc3843 ، uc3842 ۋە uc3845 نى تۇنۇشتۇرۇپ ئۈتەي.
بۇلارنىڭ ئىچكى تۈزلىشى ئاساسەن ئوخشاپ كىتىدۇ، پەقەت uc3845 لا 12 ۋولىتلىق مەنبەدە تەۋرىنەلەيدۇ. قالغانلېرى توك مەنبەسىنىڭ بىسىمى 16 ۋولىتقا يەتكەندە ئاندىن تەۋرىنىشكە باشلايدۇ، تەۋرىنىشتىن تۇختاش بىسىمى 8 ۋولىت بۇلىدۇ. R4 قارشىلىقتىن ئۈتكەن ئاز توك ئىقىمى c7-8 لەرنى زەرەتلەپ توك بىسىمى 17 ۋولىتقا يەتكەندە ئىمپىلوس سىگنال 6-پۇتتىن چىقىپ توك مەنبە خىزمەت قىلىشقا باشلايدۇ، شۇنىڭ بىلەن 9ئوراملىق n2 ئورامدىن ئىندوكسىون توك چىقىپ توپلاشتۇرۇلغان توك يۇلىدىكى c7-8 لەردە زەرەتلەنگەن مەنبە بىسىمى 13 ۋولىتلارغا چۈشۈپ بولغۇچە قايتا تۇلۇقلاشقا باشلايدۇ، ھازىر ئىشلىتىلىۋاتقان uc3843 ، uc3842 توك يوللېرىدا نورمال خىزمەت قىلىۋاتقاندىكى مەنبە بىسىمى كۈپىنچىسىدە 17 ۋولىت ئەتراپىدا بۇلۇپ، توپلاشتۇرۇلغان توك يۇلى سەل قىزىيدۇ، موس قۇۋۋەت لامپىسىنى قوزغىتىشقا ئادەتتە پەقەت 15 ۋولىت بولسىلا يىتەرلىك ، ئەمما تەۋرەنگۈچلۈك مەنبە توپلاشتۇرۇلغان توك يۇلىنىڭ بىسىمىنى تۈۋەنلىتىپ 15 ۋولىتقا چۈشۈرۈپ قويغاندا يىڭى قوزغاندا ۋە ئېغىر يۈك ئۇلانغان ۋاقىتلاردا كوندىساتوردىكى زەرەت مىقتارى داۋاملىشالماي توك يۇلى خىزمەتتىن تۇختاپ قالىدۇ، ياكى قوزغاتقۇچ ئىمپىلوس بىسىمى تۈۋەنلەپ كەتكەندە موس لامپا بەك قىززىپ كىتىدۇ ياكى كۈيۈپ كىتىدۇ. ئەگەر c7-8 لەرنىڭ سىغىمىنى ئاشۇرۇپ توك مەنبەنىڭ مۇقىملىغىنى ساقلىماقچى بولساق يىڭى قوزغالغاندا قارشىلىقتىن ئۈتىدىغان توك ئىقىمى كوندىساتوردىكى بىسىمنى17 ۋولىتقا يەتكۈزۈش ۋاختى ئۇزىراپ كىتىدۇ، قارشىلىق قۇۋۋىتىنى چوڭلىتىپ تىز زەرەتلەيمەن دىگەندە ئورۇنسىز توك ئىسراپچىلىقى يۇقىرلاپ كىتىدۇ. شۇڭا ئىسخىمىدىكى قىزىل سىزىلغان بىر دانە 220 مىكرولۇق كودىساتور بىلەن قىزىل سىزىلغان ئىككى قۇتۇپلۇق لامپا توك يۇلىغا سەپلەنسە ئورامدىن چىقىدىغان توك بىسىمىنى 15 ۋولىتقا چۈشۈرسىمۇ بۇلىۋېرىدۇ.

  قوزغاتقۇچ ئىچىدە ھەرە چىشى تەۋرەنگۈچ ، ئىمپىلوس سىگنالغا ئايلاندۇرغۇچ، ئىمپىلوس كەڭ-تارلىقىنى تىزگىنلىگۈچ، تەۋرەنگۈچ بىسىم مۇقىملىغۇچى، سىرىتتىن كىرگەن ئىقىم سىگنالى ئارقىلىق ئىمپىلوس كەڭلىكىنى تىزگىنلىگۈچكە قۇشۇمچە سىگنال بېرىش، تۈۋەن بىسىم ھالىتىدە خىزمەتنى تۇختىتىش، بىسىم كونتۇروللاش سىگنالىنى كۈچەيتىپ ئىمپىلوس كەڭلىگىنى تىزگىنلىگۈچ چىقىش ئىمپىلوس سىگنالىنى كۈزىتىش قاتارلىق ئىقتىدارلېرى بار.
ئىمپىلوسنىڭ ھاسىل بۇلىشى سىنوسلۇق مەنبە تۇغرىلىق تىمىدا بولغاچقا بۇنىڭدا تەپسىلى يازمىدىم.
  ئادەتتە نىپىز ئىكرانلىق تىلۋىزور قاتارلىقلاردا ئىشلىتىۋاتقان يۇقىرى ئۈزەك ھەجىم تەسىرى تۈپەيلى كىچىكرەك لاھىيلەنگەن بۇلۇپ كۈپىنچىسىدە مەنبە قىسمىدىلا %10 ئەتراپىدا قۇۋۋەتنىڭ ئورۇنسىز ئىسراپ بۇلىشىنى كەلتۈرۈپ چىقىرىدۇ. شۇڭا كۆپ ساندا يۇقىرى چاستوتىلىق ئۈزەك ئىشلىتىش ئارقىلىق چوڭ قۇۋۋەتلىك مەنبە قىلىپ ئىشلىتىدىغان قىلغاندا ئىنىرگىيە تىجەش ھەم ئىشلىتىش ئۈمرىنى ئۇزارتقىلى بۇلىدۇ.
  تىما يوللاش بوشلىقىنىڭ ئۆلچىمى تۈپەيلى كىيىنكى تىمامدا بۇنىڭدىن ھەجىمى كىچىك ، قۇۋۋىتى ھەسسىلەپ يۇقىرى بولغان توك مەنبەنى تۇنۇشتۇرىمەن.
