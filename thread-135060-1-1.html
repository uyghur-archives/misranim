يىقىندىن بىرى بەس-مۇنازىرلەر قىززىپ ،كەڭ تورداشلارنىڭ دىققىتىنى تارتتى ،بولۇپمۇ داڭلىق شەخىسلەرنىڭ لىكسىيەلىرى؛"ئۆمۈد پاراڭلىرى"پىروگىراممىسىدىكى بەزى نازۇك سۆزلەر ھەقىقەتەن كىشىنى چوڭقۇر ئويغا سالىدۇ.ئۆمۈد غەنى ئەپەندىنىڭ ئاساسلىق مەقسىتى بىزگە ئىنىق ئايان بولدى،جەميەتكە ،تۆھپە قۇشۇش ،كىشىلەرنىڭ ساپاسىنى يۇقرى كۆتۈرۈشكە يول كۆرسىتىش،ئەسلى ئۇيغۇر مىللىتىنىڭ مەدەنىي ساپاسىنى يۇقىرى كۆتۈرۈش دىمەكچى ئىدىم ،بىراق ئۆمۈد ئەپەندى "بىز يەرشارى پۇقراسى بولشىمىز كىرەك"دىگەنلىكى ئۈچۈن ،ناتىق جەميەتنى كۆزدە تۇتقان دەپ ھىس قىلدىم.
  ئۆمۈد ئەپەندى بۇ كىچىككىنە سەھنىنى ياراتمىدىمۇ قانداق،ھىچقانداق تەييارلىق قىلمىغان ھالەتتە پىروگىراممىغا قاتناشقان،تۆۋەندە بىر قانچە جەھەتتىن مۇلاھىزە قىلمەن
1."مەن 14 يىشىمدا قۇرئان ئوقۇغان،..."،ئۆمۈد غەنىدەك تالانتىلىق رىياسەتچى ،بۇنداق چۈشىنىكسىز بولغان ئىبارىنى قوللىنىشى ھەقىقەتەن كىشىنى ئويلاندۇردۇ،ئۆمۈد ئەپەندى قۇرئاننىڭ تەرجىمىسىنى ئوقۇغانمۇ ياكى ئەسلى نۇسخىسىنىمۇ؟تەرجىمىسىنى ئوقۇغان بولسا نىمىشقا بەك تەستە ئوقۇيدۇۋەتەستە چۈشۈنىىدۇ؟ئەسلى نۇسخىىسىنى ئوقۇغان بولسا نىمىشقا"پۈتۈن دۇنياغا تەسىر كۆرسەتكەن ئەسەر"دەيدۇ؟قۇرئاندا ئىنىقلا اللە نىڭ كىتابى،اللە نىڭ سۆزى دىيىلىدىغان تۇرسا.نىمىشقا "ئەسەر"سۆزىنى ئىشلىتىدۇ؟"ئەسەر"ئەرەبچە مەنىسى نەمۇنە دىگەنلىك،ئۇيغۇرچە ئىستىمال مەنىسى مەلۇم ئىنسان تەرپىدىن ئىجاد قىلىنغان ئەدەبى كىتاب ۋە شۇنداق نەرسىلەرنى كۆرسىتىدۇ.ئەسلى تىلغا ئىلىشنىڭ ھىچقانداق زۆرۈرىيىتى يوق مەزمۇننى كىرگۈزدى؛تىلغا ئالغان ئىكەن چوقۇم چۈشنىشلىك ،توغرا قايىل قىلارلىق سۆزلەرنى ئىشلىتىشى كىرەك،ھىلىغۇ ئۆمۈد غەنىكەن دۇنيادىكى داڭلىق كىشىلەرمۇ ناتوغرا سۆزلەرنى ئىشلىەتكەنلىكىدىن كەچۈرۈم سورغان."14 يىشىمدا قۇرئان ئوقۇدۇم،،،،،،،،ئاندىن ئىنجىلنى ئوقۇدۇم ،،،،،،،" ئاساسلىق دىمەكچى بولغىنى نىمە ؟نۇرغۇن تورداشلار ،قىرىنداشلار بۇ ئىبارىنى چۈشنەلمىدى
2."ئۇيغۇرلا يەر شارى پۇقراسى بولۇشى كىرەك"
   ئۆمۈد غەنىنىڭ بۇ سۆزگە ئالغان مىسالى بىر قىرىندىشىمىزنىڭ ئۇيغۇرنى دۇنياغا تونۇتىمىز دىگەن سۆزى ۋە ئامرىكىلىق نازاكەت خانىمدىن "سىىزمۇ ئامىرىكلىقلارنى دۇنياغا تونۇتماقچىمۇ ؟"،"-ياق بىز دۇنيانى تونۇماقچى "،"سىز قايسى مىللەتتىن ؟"،"ئۇقمايمەن،دادام ،،،،،ئانام...."بۇ مىساللارنى  ئۆمۈد غەنى پەخىرلەنگەن ھالدا سۆزلىدى؛ ئامىرىكىلىقلارنىڭ ھەممىسى شۇنداق ئويلامدۇ،ياۋرۇپاچۇ؟نازاكەت خانىمنىڭ قارشىنى پۈتكۈل ئامرىكىغا ،ياۋرۇپاغا ۋەكىللىك قىلدۇ دىسەك بولامدۇ؟بىز شۇنداق ھالەتتە بولايلى دەيلى ،سىز چەتئەلگە بارغاندا ئىسمىڭىزنى قايسى تىلدا قوللاندىڭىز؟ سىزنىڭمۇ ئوغۇل قىزىڭىز باردۇ؟ئۇلارنىڭ توينى مىللەت ئايرىماي قىلىپ بىرسىزغۇ دەيمەن ،يەرشارى پۇقراسى بولغاندىكىن.ئۇنداق بولمىسا يەرشارى پۇقراسى دىگەن ئىنىقلىمىنى قانداق چۈشنىمىز؟
3.دىننى ۋە تارىخى كىتابلارنىڭ سىتىلىشى ياخشى،،،،دىننى ۋە تارىخى كىتابلار تەپەككۇرىمىزنى چىكىندۈردۇ
ئۆمۈد غەنى ئەپەندى !سىزنىڭ 14 ياش ۋاقتىدا قۇرئاننى ئوقۇپ چۈشنەلمەسلىكىڭىز بۇ يەكۈننى چىقىرشىڭىزنىڭ ئاساسىمۇ؟ئەنە شۇ دىنغا ئىتىبار بىرىلدىمۇ؟ تەتقىق قىلىش سەھنىسى بىرىلدىمۇ؟پايدىلنىش پۇرسىتى بىرىلدىمۇ؟سىز قۇرئاننى سۆيۈپ ،ھوزۇر قەلبىڭىز بىلەن ئوقۇپ باقتىڭىزمۇ؟مەن ھەققى ھوزۇر قەلبىم بىلەن يىنىش -يىنىشلاپ ئوقۇيمەن،مەن تولۇق ئوتتۇردا فىزىكا،خىمىيە ،جۇغراپىيە قاتارلىق پەنلەرنى ياخشى ئۈگەنگەن ،ئالى مەكتەپتە بىئولوگىيە ئۈگەندىم،شۇنى بىلىپ يەتتىمكى ھازىرقى ئىلىم -پەن قۇرئاننىدىكى بىلىملارنى ئىسپاتلاۋىتىپتۇ. بىز بولساق توپا باستۇرۇپ ،ئەرەپچىسىگە ئوقۇپ قويۋاتىمىز.تارىخ--مەشھۇرلار ئىيتقان تارىخ كىشىنى زىرەك ،ئەقىللىق ،سەزگۈر قىلىدۇ دەپ. يەنە شۇ قارشىڭىزدا چىڭ تۇرامسىز؟
مەنچە ھەپتىدە 4،5 قىتىم قويلىۋاتقان سەنئەت نومۇرلىرى،ئۆزلۈكسىز ئىجاد قىلنىۋاتقان تىتىقسىز ناخشىلار ،تەپەككۇرنى ئاچامدۇ؟تارىختا نۇرغۇن خانلىق ،بەگ تۆرلەر ناخشا -ئۇسۇلنى ئۆزىگە خىزمەت قىلدۇرغان،ئىنىقى ناخشا -ئۇسۇل بىلەن شۇغۇللۇنىدىغانلارنىڭ ھەممىسى قۇل -دىدەك ،چاكارلار ئىدى،ھازىر مەدەنىلەشتۇق،بىز دۇنياغا ناخشا -ئۇسۇلچى خەلىق بولۇپ تونۇلدۇق،ئىنسان روھىنى چىرتىۋاتقان،تەپەككۇرىنى چىكىندۈرۋاتقان سەۋەب دىننى ۋە تارىىخ بولسا ،بۇلار نىمە؟
  سىز تىلۋىزىيە ئىستانسىدا بولغانكىن رەھبەرلەرگە ئىلتىماس قىلسىڭىز ،ئاشۇ ئەدەبى سەنئەتنىڭ ئورنىغا بىر تۈردە پەن -تىىخنىكىدا يىقىلىق ياراتقلانلار مۇسابىقىسى ،ئىلغار تەپەككۇر نى تاللاش مۇسابىقىسى دىگەندەك مەزمۇنلاردىكى سەھىپىلەرنى ئورۇنلاشتۇرسىڭىز ،بىز مۇ نوبىل مۇكاپات دىگەنلەرنى ئىلىپ باقساب .ھىچ بولمىغاندا سىز دىگەن ئاشۇ سوپۇنىنى ئىچكىرسىدە ئىشلەپ ،ئۇيغۇرچە ماركا بىسىپ  ساتىدىغان سودىگەرلەر "كىتاب خالتىسى"،بىكا تەلەپلەرنى ئەمەس ،ھەققى بىلمى بارلارغا ئىرىشىپ،رايۇنىمىزدا زاۋۇت خارخانلارنى قۇراتتى،ئەنە شۇ چاغدا ھەرقايسى تىلدىكى كىتابلار ئوقۇلاتتى،ياشلار خىزمەت ئۈچۈن تامىقىنى كۈتۈپخانلاردا يىگەن بولار ئىدى،يىراق -يىراقلاردىن دۇتتا -تەمبۇرلىرىنى ئەمەس ،كەشپىياتلىرىنى كۆتۈرۈشۈپ كىلىشەتتى.ئەنە شۇ چاغدا ھەپتىلىك چولپان -يىللق چولپانلار ئەمەس بەلكى ھەرقايسى كارخانلارنىڭ ئىنجىنىرلىرى بارلىققا كىلەتتى ،ئەنە شۇ چاغدا تۆت ئۇستاز كەينىگە ئۆرۈلۈپ ئەمەس ،لىدىرلار ،دىرىكتۇلار شۇ تالانتىلىقلارنىڭ ئالدىغا يۈگرەپ بىرىپ ،بىزنىڭ سىپىمىزگە قوشۇلۇڭ؛ مەن سىزنى بىناكارلىق ساھەسىدە ئەۋزەل شارئىتقا ئىگە قىلمەن.ماڭا قوشۇلۇڭ مەن سىزنى ئاپتۇمۇبىل ئىشلەپچىقىرىش ساھاسىدە ياپۇنغا ،ئامرىكىغا ئۇقۇشقا چىقىرمەن دىگەن بولار ئىدى،ئەنە شۇ چاغلاردا قاۋاقخانا،تورخانا،مەيخانا،زامانىۋى ئۇسۇلخانلاردىكى شەپەرەڭ،لايغەزەللەر تۇرمۇش بىسمىدىن كىتابخانىدا تۈنەيتى،باشلانغۇچتىن -ئالى مەكتەپ ،جەميەتكىچە بولغان "يەرشارى پۇقرالىرى"كىتابتىن بىشىنى كۆتۈرمەيتى،كەچلىك تەكرارنى باھانە قىلىپ ،ئامرىكا كىنولىرىنى ئىشلەپمۇ يۈرمەيتى،ئۆيىگە ۋاقتىدا قايتاتتى.پەقەت ھەپتىدە بىر سان،ھىچ بولمىسا ئايدا بىر سان -تالانتلىقلارنى بايقاش سەھنىسى ،ئۇ سىزنىڭ قولىڭىزدىن كىلىدىغۇ دەيمەن
  ئاخىردا دەيدىغىنىىم نەق مەيداندا سورالغان سۇئاللارغا بىرىلگەن جاۋابلار ئاساسەن قايىل قىلارلىق ئەمەس.  مەن ئوتۇتۇرغا قويغان مەسسىللەردىن باشقان  قاراشلارغا تامامەن خۇرسەن بولغان ھالدا قوشۇلىمەن ،خۇرسەن بولدۇم،"ئۆمۈد پاراڭلىرى"كىچىك سەھنە بولسىمۇ ياخشىراق تەييارلىق قىلشىڭىزنى ئۆمۈد قىلمەن،چۈنكى كۆردىغانلارنىڭ ھەممىسى تاماششىبىن ،باھالىغۇچى
