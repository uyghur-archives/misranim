ئالدىنقى تىمىدا ‹‹تەۋرەنگۈچلۈك توك مەنبە›› نى قىسقىچە تۇنۇشتۇرغان ئىدۇق، ئۇ تىمىدا دىيىلگىنى پەقەت بىر يۈنىلىشتىلا ماگىنىت مەيدانى ھاسىل قىلىدىغان يېرىم دەۋىرلىك قوزغاتقۇچلۇق مەنبە ئىدى، بۇ تىمىدا بىز ئىندوكتىپلىق توك ھاسىل قىلىشى ھەسسىلەپ يۇقىرى بۇلىدىغان كۆۋرۈكسىمان ۋە يېرىم كۆۋرۈكسىمان توك مەنبە ھەققىدە مۇلاھىزە قىلايلى. بۇ ئىككىسىدىن قايسىسىدىلا بولسۇن يۇقىرى چاستوتىلىق تىرانسىفورماتوردىن ئۈتىدىغان توك، بىر قىتىم ئوڭ يۈنىلىشتە ئۈتسە يەنە بىر قىتىم سول يۈنىلىشتە ئۈتىدۇ.
  يېرىم كۆۋرۈكسىمان توك مەنبە؛
    يېرىم كۆۋرۈكسىمان (ئەمىليەتتە ئىككى كوندىساتور قۇشۇلۇپ پۈتۈن كۆۋرۈكسىمان بۇلىدۇ) مەنبەدە ئىككى لامپىنىڭ ئۇتتۇرىدىن بىرلەمچى ئورامنىڭ بىر ئۇچى ئۇلىنىپ، يەنە بىر ئۇچى توك مەبەگە ئارقىمۇ ئارقا ئۇلانغان كوندىساتورنىڭ ئۇتتۇردىكى نۇقتىغا يەنە بىر كىچىك سىغىملىق كوندىساتور ئارقىلىق ئۇلانغان بۇلىدۇ. 220 ۋولىتلىق توك مەنبە ئۇلانغاندىن كىيىن كۆۋرۈكسىمان ئىككى قۇتۇپلۇق لامپىدىن رەتلەنگەن توك C1 گە زەرەتلىنىپ بىسىمى بارغانچە ئىشىپ بېرىش بىلەن بىرگە ئارقىمۇ ئارقا ئۇلىنىپ مەنبەگە ئۇلانغان C3 ۋە C4لەرمۇ مەنبە بىسىمىغا تەڭ زەرەتلىنىدۇ، ئىككى كوندىساتور ئۇلانغان ئۇتتۇرا نۇقتىدا توك مەنبەنىڭ يېرىمىغا تەڭ بىسىم ھاسىل بۇلىدۇ.
  قۇلايلىق ئۈچۈن ماگزىنلاردا رەڭلىك ۋولفىراملىق چىراغ ياندۇرۇشتا ئىشلىتىدىغان توك يۇلى ئىسخىمىسىنى تۇنۇشتۇراي.


  توك يۇلى ئىخىمىسىدىكى L1,L2,L3 لەر ئىسخىما يۇقىرى تەرپىدىكى رەسىمدىكىدەك ماگىنىتلىق ھالقىغا ئىككى ئورامدىن ئورالغان بۇلىدۇ. بىرلەمچى ئورامدىن ئۈتىدىغان توك ئىقىمى ماگىنىتلىق ھالقىدىكى ئىككى ئورام ئورالغان سىمدىن ئۈتكەندە ماگىنىت ھالقىدىكى قالغان ئىككى ئورامدا ئىندوكسىيلىك توك ھاسىل بۇلىدۇ. لاھىيەدە L1 ۋە L2 لەر نىڭ R2 ۋە R4 لەر ئارقىلىق كىرىستال ئۈچ قۇتۇپلۇق لامپىنىڭ ئاساس قۇتۇبىغا ئۇلىنىدىغان ئۇچلېرىدىن چىقىدىغان ئىندوكسىون توكنىڭ يۈنىلىشى بىر-بىرىگە قارشى لاھىيلەنگەن. R3 تىن ئۈتكەن ئاجىز توك ئىقىمى Q2 نىڭ ئاساسى قۇتۇبىغا قوزغاتقۇچى توك تەمىنلەپ يىغقۇچى قۇتۇبى بىلەن تارقاتقۇچى قۇتۇبىدا توك ئۈتكۈزىدۇ. Q1 بولسا بۇ چاغدا قوزغالمايدۇ. چۇنكى R3 تىن كەلگەن توك  Q1 نىڭ تارقاتقۇچى قۇتۇبىغا ئۇلانغان، ئادەتتە كىرىستال ئۈچ قۇتۇپلۇق لامپىلارنىڭ ئاساس قۇتۇپ توك بىسىمى تارقاتقۇچى قۇتۇپ توك بىسىمىدىن يۇقىرى بولمىغاندا خىزمەت قىلمايدۇ.
  Q2 خىزمەتكە كىرىشكەندىن كىيىن L1 دىكى ھاسىل بولغان ئىندوكسىون توك بىسىمى ئاشقانچە Q2 تىخىمۇ كۆپ توك ئۈتكۈزىدۇ، بۇ جەرياندا C4 كوندىساتوردىكى زەرەتلەنگەن ئوتتۇرا بىسىم توك تىرانسىفورماتور، L3 ۋە Q2 لەردىن تۈزۈلگەن ئىلىكتىر زەنجىرى ئارقىلىق زەرەتسىزلىنىدۇ ھەم بىرلا ۋاقىتتا شۇ ئىلىكتىر زەنجىرى ئارقىلىق C3 زەرەتلىنىدۇ. C4 زەرەتسىزلىنىپ ، C3 زەرەتلىنىپ ئاخىرلاشقاندا بىر دەۋىر توك ئۈتكۈزۈش ئاخىرلىشىدۇ.
  بىر دەۋىرلىك توك ئۈتكۈزۈش ئاخىرلاشقاندا تىرانسىفورماتور ئورامى ، L3 ۋە L2 ، L1 لەردە تەتۈر يۈنىلىشلىك قۇيۇن توك ھاسىل بۇلىدۇ، بۇ ئاجىزغىنا تەتۈر قۇيۇن تۇكى Q2 نى توك ئۈتكۈزۈشتىن تۇختىتىش بىلەن بىرگە Q1 لامپىنىڭ ئاساسى قۇتۇبىنى ئاجىز قوزغىتىش بىسىمى بىلەن تەمىنلەپ Q1 نىڭ يىغقۇچى قۇتۇبى بىلەن تارقاتقۇچى قۇتۇبىنى ئاجىز توك ئۈتكۈزۈش ھالىتىگە كىرگۈزىدۇ ھەم بۇ قىتىمدا C3 زەرەتسىزلىنىش بىلەن بىرگە C4 زەرەتلىنىدۇ. ھەم توك مىقتارى ئۈزلۈكسىز ئاشىدۇ. ھەم C3 زەرەتسىزلىنىپ، C4 زەرەتلىنىپ ئاخىرلاشقاندا يەنە ئوراملاردا قۇيۇن توك ھاسىل بۇلۇپ دەۋىرلىنىدۇ. بۇ توك يۇلىدىكى R3 پەقەت بىر قىتىملىق قوزغىلىشنى ئەمەلگە ئاشۇرۇش ئۈچۈنلا ئىشلەيدىغان ئىنتايىن ئاجىز بىر توك ئىقىمى تەمىنلەش ئۈچۈنلا قۇشۇلغاچ، شۇڭا قارشىلىق قىممىتى 510 كىلو ئومدىن بىر مىگا ئومغىچە بۇلىدۇ.
  بۇ خىل توك مەنبە 15 ۋات، 25 ۋات، 35 ۋات، 50 ۋاتلىق ھەر خىل بۇلۇپ، يۈكسىز ھالەتتە 10كىلو گىرتىسقىمۇ يەتمەيدىغان چاستوتىدا ئىشلەيدۇ، ئەمما لامپۇچكىسىنى ياندۇرىۋاتقاندا 25 كىلوگىرتىستىن 100 كىلوگىرتىسقىچە بۇلىدۇ، ئەگەر D1 قوش يۈنىلىشلىك لامپا بولمىسا چىقىرىش توك قۇۋۋىتىنى كونتۇرول قىلغىلى بولمايدۇ. ناۋادا چىقىش توكىنى يۇقىرى چاستوتىلىق ئىككى قۇتۇپلۇق لامپا بىلەن تۇغىرلاپ مۇقىم بىسىم قىلىپ ئىشلەتمەكچى بولغاندا مۇقىم يۈك بولمىغاندا بىسىمنى كونتۇرول قىلغىلى بولمايدۇ.
  كونتۇرول قىلىش ئۈچۈن تەۋرىنىش سىگنالىنى ئىمپىلوس كەڭلىگى كونتۇرول قىلىنىدىغان توپلاشتۇرۇلغان توك يۇلى ئارقىلىق تىزگىنلەشكە تۇغرا كىلىدۇ، كومپىوتىرلارنىڭ توك مەنبەسى دەل مۇشۇنداق توك يۇلىغا تەۋە. يەنى ھەجىمى كىچىك، قۇۋۋىتى يۇقىرى، بىز 55 ۋاتلىق رەڭلىك تىلۋىزورنىڭ تىرانسىفورماتورى بىلەن كومپىوتىرنىڭ توك مەنبەسىنىڭ ئاساسى تىرانسىفورماتورىنى سىلىشتۇرساق كومپىوتىر مەنبەسىنىڭ تىرانسىفورماتورى كىچىكرەك تۇرىدۇ، ئەمما قۇۋۋىتى 5 ھەسسە يۇقىرى...
كومپىوتىر توك مەنبەسى ATX توك مەنبەسى
      

  كۆۋرۈكسىمان توك مەنبە؛
    يۇقارقى يېرىم كۆۋرۈكسىمان توك مەنبەنىڭ پىرىنسىپىدىن ھەجىمى كىچىك، قۇۋۋىتى چوڭ بۇلۇشتا ئادەتتىكى توك مەنبەلەرگە قارىغاندا ئوڭ يېرىم دەۋىر ۋە تەتۈر يېرىم دەۋىرلىك ئىندوكسىيلىك بولغاندا نەچچە ھەسسە يۇقىرى قۇۋۋەتكە ئېرىشكىلى بۇلىدىغانلىقىنى بىلدۇق.
  كىچىك قۇۋۋەتلىك توك مەنبەلەردە رىمونىتقا قۇلايلىق، تەننەرخى ئەرزان بۇلىشى ئۈچۈن 500 ۋاتتىن كىچىك مەنبەلەردە كۈپىنچە يەككە يۈنىلىشلىك تەۋرەنگۈچ مەنبە ۋە ATX توك مەنبەسى ئىشلىتىلىدۇ.    ئىلىكتىرىك كەپشەر دىگەندەك چوڭ قۇۋۋەتلىك ئۈسكىنىلەر ۋە چوڭ قۇۋۋەتلىك تەتۈر ئۆزگەرتكۈچ ، چاستوتىسى تىزگىنلىنىدىغان ئىلىكتىرماتور قوزغاتقۇچلاردا بولسا پۈتۈن كۆۋرۈكسىمان قوزغىتىش ئۇسولى قوللىنىدۇ.
