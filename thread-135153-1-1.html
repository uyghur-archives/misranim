                                                ئاشىق يىگىت
                                                                          
                                    ئاچساملا  كۆزۈمنى  ئۇيقۇدىن  شۇئان ،
                                    قەلىب   ئىكرانىمدا   بۇلىسەن   ئايان  .
                                    تاتلىق   تۇيغۇم   خىيالەن   شۇنداق  ،
                                    يۈرەكنى   تېلغار   غىرىبلىق   ھامان  .

                                  ئۇچراشقان شۇچاغلار ھېلىھەم ئېسىمدە،
                                  بىر كۆرۈپ ئاشىقلىق قوزغالدى  مەندە .
                                  شۇ گۈزەل سىمايىڭ كەتمىدى  پەقەت ،
                                  ۋەسىلىڭگە يىتەلمەي ئۆتىمەن  غەمدە .

                                 چىقمايدۇ ئېسىمدىن تاتلىق كۈلۈشلەر ،
                                 يۈرەكنى  بىر   ئالدى  قىيا   بېقىشلەر .
                                 تۇيدۇرماي ياققانتىڭ يۈرەككە ئوتنى ،
                                تۇتقۇن بوپ ئىشقىڭدا ئاغرىق يېتىشلەر.

                                ھەيرانمەن تۈنۈگۈن شۇنچىلىك غەمسىز،
                                غەلۋىچى  شوخ  ئىدىم تۇلىمۇ  كەپسىز .
                                ئەپسۇسكى شۇ كۈنلەر كەلمەسكە كىتىپ،
                                ئۆزگەردى  بۈگۈنكى  دۇنيا بىر  رەڭسىز .

                                تۇنجى ھەم ئەبەدىي ئۇچىرىماس ئاخىر ،
                                كۈن  بۇيى  شۇ  ئىشلار  كاللامدا زاھىر .
                                تولغىنىپ   ياتتىم   سەۋدايى   كەبى  ،
                                مۇھەببەت  دەردىدە  ئايلاندىم  شائىر .

                                مۇھەببەت كوچىسدا يۈرىمەن ئارىلاپ ،
                                بەردىم  تەسەللىي  ئۆزۈمنى  ئالداپ  .
                                گوياكى   ئايغا  ئوخشاتتىم  سېنى   ،
                                بۇلۇتلار كەينىدىن كۆز قېسىپ ماراپ .



                                      مەرھۇم تاغامغا مەرسىيە
                                    ‹ئىمىنجان ئابىتنى ئەسلەيمەن›

                      بار ئىدى جان -جىگەرلەر قەلبىدە كۆپ ھۆرمەت ساڭا ،
                      سەندە بولغاچقا بىراۋدىن كەم تېپىلغان خىسلەت ساڭا .
                      كەڭ -قورساقلىق،شوخ يۇمۇرچى تولدۇراتتىڭ كۈلكىگە ،
                      ھېچ  ئۆزگەرمەس  بالا  مىجەز  قاملاشقان  زىننەت ساڭا.
                      مىننتىڭ  يوق  ياخشىلىق  كۆپ كەلسە  قۇلۇڭ ھەممىگە ،
                      ئايىمايتتىڭ   ياردىمىڭنى   ئۆرگەرمەس   ئادەت   ساڭا .
                      گۇمانلانماي  ھەم  ئۆزۈڭدىن  ئويلايتتىڭ سەن باشقىنى ،
                      ئەجەل  يىتىپ  ياۋۇز  كېسەل   بوپقالدى  ئاپەت  ساڭا  .
                      نەۋقىران   پەيتىڭدە   كەتتىڭ   ئۇشتۇمتۇتلا  ئايرىلىپ  ،
                      ياشلار   ئاقتى   تاراملايدۇ   قىلالمىدۇق   تاقەت   ساڭا .
                      يىتىم  قالدى   تىرىك  جانلار  ئاتا  مېھرىگە  تەلمۈرۈپ  ،
                      كۈچتۈڭگۈر  قاۋۇل  ئىدىڭ  كەلدى ئەجەپ كۈلپەت ساڭا .
                      قىبىلنامە   بۇلۇپ   قالدى   سېنىڭ   دېگەن   گەپلىرىڭ  ،
                      چۈشۈرمەيمىز  ئاغزىمىزدىن  ھەر  ئىنتىزار ھاجەت ساڭا  .
                      قېرىنداشلار    قاتارىدا    سەن    ئىدىڭسەن    كىچىگى  ،
                      تىنچ  ياتقىن  ئەلۋىدا  ياغقاي  چەكسىز  شۆھرەت  ساڭا .
                      سىياقدىشىڭ    بەئەينى    كۆرسەك    سېنىڭ    بالاڭنى  ،
                      داۋام   قىلىپ   ئىزىڭغا   ياندۇردىغان   پۇرسەت   ساڭا  .



                                              سۆيگۈ  ئارمانى

                                  تۇيۇقسىزلا   ئالدىمدا   بولغانتىڭ   پەيدا ،
                                  بىر    كۆرۈپ    ساڭا     بولدۇم    شەيدا .

                               ئىككىنجى رەت كۆرۈپ سېنى كەتتىم ھۇشىمدىن،
                               بىلمەسمەن  ھىچ  بىر  ئىش كەلمەس  قۇلۇمدىن .

                              قايتا  رەت  كۆرگەندە  سېنى  يۈرەك  باشقىچە  ،
                              پىراقىڭدا     تولغاشتىم     تاڭلار     ئاتقۇچە   .

                              تۆتىنجى  رەت  ئىشقىڭدا  يۈرىمەن  سەرسان  ،
                              پۇچلىنىپ    دەردىڭدە   بولدىمغۇ   ۋەيران    .

                              بەشىنجى   رەت   ھاياجاند  قارايمەن يارغا ،
                              تارتالماي      دەردىنى    قالدىم     بالاغا  .

                             ئالتىنجى  رەت  ئوقيارىڭ  يۈرەكنى  تېلغاپ  ،
                             چەكمەكتىمەن  ھەسرەتنى  ئازابىڭ  قىيناپ   .

                             يەتتىنجى  رەت  بۇلالماي  ئەقىلدىن  ئېزىپ  ،
                             ئورناتتىم    يۈرەككە    ئېسمىڭنى    يىزىپ  .



                                     ئالدىدىڭ سەن ۋاپاسىز يارىم

                                  سۆيگەن ئىدىمغۇ سەن مېنىڭ جانىم  ،
                                  قىينىدىڭ  ئەجەپ  قويمىدىڭ  ھالىم .
                                  مۇپتىلا  قىلدىڭ   مەڭگۈلۈك   ئازاپ ،
                                   ئالدىدىڭ   سەن   ۋاپاسىز   يارىم   .

                                   سېغىندىم   مەن   تاتلىقىم   سېنى  ،
                                  ئەپسۇس   لېكىن  بىلمىدىڭ   مېنى  .
                                  بەرگەن    ئاشۇ    ۋەدىلەر     قېنى  ،
                                  ئالدىدىڭ   سەن   ۋاپاسىز   يارىم   .

                                  مۇھەببتىمىز     بولدى   شۇنچىلىك  ،
                                  ئېلمىدىڭغۇ      مېنى     قىلچىلىك   .
                                  دەپ    باققىنە    ئۆزۈڭ   قانچىلىك  ،
                                  ئالدىدىڭ    سەن   ۋاپاسىز    يارىم  .

                                  كۈلمىدى   جاھان   ھايات   مەزىسىز ،
                                  ئازاپقا   دوستمەن   كۈنلىرىم  سەنسىز.
                                  ئايلاندى   تارىخ   تۈنۈگۈنكى  غەمسىز،
                                  ئالدىدىڭ     سەن     ۋاپاسىز    يارىم .

                                  ئوينىتىپ   نەلەگە   تاشلىدىڭ   قايان ،
                                  ھالىم     بەكمۇ      خاراپتۇر    يامان  .
                                  ئۇنتۇش   قۇلۇمدىن   كەلمەيدۇ  ئاسان،
                                  ئالدىدىڭ     سەن    ۋاپاسىز    يارىم  .

                                 ئۆرتەندى   ئازاپتىن    يىرىلدى   يۈرەك ،
                                 ئايرىلىش    مەندىن   ئۆلۈمدىن   دېرەك .
                                 سۆيەرمەن   قانداقمۇ   سېنىڭدىن   بۆلەك،
                                 ئالدىدىڭ     سەن     ۋاپاسىز     يارىم   .


                             
                                                     خاتىمە

                                            گۈزەل يۇرتۇم  يۇپۇرغا ،
                                           تۇغۇلغان بۇ ئەزىز دىيار .
                                           قوشاق يازغان مۇئەللىپ،
                                           كۆرەش   ئوغلى   ئىليار .
