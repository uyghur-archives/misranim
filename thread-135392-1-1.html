                          رازى قىلغىن ئاتا-ئاناڭنى
                      __ ئەلى تۇرسۇن سىرلىق....

                       كۆرەي دىسەڭ دۇنيادا بەخىت،
                          رازى قىلغىن ئاتا - ئاناڭنى.
                      ئالاي دېسەڭ جەننەتتىن تەخىت،
                          رازى قىلغىن ئاتا - ئاناڭنى.

                         ئاتا - ئاناڭنى قىلماي رازى،
                         بولدۇم دېمە ھەتتاكى قازى.
                         ئاتا - ئانىنى قىلمىغان رازى،
                         بولۇر زىمىستان باھار يازى.

                        مەيلى شاھ بول ۋەياكى مالاي،
                        مەيلى باي بول ۋەياكى گاداي.
                        ئاتا - ئانىسىز قالغانلار تالاي،
                         رازى قىلغىن ئاتا - ئاناڭنى.

                          شۈكىر قىلغىن ئاتاڭ بارىغا،
                          شۈكىر قىلغىن ئاناڭ بارىغا.
                          قولاق سالغىن ئاھۇ زارىغا،
                          رازى قىلغىن ئاتا - ئاناڭنى.

                        دەردى باردۇر بەلىكم يەتكۈچە،
                        دەتلىك يۈرەك پۈتۈن ئەتكۈچە.
                        بۇ دۇنيادىن  ئۇلار  كەتكۈچە،
                        رازى قىلغىن ئاتا - ئاناڭنى.

                       ئاپتورى : ئۆزەم 2014.10.04

                          بىۋاپاغا ئوخشاپ قالماڭلار
                          __ ئەلى تۇرسۇن سىرلىق...

                        تۇرسا ياندا سۆيگەن يارىڭلار،
                        ئۆزگىلەرگە ئاتاپ بارىڭلار.
                      تاشاپ كەتسە سۆيگەن يارىڭلار،
                       بىۋاپا دەپ يېغلاپ قالماڭلار.

                       تەلمۈرگەندە سۆيگەن يارىڭلار،
                       تىترىمەمدۇ يۈرەك تارىڭلار.
                       ئاڭلانمىسۇن ھەرچاغ زارىڭلار،
                       بىۋاپاغا ئوخشاپ قالماڭلار.

                     چېچەك ئاچسۇن سۆيگۈ گۈلۈڭلار،
                     گۈلگە تولسۇن باسقان يولۇڭلار.
                     ئاغرىمىسۇن ئاشىق دىلىڭلار،
                     جۈپ كۆزلەرنى ياشلاپ قالماڭلار.

                     پاك مۇھاببەت قەدرى بىلىڭلار،
                     يالغان ئەمەس دىلدىن سۆيۈڭلار.
                     بەخىتلىك بوپ خوشال يۈرىڭلار،
                     نامەردلەرگە ئوخشاپ قالماڭلار.

                   ئاپتورى : ئۆزەم 2015.01.31

                               كۆنۈپ كەتتىم
                      __ ئەلى تۇرسۇن سىرلىق...

                    مەندىن كېچىپ سۆيسەڭمۇ باشقا،
                      ئاشىق كۆزۈم تولسىمۇ ياشقا.
                      تالاي دەتلەر كەلسىمۇ باشقا،
                     كۆنۈپ كەتتىم سەنسىز ياشاشقا.

                     مەن يوق خۇشال يۈرەلمەيدۇ دەپ،
                     مەن يوق ۋىسال تاپالمايدۇ دەپ.
                      ئويلىمىغىن كۆنەلمەيدۇ دەپ،
                     كۆنۈپ كەتتىم سەنسىز ياشاشقا.

                      يېلىنمايمەن ھەرگىزمۇ گۈلۈم،
                      يورۇق تاڭغا ئايلاندى تۈنۈم.
                      سەن يوقتىمۇ ئۆتكەن بۇ كۈنۈم،
                      كۆنۈپ كەتتىم سەنسىز ياشاشقا.

                     مەيلى كەتسەڭ كەتمە دېمەيمەن،
                     قەدرىمگە ھەم يەتمۇ دېمەيمەن.
                     ھەممىنى بىر رەت قايتا باشلايمەن،
                     كۆنۈپ كەتتىم سەنسىز ياشاشقا.

                   ئاپتورى: ئۆزەم 2014.06.01

                     ياخشىلىق قىل قولۇڭدىن كەلسە!
                     __ ئەلى تۇرسۇن سىرلىق...

                    نامەردلەرگە بولۇپسەن ئۈلپەت،
                    ئاجىزلارغا سالىمىغىن كۈلپەت.
                     ئۆلۈمدىنمۇ كەلگۈچە نۆۋەت،
                   ياخشىلىق قىل قولۇڭدىن كەلسە.

                   ئاجىزلارنىڭ ئاقمىسۇن يېشى،
                   ئېگىلمىسۇن بەندىگە بېشى.
                   تۆكۈلمىسۇن يېتىمنىڭ يېشى،
                 ياخشىلىق قىل قولۇڭدىن كەلسە.

                  بەرگىنىڭگە قىلمىغىن مىننەت،
                  ھەممە بىزگە بىرىلگەن نىممەت.
                  ھاياتىڭدىن ياراتقىن قىممەت،
                ياخشىلىق قىل قولۇڭدىن كەلسە.

                  روھىڭ پاكتۇر ئېمانىڭ بىلەن،
                  تېنىڭ ساقتۇر ئېھسانىڭ بىلەن.
                  يولۇڭ ئاقتۇر ئېنسابىڭ بىلەن،
                ياخشىلىق قىل قولۇڭدىن كەلسە.

                  ئەل يورتىڭغا تىلە ئامانلىق،
                  قىلماس ساڭا ئۇلار يامانلىق.
                  يېتەر دوستۇم ئەمدى نادانلىق،
                ياخشىلىق قىل قولۇڭدىن كەلسە.

                ئاپتورى : ئۆزەم 2014.01.24

                      يۈرەك مەندە بار
                __ ئەلى تۇرسۇن سىرلىق...

              گۈلۈم كەتسەڭ يېراق-يېراققا،
              ئەسلەيدىغان يۈرەك مەندە بار.
              ئۆلسەممۇ گەر سەن ئۈچۈن بىراق،
              كەچمەيدىغان يۈرەك مەندە بار.

               يار يوقتىمۇ پەقەت يار ئۈچۈن،
               يۈرەكتىكى ئاشىق دىل ئۈچۈن.
               سۆيگۈ كۈتۈپ غېرىپ تۈنلەردە،
               ئۆرتەيدىغان يۈرەك مەندە بار.

                 بىر كېتىپلا يېراققا گۈلۈم،
               كەلمىسەڭمۇ مەڭگۈگە جېنىم.
                دېدارىڭنى كۈتۈپ ئۈمۈتتە،
               ساقلايدىغان يۈرەك مەندە بار.

               ئوچۇرۇڭ يوق ئۆتكەن كۈنۈمدە،
               يۈرەك يىغلار سەنسىز تۈنۈمدە.
               سۆيگۈم ئۈچۈن كۈندۈز-تۈنۈمدە،
                يىغلايدىغان يۈرەك مەندە بار.

                سۆيگۈدىكى بۇ دەت پىراققا،
                كەتسەڭمۇ گەر يېراق-يېراققا.
                يار ۋەدەڭنى ئەسلەپ ئەبەدكە،
                چىدايدىغان يۈرەك مەندە بار.

            ئاپتورى : ئۆزەم 2014.03.27

           ئۆتۈپ كەلدىم يېگىتتەك ياشاپ
           __ ئەلى تۇرسۇن سىرلىق....

          ئۆتۈپ كەلدىم يېگىتتەك ياشاپ،
          ئەگمەي ھەرچاغ شەيتانغا باشنى.
          ئۆلسەممۇ گەر رازىمەن لىكىن،
          ئاققۇزمايمەن سەندەككە ياشنى.

          ئۆتۈپ كەلدىم يېگىتتەك ياشاپ،
          تالقان قىلىپ گوياكى تاشنى.
          ئېگىلمەيمەن نامەردكە بىلگىن،
          ئايرىساڭمۇ تېنىمدىن باشنى.

         ئۆتۈپ كەلدىم يېگىتتەك ياشاپ،
         ئۆگەنمەي ياتتىن كۆزلەر بوياشنى.
          قىلىپ كەلدىم قەلبىمگە مىسال،
          ھەر تاڭدىكى نورلۇق قوياشنى.

         ئاپتورى : ئۆزەم 2014.12.31

          تونۇشمىغان بولسام ( 2015 )
       ___ ئەلى تۇرسۇن سىرلىق....

         تونۇشمىغان بولسام ئەگەردە؟
         بەلكىم مەنمۇ يۈرمەستىم يارسىز.
          سەۋەپسىزلا كەتكىنىڭدە سەن،
        يېغلاپمۇ ھەم قالماستىم سەنسىز.

          باشلاپ قويۇپ سۆيگۈڭگە ئۆزۈڭ،
           ئەزدىڭ يەنە نازۇك دىلىمنى.
           چىن يۈرەكتىن توتقانتىم ساڭا،
           يۈرەكتىكى سۆيگۈ گۈلۈمنى.

           گۈللىرىممۇ قايرىلدى سەنسىز،
           قالدى پەقەت دىلىمدا سۆيگۈ.
           سۈيەرمەن دەپ ئويلىغان سېنى،
           چىن يۈرەكتىن ھاياتتا مەڭگۈ.

           لىكىن تەقدىر ئايرىپ بىزلەرنى،
             سۆيگۈمىزنى قىلدىغۇ تامان.
           بارالمايمەن سېغىنسام گۈلۈم،
          بىزمۇ ئەمەدى ئايرىلدۇق ھامان.

         شۇڭا دەيمەن مەن ساڭا شۇنى،
         سەنمۇ ئەمدى ئونۇتقىن مېنى.
         بەخىتلىك بول يېغلىما گۈلۈم !
         مەنمۇ ئەمدى ئۇنتاي ھەم سېنى.

     ئاپتورى : ئۆزەم .2015.02.07

                 يالغان مۇھاببەت
             __ ئەلى تۇرسۇن سىرلىق...

           ئۆزۈڭ كۆيگەن ئۆزۈڭ باشلىغان،
           ئۆزۈڭ سۆيگەن ئۆزۈڭ تاشلىغان.
           دەت - ئازاپتا يۈرەك ياشلىغان،
           خەيىر ئەمدى يالغان مۇھەببەت.

           بۈگۈن بىلسەم كۆڭلۈڭنى جانان،
           ئىكەن ئەسلى نېيتىڭمۇ يامان.
            يۈرەكىمدە ئەلەملەر ھامان،
          خەيىر ئەمدى يالغان مۇھەببەت.

           كېرەك ئەمەس ئايدەك روخسارىڭ،
           كېرەك ئەمەس مالۇ - دۇنيارىڭ.
           پەقەت كېرەك ماڭا ئىمانىڭ !
          خەيىر ئەمدى يالغان مۇھەببەت.

           كۆيۈپتىمەن ئايدەڭ يۈزۈڭگە،
           ئىشەنمەيمەن ئەمدى سۆزۈڭگە.
           كۆرۈنمىگىن قايتا كۆزۈمگە،
          خەيىر ئەمدى يالغان مۇھەببەت.
