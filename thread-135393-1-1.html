                                          خەير تۇنجى مۇھەببەت ‹شئېر›
                                             ئاپتورى : ئىليار كۆرەش

                                            شۇنچە سەبىي ساددىنى ،
                                            قويماي  قىلچە  ھالىنى  .
                                            قىلدى  يىغلاپ  نالىنى  ،
                                            خەير تۇنجى مۇھەببەت .

                                            تۇنجى كۆرۈپ شۇ زامان ،
                                            بولدى ھالىم بەك يامان .
                                            سەۋەبكارى سەن ھامان ،
                                            خەير تۇنجى مۇھەببەت .

                                            گۈزەللىكىڭ لال قىلىپ ،
                                            يۈرىكىمگە ئىز  سېلىپ .
                                            تۇيدۇرمايلا بىر ئىلىپ ،
                                            خەير تۇنجى مۇھەببەت.

                                            يۈرەككە  ئېتىپ  چەنلەپ ،
                                           ئالداندىم  شېرىن  بەزلەپ .
                                          شەيتانلىرىڭ بېسىپ غەپلەت،
                                           خەير  تۇنجى  مۇھەببەت .

                                            ئىزدەپ تېپىپ مىڭ تەستە،
                                            سوۋغا  قىلدىم گۈلدەستە .
                                            قاراپ  قالدىم  ھەۋەستە  .
                                            خەير  تۇنجى مۇھەببەت  .

                                            دېدىم  ساڭا  سۆيىمەن ،
                                            مەڭگۈلۈككە  يۈرىمەن  .
                                            ۋېسال  پەيزى سۈرىمەن،
                                            خەير تۇنجى مۇھەببەت.

                                           سۆيگەنمەن تۇنجى بۇلۇپ ،
                                           قەلىبلەرگە تاتلىق قۇنۇپ  .
                                           يۈرۈكۈمنى قاتتىق قۇنۇپ ،
                                           خەير  تۇنجى مۇھەببەت  .

                                           تاشلاپ كەتتى كارى يوق ،
                                           بۇ دەردىمگە  چارى  يوق .
                                           ئەمدىلىكتە  يارى   يوق  ،  
                                           خەير  تۇنجى  مۇھەببەت .

                                           دىيەلمىدىم  بىرمۇ  گەپ ،
                                           تەقدىرىممۇ  مۇشۇ  دەپ .
                                           ئارزۇلىرىم  بولدى  چەپ ،
                                           خەير  تۇنجى  مۇھەببەت .

                                           تارتماقتىمەن  دەردىنى ،
                                           ئالالمىدىم   ئەنتىنى    .
                                           رەھىم قىلماي بەندىنى  ،
                                           خەير تۇنجى مۇھەببەت .  

                                           سۈرىتىڭگە  ئاھ ئۇرۇپ ،
                                           يۈرەكلىرىم  چاك بۇلۇپ.
                                           ھەسرەتلەردە قان تۇلۇپ،
                                           خەير  تۇنجى  مۇھەببەت.

                                           قانداق قىلاي نە سەۋەب ،
                                           ئايرىلدىمغۇ مەن ئەجەپ.
                                           باسالمىدىم  دەر  غەزەپ،
                                           خەير  تۇنجى  مۇھەببەت.

                                           يىتەلمىدىم   ئويلىنىپ  ،
                                           كېچە-كۈندۈز تولغىنىپ.
                                           قەلبىممۇ تەڭ بۇلۈىنىپ ،
                                           خەير تۇنجى مۇھەببەت .  

                                           يالۋۇرمايمەن مەن ساڭا،
                                           ساۋاق  بولدى  بۇ ماڭا .
                                           ئۆتنە  ئالەمدۇر  شۇڭا ،
                                           خەير تۇنجى مۇھەببەت.

                                           تاشلاپ  ئازاپ قوينىغا ،
                                           كەلمەس ھەرگىز ئورنىغا.
                                           تەشنا  بولدۇم  دورىغا ،
                                           خەير تۇنجى مۇھەببەت.

                                          بىمەنە سەۋەب ئويدۇرۇپ،
                                          دەرت - بالاغا قويغۇزۇپ .
                                          مۇھەببەتتە   تويغۇزۇپ  ،
                                          خەير تۇنجى مۇھەببەت  .

                                           ۋېسالىڭغا   يىتەلمەي  ،
                                          سەندىن پەقەت كېچەلمەي.
                                          ئايرىلساممۇ  كىتەلمەي  ،
                                          خەير تۇنجى مۇھەببەت .

                                          خازان بولدى غۇنچىسى ،
                                          ماڭا  قىلغان سوۋغىسى .
                                         ئۇنتۇش قىيندۇر تۇنجىسى،
                                          خەير  تۇنجى مۇھەببەت .

                                          تاپالمىدىم   بەختىمنى ،
                                          قايتا رۇسلاپ قەددىمنى.
                                          ئۇنتالمىدىم مەن ئۇنى ،
                                          خەير تۇنجى مۇھەببەت .
