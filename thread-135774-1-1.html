                                        مەن شۇنداق بىر نوچى بالىمەن ‹ساتىرا›
                                               ئاپتورى : ئىليار كۆرەش
       (مەزكۇر ئەسەر تىيىپچان ئىلىيوفنىڭ ‹مەن شۇنداق بىر ئوغۇل بالىمەن›ناملىق شېئىرىگە تەخمىس)
                                           (ساختىپەز سىتودىنتلارغا ھەجۋىي)

   
                                         
                                       مەن ئۆزۈم شۇنداق بىر نوچى بالىمەن ،
                                       خامېلىئوندەك  ھەرخىل  بۇلۇۋالىمەن .
                                       كۆزلەپ  پايدىنى  ھەر  يوللار  بىلەن  ،
                                       تويماستىن    ھەممىنى   يۇلۇۋالىمەن .

                                      دەرس    دېسىلا    بېشىم     ئاغرىپ ،
                                      ئۇخلاپ  سىنىپتا  خورەكنى  تارتىپ  .
                                      تاپشۇرۇققا  ياۋاشنى  قىلمەن كاتىپ ،
                                      كۆرەڭلەپ     ئاقسا      يۈرىۋالىمەن .

                                      كۆرگەندە مۇئەللىمنى يۈركۈم يايراپ ،
                                      ماختاپ ئۇچۇرىمەن  گەپتىلا  جايلاپ .
                                      توختىماي  ئىزدەپ كۈنلەپ - ئايلاپ ،
                                      كەتمەستىن   ئۆيىدىلا   يېتىۋالىمەن  .

                                      راداردەك بالىسىنى بايقاپ يىراقتىن ،
                                      قارا  دىمەستىن  بالىڭىز   ‹ئاقتىن› .
                                      يوقتۇر  دۇنيادا  بۇنداق  ئوماقتىن  ،
                                      ماڭقىسىغا  چوككىدە  سۆيىۋالىمەن .

                                      مۇئەللىممۇ قايىلدۇر سادىقلىقىمدىن ،
                                      تەخسىكەش ئېلىمىدە قابىللىقىمدىن .
                                      ياغلاشقا   گېلىنى   ماھىرلىقىمدىن ،
                                      مۇئەللىمگە   ئەتېۋار   بۇلۇۋالىمەن  .

                                      قەلەمسىز ئېمتىھان بىرىشكە يامان ،
                                      لاياقەتسىز  پەننى يۇقىتىش ئاسان .
                                      مۇكاپات پۇللىرى  تەييادۇر ھامان ،
                                      سەمىرىپ  بارا - بارا چېڭىۋاتىمەن .

                                      چۈشسىلا نەزىرىمگە قىزلارنى كۆرۈپ ،
                                      ئازدۇرۇپ ئۇلارنىڭ پەيزىنى سۈرۈپ  .
                                      زېرىكسەم كونىسىنى تاشلاپ يۈرۈپ  ،
                                      مەكتەپتىكى  گۈللەرنى  ئۈزۈۋالىمەن  .

                                      زىياندىن  يىراقمەن   قىلىپ  پەمىنى ،
                                      تەجىرىبەم  مولدۇر  بابلاپ ھەممىنى .
                                      ئۆرۈيمەن  يانچۇقتىن  تېتىپ تەمىنى ،
                                      چاندۇرماي    ئاستا    شۈلۈۋالىمەن  .

                                      دەسلەپتە  يىقىن  دوستۇدەك  قاراپ ،
                                      تاشلىسا   پايدىنى  ئىتتەك  غاجاپ  .
                                      يوقالسا  قىممىتى  شۇئانلا  ئاجاپ   ،
                                      چېچەنلىگىمدىن قاتتىق كۈلۈۋالىمەن .

                                      تۈلكىدەك  ماھارەت  ماڭىلا  تالىق  ،
                                      كۈچۈكلەپ غوجامدەك بابلاشقا باغلىق .
                                      قۇرۇقداپ   يىنىنى  قىلىمەن  چاغلىق ،
                                      قورقماستىن   ھارامنى   يۈتۈۋالىمەن .

                                      كۆرسەم  ياۋاشنى  شېلتىڭ  ئېتىپ  ،
                                      ئاكىسى  كەلگەندە  بەدەر  قېچىپ  .
                                      قۇرۇق     ھەيۋىنى     تاراقشىتىپ  ،
                                      يولۋاس    تېرىسىگە   ئورنۇۋالىمەن .

                                      ئېشىپ كەتسە نەتىجە ياخشىلارنىڭ ،
                                      بىرىمەن  ئەدىپىنى  ھەم  ئۇلارنىڭ  .
                                      سەۋەبچىسى  مەندۇر  ھەر بالانىڭ ،
                                      بۇشىتىپ    ئورنىغا    ئۆتۈۋالىمەن  .

                                     پۇل دېسە ۋىجداننى قورقماي  سېتىپ  ،
                                     ئۇنىڭ ئۈچۈن ئېشەكتەك ئىشلىدىم چېپىپ.
                                     ئەپتى - بەشىرىم  ھەتتا  قالدى ئېچىلىپ ،
                                      يىقىلسام     قايتىدىن     تۇرۇۋالىمەن   .

                                     بۇ  خىل   ياشاش   مېنىڭ  شۇئارىم  ،
                                     ئىنسانچىلىقتا    ياخشى    قورالىم    .
                                     قىلىمەن  قەسەم  شەك  يوق گۇمانىم ،
                                     ياشاشنىڭ     ئىپىنى     بىلىۋالىمەن  .

   
                                               مەنبە : ئۆز قەلىمم
