                  سۈيگۈ ئ‍ۇۋالى
                       (شېئىر )

       بىرجۇپ ئ‍اشىق بولغانىكەن ،
       بەخىت ئ‍ىچىدە ئ‍ۆتكەنىكەن ،
        ئۇلارنىڭ بۇ بەختىگە ،
       ئ‍ايمۇ ھەسەت قىلارىكەن ،

           بىركۈنى دەپتۇ يىگىتى ،
         تاپقىن باشقا بىرىنى ،
         سىلكىپ تاشلاپ كىتىپتۇ،
           ئ‍ۈزۈپ سۆيگۈ يىپىنى ،

            ھەيران قاپتۇ قىزچاقمۇ ،
           ئ‍اڭلاپ ئ‍ۇنىڭ گېپىنى ،
              تۇرۇپ قاپتۇ جايىدا ،
            يوقۇتۇپتۇ ئ‍ېسىنى ،

          كۆپ ئ‍ويلاپتۇ سەبىي قىز،
          خاتا قىلدىم نېمىنى ،
          ئ‍ازار بېرىپ دىلىمغا ،
        ئ‍الدىڭ يۈرەك قېنىنى ،

         ئ‍ەسلى كەتكەن يىگىتنىڭ ،
          باركەن باشقا سۆيگۈنى ،
         شۇڭا تاشلاپ كېتىپتۇ  ،
          ۋەدىلەشكەن قىزىنى ،

               خۇشال باردى بۇ يىگىت ،
            يىڭى سۆيگۈ يېنىغا ،
            كۆرۈپ باشقا بىرى بىلەن  ،
               تۇرۇپ قاپتۇ جايىدا ،

           ئاچچىقى كېلىپ سوراپتۇ ،
             ئ‍ۇ كىم زادى ئ‍ىيتقىنا ،
            چېچىلىپتۇ بۇ قىزمۇ ،
             قاراپ باقە ھالىڭغا ،
      
            كىمتىڭ ئ‍ۆزۈڭ ۋارقىراپ ،
           مىنى سوراق قىلغۇدەك ،
             يات ئ‍الدىدا چالۋاقاپ ،
             ماڭا دەرس ئ‍ۆتكىدەك ،

             ئ‍ەمدى ساڭا دەپ قوياي ،
             بولالمايسەن ماڭا يار ،
           ۋاقتىدا كەت يىنىمدىن ،
         بولماي دىسەڭ ئىتتەك خار ،

          بۇنى ئ‍اڭلاپ يىگىتنىڭ ،
            ۋىجدانى بەك قايناپتۇ ،
         ئاخىرقى قېتىم سوراپتۇ ،
          سۆيىمەن دېگىنىڭ يالغانمۇ ،
   
               ئ‍ۇ قىز يەنە پەرۋاسىز ،  
             مۇنداق جاۋاپ بىرىپتۇ ،
          ئ‍ۇ دىگەننىڭ بىر قىلتاق ،
           خوشمۇ دىمەي كېتىپتۇ ،

          كۆپ يىغلاپتۇ بۇيىگىت ،
          تاپالماي ھىچ ئېسىنى ،
           بىر قول كېلىپ سۈرتۈپتۈ ،
           بۇ پۇشايمان يېشىنى ،

           كۆتىرىپلا بېشىنى ،
              خۇشال بوپتۇ قەۋەتلا ،
             لىكىن بۇ چاغ ئ‍ۇ قىزچاق ،
         ماڭغان ئىكەن يىراققا،

          توختاڭ دەپتۇ  بۇ يىگىت ،
          قوغلاپتۇ قىز كەينىدىن ،
               سوراپتۇ كۆپ كەچۈرۈم ،
          قىلغان ئ‍اشۇ ئىشىدىن ،

               يالۋۇرۇپتۇ قايتىلاپ ،
            يۈرەيلى دەپ قايتىدىن ،
            قىز مۇ ئ‍اخىر كەچۈرۈپتۇ ،
           باشلايلى دەپ يېڭىدىن ،
  
         شۇندىن كېيىن بۇلارغا ،
           ھارام بوپتۇ ئ‍ايرىلىش ،
            بولغاچ ئ‍ۇلاردا ساداقەت ،
        نىسىپ بوپتۇ توي قىلىش ،

    بۇ ئەسەرنى ھازىرقى زاماندىكى ياش قىز ئ‍وغۇللىرىمىز ئ‍ۈچۈن يازدىم،  ئ‍ەڭ ياخشىسى ئ‍ۆزىڭىزدە بارىنى قەدىرلەڭ ،سىزنى ئازدۇرغاننڭ تۆپىدە ئ‍ۇنىمۇ ئازدۇرغان ئادەم بارلىقىنى ئۇنۇتماڭ ،

    ئ‍ەزمەت ئابدۇرەھىم دىلشەمى
