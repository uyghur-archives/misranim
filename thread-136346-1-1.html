                                             ئايەم
                             شادلىققا چۈمۈلەر بارچە كائىنات،
                             سائادەت ياغدۇرۇپ كەلگەندە ئايەم.
                            ئاداۋەت يۇيۇلۇپ مىھىرلىك باقار،
                            قايتىدىن دوسىتلىشىپ ھەممىلا ئادەم.

                           يىگىتلەر شادلىقتا چۈشەر ساماغا،
                           ياڭرىتىپ ھەر جايدا ھېيىت ناۋاسى.
                          ھەممىنىڭ چىھىرىدە كۈلكە تەبەسسۇم،
                          دىللاردىن ئاڭلىنار سۆيگۈ ساداسى.

                          سەبىلەر قەلبىمۇ شادلىنار شۇ دەم،
                          ئالغاندا چوڭلاردىن نىسىۋىسىنى.
                          بۈيۈكلەر كۆڭلىمۇ ئەمىن تاپىدۇ،
                          كۆرگەندە نەۋرىسىنىڭ ئەركىلىشىنى.


                         ئاتىلار تەلمۈرەر پەرزەنىت يۇلىغا،
                        كەل ئېچىپ باغرىنى كەلگىن بالام دەپ.
                       چوڭ ئۆيگە جەم بۇلار بارچە قېرىنداش،
                       مىھىرىبان جان ئانام بۈيۈك ئاتام دەپ.


                                    سۈيەلمىدىڭ
                      بىر سىنىڭ دىگەنتىڭ ۋەدىلەر بىرىپ،
                     ئۈزۈڭگە لال قىلدىڭ قەلبىمگە كىرىپ.
                    قېنى يار ۋەدىلەر سۈيەلمىدىڭغۇ؟
                    سۆيدۇڭغۇ باشقىنى كەتتىڭ يۈز ئۆرۈپ.

                   سېغىنىپ مىنىڭدەك ھېچ تۈنەلمىدىڭ،
                  بىلىمەن ئازاپقا سەن كۈنەلمىدىڭ.
                  تەلمۈرگەن ئىدىم مەن چىن ئەقىدەڭگە،
                ۋە لىكىن مىنىڭدەك سەن سۈيەلمىدىڭ.

                مەڭگۈلۈك سۆيگۈگە بەردىڭ خاتىمە،
                ئۆتمۈشمۇ قالدۇردى شىرىن ئەسلىمە.
                بىرىشكەن ۋەدىلەر مەڭگۈلۈك ئىدى،
                يۈرۈكۈم ئىزىلدى بولدى مىڭ پارە.

                خەيىر-خوش نىگارىم ئەمدى كىتىمەن،
                ئازاپقا ئۆلمەيمەن كۈلۈپ ئۆتىمەن.
                سۆيسەممۇ شۇنچىلىك سەن سۈيەلمىدىڭ،
               مەنمۇ ھەم ئۆزگىدىن بەخىت كۈتىمەن.

                                 خىيالىمدىكى مۇھەببەت
                         بىر كۈنلەر ئۇچراتسام مەن دىلبىرىمنى
                         ئېيىتسام دەردىمنى قەلىپ سىرىمنى.
                         تاھىرى سۆيگەندەك زۆھرەنى شۇنداق،
                          بىرەتتىم بارىمنى ھەتتا جېنىمنى.

                         ئوينايىتتىم يار بىلەن ئاپئاق قارلاردا.
                         كۈتەتتىم كىلەر دەپ گۈزەل باغلاردا.
                        باسقاندا باغرىمغا يۈرەك پارەمنى،
                        ئۇنتاتتىم ھەممىنى بەخىت قوينىدا.

                        سۈيەتتىم دىلىمدىن مەڭگۈلۈك ئەبەت.
                        ساقلايىتتىم بىر ئۆمۈر قىلماي خىيانەت،
                        ئەپسۇسىكى ئارمانغا چۇشلۇق دەرمان يوق،
                       ھەي....................
                       بۇ پەقەت خىيالىمدىكى مۇھەببەت.

                                      كەتتىڭ
                         شادلىناتتىم كۆرگەندە سىنى،
                         كۈرەلمەيمەن ئەمدى سە كەتتىڭ.
                        بەخىتلىكتىم سۆيگەندە سىنى،
                       سۈيەلمەيمەن ئەمدى سەن كەتتىڭ.

                          تەلمۈرەتتىم خۇما كۈزۈڭگە،
                         نەزەرىڭنى مەندىن ئېلىپ كەتتىڭ.
                        ئىشىنەتىتىم شىرىن سۈزۈڭگە،
                       يۈرەگىمگە دەرىتنى سەپ كەتتىڭ.

                        قانمىغانتىم تېخى مىھرىڭگە.
                       مىنى يالغۇز تاشلاپلا كەتتىڭ.
                       ئاشىق ئىدىم ئىللىق چىھرىڭگە.
                      كۆزلىرىمنى ياشلاپلا كەتتىڭ.


         مەنبە :قەلىمىم
