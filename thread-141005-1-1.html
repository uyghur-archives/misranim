نىمەئۈچۈن چىش يىلتىزىنى داۋالاپ بولغاندىن كىيىن چىشنى قاپلاش تەكلىپى بېرىلىدۇ
يىلتىزكانىلىنى داۋالاشنى قۇبۇل قىلغان چىشلار ئاساسەن چىرىگەن،يىلىكى ياللۇغلانغان ياكىتاشقى كۈچنىڭ تەسىرىدە زەخىملەنگەن چىشلار بۇلۇپ،كەمتۈك ئورنى چوڭىراقبۇلىدۇ.دوختۇر يىلتىز كانىلىنى داۋالاپ بولغىنى بىلەن،چىشنىڭ تۈزىلىشىنى كەلتۈرۈپچىقارغان توقۇلمىلار ئىنتايىن ئاجىزلىغان بۇلىدۇ.بۇ ئەھۋال ئاسانلا چىشنىڭ تاجىسىدەز كىتىش ياكى يىلتىزى دەز كىتىپ پاچىلنىپ چىشنىڭ ئىقتىدارى يوقىلىدۇ.بۇ ۋاقىتتابۇ چىشنى قاپلماقتىن باشقا ئامال يوق.
ئۇنىڭئۈستىگە يىلتىز كانىلىنى داۋالاش دورا ۋە ئەسۋاپلارنىڭ ياردىمىدە كانال ئىچىدىكىچىقىرىندىلار ۋە باكتىرىيەلەرنى ئەڭ يۇقۇرى سەۋىيىدە  دىزىنفىكسىيە قىلىپ پاكىز تازلاش مەقسەتقىلىنىدۇ،ئەلۋەتتە بۇ داۋالاش جەريانى دوختۇرنىڭ ئەستايىدىل زىھىن قۇيىشى ئاستىدابولسا، بۇ چىشنىڭ يىلتىز كانىلىدا قايتا ياللۇغلىنىش ئالامىتى بولمايدۇ.ئەمما بۇچىشنىڭ ئوزۇقلۇق ۋە ئوكسىگىن يەتكۈزىدىغان مىلىك توقۇلمىسى بولمىغاچقا بۇ چىشتاچايناش ئىقتىدارى ۋە مۇستەھكەملىكىدە چىكىنىشلەر بۇلىدۇ.شۇنىڭ ئۈچۈن ئادەتتە چىشدوختۇرى يىلتىز كانىلى داۋالاشنى قۇبۇل قىلغان چىشلارنى قاپلىتىش تەكلىبىنىبېرىدۇ،بۇنداق بولغاندا بۇ چىشنىڭ ئىشلىتىش سۈپىتى ۋە ئۆمرى ئۇزۇن بۇلىدۇ.
مەنبە:يىپەك يولى شىنجاڭ چىش دوختۇرلىرى كۇلۇبى
