تۇرمۇش ئۇنداق ئاددى نەرسە ئەمەس ، ھەر ئادەمنىڭ خارەكتىر مىجەزى ئوخشاش بولمايدۇ ، ھەممە ئادەمنىڭ ھېس تۇيغۇلىرى ئوخشاش بولىۋەرمەيدۇ ، ئەر خوتۇننىڭ بىرسىنىڭ قەلبى ئالىيجاناپ پىداكار كەلسە يەنە بىرى يۈسسىز چىقىپ قالىدىغان ئىش بار ، ئەر كىشى چوقۇم يوقىردا  تىما ئىگىسى تىلغا ئالغان تىما ئىگىسىنىڭ قەلبىدىكى قىممەت قاراشلىرنىڭ ھەممىسىگە ئىگە بولشى شەرت ئەمەس ياكى ئادەم چوقۇم ھەممە دە يىتىشكەن تولۇق تاكاممۇللاشقان  سۇمباتلىق بولىشىمۇ ناتايىن ھەم بۇ تۇرمۇشنىڭ ئالدىنقى شەرتى ئەمەس .     ، ئايال كىشى باشقا بىر دۇنيا ، ئەر كىشى مۇ ھەم باشقا بىر دۇنيا ، شۇ سەۋەپ بىرسى بىرسىنىڭ كەمچىللىكىنىلا كۆرىدۇ ھەم ئۇنى ئىسىدە ساقلايدۇ  ئەمما ئارتۇقچىلىقى ئىسىدە يوق ، شۇ سەۋەپ جەڭگى جىدەل ئەۋجىگە چىقىپ ھېچ بىرىگە قايىللىق يول قويۇش بولماي ، تۇرمۇشتىكى بۇ مۇرەككەپچىلىك چوڭلارنىڭ قولىقىغا بارىدۇ ، بۇنداق ئەھۋالدا چوڭلار مەسىلىگە چوڭ بولغاندىكىن كىچىكلەرنىڭ بۇ نادانلىقىغا قەلىپنى ئالىجاناپ تۇتۇپ سوغۇققانلىق بىلەن مۇئامىلە قىلىپ ئىشنى توغرىلىق بىلەن ھەل قىلىشى كىرەك .  ئەكسىچە شەخسىيەتچىلىك بىلەن ئۆزنىلا ئويلىسا ھەر ئىككىلى تەرەپ پاراكەندىچىلىككە ئۇچرايدۇ ، . يوقىردا تىما ئىگىسى تىغا ئالغان تىما ئىگىسىنىڭ قەلبىدىكى شەرتلەر ئىنسانى ۋە شەيتانى تۇيغىدىن كەلگەن ، تىما دا دىيىلگەن شەرتلەر بەك ئاددى ،  بەك ئازكەن ، شۇلا بولسا كۇپايىمۇ ؟ ، مېنىڭچە ھەقىقى جەلىپ قىلارلىق ئەر ۋە جەلىپ قىلارلىق ئايال ئىتىقادىغا سادىق ئەقىدىسى مۇستەھكەم بولغىنى ، بولدى مۇشۇ بولسىلا كۇپايە ، ھەزرىتى پاتىمە ئانىمىز بىر يۇتۇپ زەم-زەم گە رازى بولغان ، ئۇممۇ سەلىمە ئۇنىڭدىنمۇ ئالىجاناپ ...... ئىتىقادىغا سادىق ئەر ۋە ئىتىقادىغا سادىق ئايال ھەممىگە  بەك مەسئۇليەتچان بولىدۇ ، بۇنىڭغا گەپ كەتمەيدۇ ، تۈرگە ئايرىشنىڭ ھاجىتى يوق ،   ھەقىقى بەخىتلىك ئائىلە ، ھەقىقى بەخىتلىك ئەر ، ھەقىقى بەخىتلىك ئايال  دىگەن مۇشۇ  .
