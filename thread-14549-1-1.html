مىسرانىم تورى تور ئىچىلغاندىن بۇيان ئىزچىل باشقا تور بىكەتلەر ئالدىدا مىڭىپ كەلمەكتە، شۇڭا بىز مىسرانىمنىڭ بارلىق ئەزالىرى باشقۇرغۇچىللىرى ئەڭ مۇھىمى مىسرانىم ئەپەندىنىڭ تىرىشچانلىقى بىلەن ،بۇ مۇنبەربى ئۇيغۇر تور بەت ساھەسى بويىچە ئالدىنقى قاتارغا ئۆتكۈزۈش تىرىشچانلىق كۆرسىتىشىمىز كىرەك ،مىسرانىم ھەممەيلەننىڭ ئورتاق ئائىلىسى .
شۇڭلاشقا مىسرانىم ئەپەندى ،ۋە مىسرانىم باشقۇرغۇچىللىرىغا كىچىككنە تەكلىپ بەرمەكچىمەن .
بىرىنجىدىن باشقۇرغۇرچلىرىمىزنىڭ مىسرانىم خەلقى بىلەن ئارىلىشىشنى قويۇقلاشتۇرسا ،ئۇلار خاس تىما تەستىقلاپلا قالماي ،ئىمكان بولسا ۋاقىت چىقىرىپ نادىر تىما ،ياخشى ئىنكاسلارنى يوللاپ تۇرسا خەلق بىلەن بولغان ئالەقىنى كۈچەيتسە دىمەكچى ،مۇنداقچە ئىيىتقاندا تەھرىھ ئىشخانىسىدىن چىقىپ ھەرقايسى ئەزالارنى چۈشىنىشنى قويۇقلاشتۇرسا .
ئىككىنجىدىن مىسرانىم باشقۇرغۇچىلىرى ئۆزىنىڭ رولىدىن تولۇق پايدىلىنىپ تۇيۇقسىز يۈز بەرگەن ۋەقەلەرنىڭ ئۈنۈملۈك ئالدىنى ئالسا ۋاقتىدا تىزگىنلىسە ،تور دۇنياسىدا خىلمۇخىل ئادەملەر بولىدۇ ،يامان غەرەز،ياكى تىللىشىپ ئىنكاس يىزىشلارنى كۆرگەندە ،ۋاقتىدا ئاگاھلاندۇرۇش بىرىش ،ياكى چەكلەش كۈچىنى ئىشلەتسە .
ئۈچىنچىدىن تىما تەستىقلاشقا ئالاھىدە دىققەت قىلسا ،دۆلەت قائىدە قانۇنلىرىغا خىلاپ بولمىغان تىمىلارنى ۋاقتىدا تەستىقلىسا ،ئامال بار مۇنبەرگە پايدىلىق ،ھەممەيلەن ئورتاق مەنپەئەتلىنەلەيدىغان تىمىلارنى ئالدىن ئويلاشسا ، تىمىنى بىر قۇر كۆرۈپ ئالاھىدە ئىملا خاتالىقى بولمىسا ،بەزى ئازراق ئىملا خاتالقلىرىنى تۈزەپ مۇنبەرگە چىقارسا ،
تۆتىنجىدىن نادىر تىمىلارنىڭ رەڭدارلىقىنى ئاشۇرسا،  يەنى نادىر تىمىلارنىڭ كۆلىمىنى كىڭەيتسە بەلكىم مىسرانىم مۇنبىرى ئەدەبىيات -سەنئەت توربىكىتى بولغانلىقى ئۈچۈن شۇ مەزمۇندىكى نادىر تىمىلارنى تەستىقلىشى مۇمكىن ، ئاساستا مىسرانىمنى ئۇنۋىرسال تور بىكەتكە ئايلاندۇرۇش ئۈچۈن نادىر تىمىلار رەڭدارلىقىنى ئاشۇرۇپ ھەممەيلەننىڭ كۆزىدىن ئوت چىقارغىدەك مۇنبەر قىلىپ قۇرۇپ چىقساق ،بۇ ئىشتا مىسرانىم ئەپەندى باشقۇرغۇچىلارغا يىشىل چىراغ يىقىپ بەرسە بولىدىغان ئوخشايدۇ .

ئاخىرىدا بارلىق مىسرانىم ئەزالىرى باشقۇرغۇچى ۋە مىسرانىم ئەپەندىنى قوللاپ ،مۇنبەردە ئامال بار ئۇرۇش تالاش قىلماي (مەندىن تارتىپ)ئوخشاش پىكىرلەدە بىرلىككە كىلىپ ،ئوخشىمىغان پىكىرلەردە سىلىق مۇنازىرلىشىپ مۇنبەرنىڭ قايتا گۈللىنىشى ئۈچۈن بىر كىشلىك تۆھپە قوشسا .

تۆۋەنچىلىك بىلەن ،مىسرانىم ئەزاسى ئەلزامان .



قەشقەرنىڭ كۆزيىشى مىنىڭ كۆزيىشىم، قەشقەرنىڭ چۆكۈشى مىنىڭ قىسمىتىم .
