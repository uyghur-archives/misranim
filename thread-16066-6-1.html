تورداشلار گەرچە تىما مەزمۇنىدىن چەتنەپ كەتكەن بولسىمۇ خىلى ھەق گەپ قىپتۇ.
باشقۇرغۇچى ۋە ئالىي ئەزا دىگەنلەرنىڭ،مۇنبەردە باشقۇرغۇچى بولوشقا دىپلۇم ۋە باشقا نەرسىلەر كەتمەيدۇ دەپ جار سىلىپ ۋارقىرغۇچىلارنىڭ ساپاسى ۋە بىلىمى مانا مەن دەپ بۇ تىمىغا يازغان ئىنكاسلىرىدىنلا كۆرۈنۈپ تۇرۇپتۇ.
تىما نىمە دەۋاتىدۇ، ئۇلار نىمە دەۋاتىدۇ.بىرسى كۆڭلىگە ياقمىغان پىكىردىن بىرنى قىلىپ قالدىمۇ تىما مەزمۇنىدىن چەتنەپ ئۇنىڭغا رەدىيە بىرىشكە ئۆتۈپ كەتكەن. ئۆزلىرىنىڭ پىكىرىنىلا راس دەۋاتقان.باشقىلارنى چۈشۈنۈش،ھۆرمەت قىلىش،يول قويوش،كەڭ قورساق بولوش دىگەندىن ئەسەرمۇ يوق...
يۇقىردا تالاش تارتىش بولىۋاتقان مەسىللەر گەچە بۇ تىما بىلەن قىلچە مۇناسىۋەتسىز بولسىمۇ، مەنمۇ جاپا تارتىپ يازغان تىمامدىن چەتنەپ ئۆز كۆز قارىشىمنى قوياي،چۈنكى بۇ مىنىڭ ئەجرىمگە چۈشۈپتۇ ئەمەسمۇ.
مۇنبەردە باشقۇرغۇچى بولوش قانداقتۇر بىر شەرەپ ئىشمۇ ئەمەس، ھەم قانداقتۇر بىر نۇمۇس ئىشمۇ ئەمەس،مۇنبەردە باشقۇرغۇچى بولوشقا چوڭ چوڭ كىنىشكىلەر،ئالىي مەكتەپنى پۈتتۈرگەن دىپلۇملارمۇ كەتمەيدۇ،ئەمما نىگىزلىك مەسىلە شۇكى قانداقلا بولمىسۇن بۇ بىر باشقۇرغۇچى،مۇنبەرنى،كۆپچىلىكنى يىتەكلىگۈچى.كۆپچىلىكنىڭ مۇنبەرگە سەرىپ قىلغان ئەجرىگە،زىھنىگە،قىممەتلىك ۋاختىغا ئىگە بولغۇچىلار دەل شۇلار.بۇنداق دىيىش ئارقىلىق مۇنبەر قۇرغۇچىلارنىڭ شۈنچە مىڭ بىر جاپادا بىلىپ بولمايدىغان بىسىم،جاپا مۇشاقەتلەر بىلەن مۇنبەرنى بەرپا قىلغان ئەجرىگە،بىز قىرىنداشلارغا نەپ يەتكۈزۈش،بىزلەر ھەم كەلگۈسىمىزگە نىسبەتەن بىر كىشىلىك ئجابىي رولىنى ئۆتەش قاتارلىق ئارزۇ ئارمانلىرىنى يوققا چىقارماقچى ئەمەسمەن.ئۇلانىڭ  ئەجرىنى زىيادە قىلسۇن ئىگىمىز ئاللاھ.
ئەمما مەن يۇقىردا دەپ ئۆتكەندەك،مۇنبەر ئارقىلىق كۆپچىلىككە نەپ بىرىپ،ئۇلۇغۋار مەسئۇلىيەتنى ئۈستىگە ئالغىلىمۇ،ئۆزى ھەم كۆپچىلىككە نەپ بەرگىلىمۇ بولىدۇ،كۆپچىلىكنىڭ ئەجرىگە،ۋاختىغا خىيانەت قىلغىلىمۇ بولىدۇ.
شۇ بىر ھەقىقەتمۇ بىزگە ئايانكى مۇنبەر قۇرغۇچىلار،باشقۇرغۇچىلا بىزلەر ئۈچۈن
تىنىپ تاپماي ئەجىر قىلىدۇ،جاپا چىكىدۇ،بىز ئۇلارنىڭ ئەجرىدىن پايدىلنىمىز،ئۇلارغا ھىچ بىر ئىش قىلىپ بەرمەيمىز،ئۇلارمۇ بىزگە ئوخشاش ئىنسان يەيدۇ،ئىچىدۇ،تىرىكچىلىك قىلىدۇ،ئۇلارنىڭمۇ ئائىلىسى بار،ئەمما ئۇلار ھازىر تارتقان جاپالىرىغا ،ئەجرىگەچۇشلۇق قىممەتكە ئىرىشەلمەيلا قالماستىن بارغانچە زىينىغا مىڭىۋاتقان بولىشىمۇ مومكىن،شۇڭا بىز ئۈچۈن خالىس ئەجىر قىلغۇچى بۇ پىدائىلارغا كۆپلەپ رەھمەت ئىيتماي تۇرالمايمىز،
بۇپىداكارلانىڭ ئەجرىمۇ بىز ئويلىشىدىغان مەسىلە،شۇڭا بىز ئۇلارنى قوللىشىمىز،قولىمىزدىن كىلىشچە كۆپچىلىك ئورتاق كۈچ چىقىرىشىمىز،ئىرىشىمىز كىرەك.باشقۇرغۇچى بولوشقا كىنىشكە دىپلۇم كەتمەيدۇ ئەمما بۇنىڭغا ئەقىل،بىلىم،نەزەر دائىرە،ساپا كىتىدۇ،ئۆزىڭلار يۇقىردا كۆردۈڭلار، باشقۇرغۇچى بولوشقا دىپلۇم، كىنىشكە كەتمەيدۇ،بىلىم كىتىدۇ دەيدۇ،ئەمما ئۇلار تىمىنى چۈشەنمەمدۇ ئۇخمىدىم ئۆز كۆڭلىگە ياقمايدىغان پىكىر چۈشسىلا ئۇنىڭغا رەدىيە بىرىمەن دەپ، تىما مەزمۇنىدىن چەتنەپ،تىمىنىڭ يۆلىنىشىنى ئۆزگەرتىۋىتىدىكەن.تور دۇنياسىدا ھەر خىل سەۋىيەدىكلى،ھەر خىل كىشىلەر توپى مەۋجۇت دەيدۇ يۇ،ئۇلارغا رەدىيە بەرگىلى تەييار،رەدىيە بىرىمەن،ئۆز پىكىرنى باشقىلارغا تاڭىمەن دەپ،تىمىدىن چەتنىگەننى ئاز دەپ جىدەل تىرىپ،كۆڭۈل ئاغرىقى قىلىپ،تىمىنىڭ قىممىتىنىمۇ نۆل قىلىپ... باشقىلار خاتا سۆزلەپ قالسا،كۆڭلىگە ياقمايدىغان،پىكىرىگە ئوخشىمايدىغا پىكىردىن بىرنى قىلىپ قالسا ئۇنىڭغا يول قويوش،كەڭ قورساق بولوش دىگەندىن يىراق،ئۇنداقتا باشقۇرغۇچى بولوشقا زادى نىمە كىتىدىكەن دوسلار...
ئۆز پىكىرنى باشقىلارغا قارىتىپ دىمىسىمۇ ئۆزىنىڭ تىمىغا بولغان كۆز قارىشىنى تامامەن ئوتتورغا قويالايدۇ ئەمەسمۇ،باشقىلارغا يول قويوش،كەڭ قورساق بولوشمۇ ئىسىل پەزىلەت.
مەن خىلى بولغان مىسرانىمغا ئاياغ باسمىغىلى،مەن تور دۇنياسىغا قەدەمئالغىلى ئۇزاق بولمىغان بولسىمۇ، دەسلەپ بىر نەچچە تىما يوللىدىم،كىيىن ئىلگىركى شۇنداق قىممەتلىك تىمىدىن بىرنى يوللىدىم،ئۇنىڭ مەنبەسىنى،ياغان كىشىنى بىلمەيمەن،مەن بۇنى ئىنىق ئەسكەرتىپ يوللىدىم،بۇ تىمىنى نادىرلاپ،چوققىلاپ قويغان مۇنبەرلەر ھەم بار، ئەمما مىسرانىمدىكى قايسى ئىقتىدارلىق،پەزىلەتلىك مۇبەرەك باشقۇرغۇچىنىڭ قولىغا چۈشتىكىن تىمىنىڭ قىممىتىنىمۇ كۆزدە تۇتماستىن،مىنى كۆچۈرمىكەش،ئەمدى كۆرۈلسە ئۇ قىلىمىز بۇ قىلىمىز دەپ بىر مۇنچە سۆزلەر بىلەن ئۆچىرىۋەتتى،ئۆچەرسىمۇ مەيلى ئىدى ئەمما ئادەمنىڭ كۆڭلىگە ئازار بەرمەسلىكى كىرەكتە.ئۇلارنىڭ تىمىنى تولوق ئوقوغۇنىدىن،چۈشەنگىنىدىن گۇمانلىنىمەن،قايسى كۈنى مىسرانىمدىكى xxباشقا مۇنبەرگە يوللىغان بىر تىمامغا << تولىمۇ ياخشى پىكىر قىپسىز ئابدۇكۆرەش. تولىمۇ جانلىق ھەم ئويلاندۇرغۇدەك ئىكەن ،مەنمۇ يىڭى ئەزالارنىڭ بىرى ، مەن سىزگە قايىل بولوپ قالدىم، بولسا سىز مۇشۇ قەدىمىڭىز بىلەن مىسرانىم ئۈچۈن تۆھپە قوشىشىڭىزنى ئۈمۈد قىلىمەن ھەم ھەممەيلەن ئۈمىد قىلىدۇ،مىسرانىم ئۈچۈن ھەممەيلەن قول تۇتۇشۇپ بىرلىكتەئىلگىرلەيلى >>دىگەن ئىنكاسنى قالدۇرۇپتۇ،ئارقىدىنلار ئۇنىڭ بۇ پىكىرىگە ئاچچىق رەدىيە پىكىرلىرى يىغپ كەتتى.
بۇ نىمە قىلىق ئەمدى،بىر- بىرىگە ئازراق يول قويسا،كەڭ قورساق بولسا بىرەر يىرى ئۆزۈلۈپ قالارمۇ،بىر بىرىمىزگە بولغان ھۆرمەتنى قىسىپ نىمىلەرگەىمۇ ئىرىشەرمىز زادى.
ئاتا-بوۋىمىز تەرپىدىن ئەزىزلەنگەن،ئاتا-ئانىمىزنى ئەزىزلىگەن مۇشۇ تۇپراقتا ياشاپ،تومىرىمىزدا ئىقىۋاتقان قىنىمىز،مۇقەددەس دىنىمىز،تەقدىرىمىز ئوخشاش قىرىنداشلاردىن تۇرۇپ بىر -بىرىمىزنى ئىتراپ قىلالماي،ئازراقمۇ يول قويالماي،ھۆرمەت قىلالماي...
شۇنداق بىر ھەقىقەت باركى مەن بۇ زىمىنغا بىر ئۇيغۇر،بىر مۇسۇلمان بولوپ تۆرەلدىم،مىنىڭ مەۋجۇتلىغىم،مىنىڭ تەقدىرىم شۇ قىرىنداشلىرىم بىلەن ئوپئوخشاش،مىنىڭ ئوىشاش بولمىسا مىنىڭ پەرزەنتىمنىڭ ئوخشاش،مەيلى مەن نىمىلا قىلماي،مەيلى  قارا تەنلىك نىگىر ياكى كۆك كۆزلۈك ئوروس بىلەن توي قىلىپ ئەۋلاتلىرىم باشقا مىللەتكە چاڭلىشىپ كەتسۇن ئۆزۈمنىڭ ئۇيغۇر ئىكەنلىكىدىن ئىبارەت بىر ھەقىقەتنى مەڭگۈ ئۆزگەرتەلمەيمەن.مىنىڭ ئىززىتىمنى قىلىدىنغىنى،ھۆرمىتىمنى قىلىدىغىنى مەيلى ئاقساق- چولاق،توكور- ماڭماق بولوپ كەتسۇن يەنە شۇ قىرىنداشلىرىم تەرپىدىن بولىدۇ...
بىز جىكسۇننىڭ ناخشىلىرىنى ئاڭلايمىز،بىلگەيتىسنىڭ ئەجرىدىن بەھىرلىنىمىز ئەمما ئۇلارنى ئەزىزلەيدىغىنى يەنىلا ئۆز قىرىنداشلىرى بولىدۇ،باشقىلار ۋاقىتلىق ھۆرمەتلىگەندەك،ئەزىزلىگەندەك قىلغان بىلەن ئۆز قىرىنداشلىرىدەك مەڭگۈ بولمايدۇ...
ئابدۇرىم ئۆتكۈرنى،ئابدۇخالىق ئۇيغۇرنى،مەمەتئىلى ئەپەندىنى،ئابدۇللا ئابدۇرىمنى،ئابدۇكىرىم ئابلىزنى كىملەر بىلىدۇ؟ كىملەر ھۆرمەتلەيدۇ؟كىملەر ئەزىزلەيدۇ؟
بىر-بىرىمىزگە يول قويايلى،بىرلىشەيلى،بىر-بىرىمىزنىڭ ھۆرمىتىنى قىلايلى...
بۇ تىمامنى يۇقىردىكى دوستۇمغا بولغا ھۆرمىتىم يۈزىسىدىن يوللىغا ئىدىم،بەلكىم ئۇنداق بولمىغىنىدىمۇ يوللار ئىدىم،
ھەر قانداق بىر بىز ئۈچۈن ئەجىر قىلىۋاتقان مۇنبەر ئۈچۈن قولومدىن كىلىشچە ئەجىر قىلىش،ئۇلارنى قوللار مىنىڭ خۇشاللىغىم ھەم ھەم مەسئۇلىيتىم...
سىناپ باقماسلىقنىڭ ئۆزى ئەڭ چوڭ مەغلۇبىيەت...
ياشىساڭ ئۆز ئەقىدەڭ يولىدا جان بىرەلىگۈدەك ياشا...

مەنچىمۇ مۇنبەردە ئادەمنىڭ ئويلىنىشىغا  تىگىشلىك  ئىشلار خىلى بار .
مەن نەچچە  تىمىلاردا  دەپ ئۆتتۈم   ، بىز مۇنبەرگە نىمىشقا كىرىمىز ؟  غەيۋەت قىلغىلىمۇ ، ئۆز ئارا زىددىيەت تىرىغىلىمۇ ؟  مۇنبەرگە  يىڭىلىقلاردىن خەۋەردار بولارمەنمىكى  ، يىڭى يىڭى  مەدەنيەت خەۋەرلىرىنى بىلەرمەنمىكى ، ئەدەبىيات جەھەتتىكى  ئىھتىياجىمغا بىر زوق ئالارمەنمىكى دەپ كىرسە ، بۇ يەرگە كىرگەندىن كىيىن   يازغۇچىنىڭ  يازغان  تىمىسى بىر ياقتا قالغان ،  چۈشكەن ئىنكاسلارنىڭ تۇترۇقسىزلىقىدىن  بىشى ئايلىنىدۇ  كىشىنىڭ  .ئىنكاسلارنىڭ بەزىلىرى خۇددى چ چ غا ئوخشاپ قالاي دىگەن .
يەنە بەزىلەر  ھەتتا  نىمە مەزمۇندا يىزىلغانىلىقىنى ئوقۇپمۇ باقماستىن رەھمەت سۆزلىرىنى توشقۇزۋەتكەن  . خۇددى بىر تىما يازغان كىشى  بىكارچىلىقتا  زىرىكىپ بىرسىنىڭ رەھمىتىگە  زار  بولۇپ  بىر نىمە يازغاندەك .
يەنە بەزىلەر باشتىكىلەرنىڭ ئىنكاسىغا ئاساسلىنىپلا  قۇرۇق سۆزدىن ئىككى ئىغىز قىىلپ  خۇددى بىرسى ئىنكاس ئالۋىڭى سالغاندەك ئالۋاڭ ئىنكاس يازىدىغانلار باركەن . مۇشۇنداقلارنى  ھېچبىر كىشى زورلىمىغاندىكىن ئىنكاس  يازمىسىمۇ ئۆزىنى بىسۋالسا  .
  بۇ قىرىندىشىمىز دەپ ئۆتكەندەك  ھەممىمز ئىنكاسلىرىمىزدىمۇ  شۇ تىمىغا مۇناسىۋەتلىك  پىكىرلەرنىڭ  قىلىپ تىمىنىڭ كەم جايلىرىنى  تىخىمۇ مۇكەممەللەشتۈرسەك  ، ئاۋال تىمىنى ئوقۇپ ئاندىن باشقىلارنىڭ تىمىسىنى كۆرۈپ  بىرەر بىلىم ھەققىدە  بىر چۈشەنچىگە ئىگە بولساق ئاندىن بىز  بىر سائەت تۆلىگەن 2 يۈەن پۇلىمىزغا رازى بولىمىز  . بۇمۇ ئاندىن ھەقىقى  مۇنازىرە بولىدۇ  .
ئاندىن مۇشۇ مۇنبەرگە ناخشا  يوللايدىغان ئىشلار پەقەتلا ئىددىيەمدىن ئۆتمىدى .  يوللانغان ناخشىلارنىڭ كۆپ ساندىكىلرى  ئاساسەن ھەممە تور بەتلەرنىڭ ناخشا  مۇزىكا  بەتلىرىدە  بار ناخشىلار  ئۇنداق  ناخشىلارنى  يەنە زورلاپ مۇنبەرگە ئەكىرىشنىڭ نىمە زۆرۈرىيتى بار  ؟   مىنىڭچە   تورغا چىقىپ  مۇنبەر كۆرۈشنى   بىلىدىغانلارنىڭ ئارىسىدا  ناخشا مۇزىكا بەتلىرىگە كىرىپ ئۆزى ياخشى كۆرگەن ناخشىنى ئاڭلاشنى بىلمەيدىغانلار  يوق دىيەرلىكقۇ دەيمەن  .  بىز نىڭ ناخشا ئاڭلىغۇمز  بولسا  ھازىر ناخشا مۇزىكا ئاڭلايدىغان يەر دىگەن ساماندەك  ، ئۇنى  يەنە مۇنبەرگە كىرىپ  ئالاھىتەن مۇنازىرە ئىپ بىرىشنىڭ ھاجىتى يوق  . بولسا  مۇشۇنداق پىچەك تىمىلارنى باشقۇرغۇچىلار تەستىقلىمىسا   ،  باشقا  دوستلىرىمىزمۇ  يوللاپ  يۈرمسە  .  ئەگەرئالاھىدە ياخشى يىڭى چىققان  باشقا تورلاردا  يوق ، ئىجادىي ناخشىلار بولسا بۇ ئۇنىڭ سىرتىدا .
كىشى ھەر ۋاقىت بەخىت تۇيغۇسى ئىچىدە ياشاشنى ئۈگىنۋىلىشى كىرەك.ئۆزىدە بار بولغان بەخىتننى سىزەلمىگەن كىشى ئەڭ بەخىتسىزدۇر.

باشقىلارنىڭ ئىنكاسى سىزگە ياقمىغان بولسا،خالىمىغان بولسىڭىز  مەيلى.باشقۇرغۇچى بولسىڭز ئۈچۈرۈڭ،ئۇنداق بولمىسا ئۇنىڭ بىلەن كارىڭىز بولماي ئۆز يولىڭىزغا مىڭىپ تىما مەزمۇنىدا ئىنكاس يىزىڭ. ئەگەرتازا  بولمىغۇدەك بولسا باشقۇرغۇچى دەيدىغان قارا كۆزلەرمۇ قاراپ تۇرماس.

سىزگە ئارتۇقچە بولسىمۇ بىر تەكلىپ بەرسەم رەنجىمەڭ دوستۇم،سىزنىڭ پىكىرلىرىڭىز بىرىنجىدىن لوگىكىغا ئۇيغۇن ئەمەس بۇ ئەڭ موھىم بىر تەرەپ،سىزنىڭ بەزى پىكىرلىرىڭىز ئۆزىڭىزنىڭكىگە زىت كىلىدىكەن،ئىككىنچىدىن مىجەزىڭىز كەسكىن دادىل ئىكەن، ئەمما دىققەت قىلمىسا بۇنىڭ ئالدىراڭغۇلىقتىن ئىبارەت بىر ناچار تەرپىنى يۇقتۇرۋالىمىز،بۇنىڭ جىق ئەمەس ئازراق ئىزناسىنى سىزدە بار ئوخشايدۇ دەپ ئويلىدىم ،ئۇنداق بولمىسا سىز بىپەرۋا بولمىسا تىما مەزمۇنىدىكى ۋە ئىنكاسلاردىكى بىر دىگەن گەپنى قايتا قايتا دىگۈزمەيسىز ئادەمگە،ئۈچىنجىدىن مۇنازىرە ئەخلاقى دەيدىغان بىر سۆزمۇ بولسۇن ئىڭىمىزدا،سىز بىلەن مۇنازىرە قىلغۇچى قانداقتۇر سىزنىڭ دۈشمىنىڭىز ئەمەس،ئۇ سىزنىڭ سەبدىشىڭىز،شۇڭا ئۇنى ھۆرمەتلىشىڭىز،بەزى ئاچچىق ئارلاشقان پىكىرلىرنى قىلماسلىقىڭىز كىرەك،چۈنكى بۇ مۇنازىرە،ھەرگىزمۇ تالاش تارتىش ئەمەس.
سىزھازىر مەن ئۈچۈن جانلىق،تۈز پىكىر كەسكىن مۇنازىرەدىشىم.
