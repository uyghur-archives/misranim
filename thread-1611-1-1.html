                        چاقچاق چاقچاق تۆتىنجى قىسىم

                                  نازى قىزغا

                         تۇنۇش دۇشمەنگە چاقچاقنى ،
                         يېزىپ قۇيۇپ نازى قىز .
                         دوسلار ئارا سۆھپەتتە ،
                         بۇلۇپ قالماڭ قاچقۇن سىز .

                             كورلا قىزغا

                          كورلا قىزى كورلىدا ،
                          ھاۋارايى قانداراق .
                         چاقچاق كەتتى سىزگىمۇ .
                         ئۇقىۋېلىڭ چاقانراق .

                            مەلىكەمگە

                         غۇلجا چاقچاق ۋارىسى ،
                         دەل ئۈزىدۇر مەلىكەم .
                         يازغان ھەربىر چاقچىقىدا ،
                         خۇرۇچ تولۇق تۇزى كەم .

                                 ئايپەرىگە

                        ئايپەرى ئىكەن ئىسمىڭىز ،
                        ئايدىن گۈزەل ئوخشايسىز .
                        سىزگە چاقچاق يازمىسام ،
                        قاپىقىڭىزنى تولغايسىز .

                             ئاق بۆرىگە

                       بەكمۇ مىسكىن يېتىۋاپسىز ،
                       بېشىڭىزنى قۇيۇپ مۇنبەرگە.
                       نىمە دەتلەر كەلدىكىن ،
                       بىلەلمىدىم مەن سىزگە.

                               كۆكنۇرغا

                      چاقچاق يېزىپ ھەممىگە ،
                      ئايرىپسىزغۇ تاڭنۇرنى .
                      يەنە يازسىڭىز ئەگەردە ،
                      ماڭا يېزىڭ يەنە بىرنى .

                             تارىمغا

                      چىمەن دوپپا كۆز ئەينەك ،
                      بەك يارىشىپتۇ ئۇ سىزگە .
                     چاقچىقىمدىن ئازراقنى ،
                     نىسىۋە قىلدىم مەن سىزگە .

                            ئالدىراشقا

                      مىسرانىمنىڭ مۇنبىرىگە ،
                      كىرىپ قاپتۇ ئالدىراش .
                      ئوغوللارغا يات ئىشقۇ ،
                      بىكار بولسا چاچ تاراش .

                           پادىشاھ غا

                      ساتىراچخانىدىكى پاراڭنى ،
                      يېزىۋىتىپتۇ پادىشاھ ،
                      ئۆيىنىڭ ھەممە يىرىنى ،
                      قورشىۋاپتۇ تام -قاشاھ .

                              سايرامغا

                       ئەرلەردىكى مۇھەببەت ،
                       تۇنۇتىغۇ سايرامنى .
                       ئەسەر يېزىش دەردىدە ،
                       ئۇنۇتۇپتۇ بايرامنى .

                              ئۈشكۈرگە

                        قويمىچىنىڭ دەردىنى ،
                        بەك تارتىپتۇ ئۈشكۈرمۇ.
                        چىرايىمۇ بەك قارا ،
                        بىلەلمىدىم نىگىرمۇ ................چاقچاق ھە ...

                              خۇمپەرگە

                       چىن سۆيگەن يارىدىن ،
                       ئايرىلىپتۇ خۇمپەرمۇ .
                       قولىڭىزدا جىق كورۇت ،
                       ھەمىسىلا دەمپەيمۇ .

مەدىنەگە

                       بىر يۇرۇتلۇق ئىكەنمىز ،
                       ئەسلىدىنلا ئىككىمىز ،
                       ئاڭا بەكمۇ قاملاشقان ،
                       تەڭ دىمەتلىك يېشىمىز .

ئەركىزاتقا

                      رەسىمىڭىزگە قارىسام ،
                      بەكمۇ تاتلىق تۇرىسىز .
                      چوڭ بولدىمىكىن دىسەم ،
                      سوسكا ئىمىپ يۈرۈيسىز.

                           سۇبانەمگە

                      چەتئەللىكنىڭ ئادىتى ،
                      سىڭىپ كىردى تۇرمۇشقا .
                      چىقىرۋالغان رەسىمىڭىز ،
                      مۇڭ سىرايدۇ نىمىشقا .

                            ئۇدۇنجانغا

                     ئۇنتۇپ قالماي سىزگىمۇ ،
                     يېزىپ قويدۇم ئۇدۇنجان ،
                     يازماي قالسام ئەگەردە ،
                     بۇلۇپ قالماڭ يېرىم جان .

                           ئەركەمگە

                      ئەركەم دوستۇم سىزگىمۇ ،
                      يارىشىپتۇ چىغ قالپاق .
                      جىدەل تېرىپ ئورۇنسىز ،
                      يېرىلمىسۇن قاش قاپاق .

ئەسسالامۇ ئەلەيكۇم تورداشلار بۇرۇنقى پەدە بۇيىچە ئالدىنقى قېتىم ئىسمى چىقمىغانلارغا يوللاندى ئىسمى چىقمىغانلار يەنە بۇرۇنقىدەك خاپا بولماڭلار ھە...ئېغىر كەتكەنلەرمۇ شۇنداق .
