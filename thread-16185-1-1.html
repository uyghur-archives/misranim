ئەقىللىق تاكسى شوپۇرى.

بىر كۈنى  تارىم ئەپەندى ئۇرۇمچىگە كەلگەن كۈنى تاكسىغا ئولتۇرماقچى بوپتۇ،لىكىن تاكسىچىلارنىڭ ئۆزىنى قۇيۇپ كىتىشىدىن ئەنسىرەپ،،ئەتىركىسىنى ئېلىپ چىشى كولىغاچ مېڭىپتۇ،،  باردىغان يەرگە بارغاندىن كىيىن سوراپتۇ ،،،نەچچەپۇ؟؟؟؟؟؟؟؟بۇ چاغدا تاكسى شوپۇرى چەيدو بىلەن ئىڭكىنى قىرغاش ،،،،،،،،،،،،،،،،،،،،،،،ئۆزەڭ بىلەسەن،،،،،

يېقىندىن بۇيانقى ئۆزگۈرۈش.

بىر كۈنى كۈلقىز ماڭا بىر قۇر سەپسېلىپ قارۋەتكەندىن كىيىن،،،
يېقىندىن بۇيان سەن يەنەسەل،،،دىيشىگە مەن ئۇنىڭ سۆزىنى تارتىۋېلىپ ،،بولدى مىنى سەمرىپ كىتىپسەن دىسەڭ بولمايدۇ جۇمۇ؟دىدىم،،،،
ئۇ بېشىنى لىڭشىتىپ قۇشۇلدىغانلىقىنى بىلدۈردى،،ھەمدە مۇنداق دىدى،،،،
ئىشىشىپ قاپسىز تەلپەك كىيۋېلىڭ....

يۇشۇرۇن كۈيۈپ قېلىش،،،

ما مۇنبەردىكى ئالدىراش مەرھاباغا يۇشۇرۇن كۈيۈپ قاپتۇ،،،
بىر كۈنى غەيرەتكە كىلىپ بۇ مەرھاباغا ئۇچۇر يېزىپتۇ،،،ئۇچۇر مۇنداق يېزىلغانتى(مەن سىزگە دىققەت قىلىپ يۈرگىلى خېلى بولدى)
بىردەمدىن كىيىن مەرھابا ئۇچۇر قايتۇرۇپتۇ،،،(ئۆتۈنۈپ قالاي باشىلققا دەپ قويمىسىڭىز ئەمدى خىزمەت ۋاقتىدا ھەركىزمۇ كازىر چاقمايمەن)

تاماكا چىكىشكە بۇلدىغان قۇتقۇزۇش پاراخوتى.

پاراخوت  ئاستا-ئاستا چۆكۈپ كىتىپ باراتتى.ھەممە ئادەم قورقۇپ ۋارقىراشتى،،
خۇشخۇي520 غاڭزىسىنى چىشلىكىنچە ،،كەپسىزدىن سورىدى!
تاماكا چىكىشكە بۇلدىغان قۇتقۇزۇش پاراخوتى قايسى؟


                       ئاشخاندا
بىر كۈنى سىرلىق، كامىران  كارۋان ،مەن ئاشخاندا ئولتۇرۇپ پاراڭلىشپ قالدۇق  ئۇنىڭ ئۈستىگە بۇ بىز يىگەن تاماقمۇ بەك مېزلىك ئىكەن..
سىرلىق،،، بىر قېتىم كۈچادا تاماق يىسەممۇ شۇنداق ئوخشاپ كىتىپتىكەن، ئاغزىمدا قالغان ئىدى .ما ئاشخاندىمۇ ئوخشايدىكەن دەپ تۇرشىغا،،،،كامىران قايسى ئاشخانا ئىدى ئۇ؟؟دەپ سورىدى،،،
سىرلىق ئاشخاننىڭ ئىسمىنى دەپ تۇرشىغا كامىران  راس،،،راس ...دەل شۇ چاغدا مەن ئاشۇ ئاشخاندا ئىشلەيتىم..
بۇ كەپنى ئاڭلىغان سىرلىق ماڭا قاراپ قۇيۇپ، پو ئاتمىغىنە مەن سىنى ئۇ جاغدا كۆرمىگەن ئىدىمغۇ؟؟
كامىران كەينىدىنلا،،   مەن ئۇ چاغدا تاماق تەييارلاۋاتسام سىرتىغا چىقالمايمەندە...
سىرلىق جىم بوپ قالدى،،،
كارۋان!مەن قايسى كۈنى قەشقەرگە  بارغانتىم .بىراق بۇ شوپۇر شۇنداق ئۇستىكەن...15سائەتتىمۇ قەشقەرگە ئاپىرىپ قويدى،،،مەن بۇ قېتىم بارسام چۇقۇم يەنە ئاشۇ ماشىنىغا ئولتۇرمەن،،،،،دەپ تۇرشىغا كامىران يەنە .
قانداق رەڭلىك ماشىنا ماركىسى نىمە ئىكەن؟ دەپ سورىدى..كارۋان ماشىنىڭ رەڭكىنى ھەم ماركىسىنى دەپ تۇرشىغا....توغرا دەپلا چوزىنىڭ ئۈسىتىگە بىرنى قويدى ،،،بۇ جاغدا كارۋان،،،سەن قانداق بىلىسەن؟
دەل شۇ ماشىننى مەن ھەيدەيمەن ئەمەسمۇ دىدى كامىران!
كارۋان ئۇ جاغدا مەن سىنى ئۇ ماشىنىدا كۆرمىگەن ئىدىمغۇ؟ كامىران ...مەن ئۇ جاغدا ماشىنا ھەيداۋاتقان تۇرسام سەن مىنى قانداق كۆرىسەن؟
كارۋان بىلەن سىرلىق ماڭا قاراپلا جىم بوپ قالدى،،،،كامىران ئۆزىچە ماختانغىلى تۇردى،،،
مەن گىپىنى ئارلاپلا  دىدىمدە،،،مەنمۇ ئەينى چاغدا تۇرمىدە تاماق ئەتسەم جازا ئۆتۈگۈچىلەر شۇنداق ئىشتىھا بىلەن يەيتى...دىسەم كامىران قۇپۇلا دىمسىمۇ بەك ئوخشايتى ،،،دەپلا كەپنى توختۇتۇپ قويدى مەنمۇ  سورۇدۇمدە....سەم ئۇ يەردە نىمە ئىش قىلاتتىڭ؟؟ كامىران ئويلانمايلا،،،مەن جازا ئۆتەۋاتقان،،،،
ئۇھ ئاپلا دىدى كامىران..

مەنبە ئاڭلىغان ھەم ئۆزەم يازغانلىرىم.
ئەقلىمىزنى ئىشلەتسەك بىز ئەڭ ئەقىللىق مىللەت!

ئەقىللىق تاكسى شوپۇرى.

بىر كۈنى  تارىم ئەپەندى ئۇرۇمچىگە كەلگەن كۈنى تاكسىغا ئولتۇرماقچى بوپتۇ،لىكىن تاكسىچىلارنىڭ ئۆزىنى قۇيۇپ كىتىشىدىن ئەنسىرەپ،،ئەتىركىسىنى ئېلىپ چىشى كولىغاچ مېڭىپتۇ،،  باردىغان يەرگە بارغاندىن كىيىن سوراپتۇ ،،،نەچچەپۇ؟؟؟؟؟؟؟؟بۇ چاغدا تاكسى شوپۇرى چەيدو بىلەن ئىڭكىنى قىرغاش ،،،،،،،،،،،،،،،،،،،،،،،ئۆزەڭ بىلەسەن،،،،،

يېقىندىن بۇيانقى ئۆزگۈرۈش.

بىر كۈنى كۈلقىز ماڭا بىر قۇر سەپسېلىپ قارۋەتكەندىن كىيىن،،،
يېقىندىن بۇيان سەن يەنەسەل،،،دىيشىگە مەن ئۇنىڭ سۆزىنى تارتىۋېلىپ ،،بولدى مىنى سەمرىپ كىتىپسەن دىسەڭ بولمايدۇ جۇمۇ؟دىدىم،،،،
ئۇ بېشىنى لىڭشىتىپ قۇشۇلدىغانلىقىنى بىلدۈردى،،ھەمدە مۇنداق دىدى،،،،
ئىشىشىپ قاپسىز تەلپەك كىيۋېلىڭ....

يۇشۇرۇن كۈيۈپ قېلىش،،،

ما مۇنبەردىكى ئالدىراش مەرھاباغا يۇشۇرۇن كۈيۈپ قاپتۇ،،،
بىر كۈنى غەيرەتكە كىلىپ بۇ مەرھاباغا ئۇچۇر يېزىپتۇ،،،ئۇچۇر مۇنداق يېزىلغانتى(مەن سىزگە دىققەت قىلىپ يۈرگىلى خېلى بولدى)
بىردەمدىن كىيىن مەرھابا ئۇچۇر قايتۇرۇپتۇ،،،(ئۆتۈنۈپ قالاي باشىلققا دەپ قويمىسىڭىز ئەمدى خىزمەت ۋاقتىدا ھەركىزمۇ كازىر چاقمايمەن)

تاماكا چىكىشكە بۇلدىغان قۇتقۇزۇش پاراخوتى.

پاراخوت  ئاستا-ئاستا چۆكۈپ كىتىپ باراتتى.ھەممە ئادەم قورقۇپ ۋارقىراشتى،،
خۇشخۇي520 غاڭزىسىنى چىشلىكىنچە ،،كەپسىزدىن سورىدى!
تاماكا چىكىشكە بۇلدىغان قۇتقۇزۇش پاراخوتى قايسى؟


                       ئاشخاندا
بىر كۈنى سىرلىق، كامىران  كارۋان ،مەن ئاشخاندا ئولتۇرۇپ پاراڭلىشپ قالدۇق  ئۇنىڭ ئۈستىگە بۇ بىز يىگەن تاماقمۇ بەك مېزلىك ئىكەن..
سىرلىق،،، بىر قېتىم كۈچادا تاماق يىسەممۇ شۇنداق ئوخشاپ كىتىپتىكەن، ئاغزىمدا قالغان ئىدى .ما ئاشخاندىمۇ ئوخشايدىكەن دەپ تۇرشىغا،،،،كامىران قايسى ئاشخانا ئىدى ئۇ؟؟دەپ سورىدى،،،
سىرلىق ئاشخاننىڭ ئىسمىنى دەپ تۇرشىغا كامىران  راس،،،راس ...دەل شۇ چاغدا مەن ئاشۇ ئاشخاندا ئىشلەيتىم..
بۇ كەپنى ئاڭلىغان سىرلىق ماڭا قاراپ قۇيۇپ، پو ئاتمىغىنە مەن سىنى ئۇ جاغدا كۆرمىگەن ئىدىمغۇ؟؟
كامىران كەينىدىنلا،،   مەن ئۇ چاغدا تاماق تەييارلاۋاتسام سىرتىغا چىقالمايمەندە...
سىرلىق جىم بوپ قالدى،،،
كارۋان!مەن قايسى كۈنى قەشقەرگە  بارغانتىم .بىراق بۇ شوپۇر شۇنداق ئۇستىكەن...15سائەتتىمۇ قەشقەرگە ئاپىرىپ قويدى،،،مەن بۇ قېتىم بارسام چۇقۇم يەنە ئاشۇ ماشىنىغا ئولتۇرمەن،،،،،دەپ تۇرشىغا كامىران يەنە .
قانداق رەڭلىك ماشىنا ماركىسى نىمە ئىكەن؟ دەپ سورىدى..كارۋان ماشىنىڭ رەڭكىنى ھەم ماركىسىنى دەپ تۇرشىغا....توغرا دەپلا چوزىنىڭ ئۈسىتىگە بىرنى قويدى ،،،بۇ جاغدا كارۋان،،،سەن قانداق بىلىسەن؟
دەل شۇ ماشىننى مەن ھەيدەيمەن ئەمەسمۇ دىدى كامىران!
كارۋان ئۇ جاغدا مەن سىنى ئۇ ماشىنىدا كۆرمىگەن ئىدىمغۇ؟ كامىران ...مەن ئۇ جاغدا ماشىنا ھەيداۋاتقان تۇرسام سەن مىنى قانداق كۆرىسەن؟
كارۋان بىلەن سىرلىق ماڭا قاراپلا جىم بوپ قالدى،،،،كامىران ئۆزىچە ماختانغىلى تۇردى،،،
مەن گىپىنى ئارلاپلا  دىدىمدە،،،مەنمۇ ئەينى چاغدا تۇرمىدە تاماق ئەتسەم جازا ئۆتۈگۈچىلەر شۇنداق ئىشتىھا بىلەن يەيتى...دىسەم كامىران قۇپۇلا دىمسىمۇ بەك ئوخشايتى ،،،دەپلا كەپنى توختۇتۇپ قويدى مەنمۇ  سورۇدۇمدە....سەم ئۇ يەردە نىمە ئىش قىلاتتىڭ؟؟ كامىران ئويلانمايلا،،،مەن جازا ئۆتەۋاتقان،،،،
ئۇھ ئاپلا دىدى كامىران..

مەنبە ئاڭلىغان ھەم ئۆزەم يازغانلىرىم.
ئەقلىمىزنى ئىشلەتسەك بىز ئەڭ ئەقىللىق مىللەت!
