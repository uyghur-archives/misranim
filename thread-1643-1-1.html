                                             مىسران غا
                                       مىسران باشلىق ياخشىغۇ ،
                                      كوت-كوتىڭىز ئالدىراشقۇ .
                                       ئاچچىق بارمۇ ،تاتلىقمۇ.
                                    ئەگەريازمىسام نادىر تېما ،
                                  جازاغا تاياق بارمۇ ،توخماقمۇ ؟   راس گەپ جىڭ گەپ ھە ؟
                                    
                                           چالىقۇشى غا
                                   چالقۇشىدىگەن  بىر رومان  ،  
                                  ئوقۇغان مەن بىر زامان .
                                  مىجەزىڭىز بەك ئېغىر  .
                                 ماڭا بەكلا ئوخشايسىز . باشلىق سىزنىڭ سايىڭىز دە ماختىنىۋالدىم ھە .ھى ھى ..[s:2]

                                        كەپسىز قىز غا
                             سەلكىن دىگەن مۇنبەردە ،دوس ئىدۇققۇ ئىككىمىز .
                            ئايرىلمايمىز دىگەنغۇ ،ھىجران كەلدى خەۋەرسىز .
                           ئەگەر  قاتمىسام قوشاققا ،خاپا  بۇلۇپ يۈرمەيسىز .
                           ھىكايىگە بەك ئۇستا ،نىمىشقا شېئىر يېزىپ قويمايسىز .  ھېلىقىنى تىزرەك يوللاڭلا .[s:24]

                                          مەلىكە
                             ۋاي مەلىكەم شاشمۇ ؟
                            قېچىپ كەتلە راسمۇ ؟
                          مىنى تاشلاپ مۇنبەردە
                          سەيلە قىلىپ نەلەردە .           نىمە بولغانسلە ئادىشىم ھە ؟[s:10]
                                
                                     شاتئاي غا
                          مىسران دىگەن مۇنبەردە
                         چۇخەي دەپلا يۈرەمسىز .
                        كوت-كوتلاردا نىمىشقا ؟[s:10]
                       سالام قىلىپ قويمايسىز .              ئاغىنەم ياخشى تۇرۋاتامسىز يە ؟

                               رىۋايەت كە
                     رىۋايەت باش سۈرىتىڭز نىمە ئۇ [s:10]
                     قارام ،سىدام سىز ما قالتىسقۇ .
                    ھەرزامان يامانلايسىز مۇنبەردە
                    قوشاقچى دىلسىرىڭىز بۇيەردە .   يامانلىماڭ ماخما .
                
                            ۋېرنا غا
                   ئىسمىڭز راست ۋېرنا مۇ
                   قوشاق يازسام بۇلا مۇ [s:10]
                   باش سۈرتىڭىز بەك ئوماق
                   سىز مۇنبەردىكى بىر بوۋاق .   ھى ھى ،،ئەمدى قوشاق يېزىپ تۇنىشاي دىگەنچۇ ؟قاراڭ

                           ئۈمۈت گۈلى
                  ئۈمۈت گۈلى بارمۇ يە ؟
                  قوشاق قاتسام بۇلامدۇ ؟
                 ئاخشام قىلغان گەپلىرى ،
                 راسا كۈلسەم بۇلامدۇ .               ھى ھى ............سىز بەكلا ............

                         تۇمار
                 بىر يىل بولدى كۈرۈشمەي ،
                ياتسىراپلا قالدۇقمۇ .
                 مىسرانىم مۇنبىرىدە
                جەم بولساقمۇ بۇلامدۇ .           سىزمىنى .ئۇنتۇپ قالماڭ ھە ،يوقاپمۇ كەتمەڭ ،ئەنسىرەپ قالدىم

                        ئويغۇر تېكىن
                  ئويغۇر تېكىن ئۆزلىرى ،
                  ھەيۋەت ئىكەن باشلىرى .
                  چۈندىكىلەر نىمە دەيدۇ ،
                  تىكەن دەپلا يۈرەمدۇ .        بىر چىرايلىق ئىسىمنى سەتلەشتۈرۈپ دەيمىنا ئەمدى .
              
                         سۈزۈك ئاي  غا
                 سوراپ باقاي مەنسىز دىن ،
                 خاپا بولماڭ بۇلامدۇ
                ئەلكۈيىدىكى دىلسىر ى
                مەندىن ئوبدان تۇرامدۇ .         ھى ھى ...چاق چاق كوت -كوتنى قالدۇرۇپ قۇيامسىز يۇشۇرۇن
                
                        كەتىرەن
                 باش سۈرىتىڭز بەك كىچىك ،
                 كۆرمەي قالدىم بەزىدە .
                لوپا ئەينەك تاقىسام ،
                توپا كىردى كۆزەمگە .         سەل چوڭايتىپ ،خاپا بولمىغان بولسىز .
                    
                       كارتون
                كارتون كۆرۈپ مەن كۈندە ،
               تاياق يەپلا يۈرۈيمەن .
               ئىنكا يېزىڭ مۇنبەردە ،
               قوشاق قېتىپ قويىمەن .         ئانچە- مۇنچە ئىنكاس دەمدۇ پىنكاسمۇ يازىلى  ماندا ھە  !

                     نازى غا
               نازى خېنىم ئۆزلىرى ،
               سالامەتمۇ تەنلىرى .
              جەڭ قىلايلى بۇلامدۇ ،
              بەسلىشەيلى بەيگىمۇ .
        
                    ناز غا
              ناز ئىكەن ئىسمىڭز ،
              سەلكىن دىكى شۇمۇسىز .

              ھىكا يىنى تاشلامسىز .     خاپا بولماي گۈل-چۈشىنى ،چاقىرىپ قۇيىڭە ،قولىنى ئىشقا سېلىپ تۇرسۇن
