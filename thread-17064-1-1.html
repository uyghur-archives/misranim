                                                   كىتىپ بارسام باغچىدا،
                                                  ئۇچراپ قالدى بىر قىز،
                                                     كىيىنىشى ئۆزگىچە،
                                                   مېڭىشى ئۇنىڭدىن تىز،
يىگىت
                                                  يۈگۈرەپ بېرىپ ئۇنىڭغا،
                                                دىدىم قىزچاق ياخشىمۇسىز؟
                                                   رام قىلدىڭىز كۆڭلۈمنى،
                                                  بىر گىپىم بار ئاڭلامسىز،
قىز                                                     
                                                   قېنى ئېيتىڭ گىپىڭىزنى،
                                                   ياخشى بولسا ئاڭلايمەن،
                                                  ئاتا-ئانام ساقلاپ قالغان،
                                                    مەن ئۆيگە ئالدىرايمەن،

يىگىت                                              نىمىشقىكىن مەن سىزنى،
                                                   بىر كۆرۈپ كۆيۈپ قالدىم،
                                                      كۆڭلىڭىزمۇ ياخشىكەن،
                                                     مەن بۇنى بىلىپ قالدىم،

                                                   سوراپ باقاي ئەي قىزچاق،
                                                      بارمۇ ئوغۇل دوستىڭىز،
                                                     بولمىسا مەن دوس بولاي،
                                                       بارمۇ-يوقمۇ؟رايىڭىز؟
قىز                                                      
                                                      توۋا قىلدىم خۇدايىم،
                                                        ئاجايىپكەن گىپىڭىز،
                                                      تايىنى يوق گەپ قىلىپ،
                                                      ئەجەپ خىجىل قىلدىڭىز،

                                                    دوس تۇتمايمەن ئوغۇلدىن،
                                                        قىز دوسلىرىم كۇپايە،
                                                     ئالدىرايتىم مەن ماڭاي،
                                                       بولدى ئېيتماڭ ھىكايە،

يىگىت                                                توختاپ تۇرۇڭ گۈزەل قىز،
                                                      سىزگە تېخى گىپىم بار،
                                                   دوس بولۇشنى خالىمىسىڭىز،
                                                     بولاي سىزگە مەڭگۈيار،

                                                     مىنىڭ ئىسمىم ئالىمجان،
                                                      دوختۇرلۇقتۇر خىزمىتىم،
                                                       خالىسىڭىز يار بولۇپ،
                                                        بىرگە بولاي دىۋىدىم،
                                                     
                                                     پەرۋايىم يوق پۇلىڭىزغا،
قىز                                                     بولدى قويۇڭ سۆزىڭىزنى،
                                                      بولدى مەن مېڭىۋىرەي،\
                                                      ئويلاپ بېقىڭ ئۆزىڭىزنى،

                                                        ۋىجدانىڭىز بولسا گەر،
                                                    بۇنداق ئىشنى قىلماڭ سىز،
                                                      چوڭ سۈپەتلا بولسىڭىز،
                                                        بىزلەر كۆيۈپ قالىمىز،\
  
يىگىت                                                شۇ گەپلەرنى دەپ قىزچاق، `
                                                          كىتىپ قالدى يولىغا،
                                                         يۈرىكىمدە ئوت قالدى،
                                                           قاراپ قالدىم بويىغا،\

ھەي..........قىزچاق شۇنداق دەپ،كىتىپ قالدى،نىمىدىگەن ئىسىل قىز، مۇنبەرگە كىرىۋالدىكىن دەپ ئىزدەپ كىرىشىم.                        مەنبە؛دات باسقان قىلىچ.
