ھەر قېتىم سىنى خىيال قىلغاندا يۈرۈگۈم قاتتىق ئاغرىيدۇ .ئەمما بۇ ئاغرىقنىڭ قاچان ئاخرلىشىدىغانلىقىنى بىلگىلى بولمايدۇ .مەن يەنە سىنى قانچىلىك ئەسلەيدىغانلىقىمنى  ، يەنە قانچە ئۇزۈن ياخشى كۆردىغانلىقىمنىمۇ بىلمەيمەن .ھېچكىم ماڭا دەپ بىرەلمەيدۇ .ھېچكىم مەن ئۈچۈن قارار چىقىرالمايدۇ .



قانچە قېتىملاپ ئۆزۈمدىن چارچىدىڭمۇ دەپ سورىدىم ، ئوخشاشلا سەندىنمۇ شۇنداق سورىغانمەن .ئېيتە جېنىم، بۇ سۆيگۈنى داۋاملاشتۇرۇش كىرەكمۇ ياكى ئاخىرلاشتۇرۇش كىرەكمۇ ؟ بەلكىم بۇ سۆيگۈنى بالدۇرلا ئۈزۈل كىسىل ئاخىرلاشتۇرۇش كىرەكتۇ ......



شامال توختىدى ، بۇلۇتمۇ توختىدى .سىنى سۆيگەن بۇ يۈرەك بۇ دەقىقىدە توختاپ قالارمۇ ؟ ئېيىتقىنە  ،شۇنداق بولارمۇ ؟ مەن يىلىنغان ھالدا سەندىن سورىدىم  . سەن ئۈن تۈنسىز باشقا تەرەپكە قارىدىڭ . دىمەك سەن سۇئالىمدىن ئۆزۈڭنى قاچۇرغان ئىدىڭ .



شۇنچە ئۇزۇن سەپەرنى سەن بىلەن بىرگە ماڭدىم . بۇ سەپەردە بىزنىڭ تۇرمۇشىمىزدا نۇرغۇن كۈلكە- چاقچاق ، قايغۇ-ھەسرەت  ..........لەر بولغان .......بۇلار سىنىڭ ئىسىڭدىمۇ ؟ سەن يەنىلا شۇنچە بىپەرۋامۇ ؟



مەن بۇ ئۆمرۈمنى ئەنسىرەش ئىچىدىلا ئۆتكۈزدۈم . چۈنكى مەن ئۆزۈمنىڭ يەنە ساڭا قانچە ئۇزۇن ھەمراھ بولۇپ ماڭالايدىغانلىقىمنى بىلمەيمەن .مەن پەقەتلا سىنىڭ قولۇڭنى قويۋەتمەي توختىماستىن ئالغا قاراپ مېڭىشنىلا بىلىمەن .مەن ئالدىمىزدىكى يولنىڭ ئەگرى توقاي ياكى تۈز ئىكەنلىكىنى بىلمەيمەن .سەن يەنە مىنىڭ ئەندىشە قىلىشىمغا مۇھتاجمۇ ، ساڭا ھەمراھ بولىشىمغا مۇھتاجمۇ بۇنى بىلشىكە ئەسلا قادىر ئەمەسمەن ؟



سىنىڭ ئۈن تۈنسىزلىكىڭ مىنى بارغانچە ئۈمۈتسىزلەندۈرۋاتاتتى . ئەجىبا راستىنلا ئايرىلىدىغان ۋاقتىمىز كەلگەنمىدۇ ؟مەن ئايرىلىشتىن قورقىمەن ، ئىزچىل ئەنسىرەپ كەلگىنىممۇ شۇ .ئىلگىرى مەن سىنىڭ كۆزۈڭدىن قەيسەرلىكنى تاپقان ئىدىم .ئەمما بۇ دەقىقىدە مەن قەيسەرلىكنىڭ يىللارنىڭ ئۆتىشىگە ئەگىشىپ غايىپ بولغانلىقىنى ھىس قىلدىم . مەن ئۆزۈمنىڭ ئەمدى سىنىڭ سۆيگىڭنى قۇبۇل قىلالىغۇدەك قۈربىتىمنىڭمۇ قالمىغانلىقىنى ھىس قىلىۋاتىمەن .بۇ زادى كىمنىڭ خاتالىقى ؟؟؟؟ بۇ ئەمدى مەن ئۈچۈن مۇھىم ئەمەس .



ئۇزۇندىن بۇيان بىزنىڭ ھىسياتىمىز مەڭگۈ ئايىغى چىقمايدىغان ئۈمۈت ئارزۇ .سىنىڭ نەزىرىڭدە مۇھەببەت يالغان ، چۈنكى مەن ساڭا بىر رىياللىقنى ئاتا قىلدىم .سان ساناقسىزلىغان ئۈمۈت ئارزۇ ، ھاياجان بۇلارنىڭ ھەممىسىنى مەن ساڭا ئاتا قىلدىم .ئەمما سەن نەزىرىڭدىكى‹ مۇھەببەت يالغان › غا ئىشىنىپلا كەلدىڭ .



چۈش ھەقىقەتەن گۈزەل .مەن ھەر قېتىم چۈشۈمدە سەن بىلەن ئۇچرىشىمەن . مەن سىنىڭ چۈشۈڭدىن ئاللىقاچان قايتىپ چىقتىم ، بۇندىن كىيىن سىنىڭ چۈشۈڭدىكى ئۇ سىما ئەمدى مەن ئەمەس .ئىلگىرى سىنىڭ چۈشۈڭدىكى شۇ سىمانىڭ مەڭگۈ مەن بولۇپ قېلىشىنى تولىمۇ ئارزۇ قىلغان ئىدىم ئەمما رىياللىق بۇنىڭغا يول قويمىدى ....



مۇھەببەت ، نەپرەتنى لايىق كۆرمەيدۇ .شۇ سەۋەپ ساڭا نەپرەتلىنىپ ئاھ ئۇرۇپ  ياش تۆكۈشنى لايىق كۆرمىدىم .



مەن يەنە قانچىلىك ۋاقىتتىن كىيىن سىنى ئۇنتۇپ كىتىدىغانلىقىمنى بىلمىدىم .. مەن پەقەت بۇ كۈتۈش ،  ھەسرەت ، دىل ئاغرىقىنىڭ ماڭا يەنە نۇرغۇن ۋاقىت ھەمرا بولىدىغانلىقىنىلا بىلىمەن  . مەن بۇلارنىڭ كۆڭۈلدىكىدەك بولمىغانلىقىغا نەپرەتلىنىمەن . ماڭا ئۈمۈت ئاتا قىلىپ يەنە ئۇنى ئۆزۇڭنىڭ ئۈمۈتسىزلىككە ئايلاندۇرغىنىڭغا نەپرەتلىنىمەن .ئەگەر مىنىڭ سۆيگۈم ساڭا خوشاللىق ئاتا قىلسا بۇ خوشاللىقنى بىر ئۆمۈر ئىسىڭدە تۇت ، ئەكچىسە قايغۇ ئاتا قىلسا بۇ قايغۇنىمۇ بىر ئۆمۈر ئىسىڭدە تۇت .



ئەگەر راستىنلا تاللاش ئىھتىمالى بولغان بولسا ئىدى مەن چوقۇم سەن بىلەن تۇنۇشماشلىقنى تاللىغان بولاتتىم ، سىنى سۆيۈپ قالماسلىقنى تاللىغان بولاتتىم .ئەپسۇس بۇ تەقدىر . بۇ بەلكىم تەقدىرنىڭ مىنىڭ پىشانەمگە پۈتكەنلىرىدۇر ......ھاياتلىقنىڭ كىچىككىنە بىر سىناقلىرىدۇر ......خەيىر ئەمدى ، بەخىت ، ئامەت ساڭا يار بولسۇن .....

ھەر قېتىم سىنى خىيال قىلغاندا يۈرۈگۈم قاتتىق ئاغرىيدۇ .ئەمما بۇ ئاغرىقنىڭ قاچان ئاخرلىشىدىغانلىقىنى بىلگىلى بولمايدۇ .مەن يەنە سىنى قانچىلىك ئەسلەيدىغانلىقىمنى  ، يەنە قانچە ئۇزۈن ياخشى كۆردىغانلىقىمنىمۇ بىلمەيمەن .ھېچكىم ماڭا دەپ بىرەلمەيدۇ .ھېچكىم مەن ئۈچۈن قارار چىقىرالمايدۇ .



قانچە قېتىملاپ ئۆزۈمدىن چارچىدىڭمۇ دەپ سورىدىم ، ئوخشاشلا سەندىنمۇ شۇنداق سورىغانمەن .ئېيتە جېنىم، بۇ سۆيگۈنى داۋاملاشتۇرۇش كىرەكمۇ ياكى ئاخىرلاشتۇرۇش كىرەكمۇ ؟ بەلكىم بۇ سۆيگۈنى بالدۇرلا ئۈزۈل كىسىل ئاخىرلاشتۇرۇش كىرەكتۇ ......



شامال توختىدى ، بۇلۇتمۇ توختىدى .سىنى سۆيگەن بۇ يۈرەك بۇ دەقىقىدە توختاپ قالارمۇ ؟ ئېيىتقىنە  ،شۇنداق بولارمۇ ؟ مەن يىلىنغان ھالدا سەندىن سورىدىم  . سەن ئۈن تۈنسىز باشقا تەرەپكە قارىدىڭ . دىمەك سەن سۇئالىمدىن ئۆزۈڭنى قاچۇرغان ئىدىڭ .



شۇنچە ئۇزۇن سەپەرنى سەن بىلەن بىرگە ماڭدىم . بۇ سەپەردە بىزنىڭ تۇرمۇشىمىزدا نۇرغۇن كۈلكە- چاقچاق ، قايغۇ-ھەسرەت  ..........لەر بولغان .......بۇلار سىنىڭ ئىسىڭدىمۇ ؟ سەن يەنىلا شۇنچە بىپەرۋامۇ ؟



مەن بۇ ئۆمرۈمنى ئەنسىرەش ئىچىدىلا ئۆتكۈزدۈم . چۈنكى مەن ئۆزۈمنىڭ يەنە ساڭا قانچە ئۇزۇن ھەمراھ بولۇپ ماڭالايدىغانلىقىمنى بىلمەيمەن .مەن پەقەتلا سىنىڭ قولۇڭنى قويۋەتمەي توختىماستىن ئالغا قاراپ مېڭىشنىلا بىلىمەن .مەن ئالدىمىزدىكى يولنىڭ ئەگرى توقاي ياكى تۈز ئىكەنلىكىنى بىلمەيمەن .سەن يەنە مىنىڭ ئەندىشە قىلىشىمغا مۇھتاجمۇ ، ساڭا ھەمراھ بولىشىمغا مۇھتاجمۇ بۇنى بىلشىكە ئەسلا قادىر ئەمەسمەن ؟



سىنىڭ ئۈن تۈنسىزلىكىڭ مىنى بارغانچە ئۈمۈتسىزلەندۈرۋاتاتتى . ئەجىبا راستىنلا ئايرىلىدىغان ۋاقتىمىز كەلگەنمىدۇ ؟مەن ئايرىلىشتىن قورقىمەن ، ئىزچىل ئەنسىرەپ كەلگىنىممۇ شۇ .ئىلگىرى مەن سىنىڭ كۆزۈڭدىن قەيسەرلىكنى تاپقان ئىدىم .ئەمما بۇ دەقىقىدە مەن قەيسەرلىكنىڭ يىللارنىڭ ئۆتىشىگە ئەگىشىپ غايىپ بولغانلىقىنى ھىس قىلدىم . مەن ئۆزۈمنىڭ ئەمدى سىنىڭ سۆيگىڭنى قۇبۇل قىلالىغۇدەك قۈربىتىمنىڭمۇ قالمىغانلىقىنى ھىس قىلىۋاتىمەن .بۇ زادى كىمنىڭ خاتالىقى ؟؟؟؟ بۇ ئەمدى مەن ئۈچۈن مۇھىم ئەمەس .



ئۇزۇندىن بۇيان بىزنىڭ ھىسياتىمىز مەڭگۈ ئايىغى چىقمايدىغان ئۈمۈت ئارزۇ .سىنىڭ نەزىرىڭدە مۇھەببەت يالغان ، چۈنكى مەن ساڭا بىر رىياللىقنى ئاتا قىلدىم .سان ساناقسىزلىغان ئۈمۈت ئارزۇ ، ھاياجان بۇلارنىڭ ھەممىسىنى مەن ساڭا ئاتا قىلدىم .ئەمما سەن نەزىرىڭدىكى‹ مۇھەببەت يالغان › غا ئىشىنىپلا كەلدىڭ .



چۈش ھەقىقەتەن گۈزەل .مەن ھەر قېتىم چۈشۈمدە سەن بىلەن ئۇچرىشىمەن . مەن سىنىڭ چۈشۈڭدىن ئاللىقاچان قايتىپ چىقتىم ، بۇندىن كىيىن سىنىڭ چۈشۈڭدىكى ئۇ سىما ئەمدى مەن ئەمەس .ئىلگىرى سىنىڭ چۈشۈڭدىكى شۇ سىمانىڭ مەڭگۈ مەن بولۇپ قېلىشىنى تولىمۇ ئارزۇ قىلغان ئىدىم ئەمما رىياللىق بۇنىڭغا يول قويمىدى ....



مۇھەببەت ، نەپرەتنى لايىق كۆرمەيدۇ .شۇ سەۋەپ ساڭا نەپرەتلىنىپ ئاھ ئۇرۇپ  ياش تۆكۈشنى لايىق كۆرمىدىم .



مەن يەنە قانچىلىك ۋاقىتتىن كىيىن سىنى ئۇنتۇپ كىتىدىغانلىقىمنى بىلمىدىم .. مەن پەقەت بۇ كۈتۈش ،  ھەسرەت ، دىل ئاغرىقىنىڭ ماڭا يەنە نۇرغۇن ۋاقىت ھەمرا بولىدىغانلىقىنىلا بىلىمەن  . مەن بۇلارنىڭ كۆڭۈلدىكىدەك بولمىغانلىقىغا نەپرەتلىنىمەن . ماڭا ئۈمۈت ئاتا قىلىپ يەنە ئۇنى ئۆزۇڭنىڭ ئۈمۈتسىزلىككە ئايلاندۇرغىنىڭغا نەپرەتلىنىمەن .ئەگەر مىنىڭ سۆيگۈم ساڭا خوشاللىق ئاتا قىلسا بۇ خوشاللىقنى بىر ئۆمۈر ئىسىڭدە تۇت ، ئەكچىسە قايغۇ ئاتا قىلسا بۇ قايغۇنىمۇ بىر ئۆمۈر ئىسىڭدە تۇت .



ئەگەر راستىنلا تاللاش ئىھتىمالى بولغان بولسا ئىدى مەن چوقۇم سەن بىلەن تۇنۇشماشلىقنى تاللىغان بولاتتىم ، سىنى سۆيۈپ قالماسلىقنى تاللىغان بولاتتىم .ئەپسۇس بۇ تەقدىر . بۇ بەلكىم تەقدىرنىڭ مىنىڭ پىشانەمگە پۈتكەنلىرىدۇر ......ھاياتلىقنىڭ كىچىككىنە بىر سىناقلىرىدۇر ......خەيىر ئەمدى ، بەخىت ، ئامەت ساڭا يار بولسۇن .....
