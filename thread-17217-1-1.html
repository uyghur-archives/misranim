دىراممىنىڭ ئاخىرىنى باشلاي ئەمسە،لىكىن بۈگۈن دىرامما ھىكايە شەكىلدە باشلىنىدۇ،

ھەي شۇنداق قىلىپ ئۇ جومباق يۈرەككە ئوتنى سېپ قويۇپ كەتتى.ئۆيگە بېرىپ بېشىم قىززىپ ناھايىتى بىئارام
بولۇپ،سوغۇق ئىچىملىكتىن نەچچىنى ئىچىۋەتتىم ئۇنى بىلمىدىم.لىكىن بۇ جايىل يۈرەك بولدى قىلىدىغاندەك ئەمەس. بىر كىچە خىيال بىلەن ئۆتۈپ كەتتى دىسە،،،،،،،،،،،،
مانا تاڭمۇ ئاتتى،خىيالىم يەنىلا شۇ قىزدا،ئۇيان ئويلاپ-بۇيان ئويلاپ،ئاخىرى بۈگۈنمۇ ساقلاپ تەلەپ قويۇش
نىيىتىگە كەلدىم. لىكىن يانچۇققا قارىسام پۇل يوق، ئاپلا،ئۇ جومباقنى كەلتۈرگەن بىلەن مىھمان قىلىشقا پۇل
بولمىسا......  ئاخىرى بىرەر يەرگە بېرىپ مەدىكار ئىشلەپ بولسىمۇ ،قىزنى مىھمان قىلغىدەك پۇل تاپاي دەپ
ماڭدىم،مەندەك دات باسقان قىلىچنى خەخمۇ ئىشلەتمىدى،ئاخىرى بىر باقمىچىنىڭ 2 ئېغىلنىڭ قىغىنى100يۈەنگە
دىيىشىپ ئىشلەشكە باشلىدىم،ھەي نىمىدىگەن جاپا بۇ،چۈش بولغىچە قىلىپ بولالامەنمۇ،،،،،،،
قىز بالىغا كۆيگەنگە تويە قىلىچ،گەندىگە كىچىپ يۈرۈپ بۇ ئىشنىمۇ تۈگتتىم،دەرھال باغچىغا قاراپ ماڭدىم.
كىيىم كىچەكنىمۇ ئالماشتۇرمىدىم،تۈنۈگۈن پىچچەت كىيىنىپ بارسام ياقتۇرمىدى ئەمەسمۇ.بولدىلا سەھرالىق
بولغاندىكەن مۇشۇنداقلا باراي،ئىشتىنىمنىڭ ئايىغىنى تۈرۈپ كەش كەيمەيلا ماڭدىم..........
يولدا،ئۇ بۇ قاراپ ئادەمنىمۇ نۇمۇس قىلدۇرىۋەتتى،خۇددى بۇ خەقتە مىنىڭ قەرزىم باردەك.
باغچىغىمۇ كەپ قالدىم،كۆزۈم قىزنىڭ يولىدا،ئولتۇرۇپ قىلىدىغان گەپلەرنى ئويلىغاچ تۇردۇم،
ھا....كەلدى.بۈگۈن تۈنۈگۈنكىدىنمۇ چىرايلىق بوپ كىتىپتۇ،يۈرەكلىرىم ئويناپ كەتتى.لىكىن جىددىلىشىۋاتىمەن.
قىزچاق ئىتتىك-ئىتتىك ماڭماقتا ئىدى،مەن يۈگۈرەپ ئالدىغىلا بېرىشىمغا،نىمىگە پۇتلاشتىم،قىزچاقنىڭ
ئالدىغىلا يىقىلىپ كەتتىم،نىمە نۇمۇس بۇ،
ھۇف، قىزچاق قورقۇپ كەتكەن چېغى جايىدىلا جىم تۇرۇپ ماڭا بىر قارىدى،
ياخشىمۇ سىز گۈزەل قىز ،يۈرەك ئۇنىماي،سىزنى ساقلاپ تۇردۇم.
خىڭ،سىزگە قىلىدىغان گىپىم يوق.      قىز شۇنداق دەپ ئالدىغا قاراپ ماڭدى.
توختاڭ گۈزىلىم مەن سىزگە ئاشىق،يۈگۈرۈپ بېرىپ ئالدىغا ئۆتىۋالدىم.
لۈكچەك،نېرى تۇر ،دىدىمغۇ مەن ئالدىراش،
ۋاي توختاڭ،بۈگۈن زادى تونۇشقىنىم تونۇشقان،ئالدىغىلا ئۆتۈپ يېنىمدىن پىچاقنى چىقىرىپ،
بۈگۈن مىنى رەت قىلسىڭىز،ئالدىڭىزدا ئۆلىۋالىمەن،دەپلا پىچاقتا قولۇمنى جىجاپ ئازراق قان چىقاردىم.
ۋاي نىمە قىلغىنىڭىز بۇ،كاللىڭىزدا چاتاق بارمۇ؟تىزراق قاننى توختىتايلى،      دىدى جومباق.
چىقىۋەرسۇن،سىزسىز بۇ جاننىڭ ماڭا كىرىكى يوق. دەپ تۇرىۋالدىم،كۆزىگە قاراپ،
ھەي.بەك جاھىلكەنسىز،بوپتۇ سىز دىگەندەك بولسۇن،لىكىن قولىڭىزنى تاڭدۇرايلى،دىدى ئۇ.
بولىدۇ بولىدۇ،راستىنلا قوشۇلدىڭىز،ئەكىلىڭ قولىڭىزدىكى نەرسىلەرنى مەن كۆتۈرگەچ،سىزنى ئۆيگە ئاپىرىپ
قوياي، دەرھا ئۇنىڭ قولىكىنى تارتىپلا كۆتۈرگەچ ماڭدىم.
ئىسمىم ئالىم،سىزنىڭچۇ؟ ئىسمىم ئادىلە،     
پاھ،ئالىم ئادىلە،نىمىدىگەن ماس كەلگەن ھە.ھا ھا ھا،......
بۇنچە كۆپ نەرسىلەرنى قانداق قىلىسىز؟
ئاپامنىڭ مىجەزى يوق،شۇڭا، ئۇ تىلىنى چايناپ بوش جاۋاپ بەردى.
ھە....توختاڭ، ئىككىمىز بۇ يەردە بىر دۇئا قىلايلى. دەپلا چاپىنىمنى يەرگە سالدىم،كىلىڭ ئولتۇرۇڭ بۇ يەردە،
مەن چىملىقتا ئولتۇرۇپلا،ئاسمانغا قاراپ،كۆزۈمنى يۇمۇۋېلىپ،بىر نەچچە سۈرە ئوقىدىم.
ئاندىن دەرھاللا،يىغلاپ تۇرۇپ دۇئا قىلدىم،ھۆ ھۆ ھۆ........ خۇدايىم ھاجىم ئاپامغا شىپالىق بەرگەيلا،
كىسىلىنى ساقايتقايلا،دەپ ئۇزۇن دۇئا قىلدىم،
ئادىلە دەرھاللا بىر قولياغلىق سۇنۇپ،  يۈزىڭىزنى ئېيتىۋېلىڭ،ماڭايلى،سەت تۇرىدۇ،ئادەم جىق دىدى.
ماقۇل،دەپ قولياغلىقتا يۈز كۆزۈمنى بىر قۇر ئېيتتتىم.
قەتئىي خاپا بولماڭ،مەن تۈنۈگۈن سىزگە ھۆرمەتسىزلىك قىپتىمەن،دىدى ئادىلە يەرگە قاراپ.
نىمىسىنى دەيسىز.ئېلىشمىغىچە تېپىشماس دىگەن شۇ ،بولدى ماڭايلى،
ئىككەيلەن پاراڭلاشقاچ تاماق يىگىلى ماڭدۇق.
ئۇ پات-پات يالاڭياق پۇتۇمغا قاراپ قوياتتى.    مەن ئالدىمغا قاراپ مېڭىۋەردىم.
ئۇ ئالىي بىر ئاشخانىنىڭ ئالدىغا كەلدى.
بۇ ئاشخانىدا تاماق يىسەك،يۈز كوي پۇلۇم يەتمەيدىغاندەك تۇرىدۇ،ھەي مات بولۇپ قالامەنممۇ؟
ئادىلە بۇ ئاشخانىنى دوسلىرىم ياخشى ئەمەس دىگەن،باشقا يەرگە بارمايلىمۇ؟ دىدىم ئۇنىڭغا،
بۇ ئاشخانىنىڭ تامىقى ياخشى،ئىشەنمىسىڭىز تېتىپ بېقىپ ئاندىن گىپىڭىزنى قىلىڭ، دىدى ئۇ.
بوپتۇ ئەمسە كىرەيلى،دىدىم،
ئاشخانىغا كىرىپلا، ۋاي قورسىقىم،بەك ئاغرىۋاتىدۇ، ئادىلە سىز تاماق بۇيرۇتقاچ تۇرۇڭ،مەن ھاجەت قىلىۋالاي،
دەپ ھاجەتخانىغا ماڭدىم،. بوش قىلىچ ئەمەس مەن.ھى ھى ھى
چىقسام. دىگەندەك داپەنجى،كاۋاپ،سوغۇق سەي قىززىق سەيلەرنى راسا بۇيرۇتۇپتۇ،
تاماق يىگىچە ئەبجەش پاراڭ قىلىپ،ئۇنى قاتتىق كۈلدۈرىۋەتتىم،
قورساقمۇ راۋۇرۇس تويدى،ئۇنى ئۆيگە ئاپىرىپ قويىدىغان بولۇپ،پاراڭلىشىپ ماڭدۇق،
چايغا چۈشسە،ئىرىمەيدىغان قەن يوق،تۈنۈگۈن مىنى مات قىلغان،قىز بۈگۈن مابىتىمگە ئايلاندى ھا ھا ھا......
شۇنداق قىلىپ ئۇنىڭ ئۆيىنىڭ ئالدىغىمۇ كەلدۇق،
مەن ئۆزۈمنى تۈنۈگۈن دوختۇر،دىگەن بىلەن ئەسلىدە بىر يالاڭتۆش.
مەن ئۆيۈمگە كىرىپ كىتەي،كىيىن كۆرىشەرمىز دىدى ئۇ.
مۇشۇنداق كىتىشكە كۆڭۈل ئۇنامۇ؟ باشقا گەپ يوقمۇ جېنىم. ئۇنىڭغا قارىدىم.
سىزمۇ ئۆيىڭىزگە كەتمىسىڭىز بولماس؟دىدى ئۇ.
بوپتۇ.ئەمسە كىرىپ كىتىڭ،دىدىم ئۇنىڭغا،
لىكىن كەتكىم پەقەت كەلمىدى.....ئاخىرى كاللامغا بىر ئوي كىلىپ ناھايىتى خۇشال بولدۇم.
ئۇ يەردە يېرىم سائەتتەك تۇرۇپ،ئاندىن ئۇدۇل ئۇلارنىڭ ئۆيىگە كىردىم.
تاك تاك تاك،ئىشىكنى چەكتىم.  ئالدىمغا ياشتا چوڭ،پاكىز كىيىنگەن،ساقاللىق بىر ئادەم چىقتى،
ئەسسالامۇئەلەيكۇم  قاراجىمكا،دەپلا كۆرۈشتۈم،
ھۆ ھۆ ھۆ ...قاراجىمكا،مەن بىر مۇساپىر،بۇ شەھەرگە كەلگىلى ئىككى يىل بولدى،تاپقىنىمنى قورساققا يەپ
كۈن ئۆتكۈزىۋاتقان،بۈگۈن قورسىقىم ئاچ قالدى،مەندىن يانمىسا خۇدادىن يانا،دەپ يىغلاپ تۇرىۋالدىم.
ۋاي تۇرسىلا ئۇكام،يۈرسىلە ھەممىمىز مۇسۇلمان ئادەم،ئۆيگە كىرسىلە، ئۇنىڭغا ئەگىشىپ ئۆيىگە كىردىم،
ئادىلەنىڭ مىنى كۆرۈپ،يۈزىدە قان قالماي،ماڭا ھومىيىپ خۇددى..........
مەن دەرھاللا ئۆيگە كىرىپ،يەردىلا ئولتۇرۇپ،سۈرە ياسىننىڭ يەتتىلى موبىنىنى ئوقۇپ خەتمە قۇرئان قىلدىم،
تۈگىتىپلا،  ھۆ ھۆ ھۆ............... دەپ دۇئا قىلىشقا باشلىدىم،
خۇدايىم قاراجىمكامنىڭ تىنىنى سالامەت قىلغايلا،ھاجىم ئاپامغا شىپالىق بەرگەيلا، دەپ ئۇزۇن دۇئا قىلىۋەتتىم.

قاراجىمكام،ۋاي قاراخۇنۇم ئورنىلىرىدىن تۇرسىلا،پەستە ئولتۇرغانلىرى نىمىسى دەپ قولۇمدىن تارتىپ،
تۆگە باشلىدى.          ئۇكام ئۆزلىرى نىمە ئىش قىلىلا؟
ھۆ ھۆ ھۆ ........قاراجىمكا مەن بىر قارا مۇساپىر،بەزىدە ئۆيۈم كوچا بولسا بەزىدە يولدىمۇ ئۇخلاۋىرىمەن،
بۇمۇ خۇدانىڭ پۈتكىنى ئوخشايدۇ،قانداق قىلىمىز دەپ يىغلاشقا باشلىدىم.
ئۇكام ئاۋۋال تاماق يەيلى،    ئادىلە تاماق پىشقان بولسا ئەكىر قىزىم؟   ماقۇل دادا، ئادىلە بىردەم تۇرۇپ
لەغمەننى ئېلىپ كىردى.
پاھ،نىمىدىگەن ئوخشىغان ئاش،   
تاماق يەپ بولۇپ،قاراجىمكام،     ئۇكام ئىشىك ئالدىدىكى ئۆيدە ئادەم يوق،خالىسىلا ئاشۇ يەردە ئولتۇرسىلا،
بىرەر ئىشنىڭ پىشىنى تۇتمىغان بولسىلا،مەن ئىشقا سالاي،دىدى،
ۋاي ماقۇل،قاراجىمكا مەن ئېغىلدا بولسىمۇ يېتىۋىرىمەن،ئۈستىمگە قار-يامغۇر چۈشمىسىلا،دىدىم ئۇنىڭغا،
پەستىكى ئۆيگە كىردۇق.ئۆي مۇشۇ ئۇكام،ھېلى ئادىلە يوتقان-كۆرپە ئەكىرىپ بەرسۇن.
قاراجىمكام ئۆينى كۆرسىتىپ چىقىپ كەتتى.
ھەي............ھا ھا ھا     بۇمۇ خۇدانىڭ بەرگىنى،  بىردەم تۇرسام ئادىلە،كىگىز-كىچەكلەرنى ئېلىپ كەلدى،
ئۇف........ئادەمنى ئەجەپ قورقۇتتىڭىز،بۇنداق قىلىشىڭىزنى ئويلاپمۇ باقماپتىكەنمەن،دىدى ئادىلە،
سىز ئويلىمىغان ئىش تېخى جىق جېنىم. سىزنى دەپلا قىلىۋەتتىمغۇ ھەممە ئىشنى.
دىگەن بىلەن،راستىنلا بىلىملىك قارىمكەنسىز جۇمۇ،
ھا ھا ھا،بىلىمسىزمىكى دىگەنما، ئابدۇكىرىم ئابلىزنىڭ شاگىرتىدا ئۈچ كىلو ۋىجدان بولسا،
مانا بىزدە ئالتە كىلو ۋىجدان بار جۇمۇ،ھا ھا ھا.......ئىككىمىز قاتتىق كۈلۈشۈپ كەتتۇق،
ئەمسە مەن چىكەتتىم،يوتقان كۆرپىلەر ئەنە.
خوش جېنىم،دىدى ئۇمۇ مىنى دوراپ،
ئاھ ئۇنىڭ ئاۋۇ گىپىدىن،ما يۈرەكنىڭ ئوتى،ئون ھەسسىدىن يۈزگە چىقتى،
جېنىم،بىرنى كىلىۋەتمەي ماڭامسىز؟دىدىم قېلىنلىق قىلىپ،كۆزىگە قاراپ.
خىڭ...            ئۇ نۇمۇس ئارىلاش ھىجىيىپ ئىشىكنى يېپىپ چىقىپ كەتتى،

ھەي مەن نىمىدىگەن بەختلىك،تورۇسقا قاراپ خىيال سۈرۈشكە باشلىدىم.

دىراممىنىڭ ئاخىرىنى باشلاي ئەمسە،لىكىن بۈگۈن دىرامما ھىكايە شەكىلدە باشلىنىدۇ،

ھەي شۇنداق قىلىپ ئۇ جومباق يۈرەككە ئوتنى سېپ قويۇپ كەتتى.ئۆيگە بېرىپ بېشىم قىززىپ ناھايىتى بىئارام
بولۇپ،سوغۇق ئىچىملىكتىن نەچچىنى ئىچىۋەتتىم ئۇنى بىلمىدىم.لىكىن بۇ جايىل يۈرەك بولدى قىلىدىغاندەك ئەمەس. بىر كىچە خىيال بىلەن ئۆتۈپ كەتتى دىسە،،،،،،،،،،،،
مانا تاڭمۇ ئاتتى،خىيالىم يەنىلا شۇ قىزدا،ئۇيان ئويلاپ-بۇيان ئويلاپ،ئاخىرى بۈگۈنمۇ ساقلاپ تەلەپ قويۇش
نىيىتىگە كەلدىم. لىكىن يانچۇققا قارىسام پۇل يوق، ئاپلا،ئۇ جومباقنى كەلتۈرگەن بىلەن مىھمان قىلىشقا پۇل
بولمىسا......  ئاخىرى بىرەر يەرگە بېرىپ مەدىكار ئىشلەپ بولسىمۇ ،قىزنى مىھمان قىلغىدەك پۇل تاپاي دەپ
ماڭدىم،مەندەك دات باسقان قىلىچنى خەخمۇ ئىشلەتمىدى،ئاخىرى بىر باقمىچىنىڭ 2 ئېغىلنىڭ قىغىنى100يۈەنگە
دىيىشىپ ئىشلەشكە باشلىدىم،ھەي نىمىدىگەن جاپا بۇ،چۈش بولغىچە قىلىپ بولالامەنمۇ،،،،،،،
قىز بالىغا كۆيگەنگە تويە قىلىچ،گەندىگە كىچىپ يۈرۈپ بۇ ئىشنىمۇ تۈگتتىم،دەرھال باغچىغا قاراپ ماڭدىم.
كىيىم كىچەكنىمۇ ئالماشتۇرمىدىم،تۈنۈگۈن پىچچەت كىيىنىپ بارسام ياقتۇرمىدى ئەمەسمۇ.بولدىلا سەھرالىق
بولغاندىكەن مۇشۇنداقلا باراي،ئىشتىنىمنىڭ ئايىغىنى تۈرۈپ كەش كەيمەيلا ماڭدىم..........
يولدا،ئۇ بۇ قاراپ ئادەمنىمۇ نۇمۇس قىلدۇرىۋەتتى،خۇددى بۇ خەقتە مىنىڭ قەرزىم باردەك.
باغچىغىمۇ كەپ قالدىم،كۆزۈم قىزنىڭ يولىدا،ئولتۇرۇپ قىلىدىغان گەپلەرنى ئويلىغاچ تۇردۇم،
ھا....كەلدى.بۈگۈن تۈنۈگۈنكىدىنمۇ چىرايلىق بوپ كىتىپتۇ،يۈرەكلىرىم ئويناپ كەتتى.لىكىن جىددىلىشىۋاتىمەن.
قىزچاق ئىتتىك-ئىتتىك ماڭماقتا ئىدى،مەن يۈگۈرەپ ئالدىغىلا بېرىشىمغا،نىمىگە پۇتلاشتىم،قىزچاقنىڭ
ئالدىغىلا يىقىلىپ كەتتىم،نىمە نۇمۇس بۇ،
ھۇف، قىزچاق قورقۇپ كەتكەن چېغى جايىدىلا جىم تۇرۇپ ماڭا بىر قارىدى،
ياخشىمۇ سىز گۈزەل قىز ،يۈرەك ئۇنىماي،سىزنى ساقلاپ تۇردۇم.
خىڭ،سىزگە قىلىدىغان گىپىم يوق.      قىز شۇنداق دەپ ئالدىغا قاراپ ماڭدى.
توختاڭ گۈزىلىم مەن سىزگە ئاشىق،يۈگۈرۈپ بېرىپ ئالدىغا ئۆتىۋالدىم.
لۈكچەك،نېرى تۇر ،دىدىمغۇ مەن ئالدىراش،
ۋاي توختاڭ،بۈگۈن زادى تونۇشقىنىم تونۇشقان،ئالدىغىلا ئۆتۈپ يېنىمدىن پىچاقنى چىقىرىپ،
بۈگۈن مىنى رەت قىلسىڭىز،ئالدىڭىزدا ئۆلىۋالىمەن،دەپلا پىچاقتا قولۇمنى جىجاپ ئازراق قان چىقاردىم.
ۋاي نىمە قىلغىنىڭىز بۇ،كاللىڭىزدا چاتاق بارمۇ؟تىزراق قاننى توختىتايلى،      دىدى جومباق.
چىقىۋەرسۇن،سىزسىز بۇ جاننىڭ ماڭا كىرىكى يوق. دەپ تۇرىۋالدىم،كۆزىگە قاراپ،
ھەي.بەك جاھىلكەنسىز،بوپتۇ سىز دىگەندەك بولسۇن،لىكىن قولىڭىزنى تاڭدۇرايلى،دىدى ئۇ.
بولىدۇ بولىدۇ،راستىنلا قوشۇلدىڭىز،ئەكىلىڭ قولىڭىزدىكى نەرسىلەرنى مەن كۆتۈرگەچ،سىزنى ئۆيگە ئاپىرىپ
قوياي، دەرھا ئۇنىڭ قولىكىنى تارتىپلا كۆتۈرگەچ ماڭدىم.
ئىسمىم ئالىم،سىزنىڭچۇ؟ ئىسمىم ئادىلە،     
پاھ،ئالىم ئادىلە،نىمىدىگەن ماس كەلگەن ھە.ھا ھا ھا،......
بۇنچە كۆپ نەرسىلەرنى قانداق قىلىسىز؟
ئاپامنىڭ مىجەزى يوق،شۇڭا، ئۇ تىلىنى چايناپ بوش جاۋاپ بەردى.
ھە....توختاڭ، ئىككىمىز بۇ يەردە بىر دۇئا قىلايلى. دەپلا چاپىنىمنى يەرگە سالدىم،كىلىڭ ئولتۇرۇڭ بۇ يەردە،
مەن چىملىقتا ئولتۇرۇپلا،ئاسمانغا قاراپ،كۆزۈمنى يۇمۇۋېلىپ،بىر نەچچە سۈرە ئوقىدىم.
ئاندىن دەرھاللا،يىغلاپ تۇرۇپ دۇئا قىلدىم،ھۆ ھۆ ھۆ........ خۇدايىم ھاجىم ئاپامغا شىپالىق بەرگەيلا،
كىسىلىنى ساقايتقايلا،دەپ ئۇزۇن دۇئا قىلدىم،
ئادىلە دەرھاللا بىر قولياغلىق سۇنۇپ،  يۈزىڭىزنى ئېيتىۋېلىڭ،ماڭايلى،سەت تۇرىدۇ،ئادەم جىق دىدى.
ماقۇل،دەپ قولياغلىقتا يۈز كۆزۈمنى بىر قۇر ئېيتتتىم.
قەتئىي خاپا بولماڭ،مەن تۈنۈگۈن سىزگە ھۆرمەتسىزلىك قىپتىمەن،دىدى ئادىلە يەرگە قاراپ.
نىمىسىنى دەيسىز.ئېلىشمىغىچە تېپىشماس دىگەن شۇ ،بولدى ماڭايلى،
ئىككەيلەن پاراڭلاشقاچ تاماق يىگىلى ماڭدۇق.
ئۇ پات-پات يالاڭياق پۇتۇمغا قاراپ قوياتتى.    مەن ئالدىمغا قاراپ مېڭىۋەردىم.
ئۇ ئالىي بىر ئاشخانىنىڭ ئالدىغا كەلدى.
بۇ ئاشخانىدا تاماق يىسەك،يۈز كوي پۇلۇم يەتمەيدىغاندەك تۇرىدۇ،ھەي مات بولۇپ قالامەنممۇ؟
ئادىلە بۇ ئاشخانىنى دوسلىرىم ياخشى ئەمەس دىگەن،باشقا يەرگە بارمايلىمۇ؟ دىدىم ئۇنىڭغا،
بۇ ئاشخانىنىڭ تامىقى ياخشى،ئىشەنمىسىڭىز تېتىپ بېقىپ ئاندىن گىپىڭىزنى قىلىڭ، دىدى ئۇ.
بوپتۇ ئەمسە كىرەيلى،دىدىم،
ئاشخانىغا كىرىپلا، ۋاي قورسىقىم،بەك ئاغرىۋاتىدۇ، ئادىلە سىز تاماق بۇيرۇتقاچ تۇرۇڭ،مەن ھاجەت قىلىۋالاي،
دەپ ھاجەتخانىغا ماڭدىم،. بوش قىلىچ ئەمەس مەن.ھى ھى ھى
چىقسام. دىگەندەك داپەنجى،كاۋاپ،سوغۇق سەي قىززىق سەيلەرنى راسا بۇيرۇتۇپتۇ،
تاماق يىگىچە ئەبجەش پاراڭ قىلىپ،ئۇنى قاتتىق كۈلدۈرىۋەتتىم،
قورساقمۇ راۋۇرۇس تويدى،ئۇنى ئۆيگە ئاپىرىپ قويىدىغان بولۇپ،پاراڭلىشىپ ماڭدۇق،
چايغا چۈشسە،ئىرىمەيدىغان قەن يوق،تۈنۈگۈن مىنى مات قىلغان،قىز بۈگۈن مابىتىمگە ئايلاندى ھا ھا ھا......
شۇنداق قىلىپ ئۇنىڭ ئۆيىنىڭ ئالدىغىمۇ كەلدۇق،
مەن ئۆزۈمنى تۈنۈگۈن دوختۇر،دىگەن بىلەن ئەسلىدە بىر يالاڭتۆش.
مەن ئۆيۈمگە كىرىپ كىتەي،كىيىن كۆرىشەرمىز دىدى ئۇ.
مۇشۇنداق كىتىشكە كۆڭۈل ئۇنامۇ؟ باشقا گەپ يوقمۇ جېنىم. ئۇنىڭغا قارىدىم.
سىزمۇ ئۆيىڭىزگە كەتمىسىڭىز بولماس؟دىدى ئۇ.
بوپتۇ.ئەمسە كىرىپ كىتىڭ،دىدىم ئۇنىڭغا،
لىكىن كەتكىم پەقەت كەلمىدى.....ئاخىرى كاللامغا بىر ئوي كىلىپ ناھايىتى خۇشال بولدۇم.
ئۇ يەردە يېرىم سائەتتەك تۇرۇپ،ئاندىن ئۇدۇل ئۇلارنىڭ ئۆيىگە كىردىم.
تاك تاك تاك،ئىشىكنى چەكتىم.  ئالدىمغا ياشتا چوڭ،پاكىز كىيىنگەن،ساقاللىق بىر ئادەم چىقتى،
ئەسسالامۇئەلەيكۇم  قاراجىمكا،دەپلا كۆرۈشتۈم،
ھۆ ھۆ ھۆ ...قاراجىمكا،مەن بىر مۇساپىر،بۇ شەھەرگە كەلگىلى ئىككى يىل بولدى،تاپقىنىمنى قورساققا يەپ
كۈن ئۆتكۈزىۋاتقان،بۈگۈن قورسىقىم ئاچ قالدى،مەندىن يانمىسا خۇدادىن يانا،دەپ يىغلاپ تۇرىۋالدىم.
ۋاي تۇرسىلا ئۇكام،يۈرسىلە ھەممىمىز مۇسۇلمان ئادەم،ئۆيگە كىرسىلە، ئۇنىڭغا ئەگىشىپ ئۆيىگە كىردىم،
ئادىلەنىڭ مىنى كۆرۈپ،يۈزىدە قان قالماي،ماڭا ھومىيىپ خۇددى..........
مەن دەرھاللا ئۆيگە كىرىپ،يەردىلا ئولتۇرۇپ،سۈرە ياسىننىڭ يەتتىلى موبىنىنى ئوقۇپ خەتمە قۇرئان قىلدىم،
تۈگىتىپلا،  ھۆ ھۆ ھۆ............... دەپ دۇئا قىلىشقا باشلىدىم،
خۇدايىم قاراجىمكامنىڭ تىنىنى سالامەت قىلغايلا،ھاجىم ئاپامغا شىپالىق بەرگەيلا، دەپ ئۇزۇن دۇئا قىلىۋەتتىم.

قاراجىمكام،ۋاي قاراخۇنۇم ئورنىلىرىدىن تۇرسىلا،پەستە ئولتۇرغانلىرى نىمىسى دەپ قولۇمدىن تارتىپ،
تۆگە باشلىدى.          ئۇكام ئۆزلىرى نىمە ئىش قىلىلا؟
ھۆ ھۆ ھۆ ........قاراجىمكا مەن بىر قارا مۇساپىر،بەزىدە ئۆيۈم كوچا بولسا بەزىدە يولدىمۇ ئۇخلاۋىرىمەن،
بۇمۇ خۇدانىڭ پۈتكىنى ئوخشايدۇ،قانداق قىلىمىز دەپ يىغلاشقا باشلىدىم.
ئۇكام ئاۋۋال تاماق يەيلى،    ئادىلە تاماق پىشقان بولسا ئەكىر قىزىم؟   ماقۇل دادا، ئادىلە بىردەم تۇرۇپ
لەغمەننى ئېلىپ كىردى.
پاھ،نىمىدىگەن ئوخشىغان ئاش،   
تاماق يەپ بولۇپ،قاراجىمكام،     ئۇكام ئىشىك ئالدىدىكى ئۆيدە ئادەم يوق،خالىسىلا ئاشۇ يەردە ئولتۇرسىلا،
بىرەر ئىشنىڭ پىشىنى تۇتمىغان بولسىلا،مەن ئىشقا سالاي،دىدى،
ۋاي ماقۇل،قاراجىمكا مەن ئېغىلدا بولسىمۇ يېتىۋىرىمەن،ئۈستىمگە قار-يامغۇر چۈشمىسىلا،دىدىم ئۇنىڭغا،
پەستىكى ئۆيگە كىردۇق.ئۆي مۇشۇ ئۇكام،ھېلى ئادىلە يوتقان-كۆرپە ئەكىرىپ بەرسۇن.
قاراجىمكام ئۆينى كۆرسىتىپ چىقىپ كەتتى.
ھەي............ھا ھا ھا     بۇمۇ خۇدانىڭ بەرگىنى،  بىردەم تۇرسام ئادىلە،كىگىز-كىچەكلەرنى ئېلىپ كەلدى،
ئۇف........ئادەمنى ئەجەپ قورقۇتتىڭىز،بۇنداق قىلىشىڭىزنى ئويلاپمۇ باقماپتىكەنمەن،دىدى ئادىلە،
سىز ئويلىمىغان ئىش تېخى جىق جېنىم. سىزنى دەپلا قىلىۋەتتىمغۇ ھەممە ئىشنى.
دىگەن بىلەن،راستىنلا بىلىملىك قارىمكەنسىز جۇمۇ،
ھا ھا ھا،بىلىمسىزمىكى دىگەنما، ئابدۇكىرىم ئابلىزنىڭ شاگىرتىدا ئۈچ كىلو ۋىجدان بولسا،
مانا بىزدە ئالتە كىلو ۋىجدان بار جۇمۇ،ھا ھا ھا.......ئىككىمىز قاتتىق كۈلۈشۈپ كەتتۇق،
ئەمسە مەن چىكەتتىم،يوتقان كۆرپىلەر ئەنە.
خوش جېنىم،دىدى ئۇمۇ مىنى دوراپ،
ئاھ ئۇنىڭ ئاۋۇ گىپىدىن،ما يۈرەكنىڭ ئوتى،ئون ھەسسىدىن يۈزگە چىقتى،
جېنىم،بىرنى كىلىۋەتمەي ماڭامسىز؟دىدىم قېلىنلىق قىلىپ،كۆزىگە قاراپ.
خىڭ...            ئۇ نۇمۇس ئارىلاش ھىجىيىپ ئىشىكنى يېپىپ چىقىپ كەتتى،

ھەي مەن نىمىدىگەن بەختلىك،تورۇسقا قاراپ خىيال سۈرۈشكە باشلىدىم.
