يېڭى ئوقىغانلار،بىرىنچىسىدىن باشلاپ ئوقىمىسىڭىز چۈشىنەلمەيسىز.

ھەي،يازغانكەنمەن ئاخىرىغىچە يازاي.

تورۇسقا قاراپ خىيال سۈرۈپ، ئۆزۈممۇ بىلمەي ئۇيقۇغا كىتىپتىمەن.
تاك تاك تاك، دەرھال ئىشىك چىكىلدى، يۈرسىلە ئۇكام،نامازغا بېرىپ كىلەيلى، دىدى قاراجىمكام،
مەنمۇ دەرھاللا تاھارەت ئېلىپ نامازغا ماڭدۇق. نامازدىن كىيىن جامائەت بىلەن،بىر قاتار ئەھۋاللىشىپ،
ئۆيگە ماڭدۇق.  ئۆيدە ئەتىگەنلىك ناشتا قىلىپ بولۇپ،قاراجىمكامنىڭ ئارقىسىدىن ماڭدىم،
ئۇ بىر دۇككاننىڭ ئالدىغا كىلىپ ئىشىكىنى ئاچتى،    كىرەيلى،بۇ بىزنىڭ دۇككان قارىم ئۇكام،
ھە،بۇ زەگەرچىلىك دۇككىنى مۇشۇلارنىڭ ئىكەندە، دىدىم كۆڭلۈمدە،
شۇنداق قىلىپ،بەزىدە ئادىلەنىڭ دادىسىنىڭ دۇككىنمىغا بېرىپ،بەزىدە قۇرئان ئوقۇغان يەرگە بېرىپ،
بۇ يۇرتتا خېلى داڭقىممۇ چىقىپ قالدى،باشتا مەھەللىدىكى مەسجىدكە،ئىمام بولدۇم،بارا-بارا كىشلەر
پىكىر قىلىشىپ جۈمە مەسجىدكە ئىمام قىلىپ قويدى.لىكىن تەپسىرنى كىشلەر يىغلىغىدەك قىلىۋەتتىم دە.
كۈنلەر بىر-بىرلەپ ئۆتۈپ،ئۈچ ئايدىن ئېشىپتۇ، بىر كۈنى مەسجىدنىڭ بۇرۇنقى ئىمامى،ئالىمقارى ئۇكام،
قانچە ياشقا كىرلە؟ دىدى،
بۇ يىل،24 ياشقا كىرىپ قالدىم تەخسىر،دىدىم،
رەسۇلۇللاھ،سىلەردىن توي قىلىشقا قادىر بولالىغانلار ،توي قىلىۋالسۇن،قادىر بولالمىغانلار روزا تۇتسۇن،
دىگەن، سىلىمۇ جامائەت ھۆرمەتلەيدىغان ئىمام،توي قىلىۋالسىلا ياخشى بولاتتى،
كىمنىڭ قىزىنى دىسىلە،بىز سوراپ بىرەيلى،دىدى.
مەنمۇ تارتىنغان بولۇپ،يەرگە قارىغاچ، ئىبرايىم قاراجىمكامنىڭ قىزى،ئادىلە قانداق بولا،دىدىم.
بولمامدىغان ئاخۇنۇم،ئىبرايىم قاراجىممۇ سىلىنىڭ ئېغىزلىرىغا قاراپ قالغان،دىدى.
بىر نەچچە كۈندىلا توينى بىكىتىپ بوپتۇ، ۋاھ  خۇشاللىقىمدا بېشىم تورۇسقا تاقاشتىغۇ ھازىر.

شۇنداق قىلىپ تويۇممۇ بىكىتىلىپ بولدى. مەن دەرھال مۇنبەر چوڭلىرىنى، ئاش-سۇ ئېلىپ بېرىشقا بىللە
بېرىشىنى سورىدىم،
مىسرانىمكام باشلىق،قىزغىن،خوك،تارىم،مۇسەللەس،ئۇيغۇرۇم007.سۇلتان،بەھرام،ئارشاڭ،ھۆكۈمران،كۆكيال،كۆكيار،ھىسلىرىم،كۆكنۇر،مىلودىيە،بىلىگقۇت،
قىزلاردىن؛ كۈنقىز،گۈل-چۈشى،كەپسىز قىز،كۆڭۈل،مەلىكە،قار مەلىكە، مەلەك،سۈزۈك،ئايسىمان،گۈلسەبىر،
تومۇچۇق،نەركىس.دىلنازۇك،دىلناز،سەبدە،قاتارلىقلار ، مىسرانىم دىھقانچىلىق مەيدانىدىن،1000كالا،
1000قوي،1000 توخۇ. 10توننا بۇغداي،10توننا گۈرۈچنى ئېلىپ،ئۇزۇن كارۋان بولۇپ،يولغا چىقتۇق،
ۋە قالغان،13000 مىڭ ئەزانى توي كۈنى ئەتتىگەندە كىلىشكە،باغاق بەردۇق،
يۇرت ئەھلى،بۇنچىۋالا كۆپ چارۋىلارنى،ئۇن-گۈرۈچلەرنى كۆرۈپ،كىلىدىغان ئادەمدىن خەۋىرى يوق،
بۇنى قانداق قىلامىز دەپ،بېشى قېتىشتى.
شۇنداق قىلىپ ئەتە توي،بېشىم قېتىشقا باشلىدى،ھەي13000 كىشى بۇ كىچىك يۇرتقا پاتامۇ،
ئەتىسى تاڭمۇ ئاتتى،مۇنبەر ئەھلى،بەزىسى ئات،بەزىسى ئىشەك،ۋە بەزىلىرى مۇنبەردىكى كالىللىرىنى مىنىشىپ،
كەلدى، پاھ-شۇنداق تىزىلىپ كىتىپتۇكى،مەن بىر باشتىن كۆرۈشسەممۇ چۈشكىچە،ئاران كۆرۈشۈپ بولغۇدەكمەن،
ئاخىرى چوڭ بىر لابانى ئېلىپ، ئەسسالامۇئەلەيكۇم،مۇنبەر ئەھلى گۈزەل كورلىغا خۇش كەپسىلەر،
دەپ قاتتىق ۋارقىرىدىم،
ئېلىپ كىلىنگەن كالا-قويلارنى سويۇپ،بىر مو يەرگە، 120 قازان ئورنىتىلىپ،تاماقلار ئىتىلىشكە باشلىدى.
ھويلا،مەھەللە،ئادەم دېڭىزى،  مەن ئادىلە يۈتۈپ كىتەمۇ دەپ،يېنىغا كىرىپ ئولتۇرىۋالدىم.
دۇم تاڭ تاڭ دۇم تاڭ، سازلار چېلىنىشقا باشلىدى. بالىلار ھويلىغا پاتماي ئۇسۇل ئوينىماقتا،
بىردەمدىن كىيىن بىر مەيدان دىسكو باشلاندى،پاھ- قامچا،تاتلىقىم،كارۋان،تاڭنۇر،خەلقىم،چاپلىن،ئۆرۈكجان،
ئوتقاش،قەيسەر،ئەنۋەر،خەتەر،ئەلشىر،زامانى،ئاقسۇ سىرى،ھەسەل ھەرىسى،ئاشىق214،تەنھا204،جىمغۇر،
جىبران،جان پولات،جانپولات007،ئەسرارى.زەپەر،قەيسەر،چارتېكىن،ئۆزتىكىن،ئۇيغۇر تېكىن،ئۇيغرا،رومانتىكا،ماتىماتىكا،گۈزەل ئاتۇش،ئەھكام،ئەركەل،بۇلبۇل،ئۆرلەش،
مەلىكە،قارمەلىكە،شاتئاي،كۈلقىز،تۇمۇچۇق،ئاقمۈشۈك،سۇبات،جەلىپكار قىز،نادىرە،شىرىن،ئەتتىر،كۆزيېشى،
كەچمىشىم،دىلسىرى،گۈلسىرى،بەرگى،گۈلزىبا،ئۇيغۇرئاي،دىل، ياۋاگۈل،ئىسلامنۇر،نازىلئاي،.................قاتارلىقلار ھويلىنى چاڭ چىقىرىۋەتتى،قىزلارنىڭ بەزىلىرىنىڭ ياغلىقلىرى
چۈشۈپ قالسىمۇ،پەرۋاسىز ئوينىماقتا،ھويلىغا پاتمىغانلار،باغدا، ئىشقىلىپ يۇرتنىڭ بىكار يېرى قالماپتۇ،
بىر چاغدا،ئالا چۇقان ۆتۈرۈلۈپ كەتتى، نىمە بولغاندۇ دىسەك، نەسرۇللا،كەپسىز،بۆرە،سەنسىز،خاپىلىق،سەرۋان،
ئىزباسار،ئالدىراش،گىلادىئاتۇر،خالىد،قاتارلىقلار ئۆگزىدە دىسكو ئويناپ، كىلىن بار ئۆينىڭ ئۆگزىسى
چۈشۈپ كىتىپتۇ ئەمەسمۇ.   ۋاي مىنىڭ ئادىلەرىم قانداقراق؟دەپ يۈگۈرەپ كىرسەم، ھەممەيلەن ساق سالامەتكەن، بولدى توختىتىڭلار سازنى ،ئاران توي قىلغاندا خوتۇنۇمدىن ئايرىلىپ قالماي مەن.
توختىتىشقا ئۇلار ئۇنىماي. تاڭسا ئوينايمىز دەپ،تاڭسىغا چۈشتى،
كۆڭۈل بىلەن تاڭنۇر،كارۋان بىلەن كۈلقىز، نەسرۇللا بىلەن شوخ مەلىكە، ئۆرۈكجان بىلەن تومۇچۇق،
ئىقىلىپ بۇلار تاڭسىنىمۇ ھويلىغا پاتماي.باغدا،ئۆگزىدە،ھەتتا ئېغىلدىمۇ ئويناشتى،
سوررۇن راسا قىززىغاندا، ئىشىك تەرەپتىن،ھەمزە،ھۆكۈمران،كۆركەم،كارۋانجان،تۇز،ئاشىقجان،يارىماس،
قاتارلىقلار،كىلىپ توختىتىڭلار مۇشۇمۇ مۇسۇماننىڭ تويى بولدىمۇ، دەپ ھەمزە يۈرەككە ئۆتكىدەك،گەپتىن
بىر نەچچە ئېغىز قىلىۋەتتى،بايا ئۇسۇل ئوينىغانلار،ھازىر تەسىرلىنىپ يىغلىماقتا،
نىكاھنى ئوقۇيلى، ئىچكىركى ئۆيگە كىرىپ نىكاھ ئوقۇلدى.يۇرت ئۆلىمالىرى بىلەن مۇنبەر ئۆلىمالىرى،
تۇزغا نان چىلاپ يىسۇن دەپ، قاتتىق بىر مەيدان ئىختىلاپ بولۇپ،ھەمزە ئۇلارنى قايىل قىلدى.
كەچمۇ بولدى.يولغا چىقىپ قارىسام،كۆز يگـۆۈسىز كىشلەر،  مۇنبەرداشلار بۈگۈن قونىدىغان بولۇپ،سويغان
كالا،قوي تىرىلىرىنى تېگىغا سېلىپ يېتىشىپتۇ.
ئۆيگە قايتىپ،ئادىلە ئىككىمىز تويدا چۈشكەن پۇللارنى ساناشقا باشلىدۇق.
ساناتىمىز-ساناتىمىز تۈگەيدىغاندەك ئەمەس. تاك تاك تاك ئىشىك چىكىلدى،
كىمدۇ بۇ ئادەمنى ئارامىدا قويمىغان، ئىشىكنى ئاچسام مەھەللنىڭ مەزىنى،ئاخۇنۇم دوست-بۇرادەرلىرى
ماڭىدىغان بوپتۇ،خوشلىسىۋالسىلا، ھە  تاڭ ئاتتىما، ئاپلا توي كىچىسىنى پۇل ساناپ ئۆتكۈزىۋىتىپتىمەن ئەمەسمۇ؟
تويە ئاچكۆز قىلىچ.
مۇنبەرداشلىرىم بىلەن يىغلاپ ئۇزاتتىم.  ھەي تويمۇ تۈگىدى.
لىكىن ئەخلەت-ئۇخلەت ماڭا قالدى،بۇ ئەخلەتلارنى بىرەر ئاي توشۇمىسام،تۈگەيدىغاندەك ئەمەس.
ئەي يۇرت ئەھلى بىلەن.قازان ئېسىپلا تۈگەتسەك بوپتىكەن ئەسلى،تويە ھاماقەت قىلىچ،
تويۇم بەك قىزىدى لىكىن. ئاۋۇ تىشىلگەن ئۆگزىنى قانداق قىلامەن.
تويدا ئادىلە بىلەن،نىكاھ ئوقۇلماستا بىر مەيدان تاڭسا ئوينىغان،كىچىك ئۆيدە ئادەم كۆرمەيدۇ دەپ،
كىم كۆرۈپ قالدى. غولغولا قىلىپ مەلىنىڭ خەقلىرى مىنى ئىماملىقتىنمۇ چۈشىرىۋەتتى،ھەي.............
توي قىلغانغا تويە قىلىچ،ئەمدى سىنى دات باسقىنى باسقان.

دىگەن، سىلىمۇ جامائەت ھۆرمەتلەيدىغان ئىمام،توي قىلىۋالسىلا ياخشى بولاتتى،
كىمنىڭ قىزىنى دىسىلە،بىز سوراپ بىرەيلى،دىدى.
مەنمۇ تارتىنغان بولۇپ،يەرگە قارىغاچ، ئىبرايىم قاراجىمكامنىڭ قىزى،ئادىلە قانداق بولا،دىدىم.
بولمامدىغان ئاخۇنۇم،ئىبرايىم قاراجىممۇ سىلىنىڭ ئېغىزلىرىغا قاراپ قالغان،دىدى.
بىر نەچچە كۈندىلا توينى بىكىتىپ بوپتۇ، ۋاھ  خۇشاللىقىمدا بېشىم تورۇسقا تاقاشتىغۇ ھازىر.

شۇنداق قىلىپ تويۇممۇ بىكىتىلىپ بولدى. مەن دەرھال مۇنبەر چوڭلىرىنى، ئاش-سۇ ئېلىپ بېرىشقا بىللە
بېرىشىنى سورىدىم،
مىسرانىمكام باشلىق،قىزغىن،خوك،تارىم،مۇسەللەس،ئۇيغۇرۇم007.سۇلتان،بەھرام،ئارشاڭ،ھۆكۈمران،كۆكيال،كۆكيار،ھىسلىرىم،كۆكنۇر،مىلودىيە،بىلىگقۇت،
قىزلاردىن؛ كۈنقىز،گۈل-چۈشى،كەپسىز قىز،كۆڭۈل،مەلىكە،قار مەلىكە، مەلەك،سۈزۈك،ئايسىمان،گۈلسەبىر،
تومۇچۇق،نەركىس.دىلنازۇك،دىلناز،سەبدە،قاتارلىقلار ، مىسرانىم دىھقانچىلىق مەيدانىدىن،1000كالا،
1000قوي،1000 توخۇ. 10توننا بۇغداي،10توننا گۈرۈچنى ئېلىپ،ئۇزۇن كارۋان بولۇپ،يولغا چىقتۇق،
ۋە قالغان،13000 مىڭ ئەزانى توي كۈنى ئەتتىگەندە كىلىشكە،باغاق بەردۇق،
يۇرت ئەھلى،بۇنچىۋالا كۆپ چارۋىلارنى،ئۇن-گۈرۈچلەرنى كۆرۈپ،كىلىدىغان ئادەمدىن خەۋىرى يوق،
بۇنى قانداق قىلامىز دەپ،بېشى قېتىشتى.
شۇنداق قىلىپ ئەتە توي،بېشىم قېتىشقا باشلىدى،ھەي13000 كىشى بۇ كىچىك يۇرتقا پاتامۇ،
ئەتىسى تاڭمۇ ئاتتى،مۇنبەر ئەھلى،بەزىسى ئات،بەزىسى ئىشەك،ۋە بەزىلىرى مۇنبەردىكى كالىللىرىنى مىنىشىپ،
كەلدى، پاھ-شۇنداق تىزىلىپ كىتىپتۇكى،مەن بىر باشتىن كۆرۈشسەممۇ چۈشكىچە،ئاران كۆرۈشۈپ بولغۇدەكمەن،
ئاخىرى چوڭ بىر لابانى ئېلىپ، ئەسسالامۇئەلەيكۇم،مۇنبەر ئەھلى گۈزەل كورلىغا خۇش كەپسىلەر،
دەپ قاتتىق ۋارقىرىدىم،
ئېلىپ كىلىنگەن كالا-قويلارنى سويۇپ،بىر مو يەرگە، 120 قازان ئورنىتىلىپ،تاماقلار ئىتىلىشكە باشلىدى.
ھويلا،مەھەللە،ئادەم دېڭىزى،  مەن ئادىلە يۈتۈپ كىتەمۇ دەپ،يېنىغا كىرىپ ئولتۇرىۋالدىم.
دۇم تاڭ تاڭ دۇم تاڭ، سازلار چېلىنىشقا باشلىدى. بالىلار ھويلىغا پاتماي ئۇسۇل ئوينىماقتا،

ئوتقاش،قەيسەر،ئەنۋەر،خەتەر،ئەلشىر،زامانى،ئاقسۇ سىرى،ھەسەل ھەرىسى،ئاشىق214،تەنھا204،جىمغۇر،

چۈشۈپ قالسىمۇ،پەرۋاسىز ئوينىماقتا،ھويلىغا پاتمىغانلار،باغدا، ئىشقىلىپ يۇرتنىڭ بىكار يېرى قالماپتۇ،
بىر چاغدا،ئالا چۇقان ۆتۈرۈلۈپ كەتتى، نىمە بولغاندۇ دىسەك، نەسرۇللا،كەپسىز،بۆرە،سەنسىز،خاپىلىق،سەرۋان،
ئىزباسار،ئالدىراش،گىلادىئاتۇر،خالىد،قاتارلىقلار ئۆگزىدە دىسكو ئويناپ، كىلىن بار ئۆينىڭ ئۆگزىسى
چۈشۈپ كىتىپتۇ ئەمەسمۇ.   ۋاي مىنىڭ ئادىلەرىم قانداقراق؟دەپ يۈگۈرەپ كىرسەم، ھەممەيلەن ساق سالامەتكەن، بولدى توختىتىڭلار سازنى ،ئاران توي قىلغاندا خوتۇنۇمدىن ئايرىلىپ قالماي مەن.
توختىتىشقا ئۇلار ئۇنىماي. تاڭسا ئوينايمىز دەپ،تاڭسىغا چۈشتى،
كۆڭۈل بىلەن تاڭنۇر،كارۋان بىلەن كۈلقىز، نەسرۇللا بىلەن شوخ مەلىكە، ئۆرۈكجان بىلەن تومۇچۇق،
ئىقىلىپ بۇلار تاڭسىنىمۇ ھويلىغا پاتماي.باغدا،ئۆگزىدە،ھەتتا ئېغىلدىمۇ ئويناشتى،
سوررۇن راسا قىززىغاندا، ئىشىك تەرەپتىن،ھەمزە،ھۆكۈمران،كۆركەم،كارۋانجان،تۇز،ئاشىقجان،يارىماس،
قاتارلىقلار،كىلىپ توختىتىڭلار مۇشۇمۇ مۇسۇماننىڭ تويى بولدىمۇ، دەپ ھەمزە يۈرەككە ئۆتكىدەك،گەپتىن
بىر نەچچە ئېغىز قىلىۋەتتى،بايا ئۇسۇل ئوينىغانلار،ھازىر تەسىرلىنىپ يىغلىماقتا،
نىكاھنى ئوقۇيلى، ئىچكىركى ئۆيگە كىرىپ نىكاھ ئوقۇلدى.يۇرت ئۆلىمالىرى بىلەن مۇنبەر ئۆلىمالىرى،
تۇزغا نان چىلاپ يىسۇن دەپ، قاتتىق بىر مەيدان ئىختىلاپ بولۇپ،ھەمزە ئۇلارنى قايىل قىلدى.

كالا،قوي تىرىلىرىنى تېگىغا سېلىپ يېتىشىپتۇ.
ئۆيگە قايتىپ،ئادىلە ئىككىمىز تويدا چۈشكەن پۇللارنى ساناشقا باشلىدۇق.

ئاش سۈيى ئېلىپ بېرىشقا بىللە بارساڭلار ، تون كىيگۈزسەم دەپ شۇنداق يالۋۇرغانتى . ماقۇل بىكارلىق تون بولغاندىن كېيىن باراي دەپ ، بارسام نەدىكىنى تون تۇرماق بىرەر تال دوپپىغىمۇ ئېرىشەلمىدىم دىسە ، دۇنيادا شۇنداقمۇ ، يالغانچى ، پىخسىق ئادەملەر بولىدىكەن .
بۇندىن كېيىن مۇشۇنداق ئىشلارغا ئىشتىراك قىلساڭلار ئاۋال نەق بولسۇن ، بولمىسا مېنىڭ نىسىدىن قىلچە تەلىيىم يوق .
ھېسسىيات قارىشىڭنى ئەقلىڭنىڭ ئۈستىگە قويغان كۆنۈڭ، سېنىڭ خاتالىشىشقا يول قويغان تۇنجى كۈنۈڭدۇر.
