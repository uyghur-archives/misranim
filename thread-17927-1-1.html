    ئەسسالامۇ ئەلەيكۇم مىسرانىمدىكى ئۇششاق چوڭ پىششىق توڭ ،ئەزالىرىمىز ،ھەممىلىرى چىژ تۇرۇشلا ،مۇنبەرداشلىرىمىز قەشقەرنىڭ يەرلىك،پاراڭلىرى ،قەشقەرنىڭ سىسىق گەپلىرىنى يازدى بۈگۈن مىنىڭمۇ قولۇم قىچىشىپ قەشقەرنىڭ ئوتتەك پاراڭلىرىنى يىزىپ چىقتىم .
قاملاشمىسىمۇ ماختاپ قويۇشالا .

    قەشقەردە توي دىگەن بەك قىززىيدۇ دىسە مەن شۇ تويلاردىكى قىزىقارلىق پاراڭلاردىن باشلاي .
بىركۈنى بىر ئاغىنەمنىڭ تويى بولدىدە ،
ئۇئاغىنەم ،تەمتىرەپ قالدىمۇ،ئۇنى ئۇقمىدىم ،ئىمام ئاللىمۇ دىگەندە ،ئالدۇق دەپ سالدى دە،ئىمام ھومىيىپ قاراپ
قوياپ ئۇكام سىلى سىلى ئىشەكمۇ يا يە نى شىرىك ھەيدەيدىغان دىمەسما .

   نىكاھ تۈگەپ ھەممەيلەن پولا يىگىنى ئوتتاردۇق دە ،بىر چاغدا بىرەيلەن باللا شاسۇ بامىدۇ ماھارۋامنى توككا سىلىۋالسام دەپتى .
بايىقى خاپىچىلىقنى تۈگىتەلمەيۋاتقانلاردىن بىرسى .
شاسۇ ئىزدەپ ئاۋارە بومماي ئا بۇلۇڭۈىلا كىلە ،ئادىلجان ئەتىگەننىڭ ياغى جۇڭدۇڭدا تۇرۇپ كۆ پ توك چىكەتتى شۇنىڭ بۇرنىغىلام .... .

    ئاخشىمى ئۇسسۇل ئوينار ۋاقتىدا ،باللا ئوسسۇل ئويناپ كىتىشتى دە ئوينايدۇ ئوينايدۇ ،قوللىرى بىلىدىن كۆتۈرۈلمەيدۇ ،بىر ۋاىتا بىر ئەدەم قىشىمغا كەپتۇ .
ئۇكام باللا قۇچقاچ قورىۋاتامدا دەيدا .

     ئەتىسى يۈز ئاچقۇ دەپ بىر ئىش با ،ھەممەيلەن چاقچاقلىشىپ ئوتىراپ گەپ ساقالغا كىلىپ توختىدى دە .
بىرى قوپاپ ھەي سەندە با ساقال ئەچىكىدىمۇ با دىۋىدى يەنە بىرى شاڭخوقىلىپ .
ساقال دىگەن دادىسىنى دورىغان ئەدەمگە بۇ ئاغىنىمىز چىشى جەمەتىنى تارتىپ قاغان ئوخشايدا .
يەنە گەپ بەرمەي باشلاپ كەتتى دە .
بىر كۈنى سەتىراچخانىغا چىرىپتىم ،مۇشۇ ئاغىنەم بىلەن چىقاشىدا ،ۋاي مەن ،تۆلەيدىسەم ،ۋاي مەن تۆلەي دەپ ئاخىرى سەتىراچنىڭ پۇلىنى مەن تۆلىدىم،پوچى ئون كوينى بىرىپ چىقىپ كىتەي دەۋاتسام ئۇستام بەش كوينى تەڭلايدا ،ۋاي نىمە ئىش بۇ دىسەم ،ئاغىنىلىرىنىڭ ساقىلىنى سىلنىڭ ساقالنى ئاغان تىغدا ئاغانتىم دەيدا .

     شۇكۈنى تويدا پايپاق تاقىتىتىغان ئىش بار ئەمدى ،پايپاق تارقىتىپ كىلىپ بىر ئاغىنىمىز پايپاقچىكەن شۇنداق بولسىمۇ ئالدى ئىلىپلام پايپىقىنىڭ تۆپىگىلام كىيىۋا لدى دە .
يىنىدا ئوتتاغان بىرەيلەن ھە پايپاقچى پايپاقنى جۇپلاپ كىيىۋالدىڭغۇ،ئاخشاملىققا خوتۇنۇڭغا پايپاقنىڭ ھىسابىنى جىقراق بىرەي دەپ قالدىڭ ھە،
ۋاي سەن ئەمەس ئىدىش ئايالىدىن قوقاپ بەش موچەنلىك پىتەك كىيەمەيدىغان ،پىتىكىمنىڭ سانى ئازلاپ كىتىدۇ دەپ ئايىغىڭغا پىتەك سامماي يۈرىۋاتقىنىڭنى كىم بىلمەيدا .
ۋوي ،پايپاقنى جۇپلاپ كىيىۋالغان بىلەن يۇيىدىغاندا خوتۇن بۇرتىڭدا ئىلەڭگۈچ ئۇچامىكى .

پايپاق تارقىتىپ بولاپ يىقىنلارغا خاتىرە بولۇپ قالسۇن دەپ بىر دىن جەيناماز تارقاتتى دە ،
ئۆيدىن ئۇزايدىغان چاغدا يىگىتنىڭ دادىسى بالىلرىم خاپۇلاشماي يىنىشالا ،ئا جەيناماز قاتلىنىپ تۇرىۋەمىسۇن ئانچە -مۇنچە ناماز ئوقۇپ تۇراسىلە دىۋىدى .
باللادىن بىرسى ئوقۇيمىز ،ئوقۇيمىز خاتىرجەم بولسا ئاكا دىسە .
يىنىدا تۇرغان بىر ئاغىنىسى قوپاپ ھە ئوقۇيسەن <<خانتاجى>>نىڭ ئايرىمخانىسىغا سىلىپ ،مەيداندىن چۈشكەندىن كىيىن خۇپتەن ئۇقاسەن.

     بىر ئاغىنەم باتى چىچىنى بەك ئۇزۇن قويىدىغان بىر كۈنى بىرنىڭ ئۆيگە چىقتى دە ،
ماڭە سەھرانى بىر كۆرسىتىپ كىرەي دەپ مەلىگە ئىچىقتىم ئۇنى يولدا شودۇيجاڭ ئۇچلاپ قىلىپ مەن بىلەن كۆرۈشۈپ بولۇپ ھىلىقى ئاغىنەمنىڭ باشتىن ئايىغىغا بىر قارىۋىتىپ .
ئۇكەم سىلى غوجاممۇ ،خىنىممۇ ،ئانىلىرىغا دىسە تىزراق ياتلىقى قىلىۋەتسۇن ،بولمىسا ئوغلۇم با شۇنىڭغىلا ئەپبىرەيما .
نىمىشقا دەيلا ،باشلىرىدا قۇچقاچ ئۇۋلاپ بىر نەچچىنىڭ ئانىسى بولاپ قاممىسا يەنە .

    خاڭدىلىق ئاغىنەم باتى شۇنىڭ يىنىڭغا چىقىپ چىرەي دەپ ماڭدىم دە يىرىم يولغا بىرىپ توكسىكىلىتنىڭ توكى تۈگەپ يۆتىلەپ مىڭىۋاتسام بىر بوۋاي ئىشىكىنى ھەيدەپ ئارقىمدىن كەپ قالدى دە .
ئۇكەم ئىشەكلىرى ھىرىپ قاغان ئوخشىمامدا ؟ئىشەككىمۇ چا مانداق يەمىن جىق بەسىلىرى ئوبدان ماڭىدا
قايلىسا ما ئىشىكىمگە 30يىل بولدى ھارماي مىڭىۋاتىدا .دەپتى ،ھار كەپتىكىن .
قىتەۋە <<بەمىگەن بوغازنى يىمىگەن>>قوتانىڭ ئىشىكى دەپ قويدۇم.

    ياشلار بايرىمىغۇ دەيمەن ،باللا بىلەن سەرادىن شەگە چىرىپ قالدۇق دە ،ئايلىنىپ كىتىپ بارساق باغچىنىڭ ئالدىدا تمۇچۇقتەك پەچەت قىز بالىدىن ئىككىسى كىتىپ بامامدا ئاغىنىلىرىم ۋاي كەلتۈرىلى،ۋاي ئەگىشىلى دەپ تۇرىۋالدى ،
مەن قوپاپ باللا مىكياننى دەپ خورازدىن تاياق يىمەيلى ئاستا يولىمىزغا ماڭايلى دىسەم پەقەت ئۇنىمىدى شۇنىڭ بىلەن باغچىمۇ باغچا كۆلمۇكۆل ئايلىنىپ يۇدۇق دە بىر ۋاختا ھىلىقى قىزدىن بىرسى ئالدىمىزغا گۆلىيىپ كىلىۋاتمامدا ئو مانا كۆرۈڭ كارامەتنى دەپ قايلاپ قىز كىلىپلا ،
غوجاملا يولدىن ئىزىپ قىلىشلىما ،تولا يول مىڭىپ ئاياغلىرىمۇ ئۇپراپ كىتىپتۇ ،ئاياغلىرىنى يەڭگۈشلىۋىلىشسا ،يىڭى بازانى بىلىشۇلا ھەقىچان ،شۇيەرگە بىرىشسىرى چەتتىرەك بىر بازار بار شۇيەردە دۆلەگلىك ئاياغ بەك جىق شۇنىڭدىن بىرەنى ئىلىشالا بىلىشەملا مەن باشلاپ بارايمۇ .

   سەيچىلىك قىلىۋاتقان ۋاقتلىرىمدا تازا كاكىراپ سەي سەي دەپ كىتىپ بارىتىم بىر ئايال چىقىپ سەۋزىلىرىنى قانچىدىن بەرلە دەپتى .باغلىق سەۋزىتى كويلىرىغا ئىككى دىسەم نىمانداق قىممەت ساتلا بالام .دەيدا
شۇنىڭ بىلەن ئاچا بۇ سەۋزىنىڭ باغۇ بىر قوي تويغىدەك كۆكىمۇچىقىدا دىسەم .
ئۇنداق بوسا سەۋزىلىرىنى ئالاي كۆكىدە ئانىلىرى ئاخشاملىققا مانتا سەپبەرسۇن دەيدا .

     بىر كۈنى تورخانىدا قىزىق ئىتىشىۋاتىتىم تىلفۇن كەپ قالدى دە .
قايلىسام تاغامكەن .ھە چىژلىقما ؟چىژ تۇدۇڭ دىسەم نەدىسەن دەيدا .
ھەمەن ۋاڭبادا بەنتىيو ئىتىۋاتقان دىسەم .
نىمە دىڭ -دىڭ ئىتىۋاتقان .
ياق....
ئۇرغۇيچە سۆزلىيە گىپىڭنى چۈشەنمىدىم دەيدا .

ئوتتەك پاراڭلىرىم ئەمدى قىزىۋاتىتى ما خوتنۇم تولا چاڭخوقىلىپ زىرىكتۈرىۋەتتى ياقتۇرساڭلا ئىنكاسلىرىمىزدا ئاخىرىنى يازارمىز .
قەشقەرنىڭ كۆزيىشى مىنىڭ كۆزيىشىم، قەشقەرنىڭ چۆكۈشى مىنىڭ قىسمىتىم .

    ئەسسالامۇ ئەلەيكۇم مىسرانىمدىكى ئۇششاق چوڭ پىششىق توڭ ،ئەزالىرىمىز ،ھەممىلىرى چىژ تۇرۇشلا ،مۇنبەرداشلىرىمىز قەشقەرنىڭ يەرلىك،پاراڭلىرى ،قەشقەرنىڭ سىسىق گەپلىرىنى يازدى بۈگۈن مىنىڭمۇ قولۇم قىچىشىپ قەشقەرنىڭ ئوتتەك پاراڭلىرىنى يىزىپ چىقتىم .
قاملاشمىسىمۇ ماختاپ قويۇشالا .

    قەشقەردە توي دىگەن بەك قىززىيدۇ دىسە مەن شۇ تويلاردىكى قىزىقارلىق پاراڭلاردىن باشلاي .
بىركۈنى بىر ئاغىنەمنىڭ تويى بولدىدە ،
ئۇئاغىنەم ،تەمتىرەپ قالدىمۇ،ئۇنى ئۇقمىدىم ،ئىمام ئاللىمۇ دىگەندە ،ئالدۇق دەپ سالدى دە،ئىمام ھومىيىپ قاراپ
قوياپ ئۇكام سىلى سىلى ئىشەكمۇ يا يە نى شىرىك ھەيدەيدىغان دىمەسما .

   نىكاھ تۈگەپ ھەممەيلەن پولا يىگىنى ئوتتاردۇق دە ،بىر چاغدا بىرەيلەن باللا شاسۇ بامىدۇ ماھارۋامنى توككا سىلىۋالسام دەپتى .
بايىقى خاپىچىلىقنى تۈگىتەلمەيۋاتقانلاردىن بىرسى .
شاسۇ ئىزدەپ ئاۋارە بومماي ئا بۇلۇڭۈىلا كىلە ،ئادىلجان ئەتىگەننىڭ ياغى جۇڭدۇڭدا تۇرۇپ كۆ پ توك چىكەتتى شۇنىڭ بۇرنىغىلام .... .

    ئاخشىمى ئۇسسۇل ئوينار ۋاقتىدا ،باللا ئوسسۇل ئويناپ كىتىشتى دە ئوينايدۇ ئوينايدۇ ،قوللىرى بىلىدىن كۆتۈرۈلمەيدۇ ،بىر ۋاىتا بىر ئەدەم قىشىمغا كەپتۇ .
ئۇكام باللا قۇچقاچ قورىۋاتامدا دەيدا .

     ئەتىسى يۈز ئاچقۇ دەپ بىر ئىش با ،ھەممەيلەن چاقچاقلىشىپ ئوتىراپ گەپ ساقالغا كىلىپ توختىدى دە .
بىرى قوپاپ ھەي سەندە با ساقال ئەچىكىدىمۇ با دىۋىدى يەنە بىرى شاڭخوقىلىپ .
ساقال دىگەن دادىسىنى دورىغان ئەدەمگە بۇ ئاغىنىمىز چىشى جەمەتىنى تارتىپ قاغان ئوخشايدا .
يەنە گەپ بەرمەي باشلاپ كەتتى دە .
بىر كۈنى سەتىراچخانىغا چىرىپتىم ،مۇشۇ ئاغىنەم بىلەن چىقاشىدا ،ۋاي مەن ،تۆلەيدىسەم ،ۋاي مەن تۆلەي دەپ ئاخىرى سەتىراچنىڭ پۇلىنى مەن تۆلىدىم،پوچى ئون كوينى بىرىپ چىقىپ كىتەي دەۋاتسام ئۇستام بەش كوينى تەڭلايدا ،ۋاي نىمە ئىش بۇ دىسەم ،ئاغىنىلىرىنىڭ ساقىلىنى سىلنىڭ ساقالنى ئاغان تىغدا ئاغانتىم دەيدا .

     شۇكۈنى تويدا پايپاق تاقىتىتىغان ئىش بار ئەمدى ،پايپاق تارقىتىپ كىلىپ بىر ئاغىنىمىز پايپاقچىكەن شۇنداق بولسىمۇ ئالدى ئىلىپلام پايپىقىنىڭ تۆپىگىلام كىيىۋا لدى دە .
يىنىدا ئوتتاغان بىرەيلەن ھە پايپاقچى پايپاقنى جۇپلاپ كىيىۋالدىڭغۇ،ئاخشاملىققا خوتۇنۇڭغا پايپاقنىڭ ھىسابىنى جىقراق بىرەي دەپ قالدىڭ ھە،
ۋاي سەن ئەمەس ئىدىش ئايالىدىن قوقاپ بەش موچەنلىك پىتەك كىيەمەيدىغان ،پىتىكىمنىڭ سانى ئازلاپ كىتىدۇ دەپ ئايىغىڭغا پىتەك سامماي يۈرىۋاتقىنىڭنى كىم بىلمەيدا .
ۋوي ،پايپاقنى جۇپلاپ كىيىۋالغان بىلەن يۇيىدىغاندا خوتۇن بۇرتىڭدا ئىلەڭگۈچ ئۇچامىكى .

پايپاق تارقىتىپ بولاپ يىقىنلارغا خاتىرە بولۇپ قالسۇن دەپ بىر دىن جەيناماز تارقاتتى دە ،
ئۆيدىن ئۇزايدىغان چاغدا يىگىتنىڭ دادىسى بالىلرىم خاپۇلاشماي يىنىشالا ،ئا جەيناماز قاتلىنىپ تۇرىۋەمىسۇن ئانچە -مۇنچە ناماز ئوقۇپ تۇراسىلە دىۋىدى .
باللادىن بىرسى ئوقۇيمىز ،ئوقۇيمىز خاتىرجەم بولسا ئاكا دىسە .
يىنىدا تۇرغان بىر ئاغىنىسى قوپاپ ھە ئوقۇيسەن <<خانتاجى>>نىڭ ئايرىمخانىسىغا سىلىپ ،مەيداندىن چۈشكەندىن كىيىن خۇپتەن ئۇقاسەن.

     بىر ئاغىنەم باتى چىچىنى بەك ئۇزۇن قويىدىغان بىر كۈنى بىرنىڭ ئۆيگە چىقتى دە ،
ماڭە سەھرانى بىر كۆرسىتىپ كىرەي دەپ مەلىگە ئىچىقتىم ئۇنى يولدا شودۇيجاڭ ئۇچلاپ قىلىپ مەن بىلەن كۆرۈشۈپ بولۇپ ھىلىقى ئاغىنەمنىڭ باشتىن ئايىغىغا بىر قارىۋىتىپ .
ئۇكەم سىلى غوجاممۇ ،خىنىممۇ ،ئانىلىرىغا دىسە تىزراق ياتلىقى قىلىۋەتسۇن ،بولمىسا ئوغلۇم با شۇنىڭغىلا ئەپبىرەيما .
نىمىشقا دەيلا ،باشلىرىدا قۇچقاچ ئۇۋلاپ بىر نەچچىنىڭ ئانىسى بولاپ قاممىسا يەنە .

    خاڭدىلىق ئاغىنەم باتى شۇنىڭ يىنىڭغا چىقىپ چىرەي دەپ ماڭدىم دە يىرىم يولغا بىرىپ توكسىكىلىتنىڭ توكى تۈگەپ يۆتىلەپ مىڭىۋاتسام بىر بوۋاي ئىشىكىنى ھەيدەپ ئارقىمدىن كەپ قالدى دە .
ئۇكەم ئىشەكلىرى ھىرىپ قاغان ئوخشىمامدا ؟ئىشەككىمۇ چا مانداق يەمىن جىق بەسىلىرى ئوبدان ماڭىدا
قايلىسا ما ئىشىكىمگە 30يىل بولدى ھارماي مىڭىۋاتىدا .دەپتى ،ھار كەپتىكىن .
قىتەۋە <<بەمىگەن بوغازنى يىمىگەن>>قوتانىڭ ئىشىكى دەپ قويدۇم.

    ياشلار بايرىمىغۇ دەيمەن ،باللا بىلەن سەرادىن شەگە چىرىپ قالدۇق دە ،ئايلىنىپ كىتىپ بارساق باغچىنىڭ ئالدىدا تمۇچۇقتەك پەچەت قىز بالىدىن ئىككىسى كىتىپ بامامدا ئاغىنىلىرىم ۋاي كەلتۈرىلى،ۋاي ئەگىشىلى دەپ تۇرىۋالدى ،
مەن قوپاپ باللا مىكياننى دەپ خورازدىن تاياق يىمەيلى ئاستا يولىمىزغا ماڭايلى دىسەم پەقەت ئۇنىمىدى شۇنىڭ بىلەن باغچىمۇ باغچا كۆلمۇكۆل ئايلىنىپ يۇدۇق دە بىر ۋاختا ھىلىقى قىزدىن بىرسى ئالدىمىزغا گۆلىيىپ كىلىۋاتمامدا ئو مانا كۆرۈڭ كارامەتنى دەپ قايلاپ قىز كىلىپلا ،
غوجاملا يولدىن ئىزىپ قىلىشلىما ،تولا يول مىڭىپ ئاياغلىرىمۇ ئۇپراپ كىتىپتۇ ،ئاياغلىرىنى يەڭگۈشلىۋىلىشسا ،يىڭى بازانى بىلىشۇلا ھەقىچان ،شۇيەرگە بىرىشسىرى چەتتىرەك بىر بازار بار شۇيەردە دۆلەگلىك ئاياغ بەك جىق شۇنىڭدىن بىرەنى ئىلىشالا بىلىشەملا مەن باشلاپ بارايمۇ .

   سەيچىلىك قىلىۋاتقان ۋاقتلىرىمدا تازا كاكىراپ سەي سەي دەپ كىتىپ بارىتىم بىر ئايال چىقىپ سەۋزىلىرىنى قانچىدىن بەرلە دەپتى .باغلىق سەۋزىتى كويلىرىغا ئىككى دىسەم نىمانداق قىممەت ساتلا بالام .دەيدا
شۇنىڭ بىلەن ئاچا بۇ سەۋزىنىڭ باغۇ بىر قوي تويغىدەك كۆكىمۇچىقىدا دىسەم .
ئۇنداق بوسا سەۋزىلىرىنى ئالاي كۆكىدە ئانىلىرى ئاخشاملىققا مانتا سەپبەرسۇن دەيدا .
