                                        تورداشلارغا چاقچاق

ئەسسالام تورداشلا مۇنبەر ئەسىلگە كىلىپ بەك ياخشى بولدى شۇ مۇناسىۋەت بىلەن بۇرۇنقى خۇشال
چاغلىرىمىزنى باشلىۋالساق دىگەن ئوي بىلەن بۇنى يولىدىم ئىسمى چىقمىغانلا خاپا بولماڭلار بىرلىكتە
مۇنبەرنى قىزتايلى قىنى ئەمىسە باشلاڭلار ...... مەلىكەملەر ساقلاپ تۇردۇ ھەقاچان قايتۇرما زەربىگە .

                                  ئوتقاشقا
                         ئەيمىنمەستىن  قىز كىلىپ،
                         چوغ تەڭلەپتۇ ئوتقاشقا ،
                         سۆيگۈ ئۈچۈن ئاۋارە بۇلۇپ ،
                         يۈرەك باغرى ئۇنسىز سوققاچقا .

                               مۇسلىنەگە
                        غېرىپ ئانىنىڭ ئۈلۈمى ،
                        مۇسلىنەگە ئېغىر كەپتۇ .
                        جۇۋان قىزنىڭ ھەسرىتى ،
                        ئىرىدىن يىتىم قاپتۇ ،

                              قەھۋە گە
                        قەھۋە ئىچىپ ئولتۇرۇپ ،
                        ئەسلىمىلەرگە چۈكۈپسىز ،
                        لىرىك مۇزىكا قوينىدا ،
                        غەمكىن مۇڭغا پېتىپسىز.

                             ئەرۋاھغا
                        پۇل تېپىشنىڭ كۇيىدا ،
                        ئەرۋاھ دوستۇم ئالجىپتۇ .
                        چىچى ئالغان تورداشلار ،
                        چالما- كىسەك ئېتىپتۇ ............... چاقچاق ھە ..

                                 شاتئايغا
                        ئۇنتۇپ كىتىش شاتئايغا ،
                        تۇلمۇ ئېغىر بىلنىپتۇ ،
                        كالىسىدا چىرماش خىيال،
                        يول تاپالماي ئېزىپتۇ .

                             مۇڭسىرداشقا
                        قىزلارغا ئوبدان تەدبىرنى ،
                       چۈشەندۈرۈپتۇ مۇڭسىرداش .
                       دوسلار ئارا ئىناق بۇلۇپ ،
                       بىز ئۈتەيلى چىن دىلداش .

                            مەلىكەمگە
                       بىزنىڭ قۇشاق پىشۋاسى ،
                       مەلىكەمدۇر دەل ئۈزى ،
                       مىنى ئۈزىگە رام قىلغان ،
                       بىر جۇپ نۇرلۇق كۈزى ...........چاقچاق ھە ..

                              كەترەنگە
                       غەمكىن ئۆتكەن تۇرمۇشنى ،
                       ئەسلەپ يىغلاش بىھاجەت ،
                       كۆڭۇل چۈشمىگەن يارغا ،
                       ئەقىدە باغلاش نەھاجەت .

                             دىلسىرىگە
                        بىلە چىققان مۇنبەرلەر ،
                        ئاچمىدى بىزگە قوينىنى ،
                        بىلە كۈچەپ تۇسۋالدۇق ،
                        ھۇجۇم قىلغان بورنىنى .

                            ئۇيغۇرتىگىنگە ،
                        ئۇيۇن كۈرۈپ بىر چەتتە ،
                        كۈلۈپ تۇردى تىكىنجان .
                        كۈچۈم يەتكەن بولسىدى ،
                        قىپ قۇياتتىم يېرىم  جان ......
                        چاقچاق ھە ....ھېلقى كۈنلەر ئىسىمگە كىلۋالدى.

                             بۆرە ئوغلىغا
                        بۆرە ئوغلى بۆرىدىن ،
                        ئەيمىنەمدۇ بىلمىدىم ،
                        شئېىر يازسا توختىماي ،
                        يۇرۇتاشكەن سەزمىدىم .

                 مەنبە---- يەنە مەن تاڭنۇر ...... قېنى باشلاڭلار ئەمسە ...........
ھازىرغىچە 1 ئادەم باھالىدىياخشى باھا يىغىش سەۋەبى
axkar + 1000 يەنە 5000  قالدى ،
