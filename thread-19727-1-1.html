　　 خەلقىمىز قەدىمدىن تارتىپ ، بىزنىڭ ئوغۇللىرىمىزنى سەركەردە كۆرسە ، قىزلىرىمىزنى ئەنە شۇنىڭ ئەگەشكۈچىسى كۆرۇپ كەلگەن . ئەنە شۇنداق رولىنى رولىنى جارى قىلدۇرالىغان يىگىتلىرىمىزنى جەسور ، ھەقىقىي ئەركەك دەپ ئىتراپ قىلغان ، ئۇلار قىزلار قەلبىنىدىن چوڭقۇر ئورۇن ئىلىپ ، قەدىم زامانلاردىن بېرى قىزلارنىڭ ھاياجانلىنىپ تۇرۇپ ، گۈزەل تىللار بىلەن مەدھىيلىشىگە ئېرىشىپ كەلگەن . لېكىن جەمىئىيىتىمىزدىكى تەرەققىياتلارنىڭ تېزلىشىگە ئەگىشىپ ، غەرىبنىڭ باشقا ئىلغار مەدەنىيىتىدىن كىيىم-كېچەك مەدەنىيىتى تېزراق سىڭىپ كىرىپ ، ئۇيغۇر قىزلىرىمىز ، ئۇچۇق دېسەم بىز قىزىقىدىغان ھەمدە ئەڭ ياقتۇرۇپ كىيىدىغان ، ئۆز گۈزەللىكىنى نامايەن قىلىشتا كەم بولسا بولمايدىغان كىيىملىرىمىز قاتارىدىن ئورۇن ئالدى . بۇنىڭ بىلەن مىللىيچە كىيىملىرىمىز يىراق سەھراغا تاشلىنىپ بوينى قىسىلىپ قالدى . بىلىم دائىرىسى تار ، ئەخلاقىي ياخشى بولمىغان يىگىتلىرىمىزگىمۇ يېرىم يالىڭاچ كىيىنگەنلەر پەرىزات ، مىللىي كىيىملىرىمىزنى كىيىپ ئورىنىۋالغانلار كاللىسى ئىچىلمىغان ، شەھەر كۆرمىگەن مەتو كۆرىنىدىغان بولۇپ قالدى . بۇنىڭ بىلەن بىر بۆلەك ئەخمەق قىزلىرىمىز شۇ يىگىتلەرنىڭ قەلبىدىكى پەرىزات بولۇش ئۈچۈن ، بۇخىلدىكى كىيىملەرنى مودا قىلىپ ، بەس - بەستە لاھىيەلەپ كىيىشكە باشلىدى . بۇ ئەھۋال ئەۋج ئېلىپ كەتكەچكە ، ئۇيغۇر تىككۇچىلىرىمىزمۇ ، مىللىي كىيىملەرنى تىكسە بازار تاپمايدىغانلىقىنى ئويلاپ ، غەربچە كىيىم تىكىشنى كۇرسلارغا قاتنىشىپ بولسىمۇ ئۆگەندى . بۇنىڭ بىلەن شەھەردىكى مىللىي كىيىملىرىمىز پەقەت سەھنە كىيىمى تىكىپ ساتىدىغان يەرلەردىلا قالدى . دېمەك ، قىزلارنىڭ يېرىم يالىڭاچ كىيىنىشىنىڭ ئەۋج ئېلىشى پەقەت شۇ قىزنىڭلا ئەمەس ، شۇ قىزلارنىڭ تۇرمۇشىدا ئەڭ مۇھىم ئورۇننى ئىگىلىگەن يىگىتلىرىمىزنىڭمۇ باش تارتىپ بۇلالمايدىغان مەسئولىيىتى ئىدى . ئەمما ئاغزىدىن غۇرور ، ئەركەكلىك توغرىسىدىكى سۆزلەر چۈشمەيدىغان ، مەيدىلىرىگە « مەن يىگىت » دەپ مۇشتلاپ يۇرۇيدىغان يىگىتلىرىمىزمۇ ، شۇ يېرىم يالىڭاچ كىيىنگەن پەرىشتىلەرنىڭ ئىشقى ئوتىدا كۆيۇپ ئەس - ھۇشىنى يۇقىتىپ ، ئۆزى ياخشى كۆرگەن قىزنىڭ كىيىنىشىنىمۇ باشقۇرالمايدىغان ، شۇ قىزلارنىڭ كۆڭلى ئۈچۈن < بۇ كىيىملىرىڭىز بىلەن تېخىمۇ چىرايلىق بولۇپ كىتىپسىز > دەپ ماختايدىغان ، ھەتتا ئۆزلىرى ئىزدەپ تېپىپ شۇنداق كىيىملەرنى سوۋغا قىلىدىغان بولۇپ كەتتى . بۇنىڭ بىلەن قىزلىرىمىز چىرايلىق كۆرۇنۇش ئۈچۈن يىگىتلىرىگە يارىغان شۇ خىلدىكى كىيىملەرنى تېخىمۇ ياقتۇرۇپ كىيىدىغان بولۇپ كەتتى . دېمەك ، شۇ ئەگەشكۈچىلىرىنى ياخشى باشقۇرۇپ ، يېتەكلەپ مېڭىش مەسئولىيىتى زىممىسىگە يۇكلەنگەن سەركەردىلىرىمىز ئۆز رولىنى جارى قىلدۇرالماي ، ئۇلاردىكى جۇشقۇنلۇق يۇقاپ ، ئاجىز بىر قۇشۇن بېشى قالغاچ ، قىزلىرىمىز ئۆزى خالىغان يوللارغا ئۇلارنى باشلاپ مىڭىشقا باشلىدى . بۇنىڭ بىلەن قىزلىرىمىزغا < جەسور > دەپ تەرىپلىنىپ ، قەلبىدىن سۆيىدىغان ئاق ئاتلىق شاھزادىلەر ئازلاپ كەتكەندەك بىلىنىدىغان بۇلۇپ كەتتى . ھەتتا كۈلكىلىكى شۈكى ، يىگىتلىرىمىز قولتۇقلاپ مېڭىۋاتقان قىزلىرىنىڭ قانداق كىيىنگەنلىكىنى كۆرمىگەندەك ، ئاشۇ قىزغا ئۇخشاش كىيىنگەنلەرنىڭ ھەممىسىگە ئىپپىتىنى يۇقاتقاندەك قارايدىغان ، ئۇلار ئۈچۈن جاھاندا پاك قىز قالمىغاندەك بۇلۇپ قالدى ...
