ئەسسالام ئەلەيكۇم مىسرانىم ئەھلى ياخشىمۇسىلەر ،مىسرانىم مۇنبىرى ئېچىلغان يېرىم يىلدىن كۆپرەك ۋاقىتتا  كۈتۈلمىگەن شانلىق نەتىجىلەرنى ياراتتى ، بۇ ھال مۇنبىرىمىزدىكى ھەممەيلەننى خۇشال قىلىدۇ ئەلۋەتتە ،شۇنىڭغا ئەگىشىپ مىسرانىمنىڭ ئېلگىرىكى بەزى ئىقتىدارلىرى يۇقىلىپ كىتىپتۇ (بۇ شەخسەن مىنىڭ كۆز قارشىم،باشقا تورداشلار قانداق قارايدۇ بىلمىدىم) شۇ مۇناسىۋەت بىلەن مەن ھازىر مەزكۈر تىمىنى يېزىپ ئولتۇرۇپتىمەن ،تۆۋەندە مەن ئۆزۈم بايقىغان يۇچۇقنى كۆرسىتىپ ئۆتمەكچىمەن ،باشقۇرغۇچىلار ۋە مىسران باشلىقنىڭ ئويلىشىپ بېقىشىنى ئۈمۈد قىلىمەن
1.مىسرانىمدا باشقۇرغۇچىلار بىلەن ئالاقىلىشىش ئورنى يوقكەن
         بىز ئەزالار بىرەر مەسىلىگە يۇلۇققاندا باشقۇرغۇچىلاردىن مەسلىھەت سورايلى دىسەك ھېچقانداق ئالاقىلىشىش ئۇسۇلى يوق،بايا مەن باشقۇرغۇچىلارنى ئىزدەيدىغان ئىش چىقىپ بىر باشقۇغۇچىمىزنىڭ چ چ نۇمۇرىنى قوشسام ئۇ باشقۇرغۇچىمىز مىنى رەت قىلىۋەتتى (مۇشۇ ئىش بۇ تىمىنى يېزىشىمغا ئۈلگە قىلدى) ئېلگىرىكى مۇنبەردە بولسا ئەزالار ئارا ئۇچۇر يېزىشقىلى بولاتتى ھازىرقى مۇنبەردە ئۇنداق ئىقتىدار يوقكەن


يەنە بىرسى مىسرانىم مۇنبىرىمىزدە يېقىندىن بۇيان مۇكاپاتلىق تەلەي يۇلتۇزى پائالىيىتى توغرىسىدا ، مۇنبىرىمىزدىكى بارلىق ئەزالار بۇنىڭغا ئاكتىپلىق بىلەن قاتنىشىۋاتىدۇ  ،مەنمۇ ھەم شۇنىڭ ئىچىدە ،بۇنىڭدا ئىنكاسنى يۇشۇرۇن يېزىشنى تەلەپ قىلىنغان بۇلۇپ بىز يۇشۇرۇن يازغان ئىنكاسلار باشقا ئىنكاس يازغان تورداشلارغا ئۇچۇقتىن ئۇچۇق كۈرىنىپ تۇرىدىكەن ،بۇنداق بولغاندا باشقىلار ئالدىدا جاۋاب بىرىپ بولغان ئەزالارنىڭكىدىن كۈچۈرۈپ يېزىپ جاۋاب بىرىدىغان ئەھۋال كۈرۈلەمدۇ قانداق

مەن مۇنبەردە ئۇچراتقان ھەل بولمىغان مەسىلىلەر ئاساسلىقى مۇشۇ ئىككى خىل ،قالغىنىنى باشقا مۇنبەرداشلار كۆرسىتىپ بەرسە

مىسرانىم مۇنبىرى باشقۇرغۇچىلىرىنىڭ مۇشۇ مەسىلىنى ئويلىشىپ كۈرۈشىنى ئۈمۈد قىلىمەن      
سىنى سۆيگىنىم خاتا ئەمەس ،ئۇچراتقىنىم خاتا

كەلگۈسى ، مۇنبىرىمىزنى قوللاپ پىكىر تەكلىپ بەرگىنىڭىزگە چىن كۆڭلىمىزدىن رەھمەت ئەيتىمىز !
سورغان سۇئاللىرىڭىزغا مەن ئاز - تۇلا جاۋاپ بىرىپ ئۈتەي :
بىرىنچى سۇئالىڭىزغا ئېنىقلىما بىرىپ ئۆتسەم :
توغرا ئەيتىسىز ، بۇرۇنقى مۇنبەرلەردە كۆپ ئىشلارنى قىلغىلى بۇلاتتى . خالىغانچە ئۇچۇر قىلىشقىلى بۇلاتتى ، مەخسۇس مۇنبەر توپلىرى بار بۇلۇپ ، ئۇنىڭدىمۇ خالىغانچە ئالاقە قىلىشقىلى بۇلاتتى . ئەپسۇس.........................
ھەممە ئەزالار سىزدەك ئاڭلىق ئەمەس .   مۇنبەردىكى ئۇچۇر خۇددى ئاتقان ئوقتەكلا ، يوللانغان تىما ئىنكاسلارنىڭ ئۈچۈرۈشكە تىگىشلىكلىرىنى ئۈچۈرۈپ بىر تەرەپ قىلىپ ماڭالىغىنىمىز بىلەن  قىززىق قانلىق بىلەن يوللۇنۇپ قالغان بىنۇرمال ئۇچۇرلارنى ۋاختىدا بىر تەرەپ قىلىش  مۇمكىن ئەمەس  ، شۇڭا بەزىبىر سەۋەپلەرنى كۆزدە تۇتۇپ باشتا مۇنبەرنىڭ ئۇچۇر ئىختىدارىنى ئىتىۋەتكەنتۇق . لىكىن يەنە سىلەرنىڭ باشقۇرغۇچىلار بىلەن ئالاقلىششىڭلاغا قۇلايلىق بولسۇن ئۈچۈن يېقىندا بۇ ئىختىدارنى يەنە ئەسلىگە كەلتۈردۇق . بەزى ئەزالارنىڭ قىززىق قانلىق قىلىپ  بىنۇرمال ئۇچۇرلارنى يوللىماسلىقىنى ئۈمۈد قىلىمەن !  مۇنبەرنىڭ تاقىلىپ قېلىشىنى ھەممىمىز خالىمايمىز ئەلۋەتتە !
ئىككىنچى سۇئالىڭىزغا كەلسەك :
ئالدى بىلەن تەلەي يۇلتۇزى پائالىيتىگە ئاكتىپ قاتنىشىپ كىلۋاتقانلىقىڭىزغا رەھمەت ئەيتىمىز !
خەۋىرىڭىز بار ، بۇ پائالىيەتتىكى سۇئاللار 1-2-3- ...دىگەندەك تەرتىپ بۇيىچە ئوتتۇرغا قۇيۇلغان بۇلۇپ ، سۇئالغا جاۋاپ بەرگۈچى ئەزالارمۇ مۇشۇ تەرتىپ بۇيىچە جاۋاپ بىرىپ كىلۋاتىسىلەر .پائالىيەتكە قاتناشقان ھەممە ئەزالارنىڭ بىرلا ۋاقىتتا يوللىغان  جاۋاپلىرىنىڭ ھەممىسىنىڭ توغرا بۇلۇپ چىقىشى ناتايىن . ئالدىقى قەۋەتتە توغرا جاۋابى چىقىپ بولغان سۇئاللارغا كىيىنكى قاتاردىكىلەر  تۇلۇقلاپ جاۋاپ بەرسە  تۇلۇقلىغىنى ئىتبارغا ئېلنىدۇ . جاۋاب بىرىلىپ بۇلۇنغىنىغا گەرچە كۈچۈرمەي ئۈزى جاۋاپ بەرگەن بولسىمۇ ئۇ ئىتراپ قىلىنمايدۇ . شۇڭا بۇ < تەلەي يۇلتۇزى پائالىيتى > .
