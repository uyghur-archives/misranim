ھەممىڭلار قۇيمىچى دەيسىلە ئىنكاسلارغا قارىسام تۇلۇق چۈشەنمەيدىكەنسىلەر ...
كۆچمە خەۋەرلىشىش شىركىتى بىكاردىن بىكار قالايمىقان پۇل تۇتۇپ قالمايدۇ ، ھەممىسىنىڭ ئۆلچەملىك ھەق ئىسپاتى بار .
ئايلىق ئەڭ تۈۋەن سۆزلىشىش ھەققىگە كەلسەك مەشرەپ كارتىسىنىڭ بەش يۈەن ، بىر ئادەم بىر ئايدا بەش يۈەنلىكمۇ سۆزلىشەلمىسە ئۆزىنى قىينىماي ئاممىۋى تىلفۇن ئىشلەتكىنى تۈزۈك . رايۇن ئاتلاش ھەققى قىممەت بولسا ئون يۈەن سەككىز مو سەككىز پۇڭلىق بىر كەسپى بار شۇنى ئېچىپ بارسا كەلسە ھەقسىز ئۇرسا 0.29|مىنۇت . ھەق ئېلىنىدۇ . ھېلىقى ئۇيغۇرچە گېزىتكە كەلسەك ئادەتتە ئالتە ئايلىق ، بىر يىللىق ۋە خالىغان ۋاقىتتا ئېچىپ تاقىغىلى بۇلىدىغان ئۈچ خىل ئۇسۇلى بار . قايسىنى ئاچسىڭىز بۇلىۋېردۇ . سىز ئىلتىماس قىلمىسىڭىز بىكاردىن بىكار بىر يىللىق ئېچىۋەتمەيدۇ .
يەنە بەزىلەر ئۆزى ئېچىپ قۇيىدىكەن دەيدۇ ، بىكاردىن بىكار ئېچىپ قويمايدۇ ئۇنى ، سوۋغا قىلىدىغان سۆزلىشىش ھەققى بار ئاچامسىز دەپ ئابونىت رازى بولغاندىن كىيىن ئاندىن ئاچىدۇ . ئۇ ھەق تۈۋەندىكىچە :
ئۇيغۇرچە ئەتىگەن  كەچلىك يانفۇن گېزىتى 3+يۈەن ،سىمسىز مۇزىكا كۇلۇبى ئالى دەرىجىلىك ئەزا +5يۈەن ،كەلگەن تىلفۇن باشقۇرغۇچىسى 3+يۈەن ،مۇزىكىلىق قوڭغۇراق كەسپى 4+يۈەن . بۇلار جەمئى 15 يۈەن . كۆچمە خەۋەرلىشىش شىركىتى ھەر ئاينىڭ ئەڭ ئاخىرقى كۈنى قالدۇق سوممىڭىز غا 15 يۈەن قۇشۇپ بىردۇ ، دىمەك سىزگە سوۋغا قىلىدىغان 180 يۈەن سۆزلىشىش ھەققى مۇشۇ كەسىپلەرگە  تۇتۇپ قېلىنىدۇ بۇ كەسىپلەرنى ھەقسىز ئىشلىتەلەيسز .  شىنجاڭ كۆچمە خەۋەرلىشىش شىركىتىنىڭ ھەق ئۆلچىمىنى دۆلەتتىن بىكىتىپ بىرىدۇ . ئۆزى خالىغانچە ھەق ئېلىش ،ئۆزگەرتىش ھۇقۇقى يوق ، مۇلازىمىتى شىنجاڭ ئىچىدىكى سىمسىز ئالاقە شىركەتلىرى (بىرلەشمە ، تىلىگراف )قاتارلىقلارغا قارىغاندا يەنىلا ئەڭ ياخشى .  زارى بولمىساڭلار باشقا شىركەتنىڭ كىنى ئىشلەتسەڭلار بۇلىدۇ ، لىكىن مەن بەش يىلدىن بېرى ئۇ شىركەتنىڭ ھەق ئۆلچىمىدىن نازارى بۇلۇپ باقمىدىم .
كۆچمە خەۋىرلىشىنىڭ مۇلازىمەت ھەققى قىممەت بولسا تىلىگراف شىركىتىنىنڭ تىيەنيى تىلفۇنىنى ئىشلىتىڭلار ھازىر ئۇ تىلفۇننىڭ ھەق ئۆلچىمى بەك تۈۋەن شىنجاڭ ئىچىدە كەزمە قىلسا كەلگن تىلفۇنغا ھەق چىقمايدىكەن . يەنە ھەر ئايدا ئىتبارلىق كەسىپلىرى بار .

مەلۇم بىر بىر ئىش تۇپەيلىدىن بوپ كىتىپ يانفۇنۇمنى ئاچقۇزماي باشقا تېلفۇن ئىشلىتىپ يۈرىۋېرىپتىمەن . ئىشنى تۈگۈتۈپ يانپۇنۇمنى ئاچقۇزاي دەپ پۇل تۆلىگىنى كىرسەم ، "كوڭخۇ " ( مۇنداق نۇمۇر مەۋجۇت ئەمەس ) دەيدۇ . قاتتىق ئاچچەىغىم كەلدى . نەچچە يىل ئىشلەتكەن نۇمۇر تۇرسا ، يەنى كىلىپ نۇرغۇن ئالاقە مۇشۇ نۇمۇرغا باغلىنىشلىق تۇرسا .... شۇنداق قىلىپ كوچمە خەۋەرلىشىش زالىغا كىرىپ ، نۇمۇرۇمنى قايتا بىجىرمەن دېسەم مۇلازىمەتچى خادىم قىزچاق بىجىرگىلى بولمايدۇ دېمەسما ، شۇنىڭ بىلەن بىرمۇنچە تاكاللىشىپ كەتتىم . بىردەمدىن كىيىن بىشىغا قاسقان شەپكە كەيگەن بىرسى كىلىپ مېنى بىر تېلفۇن ئىلىپ ساتىدىغان ھايانكەشنىڭ قىشىغا باشلاپ بىرىپ ، مۇشۇ كىشى بىلەن دېيىشسىڭز نۇمۇرىڭىزنى ئىلىپ بىرەلەيدۇ دەپ تۇنۇشتۇرغان بولدى . مەن ئۇ كىشى بىلەن بىر ھەپتىدە ئىلىپ بىردىغانغا دېيشىپ 200 يۈئەنگە كىلىشىپ پۇل بەرسەم ، بىر ئاي بولدى ئىلىپ بىرەلمىدى .
كۆچمە خەۋەرلىشىش شىركىتى شۇنداق يۈزسىزمۇ ؟ ! نەچچە يىل ئىشلەتكەن نۇمۇرنى نېمە ئۈچۈن ئىگىسىگە بەرمەيدۇ ؟! نېمە ئۈچۈن ھايانكەشلەرگە بىردۇ ؟!

قىلاي دىسە شۇمۇ گەپ بولدىمۇ؟ئۆزىڭىز نەچچە يىل ئىشلەتكەن كارتىغا پۇل تۆلىمىسىڭىز؟سىزگە ئالاقە لازىم ئەمەسكەنغۇ؟جۇڭگو كۆچمە خەۋەرلىشىش شىركىتى بۇغوجام قاچان كېلىپ پۇل تۆلەر دەپ ساقلاپ تۇرسا بولامتى؟ئادەتتە 3ئاي پۇل تۆلىمىسە ھەرقانداق كارتىنى بىكار قىلىۋىتىدۇ،ئەمدى ھايانكەشكە كەلسەك قارىغاندا كۆچمە خەۋەرلىشىش شىركىتىدە تونۇشى بارئوخشايدۇ،بىراق بۇ ھايانكەش كۆچمە خەۋەرلىشىش شىركىتىگە ۋەكىللىك قىلالمايدۇ.
