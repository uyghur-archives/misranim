  كېسەل مۇناسىۋىتى بىلەن دوختۇرخانىدا داۋاىلنىۋاتاتتىم بۈگۈنكى داۋالاش باسقۇچى جەريانىدا مەسئۇل دوختۇر(خەنزۇ) بىلەن پاراڭلىشىپ قالدىم.ئۇنىڭ ئېيتىشىچە ئۆتكەنلەردە بىر ئۇيغۇر ئايال بىر قىزنى دوختۇرخانىغا ئەكەپتۇ، قىزنىڭ ئۇچىسىدا مەكتەپ فورمىسى بولۇپ ماڭغۇدەكمۇ ھالى يوق ،ئاپىسى يۆلەپ دېگۈدەك ئەكىرىپتۇ.دوختۇر نېمە بولغانلىقىنى سورىغانكەن،قىز ‹‹قورسۇقۇمنىڭ ئىچى ئاغرىپ كېتىۋاتىدۇ›› دەپلا ‹‹ۋايجان››دەپ قورسىقىنى تۇتقىنىچە قاتتىق چىرقىراپ يەردە ئولتۇرۇپ قاپتۇ، دوختۇر ئانىسى بىلەن بىللە قىزنى يۆلەپ كېسەل كارۋىتىغا ياتقۇزۇپ ،ئانىسىنى سىرتقا چىقىرىۋېتىپ تەكشۈرمەكچى بولغانكەن قىزنىڭ قورسىقىنىڭ تېڭىقلىق تۇرغانلىقىنى كۆرۈپتۇ. دوختۇر ئىشنىڭ تېگىگە سەل يېتىپ قېلىپ، قىزنىڭ ئانىسىنى چاقىرىپ‹‹قىزىڭ ھازىر بوشىنىدىكەن››دەپتۇ،قىزنىڭ ئانىسى شۇ يەردىلا دوختۇرغا چالۋاقاپ ‹‹قىزىم دېگەن ئوقۇغۇچى تۇرسا ،سەن نېمە دەپ جۆيىلۋاتىسەن ؟››دەپ ئۇرۇشقىلى تۇرۇپتۇ، ئۇنىڭغىچە قىز ۋارقىراپ –جارقىراپ ئېپ بوپتۇ.دوختۇر  قارىسا  بالىنىڭ باش سۈيى تۆكۈلۈپ بولغان، شۇنىڭ بىلەن قىزنىڭ ئانىسىغا ‹‹قىزىڭنى قۇتقۇزۇش مۇھىممۇ ئەمەسمۇ؟قىزىڭ ھازىرلا بوشىنىدۇ،تېزراق رەسمىيەت بېجىرىپ چىققىن  ››دەپتۇ.

 قىزنىڭ ئانىسى يىغلىغىنىچە رەسمىيەت بېجىرىشكە چىقىپ كېتىپتۇ،ئۇنىڭغىچە دوختۇر بىلەن سېستىرالار قىزنى تۇغۇتخانىغا ئەكىرىپتۇ،قىزنى تۇغۇتخانىغا ئەكىرىپ كارۋاتقا چىقىرىپ بولغۇچە بوشۇنۇپ بوپتۇ.تۇغۇلغان بوۋاق ئوغۇل ئىكەن،قىزنى كېسەلخانىغا يۆتكەپ چىققاندىن كېيىن قىزنىڭ ئانىسى ئۆزىنى ئۇرۇپ يىغلاپ قىزىدىن مەجبۇرى ھالدا ‹‹بالىنىڭ دادىسى كىم؟››دەپ سوراپتۇ.قىز يىغلاپلا تۇرغاندىن باشقا ئاپىسىغا ھېچنىمە دېگىلى ئۇنىماپتۇ، ئاخىرى بولماي ئانىسى دوختۇردىن قىزىدىن گەپ ئېلىپ بېقىشنى ئۆتۈنۈپتۇ، دوختۇر قىزدىن نېمە ئىش بولغانلىقىنى سوراپتۇ.

ئەسلى بۇ قىزتولۇق 1- يىللىقنىڭ ئوقۇغۇچىسى بولۇپ ،ئۇ  يىغلاپ تۇرۇپ ئۆكتەبىر بايرىمىدا سىنىپىدىكى ۋە باشقا سىنىپتىكى تۆت جۈپ قىز –ئوغۇلنىڭ باغچىغا بېرىپ ئوينىغانلىقىنى، ئۇ يەردە قىزلارنىڭ  ئوغۇللار ئېلىپ كەلگەن  ئىچىملىكنى ئىچكەنلىكىنى ،كېيىن ھەممە قىزلارنىڭ باشلىرىنىڭ قېيىپ غەلىتى بولۇپ قالغانلىقىنى، ئۇنىڭدىن كېيىن نېمە بولغانلىقىنى بىلمەيدىغانلىقىنى،بىر كەملەردە قارىسا قېشىدا قالغان قىزلارنىڭ يوقلىقى ،ئۆزىنىڭ باغچىنىڭ ئىچىدە ئەمەس باشقا بىر يەردە شۇنداق ۋەيرانە ھالدا ياتقانىلقىنى،يىغلا –قاخشا ھالدا ئۇ يەردىن چىقىپ ئۆيگە قايتىپ كەلگەنلىكىنى،ئۆزىنى كىمنىڭ ئۇ يەرگە ئېلىپ بارغانلىقىنى پەقەتلا بىلمەيدىغانلىقىنى ، بىركىمگە دەي دېسە نومۇس قىلىپ ھەم قورقۇپ دېيەلمىگەنلىكىنى ،كېيىن ھېچنىمە سىڭمەي ياندۇرۇپ يۈرگەندە ‹‹ ئاشقازىنىم ئاغرىپ كەتكەن ئوخشايدۇ››دەپ يۈرگەنلىكىنى ،قورسىقىنىڭ بارا –بارا يوغىناپ كېتىۋاتقانلقىنى، قورقۇپ نېمە قىلارنى بىلمەي قورسىقىنى تېڭىپ يۈرگەنلىكىنى ،كىمنىڭ ئۆزىنىڭ نومۇسىغا تەككەنلىكىنى بىلمەيدىغانلىقىنى،قورسىقى قاتتىق تولغاپ ، بەكلا ئاغرىپ كەتكەچكە ئىلاجىسىز قورسىقىنىڭ ئاغرىپ كەتكەنلىكىنى ئانىسىغا ئېيتىپ ،ئانىسى بىلەن دوختۇرخانىغا بىرگە كەلگەنلىكىنى نومۇس،قورقۇنۇچ ئىچىدە يىغلاپ تۇرۇپ دەپ بېرىپتۇ،دوختۇر قىزدىن ئاڭلىغانلىرىنى ئانىسىغا ئېيتىپ بېرىپتۇ، قىزنىڭ ئانىسى نومۇس ۋە ئەلەمدىن قاتتىق يىغلاپ كېتىپتۇ، دوختۇر قىزنىڭ ئانىسىغا، قىزنىڭ  قورسىقىنى تېڭىپ يۈرگەن بولسىمۇ، بالىنىڭ ساغلام تۇغۇلغانلىقىنى،كېسەللىك ئەھۋالىنىڭ بايقالمىغانلىقىنى ، ئەگەر بالىنى بېقىشنى خالىمىسا ئۆزىنىڭ بالىنى بېقىش ئۈچۈن ئادەم تېپىپ بېرىدىغانىلقىنى ئېيتىپتۇ، قىزنىڭ ئانىسى‹‹بالىنى ئۆزىمىز باقىمىز، ھېچكىمگە بەرمەيمىز››دەپ ،بالىنى ئۆيىگە ئەكىتىپتۇ.دوختۇر ماڭا ئۆزىنىڭ ‹‹ ئۇ ئوغۇل بالىلار چوقۇم قىزلارنىڭ ئىچىملىكىگە مەس قىلىش دورىسى سېلىۋەتتى ››دەپ گۇمانلىنىدىغانلىقىنى، ھازىر ئوغۇللارنىڭ قىزلارنىڭ ئىچىملىكىگە مەس قىلىش دورىسى سېلىپ مەس قىىلپ قويۇپ،ئۇلارنىڭ نومۇسىغا تېگىدىغان ئىشنىڭ كۆپىيىپ كەتكەنلىكىنى، مەس قىلىش دورىسىنىڭ ئىنتايىن ئەرزانلىقىنى ئېيتىپ بەردى.

دوختۇرنىڭ گېپىنى يالغانمىكىن دېسەم راستەكلا تۇراتتى، ئۇنىڭ بۇ گەپلىرىنى ئاڭلاپ ئۆزۈمچە نومۇس قىلىپ قالدىم،پاك بىر قىزنىڭ نومۇسى بۇلغاندى، بۇ قىز ئۆزىنىڭ قايسى ھايۋان سۈپەت يىرگخنىشلىك ئىنسان تەرىپىدىن نابۇت قىلنىپ ،بىر بالىنىڭ ئانىسى بولۇپ قالغانلىقىنىمۇ ئۇقماي قالدى،مېنىڭچە بۇ قىز بىر ئۆمۈر ھەسرەت – نادامەت ئىچىدە ئۆتىشى مۈمكىن،مۇشۇ قىزلارغا ئوخشاش ئاقىۋەتكە قالغانلار خېلى كۆپ، بىر ۋاقىتتىمۇ مۇشۇنىڭغا ئوخشاش بىر ۋەقەنى ئاڭلىغان ئىدىم.
ئۈچ جۈپ قىز -يىگىت ئوينىغىلى سىرتقا چىقىپتۇ، قىزلار ئالدىغا قويۇلغان سوغۇق ئىچىملكىلەرنى ئچىپ خۇدىنى يوقاتقاندىن كېيىن ئىككى قىزنى ئىككى مۇناپىق بىر مېھمانخانىغا ئەكىرىپ ياتاق ئاچقۇزۇپتۇ،بىرسى قىزنى تاكسىغا چىقىرىپ دۇجئىسۇغا (سۇ ئۈزۈش ئورنى)ئاپىرىپتۇ، ئۇ جاي سىرت بولغاچقا، شامالنىڭ تەسىرىدە قىز مەسلىكتىن يېشىىلپ كېتىپتۇ ۋە ئۆزىنىڭ بۇ جايدا ياتقانلىقىنى كۆرۈپ قورققىنىدىن قاتتىق ۋارقىراپتۇ،يىگىت قىزنىڭ ئاغزىنى توسۇۋېلىپ ،زورلۇق قىلماقچى بولغان ئىكەن ،قىز ۋارقىراپ -جارقىراپ قارشىلىق كۆرسىتىپتۇ، ئەتراپتا تۇرغان كىشىلەر قىزنىڭ ۋارقىرىغان ئاۋازىنى ئاڭلاپ يۈگۈرۈشۈپ كېلىپ، قىزنى ئۇ نومۇسسىز گۇينىڭ چاڭگىلىدىن قۇتۇلدۇرىۋاپتۇ.
 مۇشۇنىڭغا ئوخشاش بىر قىسىم ھايۋان كەبى ئەرلەرگە ،يىگىتلەرگە ( بۇنداقلارنى يىگىت دېگۈم كەلمىسىمۇ) شۇنى دېگۈم كېلىۋاتىدۇ،سىلەرنىڭ ئاچا -سىىڭللىرىڭلار يوقمۇ؟ ئۇلارمۇ شۇنداق كۈنگە قالسا قانداق قىىلسىلەر؟ئۆز مىللىتىڭلارنى قىزلىرىنى ئۆزەڭلار نابۇت قىلغىلى قانداقمۇ ۋىجدانىڭلار كۆتىرىدۇ؟......
 بۇنداق دېسەم،بەزىلەر <<شۇ قىزلاردىمۇ بار ، خەقنىڭ ئارقىسىغا كىرىپ ماڭغاندىكىن >> دېيىشى مۈمكىن، توغرا،قىزلاردىمۇ بار، لېكىن يىگىتلەر بىلەن سىرتلارغا بىرگە چىقىدىغان ھەممە قىزلار  ئۇنداق بۇزۇق ئەمەس، بەزى قىزلار ئوغۇل دوستلىرىغا ئىشىنىپ ،ئۇلار بىلەن سىرتلارغا چىقىپ قالىدۇ،شۇنداق ۋاقىتلاردا ئۇلارنى مەس قىلىپ قويۇپ، نومۇسىغا تەگسە ،ئۇ قىزلار ئۆمۈر بويى ئازاب ئىچىدە ئۆتمەمدۇ؟....
 بولدى ،بەك قايناپ كېتىدىكەنمەن،......
 قىزلارنىڭ نومۇسىغا تېگىدىغان نومۇسسىز،ھايۋان كەبى كىشىلىرىمىز: ھايۋان سىلەردىن ياخشىراق......سىلەر شۇنداق پەسكەشلىكلەرنى قىىلۋاتقاندا، ئۆزەڭلارنىڭ ئاچا -سىڭىلىىرىڭلارنىمۇ ئويلاپ قويۇڭلار.....
 مۈمكىن بولسا سىلەردەك ئەبلەخلەرنى دارغا ئاسسىمۇ ئازلىق قىلىدۇ.....
 چوڭ سۆزلىمەي ،توۋا دەي........
