باللا 6 ئاينىڭ 12 كۈنىنى ئەتىگەندىن مۈكچۈيۈپ ئولتۇرۇپ يازغان ھىكايەم تورغا چىقىپلا تەستىقتىن ئۆتمەي يوقاپ كەتتى ،ھەيران قالدىم ساقلىۋالماپتىكەن مەن ،قايتا يازاي ئەمدى .ئۇكۈنى مەلىكەم ماڭا ئاتاپ بىرنەرسە يېزىپتىكەن شۇنى ئوقۇيمەن دەپ نەس بېسىپ كەتتى .بوپتۇلا بالا كەلسە قوشلاپ كەپتۇ دىگەن شۇدە ،مەن بولغاندىن كىيىن دەپ شۈكىرى قىلىپ تۇردۇم ئامال يوق ،ئەمدى تۈنۈگۈن كۆكنۇرقوشاق يېزىپتىكەن ،مەن داۋاملاشتۇرمىسام مەندىن يامانلاپ قالدىمۇ چىقىپ كىتىپتۇ ،مەلىكە نىڭ ئىشى ئالدىراش بۇلۇپ كىتىپ قاراپ بۇلاماپتۇ ،تاڭنۇرى چۇ دىمەيسىلە ھېلىق شى جېي بېي نىدەپ ئۇيقۇنى ھارام قىلىپ ئىنكاس يازماپتۇ ،ئەركەم بايراق دەمدۇ ،جىم تۇرۋاپتۇ .بۇ كۆكنۇر ئەمدى بەي -بەي دەيدۇ شۇڭا دوسلىرىمغا ئەپسانە تۇقۇپ ئولتۇردۇم باللا،ئەمدى كۆڭلۈمىزنى ئىككى دەن بىر قىلمايلى .
                                                      كۆكنۇر
                                             كۆكنۇر قوشاق تۇقۇپتۇ،
                                             تاڭنۇرى بىلەن قېتىپتۇ.
                                            يېرىم ئىنكاس يازمىسام
                                            مىسرنىمدىن   قېچىپتۇ .
توۋا تۈنۈگۈن قاراڭە شۇ بىرقانچە ئىشلانى ئويلاپ كاللامدىن ئۆتكۈزەلمەي ،تۇقۇماپتىمەن،ئاستا ئىززەتلىگەندە كىرۋالاسىز .
                 
                                                   مەلىكە
                                          كۈندە ئىشلەپ ئاغىنەم
                                         سىز گە ھاردۇق يەتتىمۇ.
                                         بىرىپ قالسام ناۋادا ،
                                         قىزغىن قارشى ئالاسىز مۇ .
مەن شۇجايغا باردىغاندەكلا زادى .لىكىن يەنە قانداق ئۆزگىرىشلەر بۇلىدۇ دەڭلا ھەي [s:4]
            
                                                تاڭنۇرى
                                        ياز ماڭ ئەمدى پاجىئەنى ،
                                      ئەسلىتىپ سىزچۇڭقۇرئازاپنى.
                                       شىرىن چۈشلەردىن ئويغۇڭ،
                                       ئەمدى رىئاللىقا يۈزلىنىڭ.
سىز شېئىر يېزىپ مەنمۇ دوراپ يېزىپ باقتىم ئەزەلدىن يازالمايتىم مەن ،لىكىن ئاۋۇ ئۆزۈمگە دىگەن گەپ ھە .

                                               شاتئاي
                                       نىسى سوقسام قوشاقنى ،
                                      مەندىن خاپا بولدۇڭمۇ ،
                                      تىلفۇن قىلسام ئامايسەن،
                                      بەك ئالدىراشلا قىلدىڭغۇ .
توۋا ،خوشمۇ يوق ،قۇيۋەتتىڭغۇ ھە ،ئاخشام ،مىنىڭ بىر ئادىتىم تىلفۇندا پاراڭلىشىشقا بەكلا سۇس ،شۇڭا تىلفۇن پۇلۇم كۆپ چىقمايدۇ ..بۇنىڭغا ھەيران قالما باشقىلار ئۇرسىمۇ ئالدىراپلا قۇيۋىتىمەن .

                                               نادىرە
                                      تالاشسام ئاغزىڭدىكى ئىمزىكىنى ،
                                     بەردىڭ ماڭا نۇرغۇن جاپا ،ئازاپنى.
                                     ئۇنتاي دىدىم ئەمما ،ئۇنتالمىدىم ،
                                     تىزرەك بەرگىن ئاغزىڭدىكى شۈمەكنى.
تېخىچە ئاغزىڭدا تۇرۋاتسا ،ئۆلتۈرمەكچىما .

                                             كەپسىز قىز
                                     ئەجەپ بېرىۋېلىپ يېزىغا ،
                                    كىر مەيسىز غۇ مۇنبەرگە.
                                    يوللىۋېتىڭ يەنە ھىكايە ،
                                    ئوقۇپ چىقاي خۇماردىن .
بۈگۈن دەم ئالسىز غۇ دەيمەن ،سەل تۇرۇپ چىقارسىز تورغا .

                                          كۈنقىز
                                   نىمە دەيسىز كوت-كوتتا ،
                                   ئالداپسىزغۇ قەستەنگە.
                                   ئاچچىقمنى كەلتۈرمەڭ ،
                                   كىر مەيمە،مەن مۇنبەرگە.
سىز گە قەستەنگە دەۋاتىمەن .[s:30]ماڭا خاپا بولماي چىقىپ كىتىدىغان يولنى دەپ بىرەمسىز ،

                                          فاتىمە
                                    ئۇزۇن بوپتۇ فاتىمە ،
                                    ئەمدى سىزگە يازىمە.
                                    ئىسمىڭزنى تۇتىۋالدىم،
                                   ئۇنۇتمايمەن مەڭگۈگە.
قايسى كۈنى مەلىكە يازغان ئىشخاندىكى ئىشلارنى ئوقۇپ ،ماڭا ئىنكاس يېزىپسىز شۇنچىلىك خوشال بۇلۇپ كەتتىم .رەخمەت سىز گە .

                                           مەرھابا
                              ئۇنتۇپ قاپتىمەن ئۆتكەندە،
                              كىچىك بولغاچ مۇنبەردە.
                              خاپا بولماڭ ئوماق سىڭلىم،
                             ئەمدى يازاي سىز گە ئەپسانە .

                                        تۇمار
                              يۈتۈپ كەتكەن خاتىرە،
                              يېنىپ كەلمەس ئەسلىگە.
                             خەپ داۋامغا تاشنا مەن،
                             يېزىۋېتىڭ قەستەنگە .
گەپ تۈنۈگۈن دىگەندەك ،سىز مۇ بەكلا ئالدىراش ئوخشايسىز ،لىكىن ئاخرىمۇ چىقىپ بولدى دەڭە ،شۇنداق تىمۇ ،ئۆزقۇلىڭىز بىلەن ئاخىرلاشتۇر ڭ ،سىزگە ئامەت تىلايمەن .

                                   tumar
                            يوقاپ كەتتۇق كىيۇكيۇدا،
                            خەت قالدۇرۇڭ بەزىىدە .
                            بۇرنۇم قاندى دىدىڭز ،
                            ئەنسىرەپلا بەك ئاۋارە.
توۋا ،سىز بىلەن پاراڭلاشمىغىلى بۇيىل 12 ئاي كەلسە 2 يىل بولغىدەك ،لىكىن سىز بىرقانچە ئاينىڭ ئالدىدا كۆردۈم گەپ قىلماپتىمەن ،مەن تىلفۇندا پاراڭلىشىشقا بەكلا ئۆچ بۇلۇپ كەتتىمنۇ بىر قېتىممۇ ئۇرۇپ باقماپتىمەن ھە !

                                    تەبەسسۇم
                             تەلپەك ئادىشىم بارمۇيە ،
                             سەن بولساڭ گۈلدەستە،
                             ئەمدى بولاي ياپراقچە ،
                             قايتۇرۋەتتىم ئۆزگىچە.
ئاداش ھېلىقى مۇنبەردە ،تۇتۇلۇپ قالماي دەپ شۇڭا دىيەلمىگەن ئەمدى بۇيەردە دىۋالاي ھە ،قانداق ھى ھى
      
                                      ئېدىرسجان
                              ۋاي ئېدىرىسجان نەدەسىز .
                              قېتىق سېتىپ يۈرەمسىز ،
                              مۇنبەردىگەن بۇيەردە ،
                              خەتنى چوڭ چوڭ يازمامسىز .
ھېلىق تېمىڭزدىكى خەتنى دەيمەن ،تۇنۇش دۈشمەن بولغاندىكىن ،ئۇنتىيالاماي قېتىق دەپ قويدۇما يا .

                                         باياۋان
                              يوقاپ كەتمەڭ باياۋان،
                              بېرىۋېلىپ چۆللەرگە،
                              يەنەمىسراننى تاشلاپ ،
                             كىەتمەڭ لىكىن ئۆيلەرگە.
      
                                      تەنھا بۆرە186
                              ئىسمىڭز زادى نىىمە
                             تەنھابۆرە شۇنداىمۇ،
                             كەينىدىكى رەقەممۇ ،
                             داداڭىزنىڭ نامىمۇ .
ئانى بىر چۆرىۋەتكەن بولسىڭز ھە

                                  كەتىرەن
                         ئىسمىڭز كەتىرەنتى ،
                         كاتىران دەمدۇ بەزىلەر ،
                          لاتىنچىغۇ   ھەرىپلەر،
                          نىمە بولدى ئۆزگەردى.
باللا كەترەن بىر خىل گۈلمۇ شۇ ،ئۇنداق قىلماڭلار ھە

                                  خۇمپەر
                          چۈشنى كۆرۈپ خۇمپەرجان ،
                          جۆيلىيدىكەنسىز ھەرقاچان.
                          ئەمدى دىمەڭ دىلسىرى ،
                          مەلىكەمنى چاقىر ماڭ .
تىللىغاندەكلا ئىسمكەن ئامالسىز قالدىم شۇ جاننى قۇشۇپ ،دىمەكچى جۆيلۈپ قۇيۇڭ ،ئۇنتۇپ قالماڭ !

                                  ئەركەم 624
                           پاراڭلاشتۇق كوت -كوتتا ،
                           ھالىڭز بەكلا يامانغۇ ،
                           ھىكا يە ،شېئىر يازمامسىز ،
                           كەچۈرۈم سوراي بۇلامدۇ .
ھېلىق ئىمزايىڭز كاللامدىن ئۆتمەيدۇ ،باش سۈرىتىڭز چۇ ،شۇڭا ئۇنتۇپ قالماي قايتا دىدارلىشپتىمىز .

                                    ئەركەم
                           نىمەبولدۇق ئەركەم ،
                          ئاق بايراقنى چىقارماڭ.
                          چاقچاق قىلىپ مۇنبەردە ،
                           مىنى كۈلدۈرۈپ ئۆلتۈرمەڭ.
جىمىپ كەتمەيلى ھە .بۇيەردىن ئانچە -مۇنچە خوشاللىق تاپمىساق نەدىنمۇ تاپارمەن ،پەقەتلا مۇشۇ قالدى .

                           
                                     دىل-زىبا

                             سىزنى مەجبۇرلاپ تىمەن،
                             يەنە باشنى ئۆزگەرتىشكە،
                             خىجىل بولغان شۇ پەيىتتە .
كەچۈرۈك ئەمدى مەنمۇ شۇنداق كۆپ ۋاقىت سەرىپ قىلىپ بۇنى قوللانغان .

                                 سۈبھى

                          ئەمدى گۈلدەستە.
