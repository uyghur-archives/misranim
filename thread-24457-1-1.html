مەن بۇ تىمىنى يوللىغاندىدن كىيىن نۇرغۇنلىغان تورداشلار ئوخشاش بولىمغان ئىنكاسلىرىنى قالدۇرۇپ، ھەتتا ئۇچۇر ئارقىلىق ھاقارەت سۆزلىرىنى ياغدۇرۇپ ،مىنىڭ بەزى مەسىلىلەرگە بولغان قارىشىمنى تىخىمۇ چوڭقۇر تەپەككۇر سەيناسىغا ئېلىپ چىقتى . شۇنداقلا سوغۇققان تەپەككۇر دۇنياسىدا ئەقىل بىلەن ئىش كۆرۇشنىڭ موھىملىقىنى تونۇتتى .
     مەن ئىككى بالىغا دادا بولغان بىر كۆيۇمچان دادىمەن .چۇنكى باللىرىم مەن كەلگەندە قولىدىكى بارلىق نەرسىنى تاشلىۋىتىم ماڭا ئېسىلىشىدۇ . مەن ئۇلارنى سىرىتقا ئېلىپ چىقىپ ، ئوغلۇم ۋە قىزىمنىڭ تاشقى دۇنيادىكى بارلىق قىزىقىش ۋە ھەۋەسلىرىنى سەۋىرچانلىق بىلەن كۇزىتىمەن . كۆپ نەرسىلەرنى مەن ئۇلاردىن ئۇگىنىمەن يەنى بالىلىق دۇنيانىڭ خوشاللىقلىرىنى قايتىۋالغاندەك ھەۋەستە كۈلىمەن ... مەن بالىلىرىمنى باشقىلارنىڭ قولىغا تاشلاپ بەرمىدىم .ئايالىم ئۆزىنىڭ ئامراق كەسپى - دوختۇرلۇقنى تاشلاپ بالىلارنى تەربىلەۋاتىدۇ . مەن خىزمەت سەۋەبىدىن - دەم ئىلىشتىكى 10مىنۇت ۋاقتىمنى بولسىمۇ شۇلارغا ئاتا قىلىشقا تىرىشىتىم . مىنىڭ شەنبە يەكشەنبە ۋاقتىم بالىلىرىمنى باغچىغا ئاپىرىش بىلەن ئۆتىدۇ  ، ئورۇنسىز سورۇنلارغا سەرىپ قىلغان پۇللارنىڭ ئورنىغا ئۇلارنى ئۆزى ياخشى كۆرىدىغان ئويۇنچۇقلارغا سېلىپ ،خوش قىلىدىم ... ھازىر مىنىڭ تۇرمۇشۇم بەخىت ئىچىدە ... باياشات دۇنيا ئىچىدە ئەمەس . ئايالىم ئىككىمىز باشقىلارنىڭ ئۆيىدە بىر ئۆمۇر ئىجارىگە ئولتۇرۇشقا رازىكى بۇ بەخىتنى يوقاتمايمىز ....
      مۇشۇ سەۋەپلىك مەن بۇ تىمىنى يوللىغان . مىنىڭ ئوغلۇم بەكمۇ قەيسەر ، قىزىم پىكىر قىلىشقا بەك ماھىر ... بۇ مىنىڭ ئۇتۇغۇم ... چۇنكى مەن ئۇلارنىڭ يىنىدا دائىم بار ... مىنىڭ كىلىشىمنى ئۇلار كۇتىدۇ .. تەلپۇنۇشتە كۇتىدۇ ....چۇنكى مەن دادا ... ئائىلىسىنى سۆيىدىغان ئەر ... مەن ھەرگىزمۇ ئەرلىكى ماكچايغان ، مازلىق سەيناسىنىڭ چولپىنى بولۇشنى تەشەببۇس قىلغۇچى << تەرەققىياتچى >>ئەمەس . مىنىڭ بۇ تىمام بۇنىڭ دەلىلى :

قايسى تەرەپ تەربىيلەشكە كەلسەك، مەيلى ئوغۇل بولسۇن ياكى قىز بولسۇن دادا ئەڭ مۇۋاپىق، ئانا قوشۇمچە بولغىنى ياخشى.
ئايال كىشى دىگەن بالىغا ھەددىدىن زىيادە كۆيگەنلىكىمۇ ئىشقىلىپ بالىلارنىڭ بەزى خاتالىقلىرىنى يۇشۇرۇپ بالىسىغا بۇلۇشىدۇ،زىياردە ئەركە ئۆگۈتىپ قويىدۇ، يىقىلسا يۆلەيدۇ، چىرايلىق ياساپ قويىمەن دەپ قىممەت باھالىق كىيىملەرنى ئېلىپ بىردۇ، ئاقىۋەت ئۇ بالا يىقىلسا يۆلەشكە مۇھتاج، زىيادە كەپسىز ، بەڭباش، ئەقىلى تەرەققى قىلمىغان،  بەتخەجى، ئۆز ئالدىغا مۇستەقىل ياشىيالمايدىغان سۈنئىيلىكى يۇقىرى بولۇپ يىتىشىپ چىقىدۇ.  
جەڭگە ماڭسا جەڭگە يارىغانلىقتىن خۇشال بولۇپ يىغلىماي ئۆلۈپ كىتىشىدىن قورقۇپ يىغلايدۇ. شۇڭا بالا تەربىيلەشتە دادا باش بولغىنى مۇۋاپىق.

پوۋسىتنىڭ ئىنكاسى :
قىسقا ھەجىملىك بۇ يازمىنىڭ مەناسىنى قايتا - قايتا تەھلىل قىلدىم. تەرەققىياتنىڭ تەسەۋۋۇر قىلغۇسىز تېزلىكتە كىتىۋاتقان قەدىمى، ئېڭىمىزغا يۇشۇرۇن خىرىس قىلىۋاتقان تۈرلۈك كىرىزىس، ئىقتىسادنىڭ يەر شارىلىشىشى بىلەن قىممەتلىك بايلىقلىرىدىن ئايرىلىۋاتقان ئۇيغۇر جەمئىيىتىنىڭ ھازىرقى ھالىتىدە يالغۇز پەرزەنتلەرنى تەربىيىلەش ئىنتايىن مۇھىم ئىشلارنىڭ بىرى بولۇپ قالدى. }qv-lO  
مەزكۇر ئەسەرنىڭ ئاپتورى ناھايىتى سەگەكلىك بىلەن بۇ نۇقتىنى چالا - پۇلا بولسىمۇ تونۇپ يېتىپ قولىغا قەلەم ئالغان. شۇنداق بىر ياخشى پىكىرنى ئويلىغان - يۇ، ئەمما ئىپادىلەشتە كەتكۈزۈپ قويغان. ئاقىۋەت ئاپتورنىڭ كىچىككىنە ئالدىراڭغۇلۇقى ئاۋۋال ماۋزۇدا ئۆز ئىپادىسىنى تاپقان. كەينىدىن مەزمۇننىڭ چېچىلاڭغۇلۇقى ئاپتورنىڭ يۈرىكىدىكى پىكىرلەرنى تېرىقتەك چېچىۋەتكەن. شۇنىڭ بىلەن دىلدىكىنى تىل ئۇرۇندىيالمىغان.  
خۇلاسە كالامىم شۇكى، ئاپتورنىڭ مۇشۇنىڭغا ئوخشاش تەپەككۇرلىرى بىلەن مىسرانىم ئەھلىنىڭ ئالدىغا سوئال تاشلاپ تۇرۇشىنى ئۈمىد قىلىمىز!

ئومۇمەن خۇلاسە شۇكى :
  1)    بالىلار بىزنىڭ كەلگۇسىمىز ، ئۇلارنىڭ بىزنىڭ كىملىك بەلگىمىزنى داۋاملاشتۇرغۇچىلار شۇڭا ئۇلارغا بولغان تەربىيە بەك موھىم .

3) تۇترۇقسىز سورۇنغا بارغاندىن كۆرە ، بالىڭىزنى يۆتلەپ ، باغچەغا ياكى كەچنىڭ گۇزەل مەنزىرىسىدىن تەڭ بەھىرلىنىڭ .
