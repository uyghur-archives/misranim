    خەلقئارالىق كۈزىتىش : جۇڭگو ئىراق ئۇرۇشىنى چۆرۈۋەتكەن ئامېرىكىغا قانداق تاقابىل تۇرۇشى كېرەك
ئىراق ئۇرۇشىدىن كېيىن ئامېرىكا جۇڭگوغا قارىتا ئورۇنلاشتۇرۇش ئېلىپ بارماي قالارمۇ؟
 ئامېرىكا ئىراقتىن قوشۇنلىرىنى چېكىندۈرگەندىن كېيىن ، قولىنى بوشىتىپ جۇڭگوغا تاقابىل تۇرىدۇ -دەيدىغان بىر خىل گەپ بار ، بۇ گەپ ئاڭلىماققا داۋلىسى باردەك قىلىدۇ ، چۈنكى تارىختىن قارىغاندا ، ئامېرىكا ھامان ئاساسلىق رەقىب ئىزدەپ ئېشىپ كەلدى ، ئەمما ، بۇ خىل گەپنىڭ خاتا يېتەكلەش تەرىپىمۇ بولۇشى مۇمكىن ، چۈنكى ، ئامېرىكىنىڭ تېررورچىلارغا قارشى ئۇرۇشى تېخى ئاخىرلاشمىغان بولۇپ ، ئامېرىكا يەنىلا ئۇرۇش دەۋرىدە تۇرۇۋاتىدۇ ، ئافغانىستاندىكى ئۇرۇش تېخى ئاياغلاشمىدى. ئۇنىڭ ئاساسلىق دۈشمىنى يەنىلا تېررورچىلار ، ئامېرىكا چىكىنسىمۇ ، تېررورچىلار چېكىنمەيدۇ ، ئۇنىڭدىن قالسا ، ۋاشىنگوتۇننىڭ تۇتىدىغىنى ئىقتىسات بولۇپ ، بۈگۈنكى كۈندە ئامېرىكىنىڭ ئىقتىسادى دۇنيا ئىقتىسادىي بىلەن ، بولۇپمۇ يېڭىدىن گۈللەنگەن دۆلەتلەرنىڭ ئىقتىسادى بىلەن چىڭ باغلانغان . ئامېرىكىنىڭ ئىقتىسادى ئۆرلىشى ئۈچۈن ، پۈتكۈل دۇنيانىڭ ئىقتىسادى ئۆرلىشى كېرەك . مۇشۇ جەھەتتە ، ئامېرىكا دەرۋەقە يەنىلا جۇڭگونىڭ قوللىشىغا تايىنىدۇ.
ئامېرىكا بىراقلا جۇڭگونى رەقىبى ئورنىغا قويۇپ ، جۇڭگو بىلەن ئېلىشالمايدۇ ، ئامېرىكا ئىراقتىن چېكىنىپ چىققاندىن كېيىن ، جۇڭگوغا قارىتىدىغان سىياسەتلىرىگە ئاز بولغاندىمۇ مۇنداق ئىككى جەھەتتە تەسىر كۆرسىتىشى مۇمكىن ، بۇ ، بىزنىڭ تېخىمۇ كۆپ دىققەت قىلىشىمىزغا ئەرزىيدۇ.
1– ئىقتىساد ، ئوباما قوشۇن چېكىندۈرۈشكە مۇناسىۋەتلىك نۇتۇقلىرىدا مۇنۇلارنى تىلغا ئالدى: قوشۇن چېكىندۈرۈشتىكى بىر مەقسەت ئامېرىكا ھازىردىن باشلاپ كۈچىنى مەركەزلەشتۈرۈپ ئىچكى ئىقتىسادنى تۇتۇش كېرەك ، ئۇرۇش قىلىش ئۈچۈن پۇل خەجلەشكە توغرا كېلىدۇ ، ئامېرىكا ھازىر خەجلىيەلمەيدۇ ، تېخىمۇ كۆپ كۈچ سەرپ قىلىپ ئىچكى ئىقتىسادنى قۇتقۇزۇشقا توغرا كېلىۋاتىدۇ.
ۋاشىنگىتون ئىقتىسادىي قورۇلۇشىنى مەركەز قىلىدىغان بولغاچقا ، جۇڭگونىڭ ئورنى مۇقەررەر يۇقىرى ئۆرلەيدۇ . چۈنكى، جۇڭگو ئامېرىكىنىڭ ئەڭ چوڭ ھەقدارى ھېسابلىنىدۇ ، ئامېرىكا بولسا جۇڭگوغا ئەڭ چوڭ مەبلەغ سالغۇچى تەرەپ بولۇپ ، ئامېرىكا بىلەن جۇڭگو يەنە ئۆز ئارا ئەڭ چوڭ سودا شېرىكلىرى ھېسابلىنىدۇ.
بىراق ، بۇ ئۆرلەش ھەم ياخشى ئىش ، ھەم يامان ئىش ، ياخشى ئىش دېيىشتە ، جۇڭگو بىلەن ئامېرىكا ئىقتىسادىي ھازىر چىڭ باغلانغان بولۇپ ،  بىر تەرەپ گۈللەنسە تەڭ گۈللىنىش ، بىر تەرەپ زىيانغا ئۇچرىسا تەڭ زىيان تارتىش دېگەن گەپ بار . ئەگەر ئامېرىكىنىڭ ئىقتىسادىي ياخشىلانسا ، ئىشقا ئورۇنلىشىدىغانلار كۆپىيىپ ئىستېمال جەزمەن قايتىدىن جانلىنىدۇ، بۇ ، جۇڭگونىڭ ئېكسپورت كارخانىلىرىغا نىسبەتەن ئەلۋەتتە ياخشى ئىش ھېسابلىنىدۇ.
يامان ئىشچۇ ؟ ئوباما ئىقتىسادنى تۇتۇپ ، ئىشقا ئورۇنلىشىدىغانلارنى كۆپەيتىشتە ، ناھايىتى ئاسانلا جۇڭگونى ئامېرىكىلىقلارنىڭ تاۋىقىنى تارتىۋالماقچى بولغان بىرىنچى رىقابەت رەقىبى دەپ قارايدۇ ، ئامېرىكا ھازىر يەنە ئىكىسپۇرتنى كېڭەيتىشنى ئەڭ مۇھىم جانلىنىش رېتسىپى دەپ قاراۋاتىدۇ . ئالدىنقىسى ، ۋاشىنگىتوننى بەزى سودا قورۇقچىلىقى كونا ھېلىسنى ئىشلىتىشكە مەجبۇرلىسا ، كېيىنكىسى ، جۇڭگو بىلەن سۆھبەت ئۆتكۈزگەندە، جۇڭگودىن بەزى بازار دائىرىسىنىڭ چوڭ ئىشىكىنى يەنىمۇ كەڭرەك ئېچىۋېتىشنى تەلەپ قىلىۋاتىدۇ ، بۇلارنىڭ ھەممىسى جۇڭگو ئۈچۈن ئازدۇر–كۆپتۇر ئاۋارىچىلىق ئېلىپ كېلىدۇ.
2– سىياسەت ، ئوباما مۇنداق دەيدۇ : ئىراقتىن ئالغان ساۋاقنىڭ بىرى شۇكى ، ئامېرىكا پۈتۈن دۇنيا تەسىر كۆرسىتەلەيدىغان بولۇش ئۈچۈن ، پەقەت ھەربىي كۈچىنى جارى قىلدۇرۇشقا تايانسىلا بولمايدۇ ، يەنە بەزىلەر مۇنداق دېيىشمەكتە ، بۇ ، ئامېرىكىنىڭ بۇنىڭدىن كېيىنكى بىر مەزگىل ئىچىدە ھەربىي كۈچىدىن ئېھتىياتچانلىق بىلەن پايدىلىنىدىغانلىقىدىن دېرەك بەرسىمۇ ، ئەمما ، ئېھتىياتچان دېگەن سۆز تېخى تولۇق بولماي قاپتۇ ، تېخىمۇ ئەپچىللىك بىلەن پايدىلىنىش دېسە توغرا بولغىدەك ، بۇ يەنە ، سىياسى تەسىر كۈچىدىن تېخىمۇ ئەپچىل پايدىلىنىش دېگەننىمۇ ئۆز ئىچىگە ئالىدۇ. جۇڭگونىڭ ئەتراپىغا نىسبەتەن ئېيتقاندا، ئامېرىكا  ئاسىياغا قايتىدىن بېرىش نىمۇ تىلغا ئالدى . بۇ قايتىدىن بېرىش تەسىر كۈچىنىڭ قايتىدىن بارىدىغانلىقىنى كۆرسىتىدۇ . بەزىلەر ئامېرىكىنىڭ قايتا كەلگەنلىكىگە ئالقىش ياڭراتتىغۇ ؟ ياپونىيەنىڭ سابىق قوغدىنىش ۋەزىرى يېقىندا ئېلان قىلغان بىر پارچە ماقالىسىدا ھىلارىنى كېسنگىرگە ئوخشىتىپ ئەينى يىلى كېسنگىر جۇڭگونى زىيارەت قىلغاندا ، جۇڭگونىڭ دەرۋازىسىنى ئېچىۋېتىشنى ئىلگىرى سۈرگەن بولسا ، بۈگۈنكى كۈندە ھىلارىنىڭ ئاسىياغا قىلغان سەپىرى: ئامېرىكا جۇڭگونىڭ رايون زومىگەرلىككە ئىنتىلىشنى خالىمايدۇ دېگەن سىگنالنى چىقاردى ، دېدى.
  ئامېرىكا يەنە بىر مۇنچە قوش تەرەپ ، كۆپ تەرەپلىمىلىك ۋە زور خەلقئارالىق سورۇنلاردا ، بەزى زور ۋە رايون خاراكتېرلىك قىزىق نۇقتىلىق مەسىلىلەردە جۇڭگوغا بېسىم ئىشلىتىپ ، جۇڭگودىن ئۆزىنىڭ تەلىپى بويىچە ئىشلەشنى تەلەپ قىلىش مۇمكىن، بۇلار ، بۇنىڭدىن كېيىنكى بىر مەزگىلدە بىز دۈچ كېلىدىغان ئاساسلىق ئاۋارىچىلىقلار بولۇشى مۇمكىن.
 بىراق ، جۇڭگو بىلەن ئامېرىكا ئوتتۇرىسىدىكى بۇ ئاۋارىچىلىقلار ماھىيەت جەھەتتىن ئالغاندا قانداقتۇ يېڭى نەرسە ئەمەس ، بۇ ئاۋارىچىلىقلارنىڭ كۆپىيىشى بىلەن ، جۇڭگو–ئامېرىكا مۇناسىۋىتىنىڭ مۇشۇ يىللاردىن بۇيانقى تەرەققىيات يولىنى ئۆزگەرتىشكىمۇ بولمايدۇ.
 جۇڭگو–ئامېرىكا مۇناسىۋىتىدە بۇنىڭلىق بىلەن چوڭ بۇرۇلۇش كۆرۈلمەيدۇ ، يەنىلا بۇرۇنقىدەك ئەگرى–توقايلىق ئىچىدە ئالغا باسىدۇ . شۇڭا بىز ئامېرىكىنىڭ قوشۇنلىرىنى چېكىندۈرۈپ ، قوللىرىنى بوشىتىشىدىن بورۇن ، ئۆزىمىز جىددىيلىشىپ كەتمەسلىكىمىز ، جۇڭگو–ئامېرىكا مۇناسىۋىتىدە كۆرۈلگەن بەزى مەسىلىلەرگە قارىتا سەزگۈرلۈكىمىزنى بەكمۇ ئاشۇرۇۋەتمەسلىكىمىز لازىم . ھازىرقى پەيتتە ، سەۋرچان–تەمكىن بولۇپ ، جۇڭگو–ئامېرىكا مۇناسىۋىتىنىڭ يۈزلىنىشىنى ئايدىڭلاشتۇرۇش زۆرۈر.
