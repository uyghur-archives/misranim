    گېموروي قانداق شەكىللىنىدۇ ؟بۇ ھەققىتە نۆۋەتتە ئۈچ خىل كۆز قاراش مەۋجۇت . 
    مەقئەت،تۈزئۈچەي ئاستىدىكى قان تومۇرلۇق قوش ئەستەرنىڭ تۆۋەنگە سۈرۈلىشى : تۈزئۈچەي ئەينىكىدە تەكشۈرۈپ ، بىر قېتىمىدىلا تۈز ئۈچەي ئاستى قىسمىنىڭ ئوڭ تەرەپ ئالدى ، ئوڭ تەرەپ كەينى ۋە سول تەرەپتىن ئىبارەت ئۈچ ئورۇندا ئۈچ پارچە قېلىنلاشقان قوش ئەستەر بارلىقنى بايقىغىلى بولىدۇ .قوش ئەستەر مەقئەت كانىلىنىڭ يېپىلىشىغا ياردەم بېرىدۇ . قوش ئەستەر تۈز ئۈچەي ئاستى قىسىم شىللىق پەرىدىسى ئاستىدىكى قېلىن بىرىكىمە توقۇلمىلار ۋە ۋېنا قان تومۇر توپلىمىدىن تۈزۈلگەن . ئەگەر قەۋزىيەت بولۇپ چوڭ تەرەت قىلىش قىيىنلاشسا ، زىيادە كۈچىسە، تەرەت چىقارغاندا تەرەتنىڭ قىستاپ ئىتتىرىشى بىلەن قوش ئەستەر تۆۋەنگە سۈرۈلۈپ، ۋېنا قان تومۇرلىرىغا قان قىيىپ كېڭيشى نەتىجىسىدە گېموروي شەككىلىنىدۇ . 

    مەقئەت،تۈزئۈچەي قىسمىدىكى ۋېنالارنىڭ ئەگرى-بۈگرى كېڭىيشى : گېموروينىڭ ئاساسىي پاتالوگىيىلىك ئۆزگىرىشى ۋېنا قان تومۇرلىرىنىڭ كېڭىيشىدىن ئىبارەت . قەۋزىيەت ، ئۇزۇن مۇددەتلىك ئىچى سۈرۈش ، گېموروي ۋېنا تومۇرلىرىدا ۋېنا قاپقىسى بولماسىلىق ، ئادەم ئۆرە تۇرغاندىكى يەرشارىنىڭ تارتىش كۇچى قاتارلىق ئامىللار ۋېنا قان تومۇرىنىڭ ئىچكى بېسىمىنى ئاشۇرىدۇ-دە ، قان قايتىش ئاستىلاپ ، ۋېنا قان تومۇرلىرى ئەگرى-بۈگرى كېڭىيىپ  گېموروي شەككىلىنىدۇ .

    قان-تومۇرلارنىڭ ئۇستىدە ئۆسۈكچىلەرنىڭ پەيدا بولۇشى :ھىستولوگىيە ئىلىمىي تەتقىقات نەتىجىلىرى گېموروينى قان تومۇرلارنىڭ ئۈستىدە ئۆسۈكچىلەرنىڭ پەيدا بولۇپ ، قان تومۇر ئۆسمىسىگە ئايلانغانلىقىدىن ھاسىل بولغانلىقىنى جەزملەشتۈردى .
