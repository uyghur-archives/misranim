ئەسسەلامۇ ئەلەيكۇم ، مۇنبەرداشلار ، تۆۋەندە ئۆزەم ئىشلىگەن سۈرەتلىك ئاۋازلىق دەرسلىكنى سىلەرنىڭ پايدىلىنىشىڭلارغا سۇنماقچىمەن، تۆۋەندىكى ئاۋازلىق ھۆججەتتىن دىتالنىڭ ئالاھىدىلىكىنى بىلىۋېلىڭ، ئازراق جاپا تارتىپ قالدىم ،شۇڭا ئازبولسىمۇ پۇلغا ساتماقچى، ئەمگىكىمنى ھۆرمەت قىلغايىسىز  

بۇ دىتال بولسا قاتتىق دىسكىدىكى ھۆججەتلەرنى زىيانغا ئۇچراتمىغان ئاساستا قاتتىق دىسكىنى رايۇنغا ئايرۇيدۇ، مەن شۇ دىتالنىڭ ئىشلىتىش ئۇسۇلىنى دەرسلىك قىلىپ قويغان، خالىغانلار كۆرۈڭلار تەپسىلىي چۈشەندۈرۈلگەن،  

قاتتىق دىسكىنى رايۇنغا ئايرىش دەرسلىكىنىڭ چۈشەندۈرۈلىشى ،باشتا بۇنى تىڭشاپ بېقىڭ، قىززىقسىڭىز تۆۋەندىكى دەرسلىكنى سېتىۋالسىڭىز بۇلىدۇ، مۇنبەر پۇلىغا، بۇ خەتنى چىكىپ ئاڭلاڭ
دەرسىلىكنى مۇنبەر يۈزىدىلا كۆرەلەيسىز، چۈشۈرۈپ يۈرمەيسىز، چۈشۈرۈپ كۆرىدىغان قىلىپ قويسام، كۆرىدىغانلار ئاز چىقىپ قالغۇدەك، چۈشۈرۋالاي دىگۈچىلەر بولسا ئادىرسىنى مۇشۇ تېمىنى سېتىۋالسا يوللاپ بىرىمەن،قىسقا ئۇچۇر قىلسىڭىز ياكى چچنى قاچىلۋالسىڭزى بۇلىدۇ، ئەگەر تۆۋەندىكى دىتالنى ئىشلىتىشنى يەنىلا بىلالمىسەڭلار ياكى قاتتىق دىسكىدىكى نەرسىلەرنىڭ يوقاپ كىتىشىدىن ئەنسىرسەڭلار، مەن بىر قوللۇق قىلىپ بەرسەممۇ بۇلىدۇ، چچ ئارقىلىق يىراقتىن كومپىيۇتىرىڭىزنى كونترول قىلىپ ھەممە مەشغۇلاتنى 100% قىلىپ بىرىمەن، بۇنىڭغا ھەق ئالمايمەن، لىكىن دۇئا ئالىمەن،چچ؛872795362    649140221

مەن بۇنى نەق پۇلغا ساتمىغاندىكىن قورقماي سېتىۋېلىڭلار، مەشغۇلاتنى دەرسلىكتەك قىلساڭلار، چۇقۇم مۇۋاپىقىيەتلىك قىلالايسىلەر، بۇ مۇنبەردە پىخسىقلار بەك كۆپمۇ نىمە ،سېتىۋالغانلار ھەجەپ ئازغۇ، ئەگەر باشقىچە ئۇسۇلدا قاتتىق دىسكىنى رايۇنغا ئايرىيمەن دىسەڭلار، بەك ئاۋارچىلىق تارتىسىلەر، ھەم ئىچىدىكى بارلىق ھۆججەتلىرىڭلاريۇيۇلۇپ كىتىدۇ، مۇشۇنداق پايدىلىق بىلىم يوللىساممۇ ناھايتى ئازغىنە كىشى سېتىۋالغىنى دىسە، قارىغاندا بۇندىن كىيىن بۇنداق تېما يوللىمىسام بۇلىدىغان ئوخشايدۇ، مۇشۇ تېمىنى تەييارلايمەن دەپ 5 سائەت ۋاقتىم كەتتى،
دەرسنى ئىشلەشكە 2 سائەت، تورغا يوللاشقا ئىككى سائەت، تېما قىلىپ يىزىشقا بىر سائەت ۋاقتىم كەتتى، مەن ۋاقتىم كەتتى دەپ قىلغان ئىشىمدىن مىننەت قىلغىنىم ئەمەس،مۇشۇنداق پايدىلىق تېمىنى ناھايتى ئاز كىشىلەرنىڭ كۆرگىنىدىن ئەپسۇسلىنىمەن، بىر قېتىم كۆرسىلا ئۆگىنىۋالدىغان بىلىم تۇرسا بۇ، ھازىر 90% كىشى بازارغا بېرىپ ۇستىلارغا پۇل بېرىپ رايۇنغا ئايرىتىدۇ، ياكى فورماتلىتىدۇ، مىنىڭ ئۆيۇمنىڭ يېنىدىكى بىر كومپىيۇتىر رېمنۇتخانىسىغىمۇ مۇشۇ توغرۇلۇق كومپىيۇتىرنى رايۇنغا ئايرىتىش قا كىلىدىغان كىشىلەر خېلى بار.بۇ دەرسلىك ھەر قانداق كومپىيۇتىرى بار كىشىلەرگە كىينچە ئەسقاتىدۇ،

قارىغاندا بۇندىن كىيىنكى پىلانلىرىمنى مۇشۇ يەردە توختۇتۇپ قويسام بولغۇدەك،
