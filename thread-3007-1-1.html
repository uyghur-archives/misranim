    _ مەرھۇم ئۇستاز يازغۇچى تۇردى سامساق ئاكىغا قايغۇرىمەن      


       ئەزىم دەريالار كالىنىڭ كۆزىدەك كىچىككىنە بۇلاقلاردىن باشلىنىدۇ، ئاققانچە زورىيىپ ھەيۋەتلىك ئېقىنغا ئايلىنىدۇ. قەلەمكەشلەر قوشۇنى دەل ئۇنىڭ ئەكسىچە. بۇ ئېقىنىنىڭ بېشى چۈمۈلنىڭ ئۇۋىسىدەك قىستاڭچىلىق، دېھقان بازىرىدەك قايناق،ئاخىرى قانداقتۇر، دەپ نەزەر سېلىپ ئوتتۇرا ئېقىنغا قارىساق ئارىسالدىچىلىق، ئاخىرىغا قارىغۇدەك بولساق، تىلغا ئالغۇدەك، ئەستە قالغۇدەك، بارماق بىلەن سانىغۇدەك چوڭ ئىش يوق. ئەتىيازدا ئېتىزغا مايسا يۇڭدەك ئۈنۈپ، ھوسۈل يىغىدىغاندا خاماندا داننىڭ تايىنى بولمىغاندەك ئىش.
      مەرھۇم ئاكىمىز بالىلىق چاغلىرىدا بەشكېرەم باغلىرىغا سىغماي، تونۇردەك قىزىغان شېغىللىق سايلارنى يالاڭ ئاياغ كېزىپ، ھەق - ئادالەت ئىزدەپ، جەنۇپنىڭ تاقىر تاغلىرىدىن ئېشىپ، تەڭرىتېغىنىڭ شىمالدىكى گۈزەل يايلاقلار، قارىغايلىق، قار - مۇزلوق چوققىلاردا قاڭقىپ، ئاخىر ھەق - ئادالەتنىڭ مەيلى قەيەردە بولمىسۇن، كۇرەش، جان تىكىپ ئېلىشىش ئارقىلىق ئېرىشكىلى بولىدىغان مۇقەددەس نەرسە ئىكەنلىكىنى تونۇپ، شىمالى تەڭرىتاغ ئېتىكىدە يامغۇردەك يېغۋاتقان ئوقلار ئارىسىدا <<ھۇررا>> توۋلاپ، قان كېچىپ،سەپ يېرىپ، ئات ئوينىتىپ، غازى بولۇپ، بۈگۈنكى كۈنگە ناھايىتى تەستە ئېرىشكەندى، بىزدە ئەلەمدىنمۇ، قەلەمدىمۇ غازى بولغان بىر ئادەم ئىدى. پىلىموت، تانكا، ئايرۇپلان، زەمبىرەك ئوقلىرى ئارىسىدىن ساق چىقىپ،بىر قار نىيەت، نازۇك بارماقنىڭ تىغىدا جان بىرىش _ تەڭرى بىزگە  ئورۇنلاشتۇرغان ھەيرانلىق. ئۇنى ھازىرقى زامان ئۇيغۇر رېئالىستىك پىروزىسىغا تۈۋرۈك بولغان يازغۇچى دېيىشتىن ئايانمايمەن. ئۇ بالىلىق دەۋرىدىمۇ دۇنيانىڭ قاتتىقچىلىقى، ئۇيغۇر دىھقانلىرنىڭ دۇنيادىكى ئەڭ تۆۋەن،بىچارە، نامراتلىق كۈنلىرىنى ئۆز بېشىدىن ئۆتكۈزدى، ئالتۇندەك ياشلىقنى شۇ خورلۇق ئۇچۇن جەڭلەردە قۇربان قىلدى، قىران چاغلىرىنى شغېخىل ئازگاللىرىدا ئاچلىق،سوغۇق، ئېغىر ئەمگەكتە ئۆتكۈزسە، كېچىلىرى ئەدەبىياتىمىز ئۇچۇن يۈرەك قېنىنى سىياھ قىلىپ، ئەل - ۋەتەن غېمىدە قەلەم تەۋرەتكەنىدى.
      ھازىرقى ئۇيغۇر ئەدەبىيات قوشۇندا تۆھپىكار مويسىپىتلەر،ساھىب قەلەم ئۇستاز مۇئەللىپلەر بارماق بىلەن سانىغۇدەك ئاز قالدى. ئۇلار ئەدەبىياتىمىزنىڭ پىشۋالىرى، مۆھتەرەم ئاتىلىرى، تەۋەرۇك كىشىلىرى_ بېشىمىزغا ئېلىپ كۆتۈردىغان ئەزىز مىھمانلىرىمىز. رەنجى- مۇشەققەت ئىچىدە ئۆتكۈزدى. جىسمانىي، روھىي جەھەتتىن كۆپ خورلۇق، ئېزىلىشكە ئۇچراپ، بىر قەدەر ئىنسانىي كۈنگە ئېرىشكىلى ئۇزاق بولمىدى. <<تەڭرىتاغ باغرىدا>>، <<بەش تال ئوق>>، <<قەشقەر گۈزىلى>>. . .مەن قايتىلاپ ئۆگەنگەن سەرخىل ئۇيغۇر يازغۇچىلار يازمىلىرى ئىچىدە يۈرەكتىن ئورۇن ئالغان ئەسەرلەرنىڭ ئالدىنقى قاتاردا ئىدى. مەن دېمىسەممۇ، كىتاب ئوقۇيدىغان ئۇيغۇر جامائەتلىرى ئۇيغۇر ئەدەبىياتى ئۇچۇن ئۇ مۆھتەرەم زاتنىڭ قانچىلغان نادىر ئەسەرلىرى، توم - توم كىتابلارنى يېزىپ تەغدىم قىلغانلىقىنى بىلىدۇ. ئۇ، ھەق ئۇچۇن سەمىمىلىك بىلەن سۆزلىدى.ئۇ ھەق ئۇچۇن قان كېچىپ جەڭلەرنى قىلدى. ئۇ ھەق ئۇچۇن 50 يىل يازدى.
      بۇنىڭدىن بەش يىل بۇرۇن قېنى قىزىق بىر ياش ھەۋەسكار بىلەن تىركىشىپ قالغىنىم ئېسىمدە.
     _ تۇردى ئاكام قېرغاندا سۇپىتىنى ساقلاپ ئولتۇرماي ئىناۋىتىنى چۈشۈرۈۋالدى،
     _ نېمە بولدى؟
     _ سېتىقچى بولۇۋالدى.
     _  نېمە ساتىدىكەن؟ _ چۆچۇپ كەتكەنىدىم.
     _ تىجارەتچىلەردىن رەخىت ئېلىپ ساتىدىكەن.
     _ شۇنىڭ بىلەن ئىناۋىتى چۈشۈپ كەتتىما ؟
     _ چۇشمەي ئەمىسە .
     _  سىزچە نېمە ئىش قىلسا بولاتتى ؟
     _ ھۆرمەتلىك يازغۇچى بولغاندىكىن، يەنە ئەسەر يېزىشى كېرەك _ تە.
     _ ئۇيغۇرلار ئۇچۇن 50 يىل قەلەم تەۋرەتتى، نۇرغۇن كىتاب يازدى. قۇرۇمايدىغان بۇلاق يوق، دۇنيامۇ ئاخىرى تۇگەيدۇ. ئۇ ئادەم 60 تىن ئاشتى. يازىدىغان ئىشمۇ كۈچ - مادارغا باقىدۇ، چارچىدى.قېرىغان ئادەمدىن تەمە قىلغۇچە، سىلى _ بىزدەك ياشلار يازغىنىمىز ياخشى، مېنىڭچە ئۇ ئادەم ئەڭ ئۇبدان ئىشقا تۇردى.
     _  نېمىشقا ئۇنداق دەيسىز ؟
     _ چۇنكى، ئەمدى ياش چېغىدىدىكىدەك يازالمىغاندىكىن، ئەدەبىيات سورۇنىغا مەنمۇ ئەخلەت تۆكمەي دەپ ئۆزىنى ئايىغاندۇ. يىرىم ئەسىردىن تارتىپ ئادەملەرنىڭ روھيىتىنى گۈزەللەشتۇرەي دەپ قەلەم تۇتقان بولسا، ئەمدى ئۈستىۋىشىنى گۈزەللەشتۇرەي دەپ كىيىم _ كېچەك ساتقاندۇ. ساتسا رەخت سېتىپتىغۇ، بەزى مۇناپىق ئادەملەردەك ۋىجدان ساتماپتۇ.
    _ بۇ گېپىڭىزمۇ راست، لېكىن شۇنى دەپ قوياي: يازغۇچى ئەڭ ئاۋۋال ئادەم بولۇشى، ئادەم بولۇشى ئۇچۇن ئادەمدەك تۇرمۇش كەچۇرشى كىرەك. نامراتلىق، گادايلىق، ئىچىدە ياشاپ يازغۇچىلىق قىلىشنىڭ ئۆزى بىر ئاجىزلىق، لېكىن يازغۇچىلارنىڭ كۆپى مەدىكار بىلەن دىھقان ئوتتۇرسىدىكى تولىمۇ بىچارە بىر خىل تۇرمۇش ئىچىدە ياشاۋاتىدۇ. ئۇ گاداي، پېقىر تۇرمۇشقا پىشىپ كەتتى. ئەمدى ھازىرقى زاماندىكى باياشاتلىققا يۈزلىنىۋاتقان يېڭىچە قىممەت قارىشىغا ئىگە ئادەمنىڭ تۇرمۇشىنى تېتىپ كۆرمىسە، ئۇنىڭدىن ھازىرقى زامان ئادىمىنى يازغان نادىر ئەسەرلەرنىڭ چىقىشى مۇمكىن ئەمەس. يازغانلىرى پەقەت ئاچلىق. گادايلىق، بىچارىلىك ھەققىدىكى شىكايەتنامىدىن ھالقىـــيالمايدۇ. يازغۇچىلا ئادەمدەك ياشىشى كېرەك.
    بىزدە يازغۇچىلىق نامىراتلىق ئىچىدىكى بىر سۇنئــىي رېئاللىق. شۇڭا، جەمئىيەتتە ئۇنچىۋالا ئېتىبارغا ئېلىنىپ كەتمەيدۇ. بۇنى تۇردى سامساق ئاكا ئۆمۇر بويى قەلەم تۇتۇپ ياشاپ، ئۆمۇرىنىڭ ئاخىرىدا چوڭقۇر چۈشەنگەنىدى .
     تۇردى ئاكا، ماڭا نۇرغۇن ئارمانلىرىڭىزنى سۆزلەپ بەرگەنىدىڭىز. قەلىمىڭىز تىخىچە ياشلىقىنى يوقاتمىغان قېرى تۇلپاردەك يورغىلاپ، چىڭ سۇلالىسى دەۋىدىكى قەشقەر دېھقانلا قوزغىلىڭىنىڭ مەشھۇر سەركەدىسى توختى مانجۇ ھەققىدىكى رومان بىلەن مۇڭداشماقتا ئىدى . ھازىر قانچىلىك يازغان بولغىيدىڭىز ؟! قان كېچىپ، جەڭ قىلغان سەبداشلىرىڭىز  _ مەرھۇم زۇنۇن قادىر، تېيىپجان ئېلىيوپ ۋە ئابدىكىرىم خوجا، ئاشۇ توختى مانجولامۇ نۇرغۇن ئارمانلار بىلەن كەتكەنىدى،بۇ چاغقا ئۇلار بىلەن كۆرۈشكەنسىز . بىز تىرىك تۇرۇپ ئارمانلارنى يېزىپ بولالمايۋاتىمىز، ھەقىچان ئارمانلار ئوتىدا كۆيۇپ، بۇ دۇنيادىن سىزگە ئەگىشىپ بارىمىز.
     ئەي مويسىپىت ئاتىلار، ئۆزۇڭلارنى ئاسىراڭلار، ئۆمرۈڭلارنى بەرگەي، بىزنىڭ سورۇنىمىزدا سىلەر بولمىساڭلار يېتىملىكىمىز چوقۇم چېنىپ قالىدۇ.
