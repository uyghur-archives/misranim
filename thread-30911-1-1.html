ئەسسالامۇ ئەلەيكۇم مىسرانىم تور بېكىتىنىڭ ئەزىزانە قوللىغۇچىلىرى.. مىسرانىم تور بېكىتىمىز ،جۈملىدىن مۇنبىرىمىز نۇرغۇن تورداشلارنىڭ قوللاپ قۇۋەتلىشىگە ئېرىشكەندىن كېيىن، پۈتكۈل زېھنىمنى مۇنبەرنىڭ تېخنىكىلىق ھەم گۈزەللىك ئىشلىرىغا قارىتىپ مىسرانىم باش بېتىنىڭ رەڭدارلىقى تۆۋەنلىتىپ ھەم مۇنبەر باشلىقى بىلەن ئەزالار ئوتتۇرسىدا <ئالاقىلىشالماسلىق> تۈگۈنى پەيدا قىلىپ قويدۇم ...

شۇ سەۋەبلىك مىسرانىم مۇنبىرىگە باشقۇرۇش تالانتى بار، ئۆزىنى تۇتىۋالغان ، ئېغىر بېسىق ، ھەم بىر قىسىم مۇنبەرلەردە مۇنبەر باشلىقلىق ،باشقۇرغۇچىلىق ۋەزىپىسىنى ئۆتۈپ كەلگەن ،مىسرانىمدىمۇ ئىزچىللىقنى ساقلاپ كەلگەن ئىككى نەپەر مۇنبەر ئەزانى مۇنبەر باشلىقلىق ۋەزىپىسىگە ئولتۇرغۇزدۇق!
ئۇلارنىڭ مۇنبەردىكى نامى ئۇيغۇرجان ھەم ئۇيغۇرۇم007 ،، بىز بۇ ئىككى ئەزا بىلەن سۆزلەشكەندىن كىيىن بىر قىسىم مۇنبەر ئىشلىرىنىڭ ھۆددىسىدىن تولۇق چىقىپ كېتەلەيدىغانلىقىنى كۆزدە تۇتۇپ مۇنبەر باشلىقلىق ئۇنۋانىنى بەردۇق!





ئۇلارنىڭ ھوقوق دائىرىسى --- مەخپىي نۇمۇرى يۈتۈپ كەتكەنلارنىڭ مەخپىي نۇمرىنى قايتۇرۇپ بېرىىش ،، قايتا ئورنىتىش، مېدال تارقىتىش،مېدال قايتۇرۋىلىش، باشقۇرغۇچى سايلاش،باشقۇرغۇچى چۈشۈرۈش.. ئالىي ئەزا سايلاش ھەم چۈشۈرۈش، مۇنبەر پۇلى ،مۇنبەر نۇمۇرى بېرىش ھەم تارتىۋىلىش،، ئەخلەت تېمىلارنى بىر تەرەپ قىلىش، خاتا ئۆچۈرلۈپ كەتكەن تېمىلارنى ئەسلىگە كەلتۈرۈش ياكى مۇۋاپىق بىر تەرەپ قىلىش، ئەزا چەكلەش، ئەزا ئارخىبىنى مۇنبەر ساندانىدىن ئۆچۈرۈپ تاشلاش قاتارلىق مۇنبەردە بار بولغان بارلىق ھوقوقلاردۇر!
2011- يىلى 1- ئاينىڭ 1 - كۈنىدىن باشلاپ مۇنبەرنىڭ بارلىق ئامانلىق ھەم نازارەت ،مۇلازىمەت ئىشلىرىنى مەزكۇر ئىككى ئەزا ئۆز ئۈستىگە ئالىدۇ.

يېقىندىن بۇيان بىر قىسىم ئەزالار ئىسمىنى ئۇيغۇرچىغا ئۆزگەرتىپ بېرىش تەكلىبىنى ئېيتىپ ئۆتتى ..
بىراق سانداننىڭ زىيادە چوڭ بولۇپ كەتكەنلىكى ھەم يېڭى نەشىرگە كۆتۈرگەندە بىر قىسىم ئۇيغۇرچە ئىسىملارنىڭ سانداندىن پۈتۈنلەي چۈشۈپ قېلىشىنىڭ ئالدىنى ئېلىش ئۈچۈن تىزىملىتىلىپ بولۇنغان ئىسىملار ئۇيغۇرچىغا ئۆزگەرتىپ بېرىلمەيدۇ.

باشقا تېخنىكىلىق مەسىللەرنى مىسرانىم تور باشقۇرۇش ئىشخانىسى بىرلىكتە قىلىدۇ!

تورداشلارنىڭ بۇ ئىككى ئەزارىمىزنى قوللىشىنى ھەم خىزمىتىگە يېقىندىن ماسلىشىپ بېرىشىنى ئۈمۈد قىلىمىز!
