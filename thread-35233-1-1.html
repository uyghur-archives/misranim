                                                     مەسىتلىك پەسىتلىكتۇر
                                                           داۋامى

بىرىنجى قەدەھ ئۇسسۇزلۇقنى بېسىش ئۈچۈن ، ئىككىنجىسى خۇشاللىق ، ئۈچۈنجىسى كەيىپ بۇلۇپ ھۇزۇرلىنىش ، تۆتىنجىسى بولسا ، ئەقىلدىن ئېزىش ئۈچۈن ئىچىلىدۇ .
                                                                                           - ئانار خارسس

مەسىتلىك ئەقىلدىن ئېزىش مەشقىدۇر .
                                                                                          - پىغاگور

مەسىتلىك تازا ئەقىلسىزلىك ، ئۇ بىزنى ئىستىداتلىرىمىزدىن مەھرۇم قىلىدۇ .
                                                                                          - سولۇن

ئادەملەر ئىچىملىك ئىچىپ ، يەنىمۇ بەتتەر قۇپاللىشىپ ، تەنتەك بۇلۇپ ، رەھىمسىزلىشىپ كىتىدۇ .
                                                                                         - ل . ن . تولىستوي

ئەقىلنى كىمەيتىشنىڭ ئەڭ نەق ۋاستىسى -ئىچىملىك .
                                                                                       - ۋ . يا . دانىلوۋسكىي

بەك كۆپ ئىچكەن ئادەم سىز ساقلىيالمايدۇ ھەم ۋدىسىگە ۋاپامۇ قىلمايدۇ .
                                                                                     - م . سېرۋانتىس

ئىچىملىك چىكىدىن ئېشىپ كەتسە ، تەلۋە ئادەمنىمۇ ، دانا ئادەمنىمۇ بىردەك يىنىك قىلىپ قۇيىدۇ .
                                                                                    - فېوگنىد

مەسىتلىك ۋە مەدىنيەت - مۇز ۋە ئوتتەك ، نۇر ۋە زۇلمەتتەك ئۆز-ئارا بىر-بىرىگە مۇتلەق قارمۇ-قارشى چۈشەنچە .
                                                                                    - ن . ئا . سېماشكو

مەسىتلىك ۋە جىسمانىي تەربىيە بىر-بىرىگە زىتتۇر .
                                                                                   - ن . ئا . سېماشكو

مەن ھىچقاچان دۇنيادا ھاردۇقنى چىقىرىش ، پىكىرىمنى ئايدىڭلاشتۇرۇش ئۈچۈن ئىچىملىككە ئاشنا بۇلۇشنى خالىمايتىم .                                                                  - ت . گىگسىل

نادانلار ئىچىملىك ئىچكەندە مەسىت بۇلىدۇ . بەخىتسىزلىكتە بولسا - ئەقىل-ھوشىدىن ئاجرايدۇ .
                                                                                 - پىفاگور

ئەينەك ئادەمنىڭ يۈزىنى ئەكىس ئەتتۈرگەندەك ، مەسىتلىك ئۇنىڭ رۇھى دۇنياسىنى ئېچىپ كۆرسىتىدۇ .
                                                                                 - ئەسفىل

ئۇسسۇزلۇقنى سۇ بىلەن ئەمەس ، ئىچىملىك بىلەن قاندۇرغان ئادەم ھايۋاندىن بەتتەر .
                                                                                - دىئوگېن

ھەددىدىن تاشقىرى ئىچىملىككە بىرىلىش ھىچقانداق نەپ بەرمەيدۇ . ئەكىسچە ئەقىلنى ئاجىزلاشتۇرۇپ ، سالامەتلىكنى ناچارلىتىدۇ ، يانچۇقنى قۇرۇغداپ ۋاقىتسىز ئۆلۈمگە ئېلىپ كىلىدۇ .
                                                                               - ئى . ت . پوساشكوف

ئەقلىي ئەمگەكتە چارچىغان قانچىلىغان كىشىلەر ۋە قانچىلىغان قارا ئىشچىلار ئىچىملىك دامىغا چۈشۈپ ، ئۆزلىرىنى ئۆلۈمگە ئېلىپ بارغۇچى مۇدىھىش يولغا كىرىپ قالىدۇ .
                                                                               - ژ لوندون

ھاراقكەش ئادەم ھىچقاچان ئۆز سالامەتلىكىنىڭ غېمىنى يېيەلمەيدۇ ۋە ئۇزاق ئۆمۈر كۆرەلمەيدۇ .
                                                                             - ئا . ب . نېلوبىن

ئىچىملىك كۈچىگە كۈچ قوشمايدۇ ، بەلكى تېخىمۇ بۇشاشتۇرىدۇ . دىلنى ، ۋۇجۇدىنى ۋە خارەكتىرىنى ياشارىتمايدۇ ، بەلكى تېخىمۇ قېرىتىدۇ ۋە ئەقىلنى ھاجىزلاشتۇرىدۇ !
                                                                             - ر . كوخ

ئىچىملىك تەسىرىدە پەيدا بۇلىدى غان ، ھەممە كىشىگە مەلۇم بولغان قىزىللىق ئەسلىدە قان تۇمۇرلارنىڭ پالەچ بۇلشىدىن باشقا نەرسە ئەمەس .
                                                                           - گ . بۇڭكى

داۋاملىق ئىچىش زىيانلىق ، ئۇ جىگەر ۋە مىڭە پائاليىتىنى بۇزىدۇ ، نېرۋىنى ئاجىزلاشتۇرۇپ ، نېرۋا كېسىلى ۋە ۋاقىتسىز ئۆلۈمگە ئېلىپ كېلىدۇ .
                                                                            - ئىبىن سىنا

ئىچىملىك بەدەن ئەزالىرىنى يىمىرىپ ، ئۇنىڭ يۇقۇملۇق كىسەللەرگە بەرداشلىق بىرىش قۇۋۋىتىنى تۈگىتىدۇ .
                                                                          - ئا . بودرىراللار

ئىچىملىك ئىنسان تەن ساغلاملىقىغا پەقەت بەدەننى زەھەرلەپ زىيان يەتكۈزۈش بىلەنلا قالمايدۇ ؛ ئۇ ھاراقكەش ئادەمنى باشقا ھەر تۈرلۈك نوقسانلارغا مايىل قىلىپ قۇيىدۇ .
                                                                            - ن . ئا . سېماشكو

ئىچمەيدىغانلارغا نىسبەتەن ئىچەرمەنلەردە ھەر خىل كىسەللىكلەر ئېغىر ۋە خەتەرلىك بۇلىدۇ .
                                                                                 - ك . م . بېر

ئادەم ئۈچۈن ۋابا قورقۇنۇچلۇق ، ئەمما ئىچىملىك ئۇنىڭدىنمۇ يامان .
                                                                                 - ئو بالزاك

ھەممە يۇقۇملىق كېسەللەرنى قۇشۇپ ھىساپلىغاندىمۇ ئىنسانىيەتنىڭ ئىچىملىكتىن نابۇت بولغان قىسىمى كۆپنى تەشكىل قىلىدۇ .
                                                                                - ژ . روشار

ھاراقكەشلىك تارىخى مۇسىبەتلىك تارىختۇر . ئۇ ئاچلىق ، ۋابا ۋە ئۇرۇشقا قارىغاندىمۇ كۆپرەك ۋەيرازچىلىق ئېلىپ كىلىدۇ .
                                                                                - ئو . كىلادستون

ئىچىملىك ئىستېمال قىلىش - يامان ئۆسمە كىسەللىكلىرىنى كەلتۈرۈپ چىقىرىشنىڭ بىر سەۋەبى بۇلۇشمۇ مۆمكىن .
                                                                                 - ئا . ئا . باگومالېتس

بۇتۇلكامدا مەن دەھشەتلىك مەنزىرىنى كۆرىمەن ، ئۇ بۇتۇلكا ئىچىدىكى نەرسىدىن كىلىپ چىقىدۇ . كۆز ئالدىمدا ئىلمىي تەبىئەتشۇناسلىق مۇزېيدىكى بۇتۇلكىدا خېلمۇ خېل چايانلار ، يىلانلار ۋە يىرىڭلانغان تاماق بەزلىرىنى كۆرۈپ تۇرغانغا ئوخشايمەن .
                                                                              - گ . گېنىي

ھاراقكەشلىك ئادەمزات ئىستىقبالىنىڭ ئەڭ ئەشەددى ەۈشمىنى .
                                                                              - ئا . فورېل

ھاراقكەشلىك كىشى ئۆلگەن بىلەن تۈگىمەيدۇ ؛ ئۇ ھەددى-ھىسابسىز ۋە تۈرلۈك تۈمەن يوللار ئارقىلىق ئەۋلاتلارغا ئۆتىۋېرىدۇ .
                                                                             - ئا . فورنې

بالىلاردىكى نىمجانلىق ۋە كىسەلمەنلىكنىڭ سەھەبى ھاراقكەشلىك .
                                                                              - گېپوكرات

ھاراقكەشلەر ئۆز ئۇرۇق-ئەۋلاتلىرىغا تۈرلۈك-تۈرلۈك ھەسرەتلەرنى  ئىنئام قىلىپ قالدۇرىدۇ ، ئۇلار ئىچىدىكى ئەڭ يامىنى رۇھى كىسەللىك .
                                                                               - ۋ . يا . كانېل

نۇرغۇنلىغان تەنتەك ، ئەقلىي ناچارلار ئىچىملىككە بىرىلگەن ئاتا-ئانىلاردىن تۇغىلىدۇ .
                                                                               - ف . بېكون

ئادەمنى تەنتەك قىلىمەن دېسەڭ ، ئۇنىڭغا ھەمىشە ئىچىملىك ئىچۈرۈشتىن باشقا ئۈنۈملۈك يول يوق .
                                                                                 - ئە . كرېپېلىن

قۇلى تىترەيدىغان خىرە كۆزى بىلەن ئاران كۆردىغان ، ياخشى ئاڭلىيالمايدىغان ، دېمى سىقىلىشتىن ھاسىرايدىغان ھاراقكەشنىڭ قىلغان .ىشى نىمە بۇلەتتى .
                                                                               - ن . ئا . سېماشكو

ھاراقكەشلىك - بۇزۇقلۇق پۇراپ تۇرغۇچى قىمارخانا ، پارىخورلۇققا ، قەرىزدارلىققا ، چىدىغىلى بولمايدىغان جىنسىي باشباشتاقلىققا ۋە زوراۋانلىققا يول كۆرسىتىپ بەرگۈچى يىشىل چىراغتۇر .
                                                                               - ئى . م . يارۇسلاۋىسكىي

ھاراقكەشلىك ۋە جىنايەت ئىجىتمائىي ھاياتنىڭ بىر-بىرى بىلەن چىڭ باغلانغان ئىككى ھادىسىدۇر .
                                                                               - ئى . ب . مېرژېۋسكىي

ئىچىملىك ئادەمنى قاتىل ۋە ئوغرىغا ئايلاندۇرىدۇ .
                                                                               - ك . كولتون

ئىچىملىك تۈرمىنى توشقۇزۇپ تۇرغۇچى خىزمەتچى .
                                                                              - بورىرللار

مەسىت ئادەم ساق ۋاقتىدا ئۇخلىسىمۇ چۈشىگە كىرمەيدىغان يامان ئىشلارنى قىلىشقا قادىردۇر .
                                                                              - ژ . لوندون

ئىنسانىيەتكە داغ بۇلۇپ قالىدىغان جىنەيەتلەرنىڭ ئوندىن توققۇزى ئىچىملىك بېسىمى بىلەن قىلىدۇ .
                                                                              - ل . ن . تولىستوي

ئىچكەن ئادەمنىڭ ھەممىسى ھاراقكەش ئەمەس ، ئەمما ئىچىملىكنىڭ خۇسۇسيىتى شۇنىڭدىكى ، ئۇنى ئىچىشكە باشلىغان ئادەم ئاسانلا ھاراقكەشكە ئايلىنىپ قالىدۇ .
                                                                              - پ . ئى . كوۋالېۋسكىي

ئەگەر ئادەملەر ئىچىملىكتىن ئەقىلنى يۇقىتىپ ، ئۆزلىرىنى زەھەرلىمگەندە ئىدى ، ھاياتىمىزدا قانچىلىك تاتلىق ئۆزگىرىشلەر بۇلۇشىنى تەسەۋۋۇر قىلىشمۇ تەس بولمىغان بۇلاتتى .
                                                                               - پ . فىرانكىلىن

مەن كېچىسى باش ئاغرىقىنى كەلتۈرۈپ چىقىردىغان ھاراققا بېرىلىپ ، كۈندۈزى ئىڭراپ يۈردىغان ئائادەملەرگە ئوخشاپ قېلىشنى ئەسلا خالىمايمەن .
                                                                              - ئى . گيوتى

مەسىت ئادەمدىن قانداق قىلىپ ئىچىملىكنى قانداق قىلىپ ئىچىملىكنى تاشلاش يۇلىنى سورىماقچىمەن ؟ ئۇنىڭ ئۈچۈن مەن جاۋاپ بىرىمەن : مەسىتلىكتە قىلغان ئىشلىرىنى كۆپرەك ئويلاپ قويسۇن .
                                                                              - پىغاگور

ئادەم ئىچىشكە ئىمكانىيەت بولماي قالغان ئەمەس ، بەلكى ئۆيدە ، ئۇدۇلىدا ئىچىملىك تۇرغان ، ئۇنىڭ پۇرقىنى سەزگەن ۋە ئازىراقلا قولىنى سوزسا ئېلىش مۆمكىن بولغان ئەھۋالدا ئۇنى ئىچمىسىلا ئىچىملىكنى تاشلىيالايدۇ .
                                                                              - ل . ن . تولىستوي

ئادەم ئىچىش ۋە چەكىشنى تاشلىسا ھەممە ھايات ھادىسىلىرىنى ئۇنىڭغا يېڭىدىن ئېچىپ بەرگۈچى ئەقلىي تىنىقلىق ۋە ۋەزمىن دۇنيا قاراشقا ئىگە بۇلىدۇ .
                                                                          - ل . ن . تولىستوي

مەن چوڭ بولغىنىمدىن كىيىن ھېچقاچان ئۆزۈمدە چېكىشكە ئېھتىياج سەزمىدىم ۋە بۇ ئادەتنى ھەمىشە ۋەھشىيلىك ، پاسكىنىلىق ۋە بەكمۇ زىيانلىق ، دەپ بىلدىم .
                                                                          - م گەندى

ھەر - بىر چەككۈچى شۇنى بىلىش ۋە ئىسىدە تۇتۇش كىرەككى ، ئۇ بىر ئۆزىنىلا ئەمەس ، باشقىلارنىمۇ زەھەرلەندۈرەلەيدۇ .
                                                                          - ن . ئا . سېماشكو

< چىكىش ۋە ئىچىشنى تاشلاش بەك قىيىن > دىگەن تەسەۋۋۇر خاتا . بۇ پىكىرگە ئىشەنمەسلىك كىرەك .
                                                                          - ل . ن . تولستوي

غەم - قايغۇنى ئىچىملىكتىن باشقا ھەر خىل يوللار بىلەنمۇ بىر تەرەپ قىلىش مۆمكىن .
                                                                          - س . جونسون
