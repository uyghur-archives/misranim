مەيلى تۇرىۋەرسۇن ، بەك كۆزىڭىزگە سەت كۆرۈنۈپ كەتكەن بولسا   شۇ ئاچچىغىڭىزدا ئىلاننى تۆت -بەشنى چېكىۋېتىڭە  يىنىكلەپ قالىسىز ....

بىز نۇرغۇنلىغان يېڭى ھەم ياخشى ئۇچۇر،ئەتراپىمىزدىكى ئىشلار،ياخشى تىمىلار ۋە نادىر ئەسەرلەردىن مىسرانىم ئارقىلىق بەھىر ئېلىۋاتىمىز،شۇنداق بولغان ئىكەن سىز<<تىما ئاستىدا كۆرۈلىدىغان ئىمزانىڭ ئۈستىدىكى ئېلانلارنى چىقىرىپ تاشلىۋەتسە قانداق بولا؟>>دىگەن گەپنى قىلماي،<<تىما ئاستىدا كۆرۈلىدىغان ئىمزانىڭ ئۈستىدىكى ئېلانلارنى كۈندە بىر قېتىمدىن چىكىپ قويساق قانداق ياخشى بولار ئىدى؟>>دەپ يازغان بولسىڭىز نەقەدەر ياخشى بولاتتى...
كۆپلىگەن ئۇيغۇر توربەت ئىگىلىرى گوگىل،بەيدۇ...قاتارلىق ئېلان تەمىنلەش سۇپىسىنىڭ ئېلانلىرى ئارقىلىق ئازدۇر-كۆپتۇر كىرىم قىلىپ،يەنە بىز تورداشلار ئۈچۈن جاندىل بىلەن خىزمەت قىلىۋاتىدۇ،شۇنىڭ ئۈچۈن مۇنبەردە كۆرۈنگەن خەنزۇچە ياكى ئىنگلىزچە خەتلىك ئېلانلارنى كۈندە بىر قېتىم چىقىپ قۇيۇشنى ئۆزىمىزگە ئادەت قىلىۋالساق،بۇ مىسرانىمنىڭ تېخىمۇ تەرەققى قىلىشىنى قىلغان زەررىچىلىك ئىشىمىز بولغۇسى...ئۆز-ئارا قوللاش دىگەن بولسۇن بىزدە،بېكەت باشلىقلىرى بىكارغا قۇيۇپ قويغان ئېلان ئەمەس ئۇ!

ئەجىبا سىزنىڭ<<تىما ئاستىدا كۆرۈلىدىغان ئىمزانىڭ ئۈستىدىكى ئېلانلارنى چىقىرىپ تاشلىۋەتسە قانداق بولا>>دىگەن گەپلىرىڭىز مۇنبەر باشقۇرغۇچىلىرىنى،بېكەت باشلىقىنى بىئارام قىلماسمۇ؟كۆڭلى ئازار يىمەسمۇ؟شۇ ئېلانلارنى نىمە ئۈچۈن قۇيۇپ قويغىنىمىزنى بىلمەيدىغانلار،چۈشەنمەيدىغانلار بار ئىكەن دەپ...
ئەگەر سىز ئاشۇ خەنزۇچە ياكى ئىنگلىزچە خەتلىك ئېلانلارنى نىمىشقا قويغىنىنى بىلمىگەن بولسىڭىز،<< تىما ئاستىدا كۆرۈلىدىغان ئىمزانىڭ ئۈستىدىكى ئېلانلارنى چىقىرىپ تاشلىۋەتسە قانداق بولا>>دىگەن گەپنى قىلماي،<<تىما ئاستىدا كۆرۈلىدىغان ئىمزانىڭ ئۈستىدىكى ئېلانلارنى نىمىشقا قۇيدۇڭلار؟ئۇنىڭ نىمە پايدىسى بار؟>>دەپ سورىسىڭىز بېكەت باشلىقى ياكى باشقۇرغۇچىلار ۋە ياكى بىلىدىغان باشقا ئەزالار جاۋاب بىرەتتى قېرىندىشىم.

بولدى قېرىندىشىم،بۇ يەردە بىر-بىرىمىزگە تەنە قىلىشىپ يۈرمەيلى،مەن پەقەت بىر-بىرىمىزنى چۈشەنسەككەن،بىر-بىرىمىزنى قوللىساقكەن دىگەن ئۈمىدتىمەن!
ئەسسالامۇ ئەلەيكۇم ئەزىزلەر، كونا يېڭى مۇنبەرداشلار...ئۆزئارا پىكىر ئاملاشتۇرۇشلار بولسا ئۈندىدارغا قۇشىۋالساڭلار بولىدۇ نۇمۇرۇم:295997753
ئەسكەرتىش: قۇشىۋالغاندا كىم ئىكەنلىكىڭىزنى ئەسكەرتىپ قويۇڭ!
