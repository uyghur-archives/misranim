  تولۇق ئوتتۇرنىڭ 3 - يىللىقىغا چىقاندىن ياكى ئاندىن  چ چ دىگەن نەرسىنىڭ نىمىلىكىنى بىلگەن ئىدىم ، ئەينى چاغدا ئۆيىمىزدە كومپىيوتۇر بولسىمۇ تورغا ئۇلىمىغان ئىدى ، كېيىن ئاكامنىڭ چىڭ تۇرشى بىلەن دادام ئاخىرى تورنى ئېچىشقا قوشۇلدى ، شۇنداق قىلىپ تورنىمۇ ئاچتى ...دەسلەبتە مەن پەقەتلا توربەتلىرىدىن ناخشا ، ئىتوت . پەقەت بولمىغاندا كىنو كۆرەتتىم ، كېيىن  دوسلىرىمنى  ھەممىسىنىڭ چ چ نۇمۇرى بارلىقىنى بىلىپ مەنمۇ ئاكامغا نۇمۇر ئېلىپ بېرىشنى ئۆتۈندۈم . مەن دىگەن كۈننىڭ ئەتىسىلا ئاكام ماڭا بىر يېڭى نۇمۇر بەردى ، مەن خۇشال بولۇپ ئاكامغا رەھمەت ئېتسام ، ئاكام ماڭا كەسكىن تەلەپپۇزدا « ئۇكام ، بۇ نەرسىگە قىزىقىپ قالما ....پەقەت ئويۇنلا خالاس ، ئەسلىدە بۇ نۇمۇرنى بەرگۈم يوق ئىدى ، لىكىن بوپتۇ ....بەزىدە ئىچىڭ پۇشۇپ قالغاندا دوسلىرىڭ بىلەن تېلفوندا پاراڭلاشمىساڭمۇ چ چ دا پاراڭلىشىپ ئولتۇراسەن .» دېدى ....مەن بېشىمنى لىڭشىتىپ ئۇنىڭغا كۈلدۈم ، ئەتىسىلا مەكتەپكە بېرىپ دوسلىرىمنىڭ ھەممىسىنىڭ چ چ نۇمۇرىنى يېزىپ كىلىپ چ چ  رىمغا قاچىلىدىم ، دەسلەپتە بەك ئاچمىدىم ، 2004- يىلى يېڭى يىلنىڭ 1 - كۈنى  ئىدى ،شۇ كەچتە ئىچىم پۇشۇپ ، ئاچتىم ... قارىسام بىر قانچە دوسلىرىم باركەن ، شۇنىڭ بىلەن ئۇلارغا بىلەن مۇڭدىىشقا چۈشتۈم ، بىر چاغ بولغاندا چ چ رىمغا ئۇچۇر كەلدى ، ئېچىپ قارىسام مېنى دوسقا قوشۇش ئۇچۇرى ئىدى ، بۇ تۇنجى قېتىم مېنى باشقىلار دوسقا قوشۇشى بولغاچقا ، مەنمۇ ئۇ چ چ نۇمۇرىنى قوشتۇم ... شۇنىڭ بىلەن بۇ ناتونۇش چ چ نۇمۇر ئىگىسى بىلەن پاراڭغا چۈشۈپ كەتتىم ، ( ھازىر بۇ ئىشلارنى ئويلىسام كۈلگۈم كىلىدۇ ) باشتا بۇ چ چ ئگىسىنىڭ توردىكى ئىسمىغا قاراپ قىز  ئوخشايدۇ دەپ ئويلاپ تۇرسام ، ئىسمىنى دىگەندىلا ئوغۇل بالا ئىكەنلىكىنى بىلىپ نىمىشقا قىز بالىدەك ئىسىم قويۋالدىڭىز دەپ سورىسام كېيىن بىلىپ قالىسىز دېدى ...شۇنىڭ بىلەن ئۇ بالا بىلەن پاراڭلىشىش باشلىنىپ كەتتى ، پاراڭلىشىش ئارقىلىق ئۇ بالىنىڭمۇ مەن بىلەن تەڭ دىمەتلىك ئىكەنلىكىنى ، بۇ يىل ئالىي مەكتەپ ئىمتىھانى بېرىدىغانلىقىنى بىلدىم ...تۇنجى كۈنىلا ئىككى - ئۈچ سائەت پاراڭلىشىپ كەتتۇق ، مەن تۇنجى قېتىم توردا پاراڭلىشىشىم بولغاچقىمۇ توختىماي سوئال سورايتىم ، ئۇ بالىمۇ زېرىكمەي جاۋاپ بېرەتتى ، ئۇ بالا مەن ئەمدى ماڭاي دىگەندىلا ئاندىن  سوئال سوراشنى توختاتتىم ۋە خوشلىشىپ چ چنى ئۆچۈردۈم ... ئۆچۈرۈپ بولۇپ ئۆزۈمگە ھەيران قالدىم ، ئادەتتە باشقىلار بىلەن پاراڭلىشىپ كەتمەيدىغان مەن توردا ھەجەپ سېغىزخان بولۇپ كەتتىمغۇ ......

ھاممامنىڭ ئۆيىگە كەتتى ، ماڭا بېرىس ىۇش ياقماي ئۆزۈم  يالغۇز قالدىم ، شۇنىڭ بىلەن دەرھال كومپىيوتۇرنى قوزغىتىپ ، چ چ رىمنى ئاچتىم ، قارىسام ئۇ باركەن ، ئىچىمدىن بىر خىل خۇشاللىق يايراۋاتاتتى ۋە ئۇنىڭغا ئۇچۇر يېزىشقا باشلىدىم ، ۋە يەنە پاراڭغا چۈشۈپ كەتتىم ، لىكىن ئۇ بۈگۈن كۆڭلىنىڭ يېرىم ئىكەنلىكىنى دىدى ، مەن نىمە بولغانلىقىنى سورىسام ئۇ بۇھەببىتىدىن ئايرىلىپ كەتكەنلىكىنى ئېتىپ دەدىنى تۆككىلى تۇردى ، راستىمنى دىسەم مەن شۇ چاغلاردا نىمىنىڭ ياخشى كۆرۈش ، مۇھەببەتنىڭ نىمىلىكىنىمۇ بىلمەيتتىم ، ۋە ئۇنىڭغا :« كىچىك تۇرۇپ مۇھەببەتلىشىپ نىمە قىلارسىز ، يا چۈشەنمىسىڭىز » دەپ ئۇچۇر يازغىنىمدا ئۇ كۈلگەنلىك ئىپادىسى بىلەن  ماڭا سىز بەك نادانمۇ نىمە ؟ دەپ ئۇچۇر قايتۇردى ، بۇ گەپنى ئوقۇپ ئىچىم بىر خىل بولۇپ قالدى ، شۇ تەرىقىدە ئۇ جىددىي ئىشى چىقىپ قالغانلىقىنى ئېتىپ مەن بىلەن خوشلىشىپ چۈشۈپ كەتتى ، شۇنىڭ بىلەن ئۇ تەتىل تۈگىگىچە تورغا چىقىپ باقمىدى ، تولۇق ئوتتۇرنىڭ ئاخىرقى ئوقۇش مەۋسۈمى باشلاندى ، مەنمۇ ۋاقىتنى چىڭ تۇتۇپ ئۆگىنىش قىلىشقا كىرىشتىم ، تورغا چىقىشقا ۋاقىت چىقمايتتى ، بەزى شەنبە كۈنلىرى چ چ رىمنى ئېچىپ بىراۋنى ئىزدەيتتىم ، ئىمتىھانغا بىر ئاي قالغاننىڭ بىر شەنبە كغنى مەن ئادىتىم بويىچە چ چ رىمنى ئاچتىم ، قارىسام ئۇ باركەن ، ئەمدىلا ئۇچۇر يازاپ دەپ تۇرىشىمغا ئۇنىڭدىن ئەھۋال سوراش ئۇچۇرى كەلدى ، مەنمۇ جاۋابەن ئۇچۇر قايتۇردۇم ، پارڭلاشمىغىلى ئۇزۇن بولغاچقىمۇ مۇڭدىشىپلا كەتتۇق ، مەن ئاخىردا ئۇنىڭ نەدە ئوقۇغۇسى بارلىقىنى سورىدىم ، ئۇ ماڭا نەگە ئۆتسە شۇ يەردە ئوقۇيدىغانلىقىنى دېدى ، مەندىن بۇ سوئالنى سورىغاندا مەنمۇ ئۇنىڭغا ئوخشاشلا جاۋاب بەردىم ، شۇ كۈنى بىز كەچ 4-5 سائەت پاراڭلاشتۇق ، ئۇ ماڭا ئەمدى ئېمتىھاندىن كېيىن كىردىغانلىقىنى ئېتىپ ۋە بۈگۈن مۇشۇ ئىشنى ماڭا ئېتىپ قويۇش ئۈچۈن كىرگەنلىكىنى دەپ خوشلىشىپ چىقىپ كەتتى ، مەنمۇ چ چ نى ئۆچۈرۈپ چۈشتۈم ، لىكىن كۆڭلۈمدە بىر خىل ىۇشاللىق چىرمىۋالغان ئىدى ...بۇ قېتىمقى پاراڭلىشىشتىن مەن خېلىلا خۇش بولغان ئىدىم ، شۇنىڭ بىلەن مەنمۇ ئىمتىھان دەپلا ئۆگنىش بىلەنلا ئالدىراش يۈردۈم ، چ چ نىمۇ ئېچىپ باقمىدىم ، ئىمتىھان كۈنىمۇ ئاخىرى كەلدى ، مەنمۇ جەڭ مەيدانىغا كىردىم ، ئىككى كۈنلۈك جەڭگاھتىن ئاخىرى چىقتىم ، لىكىن ئۆزۈمگە ئىشەنچىم ئەزەلدىنلا تۆۋەن بولغاچقا ، ئىچكىردىكى ئۆزۈم ئارزۇ قىلغان مەكتەپنى يازماي ، ئارزۇيۇمغا  شىنجاڭ ئۇۋىرسىتى بىلەن پىداگوكا ئۇنۋېرسىتىنى يېزىپ قويۇپلا بولدى قىلدىم ، باشقا مەكتەپنىمۇ يازمىدىم بۇنىڭغا  داداممۇ بىر نىمە
دىمىدى ، شۇنىڭ بىلەن ئۆيدە ئىمتىھان نۇمۇرىنى كۈتۈش بىلەن ئولتۇردۇم ، شۇ كۈنلەردە نىمىشقىكىن چ چ نىمۇ ئېچىپ باقمىدىم ، قىلدىغان ئىشىم ئۆي يۇغۇشتۇرۇش ، ئاپامدىن تاماق ئېتىشنى ئۆگىنىش بولدى .........نەتىجە چىقىدىغان كۈنمۇ ئاخىرى يىتىپ كەلدى ، باشقا دوسلىرىم ماڭا تېلفون قىلىپ ئۆزىنىڭ قانچە نۇمۇر ئالغانلىقىنى ئېيتتى ،مەنمۇ غەيرەتكە كىلىپ نۇمۇرۇمنى سۈرۈشتۈردۈم ، ۋە نۇمۇرۇمنى ئاڭلاپ ىۇش بولدۇم  لىكىن يۈرۈكۈنمىڭ بىر يەرلىرى ئېچىشتى ، ئىچكىردىكى ئۆزۈم ئارزۇ قىلغان مەكتەپنى يازمىغانلىقىمغا ئۆكۈندۈم ، ئەمدى ھەرقانچە قىلغان بىلەن ئىشلار ئورنىغا كەلمەيدۇ ، ئامال يوق چاقىرىق قەغىزى كۈتتۈم ، شۇ كۈنلەردە يەنە 2 ئاي ياقى ئېچىپ باقمىغان چ چ نۇمۇرۇمنى ئاچتىم ، چ چ نى ئېچىپ بولغۇچە بىر قانچە ئۇچۇر كەلدى ، ئۇچۇر نى ئۇ يازغان ئىكەن ... بىز ئامان - مۇشۇ چ چ دىكى مۇھەببەتكە مەنمۇ ئىشەنمەيتتىم،  ئەمما يېقىندىن بېرى ئىشىنىدىغان بولۇپ قالدىم ،جۇمۇ، بىر دوستۇم   چ چ  دا پاراڭلاشقىلى بىر ئاي بولغان قىزغا كۆيۈپ   ،  كېچىسى ئۇھ تارتىپ يۈرۈيدۇ  ياتاقتا، ..     لېكىن ئۇ قىز  دوستۇمدەك  ئوغۇللاردىن نەچچىسىگە ماقۇل دەپتۇ ،   شۇنداقتىمۇ  ئۇ دوستۇم ئۇ قىزدىن ۋاز كېچەلمەيۋاتىدۇ.

  سىزنىڭ مۇھەببەت قىسسىڭزنى تەپسىلى ئوقۇپ چىقتىم. ھازىرقى ئۇنىۋېرستىتىكى مۇھەببەت راسلا بىر- بىرىگە سائەتلەپ  ۋ   ۋاڭ  پاراڭلىشىدۇ.  تاماقتا، بازادا، ،،،،يەنە ھەممىدە بىرگە  ،  تاس قالىدۇ  ھەجەتخانىغىمۇ بىرگە كىرگىلى.  ئەمما  سىزگە ئوخشاش گۈزەل ھەم ساپ مۇھەببەت  ئىزناسىنى قالدۇرغانلار تولىمۇ ئاز دېمەتلىك.
   مەنچە ھازىرقى  مۇھەتتىكى يۈزلىنىش    بىر- بىرىنىڭ ھاجىتىدىن چىقىش بولۇپ قېلىۋاتىدۇ.    زىنا  كىشىلەر ئۈچۈن ئادەتتىكى ئىشقا ئايلىنىپ كېتىۋاتىدۇ. ئېتىقادسىز مىللەت كەلگۈسىدە گۇمران بولىدۇ. ئەخلاق كىرزىسى ئىچىدە ئۆزىنى ئۆزى گۇمران قىلىدۇ.   

   چ چ  دا سائەتلەپ مۇھەببەت ئىزدەيدىغان،  چ چ  دا  قىز كەلتۈرۈش، يېگىت كەلتۈرۈش ئۈچۈن ئاۋارە يۈرگەن  < مەجنۇن > لاردىن شۇنى سەمىمىي ئۈمىد قىلىمەنكى،  بەخت ئىزدىمەڭ، ئۇ يېنىڭزدا،  ئەتراپىڭىزدا.  دىققەت قىلىڭ. چ چ دىكى تونۇشقان قىز دوستىڭىز ( مۇھەببىتىڭىز )  سىزنىڭ  قىزىڭىز،     يېگىتىڭز   سىزنىڭ دادىڭىز   چىقىپ قالمىسۇن!؟
