كۆپچىلىك ئوبدان تۇرۇپسىلەرمۇ؟ يەنە مەن  
بۈگۈن مەن قاخشىمايمەن دەپ كىرگەنىدىم ، ئەمما يەنە سەل قاخشاپ قويۇپ ، گىپىمنىڭ پوسكاللىسىنى قىلمىسام ، چۈشىنىكسىز بولۇپ قالىدىغاندەك تۇرىدۇ! دەۋالمىسام پەقەت چىدىمىدىم!!!!

مۇنداقلا دەي! ماۋۇ مۇنبەرمۇ نىمە؟ پاخۇي!!!!! مۇشۇ كۈنلەردە بىزنىڭ بۇ مۇنبەر دىگەندە ئىنكاس يازغىدەك تېما پەقەتلا كۆرۈنمىدىغۇ؟! نەگىلا قارىسا، < مۇھەببەت...> ، <قىزلىرىمىزگە نىمە بولدى؟>  ۋاي تېخى< چ چ دىكى مۇھەببەت> < قاچان مىنى يوقلاپ كىلىسەن> دىسە يەنە بىر تۇزىكام چىقىپ < ئەتە سېنى يوقلاپ بارىمەن> دەپ شېئىرلارچۇ تېخى.... < مۇھەببەت دىگەن نىمە؟> < مىنىڭ سۆيگۈنۈم ئاسماندىن چۈشەمدۇ؟ >  < يىگىتلەر ماۋۇ  ئىشلارنى قىلالامسىلەر؟> ...ۋاي...ۋاي..ۋاي....پۈتۈنلەي  قاپىيەگە زورلانغان رەتلىك يىزىلغان تۆت جۈملە سۆز!!!!!ۋوي تېخى ، 2 كۈن بىر يەرگە بېرىپ كىلىشى ئۈچۈنمۇ تېما يوللاپ قويۇشلىرىچۇ تېخى، كەلمەسكە كىتىدىغاندەك...بىر گەپ قىلاي دىۋىدىم......
بەلكىم دىيىشڭلار مۇمكىن، قىنى ئۆزىڭىز بىرنى يىزىپ بېقىڭە ؟ دەپ، گەرچە ئورۇندۇق ياسىيالمىغان ئادەم، ئورۇندۇقنىڭ قۇرۇلمىسىنىڭ ياخشى - يامانلىقى، سۈپىتىنىڭ ياخشى - ناچارلىقىنى پەرق ئىتەلەيدۇ!!!

مەن ھەرگىزمۇ بۇنى ئەزالاردىن كۆرمەيمەن، ئىختىيارى، قانداق تېما يازسا ئۆزىنىڭ ئىشى، يەنە قىزلارنى تىللامدۇ؟ مىللەتنى سۆكەمدۇ؟ < پەرۋانەڭ بۇلاي، پاشا - كۇمۇتاڭ بۇلاي> دەمدۇ؟! مەيلى ئۆزىنىڭ ئىشى! مۇشۇنداق تېمىلار يىزىلىۋەرسە، تەستىقلىنىپ مۇنبەرگە چىقىۋەرسە ، نىمە بولماقچى بۇ مۇنبەر؟؟؟؟

ئەزالارنىڭ قىزغىنلىقىغا سوغاق سۇ سەپكىنىم ئەمەس،  مەنمۇ مۇنبەرنىڭ ساپاسى يۇقۇرى كۆتۈرۈلسىكەن دىگەن ئۈمۈدتە، مەنلا ئەمەس ھەممەيلەن شۇ ئۈمىدتە شۇنداققۇ؟!

مۇنبەر دىگەنگە تىزىملاتماي تۇرۇپ ئوقىغان ئادەمنىڭ كىرگۈسى كەلگۈدەك بولسا، "توختىغىنا ، مەن بىر ئىنكاس يازايچۇ؟ "دىگىدەك قىزىقىپ ، تىزىملىتىپ تۇرسا، مانا بۇ مۇنبەرنىڭ تارتىش كۈچىنىڭ ئاشقانلىقى، ساپاسىنىڭ يۇقۇرلىقى....

خوش، ئەمدى كىمدىن كۆرىمىز؟!  مەن بۇنى مىسرانىم بىلەن باشقۇرغۇچىلاردىن كۆرىمەن !! ئەسلىدىنلا سۇنىڭ بىشى لاي!
دەپ قوياي ، باشقۇرغۇچى يولداشلار:
مۇنبەرنىڭ ساپاسىنىڭ يۇقۇرى- تۆۋەنلىكى سىزگە باغلىق جۇمۇ؟! سىزمۇ بار مىسرانىم ئەپەندى،مۇنبەر باشلىقى  باشقۇرغۇچىلارنى چىڭ تۇتمىسا، باشقۇرغۇچىلار ئەزالارنىڭ تېمىسىنى چىڭ تۇتمىسا، مۇنبەر ئەخلەتخانىغا ئايلانماي نىمە بولىدۇ؟! گەپ مۇشۇ باشلىقلاردا!!!!!
ئاددى مىساللارنى مەن ئىلىپ بېرەي :
كۈنىگە نۇرغۇن تېما يوللىنىدۇ، بۇنىڭ ئىچىدە ئەھمىيەتلىكىنى تاللاپ ئاز  ھەم ساز تېمىنى تەستىقلايسەن، تېمىسى تەستىقلانمىغان ئەزاغا ئۇچۇر قىلىپ ئەستايىدىل ، تۆۋەنچىلىك بىلەن چۈشەندۈرۈپ ئۆتىسەن!  بۇ يەردىكى مەسىلە دەل سەندە، سېنى سۆزلەش قابىلىيىتى بار دەپ باشقۇرغۇچى سايلىغان ئىكەن بۇنى سەن  قىلالايسەن،  شۇڭا تىتىقسىز تېما يوللىغان ئەزاغا چۈشەندۇرسەڭ ئۇمۇ ئادەم ، چۈشىنىدۇ،
يەنە جاھىللىق قىلىپ تۇرۇۋالسا، مۇنبەر باشلىقىغا قويۇپ قوي ، مۇنبەر باشلىقى چۈشەندۈرىدۇ، قەستەن جىدەل چىقارغۇسى بولسا، مىسرانىم ئەپەندىنڭ ئۆز ئالدىغا ئامالى بار، ئۇنداقلادىن قورقۇپ تەستىقلاۋەرسەڭ، مۇنبەر مانا بۈگۈنكىدەك ھالغا چۈشۈپ قالىدۇ!

مىسرانىم ئەپەندى: قۇلاقنى دىڭ تۇتۇپ ئاڭلاڭ:
باشقۇرغۇچىلارغا ھەر ۋاقىت يىغىن خاراكتېرلىق تېمىلارنى يوللاپ، (تەستىقلانمايدۇ، تەھرىرخانىدا باشقۇرغۇچىلار كۆرسىلا بولدى)  باشقۇرغۇچىلارنىڭ قۇلىقىغا قويۇپ تۇرۇش كېرەك، ئەگەر مىنىڭ ئۇنچىلىك ۋاقتىم يوق دىسىڭىز، سۆز قابىلىيىتى، باشقۇرۇش سەنئىتى  يۇقۇرى بىرەرسىنى ئورنىڭىزغا دەسسىتىپ قويۇڭ! باشقۇرغۇچىلار گىپىنى ئاڭلىمىسا، كۆپىسىنى كۆتەگۈزۈپ قويۇڭ، بىر ئىككى ئادەم ئىشلىمىگەنگە توختاپ قالىدىغان مۇنبەر ئەمەس بۇ!  بىر نەچچە سىسىق ئالما ، بىر سىۋەت ئالمىنىڭ ھەممىنى سېسىتىۋەتمىسۇن!!!

بىلىمەن، مەنمۇ باشقۇرغۇچى ، بىكەت باشلىقى بولۇپ باققان، توغرا ،  باشقۇرغۇچى بولماق ئاسان ئەمەس، مۇنبەر باشلىقى بولماق تېخىمۇ ئاسان ئەمەس، ئەمدى بىز شۇ مۇنبەرنى  ياخشى بولسۇن دەپ سىلەرگە دەرت ئيتمىساق كىمگە ئېيتىمىز؟!
توغرا قۇبۇل قىلساڭلار مانا، بولمىسا ، ماڭا بىر مۇنبەر تۇنۇشتۇرۇپ بەرسەڭلار ، شۇ يەرگە چىقىپ كىتەي! چوڭلار كۆپرەك ، ئىسىل تېمىلار يوللىنىدىغان مۇنبەر بولسۇن ، قانداق؟

ئاچچىقىم كەلسە، مانا ئەمسە دەپ مەنمۇ يازىمەن، كىلەمدۇ كەلمەمدۇ؟! تۇزلۇقمۇ ، تۇزىكاممۇ ، تېما ئىنكاس دىگەننى تۆكىۋىتىمەن! تەستىقلىمىساڭلار ئۇرۇشىمەن ، تىللايمەن، باشقا ئىسىمدا كىرىپ يەنە تىللايمەن...

قېنى بۇ ئىنكاسنىمۇ كۆرۈپ ئۆتۈڭ !


ئەسسالام ! پولكوۋنىك ئەپەندى .
تېمىدا دادىل ئوتتۇرىغا قويغان پىكرىڭىزدىن ھەقىقەتەن سۈيۈندۈم .
      مانا بۈگۈن توردىشىمىز پولكوۋنىك بىز باشقۇرغۇچىلار دىسەك << ئەزالارنىڭ كۆڭلىگە كېلەرمۇ>> دىگەندەك ئەندىشىلەر بىلەن دىيەلمەيۋاتقان پىكىرنى ناھايىتى دادىللىق بىلەن ئوتتۇرىغا قويۇپتۇ ، بىز باشقۇرغۇچىلارنىمۇ خېلى تەنقىدلەپتۇ . مانا مۇشۇنداق پىكىرنى ھەقىقى قوللايمىز . بىللى قاخشايلى ، قارىسام ئوي - پىكرىمىز خېلى بىر يەردىن چىقىدىغاندەك قىلىدۇ .
       ئەسلەپ ئۆتسەك تور قايتا ئېچىلغاندىن كېيىن ، مىسرانىم مۇنبىرى بىرىنچى بۇلۇپ، باش كۆتۈرۈپ چىقىپ ، تېمىلىرىنىڭ سەرخىللىقى بىلەن كۆپلىگەن كىشىلەرنى ئۆزىگە تارتىپ ، مۇنبەرگە باشلاپ كىردى . دەسلەپتە خېلى كۆپ بىلىملىك زاتلىرىمىز ، ئىلغار پىكىرلىك ياشلىرىمىز سەرخىل تېمىلارنى يوللاپ ، ئىسىل پىكىرلىك ئىنكاسلارنى يېزىپ ، مۇنبەرگە بولغان رىشتىمىزنى تېخىمۇ چىڭ چىڭىتقان ئىدى . بىزمۇ بۇنىڭدىن ئىنتايىن خوشال ئىدۇق .
      دەرۋەقە تېمىڭىزدا ئېيتقاندەك ھازىر مۇنبەر كەيپىياتى ئادەمگە بىر ئاز بىزارلىق تۇيغۇسى بېرىدۇ . بۇ بىزارلىق زادى نىمىدىن كېلىپ چىققان ؟. دەرۋەقە سەرخىل  تېمىلارنىڭ ئازلىقى ، ھە دىسىلا خوتۇن - قىزلىرىمىزنى ئەركەكلىكىنى سۆكىدىغان تېمىلارنىڭ كۆپلىگى ، شېئىرلارنىڭ سۈپىتىنىڭ تۆۋەنلىكى ، ئىنكاسلاردىكى ئۇرۇنسىز تالاش تارتىشلار . بۇلار سىز ئېيتقاندەك ئادەمگە قىلچىلىك ھوزۇر ئاتا قىلالمايدۇ .
ئاشۇ بەزى  سۈپەتسىز تېمىلارنىڭ ، قالايمىقان ئىنكاسلارنىڭ مۇنبەر يۈزىدە ئورۇن ئېلىشىغا يول قويغان سەۋەنلىك ھەقىقەتەن مىسرانىمدا شۇنداقلا بىز باشقۇرغۇچىلاردا بار، بىز بۇنى چوقۇم تۈزىتىشكە تىرىشىمىز .
خوش ئەمدى ھەقىقى گەپنىڭ ئوزىگە كېلەيلى :  
      سەرخىل تېمىلار نىمىشقا شۇنچە ئاز ؟ يازالايدىغانلار يوقمۇ ؟   بۇرۇن يېزىلغان ئىسىل پىكرلىك ئىنكاسلار نىمىشقا يېزىلمايدۇ؟ . مېنىڭچە شۇنداق سەرخىل تېمىلارمۇ خېلى بار ، يازالايدىغانلارمۇ خېلى بار، تېتىملىق ، كىشىنى ئويلاندۇرغۇدەك ئىنكاس يازىدىغان ياشلىرىمىزمۇ خېلى كۆپ . ئەمما ئۇلار نىمىشقا يازمايدۇ ؟ سەۋەبى نىمە ؟ بۇنى ئويلىنىپ باقتۇقمۇ؟ قانچىلىكىمىز بۇ ھەقتە ئويلىنىپ باقتۇق قېنى؟ بۇلارنى يېزىشتىن باشقۇرغۇچىلار چەكلىدىما ؟
ياق . جاۋابىىم يەنىلا ياق .
بۇلارنى شۇ تېما يېزىشىتىن ، ئىنكاس يېزىشىشتىن ۋاز كەچتۈرۈۋاتقانمۇ شۇ بىر قىسىم << دانا پىكىرلىك ، ھەق سۆزلەيدىغان، قالتىس >> ئەزالىرىمىز ئەمەسمۇ ؟ شۇنداق ياخشى تېما ياكى ئىنكاس يازسا  تېمىنى بىر باشتىن ئوقۇمايلا مەلۈم بىر سۆزنى تۇتۇۋېلىپ تېما ، ياكى ئىنكاس ئىگىسىگە بولۇشىغا زەھىرىنى چاچسا ، ھاقارەتلىسە ، ئۇنىڭغا يەنە بىر توپ << شاتۇتى >> لار قۇشۇلۇپ  بەس- بەستە سايراپ كەتسە كىممۇ يەنە ئىنكاس يېزىشنى خالىسۇن . ھەممىمىزنىڭ ئىزدەيدىغىنى خوشاللىق ، شاتلىق تۇرسا ، <<بىكارلىق كۆڭۈل ئاغرىقى>> تارتىۋېلىشنى كىممۇ خالىسۇن .
      ئەنە شۇنداق بىلىملىك زاتلىرىمىزنى ، ئىلغار پىكىرلىك ياشلىرىمىزنى چوڭقۇر مەنىلىك تېما، ئۆتكۈر پىكىرلىك ئىنكاسلارنى يېزىشقا قايتا دەۋەت قىلىش زادى مۇمكىن بولماسمۇ؟
تامامەن بولىدۇ. مۇھىمى ئاشۇلارنىڭ تېمىسىغا مۇنبەردىن سورۇن ھازىرلاپ بەرسەك ، ئىنكاسلىرىمىزدا تېمىغا قارىتا ئىجابىي ۋە تەنقىدى پىكىرلەرنى يازساق ، ئۇلارغا مەدەت بەرسەك ئۇلارمۇ قايتىپ كېلىدۇ ، شۇنداقلا ئۆز تالانتىنى كۆرسىتىپ ، مۇنبەر ئۈچۈن ياخشى تېمىلارنى سوۋغا قىلىدۇ . شۇڭا بارلىق ئەزالار ئىنكاسلارنى تېما ئىگىسىگە ئەمەس ، تېما مەزمۇنىغا قارىتا كۆز قارىشىمىزنى ( شۇ ئەسەرنىڭ مۇۋاپپىقىيەتلىك چىققان تەرىپى ۋە ساقلانغان نوقسانلار ) ئەدەبى نوقتىدا ئوتتۇرىغا قويساق . قالايمىقان بىلجىرلاپ ، شۇلارنىڭ ئىجادىي قىزغىنلىقىغا سوغۇق سۇ سېپىدىغانلار ، تېما كۈرەمدۇ ، ياكى باشقا ئىش قىلامدۇ ، غىت قىسىۋېلىپ ، ئىلغار پىكىرلىك ياشلىرىمىزنىڭ ئىسىل تېمىلىرى ۋە ئىنكاسلىرىدىن ئۆزىمىزگە لايىق بىلىم ئېلىپ ، كېيىنكى ئىجادىيەت يولىمىزغا مەلۇم ئاساس سالساق . 
      مۇنبەردىكى تېمىلارنى تەستىقلاشتا ئىجادىي تېمىلارغا خېلى ئەھمىيەت بېرىۋاتىمىز . مۇنبەر يۈزىگە چىققان تېمىلارنى ئارقا بەتكە يوللانغان تېمىلار ئىچىدىن ئەڭ ياخشىلىرىنى تاللاپ چىقىرىۋاتىمىز. يەنە بەزى تورداشلىرىمىز شۇنداق ياخشى ، كىشىنى ئويلاندۇرغىدەك تېمىلارنى يوللاۋاتىدۇ .ئەمما مەنبەسى ، مەنبە ئۇلىنىش ئادرىسى يوق . بۇلارغا ئۇچۇر بەرسەكمۇ جاۋاپ ئالالمىدۇق . ئىنتايىن ئەپسۇسلۇق ئىچىدە ئۈچۈرۈشكە مەجبۇر بولدۇق .
      يەنە بىر قىسىم ئەزالار ئىجادىيەتكە يېڭى كىرىشكەن بولغاچقا، بەزى ئىجادى تېمىسىنى ( لاياقەتلىك دىيىشكە بولمىسىمۇ ، ھەر ھالدا تۇنجى قەدەمنى ياخشى باشلاپتۇ دىيىشكە بولىدۇ) ئىلھام بولسۇن ئۈچۈن تەستىقلىدۇق . ئەلۋەتتە بۇنىڭ نەتىجىسىنىمۇ كۆردۇق . بىر قىسىم ئەزالار دەسلەپتە ئاددى شېئىر( گەرچە تازا ياخشى دىگۈدەك بومىسىمۇ ) يېزىشتىن باشلىغان بولسا ھازىر خېلى سۈپەتلىك شېئىر ۋە ياخشى ھىكايە ۋە ماقالىلارنى يازالايدىغان بولدى ، بۇنىمۇ مۇنبەرنىڭ شۇلارغا يارىتىپ بەرگەن شارائىتىدىن ئايرىلىپ قارىغىلى بولمايدۇ .

قانداق قىلغاندا مۇنبەرنى ھەقىقى مۇنبەرگە ئوخشاتقىلى بولىدۇ؟
      بىزنىڭ يېتەكچى ئىدىيىمىز پارتىيە ۋە دۆلەتنىڭ سىياسەت، قانۇن ـ نىزاملىرى ئاساسىدا جامائەت پىكىرىنى توغرا يۆلۈنۈشكە يېتەكلەشتىن ئىبارەت. مىسرانىم مۇنبىرى ئونۋىرسال مۇنبەر بولغانلىقى ئۈچۈن چىقىدىغان تېمىلار كۆپ خىل ساھەگە چېتىلىدۇ . شۇڭا تېما ئۇستىدە ھازىرغىچە چەكلىمە قويۇلماي كەلگەن ، لېكىن بەزى تورداشلىرىمىز ئۆز ھېسسىياتىغا تايىنىپلا ، ھاياجىنىنى باسالماي ئىنكاس يېزىپ مۇنبەر بىخىەتەرلىكىگە تەھدىت ئېلىپ كەلدى ، يوقۇرى دەرىجىلىك ئورگانلارنىڭ ئۇقتۇرۇشى بىلەن بەزى سەھىپىگە  چەك قويۇلدى ، بۇنى ھەر بىر ئەزا توغرا چۈشۈنۈشى كېرەك .
      ئەزالىرىمىز يالغۇز شېئىرىيەت ، ھىكايە، پوۋىسىت ، باشقىلار ئۈستىدىن قاخشايدىغان ، دەرت تۈكىدىغان تېمىنىلا يېزىۋەرمەي  ئىقتىساد ، مەدەنىيەت ، ئورپ ئادەت، تۇرمۇش، نىكاھ ، ئائىلە، ساغلاملىق ، قانۇن قاتارلىق تەرەپلەردىنمۇ ئويلىنىپ ، بىز گە ھەقىقى نەپ بېرەلەيدىغان ، نادىر تېمىلارنى ۋۇجۇتقا چىقىرىشقا تىرىشساق ، ياكى كىتاپلاردىن ئەھمىيەتلىك تېمىلارنى كۆچۈرۈپ يوللىساق مۇنبەر چوقۇم ھەممىزگە ھوزۇر ئاتا قىلالايدۇ .
      شېئىرلارنىڭ سۈپىتىگىمۇ ئالاھىدە ئەھمىيەت بەرسەك ، تېما يوللايدىغان ئەزالارمۇ تېمىسىنى قايتا - قايتا پىشششىقلاپ ئىشلىگەندىن كېيىن ئاندىن يوللىسا ، تېما تەستىقلىغان باشقۇرغۇچىلارمۇ ھۇشيارلىقنى ئۆستۈرۈپ ، تېمىلارغا بولغان تەلەپنى تېخىمۇ قاتتىق قۇيۇپ ، ئەھمىيەتسىز تېما ، ھىچبىر ئەھمىيىتى يوق ، ھەممە يەردىن تاپقىلى بولىدىغان ناخشا ۋە رەسىملەرنى تەستىقلاش ئۆتكىلىنى چىڭ تۇتساق ھەممەيلەن ياقتۇرغۇدەك ئىسمى جىسمىغا لايىق مۇنبەر قىلىپ قۇرۇپ چىقالايمىز دەپ قارايمەن .    
  
ئەمدى مەنمۇ بىردەم قاخشاي . 
       بىر قىسىم ئەزالار تورخانىدىلا ئولتۇرۇپ باشقا بىر توربەتلەردىن كۆچۈرۈپ كەلگەن ھېچبىر ئەھمىيىتى يوق تېمىلارنى يوللاۋاتىدۇ ، ئۇچۇر بېرىپ ئۈچۈرسەك بىر دەمدىن كېيىن يەنە يوللاۋاتىدۇ . ئاخىرى ئۇچۇر بەرمەيلا ئۈچۈرىدىغان بولدۇق ، شەخسەن ئۆزۈم شۇنداق قىلىدىغان بولدۇم . نىمىشقا ؟ بۇ ھوقۇقنى ساڭا كىم بەرگەن دەمسىز ؟ 
      ئەمسە دىققەت بىلەن ئاڭلاڭ !
تىت - تىت - تىت ، چىك - چىك - چىك ، بۇ نىمە ئاۋاز ؟
       qq نى ئاچساملا ياكى مۇنبەرگە كىرسەملا بىر مۇنچە ئۇچۇر ، قاراپ باقاي دەپ دەپ ئاچسام بەزى ئەزالار ئۆز تېمىسىنىڭ تىلى ، بەئىلىگى ، ۋەقەلىك ئورۇنلاشتۇرۇلىشى تەرەپلەردىن مەسلىھەت سورىغان ، بەزىلىرى مۇنبەرگە رەسىم يوللاش ، تېما يوللاش جەھەتتە مەسلىھەت سورىغان . بۇلارغا ۋاقتىمنىڭ زىچلىقىغا قارىماي قولۇمدىن كەلگەن مەسلىھەتلەرنى بەردىم . نوقسانلارنى ئەينەن كۆرسىتىپ بەردىم ، بەزىلىرىنى ئىنكاس شەكلىدە يېزىپ قويدۇم .( ھەممىسى تېما ئۈستىدىلا بولۇندى ) ئىشقىلىپ ئۇلارنىڭ ھاجىتىدىن چىقىشقا تېرىشتىم . ئۇلارمۇ مەلۇم جەھەتتىن قانائەت ھاسىل قىلدى . مۇشۇنداق ئەزالارنى بارلىق باشقۇرغۇچىلار قارشى ئالىمىز . قولىمىزدىن كېلىشىچە ياردەم قىلىشقا تىرىشىمىز . 

      يەنە كۆپ قىسىم ئۇچۇرلارچۇ ؟ << نىمىشقا تېمامنى تەستىقلىمايسەن ؟ كىمتىڭ سەن بۇنچە كۆرەڭلىگۈدەك ؟>> ، << دەپ قوياي : تېمامنى ئۆزگەرتمە ماقما؟ . باشقىلار بىلەن ئورۇشسام نىمە كارىڭ . ئىشىڭنى قىل ماقما؟ ، باشقىلارنىڭكى تەستىقلىنىدۇ ، مېنىڭ نىمىشقا تەستىقلانمايدۇ ، ھوشۇڭنى تاپ ماقما؟ >> ئۆرگىلىپ كېتەي بۇيرۇقلاردىن . بۇ ئەمدى قايسى ئاكا - ئاچىلىرىمدۇ دەپ قارىسام يەنە شۇ ئىنكاسىدا داۋاملىق باشقىلارنى ھاقارەتلىمىسە ، ئۇيقۇسى كەلمەيدىغان، يازغان تېمىغا ئىنكاس يازماي ماڭا تەنە قىلىپ ئىنكاس يازغان ئاشو چوڭلارنىڭ سىياقىغا كىرىۋالغان << ئۇششاق >> ئەزالاركەن  . بوپتۇلا دەپ ئۆتكۈزۈۋەتتىم ، لېكىن بۇ ئۇچۇردىلا ئەمەس ، ئىنكاسلاردىمۇ تەكرارلاندى . تېخىمۇ ئەۋزەيلىدى . ئاخىر بولماي دوستلىرىم بىلەن سىردىشىپ تۇرىدىغان بۇ ئالاقىلىشىش قورالىنىمۇ ئېچىشتىن قورقىدىغان بۇلۇپ قالدىم . بۇنداق ئۇچۇرلار يالغۇز مېنىلا ئەمەس باشقا باشقۇرغۇچىلارنىمۇ << يوقلاپ >> ئۆتۈپ تۇرىدىكەن .
      ئويلاپ باقسام بۇ ھاقارەتلەرنىڭ ، تەنە ، بۇيرۇقلارنىڭ كېلىش سەۋەبى ئاشۇ ئەھمىيەتسىز پاخال تېما ۋە ئىنكاسلارنى ئۈچۈرگۈنىنىمىز ئىكەن . قانداق تېما يوللىسا تەستىقلاۋەرسەك ، ئىنكاسلاردا خوشامەتنى بولۇشىغا ياغدۇرساق بىز << ياخشى باشقۇرغۇچى >> بولالىغۇدەكمىز . لېكىن يۈرۈكۈمىزگە قول سېلىپ باقساق بۇنداق تېمىلار، ئىنكاسلارنى تەستىقلاش بىز باشقۇرغۇچىلارنىڭ ، جۈملىدىن مېنىڭ قۇلۇمدىن كەلمەيدۇ . 
      شۇڭا دەيدىغىنىم : مەيلى كىم بولۇڭ ، چوڭ بۇلۇڭ ، كىچىك بۇلۇڭ ، مېنىمۇ ھۆرمەت قىلىپ ، ماڭا قارىتا سەمىمى تەنقىدى پىكىر بەرسىڭىز ، سىز مېنىڭ دوستۇم ، قېرىندىشىم ، مۇنبەردىشىم بولۇپ ھېساپلىنىسىز ، مەنمۇ سىزنى چوڭ ياكى كىچىك دىمەي ھۆرمەتلەيمەن . يەنە زىتىمغا ، شەخسىيىتىمگە ھاقارەت قىلسىڭىز مەنمۇ قاراپ تۇرمايمەن . بولۇپمۇ ئىنكاسلاردا ئىسمىمنى تىلغا ئېلىپ قالايمىقان بىلجىرلىسىڭىز ( ئىنكاسنى ماڭا ئەمەس تېمامغا يېزىڭ، ھەرگىزمۇ قۇرۇق خوشامەت قىلماڭ ) سەۋىر تاقەتنىڭمۇ چېكى بار . ئويلىنىپ ئىش كۈرۈڭ .
       ئەگەر بەكلا نارازى بولسىڭىز ، ئەينا  مىسرانىم . ئۇيغۇرۇم 007 لاردەك مۇنبەر باشلىقلىرى بار ئەرز قىلسىڭىز بولىدۇ . سەۋەنلىكم بەكلا كۆپ بولسا مەمۇ مۇنبەردىن چېكىنىپ ئۆزۈمنىڭ شەخسى نورمال تۇرمۇشۇمغا قايتىپ، ئۈزۈمنىڭ ئىشىنى قىلاي  . ئاندىن تىللامسىز ، غاجامسىز ، مانا مۇنبەر.

      شۇنداق باشقۇرغۇچىلارغا ھاقارەت قىلىۋاتقان توختاۋسىز ھۇجۇم قىلىپ ئىنكاس يېزىۋاتقانلارغا يەنە شۇنىمۇ دەپ قوياي : باشقۇرغۇرغۇچىلىق ئۇنداق ئاسان ئىش ئەمەس ، سىز شىرىن ئۇيقۇغا كەتكەن چاغلىرىڭىزدا يېرىم كېچىگىچە تېما تەھرىرلەش ، تېمىلارنى ، ئىنكاسلارنى كۈرۈپ نازارەت قىلىش بىلەن قانچىلىغان كېچىنى ئۇيقۇسىز ئۆتكۈزۈۋاتقان، ئائىلىسىنىڭ تۇگىمەس ئىشلىرىنى تاشلاپ قويۇپ ، ئائىلىسىدىكىلەرنى نارازىقىلىپ مۇنبەر ئۈچۈن خىزمەت قىلىۋاتقان  باشقۇرغۇچىلارنىمۇ چۈشىنىپ قويۇڭ ، ئاۋال ئۆزىڭىز يوللىغان تېما ۋە ئىنكاس ئۈستىدە ئەستايىدىل ئويلىنىپ بېقىپ سۆزلەڭ .

 قىسقىسى : مۇنبەرنىڭ توغرا يۈنۈلۈشتە تەرەققىي قىلىشى ئەزالار بىلەن باشقۇرغۇچىلار ئورتاق كۈچ چىقىرىشىغا باغلىق . شۇڭا ھەممەيلەن ئەستايىدىل ئويلىنىپ كۈرۈپ ، ساقلانغان مەسىلىلەرنى ۋاقتىدا تۈزۈتۈپ ، مۇنبەرنى ھەقىقى بىلىم باغچىسىغا ئايلاندۇرۇشقا تىرىشساق .
  
    پولكوۋنىك ، ئىككىمىز خېلى << دەردىمىز>> تۆكۈۋالدۇق ، ئەمدى <<ھالالباغ >> قا بېرىپ، بىرەر پىيالە چاي ئىچىشكەچ موڭداشمايلىمۇ .  
ھېسسىيات قارىشىڭنى ئەقلىڭنىڭ ئۈستىگە قويغان كۆنۈڭ، سېنىڭ خاتالىشىشقا يول قويغان تۇنجى كۈنۈڭدۇر.
