    ئۇستاز بىلەن شاگىرىت جاھان كىزىپ بىر خىلۋەت كىچىك بازارغا كىلىپ بىر ئائىلىگە چۈشكۈن قىپتۇ . يېرىم كېچە بولغاندا بۇ ئۆيدىن چىققان يىغا- زارە ئۇلارنى ئويغىتىۋېتىپتۇ . ئەسلىدە بۇ ئۆيدىكى ئايالنىڭ سىڭلىسى بوغۇم كىرىشىۋېلىش كىسىلىگە گىرىپتار بولغان بولۇپ ، ئاغرىققا چىدىماي دەرياغا سەكرەپ ئۆلىۋالغانىكەن . ئۇنىڭ ئاچىسىمۇ شۇ كېسەلگە گىرىپتار بولغان بولۇپ ، بالىسىنى كۆتۈرگىنىچە سىڭلىسىغا قاراپ يىغلاۋاتقانىكەن .
ئەتىسى تاڭ ئېتىپ يولغا چىققاندا شاگىرت ئۇستازىدىن :-ئۇستاز بۇ ئائىلىدىكى ئىككى ئايالنىڭ بىرى دەرياغا سەكرىۋالدى، يەنە بىرى توختىماي يىغلاۋاتىدۇ ، ئۇلار نىمە ئۈچۈن شۇنچە ئاجىز ؟- دەپ سوراپتۇ . -ئاياللار شۇنداق ئاجىزدۇر ،-دەپ جاۋاپ بىرىپتۇ ئۇستاز .


  بىر نەچچە يىل ئۈتۈپ ، ئۇلار يەنە مۇشۇ كىچىك بازارغا كىلىپ قاپتۇ . ئۇلار ئەينى چاغدا ئۆزلىرى قونالغۇ قىلغان ھېلىقى ئائىلىگە كەپتۇ .ئەسلىدىكى ئۆينىڭ ئورنىغا يېڭى ئۆي سېلىنغان بۇلۇپ ، ئىشىك ئالدىدا تۆت-بەش ياشلاردىكى بىر بالا ئويناۋاتقانىكەن . ئارىدا ئۆيدىن بىر ئايال چىقىپتۇ . ئۇنىڭ بەستى ئىگىلگەن بۇلۇپ ، بۇيى بالىسىنىڭ بۇيىدىن ئانچە پەرىقلەنمەيدىكەن . ئۇستاز بىلەن شاگىرت بىر قاراپلا ئۇنىڭ ئەينى چاغدىكى ھېلىقى ئايال ئىكەنلىكىنى تۇنۇۋاپتۇ . ئايالمۇ ئۇلارنى تونۇپتۇ ۋە ئۆز ئارا ئەھۋال سوراشقاندىن كىيىن مۇشۇ بىر نەچچە يىل ئىچىدە تاتقان جاپالىرىنى بۇ ئىككىسىگە سۆزلەپ بىرىپتۇ :-ئۇستاز ، ئۆزلىرى كەتكەندىن كىيىن ئېرىم كېسىلىمنىڭ ئۆزىگە يۇقۇپ قېلىشىدىن ئەنسىرەپ ئانا-بالا ئىككىمىزنى تاشلاپ كىتىپ قالدى . خەقلەرنىڭ ھەممىسى مىنى ئۇزۇن ياشىيالمايدۇ دىيىشتى . لىكىن مەن:<<ئەگەر مەنمۇ سىڭلىمغا ئوخشاش ئۆلىۋالسام، بالاممۇ شەكسىز ئۆلۈپ كىتىدۇ >> دەپ ئويلاپ ئۆلىۋېلىش نىيىتىدىن ۋاز كەچتىم ۋە قاتتىق ئاغرىققا ، چىداپ ھازىرغىچە ياشاپ كەلدىم ،- دەپتۇ ھەمدە كۈلىمسىرەپ تۇرۇپ سۆزىنى داۋاملاشتۇرۇپتۇ ،-مەن ئۆزۈمنىڭ مۇشۇنچىلىك ئۆمۈر كۆرەلەيدىغانلىقىمنى زادىلا ئويلىماپتىكەنمەن . بىراق ھازىرقى ئەھۋالىمنى ئۆزلىرىمۇ كۆرۈپ تۇرۇپلا .
پاكىز ، رەتلىك قورۇ-جاي بىلەن ئۇ ئايالنىڭ كۆزلىرىدىكى ئۈمىدنى كۆرگەن شاگىرىت ئۇستازىدىن : -ئۇستاز ، سىز ئاياللار ئاجىز كىلىدۇ ، دىمىگەنمىدىڭىز ؟ -دەپ سوراپتۇ .
ئۇستاز بېشىنى لىڭشىتىپ تۇرۇپ : - لىكىن ، ئانىلار كۈچلۈكلەردىندۇر ،- دەپ جاۋاپ بىرىپتۇ .
توغرا ، ئانا بۇ دۇنيادىكى ئەڭ ئۇلۇغ ئىنساندۇر ، ئۇ بالا ئۈچۈن ھەممىنى بويسۇندۇرالايدۇ . ئانا ئەڭ ئۇلۇغ كۈچتۇر ، ئۇ بەزىدە ھەتتا ئەزلائىلنىمۇ بويسۇندۇرالايدۇ .
