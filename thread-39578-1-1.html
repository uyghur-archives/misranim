كۆڭۈل قىزغا

ئاران كىرىپ مۇنبەرگە ، قوشاق تېمىسى يازدىم
كۆڭۈل خاتالاشسىمۇ ،ئەيىپلىرىنى ئاستا ياپتىم
تاڭنۇر شاقۇر چاپلىندىن، بەكلا كۈنلەپ كەتتىم
شۇ قىلغىنىممۇ يەتمەي، ھەتتا تىزلىنىپمۇ كەتتىم

ئەسلىدە كۆڭۈل قىزنى ، زادى چۈشەنمەك بەك تەسكەن
سىزنى سۆيىمەن دەپ قۇيۇپ باشقىنى سۆيۈپ يۈرەركەن
قانداق قىلاي مەن ئەمدى، قىزلارنى چۈشەنمەك قىيىنكەن
تۇنجى مۇھەببەت ئالدىسا، ئىچىم زەرداپ بۇلدىكەن

قوشاق تۇقۇپ مۇشۇنداق ،ئىچ دەردىمنى تۆكۈۋالاي،
ئالداپ كەتتى كۆڭۈل قىز ، مەن ئەمدى قانداق قىلاي
شۇنچە يالۋۇرۇپ زارلاندىم ،ئەمدى كىمگە يالۋۇراي
چىن سۆيگەننى بىلمىدى، ۋاپاسىز قىزدىن زارلانماي

تاڭنۇر ئاكامغا
ئەسسەلامۇ ئەلەيكۇم تاڭنۇركا، تىنىڭىز ساق -ئامانمۇ؟
مەن بەرگەن كۇمىلاچنى يەپ، باشقا چاچلار چىقتىمۇ؟
بورتالانى سوغۇق دەيسىز، سوغۇق ئەمدى كەتتىمۇ؟
مېھمان بۇلۇپ مەن باراي، ئۆي-ئايۋانلار كاتتىمۇ؟

مېھمان بۇلۇپ مەن بارسام، مىنى قارشى ئالامسىز؟
ياكى مىنى مالچى دەپ، ئېغىلغىلا باشلامسىز؟
ئۆيىڭىزگە مەن كەلسەم، قورقماي توپاق سويامسىز؟
خۇتۇنىڭىز گەر ئۇنمىسا، ئاغزىغا بىرنى قۇيامسىز؟

چاقچاق قىلدىم تاڭنۇركا، چاقچىقىمنى كۆتۈرۈڭ
مېھمان بۇلۇپ گەر بارسام، مىكيان بولسىمۇ ئۆلتۈرۈڭ
لازىنى راسا سېلىپ ، تۇخۇ قوردىقى كەلتۈرۈڭ
نان بىلەنلا ئۇزاتماي ،گۆشلۈك تاماق ئەتتۈرۈڭ

قېنى تاڭنۇركا بارمۇسىز، سىزنى قوشاققا قاتتىمغۇ
قىچقارمىغان بولسىڭىزمۇ، مېھمان بۇلۇپ باردىمغۇ
ئۆيىڭىز سوغۇق بولسىمۇ، بىر كېچە چىداپ ياتتىمغۇ
ئۆيىڭىزدە بار تاماقنىڭ، ھەممىسىنى يەپ كەتتىمغۇ


گۈلبەشكەر ئاچامغا
مۇنبىرمىزدىكى بۇ قىزچاق، يىتىشىپ چىققان قوشاقچى
مۇھەببەتلەشكەنلەرنى كۆرسە، ئىش بۇزىدىغان پۇتاقچى
ئۇرۇشقانلارنى كۆرسە، جىم تۇرماس بۇ دۇمباقچى
ئۇرۇشۇپ قالسا ئەگەردە، ئاغزى ھارماس كاپكاپچى

مۇنبىرمىزدىكى بۇ قىزچاق، يىتىشىپ چىققان پايلاقچى
پايدىنى كۆرسە يۈگرەيدىغان ، بىكار تەلەت ئاقنانچى
گەپ ياتماس ئىچىدە، ئۆتۈپ كەتكەن غەيۋەتچى
تېما كۆرسە جىم تۇرماس، ئىچى قۇتۇر بۇزغۇنچى

گەپ تالاشقانلارنى كۆرسە، ئارلىشۋالىدىغان سالاچى
ۋەدىسىدە پەقەت تۇرماس، دىتى يوق يالغانچى
چاقچاق قىلسام ئۇنىڭغا، كۆتۈرەلمەس بۇ پوچى
مەن نەسرۇللا ئاشۇنداق ، ئاغزىم سېسىق قوشاقچى

..........گۈلبەشكەر ، قېنى بارمۇ سىز، سىزنى شۇنداق قوشاققا تۇقۇساممۇ يەنە جىم تۇرۋالامسىز؟قېنى يامان بولسىڭىز مىنىڭ ئەدىبىمنى بېرىپ قويمامسىز، كارامەتلىرڭىزنى چىقارمامسىز، ھازىر شۇ تاپتا ئىلھاملىرىڭىز مىنىڭ قوشىقىمنى كۆرۈپ نەچچە ھەسسە كۈچۈيۈپ ئاغزڭىزدىن بۇلدۇقلاپ چىقىۋاتقاندۇ ھە، سىزگە تۇقۇغان قوشاقلىرىم ھەرگىز كەتمىسۇن، سىزنى چاقچاق كۆتۈردۇ دەپ سىزگە چاقچاق قىلدىم، ئۇنىڭ ئۈستىگە ئىككىمىز كونا مۇنبەرداشلاردىن بولغاندىكىن خاپا بولمامسسىز ھە، بۇ دىگەن بىر قوشاق خالاس. ئەگەر خاپا بولسىڭىز مىنى قانداق قوشاققا تۇقۇپ يۈزۈمنى چۈشۈرۋەتسىڭىزمۇ كۆتۈرمەن،

يوچۇن ئادەم گە

ئەسسەلام جان ئاكا ، ئىلھاملىرىڭىز كەلدىمۇ
شېئىر يىزىپ مۇنبەردە، بۇ قوللىرىڭىز تالدىمۇ
قوشاق يىزىپ مەن سىزگە ، دوستلاشساق بۇلامۇ
يازغان بۇ قوشاقلىرىم ، سىزدەك شائىرغا يارامۇ

چاقچاق قىلىپ سىز بىلەن ،يېقىن دوسلاردىن بولغۇم بار
شېئىر يىزىپ سىزگە ئوخشاش، ئۇستا شائىر بولغۇم بار
سىزگە زادى قايىلمەن، خالىسڭىز [ئاكا] تۇتقۇم بار
خالامسىزكىن ئاكىجان، سىرداش دوست بۇپ ئۆتكۈم بار


مۇسلىنە گە

مۇنبىرمىزدىكى مۇسلىنە ،ئىسمىڭىز نىمانداق چىرايلىق
قوشاققا قاتسام مەن سىزنى ، قىلماسسىز ماڭا يامانلىق
مىنىڭ ئىسمىمنى بىلمىسىڭىز ، بېشىم ئۈستىدە يىزىقلىق
تۇنۇشۇپ قويسىڭىز مەن بىلەن ، قىلىمەن سىزگە ياخشىلىق

ئاڭلىسام سىز مۇسلىنە ، ئۇستا قوشاقچى ئىكەنسىز
كارامىتىڭىزنى كۆرسەتمەي ، دائىم جىملا يۈرەمسىز
قوشاق تۇقۇپ تېمامدا، ئىنكاسلارنى يازامسىز
قوشاققا قېتىپ مىنىمۇ ،ئاز تولا داڭلاپ قۇيامسىز

ئەگەر خالىسىڭىز ئىككىمىز، بىرگە قوشاق تۇقۇيلى
قوشاق ئاتقان دوسلارغا ، بىرلىشىپ تەڭ تۇرايلى،
راسا تۇقۇپ قوشاقنى، باشقىلارنى خوشال قىلايلى
مەقسىتىم مىنىڭ شۇ، ئىككىمىز دوست بۇلايلى


ئۆرۈكچانغا
مۇنبرىمىزدە بىر شائىر بار. ئىسمى ئۇنىڭ ئۆرۈك كەن
پۇلى بولمىغاچ يانچۇغىدا، خۇتۇن ئالالماي يۈردىكەن
پۇلىڭىز بولمىسا ماڭا دەڭ. ئىككىمىز بىر يۇرتلۇق كەن
ئىككىمىز بىر تۇنۇشساق ، ياخشى دوسلار بولساقكەن

مىنى دوست دەپ بىلسىڭىز، خۇتۇن ئېلىپ بىرىمەن
قىززىتسىڭىز تېمامنى ، خۇتۇن ئېلىپ بىرىمەن
خۇتۇن ئالالماي مۇنبەردە، بىر مۇنچىۋالا قاقشاپسىز
پۇلىڭىز بولمىسىمۇ ، خۇتۇن ئېلىپ بىرىمەن.

بىزنىڭ بۇ يۇرتلاردا، خۇتۇن ئالماق بەك تەسكەن
خۇتۇن كىشىنىڭ مەھرى ھەققى، ئوتتۇزمىڭ كويكەن
قانداق قىلسىڭزى بۇلار ئەمدى، يانچۇقىڭىزدا پۇل يوقكەن
كورلا دىگەن بۇ يۇرتتا ، خۇتۇن ئالماق بەك تەسكەن


كۈن قىز غا

مۇنبىرمىزدە بىر قىز بار ، ئىسمى ئۇنىڭ كۈن قىز
تېما يوللاشقا بەك ئۇستا ، سۈرئىتى بەك تىز
قىززىمايدۇ ھەرگىزمۇ، بۇ مۇنبەر ئۇنۇڭسىز
تېمىللىرىنى كۆرسىڭىز سىزمۇ قايىل بۇلىسىز

داڭلاپ قويسام كۈنقىزچاق، ھاكاۋۇرلىشىپ كەتمەڭ
ماختاپ قويسام مەن سىزنى ، ماڭا ئامراق بۇپ قالماڭ
باي بۇپ كەتتىم مەن ھازىر، ئەمدى ماڭا پۇل بەرمەڭ
قوشاققا قاتسام مەن سىزنى، مىنىمۇ قوشاقتا داڭلاڭ

باشقۇرغۇچىمىز كۈنقىزچاق، يازدىم مەن سىزگە قوشاق
يىزىپ قويمامسىز ماڭىمۇ، بىر نەچچە ئېغىز قوشاق
قوشاق تۇقۇشۇپ ئىككىمىز دائىم مۇڭدىشىپ تۇرساق
ئارلىقىمىز يىراق بولسىمۇ، دائىم ئەسلىشىپ تۇرساق


ئىدىرىسچانغا
خېردارىم ئىدىرىسچان، تىجارىتىڭىز بەك ئوبدان
مىنىڭ ئىلتىپاتىم بىلەن، سىز يۇرتىڭىزدا باي بولغان
سۈتىڭىز تۈگەپ كەتسە، كالىنى ئۆزىڭىز ساققان
پايدىنى كۆپ ئالىمەن دەپ سۈتكە سۇنى قوشقان

دۇكاندىن بىرنى ئېچىپسىز ، سۈتچىلىكمۇ قىلىپسىز
سۈتكە سۇنى ئارلاشتۇرۇپ، سەتلىشىپمۇ يۈرۈپسىز
سۈتىڭىزنى خەق ئالمىسا ،ئۆزىڭىزلا ئىچىپسىز
شۇڭلاشقا سۈتچىلىكتىن زىيان تارتىپ كىتىپسىز

باشقا ئىش قىلىڭ ئىدىرىسچان، سۈتچىلىق قىلماق بەك تەس
زىيان تارتىپ سۈتچىلىكتىن، ھەر ۋاقىت يۈرمەڭ مەس
چۈشكۈنلىشىپ كەتمەڭ ھەرگىزمۇ، ئورنىڭىزدىن تۇرۇڭ دەس
