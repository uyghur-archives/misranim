بۇندىن كىيىنكى ئىجادىيتىڭىزگە ئۇتۇق تىلەيمەن ، ئەجرىڭىزگە تەشەككۈر  [s:166]

مەن بۇ ئەسەردىن تولىمۇ ئەپسۇسلاندىم.مەن بىر تولۇقسىز ئوتتۇرا مەكتەپ ئوقۇتقۇچىسى.مەكتىپىمىزدە بالدۇر يېتىلىپ مۇھەببەت يولىغا بالدۇر كىرىپ قالغانلارمۇ يوق ئەمەس.2%ئەتراپىدا بار.لىكىن قالغان ئوقۇغۇچىلىرىمىز ئۇنداق ئىشلارنى ئويلىماي ساددا بالىلىق ھىسسىيات بىلەن پۇتۈن زىھنىنى ئوقۇشقا قارىتىۋاتىدۇ.بۇ پوۋېست تولۇقسىز ئوتتۇرا مەكتەپ ئوقۇغۇچىلىرىنى زەھەرلەيدىغان،بالدۇر مۇھەببەتلىشىش قاينىمى شەكىللەندۈرۈپ قويىدىغان ،ھەم شۇنداق يولدامېڭىۋاتقانلارنى قىلىۋاتقان ئىشىمىز توغرىكەن دىگەن تۇيغۇغا كەلتۈرۈپ كويىدىغان ناچار ئىددىيەۋى باش تېمىغا ئىگە ئىكەن.مەن زىيالى دىگەن تورداشنىڭ ئىنكاسلىرىنى ۋە باشقا ئەسەرلىرىنى كۆرگەن بولۇپمۇ بىر غايىلىك ياشقا خاس ئىچكى دۇنياسى ئىپادىلەنگەن قەلىب سۆزلىرىدىن سۆيۈنگەن ئىدىم.بۇ ئەسىرى مېنى قاتتىق ئۈمىتسىزلەندۈردى.بىر ئەسەرنى يېزىپ چىقماق ئاسان ئەمەس.بولۇپمۇ بۇنداق خېلى يۇقۇرۇ سەۋىيەدىكى ئەسەرنى يېزىپ چىقماق ئاسان ئەمەس.مەن بۇ ئىنكاسنى يازمىسام بولاتتى.لىكىن بىز بىراۋنىڭ نوپۇزىنىلا ئويلاپ پەرزەنتلىرىمىزنى ئويلىمىساق بولمايدۇ.بۇندىن كېيىن كىمدە كىم ئەسەر يازماقچى بولىدىكەن تولۇقسىز ئوتتۇرا ۋە باشلانغۇچ مەكتەپتىكى بالدۇر يېتىلىپ قالغان بالىلارنى تېماقىلماڭلار.تالانتىڭلارنى ئىقتىدارىڭلارنى زىيانلىق تېمىلارغا ئەمەس پايدىلىق تېمىلارغا قارىتىڭلار شۇ چاغدا مەنمۇ ئالقىشلىشىپ بېرىمەن.زىيالى ئۇكام بۇ تېمىڭىزدىن بەدىئىلىك جەھەتتە ئىۋەن تاپمىدىم توغرا چۈشۈنۈڭ.باش تېمىڭىز زادىبولماپتۇ.ھازىر مېنى خاتا چۈشەنسىڭىز نەچچە يىلدىن كېيىن سىزمۇ مەكتىپىڭىزنى پۈتتۈرۈپ مەن بىلەن كەسىپداش بولۇپ قالغان ۋاقتىڭىزدا مىنىڭ ئاغرىنىشىمنى چۈشىنىپ قالىسىز. ئۆسمۈرلىرىمىزنى تىلىۋىزىيە تىياتىرلىرىنىڭ زەھەرلەشلىرىمۇ يېتەر.بولدى بەس قىلايلى.
قىزىنى باشقۇرمىغان ئانا گۇناھكار،ئايالىنى باشقۇرمىغان ئەر جىنايەتچى،ئاچا-سىڭىللىرىنى باشقۇرمىغان ئاكا-ئۇكىلار خىيانەتچى!

بىر قوشۇق تۇز بىر قازان ئاشقا تېىتىيدۇ. « كۆرگىنىڭدىن كۆپتۇر كۆرمىگىنىڭ، كۆرمىگەننى كۆرىسەن ئۆلمىگىنىڭ». « سۇغا سالساڭ سۇ كۆتۈرمەس مىسقال تۆمۈرنى، ئالتۇن بېرىپ ئېلىپ بولالماس قالغان كۆڭۈلنى». « لەيلىنىڭ جامالىنى كۆرۈشكە مەجنۇننىڭ كۆزى كېرەك»

ئاپتۇر (ziyaliy) نىڭ يازمىسىغا

سالام زىيالى !
  بىر ئەدەبياتچىغا پىكىربەرسەم ،ئوزۇمگمۇ غەيرى تۇيۇلۋاتىدۇ،ئەمما مەن ئەسەر ئوقۇسام،ئۆز خاھشىغا بىقىپ ئوزگەرتىپ باققۇم كىلدىغان،شۇڭا بۇ پۇرسەتنى چىڭ تۇتتۇم ئەمسە.....
    ئەسەردە ياش مەسلىسدە ئۇنماي تۇرۋالغۇدەك،بۇ ۋەقە بىشڭىزدىن ئۆتكەندەك قىلدۇ(مېنڭ قارشىم خالاس)  يەنە بىر مەسلە،گۇلزېبانىڭ شۇ سەرگۇزەشتىسى سەل ماس كەلمەيدىكەن،شۇنداق كۆپ ئشلارغا يولقۇشقا جريان،ۋاقىت ،ۋەقە بولشى كېرەك ئەلۋەتتە،شۇڭا،مەنمۇ جاھللىق بىلەن ياش ئوزگەرتىشنى تەشەببۇس قىلمەن،     ئەمدى،ئۆزىڭىز ياردەم دەپسىز،مەن بولسام،گۇلزىبانىڭ كىسىلنى ئۇنىڭ ۋاپاسىزلىقى ئاشكارلانغاندىن كېيىن ،باھادىر ئۇنىڭغا ئوچ بولۇپ،يىرگېنىپ،كۆرۇشمەس بولغاندىن كىيېن،ھەتتا كۆزىگەئالۋاستى بولۇپ كۆرۇنگەندىن كىيېن،ئاشكارلايتىم،گۈلزىبانىڭ ئۆز ئاغزى بىلەن ئاخىرقى قىتىم كۆرۇشۈشنى تەلەپ قلغاندىن كىيېن ئاشكارلايتم،ھەممە ئاشكار بولغاندا باھادىر ئىچ ئاغىرتىپ،ئۇنى كەچۇرەتتى،(ئەسەردىكى باھادىر تۇتامسز خاراكتىردە بولسا ئوبراز بەك خۇنۈك بىلىندىكەن)ئازىدىن ئاز كۈن ئۆتكۇزۈپ گۈلزىبانى ئۆلۇمنى كۈتۇۋىلېپ،ئۇنى ياد ېىتىپ قەبرىسنى يوقلاتقۇزۇپ،شۇ جەرياندا ئالىيەنى سەپەرداش قللاتتم.(مەن ئەسەر ئوقۇسام ،ئۆزگەرتىپ مۇنداق بولسا بەكرەك جەلىپ قىلامدۇ ياكى ئانداق بولسىمۇ دەپ ئوقۇغاچقا ئاقساقاللىق قىپ قويدۇم كەچۇرۇڭ)
