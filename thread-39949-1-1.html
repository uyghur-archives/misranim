                                                  ساراڭ ۋە سالام خەت

ساراڭلار دۇختۇرخانسدىكى بىر ساراڭ باشقا بىرسگە خەت يازماقچى بۇلۇپ خەت يېزىپ  ئولتۇرسا سېستىرا قىز  تەكشۈرۈپ كىرىپ قاپتۇ دە ،  ساراڭدىن سۇراپتۇ : " ھەي نىمىش قىلۋاتىسەن ؟ "

ساراڭ چالۋاقاپ :" ۋاي نىمە كارىڭ  .ئۆزەمگە - ئۆزەم خەت يېزىۋاتىمەن "

سېستىرا قىز قىزقىپ سۇراپتۇ :  ئەمىسە دىگىنە خەتكە نىمە دەپ يازدىڭ  ؟

ساراڭ تېخمۇ تىرىكىپ :  ماڭاۋە كالۋا . مەن تېخى خەتنى تاپشۇرۇپ ئالمىغان تورسام ، نىمە بىلىمەن نىمە يازغىنىمنى !!

                                                
                                                      بۇ خۇراز كىمنىڭكى  

بېرشوپۇر ماشېنا ھەيدەپ كېتىۋېتىپ ئېھتىياتىسىزلقىتىن بىر خۇرازنى بېسىپ ئۆلتۈرۈپ قۇيۇپتۇ .شوپۇر بېشنى قاشلىغان ھالدا ماشنىدىن ئاستا چۇشۇپ  بۇ كىمنىڭ خورۇزىدۇ  دەپ تۇرسا  بىر ئۆينىڭ ئالىدىدا ئۇيناپ تۇرغان بىر بالىنى كۆرۈپ ئونڭىدىن سۇراپتۇ :" ھەي بالا بۇ خوراز سىلەنڭىمۇ ، ئەگەر سىلەرنىڭ بولسا تۆلەپ بېرەي " دېسە  .ئۇ بالا "ياق !بۇ بىزنىڭ  ئەمەس ، سەن  ئەڭ ياخشىسى بونى  ئۆينىڭ ئاقىدىكى كاتەكتىكى مىكىيانلاردىن قايسىڭلارنىڭ ئېرى يۇقاپ كەتتى  دەپ سۇراپ باق دەپتۇ .


                                     پىخسىق قاسساپنىڭ ھىساۋاتى

بىر پىخسىق قاسساپ  دوختۇرخاننىڭ داۋالاش ھەققنىڭ يۇققىرلىقغا ئاچچقتىن يىرلىغۇدەك بۇلۇپ دۇختۇرلارغا قاقشاپ مۇنداق دەپتۇ :
" نىمىدىگەن ئۇچىغا چىققان ساراڭلىق بۇ  !  تۇغۇت بۆلۈمدىكى دۇختۇرلاغا تۆلىگەن پۇل ، ئايالمنىڭ دۇختۇرخاندا ياتقان  پۇلى ، داۋالىنىش ھەققى ، كۆتۈنۈش پۇلى ، ئۇ پۇل بۇ پۇل بۇلاپ  ھەممنى قۇشقاندا جىڭلاپ باقساق  بىر كىلۇگىراممۇ كەلمەيدىغان  بۇ بالام 5000 يىۋەنگە تۇختدىمۇ ؟


                                                    پىشمىدىم  

ئىككى ساراڭ ساراڭلار دۇختۇرخانىسىدىن قېچىپ چقىپ بىر باغقا كىرۋاپتۇ .
بىر ساراڭ  يۇغان بىر ئالما دەرەخنىڭ ئۆستىگە چقىپ ئۇيناپتۇ ، يەنە بىرسى ئالما دەرەخى ئاستىدا ئۇياقتىن بۇياققا چېپىپ ئۇيناپتۇ .

بىر ۋاقىت ئۆتكەندە  ئالما دەرەخى ئاستىدىكى ساراڭنىڭ ئىچى پۇشۇپ  ئۆستىدىكى ساراڭغا دەپتۇ : ھەي چۆشمەمسە ، نىمانداق دەرەخنىڭ ئۆستىدە ئۇيناپ چۆشمەيسەن دەيمەن ؟

ئالما دەرەخىنىڭ  ئۇچىغا چىقىۋالغان ساراڭ پەسكە قاراپ  :   ماڭاۋە ساراڭ ، ئىشڭنى قىلە  ..مەن تېخى پىشمىغان تۇرسام


                                                      كەسپىمىز ئوخشاشكەن

بىر بىمار ئۇكۇلدىن بەك قۇرقىدىكەن ، يەنە كىلىپ ئۆلمەكنىڭ ئۈستىگە تەپمەك دىگەندەك ھەر قېتىم دۇختۇرخانغا بارغاندا  مەسئۇل سېستىرا قىز ئۇكۇلنى ۋاي داد دىگىدەك ئاغىرتىپ ئۇردىكەن . بۇ قېتىم ئۇكۇلنى ئەمدى ئۇرۇپ بۇلۇپ تازا ۋايجانلاپ تۇرغاندا
   سېستىرا قىز ئۇنى زاڭلىق قىلىپ سۇراپ قاپتۇ :" سىز نىمە خىزمەت بىلەن شۇغۇلىنىسىز ؟ "
بىمار : " ئىككمىزنىڭ كەسپى ئۇخشاش "
سېستىرا :" ۋۇي ئۇنداقتا  سىزمۇ دۇختۇركەنسىز دە ؟"
بىمار چۆشەندۈرۈشكە ئالىدىراپ :" ياقەي ، كەسپىمىز ئوخشاش  لىكىن قىلىۋاتقان ئىشمىز ئۇخشاش ئەمەس "
سېسترا بارغانسىرى قىزىقىپ سۇراپتۇ : " ئەمسە سىز ؟ ...."
بىمار :" مەن ئاياغقا مىخ قاقىمەن "


                                                      دوختۇرخانىدا ياتمىسا
بىر ئىدارىدا ناھايتى پىخسىق بىر باشلىق بۇلۇپ   ماشىنا بۇزۇلۇپ قالسالا شوپۇرنى تىللاپ جاپا سالىدىكەن شۇڭا شوپۇرمۇ ماشىنا بۇزۇلغان ھامان ۋاقتىدا رىمنۇت قىلىپ تۇرىدىكەن ، بىركۇنى ماشنىغا  يېڭىدىن سالغان    بىر سايمىنى بۇزۇپ قاپتىكەن، باشلىق شوپۇرغا يەنە چالۋاقاپ ماشىنا كىچىكىنا بۇزۇلۇپ بولغىچە يېڭى سايمان ئالماشتۇرىمەن دەپلا تۇرىدىكەنسەن  ئىلگىرى ئېلىۋەتكىنىنى سالسا بولمامدىكەن دىيىشىگە شوپۇرمۇ بوشكەلمەي بۇزۇلغان نەرسىلەر     دوختۇرخانىدا  بىرەر قېتىممۇ ياتمىسا دەپ جاۋاپ بىرىپتۇ .
