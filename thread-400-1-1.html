                    بۈگۈن تېما يولىغانلار دىققەت

                          مىسرانغا

                    مىسرانجانمۇ سىز ئارلاپ،
                    ئىنكاس يېزىڭ تېمىغا .
                    بەك ۋاقتىڭىز يەتمىسە ،
                    كۈلۈپ بىرىڭ زورىغا .

                    مۇنبەرداشلار دەيدىكەن ،
                    مىسرانىمنى بەك رەڭلىك .
                    قىزلاردىن رەڭ تالشىپ ،
                    سۇرتىۋالماڭ سىز ئەڭلىك .

                      مەتتۇرسۇن ئېلىي نۇرغا

                     چاچما شئېىر پىشۋاسى ،
                     مەتتۇرسۇن ئېلىي نۇردۇر.
                     32  ھەرىپنى يەۋېلىپ ،
                     سىڭمىگىنى ئۇ بىلەن ئۈدۇر.

                           ئوتقاشقا

                     شەيتان كۆڭۈلنى يازغىنى ،
                     ئوتقاش دوستۇم دەل ئۈزى .
                     تولا ئېلشىپ شەيتان بىلەن ،
                     پىرخۇن بولغان يەنە ئۈزى .

                             ئەرۋاھغا

                      ئەرۋاھ دوستۇم بەك يامان ،
                      سودا قىپتۇ مۇنبەردە ،
                      ساتالمىغان چى چىنى ،
                      گاڭ گو قىپتۇ دىيەنشىدە .

                             فاتىمەگە

                      ھەسرەت بىلەن ئۆتمەكتە ،
                      فاتىمەنىڭ كۈنلىرى .
                      تولا يىغلاپ ھەسرەتتە ،
                      قار بېسىپتۇ تۈنلىرى .

                        dakyanos قا

                       چىقىپ قاپتۇ يەنە بىر ،
                       ھەۋەسكارنىڭ ھەۋىسى .
                       باللارغۇ دەيدىكەن،
                       ئابدۇكىرىمنىڭ نەۋرىسى .

                         تەبەسسۇمغا
                      تەبەسسۇمنىڭ قۇشقى ،
                      يۇقاپ كەتتى مۇنبەردە .
                      ئىلھام ئىزدەپ بىراۋدىن،
                      يۈرىدىغاندۇ نەلىدە .

                            شەھلاغا
                      ۋاپاسىزنىڭ دەردىنى ،
                      بەك تارتىپتۇ شەھلا قىز .
                      كۆز يېشىڭىزغا ھەمرادۇر ،
                      مۇنبەردىكى سىز ھەم بىز .

                           كەترەنگە ،
                       كەترەن قىزمۇ توختىماي ،
                       ئىنكاس يازار تېمىغا ،
                       سەيلىگە بېرىپ قالساق ،
                       باشلامدىغاندۇ بېغىغا .

                              نازىغا
                        قىمار ئويناپ نازى قىز ،
                        كىرىپ قاپتۇ ساقچىغا ،
                        قىيىن -قىستاق دەردىدىن،
                        ئۆزگۈرىۋاپتۇ باخشىغا .

                              مەلىكىگە
                        ھاياتتىكى قايمۇقۇش ،
                        تاشلىمىسۇن چۆللەرگە .
                        ھەرگىز خاپا بوپ قالماڭ ،
                        مىسران قويغان نۆللەرگە .

                                 شىرخانغا

                         ئۆيلەپ قۇيۇپ بوۋسىنى ،
                         يۇقاپ كەتتى ما شىرخان ،
                         ئەسەر توپلاش دەردىدە ،
                         چىكىپ قالماڭ سىز پىغان ،

                              مەلىكەمگە
                         مەلىكەمنىڭ دەستىدە ،
                         تۇنۇشۇپ قالدۇق بىر ئۇبدان.
                         يازە يازە ئاخىرى ،
                         پۈتۈپ چىقتى بىر توپلام .

                                شاتئايغا
                         شاتئاي قىزمۇ ئىنكاسنى ،
                         بەك يازىدۇ رىتىملىق ،
                         يازغىنىمدۇر بۇ سىزگە ،
                         ئاتاپ تۇنجى قېتىملىق .

                                   ئۈمۈت گۈلىگە
                          ئۈمۈت خېنىم بامۇ سىز ،
                          قايسى باغنىڭ گۈلى سىز .
                         چىچىرىڭىزغا خەت يازسام،
                         ئالدىراشلا مەن دەيسىز .

                          قۇشاق قاتسام سىلەرگە ،
                          ھەرگىز خاپا بولماڭلار ،
                          تاماق بولسا تالىماي ،
                          ھەمىسىنى تەڭ يەڭلار .

ئەسسالام تورداشلا مەن بۈگۈن تېما يولىغانلارنىڭ تېمىسغا قارتا ئازراق قۇشاق يېزىپ قويدۇم قاملاشمىغان
بولسا خاپا بولمىغايسىلەر ،ئىسمى چىقمىغانلارمۇ شۇنداق كىيىنكى قېتىم سىلەرنىمۇ يازىمەن .
