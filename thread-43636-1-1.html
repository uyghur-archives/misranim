ئەسسالامۇ__ئەلەيكۇم ئەزىز قېرىنداشلار !!!كەمىنە بىھزاتتىن مۇنبەردىكى ھەممەيلەنگە ئوتلۇق سالام...
ئۇزۇن بوپتۇ،مەكتەپتىكى ئوقۇش پۈتتۈرۈشنىڭ ئىشلىرى بىلەن مەشخۇل بولۇپ سىلەر بىلەن ئەسرالاشمىغىلى ،مانا بۈگۈن پۇرسەتمۇ كىلىپ قاپتۇ ،ئۇشبۇ پەيتتە تۆۋەندىكى مەسىلە توغرىسىدا كەڭ مۇنبەر ئەھلى بىلەن ئورتاقلاشساق دىگەن ئۈمد-ئارزۇ مىنى قەلەم تەۋرىتىشكە سىلەرنىڭ كۆز قاراشلىرىڭلارنى ئاڭلاشقا دالالەت قىلدى!!!
مەسئۇلىيەتچان ،ئەقلى ھۇشۇ جايىدا كىشىلىرىمىز تۇرمۇشقا قانداق مۇئامىلىدە بولۇشنى ئۆز كاللىسىدا تولۇق ئايدىڭلاشتۇرىۋىلىپ ئاندىن ئۆيلۈك-ئوچاقلىق بولۇشقا قارار چىقىرىشىدۇ،ئائىلە ئەر-ئايال ۋە پەرزەنت بىلەنلا ئۆزىنىڭ ئائىلىلىك ئۇلىنى مۇستەھكەملىيەلەيدۇ...خوش ،ئۇنداقتا مۇشۇ<<ئائىلە>>ئۇقۇمىغا سىز زادى قانداق تەبىر بىرىسىز؟سىزنىڭچە ئائىلىۋى ھاياتقا نىسپەتەن مۇھەببەتتىن باشقا يەنە قايسى تەرەپ بىزنىڭ مۇشۇ ئائىلىنى بەرپا قىلىشىمىزدىكى ئاچقۇچلۇق مەسىلە ھىساپلىنىدۇ...سىزنىڭچە ياخشى كۆرۈشتۇق دەپلا داقا دۇمباقنى بولىشىغا چىلىپ ئەر-ئايال ئاتالغانلىق بۇ ئىككى ياشنىڭ ھاياتىغا نەۋ باھارنى باشلاپ كىلەلەمدۇ!؟
قايسى كۈنى بىر سورۇندا بىر ئاغىنىمىز ئۆزىنىڭ تويغا تەييارلىق كۆرىۋاتقىنىنى ،پات-يىقىندا ئائىلە قۇرىدىغانلىقىنىنى ئېيتىپ ئۆتتى...لىكىن ئۇنىڭ كىچىكىدىن تۇغما يۈرەك كىسەل...قاتارلىق بىر قانچە خىل ئېرسىيەت خارەكتىرلىك كىسىلى بارلىقىنى بىلەتتۇق،شۇنىڭ بىلەن مەن خالى جايدا ئۇنىڭدىن توي قىلماقچى بولغان قىزنىڭ بۇنى بىلىدىغان - بىلمەيدىغانلىقى توغرىسىدا سورىغىنىمدا ،ئۇ ماڭا:<<ئۇ قىزنىڭ باشقا يۇرتلۇق يىراق تۇغقىنى ئىكەنلىكى،ئۆزىنىڭ كىسىلىنى بىلىپ قىلىپ تۇرمۇشلۇق بولالماي قىلىشىدىن ئەنسىرەپ مۇشۇ ئۇسۇلنى قوللانغانلىقىنى ،ئۇنىڭدىن باشقا ئامال قلالمىغانلىقى،يۇرتتىكى قىزلار كىسەل ئىكەنلىكىنى بىلگەندىن كىيىنلا تويغا قوشۇلمىغاچقىلا مۇشۇ قارانى چىقارغانلىقىنى،تويدىن كىيىن دىيىشنى لايىق تاپقانلىقىنى ئېيتىپ ئۆتتى،مەن ئۇنىڭدىن قايتىلاپ سورىدىم:_<<ئۇنداقتا پەرزەنلىك بولساڭ قانداق قىلىسەن،ئۇ تەرىپىنى ئويلاشتىڭمۇ ؟>>ئۇ:_ياق،ئاداش قالغىنىنى خۇدايىمغا تاپشۇرۇپ قويدۇم،ئەگەر كىسەل تۇغۇلۇپ قالسا داۋالاتسام بولىدىغۇ...؟
كۆزئالدىمدا نۇرغۇن سۇئال ،بىر ئائىلنىڭ تۈۋرۈكى بولماقچى بولغان ئەر ،تىخى ئائىلە قۇرۇشتىن بۇرۇنلا مۇشۇنداق مەسئۇلىيەتسىزلىك قىلسا ،ئۇنىڭ كىيىنكى تۇرمۇشى زادى قانداق بولار ؟؟؟مەن ئويلايمەن،ھىچ بولمىغاندا ئۆزىنىڭ كىسەللىك ئەھۋالى توغرىسىدا شۇ قىزنى ئەسكەرتىپ ھەقىقى رازىلىقى بىلەن تۇرمۇشقا چىققان بولسىمۇ،ئۇنى يەنە توغرا چۈشەنسەك،لىكىن ،پەقەت ئۆزىنڭ بويتاقچىلىق ھاياتىنى ئاخىرلاشتۇرۇش ئۈچۈن ئاتا-ئانىسىنىڭ زورى بىلەن بۇ تويغا تۇتۇش قىلىشى ھەر قانداق ئەقىل ئىگىسىنى ئويغا سالماي قويمايدۇ...يىقىندىن بۇيان دوختۇرخانىلاردا نۇرغۇن كىچىك بالىلارنىڭ تۇغۇلغاندىن باشلاپ كىسەل دەردىنى يەتكۈچە تارتىۋاتقانلىقىنى كۆرۈپ بۇ مەسىلە توغرىسىدا تولا ئويلايدىغان بولۇپ قالدىم،شۇ نارسىدىلەرنىڭ ھاياتقا بولغان تەلپۈنۈشى ،سىزگە قارىغاندىكى ھالەتلىرىدىن يۈرىگىڭىز تولىمۇ ئاچچىق ھىسلارغا چۆمىدۇ...ئۇلارغا ئىچ ئاغرىتىمىزيۇ لىكىن،قولىمىزدىن شۇ نارسىدىنىڭ كىسىلىگە شىپالىق تىلەشتىن ئۆزگە چارە قىلالمايمىز... ئۆزىمىزنىلا ئويلىمايلى قېرىنداشلار ،كەلگۈسىدىكى شۇ نارسىدىنىڭ كىسەل كارۋىتىدا ئاچچىق ئىڭراپ ...سىزگە مەڭگۈلۈك دەرت-ئەلەم ۋە پۇشايماننى ھەمرا قىلىپ قويماسلىقى ئۈچۈن ياخشىراق ئويلىشىپ بىقىشىڭلارنى ئۈمد قىلىمەن !!!
بۇ ئەلۋەتتە شەخسەن ئوغۇللارغىلا قارىتىلمىغان،مەيلى يىگىتلىرىمىز ۋە ياكى قىزلىرىمىز بولسۇن بۇ توغۇرلۇق ياخشىراق ئويلىشىپ كۆرىشىنى ،كۆز ئالدىنىلا كۆرۈپ ،يىراق كەلگۈسىنى <<دوغا>>تىكىپ بىردەملىك بەختنىڭ قۇچىقىدا ئەللەيلىنىپ ئۆزىنى غەپلەت ئۇيقۇسىنىڭ ساھىبى قىلىپ قويماسلىقىنى ئۈمد قىلىمەن!!!


ئالاھىدە ئەسكەرتىش :__ھەممىنى بىلگۈچى شۇنداقلا بىزنى تۇرمۇش مۇساپىمىزدە تۈرلۈك سىناقلار بىلەن سىنىغۇچى رەببىمىزنى، يەنى ئاللاھنىڭ قۇلى بولۇپ ياشاپ كەلگەن ئاتا-بوۋامنىڭ پۇشتى بولغان مەنمۇ ئوبدان بىلىمەن...شۇڭا بۇ توغرىسىدا ئوشۇق سۆزلىمەسلىكىڭلارنى ئۈمد قىلىمەن !!! مەقسىدىم ھەرگىزمۇ ،ئۇنداق بۇنداق كىسىلى بارلار چوقۇم توي قىلماسلىق كىرەك دىمەكچى ئەمەس،بەلكىم ئەستايىدىل ئويلىنىڭ دىمەكچى ... مەقسىدىمنى تولۇق ھەزىم قىلىپ بولۇپ ئاندىن پىكىر قاتناشتۇرۇڭ !!!

بۇ توغۇرلۇق تەپسىلى توختالماقچى ئىدىم،ۋاقىت مۇناسىۋىتى بىلەن ئاساسلىق تۈپكى مەسىلىنىلا سۆزلەپ ئۆتتۈم،قالغىنىنى كومپىيۇتۇر ئالدىدا ئولتۇرغان قارا كۆز قېرىنداشلىرىمىزغا قويۇپ قويدۇم،ئەقلى- ھۇشى جايىدا قېرىنداشلىرىمىز مەقسىدىمنى چوقۇم توغرا چۈشىنىدۇ...

ئەسسالامۇ ئەلەيكۇم تورداش؟بۈگۈن مۇنبەرگە كىرىپلا تۇنجى دېققىتىمنى تارتقىنى يۇقارقى تېمىڭىز بولدى، چۈنكى ماۋزۇنى كۆرۈپلا ئەلۋەتتە تېخى تۇرمۇش قۇرمىغان بىر ياش ئۈچۈن بۇ كۆرۈپ بېقىشقا تېگىشلىك تېما دەپ ئويلايمەن، خۇددى سىز دەپ ئېيتىقاندەك يۇقارقىدەك ئەھۋاللار ھەقىقەتەن ئەتراپىمىزدا مەۋجۇت. تېخى ئەڭ قورقۇنۇچلىقى ھەتتاكى ئۆزىنىڭ ئەيدىز كېسىلى ئىكەنلىكىنى بىلىپ تۇرۇپمۇ يەنە تېخى توي قىلىش ئۈچۈن قارشى تەرەپكە بىلىندۈرمەي توي تەييارلىقىغا چۈشكەنلەرمۇ بار...بۇنى پەقەت سالامەتلىك تەكشۈرۈشتىن ئۆتكەندە بىلگەنلەر ئاھۇ-زار قىلىپ توينى توختىتىدۇ، ئامال-چارىلەر ئارقلىق بىلىندۈرمەي توي قىلۋالغانلار  بولسا يەنە بىر ساغلام كىشىنى  ئۆزى تۈپەيلى شۇ نىجىس كېسەلگە مۇپتىلا بولىدۇ.
ئويلاپ باقايلى مەيلى توي بولماي توختاپ قالسۇن ياكى توي بولۇپ ئاقىۋىتى ئاياغلاشسۇن كىمگە زىيان ؟كىمگە پايدا؟مېنىڭچە بۇ بەكلا شەخسىيەتچىلىك...
تېما ئىگىسى دىگەندەك تۇرمۇش قۇرۇش ئىشىدا ئويلىنىشقا تېگىشلىك كېسەللىك سەۋەبىدىن باشقىمۇ نۇرغۇن مەسىلىلەر بار....شۇڭا بۇ يولغا قەدەم ئېلىش ئالدىدا تۇرغان ھەر بىر كىشى ئالدى بىلەن ئوبدان ئويلىنىپ ئاندىن ئاۋايلاپ قەدەم ئېلىشى لازىم.
ئۇنتۇپ قالمايلى ئېيتىلىشىدىنمۇ ئىنىقكى تۇرمۇشنىڭ كەينىدە بىر "مۇش"بار.شۇڭا كاللىمىزدا چوقۇم ئوبدان تەييارلىق بولىشى شەرت.
ھەر بىر خەت ، ھەر جۈملىدە ھەقىقەت بار!
