پۇتبول مۇنبىرىدىكى بىر دوستىمىز بارى بىلەن تىلفۇندا كۆرۈشۈپتىكەن، شۇڭا تىمىنى بۇيەرگە ئىلىپ كەلدىم:


2011-يىلى3-ئاينىڭ27-كۇنى جۇڭگۇ ئولىمپىك كۇماندىسى بىلەن كۇرىيە ئولىمپىك كوماندىسى بىر مەيدان بەدەن قىزىتىش مۇسابىقىسى ئىلىپ بىرىپ،بارى بىلەن مىرەخمەتجان بۇ مۇسابىقىگە قاتنىشىپ قايتىپ كەلدى.بۇ مەيدان مۇسابىقىنىڭ ئاساسلىق مەقسىتى لوندون ئولىمپىك تەنھەركەت يىغىنىنىڭ ئاسىيا رايۇنى پۇتبول تاللاش تۈرىدىكى مۇسابىقە ئۇچۇن قارشى تەرەپنى سىناشنى ئاساسى مەخسەت قىلغان ئىدى.بۇ مەيداندا جۇڭگۇ ئولىمپىك كوماندىسى 1گە قارشى 0بىلەن ئۇتتۇرۇپ قويدى.بارى بىلەن مىرئەخمەتجان ئاساسلىق ئەزا بولۇپ مەيدانغا چىقىپ كىيىنكى مەيداندا ئالمىشپ ئارام ئىلىشتى.تۇنۇگۇن ماھىرلارنىڭ ھەممىسى ئۆزلىرى تۇرۇشلۇق كۇلۇپلارغا قايتىپ كەلدى.ھەرقايسى تورلاردا بۇ قىتىمقى مۇسابىقە شۇنداقلا بارى بىلەن مىرەخمەتجان  توغرىسىدا بىر قسىم خەۋەرلەر بىرىلدى،مەن مۇشۇ ماھىرلارنىڭ ئۆز ئاغزىدىن ئىشەنچىلىك خەۋەر ئىلىش مەقسىتىدە تۈنۈگۈن بارى بىلەن تىلىفۇندا كۇرۇشۇپ،ئۇنىڭ كۆزقاراشلىرىنى ئاڭلاپ باقتىم.تۆۋەندە شۇ پاراڭلارنىڭ بىر قىسمىنى سىلەرگە سۇنماقچى.  
  ئالدىنقى كۇندىن باشلاپ ھەرقايسى تورلاردا مۇسابىقەڭلەر توغرىسىدا كۆپ خەۋەرلەربىرىلدى،سىز ئۆزىڭىزنىڭ ئىپادىڭىزھەققىدە سۆزلەپ باقامسىز؟  
  بارى.بۇ مەيدان مۇسابىقىدە مەن زادىلا ئوينىيالمىدىم.مۇسابىقە باشلىنىش ئالدىدىكى ئورونلاشتۇرۇش بويىچە پىياۋچىڭ ئىككىمىز ھۇجۇمدا ماسلاشتۇق،ئەمەلىيەتتە ئەڭ ئالدىدا ئويناش مىنىڭ ئالاھىدىلىكىمگە ماس كەلمەيدو.شۇ سەۋەپ مۇسابىقە باشلىنىشتىن ئاۋال تىرنىرغا پىكىرىمنى ئىيتتىم.تىرنىر ئاۋال ئالدى ھۇجۇم بولوپ ئويناپ،ئەھۋالغا قاراپ كەينىگە يىنىپ سول ئالدى مەيداندا ئۆزۇمنىڭ ئەسلى ئورنۇمغا يىنىپ كىلىپ ئالمىشىپ ئوينىشىمنى تاپىلىدى.تىرنىرنىڭ ئاساسلىق مەقسىدى مىنىڭ تۇختىماي يۇگۇرۇش ئالاھىدىلىكىمدىن پايدىلىنىپ،قارشى تەرەپ بىكلىرىنىڭ خاتالىقىدىن پايدىلىنىپ توپنى تارتىپ ئىلىشنى،شۇنداقلا پۇرسەت بولسىلا قاپقا توپ ئۇرۇشنى مەخسەت قىلغان.ئەمەلىيەتتەھۇجۇم ئۇينىغاندا ھۇجۇمغا پەقەت توپ يەتمىدى،پاسلارنىڭ ھەممىسى ئۇتتۇرا مەيداندا ئۈزۈلۇپ قالدى.كەينىگە يانغاندا بولسا ھۇجۇم ئاجىزلاپ قالدى،ئەسلىدە مەن ئالدى ھۇجۇمچىنىڭ سەل ئارقىسىدا تۇرۇپ ئوينىسام مىنىڭ ئالاھىدىلىكىم ئىچىلاتتى.بۇ مەيدان مۇسابىقىدە ئىپادىسى ئەڭ ياخشى بولغىنى پىياۋچىڭ،ئۇنىڭ تىخنىكىسى ئالاھىدە،ئالداپ ئۆتۇشكە ماھىر،مەن بولسام ھۇجۇمچىنىڭ كەينىدىن قىستىلىپ كىرىپ توپ ئىلىشقا بىر قەدەر كۆنگەن.شۇ سەۋەپ مىنىڭ بۇ مەيداندىكى ئالدى ھۇجۇمدىكى ئىپادەم ياخشى بولمىدى.   
سىز ئالدىنقى قىتىم خاڭجۇ كۇلۇبى بىلەن كورىيەگە بىر قىتىم بىرىپ كەلگەن،بۇ قىتىم يەنە باردىڭىز.بۇ قىتىمقى مۇسابىقە بىلەن ئالدىنقى قىتىملىق مۇسابىقىنىڭ پەرقى بارمۇ؟  
بارى.ئالدىنقى قىتىم ئاسىيا پۇتپۇلچىلار بىرلەشمىسىنىڭ ئورۇنلاشتۇرلۇشى بويىچە بولغاچقا  ياخشى مەشىق ۋە باشقا شارائىتلارغا ئىگە بولغان .بۇ قىتىم بىز كۇرىيە ئولىمپىك كۇماندىسىنىڭ تەكلىۋى بىلەن بارغان.ئالدى بىلەن ياتاق ۋە مەيدان ئورۇنلاشتۇرۇشى بىزگەئازراق قولايسىز بولۇپ قالدى،يەنى ئايدۇرۇمدىن بىز چۈشىدىغان مىھمانخانىغا تەخمىنەن بىر سائەتتە بارىدىكەنمىز،ئەتىسى مۇسابىقە بولىدىغان مەيدانغا بىز چۈشكەن مىھمانخانىدىن بولغان ئارىلىق يەنە بىر سائەتلىك يول ئىكەن.شۇ سەۋەپ مەشىق  قىلالمىدۇق.چۈشتىن بۇرۇن 11 يىرىمدا تاماقنى يەپ مەيدانغا بارغۇچە بىر سائەت كەتتى.مۇسابىقە باشلىنىشتىن ئاۋال مەيدان يىنىدىكى بىر مەكتەپنىڭ پۇتبول مەيدانىدا بەدەن قىزىتقان بولدۇق.شۇنىڭ بىلەن مۇسابىقە باشلىنىپ كەتتى.ئاقىۋەت يىڭەلمىدۇق.  
  سىز ھازىر دىيارىمىزدىكى داڭدار پۇتبولچى بولۇپ قالدىڭىز،دىيارىمىزدا يەنە سىزنىڭ نۇرغۇن بىللە توپ ئوينىغان بۇرۇنقى سەبداشلىرىڭىز بار،ناۋادا ئۇلارمۇ ئىچكىرىگە بىرىپ قالسا مەيدانغا چىقالامدۇ؟   
بارى.چوڭ كۇلۇپلار بىلەن مۇنتىزىم بولمىغان كۇماندىلارنىڭ پەرقى ھەقىقەتەن بار،شىنجاڭلىق ماھىرلارنىڭ تىرىشسىلا ئىستىقبالى بار،يەنە نۇرغۇن ماھىرلار كىلىپ مەشىققە قاتنىشىپ باقسا بولىدو.ئەمما مەشىق مۇسابىقىسى بىلەن ھەقىقىي مۇسابىقىنىڭ پەرقى بەكمۇ چوڭ.مەنمۇ ئۆتكەن يىلى كەلگەندە مەشىق مۇسابىقىللىرىدە كۆپ توپ ئۇرغان،ئەمما ھەقىقى مۇسابىقىدە بولسا تىركىشىش بەكمۇ كۈچلۈك،توپ ئۇرماق تەس،شۇ سەۋەپ مەنمۇ ئارانلا بىر توپ ئۇردۇم.مەيدانغا چىقىشنىلا قوغلىشىش مۇھىم ئەمەس ،ئەڭ مۇھىمى مەيدانغا چىققانغا لايىق ئويناش.شۇڭلاشقا  تىخنىكىسىنى كۇچەيتىشكە ئەھمىيەت بەرسە ئاخىرى چۇقۇم كىلەلەيدۇ  ۋە كەلسە ئوينىيالايدۇ،
ھازىر ئىچكىرىدە بەش بولوپ قالدىڭلار،شىنجىن كۇماندىسىكى شىنجاڭلىق بالىلارغا قانداق قارايسىز؟
بارى. دىلمۇرات،يەھيا،ئابدۇۋەلىلەرمىنىڭ بۇرۇنقى سەبداشلىرىم.ئۇلارنىڭ ئەمىلى ماھارىتى يۇقىرى. ئۇلارچىڭدۇدا بىر مەزگىل تۇرۇپ قالدى،ناۋادا ئۇلار كىلىمىز دىسىلا  باشقا كۇلۇپلار تالىشىپ ئالاتتى.مەسلەن ئابدۇۋەلىنىڭ مۇسابىقىسىنى بىزنىڭ تىرنىر كۇرۇپ بولغاندىن كىيىن مىنى ئىزلەپ كىلىپ،«ئاشۇ بالىغا خىزمەت ئىشلە،بىزنىڭ خاڭجۇغا كەلسۇن»دەپ جىق دەپ كەتتى.ئابدۇۋەلى كەلسە ئىنىقلا سول بىك بولوپ ئوينايتتى.ئەمما ئابدۇۋەلى كۇماندىمىزدىكى سۇل بىك رۇڭ خاۋنىڭ دۆلەت كۇماندىسىنىڭ ئەزاسىلىقىنى بىلگەچ كەلسە ئورۇنباسار ئەزا بولۇپ قىلىشتىن ئەنسىرىدى.ئاڭغىچە ئۇلار شىنجىنغا كىتىپ قىلىشتى.بىزنىڭ تىرنىر ھازىرغىچە ئابدۇۋەلىنىڭ گىپىنى قىلىدو.شۇنداق كۇماندىغا ئەكىلىۋالغۇسى بار ئىدى.بۇ قىتىم دۆلەت ئولىمپىك كوماندىسىنىڭ تىرنىرى بۇلازىۋىچمۇ مىنىڭدىن ئابدۇۋەلىنى سوراپ يۈرۈيدۇ.دىلمۇرات بىلەن يەھيامۇ ئىنىقلا باشقا كوماندىلارمۇ كۆز تىككەن بالىلار.شۇڭلاشقا بۇلارنىڭ چۇقۇم مۇسابىقە مەيدانىدىكى ئىپادىسى ياخشى بولىدۇ. ;
