ئەسسالامۇ ئەلەيكۇم مىسرانىم تور بېكىتنىڭ ئەزىز كۆرۈرمەنلىرى سادىق قوللىغۇچىلىرى!
مۇنبىرىمىزنى ئۇيغۇر تور دۇنياسىدىكى ئەڭ ياخشى بايراقدار مۇنبەر قىلىپ قۇرۇپ چىقىش ئۈچۈن باشتىن تارتىپ بىز بىلەن تەڭ ئىلگىرلەپ بىزگە قولمۇ قول ياردەم قىلىپ ،مۇنبەرنى ئاۋات قىلىپ كېلىۋاتقان ئومومىي تورداشلار ساداسىغا دىققەت قىلىپ ۋە يۈكسەك ئېتىبار بېرىپ يۇقارقى ئۇقتۇرش يولغا قويۇلدى.(يېقىندىن بۇيانقى ئالاھىدە ئىشلارنى كۆزدە تۇتۇپ بىر قىسىم ئۇقتۇرۇشلار كۆپرەك يوللىنىپ كەتتى،توغرا چۈشەنگەيسىلەر).

باشقۇرغۇچىلار مۇنبەردە تېمىلارنى رەتلەش ،تەستىقلاش،مۇنبەر تەرتىۋىنى ساقلاشتا مۇھىم رول ئوينايدۇ ..باشقا ھوقۇقلاردىن ئەزالاردەك تەڭ بەھرىمان بولىدۇ. بىر قىسم باشقۇرغۇچىلىرىمىز مۇنازىرىدە يېڭىلىپ قېلىش ياكى ھەل قىلالماسلىق بىلەن بىر قىسىم ئەزالارنى چەكلەش ھەم قاتتىق چارە كۆرۈش قاتارلىق مەشغۇلاتلارنى قىلغىننى ئارقا سەپ خاتىرىسىدىن ئېنىق كۆردۇق. مىسرانىم خىزمەتخانىسىدىكى خادىملار بۇ ئىشلار توغۇرلۇق قاتتىق ئويلىنىپ بىر قىسىم باشقۇرغۇچىلارغا ئالاھىدە ئەسكەرتىش بەردۇق.

مۇنبەر -- مۇنازىرە ... بۇ بىر ئەركىن پىكىرلىشىش سەھنىسى . شەخسىي ئىددىينى ھەممىگە زورلاپ تېڭىش ئەمەس،ئەركىن تەپەككۇر جەريانىدا ئۆزىدىكى يېتەرسىزلىك ،پىكىر ئاجزلىقىنى تەبىي تونۇتۇش خالاس..
شۇنداق بولغان ئىكەن مەن خاتالاشقان ئىكەنمەن،قارشى كۈچلەر ،مېنىڭ ئاجزلىقىمنى دەل ۋاقتىدا تاپالىسا شۇنىڭ ئۆزى كۇپايە. شۇنىڭغا ئوخشاش بىر قىسىم ئەزالار باشقۇرغۇچىلارنىڭ يېتەرسىزلىكلىرىنى سەمىمىي ياكى مەلۇم ۋاستىلەر بىلەن كۆرسىتىپ بېرىۋاتىدۇ.. بىراق بىر قىسىم باشقۇرغۇچىلىرىمىز بۇ ئىشلاردا سەل قېنى ئىتتىكلىك قىلىۋاتىدۇ. بىز يۇقارقى ئىشلارنى كۆزدە تۇتۇپ ،ئالدىنقى قېتىملىق چىقارغان بەزى بىر قارارلىرىمىزنىڭ بەكلا يۈزەكى بولۇپ قالغانلىقىنى ھېس قىلدۇق ھەم ئالدىنقى تېمىدا دېيىلگەن 3 نەپەر ئەزانى ئەسلى پېتى قالدۇرۇپ ،ئالىي ئەزالىق ئۇنۋانى بارلارغا شۇ ئۇنۋاننى ۋە بىر قىسىم مېداللارنى ۋاقتىدا قايتۇرۇپ بەردۇق.

ئەڭ ئاۋال ئۆزىمىزدىكى ئىللەتنى تۈزىشىمىز ،ئۆزىمىزنىڭ بىلىم قۇرۇلمىمىزنى ئۆستۈرۈپ، ئىلمىي مۇنازىرىلەر بىلەن ئەزالارنى قايىل قىلىش كېرەكلىكىنى يەنە بىر قېتىم ئىلگىرلىگەن ھالدا چۈشىنىپ يەتتۇق .

شۇنىڭ ئۈچۈن ئالدىنقى قېتىملىق ئەزالارغا بولغان تەلەپنى ۋاقتىدا ئۆچۈرۈپ ، ئەڭ ئاۋال مۇنبەر ھەركەتلەندۈرگۈچ كۈچىدە مۇھىم ۋەزىپە ئۆتۈپ كېلىۋاتقان باشقۇرغۇچىلار قوشۇنىنى سەرخىللاشتۇرۇشنى خىزمەتلەرنىڭ ئەڭ ئالدىغا قويدۇق.

ئەزالارنىڭ ئالدىنقى بىر قانچە كۈن ئىچىدە كۆڭلى رەنجىپ قالغان يەرلەر بولسا ۋە ياكى بىر قىسىم ئىشلار تۈپەيلى چېچىلىپ قالغان بولسا ، كەڭ قورساقلىق ،ئەپۇچانلىق تۇيغۇسىدا تۇرۇپ ئۆز - ئارا يول قويۇشىنى ئۈمۈد قىلىمەن. شۇنداقلا بىر قانچە كىشىلەر بىلەنلا چىقىرىلىپ قالغان بىر قىسىم يەكۈن ھەم ئۇقتۇرۇشلارنى توغرا ھەزىم قىلىشىنى تۆۋەنچىلىك بىلەن ئۈمۈد قىلىمەن.

_________________

مىسرانىم باشقۇرغۇچىلىرىنى تۈپتىن رەتكە سېلىپ ،ئەڭ ياخشى باشلامچىلارنى مۇنبەر باشقۇرغۇچىلىق ۋەزىپىسىگە ئولتۇرغۇزۇپ ،، ئىناقلىقنى كۈچەيتىش ئۈچۈن خىزمەت گۇرۇپپىمىز بىر قىسىم ئەمەلىي قوللىنىدىغان تۈزۈملەرنى تۈزۈپ چىقتۇق.

1- باشقۇرغۇچىلار ئۆز ئالدىغا تېمىلارنى نادىرلاپ ياكى تەۋسىيەلىشى قاتتىق چەكلىنىدۇ. چوقۇم مەسلىھەت بىلەن ئىش قىلىشى كېرەك.
2- بىلىم قۇرۇلمىسى ھەم ياش چەكلىمىسىگە ئاساسەن مۇنبەردىكى بىر قىسىم باشقۇرغۇچىلارغا قەرەللىك سۆھبەتلەرنى ئېلىپ بارىمىز..
3- ھەر جۈمە كەچ سائەت 7 دە باشقۇرغۇچىلار سۆھبەت يىغىنى ئۆتكۈزۈلىدۇ.
4- قەرەللىك ھالدا باشقۇرغۇچىلارغا بېلەت تاشلاش تېمىسى يوللىنىپ ئەزالارنىڭ يۈرەك سۆزلىرى بىۋاستە خىزمەت گۇرۇپپىمىزنىڭ تەھلىل قىلىشى ھەم ئورۇندىشى ئۈچۈن سۇنىلىدۇ.
5- ھەرقانداق باشقۇرغۇچىنىڭ بىرەر مۇنازىرىدە يېڭىلىپ قالغانلىقنى باھانە قىلىپ ياكى ئاقماس ئىشلار ئۈچۈن ئەزالارنى چەكلىشى قاتتىق چەكلىنىدۇ.
6- ھەرقانداق ئەزالارنىڭ باشقۇرغۇچىلارغا سەمىمىي مەسلىھەت بېرىش ھوقوقى بولىدۇ..

مۇكاپاتلاش ھەم جازالاش ئېنىق ئايرىلغان بولۇپ ئىككى قېتىملىق باھالاشتا ئۇدا ئەزلارنىڭ قوللىشىغا ئېرىشەلمىگەن باشقۇرغۇچىلارنى ۋاقتىدا قالدۇرۇش ،شۇنداقلا ئەزالارنىڭ ئىزچىل قوللىشىغا ئېرىشكەن باشقۇرغۇچىلار نى ماددىي بويۇنلار بىلەن مۇكاپاتلاش ئىشلىرىنى ئېلىپ بارىمىز.

مىسرانىم مۇنبىرىمىز ئۆزىنى توختاتقان ،شەخسي ھېسياتنى ئارلاشتۇرمايدىغان ،بىلىملىك كىشىلەرنىڭ داۋاملىق مۇنبىرىمىزدە باش بولۇپ ،ئەزالارنى يېتەكلەپ مېڭىشىنى خالايدۇ...شۇنداقلا شۇنداق ئەزالىرىمىزنىڭ دادىللىق بىلەن بىزگە ياردەملىشىشىنى سەمىمىي ئۈمۈد قىلىدۇ.

ئەزلارنىڭ بارلىق كۆڭۈلسىزلىكنى كۆڭلىدىن چىقىرۋىتىپ ،مىسرانىمنىڭ ئەتىسى ئۈچۈن تېخىمۇ كۈچىشىنى سەمىمىي ئۈمۈد قىلىمىز!

خەلق گويا بىر دەريا( دېڭىز )غا ئوخشايدۇ . ھۆكۈمرانلار بولسا دەريا( دېڭىز ) ئۈستىدىكى پاراخوت-كېمىلەرگە ئوخشايدۇ . خەلق گاھىدا پەرۋاز قىلدۇرسا گاھىدا چۆكۈرۈپمۇ ئالالايدۇ .
باشقۇرغۇچى بولغان كىشىدە شەخسىيەتچىلىك ئىدىيىسى بولسا بولمايدۇ ، بولۇپمۇ مۇنبەردە تېخىمۇ شۇنداقكەن ، نېمىشقا دېگەندە مۇنبەرگە چوڭمۇ كىچىكمۇ ، زىيالىيمۇ دېھقانمۇ ئىشقىلىپ ھەرساھە كىشىلەرنىڭ ھەممىسى كىرىپ ئۆزىگە كېرەكلىك ئۇچۇرلارنى ئىزدەيدىكەن ، يازمىلارغا ئاۋاز قوشىدىكەن . شۇنداق بولغاچقا تەڭپۇڭلۇقنى ساقلاش مۇرەككەپ بىر ئىش . ھەممە كىشى ئاقىلانە بولسا ھەممە نەرسە كۆزىمىزگە چىرايلىق كۆرۈنگەن بۇلاتتى !
ئىشقىلىپ نېمىلا بولمىسۇن ، مىسرانىم مۇنبىرىنىڭ تور دۇنيارىمىزدا كۆڭۈل بېتىمىزنىڭ بىرى بولۇپ بىز بىلەن تەڭ ئالغا ئىلگىرىلىشىگە ئىشىنىمەن .
كىتابلاردىن قانائەت تاپالىسام ئۆزۈمنى بەختلىك ھېسابلايمەن .
