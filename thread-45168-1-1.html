بۇنىڭدىن ئۈچ يىل ئىلگىرى ئۇيغۇرلاردىن يەنە بىر يېڭى كەشپىياتچى چىقىپتۇ دەپ تونۇيدىغانلار ئۆز ئارا ،تونۇمايدىغانلار بولسا توردا تارقالغان خەۋەرلەردىن كۆرۈپ ئالىمجاندىن پەخىرلىنىش تۇيغۇسىدا بولغان ئىدۇق .
   ئالىمجاننىڭ نامى  تارالغىلى   ئۈچ يىلدىن ئارتۇق بولۇپ قالغان بولسىمۇ لېكىن ھۆكۈمەت تەرەپ ياكى يۇقىرى دەرىجلىك ئاخبارات ۋاستىلىرىدا تېخىچە ئۇ  توغرىلىق ھېچقانداق مەلۇمات يوقلىقى ھەقىقەتەن ھەرقانداق ئادەمنى گاڭگىرىتىپ قويىدىغانلىقى ھەممىمىز گە مەلۇملۇق .
توردىشىمىز cctv520ئېيتقاندەك "مانا 3 يىلمۇ ئۆتۈپ كەتتى،بىراق ئۇ توغرىلىق خەۋەرلەر يوقاپ كەتتى...  
ئۇنىڭ ئەھۋالىنى بىلىپ بېقىشقا پېقىرمۇ قادىر بولالمىدى،لېكىن كۆڭلۈمگە ئۇنىڭ كەشپىياتلىرىنىڭ قىممىتى ۋە راست يالغانلىقى توغرىسىدا گۇمانلار پەيدا بولۇپ قالدى. كەشپىياتنىڭ باھالاپ بېكىتىلىش ۋاقتى ئاران ئالتە ئاي، بىراق ساق تۆت يىلدىن بېرى ئۇنىڭ ئېرىشكەن كېنىشكە ياكى باشقا نەرسىلىرى توغرىلىق زادىلا گەپ بولمىدى..."دېگىنىدەك ھەممىمىز شۇئىشلارنىڭ قانداق بولۇۋاتقىنغا قىزىقىۋاتقان بولۇشمىز ياكى تۇرمۇشنىڭ ھەلەكچىلىكىدە ئۇ ئىشلارنىڭ ئالدى كەينىنى ئويلىغۇدەك ھەپسىلىمىز يوق ۋەياكى ئۇنداق بىر ئادەمنىڭ بارلىقىنى تولۇق ئۇنتۇپ كەتكەنلىكىمىز مۇمكىن .لېكن ئۇ ئىشلار ئۇنداق تاشلىنىپ قالغىنى يوق .
ئەمدى ئالىمجاننى مىسالغا ئالساق بەزى كەشپىياتلىرىغا دۆلەتلىك پاتنېت ئىدارسى تەرىپىدىن پاتنېت كېنىشكىسى بەرگەنلىكىمۇ كۆز يۇمۇۋالغىلى بولمايدىغان ھەقىقەت.شۇنداقلا پاتنېت كېنىشكىلەرمۇ  ساق - سالامەت  بار .ئۇلار ھازىر كىشلەرنىڭ قەلبىدىكى  بىر ئىشنىپ بىر ئىشەنمەسلىك ياكى "راستمۇ يالغانمۇ ؟"دېگەن شۇ سوئاللاردىن قۇتۇلۇپ ئەمەلىي نەتىجە ئارقىلىق كىشلەرگە جاۋاب بېرىش مەقسىتىدە، شىركەت قۇرۇپ  رەسمىي مەھسۇلات ئىشلەپچىقىرىپ كىشلەرگە سۇنۇپ نەتىجە ۋە ھەقىقى ئۈنۈم ئارقىلىق كىشلەرنى قايىل قىلماقچى بولغاچقا ،شىركەتنىڭ بەزى بىر ئىشلىرى بىلەن سەل ئالدىراش بولۇپ  قالدى .سىلەرمۇ بىلىسلەر ، بىرەر ئىش قىلىش ئۇنچە ئاسانمۇ ئەمەس ،بولۇپمۇ ..... ئۇنىڭ كەشىپ قىلىپ ياساپ چىققان مەھسۇلاتلىرىمۇ يېقىن كەلگۈسىدە بازارغا چىقىپ كىشلەرنىڭ كۆز ئالدىدا نامايەن بولغۇسى .
مېنىڭ بىلشىممۇ يۇقارقىلار ،قايسى مەھسۇلاتلارنى ئىشلەپچىقىرىدۇ ؟ قاچان بازارغا سېلىندۇ ؟ئۇنى بىلىشكە مەنمۇ ھازىرغىچە قادىر بولالمىدىم ،لىكىن يىراق بولمىغان كەلگۈسىدە چوقۇم بەزى مەھسۇلاتلار  كىشلەرنىڭ كۆز ئالدىدا نامايان بولىدۇ.
