ئەڭ ياخشىسى بۇنداق ئۇنۋىرسال مۇنبەرلەرگە دىنى تىمىلارنى يوللىمىغان ياخشىكەن دەپ ئويلايمەن، چۇنكى بۇنداق ئۇنۋىرسال مۇنبەرلەردە ھەرساھەدىكى كىشىلەر بار، بىرەر دىنى تىما يوللانسا دىندىن قىلچە خەۋىرى يوق ، نەزىريە جەھەتتىن مۇسۇلمان ئەمىلىيەتتە مۇسۇلمان ئەمەس تورداشلار ئۆزى بىلىپ- بىلمەي قالايمىغان سۆزلەپ ئىسلام دىنىمىزنىڭ كۇرسىنى چۇشۇرۋىتىدىكەن، شۇڭا ئۆزىمىزنىمۇ باشقىلارنىمۇ گۇناھكار قىلماي دىنى تىمىلارنى يوللىمايلى، دىنى تىما كۆرىدىغان ئادەم مەخسۇس دىنى مۇنبەرلەر بار شۇنىڭغا كىرىپ كۆرىدۇ، ھەرگىزمۇ دىنى تىما كۆرۈش ئۈچۈن مىسرانىمغا كىرمەيدۇ. مۇشۇنداق دىنى تىمىلارغا چۈشكەن ئىنكاسلارنى كۆرۈپ ئىچىم بەك ئاچچىق بولىدۇ، ئاللاھ تەرىپىدىن توسۇلغان، ئىسلامدا ئىنىق بەلگىلىنىپ بولغان ئىشلارغىمۇ بەزىلەر ئۆزىنىڭ تۇرمۇش ئادىتىگە ئاساسەن پەتىۋا چىقىرىۋالىدىكەن، ئاددىسى ھاراق توغۇرلۇق يوللانغان بىر تىمىدا بىر ئەزانىڭ << ھاراقنى خىزمەت ئىھتىياجى ئۈچۈن ئىچىدۇ، ھاراق ئىچمىگەنلەرنىڭ مەنسىپى ئۆسمىكى تەس، شۇڭا توغرا چۈشۈنۈشكە بولىدۇ>> دىگەن ئىنكاسىنى كۆرۈپ ياقامنى چىشلەپ قالغان ئىدىم، بۇنداق ئىشلار بەك كۆپ، ھەقنى سۆزلىسە <<تورموللىسى>>، <<بۈۋم>> دەيدىغان قالپاقلارنى كەيگۈزۈپ قويغان، شۇڭا دىنى تىما كۆرمەكچى بولساق مەرىپەت، ئۇيغۇر ئىسلام تورى قاتارلىق نۇرغۇن توربەتلەر بار، شۇنىڭغا كىرىپ كۆرەيلى بۇ توربەتلەرگە ئىسلامدىن خەۋىرى بار ھەقىقى مۇسۇلمانلار كۆپ.
بەزىلەر كۆكتە ئۇچۇپ سۇدا ئۈزۈپ كەتتى يىراق،مەن مىسالى يالاڭ ئاياق دەسسەپ تىكەن ماڭارىمەن!
