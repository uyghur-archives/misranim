     ئالدىنقى قېتىملىق بارى توغرىلىق تىمىلاردا كۆپىنچە دوستلىرىمىزنىڭ بارى دىگەن ئىسىمنى ئاڭلاپ، ئۇيغۇرمۇ ياكى باشقا مىللەتمۇ؟ دەيدىغان دەيدىغان قاراشلار پەيدا بوپتۇ، شۇنداق مەنمۇ دەسلىپىدە ئاڭلىغاندا قازاق مىللىتىدىن ئوخشايدۇ دەپ ئويلاپ قاپتىمەن ( چۈنكى بىزنىڭ ئۇيغۇرچە ئىسىملاردا بارى دىگەن ئىسىم كۆپ قوللانمايدۇ) شۇڭا ئۇنىڭ تارىخىنى پۇتبول مۇنبىرىدىن ئىزدەپ تىپىپ بۇ يەردىكى ئەزالار بىلەن ئورتاقلىشىش ئارزۇسىدا بولدۇم.

2008-2009  شىنجاڭ تەنتەربىيە لاتارىيسى كوماندىسى
2010-يىلى   جىجىياڭ  خاڭجۇ يىشىل شەھەر كوماندىسى

     بارى 14 ياشقا كىرگەن يىلى تېلۋۇزۇردىن جۇڭچۇۋەن پۇتبول مەكتىۋىنىڭ ئۇوقۇغۇچى قۇبۇل قىلىش ئۇچۇرىنى كۆرىدۇ، ۋە شىنجاڭدىكى ئوقۇغۇچى قۇبۇل قىلىش ئورنىغا بىرىپ سىناق مۇسابىقىسىگە قاتنىشىپ تالانتىغا ۋە تېخنىكىسغا تايىنىپ، شىنجاڭ رايۇنىدىن 1- بۇلۇپ ئىمتىھاندىن ئۆتۈپ تاللىنىدۇ. 16 ياشقا كىرگەن يىلى 10-نۆۋەتلىك مەملىكەتلىك تەنھەركەت يىغىنىغا قاتنىشش پۇرسىتىگە  ئىرىشىپ، شىنجاڭ كوماندىسغا ۋاكالىتەن مۇسابىقىگە چۇشىدۇ . ۋە مۇسابىقە مەيدانىدىكى نۇرلۇق يۇلتۇزغا ئايلىنىدۇ .  شىنجاڭ كوماندىسى قەدەممۇ - قەدەم  غەلبە قىلىپ ھەل قىلغۇچ مۇسابىقىگە تاللىنىش مۇسابىقىسنى ئويناۋاتقاندا، بارى ھالقىلىق بىر توپنى كىرگۈزۈپ  ھەممىنىڭ دىقىتىنى تارتىدۇ ۋە شۇ مۇسابىقىدە بارىنىڭ ھالقىلىق كىرگۇزگەن توپىغا تايىنىپ شىنجاڭ كوماندىسنىڭ 46 يىللىق  مەملىكەتلىك تەنھەركەت يىغىنى مۇسابىقىسنىڭ پۇتبول تۈرىنىڭ ھەل قىلغۇچ مۇسابىقىگە قاتنىشىش چۈشىنى رىئاللىقا ئايلاندۇرىدۇ . بىر توپ بىلەن نام چىقىرش يۇلىنى كۆپ قىسىم پۇتبول مەستانىلىرى يىقىندا ئۇنىڭ ئاسىيا سەپىرىدە باشلانغان دەپ قارىشى مۇمكىن. دەل شۇ مۇسابىقىدە شۇ ھالقىلىق توپى بىلەن شىنجاڭنىڭ  يۇقىرى سەۋىيدىكى پۇتبول تۇرىدە ھەل قىلغۇچ تارىخىنى يارىتىپ شىنجاڭدىكى   نۇرغۇن توپ مەستانىلىرىنىڭ جۈملىدىن، مەملىكەتتىكى ھەرقايسى ئاخبارات  ۋاستىلىرى، ھەرقايسى كۇلۇپ ۋە تېرنىرلارنىڭ دىققىتىنى تارتىدۇ  ۋە بىر توپ بىلەن نام چىقىرىش سەپىرىنىڭ تۇنجى قەدىمىنى  باسىدۇ .

     ئىككى يىلدىن كىيىن يەنى بارى 18 ياش ۋاقتىدا شىنجاڭ  تەنتەربىيە  لاتارىيسى پۇتبول كوماندىسنىڭ ئاساسلىق ئەزاسى سۇپىتىدە مۇسابىقىگە چۈشىدۇ ،ئىككى يىللىق جۇڭگۇ Bدەرىجدىكى پۇتبول كوماندىسىدا توپ  تەپكەن بارى بىرلەشمىدە  B دەرىجىدىكى يەنە بىر كوماندا خاڭجۇ كوماندىسى بىلەن بولغان مۇسابىقىدە بارىنىڭ ئىپادىسى ناھايتى ياخشى بۇلىدۇ ۋە مۇسابىقىدە قارشى تەرەپنىڭ 2 مۇداپىيچىسىنى چىرايلىق ئالداپ ئۇتۇپ  سەبدىشىغا چىرايلىق پاستىن بىرنى بىرىدۇ ، توپ كىرىدۇ، شىنجاڭ كوماندىسىمۇ مۇسابىقىنىڭ غەلبىسىنى قولغا كەلتۇرىدۇ . شۇ ۋاقىتتا خاڭخۇ كوماندىسنىڭ باش تېرىنىرى  بۇيى ئانچە ئىگىز بولمىغان،  لېكىن تېخنىكىسى ۋە  پارتلاش كۇچى يۇقىرى بۇ ئۇيغۇر يىگىتىگە  قىزىقىپ قالىدۇ . بىر پەسىللىك مۇسابىقىدىن كىيىن شەنشى كوماندىسى بارىنى قۇشىۋىلىش تەكلىۋىنى سۇنىدۇ لىكىن  خاڭجۇ كوماندىسىنىڭ كۇچلۇك  تەلىۋى بىلەن بارى خاڭجۇ كوماندىسنىغا  قىتىلدۇ .
     2010 - يىلى 4-ئاينىڭ 26 - كۇنى جۇڭگۇ بىرلەشمىسى  جىجياڭ يىشىل شەھەر كوماندىسىغا ۋاكالىتەن  ئاساسلىق ئەزا سۈپىدە مەيدانغا چۈشكەن.
     بارى كوماندىغا ۋاكالىتەن توپ كىرگۈزۈپ غەلبىنى قولغا كەلتۇرەلمىگەن بولسىمۇ،  لىكىن ئۇ جۇڭگۇ بىرلەشمە مۇسابىقىسدە  مەيدانغا چۇشكەن تۇنجى ئۇيغۇر ئوغلانى ۋە جۇڭگۇA دەرىجىلىك بىرلەشمىسدە ئاساسلىق ئەزا سۇپىتىدە مەيدانغا چۇشكەن تۇنجى ئۇيغۇر ئوغلانى بۇلۇپ قالغاندى . باشقا يۇرتتا ئۆزىنىڭ مەۋجۇتلىقىنى تۇنىتىش، ئۆز تىرىشچانلىقىغا تايىنىپ،  ئۆز قىممىتىنى يارىتىش ھەر قانداق ئادەمنىڭ قايىللىقىنى قوزغايدۇ  .
 
     2010-يىلللىق جۇڭگۇ بىرلەشمىسى 5- ئايلانما  مۇسابىقىسدە  جېجياڭ خاڭجۇ  يىشىل شەھەر كوماندىسى بىلەن تىيەنجن كوماندىسى  مۇسابىقىلىشىدۇ، مۇسابىقىدە بارى 32-نۇمۇرلۇق كىيىمنى كىيىپ ئاساسلىق ئەزا سۈپىتىدە  مەيدانغا چۇشىدۇ  . ئالدىنقى مەيداندا  ئىككى كوماندا تەڭلىشىپ قالىدۇ. كىيىنكى مەيدان باشلىنىپ 53 مىنۇت بولغاندا بارى  ئوتتۇرا مەيداندىن  چەكلەنگەن رايۇنغىچە تىز سۈرئەتتە توپ ئىلىپ كىرىپ  چەكلەنگەن رايۇن ئىچىدىكى  مۇداپىئەسىز تۇرغان سەبدىشىغا چىرايلىق پاستىن بىرنى بىرىدۇ، لىكىن سەبدىشى قۇرۇق ۋارتاغا بۇلۇڭ تالىماي توپ ئۇرۇدۇ، توپنى ۋارتار تۇتىۋالىدۇ . بۇ بىر قىتىملىق مۇسابىقە جېجياڭ كوماندىسنىڭ مۇسابىقىدىكى ئەڭ ئۈنۈملۈك ھۇجۇم قوزغىشى ئىدى.  مۇسابىقە 0-0 ئاخىرلاشقان بولسىمۇ بارى مۇسابىقىدىكى ئەڭ يۇقىرى نۇمۇرغا ئىرىشىدۇ . بارىنىڭ شىنجاڭدىكى تىرنىرى پەرھات ئاخبارات ۋاستىلىرنىڭ زىيارىتىنى قۇبۇل قىلىپ بارىنىڭ تىخنىكىسى ، بۈسۈپ كىرىش ئقتىدارى ياخشى. ئۇنىڭ دۆلەت كوماندىسغا كىرىش ئىقتىدارى بار دەپ بىلدۈرگەن ئىدى .

 
     2011-يىلى 3-ئاينىڭ 1-كۇنى ئاسىيا چىمپىيۇنلۇق بىرلەشمىسى كىچىك گۇرۇپىدىكى مۇسابىقە يەنى F  گۇرۇپىدىكى مۇسابىقە ئوينالدى ،سىرىق ئەجداھا تەنتەربىيە سارىيدا  يەنى جىجىياڭ كوماندىسىنىڭ مەيدانىدا  جىجىياڭ كوماندىسى ياپۇنىينىڭ چىمپىيۇن كوماندىسى ئەمىلى كۇچى كۇچلۇك  ئاسىيا  چىمپىيۇنلۇق بىرلەشمىدە  چىمپىيۇن بولۇش ئىھتىمالى چوڭ دەپ قارىلىۋاتقان  كوماندا بىلەن مۇسابىقىگە چۇشىدۇ . مۇسابىقە ئالدىنقى مەيداندا ئىككى كوماندا تەڭلىشىپ قالغان ئەھۋالدا كىيننكى مەيدان باشلىنىپ 59 مىنۇت بولغاندا جىجىياڭ كوماندىسى ئادەم ئالماشتۇرۇپ بارىنى مەيدانغا چۈشىرىدۇ  . 1 مىنۇت بولا بولماي بارى  سەبدىشى راملىسقا چىرايلىق ياردەمچى پاستىن بىرنى تاشلاپ سەبدىشنىڭ نۇمۇر ئېلىشغا كۈچ قۇشىدۇ .  86 مىنۇت بولغاندا بارى  ئوڭ تەرەپتىن  توپ ئىلىپ مىڭىپ تۇيۇقسىز ۋارتارغا توپ ئۇرىدۇ  ئۇ توپنىڭ بۇلۇڭ تاللىشى  ئادەمنىڭ قايىللىقىنى قوزغايدۇ  شۇنىڭ بىلەن جىجىياڭ كوماندىسى ئاسىيا چىمپىيۇنلۇق بىرلەشمە مۇسابىقىسنىڭ  غەلبىسنى قولغا كەلتۇرىدۇ شۇنىڭ بىلەن بىرگە بارى جۇڭگۇ تەنھەركەت  ئاخبارات  ۋاستىلىرنىڭ  ۋە ئاسىيا رايۇنىدىكى قىزىق نۇقتىغا ئايلىندۇ ۋە بىرتوپ بىلەن نام قازىنىش  سەپىرىنىى داۋاملاشتۇرىدۇ .  بۇ دەل كىلىپ قىلىش ياكى  تەلەيمۇ ؟ ياق بۇ  بارىنىڭ  كىچىكىدىن جاپالىق مەشىقىنىڭ ۋە تىخنىكىسنىڭ پۇتبول مەيدانىدىكى  ھەقىقى نامايەن بولىشى ئىدى.
  
 
، بۇنىڭ بىلەن بارى ئەگەر  يىتەرلىك ۋاقىت بولسىلا ئۆز تىخنىكىسىنىڭ ئەمىلى كۈچىنى ھەقىقىي كۆرسىتەلەيدىغانلىقىنى ئەمىلىي ھەركىتى بىلەن ئسپاتلىشى ئىدى . ئۆتكەن يىلى 10-ئاينىڭ 31-كۇنى جېجياڭ كوماندىسغا ۋاكالىتەن تۇنجى توپنى كىرگۇزگەنگە قەدەر ھازىرغىچە بارى كەسپى مەشىقكە ئەستايىدىل مۇئامىلە قىلىپ تىرىشىپ  شىنجاڭ رايۇنىنىڭ  پۇتبولچىلىقىنى ئچىكىرى ئۆلىكىلەردە ئاسيا رايۇنىدا  نۇر چاقنىتىش يۇلىدا تىرىشىپ مەشىق قىلماقتا. ئۇنىڭ جاپالىق مەشىقى ۋە تىرىشچانلىقىغا تاينىپ، يىقىندا جۇڭگۇ ئولىمپىك دۆلەت كوماندسنىڭ باش  تىرنىرى بۇلازىۋىچ تەرپىدىن  جۇڭگۇ  ئولىمپىك كوماندىسغا تاللىنىپ دۆلەت كوماندىسغا تاللىنىش ئارزۇسىنى بىرقەدەم ئالدىغا سۇردى . ۋە بىزدەك  شىنجاڭ پۇتبول مەستانىلىرىنى چەكسىز ئىپتىخارلىق تۇيغۇسىغا چۆمدۈردى . بارى  ئاددى تۇرمۇش  كەچۈرۈش ئەھۋالىغا ماسلاشقان تىرىشچان ياراملىق ئىش ئۇقىدىغان يىگىت ئۆز مۇئاشىنىڭ %70 تىنى ئائىلىسگە ئەۋەتىدۇ  . يىقىندا  150 مىڭ يۈئەننى ئۇرۇمچىدە ئۆي ئىجارە ئىلىپ ئولتۇرۇۋاتقان  ئاكىسغا ئەۋەتىپ بىرىدۇ ،بۇ ئش ئاخبارات ۋاستىلىرى ئارقىلىق ئاشكارلانغاندىن كىيىن  مۇخپىرلار ئۇنى زىيارەت قىلغىلى كىلىدۇ  ۋە ئۇنىڭدىن كىيىنكى  ئشلىرى توغرىسىدا سۇئال سورىغاندا، بارى  مەن ھازىر پۇتۇن زىھنىمنى  پۇتبولغا قارىتىمەن،  قالغان ئشلارنى  كېيىنچە ئويلىشىمەن دەپ ئىپادە بۈلدۈردى.
  بارى، خەلقىڭنىڭ سىنىڭدىن كۈتكەن ئۈمۈتلىرىنى ئاڭلاۋاتامسەن؟ بىز  مىرئەخمەت  ،بارى ..... قاتارلىق ئۇيغۇر ئوغلانلىرىنىڭ زەپەر ساداسىنى ئاڭلىغىنىمىزدا قانچىلىك خوشاللىقا چۈمىدىغانلىقىنى ھىس قىلامدىغانسىلەر؟ قىرىنداشلىرىم  تىرىشىڭلار ، بىز سىلەر ئارقىلىق  پەخىرلىنىش تۇيغۇسىنى ئىزدەۋاتىمىز،  بىز سىلەردىن پەخىرلىنىمىز . بۇ ئوغلانلىرىمىز ھازىر باشقىلارغا نىسبەتەن شنجاڭنىڭ پۇتبولچىلىق تەرەقىياتىنىڭ  مەسئۇلى ۋە نامايەندىسى  شىنجاڭنىڭ پۇتبولچىلىق ئىشلىرىنىڭ تەرەقىياتى ياش نوتىلارنىڭ  تەرەقىيات يۆنىلىشى سىلەر ئارقىلىق پايدىلىق شارائىتقا ۋە تىخىمۇ چوڭ پۇرسەتكە ئىرىشكۈسى !!!
