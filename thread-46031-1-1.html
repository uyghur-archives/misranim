    گەپنى قەيەردىن باشلىشىمنى ئۇقمايلا قالدىم، تېما يېزىش ئۈچۈن كومپىيۇتېر ئالدىغا كەلمىگەن بولساممۇ مۇشۇ يازغانلىرىمنى تېما قىلىپ يوللىشىم ئېنىق . ھەربىر ئادەمنىڭ بۈگۈنى ھەم كەلگۈسىگە ئۆتمۈشى چىڭ باغلانغان بولىدىكەن ، ئۆتكەن ئىشلار ئۆتتى ، ئالدىمغا قاراپ ماڭايچۇ دېگەن بىلەن يەنىلا بىلىپ-بىلمەي كەينىمىزگە بۇرۇلۇپ قاراپ قالىمىزكەن ، مەيلى ئۆتمۈش تاتلىق بولسۇن ياكى ئاچچىق بولسۇن بىر قېتىم بولسىمۇ ئەسلەشكە يۈرەكلەر ئىنتىلىدىكەن ، ئىڭەككە ساقال ياماشقاندا سەبىي بالىلىقنى ئەسلىگەن بولساق، ھاسىغا ئېھتىياجىمىز چۈشكەن چاغدا ياشلىق ھەم نەۋقىرانلىقنى باغاش ئەتكۈمىز كېلىدىكەن . ئاخىرىغا كەلگەندە بىزگە كۆپىنچە ھاللاردا ئۇھسىنىش جاۋاب بولۇپ بۈگۈنىمىزگە يېزىلىدىكەن .پۇشايمان قىلىمىزكەن ، خۇددى بۈگۈن تۈنۈگۈن ئۈچۈن پۇشايمان يېگىنىمىزدەك ئەتە چوقۇملا بۈگۈنىمىز ئۈچۈن پۇشايمان قىلىمىزكەن .داڭلىقلارمۇ دەپتىكەن :" خاتالىق ئۆتكۈزۈش نومۇس ئەمەس ، گەپ خاتالىقنى تۈزىتىشتە " ئەمما ھورۇننىڭ پايتىمىسىدەك چۇۋۇلۇپ چىقىۋاتقان خاتالىقلارنىڭ بىرىنى تۈزەتكۈچە يەنە بىرسى ھىجاراپ قاراپ تۇرغان، ئادەم دېگەننىڭ ئىدىيىسىنى كومپىيۇتېرغا سىستېما قاچىلىغاندەك قاچىلىۋەتكىلى بولمىسا يا ، ئەلۋەتتە ئۆمۈرنىمۇ قايتا قوزغاتقىلى بولمايدۇ ، ئۆچتۈق ، تۈگىدۈق ، كۆمۈلدۇق ، جىسمىمىز تۇپراق ، ئوردىمىز قەبرە... 

    ھازىرقى دەۋردىكى تورخۇمارلارنىڭ ئازدىن بىر قىسمى مۇنبەرگە بەند بولماقتا ، ھەر ھالدا كۆپ ساندىكى ئويۇنغا بەند بولغانلاردىن مىڭ ياخشى ، شۇنىڭلىق بىلەن بىرگە مۇنبەرگە ساپلا ساپالىق ، بىلىملىك كىشلەر توپلانغان دېيىشنىڭ ئۆزىمۇ بىر ئەخمەقلىق ، ھەممە بىلىملىك كىشى مۇنبەرگە توپلانغان بولسا ئايرىم -ئايرىم تېما يوللاشنىڭ نە ھاجىتى دەيسىلە . بىرلا تېما يوللاپ قالغىنىنى ئىنكاس ئورۇنلىرىغا يازساق بىر چېكىش بىلەنلا بارلىق يازمىلاردىن بەھىرلەنگىلى بولاتتى ئەلۋەتتە . بۇ يەردە مۇنبەرنىڭ گېپى چىقىپ قالدى ، قاچانلاردا بۇ مۇنبەر ئاتلىق نەرسىگە قىزىقىپ قالغىنىمنى بىلمەيمەن ، مۇنبەرگە مەست ھالدا چىققان ۋاقىتلىرىممۇ بولغان ، تاماكا چىشلەپ تۇرۇپ پەند-نەسىھەتلىك تېمىلارنى ئوقۇغان ، كۆڭۈل ئاچچىق بولسىمۇ يۇمۇرلارغا زورىغا كۈلگەن چاغلىرىممۇ بار ، مەنىۋىي دۇنيادا باشقىلار مېنىڭ قانداق ھالەتتە مۇنبەر كۆرۈۋاتقانلىقىمنى بىلمىگەندىكىن خاتىرجەم ھالدا ئۆزۈمنىڭ مۇقامىمدا توۋلاپ يۈرۈۋەردىم . باش سۈرىتىمنىڭ ئۈستىگە باشقىلار ھازىر كىتاب كۆرۈۋاتىدۇ! دەپ يېزىپ قويۇپ ئۆزۈمنىڭ تۇتامىغا چىققۇدەك بىر قانچە پارچە كىتابنى ئوقۇپ چىقمىغانلىقىمنى يوشۇرۇپ يۈرگۈممۇ يوق . ساماۋىي ھېكايىلەرنى يېزىپ باشقىلارنىڭ ئىنكاسى بويىچە ئۇلارنىڭ كۆزلىرىنى نەمدىگەن بولسام، ئەمما ئەمەليەتتە گۈلۈمگە قانچىلىك بەخت ئاتا قىلغىنىمنى ، ئۇنىڭ ئاھۇ-زارىغا قانچە قېتىم مۇڭدىشىيالىغانلىقىمنى ، ئۇنىڭغا يۆلەك كېرەك بولغاندا قانچە قېتىم مۈرەمگە بېشىڭنى قوي دەپ جىمجىت يۆلەك بولغىنىمنى بىلمەيمەن . ئاھ ئاتام ، ئاھ ئانام دەپ نەچچە يۈزلىگەن مىسرالارنى مۇنبەر يۈزىگە تۆككىنىم بىلەن شۇ قەدىردانلاردىن بىرەر قېتىم :" ئاپا ، دادا ئوغلۇڭلاردىن رازىمۇسىلەر ؟" دەپ سوراپ باقمىدىم . ئۇيغۇرۇم ، مىللىتىم دەپ داۋراڭ سالغىنىم بىلەن پەقەت قۇرۇق ۋىجداننى چىڭ قۇچاقلاپ يۈرۈۋاتىمەن . ئەمەلىيەتتە مەنمۇ بەزىلەردەك شەكىلۋازلىققا چۆكۈۋاتىمەن . بىردىن -بىردىن چۆكۈپ ھالاك بولمىساقلا بولاتتىغۇ ....

   ھەي ، نېمىمۇ يازارمەن، يازغىنىمنى ئۆزۈم قىلمىسام ، يا سىلەر قىلمىساڭلا . بىر تېما يوللىسام 15 مۇنبەر پۇلىغا ئېرىشىدىكەنمەن ، ئۇنىڭدىن كۆرە ئىنكاستىن 4 نى يازسام 1 مۇنبەر پۇلىنى جىق تاپىمەنكەن ئەمەسمۇ . رەسىمنىمۇ رەسىمدىن خەۋىرى بارلار كۆرمىسە، ھىچقانداق قىزىقىشى يوق بىرسىنى مۇنالىزانىڭ ئالدىغا زورلاپ ئولتۇرغۇزۇپ قويغان بىلەن ئۇنىڭدا ھېچقانداق ھۇزۇر ئېلىش تۇيغۇسى بولمايدۇ ، چەتئەل ،چەتئەل دەپ بىرخىل ھەۋەستە تىلغا ئالىمىز ، چەتئەللىكلەر ئەركىن ، ئازادە ؟ ئەجەبا بۆشۈكتىكى بوۋاقنىڭمۇ دەردى بار يەردە ئاشۇ چەتئەللىكلەرنىڭ دەردى يوقمىدۇ . نەچچە يىل ئىلگىرى تېلىۋۇزۇردا بىر چەتئەللىكنىڭ كەينىگە بىر ھارۋىنى چېتىپ دۇنيانى پىيادە ئايلىنىش ئۈچۈن ئاتلانغانلىقىنى كۆرسەتكەندە مەندىكى تۇنجى تەسىرات" چەتئەللىكنىڭ قىلمايدىغان ئىش يوقكەن " بولغان . ھازىرمۇ تۇرۇپ-تۇرۇپ شۇ يېتىمنى ئەسلەپ قالىمەن . ھارۋىسىدا نان، يەنە ئەتكەن چاي ئېتىدىغانغا قايماق ، سۈت تەييارمىدۇ دەپ قالىمەن ، ئۇ چەتئەللىكنىمۇ ئۆزۈمگە ئوخشاش كۈندە بىر قېتىم ئەتكەن چاي ئىچمىسە بېشى ئاغىرىيدۇ دەپ ئويلامدىم ياكى ئۇنىڭغا ئۇيغۇرچە كۆيۈنەمدىم بۇنىسى نائېنىق . 

      ئۇيغۇرنىڭ قورسىقى توق ، كىيمى پۈتۈن بولدى دېگۈچىلەردىن قانچىسى باردۇر دەپ قالىمەن ، مىسالى ئۆزۈمنى ئالسام مائاشلىق تۇرۇپ بەزى چاغدا دەماللىققا كەيگۈدەك سورۇنلۇق كىيىمىمنىڭ يوقلۇقىدىن بېشىمنى تاتلاپ تۇرۇپ قالىمەن ، كىنولاردىكى مەخسۇس كىيىم -كېچەك قويىدىغان ئايرىم ئۆيدىن بىرەرسى بەلكىم تېخى بىرەر ئۇيغۇرنىڭ ئۆيىگە ئورۇنلاشقىنى يوققۇ دەيمەن . نەچچە يىل ئاۋال ئۆيىمىزدىمۇ تاماققا سالىدىغان گۆش قالمىغان كۈنلەر بولغان ، نەچچە ئون يىل ئىلگىرى دادامنىڭ مائاشى چىققۇچە كۈن ساناپ ئولتۇرغان كۈنلىرىمىزمۇ بولغان . مېنىڭ ھەم سىزنىڭ نەچچە ئون يىل ئاۋالقى تۇرمۇشىنى ھازىر داۋام ئېتىۋاتقانلار تېخى ئۇيغۇر جەمئىيىتىدىن يوقالغىنى يوق . ھەر قېتىم ئاشپۇزۇللاردا تاماقلىنىپ قالسام تاماقنى چوقۇم پاكىز يېيىشكە تىرىشىمەن . قورسىقىم بەك ئاچمىغان بولسا تاماقنى جىق ئۇسۇۋەتمەسلىكىنى تاپىلاپ قويىمەن . ئەمما رېستۇرانلاردا ئۈستەلدىكى نېئمەتلەرنىڭ شۇ پېتى قالغانلىقىنى كۆرۈپ تۇرۇپمۇ كۈلۈپ تۇرۇپ يېنىپ چىققان چاغلىرىممۇ ئاز ئەمەس ، ئۇيغۇرنىڭ يۈزى تەڭرى تېغىنىڭ چوققىسىدا ئەمەسمۇ . بوينى بىلەن يۈزى ئاپتاپتا كۆيگەن دېھقانلارنىڭ قان-تەرىنى مەنلا ئەمەس سىزمۇ چەيلەۋەتىسىز . داۋاملىق چەيلەيمىز ، چەيلەپ بولۇپ ناخشا ئېيتالايدىغانلىرىمىز << دېھقان ئاكا >> دەپ ناخشا توۋلىساق، ئۇسسۇل ئوينىيالايدىغانلىرىمىز << داۋانچىڭنىڭ تاۋۇزى تاتلىق >> دەپ پىرقىرايمىز ، ئەلۋەتتە يازغۈچى ئەدىبلىرىمىزمۇ بىر نەچچە مىسرا بىلەن بىرقانچە بۆلەكلىك ھېكايىلەرنى يېزىپ ئۈلگۈرىمىز . چۈنكى بىز خۇدادىن قورقىمىز دەپ قويۇپ ، يۈز-ئىناۋەتنىڭ تۆكۈلىشىدىن ، باشقىلارنىڭ مەسخىرىسىدىن بەكرەك قورقىمىز . يەنە مەندەك << قورققان ئاۋال مۇشىت كۆتۈرىدىغان >> << سەن تاز دېگۈچە ، مەن تاز دەۋالاي >> دەيدىغانلار سەكرەپ چىقىپ مۇشۇنداق يازمىلارنى بەزەنلەرنى مەدھىيلەپ بەزەنلەرنى قامچىلاپ تاتلىق ئىبارىلەرگە ئاچچىق تەنىلەرنى قېتىپ گۈل قىلىپ كەشتىلىۋالىمىز ، ۋاھ مانا ئېسىل خىسلەت ، مانا ھەقىقىي غۇرۇر... 

    ئۆتكەندە مۇنبەردە "سىز نېمە ئۈچۈن توي قىلىسىز ؟ ''دېگەن سوئالدا خىلى ئۇزاق توختاپ قالدىم ، ئۆزۈمگە مۇشۇ سوئالنى قويدۇم ، شۇنچە ئويلىنىپمۇ جاۋابىنى مىللەتكە ياراملىق ئەۋلاد ئاپىرىدە قىلىش ،ئۇيغۇرنىڭ ئۇرۇقىنى بىخلىتىش دېگەندەك يىغىنچاقلىيالمىدىم ، بەلكىم نۇرغۇن كىشلەرنىڭ دەسلەپكى ئويى خاتىرجەم ھالدا جۈپتىنى قولتۇقلاپ مېىڭىش ، خاتىرجەم ھالدا بىرگە ئۇخلاش ، يەنە خاتىرجەم ھالدا بىز دېگەن ئەر-خوتۇن دېيىش بولسا كېرەك . خۇددى بۇرۇن باشلانغۇچ مەكتەپتىكى چاغلاردا ئوقۇتقۇچىلار : " نېمە ئۈچۈن بىلىم ئالىسىلەر؟ " دەپ سورىسا :" ۋەتەن ، خەلق ئۈچۈن " دەپ جاۋاب بېرىپ ، ئازراق ئەقىلگە تولۇپ باشقا تۇرمۇش ھەلەكچىلىكى چۈشكەن چاغدا بۇ جاۋابنىڭ ئورنىغا پۇل تېپىش ، خىزمەتكە چىقىش ، ياشاش ، ئاچ قالماسلىق دېگەندەك جاۋابلارنىڭ ئالماشقىنىغا ئوخشاش . 

    << پۇل تاپقىچە ، يۈز تاپ >> دەيدىغانلاردىن يەنە قانچىسى باردۇ ھازىر ، بىر تال خىش 50 تىيىنگە چىققان، بىر كىلو قوي گۆشى 50 سومغا يېقىنلاشقان مۇشۇ كۈنلەردە مېنىڭ يۈزۈم بارتى ، ماڭا بىر كىلو گۆش بېرىڭ دېسىڭىز قاسساپ يۈزىڭىزگە پىچىقىنى سۈركەپ قويىدۇ . ئەلۋەتتە دەۋرىمىزدىكى ئۇقۇمۇشلۇقلارنىڭ گۆشنى بىكارغا يەپ ئۇزۇن يوللۇق ئاپتۇبۇس ياكى ئايرۇپىلانلارغا بىكارلىق ئولتۇرغىنىنى ئاڭلاپ باقمىدىم . دەرۋازىنىڭ يوچىقىغا قىستۇرۇپ قويۇلغان قېلىن باغاقلارغا يۈزنى قويغىلى بولمىسا يا ، پەقەت ئۇ يۈزىڭىز بىلەن مېھمانغا بارغاندا ئۆينىڭ تۆرىگە قوڭىڭىزنى قويالايسىز . بەلكىم مۇشۇ بۆلەككە قارشى چىقىپ ، ئېسىل ئادەملەر ھۆرمەتلىنىدۇ ، سەن پۇلغا دۈم چۈشۈپسەن دەيدىغانلار چىقىدۇ ، ئەلۋەتتە شۇنداق بولىدۇ ، ئەمما سىزنىڭمۇ مۇشۇ يۈزلۇك كىشلەرگە پۇلىڭىزنى ئىنئام قىلغىڭىز كەلمەيدۇ . مەن پەقەت ماددىي جەھەتتىن سۆزلىدىم ، مەنىۋىيلىك ئۇ بىر كەڭ دۇنيا ئۇنىڭ قول تىققۇدەك قابىليەت مەندە يوق . 

    تاڭمۇ ئېتىپ قالدى ، يېرىم كېچىدە يازما يازسا شۇنداق قالايمىقان سۆزلەيدۇ دەپ قالماڭلا . كۆزلىرىم ئۇيقۇغا ئەسىر بولاي دېگەن بولسىمۇ كاللا يەنە شۇ پېتى ئويغاق ھالەتتە . بەزى چاغلاردا مۇشۇنداق يۈرەكتىكى سۆزلەرنى تۆكۈۋالغاننىڭمۇ زىيىنى يوق . بەرىبىر بىز كوچىلاردا ئۇچرىشىپ قالساقمۇ "ماۋۇ ھېلىقى ئاشكار دېگەن بالا "دەپ تونۇيالمايسىلە ، مېنىڭ خىجىل بولۇشۇمنىڭ زۆرۈرىيىتى يوق، جاھان داپشاقنىڭ ئەمەسمۇ، قېلىنلىق بىلەن نى -نى ئىشلارنى پۈتكۈزگىلى بولىدۇ ، بەلكىم ھە بەلكىم ئۇشبۇ يازمىنى ئوقۇغانلاردىن نەچچىسىدە ، ئون نەچچىسدە، ۋە ياكى يۈز ، مىڭ نەچچىسىدە ماڭا ئوخشاش قۇرۇق ۋىجدان ، غۇرۇر باغرىغا پاتماي تېپىچەكلەۋاتىدۇ ، قويۇپ بېرىڭلا ، مەيلى ئەمەسمۇ ، قويۇپ بېرىڭلا، جاھاننىڭ پەيلى شۇ ، ئىسسىق يوتقانغا كىرىۋالساملا مەن ئۈچۈن ئەتە بولىدۇ ، ئۇخلىۋالساملا مېنىڭ بۈگۈنۈم ئۆتۈپ كېتىدۇ...

   قۇرۇق پاراڭ، ئەمما قىلمىسا بولمايدىغان پاراڭ.
   توۋا ھەممىمىز ئۆزىمىزنىڭ ھەلەكچىلىكىدە ئالدىراش يۈرگەن كۈنلەردە، مىللىي توپ، بىرلىك دېگەنلەر، بىز ۋە ئۆزگە دېگەنلەر ھەققىدە سۆزلەيدىغانلار، ئانچە مۇنچە قۇرۇق پاراڭ سالىدىغانلارمۇ ئازلاپ كەتكىلى تۇردى. لېكىن قىزىق يېرى غەيۋەت قىلىدىغانلار كۆپىيىۋاتىدۇ. بەزىلەر بىر ئالىمىمىز ۋەھشىي بىر نەتىجە يارىتىپتۇ دېسە، قۇرۇق گەپكەن دەپ قاراپمۇ قويمىغان بىلەن، بىر قىسىم ناخشىچىلارنى بولۇشىچە ماختاپ مايكولدىنمۇ ئاشۇرىۋېتىدۇ. توۋا، تۇرمۇش دېگەن ئاجايىپ نەرسە، نۇرغۇن كىشىلەرگە ئىشىنىمىزكەن، دوست بولىمىزكەن. ئەمما شۇ كىشىلىرىمىزنىڭمۇ بىزدىن باشقا دوستلىرى بولىدىكەن. بىز سىرىمىزنى ھېچكىمگە دېمە دەپ ئېيتساق ئۇمۇ دوستلىرىغا ھېچكىمگە دېمە دەپ قويۇپ سىرىمىزنى دەۋېرىدىكەن. مۇشۇ كۈنلەردە سىر ساقلاش قىز ساقلاشتىنمۇ بەك تەس بولۇپ كەتكەندەك ھېس قىلىمەن.
    قانداق قىلىمەن ئەمدى بۇ يەردە قىزلارنىڭ گېپى چىقىپ قالدى، دېسەم گەپ تولا. ئاخىرىغا ئۇلاپ يەنە يىگىتلەرنىڭ گېپى چىقىدۇ. بۇنى باھالاپ ئەخلەت تېمىلار دېگەن تېمىدىكى ئەخلەت تېمىلار...ۋاي-ۋۇي ئىشقىلىپ تولا، تور تۇرمۇشىمىز خۇددى سائەتتەك بىر ئوقتىلا ئايلىنىپ ئۆتىدىغان ئوخشايدۇ، قايسى كۈنى ئەپسۇسلىنىپ چىكىلدىسام، بىر ئاغىنەم، چىكىلداۋېرىدىكەنسەن، سائەتتەك دېۋىدى، ھازىر ئويلىسام بۇ ھەقتە بىر ئىلھام كەلگەندەك قىلىدۇ، غاچچىدە قايتىپ بىر ئىككى مىسرا يېزىپ قوياي. ئۆتۈپ كەتسە بىكار بولىدۇ. https://uyghur-archive.com/misranim/read.php?tid-46034.shtml
   ھە توغرا، ئۇنتۇپ قالغىلى تاس قاپتىمەن، ئاشكار ئەپەندى، سىز غۇلجىلىق ئوخشايسىز، «-ـىزكەن» قوشۇمچىسىنى شېۋە بويىچە ئاپسىز، مەنمۇ خېلى ياقتۇرىمەن، تېخى ئاغىنىلىرىمنىڭ بۇ خىل گەپلىرىگە ئامراقلىقىم كېلىپ دوراپمۇ قويىمەن. بۇنىڭدىن كېيىن يېزىققا ئېلىپ كىرمەڭ. مېنىڭ يازغىنىم چاقچاق قىلىپ، سىزنى دوراپ قويدۇم.
جىممىدە ئۆتكەن ھاياتنىڭ قېتىدا ھېكمەتلىرى. شادلىقى ئوچۇممۇ چىقماس بىر تاغار قىسمەتلىرى.

ئاشكارنىڭ ئاشكارا گەپلىرىگە ئىككى ئېغىز ئاۋاز قوشاي:
   بۇ مەۋھۇم دۇنيادا ۋاقتىنى مۇنبەر ، ئەھمىيەتلىك تور بەتلىرىنى زىيارەت قىلىپ كېلىۋاتقانلار ھەقىقەتەن ئاز دەپ ئېيتىشقا بولىدۇ. مەن تورخانىغا ئانچە كىرىپ كەتمەيمەن، لېكىن،ئۆز جايىمىزدىكى ۋە باشقا جايدىكى تورخانىلارغا كىرىپ ، باشقىلارنىڭ كومپيۇتېردا نېمە ئىش قىلىۋاتقانلىقىنى ئىختىيارسىز كۆرۈپ قالغىنمىدا بايقىشىمچە يۇقىرىدا دېگىنىمدەك ئەھمىيەتسىز بەتلەرنى، بولۇپمۇ ئويۇننى ئويناۋاتقانلارنىڭ كۆپ ساندا ئىكەنلىكىنى جەزىملىدىم. بۇ مەسلىگە قارىتا ، مۇنبەر كۆرىۋاتقان(مەيلى قايسى مۇنبەر بولسۇن) دوسلار ئۆزىمىزگە بىر ئادەمنى قوشىۋالايلى، ئۇلارنى مۇنبەردەك ئەھمىيەتلىك بەتلەرگە باشلايلى، مىسال ئالساق مىسرانىمىدىكى نەچچە ئون مىڭ ئەزا، بىردىن باشلىساق، يەنە نەچچە ئون مىڭ ئادەم ياخشى تەرەپكە ماڭىدۇ دەپ قارايمەن. شوئارنى كۆپ توۋلىغاندىن، ئەمىلىي ئىشتىن بىرنى قىلغان ياخشى.
   مۇنبەر بولسۇن ۋە ياكى باشقا ئالاقە تورى، سۆزلەشكۈلەر بولسۇن، مەۋھۇم ئالاقىدە بىز قارشى تەرەپنىڭ قانداق ئادەم ئىكەنلىكىنى ئۇنىڭ بىزگە قىلغان گەپ - سۆزلىرىدىن بىلىپ بولالمايمىز، دېمەك، بىلمىگىنىمىز ئۇنىڭ يامان ئىكەنلىكىدىن دېرەك بەرمەيدۇ. قارشى تەرەپ ئالىي بىلىملىك، ئەخلاق - پەزىلەتلىك، ھۆرمەتكە سازاۋەر ئادەم بولىشىمۇ، قارا نىيەت، شاللاق، مەدەنىيەتسىز ئادەم بولىشىمۇ مۇمكىن. شۇنىڭ ئۈچۈن مېنىڭ تەشەببۇس قىلىدىغىنىم، مەيلى كىمگە مۇئامىلە قىلايلى، سىلىق، ئىلمىي، ئادىل ، باراۋەر، ياخشى تەرەپنى كۆپرەك ئويلاپ ياخشى پوزىتسىيىدە بولايلى. رەت قىلىش ، ئاچچىقلىنىش توغرا كەلگەندىمۇ، ئەڭ مۇۋاپىق ، ياخشى بولغان ئۇسۇللار بىلەن ئالاقىنى ئۈزسەكمۇ بولىدۇ، تىللىشىپ، ھاقارەت قىلىشىپ ئۈزلىشىشنىڭ ھىچقانداق پايدىسى يوق. چۈنكى، بۇ سىزنى قارشى تەرەپ ھىچنىمە قىلالمايدىغان مەۋھۇم دۇنيا بولغىنىدەك، سىز قارشى تەرەپكە ياخشى مۇئامىلىدە بولۇپ رەت قىلسىڭىزمۇ سىزنىڭ بويۇن ئەگكەنلىكىڭىزنى ھىچكىم بىلمەيدىغان مەۋھۇم دۇنيا ئىكەنلىكىنىمۇ ئۇنتۇپ قالماڭ. يىغىپ ئېيتقاندا ئىشنىڭ ياخشىسىنى قىلغان ئەلا.
   مۇنبەردىكى بارلىق دوسلارنىڭ ئىناق، ئىتتىپاق، ئويۇشقاق، ياخشى ئوي-تەپەككۇردا بولىشىنى ئۈمىد قىلىش بىلەن بىرگە، ھەممىڭلارنىڭ سالامەت، خىزمەت، ئۆگىنىشىڭلارنىڭ ئۇتۇقلۇق، مۇھەببەت، ئائىلەڭلارنىڭ خۇشاللىقتا، بەختتە بولىشىنى تىلەيمەن !
