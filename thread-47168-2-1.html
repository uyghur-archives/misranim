<<ئېتىراپ قىلىشقا بولمايدىغان ئىككى ئوبزور>> دىگەن تېمىغا يازغان ئىنكاسىم :

<<رەسسام غازى ئەھمەت ئاكىمىز ھەممىمىزگە تونۇشلۇق ، ئەل ھۆرمىتىگە سازاۋەر كىشىلىرىمىزنىڭ بىرسى . ئۇ سىزغان ئالىملىرىمىزنىڭ پورتىرېت رەسىمىنى دۇنيا خەلقى ئېتىراپ قىلغان . مېنىڭمۇ چوڭقۇر ھۆرمىتىم بار .
ئىلىم پەننىڭ تەرەققى قىلىشىغا ئەگىشىپ ، ئالىملىرىمىز توغرىسىدا ئىزدىنىش مەيلى چەتئەلدە بولسۇن ياكى رايۇنىمىزدا بولسۇن تېخى تۈگىگىنى يوق . شۇنداق بولغان ئىكەن يارمۇھەممەت توغلۇق ئەپەندىممۇ نۇرغۇن ئىزدىنىشلەر ئارقىلىق مەخمۇد قەشقىرى، يۈسۈپ خاس ھاجىپتىن ئىبارەت ئىككى بۈيۈك ئالىمىمىزنىڭ ھاياتى پائالىيەتلىرى توغرىسىدا نۇرغۇن تەتقىقات ئېلىپ بېرىش ئارقىلىق بۇ ئىككى ئالىمنىڭ ھازىرقى ئىشلىتىۋاتقان پورتىرېتى ھەققىدە ئوخشاش بولمىغان كۆز قارىشىنى ئوتتۇرىغا قويغان . يارمۇھەممەت توغلۇق ئەپەندىنىڭ ئىككى پورتىرېتى ئۈستىدە ئېلىپ بارغان تەتقىقاتىنى باشقىلارنىڭ تۆھپىسىنى ئىنكار قىلىش غەرىزى بىلەن قىلىنغان دەپ ئېيتىش تولىمۇ بىمەنىلىك .  
قانداقلا بولمىسۇن ھەر ئىككى خىل پورتىرېتنى ۋۇجۇتقا چىقارغان ئىلىم ئەھلىلىرىنىڭ ئىزدىنىشىنى ، تەتقىقاتىنى قوللاش ، ۋە ئۇلارغا مەدەت بېرىش كېرەك . بۇ ھەقتە مەلۈم تالاش تارتىش ئوتتۇرىغا قويۇلغان ئىكەن . نەتىجىسى ھامان چىقىدۇ . شۇڭا تورداشلارنىڭ ئەستايىدىل ئويلىنىپ پىكىر قىلىشىنى ئۈمىت قىلىمەن .
بۇ يەردە ئارتۇقچە تاكاللىشىپ ، ئاق تاغلىق ، كۆك تاغلىق بولۇشۇپ كەتمىسەك . >>
   يەنىلا ئۆز پىكرىمدە قالىمەن .
   
تۇران ئەپەندىنىڭ شۇ ئىككى پورتىرېت ھەققىدە ئىلگىرى ھىچكىم خىيالىغا كەلتۈرۈپ باقمىغان يېڭىچە كۆز قاراشنى دادىل ئوتتۇرىغا قويغانلىقىدىن، ئاپتۇرنىڭمۇ مەسىلىلەرنى كۆزىتىش ئىقتىدارىنىڭ يوقۇرىلىقى . ئەستايىدىللىقىغا قايىللىقىمنى بىلدۈرىمەن . ئىلىم - پەن ساھەسىدىكى يېشىلمىگەن نۇرغۇن تۈگۈنلەرنى مانا مۇشۇنداق ئىزدىنىش ، چوڭقۇر تەپەككۈر قىلىش ئارقىلىقلا يەشكىلى بولىدۇ .
ھېسسىيات قارىشىڭنى ئەقلىڭنىڭ ئۈستىگە قويغان كۆنۈڭ، سېنىڭ خاتالىشىشقا يول قويغان تۇنجى كۈنۈڭدۇر.

مىنىڭ  قارشىمچە غازى  ئەھمەد  ئاكا  ھەقىقەتەنمۇ  خەلىق  ئېتراپ  قىلغان  رەسىم  سەنئەتكارى  بۇ  نۇقتا  ھەممىمىزگە  ئايان.ئۇمۇ  بۇ  ئككى  ئوبرازنى  خەلىق  قۇبۇل قىلغۇدەك  دەرجىدە  يارتىش  ئۇچۇن  ئاز  جاپا  تارتمىدغۇ  دەيمەن.بڭراق  ھىچقانداق  نەرسە  سەۋەنلىك  نوقسانلادىن  خالى  بولمىغاندەك بۇ  رەسىملەردەرىمۇ  نوقسانلار  ساقلانغان.بىراق  غازى  ئەھمەد  ئاكا  تارىخچى  ئەمەس  شۇنداق  بولغاندىكىن  تارىخى  چىنلىققا  ئۇيغۇنلاشمىغان  نوقتىلانى ئەسكەرتىش  ئورۇنلۇق  دەپ  قارايمەن.مەن  گەرچە رەسسام  بولمىساممۇ   غازى  ئەھمەد  ئاكنىڭ  رەسىملىرنى  كۆرۈشنى  ياخشى  كۆرمەن.ئۇزۇندىن  بىرە  ئويلاپ  كەلگەن  بىر  خىيالىم  بار  ئېدى  تورداشلار  بىلەن  پىكىرلەشسەم.غازى  ئەھمەد  ئاكنىڭ  يەنە  بىر  داڭلىق  رەسمى  بار  ئۇ    ئۇيغۇر  12 مۇقامى  دىگەن  رەسىم   بەلكىم تورداشلار  كۆرگەندۇ    شۇ  رەسىمدە  ھىس  قىلشىچە  بۇ رەسىمدىكى  ئادەملەنىڭ  ھەممىسنىڭ  چىرايى  بىر خىل  سۇلغۇن. ئادەتتە  ئۇيغۇرلار  ناخشا-  ساز  دىسە  خۇشال  بولۇپ  يايراپ  كىتدۇ، بۇ  يەدە  ئۇلانىڭ  چىرايى  بەك  سۇلغۇن . ھىچ  چۈشەنمىدىم  بۇ  رەسىمدىكى  ئوبرازلانى  بۇنداق  يارتىشتىكى  مەقسەت  نىمە؟بۇ ئىنكاس  قانداقتۇر  غازى  ئەھمەد  ئاكنىڭ  ئەسەرلىرنى  چۈشرۈپ  ئەمەس  بەلكى  ئۆزۈمنىڭ  چۈشەنمىگەن  يەرلىرىمنى  تولۇقلاش  مەقسىتدە  يىزىلدى.

يەنە شۇ قۇرۇق گەپ،ئىلمىيلىك، ئىلىميلىك، ئىلمىيلىكنىڭلا گېپى،  قوڭامدا ئىشتىنىم يوق ئىسمىم مارجان بۈۋى دەپ،يا شۇ پۇتاق تاپقانلار بىرەر پارچە سۈرەتنى ئىلمىيلىككە ئۇيغۇن ھالدا سىزىپ ياكى باشقا بىر رەسسسامغا سىزدۇرۇپ يوللىمىدى. بىزمۇ شۇنى كۆرۈپ باھا بىرىپ باقساق،.  ئۆلگەن ئادەم دېگەن  خىيالىڭىزدا قانداق تەسەۋۋرۇ قىلسىڭىز شۇنداق ھالەتتە ئېڭىڭىزدا ئەكسى ئېتىدۇ. يەنە بىرسى سىزگە سىزىپ بەرگەن تەقدىردىمۇ ئۇنىڭدىن ئىۋەن تاپىمەن دېسىڭىز، تاپقىلى بولىدۇ،خۇدا ياراتقان ئەڭ گۈزەل لىۋەن ساھىپجاماللاردىن ئىۋەن تېپىلغان يەردە مىڭ يىلدىن ئىلگىرى ياشىغان ئادەمنىڭ ھازىرقى دەۋردە سىزىلغان سۈرىتىدە ئاندا-ساندا ئىۋەنلەر بولماي قالارمىدى؟ ماقالىنى يوللىغان تاھىرجان بېكار قالغان چېغىۋا.

توغرا دەيسىز،ھە دەپ ئىلمىلىكتىن چۈشىدىكەنمىز،ئىككى كىتاپنى قانچە ئوقۇسىڭىزمۇ ئىككى ئالىمنىڭ پورتىرىتىى توغرىسىدا ئېنىق  مەلۇمات يوق،ئەجىبا مىڭ يىل بۇرۇنقى ئۇيغۇرلارنىڭ تىپىك پورتىرىتى=مۇشۇ ئىككى ئالىمنىڭ پورتىرىتىما؟ مىڭ يىل بۇرۇنقى ئۇيغۇرلارنىڭ ئىرق ئالاھىدىلىگىنى تەتقىق قىلىشتا ئىككى كىتاپنى ئوقۇشلا كۇپەيە قىلمايدۇ،بۇنىڭدا نۇرغۇن تارىخى سېلىشتۇرما تەتقىقاتلار،ئىنسانشۇناسلىق،ئىرقشۇناسلىق تەتقىقاتلىرى  دەلىل-ئىسپاتلىرى بىلەن چوڭقۇر ئىشلىنىشى كېرەك،دۇنيا ئىلىم ئەھلىنى قايىل قىلارلىق ئىلمى نەتىجىلەر بولىشى كېرەك،ئەمما بىزدە بۇ ھەقتە تېخى مۇكەممەل تەتقىقاتلار ئېلىپ بېرىلمىدى،باشقىلارنىڭ تەتقىقات نەتىجىسىگە قارىسىڭىز،بىزنى تارىختىن  موڭغۇل ئىرقىغا ئايرىيدۇ،ھازىرمۇ سىزنى موڭغۇل ئىرقىغا ئايرىيدۇ،مەيلى سىز قايىل بولۇڭ بولماڭ،ئەھۋال شۇنداق،شۇڭا قەغەز يۈزىگە تايىنىپ،ھەرقانچە گەپدانلىق قىلغىنىمىز بىلەن،ھىسسىياتلانغان بىلەن،ئاقىۋەت ھىچكىم ھىچكىمنى قايىل قىلالمايدۇ.
قايسى بىر ھەۋەسكارنىڭ تۇغلۇق ئەپەندىنىڭ يازمىسىدىن ئىلھام ئېلىپ سىزغان بىر رەسىمى،مىسرانىمغا يوللاندى،ئەجىبا ئىسلام دىنىدىكى ئوردا يۇقۇرى قاتلام زىيالىلىرىنى قانداقتۇ ئۇزۇن پاخما چاچ قويۇپ،كۇلا كەيگۈزۈپ ئوۋچىلاردەك تەسۋىرلەش ئەقىلگە سىغامدۇ؟
تارختىن خەۋىرى بارلار شۇنى بىلىدۇ، قاراخانىلار دەۋرى ئۇيغۇر تارىخىدىكى دەۋر بۆلگۈچ بىر دەۋر،ئۇيغۇرلارنىڭ تارختىكى غايەت زور مىللى يۇغۇرلۇش باسقۇچى.قاراخانىلار ھۆكۈمرانلىق قىلغان زېمىندا نۇرغۇن مىللەت.ئىرق ئورتاق ياشايدۇ،ئۇيغۇر،قارلۇق،تاجىك،پارىس،قىپچاق،موڭغۇل،خەنزۇ....ۋە باشقىلار،مۇشۇ كەڭ زېمىندا ياشاۋاتقان ھەرمىللەت خەلقىنىڭ شۇ دەۋردە ئۇيغۇرلىشىپ مېڭىشى ئەڭ تېز بولغان.شۇنىسى ئايانكى ئۇيغۇرلار ئىرق جەھەتتە قانداقتۇ مەلۇم ساپ ئىرق ھالىتىنى ساقلاپ قالالمايدۇ،ھەم مۈمكىن ئەمەس.شۇڭا ئايرىملىق ئارقىلىق ئومۇمىلىققا ھۆكۈم قىلىش پۇت دەسسەپ تۇرالمايدۇ،ھەم ئومۇمىلىق ئارقىلىق ئايرىملىقنى ئىسپاتلاشمۇ ئىلمىلىك ئەمەس. ئەمما تارىخى شەخسلەرنىڭ پورتىرىتىنى ياراتقاندا ،ئومۇمىلىقنى ئاساس قىلىپ،ھەرخىل تىپلاردىن ئۆرنەك ئېلىپ،ھەر جەھەتتىن مۇۋاپىق بولغان تىپىك ئوبراز يارىتىش خاتا ئەمەس.
ئىككى ئالىمنىڭ پورتىرىتى ئارقىلىق 1000يىل بۇرۇنقى مىللەتنىڭ ئىرقى ئالاھىدىلىگىنى ئىسپاتلايمەن دېيىش سەل ئاشۇرۋېتىلدۇمۇ قانداق؟
ئىككى ئالىمىمىزنىڭ رەسىمىنى پۈتۈنلەي ئەينەن سىزىپ چىقىشمۇ مۈمكىن ئەمەس،پەقەتلا رەسساملارنىڭ ئىجادى ئەمگىگىدىن بىز ئۇلارنى خەلققە تونۇتىمىز.ئىككى ئالىمنىڭ غازى ئەھمەد ياراتقان پورتىرىتى ئومۇمى جەھەتتىن مۇۋەپپىقىيەتلىك چىققان،خەلق بىر قاراپلا قوبۇل قىلالايدۇ.روشەن ئاشۇرۋېتىلگەن،ياكى كەمتۈك يەرلەر يوق،ئالىملىرىمىزنىڭ سالاھىيىتىگە تولىمۇ ماس كەلگەن.  
ئەگەر ئىككى ئالىمىمىزنىڭ پورتىرىتى تارىخى چىنىلىقتىن چەتنەپ كەتتى،بىز ئۇ چاغدا قىزىل چاچ،كۆك كۆز ئىدۇق دېگۈچىلەر،ھازىر ئۆزىڭىزنىڭ چىرايىغا قاراپ بېقىڭ،ئەتراپتىكى كىشىلەرنىڭ چىرايىغا قاراپ بېقىڭ،بۇلارنىڭ  مۇشۇ ئىرقى ئالاھىدىلىگى شەكىللەنگىلى قانچىلىك بولغاندۇ؟ئىلمىلىك بىلەن ئالدىغا سۈرۈپ باقمامسىز؟ھازىرقى ئۇيغۇرلار پۈتۈنلەي ئەسلى ھالىتىنى يوقاتقان شالغۇتلارمۇ؟شەرقى شىنجاڭدىكى قېرىنداشلىرىمىز قانچىلىك تارىخقا ئىگە؟
پارىسلار،تاجىكلار ئاقتەنلىك ئىرقىغا مەنسۇپ،ئۇلار تارىختا موڭغۇل ئىرقىنىڭ تەسىرىگە بىزگە قارىغاندا ئاز ئۇچرىغان،ئەجىبا ئۇلارنىڭ ئوتتۇرا ئەسىردىكى ئالىملىرىنىڭ چىراي تۇرقىدا قانداقتۇ قىزىل چاچ،كۆك كۆزلەر يوققۇ؟
تۈركلەرنى مىسالغا ئالايلى،تۈركلەر قاراخانىلاردىن بۇرۇن غەرپكە كۆچۈپ كەتكەن،دىمەك ئۇلار بىز بىلەن قېرىنداش بولۇش سۈپۈتى بىلەن ساپ ھالەتتە كەتكەن بولىشى مۈمكىن(موڭغۇل ۋە باشقا مىللەتلەرنىڭ تەسىرىگە ئاز ئۇچرىغان).ئۇلار كۆچكەن زېمىن ئاقتەنلىكلەرنى ئاساس قىلىدۇ،تارىختا ئۇلار گرېك،بۇلغار،سلاۋيانلارنى كۆپلەپ ئاسسىمىلاتسىيە قىلىش ئارقىلىق ھازىرقى تۈرك مىللىتى شەكىللەندى،ئەسلى ئۇلار ئىرىق جەھەتتە تېخىمى ئاق يۈز،سېرىق ياكى قىزىل چاچ،كۆك كۆز بولۇپ كېتىشى كەرەكئىدى،لېكىن ئۇلار يەنىلا تارىختىكى ئىرقى ئالاھىدىلىگىنى ئاساسەن ساقلاپ قالدى.تۈرك ئەركەكلىرىدە يەنىلا قارا چاچ،قوي كۆز،بۇغدايئۆڭ،ساقال-بۇرۇتلىرى تەرەققى قىلغان كىشىلەر ئاساسى سالماقنى ئىگەللەيدۇ.  
ئەگەر ئىككى ئالىمىمىزنى قازاقىستان،قىرغىسىتان يېقىندا نەشر قىلىپ تارقىتىۋاتقان تارىخ ،دەرسلىك كىتاپلىرىدا،تەشۋىقاتلىرىدا،پۇللىرىدا ياراتقان پورتىرىتلىرىدەك ،ياپما قاپاق،پىت كۆز،مەڭزى سۆڭىگى بۆرتۈپ چىققان،سەللىنى پايتما ئورىغاندەك ئورىۋالغان،سىم ساقال قىپچاق ئاقساقىلىدەك سىزغان بولسا ئىدى.
مانا بۇ تارىخقا خالىغانچە چاخچاق قىلغانلىق،ئىغۋاگەرچىلىك قىلغانلىق بولىدۇ. بىۋاستە ئۇيغۇر مىللىتىگە قىلىنغان مۇشۇنداق ئىغۋانى تارىخچىلىرىمىز،تەتقىقاتچىلىرىمىز قىلچىمۇ تەخىر قىلماي ۋاختىدا قايتۇرما جاۋاپ بېرىشىلىرى كېرەك!ئۆزىمىزنىڭ تارىخى مىراسلىرىمىزنى،قوغدىشىمىز، قۇدرەتلىك دۆلىتىمىزنىڭ دۇنيادىكى تەسىر كۈچىدىن پايدىلنىپ ئېتىرازنامە بىلدۈرشىمىز كېرەك،بىز ئىگە چاقىسىز مىللەت ئەمەس!!!
ئاخىردا دەيدىغىنىم مۇشۇ ئورۇنسىز تالاش-تارتىشنى قىلىشماي،تارىخىمىزنى ئۆزىمىزگە تەۋە مىراسلىرىمىزنى باشقىلارنىڭ خالىغانچە تالان-تاراج قىلىشى،تارتىۋېلىشىغا  ئىلمى ئۇسۇلدا،ئېنىق پوزىتسىيىمىزنى بىلدۈرۈپ قويىشىمىز كېرەك!يەنە بىر تەكىتلەپ قوياي،مىللەتلەر مەسىلىسىگە تاقىلىدىغان مەسىلىلەردە ئىلمى مۇھاكىمە،دەلىل-ئىسپاتنى ئاساس قىلايلى(چۈنكى ھەقىقەت سىزنىڭ قولىڭىزدا)،ھەرگىزمۇ ھىسسىياتقا تايىنىپ ئاغزىمىزغا كەلگەننى دەپ قويۇپ باشقىلارنى بالاغا ئىىتىتىرمەيلى!بىلمىسىڭىز سۈكۈت قىلسىڭىز ئەڭ توغرىسى شۇ!
بىلمىگەننى بىلدىم ، كۆرمىگەننى كۆردۇم، ئاڭلىمىغاننى ئاڭلىدىم دېمە!
