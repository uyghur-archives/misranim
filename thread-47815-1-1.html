    بۇنداق دەپ يازسام نۇرغۇن كىشىلەر ئىككىلەنمەي بىلىم مۇھىم دەپ بولىشىغا قاتتىق توۋلايدۇ، بىراق ئالدىغا پۇل كەلسۇنچۇ ئىككىلەنمەي ئۆزىنىڭ ھەقىقىي ئەپتى-بەشىرسىنى ئاشكارلايدۇ. ئەمما ھەممە ئادەمنى بىر تاياقتىلا  ھەيدەش تېخىمۇ توغرا كەلمەيدۇ. تارىخ بېتىنى ۋاراقلايدىغان بولساق يۈزدە 95 پىرسەنىت ئۇلۇغ مۇتەپپەككۇر ئالىملىرىمىزنىڭ ئىسمىنى ئۇچىرىتىمىز مەسلەن ( بىزدىن:مەھمۇت قەشقەرى، يۈسۈپ خاس ھاجىپ، ئەل فارابى، رابغوزى، ئەلشىر نەۋائى، زەلىلى، ھىرقىتى، گومنام، ئاماننىساخان، ئابدۇرىشىتخان، موللا بىلال بىن موللا يۈسۈپ، تەجەللى، ئابدۇرىھىم ئۆتكۈر.........خەنزۇلاردىن كوڭزى، مىڭزى، جۇگېلىياڭ، لى بەي، دۇفۇ، بەي جۇيى، خۇاتۇ، بىيەن چۆ، شى لەيلى...........چەتئەللىكلەردىن ئارىستوتىل، پىلاتون، سوقرات، لوقمان ھەكىم، ئىبىنسىنا، نىيۇتون، گالىلى، بىرونۇ، فارادىي، ئىدىسون. ئېينىشتىيىن........قاتارلىقلار) ئەلۋەتتە بۇلار ساناپ تۈگەتگۈسىز. ئەمدى پۇلدارغا كەلسەك بۇلار يۈزدە 5پىرسەنىتمۇ چىقمايدۇ. مەسلەن(ئىسكەندەر زۇلقەر نەيىن، خې شىن، ئاپپاق خوجا، شېڭ شى سەي،...........قاتارلىقلار) مەن شۇنى كېسىپ ئېيتالايمەنكى بۇلارنىڭ كۆپۈنچىسىمۇ ئۆز زامانىسىدىكى ياخشى ئادەملەر ئەمەس .
    بىراق ھازىرقى زاماندىچۇ؟ ھازىرقى زاماندا بىز قانچىلىك ئالىم تونۇيمىز؟ يەنە قانچىلىك پۇلدار ياكى ئەمەلدارنى تونۇيمىز؟  مېنىڭچە ھازىر بىز ھېچقانچە ئالىم، ئەدىپ ۋە ياكى كەشپىياتچىلارنى تونۇمايمىز. لېكىن بايلارنى، كارخانچىلارنى، باشلىقلارنى بەك تونۇيمىز. بۇنىڭدىن تورداشلارنىڭ كۆپىنچىسى مۇستەسنا ئەمەسقۇ دەيمەن.
   راستىنى ئېيىتقاندا مەن بىر ئادەتتىكى ئالىي مەكتەپ ئوقوغۇچىسى، بىلىم سەۋىيەم يۇقىرىمۇ ئەمەس. بىراق بىر ۋاقىتتا سىنىپتا مەن سىلەرگە قويۇۋاتقان مۇشۇ مۇنازىرنى دەتالاش قىلىپ تاس قالغان بىر-بىرىمىزگە ھۆرپىيىپ قوپقىلى. شۇڭا مەن مۇشۇ تېمىنى يوللاش ئارقىلىق جەمئىيتىمىزدىكى مەندىن سەۋىيەسى يۇقىرىلارنىڭ سەمىمىيلىك بىلەن كۆز قارىشىنى ۋە تەكلىپىنى قوبۇل قىلسام، مەن بىلەن سەۋىيەسى ئوخشاشلارنىڭ مۇشۇ مەسىلىگە قارىتا كۆز قارىشىنى توغىرلىۋىلىشىنى چىن دىلىمدىن ئۈمىد قىلىمەن.
           قېنى ئەمىسە تورداشلار قىممەتلىك پىكىر تەكلىپىڭلارنى ئايىماي ئۆز نۇقتىنەزىرىڭلارنى بايان قىلغايسىلەر.
