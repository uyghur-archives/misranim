               ئاپا ، بۈگۈن مىنىڭ تۇغۇلغان كۈنۈم . 18 يىل ئاۋالقى سىزنىڭ مىنى تۇغقان كۈنىڭىز . مانا شۇ كۈندىن كىيىن پەقەت مەن ئۈچۈنلا  قۇربانلىق بېرىپ كەلدىڭىز  ، ئۆزىگىز يىمەي ئىچمەي ، ماڭا يىگۈزۈپ ئىچكۈزۈپ مىنى پەپىلەپ باقتىڭىز .... 

               ئاپا ، سىز مەن ئۈچۈن ئاز جاپالارنى تاتمىدىڭىز  ، مەن ئۈچۈن ياشلىق باھارىڭىزنى خازان قىلدىڭىز ، مەن تۇغۇلغان شۇ كۇندىن تاكى بۈگۈنگۈچە پەقەت سىزگىلا تايىنىپ ياشاپ كەلدىم . سىز بۇلارنى قىلچە ئېغىر ئالمىدىڭىز  ، رازىلىق بىلەن  ، خۇشاللىق بىلەن مىنىڭ تەلەپلىرىمنى قاندۇرۇپ كەلدىڭىز .... 

             ئاپا ، ھېلىمۇ ئىسىمدە مەن ئاغرىپ قالغان شۇ ۋاقىتلاردا سىز تولىمۇ جىددىيلىشىپ كەتكەن ئىدىڭىز . مىنى ئولتۇرغۇزۇپ قۇيۇپ ھەر تەرەپلەگە قاتراپ دوختۇرخانىنىڭ رەسمىيەتلىرىنى ئۆتىگەن ئىدىڭىز . بەلكىم ئۇ ئىشلارنى سىز ئاللىقاچان ئۇنتۇپ قالغانسىز ، چۈنكى ئۇنداق كېچىلار سىزنىڭ بېشىڭىزدىن تالاي - تالاي قېتىملاپ ئۆتكەن ئەمەسمۇ ...  تاكى ھازىرغىچە كىچىككىنى بېشىم ئاغرىپ قالسىمۇ مەن ئۈچۈن جىددىلىشىپ كىتىسىز ... بۇنداق ۋاقىتلادا ئۆزۈمنى تولىمۇ ئاشۇرۋەتكەندەك ، تولىمۇ رەھىمسىزلىك قىلغاندەك ... ھېس قىلىپ قالىمەن . چۈنكى مەن بالا ئەمەسمۇ ، ئەسلىدە مەن سىز ئۈچۈن خىزمەت قىلسام بۇلاتتى . لىكىن ھازىرچۇ ؟؟ مەن سىزگە خىزمەت قىلىش ئۇياقتا تۇرسۇن بەزىدە كۆڭلىڭىزگە ئازار بېرىمەن . كېيىن خاتالىقىمنى تونۇپ تۇرۇپمۇ سىزدىن كەچۈرۈم سورۇيالمايمەن . بەلكى بۇ مىنىڭ تولىمۇ ۋاپاسىزلىقىم بولسا كېرەك . 

        ئاپا ، مىنى كەچۈرۈڭ ... 18 يىلدىن بۇيان قىلغانلىرىمنى بىرمۇ بىر ئويلۇنۇپ چىقتىم مىنىڭ خاتا قىلغانلىرىم راستىنلا بەك كۆپكەن . لىكىن سىز ئاقكۆڭۈل ، مېھرىبان بولغاچقا ھەر قېتىم مىنى كەچۈرۈپ ماڭا يول قۇيۇپ كەپسىز ...  سىز تولىمۇ ئۇلۇغ  ، مىنى چوڭ قىلىش جەريانىدا قاپقارا توم چاچلىرىڭىزنىڭ ئارىسىغا ئاق چاچلار يۇشۇرنۇپتۇ . قىپقىزىل مەڭزىڭىز ، ئاينىمۇ خىجىل قىلغۇدەك گۈزەل چېھرىڭىز سارغىيىپتۇ . يۇمران قوللىرىڭىز قاتتىق بوپ كىتىپتۇ . نۇرلۇق كۆزلىرىڭىز خۈنۈكلىشىشكە باشلاپتۇ .... سىزنىڭ چېھرىڭىزگە ھەر قېتىم قارىغىنىمدا ئۆزۈمنى تولىمۇ گۇناھكار ھېس قىلىمەن . سىز مىنى دەپ شۇنچە جاپا چىكىۋاتسىڭىز مەن سىزگە ھىچ نەرسە قىلىپ بىرەلمىدىم . ئەسىچە چوڭ بۇلۇشقا ئەگىشىپ سىزگە تېخىمۇ كۆپ باش قېتىنچىلىقى ئەكىلىپ بېرىۋاتىمەن ... 

            ئاپا ، مىنى كەچۈرۈڭ ... سىزگە جاپا سالدىم . 18 يىل ئاۋالقى بۇگۈنكى كۈندە سىز ئاچچىق تولغاق ئازابىدا جاپا چىكىۋاتاتتىڭىز . 18 يىلدىن كىيىنكى بۈگۈنكى كۈندە مىنىڭ خۇشاللىقىم ئۈچۈن داستىغان تەييارلاش بىلەن ئالدىراش بولۇپ يەنە جاپا چىكىۋاتىسىز ... ئاپا ، مەندەك نادان قىزىڭىزنى كەچۈرۈڭ . سىزنى كۆپ جاپالارغا سالدىم . دىلىڭىزغا ئازار بەردىم مىنى كەچۈرۈڭ ...

                  ئاپا ، بۈگۈن مىنىڭ تۇغۇلغان كۈنۈم . ئەلۋەتتە بۇ خوشاللىق ماڭىلا تەئەللۇق بولسا بولمايدۇ . سىز بولمىسىىز ماڭا بۇ خۇشاللىق نەدە ؟؟ شۇڭا مەن ئاۋال سىزنى مۇبارەكلىمەكچى .

                 <<  ئاپا ، مىنى تۇغقان كۈنىڭىزگە مۇبارەك بولسۇن . بۇ كۈن سىزگە تەئەللۇق ، سىزنىڭ خۇشاللىققا چۆمىدىغان كۈنىڭىز . ھەرگىزمۇ مەن ئۈچۈن جاپا چىكىدىغان كۈنىڭىز ئەمەس . ئاپا ، سىز 18 يىل ئاۋالقى مۇشۇ كۈندە مەن ئۈچۈن ئۈلۈم ئاغزىدىن قايتىپ كەلگەن ئىدىڭىز ، سىز تارتقان جاپانى مەن تىل بىلەن تەسۋىرلەشكە ئاجىزلىق قىلىمەن . شۇڭا 18 يىلدىن كىيىنكى  بۈگۈنكى كۈندە مەن سىز ئۈچۈن بىرەر قېتىم بولسىمۇ جاپا تارتىپ باقاي،  قولىڭىزدىكىنى ماڭا بېرىڭ ئاپا ... >>

                 گەرچە بۇنىڭلىق بىلەن سىزنىڭ ماڭا سىڭدۇرگەن ئەجرىڭىزگە جاۋاپ قايتۇرالمىساممۇ ، مەن ئەمدى سىز ئۈچۈن خىزمەت قىلىشقا تەييارمەن ... ئەمدى مەنمۇ خېلە چوڭ بۇلۇپ قالدىم دەردىڭىزگە دارمان بولسام ، قولىڭىزغا قول ، پۇتىڭىزغا پۇت بولسام دەيمەن .

                  ئەقىلسىز قىزىڭىزنىڭ گۇناھلىردىن ئۆتكەيسىز ئاپا ... سىزنى سۆيىمەن .
__________________   

          گەرچە بۇ ئاددى سۆز جۈملىلەر بىلەن دىيىلگەن سۆزلەر بولسىمۇ مىنىڭ قەلبىمدىن چىققان سۆزلەر ... ئاپا ، سىزنىڭ بۇلارنى ئوقۇپ بېقىشىڭىزنى ئۈمۈد قىلىمەن .
