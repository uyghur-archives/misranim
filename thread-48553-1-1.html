ئەسسالامۇ ئەلەيكۇم ،مۇنبەرداشلار كەمچەت تۇماقنى بىلەمسىلەر ؟ ئىسىمدە قېلىشىمچە كىچىكىمدە تەلەپ ۋە تەغدىم دىگەندەك سەنئەت پىروگراممىلىرىدا پاتىگۈل سەئىدى،مۇنىرە غۇپۇر دىگەندەك داڭلىق رىياسەتچى دىكتورلىرىمىز تېلىۋىزور ئىكرانىدا يارىشىملىق كەمچەت تۇماق كەيگەن ،"سۇمبۇل"چاچلىرى چىرايلىق ئۆرۈلگەن ھالدا نامايەن بولاتتى.شۇچاغلاردىكى گۆدەك قەلبىمدىمۇ بۇرۇنقى ئەجداتلارمۇ كەمچەت تۇماقنى مۇشۇنداق كىيەر بولغىيمىتى ،بۇنداق چاچلىرىنى ئوچۇق قۇيۇپ يۈرۈش مۇسۇلمانلىق ئەقىدىمىزگە زىت كەلمەمدىغاندۇ دەپ ئويلايتىم ،كىيىن بىلىپ يەتتىمكى ھازىر ئەسلى ئاشۇ كەمچەت تۇماق  ۋە ھەرخىل دوپپىلارنىڭ ئاستىدىكى كەڭرى يېيىلىپ تۇرىدىغان رومال ئۇنتۇلغان ئىكەن.چوڭنامنىڭ ئېيتىپ بىرىشىچە ئۇنىڭ تويلۇقىغا كەمچەت تۇماق بىرىپتىكەن.ئەمما ئاشۇ تۇماقنىڭ تېگىدە رومال بارلىقى ئېنىق ئىدى. ئۆتكەندە بىر قەشقەرلىك ساۋاقدىشىمنىڭ ئائىلە رەسىمىنى كۆردۈم ،ساۋاقدىشمنىڭ ئانىسى 40 ياشلار ئەتراپىدىكى ئوتتۇرا ياش ئايال ،بېشىغا كىمخاپ تەتىللادىن تىكىلگەن شۇنداق ئۆزگىچە دوپپا كەيگەن ،دوپپىنىڭ ئاستىدا چوڭ ئاياللار بېشىدىن ئايرىمايدىغان تور رومال ئارتىلغان .يەنە بىر خوتەنلىك ساۋاقدىشىمنىڭ چوڭئانىسىنىڭ رەسىمى ئېسىمگە كېلىپ قالدى ،70 ياشلاردىن ئاشقان ،پاكىزغىنە قېرىغان ئايال ،ئوچىسىغا  قارا ئەتلەستىن ئۇزۇن كۆڭلەك كىيگەن ،بېشىدا يوغان ئاق رومال ،ئۈستىدە قارا دۇخاۋىدىن تىكىلگەن چىمەن دوپپا .ساۋاقدىشىمنىڭ ئېيتىشىچە بۇ رەسىم 90-يىللارنىڭ بېشىدا تارتىلغان ئىكەن.ھازىر ئەمەلي تۇرمۇشىمىزدا چوڭ ،ئوتتۇرا ياشلىق ،ياكى يېڭى توي قىلغان قىزلارنىڭ دوپپا ،كەمچەت تۇماق  كىيىپ يۈرۈيدىغىنى بەكمۇ ئاز بولسا كېرەك.دوپپا ئاللاقاچان چاچلىرىنى  40 ئۆرۈپ يۈرۈيدىغان مەكتەپ  بالىلىرىغا ،سەھنىدىكى ئۇسۇلچى ئارتسلارغا "ئۆتۈنىۋىتىلگەن " بولسا كېرەك . ھازىرقى ئارتىسلىرىمىزنىڭ دوپپا كەيگىنىنى خېلى كۆرگىلى بولىدۇ ،ئەمما نەچچە ئون يىللار ئالدىدا ئارتسلارمۇ دوپپا ،كەمچەت تۇماقلارنىڭ ئاستىغا نىپىز شايىدىن بولسىمۇ رومال ئارتىپ چىقاتتى ،ھازىر ئۇ شايى روماللارمۇ ئاز قالغان بولسا كېرەك.

تۆۋەندىكىلەر مەن ئۆزۈم يىغىپ ساقلىغان رەسىملەر بۇلۇپ ،مۇنبەرداشلارنىڭ كەمچەت تۇماقنى تېخىمۇ ياخشى چۈشىنىشگە پايدىسى بولسا ئەجەپ ئەمەس.
