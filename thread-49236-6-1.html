مەن بۇ تېمىنى ئوقۇش ئارقىلىق بۇ تېمىدىكى 90پىرسەنت قىزلارنىڭ ئۇنداق كىيىندىغان قىزلاردىن نەپرەتلىدىغانلىقىنى ، 99پىرسەنت ئوغۇللارنىڭ ئۇنداق قىزلارغا ئۆچلىكىنى بىلىۋالدىم ، مۇنبەردىكى قىز-ئوغۇللار بىر بىرىڭلارغا بەك ماس كېلىدىكەنىسىلەر بەرنا يىگىتلەر بىلەن ئۇياتچان ئەخلاقلىق قىزلار....بىر بىرىڭلارغا توپنى بەك پىچەت پاس بېرىۋاتقان مۇنۇ تارىخىي دەۋردە ، تور ئەسەرلىرىدە تەسۋىرلىنىدىغان ۋەھشىي ئۇيغۇر ئىپىتدائىي كىيىم جەمئىيتىدىكى يۈزى قېلىن يوپۇرماقلىق كىيىمىلەرنى كىيۋالغان كىيۋالغان ھاياسىز ئۇيغۇر خوتۇنلىرىغا ئوخشىمايدىغان پەرىشتە چىراي گۈزەللەرنى كۆز ئالدىمغا كەلتۈرۈپ ھۈم ھۈم يىغلاپ كېتىپ بارىمەن ... تەسرلىنىپ ئەمەس بەلكى  مۇنۇ ھەممە مىللەتتىن تېپىلىدىغان، بۇلۇڭ پۇشقاقلاردا مارىلاپ پۇراپ يۈرمىسە كوچا كويلاردا تۆت كۈن ساقلىسىمۇ ئۇچرىمايدىغان كىندىك ئوچۇق ئۇيغۇر قىزى توغرىسدىكى تەتۈر تاناسىپ شەكلىدىكى تور تۆكملىرىنى .... يەنە باشقا ئاتالمىشى   ئەيىپدىن بىرنى مەھكەم چاڭگاللاپ مۇنۇ يەجۇجى _مەجۇجى ،پەي -كىگىز قالپاقلارھەممسى  كىرىدىغان مۇنبەردە ئۆزىمىزنى سازايى قىلۋاتقىنىمىزغا ئېىچىنىپ ....
گوگىلغا ئۇيغۇرچە ئۇيغۇر قىزى دەپ كىرگۈزسەم يېرىم  ئاھانەت چىقىدۇ ،بەيدۇغا خەنزۇچە ئۇيغۇر قىزى دەپ كىرگۈزسەم ئوقۇلغان ئاپىرىنلار چىقىدۇ...بولدىلا تولا سۆزلەپ ھېرىپ قالدۇق ،سۆزلىگەنلەر "سەن شۇنداق قىزكەنىسز،بىز تەربىيە قىلىۋاتىمىز"دەپ پېشانىغا تامغا چېكىۋالىمىز...شۇڭا    ...
  تۇرغان ياخشىكەن جاھان رەپتارى ................ سۆكۈڭلا ،سازايى قىلىڭلار...........لېكىن مېنى تىللماڭلا ئىنكاس يېزىپ بولۇپ قاچىمەن .......... [s:181]
