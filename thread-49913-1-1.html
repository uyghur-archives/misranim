     ئۆسمۈرلىكىمىزدە كىچىككىنە خۇشاللىق ئۇچرىسا ئالەمچە خۇشال بولۇپ كېتىدىكەنمىز ،شۇ خۇشاللىقلار بىزگە ئىلھام بەخىش ئىتىپ بىزنى ئالغا ئىنتىلدىرگۈچى بولىدىكەن ،بىزنىڭمۇ مەكتەپ ھاياتىمىزدا ئاشۇنداق ئازراق خۇشاللىق كۈنلىرىمىز بولغانىدى .
ئۇكۈنىمۇ نەچچە ۋاقىتتىكىگە ئوخشاش بىرىنجى سائەتلىك دەرسنى تۈگىتىپ ،تەنەپپۇس قىلىپ ئولتۇراتتۇق ،تۇيۇقسىز ئارقا تەرەپتىن كاككۇكتەك ،زىل ئاۋازلار ئاڭلىنىپ قالدى .
-راستما مەريەمگۈل ،<<تارىم غۇنچىلىرى>>غا ئېلان قىلىنىپتىما؟
-راست ،مېنىڭ ھىلىقى <<سەھەردىكى شەبنەم تامچىسى >>دېگەن ھېكايەم بارغۇ؟<<تارىم غۇنچىلىرى >>ژورنىلىغا بىسىلىپتۇ .
-ۋاھ!ۋاھ!ئاداش ئەمدىزە داڭلىق شائېرە بولۇپ كېتىكەنسەن دە .
-ۋىيەي ،دەۋاتقان گېپىڭنى قارا ،نەدىكى شائېرە مەن ،كېيىنچە توختاپ قالماي يازالىساملا بولىدۇغۇ .
-چاقچاق قىلىپ قويدۇم ،ھەشقىپىچەكتەك تۇتىلىۋالمىغىنە .ھەشقىپىچەك خىنىم دەمدۇق يا سېنى .
بۇ سىنىپىمىزدىكى قىز <<يازغۇچىلارنىڭ >>مەريەمگۈلنىڭ ئېلان قىلىنغان ھېكايىسى توغرىسىدا ئاغزى-ئاغزىغا تەگمەي قىلىۋاتقان پاراڭلىرى ئىدى .
مەندەك بىر ئوغۇل <<يازغۇچىمۇ >>بىرىپ تەبرىكلەپ قويمىسام بولماس ئەلۋەتتە .
-مۇبارەك بولسۇن مەريەمگۈل ،سىنىپىمىزدىكى ھەۋەسكارلار ئەمدى سېزنى ئۇستاز تۇتساق بولغۇدەك .
-ئۇنداق دېمىڭە قەيسەر خىجىل قىلىپ ئادەمنى ،تۇنجى قىتىم ئۆزىڭىز قوزغىغانغۇ بۇ شامالنى ،ئەمدىزە مېنى ئۇستاز تۇتاملا؟
قىزنىڭ ئايدەك يۈزلىرى قىزىرىپ ،ئۇزۇن كىرپىكلىرى ئاستىدىكى ئويچان كۆزلىرى يەرگە قارىۋالدى .
راست ،مەن تولۇقسىزنىڭ بىرىنجى يىللقىدىلا <<مومام ئىيىتقان ھېكايە>>ناملىق يىرىك يازمىلىرىم بىلەن سىنىپىمىزدا بىر مەزگىل يىزىقچىلىق دولقۇنىنى قوزغىۋەتكەن ئەمەسمۇ ؟
ئىككىنجى سائەتلىك دەرس ئەدىبىيات دەرسى ئىدى ،ئەدىبىيات مۇئەللىمىمىز 27ياشلار چامىسىدىكى ،ئوتتۇرا بوي ،ئۆتكۈر كۆز ،ساپالىق مۇئەللىم ئىدى ،بىزنى يالغۇز ئەدىبىياتتىلا تەربىيلەپ قالماي باشقا جەھەتلەردىنمۇ ئۇنۋىرسال تەربىيە بىرەتتى ،تۇنجى قىتىملىق يازغان نەرسەمنىمۇ بۇ مۇئەللىمگە كۆرسەتكەندىن كېيىن بۇ ئىككىيلەننىڭ مۇناسىۋىتى ئالاھىدە يىقىنلىشىپ كەتكەنىدى ،شۇڭا بۇ مۇئەللىمگە ئالاھىدە ھۆرمىتىم بار ئىدى .
بۇ سائەت بىر سائەت دەرس ئۆتۈلمەي ،مەريەمگۈل يازغان <<سەھەردىكى شەبنەم تامچىسى >>دېگەن ھېكايسىنى مەريەمگۈل زىل ئاۋازى بىلەن ئوقۇپ ئۆتتى .
ھېكايىدە ئوقۇشقا ناھايىتى تەلپۈنىدىغان ،بىردېھقان بالىسىنىڭ كەچۈرمىشى تەسۋىرلەنگەن بولۇپ ،بۇ بالا ئۇقۇشقا شۇنچىلىك تەلپۈنەتتىكى ،كەچلىرى باشقىلار ئويۇن بىلەن مەشخۇل بولسا ،تاپشۇرۇق ئىشلەيتتى ،ئەتىگخنى بولسا ئورنىدىن سەھەر تۇرۇپ ئائىلە ئىشلىرنى قىلىپ بولۇپ ئاندىن مەكتەپكە ماڭاتتى ،شۇم تەقدىر بۇ بالىنىڭ ئاتىسى ئاغرىقچان بولۇپ ئائىلە ئىقتىسادى بەك ناچا ربولغاچقا بۇ بالا ئۇقۇش پۇلى تۆلىگۈدەك پۇلمۇ تاپالمايدۇ ،مەكتەپكە بارسا ئوقۇتقۇچىنىڭ ئوقۇش پۇلى توغرىسىدا قىلغان دوق كىنايىلىرى ئۆيگە بارسا بىچارە دادىسىنىڭ سۇلغۇن چىرايى ،بۇ بالا ئاخىرى مەكتەپكە مېڭىش يولىدا چوڭ بىر تۈپ تىرەك بويىدا يىغلاپ ئۆز ھالنى بايان قىلغانلىقى ناھايىتى تەسىرلىك قىلىپ تەسۋىرلەنگەنىدى .
مەريەمگۈل ھېكايىنى ئوقۇپ بولغاندىن نكېيىن سىنىپ ئىچىدە گۈلدۈراس ئالقىش سادالىرى كۆتۈرۈلدى
ئەدىبىيات مۇئەللىمىمىزمۇ ئۆزىنى بىسىۋالالماي سۆزلەپ كەتتى .
-ياشاڭ سىڭلىم ،ئەگەر سىز ئوغۇل بالا بولۇپ قالغان بولسىڭىز ،قول ئىلىشىپ سېزنى تەبرىكلەر ئىدىم ،مۇشۇ پىتىلا يازسىڭىز كەلگۈسىدە خالىدە ئىسرائىلدەك داڭلىق يازغۇچىلاردىنى بولۇپ يىتىشىپ چىقىسىز ،يىزىڭ سىڭلىم ،تېخىمۇ ياخشى يىزىڭ .
مۇئەللىم گىپىنى تۈگىتىپ ئۆتكۈر كۆزلىرىنى ئەڭ ئالدىدا ئولتۇرغان ماڭا تىكتى .
-ۋوي ،ئۇنداق دىسەم مايەردىكى ئوغۇل يازغۇچىمىزنىڭ كۆڭلىگە كېلىپ قالمىسۇن ،سەنمۇ ئۈمىدسىزلەنمە قەيسەر ،ئۇيغۇر تارىخىدا شۇنداق داڭلىق شائېر يازغۇچىلىرىمىز ئۆتكەن ،سەنمۇ تېرىشساڭلا چوقۇم ھىرقىتى،زەلىلى،تەجەللى دەك داڭلىق ئالىملىرىمىزدەك بولۇپ يىتىشىپ چىقالايسەن .
مۇئەللىم ئىيتقان شەخىسلەرنى گەرچە تونۇمىساممۇ بۇ كىشىلىرىمىز ناھايىتى داڭلىق بولسا كېرەك دەپ ئويلىدىم ئىچىمدە .
14ياشلىق ئۆسمۈرلۈك دەۋرلىرىمىز ،ئەنە شۇنداق خۇشاللىق ،شاتلىق ئارىلاشقان كۈلكە تەبەسسۇمى بىلەن ئۆتىۋەردى ،مەنمۇ مۇئەللىم ساۋاقداشلىرىمنىڭ ئىلھامى بىلەن خىلىلا كۆزگە كۆرىنىپ ياخشى ئوقۇشقا باشلىدىم .
بۇ ياشلىرىمىزدا خۇددى شائېرنىڭ قايناق ھېسلىرىدەك ،مۇھەببەتنىڭ قايناق ھېسلىرىمۇ بىخلامدىكىن تاڭ ،ھىلىقى <<سەھەردىكى شەبنەم تامچىسى>>نى يازغان مەريەمگۈل دېگەن قىزغا كۆزلىرىمنىڭ قىرىدا قارايدىغان ،گەپ قىلغاندا تەمتىرەپ تىترەيدىغان ،قىزىرىپ تۇرۇپ گەپ قىلىدىغان بولۇپ قالدىم .
ھاياتىمدا ھازىرغىچە بۇنداق قىزنى ئۇچراتمىغاندىمە بەلكىم .
شۇنداق شەرمى-ھايالىق ،يۈزى تۆۋەن ،ئەخلاقلىق ،ئەقىللىق،كەمتەر ،سىپايە بىر قىزئىدى .چىرايىنى دىمىسەممۇ كۆز ئالدىڭلارغا كېلىۋاتقاندۇ بەلكىم چۈنكى خۇلقى پەزىلىتى چىرايلىقنىڭ چىرايىدىن نۇقسان تىپىش ھەقىقەتەن بىمەنىلىك ،دەرسلەردىمۇ يالغۇز يىزىقچىلىقلا قىلىپ قالماي ،يەنە رەسىم سىزىزش خۇشخەت يىزىش قاتارلىقلارغا بەكلا قىزىقاتتى ،ھەممە دەرسلەردە سىنىپنىڭ ئالدىدا تۇرغاچقا ئۈگىنىش باشلىقىمۇ بولغانىدى .

توۋا! مۇھەببەت دېگەن مۇشۇنداق بولامدىغاندۇ ؟دەپ ئويلايتتىم مەن ئۇ ۋاقىتلار .تېلۋۇزۇرلاردىن دورىۋالدىممۇ ؟سىنىپنىڭ ئەڭ ئالدىدا ئولتورىۋالغىنىمغا قارىماي ،ھىلىقى قىزغا بۇرۇلۇپ قاراپ كېتىشلىرىم ،تەنەپپۇس ۋاقىتلىرىمدىمۇ كۆزلىرىم شۇ قىزدا ،ئەگەر كۆزلىرىمىز ئۇچرىشىپ قالسا تەنلىرىم ئوت ئىلىپ كېتەتتى ،تېخى دەرستىن چۈشۈپلا ئۇقىزنىڭ ئارقىسىدىن تۇيدۇرماي مېڭىشلىرىم ،ھاياتىمدا بۇنداق نۇقسانسىز قىزنى بەلكىم ئىككىنجى ئۇچراپ باقمىغاندىمەن ،ۋەيا ئۇچراتساممۇ شۇقىزغا سىلىشتۇرۇپ ھىچقايسىسىنى بۇ قىزغا تەڭ قىلمىغاندىمەن .
مانا بۇ بالىلىقتىكى يۈرەكنىڭ ئۈنسىز شىۋىرلىشى ،شوخلۇقلارچە تەلپۈنىشى ،ھاياتىمىدىكى تۇنجى مۇھەببەتنىڭ بىخلانمىسى بولسا كېرەك .
بىركۈنى مەريەمگۈلنىڭ ئارقىسىدىن كېتىۋىتىپ ،توساتتىن بىر ساقاللىق چوڭ كىشىنىڭ مەريەمگۈل بىلەن پااڭلىشىپ تۇرغىنىنى كۆرۈپ قالدىم .
كۆڭلۈمدە بۇ مەريەمگۈلنىڭ دادىسى بولسا كېرەك دەپ ئويلىدىم ئىچىمدە ،قويۇق ساقاللىق كىشىنىڭ ئۆتكۈر كۆزلىرى ۋالىلداپ تۇراتتى ،مەن بۇكىشىنىڭ ئۆتكۈر كۆزلىرى بىلەن مېنىڭ كۆزلىرىم ئۇچرىشىپ قالمىسۇن دەپ چەتنەپ ئۆتۈپ كەتتىم .
ئۆسمۈرلۈك دېگەن ،ئەشۇنداق قىزىقارلىق بولىدىكەن ،ھەممە ئىشقا قىزىقىپ قالدىكەنسىز ،لېكىن كۆپ ئۆتمەيلا سوۋۇيدىكەنسىز ،بىرقىزنى ياخشى كۆرىدىكەنسىز يۇ ،ئاغزىڭىزدىن چىقىرالمايدىغان ،كۆڭلۈڭىزنىڭ بىر يەرلىرىدە سىغىنىدىكەنسىز .

بۇ قانداق مۇھەببەت دەپ قالماڭ.
بۇكۆڭۈلدىكى مۇھەببەت ،يۈرەكتىكى مۇھەببەت ،ياخشى كۆرسىمۇ ئىيتالمايدىغان گاچا مۇھەببەت، بالىلىقتىكى ساراڭ مۇھەببەت .
بۇ مۇھەببەت دەپ ئاتالغان رەڭگارەڭ دۇنيانىڭ تۇنجى بىخلىنىشى ،كەلگۈسىدىكى چوڭ بىر تۈپ گىياھنىڭ قەلبىڭىزگە يىلتىز تارتىشى ،بەلكىم ئەشۇ گىياھنىڭ سايىسى ئاستىدا كەلگۈسى بىر ۋاقىتلاردا سىز سايىداپ قالاسىز ، بۇ ەىياھنى قاغا-قۇزغۇن چوقىسىمۇ ،شاماللار سوقسىمۇ قەلبىمىزدەكۆكلىسە كۆكلەيدۇكى ،ھەرگىز ئۆرۈلمەيدىكەن .

قىززىق دەيمەن شۇ ۋاقىتلاردا <<ئۈچتە ياخشى ئوقۇغۇچى >>دەپ ئۈچ-تۆت تال دەپتەر ،بىر باغلام قىرىنداش،بىر قۇرۇق تەقدىرنامىنى تۇتقۇزۇپ قويسا قىن-قىنىمىزغا پاتماي كېتىدىكەنمىز .
-تولۇقسىز ئىككىنجى يىللىق ئىككىنجى سىنىپ ئوقۇغۇچىسى قەيسەر بارات سەھنىگە چىقىڭ .
كانايدىن ئۆز ئىسمىمنى ئاڭلىغان ۋاقتىمدا ھاياجاندىن ئۆزۈمنى ئاران بىسىپ كۆپچىلىكنىڭ ئوتتۇرىسىغا چىقتىم ،بۇ ۋاقىتتىكى خۇشاللىقنى تەسۋىرلەشكە تىل ئاجىزلىق قىلاتتى ئەلۋەتتە .
ھىلىقى دەپتەر قەغەزلەرنى كۆتۈرگەنچە سىنىپقا كېرىشىم مەريەمگۈلنىڭ ئاۋازى ئاڭلاندى .
-قەيسەرجان <<ئۈچتە ياخشى ئوقۇغۇچى >>بولىۋالدىڭىزھە!راستىنلا سىزگە قايىل جىما مەن.
خىجىللىقتىن يۈزلىرىم قىزىرىپ بۇ قىزنىڭ ئالدىدا گەپ قىلالماي قالدىم ،شۇنداق بولسىمۇ ئاۋازىمنى بوش چىقىرىپ ،
-سىزمۇ ئۈگىنىش باشلىقىغۇ ؟مەنمۇ سىزنىڭ ھەممە تەرەپلىرىڭىزگە بەك قايىل ،سىزگە بەك ھەۋەس قىلىمەن دەڭە .كەلگۈسىدە داڭلىق شائېرە بولۇپ قالسىڭىز بۇنداق پاراڭلاشقىنى بولارمۇ ھە سىز بىلەن .
-خىجىل قىلىۋەتلىغۇ ئادەمنى ،سىزبىلەن پاراڭلاشماي يەنە كىم پاراڭلىشاتتىم ھە؟. خۇدايىم بۇيرىسا ئالى مەكتەپلەردە بىللە ئوقۇپ قالامدۇق تېخى قەيسەر .
-دېگىنىڭىزدەك بولار مەريەمگۈل ،تەلىيىمىزگە بىر سىنىپتا ئوقۇپ قالساق ھە،قالغان خىياللارىمنى روھى دۇنيارىمدا ئۆزۈملا قىلدىم .....

ئالى مەكتەپتە ئوقۇش ئارزۇسى ،ئوقۇشقا ئىنتىلىدىغان ھەربىر كىشىنىڭ ئۇلۇغۋار چۈشى بولسا كېرەك ،لېكىن رئاللىق ،بىزنى قەيەرگە تاشلا <<ئۈچتە ياخشى ئوقۇغۇچى >>بولۇپلا ئۇزۇن ئۆتمەي بولغان بىر ئىش بۇ ئۇلۇغۋار چۈشۈمنى قاباھەت پاتقىقىغا سۆرەپ كېردى ،شۇ پۇل ،....يەنە شۇ پۇل ....
كىتاب ئوقۇش پۇلى تاپشۇرۇڭلار دېگىنىگىمۇ مانا بىرەر ئايدىن ئاشتى ،سىنىپ مۇدىرىمىز قانچىلاپ كاپشىپ كەتمىدى ،ئەلۋەتتە ئۇلارغىمۇ ئاسان ئەمەس دە ،ئاخىرى ئىسمىمىزنى ئاتاپ تۇرۇپ ،كىتاب پۇلىنى ئەتە ۋاقتىدا ئەكىلىشىمىزنى تاپىلىدى .
بىشىم قىتىپ پۇتلىرىمنى تەستە يۆتكەپ غىرىبانەكەپەمگە خوشياقماسلىق بىلەن يىتىپ كەلدىم ،بۇئائىلە ماڭا شۇنچىلىك سوغۇق ،شۇنچىلىك مېھىرسىز بىلىنەتتى .بەلكىم قەدىردان ئانام بۇدۇنيادىن كېتىش بىلەنلا بۇ ئائىلە مۇشۇنداق بىلىنىدىغان بولۇپ قالسا كېرەك .
دادامنىڭ سۇلغۇن خىرە چىرايى ،ئۆگەي ئانامنىڭ نارازىلىق ئارىلاشقان سوغۇق چىرايى ،كىچىككىنە ئىنىمنىڭ ئاپپاق غۇبارسىز قەلبى .بۇلارنى كۆرۈشكە راھىم قالمىغانىدى ئۇزۇن ۋاقىت بىر كىشىگە ئارىلاشماي ياشىغاچمىكىن مىجەزىممۇ تەرسا ،جاھىل ،تىرىككەك بولۇپ چوڭ بولغانىدىم ،ئۆيگە قارىسام دادام تېخى كەلمىگەن شۇ ڭا بۇ ئىككىيلەنگە بىرەر ئىغىزمۇ گەپ قىلماي ئۆزۈمنى ھوجرامغا ئاتتىم .
ئەتىسى چاينى ئىچىپ بولۇپلا دادامغا كىتاب پۇلىنىڭ گىپىنى قىلدىم .
-دادا مۇئەللىم كىتاب پۇلىنى بەك سۈيلەپ كېتىپ بارىدۇ ؟پۇل بولسا بەرسەڭ .
دادام ئەمدى گەپ قىلاي دەپ تۇرىشىغا ئۆگەي ئانامنىڭ سۆرۈن ئاۋازى ئاڭلاندى .
-تخى ئىكۈنىلا سەكسەن كوي ئىلىپ ماڭغانغۇ؟قانداق مۇئەللىم ئۇ پۇل دەۋىرىدىغان .
-بىر يۈز ئوتتۇز كوي كىتاب پۇلى كەپتىكەن ،يەنە ئەللەك كوي قالغان شۇنى ....
گىپىمنىڭ ئاخىرىسىنى يۈتىۋەتتىم .
ئۆگەي ئانام مېنىڭ تەرسا مىجەزىمنى بىلگەچكىمىكىن تاڭ يەنە گەپ چىقىپ قالمىسۇن دەپ جىم بولدى .
دادام تەستە يانچۇقىدىن پۇلنى ئىلىپ ماڭا ئۇزاتتى .
مەن پۇلنى ئىلىپ ھويلىغا چىقا-چىقمايلا ئۆگەي ئانامنىڭ قاتتىق ئاۋازى ئاڭلاندى .
-داۋاملىق مۇشۇ بالاڭلا خەجلەيدۇ پۇلنى مېنىڭ بالامچۇ؟مەنچۇ ؟
دادامنىڭ بى ئىلاج زەئىپ ئاۋازى ئاڭلاندى .
-تېخى نۇئەخمەتجان كېچىك ،پۇل دېسەڭ ساڭىمۇ بىرىۋاتىمەنغۇ؟
-ھەبىرىۋاتىسەن ،.....مۇشۇ بالاڭلا بولىدىكەن ،بۇ ئۆيدە يا مەن تۇرىمەن ،يا ئۇ بالاڭ تۇرىدۇ .

ھەي <<داداڭ ئۆگەي بولسىمۇ ،ئاناڭ ئۆگەي بولمىسۇن>>دېگەن گەپنى كىم ئىيىتقان بولغىيتتى ،ئانا ھەرقانچە تۇللۇق تارتسىمۇ يىتىم قالغان بالىسىنى بىقىپ قاتارغا قوشمىغىچە ئەرگە تەگمەي شۇ بالىنى چوڭ قىلىدىكەن ،ئاتىچۇ؟بىر يىلغا قالماي بويتاقلىققا چىدالماي ئۆيگە يات بىر ئانىنى باشلاپ كېرىدىكەن .
ئۆگەي ئانا مىڭ ككيۈمچاندەك قىلسىمۇ ،يەنە بىر يەرلىرىدىن بالىغا چاندۇرۇپ قويىدىكەن ،بالا مىڭ ئۆز ئانامدەك كۆرەي دېسىمۇ قىلىنغان مۇئامىلە ،گەپلەر ئادەمنىڭ يۈرىكىنى ئىزىدىكەن .
مەنمۇ ئۆگەي ئانام بىلەن ياشىغاندىن بىرى گەرچە ئازراق تالاش-تارتىش قىلىپ قالغان بولساقمۇ بۈگۈنكىدەك گەپلەرنى ئاڭلارمەن دەپ ئويلىمىغانىدىم .
ئۆزۈمنى پەقەتلا باسالماي ئارقىمغا يىنىپ ئۆيگە كېردىم .
دادامنىڭ نائىلاجلىق ئارىلىشىپ كەتكەن بىچارە چىرايى ،ئۆگەي ئانامنىڭ چاچراپ چىقىپ كېتەي دەپ قالغان كۆزلىرىگە دۇچ كەلدىم شۇنداقتىمۇ ئاچچىغىمدا ئۆگەي ئانامنى سەنلەپلا چۈشتۈم .
-ھەي ساڭا پۇل لازىممىتى ،چىكەڭگە تاڭەۋە بو پۇلنى شۇنداق دەپلا پۇلنى ئۇنىڭ ئالدىغا ئاتتىم .
ئۆگەي ئاناممۇ ماڭا تىل سىلىپلا چۈشتى .
-ماڭا پۇل لازىم ،ھە نېمە دەيسەن يىتىم ئوغلاق ،ئاناڭ دېگەن جادىمۇ بۇ دىۋانىنى روناق تاپقۇزماي شۈلۈپ كەپتىكەن ،ئەمدى سەنمۇ شۇنداق قىلاي داۋاتەمسەن ،ئاڭلاپ قوي پۇلنى سەنلاخەجلىمەيسەن ماۋۇيەردە مېنىڭ بالاممۇ بار ئاڭلاپ قوي .
-ئانامنى تىلغا ئالماۋۇ.....ئاغزىمنى بۇزۇپ تىللاپلا چۈشتۈم .
-چىقەۋە ئۆيدىن يىتىم ئوغلاق .
...
ۋاھ خۇدا ساقلىسۇن بۇنداق بىشەم خوتۇنلاردىن ،ئىغىزلىرى پىلموتنىڭ ئوقى ،بۇلارنىڭ ئالدىدا مەندەك بىر سويما تەڭ كىلەلەيتىمۇ؟
ئۆيدىن ئوقتەك چىقىپ كەتتىم .
مانا شۇكۈنى تۇنجى قىتىم دەرس تاشلاپ ،بىشىم قايغان پۇتۇم تايغان يەرلەردە چۆرگىلەپ يۈردۈم ،بۈكۈنى نەقىلسۇن مەندىكى گۈزەل ئارزۇلار ،نەقىلسۇن ھىلىقى بىخلانغان مۇھەببەت ،ھەممىسى قۇرۇق چۈش ھەممىسى كۆپۈككە ئايلانغاندەك بىلىندى ،كەچكىچە خامۇش خىياللارنى قىلدىم ،چىگرىسى يوق خىياللار پايانى يوق دېڭىزدەك ،مېنى ئۆزىدە شۇڭغىتىپ شۇ چوڭقۇر دىڭىز تەكتىگە جەسىتىمنى تاشلىماقتا ئىدى .بۈكۈنى قاراڭغۇ چۈشكەننى تۇيمايلا قاپتىمەن .
14ياشلىق بىر بالا كېچىنى قەيەردىمۇ ئۆتكۈزەلەيدۇ ،كۈزنىڭ سوغۇقلىرىدىن قەيەردىمۇ پاناھلىنالايدۇ ،ئاخىرى ئىغىر قەدەملىرىمنى يۆتكەپ ،ھىلىقى ئارزۇلىرىمنى قەبرىسىگە كېردىم .
كىردىميۇ ،ماڭا تەككەن تىل -ئاھانەت ،دەككە-دەشنام ،تاياق -توقماقتەك يۈرىكىمگە تەگمەكتە ئىدى .بۇ ئۆيگە كەچتە كېرگىنىمگە مىڭ پۇشايمان قىلدىم ،ئۆيگە كېرگۈچە ماشىنىنىڭ ئالدىغا ئۆزۈمنى ئىتىۋالسامچۇ؟مۇزلاپ قالساممۇ تالادا ياتسامچۇ دېگەنلەرنى خىيال قىلىپ كەتتىم ،ئىسىت .
بۈكۈندىكى مەنزىرىنى بىر ئاللا بىر مېنىڭ قەلبىمىدىلا قالسۇن ،ئاھ شۇ كېچىسى ئۇيقۇم ھارام بولدى ،بىچارە دادام ئارىغا كېرىۋالمىغان بولسا ئۆگەي ئانام مېنى يەپ كېتەرمىكىن تاڭ .
تاڭ يورىمايلا ئۆيدىن ئۆزۈمنى سىرتقا ئاتتىم ،كېچىدىن بىرى ئاڭلىغان تىل -ھاقەرەتلىرىم ،ئانام ئۈستىدە توقۇلغان بەتناملار يۈرىكىمگە نەشتەر سانجىغاندەك سانجىلىپ زەردابقا تولۇپ كېتىپ باراتتى .
بۇنداق ۋاقىتتا ئازابلىرىمنى كىمگىمۇ ئىيتالايتتىم ،دەردلىك كۆز ياشلىرىمنى نەگىمۇ تۆكەلەيتتىم .
ئۆزۈمنى مېھرىبان ئانامنىڭ قەبرىسىگە تەرەپكە ئاتتىم ...
ئانامنىڭ كىچىككىنە قەبرىسىنى كۆرۈپلا ئۆزۈمنى تۇتالماي يىغلاپ كەتتىم .
ئانا ،ئازاب تارتقان بىچارە بالاڭ كەلدى ،كۆرىۋاتامسەن ئانا ،سەن تاشلاپ كەتكەن يىللاردىن بىرى دەككە-دەشنام مېنىڭ ،غەم-غۇسسە مېنىڭ بولۇپ كەلدى ئانا .ئىھ ئانا قوينۇڭنى ئاچساڭچۇ؟سېنىڭ ئىللىق مېھرىڭگە بىر قىنىۋالسام .ئانا غىرىب بولغان بالانى كۆرىۋاتامسەن ،كۆزۈمدىكى ھەسرەتلىك كۆز ياشلىرىمنى كۆرىۋاتامسەن ئانا .....ھىھ ....ھىھ....ھىھ....كۆز ياشلىرىم يۈزلىرىمنى بويلاپ سىرغىپ چۈشتى ،چۈشتىيۇ كۈز شاماللىرى ئۇچۇرۇپ تاشلىغان ،غازاڭلارنىڭ ئۈستىگە تامچىلىدى .
قەبرە بىشىدا قانچىلىك ئولتۇرغىنىمنى بىلمەيمەن ،بىر ۋاقىتتا ياش يۇقى ككزلىرىمنى سۈرتۈپ ئاستا ئورنۇمدىن تۇردۇم ،كۈزنىڭ سوغۇق شاماللىرى ،يۈزلىرىمنى يەڭگىل سىيپاپ خۇدى مېھرىبان ئانامدەك سۆزلىمەكتە ئىدى .
-ئاھ !بالام ،كۆزۈڭنى يىراققا تىك ،يىراقلارغا نەزەر سال بالام ،يوللۇرۇڭ ئوچۇق بولغاي .......
بۇ تەسۋىرلەر خۇددى مەريەمگۈل يازغانغا ھەجەب ئوخشايدۇ ھە!؟
سەھەردىكى شەبنەم تامچىسى .
تۇنجى بىخلانغان مۇھەببەت .




ئارىدىن نەچچە يىللار ئۆتتى ،خىلى كۆپ ئەسلىمىلەر قەلبىمنىڭ قات-قىتىلىرىغا كۆمۈلۈپ كەتكەن بولسىمۇ شۇ قان-ياشلىق ئەسلىمە تاھازىرغىچە قەلبىمدە ئۇيۇپ قالغان داغدەك تۇرۇپ كەلدى .
يۈرىكىمنىڭ بىر يەرلىرىدە ھىلىقى ئۆچمەس سىما مەريەمگۈلمۇ تېخى كۆز ئالدىمدىن كەتمىگەنىدى .
تولۇقسىزنى پۈتتۈرۈش ئالدىدا مەريەمگۈلنىڭ دېگەن ئاخىرقى بىر سۆزى ھىلىمۇ قۇلىقىمدا پىچىرلىماقتا .
-خۇدايىمغا ئامانەت قەيسەر ،ئامان بولۇڭ .
بۇ سۆز ھازىرمۇ خۇددى شامال پىچىرلىغاندەك قۇلاق تۈىمدىن ئەگىپ ئۆتۈپ كەتتى ،شۇ ئۆتكەنچە پەشتاقتىن ئاستا چۈشۈپ كەتتى .شۇندىن بىرى ئۇسىما مەندىن بار-بارا يىراقلاشتى ،بەلكىم ناھايىتى يىراققا قىيامەتكىچە يىراقلاشقاندۇ؟

كېيىن ئاڭلىسام ئۇقىزمۇ ئوقۇشتىن مەھرۇم بولۇپ كۆپ ئۆتمەيلا تۇرمۇشلۇق بوپتۇ ،ھىلىقى ساقاللىق ھاجىم دادىسى قىز بالىنىڭ ئۇقۇشىغا قارشى بولغاچقا ئۇنى ئوقۇغىلى قويماپتۇ .
سەھەردىكى غەبنەم مېنىڭ كۆزۈمدىلا ئەمەس ئۇقىزنىڭ كۆزىدىمۇ تامچىلاپتۇ دە!؟
ھىلىمۇ ئىسىمدە شۇ ئىشتىن كېيىن ئوقۇشتىن بەكلا چىكىنىپ كەتتىم ،ھەتتا خەنزۇتىلى بىرىدىغان مۇئەللىمىم بىلەن سوقۇشۇپ سىنىپ مۇدىرىمىزنىڭ تەنقىدلىشىگىمۇ ئۇچرىدىم .
بىر كۈنى ئۇ مۇئەللىم مېنى قەستەن چۈشۈرمەكچىمۇ بولدىمۇ تاڭ ،<<ئۈچتە ياخشى ئوقۇغۇچى >>دېگەننى ئالاھىدە ئۇرغۇلۇق قىلىپ خەنزۇچە :
-ھە ،<<ئۈچتە ياخشى ئوقۇغۇچى >>كەلگۈسىدىكى ئارزۇيۇڭ نېمە؟.
مەن جاۋاب بىرەلمەي بىر ھازا تۇرۇپ قالدىم .ھازىرغىچە جاۋاب بىرەلمەي ئويلىنىپ قالدىم ،مېنىڭ ئارزۇيۇم زادى نېمە؟
بىر ھازادىن كېيىن ئارقىمدىن يەڭگىل ئاۋاز ئاڭلاندى .
-يازغۇچى .
ئارقىمغا لەپپىدە قارىدىم ،ھىلىقى شەرمى-ھايالىق قىز مەن ئۈچۈن جاۋاب بەرگەنىدى .
خەنزۇتىلى مۇئەللىم ھەيرانلىق ئارىلاشقان مەنسىتمەسلىك نەزىرىدە ماڭا قاراپ قويدى .
ھەي ھاجىم داداڭنىڭ قۇربانى بولغان قىزچاق ،ساڭا دېسەم مەن ھازىرغىچە ئويلاپ قالىمەن ،ھاجىمنىڭ قىزى ئوقۇسا بولمامدىغاندۇ؟
شەرىئىتمىزدە قىزلارنى ئوقۇتۇش گۇناھمىدۇ يا؟بۇ سۇئاللارنى ھازىرغىچە تاپالمايۋاتىمەن .
راست مېنىڭ ئارزۇيۇم يازغۇچى بولۇش ئاۋۋال ئاۋۋال سېنى ھەم ئۆزۈمنى يازاي ئارزۇيى گۇمران بولغان ،شىۋىرلىغان قەلبى يۈرىكىدىلا قىتىپ قالغان ناتىۋان قىزنى يازاي .


ئۈمىدسىزلەنمە ئەي ئىنسان شۇكۆزىمىزدىن تۆكۈلگەن شەبنەم تامچىسىدىنمۇ قۇياشنىڭ زەررىسىنىمۇ كۆرگىلى بولىدۇ ،ئۈمىد ماڭا ،ھەم ساڭا ،ھەممىمىزگە ھەمراھ بولسۇن .
