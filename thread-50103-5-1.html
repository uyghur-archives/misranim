   مۇشۇ تېمىغا كىرىپ ئىنكاس يېزىپ قاقشاۋاتقان ئاياللارنىڭ ھەممىسى گۇناھنى ئەرگە قۇيىدىكەن . مەن نىمە قىلدىم دەپ ئۆزىدىن ھىساپ ئالىدىغان ئاياللار بەك ئازكەن . ئەڭ بولمىغاندا مەن ئۆزەمنىڭ ئايالىنىڭ ئەڭ كىچىك يولسىزلىقىنى دەپ بىرەي : مەن نىمە ئالىمەن دىسەم شۇنى ئېلىپ بىرىسىز ، مىنى بىر ئۆمۈر پۇلدىن قىسمايسىز . ئاتا - ئانام نىمە دىسە ماقۇل دەيسىز . كىيىنىشىم بىلەن كارىڭىز بولمايدۇ .
   مۇشۇنداق تەلەپكە دىگەن لەۋزىنى جېنىدىن ئارتۇق كۈرىدىغان كىم ماقۇل دىيەلەيدۇ ؟( ھەممە ئىشنىڭ ئۆز يۇلى بار ئەمەسمۇ ؟)
  ئاشۇنداق چاغدا  ئاغزىمنى بۇزۇۋەتكۈم كىلىدۇ :  ماڭا ئۇنداق شەرت قۇيىدىغانغا مەندە ئاناڭنىڭ ھەققى بارمۇ ؟ دەپ ، لېكىن ئۇنداق دىمەيمەن ، مۇشۇنداق كىتىلىۋەرسە ھامان بىركۈنى ئۇنى تاشلىۋىتىدىغىنىمنى بىلىمەن .
  بەزىدە ئويلاپ قالىمەن ، بۇنداق قىينالغاندىن ئۇنىڭدىن مۇناسىۋىتىمنى ئۈزۈپلا مىسرانىمدىكى ئوبدان قىزدىن بىرنى تاپايمۇ دەپ . ئۆز ئاچا-سىڭىللىرىمغا راۋا كۆرگەن مۇئامىلىنى ئايالىمغا قىلىمەن دەپ تىرىشقانلىرىمنىڭ ئۇنىڭغا  مۇشۇنداق تەسىر بەرگەنلىكىگە ئەپسۇسلىنىمەن ! مىنىڭ مەسئۇلىيەتچانلىقىمنى ئاجىزلىق دەپ چۈشەنگىنىدىن  ساپاسىزلىقنىڭ قانداق بۇلىدىغىنىنى چۈشىنىپمۇ قالدىم . ئەڭ ياخشىسى ئەر كىشى قويغا قويچە ، بۆرىگە بۆرىچە مۇئامىلە قىلىشى كىرەككەن دەپ قالىمەن .
كىشى كۆركى يۈزدۇر ، ئول يۈز كۆركى كۆز ، ئەقىل كۆركى تىلدۇر ، ئول تىل كۆركى سۆز !
