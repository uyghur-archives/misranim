ياراتقان بۈيۈك ئاللاھ ھىدايەت ئاتا قىلغان كىشىلا ھاياتىدا ئۇچرىغان ھەر قانداق
سوئالغا ئاسانلا توغرا جاۋاب تاپالايدۇ.مۇشۇ تۈردىكى سوئاللارنىڭ ھەممىنى يىغىپ
بىرلا جاۋاب بېرىش ھەم مۇمكىن،مەن بۇ يەردە ئۆزبېك ئەللامە ئابدۇۋەلىي قارى ئا-
كىمىزنىڭ مىساللىرى بىلەنلا كۇپايە قىلاي:
  چۆچەك،رىۋايەتلەردە كۆپ ئۇچرىتىمىز،قەھرىمانلار كۆپىنچە جادۇگەر،يالماۋۇزلار
بىلەن تۇتۇشۇپ قالسا ئۇنىڭ بىر قولىنى چاپسا،ئورنىغا يەنە بىر قول چىقىدۇ،بېشىنى
چاپسا ئورنىغا يەنە بىر باش چىقىدۇ،ئاخىرى بىر ئاقساقال بوۋاي چىقىپ يالماۋۇزنى
ئۆلتۈرۈش يولىنى كۆرسىتىپ بېرىدۇ،ئۇ قانچە تاغ-داۋانلار ئېشىپ،ئورمانلىقتا مەلۇم
دەرەخنى تېپىپ،ئۇندىكى ئۇۋىدىكى قۇشنى تۇتۇپ يۈرىكىنى مىجىسا ئاندىن ھېلىقى
يالماۋۇز ئۆلىدۇ...
   بىز ئەر بۇزۇلدى،خوتۇن بۇزۇلدى دەپ قانچە قاخشىغان بىلەن،قانچە ئامال قىلغان
بىلەن تۈزەپ بولالمايمىز،خۇددى ھېلىقى يالماۋۇزنىڭ بېشىنى چاپقاندەك...بىرنى تۈ-
زىسىڭىز بىرى چىقىپ  تۇرىدۇ،تۈزەپ بولالماي ھېرىپ قالىسىز،زېرىكىپ قالىسىز،ئاخىرى
شۇ يالماۋۇز بىلەن دوست-ئاغىنە بولۇپ كېتىسىز...يوقىتىمەن دېسىڭىز چوقۇم ھېلىقى قۇشنى تېپىپ يۈرىكىنى مىجىشىڭىز كېرەك،ئۇ بولسىمۇ ئىسلامىي دۇنيا قاراش،ئەگەر سىز
ھەقىقىي يوسۇندا ئىسلامغا بويسۇنماي تۇرۇپ قانداقتۇر ئەخلاق،ھايا دېگەنلەردىن
گەپ ساتسىڭىز بىر ئۆمۈر ئاۋارە بولىسىزكى،قىلچە نەتىجىگە ئېرىشەلمەيسىز،ئەگەر
ئىسلامغا بويسۇنۇپ،24 سائەتلىك ھەرىكىتىڭىزنى مۇشۇ مىزانغا سالسىڭىز،يالماۋۇزنىڭ
ھالاك بولغىنىنى كۆرىسىز،ئەگەر قاچانكى ھىدايەتتىن ئارقىغا زەررىچە بۇرۇلىدىكەن،
يالماۋۇزنىڭ جان كىرىپ مىدىرلىغىنىنى كۆرىدۇ...

ئاپتور جاۋابىنى بويلاپ 28قەۋەت QIQAK0903 كە 2011-05-17 10:39 PMئەۋەتىلدى :
ئەرلەرنىڭ تالاغا قاراپ قېلىشىنى ئىككىلا تەرەپتىن ئىزدىگىنىمىز  ياخشى . ئەر ناچار ئۈژمە كۆڭۈل بولسا ، ئايال ئۇنى ئۆزگەرتىشكە تىرىشىشى كىرەك . ئەقىللىق ئايال ئىرىنى ئۆزگەرتەلەيدۇ . ئەرنى ئەر قىلغانمۇ خوتۇن  ، يەر قىلغانمۇ خوتۇن .


بۇ مۇنبەرداشنىڭ پىكرىگە مەنمۇ قوشىلىمەن . مەن مېنىڭ كۆز ئالدىمدا بولغان بىر ئەھۋالنى ئېيتىپ بېرەي :
مېنىڭ بىر دوستۇم بار ئىدى ، ئۇ ئەر-ئايال ئىككەيلەننىڭ بىر قىز ، بىر ئوغلى بار . تۇرمۇش ئەھۋالى ئىنتايىن ياخشى . ئۇلارنىڭ ئۆزىنىڭ رىستۇرانى بار ئىدى .  شۇنىڭ ئۈچۈنمىكىن دوستۇمنىڭ يولدىشى ئاساسىي جەھەتتىن كۈندە دىگۈدەك سىرتتا سورۇنلاردا يۇرەتتى . ھەتتا نەچچە ئاشنىسىنى بۇلار يېڭى كەلگەن ئۇسۇلچىلار دەپ ، رىستۇرانىغا ئەكىلىپ ئىشلىتەتتى ، بەزىدە مەس بولۇپ قالسا ، ئۇ قىزلارغا ئايالى بار يەردىلا چاقچاق قىلاتتى . لېكىن مېنىڭ دۇستۇم ئەزەلدىن ئېرى ئالدىدا ياكى بىزنىڭ ئالدىمىزدا دەرت ئېيتىپ زارلانمايتتى ، يا ئېرىنىڭ گېلىنى سىقمايتتى . شۇ تەرىقىدە نەچچە يىل ئۆتتى ئاللاھنىڭ قۇدرىتى بىلەن ئۇ ئەر ھازىر شۇنداق ياخشى (بىلگۈچى ئاللاھ تائالا مەن شۇنداق دەپ ئويلايمەن . تىلىم تېگىپ كەتمىسۇن ) . ئوغلى چەتئەلدە ئوقۇپ كەلدى ، قىزىمۇ يېقىندا ئالى مەكتەپكە ئىنتىھان بېرىدۇ . مەن يېقىندا ئۇ دوستۇم بىلەن كۆرۈشۈپ قالدىم . ئۇنىڭدىن سورىدىم ، سەن بۇرۇن ئېرىڭنىڭ ئالدىڭدا قىلغانلىرىنى بىلەمتىڭ ياكى بىلمەسكە سالغانمۇ دىسەم ، نىمە دىدى دىمەمسىز ؟
بىلىمەن ئاداش ، مەن ئەقىلسىز دۆت بولمىسام ، بىلسەم قانداق قىلىمەن ، بىلدىم دەپ ئۇنىڭ بىلەن ئۇرۇشامدىم ، ئۇنداق قىلسام ، خەلقى ئالەم ئالدىدا رەسۋا بولىدىكەنمىز ، بالىلىرىم يىتىم بولىدىكەن ، ئۇنى ئاز دەپ ئۇنىڭغا يولنى ئۆزەم ئېچىپ بەرگەن بولمامدىم . مەن شۇ ۋاقىتلاردا يىغلاشتىن ، ئاللاھ تائالادىن ئۇنىڭغا ئىنساپ تىلەشتىن باشقىنى لايىق كۆرمىدىم ، بۇنداق قىلىشىم بالىلىرىم ئۈچۈن دىگەن ئىدى . مەن بۇ دوستۇمنىڭ سۆزىدىن بەك سۆيۈنۇپ كەتتىم .
شۇنىڭغا بۇ مەسىلىنىڭ جاۋاپكارلىقىنى بىر تەرەپلىمە قاراشقا بولمايدۇ .
ئەر-ئاياللارنىڭ بۇزۇلىشى ، ئۇلارنىڭ ئىرادىسىزلىكىدىن ، ئىمانننىڭ سۇسلىقىدىن بولىدۇ .
ئاللاھ تائالادىن مىللىتىمگە ھىدايەت تىلەيمەن .
نىمىلەرنىدۇ تالىشار ئىكەنمىز ئۆز - ئارا، بىر-بىرىمىزگە يول بەرمەستىن كولاپ ئورا. ئىنسان دىگەن تويماس ئىكەن قارنى بالا، ئالەممۇ ئۇنىڭ نەپسىگە ئازلــىـــق قىلا.

چاۋاك بىر قولدىن چىقمايدۇ،  تۇرمۇش قۇرۇپ  ئۆي تۇتۇش مۇھاببەتلىشىپ يۈرۈشكە ئوخشىمايدۇ، ئەر -ئاياللار ئوتتۇرىسىدىكى ئىشلار پەقەت شۇ ئەر -ئاياللارنىڭ ئۆزىگىلا مەلۇم ،.......<<تالاغا قاراپ قېلىش>>تىكى مەسئۇلىيەتنى بىر تەرەپكىلا ئايرتىش توغرا ئەمەىس،ئەرمۇ ،ئايالمۇ ئۆز مەجبۇرىيىتىنى تولۇق ئادا قىلغاندا،بىر -بىرىگە ھەقىقىي مېھرى -مۇھاببىتىنى بېغىشلىغاندا تالاغا قاراپ قېلىشتىن ساقلانغىلى بولىدۇ......مەيلى ئەر بولۇڭ ياكى ئايال بولۇڭ يولدىشىڭىز،بالىلىرىڭىز ، ئائىلىڭىز ئۈچۈن ئۆزىڭىزنى بېغىشلاڭ،شۇنداق قىلغاندا  <<تالاغا قارىماي>>ئۆيگە قارايدىغان بولىسىز......اللاھ ھەرقانداق بەندىسىنىڭ كۆڭىلگە ئىنساپ بەرسۇن......


ئەرلەرنىڭ تالاغا قاراپ قېلىشىنى ئىككىلا تەرەپتىن ئىزدىگىنىمىز  ياخشى . ئەر ناچار ئۈژمە كۆڭۈل بولسا ، ئايال ئۇنى ئۆزگەرتىشكە تىرىشىشى كىرەك . ئەقىللىق ئايال ئىرىنى ئۆزگەرتەلەيدۇ . ئەرنى ئەر قىلغانمۇ خوتۇن  ، يەر قىلغانمۇ خوتۇن .

مۇشۇ ھەقىقى ئىنكاس بۇپتۇ ، كۆردىڭىزمۇ؟ كۈڭۈل خان ئەمدىغۇ بىر يەلىرىڭىز قانغاندۇ.
