    ئەسسالامۇ ئەلەيكۇ م:
   بۇ ئەسلى  < ئەگەر مەن باشقۇرغۇچى بولسام > دىگەن تىمىغا يېزىلغان تىما ئىدى، مەزكۇر ماقالىمۇ مۇنبەر توغرىسىدا بولغاچقا ئەشۇ ئىنكاسنىڭ بىر قىسمىنى بۇ يەرگە ئەكىلىپ چاپلىدىم.
    مۇنبەر ئۈچۈن بەرگەن تەكلىۋىڭىزگە رەھمەت. بىز ،  ئەزالارنىڭ ئىلمى، ئەمەلى تەكلىپلىرىنىڭ مۇنبەردە كۆرۇلۇپ تۇرىشىغا تەشنا.
     مۇنبەرگە يېڭى كىرگەن چاغلىرىمدا مەندىمۇ  مۇنبەر توغرىسىدا، باشقۇرغۇچى توغرىسىدا ، ئەزالار توغرىسىدا نۇرغۇن تەكلىپلەر شەكىللەنگەن، كېيىنچە ۋاقىتنىڭ ئۆتۇشى، مۇنبەر تەجرىبىسىنىڭ بارا- بارا شەكىللىنىش جەريانىدا  بەزى قاراشلىرىمنىڭ ئەمەليەتكە ماس كەلمەيدىغانلىقىنى ھىس قىلغان ئىدىم.
    ھەممىمىز مۇنبەرگە نادىر تىمىلارنىڭ، ئەھمىيەتلىك تىمىلارنىڭ كۆپلەپ چىقىشىغا ، نادىر، تېتىملىق ئىنكاسلارنىڭ كۆپلەپ بولىشىغا تەشنا. ئەمما ئەمەليەت ئۇنداق بولمايۋاتىدۇ. ئەمىسە مۇشۇ مۇنبەردىكى ئەمەليەت قانداق ؟
    نۇرغۇن تورداشلار يازما يوللاشنى بىرىنچى ئورۇنغا قويىۋالغان، شۇڭا ئۆزى يازالمىسىمۇ باشقا تور بەتلەردىن يۆتكەپ ئەكىلىشكە ئامراق، بۇنداق يازمىلارنىڭ كۆپۇنچىسى ئاساسەن تەستىقلانمايدۇ، ئەزالار كۆپۇنچە كۈچىنى يازما يوللاشقا قارىتىۋالغاچقا  ، يوللانمىسى تەستىقلانسا خوش بولىدۇ، تەستىقلانمىسا باشقۇرغۇچىدىن ئاغرىنىدۇ، شۇڭا قوساق كۆپىكىنى بىر قىسىم يازمىلارغا ئىنكاس يېزىش جەريانىدا چىقىرۋالماقچى بولۇشۇپ، بەزىبىر تىمىلارنى بىردەمدىلا تاقاشقا مەجبۇرلايدۇ، شۇڭا تورداشلاردىن كۈتىدىغىنىمىز ، ئامال بار ئەھمىيەتكە ئىگە ، مەلۇم بەدىئىلىكى بار يازمىلارنى يوللىشىنى ئۈمىت قىلىمىز.
    ئەمدى ئىنكاس مەسىلىسىگە كەلسەك: مىسرانىم مۇنبىرىدە يوللىنىلىدىغان ۋە  ئېلان قىلىنىدىغان تېمىلار كۆپ بولغاچقا ئەزالار  تېمىلارغا  ئىنكاس  يېزىشتا    <يىتىشىپ >بولالمايدۇ، شۇڭا  ، ئاسان چۈشۈنەلەيدىغان، كاللىسىنى غىدىقلىيالايدىغان تېمىلارغا ئىنكاسلار كۆپرەك يېزىلىۋاتىدۇ، باش قاتۇرۇشقا تىگىشلىك، مەلۇم سەۋىيە تەلەپ قىلىنىدىغان تىمىلار بويۇن قىسىپ قېلىۋاتىدۇ،  بۇ ، خۇددى ۋالىبونى كۆرگەنلىكى ئادەم ئوينىيالمىغاچقا توپ مەيدانغا كىرەلمىگەن بىلەن ، پۇتبولنى تېپىپ قويسىلا بولىدۇ دەپ چۈشۈنىۋېلىپ ،ئۆزىدە توپ تېپەلەيدىغان ئىككى پۇتنىڭ بولۇشى سەۋەپلىك  پۇتبول مەيدانىغا كىرىۋالغاندەكلا ئىشلار كۆپ بولىۋاتىدۇ، ئەمما تېپىلگەن توپنىڭ نەگە كىتىۋاتقانلىقى بىلەن چاتىقى يوق، ئىش قىلىپ توپ تەپسىلا بولدى.
   ئادەتتە ئىنكاس يېزىش ئۇنچە ئوڭاي ئىشمۇ ئەمەس، توردىكى دوستلارنىڭ بىلىم قۇرۇلمىسى ئوخشاش بولمىغاچقا ، مۇنبەردىكى يازمىلارنى تورداشلارنىڭ چۈشۈنىش دەرىجىسى ئوخشاش بولمايدۇ، بەزىلەر چۈشەنسىمۇ لايىق ئىنكاس يازالمايدۇ.
     ئۆزەمنى ئېلىپ ئېيىتسام مەن مۇنبەرگە يېڭى كىرگەن ۋاقىتلاردا كۆپۇنچە يازمىلارغا ئىنكاس يازالايتىم لېكىن كۇنلەرنىڭ ئۆتۈشىگە ئەگىشىپ ،  مۇنبەردە ھەرخىل تېمىلارنىڭ ئۇچرىشىغا ئەگىشىپ ھە دىگەندىلا ئىنكاس يازالمايۋاتىمەن  ،ئىنكاس يېزىشمۇ يېزىقچىلىقنىڭ بىر خىل تۈرى بولغاچقا ئۇ ، ئادەمدىن ئىلھام تەلەپ قىلىدىكەن، يازمىنى ئوقۇپ سىزدە ئىلھام قوزغالمىسا زورۇقۇپ ئىنكاس يېزىشقا مەجبۇر بولىدىكەنسىز، مۇنداق زورۇقۇپ، مەجبۇرى يېزىلغان ئىنكاسنىڭ ھىچقانچە قىممىتى بولمايدىكەن، شۇڭا بىر قىسىم تورداشلاردا < ياخشى يېزىپسىز، يەنىمۇ تىرىشىڭ > دىمەكتىن باشقا ئامال يوقكەن، شۇڭا ئاددى ئىنكاسلارنىمۇ توغرا چۈشۈنىش كېرەككەن دەپ ئويلاپ قالدىم.
   ئەمدى باشقۇرغۇچىلارنىڭ  < مۇھەررىرلىك > خىزمىتى مەسىلىسىگە كەلسەك بۇ ئويلىنىشقا تىگىشلىك  مەسىلىكەن، باشقۇرغۇچىلارنىڭ تېمىلارنى ئوقىماستىنلا تەستىقلاش مەسىلىسى ئادەتتە مەۋجۇت ئەمەس، ئەسەرلەرنى تەھرىرلەشكە كەلگەندە بۇ ،  قول تۇتىدىغان ئىش، ناۋادا تورداشلارنىڭ ئۆزى يازغان ئەسەرلىرىگە مۇشۇ تەھرىرلىك نۇقتىسىدىن مۇئامىلە قىلساق بەلكىم خېلى كۆپ يازمىلار لاياقەتسىز بولغاچقا ئۇنى باشقۇرغۇچىلار تەل- تۆكۇس تەھرىرلىمەكچى بولسا باشقىدىن ماقالە  قىلىپ يېزىشقا توغرا كىلىدۇ، تور ئەدەبىياتىدا مەتبۇئات ئەدەبىياتىغا قويۇلىدىغان ، ئەسەرنىڭ بەدىئىلىك ئۆلچىمى ئانچە يۇقۇرى بولمايدىغان بولغاچقا  بىر قىسىم ئەسەرلەرگە توغرا مۇئامىلە قىلىش كېرەك.
  ئادەتتە مەتبۇئاتتا مەلۇم بىر ژورنال ئايدا ياكى ئىككى ئايدا بىر چىقىدۇ، ئۇنىڭغا تەھرىرلەش ئېلىپ بارغۇدەك ۋاقىت يىتەرلىك بولىدۇ، تور ئەدەبىياتىدا بولسا مەلۇم يازما كۆپۇنچە ۋاقىتلاردا بىر ئىككى سائەتكە قالماي مۇنبەرگە چىقىرىلىدۇ، .ۇنى تەلەپكە لايىق تەھرىرلەيمەن دەپ ئولتارسىڭىز ، تەھرىرلەشنى كۈتۈپ تۇرغان شۇنچە كۆپ يوللانمىغا ۋاقتىڭىز يىتىشمەيدۇ، مۇنبەرنىڭ باشقۇرغۇچىلىرى مۇنبەر باشقۇرۇشنى ئىشتىن سىرىتقى ۋاقىتلىرىدا قىلىدۇ، ئۇلارنىڭمۇ شەخسى تۇرمۇشى  بولىدۇ،قىلىشقا تىگىشلىك خىزمىتى بولىدۇ، مۇنبەرداشلارنىڭ بۇنى چۈشۈنىشىنى ئۈمىت قىلىمەن.
     مەن ئۇيغۇرچە  نۇرغۇن مۇنبەرلىرىنى ئارىلاپ باقتىم، مىسرانىم مۇنبىرىدەك < چاپ- چاپ > مۇنبەرنى ئۇچراتمىدىم، مىسرانىمدا ھەر كۇنى ، ھەر سائەتتە چوقۇم يېڭى تىما بار، مانا بۇ مىسرانىمنىڭ ئۆزىگە خاس ئالاھىدىلىكى، ھەر كۇنى ئوتتۇرا ھىساپ بىلەن كۇنىگە 30 دىن تۆۋەن بولمىغان ماقالە مۇنبەر يۈزىگە چىقىدۇ، شۇڭىمۇ بىر قىسىم تورداشلار ئۇ بۇ مۇنبەرلەرنى ئايلىنىپ بېقىپ يەنىلام مىسرانىمدىن مىھرىنى ئۈزەلمەي مىسرانىمغا قايتىپ كىلىدۇ.
   مىسرانىم مۇنبىرىدە ھەر كۈنى توختاۋسىز يېڭى تىمىلار ئېلان قىلىنىپ تۇرىلىدۇ، بۇ مىسرانىم  مۇنبىرىنىڭ ئالاھىدىلىكلىرىدىن بىرى. بىزنىڭمۇ سەر خىل تىمىلارنى تاللاپ ئېلان قىلغۇمىز بار، ھەر سائەتتە قارايدىغان بولساق ئارقا بەتكە يېڭىدىن يوللانغان تىمىلار توشۇپ كىتىدۇ، شۇنىڭ ئىچىدىن بەزىبىر ئەھمىيەتلىك تىمىلارنى تاللاپ مۇنبەرگە چىقىرىمىز، تىمىلار تەستىقلانماي كۆپ ئۆچۈرىلىدىغان بولغاچقا تورداشلار بىزدىن شۇنچە خاپا، باشقۇرغۇچىلار بىلەن ئەزالار ئارىسىدا تام شەكىللىنىپ قالغاندەك ھالەت شەكىللىنىدۇ ، بەزىدە بىر نەچچە قۇر تەسىراتنى ماقالە قىلىپ يوللايدىغانلارمۇ بار تېخى، ئەسلىدا ئىنكاس شەكلىدە يازسىمۇ بولىدىغان سۆزلەرنى تىما قىلىپ يوللايدۇ، ئۇندىن سىرىت بەزى خەۋەر تەرەقىسىدىكى، سىياسى تىمىلارمۇ كۆپ يوللىنىدۇ، بۇنداق تىمىلارنىڭ بەزىلىرى قۇلۇپلىنىپ تەستىقلىنىدۇ، بولمىسا ھېلىقىدەك < توپ تېپىش > كە ھىرىسمەن تورداشلار بۇنداق تىمىلاردا يۈزەكى قاراشلىرى بىلەن ئىنكاس يېزىپ مۇنبەرگە خەۋىپ ئېلىپ كىلىدۇ.
  بىر دادا كۆپۇنچە  بىر ئۆينى ئاساس قىلىپ پىكىر قىلىدۇ، بىر رەھبەر كۆپۇنچە شۇ تەشكىلنى چۆرىدىگەن ھالدا پىكىر قىلىدۇ، بىر دۆلەت باشلىقى بىر دۆلەتنى ئاساس قىلىپ پىكىر قىلىدۇ، دىمەك چىقىش نۇقتا ئوخشاش بولمىغاچقا ئوخشاش ئوبىكىتكا بىرىلىدىغان باھامۇ ئوخشاش بولمايدۇ،  مۇنبار باشلىقىنىڭ ئويلايدىغىنى مۇنبەرنىڭ ساغلام تەرەققى قىلىشى ۋە ئۇزۇنغىچە مەۋجۇت بولۇپ تۇرىشىدۇر.
    بەزى تورداشلار < دىنى تىمىلار> نى كۆپرەك ئېلان قىلىش تەكلىپىنى بىرىپتۇ، مۇنبەردە بۇنداق تىمىلارمۇ يوق ئەمەس، مۇنبەر بىر خىل سەزگۈر ئورۇن، سىياسى پۇرىقى كۈچلۇك ئورۇن، بىز نەچچەلىگەن ئەزانىڭ تەلىپىنى قاندۇرىمىز دەپ بەزى ئۆتكۈر تىمىلارنى ئېلان قىلساق، ئۇنىڭدىن  چاتاق چىقسا ئاقىبىتىگە كىم مەسئۇل بولىدۇ ؟ شۇڭا پىكىر قىلغاندا ئەتراپلىق پىكىر قىلىش كىرەك، بىزنىڭ مۇنبىرىمىزدىكى دىنى تىمىلار تەلىپىڭىزنى قاندۇرالمىسى مەخسۇس دىنى مۇنبەرلەر بارغۇ ؟
    شۇڭا مۇنبەرداشلارنىڭ مۇنبەرنى چۈشۈنىشىنى ، ۋەزىيەتنى دەڭسەپ كۆرۇشىنى ، سەزگۈر تىمىلارغا ئەستايدىل بولىشىنى ئۈمىت قىلىمەن، ئېسىمدە تۇرۇپتۇ، بىر دىنى تىمىدا بىر ئايال تورداش < ناماز ئوقۇمىغان ئايالنىڭ ئەتكەن تامىقىمۇ ھارام > دەپ ئىنكاس يازغانتى، دىنى جەھەتتىن بۇنى چۈشۈنىشكە بولسىمۇ لېكىن دۆلەتنىڭ سىياسىتى مۇشۇ خىل قاراشلارنىڭ يامراپ كىتىشىگە يول قويارمۇ ؟ مۇشۇ خىل ئەھۋالنىڭ ئالدى ئېلىنمىسا ئاخىرى مۇنبەر < رەھمەتلىك > بولۇپ قالماسمۇ ؟مۇنبەرلەردە ھۆكۇمەت نازارەتچىلىرىنىڭ بارلىقىنى ئۇنتۇپ قالمايلى دوستلار.مۇشۇنداق چوڭ ياخشى ۋەزىيەتنى قەدىرلىمەي جەمىيەتتە ئۇچرىغان ئەرزىمەس ئىشلارنى مۇنبەرگە سۆرەپ كىرىشتىن ساقلىنالايلى ،دۆلەتنىڭ غەمخورلىقىنى ، سوتسىيالىزىم تۈزىمىنىڭ ئەۋزەللىكىنى  ئۇنتۇپ قالساق بولمايدۇ.
  مۇنبەر توغرىسىدا تەكلىپ پىكىرلەرنى داۋاملىق بىرىپ تۇرىشىڭلارنى ئۈمىت قىلىمىز، بەزىلىرى تەستىقلانمىسىمۇ لېكىن ھىچ بولمىسا باشقۇرغۇچىلار كۆرىدۇ، مىسرانىم كۆرىدۇ، تېما يوللىغۇچىغا رەھمەت، جاپا چىكىپسىز.
ھەقىقەت ئىشىكىنى داۋراڭ سېلىپ پو ئاتىدىغانلار ئەمەس، ئۈن -تۇنسىز  ئىشلەشنى ئادەت قىلغان تاماسىز كىشىلەر ئاسان تاپالايدۇ.
