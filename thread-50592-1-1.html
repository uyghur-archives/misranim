   تەبەسسۇم --قىز -ئاياللىرىمىزدىكى ئەڭ گۈزەل ، ئەرلەرنى ئەڭ جەلپ قىلىدىغان بىباھا زىننەت بويۇمى .
نۇرغۇن قىز-ئاياللىرىمىز گۈزەللىكنى قوغلىشىپ خىلمۇ-خىل گىرىم بويۇملىرى ۋە قىممەت باھالىق زىننەت بويۇملىرىنى سىتىۋالغان بىلەن چىرايىدا تەبەسسۇم ، ئوچۇق پىكىر بولمىغانلىقتىن ئەرلەر ئۇنداقلارغا قىزىقىپمۇ كەتمەيدۇ .
ئۆزى كۆرۈمسىز بولسىمۇ چىھرىدىن خۇش تەبەسسۇم ، جىلۋە كەتمەيدىغان ئايال يولدىشىغىمۇ ئۈمىت ۋە شاتلىق بىغىشلايدۇ .
ھەردائىم چىرايى ئىچىلمايدىغان ،دائىملا باشقىلاردىن ئاغرىنىپ ئۆتىدىغان بۇرۇقتۇرما ئاياللار قارىماققا گۈزەل
بولسىمۇ ئائىلگە غەم -قايغۇ ، ھەمراھىغا ئىچ -پۇشۇقى ئاتا قىلىدۇ .
   ئەگەر سىز بىر چىرايلىق قىزنى ياخشى كۆرۈپ قالغان بولسىڭىز ، ئۇ شۇنچە چىرايلىق بولسىمۇ تويدىن كىيىن ئۇنىڭدا باشقىلاردىن ئاغرىندىغان ، بۇرۇقتۇرما ،باشقا ئەرلەرنى سىز بىلەن سىلىشتۇرۇپ سىزنى تۆۋەن كۆرىدىغان ،
كۈن -كۈنلەپ چىرايى ئىچىلمايدىغان ،كوت-كوت مىجەزلىك بولۇپ ھەردائىم چىرايىدىن مۇز چاچرىتىپ يۈرسە
كۈنلەرنىڭ ئۆتۈشى بىلەن سىزنىڭ نەزىرىڭىزدە دۇنيادا شۇ ئايالدەك سەت ئايال يوقلۇغىنى ئازاپ ئىچىدە ھىس قىلىپ يىتىسىز .
مۇنداق ئاياللار ئەتراپىمىزدا خىلىلا كۆپ ئۇچرايدۇ .
مۇشۇنداق ناچار مىجەزنى يۇقتۇرۋالغان ئاياللىرىمىز ئەرلىرىنى تالاغا قاراشقا مەجبۇر قىلىدۇ .
چۈنكى ھەرقانداق بىر ئەر ئائىلسىدە تاپالمىغان شاتلىقنى ، خۇشاللىقنى باشقا ئاللىقانداق جايلاردىن ئىزلەشكە
مەجبۇر بولىدۇ .
                        
                    
                                 ئەرلەر قەلبىدىكى ئەڭ گۈزەل ئايال

                    
خاتىمە ؛ مىنىڭچە چىراي گۈزەللىگى ۋاقىتلىق نىقاپ .
قىپقىزىل پورەكلەپ ئىچىلغان قىزىل گۈلمۇ مەلۇم ۋاقىتتىن كىيىن توزۇپ كىشىگە ھۇزۇر بىرەلمەس بولۇپ قالىدۇ .
ئەرلەر قەلبىدىكى ئەڭ گۈزەل ئايال --ئەڭ ئىغىر كۈنلەردىمۇ ، باشقىلاردىن ناھەق خورلاشلارغا ئۇچرىغاندىمۇ
تاپا --تەنە بىلەن ئىرىنىڭ بىشىنى ئوچاق قىلدىغان ئەمەس بەلكى يولدىشىغا خۇش تەبەسسۇم بىلەن يارىلانغان قەلىبگە مەلھەم بولۇپ ئائىلىگە ئۇمىت ، شاتلىق ئىلىپ كىلدىغان ئايال ئەڭ سۆيۈملۈك ئايال .
ئايالىدىن قاتتىق بىزار بولغان بىر دوستىمىز مۇنداق دىگەن ؛ --كوت --كوتلۇق ئاياللارنىڭ پاتنىنتى .
مەن بىر يىل ئىچىدە 2 -ئايغا ئامراق .  چۈنكى بۇ ئاي باشقا ئايلارغا قارىغاندا 2--3 كۈن قىسقا .دەپ چاقچاق قىلغان ئىدى .
ئىرپانىنىڭ قارىشىچە؛  قىز --ئاياللىرىمىزنىڭ گۈزەللىگى ھەرگىزمۇ چىرايىدا ئەمەس .
بەلكى ئوچۇق چىراي خۇشخۇيلۇقى ، ئائىلە ئىشلىرىدىكى چىۋەرلىگى ، چاققانلىقى ھەم  بىلەن ھاياتىتىن
ئىبارەت بۇ ئۇزۇن سەپەردە يولدىشىغا مىھرىبانلىق،سۆيگۈ -مۇھەببەت بىغىشلاپ ، ئائىلسىگە شاتلىق ، خۇشاللىق
ئاتا قىلالايدىغان ئايال بولغانلىقىدا .
مۇنبەردىكى يىگىتلەر  سىلەرمۇ شۇنداق قارامسىلەر ؟
ياكى مۇنبەردە كوت --كوت ئاياللارنى ياقتۇرىدىغانلارمۇ  بارمۇ ؟
