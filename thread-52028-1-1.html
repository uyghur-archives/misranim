ئەسسالامۇ ئەلەيكۇم تورداشلار ياخشىمۇ سىلەر ، بەلكىم ماقالەمنىڭ تېمىسىنى كۆرۈپ ئەركەكمۇ يېرىم بولامدۇ دەپ ھەيران قالغانسىلەر ؟ ...........
ئەركەك ئەرلىك تۇرقى بىلەن كۆرۈنىشىدە يېرىم بولمايدۇ ، بەلكى يېرىملىق ئۇنىڭ ۋۇجۇدىدا ، ئەرلىك غۇرۇرىدا ، ئىمانىدا بولىدۇ ، ئۇنداقتا << يېرىم ئەركەك >> دېگەن زادى قانداق ئەركەك ؟
بىز جەمىيەتكە قاراپ باقايلى ، دۇنيا ھاياتلىقى ئۈچۈن تىنىمسىز ئىزدىنىۋاتقان ئۆز نەپسىمىز ئۈچۈن ئۆزگىنىڭ نەپسىنى دەپسەندە قىلىۋاتقان ، ئادىمىيلىك ، ئىنسانپەرۋەرلىك روھىدىن يىراقلىشىشقا باشلىغان مۇنۇ
زېھنىمىز ، ۋىژدان - غۇرۇرىمىزغا ئادىمىيلىك نۇقتىسىدىن مەسلىلەر ئۈستىدە تەپەككۈر قىلىپ بېقىشقا بىر ئاز رۇخسەت بېرەيلى ! .......... كۆردۇقكى بىز بىز بولماي قېلىۋېتىپتىمىز ، ئائىلىنىڭ ، جەمىيەتنىڭ تۈۋرۈكى بولىمىش ئەركەكلىرىمىز ئاجىزلاشقا ( مەنىۋىي جەھەتتىن ) باشلاپتۇ . مېنىڭچە ئەمدى قۇرۇق پەلسەپە سۆزلىمەي ئەمەلىي مىساللارنى سۆزلىگىنىم ياخشىدەك قىلىدۇ ،
كوچىغا چىقىپ قارايدىغان بولساق ئاياللىرىمىز كىيىم كەيگەن يالىڭاچ ، كۆپۈنچە قىز -ئاياللىرىمىز ئاياللىقىمدىكى بار دەسمايەم مۇشۇ شۇ ، كۆرۈپ بېقىڭلار ، باھالاپ بېقىڭلار دەپ ئۆزىنى خۇددى كۆرگەزمە بۇيۇمىدەك باشقىلارغا نامايەن قىلىدۇ ، تاس قالىدۇ مال - باھا ئىدارىسىگە باھا بېكىتكۈزگىلى بارغىلى . ئۇلارنىڭ كۆپىنچىسىدە شەرمىي - ھايا دېگەندىن ئەسەرمۇ يوق ، يېڭىچە زامانىۋى جەمىيەتتە ئۇلار شۇنداق ئېچىلدىكى :
ئالدى بىلەن ئۇلارنىڭ مەسلىلەرنى تەپەككۈر قىلىشتىكى مەركىزىي ئاپاراتى بولىمىش مېڭىسى جايلاشقان كاللىسى ئېچىلدى ، ئېچىلدىيۇ بېشىدىكى چاچلىرى پاخپىيىپ كۆرۈنۈشكە باشلىدى ، ئۇلار زىيالىي ئىسلاھاتنىڭ باشلامچىسى دەۋىرنىڭ ئاۋانگارتلىرى دېگەنگە ئوخشاش چىرايلىق ناملار ئاستىدا ، ئۇلارنىڭ قىبلىنامىسى ، يولباشچىسى بولغان دارىۋېننىڭ تەدرىجىي تەرەقىيات پەلسەپىسىنى مەركىىي نۇقتا قىلىپ ئېچىلىش دائىرىسىمۇ كېڭىيىپ باردى . مەندەك بىر قۇرامىغا يېتىپ قالغان توي قىلمىغان ئەركەككە نىسبەتەن ئۇلارغا ئىمان - ھايا ، غۇرۇر - ۋىژدان دېگەنلەرنى قايرىپ قويۇپ تۇرۇپ باقسا بۇمۇ ياخشى ئىكەن ،
چۈنكى ئۇلاردىكى بار نەرسىلەرنىڭ ھەممىسى مانا مەن دەپ چىقىپ تۇرغان مەخپىيەتلىكىنى كۆرىۋالغىلى بولىدىكەن . ئەركەكلەرنى قىزلىق لاتاپىتى بىلەن ئۆزىگە جەلىپ قىلىپ ئاشىق - بىقارا قىلىش ئورنىغا ، يىگىتلەرنى ئۆزىدىن پەقەت ۋە پەقەت ۋاقىتلىق پايدىلىنىشقا قىززىقتۇرىدىغان ئاشۇ مايمۇن سۈپەت گۈزەل خېنىملارنىڭ ئوچۇق بەدەنلىرىنى تولا كۆرۈپ كەتكەچكە ئادەمدە ئۇلارغا قىززىقىش بولمايدىكەن ، قوزغىلىدىغىنى پەقەت ھايۋانلاردىمۇ بولىدىغان شەھۋەتلا بولىدىكەن خالاس . ( اللە ساقلىسۇن ئۇنداق نائەھلى ھاياسىزلاردىن بولۇپ قېلىشتىن ) . ھەيرانمەن بۇ ئۈرۈمچى دېگەن ئاجايىپ بىر يەر ئىكەن ، سوغوق ئەرلەردىن ، ئىسسىق ئاياللاردىن باشلىنىدىكەن . ئەرلەر چاپان بىلەن يۈرىيدىكەنۇ ئاياللار بەدىنىنى يېپىش ئۈچۈن ئەمەس بەلكى سەتلىكىنى يۇشۇرۇش ئۈچۈن بەدىنىگە چۈمبەل قىلىپ تارىۋالغان تار ، كالتە ، يىرتىق ئىشىك - دەرىزىلىك كىيىملەر بىلەن يۈرىۋېرىدىكەن . بۇنىڭدىنمۇ ئويلاپ باقسام بىز ئەركەكلەر ئاياللاردىن ئاجىزدەك ھېس قىلىپ قالدىم ، سوغوقتىن قورقىدىكەنمىز ئەمەسمۇ ؟ يا بۇ بىزگە كەلگەن يوقىلىش ، خارابلىشىشتىن ئالدىن سىگنالمۇ ؟ ئۆز ۋاقتىدا ئوسمان تۈرك ئىمپىريىسىنى مەغلۇپ قىلىشتا غەرب مۇستەملىكچىلىرى ئاتا تۈرۈك مۇستافا كامالدەك بىر زىيالىينى تەربىيلەش بىلەن بىرگە << ئەگەر سەن بىر قەۋىم ئۈستىدىن غالىب كېلىمەن دەيدىكەنسەن ، ئالدى بىلەن ئىشنى ئاياللىرىدىن باشلا ، ئايالى بۇزۇلغان قەۋىم تەبىي زاۋاللىققا يۈز تۇتىدۇ >> دەپ ئۆگەتكەن ئىكەن ، جەمىيەتنىڭ ئانىسى بولغان ئاياللىرى بۇزۇلغان قەۋىم قانداقمۇ ئالغا ئىلگىرلىيەلىسۇن ، بۇنداق مەدەنىيەت ( كۈلتۈر ) دە باشقىلارغا بېقىنىپ ، مەدەنىيەتنى باشقىلاردىن ئېلىپ ئۆز خاسلىقىنى ساقلىمىغان قەۋىمنى كىممۇ كۆزگە ئىلسۇن ، قېنى بىزنىڭ ئۆز ئىپپەت - نۇمۇسى ئۈچۈن جاندىن كېچەلەيدىغان سۆيۈملۈك ھەمشىرىلىرىمىز ؟ قېنى بىزنىڭ مەرىپەت ، مەدەنىيەت ئۈچۈن ئۆزىنى بېغىشلىيالايدىغان ، قەۋىمنىڭ كەلگۈسى ئۈچۈن ئىمانلىق ، دىيانەتلىك ، ۋىژدانلىق ، غۇرۇرلۇق ئەۋلادلارنى تەربىيلەپ قاتارغا قوشالايدىغان ئانىلىرىمىز ؟ ئاياللىرىمىز بۇزۇلدى ، قىزلىرىمىز بۇزۇلدى ، دېمەك ئەۋلادلىرىمىزنىڭ ئەڭ دەسلەپكى مەدەنىيەت تەربىيچىسى بۇزۇلدى ، نەتىجە شۇ بولدىكى ئاشۇ بۇزۇق ئانىلارنىڭ قولىدا بىر توپ نان قېپى ، داشقاللار مەيدانغا كەلدى .
ئەمدى بۇنىڭغا كىم مەسئۇل ؟.......... ، مەنمۇ ؟......... ، سىزمۇ ؟..... ۋەياكى ؟؟؟ ..............
بۇنىڭغا دەل سىزگە ۋە ماڭا ئوخشاش ئەركەكلەر مەسئۇل ! چۈنكى پەيغەمبىرىمىز ساللاللھۇ ئەلەيھى ۋەسەللەم << سىلەر يەنى ( ئەرلەر ) ھەممىڭلار پادىچى ھېسابلىنىسىلەر ، سىلەر ئۆز پاداڭلاردىن ھېساب بېرىسىلەر>> دېگەن . دېمەك بىز بۇ ئېغىر مەسئۇلىيەتتىن قېچىپ قۇتۇلالمايمىز .
مەسلەن : ئاددىي بىر مىسال ئالايلى ، سىز بىر ئەر ئايالىڭىزغا ئىگە بولالىدىڭىزمۇ ؟ قىزلىرىڭىزنى باشقۇرالىدىڭىزمۇ ؟ سىز بىر ئوغۇل ئاچا - سىڭىللىرىڭىزنى يامان يولدىن توسالىدىڭىزمۇ ؟
سىز بىر پۈتۈن ئەركەك بولۇپ ئايالىڭىزنى ياخشى يوللارغا باشلىيالىدىڭىزمۇ ؟ سىز بىر زىيالىي ، خىزمەتتىن كېلىپ ئايالىڭىز گىرىم جازىسىنىڭ ئالدىغا بارىدۇ ، سىز ئاشخانىغا كىرىپ تاماق ئېتىسىز ( تاماق ئېتىشىڭىز ئەيىپ ئەمەس ) بىراق سىز ئۆيدە ئۆزىڭىز ئەتكەن تاماقلارنى بالىلىرىڭىز بىلەن يەيسىز ، ئايالىڭىز بولسا چىرايلىق ياسىنىپ مېزىلىك ئەتتىرلەرنى سۈرتۈپ باشقا ئەرلەرگە ھەمراھ بولۇش ئۈچۈن ( ھەرخىل ئولتۇرۇشلار مەقسەد ) ماڭىدۇ ، سىز سويمەك ئەركەكلىك ئىمتىيازىڭىزدىن پايدىلىنىپ ئايالىڭىزنى توسالىدىڭزمۇ ؟ ۋە ياكى ئۆيدە سىزنىڭ ئالدىڭىزدا تازىلىقسىز مەينەت ھالەتتە تۇرىدۇ يۇ ، سىرتلارغا ، كوچىلارغا ياسىنىپ گۈزەل قەددى - قامىتى ، ئاياللارغا خاس جەزبىدارلىقى بىلەن چىقىدۇ ، ئايالىڭىزنىڭ گىرىم بىلەن ئارلاشقان ئاشۇ گۈزەل رۇخسارىدىن بولسا باشقىلار ھۇزۇلىنىدۇ ، سىز پەقەت ئايالىڭىزنىڭ ئالدىڭىزدىكى ئاددىي تۇرقىدىنلا بەھىرلىنىسىز . بىرەر قېتىم بولسىمۇ مېنىڭ ھەقدارىم سەن بولغىنىڭدەك ، سېنىڭ ھەقدارىڭ مەن تۇرسام يەنە كىم ئۈچۈن ياسىناتتىڭ ؟ گۈزەللىكىڭ مەن ئۈچۈن بولسا كۇپايە ، دېگەندەك گەپلەرنىمۇ دېيەلمەيسىز ھەقاچان ، بەلكىم سىزمۇ ئايالىڭز كوچىلاردا راۋۇرۇس ياسىنىپ ماڭسا ، باشقا ئەرلەر ئايالىڭىزغا ئاچ قالغان بۆرىدەك شەھۋەتلىك كۆزى بىلەن تىكىلسە ئىچىڭىز ئاچچىق بولماقتا يوق ، ئايالىڭىزنىڭ گۈزەل ھۆسىن - جامالىدىن پەخىرلىنىپ كېتەرسىز . ئايالىڭىزغا مەنىۋى جەھەتتىن ئەمەس بەلكى ماددىي جەھەتلەردىنلا ئىگە بولارسىز ؟ مانا مۇشۇ سىزدىكى مەسئۇلىيەت .
ھە راست ئۇنتۇپتىمەن ، سىز دېگەن زىيالىي زىيالىيلار ئۇنداق قاتمال ، مۇتەئەسسىپ بولسا بولمايدۇ . ئۇ دېگەن بىلىمسىز ، نادان سەھرالىق تومپايلارنىڭ ئىشى .
لېكىن ئويلىمايسىزكى : سىزگە ئوخشاش بىر مىللەتنىڭ ئاۋانگارت زىيالىيسى ئېچىپ بەرگەن يوللاردا مەدەنىيەت جەھەتتە سىزدىن تۆۋەن تۇرىدىغانلارنىڭ سىز گە ئەگىشىدىغانلىقىنى ،
سىز بەلكىم غەرب مەدەنىيىتىنىڭ نىگىزى بولغان دارۋېن پەلسەپىسىنى جىق ئۆگەنگەچكە ، ئادەم مايمۇندىن تۆرەلگەن دېگەن پەلسەپىگە ئىشىنەرسىز ، شۇڭا ئەسلىڭىزنى ئۇنۇتقىڭىز كەلمىگەندۇ ،
سىز بىر ئىشچى ، سىزنىڭمۇ خەقتىن قالغىڭىز كەلمىگەندۇ ، شۇڭا تاپقان - تەرگىنىڭىزنى ئايالىڭىزنى بېزەشكە سەرىپ قىلغانسىز .
سىز بىر ئوقۇتقۇچى ، بىلىپ قويۇڭ مەسئۇلىيىتىڭىز ھەممىدىن ئېغىركى باشقىلاردىن يېنىك ئەمەس ، لېكىن سىزمۇ خەقتىن كەم ئەمەستە ؟ سىزمۇ قالمايسىز ؟
مەيلى سىز ئەر ئوقۇتقۇچى بولۇڭ مەيلى ئايال ئوقۇتقۇچى بولۇڭ سىزنىڭ ئاشۇ سەبىي ئوقۇغۇچىلىرىڭىزنىڭ قەلبى گويا ئۇرۇق سېلىنىشىنى كۈتۈپ تۇرغان مۇنبەت زېمىن ، ئۇلارنىڭ قانداق مېۋە بېرىشى سىزنىڭ قانداق ئۇرۇق ئىشلىتىشىڭىز ، قانداق پەرۋىش قىلىشىڭىز بىلەن زىچ مۇناسىۋەتلىك ( بۇ مەزمۇننىڭ تېما بىلەن ئالاقىسى بولمىسىمۇ سۆزلەپ ئۆتۈش مۇۋاپىق دەپ قارايمەن ) .
سىز بىر سودىگەر سودا قىلىپ ئىككى - تۆت تال پۇل تاپتىڭىز ، سىزنىڭ نامىڭىز (老板) شۇڭا سىزمۇ خەقتىن قالسىڭىز بولمايدۇ ، ئىناۋەت قوغلاشمىسىڭز بولمايدۇ .
سىز بىر دېھقان ، سەھنىلەردە ، سەھىپىلەردە سىزنىڭ نامىڭىز ماختاپ كۆككە كۆتۈرۈكلۈك ، شۇڭا سىزمۇ يانمايسىز ، ئويۇن - تاماشاغا قانمايسىز . سىزنىڭمۇ شەھەرلىكلەردىن ھېچ نەرىڭىز كەم ئەمەس ، زادى دورىمىسىڭىز ، ئەگەشمىسىڭىز بولمايدۇ .
دېمەك ھەممەيلەننىڭ خەقنىڭ ئالدىدا مۇتەئەسسىپ ، قاتمال ، سەھرالىق ، خۇراپات بولغۇسى يوق شۇڭا ئاياللىرىنى ئاچا - سىڭىللىرىنى باشقۇرمايدۇ ، ئەر بولغانلارنىڭ ئايالى سىرتقا چىقىپ قايتىپ كەلسە ئېغىز بۇرۇنى جايىدا بولسىلا بولدى . ئاكا - ئىنى بولغانلار ئاچىلىرىمىز ياكى سىڭىللىرىمىز بۇلۇڭ - پۇچقاقلاردا باشقا ئەركەكلەر بىلەن ئىش - پەش تارتىشىپ يۈرسە بېرىپ تەستەك بىلەن سېلىش ئورنىغا خەقلەرنىڭ ئالدىدا ئۇياتقا قالماسلىق ئۈچۈن ئامال بولسا ئايلىنىپ ، ئامال بولمىسا يەرگە قاراپ كۆرمىگەن بولۇپ ئۆتۈپ كېتىمىز . ئۇلاردىن ھېساب ئېلىپ باقمايمىز ، جەمىيەتنىڭ داشقىلى ، قاتمال ، مۇتەئەسسىپ ، خۇراپات كىشىلەردىن بولۇپ قېلىشتىن قورقۇمىز .

ئەھۋال شۇنىڭدەك بولدى . ئۆزىنىڭ ئەرلىكىنى تۇنىيالمىغان ئەرلەر ئۆزىنىڭ ۋە ئائىلىدىكىلىرىنىڭ ئەرلىك ، ئاياللىق ، بالىلىق بۇرچىللىرىنى تۈگۈتۈش بىلەن بىرلىكتە ھەم شۇ ئىبلىسنىڭ قۇشۇنى بولۇپ باشقىلارغا ۋىرۇس تارقىتىپ جەمىيەتنى بار كۈچى بىلەن بۇزۇشقا ، كىشىلەرنى ئۆز سەنىمىگە دەسسىتىشكە تۇتۇشۇپ كەتتى . ئۇلار پەقەت ئۆز كىملىكىنى يوقاتقانلارنىلا ئازدۇردى ۋە شۇنىڭدەكلەرگە كۈچى يىتەتتى !
ئاللاھدىن تىلىدىم ؛ ماڭا ئەڭ بۈيۈك ئىلمىڭنى بەرگىن . مېنى ئەڭ يۈكسەك پەللىگە كۆتۈرگىن . مىنى بۇ پىتنە - پاسات ئىچىدە قالدۇرمىغىن ! مېنى ئازغۇنلار قاتارىدىن قىلمىغىن . سەن ماڭا توغرا ئىلمىڭنى ، توغرا يولۇڭنى بەرگىن مەن ئاندىن سېنىڭ رازىلىقىڭ بىلەن سېنىڭ رازى بولغان بەندىللىرىڭگە توغرىلىقنى سۆزلەي !
ئەمما ، ئاللاھنىڭ ئىرادىسى شۇ بولدىكى ، ھازىرچە ئۆيلەنمەسلىك ! ھىچ بىر ئايال زاتىغا كۆڭۈل بەرمەسلىك ! ئاشىقلىقنى پەقەت ۋە پەقەت بىر ئۆزىگىلا بىرىش !
جەمىيىتىمىزگە قارايدىغان بولساق كۆپىنچە ئەرلەر ئاياللىرىنى ۋە باللىرىنى تىگىشلىك رەۋىشتە توغرا يولدا ئېلىپ ماڭالمىغان . نىمە ئۈچۈن دەمسىز ؟ ئۇلار شۇ توغرىلىقنى تۇنىمىسا ، توغرا - خاتالىقلارنى ئايرىيالمىسا قانداقمۇ ئۆزىنى ۋە باشقىلارنى توغرىلىقتا ئېلىپ ماڭالىسۇن ؟
جەمىيتىمىزگە قارايدىغان بولساق كۆپىنچە ئاياللار ئەرلىرىنىڭ ۋە بالىلىرىنىڭ ئەرلىك غورورىنى ئۆزلىرى توغرا دىگەن ئۇقۇمىي بىلەن كۈچەپ دەپسەندە قىلماقتا ! نىمە ئۈچۈن دەمسىز ؟ ئۇلار غورورنىڭ ۋە غورورنى قانداق تۇتۇپ تۇرۇش ، بارغانچە داۋان ئاشۇرۇشنىڭ نىمە ئىكەنلىكىنى بىلمىگەن ، تۇنىمىغان تۇرسا ، پەقەت ۋە پەقەت ئۆزىنىڭ نەپسىنىلا توغرا دەپ تۇنىغان تۇرىۋالسا ئۇلار ھەقىقىي رەۋىشتە بولغان ئىنسانىيلىككە قانداقمۇ ياردەمچى يولداش بولالىسۇن ؟
مۇشۇ ۋەج بىلەن ئازراق ئاچچىقىمنى چىقىرىۋالاي ھە ! سۆزلەپ .
مۇشۇ دىيننى ئۇيۇن قىلىۋالغان ناكەسلەر بىلەن بىرەر سۇرۇندا ، يىغىنچىلىقلاردا بىللە ئولتۇرۇپ قېلىشتىن ئۆزەمنى تارتىدىغان بولۇپ قالدىم . نىمە ئۈچۈن دەمسىز ؟
‹‹ -قارىم ! سىلى كىزىك كېسىلى بولۇپ بىر ساقايغانما ؟ ئەمدى دەيمىنا ، سىلى بەكلا ئۇرۇق  .  - مۇشۇ قارىم تىگى - تىگىدىن ئەجەبمۇ ئاجىز . چوكانچاققا قانداقمۇ كۈچى يىتىدىغاندۇ ؟ ›› - ئۇلارنىڭ ماڭا بىر قارىشىدا دەرۋەقە ئايالىم ۋە 2 - 3 بالام بارلىق ھالىتىم بايان بولىدۇ . بۇنداق گەپنى ئاڭلىغاندا مەن ئازراق بولسىمۇ ئۆزەمنى ئاخلىمىسام بولمايدۇ - دە !
‹‹ مەن تېخى ئۆيلەنمىدىم ! ››
بۇ خىل كىشىلەر مەيلى قايسى جەھەتتىن بولسۇن بىرەر ئاجىزلىق تەرەپنى ئاڭلاپلا قالسا بولدى  كوچىلاپ ، سۆكىدىغان يەر بولسا سۆكۈپ ، دەرس بىرىدىغان يىرى بولسا دەرىس بىرىپ مەيلى قايسى تەرەپتىن بولسا مەيلىكى ، بىز ‹ قارىي › لارنىڭ ئەدىپىنى بىر بىرىۋالمىسا ، شۇ ئارقىلىق كۆڭلىنى بىر ئېچىۋالمىسا بولدى قىلمايدۇ .
شۇ خىل ئاجىزلىقلىرىمغا قارىتا تېخى ئۆيلەنمىگەنلىكىمنى بايان قىلمىسام شۇ كۈيۈمچان ‹ ئەركەك › لەر ماڭا ، بولۇپمۇ مېنىڭ ئايالىمغا بەكلا ئىچى ئاغرىغىنى تۇرغانلا گەپ . ئۆيلەنمىدىم دىسەم ئايەت - ھەدىسنىڭ ئۆز نەپسىگە توغرا كېلىدىغان تەرىپىنى سۆزلەپ تولا گەپ قىلىپ يادىتىۋەتكەنلىكىدىن ، شۇنىڭدەك بىللە ئولتۇرۇشتىن يادايدىغان ۋە بەزى يىغىنچىلىقلاردىن ئۆزەمنى تارتىدىغان بولۇپ قالدىم . ئىككى مەسجىدتىكى پىتنىخۇر - تاماشىبىنلاردىن تېنىپ ياخشى بولغان شۇ مەسجىدكە بارساممۇ ، گەرچە مەسجىد ئىچىدە پىتنە - پاراڭ بولمىسىمۇ شۇنىڭدەك سۇرۇنلاردا ئەتەي چېقىۋالاتتى .  چۈنكى جەننەتنى توغرا تۇنۇپ ، توغرا رەۋىشتە يول تۇتۇپ ماڭغانلىق ؛ شۇنىڭدەك ماددى - مەنىۋى جەھەتلەردىكى ئوق - بومبىلارغا ئۆزىنى تۇتۇپ بىرىش - دىگەنلىكمۇ بولىدۇ .
شۇنىڭدەك دىشۋارچىلىقلارغا يولۇققاندا ئىچ - ئىچىمدىن ‹‹ ئى رەببىم ! ماڭا ئايال بولۇپ سېنىڭ يولۇڭنى داۋان ئېشىشىمغا يولداش  بولالايدىغان بىرەر ئايال زاتى يوقمىدۇ ؟ ›› دەپ ئۇنىڭغا نىدا قىلىپ كېتىمەن ! مەبۇدىم ، ئىلاھىم ئاللاھ كۆڭلۈمگە سالىدۇكى ‹‹ يوق . ھازىر گەرچە بار بولسىمۇ يەنىلا يوق . سەن بۇ خىل ئىنتىلىشتىن قايت ! ››
تۇنۇپ يىتىمەنكى ، مەن ھازىر بەزى بىر ئەركەكلىكىمنى ئايالىم ۋە مەن شۇنىڭدەك نوقۇل بولغان پادىلىرىمغا تۇلۇق رەۋىشتە ئىشلىتەلمەيدىكەنمەن ۋە ئايالىممۇ ماڭا بۈگۈنكى ئىسلامىي تۇنىشى ئىچىدە  تىگىشلىك رەۋىشتە ئاياللىق بۇرچىنى ئادا قىلالمايدىكەن . بولۇپمۇ ماڭا ئايال بولغۇچى مېنىڭ داۋاملىق دىيندا داۋان ئېشىشىمغا يولداش بولالماستىن بېشىمنى ئاغرىتىپ ، دىققىتىمنى چېچىپ ھەر 3 ئايدا بىر قېتىم بىر پەللە ئۆرلىشىمگە توسالغۇ بولىدىكەن . ( ماڭا بۇ يەردىمۇ  يەنە دەرس بىرىپ مۇنداق ئاياللارمۇ بارغۇ ! دەپ جاۋاب بەرمىگەيسىز ! )
ئاللاھ ھەر بىر زاماندا شۇ زامان ئىچىدىكى كىشىلەرنىڭ ئۆز ھاللىرىغا قارىتا توغرا شەرىئەتنى ، توغرا ئەمەل - ئىبادەت يوللىرىنى كۆرسىتىپ بەردى ۋە ئالدىدا ئاشكارا ئايان قىلدى . ئەمما كىملەركى ئىبادەتتە ئاللاھدىن باشقا شەيئىلەرگە بۇرۇلۇپ كىتىدىكەن ، ( ئىبلىسلار )نىڭ كۆرسەتمىللىرىگە كىرىپ كىتىشى - بۇ ھەقىقەتتۇر ! ئاللاھ ھىچ كىشىنى كۈچى يەتمەيدىغان ئەمەللەرگە تەكلىپ قىلغىنى يوق . شۇنىڭ ئۈچۈن ئىنسان ئاۋال ئۆزىنى ئاندىن ئۆزىنى نىمىلەرنى قىلالايدىغانلىقىنى تەكشۈرۈپ كۆرىشى ئاندىن ئەمەللىرىگە قەدەم تاشلىشى كېرەك بولىدۇ . بولمىسا ؛ ئاللاھ چاھارپايلارنى بىكارلا يارىتىپ قويغىنى يوق . بەلكى ئۇلاردىن ئىبرەت ئالسۇن ئۈچۈن ، بىزلەرنى ھايۋانلارنى باشقۇرۇپ تۇرىدىغان ۋە شۇنىڭدەك ھالىمىزنى بىر ھالدىن يەنە بىر ھالغا ئۆزگەرتىشكە شۇ خىل ئىمتىيازنى بىرىش بىلەن تېخىمۇ ئەقىل تېپىشىمىزغا ، ئىنسانلىق خۇسۇسىيىتىمىزنى ، تەبىئىتىمىزنى چۈشىنىشىمىزگە قىستايدۇ .
نۇرغۇنلىغان كىشىلەر شەھەردە ياشىغاندىن كۆرە يېزا - قىشلاقلارغا كەتسە ، شۇ يەردە دېھقانچىلىق ، باغۋەنچىلىك قىلسا ئەڭ توغرا ياشاش يولىنى تاپالايتتى . كۆپىنە كىشىلەر تاغقا چىقىپ قوي باقسا دۇنيا ۋە ئاخىرەتتە رەسۋا بولىدىغان ئاقىۋەتلەردىن ساقلىنالايتتى . ئەمما ئۇلار ئەنە شۇ نادانلىقى ، گۇمراھلىقى سەۋەبىدىن بۇ خىل تاللاشنى توغرا تاپمايدۇ بەلكى ، تېخىمۇ ئەزۋەيلەپ مېڭىۋىرىدۇ .
ئەگەر شۇلار ھەقنىڭ نىمە ئىكەنلىكىنى تۇنىسا ئىدى ، قانخور يىرتقۇچلارنىڭ ، دورامچى مايمۇنلارنىڭ ، قىلمىشىغا ئىگە بولمايدىغان  ھاڭگا ئېشەكلەرنىڭ ، ئەر تاللىماي تۇتۇپ بىرىۋىرىدىغان بېلىقلارنىڭ ............... سۈپىتىدە بولماستىن ئۆزىنىڭ دۇنيادىكى تىگىشلىك ئورنىنى تاپقان بولاتتى .
ھەر بىر شەيئىلەرنىڭ كۈچىيىشىدە شۇ ساھەلەردىكى ئۈلگىلىك ئالدىن قۇربانلىق بەرگۈچى كۆرەشچان ، دائىم ئىنتىلىشچان ، سائادەتمەن قەلىب ئىگىللىرىنىڭ ئالدىن قەدەم تاشلىشى ۋە ئۈنۈملۈك يول ھازىرلىشى ئاساسىدا ئۈزلۈكسىز داۋان ئاشىدۇ . مۇزىكانتلارنىڭ يول باشلىغۇچىللىرى كومپوزىتۇرلاردۇر ! ناخشىچىلارنىڭ شائىرلاردۇر ! گەپدانلارنىڭ ئىتۇتچى ، يۇمۇرچىلاردۇر ! ئەدىبلەرنىڭ يازغۇچىلاردۇر ، مۇتەپەككۇرلاردۇر ، تىلشۇناسلاردۇر ! مۆئىنلەرنىڭ خەلىپىلەردۇر ! ..................... شۇنىڭدەك ھەر خىل ساھەلەرنىڭ ئۆزىگە خاس بولغان ياخشى - يامان باشلامچىللىرى ، يول سالغۇچىللىرى بولىدۇ . كىملەركى قايسى بىر يىتەكچىلەرگە ئەگىشىدىكەن ، ئۇ شۇنىڭ ۋە شۇلارنىڭ جۈملىسىدىندۇر !
ئاللاھ ھەممىنى ھېكمەت بىلەن ياراتتى . ئەمدىكى شۇلارنىڭ قايسىنى تاللاپ يارىلىش ، ئۆزىڭىزنى ئۆزگىلەرگە قېلىپلاشتۇرۇپ قايسىلارنىڭ جۈملىسىدىن بولۇشنى تاللاپ مېڭىشىڭىز ئۆزىڭىزدىندۇر . ئاللاھ مەبۇددۇركى ، سىز قايسىنى تاللاپ ماڭسڭىز شۇنى خاس بىر - بىرلىڭىزنى بىر بىرلىرىڭىزگە ئۇچىراشتۇرۇپ مۇلاقات قىلىپ بىرىشتۇر !
ئەھال شۇنىڭدەك بولدى . سىز كىم ؟ سىز نىمىلەرنى قىلالايسىز ؟ ئەمدى نىمىلەرنى قىلماقچى ؟ سىز ئۆزىڭىزنى تۇنۇمسىز ؟ نىمە ئىش قىلىۋاتقىنىڭىزنى بىلەلەمسىز ؟ ھەق قايسى ناھەقچۇ ؟ ئايرىيالامسىز !
مانا بۇلار ھەر بىر ‹‹ ئەركەك ›› دەپ تۇنىۋاتقان ۋە شۇ خىل ‹‹ ئەركەك ›› لىكىنى نامايەندە قىلماقچى بولغان كىشىلەرنىڭ ئويلىنىشقا تىگىشلىك بولغان ئالدىنقى قەدىمىدۇر ! بۇنىڭسىز كىملەركى داۋان ئاشىمەن دەيدىكەن ، ھەقىقەتەن كېلەچەكتە قاتتىق پۇشايمانغا قالىدۇ !
ئاخىرەتكە ئىشىنىدىغان كىشىلەر ۋەز - نەسىھەت ئالىدۇ . ئىشەنمەيدىغانلار بولسا ئۆزىنىڭ ئۇيىنىنى ئوينايدۇ - گەپ تامام !

بىر قالماق ئېيتتى : - پالانى ياخشى كۈن كۆرمەيدۇ ! چۈنكى ئۇ ئاتا - ئانىسىنى بەكلا قاخشاتقان .
بۇنى ئاڭلىغان بىر مۇسۇلمان ئېيتتىكى : سەن خۇدامىتىڭ ؟ ئۇنىڭ قانداق كۈن كۆرىدىغانلىقىنى ئالدىن بىكىتىدىغان ياكى ئالدىن بىلەلەيدىغان !
قالماق ئېيتتىكى : سەن ئۇنداق دىمە ! بۇ بۇرۇندىن تارتىپ ئۆزگەرمەي كېلىۋاتقان بىر قانۇنىيەت . يامان ئىشنى قىلغۇچى تىگىشلىك جازاسىنى تارتىدۇ !
- ماڭغىنەۋە ، ھەممىنى خۇدايىم بىلىدۇ - دىيىش بىلەن مۇسۇلمان ئۆز يولىغا مېڭىۋەردى !
كىمكى يېلەڭ ھالدا قەھرىتاندا سىرتقا چىقىدىكەن ۋە ئۇزۇن تۇرۇپ كىتىدىكەن ، ئاخىرەت قانداق بولىدۇ ؟
كىملەركى ئاچ يۈرىۋىرىدىكەن ، ئۇنىڭ ئاقىۋىتى قانداق بولىدۇ ؟

قىيامەت دىگەن پەقەت مەھشەر كۈنىلا ئەمەس بەلكى مۇشۇ دۇنيادىمۇ ئاخىرقى ھېسابنى قىلىدىغان كۈن ھەم قىيامەتتۇر !
