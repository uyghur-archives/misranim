مۇنبەر نەشىرى يۇقىرى كۆتۈرۈلگەندىن كېيىن مىسرانىم ئەپەندى ئەزالاردىن ۋاقتى-ۋاقتىدا تەكلىپ-پىكىر ئېلىپ تۇرۇش بىلەن بىرگە، ئەزالارنىڭ بەرگەن تەكلىپ-پىكىرلىرىگە ئەستايدىل، توغرا مۇئامىلە قىلىپ كېلۋاتىدۇ، بۇنىڭدىن شۇنداق سۆيۈندۈم، باشقىلارنىڭ (مەيلى ئۇ كىشى ئۆزىدىن يۇقىرى ياكى تۆۋەن تۇرسۇن ئۆزىدىن چوڭ ياكى كىچىك بولسۇن) سۆزىنى ئىلىك ئېلىشمۇ بىر ئالىيجاناپ خىسلەتكەن دەپ قالدىم، بىر خىزمەتدىشىم بولىدىغان، ھچقانداق ۋاقىتتا ئۆزىدىن كىچىكلەرنىڭ ياكى ئۆزى بىلەن تەڭ دېمەتلىكلەرنىڭ بەرگەن تەكلىۋىنى قوبۇل قىلمايدۇ، ھەتتا پىسەنتمۇ قىلمايدۇ، ئۇ باشلىقنىڭ گېپىدىن باشقا ئادەمنىڭ گېپىنى ئاڭلىمايدۇ، خاتالىقىنى ئېنىق كۆرسىتىپ بەرسەكمۇ ھېچقانداق ئىپادە بىلدۈرمەي يەنىلا ئۆزىنىڭ بىلگىنىنى قىلىۋېرىدۇ، شۇڭا بۇ يەدە مىسرانىم ئەپەندىنىڭ ھەر بىر ئەزانى شۇنداق چوڭ كۆرگىنىگە، تەڭ كۆزدە قارىغىنىغا رەھمەت. ھەر قايسى باشقۇرغۇچىلارنىڭمۇ مىسرانىم ئەپەندىدىن ئۆگىنىپ ھەممىگە بىردەك مۇئامىلە قىلىشىنى، ئەزالار تېما يوللىغاندا دەرىجىسىگە قارىماي، تېمىنىڭ مەزمۇنىغا بەكرەك ئەھمىيەر بېرىشىنى ئۈمىد قىلىمىز.

ئەر-خوتۇنلار بىر-بىرىگە مەرھەممەت ۋە مۇھەببەت نەزىرى بىلەن قاراشسا، بىر-بىرىگە مۇھەببەت ئىزھار قىلىشسا پۈتۈن گۇناھلىرى بارماقلىرى ئارىسىدىن تۆكۈلۈر.
