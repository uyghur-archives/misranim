ئەسسالامۇ ئەلەيكۇم تورداشلار ، بۇ تىمىنى يىزىش يازماسلىق توغرىسدا ئويلىنىپ ئۆزۈمنىڭ مۇسۇلمانلىق بۇرچىنى ئادا قىلىشىم ئۈچۈن يەنىلا يىزىش قارىغا كەلدىم ،خاتا سۆزلىگەن بولسام ئاۋۋال ئەپۇ سورىۋالاي  ....
      ئاللاھنىڭ يولى بارچە يوللاردىن ئۈستۈندۇر ، پۈتۈن ئالەملەرنىڭ ئۇستازى بولغان ئۇھەممەد مۇستاپا سەللەللاھۇ ئەلەيھى ۋەسسەللەم  ئادەملەرنى  ھەرقانداق دەۋىردە ھەم زۇلمەتتىن ئىلىپ چىقالايدىغان يوللارنى قويۇپ كەتكەن . مۇسۇلمانلار ھەر زاماندا مەنىۋىيەت ، ئەدەپ - ئەخلاق ۋە خۇلقتا دۇنيا كىشىلىرىگە ئىبرەت بولۇشى كىرەك .   بۈيۈك دىن ئىسلام دىنى تەرەققىياتنىڭ دۈشمىنى ئەمەس . ئەمما ئۇ تەرەققىيات قاچانكى يامانلىققا  ، زۇلۇمغا ، ئادالەتسىزلىككە يۈزلەنسە ئۇنىڭغا قارشى تۇرماي بولمايدۇ .  ھەربىر دەۋىر ئۆزىنىڭ يىڭىلىقى بىلەن ئادەملەرنى جەلىپ قىلغانغا ئوخشاش بىزنىڭ بۇ دەۋىرىمىزدە ھەم كۆپ يىڭىلىقلار ، تەرەققىياتلار بارلىققا كەلمەكتە . بولۇپمۇ ئىلىكتىرون ساھەسىدە بۇ خىل ئەھۋال تىخىمۇ روشەن ئىپادىلەنمەكتە . بۇنى بىز سۆزلىمىسەكمۇ كۆز ئالدىمىزدا كۆرۈنۈپ تۇراۋاتىدۇ .  ھەرقانداق يىڭىلىقنى ياخشىلىققا ھەم يامانلىققا ئىشلىتىش ئەلۋەتتە تامامەن مۈمكىن بولغىنىدەك يىقىنقى يىللاردىكى ئىشلار ئادەمنى سەل ئويلاندۇرۇپ قويماقتا .  
    رادىئو ، تىلۋىزور ، كومپىيوتىر ، تىلىفۇنلارنىڭ ئوموملىشى ، تىلىفۇنلارنىڭ ئىقتىدارلىرىنىڭ كۆپىيىشىگە ئەگىشىپ ھەرخىل ياخشى يامان ئۈن - سىن ئەسەرلىرى ، رەسىملەر تارقىلىپ يۇرمەكتە . ئەلۋەتتە ياخشى نەرسىنى كۆپەيتىش كۆچۈرۋالغۇچى ۋە كۆچۈرۈپ بەرگۈچىنىڭ ئەجىر ئىلىشىغا سەۋەپ بولىدۇ ، ئۇ نەرسىلەر جەمىئىيەتكە ، كىشىلەرگە ، ئائىلىگە پايدىلىق بولسا نۇر ئۈستىگە نۇر بولىدۇ ، لىكىن  ئۇنداق ئەمەسلىرىچۇ ؟؟؟؟  بۇنداق ئۇن - سىن ۋە رەسىملەر  كىشىلەرنىڭ ئىڭىنى ئۆزگەرتىۋىشىكە سەۋەپ بولۇپ قالماسمۇ ؟؟؟
       ئاللاھتائالا ئۆز كالامى قۇرئان كەرىمدە مۆئمىنلەرنى بىر بىرىگە  ھەقنى ، ياخشىلىقنى تەۋسىيە قىلىشقا ، توغرا يولغا باشلاشقا چاقىرىدۇ . يامانلىقنى ئالدىنى ئىلىشنى  توسۇشقا بۇيرىيدۇ .     باشقا دىندىكى ، باشقا ئىقلىمدىكى كىشىلەرنىڭ ھەرخىل ھالەتلەردە تۇرۇپ  ناخشا ئىيتىشلىرى ، ئۇياتسىز ھالدا چۈشكەن رەسىملىرى  كىشىلەر ئارىسىدا خىلىلا كۆپ تارقالماقتا ، بۇلارغا نىمە دىگۈلۈك ؟ بۇلارنى بىز باشقۇرالمايمىز دەپ ئولتۇرامدۇق ؟ ئۇلارغۇ ئۆزلىرىنىڭ ئاشۇ ھالەتلىرىدىن ئۇيالمىسۇن ، لىكىن تىلىفۇن ، كومپىيوتىر ، تىلۋىزورنىڭ تىزگىنىكى ئۆز قولىڭىزدا تۇرۇپ يەنە شۇ نەرسىلەرگە قاراپ ئولتۇرامسىز ؟ سىز قاراشتىن ئۇيالمامسىز ؟ ياكى خەخ ئاچسا نىمە كارىم ، مەن ئېچىۋالمىغاندىكىن دەپ كۆرۈپ ئولتۇرىۋىرەمسىز ؟ قاچانغىچە شۇ نەرسىلەرگە قاراپ ، ئاڭلاپ ئولتۇرماقچى ؟  
       مۇسۇلمان قىرىندىشىم ، ئۆزىنى مۆئمىن ، مۇسۇلمان ئاتىغان ئۇيغۇر بالىسى قىرىندىشىم ، بۇ نەرسىلەرنىڭ جەمىئىيەتكە بولغان ، بولىۋاتقان زىيىنىنى ئويلىشىڭىز ۋە تارقىلىشىنى توسۇشىڭىز كىرەك ~!!!!!! مانا بۇ  سىزنىڭ مەسئۇلىيتىڭىز ، ۋەزىپىڭىز ... قىيامەت كۈنى ياخشىلىققا بۇيرۇپ يامانلىقتىن توسۇش سورىقى سورالغاندا  ئاللاھ ھەق سۇبھانەۋەتائالا غا جاۋاپ بىرىش ئۈچۈن قىلىشقا تىگىشلىك ئامانىتىمىز  .....  
         خوش ، قانداق توسىمىز ؟  
     بىردىنبىر يول ، ئۆزىمىزدىن ، ئەتراپىمىزدىن باشلاش .  ئەتراپىمىزدىكى يىقىنلىرىمىزنىڭ ، دوست - بۇرادەرلىرىمىزنىڭ كومپىيوتىرلىرىدا ،  تىلىفۇنلىرىدا ساقلىنىۋاتقان بىھايا سۈرەتلەر ، سۈرەتلىك ناخشىلارغا كۆزىمىز چۈشكەندە ئۇنى سورىماستىن يۇيىۋىتىشتە مەنىۋىيەت جەھەتتىن ھوقوقىمىز بار ، گەرچە بۇنداق قىلىش سەل قاتتىق بولسىمۇ لىكىن توغرا يول ئۈچۈن ئەرزىيدۇ . ئەلۋەتتە باشقىلارغا تارقىلىشنىڭ ئالدىنى ئىلىشنى توسۇشتا چىرايلىق نەسىھەت ،  ياخشى سۆز بىلەن يۇيۇش ئەجرىمىزنى كۆپەيتىدۇ . يىقىنلىرىمىزنىڭ ھەم بىزنىڭ قىيامەت كۈنىدىكى سورىقىمىزنىڭ يەڭگىللىتىدۇ .   پەيغەمبەر سەللەللاھۇ ئەلەيھى ۋەسسەللەم مۇبارەك ھەدىسلىرىدا تا  قىيامەتكە داۋام ئىتىدىغان تىخنىكىغا جاۋاپ بىرەلەيدىغان كۆرسەتمە بىرىپ كەتكەن . يەنى  باشقىلارنىڭ ئەيىپىنى يىپىش ، باشقىلارنى ئىسلامغا قىزىقتۇرۇش ئارقىلىك توغرا يولغا باشلاش  توغرىسىدىكى نۇرغۇن ھەدىسلىرى بۇنىڭغا گۇۋاھلىق بىرىپ تۇرماقتا . مۇشۇ تەرەققىيات تىخنىكىسىدىكى  تارقىلىشمۇ «ياخشىلىققا بۇيرۇپ يامانلىقتىن توسۇش » پەزىپىسىگە تەۋە .  بۇنداق قىلىش ، يەنى قىرىندىشىمىزنىڭ ئەيىپىنى ئاچماستىن ئۇنى چىرايلىقچە توغرا باشلاش ، خاتا ئىشلاردىن توسۇش  ھەقىقىي مۆئمىن مۇسۇلماننىڭ ئەخلاقى . بىز باشقىلارنىڭ نەزىرىدىكى ياخشى مۇسۇلمان بولۇش ئۈچۈن ئەمەس ، بەلكى ئاللاھنىڭ دەپتىرىگە ھەقىقىي مۇسۇلمان دەپ يىزىلىدىغان مۇسۇلمان بولۇش ئۈچۈن ئەڭ ئاۋۋال ئۆزىمىزنى تۈزىشىمىز كىرەك . باشقىلارغا « ھە ، ئەسلى مۇسۇلمان دىگەن مۇشۇنداق بولىدىكەندە » دىگەن ياخشى تەسىرنى بىرىپ ئۇلارنى ئىسلامغا قىزىقتۇرىشىمىز  كىرەك . مانا بۇ ھەقىقىي مۇسۇلماننىڭ بۇرچى ، ئامانىتى .....
       ئەسلى گەپكە قايتساق ،  شۇ ھالەتتە ناخشا ئىيىتقۇچىلار ، شۇنداق ھالەتتە رەسىمگە چۈشكۈچىلەر نىمە مەخسەت بىلەن شۇنداق قىلغان بولمىسۇن بۇ بىز بىلەن مۇناسىۋەتسىز ، ئۇلارنىڭ بىھايا ھالەتلىرى ئۈچۈن بىز سوراققا تارتىلمايمىز ، لىكىن شۇلارنى ساقلىغىنىمىز  ، تۈرقاتقىنىمىز ئۈچۈن ئەلۋەتتە ھىساپ بىرىمىز . قىرىندىشىم ، مەنىۋىي جىنايەت ئىسلامدا قاتتىق سۈرۈشتۈرىلىدۇ  ،   كوكا كولا ئىچىملىكى ، NIKE  كىيىملىرى باشقا ئىسپىرىت تەركىبى بولغانلىقتىن ياكى كىيىمدە باشقا خىمىيلىك ماددا بولغانلىقتىن ئەمەس بەلكى شۇ شىركەتلەرنىڭ ئىقتىسادىي پايدىسى  ئىسلام  دىنىنى خۇنۇكلەشتۇرۇش يوللىرى ئۈچۈن سەرپ قىلىنىدىغان بولغانلىقى ئۈچۈنلا  چەكلەنگەن .  مۇشۇ ئىشلارمۇ شۇنىڭغا ئوخشاش ، ئاشۇ كىشىلەرمۇ ئىسلامنى خۇنۇكتۇرلەشتۇرىۋاتىدۇ ، ئۇلار ئۆزلىرىنىڭ بىھايا رەسىملىرىنى تارقىتىش ئارقىلىق مۇسۇلمانلارنى مەنىۋىي جەھەتتىن خۇنۈكلەشتۈرمەكچى ، مۇسۇلمانلارنىڭ ئەقىدىسىنى سۇسلاشتۇرماقچى .  بۇنى ئويلىنىپ بىقىڭ ...
     باشقىلارمۇ قىلىۋاتىدىغۇ ، مەن  قىلسام نىمە بوپتۇ دەپ قىلىۋاتقانلار تىخىمۇ قاتتىقراق جازاغا مۇناسىپتۇر ، چۈنكى باشقىلارنىڭ خاتاسىنى تۈزەتمەي ئۆزىنى خاتاغا باشلىدى ، بۇ يولدا سىز مىڭىپ قالمىغايسىز قىرىندىشىم .....
ھەركىمنىڭ يۈرىكىدە نىمە بولسا كۆزىگە شۇ نەرسە چىرايلىق كۆرۈنىدۇ ، ئىغىزىمۇ شۇ نەرسىنى سۆزلەيدۇ ، دىمەك قەلبىڭىزدە ئىماننىڭ ئۇچقۇنلىرى بولسا سىز ئۈچۈن ساقلاشقا ، تارقىتىشقا ياخشى نەرسىلەر ئاز ئەمەس ، شۇلار بىز ئۈچۈن يىتەرلىك جاۋاپ بولغۇسى ...
      ئاللاھ ھەممىمىزگە ياخشىلىق ، ئاخىرەتتىكى ئاسانلىقنى بەرسۇن . ئامىن .

ئىشلەپ چىقىرىپ كىمگە سۇنىۋاتىدۇ؟ ئۇلارنىڭ ئىش ھەركىتى قانداق؟ ئۇلار نىمىنىىمىگە قوشىۋاتىدۇ؟بۇ جەھەتتە گەپلىرىمگە قۇلاق سېلىڭ!شۇنداق ،ئۇندىن باشقا بىزدە يەنە ھالالنى ھارام قىلىدىغان، يەنى يەرلىكتىن چىقىدىغان ،ئۆزىمىز تېرىپ ئۆستۈرۈپ ھالال دەپ يەپ كۈنۈپ كەتكەن بۇغداي قۇناق، ھۆل قۇرۇق يەل يىمىشلەرنى ئالا قانداق پىشقلاپ ئىشلەش بھانىسى بىلەن،قۇرۇلغان ئاتالمىش مىللى كارخانىلار بارلىققا كىلىپ، ئەسلى  پاكىزە  بولغان ئەشۇ يەرلىك مەھسۇلاتلارغا، ئاللا قانداقتۇر «پودافىن»ۆپتۈرگۈچى خۇرۇچ، «تىئاۋلىاۋ»تەم تەمشىگۈچ؛«مىڭجاۋ»؛ئېجاۋ» ئىشەك تىرىسىدە ئىشلەنگەن بىر خىل يىمەكلىككە ئېلىشتۇرىدىغان ماددا؛قانداق مايلاردىن ئايرىۋالغانلىقىنى بىلگىلى بولمايدىن سالات مايلىرىنى قىممەتلىك ھەم ئاساسلىق ماددا ھىداپلاپ بۇ بولمىسا بولمايتتى دەپ قاراپ، ئۇنىڭ بىلەن ھەر خىل «پىچىنە -پىرەنىك»، «ئاي توقاچ»،« تومارىس»،« باقالى»،«گۈلقەن پىرەنىك»، «مېغىزلىق پىرەنىك»، «مىنىڭ پىرەنىكىم»،« شوخ- شوخ »قاتارلىق يىمەكللەرنى ئىشلەپ چىقىرىدىغان بىر قىسىم مىللى كارخانىلارنىڭ ئە شۇ قىممەتلىك مەھسۇلاتلىرى بىزگە ۋە مۇسۇلمان ئۇيغۇرلارغا قىلىنغان ئىسىل سوغا سۈپىتىدە ھەر خىل ناملار بىلەن ئاتالغان ئاتالمىش تاللا بازارلىرىدا تالىشىپ سېتىۋالىدىغان بويۇملار سۈپىتىدە سېتىلماقتا، بۇنى ئۆز كۆزى بىلەن كۆرمىگەن ئادەم ئۇنى ئېلىپ يەۋىرىدۇ. لىكىن بىلگەن ئادەم ئۇنى «بىر دۆۋە ئەخلەت يىمەكلىك»دەپ ئاتايدۇ.ئەشۇ مىللى كارخانىلارنىڭ مەسئۇللىرى تېخى ئاللانىڭ بەيتىگە بېرىپ ھەججى ئەرپات قىلىپ كەلگەن قارى سۈپەت ھاجىلار بولۇپ، ئۇلار پەقەت پۇلنى دەپلا باشقىلارنىڭ كىچىككىنە قالغان ئىمانىنى بۇلغاشنى دۇرۇس دەپ قاراۋاتىدۇ. ئۇلار ئۆزىنىڭ قىلغانلىرىنى بىلمەي قىلىۋاتامدۇ ياكى بىلىپ تۇرۇپ قىلىۋاتامدۇ؟ ئەلۋەتتە بىلىپ تۇرۇپ قىلىدۇ. ئۇلارنى پۇل دىگەن نەرسىنىڭ ئەشۇنداق قىلىپ قويغانلىقى كىشىنى تولىمۇ ئېچىدۇرىدۇ.شۇڭا مەن بىلگەچكە ئۇلارنىڭ قىلمىشىدىن ۋە ئۇلارنىڭ شۇ ھالىتىدىن تولىمۇ نەپرەتلىنىمەن، ئۇلارنى «بۇنداق قىلمايلى دىسەم ، ئۇلار ماڭا قاراپ تۇرۇپ « بۇنداق قىلمىساق بولمايدۇ، بۇ نەرسىلەرنى مىللىلار ئىشلىمەيدۇ، ئىچكىردىن كەلتۈرمىسەك باشقا جايدا يوق،ئۇنى سالمىساق مەھسۇلات ئىشلەپچىقارغىلى بولمايدۇ» دەپ جاۋاپ بەردى.شۇنىڭ بىلەن مەن بۇ ئاتالمىش مىللى كارخانىلارنىڭ « مىللەت ئۈچۈن ئىشلەپچىقارغان ئەشۇ قىممەتلىك مەھسۇلاتلىرىغا مىڭ لەنەت دىدىم.»ئويلاپ بېقىڭلار! بىز ئۇ خىل يىمەكلىكسىزمۇ ياشاپ كەلگەن.يەنە ئۇنى يىمىسەكمۇ ياشاپ كىتەلەيمىز. شۇڭا دوسلارنىڭ بۇ خىل مىللى مەھسۇلاتلارنى سېتىۋېلىپ يىيىشتىن ۋە باشقىلارغا سوغا قىلىشتىن، ھەتتا ئۇنى ئۆز دۈشمىنىڭگىمۇ سايە قىلىشتىن توسىشىنى تەۋىسيە قىلىمەن.بۇمۇ خۇددى يۇقارقى تىمىنى يازغان دوستۇم دىگەنگە ئوخشاش ھارامنى يىيىشكە بولمىغانغا ئوخشاش ئونى قوللاشقىمۇ بولمايدۇ.سېتىۋېلىشمۇ قوللاش ياردەم بىرىشتىن ئايرىلالمايدۇ.شۇڭا يىمەك ئىچمەككە ئالاھىدە دىققەت بىلەن قاراپ، يىمەك ئىچمەكنىڭ،كىيىم
