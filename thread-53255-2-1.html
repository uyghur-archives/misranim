    شۇنداق، توپا بېسىپ غېرىپسىنىپ تۇرغان بۇ قەۋرە موللا مۇسا سايرامىنىڭ قەۋرىسى.
    ئورنى كۇچاغا قاراپ ماڭغاندا باي ناھىيەسىدىن 18 كىلومىتىر يىراقلىقتىكى توقسۇن يېزىسىنىڭ كۇچاغا ماڭىدىغان يول ئۈستىدە... .
    90- يىللاردا تاسادىپى بۇ يولدىن ئۈتۈپ قالدىم. شوپۇر ئاپتۇۋۇزغا ئازىراق يۈكى بار ئىككى يېزىلىق كىشىنى ئېلىۋالدى. بۇلارنىڭ يۈكىنى باسقىچە بىر سائەتكە يېقىن تۇرۇپ قالدۇق. دەل شۇ ۋاقىتتا بىر خەنسو ياشنىڭ مەخسەتسىز ھالدا (بەلكى) بۇ قەۋرىگە كەينىنى قىلىپ ئۇ كۆرۈنۈپ تۇرغان شادىغا ئىككى قۇلىنى كېرىپ ئۆتكۈزۈپ، پۇتى بىلەن شادىنى ئالدى- كەينىگە مىدىرتىۋاتقانلىغىنى كۆرۈپ ئاچچىغىم كېلىپ ئالدىغا بېرىپ ئىككى، ئۈچ ئېغىز گەپ قىلىپ بولغىچە يولنىڭ يېنە بىر تەرىپىدە تۇرغان 3-4 نەپەر يېزىلىق ياشلار يۈگۈرۈپ كىلىپ، بىزنىڭ مەلىلىك خەنسو يولداشنى نىمىشقا تىللايسەن دەپ تاياق يىگىلى تاس قالغان ئىدىم.... .

باي ناھىيسىنىڭ باللىرىنىڭ ئىدىيىسى بە ئىلغارلىشىپ كەتكەن  گەپ  ... مىللەتلەر  ئىتپاقلىغىنى جېنى بىلەن قوغدايدىكەندە  ..
