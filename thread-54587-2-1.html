ئۇيغۇربوي ئەپەندىم ،بۇ بىر ئېغىز سۆزىڭىز ،سىزنىڭ مېنىڭ قەلبىمدىكى ئوبرازىڭىزغا خېلىلا تەسىر يەتكۈزدى.سىزنىڭچە ئادەمنىڭ ئىتىقادى ئۇنىڭ كېيىنىشى بىلەن ئۆلچىنەمدۇ ؟مۇبادا ،ئىسلام دىنى پەيدا بولغان ۋە رەسۇلىمىز تۇغۇلغان جاي سەئۇدى ئەرەبىستان بولماستىن كېلوماتى ئىللىق ياكى نەم باشقا جايلار بولغان بولسا ،تەقۋادار مۇسۇلمانلار قانداق كېيىنەر بولغىيتى ؟  
مۇسۇلمان ئاياللار مەيلى قەيەردىلا بولمىسۇن ئىسلامچە كىيىم كىيىشى پەرىز، ئىسسىق جايدا نىپىز كىيىم كىيىپ ، سوغۇق جايدا قىلىن كىيىم كىيىدىغان ئىش يوق، ئوتتۇرا شەرىقنىڭ تىمپۇراتۇرسى باشقا جاينىڭكىدىن يۇقرى تۇرۇقلۇق ئاياللىرى ھەقىقى ھىجاپلىنىپ يۈرىدۇ.
بەزىلەر كۆكتە ئۇچۇپ سۇدا ئۈزۈپ كەتتى يىراق،مەن مىسالى يالاڭ ئاياق دەسسەپ تىكەن ماڭارىمەن!

مۇسۇلمان ئاياللار مەيلى قەيەردىلا بولمىسۇن ئىسلامچە كىيىم كىيىشى پەرىز، ئىسسىق جايدا نىپىز كىيىم كىيىپ ، سوغۇق جايدا قىلىن كىيىم كىيىدىغان ئىش يوق، ئوتتۇرا شەرىقنىڭ تىمپۇراتۇرسى باشقا جاينىڭكىدىن يۇقرى تۇرۇقلۇق ئاياللىرى ھەقىقى ھىجاپلىنىپ يۈرىدۇ.
ئۇتتۇرا شەرىقنىڭ ھەممىلا ئاياللىرى شۇنداق كىيىنەمدۇ ؟يىقىننىڭ ياقى ئوتتۇرا شەرىقتىكى مۇبارەك ۋە كازافېينىڭ قىزىغا ئوخشاش بەزى ئايال داڭدارلار  تېلېۋىزوردا پار-پات پەيدا بولۇپ قالدىغان بولدى،مەن قارىسام ئۇلار ھىجاپلانماپتۇغۇ ؟ سىزنىڭچە ھېجاپلانغانلىكى ئايال مۇسۇلمان تەقۋادارمۇ ؟ھىجاپنىڭ ئىچىدىكى دۇنيانى چۈشۈنەمسىز؟
ماتېماتىكا ئىنسان تەپەككۈرىنىڭ گېمناستىكىسى

مەن مۇستاپا كامالغا بەك ھۆرمەت قىلمەن.مەيلى ئۇ قانداق ئىش قىلمىسۇن تۈركىيەنى قۇتقۇزدى.ياۋرۇپالىقلارنىڭ ھەيۋىسنى يەرگە ئۇردى.تۈركى مىللەتلەرنىڭ قەدىر قىممىتىنى ئاشۇردى.مۇستاپا كامالمۇ مۇسۇلمان.ئۇ پەقەت يىڭلىقنى ياقىلغانلىغى،تارىخ تەرەققىياتىغا يۈزلەنگەنلىگى ئۈچۈن بەزىلەر تەرپىدىن مۇسۇلمانلارنىڭ دۈشمىندەك قارالدى.ئەمىليەتتە شۇ دەۋىرگە نىسبەتەن مۇستاپا كامال چوقۇم مۇشۇ يول بىلەن مېڭىشى كىرەك ئىدى.بولمىسا بۈگۈنكى تۈركيەمۇ بولمايتتى.ئوسمان ئىمىپىريسى دىگەن ئىسىممۇ تارىختىن چىقرلىپ تاشلىناتتى.تۈركىيە بولغاچقىلا ئوسمان ئىمىپرىيسىنڭ تارىخى تارىختىن ئۆچمىدى.
ئنساننىڭ ئاساسى ھوقۇقى كۆز يېشى بىلەن ئەمەس،ئىسسىق قېنى بىلەن قولغا كېلىدۇ.

بەلكىم ، ئاللاھ سىزنىڭ ئىرادىڭىزنىڭ قارشىسىدىن چىقىپ ، ئۇنىڭ ئۈچۈن ئۇ دۇنيادىمۇ ھەشەمەتلىك سوۋغاتلارنى تەييارلاپ قويغان بولشىمۇ مۇمكىن. بىلگۈچى پەقەت بىر ئاللاھ ئەمەسمۇ ... توۋا دىمىسە بىكار جۇمۇ !!!!  نىمىشقىكىن مۇشۇ  ۋەتەنپەرۋەرلىك تېمىلىرى چىقسىلا دىننى ئارىغا قىستاپ ، بىر قىسىلچاقلاردىن كىرگۈزىۋالمىساق چىدىمايمىز... جاھانغا ئالەمشۇمۇل تەسىرلەرنى بەرگەن بىر ئادەمگە  نەدىكى بىر  ( بەك يۇقۇرى بولسا كومپىتۇر مۇلازىمەتچىسى، بەلكىم ئۇنىڭدىن سەل يۇقۇرراقتۇ، ئېنىقكى مۇستاپا كامالچىلىك ئەمەس )  ئادەملەرنىڭ باھا بىرىپ ، ئۆزىچە ئاچچىق تەنە باھا  بېرىپ كېتىشلىردىنمۇ قىززىق، بىئارام ئىش بولمىسا كېرەك....
