ئەسكەرتىش:  بۇ بىر ئىجادى ھېكايە خالاس . مۇشۇ  كەسىپتە ئىشلەۋاتقان ياكى جەمىيەتنىڭ مەلۇم قاتلاملىرىدا ياشاۋاتقان بىراۋلارنىڭ تۇرمۇشىغا ئوخشاپ قالسا  باشقىچە چۈشۈنۈپ قالماسلىقىڭلارنى ئۈمۈد قىلمەن .

تۈرمە ساقچىسىنىڭ مۇھەببىتى  (1)



    بۈگۈن ئايغا 20 ، جىنايەتچىلەرنى تۇققانلىرى بىلەن كۆرۈشتۈرىدىغان كۈنىمىز ،ئالدىراش كۈنىمىز يەنە باشلاندى، ئەترەت باشلىقىمىزنىڭ ئۇرۇنلاشتۇرۇشى بويىچە ھەممە تۈرمە ساقچىسى بىكىتىلگەن ئۇرۇندا تۇرىشىمىز، ئۇرۇنلاشتۇرۇش بويچە خىزمەت قىلىشىمىز كېرەك، ئۆز مەيلىمىزچە ھەرىكەت قىلشقا ئەسلا يول قۇيۇلمايدۇ، بۈگۈن يەنە بىزنىڭ ساقچىلارنىڭ چاقچىقى بۇيۇنچە ئېيتقاندا ئادەم تۇرماس چۆلدە گۈزەللەرنى بىر قېتىم كۆرۈۋالىدىغان كۈنىمىز، نېمشقا دەمسىز ؟  بۈگۈن بۇ يەردىكى مەھبۇسلارنى يوقلىغىلى  نۇرغۇن كىشىلەر كېلىدۇ، ئۇلارنىڭ ئىچدە ئانچە مۇنچە چىرايلىق قىزلارمۇ بار، ئۇلار بەلكىم مۇشۇ يەردىكى مەھبۇسلارنىڭ سىڭلىسى، قىزى، ئايالى بۇلىشىمۇ مۇمكىن، ئۇلارنى كۆرسەك قاپاقلىرىمىزنى تۈرۈپ سۈرلۈك قىياپەتتە تۇرغىنىمىز بىلەن كۆڭلىمىزدە بولسا باشقىچە خىيال، بۇ ئادەم دېگەن ھېسياتلىق مەخلۇقتە، مەيلى ساقچى بولسۇن، مەيلى دېھقان ھەممىسىدە ھىسيات بار، يات جىنسلارغا قارىتا  قىزقىش بۇ بەلكىم شەرتسىز رېفلىكىس بۇلۇشى مۇمكىن، بۇلۇپمۇ ھەر ئايدا ئانچە مۇنچە چىرايلىق  قىزلارنى بىرەر  قېتىم كۆرىدىغان بۇ يەردە بۇ ئالاھىدە بىر ئىشمۇ ئەمەس، تۈرمىمىز چەت ،شارائىتى ناچار ،تۆت ئەتراپى قۇملۇق  بىلەن تەبىئىي قورغان ياسىغان، قېرى توغراق ۋە يۇلغۇنلاردىن باشقا ئالاھىدە مەنزىرىسىمۇ يوق جايدا بولغاچقا بۇيەردە ئولتۇراقلاشقان  ئائىلىكلەرمۇ ناھايىتى ئاز ، تۈرمىمىزدە ئىشلەيدىغان قىزلارنىڭ كۆپىنچىسى  كېلىپلا ئۇزۇن ئۆتمەي بۇ ناچار شارائىتتىن قانداق قىلسا قۇتۇلۇپ يۆتكىلپ كېتىدۇ، بۇ يەردە مۇقىم ئىشلەپ قالغانلار بولسا تۈرمىدىكى بىرەرى بىلەن توي قىلىپ ئولتۇراقلىشىپ مۇقىم ئىشلىگەنلەر، بويتاق قىزلار تۇلىمۇ ئاز. ئۇلار بويتاق بولغان بىلەن جۈپسىز ئەمەس ھەممىسىنىڭ يۈرگىنى بار، بۇ يەردىكى بويتاق قىزلارنى تەقسىم قىلسىمۇ 9 ئەر كىشگە بىرسىمۇ توغرا كەلمەسمىكىن دەپ ئويلاپ قالىمەن. ئىش قىلىپ قىز بالا كەمچىل ،شۇنداق بولغاچقا بىز پەقەت دەم ئالغان كۈنلىرىمىز يۇرتىمىزغا، ئۆيىمىزگە بارغاندىلا  چىرايلىق قىزلار بىلەن كۈرۈشۈش پۇرسىتىگە ئىگە بولىمىز، بۇ يەردىكلەر توي قىلماقچى بولساقمۇ ئۆيىمىزدىكىلەر ياردەم قىلىپ تېپىپ بەرگەن قىزلار بىلەن توي قىلىپ بۇ يەرگە ئەكەلمىسەك بۇ يەردىن قىز بالا تېپىش تۇلىمۇ تەس، ماۋۇ ئىشلەۋاتقان قىزلارمۇ قىزىق ئۇلارغا شۇنداق ھۆرمەت قىلىپ ۋاي دەيدىغان بىزدەك كېلىشكەن يىگىتلەرگە قارىماي ئالى مەكتەپلەردە بىللە ئۇقۇغان مۇھەببەتلىرىگە ئىشتەي ساقلىشىپ يۈرۈيدۇ، ئۇلار بىلەن توي قىلغان تەقدىردىمۇ خىزمەت ئورنىنى ئالماشتۇرالماي بىرمۇنچە ئاۋارىچىلىق . قارىسىڭىز ھەپتىدە بىرەر قېتىم يولدىشىنىڭ ئۆيگە بېرىپ بىر ئىككى كۈن تۇرۇپ قايتىپ كىلۋاتقان،توي قىلدىمىكىن دىسە قىلمىغاندەك، قىلمىدىمىكىن دىسە يىراقتا ئەرلىرى بار، ئۇلارنىڭ ئەرلىرىمۇ قىزىق كۈندە بىللە بۇلالمىغان خوتۇنىمۇ خۇتۇنۇم دەپ يۈرۈشكەن، ئەر ئايال دېگەن بىر ئۆيدە تۇرۇپ ئىسسىق تاماقنى بىللە يەپ، بىر بىرنىڭ ھالىدىن خەۋەر ئېلىپ، خۇشالىق خاپىلىقلارغا
ئورتاقلاشمىسا قانداقمۇ ئەر خوتۇن بولىدۇ دەيسىز، بىرسى قارىسىڭىز بۇ يەردە بىرسى  ئۇ يەردە بىرسى، تۇتقان ئۆيى يا ئۆيگە ياكى ياتاققا ئوخشىمغان. بۇ بەلكىم مەندەك بويتاقلارنىڭ قارىشىدۇ. مەنمۇ بىر كۈنلەردە شۇلاردەك بوپ قالسام بۇنىڭغا بىر نېمە دېمەك بەك تەس.
   سىز بەلكىم مىنڭ تۈرمە ساقچىسى ئىكەنلىكىمنى بىلپ ماڭا ھەۋىسىڭىز كەپ قېلىشى مۇمكىن،توغرا چوقۇم كېلىدۇ .ئەينى ۋاقىتتا مېنىڭمۇ ساقچى فورمىسى كەيگەنلەرنى كۆرسەم ياكى ساقچىكەن دىگىنىنى ئاڭلىساممۇ ئۇلارغا ھەۋىسىم كىلەتتى، كىچىكىمدىن ساقچى بۇلۇشنى ئارزۇمۇ  قىلغان ئىدىم. شۇ ئارزۇيۇمغا ئاساسەن،تۇلۇق ئوتتۇرىنى پۈتتۈرگەندە دادام ئوغلۇم دوختۇر بولغىن تىببى ئونۋېرسىتىتىنى تاللا دىسە، ياق مەن دېگەن ساقچى بولىمەن، دەپ مۇشۇ كەسپنى تاللاپ ئوقۇدۇم، ئوقۇغاندىمۇ تازا تىرشىپ تۆتنىڭ بىرى بولۇپ ئوقۇدۇم ،مەكتەپتىمۇ نەتىجەم خىلى يامان ئەمەس بولدى، بۇرۇن ئالى مەكتەپنى پۈتتۈرگەنلەرنى ئۆزىنىڭ يۇرتلىرىغا بىر تۇتاش تەقسىم قىلاتتى ،لېكىن ماڭا كەلگەندە ئىمتىھان ئارقىلىق خىزمەت ئورنىنى ئۆزى تاپىدىغان ئىش چىقتى، مەنمۇ مەمۇرلۇق ئىمتىھانىغا قاتناشتىم، ئىمتىھان نەتىجەم چىقتى، مەن مۇشۇ چەت تۈرمىگە بۆلۈندۈم، ئۈزۈمنىڭ تۇغۇلۇپ ئۆسكەن يۇرتۇمنى تاشلاپ بۇ يەرگە كەلمەك سەل تەسكە توختىدى، نېمىشقا دەمسىز؟ مەن شەھەردە چوڭ بولغان، كىچىكىمدىن جاپا تارتىپ باقمىغان دېسەممۇ ئاشۇرۋەتمەيمەن،ئۈزۈمنىڭ ئابرۇيپەرەسلىكىممۇ كۈچلۈك، مېنىڭ ئۈزۈمنىڭ يۇرتىدا بىرەر كىچكرەك بولسىمۇ ساقچىخانىلاردا ئىشلەپ، ئاشۇ سۈرلۈك فورمامنى كىيىپ باشقىلارغا ئۈزۈمنى تازا كۆرسۈتۈۋالغۇم بارئىدى، بۇرۇن مىنى ساۋاقداشلىرىم كىتاب پۇرۇچ، ھايات ماركس دىگەندەك گەپلەرنى قىلشىپ ئانچە ئارىسىغا ئالمايتتى، مەنمۇ دەرسلەردە نەتىجەمنىڭ ياخشىلىقىدىنمۇ، ياكى كېلىشكەن بوي بەستىمنىڭ بارلىقىدىنمۇ ۋە ياكى ئائىلىمىزنىڭ ياخشى شارائىتىدىنمۇ ئىش قىلىپ بىر خىل پەخىرلىنىش تۇيغۇم بۇيۇنچە باشقلارغا ئانچە بەك ۋايمۇ دەپ كەتمەيتىم،ئۈزۈم ھاكاۋۇرراق بولغاچقا قىزلارنىمۇ ياراتماي، پۈتۈن زېھنىمنى پەقەت دەرسكىلا قارىتاتتىم، ماڭا يېقىنچىلىق قىلدىغان قىزلارغا تەكەببۇرلۇق بىلەن مۇئامىلە قىلغاچقىمۇ ئۇلار بىر مەزگىلدىن كېيىن مۇئامىلىسىنى ئۆزگەرتىپ مەن بىلەن كارى يوق يۈرۈشەتتى، مىنى ھېسسىياتسىز ،توڭ دىيىشىپ ئانچە مۇنچە غەيۋىتىمنى قىلشاتتى.
    مەن ئاشۇ مىنى چۈشەنمەيدىغانلارغا ئۈزۈمنىڭ قانچىلىكلىكمنى،قانچىلك قابىليىتىمنىڭ، ئىستىقبالىمنىڭ بارلىقنى بىلدۈرۈپ قويماقچى ئىدىم، لېكىن بۇ تەلىيىمنىڭ كاجلىقنى مۇشۇ چەت تۈرمىگە ساقچى بولۇپ كىلىپ قالدىم دەسلەپ كەلمەي كىيىنچە يەنە ئىمتىھان بىرەيمۇ دەپ ئويلىغان ئىدىم، لېكىن مۇشۇ چىرايلىق فورمامنى كىيىشكە ئالدىراپ، ئۆز-ئۈزۈمگە تەسەللى ئېيتىپ،كەسپكە بولغان قىزغىنلىقىم تۈپەيلى بۇ يەرگە كىلپ بىر مەزگىل ئىشلەپ بېقىش نىيتىگە كىلپ تۈرمىگە كىلپ ئىشقا چۈشتۈم، بۇ يەرگە كەلگەندىن كېيىن بۇ يەرنىڭ شارائىتىنى كۆرۈپ كۆڭلۈم بىر قىسىم بولدى، ئەتراپى قۇملۇق، يۇلغۇن، توغراقتىن باشقا دەل- دەرەخلىرىنىڭمۇ تايىنى يوق، قاتناش قۇلايسىز ،تاماقنىڭ ناچارلىقى ،مېنىڭچە بەلكىم شۇنداقتۇر مەن نېمە دىسەم شۇنى ئىتىپ ھالىمدىن خەۋەر ئالىدىغان ئانام يىراقتا، يىگۈم بار نەرسىنى سېتۋېلىپ يەي دىسەممۇ يوق، كۈندۈلۈك تۇرمۇش بۇيۇملىرىدىن باشقا نەرسىلىرىنىڭ تايىنى يوق بىر نەچچەماگىزىن، ھاراق تاماكا ساتىدىغان  كىچككىنە بىر قاۋاقخانا، بۇلارنى كۆرۈپ ئۈزۈمنىڭ بۇ يەرگە كەلگىنىمگە ،مىڭ پۇشايمان قىلدىم، قايتىپ كېتەي دىسەم ھېلىقى ھاكاۋۇرلۇقۇم يول قويمايتى، ساۋاقداشلىرىم تونۇش بىلىشلىرىمنىڭ، قاراڭلار ئۆردەك مۇشۇ ساغرام بولسا ھەممە كۆلنى سېسىتىمەن. دەپ ھەممە يەرنى سېسىتقاندەك ماۋۇ ئاتىكاچا ،خانزادە بىر دەمدىلا خىزمەت ئورنىنى سېسىتىپ كەپتۇ دەپ مىنى مازاق قىلىشىدىن ئەنسىرەيتىم، ئۇلار ھەتتا مىنى خىزمەتتىن ھەيدىلىپ قايتىپ كەپتۇ دىيشمەيمۇ قالمايدۇ ئەلۋەتتە، بۇ خەقنىڭ ئاغزىنىڭ دەرۋازىسى بولمىغاندىكىن نېمە دېگۈسى بولسا دەيدۇ،ئاخرى بۇ ناچار شارائىتتىن مىنڭ تەكەببۇر مىجەزىم ئۈستۈن كىلپ بۇ يەردە تۇرۇپ قالدىم،خىزمەتداشلىرىم بىلەن بىر ئوبدان چىقىشىپ قالدىم، مىنڭ كەم سۆز ،چوڭ سۈپەت ھالىتىم باشلىقىمىزغا خىلىلا يېقىپ قالغان بولسا كېرەك بۇ يەردە زېرىكىپ قالسىز ،دەپ ياتىقىمغا كونىراق بولسىمۇ تېلېۋىزۇر ئورۇنلاشتۇرۇپ بەردى. زېرىككەندە پات-پات تېلۋىزور كۆرۈپ قۇيىمەن.
  كۈنلىرىم ئالدىراش خىزمەت ھەلەكچىلىكىدە ئەتىگىنى كاناي چېلىنىشى بىلەن ئورنۇمدىن تۇرۇپ ھېلىقى مىنڭ ئامراق فورمامنى كىيىپ ئەينەكتىكى سىمايىمغا بىر پەس قاراپ مەغرۇرلىنىش تۇيغۇسى بىلەن تۈرمىگە ماڭىمەن، مەن مەسئۇل خۇزىدىكى مەھبۇسلارنىڭ تۇرمۇشىدىن ئانچە مۇنچە خەۋەر ئېلىپ ئۇلارنىڭ ۋەزىپىلىرىنى ئۇرۇنلاشتۇرمەن، كەچكىچە ئۇلار بىلەن بىرگە، كۈنلىرىم قاپاق تۈرۈپ سۈرلۈك ئۆتۈش بىلەن، بىزنىڭ قائىدىمىزدىمۇ جىنايەتچىلەرگە ئانچە يېقىنلىشىپ كىتىشكە رۇخسەت قىلىنمايدۇ، بۇ خىزمەت ھەقىقەتەن پەقەت مەن ئۈچۈنلا يارالغاندەك ماڭا تۇلىمۇ ماس كەلدى، بۇرۇنمۇ ئېچىلمايدىغان قاپىقىم ئەمدى تېخمۇ ئېچىلماس بولدى، مەن مۇشۇنداق قاپىقى يامان، توڭ ئىدىم، گېپىم كەسكىن بولۇپ بىرەرسى بۇيرۇلغان ۋەزىپنى ئۇرۇنلىيالمىدىمۇ بولدى، ئۇنىڭ تۇتۇلغىنى شۇ. تۈگمەس قائىدە مىزانلارنى دەرس ئۆتكەن كەبى سۆزلەپ قۇلاق مېڭىسىنى يەپ بۇلمەن،ئۇلار باشلىرىنى تۆۋەن سالغىنىچە گېپىمنى ئاڭلىشىدۇ، ئاڭلاۋاتقاندا دىققەت قىلمايدىغان ئىشلار ئاساسى جەھەتتىن يوق، بار بوپ قالسا مېنىڭ ئاچچىقلىنىدىغىنىمنى ئۇلار ئوبدان بىلىشىدۇ، مەن دېگەن ئۇلارنى تەربىيلەپ ئۆزگەرتىدىغان ساقچى ئۇلار بولسا خاتالىقنى تۈزەتكۈچىلەر ئەمەسمۇ؟، ئۇلار ماڭا بويسۇنۇشى مىنڭ تەربىيەمنى قوبۇل قىلشى كېرەك .
     كەچلىرى بەزىدە تېلېۋىزوردىن پات-پات كىنو كۆرۈپمۇ قۇيىدىغان بولدۇم، بەزى ۋاقىتلاردا مۇھەببەت تەسۋىرلەنگەن كىنولارنى كۆرسەم ئۈزۈممۇ بىلمەي  ،مۇھەببەتلىشىشنى ،قىزلارنى ،ئائىلىنى خىيال قىلىپمۇ باقىدىغان بولدۇم، لېكىن كىمنى خىيال قىلىدىغىنىمنى ئۈزۈممۇ بىلمەيتىم، مەن ئەزەلدىن تونۇيدىغان قىزلارغا ھەۋەس قىلىپ، ئۇلارغا كۆڭلۈم چۈشۈپ باققان ئەمەس، مەن بەلكىم ھىسيات جەھەتتە ئانچە بەك قىزغىن بولمىسام كېرەك، بەزىدە مەھبۇسلارنىڭ ئائىلىسگە،ئايالىغا يازغان خەتلىرىنى ئۇقۇپ قالساممۇ ئۆزۈمچە باشقا بىر غەلىتە نەرسىلەر يېزىلغان ئەسەرلەرنى ئوقۇغاندەك بوپ قالاتىم،خەقنىڭ خېتىنى نېمشقا ئوقۇيسىز دەمسىز؟ تۈرمىمىزنىڭ قائىدىسىدە مەھبۇسلارنىڭ ھەر قانداق خەت چېكى بىزدەك كادىر ساقچىلارنىڭ بىر قېتىملىق تەكشۈرشىدىن ئۆتمىسە ئىۋەرتىلمەيدۇ، كەلگەن خەتلەرنىمۇ ئەلۋەتتە بىز بىر قۇر ئۇقۇپ چىققاندىن كېيىن ،مەھبۇسلارغا رۇھىي جەھەتتىن زەربە بېرىدىغان خەت چېكىدىن باشقا خەتلەرنى ئۇلارنىڭ قۇلغا تاپشۇرۇپ بىرمىز،مەھبۇبسلارنىڭ ئايالىغا يازغان مۇھەببەت خەتلىرىنى كۆرگەن
چاغلىرىمدا خىياللىرىمغا يەنە شۇ خوتۇن خەقنىڭ ئېزىتقۇ خىياللىرى تېخمۇ ئىچكىرلەپ كىرىپ مىنى تېخمۇ بىئارام قىلاتتى،خىزمەتداشلىرىمنىڭ قۇلىقىغا تېلېفونلىرىنى مەھكەم يېقىپ مۇھەببەتلىشىۋاتقانلىقلىرى كۈزۈمگە چېلىقىپ قالغاندا،ئۈزۈمنىڭ بىرەر قىز بالا بىلەن نىمشقىدۇر مۇھەببەتلىشىپ قويمۇغۇنۇمغا ئاچچىقىممۇ كىلەتتى. مانا مۇشۇنداق كۈنلەردە خىزمەتداشلىرىم مىنى شاڭخو قىلدىغان دارتمىلاپ گەپ قىلىپ چېقىۋالىدىغان بولۇشتى.
-ئەكبەرجان نىمانداق يالغۇز يۈرىۋېرىسىز  كاككۇكتەك، تېلېفۇنىڭىزغا قىزلارنىڭ تېلېفۇنى ئۇلانمايدىغان بوپ قالدىمۇ يە؟ قارىڭە مۇشۇ تۈرمىدىكى مەھبۇسلارنىڭ جېنىدا ئايالى، تېخى بەزىلىرىنىڭ خەت بىلەن مۇھەببەتلىشىۋاتقان مۇھەببەتلىرى بار، چىقساق توي قىلىمىز دەپ ۋەدىلشۋاتقان،سىز نىمانداق يۈرىۋېرىسىز ئەركىن ئادەم تۇرۇپ ماۋۇلارچىلىك بۇلالماي........
    بۇنداق تۈگمەس شاڭخو قىلىشلار مىنڭ جىمجىتلا ئۇلارنىڭ يېنىدىن كېتىپ قېلىشىم بىلەن توختاپ قېلىشاتتى، مەن ئاستا سىرتقا چىقاتتىم دە يىراقلاردىكى توغراقلارغا قارىغىنىمچە ئۈزۈممۇ بىلمەيدىغان كۆڭلۈمدىكى قىزلارنى ئەسلەپ قالاتتىم ،مۇنداق چاغلاردا يىراقتىكى توغراقلارمۇ جۈپ جۈپ تۇرغاندەك، ئۇچۇۋاتقان قۇشلارمۇ جۈپلىرى بىلەن ئۇچۇۋاتقاندەك تۇيلاتتى،پەقەت بۇ تەبىئەت دۇنياسىدا مەنلا يالغۇزدەك يالغۇزلۇق ھېس قىلاتىم،مانا يېشىممۇ 26ئالتىدە، ئۈزۈم ئالى مەكتەپ تەربىيىسى كۆرگەن ،ھەممە كىشى ھەۋەس قىلدىغان مۇقىم خىزمەت، كەسىپ، ئائىلەمدىكلەرنىڭ ئەۋزەل شارئىتى، كىلشكەن بوي بەستىم...مەندە زادى نېمە كەم، مەن مۇشۇ ئۆمرۈمگىچە قىزلارغا مۇھەببەت ئىزھار قىلىپمۇ باقمىدىم، ئۇلارنىڭ چىرايى ياكى خۇي مىجەزىگىمۇ ھىرىس قىلىپ دىققەت قىلغىنىمنىمۇ ئۇقمايمەن،گۈزەل مۇھەببەت لىرىكىلىرىنى مەن پەقەت كىتابلاردىنلا ئوقۇدۇم، كۆڭۈل بوستانىمدا مۇھەببەتتىن ئىبارەت بۇ  خۇشپۇراق گۈلنىڭ جىسمى ھەم پۇرىقىمۇ يوق،ئەڭ ئاددىيسى قىز بالنىڭ قۇلنىمۇ تۇتۇپ باقمىغاچقا قىزلارنىڭ قۇلنى تۇتسا قانداق بۇلدىغىنمۇ بىلمەيمەن، توۋا دەيمەن ،ئىچىمدە مىنڭ كاللامدا مەسلە بارمىدۇ؟ دەپمۇ قالىمەن، ئىشقىلىپ چوقۇم بىر كۈنى مىنڭ كۆڭلۈمدىكى قىزنى چوقۇم ئۇچرىتىمەن ،دېگەن ئۈمىدتە كۈنلىرىمنى زىرىكىش ئىچىدە ئۆتكۈزىمەن.
    ماڭا مۇشۇنداق قالايمىقان خىياللار كىرگەن كۈنلەردە، بىر قېتىملىق كۆرۈشتۈرۈش كۈنى يىتىپ كەلدى، ئادەتتە كەلگەن كەتكەنلەرگەئانچە ئىرەن  قىلمايدىغان مەن بۇ قېتىم كەلگەنلەرنى تىزىملاشقا مەسئۇل بوپ قاپتىمەن،كۈرۈشكىچلەرنى تىزىملاۋېتىپ .تىزىملاتقىلى كەلگەن كىشىلەر ئىچدە تۇرغان قىزلارغا قاراپ قاپتىمەن، ئۇلار بىچارىلەرچە تەلمۈرۈشۈپ، ئائىلىسدىكى مەھبۇسلار بىلەن كۆرۈشۈۋېلىش ئۈچۈن، رەت بۇيۇنچە ئۆچرەتتە تۇرۇپ نۆۋەت كۈتۈشەتتى، قوللىرىدا مەھبۇسلارغا ئېلىپ كەلگەن تۇرمۇش بۇيۇملىرى، يىمەكلىك، ھەتتا مەھبۇسلار ئەڭ ئامراق تاماقلارنى ئېتىپ ئەكىلىدىغانلارمۇ بار ئىدى.
   مانا مۇشۇ تۈرمىدىن ئىبارەت ئالاھىدە مەكتەپتە مەھبۇسلار باشقىلارنىڭ ئۈزىگە قانچىلىك كۆيۈنىدىغانلىقىنى ھەقىقى ھېس قىلالامدىكىن دەيمەن، ئۇلار ئائىلىسىدىكلەر ئېلىپ كەلگەن نەرسىلەرنى رەھمەت ئېيتىشىپ قوللىرىغا ئېلىپ، كۆزلىرىگە ياش ئېلىپ رەھمەت ئېيتقىنچە  نازارەت بىلەن ياتاقلىرىغا قايتىشاتتى، تۈرمىمىزدە كۆرۈشتۈرۈش ئۆيىدە تېلېفون ئارقىلىق كۆرۈشتۈرىدىغان بولغاچقا كەلگەنلەر ئەينەك ئىچىدىكى مەھبۇسلارغا تەلمۈرۈپ مىھىر بىلەن قارشىپ سېغىنغىنىنى، ئۇلارنىڭ بالدۇر ئۆزگۈرۈپ ئائىلىگە قايتىپ كىلىشنى ئارزۇ قىلدىغانلىقلىرنى،ئائىلىسدىكى يېڭلىقلارنى،ئائىلە تۇرمۇشنى، ھەممىسىنىڭ تىنچ ئامانلىقنى ئۇلارغا يەتكۈزۈپ يىغلاپ قىزارغان كۆزلىرىنى ئۇۋۇلۇشۇپ قايتاتتى، بۇ كۆرۈنىشلەر ماڭا خۇددى تەسىرلىك كىنو كۆرۈنىشىدىكىدەك تۇيۇلماقتا،بۇرۇن ئەجەب ھېس قىلماپتىمەن،ئەسلى بۇمۇ بىر خىزمىتىمنىڭ ئالاھىدە تەرىپى بولسا كىرەك،بۇنداق ھەقىقى تەسىرلىك كۆرۈنىشلەرنى سىز ھاياتىڭىزدا نەچچە قېتىم كۆرەلەيسىز، مەن مۇشۇ تەسىرلىك كۈرنىشلەرگە،مەپتۇن بولۇپ قارىغاچ كەلگەن چىرايلىق قىزلارغىمۇ نەزەر تاشلايتتىم، چىرايلىقراق بىرەر قىزنىڭ يىغلاپ تۇرۇپ تۈرمىدە تەربىيلىنىۋاتقان مەھبۇسلارغا بولغان سېغىنىشىنى، ھىسداشلىقنى، بىلدۈرۈۋاتقانلىقىنىنى كۆرسەم ئىچىمنى بىر نەرسە تاتىلىغاندەك بىئارام بۇلاتىم،نىمشقا دىدڭىما؟قىزىقكەنسىز،مەن بىر كادىر ساقچى تۇرۇپ، ھەممە شارائىتىم تولۇق تۇرۇپ مۇشۇ تۈرمىدىكى مەھبۇسچىلىك باشقا بىرنىڭ دېمەكچى بولغىنىم چىرايلىق قىزلارنىڭ كۆيىنىشكە، مىھىر مۇھەببىتىگە، كۆڭۈل بۆلۈشىگەئېرىشەلمەيۋاتسام، مۇشۇنداق چىرايلىق قىزلارنىڭ ئاشۇ تۈرمىدىكى مەھبۇس ئۈچۈن ياش تۆكۈۋاتقان كۆرۈنىشى مىنى ئەلۋەتتە بىئارام قىلماي قالمايدۇ دە، شۇ چاغدا ئاشۇ مەھبۇسلارغىمۇ سەل ھەسىتىم قوزغىلاتتى.
