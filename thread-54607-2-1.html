تۈرمە ساقچىسنىڭ مۇھەببىتى(2)بارچۇق قىزى



كۈرۈشتۈرۈش زالىنىڭ  ئىشكى ئالدىغا قۇيۇلغان ئۈستەل ئۇرۇندۇقتا ئولتۇرۇپ،كۈرۈشكۈچلەرنى تىزملاپ رەت تەرتىۋى بۇينچە ئىچگە كىرگۈزگەچ بۈگۈن كەلگەنلەرگە دىققەت بىلەن قاراپ قالدىم،بۇ كۆڭلۈمگە قىزلارنىڭ ئېزىتقۇ خىياللىرى كىرۋالغان كۈنلۈرۈم بولغاچقا  ئۈزۈممۇ بىلمەي ئۇلارغا قاراپ قاپتىمەن،قارسىڭىز بۇ قىزلارمۇ ھەمىسى ئوخشاش بىر جىنىس بولسىمۇ لىكىن سىرتقى كۈرنىشى،چىرايى ئادەمگە ھەرخىل تۇيغۇ بىردىكەن،بىرسى قارسىڭىز ئۇزۇن،كىيملەرنى كىيشىپ ياغلىقلارنى ئەرەپ قىزلىردەك ئۇراپ خۇددى ئەرەبلەردەك بۇلۋالغان  ،بىرسى قىسقا كىيملەرنى كىيشىپ چاچلىرنى ھەرخىل رەڭدە بۇيشۇپ  ياۋرۇپا قىزلىردەك قېنىق گىرىم قىلپ چاچلىرنى بۈدرە قىلۋالغان ،بىرسى ئۇرۇق بىرسى  سىمىز،بىرسى ئادەتىكىچە بىرسى تۇلمۇ چىرايلىق،بىرسى ئاپئاق بەزىسى بۇغداي ئۆڭ ،بىرلىرى گىرىممۇ قىلماي تەبئىيلا،بىرسى ياراشتۇرۇپ گىرىم قىلۋالغان،مەن مۇشۇ يېشىمغىچە بۈگۈنكىدەك قىزلارنى كۈزتۈپ باققىنمنى بىلمەيمەن،شۇنچە چوڭ شەھەردىكى قىزلارغا ھەۋەس بىلەن قاراپ باقمىغان ئادەمنىڭ مۇشۇ ئادەم تۇرماس چۆلدىكى مەھبۇبسلارنىڭ ئۇرۇق تۇققان،خۇتۇن،قىزلىرغا ھەۋەس بىلەن قاراپ قېلشىم تۇلمۇ قىززىق ئىش ......

    ئالدىمدا ئۆچرەت تۇرغان كىشلەر بىر بىرنى قىستىشىپ،يوقلاپ كەلگەن كىشىلىرنى بىر مىنىۇت بولسىمۇ بالدۇراق كۈرۋېلىش ئۈچۈن ئالدىرايتتى،ئۇلارنىڭ قوللىردا مەھبۇبسلارغا ئەپكەلگەن تۇرمۇش بۇيۇملىرى،ئىچ كىيم دىگەندەك نەرسىلەر،ئاندا ساندا كىشلەرنىڭ قوللىرىدا يىمەكلىك،ھەتتا مۇشۇ ئۆزگىرۋاتقان جىنايەتچىلەرنىڭ ئەڭ ئامراق تامقىنى ئىتىپ ئەكەلگەنلەرمۇ بار ئىدى،زالنى بىر پەس ۋاراڭ چۇرۇڭ قاپلىدى،مۇشۇ چىرايلىق قىزلارنى كۈزتۈش بىلەن ئاۋارە بۇلۋاتقىنمدا ئاچىقىم راسلا تۇتتى،ئورنۇمدىن تۇرۇپ قاپقىمنى تۈرۈپ،يۇقرى ئاۋازدا،

-قىستاشماڭلار،پاراڭ چۇرۇڭ قىلماڭلار! يەنە كىم مۇشۇنداق قىلسا  بۇ يەردىكلەر بىلەن كۈرۈشتۈرمەيمەن-دىدىم ،مىنىڭ قاپقىم بىلەن سەل ئۈركىگەن  كىشلەر جىم بۇللۇشتى،ئۇلارمىنىڭ يوقلاپ كەلگەن كىشىلىرى بىلەن كۈرۈشتۈرمەسلىكىمدىن ئەنسىرگەن چېغى زالنىڭ ئىچى تىنچىلىنىپ قالدى،كۈزۈم يەنلا قىزلارنىڭ  ئۈستى بېشىدا،يوقلىغۇچلارنىڭ ئارسىدا  ،ئانچە مۇنچە چىرايلىق،مەدىنى ياسنىپ،گىرىملىرنىمۇ جايىدا قىلپ ،كىيملىرنى ياراشتۇرۇپ كىيگەن زىيالى سۈپەت قىزلارمۇ بار ئىدى،لىكىن بۇنداقلار ماۋۇ،پۈتۈن چىرايىنى  يۇغان قارا ياغلىق بىلەن ئۇراپ كۈزىنلا ئۇچۇق قويغان قىزلاردەك كۆپ ئەمەس،بەلكىم زىيالىلارنىڭ ئارسىدا جىنايەت ئۆتكۈزۈپ بۇ يەرگە كىلدىغانلارنىڭ سانىنىڭ ئاز بولغانلىقدىن بولسا كىرەك،مۇشۇنداق ياغلىقتا يۈزلىرنى مەھكەم ئۇرۋېلىشقان قىزلارنىڭ كىيگەن كىيمىدىن،زىلۋا بۇيدىن ياش قىز،چوكان ئىكەنلىكنىپەرىقلەندۈرمىسىڭىز  پەرىقلەندۈرمەك تۇلمۇ تەس ئىكەن،ئۆچرەتتە قاتارلىشىپ كۆزلىرنى بىچارىلىك بىلەن ماڭا تىككەن ئانىلارمۇ مىنىڭ دىققىتمنى ماۋۇ چىرايلىق قىزلاردەك تارتمايۋاتاتتى،ئۇ قىزلارغا قاراپ   كۈزۈمنىڭ پىتىنى ئۆلتۈرۋاتقان شۇ چاغدا بىر كىچىك بالنىڭ يىغلىغان ئاۋازى دىققىتمنى بۆلدى،ئالدىمدىلا3-4ياشلار چامسىدىكى بىر ئۇماق  ئۇغۇل بالا بارچە ئاۋازى بىلەن-سۇ..سۇ..دىگىنچە قۇلىنى مىنىڭ ئۈستۈلۈم  ئۈستىگە قۇيۇلغان پۇتۇلكىدىكى مىنرال سۇغا سۇزۇپ  ۋارقىراپ يىغلاپ،زالنى بېشغا كىيگەن ئىدى،يېنىدا22-23ياشلار چامسىدىكى ،ئاق پىشماق،قاپقارا قاشلىرى تۈكلۈپلا تۇردىغان  ياراشتۇرۇپ كىينگەن بىر قىز خىجىلچىلىقتا نىمە قىلارىنى بىلمەي بالسىنى بەزلەۋاتاتاتتى
،-چىرايلىق ئوغلۇم خاپا قىلماڭ ھە؟!دادىڭىزنى  كۈرۈپ بۇلۇپلا سىزگە سۇ ئەپ بىرمەن،ئۇيۇنچۇق ئەپ بىرمەن،يىغلىماڭ جېنىم بالام.........ئۇ قىزنىڭ سۆزلىردىنلا بۇ يەرگە يولدۇشنى يوقلاپ كەلگەنلىكىنى بىلگىلى بۇلاتتى،ھېلقى ئۇماق بالا ھىچ ئاپسىنىڭ گىپىنى ئاڭلايدىغاندەك ئەمەس،جاھىللىق بىلەن تېخمۇ كۈچەپ يىغلاپ ئاپسىنىڭ قۇلدىن قاچماقچى بۇلاتتى،زالنىڭ ئىچى بۇ بالنىڭ يىغسى بىلەن تەرتىپسىزلىشىپ كەتكەندەك بولدى.

   زالنىڭ تىنچىلىقنى ئەسلىگە كەلتۈرۈش ئۈچۈن ئالدىمدىكى مىنرال سۇنى قۇلۇمغا ئېلىپ ھېلقى بالغا تەڭلىدىم
،-ئېلڭە ئوغلۇم ئېلىڭ،ئاپڭىزنى خاپا قىلماڭ ھە؟يەنە يىغلىسىڭىز،قۇلقىڭىزنى كىسىپ تامغا مىقلاپ قۇيمەن-،قانداقلارچە بۇ بالىنى بۇنداق بەزلەپ سالغىنمنىۇ بىلمەيمەن،كىچكىمدە مەن تۇلا يىغلاپ ئاپامنى خاپا قىلسام دادام مىنى ئاشۇنداق دەپ قورقۇتاتتى،مەن بىرجۈپ چىرايلىق قۇلقۇمنى دادامنىڭ كىسىپ تامغا مىقلاپ قۇيشىدىن قورقۇپ دادام بار يەردە يىغلىماسلىققا تىرشىپ ئاۋازىمنى پەسلىتىپ يىغامنى توختۇۋالاتىم،دە ئاپامنىڭ كۆينىكنىڭ دالدىسدىن دادامغا مارايتىم،ئۇنىڭ يېنىدىكى پىچقىنى چىقرىپلا  قۇلقۇمنى كىسۋىتىشتىن بەك قورقاتتىم.بۇ ئۇسۇلۇمنىڭ  ئۈنمى بولدىمۇ ھېلقى ئۇغۇل بالا قۇلۇمدىكى مىنرال سۇنى قۇلغا ئېلىپ  چىرايلىق ئېغزىنى ئۆمچەيتكىنچە پۇتۇلكىنىڭ ئېغزىنى ئېچىپ سۇ ئىچىشكە باشلىدى،زالنىڭ ئىچىمۇ خلى تىنچلىنىپ قالدى،بالنىڭ ئاپسى بوش ئاۋازدا ماڭا رەھمەت ئېيتقىنچە بالسىنىڭ ياش يۇقى يۈزلىرنى سۈرتۈشكە باشلىدى،دىققىتىم ئۇ قىزغا مەركەزلەشتى،جەنۇبى شىنجاڭدا قىزلار كىچىك تۇرۇپلا توي قىلغاچقىمكىن  يېشى 24لەرگە كىرمەي تۇرۇپلا نەچچە ياشلىق بالغا  ئانا بوپ قېلشاتتى،بۇ قىزمۇ دەل شۇنداقلارنىڭ ۋەكىلى بولسا كىرەك،يېشى ئانچە چوڭدەك قىلمايتتى،ئۇنىڭ قارلىغاچ قۇيرىقى شەكىللىك قاپقارا قۇشۇما قېشى،ئاپئاق يۈزى،ياغلىقنىڭ ئالدىدىن قۇلاق تۈۋدىن چىقىپ قالغان قۇڭۇررەڭ دولقۇنسىمان چاچلىرى،ھىللا يۈرەككە يا ئۇقى بۇلۇپ سانچىلدىغاندەكلا تۇردىغان ئۇزۇن تال -تال  كىرپىكلىرى،خۇشپىچىم غۇنچە بۇيى بۇ قىزنىڭ ساپ قەشقەر قىزى ئىكەنلىكنى  ھىس قىلش تەس ئەمەس ئىدى ئەلۋەتتە،قىز بالسىنىڭ ياشلىرنى سۈرتۈش بىلەن ئاۋارە مەن بولسام ئۇنىڭ چىرايلىق چىرايى،گۈزەل قامىتىنى كۈزتۈش بىلەن ئاۋارە ئىدىم،قىز بىر چاغدا بالسىنىڭ يۈز كۈزنى سۈرتۈپ بۇلۇپ بېشىن ئاستا كۈتەردى،شۇ ھاغدا كۈزۈم قىزنىڭ بىر جۈپ قۇڭۇررەڭ يۇغان .ئوتلۇق كۆزلىرىگە چۈششى بىلەن توك تەپكەندەك بىر سىزىم  بېشىمدىن كىرىپ پۇتۇمدىن چىققاندەك تۇيۇلدى،دۇنيادا مىنىڭ كۈزۈمدەك چىرايلىق كۆز يوق دەپ ماختىنىپ يۈرۈپتىمەن تېخى ،مىنىڭكىدىنمۇ چىرايلىق كۆزنىڭ بارلىقىغا بۇ قېتىم ھەقىقى قايىل بولدۇم،كۆزلىرىمنى ئۇنىڭدىن ھىچ ئۈزەلمىدىم،ئۇنىڭغا خۇددى نادىر  پورتىرت ئەسىرىگە قارغاندەك قارايتىم،قىز مىنىڭ ئۈزگە تىكلىپ قارغىنمدىن خىجل بولدۇمۇ قانداق چىرايلىق كۆزلىرنى مەندىن ئېلىپ قېچىپ باشقا تەرەپكە قارۋالدى،شۇ چاغدىلا ئۈزۈمنىڭ خىزمەت ئورنۇمدا ئىكەنلىكمنى ھىس قىلدىم،دىققىتمنى يىغىپ ئەتراپىمغا بىر قۇر كۆز يۈگۈرتۈم.

   ئالىدا كىرىپ كەتكەنلەر بىر بىرلەپ يېنىپ چىقتى،مەنمۇ تىزملاشنى باشلاپ بىر بىرلەپ زالغا كىرگۈزۈشكە باشلىدىم،نۆۋەت ھېلقى چىرايلىق قىزغا كەلدى،ئۇنىڭ قۇلدىكى كۈرۈشتۈرۈش كىنشكىسى بىلەن،كىملىكنى ئېلىپ تىزملاش خاتىرسىگە تىزملىغاچ ئۇنىڭ ئىسمىنىڭ ئىپپار قىز ئىكەنلىكنى،مىنىڭ قول ئاستىمدىكى مۇھەممەتنىڭ ئايالى ئىكەنلىكنىمۇ بىلۋېلىشقا ئۈلگۈردۈم،ئۇنىڭغا كىنشكىسنى بىرىپ كىرگۈزۋەتكەندىن كىيىن باشقىلارنى خوشياقماسلىق بىلەن تىزملاشقا باشلىدىم،تىزملاتقانلارغا يەنە ھېلقى قۇپاللىقىم بىلەن مۇئامىلە قىلدىم،مىنىڭ پۈتۈن نىرۋىلىرىم،ئەس يادىم شۇ ئىپپار قىزدىلا قالغان ئىدى،ئىپپار قىزنىڭ يولدىشى  مۇھەممەت مىنىڭ قول ئاستىمدا ئۈزگىرۋاتقان ،مەندىن نەچچە ياش چوڭ چىقدىغان،ئاچچىقى يامان،مىجەزى چۇسراق،قاپقارا قاشلىرى،يۇغان قارا كۆزلىرى داۋاملىق  ئويچان يۈرۈيدىغان  ساپ قەشقەر يىگىتلىرىدەك كىلشكەن،تۈرمىمىزگە خەق بىلەن مۇشتلۇشۇپ قېلىپ  جىدەل چىقرىپ كىرىپ قالغان ،مىنىڭ قول ئاستىمدا ئۆزگىرۋاتقان،چىقىپ كىتشكە نەچچە ئايلىق ۋاقتى قالغان مەھبۇس ئىدى،ئۇنىڭ مۇنچىۋالا چىرايلىق خۇتۇنىنىڭ بارلىقىغا ھەقىقەتەن  ئىچىم ئاچچىق بولدى ،ئىپپار قىزنىڭ  شۇ چىرايلىق كۆزلىرى مىنىڭ ئارامىنى بۇزماقتا ......

        كۈرشۈپ بولغان كىشلەر   ئىچدىن بىر بىرلەپ يېنىپ چىقىشقا باشلىدى . چىقۋاتقانلار ئارسىدىن ئىپپار قىزنى ئىزدەيتىم ،ئاردىن 20مىنۇتچە ۋاقىت ئۆتكەندىن كىيىن ئىپار قىز ئوغلىنىڭ قۇلىنى چىڭ تۇتقىنچە  يىغدىن قىزارغان كۆزلىرىنى ئېيتىپ زالدىن يېىنىپ چىقتى،ئۇنىڭ جاھىل ،ئۇماق ئوغلى مىنىڭ يېنىمغا كەلگەندە مىنىڭ قۇلقىنى كىسۋىتشىمدىن ئەنسىرگەندەك ئاپسىغا تاپ بېسىپ يۈگرەپ دىگدەك ئۈتۈپ كەتتى،ئۇنىڭ شۇ چىرايلىق كۆزلىرنىڭ مۇشۇ يەردىكى  ئەركىنلىكتىن مەھرۇم بولغان بىر جىنايەتچى ئۈچۈن ياش تۈكۈپ قىزرىپ كەتكىننى كۈرۈپ كۆڭلۈم بىر قىسملا بوپ قالدى،ئۇنىڭ يولدىشى مۇھەممەتكە ھەستىم قوزغۇلۇپ ،ئۈزۈمنىڭ مۇشۇنداق يالغۇز قۇشتەك يۈرگۈنۈمگە بەكلا ئاچىقىم كەلدى،مۇھەممەت بىلەن ئۈزۈمنى سېلىشتۇردۇم،مەن ئەركىن تېخى ساقچى ماڭا كۈينىدىغان بىر قىزنىڭ تايىنى يوق،نەدىكى بىر جىنايەتچىنىڭ  شۇنداق چىرايلىق ئايالى،ئۇماق ئوغلىنىڭ بارلىقىغا ئىچىم تارلىق قىلاتتى.بۈگۈنكى كۈرۈشتۈرۈش ئىشلىرمىزمۇ ئاخىرى ئاخىرلاشتى،ياتاققا كىرىپ كارۋاتقا ئۈزۈمنى تاشلاپ تۇرۇسقا قارغىنمچە ھېلقى چىرايلىق قىز ئىپپار قىزنى ئەسلەپ قالدىم،ئۇنىڭ شۇ چىرايلىق كۆزلىرى،ئويماقتەك ئېغزى كۆز ئالدىمغا كىلپ مىنى بىئارام قىلدى،ئۇنى نىمە ئۈچۈن خىيال قىلپ قالغانلىقمنىمۇ بىلمەيمەن،بەلكىم مەن بۇ يەردە  ئىشلەۋاتقان بۇ بىر يىل ئىچدە بۇنداق چىرايلىق قىزنى كۈرۈپ باقمىغىنىم سەۋەب بولسا كىرەك،يەنە شۇ سېلىشتۇرما،بىر ساقچىنىڭ قىز دوستىنىڭمۇ يوقلىقى،بىرجىدەلخور جىنايەتچىنىڭ شۇنداق چىرايلىق  ئايالى بارلىقى...مىڭەم پارتلاپ كىتەيلا دەپ قالدى،ئېغىر ئۇھ تارتقىنمچە ئۇيان بۇيانغا ئۆرلۈپ  ئۇيقۇغا كىتپتىمەن،تۇخۇ داڭگال،ئۆچكە جاڭگال چۈشەر دىگەندەك كىچىچە چۈشۈمدىن شۇ ئىپپار قىزنىڭ بىر جۈپ چىرايلىق كۈزى چىقمىدى،تۈرمىمىزدە چېلنىدىغان ئەتگەنلىك كاناي بىلەن ئۇيقۇمنى ئېچىپ يۈز كۈزۈمنى يۇيۇپ ھېلقى چىرايلىق فورمامنى  كىيىپ تامغا ئېسىقلىق ئەينەككە قاردىم،ئۈزۈمنىڭ چوڭ  قاپقار كۈزۈمنى كۈرشۈم بىلەن  تۈنگۈنكى ئىپپار قىزنىڭ چىرايلىق كۈزى ئىسمگە كەلدى،ئەينەككە قاراشتىنمۇ يالتىيىپ ئۇنىڭ شۇ چىرايلىق كۈزىنى خىيال قىلغىنمچە ئىشقا ماڭدىم،ئاشخانغا كىرىپ ھېلقى يۇغان موما بىلەن،شىپەندە ناشتا قىلپ ئىشخانام تەرەپكە يول ئالدىم،كۈرۈشتۈرۈش زالنىڭ ئالدىغا كىلپ ئىختىيارسىز قەدەملىرمنى ئاستىلتىپ ئۈزۈممۇ تۇيمىغان ھالدا  زالغا بىر پەس قاراپ ئىپپار قىزنىڭ بالسىنىڭ قۇلىنى چىڭ تۇتۇپ ئالدىمدىن يىنىك قەدەم ئالغىنچە ئۈتۈپ كىتۋاتقان ھالتىنى ئەسلىەپ ھاڭۋاقتىلارچە  تۇرۇپ قاپتىمەن كەينىمدىن ئاڭلانغان ئاياق تىۋشىدىن ئىسىمنى يىغىپ ئىشخانامغا كىردىم،بۈگۈن تۈرمىمىزدە چوڭ تازلىق ئۇرۇنلاشتۇرۇلدى،ھازى ئەتىياز پەسلى بولغاچقا تۈرمىمىزنىڭ تازلىق ئىشلىرى بەكمۇ چىڭ تۇتلىدۇ،ئۇنىڭ ئۈستىگە تۈنگۈن كۈرۈشتۈرۈش كۈنى بولغاچقا تېخمۇ شۇنداق،سەللا دىققەت قىلمىساق يۇقۇملۇق كىسەللەر ئاسان تارقىلدۇ،ئۇرۇنلاشتۇرۇشنى بىر قۇر ئاڭلىغاندىن كىيىن جىنايەتچىلەر تۇردىغان ئىگىز قورشاۋ تاملىق تۈرمە ھويلىسغا كىردىم،ئادەتتە كۈندە باشقا ئۇرۇنلاشتۇرۇش بولمىغان ئەھۋالدا  جىىنايەتچىلەرنىڭ يوتقان كۆرپىلىرنىڭ رەتلىك يىغلغان يىغىلمىغانلىقى،ياتقىنىڭ تازلىقى ،ئۈزنىڭ شەخسى تازلىقدىن تارتىپ چاچ ساقالىرنىڭ پاكىز ئېلىنغىنغىچە تەكشۈرۈش مىنىڭ ۋەزىپەم ئىدى،بۈگۈن ئالاھىدە چوڭ تازلىق بولغاچقا مىنىڭ مەسئۇلىيتىمنىڭ دائىرسىمۇ كىڭەيدى،ئۇلارنىڭ قاچا قۇچىلىرنى دىزنفىكسىيەلەش كىرەك،ئۇلار تۈنگۈن ئائىلىسدىكلەر ئەپكەلگەن يىمەكلىكلەرنى يىدى،ۋاقتىدا دىزنفىكسىيەلىمسە يۇقۇملۇق كىسەلنىڭ يىمەكلىك ئارقىل يۇقشنى كەلتۈرۈپ چىقشى مۇمكىن،ياتاقنىڭ ئاستىنى خۇددى يالاپ پارقىراتقاندەك ھالەتكە ئەكلىش ئۈچۈن كۈڭۈل قۇيۇپ سۈرتۈپ،يۇيۇش تەلەپ.قىلنىدۇ،شامالداش ئۈينىڭ تېمدىن  تارتىپ يۇيۇپ تۇرمۇش بۇيۇملىرنى رەتلىك يىغۇشتۇرمىسا تېخى بولمايدۇ.

    مەن مەسئۇل  جىنايەتچىلەرنىڭ ياتقىغا كىرىپ ئۇلارغا بىر تۈرلۈك ۋەزىپلەرنى  ئۇرۇنلاشتۇرۇپ بولغاندىن كىيىن،سىرتقا ماڭدىم ،ئىشىك تۈۋدە تۇرغان ،چاچلىرى چۈشۈرۋىتلگەن جىنايەتچىلەر كىيمى كىيگەن ،تاقىر باش مۇھەممەتنى كۈرشۈم بىلەنلا ئۇنىڭ چىرايلىق ئايالى ئىپپار قىز يەنە بىر قېتىم پۈتۈن نىرۋىلىرمنى قالايمىقان قىلدى.ياتاقتىن يېنىپ چىقىپ كارىدوردا سىرتقا قارغاچ خىيال بىلەن ماڭدىم،كەينىمدىن بىرنىڭ چاقىرغان ئاۋازى ئاڭلاندى،

-دۈيجاڭ،-ئارقامغا قاردىم،مۇھەممەت كەينىمدىلا ماڭا  قاراپ تۇراتتى،ئۇنىڭ ئايالىنى خىيال قىلۋاتقىنمدا ئۇنىڭ مىنى چاقىرىپ خىيالمنى بۇزۋىتشى بەكلا ئاچىقمنى كەلتۈردى،بىر قەۋەت بۈرە  تىرسنى كىيىپ ئۇنىڭغا- نىمە دەيسەن؟بىر ئىشىڭ بارمىتى،-دىدىم يۇقرى ئاۋازدا،ئۇ سەل تەمتىرەپ تۇرۇپ ئالدىمغىلا كەلدى،دۇدۇقلىغىنچە،

-بۇ،بۇ ..خەتنى ئۈيۈمگە ئەۋەتمەكچىدىم،سىلنى ئاۋارە قىلدىغان بولدۇم،خاپا بولماي،ئەۋەرتىپ بەرگەن بولسىلا...-،ئۇ بېشىنى تۈۋەن قىلپ ئۈتۈنگەن قىياپەتتە  لىتاپقا  سېلنىپ ئاغزى يەملەنمىگەن بىر پارچە خەتنى ماڭا ئۇزاتتى،خەتنى قۇلۇمغا  ئېلىپ ئادرىسقا قاردىم ئادىرسى تۇلۇق بولمىسا قايتۇرلىدۇ ئەلۋەتتە،ئادىرس كەينىدىكى ئىپپار قىزغا دىگەن خەتلەرنى كۈرۈپ ئۈزۈممۇ تۇيمىغان ھالدا،

قىزلارنىڭ سېنى ياخشى كۆرمەسلىكىدىكى بىرىنچى سەۋەپ شۇكى ئادەملەر ئارىسىدا دەرىجە سېلىشتۇرمىسى پەيدا قىلغىنىڭدۇر .

يۇقارقى پارچە مەيلى توقۇلما ۋەقەلىك بولسۇن ۋە ياكى راست ئىشلار بولسۇن . بىر مىللەتنىڭ پىسخىك بۇزۇلىشىغا سەۋەپ بولىدىغان ھەم كومپارتىيە ساقچىلىرىغا بولغان خاتا چۈشەنچە پەيدا قىلىدىغان بۇ تېمىنى ئۆچۈرۈشنى تەۋسىيە قىلىمەن . كومپارتىيىمىز قۇرۇلغانلىقىنىڭ 90 يىللىق خاتىرە يىلىدا بۇنداق تېمىلارنىڭ مىسرانىمدا پەيدا بولۇشى ئانچە ياخشى ئاقىۋەت بىلەن نەتىجىلەنمەيدۇ . ھەممىلا جايدا كومپارتىيەنىڭ مىھىربانلىقى ۋە خەلققە بولغان بولغان غەمخورلۇقى سۆزلىنۋاتقان مۇشۇ پەيتتە بۇ تېمىنىڭ باش كۆتۈرۈپ چىقىشى تازا ياخشى ئىش ئەمەس .

                                                     ئەسسالامۇ -ئەلەيكۇم

مىھرىبان ئايالىم ئىپپار قىز،ئوغلۇم ئالىمجان،كۈيۈمچان ئانا قانداق ئەھۋالىڭلار تىنڭلار سالامەت،كۈنلىرىڭلار خۇشال خۇرام ئۈتۋاتامدۇ ،مەن اللادىن ھەمىڭلارغا سالامەتلىك،خاتىرجەملىك تىلەيمەن،مەنمۇ كادىر ساقچىلارنىڭ غەمخورلىقى بىلەن  ناھايتى ياخشى  تۇردۇم،مەندىن ئەنسىرمەڭلار!

  ئىپپار قىز سەن بەلكىم تۈنگۈن كۈرۈشكەن تۇرساق بۈگۈن يەنە  خەت يازغىنمدىن ھەيران قېلۋاتقانسەن،بىلەمسەن،ھەر دئىم سىنىڭ بىلەن كۈرشۈشتۈن بۇرۇن نۇرغۇن گەپلەرنى ساڭا يۈز تۇرانە دىيىشنى ئويلايمەن،لىكىن ۋاقىتنىڭ قىسقىلقى بىلەن ساڭا بۇ گەپلىرمنى دىيشكە پۇرسەت بولمايدۇ،جېنىم ئىپپار مىنىڭ سىنى قانچىلىك سېغىنغىنمنى سەن تەسەۋۇر قىلاليسەن،مەن سىنى بەكمۇ سېغىندىم،تۈنگۈن كۈرشۈپ بۇ ەەپلىرمنى دىيەلگەن بولساممۇ لىكىن يەنە بەزى گەپلىرمنى دىيشكە پۇرسەت بولمىدى،سەن ھەر قېتىم كەلگىنڭدە،ئاشۇ چىرايلىق كۆزلىرڭدىن ياش تۈكۈپ مەن بىلەن تىنچلىق سۇرغان چېغىڭدا مىنىڭ دەيدىغان گەپلىرمنىڭ ئورنىغا ساڭا تەسەلى بىرىش،سىنى ئالداپ سالداپ كۆڭلۈڭنى ياساش بىلەن كۈرشۈش ۋاقتىمىز تۇشۇپ قالدۇ،ياتقىمغا كىرگەندىن كىيىن ساڭا دىمەكچى بۇلۇپ دىيەلمىگەن گەپلىرىم ئىسمگە كىلپ مىنىڭ يۈركۈم سىقلىدۇ.

  ئىپپار سەن ھەر قېتىم  مىنى يوقلاپ كەلگىنڭدە ماڭا نۇرغۇن   نەرسىلەرنى كۈترۈپ كىلسەن،بىلمەن ئائىلىمىزنىڭ شارائىتىنى ،مۇشۇ نەرسلەرنى ئېلىپ كىلش ئۈچۈن قانچىلىك جاپا تارتىپ خەقنىڭ  ئالدىدا ئىشلەپ ،پۇل تاپدىغىنڭنى،ئاشخانىدا ئىشلەۋاتمەن دىگەن ئىدىڭ،ئىشلىرىڭ چۇقۇم جاپالىق،خەققە بىر قاچا تاماقنى ياراتقۇزۇپ،مۇلازىمىتنى ياخشى قىلش شەھەرنىڭ شارائىتدا تۇلمۇ تەس،بىلمەن بۇ جاپالارنىڭ ھەمىسى مەن ئۈچۈن مەن سىنى قىيىن كۈنگە قويدۇم،مەن ساڭا يۈز كىلەلمەيمەن،قانچە قىلساممۇ سىنىڭ ماڭا قىلغانلىرىڭنى ياندۇرۇپ بۇلالمايمەن،سىنى جاپا تارتىپ قالمىسۇن دەپ قانچە قېتىم مىنى يوقلاپ كەلمە دىسەم،سەن بەلكىم باشقىچە چۈشەنگەن بۇلشۇڭ مۇمكىن،مەن سىنى ماڭا نەرسە كىرەك ئالدىغان پۇللىرنى ئۈزگە ئۆيگە ئىشلتەمدىكىن دەپ شۇنداق دىگەن ئىدىم،مانا ۋاقتىممۇ تۇشۇشقا نەچچە ئاي ۋاقىت قالدى كۆڭلۈمدىكى گەپلىرىمنى ساڭا دەۋالمىغچە  كۆڭلۈم ئارامىغا چۈشمەيدىغان ئوخشايدۇ.

   ئىپپار مەن جەميەتىكى ۋاقتىمدا ساڭا تۇلمۇ قاتتىق قوللۇق قىلدىم،بىرەر قېتىم ساڭا ئىشنىپ باقىنمنى،سىنىڭ ماڭا بولغان مىھرىڭنىڭ ھەقىلىقنى ئىتراپ قىلغان بەندە ئەمەسمەن،داۋاملىق چىرايلىقتا ۋاپا يوق دىگەننى دەستەك قىلپ سىنى ئۇرۇنسىز كۈنلىدىم،سىنىڭ مىنى تاشلاپ باشقا بىرى بىلەن كىتىپ قېلشىڭدىن ئەنسىرەپ سىنى داۋاملىق نازارەت قىلدىم، مۇشۇ كۈنلەمچىلكىمنىڭ دەستىدىن سىنى قانچە قېتىم ئۇرغۇنۇم،چىرايلىق شۇ كۆزلىرىڭنى قانچە قېتىم ياشقا تولدۇرغۇنۇم ئىسىمدە يوق،مەن سىنى ھەقىقەتەن قىينىدىم،سەن تىرىك تۇرۇپ دۇزاق ئازابىتنى مىنىڭ قۇلۇمدا كۆردۈڭ،ئاشۇ كۈنلەمچىلكىمنىڭ دەستىدىن مۇشۇ قوشنىمىزنىڭ تېخى توي قىلمىغان ئوغلىدىن سىنى كۈنلەپ قېلىپ  قەستەن جىدەل چقىقرىپ ئۇنى ئۇرۇپ يارلاندۇرۇپ قويدۇم،شۇنىڭ بىلەن بۇ يەرگە  كىلپ قالدىم،سىنى جاپاغا قويدۇم،ئۈزۈم بۇ يەردە يەيدىغىنىم كىيدىغنىم بار سىلەرچۇ؟ئوغلۇم بىلەن ھەقىقەتەن قىينالدىڭ،شۇنداق بولسىمۇ ماڭا چاندۇرمۇدۇڭ،مىنى يوقلاپ تۇردۇڭ،مەن دەسلەپ بۇ يەرگە كىرىپ قالغان كۈنلۈرمىدە سىنى مىنى تاشلىۋىتدۇ،ئەمدى تۇرمۇشنى قايتىدىن ئۇرۇنلاشتۇردۇ،دىگەنلەرنى ئويلىغان ئىدىم،سىنىڭ مىنى تاشلىۋەتمەي ھازىرغا قەدەر ھالىمدىن خەۋەر ئېلشىڭغا ئىشەنچىم يوق ئىدى،ساڭا يەنە بۇرۇنقىدەك ئىشەنمەسلىك نەزرىم بىلەن قارايتىم،لىكىن سەن مىنى تاشلىمدىڭ،ھالىمدىن خەۋەر ئالدىڭ،ئاشۇ كىسەلچان ئانامنىڭ ھالدىن خەۋەر ئالدىڭ،سەن بولمىغان بولساڭ بىچچارە ئانام قەيەرلەردە سەرسان بۇلار ئىدى،ئوغلۇم ئالمجانچۇ،ئۇمۇ دادسىزلىقنىڭ دەردىنىلا تارتىپ قالماي ئانسىزلىقنىڭ دەردىنىمۇ تارتار  ئىدى،جېنىم ئىپپار سەن ھەقىقەتەن پەرشتە،ماڭا اللا ئىگەم ماڭا نىسىپ قىلغان پەرشتە،مەن ساڭا قانچىلىك قىلساممۇ قىلغانلىرىڭنى قايتۇرۇپ بۇلالمايمەن،مەن خاتا قىلغىنمغا،سىنى خاتا چۈشەنگەنلىرمگە  پۇشايمان قىلدىم،چۇڭقۇر ھىس قىلدىم،مىنى كەچۈر،مەن چىققاندىن كىيىن سىنى ھەرگىز جاپا تارتقۇزمايمەن،ھالىڭدىن خەۋەر ئالمەن،بىزنىڭ يەنە  بەختلىك مىنۇتلىرمىزغا ئاز ۋاقىت قالدى بەرداشلىق بەرگىن.اللاغا ئامانەت،ئانامغا مەندىن سالام ئېيىت،مىنىڭ ئورنۇمدا ئوغلۇمىزنىڭ پىشانىسگە سۈيۈپ قويغىن،سىنىڭ ئاخىرقى قېتىم كەچۈرۋىتشىڭنى تىلەپ،------

                2008-يىلى 3-ئاينىڭ -20-كۈنى





خەتنى يىش يېنىشلاپ ئۇقدۇم،ئىپپار قىزغا بەكمۇ ئىچىم ئاغرىدى،سەتنىڭ ھالى چىرايلىقنىڭ دەردى دىگەندەك ئۇنىڭ دەردى يەتكىچە ئىكەنلىكنى ھىس قىلپ قالدىم،مۇشۇنداق ئۈزنىڭ سايسىدىنمۇ كۈنلەيدىغان بىر ئۆكتەمنىڭ ئايالى بوپ قالغانلىقىغا ئىچىم ئاچچىق بولدى،ئۇنىڭغا بولغان ھىسداشلىقىم قوزغالدى،ئۇنىڭ كەڭ قۇرساق،ئەپەۇچان،غەيرەتلىك،مىھرىبان مىجەزى  مىنى ئۇنىڭ چىرايلىق كۆزلىردىنمۇ بەكرەك  قىزىقتۇردى،ئۇنىڭغا بولغان ھۆرمىتىم قوزغالدى،ھازىرقى جەمىيەتتە مۇنداق ۋاپادار ئاق كۈڭۈل ئاياللارنىڭ سانى قانچىلىك دەيسىز،ئۇنىڭ قۇربان بىرىش رۇھى بەكمۇ كەم ئۇچرايدۇ،مىنڭچىمۇ مۇھەببەت بىر بىرگە بولغان ساداقەت،مىھىر ۋاپا،ۋە قۇربان بىرش ئارقىلق ئىپادىلنىشى كىرەك،مىنىڭ كۆڭلۈمدىكى قىزنىڭمۇ دەل ئىپپار قىزدەك مۇلايىم ئاق كۈڭۈل،مىھرىبان ۋاپادار،يولدىشى بىلەن جاپادىمۇ راھەتتىمۇ بىللە بۇلدىغان،قۇربان بىرىش رۇيىغا ئىگە قىزلار بۇلشى كىرەكلىكىنى ھىس قىلدىم،خەتنى لىتاپقا سېلىپ ئېغزىنى ھىملەپ ئەتتىم،خەت چەك ئىشخانسىغا بىرىپ قۇيۇش ئۈچۈن خەت چەك ئىشخانسىغا ماڭدىم،ئىشخاندا ئادەم يوق،خەتنى يانچۇقۇمغا سېلىپ ئىشخانامغا قايتىپ كىردىم،دەرىزىدىن سىرتقا قاراپ  خيالغا پاتتىم،دەرىزە ئالىدىكى يېڭىلا بىخ چىقارغان سۈگەت نوتىلىرى ،ئەتىيازنىڭ مەيىن شامالىرىدا  ھەر تەرەپكە چايقىلاتتى،مۇشۇ ھالەتتە قانچىلك تۇرغۇنۇمنى بىلمەيمەن،خىيالمىغا ئىپپار قىزنىڭ شۇ چىرايلىق تۇرقى ،كەپسىز جاھىل ئوغلى كىرۋالغان ئىدى،چايقىلۋاتقان سۈگەت نوتىلىرى بىلەن  تەڭ  يۈركۈم چايقالغاندەك،ئىچ ئىچىمدىن بىر سادا ...ئىپار قىز..ئىپار قىز....دىگەن سادا چىقۋاتقاندەك قىلاتتى،پۈتۈن ھىس تۇيغۇم ،خىيال كەپتىرىم ئۇنىڭ بېشدا ئەگىپ يۈرەتتى،بىر چاغدا قۇرسۇقۇمنىڭ ئېچىپ  كوركىرغان ئاۋازى  بىلەن ئىسىمنى يىغىپ  سائىتىمگە قاردىم،سائەت 2گە ئازلا قالغان بۇلۇپ ئالىقاچان چۈشلۈك تاماق ۋاقتىمۇ ئۈتۈپ كەتكەن ئىدى،ئىشخانامدىكلەر تاماققا چىقىپ كەتكەن بولسا كىرەك،مەنلا يالغۇز قالغان ئىدىم،ئۇلارنىڭ مىنى چاقىرۋالمايلا تاماققا چىقىپ كەتكىنگە ئاچىقىم كىلپ ئۆزئۈزۈمگە غوتۇلدۇغۇنمچە كادىرلار ئاشخانسىغا تاماققا ماڭدىم.

       ئاشخاندا ئالىقاچان تاماق تارقىتلىپ بۇلۇنغان بۇلۇپ ،ئاشپەزلەر ئاشخاننىنى يۇغۇشتۇرۋاتقان ئىكەن،ئاشپەز ئابلەتكام مىنى كۈرۈپ خجل بولغان قىياپەتتە،مەندىن كەچرۈم سۇراپ،تاماقنىڭ تۈگەپ كەتكىننى چىرايلىق گەپلەر بىلەن چۈشەندۈرگىلى تۇردى،چىرايلىق گەپكە قۇرساق تۇيامدۇ،ئۇنىڭ چۈشەندۈرشلىرىگە قۇلقۇمنى يۇپرۇپ،سىرتقا ماڭدىم،سىرىتتىن بىرەر نەرسە تېپىپ يىمەكچى بولدۇم،بۇ تۈرمىمىزنىڭ يېنىدا تۈزۈكرەك ئاشخانمۇ يوق،بىر ناۋايخانلا بار،ناۋاي ئانچە مۇنچە سامسا يېقىپ قالاتتى،بار  ئۈمۈدىمنى ئۇنىڭغا باغلاپ سىرتقا ماڭدىم،تۇيۇقسىز چېلىنغان  يىغلىش كانىيى بۇ ئۈمۈدىمنى يوققا چىقاردى،دەرۋازنىڭ يېنىغا بارمايلا ئارقامغا ياندىم،قۇرسۇقۇم ئېچىپ كوركىراپ ئارام بەرمەيتى،ئىشخانغا كىرىپ ،تۈرمە باشلىقمىزنىڭ ئەتىياز پەسىلىك خىزمەت ئۇرۇنلاشتۇرشنى ئاڭلاپ،چوڭ تازلىقنى ئۆتكۈزۋېلىش ئۈچۈن تۈۈرمە رايۇنغا كىردىم،قۇرسىقىم ئېچىپ بەكلا بىئارام بولدۇم،كىچكىمدىن قۇرساق ئاچلىقىغا چىديالمايدىغان كىسلىم قوزغۇلۇپ قالدى.كۈزۈمگە ھەممە يەردە يەيدىغان نەرسە تۇرغاندەكلا كۈرنەتتى،پەقەت چىديالمىدىم.قۇرساق ئاچلىق ئازابى مىنى قىينىغلى تۇردى.

     مەھبۇسلار ياتقىغا كىرىپ بىر قۇر كۆز يۈگۈرتۈپ تازلىق تەكشۈرەشكە باشلىدىم،شامالداش ئۈينى تەكشۈرۋاتقىنمدا ،مەھبۇسلارنىڭ تۇرمۇش لازمەتلىكلىرنى قۇيدىغان ئىشكاپ ئۈستىدە بىر سۇلياۋ خالتا ئىچدە گۆش ناندەك بىر نەرسە كۈرۈنگەندەك قىلدى،ئالدىراپ قۇلۇمغا ئېلىپ قاردىم ،قۇرسىقىڭىز ئېچىپ كەتسە كۈزىڭىز ئۆتكۈرلىشىپ قالدىغان ئوخشايدۇ،راسلا  خالتىدا قىزارتىپ پىشرۇلغان گۆش نان تۇراتتى،كىچكىمدىن ئانامنىڭ قىزارتىپ پىشۇرغان گۆش نېنىغا بەكمۇ ئامراق ئىدىم،بۇ  يەرگە ئىشقا چۈشكەندىن بىرى گۆش نان يەپ باقمىغىلى خىلى ئۇزۇن ۋاقىت بولغاچقىمۇ ياكى بۇ قۇرساقنىڭ ئاچلىقمۇ بۇ گۆش نان كۈزۈمگە شۇ تاپ گۈل بۇلۇپ كۈرۈندى.

-دۈيجاڭ بۇ ،مىنىڭ نەرسەم ئىدى،ئۆيدىكلەر ئەكەپتىكەن،باياتىن يۇغۇشتۇرمەن دەپ ئۇنتۇپ قاپتىمەن،مىنى  كەچرۈڭ.....،-يېنىمدىلا مۇھەممەت،ئاپسىنىڭ ئەيپلىشدىن قورىققان كىچىك بالدەك يەرگە قاراپ مەندىن كەچرۈم سۇراۋاتاتتى،ئۇنىڭ باشقا گەپلىرى قۇلقۇمغىمۇ كىرمىدى ،سۇلياۋ خالتىنى كۈترۈپلا سىرتقا ماڭدىم،ئۇ مىنى ئاچىققى كىلپ بۇ نەرسلەرنى تاشلىۋەتكىلى ماڭدى دىدى بولغاي ئارتۇق گەپ قىلماي ئىزدىلا قالدى،دىجۇرنىلىك ئىشخانسىغا كىرىپ خالتىنىڭ ئېغزىنى ئاچتىم دە- نەچچە كۈن ئاچ  قالغاندەك ئاچ كۆزلۈك بىلەن گۆش ناننى ئىككى يالماپلا يەۋەتتىم،قۇرساقمۇ تويدى،بۇ گۆش نان ھەقىقەتەن ئوخشىغان ئىدى،بەلكىم ئىپپار قىزنىڭ پىشۇرغىندىن بولسا كىرەك،شۇنچىلىك تېتىپ كەتتى يەنە نەچچىسى بولسىمۇ يەۋەتكىدەك دەرجىدە ئىدىم،ئۇنىڭ چىرايلىق ھەم تاماققا ئۇستا ئىكەنلىكى ماڭا يەنە بىر قېتىم چۇڭقۇر،تەسىر قىلدى.

تازلىق ئىشلىرنى ئۆتكۈزۋېلىپ بۇلۇپ بالدۇرلا ئىشتىن چۈشۈپ ياتقىمغا يېنىپ كىردىم،ھېلقى كونا تىلۋىزۇرنى ئاچتىم،تۈگمەس ئىلانلاردىن بىزار بۇلۇپ  چاپنىمنى سېلىپ كارۋات يېنىدىكى ئىلغۇچقا ئاستىم.يەرگە بىر نەرسە چۈشكەندەك قىلدى،بۇ مۇھەممەتنىڭ ئىپپار قىزغا يازغان خىتى ئىدى،بايا خەت چەك ئىشخانسىغا بىرىپ قۇيمەن دەپ ئۇنتقىنىم ئىسمگە كەلدى،خەتنى ئۈستەل ئۈستىگە  تاشلاپ قۇيۇپ يەنە شۇ ئىپپار قىزنى ئەسلەشكە باشلىدىم...

يەنە كاللام بۈلۈندى،پۈتۈن خىيالىم ئۇنىڭدىلا قالدى،ئۇنىڭ شۇ ئوتلۇق كۆزلىرى كۆز ئالدىمدىلا،مەن
  ئۇنى ياختۇرۇپ قالغاندىمەنمۇ؟ياق! مۇمكىن ئەمەس مەن دىگەن تۈرمە  ساقچىسى ئۇ بولسا مەھبۇسنىڭ ئايالى..
ئەتىيازلىق تېرىلغۇ ۋاقتى بولغاچقا،كىۋەز  تېرىيدىغان يەرلەرنىڭ ئەخلىتنى   تىريدىغان،تۈزلەيدىغان،تەييارلىق خىزمىتى جىددى ئىشلەنمەكتە،مىنڭمۇ ۋەزىپەم ئېغىرلاپ   زىركىشىلك،بۇ كۈنلىرىمگە ھەمدەم بولدى،ئېغىر  جىسمانى ئەمگەكنى مەھبۇبسلار قىلسا سىز   نىمگە ھېرىپ قالسىز دىدڭىزما؟-ھەي شۇ دەپ كەلسەم  يەنە شۇ ئۇلار ئەمگەك قىلدۇ بىز بولساق ئۇلارغا  نازارەتچىلىك قىلمىز،سەل  بىخەستىلىك قىساق قاچدىغانلارمۇ يوق ئەمەس شۇنداق  بوپلا قالسا ئىش چاتاق،بىزنىڭ مەھبۇسلار بىلەن   ئوخشاش بوپ  قېلشىمىزدا گەپ يوق،مەسئۇلىيتىمىزنىڭ ئېغىرلىقدىن ئاسان چارچاپ قالدىغان بولساق كىرەك،چۈشلۈك  تاماقنىمۇ ئېتىز بېشىدىلا  مەھبۇسلار بىلەن بىرگە يەيمىز،قىززىق بىر پىيالە  چاينىڭ قەدىرى تۇلمۇ ئۈتلىپ كىتدۇ،ئېتىز بېشىدا  يىگەن تاماق ئىسسىق  باسمايۋاتقان،قارسىڭىز قۇرسىقىڭىزنىڭ بىر يىرى توق  بىر يىرى ئاچ،ئەتراپىڭىز كەڭرى كەتكەن كىۋەز  ئېتىزى،يىراقلاردا پەلەككە بوي تالاشقان كۈجۈم كۈجۈم توغراق،زۇمچەك زۇمچەك يۇلغۇنلارمۇ ئانچە مۇنچە كۈزىڭىزگگە چېلقىپمۇ قالدۇ،ئېرىق بۇيدىكى قاغىلەك جىگدىلەرنىڭ شېخىغا قۇنۋالغان قارا  قاغلار،قاقىلدىشىپ بۇ ئادەم تۇرماس چۆلىكتىكى ھاياتلىقنىڭ بىشارتىنى بىرپمۇ  قۇيشىدۇ،مەھبۇبسلار بولسا گەز باغلاپ كەتكەن لەۋلىرنى يالشىپ قۇيۇپ ئېتىز ئىشلىرى بىلەن مەشخۇل بۇلدۇ  .
ئالدىمدىلا دۈمچىيىپ سۇلياۋ يۇپۇقلارنىڭ پارچىلىرنى  تىرۋاتقان مۇھەممەتنى كۈرۈپ نىمە  ئۈچۈندۇر ئۇنىڭ چىرايلىق خۇتۇنى ئىپپار قىزنى ئەسلەپ قالدىم،ئۇ نىمە ئىش  قىلۋاتدىكنە؟ئۇنىڭ كەپسىز جاھىل ئوغلى ئۇنى يەنە قاقشىتىپ يۈرەمدىغاندۇر،ئايالى بولسا نىمدىگەن ياخشى ،ئىسسىقنە چاي،ئوخشىغان تاماق....بۇلارنى ئويلاۋېتىپ قايسى كۈنى يىگەن گۆش نان كۆز ئالدىمغا كىلپ ئاغزىمغا سېرىق سۇ يىغلىپ كەتتى،مۇھەممەتنىڭ  شۇنداق چىرايلىق ،تاماققا ئۇستا خۇتۇننى مىنڭ   كاللامنى ئېلشتۇرۇپلا ۋەتتى،تۇرۇپلا مۇھەممەتكە ، ھەسەتخورلۇق ۋە كۈنلەش ئارلاشقان نەزەردە قاراپ قويدۇم،ئىچمنى بىر نەرسە تاتلىغاندەك بىئارام  بۇلاتىم،مۇھەممەت كۈزۈمگە مىنڭ رەقىبىمدەكلا   كۈرنەتتى،ماڭا كۈزۈمگە سانچىلغان مىختەك  تۇيلاتتى.قارغانسىرى ئۇنىڭغا  ئۆچمەنلىكىم تۇتۇپمۇ قالدىغان بولدى،بەزىدە ئۇ  خۇددى مىنڭ ئالدىغان قىزمنى ئېلۋالغاندەكلا...توۋا  مەن نىمە بولدۇم،ئۇ
دىگەن ئىپپار قىزنىڭ يولدىشى تۇرسا.بالسى تۇرسا،  مەن كىم،ئۇ كىم،مەن ساقچى ئۇ مەھبۇسنىڭ ئايالى ،ئۇ نىمگە ھىساپ،مەن نىمگە ھىساپ.....ئۈزۈمگە مۇشۇنداق ئايقى يوق سۇئاللارنى قۇيۇپ كۈننى كەچ قىلدىغانمۇ بولدۇم.
ئىپپار قىز ھىچ خىيالمدىن چىقدىغاندەك  ئەمەس،ئاخشىمى ئۇيقۇ يوق،كۈندىزى ئۇنىڭ يولدىشىنى  كۆرسەملا ئۇ يەنە چاقماق   تىزلىكدە پۈتۈن ۋۇجۇدۇمنى ،خىيالىرمنى ئۈزگە   تارتىدۇ،مەن بارغانسىرى ئۇنى سېغىندىغان ئەسلەيدىغان بولدۇم،ئۇ ئىسمگە كەپ
  قالغاندا تاماقمۇ يىگۈم كەلمەيتى،بۇنداق چاغلاردا مۇشۇ يەردىكى مەھبۇسلارنىڭ ئايالىنى باللىرنى  قانچىلىك سېغىندىغىننى
چۈشەنگەندەكمۇ بوپ قالاتىم،مۇشۇ كۈنلەردەمۇھەببەت  دىگەن شۇ بولسا كىرەك،يوق چاغدا ئەسلەيدىغان كۆرگىسى   كىلدىغان،دەپمۇ قالدىم،ئىشقىلپ مىنىڭ ئۇخلاۋاتقان  ھىس تۇيغۇللۇرۇم ئىپپار قىزنىڭ ئۇچۇرشى بىلەن ئويغانغاندەكلا،مەندەك تەرى سوغوق،ھاكاۋۇر تۈرمە ساقچىسنىڭ بىر
مەھبۇبسنىڭ خۇتۇننى ياخشى كۈرۈپ قېلشى ،يېڭى زامانىڭ يېڭى چۆچكىدەكلا تۇيلىدىغان مەن مۇشۇ چۆچەكنىڭ ئەر باش قەھرىمانى بوپ قالغاندەك ھىسياتتا ئىدىم،چۈنكى كەمبەغەلنىڭ قىزنى شاھزادىلەرنىڭ ياخشى كۈرۈپ قېلشى ئاللىقاچان كونا چۆچەك ئىدى.مەن  بارغانسىرى ئىپپار قىزنى ياخشى كۈرۈپ قالغىنمنى تەن ئالدىغانمۇ بولدۇم،ئۇنى ھەقىقەتەن ياخشى كۈرۈپ قالغان  ئىدىم،مەن ئۇنى سېغناتتىم،ئۇ ھەر قاچان مىنڭ خىيالىمدا ئىدى.  بۈگۈن ۋەزىپىنى بالدۇرلا تۈگتۈپ ياتقىمغا  بۇرۇنراق قايتىم،ياتقىم نەچچە ۋاقىتىن بىرقى ئالدىراش كۈنلەردە تازلانمىغاچقا بەكمۇ  قالايمىقان بوپ كەتكەن ئىدى،ياتاقنى تازلىۋەتمەكچى  بولدۇم،ئۈستەل ئۈستىدىكى كىتابلار ئارسىدىكى بىر  پارچە خەتنى كۈرۈش   بىلەنلا تەلۋىلگىم تۇتتى،زەردە بىلەن خەتنى يىرتىپ   پارچە پارچە قىلپ ئەخلەت سىۋتىگە تاشلىدىم بۇ مۇھەممەتنىڭ ئىپپار قىزغا  يازغان خىتى ئىدى،قايسى كۈنى ئۇنى خەت چەك بۈلمىگە   بىرمەن دەپ يانچۇقۇمغا سېلىپ قۇيۇپ ئۇنتۇپ قالغان  ئىكەنمەن،مۇھەببەتنىڭ ئادەمنى مۇنچىۋالا  شەخسىيەتچى قىلپ قۇيدىغىننى ئەمدى ھىس قىلماقتا  ئىدىم،ئىپپار قىزنىڭ  مۇھەممەتنىڭ ئايالى ئىكەنلىكنى ئويلىساملا ئاچقىم  كىلپ بېشىم ئاغرىپ كىتەتتى،شۇ سەۋەپلىكمۇ ئىش قىلپ مۇھەممەتكىمۇ ئۆچ   بوپمۇ قالدىم، ئۇنى بىرەر كىچىكىنە ئىشلىرنى  باھانە قىلپ  ئەيپلەپمۇ قۇيدىغانمۇ بولدۇم،ئۇ  ئۈنسىز بېشنى تۈۋەن سېلىپ  تەنقىدىمنى قۇبۇل قىلشقا مەجبۇر. كەچتە  ئىپپار قىزنى خيال قىلپ ئۇخلاپ  قاپتىمەن،ئۇنى بىر تاتلىق چۈشەپمۇ  كىتپتىمەن،سەھەردىلا كەيپىياتىم خىلىلا  كۈترەڭگۈ بوپ قالدىم،ئىشخانغا كىرىپ ئۈستەل ئۈستىدىكى خەت چەكلەرنى كۆردۈم ،كىملەرگە كەلدىكىن دەپ قاراپمۇ  باقتىم،ئويلىغىنمدەكلا  ئىپپار قىزنىڭ مۇھەممەتكە يازغان بىر پارچە  خېتى،بار ئىدى خەتنىڭ ئاغزىنى ئېچىپ خەتكە كۆز  يۈگۈرتۈم،باش قۇردىكى  مۇھەممەت سىزنى سېغىندىم ،دىگەن خەتلەرنى كۈرۈپ
ئاچقىمدا يېرلىپ كىتەيلا دەپ قالدىم ،خەتنى يەنە   يىرتىپ سىۋەتكە تاشلىدىم،مىنىڭ خىيالمدىكى ئاشغىمنىڭ يات بىرگە  ھەتتا يولدىشى بولسىمۇ يات بىر مەندىن باشقا بىر   ئەرگە شۇنداق خەتنى يازغىنغا ھەقىقەتەن چىدىمىغان ئىدىم،مانا مۇشۇنداق قۇرۇق ئۆستەڭگەمىراپ بۇلۇپ كۈرۈرشتۈرۈش كۈنىنىڭ يىتىپ كەلگىنمۇ بىلمەيلا قاپتىمەن،ئەتە يەنە ئايغا 20 ئىكەنلىكنى ئىشخانا تېمىغا ئېسىقلىق كالىندارنىڭ ئۇچۇق  ۋارقىدىنلا بىلدىم،كەچمۇ كىردى  ،ياتقىمغا قايتىپمۇ كىردىم،بۈگۈنكىدەك تاڭنىڭ  تەستە ئاتىدىغننى پەقەت ھىس قىلپ
باقمىغانكەنمەن،تاڭنىڭ ئېتىشنى بالدۇراق ئىپپار قىزنى كۈرۈشنى ئارزۇ قىلاتىم،ئاخىرى تاڭمۇ
ئاتتى،بۈگۈن سالامغا باردىغان ئۇغۇلدەك ،چاچلىرمنى  يۇيۇپ،رەتلىك
تاراپ،يۈزلىرمنى مەخسۇس ئەرلەر مېيدا مايلاپ ئىشقا چىقتىم،چۈنكى بۈگۈن كۈرۈشتۈرۈش كۈنى تۇرسا مىنىڭ خىيالى ئاشغىم ئىپپار   بۇ يەرگە كىلەتتى،مەن خۇددى ئۇ مىنى يوقلاپ كىلدىغاندەك جىدىيلىشەتتىم.
بۈگۈن مەن تىزملاشقا ئەمەس،بەلكى   مەھبۇبۇسلارنىڭ ئائىىسدىكىلىرى بىلەن قىلغان   سۆزلىرنى ئاڭلاشقا   ئۇرۇنلاشتۇرۇلدۇم،قاپقار ئەينەك بىلەن تۇسۇلغان بۇ ئۆيدە ئولتۇرۇپ كامىرا ئۇلانغان كومپىيوتىردىن  زالنىڭ ئىچنى كۈزەتكەچ ئۇلارنىڭ پاراڭلىرغا قۇلاق سالاتىم ،ئەگەر بىرەر مەھبۇس ئائىلىسدىكلەرگە ياكى ئائىلا تاۋاباتلىرى مەھبۇسلارغابىزنىڭ قائىدىمزدىن  ھالقىپ كەتكەن سۆزلەرنى قىلسا شۇ ھامان شۇ مەھبۇسنىڭ تىلفۇننى ئاپتۇماتىك ئىتۋىتىش مىنڭ  ۋەزىپەم ئىدى،مىنىڭ قۇلقۇمدا ئاڭلاش ئەسۋابى بولسىمۇ كۈزۈم كومپىيوتىردىكى  سنئالغۇ ئارقىلق قۇيلۋاتقان سىرتىكلەرنىڭ كۈزتىش خاتىرسىدە ئىدى،ھىچ بولىسا ئاشۇ ئىكران ئارقىلق بولسىمۇ ئىپپارنى كۈرۈشنى  ئۈمۈد قىلاتتىم،لىكىن كەچ كىرگىچە ئۇنىڭ قارسىمۇ كۈرۈنمىدى،ئۇ كەلمىگەن ئىدى  ،تۇلمۇ روھسىز ئىشتىن چۈشتۈم،ياتاققا كىرىپ  تۇرۇسقا قارپ خىيال سۈرشكە باشلىدىم ،ئۇ ئاغرىپ قالغانمىدۇ؟،بىرەر ئىش  بولغانمىدۇ،؟نىمە ئۈچۈن كەلمىگەندۇر؟...تۈگمەس  سۇئاللار مىڭەمدە ئەكىس ئىتەتتى،لىكىن ھەمىسى جاۋابسىزلا.   كۈنلىرىم يەنە بىر خىل رىتىمدا ،مەھبۇسلارنى  ئېتىزغا ئەپچقىمەن،ئەكرىمەن،ئىشلىرنى  ئۆتكۈزۋالمەن،كەچتە ياتاققا كىرسەم يەنە   شۇ ئىپپارنى ئەسلەيمەن،كۈندىزلىرى مۇھەممەتكە  نەپرەت كۈزۈم ئۆچمەنلىكىم بىلەن قارايدىغان
ئىشلارمۇ مىنىڭ زىممەمدە،يەنە  بىر ئايمۇ ئۆتتى،كۈرۈشتۈرۈش كۈنى يىتىپ   كەلدى،ئالدىنقى قېتىمقىدەكلا چۇڭ ئۈمۈد بىلەن ئۇنى  ساقلىدىم،بۈگۈن  مەن  كۈرۈشتۈرۈش زالىنىڭ ئىچدىكى تەرتىپ ساقلاش  خىزمىتگە مەسئۇل بولدۇم،ئەينەك ئىچدە تۇرۇپ  سىرتىكى كىشلەرگە ئۈمۈد بىلەن قارايمەن،ئۇلارنىڭ ئىچدىن ئىپارنى ئىزدەيمەن،ئۇنىڭ   شۇ گۈزەل چىراينى مەن پەقەت بىر قېتىملا كۈرۈش  پۇرسىتگە ئىرشكەن بولساممۇ لىكىن ئۇ مىنڭ قەلبىمگە مۆھۈردەكلا  بېسلىپ ئورناپ كەتكەن ئىدى،ئۇنى قانچىلىك   سېغىنۋاتقىنمنىمۇ تىلمدا  ئىپادىليەلىشىم تۇلمۇ مۈشكۈل،ئىشتىن چۈششىمىزگە   ئاز قالغان ۋاقىت ئىدى،تۇيۇقسىز كۈرۈشتۈرۈش زالىنىڭ تۈرمنىڭ ئىچگە  قارايدىغان دەرزىسى تۈۋدە مۇھەممەتنى كۈرۈپ قالدىم   ئۇ بويننى سۇزۇپ زالغا قاراۋاتاتتى،ئادەتتە  مەھبۇسلارنىڭ ئائىلىسدىكلەر   كەلسە بىز ساقچىلار ئۇلارنى چاقىرىپ كىرمىگەچە بۇ  زالغا قالايمىقان كىرشى مەنئى قىلنغان بولغاچقا
،ئۇ زالغا دەىزىدىن   ماراۋاتقاندەك قىلاتتى،ئۇنىڭ بۇ ھەركىتنى كۈرۈپ   ئەرۋايىم قىرىق گەز ئۇچتى،ئاچىقىمدا ئۇنى دەرىزنىڭ  يېنىدا تۇرما دەپ   ھەيدىۋەتتىم،مىنىڭ كۈنلەمچىلك كىسلىم قوزغالغان    ئىدى،سۈيۈملۈكۈم ئىپپارنى مەن ئۇنىڭدىن بۇرۇن   كۈرشىم كىرەكتەك ھىس  قىلاتىم،مەيلى ئۇ ئۇنىڭ يولدىشى بولسۇن. ئىشتىن چۈشۈشكە بەش مىنۇت قالغاندا ئىپپار  زالدا پەيدا بولدى. ئۇ يالغۇز كەلگەندەك قىلاتتى،   يېنىدا ھېلقى جاھىل ئوغلىمۇ  كۈرۈنمەيتى،ئۇنى كۈرۈپ يۈرەكلىرىم ئويناپ كەتتى،ئۇ  ئەينەك ئىچگە تەشنالىق بىلەن قاراپ بىرنى ئىزدەۋاتاتتى،ئۇ مۇھەممەتنى مىنىڭ رەقىبىم مۇھەممەتنى ئىزدەۋاتاتتى،ئاچقىم   كەلدى،ئۇنڭغا ئۈمۈدلىك كۆزلىرىمنى تىكتىم،ئۇ يەنلا

بىپەرۋا ،بىرسى مۇھەممەتنى

  باشلاپ كىردى.

مۇھەممەت زالغا كىرپلا ماڭا ئەيمىنىپ قاراپ  قۇيۇپ ،بېشىنى تۈۋەن قىلغىنچە كۈرۈشتۈرۈش  كۆزنىكدىكى تىلفۇننى قۇلغا  ئالدى،ئىپپارمۇ تىلفۇننى قۇلغا ئېلىپ ئۇنىڭدىن  ئەھۋال سۇراۋاتاتتى،يەنە بىر قېتىم  ئىپپارنىڭ  پۈتۈن چىراينى كۈرۈش ماڭا نىسىپ   بولغانلىقدىن تۇلمۇ خۇشال ئىدىم،ئۇنىڭغا زەن سېلىپ  قاردىم،ئۇنىڭ ئوتلۇق كۆزلىرى سەل ئولتۇرۇشقان،سۈزۈك چىرايىمۇ مەن  تۇنجى قېتىم كۆرگەن ۋاقىتىكىدىن سەل تاترىپ قالغان  ئىدى،قارماققا كىسەل تارتقان ئادەمدەك  كۈرنەتتى،ئۇنىڭ بۇ ھالىتىنى  كۈرۈپ يۈركۈم ئېچىشتى،ئۇنىڭغا نىمە بولدىكىنە،كەيپىياتىمۇ بەكلا سۇلغۇن،بىر چاغدا مۇھەممەت تىلفۇننى قۇيۇپ سىرتقا  ماڭدى،پەرزىمچە ئىشتىن چۈشۈشكە ئازلا ۋاقىت  قالغاچقا باشقا كادىرلار ئۇلارنىڭ كۈرشۈش ۋاقتىنى   سەل قىسقارتىۋەتكەن بولسا كىرەك،ئۇلارنىڭ شۇنچە قىسقا ۋاقىت  كۈرۈشتۈرۈلگىندىن مەن تۇلمۇ خۇشال بولغان ئىدىم،مىنىڭ ئاشىغىمنىڭ مۇشۇ  مەھبۇس بىلەن  ئۇزۇنغىچە كۈرشىپ ھال ئېيتىشنى خالمايتىم ئەلۋەتتە،ئىپپار كىرىپ كىتۋاتقان مۇھەممەتنىڭ  كەينىدىن بىر ھازا قاراپ تۇرۋەتكەندىن كىيىن،ئاستا قەدەملەر بىلەن  زالدىن   سىرتقا ماڭدى،مەن ئۇنىڭ ئارقىسدىن ھاڭۋاقتىلارچە قاراپ تۇرۇپ قاپتىمەن،ئۇ    مىنىڭ بۇ ھالىتىمدىن بىخەۋەر ئۆز يۇلغا راۋان بولدى.

بۈگۈن قانداقلا بولمىسۇن ئۇنى كۈرۋالغىنمدىن  تۇلمۇ خۇشاللىق ھىس قىلاتىم،ئۇنىڭ  سۇلغۇن چىرايى كۆز ئالدىمغا كەلسە ئۇنىڭدىن ئەنسىرەپ قالدىم،ئۇنىڭغا نىمە ئىش  بولغانلىقنى بەكلا بىلگۈم كەلدى،لىكىن مەن بۇنڭغا ئامال قىلالمايمەن،تۈگمەس  پەرەزلەر بىلەن بۈگۈنمۇ  ئۇخلىغان بىلەن   ئۇخلىمغاننىڭ ئارلىقدا تاڭ ئاتقۇزدۇم،يەنە شو كونا  كەسىپ،كىۋەزلىكنىڭ ئۇتنى ئېلىش،تۈۋنى بۇشتۇش،ئۇچنى ئۈزۈش،خىمىيۋى ئۇغۇت بىرىش بۇلارنىڭ ھەمىسسى مەھبۇبسلارنىڭ قىلدىغان يازلىق   ئەمگىكى،ئەمدى بىرنى تۈگەتسىڭىز يەنە بىرسى  تەييار،بۇلارمۇ تۈگسە كىۋەز تەرمىسە تېخى   بولمايدۇ،ئىش قىلپ بۇ يەردە ئەمەەك  كۈچى بىكارلىق بولغاچقىمكىن بىردەم بىكار  تۇرغۇزغىمىز كەلمەمدىكىن دەيمەن،مۇشۇنداق ئەمگەكلەر،ۋە ئىپپارنى ئەسلەش  بىلەن   بۇلۇپ كۈنىڭ قانداق ئۈتۈپ كەتكىننمۇ بىلمەپتىمەن،مۇھەممەتنىڭ ۋاقتىنىڭ تۇشۇپ   قالغانلىقنمۇ بىلمەي يۈرىۋىرپتىمەن،ئۈزۈمنىڭ  مۇنچىۋالا بىپەرۋا بوپ قالغننى ئەمدى ھىس قىلشىم  ئىدى.
  شۇ كۈنى مۇھەممەتنىڭ قۇيۇپ بىرلىشگە ئىككى كۈن قالغان چاغ  بۇلۇپ تۈرمە باشلىقمىز ماڭا ئۇنىڭ ئائىلىسگە خەۋەر   بىرىپ قۇيشىمنى تاپشۇردى،ئۇنىڭ ئەركىنلىككە چىقىپ   شۇ گۈزەل مەشۇغۇم ئىپپارنى ئىگەلىۋالدىغانلىقنى،.ئىپپار ئەمدى مەڭگۈ   ئۇنىڭ بوپ كىتىدىغننى ئويلىسام يۈركۈمگە پىچاق   تىققاندەك بىئارام بۇلۇشقا باشلىدىم،مۇھەممەتنىڭ  نىمشقىمۇ ئۇزۇنراق كىسىمگە  ھۈكۈم قىلنمىغنغىمۇ ئاچىقىم كەلدى،ئاخرى شەخسى  ھىسياتىمدىن خىزمەتنىڭ ئىھتىياجى يۇقرى ئۇرۇنغا  چىقىپ ئۇنىڭ ئائىلىسگە تىلفۇن قىلدىم،تىلفۇننى ئىپپار ئالغان بولسا كىرەك ئاۋازى تۇلمۇ يېقىملىق ئاڭلاندى،ئۇنىڭ ئاۋازنى  ئاڭلاش بىلەن جىديلىشىپ   تىلىم گەپكە كەلمىگەندەك  بوپ قالدىم،ئۇنىڭ ئىككىنچى قېتىم ۋەي! دىگەن ئاۋازى بىلەن ئىسىمنى
