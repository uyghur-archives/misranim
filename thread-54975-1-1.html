بۇ ماقالىنى يېزىش ئارقىلىق قىسمەن كىشلەرنىڭ زىتىغا تىگىپ قويىدىغىنىمنى بىلىمەن . لىكىن ، بۇ گەپلەرنى دېمىسەم بوغۇلۇپ قالىدىغان ئوخشايمەن . بۇ ماقالىنى شۇنداقلارنىڭ بەكراق كۆرۈشىنى ، يۈزىنىڭ چىمىلدىشىنى ، ۋۇجۇدىدىكى ئەركەكلىكنىڭ قايتا ئويغىنىشىنى ئۈمىد قىلىمەن .
ھەممىمىزگە مەلۇم ، بىر كىشىنىڭ ھالاللاپ ئالغان ئايالى شۇ كىشىگىلا خاس ، ئايال كىشىنىڭ بەدەنلىرىنى كۆرۈش ، سىيلاش ۋە قۇچاغلاش پەقەت شۇ ئەرگىلا خاس بىر ئىشتۇر . بولۇپمۇ بىز ئۇيغۇر ئەركەكلىرى بۇ ئىشقا كەلگەندە تولىمۇ سەزگۈر ھەم نازۇك كېلىمىز . بۇ ئىشقا نىسبەتەن مۇرەسسە قىلىش يول قۇيۇش دىگەنلەر بىز ئۈچۈن ئەرلىك غۇرۇرغا چېتىلدىغان ئەڭ نازۇك مەسىلىنىڭ بىرى ھىسابلىنىدۇ . تارىختىن بۇيان ئۆزىنىڭ ھالال جۈپتى ۋە ئىگىچە – سىڭىللىرىنىڭ نومۇسى ئۈچۈن جېنىنى پىدا ئەتكەن باھادىر سۈپەت ئەركەكلىرىمىزمۇ ئاز ئەمەس .
ئەمما ، دەۋرىنىڭ تەرەققىياتىغا ئەگىشىپ ، سىرتقى مەدەنىيەتلەرنىڭ غىدىقلىشى تەسىرىدە ، ئۆز قەۋمىمىزدا تەرەققىيات بىلەن خۇنۈك مەدەنىيەتنى ، ئىلغارلىق بىلەن زەھەرلىك ئەبجەش ئىدىيەنى پەرق ئىتەلمەيدىغان قىسمەن لەنەتكەردى ئەرلەرنىڭ پەيدا بولۇۋاتقانلىقىدەك بۇ ئەمەلىيەتكىمۇ كۆز يۇمالمايمىز . ئۇلارنىڭ قارىشىچە ئايالىنىڭ كىيىم – كىچىكىگە چەك قۇيۇش ، باشقا ئەرلەر بىلەن بولغان پائالىيەتلەرنى چەكلەش قاتارلىقلار ئۇششاق ئىشلارغا ئارلاشقانلىق ، كونىچە ئېقىم بىلەن مۇتەئەسسىپلىك قلىغانلىق ھىسابلىنىپ ، ئاياللارنى مەيلىگە قۇيۇپ بېرىش ۋە باشقا ئەرلەر بىلەن بولغان پائالىيەتلىرىگە يول قۇيۇش ، دەۋىر تەرەققىياتىغا ماسلاشقانلىق ھىسابلىنىدىكەن . لەنەت شۇنداق ئەرلەرگە !!!
ئالدىنقى كۈنى بىر دوستۇم ئىككىمىز ماشىنىدا پاراڭلىشىپ ئولتۇرغان ئىدۇق . ئالدىمىزدىن بىر جۈپ ئەر-ئايال قولتۇغلاشقىنىچە ئۆتۈپ كەتتى . ئايال كىشى مەيدىلىرى ئۇچۇق دىگۈدەك نىپىز كوپتا كىيىۋالغان بولۇپ ، جىڭبو-جاۋەنلەرنىڭ ھەممىسى مانا مەن دەپ كۆرۈنۈپلا تۇراتتى . كىيىۋالغان يوپكىسىمۇ شۇنچىلىك قىسقا بولۇپ ، ئەۋرىتىنى ئارانلا يېپىپ تۇراتتى . ياندىكى قەرت ئويناۋاتقان باشقا ئەرلەرنىڭ ، ھېلىقى ئايالنىڭ مەيدىسى ۋە يوتىلىرىغا ئاچكۆزلۈك بىلەن تىكىلىپ تۇرۇشقىنىغا كۆزۈم چۈشۈپ قالدى – دە ، دەررۇ ھېلىقى ئەرنىڭ چىرايىغا قارىدىم . توۋا … بۇ ئەر شۇنچىلىك خاتىرجەم ئىدىكى ، ئۆزىنىڭ ئايالىغا شەھۋەت بىلەن ئاچكۆزلەرچە تىكىلگەن كۆزلەر قېتىغا كىرىپ چىقمىغان . ئەكسىچە ، ئايالىدىن پەخىرلەنگەندەك ئىپادە بىلەن كېتىپ باراتتى . توۋا !!! ، – دېدىم دوستۇمغا قاراپ ، – بۇ ئادەممۇ يا ، ئايالىنىڭ كىيىنىشىدىن نومۇس قىلمىغانمىدۇ ؟- دېدىم . دوستۇم بېشىنى چايقاپ قۇيۇپ – ئاداش، سەن ياسىغان تور بەتنى قانچە جىق ئادەم كۆرسە ، شۇنچە خوشال بولغانغا ئوخشاش ، بۇنداق ئەرلەر خوتۇنىغا ئەرلەر قانچە جىق قارىسا ، خوتۇنۇمنىڭ 人气(كۆرۈلۈش قېتىم سانى)سى ئاشتى دەپ خوشال بولىدۇ ، – دەپ كۈلۈپ كەتتى . بۇ گەپنى ئاڭلاپ مەنمۇ كۈلۈپ كەتتىم . خېلى ۋاقىتلارغىچە بۇ گەپ ئىسىمدىن چىقماي شۇنداق ئەرلەرنى كۆرسەملا ئىچىمدە «人气(كۆرۈلۈش قېتىم سانى)غا ئەھمىيەت بېرىدىغان ئەرلەر » دەپ ئويلاپ كۈلۈپ يۈردۈم . بۇرۇن ، كوچىلاردا يېرىم يالىڭاچ كىيىنىپ يۈرگەن ئازغۇن ئاياللارنى كۆرگىنىمدە ، ئىرىنىڭ ئۇلارغا گىپى ئوتمەمدىغاندۇ ، ئەرلىك بولغان خوتۇنمۇ ئۆزىنىڭ كىيىم – كىچىكىگە دىققەت قىلمامدىغاندۇ ؟ دەپ ئويلايتتىم . ھازىر ئۇلارنىڭ ئەرلىرىنىڭ چوقۇم ئۆتۈپ كەتكەن مازلىقىنى تەسەۋۋۇر قىلىدىغان بولدۇم . لەنەت شۇنداق ئەرلەرگە !!!
يېقىندا بىر تويغا بېرىپ قالدىم ، ئىشىم چىقىپ قالغانلىقى سەۋەپلىك سەل كىچىكىپراق بېرىپ قالدىم . تويغا كىرىپ سالام – سائەت قىلىپ بولۇپ ، چاينى بىر ئوتلاپ سەھنىگە قارىسام جۈپ – جۈپ بولۇشۇپ تانسىغا چۈشۈپلا كېتىپتۇ . بىرسىنىڭ ئايالى باشقا بىرسى بىلەن يېقىن قۇچاغلىشىپ تانسا ئوينىغىلى تۇرۇپتۇ . نىمىلەرنى دىيىشتىكىن تاڭ خوشاللىقتىن ئاغزى يۇمۇلمايلا قېلىشىپتۇ . يولدىشىنى بىلەن بايا تويغا كىرىپلا كۆرۈشكەن كىشلەر قاتارىدا كۆرگەن ئىدىم . بۇ نىمە ئىش بولۇپ كەتتى ئەمدى دەپ ئۈستەلگە قارىسام ، ئىرى يېنىدىكى بىرسى بىلەن بەخىرامان پاراڭلىشپ ئولتۇرۇپ كېتىپتۇ . قارىغاندا تانسا ئويناۋاتقان ئەرگە ھاراق نۆۋىتى كەلگەندە تانسىغا قوپۇپ كەتكەن چېغى ، مۇزىكا توختاپ ھەممىسى ئۈستەلگە قايتىپ كېلىپ ، ھېلىقى ئايالىنىڭ ئىرى بولغان ئاداشنىڭ قېشىغا كىلىپ ئولتۇرۇپ ھارىقىنى سوقۇشتۇرۇپ كۆتۈرۋەتتى . ئارلىقتا مۇزىكا باشلىنىپ يەنە بىر ئەر شۇ ئايال تانسغا چۈشۈپ كەتتى . باللىرى بولسا نېرىدا قىيا – چىيا قىلىشىپ ئويناپ يۈرەتتى . ئۇ ئايالنىڭ يولدىشىغا قارىسام يەنىلا ھىچ ئىش بولمىغاندەك بەخىرامان ئولتۇراتتى . كۆڭلۈمدە خېلى ھۆرمىتى بار بۇ كىشى نەزىرىمدىن شۇنداق بىر چۈشۈپ كەتتىكى كۆزۈمگە شۇنچىلىك ئەبگا كۆرۈنۈپ كەتتى . تويدىن يېنىپ يېقىن دوستۇمغا بۇ ئىشنى دەپ بەرسەم ، – بۇنىغۇ قۇيۇپ تۇرە ، ئۆتكەندە بىر سورۇنغا بارسام ، بىر ئاداش مەس بولغاندا ، ھۆرمىتىمنى قىلىپ ئايالىم بىلەن تانسا ئويناپمۇ قويمىدىڭىز دەيدۇ ، ھۆرمىتىڭىنى قىلمىغان بولسام خاپا بولما ، لىكىن مىنىڭ ھۆرمىتىمنى ۋە ئۆزەڭنىڭ ھۆرمىتىنى قىلاي دىسەڭ ئايالىمنى تانسىغا تارتىپ سالما دېدىم ، – دېۋىدى تېخىمۇ ھەيران قالدىم . قانداق تەتۈرلۈك بۇ ؟ خوتونىنى تانسىغا تارتسا ھۆرمىتىنى قىلغان بولارمىش تېخى . مۇشۇنىڭغا ئوخشاش خوتۇنى باشقىلار بىلەن قالايمىقان يېقىنلىشىپ يۈرسە قېتىغا كىرىپ چىقمايدىغان تېخى خوتۇنىنىڭ باشقا ئەرلەر بىلەن تانسا ئوينىشىنى ئارزۇلايدىغان ئەرلەرنى قانداقمۇ ئەركەك دىگىلى بولسۇن ؟؟؟ لەنەت شۇنداق ئەرلەرگە !!!
بىر قانچە يىل بۇرۇن ، ئايالىم بىلەن بىر تويغا باردىم . تاماق تارتىلىپ بولۇپ ساز چېلىندى . ئەتراپتا چوڭلار بولغاچقا تارتىنىپ ئولتۇرۇدۇق . بىر چاغدا قارىسام بىرسى ئايالىمنىڭ ئالدىغا بېرىپ قولىنى تەڭلەپ تۇرۇپ كېتىپتۇ . ئايالىمغا قارىسام ئۇ ماڭا «بۇ ئىشنى تىزراق بىر تەرەپ قىلسىڭىز » دىگەن مەنىدە قاراپ تۇراتتى . ئاچچىقىم يۈز سەكسەنگە چىقىپ ، ئاغزى – بۇرنىنى يالاق قىلىۋىتەي دىگەن ، توينىڭ ھۆرمىتىنى قىلىپ ئۆزۈمنى تۇتۇۋېلىپ بېرىپ ، ھېلىقى ئاداشنى تارتىپ كىلىپ ئۆزى ئولتۇرۋاتقان ئۈستەلگە ئولتۇرغۇزۇپ ، ئۇنىڭغا ۋە سورۇنداشلىرىغا ، تويدىكى ئىرى بار ئاياللارنى كۆرسىتىپ دىققەت قىلىشنى ئېيتىپ قۇيۇپ جايىمغا قايتتىم . يەنە بىر قېتىملىق مۇزىكا چېلىنغاندا قارىسام شۇ ئاداش يەنە ئايالىمغا قول تەڭلەپ تۇرۇپتۇ . چاچراپ قۇپۇپ «كېلە ، سەن بىلەن تانسىنى مەنلا ئويناي» دەپ ياقىسىدىن سۆرىگەن پېتى رېستوراننىڭ سىرتىغا ئاچىقىپ ، قۇلاقتىن قۇلاق تۇۋىگە ئىككى شاپىلاق قۇيۇپ ، تۇمشۇققا مۇشتتىن تۆتنى قويسام بۇرنىدىن ئوقتەك قان كەتتى . ئاندىن «دەيدىغان گىپىڭ بارمۇ» دېسەم ، نومۇس قىلماي «ئۇقماپتىمەن خاپا بولماڭلا» دەيدۇ . « ھېي بايا ساڭا نىمە دېدىم ؟ مۇشت يىمىسەڭ ئۇقمامتىڭ ؟» دەپ شاپىلاق بىلەن يەنە بىرنى سالدىم . شۇ ئەسنادا ئىچىدىكىلەر چىقىپ مېنى تارتىپ ئەكەتتى . تويدىكى چوڭلارنىڭ ئالدىدىمۇ شۇنچىلىك خىجىل بولۇپ كەتتىم . شۇ ئاچچىغىمدا رېستۇرانغا كىرىپ ئولتۇرسام ، بىر رومكا ھاراق سۇنۇپ كەپتۇ ئالدىمغا «خاپا بولماڭلا» دەپ ، «مېنى چۆپچى كۆرۈۋاتامسەن ؟» دىسەم «ياق» دەيدۇ «يەنە تاياق يەۋالماي دېسەڭ يوقىلە» دەپ ۋارقىرۋىدىم ئاستىلا كېتىپ قالدى . بىر چاغدا قارىسام ، دوستلىرىم مەست بولۇپ قاپتىكەن ھېلىقى گۇينى تارتقۇچلاپ كۆزىگە قولىنى تىققىلى تۇرۇپتۇ . چىرايلىق توي بۇزۇلمىسۇن دەپ دوستلىرىمنى ئايرىۋالدىم . ئۇ مازمۇ شۇنچىلىك پۇشايمان قىلىپ كەتتىكى ، ئىككىنچى قېتىم تويغا بېرىپ خەقنىڭ خوتۇنىنى تانسىغا تارتىپ سالماس دەپ ئويلايمەن . بىر دوستۇم خىزمەتداشلىرى بىلەن رېستۇرانغا ئوينىغىلى بېرىپتىكەن . بىر چاغدا بىر خىزمەتدىشى كەپتۇ دوستۇمنىڭ قېشىغا « XXXگۈل بىلەن تانسا ئوينىسام خاپا بولمايدىغانسىز؟» دەپ ، دوستۇم ئۆزىنى تۇتۇۋېلىپ ، «كىمۇ ئۇ ؟» دېسە تەپتارتماي «سىزنىڭ ئايالىڭىز» دىگۈدەك ، دوستۇمىڭ ئاچچىقى كىلىپ «مېنىڭ خوتۇنۇم بىلەن نىمە بولغانغا سەن تانسا ئوينايتتىڭ ؟ ما گەپنى ئىككىنچى ماڭا قىلىپ سالما ئۇقتۇڭمۇ ؟ » دەپ كۆزىگە قولىنى تىقىپ ھەيدىۋېتىپتىكەن . كىيىن كەچلىك بازاردا ئۇچراپ قېلىپ ، تىلىنى چايناپ يەنە شۇ گەپنى تىلغا ئېلىپ «ئۇ چاغدا باشقا مەقسىتىم يوقتى ، خاتا چۈشىنىپ قالدىڭىز» دەپ تىلىنى چاينىۋىدى ، شۇنچە ئادەم بار يەردە ئوبدانلا يىدى تۇمشۇققا . بىزمۇ بىكار تۇرماي بىر ئىككىدىن دەسسەپ قويدۇق بېقىنغا(ھى ھى ھى) . ئەرلىك ئايالغا بۇنداق بىزەڭلىك قىلىش ئادەمنى ئاممىۋى سورۇندا ئاشۇنداق خار قىلىدۇ ، چۈنكى ھەممە ئەركىشى ماز ئەمەستە . لەنەت شۇنداق ئەرلەرگە !!!

خۇلاسە كالام ، بۇنىڭغا ئوخشاش ئۆزىنىڭ ئەرلىكىنى ، ئەرلىك مەجبۇريىتىنى ئۇنتۇپ قالغان ئەرلەر پەقەت قىسمەن ھالدا . ئەمما ئىدىيەدىكى كىسەلنىڭ يۇقۇشى بەدەندىكى كىسەلگە يۇقۇشىدىن نىسبەتەن تىز ھەم كۆلەملىكراق بولىدۇ ، شۇڭا ھەممەيلەننىڭ ئۆزىگە بۇنداق لەنەتكەردى كىسەلنىڭ يۇقۇپ قېلىشىدىن ئاۋۋال ۋاكسىنا ئۇرۇۋېلىشىنى ئۈمىد قىلىمەن .

مىللىتىمىزدە بۇنداق  ئەرلەر باردۇر ، لىكىن بەك ئاز سالماقنى ئىگىلىشى مۇمكىن(بۇلۇپمۇ بىزنىڭ مىللەتتە). توۋا ، خۇتۇنىنى بازارغا سېلىش، ئۇنىڭدىن پەخىرلىنىش، ئۇنىڭغا يات جىنىسلەرنىڭ يېقىنىچىلىق قىلىشىغا يول قۇيۇش بۇ ھەرگىزمۇ ئەركەركلەر قىلدىغان ئىش ئەمەس . ئۇلارمۇ ھەم ئەركەك ئەمەس. ‹‹ئەرلەر ئاياللارنىڭ ھاميسىدۇر››(نىسا سۈرىسى 34 - ئايەت) ئاياللارنى ئاللاھ تائاللا ئەرلەر ئۈچۈن ياراتقان. ئەرلەرنىڭ ئاياللارنى قوغدىشى ، ئۇلارنى ياخشى يولغا باشلىشى ئۇلارنىڭ مەجبۇرىيتى ۋە مەسئۇلىيتى. جەمىيەتنىڭ بۇزۇقچىلىققا تۇلۇپ كىتىشى ھەرگىز ئاياللارنىڭلا مەسئۇلىيتى ئەمەس. چاۋاك بىر قولدىن چىقمايدۇ. ئەرلەرنىڭ مازلىشىپ كىتىشى، ئۇلاردا ئەركەكلىك قاننىڭ سۇسلاپ كىتىشى بۇ نىمە ئۈچۈن؟ بۇ دەل ئىمانسىزلىق، ۋىجداننىڭ يوقلىقى ، ئاياللارنىڭ دىننى ھەقىقى مەنسى بىلەن چۈشەنمەسلىكى. ‹‹مۆمىنلەرگە ئېيتقىنىكى ، نامەھرەملەرگە تىكىلىپ قارمىسۇن، ئەۋرەتلىرىنى ياپسۇن، تاشقى زىننەتلىرىدىن باشقا زىننەتلىرىنى ئاشكارلىمىسۇن. لىچەكلىرى بىلەن كۆكىرەكلىرىنى ياپسۇن. (تاشقى زىننەتلىرىدىن باشقا زىننەتلىرىنى ئەرلىرىدىن، ئاتىللىرىدىن، قېينىئاتىللىرىدىن، ئوغوللىرىدىن ، يا ئەرلىرىنىڭ ئوغوللىرىدىن، يا ئۆز قېرىنداشلىرىدىن، ياقېرىنداشلىرىدنىڭ ئوغوللىرىدىن، يا ھەمشىرلىرىنىڭ ئوغوللىرىدىن ، يا دىنداش ئاياللاردىن، يا قول ئاستىدىكى چۆرىلەردىن، يا خۇتۇنلارغا ئىھتىياچى چۈشمەيدىغان خىزمەتچىلەردىن بالىلاردىن) باشقا كىشلەرگە كۆرسەتمىسۇن. زىننەتلىرى كىشلەرگە بىلدۈرۈش ئۈچۈن ئاياغلىرىنى يەرگە ئۇرمىسۇن. ئى مۆمىنلەر! بەخىتكە ئىرىشىشىڭلار ئۈچۈن ھەممىڭلار ئاللاھقا تەۋبە قىلىڭلار››(نۇر سۈرىسى 31- ئايەت)
ئەركەك دىگەن بىر مىللەتنىڭ ئاتىسى، بىر ئائىلىنىڭ تۈۋرىكى، بىر ئايالنىڭ ھامىيسى، بىر بالىنىڭ دادىسى . ئەرلەر ۋىجدانىنى ئويغۇتىپ، ھاراقتا قاپاق بۇلۇپ قالاي دىگەن كاللىسىنى قاتتىق سىلكىۋىتىپ، ھەۋەسكە- شەھۋەتكە بىرىلىپ كەتكەن غورورىنى سەگىتىپ ئەتىراپىغا  ئۆزىنىڭ قوغدىشىغا مۇھتاج بۇلىۋاتقان ئاياللارغا، قاراڭغۇلۇقتا قېلىۋاتقان ، كەلگۈسىدىن ئۈمۈدلەر قالمايۋاتقان مىللىتىگە، قاراڭغۇلۇق كوچىسىغا كىرىپ كىتىۋاتقان كىينىكى ئەۋلادلىرىمىزغا ئەقىل كۆزى بىلەن ئوبدان زەن سېلىپ باقسۇن. قانچىلىك ئاياللارنىڭ ھەقىقى ئەركەكلىرىمىز قېنى دەپ يىغلاۋاتقانلىقلىرىغا قۇلاق سېلىپ باقسۇن. كوچىدىكى شۇنداق ئاياللارغا زاڭلىق قىلىش ، مەسخىرە قىلىش بىلەن قاراۋاتقان ئاشۇ قارتا ئويناۋاتقان ئەرلەرگە لەنەت بولسۇن!!! ئۆز مىللىتىنى ئۆزى مەسخىرە قىلىۋاتقان ئەركەكلەرگە لەنەت بولسۇن!!!  شۇ ئىرى بار ئايالنى تانسىغا تارتقان ئەرلەرگە لەنەت بولسۇن!!! ئايالىنى بالىلىرىنى تەربىيلەشكە قويماي رىستۇرانمۇ رىستۇران سۆرىۋالغان ئەرلەرگە لەنەت بولسۇن!!! ئۆيدە ئېرىنى ھۆرمەت قىلماي، كوچىغا چىقىۋالىدىغان ئاشۇ نادان ئاياللارغا لەنەت بولسۇن!!! بىچارە پەرزەنتلەرنى يىتىم قالدۇرۋاتقان ئاشۇ ئازغۇن ئاتا-ئانىلارغا لەنەت بولسۇن!!!
گەپ ئېغىزدىن چىققۇچە ئادەم باشقۇرىدۇ، ئېغىزدىن چىققان گەپ ئادەمنى باشقۇرىدۇ!
