ئايرۇپىلاندىنمۇ چۈشتۇق مانا خىياللار بىلەن ئۇلارغا ئەگىشىپ مىڭىۋاتىمەن، پۇتۇمغا تاش ئىسىپ قويولغاندەك مىڭىشىم تەسكە توختىماقتا ئىدى.
ئاخىرى چىقمايدىغان خىياللار بىلەن ئايرۇدۇم ئالدىغا قانداق چىقىپ قالغانلىقىمنىمۇ تۇيىماي قاپتىمەن، دادام ئەۋەتكەن بالىلار قولومدىن نەرسە كىرەكلىرىمنى ئالغاندىلا ئاندىن ئىسىمگە كەلدىم، دادامنىڭ قەشقەردىكى شۆبە شىركىتىدە ئىشلەيدىغان ئابدۇسوپۇر، ئەكپەر قاتارلىق ئاغىنىلىرىم ۋە مەن تونىمايدىغان ئىككى يىگىت ئىككى ماشىنىدا مىنى ئايرۇدۇرۇمغا ئالغىلى چىققان ئىدى، ئۇلار بىلەن بىر قۇر سالاملاشقاندىن كىيىن سەل ھۇشۇمغا كەلگەندەك بولدۇم، مىھرىئاي بىلەن ئۇيغۇرنىسا ئالدىمدىلا ماڭا قاراپ تۇراتتى، مەن ئىغىز ئىچىپ گەپ قىلىپ بولغۇچە مىھرىئاي ئىغىز ئاچتى.
پاكزات خوش ئەمىسە ، تىلىغۇنلىشىپ تۇرايلى، دوستۇم ئۆيىگە ئالدىرايتى، مەنمۇ ئۇنى ئۆيىگە ئاپىرىپ قويوپ ئۆيگە قايتاي، ئىش بولسا تىلىغۇن قىلارلا.
خۇددى ھازىرلا گەپ قىلمىسام ئۇلار كىتىپ قالىدىغاندەك جىددىيلىشىپ كەتتىم.
تەييار ماشىنا تۇرسا مەن ئاپىرىپ قوياي، ماشىنا توسىمىز دەپ ئۇزۇن تۇرۇپ قالارسىلەرمىكىن دىدىممەن ئۇلارنىڭ كىتىپ قىلىشىدىن ئەنسىرەپ.
كەچۈرۈڭ ھىلىمۇ بايىدىن بىرى كۆپ ئاۋارە قىلدۇق سىزنى، سىزنىمۇ يولىڭىزدىن قويمايلى دىدى ئۇيغۇرئاي يەرگە قاراپ تۇرۇپ.
ئۇيغۇرئاي بىلەن بۇنداقلا خوشلاشقىم يوق ئىدى، لىكىن نىمە ئامال.
بولىدۇ ئەمسە مەنمۇ مىڭىۋىرەي ، بۈگۈن ياخشى تونۇشۇپ قالدۇق، مەن ياردەم قىلغۇدەك ئىش بولسا تىلىغۇن قىلارسىلەر دىدىمدە تىلىغۇن نۇمۇرۇمنى قالدۇرۇپ قويوپ ماشىنىغا ئولتۇردۇم،
ئاداش سەن مىنى ئۆيگە ئاپىرىپ قويغىن ، قالغانلار قايتىپ ئارام ئالسۇن دىدىم ئابدۇسوپۇرغا قاراپ.
ئابدۇسوپۇر بالىلارنى يولغا سىلىۋەتكەندىن كىيىن ماشىنىغا چىقتۇق لىكىن پەقەتلا ماڭغۇم يوق ئىدى ئەمما ماڭماسلىققا باھانەممۇ يوق ئىدى.
ماشىنا قوزغىلىش بىلەن تەڭ يۈرۈگۈم ئەنسىز سوقوشقا باشلىدى،شۇنداق ئارقىغا قارىغۇم كەلدى، لىكىن ئۇنداق قىلالمايتىم، ئابدۇسوپۇرنىڭ ھازىرقى روھى ھالىتىمنى بىلىپ قىلىشىنى خالىمايتىم، شۇنداقتىمۇ ئۇنىڭغا بىلىندۈرمەي ئۇيغۇرئايغا قاراپ بىقىش ئۈچۈن ماشىننىڭ يىنىدىكى ئەينىكىدىن ئارقىغا قارىدىم.
ئۇيغۇرئاي يەنىلا بايىقى جايىدىن مىدىرلىماي ماشىنىنىڭ ئارقىدىن قاراپ تۇراتتى.
پاكزات ئەپەندى ئارقىغا قارىغۇلىرى بولسا ئوغۇرلۇقچە قارىماي ئۆرۈلۈپ قاراۋەرسىلە، مەن ھىچقانداق ئويلىمايمەن، بويانلىرىغا قوقاق تۇرۇپ قالمىسۇن يانا، دىدى ئابدۇسوپۇر مەنىلىك كۈلۈپ قويوپ.
  بۇ نىمە دىگىنىڭ مەن ئارقىغا قارىمىدىم ، ئەينەككە مۇنداقلا قاراپ قويدۇم شۇ دىدىممەن، ھۇدۇقۇپ.
ئۇ بارلىق ئاۋازىنى قويوپ بىرىپ قاقاھلاپ كۈلۈپ كەتتى.
ئۇنچىلىك ئىشنىمۇ بىلمەيدىغان كىچىك بالا ئەمەسمەن، بولدى، خىجىل بولمىسىلا ، بەك قارىغۇلىرى كەپكەتكەن بولسا قاراۋەرسىلە مەن توغرا چۈشىنىمەن، بىشىمىزدىن جىق ئۆتكەن بۇنداق ئىشلار بىزنىڭ ، ماشىنىنى ئارقىغا قايتۇرايمۇ يا؟
نىمە دىيىشىمنى بىلەلمەيلا قالدىم، ئۇ كۆڭلىمدىكىنى تىپىۋالغان ئىدى، دەقىقىدىن كىيىن ئىككىلىمىز تەڭلا كۈلۈشۈپ كەتتۇق .
بولدى قىلسىلا كۆرۈشكىلىرى بولسا ئادىرىسى ۋە تىلىغۇن نۇمۇرىنى مەن تىپىپ بىرىمەن، بۈگۈن ۋاقتىدا ئۆيگە قايتىپ ئارام ئالسىلا ئاتا ئانىلىرىمۇ يوللىرىغا قاراپ قالدى، دىدى ئابدۇسوپۇر ئارقىغا قارىغاچ.
مەنمۇ ماقۇل دىگەن مەنىدە بىشىمنى لىڭىشتىپ قويدۇم .
ئوقوتقۇچىلارنىڭ ئەڭ ياخشى ماتىرىيال ئامبىرى شىرنە ماتىرىيال ئامبىرى http://www.xirna.com ،ھەقسىز ماتىرىياللار چۈشۈرىۋىلىڭ.

مانا ئۇيغۇرئاي بىلەن ئايرۇدۇرۇمدا خوشلاشقىنىمىزغىمۇ بىر ھەپتىدىن ئىشىپتۇ، مەيلى نىمىلا  ئىش قىلاي ئۇيغۇرئاينىڭ ئوتلۇق سىماسى كۆز ئالدىمدىلا، قولوم ھىچقانداق ئىشقا بارمايتى، قىلالايدىغان بىردىنبىر ئىشىم ئۇ ھەم مەن ياخشى كۆرىدىغان ئاشىق دىگەن ناخشىنى قايتىلاپ ئاڭلاش ئىدى.

سىنىڭ كۆڭلۈڭ كىمگە باقتى جان.
مىنىڭ كۆڭلۈم ساڭا باقتى.
بىلىپ قويغىن ، ساڭا مەن ئاشىقلىقىمنى
سەنسىز ھەرگىز يۈرەلمەسمەن.
........................

تۇرۇپ سەۋدالارچە خىيالغا چۆكۈپ كەتسەم، تۇرۇپ كۈلۈپ كىتەتتىم، بەزىدە تۇرۇپلا ئۆزۈمگىمۇ ئۆچلىكىم كىلەتتى.
توۋا خىلى بەردەس يىگىت ئىدىم، ئىسمىنىلا بىلىدىغان باشقا ھىچقانداق ئىشلىرىنى بىلمەيدىغان بىر ناتونۇش قىز ئۈچۈن نىمە بولوپ كىتىپ بارىمەن ، ئۇ قىز يا مىنىڭ كىملىكىمنىمۇ ، ئۇنى ياخشى كۆرىدىغانلىقىمنىمۇ بىلمىسە نىمە ئۆزۈمنى قىينايمەن دەپ ئويلايتىم ، لىكىن يەنە دەرھاللا ئۆز ئۆزۈمگە رەدىيە بىرەتتىم.
ئۇنى چوقوم تىپىشىم ، ئۇنىڭغا ئۆزىنى ياخشى كۆرىدىغانلىقىمنى ، ئۇنى بىر ئۆمۈر ياخشى كۆرۈپ، بەخىتلىك قىلىدىغانلىقىمنى ئىيتىشىم كىرەك.
ئابدۇسوپۇرنىڭمۇ يوقاپ كەتكىننى دەيمەن، مانا مەن ئۆزۈم خەۋىرىنى بىرىمەن دەپ يوقاپ كەتكىلىمۇ ھەپتىدىن ئىشىپتۇ، ھەجەپ جىمپلا كەتتى دىسە، بەرگەن ۋەدىسىنى ئۇنتۇپقۇ قالمىغاندۇ ھە؟
تىلىغۇن قىلايمۇيا؟ ياق ساقلاپ باقاي، يەنە بىر ئىككى كۈن ساقلاپ باقاي يەنە خەۋىرى بولمىسا ئاندىن بىرگەپ بولا.
تۇيۇقسىز چىكىلگەن ئىشىك ئاۋازىدىن چۈچۈپ خىيالدىن ئويغاندىمدە ھوجرامنىڭ ئىشىكىنى ئاچىتىم، ئىشىك ئالدىدا ئابدۇسوپۇر توراتتى.
توۋا دەۋەتتىم ئىچىمدە ، ئادەم بىرەر ئىشنى خىيال قىلغىلىمۇ بولمامدۇ نىمە ، بىردىنلا پۇت قولومغا جان كىرگەندەك بولۇپ تىزلىك بىلەن قولومنى ئۇزىتىپ ئۇنىڭ بىلەن كۆرۈشتۈم ۋە ئۆي ئىچىگە تەكلىپ قىلىپ، ھوجرامنىڭ ئىشىكىنى يىپىۋىتىپ  تاماكا سۇندۇم.
ئاداش تاماكا چىكە،چاي قۇيۇپ بىرەيمۇ؟ ئۇسساپ كەتتىڭ ھەقاچان؟ ھەجەپ يوقاپلا كەتتىڭغۇ؟
بولدى ئۆزلىىرنى  ئاۋارە قىلماي، قىزىل تۈگە بولسا بىرەنى ئىچەي، راسلا بەك ئۇسساپ كەتتىم ، قانداق قىلىمىز دەيلا، سەككىز سائەت ئىشلەپ جان باقىدىغان ئادەم شۇ خىزمەت بىلەن ئالدىراش بولوپ قالدىغان گەپكەن، ئۇنىڭ ئۈستىگە بۇ بىر ھەپتە شىركەتتىمۇ ئىش جىق بوپ قالدى، شۇڭا سىلىنىمۇ يوقلاپ كىلەلمىدىم، قانداق ھاردۇقلىرى چىققاندەكمۇ؟ ھاردۇقلىرى چىققان بولسا شەھەرنى بىر ئايلاندۇرۇپ كىلەي دەپ كىلىشىم ئىدى؟
توڭلاتقۇدىن بىر تال قىزىل تۈگىنى ئىلىپ ئۇنىڭغا سۇندۇمدە ئۇنىڭ ئۇدۇلىدىكى كىرىسلوغا كىلىپ ئولتۇردۇپ قىسقىلا جاۋاپ بەردىم.
شۇنداق ھاردۇقۇممۇ چىقىپ قالدى، لىكىن سىرىتقا چىقىپ ئويناشقا رايىم يوق، قانداق يىڭى خەۋەرلەر بار؟
ئۇيغۇر ئاي توغرىسىدا ئىگەللىگەن ئەھۋاللارنى تىزراق دەپ بىرىشىنى كۈتۈپ خىلى ئۇزاق ئۇنىڭغا قاراپ ئولتۇرغان بولساممۇ، ئۇ ھىچ گەپ قىلاي دىمەيتى، تۇيۇقسىزلا گاچا بولوپ قالغاندەك ئىدى، ئۇ گەپ قىلمىغانسىرى ۋاقى تىخىمۇ ئاستا ئۆتىۋاتقاندەك بىلىنىپ كىتىۋاتاتتى، ئۇ قولىدىكى تاماكىنى ئالدىرىماي چىكىپ تۈگەتتى دە قىزىل تۈگە ئىچىملىكىنىڭ ئاغزىنى ئالدىرىماي ئېچىپ ئىچكەچ گەپنى باشلىدى.
ھوجرىلىرىنى قايتىدىن تۈزۈشتۈرۈپلا ھە؟ ھەجەپ چىرايلىق قىلىۋاپلا، قىزلارنىڭ ھوجرىسىدىن قىلىشمايدىكەن، ئۈرۈمچىدىن كەلگەندىن بىرى خىلى سەمرىپ قاپلا ، دىمىسىمۇ غەم بولمىغاندىنكىن ، يىتىپ يىسە ھەممە ئادەم سەمرىيدۇدە شۇنداقمۇ؟ دىدى ئۇ ماڭا سۇئال نەزىرى بىلەن قاراپ.
بەكلا ئاچچىقىم كەلدى،مەن نىمە كويدا، بۇ نىمە ئويدا، مىنىڭ خىيالىمدىكى ئىشلار ئۇنىڭ خىيالىدا يوقتەكلا ئىدى، ۋاي ئىسىت بىر ھەپتىدىن بىرى سىنىڭ يولوڭغا قاراپ تاتلىق خىياللارنى سۈرۈپ كىتىپتىمەن، سىنى كۆرۈپ خوشلۇقتا تاماكا تۇتۇپ كەتكەنلىرىمچۇ تىخى، مەيدەڭگە ئۇرۇپ لاپ ئاتقان بىلەن ئىسىمدىمۇ يوق دىگىنە خەپ.
شۇ ، توي قىلىپ قالسام دەپ تۈزەشتۈرۈپ قويدۇم، بولمىسا قىزلارنى ھۈركۈتۈپ قويمامدىم دىدىم تەنە ئارىلاش.
توغرا ، توغرا توينىڭ گىپىنى قىلمىغان بولىسىلا مۇھىم ئىشتىن بىرنى ئۇنتۇپ قالىدىكەنمەن.
مۇنداق دىمەمسەن، مەنمۇ سىنىڭ مۇشۇ گىپىڭنى ساقلاپ تۇرغان قىنى تىزراق دىگىنە دىدىممەن، ئۇنىڭغا يىقىنراق سۈرۈلۈپ زورىغا كۈلۈمسىرەپ.
مۇنداق ئىش ، بايا شىركەتتە تۇرسام دادىلىرى مىنى بىر كىلىپ كەتسۇن دەپتىكەن شۇڭا ھازىر كەلگەن ئىدىم، دادىلىرى سىلىنى ئۆيلەپ قويوشنىڭ تەييارلىقىنى قىلساق دەپ سىلىدىن گەپ ئىلىشقا مىنى چاقىرتىپتۇ، شۇ سىلىنىڭكى كۆڭۈلدىكىلىرىنى دەپ بەرسىلە ، مەنمۇ جاۋابىنى دادىلىرىغا بىرىۋىتىپ قايتىپ ئىشىمنى قىلاي؟ سىرىتلارنى ئايلاندۇرۇپ كىلەي دىسەم خوشياقمايدىغان ئوخشايدۇ سىلىگە.
مۇشۇ گەپمىتى؟ دىدىممەن زەردە بىلەن
ھەئە شۇگەپ يەنە نىمە گەپ بولماقچىتى، توي ئىشى ياخشى ئىش تۇرسا خوش بولماي نىمە چىچىلىپ يۈرىلا؟
مەن نەدە چىچىلدىم، دىدىممەن ئۆزۈمنى تەستە بىسىپ
ھازىچ چىچىلدىلىغۇ ؟
مەن چىچىلمىدىم
راسلا چىچىلمىلىما
ياق چىچىلمىدىم
پەقەت چىچىلمىدىلىما؟
ياق پەقەت چىچىلمىدىم، ئاچچىقىمدا ئاۋازىمنىڭ يۇقىرى چىقىپ كەتكەنلىكىگىمۇ دىققەت قىلماپتىمەن.
ئابدۇسوپۇر قاقاھلاپ كۈلۈپ كەتتى دە كۆزلىرىدىن چىققان ياشلىرىنى سۈرتكەچ دىدى.
راس خاپا بولماپلا، مەن ئىشەندىم، بولدى چاقچاق قىلىپ قويدۇم خاپا بولمىسىلا ياخشى خوش خەۋەر بار، قىشىمغا كىلىپ ئولتۇرسىلا دىدى يىنىدىكى ئوروندۇقنى كۆرسەتكەچ.
بولدى ئەخمەق بولىۋىرىدىغان ئادەم يوق گىپىڭ بولسا دەۋەگىن مەن ئاڭلاۋىرىمەن دىدىم كومپىيوتىر ئۈستىلىمنىڭ ئالدىدا ئولتۇرغاچ.
ئاۋازىم يۇقۇرى چىقىپ كەتسە ئۇيغۇرئاينىڭ ئىسمنى ئانىلىرى ئاڭلاپ قالارمىكىن دىدى ئابدۇسوپۇ كۈلۈپ تۇرۇپ.
ئۇيغۇرئاينىڭ ئىسمىنى ئاڭلاپلا يۈزلىرىممۇ ئوت بولوپ كەتتى
توۋا ماۋۇ بالىنى ئىسمىنى ئاڭلىسىلىرىمۇ قىزىرىپ كىتىدىكەنلا ھە ؟
بولدى قۇرۇق گەپنى ئاز قىلىپ گەپكە كىلەيلى دىدىممەن ئۇ كۆرسەتكەن ئوروندۇققا كىلىپ ئولتۇرۇپ.
بولدى دەپ بەسەملا بولىدىغۇ ، خاپا بولمىسىلا، مۇنداق ئىش مەن بۇ بىر ھەپتىدىن بىرى ئۇنىڭ ئۆيى بار جايدىكى دوسلىرىمدىن ئۇقۇشۇپ ئاخىرى ئۇنىڭ ئۆيىنى تاپتىم ، لىكىن ئۆيدە ئادەم يوقكەن بۇ بىر قانچە كۈندىن بىرى شۇ ئۆينىڭ ئەتراپىدا كۆز قۇلاق بولوپ يۈردۈم لىكىن ئۆيگە ئادەم كەلمىدى، ئەسلى خوشنىلىرىنى باشقىچە ئويدا بولوپ قالمىسۇن دەپ سورىماي دىگەن ئىدىم ،لىكىن سىلىنى بەك ساقلاپ كاللىسىدىن كەتمىسۇن دەپ بۈگۈن ئاخىرى قوشنىىلىرىدىن سۈرۈشتۈردۈم .
ئۇ گىپىنى تۈگۈتۈپ بولوپ قولىدىكى قىزىل تۈگە ئىچىملىكىنى ئىككى يۇتۇم ئىچىپ بولوپ ئىچىملىكنىڭ ئىچىگە قاراشقا باشلىدى.
ۋاي بولدى قاراۋەرمە، ئىچىپ بولدۇڭ، ئىچىدە قالمىدى، تۈگە چىقمايدۇ ئىچىدىن ئۇنىڭ، ئادەمنى تەقەززا قىلماي دىگىنە گەپنىڭ ئاخىرىنى ئىچىدىغان ئىشىڭ بولسا بۇ بىرنىڭ ئورنىغا مىڭنى ئەڭ بىرەي دىدىممەن جىلە بولوپ.
گەپلىرىدە تۇرسىلا ئەمسە؟
بولدى گىپىم گەپ ، ئاخىرىنى دىگىنە ئىتتىك
قانداق بولىتى، يەنە ئۈرۈمچىگە كىتىپتۇ.
نىمىگە كىتىپتۇ؟ ئۈرۈمچىنىڭ نەدىكەن؟ نىمە قىلىدىكەن ئۈرۈمچىدە؟
ئالدىرىماي تۇرسىلا مەن تەپسىلى دەپ بىرىمەن، ئانىسىنىڭ سالامەتلىكى ياخشى ئەمەسكەن ئۈرۈمچىگە داۋالاتقىلى ئىلىپ كىتىپتۇ، قايسى دوختۇرخانىغا ئىلىپ كەتكەنلىكىنى قوشنىلىرىمۇ ئىنىق بىلمەيدىكەن.
مۇشۇ بىر ھەپتىدىن بىرى مىنى ساقلىتىپ ئىلىپ كەلگەن خوش خەۋىرىڭ مۇشۇمۇ ؟ سىنىمۇ ئاغىنەم دەپ يۈرۈپتىمەن ، بولدى ئۆزۈم ئىزدەيمەن دىدىممەن ئاچچىقىمدا.
مەنمۇ مۇشۇ گەپنى دىمەكچى ئىدىم، بىز بىرلىكتە بارايلى
سەن نىمە قىلىسەن بىرىپ، بولدى ئۆزۈم بارىمەن.
بىللە بارىلى، مىنىڭ ياخشى ئامالىم بار، ئەگەر قوشولمىسىلا ئۆزلىرى ئامال قىلسىلا، بىللە بارسام ياردىمىم تىگىدۇ، جەمىيەتتە ئوينايدىغان بىر قانچە ياراملىق دوسلىرىم بار ئۇلارنىمۇ ئىشقا سالايلى، ئادەم كۆپراق بولسا تىپىشمۇ ئاسان بولىدۇ، قانداق دەيلا؟
بولىدۇ ئەمسە قىنى دەپ باقە قانداق ياخشى ئامالىڭ بار؟ دادامنى بىزنى ئۈرۈمچىگە ئەۋەتىشكە قايىل قىلىدىغانغا ؟
بۇ ئاسان ، بايا دادىلىرى مىنى سىلىنى ئانچە مۇنچە شركەتنىڭ ئىشىغا ئارلىشىپ ، شىركىتىمىزنىڭ سودا ئىشلىرىنى چۈشەنسۇن، سەن كىرىپ سۆزلىشىپ باق چاقىرتىپتىكەن، مەن چىقىپ سىلىنى ئۈرۈمچىدىكى شىركەتنى كۆرۈپ كەلگەچ مەكتەپتىكى ئوقوتقۇچىلىرىم بىلەن كۆرۈشۈپ كىلەي دەيدۇ دەي قانداق بولامدۇيا؟
كۈلۈپ كەتتىم، بەزى ئىشلارنى ئەپلەشتۈرۈلمىگىنىڭ بىلەن ھىلىگەرلىكتە خىلى نامىڭ بار جۇما دىدىممەن ئىچىمدىن كۈلۈپ تۇرۇپ.
بولدى سىلە قالغان ئىشنى ئويلىماي، مىڭىشنىڭ تەييرلىقىنى قىلسىلا ، قالغان ئىشلارنى ئۆزۈپ ئەپلەشتۈرىمەن، ھە راسىت قاچان يولغا چىقىمىز؟
ئەتە سەھەردە، بولسا تىخىمۇ تىز بولسا ياخشى بولاتتى.
بولىدۇ ئەمسە ئەتە سەھەر يولغا چىقايلى ؟
ناخشىنىڭ ئاۋازىنى يۇقىرى كۆترىۋىتىپ، كۆزۈمنى يۇمدۇم، قاچانمۇ تاڭ ئاتار.
