          ئالىي مەكتەپكە ئوقۇغۇچى قوبۇل قىلىش ئالدامچىلىقىدىن ئاگاھ بولۇڭ

                                پاتىمە ئەيسا ، نىجات ئىمىن
    ھەر يىللىق ئادەتتىكى ئالىي مەكتەپ ئىمتىھانىنىڭ نەتىجىسى ئېلان قىلىنىشى بىلەن تەڭلا قىسمەن ئوقۇغۇچىلار نومۇرىنىڭ ئالىي مەكتەپنىڭ ئوقۇغۇچى قوبۇل قىلىش نومۇر چېكىگە يەتكەنلىكى ياكى خېلى يۇقىرى بولغانلىقىدىن خۇشال بولسا، يەنە قىسمەن ئوقۇغۇچىلار غەمگە پاتىدۇ، ئاتا – ئانىلىرىمۇ ھەم شۇنداق. ئىمتىھاندىن ئۆتەلمىگەن بەزى ئوقۇغۇچىلار تولۇق 3 – يىللىقتا قايتا ئوقۇشنى تاللايدۇ ياكى باشقا چىقىش يولى ئىزدەيدۇ، بەزى ئاتا – ئانىلار بالىسىنىڭ بۇنداق تاللىشىغا ھۆرمەت قىلىدۇ. يەنە بەزى ئوقۇغۇچىلار چوقۇم شۇ يىلى ئالىي مەكتەپكە ئوقۇشقا بېرىشنى بەك ئارزۇ قىلىدۇ، بۇنىڭ بىلەن قىسمەن ئاتا – ئانىلار بالىسىنىڭ ئالىي مەكتەپتە ئوقۇش ئارزۇسىنىڭ كۈچلۈك بولۇشى ياكى بالىسىنى تونۇش – بىلىشلىرىنىڭ بالىسى بىلەن سېلىشتۇرۇش، بەسلەشتۈرۈش قاتارلىق سەۋەبلەر تۈپەيلى ئۆز ئالدىغا تونۇش ئىزدەپ، بالىسىنى ئالىي مەكتەپكە قوبۇل قىلدۇرماقچى بولۇپ، تەرەپ – تەرەپكە قاترايدۇ. ئوقۇغۇچى قوبۇل قىلىشتىكى ئالدامچىلار ئوقۇغۇچىلارنىڭ، ئاتا – ئانىلارنىڭ گاڭگىراپ يۈرگەن ئاجىزلىقىدىن پايدىلىنىپ ھەر خىل ئالدامچىلىق ۋاسىتىسىنى قوللىنىدۇ، پىسخىك جەھەتتىن ئاجىزلاشقان بەزى ئوقۇغۇچىلار، ئاتا – ئانىلار ئالدامچىلارنىڭ ‹‹قارمىقى››غا ئاسانلا ئىلىنىپ قالىدۇ.
    2010 – يىللىق ئادەتتىكى ئالىي مەكتەپكە ئوقۇغۇچى قوبۇل قىلىش مەزگىلىدە مائارىپ تارماقلىرى ئاپتونوم رايون بويىچە ئوقۇغۇچى قوبۇل قىلىش ئالدامچىلىقىغا زەربە بېرىش ھەرىكىتىنى قانات يايدۇرۇپ، ئۈرۈمچى، قۇمۇل، بايىنغولىن، ئاقسۇ، سانجىلاردا 21 قانۇنسىز ئوقۇغۇچى قوبۇل قىلىش نۇقتىسىنى پاش قىلدى. شىئەندىكى بىر خۇسۇسىي مەكتەپ ئاپتونوم رايونىمىزدىن 48 ئوقۇغۇچىنى ئالداپ قوبۇل قىلىپ ئېلىپ ماڭغاندا، ئاپتونوم رايونىمىز بۇ ئوقۇغۇچىلارنى پويىز ئىستانسىسىدا توسۇپ قېلىپ، ئۇلارنى ۋاقىت ۋە ئىقتىسادىي زىياندىن ساقلاپ قالدى. ئالدانغان بۇ ئوقۇغۇچىلارنىڭ كۆپ قىسمى ئاز سانلىق مىللەت ئوقۇغۇچىلار بولۇپ، 32سى چەت، يىراق يېزىلاردىن ئالداپ ئېلىپ مېڭىلغان، بۇنىڭدىن ئالدامچىلارنىڭ ئالدامچىلىق تىغ ئۇچىنى ئۇچۇر ئالاقىسى تۆۋەن يېزىلارغا، دېھقان – چارۋىچىلارنىڭ بالىلىرىغا قارىتىدىغانلىقىنى كۆرۈۋالالايمىز.

    ئاپتونوم رايونلۇق ئوقۇغۇچى قوبۇل قىلىش ئىشخانىسىدىن ئادىل مۇھەممەت: ئوقۇغۇچىلار ئالىي مەكتەپ چاقىرىق قەغىزىنى تاپشۇرۇۋالغاندا، ئۇ مەكتەپنى ئارزۇسىغا يازغانمۇ يوق؟ قوبۇل قىلىنغان مەكتىپى ئاپتونوم رايونلۇق ئوقۇغۇچى قوبۇل قىلىش ئىشخانىسى بېسىپ تارقاتقان ‹‹2011 – يىللىق ئادەتتىكى ئالىي مەكتەپلەرگە ئوقۇغۇچى قوبۇل قىلىش ۋە ئىمتىھان›› دېگەن قوللانمىدا بارمۇ – يوق؟ قوبۇل قىلىنغانلىق ئۇچۇرى ئاپتونوم رايونلۇق ئوقۇغۇچى قوبۇل قىلىش ئىشخانىسى ئېلان قىلغان سۈرۈشتۈرۈش تورلىرىدا بارمۇ – يوق؟ بۇ نۇقتىلارغا ئالاھىدە دىققەت قىلىشى كېرەك، ئۇنىڭدىن باشقا، شۇ جايىدىكى ئوقۇغۇچى قوبۇل قىلىش ئىشخانىلىرىدىن سۈرۈشتۈرسىمۇ بولىدۇ، دېدى.
    ئوقۇغۇچى قوبۇل قىلىش پىلانىنى دۆلەت بىر تۇتاش بەلگىلەيدۇ ھەم جەمئىيەتكە ئاشكارا ئېلان قىلىدۇ، دۆلەت تەستىقلىمىغان، جەمئىيەتكە ئاشكارا ئېلان قىلىنمىغان ئوقۇغۇچى قوبۇل قىلىش پىلانىنىڭ ھەممىسى ساختا، مەكتەپلەرنىڭ ئۆز ئالدىغا ئوقۇغۇچى قوبۇل قىلىشىغا، تولۇقلاپ ئوقۇغۇچى قوبۇل قىلىشىغا يول قويۇلمايدۇ ھەم دۆلەت ئېتىراپ قىلىش دائىرىسىدە يوق مەكتەپلەرگە ئالدىنىپ قوبۇل قىلىنىپ ئوقۇغان ئوقۇغۇچىلارنىڭ تىزىملىكى ئېلېكترونلۇق تىزىملىككە كىرگۈزۈلمەيدۇ، ئېلېكترونلۇق تىزىملىككە كىرگۈزۈلمىگەن ئوقۇغۇچىلارنىڭ ئوقۇش پۈتكۈزۈش گۇۋاھنامىسى كادىرلار تارماقلىرىنىڭ تەكشۈرۈشىدىن ئۆتمەيدۇ ۋە ئېتىراپ قىلىنمايدۇ. ھازىر ئېلىمىزنىڭ ئادەتتىكى ئالىي مەكتەپكە ئوقۇغۇچى قوبۇل قىلىش خىزمىتى تۈزۈمىنىڭ قېلىپلىشىش دەرىجىسى، باشقۇرۇش تەرتىپى پىشقان، قېلىپلاشقان باسقۇچقا قەدەم قويدى، شۇڭا ئوقۇغۇچىلار ھوشيارلىقىنى ئۆستۈرۈپ، ئالدامچىلارنىڭ ئالدام خالتىسىغا چۈشمەسلىكى، ئازراقلا گۇمان بولسا، ئۆز جايىدىكى ئوقۇغۇچى قوبۇل قىلىش ئىشخانىسىغا ياكى ئاپتونوم رايونلۇق ئوقۇغۇچى قوبۇل قىلىش ئىشخانىسىنىڭ مەخسۇس سۈرۈشتۈرۈش تېلېفونى(8750000 – 0991)غا تېلېفون بېرىپ ئېنىقلىشى كېرەك، ئاپتونوم رايونلۇق ئوقۇغۇچى قوبۇل قىلىش ئىشخانىسىنىڭ بۇ قىزىق لىنىيەلىك تېلېفونى ئوقۇغۇچى قوبۇل قىلىش مەزگىلىدە ئوقۇغۇچىلارغا، ئاتا – ئانىلارغا ھەر ۋاقىت مۇلازىمەت قىلىدۇ.
