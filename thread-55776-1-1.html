     مەن دادامنى يىغلىمايدۇ ،ئەزەلدىن كۆز يېشى تۆكۈپ باققان ئەمەس ،بۇلۇپمۇ ھاياتىدا ئۆزىنىڭ بىر ئوغۇلبالا ، ھەقىقى ئەركەك ئىكەنلىكىنى ھېس  قىلغىنىدىن تارتىپ، مەيلى قانداق ئەھۋال بولۇشىدىن قەتئىنەزەر كۆز يېشىنى تۆكۈپ يۈرمىگەن ، مىنىڭ دادام ئەنە  شۇنداق ئادەم دەپ يۈرۈپتىكەنمەن .ئەمما مەن كېيىنكى كۈنلەردە مۇشۇ قارىشىم بىلەن قىپقىزىل خاتالاشقان ئىكەنمەن . شۇ خىل قاراشتىكىلەر ئىچىدە ئەڭ ناچىرى مەن ئىكەنمەن . ھاياتىدا نۇرغۇن ئىسسىق -سوغۇقنى ،جەۋرى -جاپانى تارتىپمۇ دەس يۈرگەن شۇ دادام مېنى15 مىنۇت ساقلاشقا تاقەت قىلالماي يىغلاپ كېتىپتۇ ئەمەسمۇ ...
دادامنىڭ ياش تۆككىنى كۆرگەن شۇ كۈنۈمدە مەن سىلەر ئويلىغاندەك دادامنىڭ قەلبىمدىكى ئوبرازىنى شۇ تۈپەيلى خۇنۈكلەشتۈرىۋالغىنىم يوق ،ئەكسىچە ، مىنى ھاياتىمدا تۇنجى ياخشى كۆرگەن ، شۇنداقلا بۇ ياخشى كۆرۈشنى مەڭگۈ داۋام قىلىدىغان ، دۇنيادىكى ئەڭ ئېسىل ئەرنىڭ دەل دادام ئىكەنلىكىدىن تولىمۇ تەسىرلەندىم.
      ناھايىتى ئالدىراش ئىشلەۋاتىمەن ، قۇلۇمدا قىلىپ تۈگىتەلمىگەن نۇرغۇن ئىشلار ئۈستىلىمدە ئۆچرەت ساقلاپ تۇرىدۇ . شۇ تۇرقۇمدا  ئەتە زامان ئاخىرى بولغۇدەك دىسىمۇ ،قۇلۇمدىكى ئىشنى تاشلاپ قويۇشقا كۆزۈم قىيمايدۇ ،.بۇ - بىزدىكى ئومۇمىيلىشىپ كەتكەن قانۇنىيەت.
  تۇيۇقسىزكەلگەن تىلفۇن كەيپىمنى ئۇچۇردى . قارىسام ئاكام :
-سەن ھازىر سەل بىكارمۇ ؟-دىدى ئۇ ئالدىراشلا .
- مەن بەك ئالدىراش .
-دادام تۇيۇقسىز دوختۇرخانىدا يىتىپ قالدى . تېخى بايىلا ئېلىپ كەلدۇق ،ئەھۋالى سەل ناچار ، سەن تېزدىن رۇخسەت سوراپ كەل،ئۇدۇل جىددىي قۇتقۇزۇش بۆلمىگە كىر.
ئاكام تېلفوننى قۇيىۋەتتى،بېشىمدىن تۈتۈن چىقىپ تۇيۇقسىز ئالەم بىر مىنۇت جايىدا توختاپ قالغاندەك ھىسسىياتقا كىلىپ قالدىم .قانداقلارچە ما جۇرىننىڭ ئىشخانىسىغا كىرىپ قالغىنىمنى بىلمىدىم :
- مەن رۇخسەت سورايمەن .
-رۇخسەت سوراش خېتى يازىسىز ،-دىدى جۇرىن چىرايىمدىكى ئۆزگىرىش ئالامەتلىرىگە قاراپ .
رۇخسەت سوراش خېتىنى قانداق تېز يازغان بولسام ،شۇنداق تېز ئېلىپ كىردىم . ئىشخانىدا دادام بىلەن بۇرۇن بىر ئىدارىدا ئىشلىگەن پىشقەدەم خەنزۇ خىزمەتدىشىمىز بار ئىكەن :
- نىمە بولدى رۇخسەت سوراپ قاپسەنغۇ؟
دادام دوختۇرخانىدا يېتىپ قاپتۇ .مەن گىپىمنىڭ ئاخىرىنى دىيەلمىدىم ،كۈزۈمگە يىغىلغان لىققىدە ياشتىن ھىچنىمىنى كۆرەلمىدىم ،ئەمما ئۇنى سىڭدۈرۋىتىشكە مىنىڭ ئامالىم بولمىدى .
-نىمە ئاغرىقتى ؟
-يۈرەك كىسىلى ،- مەن شۇنداق دىگەنچە سىرتقا قاراپ ئۇچتۇم.مەن ئىشلەۋاتقان قاتناش ئەترىتى ناھىيە بازىرىدىن 7 كىلومىتىر يىراقلىقتىكى ناھىيە سىرتىدا ئىدى . بۇ ئەتراپتا ئۇزۇن يول ماڭىدىغان ئاپتوبۇس ۋە كىچىك ماشىنىلار ئۆتۈپ تۇرمىسا تاكسى ۋە مەخسۇس ئابتوبۇسلار بۇلۇپمۇ سىز ساقلىغاندا تولىمۇ تەستە كىلەتتى . توسقان ئىككى تاكسىدا ئادەم بولغاچقا ئۆتۈپ كەتتى .ئىسىت مىنىڭ قاتناش ئەترىتىدە ئىشلەپ قالغىنىم ، ئۆز -ئۆزەمدىن تۇرۇپلا نەپرەتلەندىم .ئاخىرى بىر ئابتوبۇس كەلدى .بۇمۇ بولغىنى ،بۇ ئابتوبۇس  دەل دوختۇرخانا ئەتراپىدا توختايتتى .
بىردەملىك خىيالغا پاتتىم ‹‹ھەي ئالمىرە ،سېنىڭ داداڭ يىغلامدۇ ؟نۇرسىماننىڭ  بوۋىسى تۈگەپ كىتىپتىكەن ،دادىسى ھۈ..ھۈ...ھۈ... قىلىپ يىغلاپتۇ .
-ئۇ-مىنىڭ بوۋام ،دادامنىڭ دادىسى بولغاندىكىن يىغلىماي كۈلەمتى ، ساراڭ!
-ئۆزەڭ ساراڭ ، قارا مەخرۇزە ، ئۇنى دوستلۇقتىن چىقىرۋىتەيلى ،ئۇنىڭ بىلەن دوست بولمايلى -ھە ،ھەي ،سېنىڭ داداڭ يىغلامدۇ ؟
-مىنىڭ دادام يىغلىمايدۇ ،مەن دادامنىڭ بوۋام تۈگەپ كەتكەندىمۇ ماڭقىسىنى تارتىپ يىغلىغىنىنى كۆرمىگەن ،چۈنكى مىنىڭ دادام ساقچى ،ساقچىمۇ يىغلامدۇ ؟ سەن دەپ باقە ،ساقچى فورما كىيىۋىلىپ يىغلىسا نىمىگە ئوخشاپ قالىدۇ ؟ئالمىرەنىڭ  دادىسىغا ئوخشاپ قالمامدۇ ؟
-سېنىڭ داداڭ بوۋاڭغا ئۆگەيدۇ شۇ ،شۇڭا يىغلىمىغاندۇ يە ،
-ھەي سېرىق ، نىمە قېرى خوتۇندەك بىلمەيدىغىنىڭ يوق سېنىڭ ،مىنىڭ دادام بوۋامنىڭ چوڭ ئوغلى ،بىلىپ قوي!....
كىچىك چاغلىرىمدىكى شۇ ئەسلىمىلىرىم مىنى سەگىتتى . قارىسام ئالدىمدا دوختۇرخانا بىناسى . يۈرىكىم ئەنسىز خىياللار ئىلكىدا تېپىرلايتتى ،ئاكام دوختۇرخانا  ئىشىكى ئالدىدا تاماكا چېكىپ تۇراتتى :
-دادام قانداقراق ؟
-تېز كىرىپ كۆرۈشۈپ چىق .
كىرسەم دادام ئاسما ئوكۇل سېلىنغان ھالەتتە ئاران مىدىراپ يېتىپتۇ ،ئاپام بولسا يېنىدا غەمكىن ئولتۇرىدۇ .
-دادا ،مەن كەلدىم خېلى ياخشى يېتىپسىزمۇ ؟
ئىختىيارسىز كۆز ياشلىرىمنى توختىتىۋالالمىدىم . ئادەم بار يەردە يىغلىمايمەن دىگەنلىرىم بىر ياقلاردا قىپقالدى ،ئەمما دادام مەندىن بۇرۇن يىغلاپتىكەن ئەمەسمۇ ؟
-قارا داداڭنى ،ساڭا تارتىشىپ ،تولا ئىشىككە قاراپ ،سەن كەلگۈچە كۆز يېشى قىلىپ كەتكىنىنى ،يىغلاپ باقمىغان ئادەم يىغلىسا ئەجەپ بىر قىسما بولىدىكەن ،-دىدى ئاپام زورىغا كۈلۈمسىرەپ .
-مەن شۇ سەل كىچىكىپ قالدىم .
دادام بىشىنى بوش لىڭشىتتى ۋە بوش ئاۋازدا ئارانلا :
-تاماق يىدىڭمۇ ؟
- يىدىم .
مۇشۇ يېشىمدىمۇ ئۆز ھالى بىلەن بولماي مىنىڭدىن غەم يەپ يۈرۈشىنى ،مەن يا بەش ياشلىق بالا بولمىسام ...
دادام شۇ تۇرقىدا ماڭا بىر كۈندىلا كىچىلكلەپ يىگىلەپ قالغاندەكلا تۇيۇلدى ،چارلاشقان ساقاللىرى ئۆسۈپ ،يۈزلىرى قورۇلۇپ كەتكەندەكلا بىلىندى ،يۈرەك تىڭشىغۇچنىڭ سىترىلكىسىغا قارىدىم ،ئاجىز پىلىلداپ ماڭماقتا ،بۇ ئەلۋەتتە خۇش خەۋەر ئەمەس ئىدى .دادامغا چاندۇرماي سىرتقا چىقىپ مەسئۇل دوختۇرنىڭ ئىشخانىسىغا كىردىم
-دادامنىڭ ئەھۋالى قانداقراق ؟ماڭا راست گەپ قىلىڭ ،مەن ئۇنىڭ قىزى .
-ھازىرچە چوڭ چاتاق يوق ،بىز دادىڭىزغا يۈرەكنى تىنچلاندۇرۇدىغان ،قان تۇمۇرلارنى كىڭەيتىدىغان ئوكۇللارنى سىلىۋاتىمىز ،يۈرەك كىلاپانى توسۇلۇپ قاپتۇ ،ئەگەر بۇ سىناق ئۈنۈم بەرمىسە چوڭ دوختۇرخانىلارغا ئاچىقساڭلار بولىدۇ ، بەلكىم يۈرەككە كۆۋرۈك سېلىشى مۇمكىن ،ئۆزىنىڭ كەيپىياتىنى تەڭشىشى ئىنتايىن مۇھىم ،سوغاق نەرسىلەرنى كۆ پ ئىستىمال قىلمىسۇن ،مەڭگۈ ئاچچىقلانمىسۇن .
-ماقۇل ،چوقۇم .
ياتاققا كىرمەي ،كارىدوردا دەلدەڭشىپ يۇردۇم .بىزگە ،ئاكامغا ماڭا دادام بولمىسا بولمايدۇ ،زادىلا بولمايدۇ ،دادام ياشىشى كېرەك .نىمىلا بولمىسۇن ،ئاپام ئۈچۈن بولسىمۇ ،چۈنكى بىزنىڭ  ئاپامنى دادامدەك چوڭقۇر چۈشىنىپ ،ئۇنىڭغا يۈلەك بولالالىشىمىزغا كۈزۈم يەتمەيدۇ ،گەرچە ئاپامغا ئېچىل داستىخىنىم ،ئۇچار گىلەم ،ئۇر توقماققىچە ئالدىغا ئەكىلىپ بەرگەندىمۇ ،مەڭگۈ بىز بالىلار ئۇلارنىڭ ئورنىنى باسالمايمىز،چۈنكى دادام ئاپامغا دۇنيادىكى ئەڭ ئېسىل دادىلاردىنمۇ بەكرەك كۆيىنىدۇ ،ئاغرىپ قالسا،دوختۇرغا ئاپىرىپ ،ئۆزى تاماق ئېتىپ ئاپارىدۇ ،ئۇ بۇ ئىشنى بىر نەچچە يىللا داۋام قىلغىنى يوق ،چۈنكى ،مىنىڭ ئاپام كۆپ كىسەل تارتقانىدى ...ئۇلار 30نەچچە يىلنى شۇنداق بىر -بىرىگە يار -يۈلەك بۇلۇپ ئۆتكۈزدى ،دادام 10 يىل چىگرادا تۇرغان چىغىدا ،ئاپام يالغۇز بىزنى باقىمەن دەپ خىزمىتىدىن ئىستىپا بەردى ،بىزنى باقتى.. .
  بۇ شۇنداق قانۇنىيەت ئىكەن ،دىمەك ھاياتنىڭ لوگىكىسى ئەنە شۇنداق فورمىلادا داۋام ئىتىدىكەن .نۈۋەت بىزگە كىلىپ ،ئەمدى ئۇلار ئۇھ ،دەپ ،بىز بىقىش ۋەزىپىسىنى ئادا قىلغىچىلىك ئۇلار ماغدۇرىدىن كېتىپ ئاستا- ئاست بىر -بىرىنى تارتىپ ئارىمىزدىن كېتىپ قالىدىكەن ،،ئۇ دەھشەت كۈندە سىزنىڭ ئازابىڭىز پۇشايماندىن ئىبارەت خالاس.
دادامنىڭ بىزگە تارتىشقان ئىممونىت كۈچى غالىپ كىلىپ ،ئۈمىد -ئىشەنچى بىلەن بىرىنچى ئاللانىڭ مەرھەمىتى بىلەن ئۆيىمىزگە خۇشال ھالدا قايتىپ كەلدى ،بۇ تەلەينى ھەممىلا كىشىگە نىسىپ بولمايدۇ دەپ قاراپ ،ئۇنى قەدىرلەشكە تىرىشىمەن .
-شۇ چاغدا نىمىدەپ يىغلىدىڭىز ؟بالىلارغا دەپ بەرمەمسىز ؟-دىدى ئاپام بىر كۈنى .
-يالغان سۆزلىمە ،مەن نەدە يىغلىدىم ؟،-دادام كۈلدى
- ھەي ،يىغلىغىنىڭىزنىچۇ ،يېنىڭىزدا ياتقانلارمۇ كۆرگەن تۇرسا.
-مەنچۇ -دىدى دادام تەن بېرىپ -مۇشۇ ئىككى بالامنى مەندىن كىيىن قالسا قانداقمۇ قىلار ...دىگەننى ئويلىدىم ،يەنە بىرسىگە مەخرۇزەنى كۆرەلمەسمەنمىكىن دەپ ئەنسىرىدىم .
كۆزلىرىمگە يەنە ياش كەلدى .نىمىدىگەن تۈگىمەس ئەنسىرەش ،تۈگىمەس غەملەر بۇ ،ئاتا ئانىنىڭمۇ بۇنچىلىك غىمىنى يىمىسە كېرەك كىشى ،بالام ،بالام دەپ ئۆتىدىغان جاھانمىدۇ -بۇ ؟بىزمۇ كېيىن شۇنداق بولارمىزمۇ ؟...
    دوستۇم ،سىزنىڭ دادىڭىز يىغلامدۇ ؟مىنىڭچە ،بۇنى سوراشتىن ئاۋۋال ،ئۇنى ھەرگىز يىغلىتىدىغان ئىشنى قىلمايلى . ئۇلارنىڭ كۆز يېشىغا قالمايلى ،دوستۇم ،سىزدىن يەنە بىر ئىشنى سوراي ، سىز توي قىلغانمۇ  ؟ ئۆيىڭىز دادىڭىزنىڭ ئۆيىگە يېقىنمۇ ؟ئەگەر يېقىن بولسا ،كۈندە بولمىسا ھەپتىدە چوقۇم يوقلاشنى ئۇنۇتماڭ ،مەيلى قانداقلا مۇھىم ئىشىڭىز چىقىپ قىلىشىدىن قەتئىنەزەر  بىرىنچى بۇلۇپ ئۇلاردىن تىلفۇندا ئەھۋال سوراشنى ئۇنۇتماڭ ،پەرىزىمچە سىز ھاجەتخانىغا بېرىشنى ھەرگىز ئۇنۇتمايسىز ،شۇ ۋاقتىڭىزدىن كىتىۋىتىپ بىر مىنۇت ۋاقتىڭىزنى ئاجراتسىڭىزمۇ كىچىكمەيسىز .ئۇلارنى چۈشلۈك تاماقتا  ساقلاتماڭ ،كەچلىك ياتاردا ساقلاتماڭ،بولمىسا ئۇلار كۆز يىشى قىلىدۇ ، يەنە بىر ئەسكەرتىشىم ،ئۇلار بۇ دۇنياىدن كېتەرىدە تېخىمۇ ساقلاتماڭ...
