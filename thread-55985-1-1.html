             رايۇنىمىزدىكى قىسمەن ئۇيغۇرلارنىڭ توي- تۆكۈنلىرىدىكى بىيىت ياكى قوشاق ئېيتىشىشى ۋە ھازىرقى ھالىتى
       
توي ئىنسانلار ھاياتىدىكى ئەڭ چوڭ ۋە ئەڭ ئەھمىيەتلىك ئىشلارنىڭ بىرى بولۇپ ،ئۇ تارىختىن بۇيان شەخسلەر ،ئائىلە ۋە جەميەتنىڭ يۈكسەك ئېتىبارىغا ئېرىشىپ كەلمەكتە.
دىمەك ئەر- ئايال ئوتتۇرسىدىكى توينىڭ تەرەقىيات جەريانلىرىدا ئوخشاش بولمىغان ئۆرۈپ ئادەت ۋە قائىدە يوسۇنلار شەكىللەنگەن.
مېنىڭ بۇ يەردە دوسلارنىڭ ھوزۇرىغا سۇنماقچى بولغىنىم دەل رايۇننىمىزنىڭ قىسمەن جايلىرىدىكى ئۇيغۇرلىرىمىزنىڭ  توي بولۇپ يەنى قىزنى يۆتكەپ ئەكىلىپ بولغاندىن كېيىنكى    قائىدە يۇسۇنلىرىدىن بىرى بولغان بىيىت ئېيتىشى ۋاقتىدا ئىككى تەرەپ ئۆزارا ئېيتىشقان بىيىتلىرىدىن ئىبارەت.
دىمەك رايۇنىمىزنىڭ قىسمەن جايلىرىدىكى ئۇيغۇرلىرىمىز توي كۈنى قىزنى يۆتكەپ ئەپكېلىپ نولغاندىن كېيىن يىگىت تەرەپ قىز تەرەپنىڭ ئاتا ئانىسى ئۇرۇق تۇققانلىرى ،خۇلۇم خوشنىلىرى دوست بۇرادەرلىرىنى چىللايدۇ.
قىز تەرەپتىكى چىللاققا كەلگەن مېھمانلار كېلىپ بولغاندىن كېيىن مېھمانلارغا قەن كېزەك سېلىنغان ئۇسۇزلۇقلار تۇتۇلىدۇ. ئاندىن قىز ئۆزىنىڭ قولداشلىرى بىلەن بىرگە ئۇرۇق تۇققانلار خۇلۇم خوشنىلار ئولتۇرغان مېھمانخانىغا باشلاپ چىقىلىدۇ .
بۇ چاغدا ئۆينىڭ ئوتتۇرسىغا چىرايلىق كۆپە ئىككى قاتلاپ سېلىندۇ قىز شۇنىڭ ئۈستىگە دەسىگىنىچە ئۇرۇق تۇققىنى ۋە چوڭلارغا قارىغىنىچە قولداشلىرىنىڭ ھەمرالىقىدا تۇرۇدۇ بۇ چاغدا تېخى يۈزى ئېچىلمايدۇ ،مېھمانخاىنىغا بولسا قىز تەرەپتىن چىللاققا كەلگەن مېھمانلاردىن باشقا يەنى يىگىت تەرەپنىڭ ئايال ئۇرۇق تۇققانلىرى خولۇم خوشنىلىرى چوڭلارمۇ كىرىدۇ ئۆ ئىچى    قىز- يىگىت ئىككى تەرەپنىڭ قىز ئايال مېھمانلىرى بىلەن لىقمۇ لىق تولۇپ كېتىدۇ.
قىزنىڭ ئاستىغا سېلىنغان كۆپىنىڭ ئۈستىگە يوغانراق پاكىزە تاش قويۇپ قويۇلىدۇ سەۋەپ قىزنى ئاشۇ تاشتەك ئېغىر بېسىق ھالدا ئۆگۈنۈپ كەتسۇن دىگەننى بىلدۈرۈدىكەن.
مېھمانلار تۇلۇق كىرىپ ئورۇنلۇشۇپ بولغاندىن كېيىن قىز ئۆيدىكى چوڭلارغا ۋە ئۇرۇق تۇققانلىرىغا ھەممىسىگە سالام بېرىدۇ   سالام بېرىش ئاخىرلاشقاندا بولسا قىزنىڭ قولداشلىرى(يەڭگىلىرى دەپ ئاتىلاتتى) ۋاي توي قىزىمىزنىڭ بارماقلىرى قۇلاقلىرى بۇيۇنلىرى ئاغرىپ كېتىپتۇ دىيىشىدۇ بۇ چاغدا قىز تەرەپ يىگىت تەرەپ توي قىزىغا ئاتاپ سالغان ئالتۇن ھالقا،ئالتۇن ئۈزۈك، ئالتۇن بلەيزۈكلەرنى قىزغا تاقاپ قويۇشۇدۇ.شۇنىڭ بىلەن قىزنى كۆرپىدە ئولتۇرغۇزۇدۇ.
ئاندىن قىز تەرپ ئايرىم بىر پارچە رەختنى قىزنىڭ بېشىغا ياپىدۇ بۇ رەختنى يىگىت تەرەپتىكى مېھمانلارنىڭ كىچىك بالىلىرىدىن بىرى رەخمەت دەپ ئالىدۇ.
ئارقىدىن يىگىت تەرەپمۇ ئوخشاش ئۇسولدا رەخت ياپىدۇ بۇ قېتىم يېپىلغان رەختنى قىز تەرەپتىن كەلگەن مېھمانلارنىڭ كىچىك بالىلىرىدىن بىرى رەخمەت دەپ رەختنى ئالىدۇ.  
بۇ چاغدا پەتنۇسقا ئىككى پىيالە ئۇسۇزلۇق سېلىنغان پەتۇنۇس تەييار قىلىندۇ يەنى قىز تەرەپ ئوغۇل تەرەپ تەييارلىغان چاچقۇلارنى پەتنۇسقا  سېلىنغان چىرايلىق رەخت ئۈستۈگە تۆكۈپ ئارلاشتۇرۇلۇپ ئاندىن تۆرۈدە ئولتۇرغان ھۆرمەتكە لايىق ھاجىم بۈۋۈم ئاناشلار تەرىپىدىن سۆرۈلەر ئوقۇلۇپ بارلىق مېھمانلارغا تەكشى چېچىلىدۇ بۇ چاغدا مېھمانلار چاچقۇنى تالىشىپ تېرىشۋالىدۇ ،چۈنكى ئۆتمۈشتىن بۇيان چاچقۇدا چېچىلغان نىمەتلەرنى ئىستىمال قىلغان كىشىنىڭ چىشى ئاغرىماسمىش چىش ئاغرىقىغا دورا بولارمىش دىگەندەك ئەنئەنىلەر ساقلانغان ئىكەن .
چاچقۇ چېچىش ئاخرلىشىپ تەييار قىلىنغان پەتنۇستىكى چاي بىيىتنى باشلىماقچى بولغان مېھمانغا سۇنۇلدۇ.ھەشقاللا دەپ ئۇسۇزلۇقنى قۇلىدا تۇتقۇنىچە بىيىتنى باشلاپ كىتىشىدۇ.
بۇ يەردە ئىككى تەرەپتىكى ئاياللارنىڭ بىيىت ئېيتىشىشى بولسا ئۆزارا قوشاق ئېيتىشىپ كۆڭۈل ئىزھارلىرىنى بىلدۈرگەنلىك ھىساپلىندۇ.

مانا مەن يۇقۇرىدا دوسلارغا بايان قىلىپ  ئۆتكەنلىرىم رايۇنىمىزنىڭ قىسمەن جايلىرىدىكى ئۇيغۇرلارنىڭ توي تۆكۈن قائىدىلىرىدىن بىرى بولغان قىزنى سالامغا ئەپچىقىش ۋە ئىككى تەرەپ ئاياللىرىنىڭ بىيىت ئېيتىشىپ ئۆز كۆڭۈل خوشاللىقلىرىنى  ئىزھار قىلىشىدىغان قائىدە يوسۇنلىرىنىڭ بىرى ئىدى..........
ئەپسوس ئەمما ھازىرچۇ توي تۆكۈنلىرىمىز تەرەقىلىشىپ  توي تۆكۈنلەر رىستۇرانلاردىلا بولىدىغان بوپ كەتتى.ئۆرۈپ ئادەتلىرىمىز مەدىنىلىشىپ قائىدە يوسۇنلىرىمىز ئاددىلىشىپ تويلاردىكى مانا مۇشۇنداق قىزنى سالامغا ئېلىپ چىقىدىغان ئىككى تەرەپ بىيىت ئېيتىشىدىغان ئەنئەنىۋى ئۆرۈپ ئادەتلىرىمىزمۇ ئەمەلدىن قېلىشقا باشلىدى.
ئەلۋەتتە ئۆرۈپ ئادەتلىرىمىزنى داۋاملاشتۇرۇشۇمىز كىرەك ئىدى.سۇسلاشتۇرۋەتمەسلىك كىرەك ئىدى.ئۆزۈمىزنىڭ مىللى خاسلىقىمىزنى،يەرلىك ئالاھىدىلىكىمىزنى ئۇزۇندىن بۇيان داۋاملىشىپ كەلگەن ئۆرۈپ ئادەتلىرىمىزنى مەڭگۈ ساقلىشىمىز كىرەك ئىدى.

مەنبە: توي تۆكۈنلەردە كۆرگەنلىرىم ،ئاڭلىغانلىرىم ،شۇنداقلا ئوقۇغان كىتاپلىرىمدىن ئېسىمدە قالغانلىرىدىن  ئىبارەت..
مەسخىرە قىلمىغايسىلەر تۇنجى قېتىم يىزىپ بېقىشىم ئىدى.
