بىر كۈنى مەن سودا ئىشى بىلەن قوشنا ناھىيەدىك دوسلارنىڭ يىنىغا بارغان ئىدىم ،مەن ئىشلىرىمنى پۇتتۇرۇپ بولغاندىن كىيىن ئۇلارنى مەندىن رازى قىلىش ئۇچۇن شۇ ناھىيەدىكى بىر رىستۇرانغا تەكلىپ قىلدىم،بىز پەقەت تۆتلا ئىدۇق ،بىردەمدىلا تىلىفۇنلا بەس-بەستە سايراپ بىردەمدىلا خىلا كۆپ كىشىلىك بىر سورۇنغا ئايلاندۇق،تاماقلا چىقتى ،بەزى دوسلۇرۇمنىڭ بەكلا پو ئاتقىسى كەلدى بولغاي ئارىدىن تۇرغۇن ئىسىملىك بىر دوستۇمىز ئاغزى-ئاغزىغا تەگمەي سۆزلەشكە باشلىدى،(ئاغىنىلە مىنىڭ شۇنداق چىرايلىق قىزلىرىم باركى ئۇلارنىڭ گۈزەللىكىدىن بىھۇش بولىسىلەر) دەپ سۆزلەپلا كەتىتى،يىنىمدا ئولتۇرغان قادىر ئىسىملىك 40!~50ياشلا چامىسىدىكى بىر سودىدىكى ھەم شىرىك دوسترىمىز(ھە ئۇكا راس شۇنداق بولسا مىنىڭ بەكمۇ ھۇشۇمدىن كەتكۇم بار)سورۇندىن قاتتىق كۇلكە ئاۋازى كۆتۇرۇلدى،(ھە ئۇكا گەپ قىلمايسەنغۇ بايىقى گەپلىرىڭ يالغانمىدى يە)ھاھاھا .....تۇرغۇن(دەپ قاپسىز بىر تىلىفۇن قىلىدىغان بولسام ئۇلار خوشلىقىدىن ھازىرلا بۇ يەرگە ئۇنىدۇ) قادىر(شۇگىپىڭ راس بولسا تىلىفۇن قىلە راسلا ئۇ قىزلا بۇ يەرگە كەلسە بۇگۇن سىلەرنى ئۆزۇم مىھمان قىلىمەن)تۇرغۇن(ئوھوي مەرتلىكىڭىز تۇتۇپ قالغان ئوخشىمامدۇ بوپتۇ تىلىفۇن قىلساممۇ قىلاي لىكىن ۋەدىڭىزدە تۇرىسىزھە)قادىر(ئوھوي ئۇكا ئەركەك بىز  ئەرنىڭ ئىغزىدىن چىققان گەپ ئاتقان ئوق ) تۇرغۇن (بولىدۇ ئەمسە )ئۇلارغا قارىشىپ بىزمۇ قايسى پۇ ئىتىۋاتىدۇ ئايرىيالمايلا قالدۇق،قىنى بىردەم تۇرايلى ،قىنى قايسى دىگىنىنى قىلار دىگەن ئوي بىلەن 10~15مىنۇتلا ئەتراپىدا ئۇقىزلارنى ساقلىدۇق ،ئارىدا بىرى (باللا سىلەر مۇشۇنداق قىلىپ كۇنننى كەچ قىلامسىلە  نىمە؟ )  
          
      كەيپىيات

تۇرغۇن(بىردەم ساقلا ئاداش،،،ئوھوي گۈزەللىرىمىز كەلدى)دەپلا ئورنىدىن دەسلا تۇرۇپ كەتتى ،بىزمۇ تۇرۇپ ئۇلارنى قىزغىن كۇتىۋالدۇق،لىكىن نىمىشقىكىن قادىر ئىسىملىك بۇ دوستۇمىزنىڭ چىرايى ئۇڭۇپ ئۇنى ئىچىگە چۇشۇپلا كەتتى ،تۇرغۇن كۆڭلىدە بۇگۇنكى قىلىۋالغان پوچىلىقىغا خۇشال بولۇپ(ھە قادىركا شۇكلا بوكەتتىڭىزغۇ سۆزلىگەچ ئولتۇرمامسىز؟ ) قادىر يەنىلا سۇكۇت ئىچىدە بىشىنى تۆۋەن چۇشۇرۇپ نىمە دىيىشىنى بىلەلمەي قالغاندەكلا  قىلاتتى،بىزمۇ تۇزلا ئۆزى قىلغان ۋەدە ئەمدىلىكتە ئىغىر چۇشكەن ئوخشايدۇ ،ھاھاھا...قادىر يەنىلا ئۇندىمەي ئولتۇراتتى،لىكىن ئۇ قىزلارنىڭ ئىچىدە -18!19ياشلا چامىسىدىكى چاشلىرىنى قويۋەتكەن چىرايلىق بىر قىز خۇددى قادىرغا ئوخشاش ئولتۇراتتى،بىزمۇ ئۇلارغا قاراپ بولار ئوتتۇرسىدا نىمە ئىش بولغانلىقىنى ئاڭقىرالمايلا  قالدۇق،سورۇن مۇشۇ كەيىياتتا 20 مىنۇت ئەتراپىدا داۋاملاشتى،ئارىدىن بىر قىز چىقىپ بۇ جىمجىتلىقنى بۇزۇپ، تۇرغۇنجان بىزنى بۇيەرگە مۇشۇنداق ساتىراچخانىدا ئولتۇرغاندەك  ئولتۇرۇشقا چاقىرغان ئوخشىمامسىز تۇرغۇن(ئۇنداق دەپمۇ كەتمەڭ ئاغىنىلىرىم سىلەرنى كۆرۇپ خۇشاللىقىدىن نىمە دىيىشىنى بىلەمەي قالدى) بىز بىردەم سۇكۇتتىن كىيىن بىر-بىرىمىزنى چاقچاق ئارلاش تونۇشتۇرۇشقا باشلىدۇق،لىكىن قادىر بىلەن ئۇ قىز يەنىلا بىشىنى تۆۋەن سىلىپ ئولتۇراتتى،بىز تاماق يىگەچ قىزلار بىلەن پاراڭلاشقاچ ئولتۇردۇق،توساتتىنلا  قادىر ئىسىملىك بۇ كىشى ئورنىدىن تۇرۇپلا (ھىساۋاتنى قىلىڭە)دەپ كەينىگە ئۆرۇلۇپلا چىقىپ كەتتى ،باللا  قارىشىپلا  قالدى  ،بىزمۇ 10~15مىنۇتلا ئەتراپىدا نىمە ئىش بولغانلىقىنى بىلەلمەي تۇرۇپ قالدۇق ،قادىر چىقىپ كەتكەندىن كىيىن ھىلىقى چاشلىرىنى قويۋەتكەن چىرايلىق قىزمۇ كۆزلىرىگە ياش ئالغىنىچە يۇگۇرۇپ چىقىپ كەتتى ،تۇرغۇن(ئادىلە ئادىلە)دەپ توۋلىغىنىچە ئۇ قىزنىڭ ئارقىسىدىن يۇگۇردى،شۇنداق قىلىپ بىز ئۇلارغا نىمە بولغىنىنى چوشىنەلمىگەن ئەھۋال ئاستىدا بىز ئۇلارنى يىرىم سائەت ئەتراپىدا ساقلىدۇق،يىرىم سائەتتىن كىيىن تۇرغۇن بىشىنى تۆۋەن ساڭگىلىتىپ بىر تال تاماكىنى تۇتاشتۇرۇپ چەككەچ كىرىپ كەلدى ،بىز ئۇنىڭدىن  نىمە بولغىنىنى سوراپ بولغىچە ئۇ سۆزلەپ چۇشۇپ كەتتى.
             ئىبرەت
(توۋا   بۇ دۇنيا نىمە بوپ كىتىۋاتىدۇ ،مەن بونداق بولار دەپ ئويلاپ باقمىغان ئىدىم كىم بىلسۇن،)دەپ بىر پەس توختىدى،مەن (تۇرغۇن ئاداش نىمە ئىش بوپ كەتتى)دەپ سورىۋىدىم تۇرغۇن(تۆۋەنچىلىك بىلەن مەن بەكمۇ خىجالاتچىكىتە قالدىم،ئەگەر بۇنداق بولىشىنى بىلسەم ئۇلارغا تىلىفۇن قىلارمىدىم )دەپ بىر پەس توختىغاندىن كىيىن  ئەسلى ھىلىقى قادىر ئاكا بىلەن ھىلىقى چاشلىرىنى قويۋەتكەن چىرايلىق ئادىلە ئۇنىڭ قىزى ئىكەنلىكىنى ئىيىتتى ،ھەممىمىزنىڭ ئۇنىڭدىن بۇ گىپىنىڭ راس ئىكەنلىكىگە ئىشەنمەيلا قالدۇق، لىكىن بۇ ئەمىليەتتە ،بۇنى كىمدىن كۆرگۈلۈك
  مەن ئۇچۇن بۇ قىتىملىق تىجارەت ۋە بۇ بىر ۋاق لىق تاماق ماڭا قاتتىق  ئىبرەت بولدى ،بۇ ئىشتىن كىيىن ئۇ قوشنا ناھىيەگە بارماس بولدۇم ھەم قىزلارنى كۆرسەم ئۇلارغا نەسىھەت قىلغۇم كىلىدىغان بوپ قالدىم.

دۇنيادا ھەر خىل ئىشلار بولىدۇ . ئىسلامسىز ھايات شۇنىڭدەك پەۋقۇلادە رەسۋاچىلىقلارغا يىتەكلەيدۇ .
مەنبەنى ئۆزەم دىمەي يەنە كىم دىسە بولاتتى ؟ ئىبرەت ئېلىڭ دەپ بىر ياخشى تىما يېزىپ بەرسە ئۆزىنى ئۆزى رەسۋا قىلغانلىق بولامدا ؟
بۈگۈنكى جەمىيەتتە شۇنىڭدەك پەۋقۇلادە تېراگىدىيەلەر يۈز بەرمەيدۇ دەپ كىم ئېيتالايدۇ ؟
نۇرغۇنلىغان بېشەم كىيمىگە ئورىنىۋالغان كىشىلەر شۇنىڭدەك ئىشلارنى ئۆز ھاياتىدا ياكى تىما ئىگىسىدەك ئۆز سۇرىنىدا كۆرسە تۈزىلىنىش ئەلۋەتتە ياخشى بولاتتى . ئۆزىمۇ ئېيتتىغۇ ئەنە ‹‹ شۇ يەرگە ئۆتمەس بولدۇم ›› دەپ .
ياخشى تىما . مۇشىنىڭدەك پەۋقۇلادە تىمىلارنىڭ كۆپلەپ چىقىشىنى كۈتىمىز ۋە شۇ خىل يازمىلار بىلەن گەپ ئۆتمەس قاشاڭلارغا تاياق - توقماق بىرىمىز !
ئۇلارنىڭ ھالى ئەنە شۇ دەرىجىگە يەتكەنلىكتىن !
سۆزلىرىمنى چۈشەنمىسىڭىز چاپسانلا ئىنكار قىلماڭ . بىر بۇلۇڭغا قويۇڭكى ، بىر ۋاقىتلاردا پايدىسى تىگىدۇ !
