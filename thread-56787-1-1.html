                                                           رەھمەت ساڭا ئۇيغۇر مۇنبىرى




                                                         ئاپتور:تاتلىق ئۇيغۇر




    ئىچكىردىكى ۋاقىتلىرىم بولۇپ، شۇ جايدىكى مەن بىلەن تەڭ قۇرداش ئۇيغۇر دوستلىرىمنىڭ قىزىقتۇرشى، بىلەن تورخانىغا چىقىشقا قىزىقىپ قالغان ئىدىم . دوستلىرىم بىلەن نۇرغۇن ئويۇنلارنى ئوينايتۇق. بەنتىيو، خۇڭجىڭ، دىگەندەك بىرلىشىپ ئوينايدىغان ئويۇنلانى، بەكلا قىزىقىپ ئويناپ كېتەتتۇق. بەزىدە خەنزۇ دوسلار بىلەن ئويناپ يېڭۋالغان، ۋاقىتلىرىمىزدا ئەكىلە ئاداش قولۇڭنى، دەپ كۈرىشىۋىتەتتۇق، توۋا بىزنىڭ شۇ كۈنلىرىمىز شۇنداق مەنىلىك ئۆتەتتى. بىز بىر ئاكاشنىڭ كاۋاپخانىسىدا خىزمەت قىلاتتۇق ئۇ ئاكاشنىڭ بىر ئوغلى بار بولۇپ، ئاشۇ بالا مېنى تورخانىغا باشلاپ ،ئەكىرەتتى. بۇ ئادىشىم ماڭا بەكلا كۈينەتتى. مەنمۇ ئۇنۇڭغا نېمىلا يىسەم ئېلىپ قوياتتىم، چۈنكى. ئۇ يەردىكى مېنىڭ ئەڭ يېقىن، دوستۇم ئاشۇ ئىدى. بىز كىچىلىرى، ئۆيدىكىلەر ئۇخلاپ بولغان چاغدا ئوغۇرلۇقچە، تورخانىغا چىقىپ كىچىچە، ئوينايتۇق. تاڭ ئاتقاندا بولسا ،بىز يۈگۈرۈپ مەشىق قىلىپ كەلدۇق. دەپ ئالداپ قوياتتۇق
كۈنلىرىمىز ،شۇ تەرىقىدە ئۈتىۋاتاتتى،مەلۇم بىر كۈنى بېيجىڭدىن بىر ھەدىمىز كەلدى. ئۇ ئاچاشمۇ تورغا چىقىدىغان بولۇپ، بىز بىر تورخانىغا، باشلاپ ئەكىردۇق. مەن دېدىم:
   - ھەدە قايسى ئويۇننى ،ئوينايمىز
     -مەن ئۇيۇن ئويناشنى بىلمەيمەن. سىلەر ئويناڭلار،مەن بىردەم ناخشا ئاڭلاي،- دېدى .
    -ماقۇل ،-دەپ بىز بۇرۇنقىدەكلا ئۇينىمىزغا كىرىپ كەتتۇق . بىر چاغدا:
  - قانداق ناخشا ئاڭلاۋاتسىز؟ -دەپ يېنىغا باردىم، باردىميۇ كۆزلىرىمگە ئىشەنمەيلا قالغان ئىدىم. مېنىڭ كۆرگىنىم بىزنىڭ ئۇيغۇر يېزىقى،ئۇيغۇر خەلقىمىزنىڭ داڭلىق ،ناخشا چولپانلىرى،مەن ناخشىلىرىنى ئەڭ ياختۇردىغان .ئالىمجان تۇرسۇن،ئەخمەت ئېلى دىگەندەك.
    ناخشىچىلانىڭ سۆرەتلىرى، ناخشىنىڭ ئىسىملىرى تۇراتتى. مەن راسلا ھەيران ئىدىم خوشاللىقىمدىن ۋاي ھەدە بۇنى قانداق چىقاردىڭىز ؟ ماڭىمۇ ئۆگۈتۈپ قۇيۇڭ دەپ تۇرۇۋالدىم. شۇنداق قىلىپ ماڭا ئىزباسار تورىنىڭ ئادىرسىنى ،تېلفۇن دەپتىرىمگە پىنيىنچە يېزىپ بەردى .مەن ھەرقانچە قىپمۇ ئۇنداق چىقىرالمىدىم،ئۇ ئاچاش ماڭا ئەستايىدىللىق بىلەن ئۆگۈتىشكە باشلىدى. مەن پۈتۈن ھۇجۇدۇم بىلەن ئۆگۈنىۋاتاتتىم توۋا ما ئىشنى مەن تېخى ئۇيۇنلا، چىقىدىغان بۇ نەرسگە،ئۈزەمچە بەك ئۇستا دەپ قاراپ تىكەنمەن،لېكىن مەن بۇنداق ناخشا،كىنو كۆرگىلى بۇلىدۇ دەپ ئويلاپ باقمىغان ئىدىم. شۇنداق قىلىپ تورخانىغا قىزىقىشىم ھەسىلەپ ئاشقان ئىدى، مەن ھەركۈنى تورغا چىقىپ ناخشا ئىتوت كۈرەتتىم يەنە تېخى نۇرغۇن
توررلانىڭ ئادىرسىنى ،تېلفۇن دەپتىرىمگە يېزىۋالغان ئىدىم . كۈنلەرنىڭ ئۆتىشى بىلەن تورغا چىقىشقىمۇ خېلى ئۇستا بوپ قالغان ئىدىم. بىر كۈنى يەنە باشقا شەھەردىن، بىرنەچچە مەن بىلەن تەڭ دېمەتلىك باللار كەلدى، ئۇلار بىلەنمۇ تورخانىغا چىقماقچى بۇلۇپ يول ئالدۇق. مەن دوستۇم بىلەن ئىككىمىزنىڭ تورغا چىقىپ ناخشا چىقىرالايدىغىنىمىزنى، يەنە ئۇيۇنغىمۇ بەك ئۇستىلىغىمىزنى توختىماي دىدۇق كىرگەندە ئۇلار بىلەن ئۇيۇندا بەسلىشىپ باقماقچى بولدۇق، تورخانىغىمۇ يىتىپ كەلدۇق ، شۇنداق قىلىپ مەن كومپىيوتېر قوزغۇلۇپ بولغاندن كىيىن ناخشا چىقىرىشقا كىرىشتىم، ناخشىنى چىقىرىپ بۇلۇپ ھە باللارئەمسە بەنتىيونى مەن مەيدان قۇراي دەپ قۇيۇپ مەيدان،قۇرۇشقا كىرىشتىم ئۇ باللامۇ تېزلا كىردى مەن دوستۇم بىلەن ساخچى بولدۇم شۇنداق قىلىپ كەسكىن بىر مەيدان ئۇيۇن باشلاندى. بىز دەرۋەقە ئۇتۇپ چىقتۇق بىردەم داۋاملاشقاندىن كېيىن ئوينىماسلىقنى قارار قىلدۇق مەن يەنىلا تورغا چىقىپ ناخشامنى ئاگلاپ كەتتىم. بىرچاغدا مەن بىلەن بىرىگە كەلگەن بىرسىگە قارسام ھە دەپ تىڭشىغۇچتا پاراڭلىشىۋاتقىنى كۆردۈم مەن ئەزەلدىن ئۇنداق پاراڭلىشىپ باقمىغان ئىدىم ، مەن بېرىپ نىمىش قىلۋاتىسلە؟ دەپ سوردۇم توردا پاراڭلىشىۋاتىمەن دېدى، ئۇ ۋۇي ماڭىمۇ ئۆگۈتۇپ قۇيۇڭلار ئاداش دېدىم ماقۇل چاتاق يوق دەپ ھەممىنى ئۈچۈرىۋېتىپ باشتىن كىرىشكە باشلىدى. مەن يېنىدا قاراپ تۇردۇم، ئۇ ئاداش مەن كۈندە بىرنەچچە رەت كىرىپ ناخشا ئاڭلايدىغان شەبنەم تورىغا چىقتى مەن دائىم كۈرۈپ تۇردىغان چايخانا دېگەن بوشلۇققا شوخ-شوخ0997دەپ پىنيىنچە كىرگۈزدى، مەخپى نۇمۇرغا 123 دەپ توشتۇرۇپ كىرىش دېگەن يەرنى، باسقاندىن كېيىن بىردەمدىن كېيىن مەن چۈشەنمەيدىغان نۇرغۇن پىنيىنچە خەتلەر ئاستىدىن ئۈستىگە چىقىۋاتاتتى. ئۇ ئاداش ماڭا تەپسىلى چۈشەندۇرۇشكە باشلىدى.مەنمۇ خېلىلا چۈشنۈپ قالغان ئىدىم مەن شۇنداق قىلىپ كۈندە چايخانىغا كىرىپ باللار بىلەن پاراڭلىشىدىغان، تېخى چايخانىلار ئارا خېلى نامىم چىقىپ قالغان ئىدى. بىر كۈنى بىر باللا بىلەن پاراڭلىشىپ قالدىم ئۇ بالىمۇ ئىچكىردە بۇلۇپ دوست بۇلۇپ ،قېلىشنى ئېيتى، مەن ئەلۋەتتە قۇشۇلدۇم، چۈنكى مەن ئۇيەردە ئۇيغۇرغا زار ئىدىم. ئۇ باللا مەندىن QQرىم بار يوقلىقىنى ئەگەر بولسا قۇشۇۋېلىشىمنى ئېيتتى. مەن QQنىڭ نىمە ئىكەنلىكىنىمۇ، بىلمەيتىم مەن ئۇ توردىكى دوستۇمدىن قىزىقىپ سورىدىم ماڭا ئۇ دوستۇم QQنۇمۇر بەردى،مەن باشقۇرغۇچىنى چاقىرىپ ئاران تەستە ئاشتىم. ئېچىشىمغىلا بىرسى خەت يازدى ئۇ خەتتە مۇنداق دېگەن ئىدى. بۇ مەن ئاداش بىز بىر- بىرمىزنى كۈرەيلى دېدى. مەن ھەيران قالدىم، قاندق گەپ قىلۋاتىدۇ؟ ماۋۇ ئاداش دەپ تۇرشۇمغىلا،بىر كامىرانىڭ سۇرىتى چىقىپ تېلفۇندەك ئاۋاز كەلدى. ئۇ ئاداش ئېلىڭلا دەپ خەت يازغان ئىدى ، كىچىك كامىرانى ،باستىم بىردەمدىن كېيىن بىر ئۇيغۇر بالا ئىكرانىمدا كۈرۈلدى،بۇ خىل خوشاللىقىمنى خەت بىلەن تەسۋىرلەشكە ئاجىز مەن ئىشقىلىپ مەندەك، بىر ئۇيغۇرغا زار بىر بالىغا بۇ ھەقىقەتەن بىر خوش بولغۇدەك ئىش ئىدى. ئۇ ئاداش بىلەن خېلى ئۇزۇن پاراڭلىشىپ كەتتۇق،شۇنداق قىلىپ ئۇ دوستۇمنىڭ تۈرۈتكىسىدە،چ چ نىمۇ خىلى ئۇبدانلا چۈشۈنۈپ قالدىم. چايخانىلارغا كىرىپ مەن ياختۇردىغان،دوستلىرىمنى قاچىلدىم شۇنداق قىلىپ تورغا چىقمىسام كۈنۈم ئۆتمەيدىغان، دەرىجىگە بارغان ئىدى. كۇنلىرىم،شۇ تاپتا لوبىنىمدىن گەپ ئاڭلاپ ئۆتمەكتە ئىدى .ماڭا داۋاملىق نەسىھەت قىلاتتى ئۇبدان ئۇكام سەن تورغا بەكلا بېرىلىپ كەتتىڭ بۇ دېگىنىم كىرمە دېگىنىم ئەمەس ،سېنى چۈشىنىمەن بىزنىڭ بۇ ئۆيدە بەقەت 5لا، ئادەم بار سىنىڭمۇ باشقا ئۇيغۇرلار بىلەن كۆرۈشكۈڭ پاراڭلاشقىڭ كېلىدۇ. لېكىن ئۇكام سەن كىچىك سېنىڭ كىچىك كاللاڭ يوغان قاقۋاش كاللاڭ بوپقالمىسۇن دەيتى. مەن يەنىلا قىلدىغىنىمنى قىلىۋىرەتتىم كۈندە QQرىمگە ئۇيغۇر قىزلانىڭ نۇمۇرنى قاچىلاپ پاراڭلىشىپ ئوينايتىم بىر قىزنى ئىكراندا كۆرگەن كۈنىسى،ئېقىتماي- تېمىتماي دوستۇمغا دەپ ئۇ قىزنىڭ چىرايىنى تەنتەنە قىلىپ ماختاپ كېتەتتىم. تېخى بەزىدە چىرايلىق خەنزۇ قىزلىرنى كۆرەپ قالسام توردا كۆرگەن چىرايلىق ئۇيغۇر قىزلا كۆز ئالدىمغا كىلىپ يەنىلا بىز ئۇيغۇرلانىڭ قىزلىرىگە يەتمەيدىكەنغۇ دەپ ئويلاپ كىتەتتىم. بۇ دېمىسەممۇ راس ھەقىقەتقۇ؟ بىزنىڭ ئۇيغۇر قىزلىرى ھەقىقەتەن چىرايلىق گۈزەل شۇڭا بىز ئۇيغۇر يىگىتلىرى ھىندىستانلىق كورىيەلىك،چولپان قىزلانىمۇ ،ئۇيغۇر قىزلىرىغا تەڭ قىلمايمىز ئەمەسمۇ.شۇنداق قىلىپ ئەش-پەش دېگۈچە بىر يېرىم يىل ئۆتۈپ كەتتى.مەن شىنجاڭغا چىقماقچى بولدۇم دوستۇم بىلەن بىرگە چىقتۇق ،ئۈرۈمچىگە كەلگۈچە كۆزىمىز تۆت ئىدى ياراتقان ئۇلۇغ ئاللانىڭ ياردىمىدە تىنىچ ئامان ئۈرۈمچىگىمۇ چۈشتۇق بۇ خوشاللىقىمغا ھىچ نىمە توغرا كەلمەيتى ھەم كەلتۈرەلمەيتىم.قايسى تەرەپكىلا قارىسام ئۇيغۇر، نەرسەكېرەك ساتقۇچىلاردىن، تارتىپ ساقچىلارغىچە ئۇيغۇر ئىدى يۈركۈم شۇنداق ئىللىدى . بۇ مىنىڭ يۇرتۇم شىنجاڭ ھە؟ دەپ سورايتىم ئۈزەمدىن شۇنداق بۇ مىنىڭ يۇرتۇمنىڭ مەركىزى ئىدى. ھەي توۋا بۇرۇن يۇرتۇمنىڭ بۇنچىلىك ئىللىق بىر ماكان ئىكەنلىكىنى ھەجەپ بىلمەپتىكەنمەن دېسە ئەمدى راسلا قەدىرگە يەتكەن ئىدىم. شۇنداق مىللىتىمنىڭ قەدىرگە تېنىمدىكى يۈرىكىم بىلەن يەتكەن ئىدىم . مەن بىر كۈندىن كېيىن ئۆزەمنىڭ كىندىك قېنى تۈكۈلگەن يۇرتۇمغا يولغا چىقتىم يۇرتۇمغا چۈشۇپ ئۆيدىكىلەر بىلەن يىغلاپ تۇرۇپ كۆرەشتىم بىچارە ئاپام مېنىڭ دەردىمدە كىچىككىنە بىر تېرە بىرئۇستىخان بۇلۇپ قاپتۇ يۈركۈمدىن ئاپامنى چىڭ قۇچاقلاپ تۇرۇپ جېنىم ئاپا بولدى يىغلىمىغىن مەن كەلسەم خوشال بولماي يىغلامسەن، دەپ ئاپامنىڭ يىغلىماسلىقىنى ئۈتۈندۈم شۇنداق قىلىپ ئۇرۇق تۇققانلارنى دوست بۇرادەرلىرىمنى يوقلاش بىلەن بىرنەنچە ھەپتە ئۆتۈپ كەتتى. مەن يەنىلا تور خانىدىن،مېھرىمنى ئۈزەلمىگەن ئىدىم. مەن ئۈيۈمگە يېقىن بىر تورخانىغا كىرۋېلىپ شۇ تورخانىغا كىرگەن قىزلانىڭ QQنۇمۇرنى قاچىلاپ خەت يېزىپ ئەخمەق بۇلۇپ بەزىدە ئەخمەق قىلىپ ئوينايتىم كۈنلىرىم تور سارىڭى ھالىتىدە داۋام قىلۋاتاتتى. مېنىڭ بۇ كاللامدا نە تۇرمۇش نە خىزمەت نە ئارزۇ ئارمان يوق ئىدى. ئويلايدىغىنىم مۇشۇ تورخانا مېنىڭ مەڭگۈ ئايرىلماس بىر قىسمىم ئىدى. مانا بۇ مېنىڭ باسقان خاتا يۇلۇم ئىدى. كۈنلەرنىڭ ئۆتىشىگە ئەگىشىپ ئۇيغۇر تورچىلىغىمۇ بارا - بارا تەرەققى قىلىشقا ئەگىشىپ مەزمۇنلۇق پايدىلىق توربەتلەر بارلىققا كېلىشكە باشلىدى. ....،شەبنەم،تەۋپىق،يەنە مەن ئەڭ ياختۇردىغان كومپىيوتېر ھەۋەسكارلىرى يىغىلىدىغان ئىنتىل سەيناسى دېگەندەك نۇرغۇن،توربەتلەر مۇنبەرنى يولغا قۇيۇپ نۇرغۇن مەندەك ئۇيۇننىڭ ۋەس-ۋەس چىلىگىدە ئۆزىنىڭ بىركىشلىك ئورنىنى تاپالمايۋاتقان ياشلارغا تاغدام يولنى كۆرسىتىپ بەردى. مۇنبەردە مىللەت ،ئاتا-ئانا ،تارىخ،مۇھەببەت،تۇرمۇش، يەنە نۇرغۇنلىغان مۇنازىرە ئۆتكۈزۈلدىغان بۇلۇپ مۇنبەردىكى پىكىرى ئۆتكۈر كۆز قارشى توغرا ياشلار بىلەن پىكىر ئاملاشتۇرۇشقا قىزىقىپ قالدىم شۇنداق قىلىپ تىرشىپ تىرمىشىم، ئۇلانىڭ QQ نۇمۇرنى قاچىلاپ ئۇلار بىلەن پىكىرلىشىپ دوست بۇلۇپ ئۈتەتتىم. شۇنداق قىلىپ مېنىڭمۇ تۇرمۇشۇم بارا - بارا ئۆزگىرىشكە باشلىدى.
مەن كەسپىمنى تىجارەتكە قاراتقان ئىدىم. ھازىرقى كۈنلەردە تىجارەت قىلىپ يول تاپماق تەس بولسىمۇ ياراتقان ئۇلۇغ ئاللا بىر يولغا سالىدىكەن، ئۆزەمنىڭ گەپتانلىقى بىلەن تىجارەتنى خېلىلا ماڭغۇزۇپ كەتتىم. مانا مۇشۇنداق بىزگە جەمىيەتنى ئەينەك قىلىپ كۆرسىتىپ بېرىۋاتقان ئۇيغۇر تورچىلىغى ئۈچۈن توختىماي تىرشىۋاتقان، ئىزدىنۋاتقان جەسۇر پىكىرى ئۆتكۈر ياشلىرىمىزغا چىن كۆڭلۈمدىن رەخمەت ئېتىمەن.
   مېنىڭ بۇ يازمام بەكلا چېچىلاڭغۇ،چىقىپ قالغان بۇلۇشى مۇمكىن، مەن پەقەت باشلانغۇچنىڭ ئىككىنچى يىللىغىنى پۈتتۈرگەن. بۇ دېگىنىم مەن ساۋاتسىز دېمەكچى ئەمەس.مەن تېرىشىمەن مەنمۇ مۇشۇ تورنىڭ پايدىسى بىلەن،تۇرمۇشنى جەمىيەتنى چۈشەندىم، مىنىڭمۇ مۇشۇ يازمام مەندەك بىر چىقىش يۇلىنى تاپالماي قالغان، بىر ئادەمگە بولسىمۇ تەسىر قىلغان بوسىكەن دېگەن،ئۈمۈتتەمەن.

ئاخىردا دەيدىغىنىم،ئۇيغۇر تورچىلىغىدىكى ماڭا نۇرغۇن پايدىلىق يولنى كۆرسىتىپ بەرگەن مۇنبەر ئەزالىرىغا كۆپ رەخمەت.

بۇھەقىقى ئىش، ئازراق ئۆزگرىپ كەتكەن بۇلۇشى مۈمكىن. بەزى يەلىرى ئىسىمدىن چىقىپ كىتىپتۇ،كەچۇرگەيسىلەر.
