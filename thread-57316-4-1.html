ئەپەندىم!!!
بۇخىل كىسەلنى داۋالىماق سىز دىگەندەك ئۇنداق ئاسان ئەمەس
شۇ بىماردىكى كىسەللىكنىڭ پەيدا بولىشىغا،تەرەققىياتىغا،ئىغىر-يىنىكلىكىگە،قايسى ئۇمۇرۇتقىدا قانداق شەكىلدە چىقىپ كەتكەنلىكىگە،كىسەللىك تەسىر دىن بولغان ئەگەشمە كىسەللىكىگە،تەسىر قىلغان نىرىپ ،توقۇلما،پەي-مۇسكۇل،يۇلۇن،قىسىملىرىغا قاراپ قانداق داۋالاش كېرەكلىكىگە قاراڭ  سىز بەلكىم ناھايىتى ئاددى خىلدىكى بەل ئۇمۇرۇتقا تەخسىسى چىقىپ كىتىش كىسىلى  ئۇستىدە پىكىر يۇرگۇزگەن داۋالىغان بولۇشىڭىز مومكىن سىز مەنچە بۇ كىسەلنى ھەقىقى تۇر دە تەتقىق قالغان بولسىڭىز يۇقاردىكى ئۇ يەكۇننى چىقارمىغان بولاتتىڭىز قالايمىقان بىلجىرلىمايتىڭىز ئۇمۇمىيلىقنى كۆزدە تۇتۇپ پىكىر بىرىڭ بىر نەرسىگە قارىغۇلاچە ئىشىنىپ ئېسىلىۋالماڭ.(بۇ دۇنيادا بەزىدە ئامال بولمىغاندا شۇنداق ناننى بەلگە تاڭىدغان ئىشمۇ با)!!
ئەگەر سىزنىڭ راستىلا شۇنداق كارامىتىڭىز بولسا مەنمۇ abdukorax قېرندىشىمىزغا ئوخشاش سىزگە ئەگىشىمەن دوختۇرخانا ئېچىپ ئاۋارە بولغىچە  سىزنىڭ مەھسۇلاتلىرىڭىزنىلا ئىشلىتىمەن بۇ كىسەلنى 12يىلدىن بېرى داۋالاپ كىلۋاتمەن سىزدەك ........(ئاھ  قېرىداشلىرىم  مىنى كەچۇرۇڭلا ! سىلەر كېسەلىك ئازاپدا ئڭراۋاتساڭلا مەن كارىم يوق مەسۇليۇتۇمنى ئۇنتۇپ نەلەردە يۇرۇپتىمەن، ) بۇنداق مللەتنى خەلىقنى سۆيدىغان خەلىقنىڭ ھالىغا يىتىدىغان، ۋەتەنپەرۋەر كىشىمىزدىن تېخنىمۇ پەخىرلىنىمىشىمىز ئۇچۇن سىناقنى بۇگۇنلا ھازىرلا باشلايلى
مۇشۇ تېمىدىكى پۇتۇن تورداشلار گۇۋاچى بولسۇن مىنىڭ دوختۇرخانام بەك مۇۋاپىق سىناق نوقتىسى بولالايدۇ چۇنكى مەن مەخسۇس مۇشۇ خىلدىكى كىسەللىكلەرنى داۋالايمەن كىسەلدىگەن بەك كۆپ خالىغانچە سىناق قىلىپ بىر كۆزىمىزنى ئېچىپ قويۇڭ پۇتۇن چىقىمغا مەن مەسئۇل.....
سىز دىگەندەك بولمىسا مەنلا ئەمەس پۇتۇن تورداشلا ر.ئالدىدا شەرمەندە بولسىز......
دوسقا زار، دۈشمەنگە خار، نامەرىتكە ئىنتىزار بولما ئۇيغۇر!!!

ئەسسلامۇ  ئەلەيكۇم؛abdukorax،arhu ئەپەندىلەر ئۇلۇق اللە نىڭ سالامى رەھمىتى سىلەرىگە بولسۇن! شۇ بىماردىكى كىسەللىكنىڭ پەيدا بولىشىغا،تەرەققىياتىغا،ئىغىر-يىنىكلىكىگە،قايسى ئۇمۇرۇتقىدا قانداق شەكىلدە چىقىپ كەتكەنلىكىگە،كىسەللىك تەسىر دىن بولغان ئەگەشمە كىسەللىكىگە تەسىر قىلغان نىرىپ ،توقۇلما، پەي-مۇسكۇل،يۇلۇن قىسىملىرىغا قاراپ قانداق ئۇزۇقلۇق يىتشمەي ماس ھالەتتە بىر - بىرىگە تەسىر كورسەتتكەن ؟ نىمە ئۇچۇن قېرىنداشلىرىمىز شۇنچىۋالا بەدەنلەرنى تولەپمۇ بۇ كېسەلىك ئۇسىتىدىن غالىپ كېلەلمەيدۇ ؟ نىمە ئۇچۇن، ئازاتلىقىتىن كىينكى 20 يىلنىڭ مابەينىدە دۇرا ھۇكۇلنىڭ دوختۇرخانلارنىڭ تەرەقىياتى شۇنچە زور بولسىمۇ بۇ مەسلنى ھەل قىلالماي شۇنچە ئېغىزلىشىپ كېىتدۇ ؟ مەيلى بەل ئۇمۇرتقا سوڭەك كېسىلى بولسۇن. مەيلى ياللۇغ كېسەلىكلىرى بولسۇن. مەيلى ئەر -ئايالار كېسەلىكلىرى بولسۇن. ۋە باشقا راك- ئوسمە يۇرەك مېڭە قان تۇمۇر كېسلى بولسۇن. ئوخشاشلا زۇرۇر ئۇزۇقلۇق يىتشمەسلىك زۇرۇر بولمغان ئۇزۇقنىڭ ئشىپ كىتىش تۇپەيلى تەسىر كورسەتكەنلىگنى ھىس قىلدىم. مەنمۇ بىر سىلەردەك باشقىلارنى داۋالايدىغان ئادىغىنە دوختۇر ئدىم تۇساتىن سالامەتلىگىمدىن مەسلە كۆرۇلدى. بۇ ئادىغنە بىر ياللۇغ كېسەل ئدى. تىبىبى ئۇنۋىرستىتى تارمىغدىكى بىرنچى دوختۇرخانغا بېرىپ ئۇلارنىڭ داۋالشنى قۇبۇل قىلدىم، ئۇيەردە 6 ئاي داۋالانسا قانچەلىك ئختسات كېتىدغالغنى بۇلسىلە، لىكىن مەن ياخشى بۇلۇش ئۇياقتا تۇرسۇن بىر كېسەل بەچىچىگە كوپەيدى، ھەتتا بىر چېلەك سۇ كوتەگسىز ھالغا بېرىپ ئۇرۇن تۇتۇپ يېتىپ قالدىم، ياپ -ياش تۇرۇپ، شۇ ۋاقىتلاردا بىر قېرىندىشمىز  ماڭا ئۇزۇق ئارقىلىق دالۋنىپ بېقشمنى تەۋسيە قىلغاندا مەن پەقەت قۇبۇل قىلمغان كېسەلنى دۇرا ھۇكۇل داۋالايدۇ دەپ چۇنكى مەنمۇ بىر ھاماقات تىببى خادىم ئدىم. ئاخرى بىرى ماڭا ئەگەر ياخشى بولامساڭلار پۇلۇڭلارنى قايتۇرۇپ بېرمەن دىگەندە قۇبۇل قىلدىم. بىر قۇتا ئاقسىل، بىر قۇتا كالتسى، بىر قۇتا ۋىتامىن   vb بىر قۇتا ۋىتامىن  vc يەپ، مەندىكى كېسەلىكلەر ھەل بولدى. شۇ چاغدا توختا ئۇزۇق بەدەنگە كىرىگەندە قانداق رول ئوينايدۇ؟ دۇرا ھۇكۇلچۇ ؟ دىگەن ساھەدە ئزدەندىم. ئاخرى سەۋەبنى تاپىقاندەك بولدۇم. مەندەك كېسەل ئازابى تاتۋاتقان تالاي قېرنداشلىرم ئۇچۇن ئاخرى ئۇز كەسپىمنى تاشلاپ ئۇزۇقلۇق كەسپى بىلەن شۇغۇلاندىم، ھەمدە ئۇزۇقلۇق ھەقىدە نۇرغۇن ۋاقىت تەتقىقات ئشلىرى بىلەن شۇغۇلاندىم، ھەمدە قانداق داۋالاش كېرەك قانداق ئالدنى ئالساق كېسەلىك ئۇستىدىن غەلبە قىلمىز  دەپ خەلقىم ئۇچۇن تىرشۋاتمەن مىنى ئەيىپلىمەڭلا دوسلار !، مەن سىلەردىن بەكمۇ خوش بولدۇم بۇگۇن . مەن ھاياتمدا خەلقىم ئۇچۇن ياشاپ كەلگەن بۇ ھاياتمدا ئۇلارنىڭ مۇڭ زارى بولامدىم، ئەمدى بولسىمۇ ئۇلارنىڭ جسمانى جاراھتنى داۋالاپ زىمسدكى يۇكنى ئازلىتاي دەپ تىرشچالىق كوستۋاتمەن. بىر ئش ئسڭلاردا بولسۇن مەن ساتقان مەسلاتىمغا  ئشنمەن چۇنكى مىنىڭ مەسىۇلاتىم خەلقىمنىڭ  ئالدىدا يەرگە قاراتمايدۇ، لىكىن كەچۇرۇڭلا تۇرغا چىقىرشىمغا بومايدۇ شىركەتنىڭ بەلگلمىسى شۇنداق
