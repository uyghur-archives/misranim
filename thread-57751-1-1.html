ماركىلىق كومپىيوتېر بىلەن قۇراشتۇرما كومپىيوتېرنىڭ قانداق پەرقى بار؟
قۇراشتۇرۇلغان كومپىيوتېرنىڭ باھاسى بىر قەدەر تۆۋەن، ماركىلىق كومپىيوتىرنىڭ سەپلىمىلىرى ماس، ئىقتىدارى مۇقىم بۇلىدۇ. ئۇنداقتا ماركىلىق كومپىيوتىر بىلەن قۇراشتۇرۇلغان كومپىيوتىرنى قانداق پەرىقلەندۈرىمىز.
ئالدى بىلەن بۇ ئىككى ئۇقۇمنى چۈشۈنەيلى
قۇراشتۇرۇلغان كومپىيوتىر: CPU، ئاساسى تاختا، ئىچكى ساقلىغۇچ، قاتتىق دىسكا، كۆرسەتكۈچ، دېسكا ئوقۇغۇچ، كومپىيوتىر قېپى، توك مەنبەسى، كونۇپكا تاختىسى، مائۇس، ئىكران قاتارلىق سەپلىمىلەردىن تۈزۈلگەن بولۇپ، قۇراشتۇرما كومپىيوتىرنى ئۆزى خالىغانچە تاللاپ قۇراشتۇرغىلى بولىدۇ.
ماركىلىق كومپىيوتىر: مەلۇم بىر ماركىدىكى كومپىيوتىر بولۇپ، شىركەت ياكى زاۋۇت تەرپىدىن قۇراشتۇرۇلغان بۇلۇپ، ماسلىشىشچانلىققا ئىگە، ھەمدە سىرتقا پۈتۈن سەپلىمىسى بىلەن سېتىلىدىغان كومپىيوتىرنى ماركىلىق كومپىيوتىر دەيمىز. ئۇنىڭ سۈپىتىگە كاپالەتلىك قىلنىدۇ، ھەمدە سېتىشتىن كىيىنىكى مۇلازىمەتتىن تۇلۇق بەھىرمەن بولغىلى بولىدۇ.
بۇ ئىككى خىل كومپىيوتىرنىڭ قانداق پەرقى بار
1. باھاسى:
ماركىلىق كومپىيوتىرنىڭ باھاسى قۇراشتۇرۇلغان كومپىيوتىردىن قىممەترەك بولىدۇ. چۈنكى ئۇنىڭ قىممىتى ئۇنىڭ ئىچكى سەپلىمىسىدە، ھەم سېتىشتىن كىيىنكى مۇلازىمىتى ياخشى، قانچە داڭلىق ماركىدىكى كومپىيوتىر بولسا ئۇنىڭ پايدىسى شۇنچە چوڭ بولىدۇ، مەسىلەن؛ 5000 يۈەنلىك ماركىلىق كومپىيوتىرنىڭ زاۋۇتىىن چىقىش باھاسى 4000 يۈەن بولسا قالغان 1000 يۈەن ئۇنىڭ ماركىسىنىڭ قىممىتى، تالون، سېتىستىن كىيىنكى مۇلازىمەت، سودا ساراينىڭ پايدىسى قاتارلىقلارنى ئۆز ئىچىگە ئالىدۇ. ئەگەر بىز ئۆزىمىز قۇراشتۇرۇپ ئالساق 4000 دىن تۆۋەن ئالغىلى بۇلىشى مۇمكىن، ئەمما 4200 دىن ئېشىپ كەتمەيدۇ. چۈنكى ساتقۇچىلار توپ كىرگۈزگەن بولغاچقا باھا جەھەتتىمۇ ئىتىبارغا ئىرىشەلەيدۇ. ئەمما قۇراشتۇرۇلغان كومپىيوتىرنىڭ باھاسى ئۇنىڭ تالوننى ئۆز ئىچىگە ئالمايدۇ. ئادەتتە تالونغا %4 باج تاپشۇرىلىدۇ.
2. سىرىتقى كۆرىنىشى ۋە ئىقتىدارى:
نۇرغۇن ماركىلىق كومپيۇتېرنىڭ سىرتقى كۆرۈنۈشى مودا، رەڭگىمۇ كۆركەم بولىدۇ. قۇراشتۇرۇلغان كومپيۇتېرنىڭ تاشقى قېپىغا قارتا تاللاش دائىرىسى بىرقەدەر كىچىك، ھەم سەپلىمىلىرى ئانچە ماسلاشمىغان بولىدۇ. ئىقتىدار جەھەتتىن ئېلىپ ئېيتساق، ماركىلىق كومپيۇتېر نىسبەتەن گەۋدىلىك بولۇپ، بەزىلىرىدە بىر كۇنۇپكا ئارقىلىق ئەسلىگە كېلىش، مەخپىي قوغداش قاتارلىق ئىقتىدارلار بار بولغان بولىدۇ. ئىشلىتىشكە ئىنتايىن قۇلايلىق. قۇراشتۇرۇلغان كومپيۇتېرنىڭ سەپلىمىلىرىنى ماسلاشتۇرۇپ بولۇش سەل تەسكە چۈشىدۇ.
3. ئىشلىتىش جەھەتتە
سىز سېتىۋالغان ماركىلىق كومپيۇتېرغا مۇكەممەل مەشغۇلات سىستېمىسى سەپلەنگەن بولسا، بۇ مەشغۇلات سىستېمىسىنىڭ پۇلىمۇ كومپيۇتېرنىڭ سېتىلىش باھاسى ئىچىدە بولغان بولىدۇ، ھازىرقى ماركىلىق كومپيۇتېرلارنىڭ كۆپىنچىسى مۇكەممەل مەشغۇلات سىستېمىسى قاچىلانغان بولىدۇ. ئەمما قۇراشتۇرما كومپيۇتېرغا ئۆزىڭىز قاچىلايسىز.
4. تۇراقلىقلىقى:
ماركىلىق كومپيۇتېرنىڭ سەپلىمىلىرىنىڭ مۇستەقىل ھالدا ماسلىشىش ئىقتىدارى بولغان بولىدۇ، ئىقتىدارىمۇ بىر قەدەر تۇراقلىق بولىدۇ، قۇراشتۇرۇلغان كومپيۇتېرنىڭ سەپلىمىلىرى تازا ماسلىشىپ بولالماي ئاسانلا ئۇنداق مۇنداق مەسىلىلەر كۆرۈلىدۇ.
5. جانلىقلىقى:
ماركىلىق كومپيۇتېرنىڭ سەپلىمىسىنى تاللاش ئىمكانىيىتى ئاز بولىدۇ. ئەمما قۇراشتۇرما كومپيۇتېرنى ئۆزىڭىزنىڭ تەلىپى ۋە ئىقتىسادى ئەھۋالىڭىزغا قارتا تاللاپ سەپلىسىڭىز بولىدۇ.
6. سېتىشتىن كېيىنكى مۇلازىمەت:
ماركىلىق كومپيۇتېرنىڭ ئادەتتە بىر قانچە قېتىم ھەقسىز سېتىشتىن كېيىنكى مۇلازىمىتى بار ھەمدە تېخنىكىلىق مەسلىھەت سوراش لىنىيىسى ئارقىلىق ئىشلەتكۈچىلەرگە قۇلايلىق ياراتقان، ئەمما قۇراشتۇرما كومپيۇتېرنىڭ سېتىشتىن كېيىنكى مۇلازىمىتى ئاساسەن بەك ئاز، ھەم ۋاقىت جەھەتتىنمۇ ماركىلىق كومپيۇتېرغا قارىغاندا قىسقا.
