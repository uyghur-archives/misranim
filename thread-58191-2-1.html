دىلشاتقا مۇناسىۋەتلىك ئۇچۇرلار بولسۇن ياكى قىسقا فىلىملار بولسۇن مىنىڭ دىققىتىمنى تارتقىنى، ئۇنىڭ يېنىدىكى كىچىك بىر بالا. تورداشلار نىمىشقا بۇ بالا ھەققىدە چۈشەنچە بەرمەيدۇ...قارسام ئۇ بالىمۇ دىلشاد بىلەن سەپەردە بىرگە بولغاندەك تۇرىدۇ. ئۇ بالا كىم؟ ئۇمۇ جاپالىق ھالدا دىلشاد بىلەن بىرگە ئايلىنىپتۇ، ئۇنىڭمۇ كىچىك تۇرۇپ قىلغان بۇ روھىدىن ئۈگۈنىشكە ئەرزىيدۇ...توغىرا دىلشاد ھەربىر ئۇيغۇر ئوغلانىنىڭ پەخىرى ! بىراق مىنى بەك ئەپسۇسلاندۇرغىنى ئاۋۇ كىچىك ئەزمەت ھەققىدە بىرسىنىڭ بولسىمۇ سۆزلەپ قويمىغىنى.....

دىلشاتنىڭ يېنىدا ماڭغىنى تولۇقسىز مەكتەپتە ئوقۇۋاتقان، دىلشاتنىڭ نەۋرە ئىنىسى دىليار، 15 ياشلىق جەسسۇر ئەزمەت ، ئەسلى پىلاندا بۇ سەپەردە دىليار يوق ئىدى ،
ئەسلى دىلشات بىلەن بۇ سەپەرگە باشقا بىر دوستى بىللە چىقماقچى بولۇپ ئىشلار ياخشى ئورۇنلاشتۇرۇلغان ، لىكىن سەپەرگە ئاتلىنىدىغان ۋاقىتتا بۇ دوسنىڭ بىللە بېرىشى مەلۇم ئىشلار تۇپەيلى بارالماسلىققا ئۆزگۈرۈپ دىگەن ۋاقىتتا ، قوزغىلىدىغان ئورۇنغا كەلمىگەن ، شۇنىڭ بىلەن تېلفۇن قىلىپ ، بارالمايدىغانلىقىنى ئۇققان ، شۇنىڭ بىلەىن دىلشات قىلچە بوشىشىپ قالماي بۇ سەپەرنى داۋاملاشتۇردىغانلىقىنى دىگەن ، لىكىن دىلشاتنىڭ تاغىسى دادىللىق بىلەن بۇ قەيسەر ئەزمەتنى دىلشاتقا قوشۇپ بۇ جاپالىق لىكىن شەرەپلىك سەپەرگە يولغا سالغان ، دىليار ۋەلسىپىت بىلەن تاكى بۇ سەپەرنىڭ مۇشۇ باسقۇچقا كەلگۈچە كىچىكلىكىگە تەن بەرمەي ئۆزىدىكى قەيسىرانە روھىنى داۋاملاشتۇرۇپ كەلمەكتە ،
بۇ سەپەر دىليارنىڭ يېقىندا مەكتەپكە چىقىشى تۈپەيلى ھەم بەزى شارائىت تۈپەيلى شىمالى شىنجاڭ سەپرى بىلەن ئاياغلاشماقچى ، بەلكىم 0ئىككىنجى يۈرۈش پىلانى تۈزۈلۈپ قالسا جەنۇبى شىنجاڭنى ئايلىنىپ قېلىشى مۈمكىن .

مېنىڭ بىلىدىغىنىم مۇشۇنچىلىك ئىكەن ، خاتالىق يەرلىرى بولسا كەچۈرۈڭلار ، بۆرتالادا دىلشات بىلەن پاراڭلاشقاندا مۇشۇنداق پاراڭلارنى قىلىپ بەرگەندەك قىلغان .
ئانا تىلغا كەلگەندە خىرىس ، ئۆزگە تىلغا بولغاندا ھىرىس  ، شۇنچە يىللىق ئىسىل ئەنئەنە ، داۋام ئىتىپ بولغاندا چىرىش . ئاھ ! تىلىم دەپ يۈرەك يىغلايدۇ !؟
