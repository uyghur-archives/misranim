ئەر كىشى بىر نەچچە مىنۇت سەرپ قىلىپلا دادا بولالايدۇ ، ئەمما ئايال كىشى بىر ئۆمۈر سەرپ قىلىپ ئانا بولىدۇ ؛
ئەر كىشى بىر كۈن ناشتىلىق تەييارلىسا خۇش بولۇپ كېتىمىز ، ئەمما ئايال كىشى بىر كۈندە ئۈچ ۋاخ ، بىر ئۆمۈر تاماق تەييارلايدۇ ؛
ئەر كىشى ئىشتىن چۈشسە كارىۋاتقا چىقىپ ئارام ئالىدۇ ، ئايال كىشى ئىشتىن چۈشسە ئاشخانىغا چىقىپ تاماق ئېتىدۇ؛
ئەرلەرنىڭ سىرتقا چىققاندا چىرايلىق ئاياللارغا قارىشى ئۇلارنىڭ فىزىئولوگىيىلىك ئېھتىياجى ھېسابلىنىدۇ ، ئاياللارنىڭ ئېرىنى سىرتتىن تارتىپ ئائىلىگە قايتۇرۇپ ئۆيدىن مېھىر تىلىشى ئۇلارنىڭ ئاياللىق ئەخلاققا رىئايە قىلمىغانلىقى سانىلىپ قالىدۇ ؛
ئايال كىشى نىكاھى ئوقۇلغاندىن باشلاپ بارلىقىنى ئېرىگە ئاتايدۇ ئەمما بىر ئۆمۈر يولدىشىنىڭ بارلىقىنى ئاتىشىغا ئېرىشەلمەيدۇ ؛
ئاياللار ئەرلەرنىڭ سىرتقا چىققاندا گۈزەللەرگە كۆز سېلىشىنى كەچۈرەلەيدۇ ، ئەمما ئەرلەر ئاياللارنىڭ قاملاشقان ئەرلەرگە قارىشىنى ھەرگىز كەچۈرەلمەيدۇ؛
ئاياللارنىڭ ئەرلەرگە   پۇت يۇيۇش ئۈچۈن سۇ ئەكېلىپ بېرىشى ۋە دولىسىنى تۇتۇپ قويۇشى قىلىشقا تېگىشلىك ئىش ئەمما ئەرلەرنىڭ ئاياللارغا ياردەملىشىپ ئۆي ئىشلىرىنى قىلىشىپ بېرىشى يۈز تۆكۈلىدىغان ئىش ؛
ئاياللار ئارتۇقچە ئارزۇ – ئارمان قىلمايدۇ ، پەقەت ئىككى تەرەپنىڭ ئاتا- ئانىسى ساق – سالامەت بولسىلا ، بالىلىرى مەكتەپتىن كېلىپ ئاپا ! دەپ چاقىرسىلا كۆڭلى كۇپايە تاپىدۇ   ، ئېرى سىرتقا چىقىشتىن ئىلگىرى ئۇنىڭ مەڭزىگە يېنىككىنە سۆيۈپ قويسىلا رازى بولىدۇ ، ئۇ ئاچچىقلانغاندا چاپىنىڭىزنى ئېلىپ چىقىپ كەتمەي ئۇنىڭغا ھەمراھ بولسىڭىزلا ئۇ رازى بولىدۇ ، ئۇنىڭ كۆڭلى يېرىم بولۇپ بىئارام بولغاندا ئۇنىڭغا ھەمراھ بولۇپ كۆڭۈلدىكى سۆزلىرىڭىزنى ئېيتىپ بەرسىڭىز ئۇنىڭ ناتونۇش ئادەملەرنى تېپىپ ئەرز ئېيتىشقا مەجبۇر قىلمىسىڭىزلا ئۇ رازى بولىدۇ ، ئۇ خەتەرگە يولۇققاندا كۆكرەك كېرىپ ئوتتۇرىغا چىقىپ قوغداپ،  سىزنى تاللاپ خاتا قىلمىغانلىقىنى ھېس قىلدۇرسىڭىزلا رازى بولىدۇ ، ئۇ يىغلىغاندا ئۇنى كەڭ كۆكرەكلىرىڭىزگە ئېلىڭ ، ئۇ ئۆزىنىڭ سىز ئۈچۈن قانچىلىك مۇھىم ئىكەنلىكىنى ھېس قىلسۇن ؛ ئۇ بىئارام بولغان ئاشۇ بىر نەچچە كۈندە ئۇنىڭغا ئىسسىق سۇ ئەكېلىپ بېرىڭ ، بۇ ئىشنى ئۇ بەلكىم بىر ئۆمۈر ئېسىدە ساقلىشى مۇمكىن ؛
ئاتا – ئانىسى ئۇنى جاپا – مۇشەققەتتە بېقىپ چوڭ قىلغان ، توي كۆينىكى كىيگەن شۇ كۈنى ئۇ ئۆزىنى سىزگە ئاتايدۇ ، شۇنىڭدىن باشلاپ سىز ئۇنىڭ ئۆمرىدىكى يۆلىنىدىغان ئىشەنچ تۈۋرۈكى بولىسىز ؛شۇنىڭدىن باشلاپ ئۇ سىزگە قىز – ئوغۇل پەرزەنت تۇغۇپ بېرىدۇ ، سىزگە ۋاكالىتەن ئاتا – ئانىڭىزنىڭ ھالىدىن خەۋەر ئېلىپ ۋاپادارلىق يەتكۈزىدۇ ، سىز ئۈچۈن ئائىلە ئىشلىرىنى تەل – تۆكۈس قىلىدۇ ، ئاۋۋال قېيىن ئاتا قېيىن ئانىسىنى كۈتۈش ئەزەلدىن ئۆزگەرتىش قائىدە ، ئاۋۋال ئۆز ئاتا –ئانىسىنى كۈتۈش بولسا ئۆي تۇتۇشنى بىلمىگەنلىك ھېسابلىنىدۇ ؛
ئەمما سىز ئويلاپ باقتىڭىزمۇ ؟ ئۇنىڭ ئاتا – ئانىسى سىلەر توي قىلىپ يەنە نېمىگە ئېرىشتى ؟
ئۇلار قىزىنىڭ سىز بىلەن بىللە بەختكە ئېرىشىشىنى ئۈمىد قىلاتتى ، ئەمما سىز بۇنى قىلالىدىڭىزمۇ ؟
ئەرلەر ، سىلەر ئۆزۈڭلارنى ھېرىپ كەتتىم دېمەڭلار ،سىلەرنىڭ سىرتتا چېپىپ يۈرۈپ ئائىلىنى قامدىشىڭلار سىلەرنىڭ مەسئۇلىيىتىڭلار ، ئائىلە ئۈچۈن تىرىشچانلىق كۆرسىتىشىڭلار سىلەرنىڭ مەجبۇرىيىتىڭلار ؛
ئەرلەر ، سىلەر ئۆزۈڭلارنى جاپا چەكتىم دېمەڭلار ، ئاياللارنىڭ سىلەر ئۈچۈن پەرزەنت تۇغۇپ بېرىش ئالدىدا تارتقان ئازابىنى سىلەر مەڭگۈ ھېس قىلالمايسىلەر ؛
سىلەر كۈلمەڭلار ، ئاياللارنىڭ كۆڭلى – كۆكسى سىلەرنىڭكىدىن كەڭ ، ئۇلار ئائىلىسى ۋە بالىلىرىڭلار ئۈچۈن سىلەرنىڭ سىرتتا ئۈچىنچى شەخس تېپىۋېلىشىڭلارغا چىدىيالايدۇ ، ئەرلەر ئاياللاردىن چىدىماس ، ئازراق غۇرۇر ئۈچۈن بارلىقىنى سىزگە ئاتىغان ئايالىڭىزدىن ۋاز كېچىشكە رازىكى ئۇنىڭ كىچىككىنە بىخەستەلىكى سەۋەبىدىن كېلىپ چىققان سەۋەنلىكىنى كەچۈرەلمەيسىز .
ئەرلەر ئاياللىرىڭلارغا قاراپ بېقىڭلار ، قورۇقلار ئۇنىڭ پېشانىسىگە يامىشىپتۇ ، ئاق چاچلار ئۇنىڭ چېكىسىگە ئىز بېسىپتۇ ، ئۇنىڭ قوللىرىمۇ بۇرۇنقىدەك نازۇك ئەمەس ؛ ئەرلەر مۇشۇنداق ۋاقىتتا ئايالىڭلاردىن ۋاز كەچمەڭلار ، ئۇلار سىلەرگە ياتلىق بولغاندا نازۇك قىزلاردىن ئىدى ، ئۇلار سىلەر ئۈچۈن ئۈن – تىنسىز بارلىقىنى ئاتاپ بۈگۈنكى ھالەتكە كېلىپ قالغان ،ئۇلار ئەڭ گۈزەل ۋاقىتلىرىنى سىلەرگە ئاتىغان ؛
نىكاھتا بىرەيلەن يەنە بىرەيلەنگە يۈز كېلەلمەيدىغان ئىش يوق ، بەلكى خۇشال – خۇرام بەختلىك ئائىلە ئۈچۈن ئۆز ئارا تەقدىم قىلىشتىن ئىبارەت ؛ ئەگەر قارشى تەرەپنى قەدىرلىمىسىڭىز ئۇنىڭ ئايرىلىپ قالغاندا ئۇنىڭ سىز ئۈچۈن شۇ قەدەر مۇھىملىقىنى پۇشايمان ئىچىدە ھېس قىلىسىز؛
ئەرلەر بىردەملىك ئاچچىقىڭلار ئارقىلىق ئاياللىرىڭلارنىڭ غۇرۇرىغا تەگمەڭلار ،ئۇلار سىلەرنىڭ قۇلۇڭلار ئەمەس ؛
ئەرلەر ، سىلەر ئۆز خاراكتېرىڭلار بىلەن ئاياللىرىڭلارنىڭ سەۋرچانلىقىغا جەڭ ئېلان قىلماڭلار ، چۈنكى ئۇلار سىلەرگە قەرز ئەمەس ؛
ئەرلەرنىڭ كۆز يېشى تۆكۈشى ئائىلىسىنى سېغىنغانلىقتىن ، ئاياللارنىڭ ياش تۆكۈشى بولسا ئائىلىدىن ئايرىلىشنى قارار قىلغانلىقتىن ؛
ئايالىڭلار ئائىلىدىن ئايرىلغاندا ئاندىن ئۇلارنى قانداق قانداق ياخشى كۆرىدىغانلىقىڭلارنى سۆزلەپ يۈرمەڭلار ، ئۇ ۋاقىتتا ئۇلارنىڭ كۆز يېشى ئاللىقاچان قۇرۇپ يۈرىكى مۇزلاپ بولغان بولىدۇ ؛
ئاياللارنىڭ ئەرلەرگە يول قويۇشى كەڭ قورساقلىق ، ئەرلەرنىڭ ئاياللارغا يول قويۇشى قورقۇنچاقلىق ،بىر – بىرىڭلارغا كەڭ قورساقلىق بىلەن مۇئامىلە قىلىپ ئۆز ئارا قەدىرلەڭلار !

ئۇنىڭ كۆڭلى يېرىم بولۇپ بىئارام بولغاندا ئۇنىڭغا ھەمراھ بولۇپ كۆڭۈلدىكى سۆزلىرىڭىزنى ئېيتىپ بەرسىڭىز ئۇنىڭ ناتونۇش ئادەملەرنى تېپىپ ئەرز ئېيتىشقا مەجبۇر قىلمىسىڭىزلا ئۇ رازى بولىدۇ ، ئۇ خەتەرگە يولۇققاندا كۆكرەك كېرىپ ئوتتۇرىغا چىقىپ قوغداپ،  سىزنى تاللاپ خاتا قىلمىغانلىقىنى ھېس قىلدۇرسىڭىزلا رازى بولىدۇ ، ئۇ يىغلىغاندا ئۇنى كەڭ كۆكرەكلىرىڭىزگە ئېلىڭ ، ئۇ ئۆزىنىڭ سىز ئۈچۈن قانچىلىك مۇھىم ئىكەنلىكىنى ھېس قىلسۇن ؛ ئۇ بىئارام بولغان ئاشۇ بىر نەچچە كۈندە ئۇنىڭغا ئىسسىق سۇ ئەكېلىپ بېرىڭ ، بۇ ئىشنى ئۇ بەلكىم بىر ئۆمۈر ئېسىدە ساقلىشى مۇمكىن ؛

ئەرلەرنىڭ ئاياللار ئۈچۈن قىلغان كىچىككىنە ئىشلىرىمۇ ئايال كىشىگە مەڭگۈلۈك خۇشاللىق ئاتا قىلالايدۇ ۋە ئۇلاردىن رازى بۇلىدۇ. چۈنكى ئاياللار بەكلا ئاجىز يارتىلغان.
سېنى بويسۇندۇرغان كىشنىڭ ساڭا قىلغان ۋەدىسى ۋە تەسەللىسى، غەلبىدىن كېيىن سىنىڭ ئالدىڭدا كۆرەڭلىگىنىدۇر!
