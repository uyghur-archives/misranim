يوللىغان ۋاقتى 2011-8-24 11:55:42 AM |ھەممە قەۋەتنى كۆرۈش |
قەدىرلىگىم، سىزگە دەيدىغان سۆزلىرىم يىغىلىپ قالدى، ئالدىڭىزدا دىيىشتىن، ئايالىڭىز تۇرۇپ سىزدىن ئەيىپ ئىزدەشتىن، سىزگە يول كۆرسىتىشتىن ئەيمەندىم، خىجىل بولدۇم، چۈنكى ئەۋىلى ئاللاھ ئاندىن سىز مېنىڭ يول باشچىم نىجاتكارىم تۇرسىڭىز.يۈز تۇرانە سىزگە گەپ تاشلىسام بولمىغۇدەك. شۇڭا مۇشۇ ئۇسۇلدا ئويلىغىنىمنى ،ھېس قىلغىنىمنى سىزگە يېزىپ قالدۇردۇم.بۇ پىكىرلىرىمگە قوشۇلامسىز يوق ئوقۇپ بولۇپ بىرنىمە دەرسىز.
دادىسى، بىز ئۆي تۇتۇپ مانا10يىللىق قەدىناسلاردىنمۇ بولۇپ قالدۇق،ئىككى ئۇماق قىزىمىز بار. تۇرمۇشىمىزمۇ خاتىرجەم، كىشىدىن ئېلىشىمىز بولسا باركى كىشىگە بېرىشىمىز يوق، ئەل قاتارى كېتىۋاتىمىز.ئەمما مۇشۇ تەرىقىغە يەتكۈچە بىزمۇ كۆپ ئەجىر قىلدۇق، ئۆزئارا ياخشىمۇ بولۇشتۇق يامانمۇ بولۇشتۇق. قىزىل خەتلىرىمىز يېشىلغا ئۆزگەرمىگەن بولسىمۇ ئەمما يىرتىلدى يېڭىلاندى،تۇرمۇش قۇرغانغا بەزىدە پۇشايمانمۇ قىلدۇق، بۇرۇن ئەجەپ بىلمەپتىكەنمەن سېنى، تونىمايلا قېلىۋاتىمەن، ۋاي بولدى تۇتماپتىمەن شۇ ئۆينى دەپ كۈدە-كۆرپىمىزنى كۆتۈرۈپ ئانىمىزنىڭكىگە يامانلاپمۇ كەتكەن بولدۇق، ئەمما ئاتا-ئانىلىرىمىزنىڭ دۇئاسى بىلەن، ئۆيىمىزگە سىڭدۈرگەن ئەجرىمىزنىڭ ھەققى-ھۆرمىتى ئۈچۈن ئائىلىمىز شۇ ئۆتكەللەردىن ئامان-ئىسەن ئۆتۈپ مانا بۈگۈن ئىككى بالىغا ئاتا-ئانا بولدۇق. ئاستا ئاستا تۇرمۇشقا مۇئامىلە قىلىشنى ئۈگەندۇق. يول قويۇشنى بىلدۇق. سىزمۇ مېنى كۆپ ئىشلاردا خەلقى-ئالام ئالدىدا ئائىلىدە ئەزىز قىلدىڭىز. لېكىن يەنە نىمىشقىدۇر ھەممە نەرسىگە ئېرىشكەنسىرى، يوق نەرسىلىرىم بار بولۇپ، پۈتۈن ئېھتىياجىم قامدالغانسىرى ئەڭ مۇھىم،ئەڭ ئەڭ مۇھىم قىممەتلىك بىر نەرسەمنى يوقىتىپ قويىۋاتقاندەك، رېلىستىن چىقىپ كەتكەن پويىز نىشانسىز مېڭىۋەرگەندەك مەنمۇ نىشانىم يوق كۈچلۈكلەر قاتارىدىن ئورۇن ئېلىپ سىزنىڭ ئائىلىدىكى ئورنىڭىزغا، ھوقۇقىڭىزغا خىرىس قىلىۋاتقاندەك تۇيغۇدا بولۇپ قېلىۋاتىمەن.مەن زادى نىمىنى يوقۇتۇپ قويغاندىمەن، سىز يېنىمدا، بالىلىرىم باغرىمدا، قېرىنداشلىرىم ،دوستلىرىم، ئەل-جامائىتىم مېنىڭ ھاياتىمدا ماڭا ھەمراھ. ئەمسە نىمىدۇ ئۇ؟؟؟؟؟؟
مانا مۇشۇنداق چىگىش خىياللار بىلەن ئەمدىلا ئىزىغا چۈشكەن تۇرمۇش خاتىرەمنى يەنە قايتىدىن ۋاراقلىشىمغا توغرا كەلدى، بۈگۈنۈمدىن تاپالمىغان جاۋاپنى تۈنۈگۈندىن ئىزدىمەك بولدۇم.بەرھەق مەن جاۋاپقا ئېرىشتىم. يۈتكەن بايلىقىمنى تېپىۋالدىم، ئەمما بۇ بايلىقنى يوقۇتۇشتا سىزنىڭمۇ مېنىڭمۇ جاۋاپكارلىقىم بار ئىكەن. دىمەك مەن يالغۇز ئۇنى قايتۇرۇپ كېلەلمىگۈدەكمەن.شۇڭا سىزگە مۇراجەت قىلاي، نىمىلىكىنى ئېيتاي، ماڭا ياردەم قىلىڭ.
دادىسى،قاراڭ بىز يېڭى توي قىلغان چاغلىرىمىزدا ئولتۇرىدىغانغا ئۆيىمىز ،يانچۇقتا تۇرمۇشقا يەتكۈدەك ئىقتىسادىمىز يوق ئىدى. سىزنىڭ ئىدارىڭىز ۋەيران بولغانلىقىنى جاكارلاپ سىز ئۆيدە تۇرمۇش پۇلى بىلەن خىزمەتكە قايتا تەقسىملەشنى كۈتۈپ ئولتۇراتتىڭىز،مەن ئوقۇش پۈتتۈرۈپ تېخى خىزمەتكە ئورۇنلاشماي تۇرۇپلا بىر يىل ئۆيدە ئولتۇرغاندىن كىيىن سىز بىلەن تونۇشۇپ، شۇ چاغدىكى ئەمەلىيىتىمىز بىلەن ئويلاشمايلا ئالدىراپ-سالدىراپ بۇ توي قىلىپ،ھەتتا ئۆي ئىجارىگە ئالىدىغان ئىقتىسادىمىزمۇ يوق سىزنىڭ ئىدارىڭىزدىن تەمىنلەپ بەرگەن بىر كونا ئىشخانىدا ئىككى يىلچە ئولتۇردۇق، مەن ئۆزۈمگە بەك ئىشىنىپ كەتكەن چېغىم،ئالىي مەكتەپ ئوقۇغان تۇرسام چوقۇم يەردە قالمايمەن دەپ بەك ئىشەنچ قىلغان ئىدىم. ئەمما شۇ ئولتۇرۇشۇمدا 7يىل ئۆيدە ئولتۇردۇم، بۇ جەرياندا سىزنىڭ خىزمەتلىرىڭىزمۇ تولا ئىگىز پەس بولۇپ ئىقتىساددا باشقىلار ئويلاپ يىتەلمىگۈدەك دەرىجىدە قىينالغان ئىدۇق. سىرتتا كۈلۈپ يۈرگىنىمىز بىلەن ئۆيدە مېنىڭ كۈنۈم يىغلاش بىلەن سىزنىڭ كۈنىڭىز ئۆينى قانداق قامدارمەن دىگەن غەم بىلەن ئۆتتى، شۇ ھالىتىمىزدە بەش يىلغىچە بالا ئالغۇمىز شۇنچە بولسىمۇ شارائىتىمىز تۈپەيلى ئالماي تۇردۇق. بىزنىڭ تۇرمۇشىمىزدىن چوڭلار ئۆمۈدىنى ئۈزدى، سىزنىڭ ئاتا -ئانىلىرىڭىز ئۆز ۋاقتىدا ئويلاشمايلا مېنى ئېلىشىڭىزغا قوشۇلغانلىقىدىن پۇشايمان قىلىشتى، مېنىڭ ئۆيۈمدىكىلەر مېنى سىزدەك بىر ئائىلىنىڭ ھۆددىسىدىن چىقالمايدىغان، قولىدىن ئىش كەلمەيدىغان(بۇ ئۇلارنىڭ شۇ چاغدىكى سىزگە بولغان كۆز قارىشى ئىدى)بىرىگە بېرىپ بەختسىز قىلىپ قويغىنىدىن ئۆكۈنىشەتتى. ھەتتا بىزنى ئايرىۋەتمەكچىمۇ بولۇشتى. ئەمما سىز مېنى تاللىغىنىڭىزغا پۇشايمان قىلمىدىڭىز، مېنى قانچە قىينالساممۇ بېقىپ كېتەلەيمەن مەن دىگەن ئەركەك، دىدىڭىز. مەنمۇ سىزدەك خىزمىتى مۇقىم بولمىغان ھەرۋاقىت ئىشسىز قېلىش خەۋپىگە دۈچ كېلىدىغان بىر ئەرنى ياخشى كۆرۈپ قالغىنىمدىن، بىللە ئۆتكۈنۈمدىن پۇشايمان قىلمىدىم، ئەكسىچە شۇنچە ئوقۇپ كېلىپمۇ قولۇمدىن ھىچ ئىش كەلمەي سىزگە يۈك بولۇپ قېلىۋاتقىنىمنى ئويلىغىنىمدا خىجىل بولدۇم، ئەمما ھەر قېتىم مەنىۋى جەھەتتە ماڭا زور يار-يۆلەك بولىۋاتقىنىڭىزدىنمۇ سۆيۈندۈم، بېقىلىشنىڭ ھوزۇرىنى سۈردۈم. شۇ چاغلىدا ئائىلەم ئۈچۈن سىز بەك مۇھىم ئىدىڭىز، سىز مېنىڭ ھاياتلىق مەنبەيىم ئىدىڭىز، سىزسىز مېنىڭ ھىچقانداق ئىشىم ئەمەلگە ئاشمايتتى. مۇشۇنداق شارائىتتا بىزدىمۇ كۆپ دەتالاش، قىيىنچىلىق تۈپەيلى ئۆزئارا رەنجىش، بىرىمىز سەن ئاز تاپتىڭ دىسەك، سەن نىمە تېپىۋاتىسەن دەپ تەنە سۆزلەشلەرمۇ بولدى. ئەمما شۇ جاپادا كۈن ساناپ ئۆتكۇزىۋاتقان تۇرمۇشىمىزنى ئويلىغاندا مىڭ تەستە قۇرغان ئائىلىنى بۇزغىمىز كەلمىدى، لېۋىمىزنى چىشلىدۇق، تىرىشچانلىق كۆرسەتتۇق، مەنمۇ تىنىم تاپماي ئۈگۈنۈشۈمنى تاشلاپ قويماي ئېلىشقا تېگىشلىك ئۇنۋانلارنى، كېنىشكىلارنى، ئېلىپ تۇردۇم، دۆلەت مەمۇرلىقىغا ئىمتىھان بېرىپ تۇردۇم. ئويلايدىغىنىم، ياخشى -يامان كۈنلىدە سىزنىڭ ئائىلىدە ئەرلىك بۇرچىڭىزنى ئادا قىلىپ كەلگىنىڭىز بولدى. ئۇ چاغلىدا، ئۆيۈمدە گاز تۈگىسە ئىشىككە تەلمۈرەتتىم، ئۇن-گۈرۈچ تۈگىسە ئىشىككە تەلمۈرەتتىم، ئۇ-بۇ بۇزۇلسا ئىشىككە تەلمۈرەتتىم، يالغۇز قالسام قورقۇپ يەنە ئىشىككە تەلمۈرەتتىم، نىمە قىيىنچىلىققا يولۇقسام،سىزدىن باشقا يۆلەنچۈگۈم يوق ئىدى.شۇنىڭ ئۈچۈن سىز ماڭا شۇنچە مۇھىم ئىدىڭىز. كىيىن كۈنلىرىمىزدە ياخشىلىنىش بولدى،تۇرمۇش قۇرۇپ6يىلدىن كىيىن مەنمۇ ئاخىرى دۆلەتلىك مەمۇرى ئىمتىھاندىن ئۆتۈپ خېلى نوپۇزلۇق ئورۇنغا خىزمەتكە چۈشتۈم، سىزنىڭمۇ خىزمىتىڭىز كۈندىن كۈنگە ياخشىلىنىپ تەمىناتى يۇقۇرى خىزمەت ئورنىغا ئىگە بولدىڭىز . شۇنداق قىلىپ ئىككى قىزىمىزمۇ تۇغۇلدى،ئۆي ئالغۇچە گەدەنگىچە بوغۇلغان ھېلىقى قەرزلىنىمۇ تۈگەتتۇق.ئولتۇرىدىغانغا ئۆيىمىز ، ھەيدەيدىغانغا ماشىنىمىز ،بانكىدا ئازدۇر-كۆپتۇر پۇلىمىز بولدى. زامانمۇ ئەجەپ تېز تەرەققى قىلدى. لېكىن……….مېنىڭ يانچۇغۇمدا پۇلۇم بولدى، سىزنىڭ ئىقتىسات ئېلىپ كىرىشىڭىزگە ئۇنچە مۇھتاج بولمايدىغان بولدۇم، بازار سىرتقا ئېچىۋېتىلدى، مۇلازىمەت كەسپى راۋاجلاندى، نىمە كەم بولسا سىزگە تايانمايلا بىر تېلىفون بىلەن يوقنى بار قىلىدىغان بولدۇم، سىزنىمۇ بۇرۇنقىدەك ئويلىماس بولدۇم، نەگە بارغۇم كەلسە تالاغا چىقىپلا تاكسى، سىز بولمىسىڭىزمۇ بولىۋېرىدۇ، خىزمەت مۇناسىۋىتى بىلەن ئۇ شەھەرمۇ بۇ شەھەر يۈردۈم، قوشۇمچە ساياھەت، سىزگە مېنى سىرتقا ئېلىپ چىقىڭمۇ دىمەس بولدۇم. نىمە كېرەك بولسا قولىڭىزغا تەلمۈرمەيمەن. باشقىلار كىيەلىگەننى كەيدىم، يىيەلىگەننى مەنمۇ يىدىم، يەنە شۇ سىزسىزمۇ بۇ ئارمانلارغا يېتىۋەردىم. ماڭا ماسلىشىپ سىزمۇ پەقەت ئۆزىڭىزدىن ئاشمىدىڭىز، ئۆيدە نمە بار نىمە يوق دىمەيدىغان بولدىڭىز، ھەتتا ئۆيدە بىرەر نەرسىدىن كاشىلا چىقسا بىر تېلىفون بىلەن رېمونتچى چاقىرىپ بەرسىڭىز بەردىڭىزكى ئۆزىڭىز قول تۇتۇپ ياساشتىن ئېرىندىڭىز، سىرتلاردا ئىشلىرىڭىز كۆپەيدى، دىمەك سىزنىڭ مەۋجۇتلىقىڭىز پۇلنىڭ كارامىتى بىلەن ئۆز كۈچىنى يوقاتتى، بەزىدە ھەتتا سىز ئارتۇقچە ئادەمدەك، بۇ ئۆي پەقەت سىزنىڭ تاماق يەيدىغان ئاشخانىڭىز ،كەچتە كىرىپ ئۇخلايدىغان ياتىقىڭىزلا بولۇپ قالدى، خالاس.
دىمەك، دادىسى مەن سىزنى يوقۇتۇپ قويۇپتىمەن، مەن قانچە تولۇقلانغانسىرى سىز شۇنچە ئاجىزلاپ كېتىۋېتىپسىز. مېنىڭ تۇرمۇشۇمدا سىز بولمىسىڭىزمۇ بۇرۇنقىدەك قورقمايدىغان،سىز بولمىسىڭىزمۇ ھەممە ئىشىم ئۆز يولىدا يۈرۈشىدىغان، ھەتتا بەزىدە خاپا بولۇشۇپ قالساق بۇرۇنقىدەك مەن سىزگە ئەرلىك ھۆرمىتىڭىزنى قىلىپ بويۇن ئىگىپ تۇرۇش ئۇياقتا تۇرسۇن، سىزگە يانىدىغان، سەن تاپقاننى مەنمۇ تېپىۋاتىمەن، ھەتتا ئاشۇرۇپ تېپىۋاتىمەن. مەن ئاياللىق بۇرچۇمنى ئادا قىلىپلا قالماي يەنە سېنىڭمۇ رولۇڭنى ئۆتەپ مەنمۇ سىرتتا رىزق تېپىۋاتىمەن. بۇ ئۆيدە شۇ ئەكىرگەن ئىككى تۆتتال تەڭگەڭدىن باشقا نىمە رولۇڭ قالدى؟ بىز ئۈچۈن نىمە قىيىنچىلىق ھەل قىلىپ بەردىڭ؟ ئوتۇن ياردىڭمۇ؟ گاز توشۇدۇڭمۇ؟ لامپىنى ئالماشتۇردۇڭمۇ؟ ياق، ھەممىنى مەن قىلىۋاتىمەن، سەنسىزمۇ كۈنۈم گۈلدەك ئۆتىۋاتىدۇ دىيەلەيدىغان بولدۇم. سىزنىڭمۇ تىلىڭىز تۇتۇلدى. مانا مەن بۇ ئۆيدە ئۆزۈم خان ئۆزۈم بەگ بولدۇم. شۇنداق ئۆتىۋەردۇق. مانا ئەمدى ئالدىنقى كۈنى ماڭا بېرىپ ماشىنا ھەيدەشنى ئۈگۈنۈپ پىراۋا ئېلىۋال، ئەتە-ئۆگۈن باھاسى ئۆسۈپ كەتسە ۋاي ئېسىت دەپ قالما دەيسىز؟ نىمە قىلىمەن پىراۋىنى؟ دىدىم . ئىشىڭ چىققاندا مېنى ئاۋارە قىلماي ئۆزۈڭ ھەيدىسەڭ بولىدىغۇ دىدىڭىز. ئاللاھ ماۋۇ گەپنى، نەچچە ۋاقىتتىن سىزدە مەن ئۈچۈن ئائىلە ئۈچۈن قىلىپ كەلگەن ئىشتىن شۇ ماشىنا ھەيدەپ بىزنىڭ ھاجىتىمىزدىن چىقىش قالغان ئىدىغۇ؟ شۇ ئىشتا بولسىمۇ مەن سىزگە تايىناتتىم، سىزنى ئەسلەيتىم، سىزگە ھاجەتمەن بولاتتىمغۇ؟ نىمە ئەمدى شۇمۇ ئېغىر كېلىۋاتامدۇ؟ تۇرمۇشنىڭ رولىنىغۇ ماڭا خېلى بۇرۇنلا ئۆتكۈزۈپ بەردىڭىز، ئەمدى، ماشىنىنىڭ رولىنىمۇ ماڭا ئۆتكۈزۈپ بەرسىڭىز، نىمە؟ماڭا سىز خوتۇن بولماقچىمۇ ياكى ………ئەمدى مەن بۇنىڭغا نىمە دەي، ھېلىمۇ ئەتراپىمىزدا بۇنداق مىساللار بەك كۆپ، ئاياللارنىڭ ماشىنا ئۈگۈنىشىنىڭ كۆپىيىشىگە ئەگىشىپ بىز ھەقىقەتەن ئاياللار مۇستەقىللىقىنى تولۇق قولغا كەلتۈردۇق. بىز مۇستەقىل بولغاندىمۇ ئاجايىپ مۇستەقىل بولدۇقكى، ئەرلەرگە ئەۋلات قالدۇرۇشتىن ئىبارەت مۇشۇ ئىشتىلا قولىغا قارايمىزكەن ئۇندىن باشقىدا ئۇلارسىز گۈلدەك ياشايدىغان بولۇپ كەتتۇق. مېنىڭچە ئەۋلات قالدۇرۇش ئىشىدىمۇ ئاز قالدى ئەرلەرنىڭمۇ رولى قالمايدىغانغا ، ھازىر ياۋرۇپا دۆلەتلىرىدە توي قىلىشنى خالىمايدىغان ئەمما ئانا بولۇشنى ئويلايدىغان ئاياللار ئۈچۈن مەخسۇس ئەرلىك سىفىرما بىلەن تەمىنلەيدىغان ئورۇنلار بارلىققا كېلىۋېتىپتۇ، بىز تەرەققىياتنى بەك تېز ئۆزلەشتۈرىدىغان خەق، ئەجىبا شۇنداق بولۇپ قالسا، بۇ دۇنيادىن سىلەرنى سىقىپ چىقىرىۋېتىمىزمۇ؟ دادىسى ئاخىرقى سۆزلەرنى سەل قوپالراق قىلىپ سالدىم، ئەمما ئۆزۈمدىكى بۇ چوڭ يوقۇتۇشتىن قاتتىق چۆچۈگۈنۈم ئۈچۈن مۇشۇنداق ھاياجانلىنىپ تۇرۇپتىمەن. ماڭا ئېرىڭ بارمۇ دىسە بار دەيدىغانغا سىزدەك سالاھىيەت كېنىشكىسى كېرەك ئەمەس، ماڭا بۇرۇنقى ھەممە ھاجىتىمدىن چىقىدىغان، مېنى ئۆزىگە بېقىندۇرۇپ ياشىتىدىغان، قانات ئاستىغا ئالىدىغان سىز كېرەك.سىزسىز كۈنۈم ئۆتمەيدىغان، سىزسىز بۇ ئۆي ئۆي بولمايدىغان سىز كېرەك. شۇڭا خۇلاسە كالامىم شۇ بولدىكى قەدىرلىكىم ماشىنىڭىزنىڭ رولىنى يەنىلا سىز تۇتۇڭ، تۇرمۇشۇمنىڭ بىلىپ بىلمەي سىزدىن تارتىۋالغان سىز قەدىرلىمەيلا ماڭا ئۆتۇنگەن رولىنىمۇ سىزگە گۇناھىمنى تىلەش ئورنىغا قايتۇراي. ئېلىڭ ماڭا رولچى بولۇڭ، كومپاس بولۇڭ، ھاياتلىقتىكى مەشئىلىم بولۇڭ.
ھاھاھاھا………بىرگەپ ئېسىمگە كېلىپ قالدى، ئالدىنقى كۈنى خوشنىمىز ئەر-ئايال سوقۇشۇپ قالغان ئىدى، ئايالى ئۆينى بېشىغا كەيگىنى تۇردى، ئېرى ھە دەپ ئۇراتتى، ئايال بىرياقتىن يىغلاپ، بىرياقتىن سەت گەپلەر بىلەن ئېرىگە تىل ياغدۇراتتى.(سېنىمۇ ئەر دەمدۇ؟ سېنىڭ نىمە رولۇڭ قالدى، پەقەت قولۇڭدىن خوتۇنۇڭنى ئۇرۇشلا كېلىدۇ، بولمىسا قىلىدىغانغا ئىشىڭ يوق ئۆيدە سېسىق ئوسىراپ ياتىسەن، سىرتتا ئىچىپ-چېكىپ يۈرىسەن، ئەگەر ئۆزەڭنى ئەر دىسەڭ “ئېرىم” دىسەم لەببەي دە، بولمىسا “بەي بەي” دە دەيدۇ. تازا قاقاقلاپ كۈلۈپتىمەن. تاياق يەپمۇ يەنە بۇ گەپلەرگە ئۈلگۈردى دىسە. قورقۇپ قالدىم بىركۈنلىدە مەنمۇ ھەددىمدىن ئېشىپ سىزگە شۇنداق دەپ سالامەنمۇ؟ ياق….. اللاھ ئۇ كۈنلەردىن يىراق قىلغايسەن. ئائىلىمىزدە شۇنداق قىيامەت بولۇشتىن ئىلگىرى ئىككىمىز دەرھال ئورنىمىزنى ئالماشتۇرىۋالايلى. بولامدۇ؟”دۇنيا ئەر دەپ يىغلايدۇ”دىگەن سۆز ھەقىقەتكە ئايلىنىپ قالمىسۇن،قىزلىرىم سىزدىن پەخىرلەنسۇن، قىممىتىڭىزنى بىلسۇن، بىز سىزگە مۇھتاج. بولۇپمۇ مەن…….
