ئەسسىلامۇ ئەلەيكوم ! مىسرانىم شەھرىدىكى بارلىق ئاھالىلەر ! كەمىنە كۈلقىز يەنە كەلدى.
_ نىمىگە كەلدىڭىز دەمىسىز ...؟
_ ۋوي قارى ھاجىكا ئەتىگەندىن بىرى ھىيتتىمۇ ئۇخلامدۇ ؟بىزنىڭ ئۆيگە كىلىپ ھىيتلاپ كىتىڭ دىگەن سىزدۇ ؟؟ كەلسەم بەرگىنىڭىز ئاران ئىككى سوم ،يەنە بۇ يەردە ئىغىز غىرىچلىماي بىرىپ چاچ پاراشۇكىڭىزنى سىتىڭە ...
توۋا دەيمەن ، بۇ ئۇدۇن ئوغلى بەك چۈپەيكەن جۇمۇ ،ئەرگە تىگەي دەپ قالغان بىر قىزغا ھىيتلىقىم دەپ ئاران ئىككى سوم تەڭلاۋاتىدۇ ، توۋا توۋا ،قىنى ما ياقام ؟؟؟ كاسسىدە بىرنى چىشلىۋىتەي ...
       ئەتىگەندىن بىرى بەرىبىر ھىيتلىغۇدەك يىرىم يوق ئۇخلايچۇ دىسەم چ چ رېمغا تولا ئۇچۇر يوللاپ مىسرانىمغا كەلسىڭىز بىز ھىيتلىق بىرىمىز دىگەن قىز چوكان ، يىگىت داداشلا  قىنى ، ماڭا بىرىدىغان ھىيتلىقىڭلا يانچۇقتا تۇرۇپ پاختىلىشىپ قالمىسۇن يەنە،قىنى قىنى ...؟
__ ۋويت شاتئاي نەگە قاچتىڭ ، كىلە مايارگە ئاۋال سەندىن ئالىمەن ، بەرمىسەڭ بىشىڭنىڭ پىتلاپ كەتكىنىنى ھەممەيلەنگە دەپ قويىمەن جۇمۇ ...؟
__ تاڭنۇر چوڭ دادا قىنى سىز ؟ ئاللا  نىمە ئەخلەت تۇڭىنىڭ كەينىدە يۈرىيسىز ؟ چىقىڭ بۇ يەرگە ؟ ماڭا بىرىدىغان ھىيتلىقنى نەقلەشتۈرىۋەتكەندىن كىيىن ئەخلەت تۇڭىنىڭ ئىچىگە كىرىۋالسىڭىزمۇ كارىم يوق . تىز تىز ...ما قولوم تىلىپ كەتتى دەيمەن ،ھىلى باكا جۇماااااااااا ... تاڭ خانىم دەۋېتىمەن باكا ...
______  مومااااااااااااا  قىنى سىز جىنىم مۇسلىمە موماااااااااا، سىزنى بەك سىغىندىم ،سىزدىنمۇ بەك يانچۇقتىكى ئاۋۇ ماۋزىدۇڭلارنى سىغىندىم ،نىمە ؟؟؟ گۈلبىكى ھەممىنى ئىلىپ كەتتىما ؟ ئۇنداق ئەمەستۇ خەپ جۇۋاينىمەك ،كوچىدا ئۇچراپ قالما ....!
__ ۋوي  بۇ كىم ئەمدى ؟؟؟ ئۇيغۇرۇم 007غۇ .. قىنى ئاكا بۇلتۇر بىرىمەن دەپ قويغان ھىيتلىقنى بىرىۋەتمەمسىز ؟ قەرىزنى بەك ئۇزۇن سوزىۋەتسە بولمايدۇ دەيمەن .
_  تايىر ئاكا  قىنى سىز ؟؟؟  
_ ھەسەل ھەرىسىمۇ كۆرۈنمەيدىغۇ ،ھىيتتا ماڭا كاۋاپ ئىلىپ بىرىدىغان بولغان ...
_ ئەنسەرى قاق بامداتتا مەندىن ھىيتلىق سوراۋاتاتتىڭىز ئەمدىز سىزدىن سوراپ كەلدىم ،قىنى چىقىرىڭ ...نىمىنى دەمسىز ؟ نىمە بولاتتى پۇل پۇل ....
  ۋاي بەك جىق كوتولداپ كەتتىم ،ئاران كەلگەندە چۈپەيلىك قىلماي ھەممىڭلار ھىيتلىق بىرىڭلا جۇمۇ ؟ يات شەھەردە يالغۇز قالدىم ،خەجلەي دىسەم ھىچكىم ھىيتلىق بەرمىدى ،ھىيتلاي دىسەم بارغۇدەك ئۆيۈممۇ يوخكەن ، ئۇ تاغ يىقىن بۇ تاغ يىقىن ،كۆڭلۈم مىسرانىمغا يىقىن دەپ زور ئىشەنىچ بىلەن كىردىم بۇ يەرگە ،ئىشەنچىمنى يەردە قويماي ماڭا ھىيتلىق بىرىڭلا ،بولمىسا يىغلاپ مۇنبەرنى بىشىمغا كىيىمەن ....

Re:مىسرانىمغا ھىيتلاپ كەلدىم ( كۈلقىز)

مىسرانىمنىڭ تاماقلىرى ئوخشاپ قاپتۇ،مۇسلىمە مومايمۇ بىر ئوخشىغان تامىقىنى بۈگۈن ئالغاچ كېپتۇ كۈلقىز،مەن تېخى بۇ يەرگە يېڭىدىن ئەزا بولدۇم،شاتئاينىڭ بۈگۈنكى چارشەنبىدىكى قاپىغى بەك يامان،ئۇلۇغ ھېيت بولسىمۇ شەنبە،يەكشەنبە،جۈمە كۈنلىرى ئېچىلىپلا يۈرۈيدۇ ئۇ قىزچاق... ئەخلەت تۇڭىنىڭ گېپىنى قىلىپ،ئەجەپ يۈرەك ئالدى قىلىۋەتتىڭىز،مەن بەك ئۆچ كۆرەتتىم،ئەخلەتنىڭ ئارىسىدىمۇ ئالتۇن چىقار،ھېيتقا ئاز بولسىمۇ بىر شېئىرنى ئاتا قىلاي،ئابدۇخالىق ئاكام 90يىل بۇرۇن يازسىمۇ،قىممىتى يوقالمىدى دەپ ئويلايمەن،تۈنلۈك قاراڭغۇدا بىر گۆھەرنى تېپىۋىلىڭ:
توپىدا ياتسا گۆھەر،
بولماس باھاسىغا زىيان.
سۇ ئاقۇر ئاققان ئېرىقتىن،
سايدا ھەم تاشقىن قىيان.مەن مۇنبەردە سۆڭەك غاجىلاپ ئۆگىنىپ قالغان،مەنمۇ قېرى ماكماك بوۋاي بوپ قالاي دېدىم.... كۆپرەك ئاتىدارچىلىق قىلارسىلەر،چىشىم بەك قالمىغاچقىمۇ،كۆپ گەپ قىلالمىدىم،رۇزى ھېيتىڭىزغا بەرگەن نەرسەم ئاددىي بوپ قلادى،قۇربان ھېيتتا ئىزدەڭ،شۇ چاغدا بېرەي قالغىننى......
