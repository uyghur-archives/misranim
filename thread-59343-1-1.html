ئەسسالامۇ ئەلەيكۇم! مىسرانىم مۇنبىردىكى ئەزىز مۇنبەرداشلار ھەممىڭلارنىڭ رۇزا ھىيتىغا مۇبارەك بۇلسۇن!
ھەممىڭلارنىڭ رۇزا ھىيتنى خۇشال-خۇرام ئۆتكۈزۈشىڭلارنى، ئايەم خۇشاللىقىنىڭ بارچە قەلبىنىڭ دىلىغا ئۇزۇق بۇلۇپ، چىرايىھىدىن ئايەم شادىلىقىنىڭ خۇشاللىقى بولغان كۈلكىنىڭ جىلۋىلىنىپ تۇرۇشنى ئۈمىد قىلمەن!

سوتتا ھاراق جازالانسۇن!

بۈگۈن ھەممەيلەننىڭ قۇتلۇق رۇزا ھىيىت كۈنى بولغاچقا، بۇرۇنقى ئۆرۈپ-ئادىتىمىز بۇيىنچە بۇ قىتىمقى ھىيت كۈنىممۇ ئۆز نۆۋىتىدە شۇنداق كۈڭۈل ئۆتتى، ئەتىگەندىن بىرى ئۆيدىكىلەرنىڭ ئىشلىرىغا ياردەملىشىپ، ئاندىن ئۆيدىكىلەر بىلەن بىللە ھىيىت ئۆتكۈزدۈم، چۈشتىن كىيىن ئەڭ يىقىن ئادىشىم تىلفۇن قىپتىكەن، شۇنىڭ بىلەن يىقىن دۇسلار بىريەرگە يىغلىپ پاراڭلىشىپ ۋاقىت بىريەرگە بىرىپ قاپتۇ، مەن دۇسلۇرۇم بىلەن خۇشلۇشۇپ ئۆيگە ماڭدىم. بۈگۈن مەن ناھايىتى خۇشال ئىدىم، چۈنكى بۇگۈن كۈرۈشمىگىلى ئۇزۇن بولغان ئاغىنەم ئابدۇللا بىلەن خىلى ئۇزۇنغىچە پاراڭلىشىپ، خىلى ئەھۋال سۇراشتۇق. شۇڭا بۈگۈن ھىيىت خۇشاللىقىمنىڭ ئۈستىگە يەنە بىر ھەسسە خۇشاللىق قىتىلىپ، مەن خۇددى ئۆزۈمنى كۆكتە پەرۋاز قىلۋاتقان قاناتلىق قۇشلاردەك، يەڭگىل سامادائۇچۇپ، ئەركىن پەرۋاز قىلۋاتقان كۆك ئەركىسى بۈركۈتتەك ھىس قىلىپ، شۇنداق ھاياجانلىنىپ، ئۆزەممۇ بىلىپ بىلمەيلا خىيال دىڭىزغا غەرىق بۇپتىمەن، خىيال ئىكرانىمدىن بىر-بىرلەپ دۇستۇم ئابدۇللا بىلەن كىچىكىمىزدە بىللە ئوينىغان، بىللە مەكتەپكە ماڭغان، بىللە دەرس تەكرارلىغان، بىرلىكتە توپ ئوينىغان... ئىشقىلىپ ئابدۇللا بىلەن ئىككىمىز ئەڭ يىقىن دۇسلاردىن ئىدۇق. مانا كىمنى ئويلىسا شۇ كىلىدۇ دىگەندەك ئۇنىڭ بىلەن يەنە ئۇچىرىشىپ قالدىم، ئۇنىڭ ئىيتىشچە ئۇ بۈگۈن چوڭنىسنىڭ ئۆيدە قۇندىغان بۇلۇپ كەلگەن ئىكەن، ئۇنىڭ چوڭنىسنىڭ ئۆيى بىزنىڭ مەھەللىدە ئىدى، ئۇنىڭ ئاتا-ئانىسى ئۇقۇتقۇچى بۇلۇپ خىزمىتى بەك ئالدىراش بولغاچقا كىچىكىدە ئۇنى چوڭنىسى باققان ئىكەن، شۇنداق بولغاچقىلا بىز كىچىكىمىزدىن باشلاپ بىللە يۈرۈپ، مانا بۈگۈنكى كۈنلۈكتە ئەڭ يىقىن دۇسلاردىن بۇلۇپ قالدۇق، ئىككىلىمىز شۇنداق خۇشال ھالدا پاراڭلىشىپ كىتىپ بارىمىز، ئۆيگە يىقىن قالغان ئىدى، مۇشۇ چوڭ يولدىن ئۆتۈپ ئازراقلا ماڭساق مەھەللىمىزگە كىلەتتۇق.
بۇ نىمىش ئەمدى، كۆز ئالدىمىزدىكى مەنزىرنى كۆرۈپ، ئىككىلەيلەن ھاڭۋىقىپ، جايمىزدا قاققان قۇزۇقتەكلا تۇرۇپ قالدۇق، ،مانا بىزدىن 100 مىتىرلا نىرىدا يۈز بەرگەن ۋەقەدىن بىز قاتتىق چۈچۈپ كەتتۇق. ئىككى قىز بىلەن بىر ئۇغۇل تۇرغاندەك قىلاتتى، بىردەمدە ئۇغۇل يەرگە يىقىلدى، ئۇنىڭ مەيدىسدە پىچاق سانجىقلىق تۇرۇپتۇ، ئارلىق يىراق بولغاچقا قايسى قىزنىڭ ئۇنىڭغا پىچاق تىققانلىقىنى كۈرمىدۇق، يىقىنلا يەردە خەلق ساقچىخانسى بار ئىدى، بىز بۇ ئالدىمىزدا يۈز بەرگەن ئىشقا تەئەججۈپلىنىپ، ھاڭۋىقىپ تۇرۇپ قاپتىمىز ئىسىمىزنى بىلگەندە بولسا ھىلقى يارلانغان ئۇغۇل قۇتقۇزۇش ماشنىسغا، ھىلىقى ئىككى قىز بولسا قولغا كويزا سىلىنىپ ساقچى ماشنىسغا چىقىرىلىپ بولىنغاتتى. بىز ئورنىمىزدا يەنە بىردەم تۇرۇپ قالدۇق. ئىككىلەيلەننىڭ كالىسدا بۇ ئىشقا بولغان ھەيران قىلىش، ھەم قاتتىق ئەپسۇسلىنىش خىياللىرى تۇغۇلدى، تۇيۇقسىز بۇ چاغدا بىر ئەر كىشنىڭ يىنىمىزدىن ئۆتۈپ كىتىۋىتىپ:
_ توۋا خۇدايىم، بۇ ئۇلۇغ ھىيتتا نىمە ئىش بۇلۇپ كەتتى، مەن ئۇلارنىڭ يىنىدىن ئۆتۈپ كىتۋاتقاندا ۋارقىرشىۋاتاتتى بىردەمدە مانا بۇ پاجىئەلىك ۋەقە يۈز بەردى، ئۇ ئوغۇل بالا مەسكەن، بەلىكىم ئۇ ئاۋۇ ئىككى قىزغا يامان گەپ قىلغان بۇلسا، ياكى ئۇلارنىڭ غۇرۇرغا تەككۈدەك گەپ قىلغان بۇلسا ئۇ قىزلار ئاچچىقىدا ئۇ بالىغا پىچاق تىققان بۇلشى مۇمىكىن-دەپ ئىغىر ئۇھ تارتىپ قۇيدى. ئۇنىڭ يىنىدا ئۇنىڭ بىلەن پاراڭلىشىپ كىتۋاتقان يەنە بىر ئادەم:
_ بۇ بالىلارمۇ ئۆيدە ھىيتنى ئۆتكۈزمەي، نىمە بار سىرتتا كەچ بولغان تۇرسا، ئۇنىڭ ئۈستىگە ئۇلۇغ ھىيتتا ھاراق ئىچىپ مەس بۇلۋالغان بارمۇ؟ ئاۋۇ ئىككى قىزمۇ تازا بەڭۋاش نىملەركەن، كەچ بولغان تۇرسا ئۇنىڭ ئۈسىتىگە بۈگۈن دىگەن ھىيتنىڭ بىرىنجى كۈنى ئۇيدە ئاتا-ئانىسغا يارەملەشمەي نىمە بار سىرىتتا؟ ئۇ قىزلارنىڭ ئاتا-ئانىسى بۇ ئىشنى ئۇقسا نىمە بۇپ كىتەر ھە!
بۇ كىشىمۇ خىلى ئاڭلانغۇدەك ئىغىر ئۇھ تارتتى، مانا بىز ماڭماقتىمىز، ئەنە كۆز ئالدىمىزدا ئابىياتىنقى بىزدىن 100 مىتىر نىرىدىكى ۋەقە يۈز بەرگەن جاي، مانا بۇ يەردىكى كىشلەر تىخىچە تارقىلىپ بۇلالماپتۇ، ئۇلار بۇ ئىش توغرۇلۇق خىلمۇ-خىل بەس مۇنازىرگە چۈشتى:
ئا:  _ ھەي ئۇ بالا ساق قالارمۇ؟
ب: _ نىملا بۇلسۇن ساق قالسا بۇلاتتى، ئەگەر ئۇ بالا ئۈلۈپ قالسا، ئۇنىڭ ئاتا-ئانىسى تۈگىشىپ كىتەر، ياشكەن تىخى ئۇ!
س: _ نىملا دىگەن بىلەن ئۇ بالا قۇتقۇزۇپ قىلىنسا ياخشى بۇلاتتى، بولمىسا ئاۋۇ ئىككى قىزغا ئىغىر گۇناھ ئاتىلىپ كىتىدۇ. ھەي!
د:  _ بۇ زاماننىڭ باللىرى قىلغىلى ئىش يوق، بىكارچىلىقتا يىگەن ئىشى مەيدىسنى ئاغرتقان گەپ ئۇلارنىڭ ئاتا-ئانىسى كۈتۈرەلەمدۇ بۇ دەرتىنى، ئۇ بىچارە ئاتا-ئانىلار نىمە بوپ كىتەر؟
ي: _ ھە ئۇلارنىڭ ئاتا-ئانىسى ئۆز بالىلىرنى ياخشى باشقۇرماپتۇ، بالسىنى ياخشى تەربىيلىگەن بولسا بۇ ئىش بولمايتى!
گ: _ بۇ بالىلارغىمۇ، ئۇ ئاتا-ئانىلارغىمۇ گۇناھنىڭ ھەممىنى ئارتىپ قويغىلى بولمايدۇ، مۇشۇ جەميەتمۇ ياخشى تەربىيە كۆرگەن بالىلار بولسا بەزىللىرى مۇشۇ ياشاش شارائىتىنىڭ ئوخشىماسلىقىدىن يامان بالىلارغا ئۆزگىرىپ كەتتى ئەمەسمۇ!

ئۇلارنىڭ گىپىنى ئاڭلاپ ئىككەيلەن ئويغا پاتتۇق، بىر جان دۇختۇرخاندا ھالسىراپ ھايات-ماماتلىق ئارسىدا جان تالىشىۋاتىدۇ، يەنە ئىككى جان بولسا قوللىرىغا كويزا سىلىنىپ، سوراخانىدا ئۆز جىنايتىنى تاپشۇرۋاتىدۇ، بۇ يەردىكى كىشلەر بولسا بۇ ئىش توغرۇلۇق مۇنازىرە قىلىپ بۇ بالىلارغا، ئۇلارنىڭ ئاتا-ئانىسغا ئىچ ئاغرىتۋاتقان ئادەملەرمۇ، بۇ بالىلاردىن، ئۇلارنىڭ ئاتا-ئانسىدىن نارازى بۇلۇپ، ئۆز كۆز قاراشلىرىنى يامغۇردەك تۆكمەكتە، مانا بۇ بالىلارنىڭ ئاتا-ئانىللىرىغا بۇ ۋاققا ساقچىلار خەۋەر قىلغان بولسا، ئۇلار ئۆز يۈرەك پارىلىرىنىڭ بىشغا كەلگەن بۇ قىسمەتتىن ئازابلىنىپ، يۈرەكلىرى پارە-پارە بۇلۇپ كەتكەندەك ئاغرىپ كەتكەندۇ ھە!
ئۇنداقتا بۇ جىنايەتنىڭ جاۋابكارلىقىنى كىم ئۈستىگە ئالىدۇ!
ئاۋۇ قۇتقۇزۇش كارۋىتىدا جان تالشىۋاتقان ئۇغۇلمۇ ياكى ئاۋۇ ئاچىغىقىغا پايلىماي ئادەم يارلاندۇرغان ئىككى قىزمۇ؟
ياق، ياق، ياق....ئۇنداقتا ھەققى جاۋابكار قانۇننىڭ سىرتىدا يايراپ يۈرۇپ يەنە قانچىلغان جانلارنىڭ جىنىغا زالىم بۇلۇپ، قانچىلغان جانلارنى تىرىك پىتى قىينايدۇ!
بۇ سوتتا مەن گۇۋاھچىلىققا چىقمەن، ئۇنداقتا مەن كىمنىڭ جاۋابكار دەپ ئەيبلەيمەن ئۇنى بىلەمسىز؟ بۇ مەخپى؟
بۇ دىلودىكى جاۋابكار: ئاشۇ نىجىس ھاراقتۇر.

نىمىشقا دەمسىز، ھەممىنى قىلغان ئاشۇ، ئۇ ھەممە جىنايەتنىڭ باشلامچىسى، مانا ئۇ ھەققى قاتىل، دۇنيانىڭ ھەرقايسى جايلىرىدا نۇرغۇنلىغان جىنايەتنى ئۆتكۈزگەن، لىكىن ھىچقانداق جازاغا تارتىلىپ باقمىغان، قانخور ئالۋاستى، تىخى جىنايتىدىن جازالىنىپ باقماي، نۇرغۇن جانلارنىڭ مىيىتىدىن كىيىن تىخىمۇ شۆھرەت تىپىۋاتقان، كۈرەڭلەۋاتقان، ھەر بىر بايرام، ئۇلتۇرۇش، يىغلىشلاردا سۇرۇننىڭ تۆرىدىن ئۇرۇن ئېلۋاتقان يالماۋۇز، نۇرغۇنلىغان جانلارنىڭ جىنىغا زالىم بۇلۇپ تۇرۇپمۇ، ھىچقانداق جىنايتى سۈرۈشتۈرۈلمىگەن، يەنىلا تىخى داڭلىق چولپانلاردىنمۇ بەكرەك تەشۋىق قىلنىۋاتقان دەل شۇ نىجىس ھاراق________ بۇ دىلودىكى ھەققى قاتىلدۇر.

مەن قەتئى نىيەتكە كەلدىم، بىكاردىن-بىكار ئۈچ جاننىڭ بىشغا قاراڭغۇ زۇلمەتلەرنى ئېلىپ كەلگەن. ياق... ياق...
ئۈچ جان ئەمەس، ئۈچ يۈز، ئۈچ مىڭ، ئۈچ مىليون....
ياق...ياق... ئۇنىڭدىنمۇ نۇرغۇن ھاراق ئىچىپ يۈز بەرگەن جىنايەتنىڭ ھەققى قاتىلى دەل شۇ مۇشۇ نىجىس ھاراقتۇر.


نىمىشقا دەمسىز، ئادەمنى ئوپراتسىيەدە نىمىشقا ناركوز قىلىدۇ، ئاغرىقنى توختىتىش ئۈچۈن، شۇنداق بۇنى ھەممە ئادەم بىلىدۇ. ناركوز قىلنىغان ئادەم ھىچ ئىشنى بىلمەيدۇ، پەقەت ھىچ نەرسىنى سەزمەيلا بىھوش يىتىپ ئوپراتسىيىدىن قىينالمايلا ئۆتۈپ كىتىدۇ.
ئۇنداقتا ھاراق ئىچكەن ئادەمچۇ؟ ئۇ ئۆزنىڭ ئەقىلىنى يوقىتىدۇ، ئۇنىڭ ئۆزنى كونتىرول قىلىش ئىقتىدارى ئازراق تەسىرگە ئۇچىرايدۇ.

توغرا بۇ دىلودا جاۋابكارلىقنى ئەلۋەتتە، ساقچىلار ئاشۇ قىزلاردىن كۈرىدۇ، ھەم ئۇغۇل بالىنىڭ مەس بولغانلىقىدىن كۈرىدۇ. توغرا ئۇلاردىمۇ جاۋابكارلىق بار.
لىكىن ھەققى قاتىل، پەرە ئارقىسدا تۇرۇپ قوماندانلىق قىلغۇچى كاتىباش_______ شۇ يالماۋۇز، قانخور ھاراقتۇر.

ئوبدان ئويلاپ باقايلى! ئەگەر ھاراق بولمىغان بولسا:
1-ئادەملەر ھازىرقىدىنمۇ بەكرەك ساغلام ياشايتتى.
2-ئادەملەر ھازىرقىدىنمۇ بەكرەك ئىناق ئۆتەتتى.
3-دۇنيادىكى جىنايەتلەرنىڭ سانى ئازايغان بۇلاتتى.

ھاراق قاتىل ئەمەس...ھاراق دېگەن بىر ئاڭسىز، جانسىز ماددا. ئادەم دېگەندە ئاڭ بولىدۇ، تەپەككۇر بولىدۇ. سىز ئىچەمسىز، ئىچمەيسىز؟ نۇرغۇن ئادەملەر ئىچمەيدۇ...ھاراق مېنى ئىچ دەپ كېلەمدۇ ئالدىمىزغا، ئۇنداق ئەمەسقۇ، ئىنسانلار ئىزدەپ سوراپ، چاپانلىرىنى ئىچىگە تىقىپ يۈرۈپ، ئۆيىگە ئەكىرىپ ئىچىدۇ. گۇناھ كىمدىكەن...يەنە شۇ پۇلغا تويمايدىغان ئادەملەردە، پۇل ئۈچۈن ياسايدۇ، ھاراقكەشلەر پۇلغا ئېلىپ ئىچىدۇ...ھاراقنى ھاراق ياسىمىغان ھاراقنى ئادەم ياسىغان.
ئەتىگەندە ئىشقا كىرسەم، مەندىن بۇرۇن كەلگەن خەن خىزمەتداشلىرىم، تۈنۈگۈن كىچە باغچا ئەتراپىدا ئادەم ئۆلۈپتۇ، ئۇيغۇرلارنىڭ ھېيتى بولغاندىكىن ئۇلار كۈندە ھاراق ئىچىپ جىدەل چىقىرۋاتىدۇ. ھازىر كەچ ۋاقتى سىرتقا چىقىش بىخەتەر ئەمەس دەپ بىر مۇنچە كوتۇلدىدى. مەنمۇ ئۆزۈمنىڭ كۆز قارشىنى سۆزلەپ قويدۇم. يەنە نېمە دەيمەن. ھېيتتىمۇ ھاراق ئىچىپ ئادەم ئۆلتۈرۈپ، جېدەل چىقىرۋاتقىنى، كوچىلاردا ئويۇن قويىۋاتقىنى يەنە شۇ ئۇيغۇر تۇرسا.
بىرسى چىكىۋېلىپ، بىرسى ئىچىۋىلىپ يۈرسە، كېيىن كۆردىغان كۈنىمىز با بىزنىڭ.
تۈنۈگۈن كەچتە ھېيتلىق چايغا ماڭغاچ ئىككى يەردە جىدەل كۆردۈم، ھەر ئىككىلىسى ياش بالىلار....نېمىشقا بەزى ئوغۇللار نېمىنى ئويلايدىغاندۇ.

     ئەگەر ھاراقنى ھەقىقى قاتىل دىيىشكە توغرا كەلسە ئاشۇ ھاراقنى خالاپ ئىچكەنلەرنى زادى نىمە دىيىش كېرەك .
   نىمىشقا؟ بارلىق ئۆتكۈزگەن گۇناھلىرىمىزنىڭ سەۋەپچىسىگە شەيتان بىلەن ھاراقنى چىشلەپ تارتىمىز . ناۋادا ئۇلار بولمىغان بولسا بىز يەنە سەۋەپ ئىزدەپ نەلەردىمۇ پىقىراپ يۈرەرمىز ھە!
  ئاللاھ ئىنسانلارنى بارلىق مەخلۇقاتلاردىن ئەقىللىق قىلىپ ياراتتى. ئىنسانلارغا باشقا ھايۋانلاردىن روشەن پەرق قىلىدىغان ئاڭنى، تەپەككۇر قىلىشتىن ئىبارەت ئەڭ ئېسىل نىممەتنى ئاتا قىلدى. ئەقلىمىزنى ئىشلىتىپ ئىشلارنىڭ ياخشى - يامان تەرىپىنى دەڭسىمەيلا، يامان ئىشنى قىلىپ بولۇپ سەۋەبنى باشقا تەرەپتىن ئىزدەش ئەقىلگە مۇۋاپىق كېلەمدۇ ؟
   مېنىڭچە تېمىدا ئىلگىرى سۈرۈلگەن ئىدىيە ياخشى ئوتتۇرىغا قويۇلمىغان . ئاڭلىق ئىنساننىڭ جانسىز نەرسىنىڭ قۇربانىغا ئايلىنىشنى شۇ ئىنساننىڭ نادانلىقىدىن، دۆتلىكىدىن كۆرۈش ئەقىلگە ئەڭ مۇۋاپىق. گۇناھنى جانسىز نەرسىگە دوڭگەش خۇددى ‹‹ ئوسۇرغاققا ئارپا بانا بوپتۇ ›› دىگەندەكلا ئىش.
   ئىنسان ئاللاھ بەرگەن ئەقىلدىن ئىبارەت بۇ ئىسىل نىممەتنى قەدىرلەپ، ئەھمىيەتلىك ئىشلارغا ئىشلەتكەن بولسا يوقارقىدەك پاجىئەلەر كېلىپ چىقمىغان، رەزىللىكلەر ئازلىغان ۋە نۇرغۇن كىشىلەرنىڭ يۈرىكى قان - ياش بىلەن تولمىغان بولاتتى .
  دىمەك ئادەمدە ئاڭ بولغاچقىلا، ئەقىل بىلەن ئىش قىلالىغاندىلا ئادىمىيلىكىنى ساقلاپ قالالايدۇ . ئۇنداق بولمايدىكەن ھايۋانچىلىكمۇ قەدىر قىممەتكە ئىگە بولالمايدۇ.
ھېسسىيات قارىشىڭنى ئەقلىڭنىڭ ئۈستىگە قويغان كۆنۈڭ، سېنىڭ خاتالىشىشقا يول قويغان تۇنجى كۈنۈڭدۇر.
