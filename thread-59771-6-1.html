ئۆلۈككە قانچىلىك يىغلىساق توغرا بولىدۇ؟     
   ئۆلۈم > پەتىۋالار تارىخ: 20.07.2009     
سوئال: ئۈلۈم - يىتىم بولغاندا بەزىلەر قاتتىق ئاۋازدا توۋلاپ يىغلىسا ئۆلگەن كىشىگە ياخشى ئەمەس دەيدۇ. يەنە بەزىلەر ئاتا ـ ئانىغا ئۈچ كۈن مۇسىبەت ئىچىدە كۆزنىڭ گۆھىرى ئاققۇدەك يىغلاش لازىم دەيدۇ. بۇنىڭ قايسى توغرا؟ ئۆلگەن كىشىگە قانداق مۇسبەتتە بولسا ئەڭ ياخشى؟ `  جاۋاب:   ئۆلۈككە ئاۋاز كۆتۈرۈپ يىغلاشقا بولمايدۇ. بۇ ھەقتە ھەدىستە مۇنداق دېيىلىدۇ:   
«ئۆلۈككە ئاۋاز كۆتۈرۈپ يىغلىغان كىشى، ئۆلۈپ كېتىشتىن بۇرۇن تەۋبە قىلمىسا، قىيامەت كۈنى قاتتىق ئازاپقا دۇچار بولىدۇ» – (مۇسلىم رىۋايىتى)   
پەيغەمبەر ئەلەيھىسسالام مۇنداق دەيدۇ: «ئۆلۈككە ئاۋاز كۆتۈرۈپ يىغلىغان كىشىدىن مەن ئادا – جودامەن» – (بۇخارىي ۋە مۇسلىم رىۋايىتى)  ئىسلامىيەتتە ئاۋاز چىقىرىپ يىغلىماسلىق شەرتى بىلەن ئۆلۈككە ئۈچ كۈنگىچە ھازا تۇتۇشقا رۇخسەت قىلىنىدۇ. لېكىن ئېرى ئۆلۈپ قالغان ئايال 4 ئاي 10 كۈن ئىددەت تۇتىدۇ.  ھازا تۇتماق دېمەك- ياخشى ۋە رەڭلىك كىيىم – كېچەكلەرنى، ئۇپا – ئەڭلىكنى تەرك قىلىشتۇر. ئەمما يىغلاپ – قاخشاش، ياقا يىرتىش، داد – پەرياد قىلىش، بەللىرىگە ئاق پۇتا باغلاش، توۋلاپ يىغلاش ئۈچۈن ئىجارىگە ئادەم ئېلىش، ئەرلەرنىڭ ھازا ئۈچۈن چاچ ۋە تىرناقلىرىنى ئۆستۈرۋېلىشى، ئۈچى، يەتتىسى، يىگىرمىسى، قىرىقى ۋە يىلى … دېگەندەك مۇراسىملەرنى ئۆتكۈزۈش ھازا تۇتقانلىق ئەمەس. بەلكى بۇ، غەيرىمۇسلىملەردىن ۋە يات مىللەتلەردىن بىزگە ئېقىپ كىرگەن قالاق ئىللەتلەردۇر.  دىنىمىز بىزنى مېيىتنىڭ قەرزى بولسا ئادا قىلىشقا، ئۇلار ئۈچۈن دۇئا ۋە ئىستىغفار ئوقۇشقا، ئۇلارنىڭ ھەققىدە سەدىقە قىلىشقا ۋە ئۇلارنىڭ ياخشى ئىشلىرىنى داۋاملاشتۇرۇشقا بۇيرۇيدۇ. بىز مۇسۇلمانلار ئۆيىمىزدە ئۆلۈم – يېتىم بولغاندا بۇلارنى قىلىشىمىز لازىم

مەن بېغىمغا بىر تۈپ ئالما كۆچىتى تىككەن ئىدىم ،ئۈچ يىلدىن كىيىن يوغىناپ غولى بىلىكىمدەك بولدى ھەم مىۋە بەردى .شۇ يىلى مەن ئۇنىڭ ئۇچۇدىن چۇرتلا كېسىۋەتتىم .كىيىنكى يىلى قارىسام بىرقانچە ياش نوتىلار ئۆسۇپ چىقىپتۇ .مەن بۇ نوتىلارغا نەشپۇت نوتىسىنى ئۇلىدىم .يەنە ئۈچ يىل ئۆتتى ،مەن ئۈزلۈكسىز تۇردە ئۇلاقنىڭ تۆۋۆنىدىكى شۇڭلارنى پۇتاپ تۇردۇم .نەتىنجىدە ئالما كۆچۇتىمدىن نەشپۇت يېدىم .شۇ يىلى نەشپۇتنىڭ ئۈچىدىن يەنە كېسىۋەتتىم .كىيىنكى يىلى چىققان نوتىلارغا ئامۇت نوتىسىنى ئۇلىدىم ھەم ئۇلاقنىڭ تۆۋىنىدىكى نوتىلارنى پۇتاپ تۇردۇم .مانا ھازىر ئالما دەرىخىمدىن ئامۇت يەۋاتىمەن .يىللارنىڭ ئۆتۇشى بىلەن بۇ ئامۇتنى يېسەم بەزىدە نەشپۇتنىڭ تەمى ،بەزىدە بولسا ئالمىنىڭ تەمى تىلىمغا تىتىيدۇ.بىخەستىلىق قىلىپ ،ئۇلاقنىڭ تۆۋىنىدىكى شۇڭلارنى پۇتىمىغان يىللىرىمدا بولسا ،بۇ كۆچەتتىن ئالما ،نەشپۇت ،ئامۇت مىۋىلىرى پىشىپ قالىدۇ.مەن بۇ مىسالنى ئېلىش ئارقىلىق نېمە دېمەكچى ؟ئۇيغۇرنىڭ مەدەنىيەت تارىخىنى كام دېگەندىمۇ 3مىڭ يىل ئالدىغا سۇرگىلى بولىدۇ .ئەجداتلىرىمىز مۇشۇ ئۈچ مىڭ يىللىق مەدەنىيەت  تارىخىدا «تەڭرى دىنى» ،«مانى دېنى » ،«شامان دېنى »،«بۇددا دېنى » ۋە ئۇلۇغ توغرا «ئىسلام»دېنىنىڭ ۋە شۇ دىنلار ياراتقان مەدەنىيەتنىڭ تەسىرىگە ئۇچرىغان .شۇڭلاشقان ئۈلۈم ئۇزۇتۇشتەك بەزى بىر مەدەنىيەت ھادىسىلىرىمىز تەركىۋىدە ،بۇ دىنلارنىڭ تەسىرى تاكى بۈگۈنكى كۈنگىچە ساقلىنىپ كەلمەكتە .بۇنى تۈگۈتۈش ئاسانغا چۇشمەيدۇ .سەۋەبى بىزنىڭ ئىسلامنى تەرغىپ قىلغۇچى ئىمام موللىرىلىمىزنىڭ ساپاسى ئەلمىساقتىن ئىنتايىن تۆۋەن .ئۇلار مەدرىسلەردە ياتقا ئېلىۋالغان قورئان ئايەتلىرىنى ئوقۇپ ،مۇدەرىس ئۆگىتىپ قويغان مەناسى بۇيىچەلا تەبلىغ قىلالايدۇ .ئەزەلدىن ئۆزىنىڭ تەكەپپۇرى بولۇپ باقمىغان .شۇڭا يۈز پىرسەنت دىنى تۈس ئالغان ئۈلۈم ئۇزۇتۇشتەك ئۆرۇپ-ئادىتىمىزدىمۇ تۇرلۇك پۇتاقلار ئۆسۇپ چىققان.

گەپ ئۆزىمىزدە!

بىز زادى ئىنساندىن، يەنى باشقىلارنىڭ سۆز چۆچىكىدىن قورقامدۇق ياكى
ئاللاھتىن، يەنى، ئاللاھنىڭ بۇيرىغىنىنى قىلماي، ئۇنىڭ غەزىپىنى كەلتۈرۈپ
قويۇشتىن قورقامدۇق؟

يۈرىكىمىزنى سىلاپ بېقىپ جاۋاپ بېرىپ باقايلى.

ئىشىنىمەنكى، كۆپىنچىمىزنىڭ جاۋابى، كېيىنكىسى بولۇپ چىقىدۇ. لېكىن،
ئەمىليىتىمىزچۇ؟

مەن مۇنداق بىر ئىشنى سۆزلەپ بېرەي:
مېنىڭ بىر دوستۇم چەت ئەلدە بىلىم ئاشۇرۇۋاتقان بولۇپ، ئۆتكەن يىلى 9 -ئايدا، يۇرتنى كۆرگىلى بېرىپتۇ.
ئۇنىڭ كونا يېڭى تونۇش بىلىش ، ئەل ئاغىنىلىرى كۈتىۋالغانمۇ بوپتۇ. لېكىن، سورۇندا، ....لەرنى تىكلەپ،
‹‹قىزىتىپ›› ئولتۇرماقچى بوپتىكەن، بۇ دوستۇم بۇنداق قىلىشنىڭ ئىسلامدا چەكلەنگەن ئىش ئىكەنلىكىنى
ئېيتىپ رەت قىپتۇ ۋە ئۇلارغا، نەسىھەت قىلغانمۇ بوپتۇ...

شۇنداق قىلىپ ئۇ سورۇنمۇ ئانچە ‹‹قىزىيالماي›› راق ئۇزاپتۇ. كېيىن، ئۇ دوستۇمنى ئەكىلىپ قويغان يېقىن
ئاغىنىسى، ‹‹ بىز مۇشۇنداق ياشاپ كېتىپ بارىمىز ئاداش، تېخى بايا ئەسىر نامىزىنى ئۆيدە ئوقۇپ چىققان،
ئەمدى بۇ يەردە.....ئولتۇرۇپتىمىز. نىمە ئامال دەيسەن؟ ›› دەپ دەرت تۆكۈپ يىغلاپ كېتىپتۇ...
ئەلۋەتتە، بۇ يىغىلغانلار ھەممىسى، ئوقۇغان، ئاز تولا مەلۇماتى بار ئادەملەر ئىكەن.

دوستۇم ماڭا شۇ گەپلەرنى دەپ بېرىۋېتىپ، كۆزىگە ياش ئېلىپ، تاس قالدى يىغلاپ تاشلىغىلى.
ئۇنىڭ ئۈمۈتسىزلەنگىنى، بىر كىم زورلىمىسىمۇ، (بەزى ئىشلارنىغۇ بىر كىملەرگە دۆڭگەپ، مۇشۇنداق قىلمىساق،
ئۇ بولىدۇ، بۇ بولىدۇ، دەپ كۆنۈپ كەتتۇق)، ئاللاھ توسقان ئىشلارنى ئۆز ئىختىيارلىقىمىز بىلەن قىلىپ،
ئەۋجىتىپ، ئاللاھتىن تەپ تارتىش، قورقۇش ئۇقۇمىنىڭ بىزدە ئويغانمايۋاتقانلىقى، ئازغۇنلۇقتىن قول ئۈزەي
دىمەيۋاتقانلىقىمىز ئىكەن...

قىسقىسى، گەپ ئۆزىمىزدە!!!!
ھاياتنىڭ مەنىسى سۆيۈش ۋە سۆيۈلۈشيۇر! يەنى ئاللاھ ئۈچۈن سۆيۈش ۋە ئاللاھ ئۈچۈن سۆيۈلۈشتۇر !
