ئەي تورداشلار دەردىم بار قۇلاق سېلىڭلار
ھەقىقى ئىشنى سۆزلەيمەن دىققەت قىلىڭلار
ئەركەكلەرنىڭ بېشىنى قاتۇرار مەندىكى بۇ غەم
قايسى غەمدۇ دىسەڭلار، خۇتۇن ئالماق تەس ئىكەن

يېشىم يىگىرمىگە ئۇلاشقاچ،مۇھەببەتلەشكۇم كەپ كەتتى
تەنھا ئۆتۈپ زىرىككەچ، يۇرەك زەردە بۇپ كەتتى
ئاشىق مەشۇق بايرىمىدا ، بىر جۈپلەرنى كۆپ كۆرۇپ
مىنىڭمۇ مۇھەببەتلىشىپ،خۇتۇن ئالغۇم كەپ كەتتى

خۇتۇن ئېلىش ئۈچۈن مەن، قىز ئىزدەشكە باشلىدىم
تورغا ئېلان چىقىرىپ، ئۆزەمنى داڭلاپ ماختىدىم
سوراپ يۇرۇپ دوسلاردىن، قىزلارنىڭ كىيۇ كىيۇ سىنى
تورخانىدا باۋيى قىپ ،كېچىچە ئۇخلىماي قاچىلدىم\

شۇنداق قىلىپ كىيۇكىيۇدا، نەچچە قىز بىلەن تۇنۇشتۇم
كۆرمەي تۇرۇپ ھۆسنىنى ،نەچچىسى بىلەن ئارلاشتىم
لايىق ئىزدەپ مۇنبەردە، ئۆزەمنى داڭلاپ شېئىردا
قوشاق تۇقۇپ قىزلارغا، لايىق ئىزدەشكە تىرىشتىم

شۇنداق قىلىپ مۇنبەردە، بىر قىز بىلەن تۇنۇشتۇم
ھۆسنىنى كۆرمەي مۇڭدىشىپ، ئالتە ئايغىچە ئارىلاشتىم،
يېشى مەندىن بەك چوڭكەن، لىكىن قەلىمى بەك ئۆتكۈر
يازغۇچى ،شائىر بولغاچقا ، خۇتۇن قىلىشقا تاللىدىم

كىيىن بىلسەم بۇ قىزغا، مۇنبەردە نەچچىسى ئاشىق كەن
يۇشۇرۇن ئۇچۇر يوللۇشۇپ،مەندىن تالىشىپ يۇرەركەن
بىلىپ قېلىپ بۇ ئىشنى، سوراق قىلدىم ئۇ قىزنى
مىنى سېسىق مالچى دەپ، مەندىن يۈز ئۆرۈپتىكەن

شۇندىن بىرى مۇنبەرگە،كۆپ كىرمەسمۇ بۇپ كەتتىم
توردىكى مۇھەببەتكە، ئىشەنمەسمۇ بۇپ كەتتىم
ھىچ ئىش خوشياقمىدى، ئالدانغاچقا قىزلارغا
توردىن لايىق ئىزدەپ، خۇتۇن ئالماق تەس ئىكەن

شۇنداق قىلىپ خۇتۇننى ، ئاتا-ئانامغا ئىزدەتتىم
خۇتۇن ئەپ بەر ئاتا دەپ، كۈندە جىدەل قىلاتتىم
خۇتۇن ئېلىپ بەرمىسەڭ. كالا باقمايمەن دەپ يۈرۈپ
ئاتا-ئانامنى ئامالسىز، خۇتۇن ئىزدەشكە قىستاتتىم

بىر كۇنى ئاپام تېپىپتۇ، كورلىدىن بىر قىزچاقنى
لىكىن مەن ياراتمىدىم، مۇڭغۇل چىراي ئۇ سەتنى
دىدىم مەن ئاپامغا ،  چىرايلىقنى تاپقىن دەپ
لىكىن ھەرگىز ئالمايمەن، سەن تاپقان بۇ جىنتەكنى

بىر كۈنى ئاتام تېپىپتۇ، تۇققىنىمىزنىڭ قىزىنى
خېلى گۈزەل قىز ئىكەن ، ئالماقچى بولدۇم ئۇ قىزنى
قىز بىلەن كۆرۈشمەك بۇپ، ئۇچراشتۇق بىز باغچىدا
كۆرۈپ شۇئان قىز مىنى، ئېتىۋالدى  بۇرنىنى

سېسىق پۇرىغاچ سەۋەپلىك، بۇ قىزمۇ ياراتمىدى
مىنى سېسىق مالچى دەپ، پەقەت كۆزگە ئىلمىدى
شۇنداق قىلىپ بۇ ئىشمۇ ، پەقەتلا قاملاشمىدى
راستىنى دىسەم تورداشلار ، خۇتۇن ئالماق تەس ئىكەن

بىر كۈنلەرنى بىرىدە، بىر قىز بىلەن تۇنۇشتۇم
مىجەزىمىز كىلىشىپ، خېلى ياخشى ئۆتۈشتۇم
شۇنداق قىلىپ بۇ قىزنى، خۇتۇن قىلىش ئۈچۈن مەن
ئاتا-ئانامغا يالۋۇرۇپ، ئۆيىگە ئەلچى ئەۋەتتىم

ئەلچى بېرىپ قىزنىڭ ئۆيىگە، قاپاقلىرى ساڭگىلاپ كەپتۇ
قىزىمىزنىڭ مەھرى ئۈچۈن . 40 مىڭ يۇەن بەر دەپتۇ
قىزىمىز جاپا تارتمىسۇن، ئۇرمىسۇن ھەتتا بالاڭمۇ
قوي كالا ھەرگىز باقمىسۇن، شەھەردىن بىنا ئال دەپتۇ

شۇنداق قىلىپ بۇ قىزدىن ، ئۇمىدىمنى ئۈزدۈم مەن
خۇتۇن ئىزدەپ تاپالماي ، بەكمۇ ھېرىپ قالدىم مەن
ياراتقاننى ياراتماي، ياراتقان مىنى ياراتماي
راستىنى دىسەم تورداشلار، خۇتۇن ئالماق تەس ئىكەن

شۇڭا دەيمەن تورداشلار، خۇتۇن ئالماق بەك تەسكەن
تەنھا يۇرەي دىسەم مەن، بويتاقلىقمۇ بەك تەسكەن
خۇتۇن ئىزدەپ تاپالماي، توردىن، رىئال دۇنيادىن
بەكمۇ ھېرىپ قالدىممەن، خۇتۇن ئالماق تەس ئىكەن

مەن ئاتقان تاشلىرىم، بىرسىگىمۇ تەگمىدى،
تەلەپ قويغان قىزچاقلار،مىنى كۆزگە ئىلمىدى   
مىنى سەت مالچى دەپ، بىر قاراپمۇ قويمىدى   
بويتاقلىق بەك يامانكەن ، خۇتۇن ئالمىسام بولمىدى

......................................................................................   
قىزلارنى كۆرسەم ھەر ۋاقىت ،يۈرۈگۈم دۈپۈلدەپ سۇقىدۇ   
گەپ قىلالماي شۇڭلاشقا،تىلىم دۇدۇقلاپ قالىدۇ     
قىزلار تۇرسا ئالدىمدا ، بېشىم ئەسلا قايىدۇ   
بەك ئۇياتچان بولغاچقا ، بويتاقلىقتا يۈرىمەن.     
  
قىزلار كەلسە ئالدىمغا ، نىمە دىيىشنى بىلمەيمەن.  
ئۇچراشمىسۇن كۆزلەر دەپ، تۈپتۈز قاراپ ماڭمايمەن   
قىز دوست بولمىغاچ شۇڭلاشقا،تورخانىدىن چىقمايمەن     
بەك تارتىنچاق بولغاچقا ، بويتاقلىقتا يۈرىمەن   
     
دوستلار سوراپ ھەر ۋاقىت ، مۇھەببىتىڭ بارمۇ دەپ   
تارتىنىپ يۈرمىگىن سەن ئەمدى،جۈرئەتلىك بولغىن دەپ
قىزلارنى كۆرسەڭ جىم تۇرما ، گەپدانراق بولغىن دەپ
ئەمدى شوخراق بولغىن سەن ، بويتاقلىقتا يۈرمە دەپ
     
بويتاقسىزلىق بۇ ياشاش، زىرىكىشلىك ئۆتىدىكەن   
تەنھالىقتا بۇ ياشلىق ، غېرىپسىنىپ يۈرىدىكەن   
مۇھەببەتلىشىش بىزلەرگە،كالا باقماقتىنمۇ تەس ئىكەن   
قانداق قىلاي مەن ئەمدى،بويتاق يۈرمەك بەك تەسكەن .   

بويتاقلىقنى تۈگىتىدىغان ، ئامال -چارە بارمىدۇ     
ياكى ئۇنى ساقايتىدىغان، دورا-دەرمەك بارمىدۇ     
قانداق قىلاي مەن ئەمدى، ماڭا ئامال بارمىدۇ `  
بۇ دۇنيادا ماڭا ئوخشاش،بىچارە بويتاق بارمىدۇ     
  
ساقال بۇرۇتلار خەت تارتتى،دائىم مۇشۇنداق يۈرەمدىم   
يېشىم يىگىرمىگە ئۇلاشتى ،مۇھەببەتلەشمەي يۈرەمدىم     
خىلى كىلىشكەن يىگىتمەن ، يەنە تارتىنىپ يۈرەمدىم     
تەڭ دىمەتلەر توي قىلدى ، مەنلا بويتاق يۈرەمدىم   
     
قىزلار چېغىدا يۈرەكلىك ، تەلەپ قۇيار يىگىتكە  
ھايا دىگەن ئوقۇمنى، قايرىپ قۇيار بىر چەتكە     
ھازىر زامان ئۆزگەردى، قىزلىرىمىز بەك ئەركە   
سالام خەتنى يىزىشىپ ، توشقۇزۋېتەر 10 بەتكە   

مىڭ ئۆزگىرەي دىسەممۇ ، بۇ مىجەزىم كاج ئىكەن   
شوخراق بۇلاي دىسەممۇ، مەندە يۈرەك يوق ئىكەن     
ھاراق ئىچەي دىسەممۇ ، بۇ ھاراق ئاچچىق ئىكەن   
قانداق قىلاي مەن ئەمدى،يار تۇتماقمۇ تەس ئىكەن   
     
مۇشۇنداق يۈرىۋېرىپ ، ئىچىم بەكمۇ سىقىلدى
بۇنداق يۈرمەي دىسەممۇ،يار تۇتماق تەس بىلىندى   
يار بولمىغاچ يانىمدا ، كۈنلىرىم مەنىسىز بىلىندى   
بويتاقلىق بەك يامانكەن، خۇتۇن ئالمىسام بولمىدى     
  
دوسلىرىم كۆرسە ھەر يەردە ،مۇھەببەت تۇتمامسەن دەيدۇ   
ئۆلمەكنىڭ ئۈستىگە تەپمەك دەپ،ئانام خۇتۇن ئال دەيدۇ     
قىزلار خەت يىزىپ كيۇكيۇدا ، تۇنۇشمامدۇق بىر دەيدۇ     
بويتاق ئىكەن دەپ مىنى ، ئەتراپىمدىن كەتمەيدۇ     
  
نەسىھەت قىلىپ دوسلىرىم، مۇھەببەتسىز يۈرمە دەپ     
سەنمۇ چوڭ بۇپ قالدىڭ،ئەمدى بويتاق يۈرمە دەپ   
ئاتا-ئاناڭ ياشاندى، ئەمدى خۇتۇن ئالغىن دەپ     
بويتاقلىقتا ئۆتمىگىن ، ئەمدى تۇرمۇش قۇرغىن دەپ   

مەن دىگەن بىر ئەركەك،بۇنداق تارتىنىپ يۈرمەي ,  
ئىزا تارتىپ قىزلاردىن ، يەرگە قاراپلا يۈرمەي     
بۇ ماڭا ياراشمايدۇ ، ھەقىقى يىگىتتەك بۇلاي     
يۈرەكنى قارام قىلىپ ، قىزلارغىمۇ گەپ قىلاي     

مەن مۇشۇنداق يۇۋاش بولغاچقا
