                            سىز كىشلىك تۇرمۇشتىكى قېينئانا ۋە كىلىن مۇناسىۋىتىگە قانداق قارايسىز؟  
                  قېينئانلار كېلىنلەرگە نىسبەتەن ياشتا چوڭ، تەجرىبىلىك، ھۆرمەتلىك كىشلەر بولغاچقا، ئۆي تۇتۇشنىڭ قائىدە –يۇسۇن ،چارە-تەدبىر لىرنى پىششىق بىلىدۇ،ۋە شۇ ئارقىلىق كىلىنلەرنى ياخشى ئادەتلەرگە، قىزلىرىنى گۈزەل ئەخلاق پەزىلەتكە يىتەكلەيدۇ.ئەگەر ياخشى تەربىيە ئالمىغان كىلىنلەرۋە ياكى قىزلار كەلگۈسىدەبەخىتسىزلىككە يۇزلىنىپ قالىدۇ.
    شۇ كۈنلەردە كۆز قارچۇقىدەك كۆرۈپ  چوڭ قىلغان قىزىڭىز بىر كۈنلەردە سىزنىڭ تەربىيىڭىزدىن ئايىرىلىپ چەت يۇرۇتلارغا ،بىلمىگەن كۆرمىگەن كىشلەرنىڭ ئارىسىغا بىرىپ كىلىن بولۇپ كىتىدۇ. ۋە كەتكەندۇ. شۇنداق قىزىڭىزنىڭ بەخىتلىك تۇرمۇش قۇرغانلىقىنى ۋە ياكى ياشاۋاتقانلىغىنى كۆرسىڭىز نەقەدەر شاتلىنىپ ،نەقەدەر سۆينۈپ كىتىسىز .ھالبۇكى ئۇلارنى جەبىر- جاپا زۇلۇم ۋە خاپىلىق ئىچىدە كۆرگۈدەك بولسىڭىز قان ياشلىرىڭىزنى تۆكۈپ قايغۇرسىز ،شۇنداق ئەمەسمۇ؟سىزگە  كىلىن  بولۇپ چۇشكەنلەرمۇ بىراۋلارنىڭ كۆز قارچۇقى ،يۇرەك پارىسى ئەمەسمۇ؟ ئۇلارمۇ  سىزگە  ئوخشاش قىزلىرىنىڭ بەخىتلىك تۇرمۇش كۆچۇرۇشنى ئىزدەيدۇيۇ،-ھىچقاچان قايغۇلۇق خەۋەرنى ئاڭلاشقا رازى ئەمەس!شۇڭا سىز قىينئانىلار ياخىشى كۆرگەن ،سۆيگەن نەرسىنى باشقىلارغا ھەم سۆيۇشنى ئۈگۈتۈشىڭىز  لازىم .كونىلار ئېيتقاندەك «ئالتۇن بىرىپ ئىلىپ بولماس ،سۇنغان كۆڭۈلنى » بىلىڭكى ئۇلارنىڭ سىزدىن كۆڭلى قالمىسۇن .
        بەزى بىر قېينئانلار باركى كېلىنلەرنى ھەر ۋاقىت قىينايدۇ ۋە خارلايدۇ ،ئۇلارنىڭ قىلغان ئىشلىرى ۋە سۆز ھەركەتلىردىن ھەدىسلا قۇسۇر ئىزدەيدۇ ۋە ئەيىپلەيدۇ ،پۇرسەت كەلگەندەئوغۇللىرىغا يامانلاپ بولار بولماس سۆزلەپ ،يامان كۆرسۈتۈشكە تىرىشىدۇ ،نەتىجىدە بىر چىرايلىق ئۆي تۇتىۋاتقان ئەر- ئايالنىڭ مۇناسىۋىتى بۇزۇلىدۇ –دە،ئوغۇللىرىنى ئاجىرىشىشقا دەۋەت قىلىدۇ. ئاجراشمىسا ئۆي ئايرىشقا زورلايدۇ.تايىنى يوق سەۋەپلەر بىلەن بىر چىرايىق ئائىلە خانۇ-ۋەيران بولىدۇ.ئوتۇردا دۈشمەنلىك پەيدا بولۇپ كىتىدۇ.ھالبۇكى بۇنىڭ ئاساسلىق سەۋەبى قىينئانىنڭ پەزىلەتسىزلىگىدىن باشقا نەرسە ئەمەس.دۇنيادا بولۇپمۇ تۆت بەش كۈنلۈك قىسقىغىنە ئومۈردە ،ئەمدى چىچەكلەشكە ،مىۋە بىرىشكە تەييارلانغان ياشلارنىڭ شىرىن ھاياتىغا زەھەر قوشۇپ بەخىتگە زامىن بولغان ئەقىلسىز خوتۇنلارغا مەيلى ھايات ۋاقىتىدا بولسۇن ۋە ياكى ئۆلگەندىن كىيىن بولسۇن ھىچكىم سىغىنىش نەزىرى بىلەن قارىمايدۇ ،رەھمەت بىلەن قارىمايدۇ .چۈنكى ئۇنى سىغىندىغانلارنىڭ كۆڭلىنى سوۋۇتۇپ مۇزلۇتۇپ  كەتكەن .
   ئەمما ياخشى پەزىلەتلىك قىينئانلارمۇ يوق ئەمەس،ئۇلارنى ھەرقانداق كىشى ئەسلەش ،سىغىنىش ۋە مىنەتدارلىق بىلەن يادلاپ تۇرىدۇ باشقىلارغا ئىبرەت ،ئۆرنەك قىلىپ كۆرسىتىدۇ .
      پەزىلەتلىك ياخشى قېينئانىلار كېلىنلىرىنى ئۆز قىزلىرىنى كۆرگەندەك ياخشى كۆرىدۇ ھەر قانداق ئىشلىرىغا يارىدەم قىلىپ ،قىلغان ئىشلىرىدىن قۇسۇر ئىزدىمەيدۇ،ئوغۇللىرىغا كىلىنىنىڭ ياخشى تەرىپىنى قىلىپ ئۇلارنىڭ ئىناق ئائىلىسىگە تىخىمۇ ئىناقلىق ئۇرۇغىنى چاچىدۇ،ئۆزى قىيىنچىلقىتا قالسىمۇ باللىرىغا يۈك بولۇپ قىلىشىدىن ئەنسىرەپ ھەرگىزمۇ قىيىنچىلىقىنى بىلدۈرمەيدۇ .ئۆي ئىشلىرىنى كىلىنگە تاشلاپ قويماي ئۆزى قىلىپ تۈگىتىدۇ،كىلىنلىرگە شۇنداق ياخشى تەربىيە ،قائىدە يۇسۇنلارنى ئۈگىتىدۇ ،كىلىنلىرىنىڭ ئازغىنە قىلىپ بەرگەن ئىشلىردىن شۇنچە مەمنۇن بولۇپ سۆينۈپ كىتىدۇ .مانا مۇشۇنداق قىينئانلارنى قانچە –قانچە تەرىپلىسەكمۇ،تىللاردا داستان قىلىپ سۆزلىسەكمۇ ئازلىق قىلىدۇ .بۇنداق قىينئاننىڭ  تەلىمى ئاستىدا بولغان كىلىنلەرمۇ ئەخلاق پەزىلەتلىك مىھرىبان ياخشى كىلىن بولۇپ چىقىدۇ .قىينئانا بولغان كىشى كىلىننى ئۆز قىزىدەك كۆرگەن يەردە ،كىلىن ئۆز ئانىسىدەك كۆرمەسمۇ ؟ ئەلۋەتتە كۆرىدۇ.
      دىمەك قېينئانلار شەپقەتلىك ۋە مىھرىبان بولسا ،كىلىنلەرمۇ قېينئانلىرغا ئىتائەتمەن بولىدۇ ،ۋە بىر ئۆمۈر خىزمىتىنى قىلىشتىن باش تارىتمايدۇ .بۇ جاھان شۇنداق جاھان سىز بىر كۈنلەردە كىلىن بولسىڭىز ئازغىنە كۈندە سىزمۇ قىينئانا بولسىز، قىينئانىسغا ياخشى خىزمەت قىلىپ دۇئاسىنى ئالغان كىلىنلەرگە نۆۋىتى كەلگەندە ئۆز كىلىنلىرىمۇ  ئۇنىڭدىن ئارتۇق خىزمىتىنى قىلىدۇ .
     ياخشى تۇرمۇش كەچۈرۈشنى كىلىنلەرمۇ ئىزدىگەندەك ،قىينئانلارمۇ نۇرغۇن يىللار كۆز قارچۇغۇدەك ئۆستۇرۇپ تەربىيلىگەن ئوغۇللىرى ئۈچۈن ياخشى تۇرمۇش ،ئىناق ئائىلە بولىشىنى ئويلىسا كىرەك .كىلىنلەر قىينئانلىرىنى تۇققان ئانىلىرى سىياقىدا كۆرۈپ خىزمىتىنى قىلىپ ئۇنى رازى قىلسا ،ھەر ئىككى تەرەپنىڭ ئارزۇسى كۆڭلىدىكىدەك ئەمەلگە ئاشقان بولىدۇ-دە ،شۇ ئۆينىڭ رىسقى كەڭرى،تۇرمۇشى باياشات،كۆڭۈلدىكىدەك بەخىتلىك ئائىلە ۋۇجۇتقا كىلىدۇ.    
    مەنبە: «سۇغات» ناملىق كىتابقا ئاساسەن ئۆز تۇرمۇشۇمغا بىرلەشتۇرۇپ يېزىپ چىقتىم.

قېينئانا بىلەن ئىناق ئۆتۈش ئۈچۈن ، كىلىن مۇھىم رول ئوينايدۇ . قېينئانىمىزنى ئۆز ئانىمىزدەك كۆرسەك ، ھەرقانداق ئىشتا ئەستايىدىل بولساق ، ئىرىمىز بىلەن ئانىسى ئوتتۇرسىدا زىدىيەت تېرىمىساق ، قېيىنئانىمۇ بىزلەرگە ئۆچلۈك قىلمايدۇ . ئەرلەرمۇ ئانىسى بىلەن ئايالى ئوتتۇرسىدىكى مۇناسىۋەتتە مۇھىم رول ئوينايدۇ . ئانىسى ئايالىڭ ، ئۇنداق ياكى مۇنداق دىسىمۇ ، ياخشى كۆرگەن ئايالى بولغاندىكىن ئانىسىغا چىرايلىق چۈشەندۈرۈپ ، ئايالىغا تەربىيە قىلىدىغانلىقىنى ئېيتسا ، ئايالى ئانىڭىز ئۇنى قىلدى ياكى بۇنى دىدى دىسىمۇ ،ئانىسى تەرەپتە تۇرۇپ چىرايلىق چۈشەندۈرسە ، ئايالىنىڭ گېپىنى ئانىسىغا ، ئانىسىنىڭ گېپىنى ئايالىغا يۆتكىمەي ، ئۇلارنىڭ مۇناسىۋىتىنى ياخشىلاشقا تىرىشسا ، قېينئانا كىلىن مۇناسىۋىتى ياخشى ئۆتۈپ كىتىدۇ .
ئۆمرۈمنى ئاللاھقا قۇلچىلىق قىلىش بىلەن بىرگە ، يولدىشىمنى رازى قىلىپ ، ئۈچ ئوغلۇمنى ئىسلام ئىچىدە تەربىيلەپ ، ئاتا _ ئانامنىڭ ئۈمىدىنى ئاقلىسام دەيمەن .
