دوختۇرخانىغا بىر دوستۇمنى يوقلاپ بىرىپ يىنىشمدا بىر تونۇشۇم ئۇچىراپ قالدى،كارىدوردا بىردەم پارڭلىشىپ تۇرساق ئاڭغىچە ئىككى ئايال بىر ئەرنىڭ پارىڭى دىققىتمنى تارتىتى،
  
-ئاداش كىرىپ قاپسىلە،دوختۇردا بىرەرسى بارمىتى؟
--بالىمىزنى ئەكىرگەن،
--نىمە بولغانتى؟
--ۋىييەي ئاداش،خەۋرىڭ يوقما،بالىمىزنى خەتنە قىلغىلى ئەكىرگەن،بىلەمسەن بۇ دوختۇرخانا كورىيەچە خەتمىنى بەك ياخشى قىلىدىكەن دەپ ئاڭلاپ ئوغلىمىزنى ئەكەلدۇق،دىگندەك كورىيەچە خەتنە قىلدۇردۇق،
-خەۋرىم يوقكەن،ئەتىگىچە ئۆيدىكىلەرنى ئىلىپ يوقلاپ كىلەرمەن،
بۇ ئەر-خوتۇنلار بەكلا خۇشال كۆرۈنەتىتى،يەنە تېخى بۇ خوشخەۋەرنى دوستلىرىغا ئۇقتۇرۇش ئۈچۈن تىلفۇن خاتىرسىنى ئاختۇرغىلى تۇردى،قارىماققىمۇ بۇ ئىككى زات بىلملك كادىردەك قىلاتىتى،مەن ھەيرانلق ئىلكىدە سىرتقا ماڭدىم،بىر نىيىتمدە ئۇ ئىككسىنى تۇتىۋىلىپ قۇلاققا تازا سالاي دىدىم،ئەمما مۇمكىنمۇ؟بۇ قانۇنغا خىلاپ ئىدى،مانا مەن تىما يازسام كۆپتۈرمىچى،تۆھمەتخور،تىخى نەچچە تىمام ئۆچۈرۈلدى،بۇغۇ مەيلى،بىز ئەزەلدىن دورامچىلق،مەنمەنلىك قىلىدىغان ئادىتىمىز تۈگىمەكتە يوق ئەدەپ كىتىپ بارىدۇ،خوش كورىيەچە خەتنە دىگەن زادى قانداق خەتنە؟ئۇلامدۇ؟ياكى ساقلا كىسەمدۇ؟ئەجدادىمىز ئوغۇللارنڭ خەتنىسىنى ئىسلامى قائىدە بويچە قىلىپ كەلگەن،بىراق زامان ئۆزگەردى دەپ ھەممىنى ئۆزگەرتىمىزمۇ؟شەرمى-ھايادىنغۇ ئاللا بۇرۇن چەتنەپ بولغان  بىز،ھىلىمۇ ئىسلام دىنىغا ئىتقات قىلىمىز،بولمىغان بولسا مەينەتچىلكتە ھەممىنىڭ ئالدىغا ئۆتىدىكەنمىز،خوش كورىيەچە خەتنە قىلىنغان ئاۋۇ شۇم چوڭ بولغاندىمۇ باشقىلار بىلمەي قالمىسۇن دەپ ئىچىپ ماڭارمۇ؟ئاۋۇ ئاتا-ئانا بولغۇچى لەنەتكەردىلەر بالىسنى يا ھىجقىز يا بەچچە قىلمىسا خىلى ئىش،توسۇش كىرەكتى ئەلۋەتتە،مەن چۈشەندۈرەي دىگەن ئەمما كىسىپ بوپتىغۇ؟
ناھىمىزنىڭ 100كۈنلۈك مەيدان پائالىتى باشلاندى،ھەر جۈمە-شەنبە كەچ ئىسىل نۇمۇرلار خەلىققە سوۋغا قىلىنىدۇ،بۇ مەيدان پائاليىتىگە مىڭلىغان ئامما ئىشتىراك قىلىدۇ،پېقىرمۇ شۇ قاتاردا ئويۇن كۆرىۋاتقان پەيتتە مەندىن 4-5مىتىر نىرىدا تۇرغان  3قىز دىققىتمنى تارتىتى،ئۇلار نۇرمال كىينگەن بولۇپ بىرى تىخى ياغلقنى ئىڭەك ئاستىدىن ئارتىپ مۇسۇلما قىزىنىڭ ئوپرازىنى يارىتىپتۇ،دەل شۇ چاغدا يىشى 40-50لەردە بار بىر ھاجىم سۈپەت،ساقال شۇنداق ياراشقان،كىيگەن كىيىممۇ ساپلا دىننى تۈس ئالغان بىر مويسىپىت پەيدا بوپ قالدى،ئاندىن قىزلارنىڭ يىنىغا كىلىپ سۈركىلىپ گەپلەشتى،گەپ ئۈنلۈك،ئاشكارە ھالدا  قىزلارنىڭ كىچىلىك باھاسى توغىرسىدا سۆزلىنىپ 180 كويدىن دىيشتى،ماڭا مۇنداقلا قاراپ قويۇپ كىتىپ قالدى،ئۇھ!!!!ياقامنى چىشلەپ توۋا دەپ قالدىم،ئاۋۇ تىشىدىن قارىسا ئەھلى -ئىمانلق كۆرۈنىدىغان ھەزىلەك ساقالنىڭ قىلغان ئىشى ھەم 3بۇزۇق پاسكىننىڭ مال بازىرىدىكى ئات-ئىشەك سودىسىدەك ھىيقماي سودا قىلىشىدىن تاس قالدىم كەينىمگە ئۇچۇپ كەتكىلى،ئا كۈنى بىر قىز بىلەن بىر ئەر پاھىشۋازنى سوراق قىلدۇق،ئوغۇل 25ياشلاردا بولۇپ ساقال قويىۋالغان ئىدى،ئاچچىقىمدا ھىلقى مازنى تەستىكىگە 5-6 نى سالدىم،
  
--ساقالنى نىمىگە قويغان،ھۇ-ھايۋان،مۇشۇ ساقال بىلەن پاھىشۋازلىق قىلىغىچە خوتۇنۇڭنى ساتساڭ بولمامدۇ؟سەن خەخنىڭ خوتۇنى بىلەن ياتساڭ دىنىڭ خوتۇنۇڭ بىلەنمۇ خەخ ياتىدۇ،بىلدىڭمۇ ماز،
ئىغىزىمدىن نىمىلەر چىقتىكىن بەك قايناپ كىتىپتىمەن،مانا ئىسلام ئەقىدىلىرىنىڭ بەلگىلىرىنى قويىۋالغان خۇنۇكلار مۇشۇنداق ئىش قىلىپ يۈرسە باشقىلار نىمىمۇ قىلار؟

ئۇلۇغ ئاللاھ بىزنى جىمى مەخلۇقاتلاردىن گۈزەل قىلىپ ياراتتى،بىلمىگەنىنى بىلدۈردى،قۇلقىمىزنى سەزگۈر قىلدى،دۇنياغا ھۆكۈمىران قىلدى،ياخشى-ياماننى ئىنق ئايىرىپ بەردى،ھالال-ھارامنى ،نەمنى قىلىشنى نىمنى قىلماسلقنى ئۆگەتتى،ئاللاھنىڭ دوستى بولغان مۇھەممەد ئەلەيھىسالامنى(ھەمدۇ-سانالاربولسۇن)بۈيۈك،مۇقەدەس قۇرئانى كەرىمنى نازىل قىلىپ بىزگە يەتكۈزۈپ بەردى،بۇنىڭدىن ئارتۇق يەنەقانداق نىئىمەتلەر بولسۇن،ئازازۇل(شەيتان)پەرشتىلەرگە دەرس بىرىدىغان مۇدەررس بولۇپ ئادەم ئاتىمىزغا ئۆزىنى چوڭ تۇتۇپ ئاللاھىنىڭ ئەمىرىگە بويسۇنماي،سەجدە قىلغىلى ئۇنىماي ئۆزىنىڭ قىلغان ئىبادىتىنىڭ مىننتىنى قىلىپ ئۇ دۇنيادا ھەرنىمە بولۇشقا رازى بوپ بۇ دۇنيادا بىز ئنسانلاردىن ئۆچ ئىلىش ئۈچۈن كۈچىنىڭ بارىچە ھازىرغىچە ھايات ياشاپ ھەركەت قىلماقتا،مانا بىزنىڭ كاللىمىزنى سىلكيىدىغان ۋاقتىمىز كىلىپ قالدى،قىيامەتىن ،ئاللاھنىڭ جازاسىدىن قورقايلى،تۆۋبە قىلايلى،يەنە جىق سۆزلىسەم ئوقۇرمەنلەرنى بىزار قىلىپ قويماي،ئامىن،
