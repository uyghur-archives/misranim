     بىر قارلىغاچ ئاق رەڭلىك بىر ئەتىرگۈلنى ياقتۇرۇپ قاپتۇ،قارلىغاچ ئۆز كۆڭلىنى گۈلگە ئىپادىلىيەلمەي نۇرغۇن كۈنلەر پىراق ئىچىدە ئۆتۈپتۇ،بىر كۈنى قارلىغاچ غەيرەتكە كېلىپ گۈلگە كۆڭلى بارلىقىنى ئىنتايىن تەسىرلىك قىلىپ بايان قىپتۇ....قارلىغاچنىڭ كۆڭلى بارلىقىنى بىلگەن ئاق گۈل قارلىغاچقا؛  -ئەگەر سىز مېنى راستىنلا سۆيسىڭىز  ،مېنىڭ رەڭگىم قىپقىزىل رەڭگە كىرگەندە تەلپىڭىزنى قوبۇل قىلىمەن ؛-دەپ جاۋاپ بېرىپتۇ.
     بۇنى ئاڭلىغان قارلىغاچ ئىنتايىن تەسىرلەنگەن ھالدا ،گۈلنى مەھكەم قۇچاقلاپتۇ،گۈل ھەر قانچە قىلسىمۇ قارلىغاچ ئۇنى قويۋەتمەپتۇ،گۈلنىڭ نەشتەردەك تىكەنلىرى قارلىغاچنىڭ ھەممە يىرىنى قانغا بوياپتۇ،ئەمما قارلىغاچ گۈلنى يەنىمۇ چىڭ قۇچاقلاپتۇ،ئاخىرىداگۈلنىڭ ئۆتكۈر تىكىنى قارلىغاچنىڭ نەق يۈركىگە سانجىلغاندا،يۈرەكتىن ئېتىلىپ چىققان بىر تامچە قان گۈلنىڭ ئاپئاق بەرگىللىرىنى قىپقېزىل رەڭگە كىرگۈزۈپتۇ،،،،،،،.
    بۇنىڭدىن قاتتىق تەسىرلەنگەن گۈل قارلىغاچنىڭ تەلىپىگە ماقۇل بولغانلىقىنى،ئۆزىنىڭمۇ كىپىنەكنى ئەمەس ۋاپادار قارلىغاچنى مەڭگۈ سۆيىدىغانلىقىنى ئىيتىپتۇ. قارلىغاچ بۇنى ئاڭلاپ كۆزىدىن بىر تامچە ياش سېرغىپ چىقىپتۇ ۋە ئالەمچە بەخىت تۇيغۇسى بىلەن گۈلگە تېكىلىپ باققىنىچە ئۇز مەشۇقى گۈلنىڭ قۇچىقىدا جان ئۈزۈپتۇ.
