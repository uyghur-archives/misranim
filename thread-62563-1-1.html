windowsمەشغۇلات سېستىمىسىدىكى Escكونۇپكىسىنىڭ سىز بىلمەيدىغان يەنە 5تۈرلۈك  رولى.....
ئادەتتە سىز كونۇپكا تاختىسى سول تەرەپ ئۈستىدىكى Escكونۇپكىسىنى بەك ئىشلىتىپ كەتمەسلىكىڭىز مۇمكىن
،،لىكىن ئىشىنىشىڭىز كېرەككى بۇ كونۇپكىنىڭ سىزنىڭ كۆپ خىل مەشغۇلاتلىرىڭىزنى قولايلاشتۇرۇش ئىقتدارى بار يەنى
بۇنى تىز ئىجرا كونۇپكىسى قىلىپ ئىشلەتسىڭىز بولىدۇ،،،،،

1.تور بەت كۆرىمەكچى بولغىنىڭىزدا ياكى كۆرىۋاتقان ئەھۋالدا،،، ناۋادا سىز بىرەر توربەتنىڭ ئېلانىنى ياكى ئادرىسىنى
خاتا چىكىپ سالدىڭىز،، ياكى سىزنىڭ ئاچقۇڭىز يوق ئەمما ئېچىلىش ئالدىدا قاچىلاۋاتقان بولسا  دەرھال Escكونۇپكىسىنى بېسىڭ
،، بۇ چاغدا شۇ توربەت دەرھال ئېچىلىشتىن توختايدۇ،،
2.تورغا چىققان ۋاقىتلىرىمىزدا ھەمىشە تىزىملىتىش ياكى ئىسىم تولدۇرۇش دىگەندەك ئىشلارغا يولۇقۇپ تۇرىمىز ،، ئاشۇنداق
تىزىملىتىدىغان ياكى ئىسىم ............ تولدۇردىغان ۋاقىتلىرىمىزدا ئېھتىياتسىزلىقتىن بىرمۇنچە خەتنى خاتا توشقۇزۇپ قويساق ،
، بىۋاستە Escنى باسساق ئاشۇ كاتەكچە ئىچىدىكى پۈتۈن مەزمۇن ئۆچىدۇ،،،،ئۆچۈرۈش كونۇپكىسىنى ئاۋارا قىلىش ھاجەتسىز،،
،، يەنەسىز خەنزۇچە خەت تولدۇرۋاتقاندا بىرمۇنچە پىڭينىچە خەتلەر خاتا بېسلىپ قالغان بولسا بىۋاستە Escنى باسسىڭىز خەت
كىرگۈزۈش قورالى رامكىسىدىكى ھەرىپلەرنىڭ ھەممىسى ئۆچىدۇ،،
3.ctlr+alt+deleteكونۇپكىسىدىن ۋەزىپە باشقۇرغۇچ ( 任务管理器) نى قوزغۇتۇشتىن باشقا يەنە ctrl+shift+Escنى تەڭ
بېسىش ئارقىلىقمۇ قوزغاتقىلى بولىدۇ..
4.مۇبادا بىز بىر قانچە كۆزنەك مەشغۇلات قىلۋاتقان بولساق، يەنە بىر مەشغۇلات كۆزنىكىنى دەرھال قوزغىتىش ئۈچۈن alt+Escنى
باسساق باشقا  مەشغۇلات كۆزنىكىمىز قوزغىلىدۇ،،،
5.ناۋادا  بىرەر مەشغۇلاتتا جەزىملەشتۈتۈش ۋە ئەمەلدىن قالدۇرۇش دىگەن خەت بار رامكا بولسا  سىز ئەمەلدىن قالدۇرماقچى
بولسىڭىز بىۋاستە Escنى بېسىڭ،،،ئاشۇ ئەمەلدىن قالدۇرۇش (取消)خېتى بار كۆزنەكلا بولسا ئىشلەيدۇ
