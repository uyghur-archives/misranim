                                                                             ئۇچۇق يېزىلغان كۈندىلىك خاتىرە              
بۈگۈن2011-يىل1-نۇيابىر ھەپتىنىڭ ئىككىنچى كۈنى
        گەرچە تۈنۈگۈن دەرسىم بىر قەدەر كۆپ بولۇپ بەك چارچاپ كەتكەن بولساممۇ،كۈزۈمگە ھېچ بىر ئۇيقۇ كىلىدىغاندەك قىلمايىتتى. شۇندىمۇ يەنىلا ئەتىكى دەرىسىمگە دەخلى بولمىسۇن دەپ كومپىيوتىرىمنى ئۈچۈردۈم دە،ئۇخلاش ئۈچۈن كارۋىتىمغا چىقتىم.ئەمما يەنىلا ئۇخلىيالمايۋاتاتتىم،چۈنكى كۆز ئالدىمغا بىرسىنىڭ سىماسى كىلىۋالغان بۇلۇپ ئۇنىڭدىن كۆز نۇرۇمنى ئۈزەلمەيۋاتاتتىم،نەچچە رەت كۈزۈمنى مەجبۇرى يۇمۇپ ئۇخلاشقا تىرىشتىم،ئەمما بۇنداق قىلساممۇ ئۈزۈمنى ئۇخلىتىشقا قۇدرىتىم يەتمىدى. چۈنكى كۆز يۇمۇلغان بىلەن يۈرەك توختىمايدىكەن،ئۇ تاكى بىز مەڭگۈگە كۆز يۇمغانغا قەدەر توختىماستىن سۇقۇپ بىزنى ھاياتنىڭ بىر-بىرىدىن ئېسىل مىلودىيەلىرىگە  جۆر قىلدىكەن.
      مانا ھازىر ۋاقىت 1-نۇيابىر سەھەر سائەت2:35 مىنۇت ئۈتۈپتۇ،دىمەك مەن ئۇخلاش ئۈچۈن كارۋاتقا چىققىلىمۇ تەخمىنەن ئۈچ يېرىم سائەت بۇلۇپ قاپتۇ،تېخىچە ئۇخلىيالىغىنىم يوق،شۇڭىلاشقا ئورنۇمدىن تۇرۇپ باشقا بىر ئىش قىلىپ  خىيالىمنى بۇراش ئۈچۈن كارۋىتىمدىن پەسكە چۈشتۈم.ۋە  ئەزەلدىن كۈندىلىك خاتىرە يېزىش ئادىتىم يوق ۋە ئۇنى يېزىشقىمۇ قىزىقمايدىغان بىر ئادەم تۇساتتىن  مىنى ئۇيقۇدىن قۇيۇۋاتقان بۇ ئىشنى بىر پارچە كۈندىلىك خاتىرە تەرىقسىدە خاتىرلەپ قۇيۇش ئۇيىغا كەلدىم ۋە كومپىيوتىرىمنى قايتىدىن ئېچىپ بۇلارنى يېزىشقا كىرىشتىم.
        كونىلاردا :"تۇخۇ داڭگال چۈشەيدۇ،ئۆچكە جاڭگال"دەيدىغان بىر تەمسىل بار.شۇنىڭغا ئوخشاش مەندىمۇ يېقىنقى بېر نەچچە يىللادىن بىرى شۇنداق بىر خىيال باركى ئۇ بولسىمۇ بىر قىزنىڭ(يەنى ئۈزۈم ياخشى كۆرگەن شۇ قىزنىڭ)ھەقىقى مۇھەببىتىگە داخىل بۇلۇش ،ۋە ئۇنىڭ بىلەن بىر ئۆمۈر خۇشال –خۇرام، بەخىتلىك تۇرمۇش كەچۈرۈشتۇر.گەرچە مەن بۇرۇن قىزغا نەچچە رەت سۆيگۈمنى ئىزھار قىپمۇ ئۇنىڭ:"بىز تېخى كىچىك،بۇ ئىشلارغا ئالدىراش يوق كىيىنچە بىر گەپ بۇلار.""ھازىر ئۇقۇشىمىز بەك ئالدىراش ،دەرىسلەرمۇ قىيىنلىشىپ كىتىپ بارىدۇ،ھەر ئىككىلىمىز ياخشى تىرشىشمىز لازىم." دىگەندەك سۆزلىرىنى ئاڭلاپ  ئۇنى ماڭا مەيلى بار بىراق ئۇقۇش تۈپەيلى ۋاقتىنچە ئۈزىنى قاچۇرۋاتىدۇ دەپ ئۇنى كۈتۈش نىيتى بىلەن ئۆز-.ۈزۈمنى بەزلەپ يۈرگەنتىم.ئەمما يېقىندا ئۇنىڭ چ چ بوشلىقىغا كىرىپ رەسىملىرىنى كۈرگۈم كەلدى ۋە ئەڭ يېقىن ۋاقىتتا چىقارغان بىر رەسىم توپنى تاپتىم ،ئەمما ئۇنىڭغا مەخپى نۇمۇر لاھىلەنگەن بۇلۇپ  سۇئالنىڭ جاۋابىنى تاپالىغاندىن كىيىن ئاندىن كۆرگىلى بولدىكەن،ئارلىقتا ئۇنىڭ يېڭى قىياپتىنى كۈرۈش تەقەززاسىدا بىلىپ_ بىلمەي بىر سائەت ۋاقىت ئۈتۈپتۇ.شۇنداق قىلىپ ئاخىرى مەخپى نۇمۇرنى يىشىپ خۇشاللىق ئىلكىدە رەسىملەرنى كۈرۈشكە باشلىدىم.
              كاللامغا توقماق بىلەن بىرنى قويغاندەك كۆز ئالدىم قاراڭغۇلىشىشقا باشلىدى،نىمە قىلارىمنى بىلەلمىدىم، يىغلايمۇ؟ياكى كۈلەيمۇ؟سۈكۈت ئىچىدە تۇرۇپ قالدىم.
                                                                              يىغلاش يا كۈلۈشنى بىلەلمەي قالدىم،
                                                                                چىرمىغاچ  قەلبىمنى چەكسىز نادامەت.
                                                                              ئاھ تەڭرىم جاۋاب بەر  ئۈتۈنەي سەندىن،
                                                                               بۇمىدى سەن ماڭا قىلغان ئادالەت؟!
     گەرچە كۈزۈمدىن تامچە ياش ئاقمىسىمۇ ئەمما يۈرۈگۈمدىن دەريا –دەريا ياشلار ئېقىپ ۋولقاندەك قىززىق تۇمۇرۇم ئىچدە ئېقىپ ئۇنى سۇۋۇتماقتا ئىدى. ئاھ......خۇدا مىنىڭ كۆرگىنىم باشقا ئادەممۇ ياكى ئۇلار ئىككىسىمۇ؟!
     مەن كۆزلىرىمگە ئىشەنمەي رەسىمنى يەنە قايتىلاپ كۈرۈپ چىقتىم.ئەمما مىنىڭ كۆرگەنلىرىم بولسا مىنىڭ ئەڭ يېقىن دوستۇم دەپ ھىساپلىنىدىغان ۋاختى كەلسەئۇنىڭ ئۈچۈن جېنىمدىن كىچىشكىمۇ  ئەرزىيدۇ دەپ ئويلايدىغان ئاغىنەم قادىر ۋە مەن ئەزەلدىن ياخشى كۆرگەن ھەم چىن مۇھاببىتىمنى ئۇنىڭ ئۈچۈن بېغىشلىغان قىز گۈلنۇر ئىككىسىنىڭ قۇچاقلىشىپ تۇرۇپ چۈشكەن بىر قاتار سۈرەتلىرى بۇلۇپ نىمە قىلىشىمنى بىلەلمەي ئاستا گۈلنۇرنىڭ بوشلىقىدىن چىكىنىپ قادىرنىڭ بوشلىقىغا كىردىم دە ئۇدۇللا ئالبۇمنى ئاچتىم.قارىسام< 10-ئاي دۆلەت بايرىمى> دەپ بىر سۈرەت توپى بار ئىكەن،مەن دەرھاللا  ئۇ رەسىملەرنى كۈرۈش نىيتىدە مائۇسنى چەكتىم،ئوخشاشلا بۇنىڭغىمۇ مەخپى نۇمۇر لاھىلەنگەن ئىكەن.يەنە كىلىپ ئۇنىڭ سۇئالى بولسا<مىنىڭ ئەڭ يېقىن دوستۇم كىم>دەپ قۇيۇلۇپتۇ،مەن ئۇنىڭ بۇرۇن ماڭا :"ئاداش مەيلى نىمىلا بولمىسۇن سەن مىنىڭ بۇدۇنيادىكى ئەڭ يېقىن دوستۇم ."دىگەن سۈزىگە ئاساسەن ئۈزۈمنىڭ ئسمىمنى كىرگۈزدىم.راس دىگەندەك بۇ سۈ رەتلەرمۇ ئېچىلدى.
                                                    دەرىتلەرگە تولدىغۇ  يۈرۈگۈم بۈگۈن،                                
                                                    نە سەۋەب بىلمىدىم قىينالدىم شۇنچە.
                                                  ئېيىتقىن سەن مىنى دوستۇڭ چاغلامسەن؟
                                                راسىت سۆزگە تەشنادۇر يۈرۈگۈم شۇنچە!
          
راسىت ! مىنىڭ گۈلنۇرنىڭ بوشلۇقىدا كۆرگەنلىرىم راسىت ئىكەن!دەرھال كومپىيوتىرنى ئەتتىم ۋە پويىز ئىستانسىسىغا قاراپ يولغا چىقتىم.
            _شاڭخەيگە باردىغان پويىزنىڭ بۈگۈننىڭ بىلتى بارمۇ؟ سۇرىدىم مەن بىلەت ساتقۇچى خادىمدىن.
          _بۈگۈننىڭ چۈشتىن كىيىن سائەت 6:00دە قوزغىلدىغان پويىزنىڭ ئۇرۇنسىز بىلتى بار ئىكەن.دەپ جاۋاب بەردى ئۇ توردىن ئاختۇرۇپ كۆرگەندىن كىيىن.
     _مەيلى ئۇرۇنسىز بولسىمۇ  بىر تال بىلەت بىرىڭ. دەپ ئۇنىڭغا بىلەت پۇلىنى سۇندۇم.
      بىلەتنى قۇلۇمغا ئالدىم دە قول سائىتىمگە قارىسام پويىز قوزغىلىشقا يەنە ئىككى سائەت ۋاقىت بار ئىكەن.پويىزدا يىگۈدەك بىر نەرسە ۋە ئىچىملىك سۇ ئالغىلى سىرىتقا چىقتىم،ئۇلارنى .ېلىپ بۇلۇپ يەنە بىر كىچىك ئۇرۇندۇقمۇ  سېتىۋالدىم. پويىز كۈتۈش زالىغا كىرگىنىمدە پويىز قوزغىلىشقا يېرىم سائەت ۋاقىت قالغان بۇلۇپ كىشىلەر ۋوگزالغا كىرىۋاتقان ئىكەن.مەن خەقلەرگە ئەگىشىپ ۋوگزالغا كىردىم ۋە قۇلۇمدىكى بىلەتكە ئاساسەن پويىزغا چىقىپ پويىزنىڭ قوزغىلىشنى تەقەززالىق بىلەن كۈتتۈم.
       گۇاڭجۇدىن شاڭخەيگە بېرىش ئۈچۈن بىر سوتكىدىن ئارتۇقراق ۋاقىت كىتىدىغان بۇلۇپ،پۈتۈن سەپەرنىڭ تىزرەك ئاياغلىشىشىنى ۋە شاڭخەيگە تىزرەك بېرىپ ئىشنىڭ ھەقىقى ئەھۋالىنى بىلىشنىلا ئويلاپ  ئەتىسى ئولتۇرۇپ بىر كىچە كىرپىك قاقماي تاڭ ئاتقۇزۇپتىمەن.يولدا يۇلۇچىلار چۈشۈشكە باشلىدى،پويىز خادىمىمۇ  بىزنى بىكار ئۇرۇنلارغا بېرىپ ئولتۇرۇڭلار دەپ ئىشىك تۈۋدىن قوغلىدى.مەنمۇ بىر ئۇرۇن تاپتىم ۋە ئۇيەردە بىر ئۇزۇنغىچە شۇ خىياللار بىلەن ئولتۇرۇپ تۇيماستىن ئۇيقۇغا كىتىپتىمەن.
       _بۇرادەر!  ئويغۇنۇڭ،شاڭخەيگە كەلدۇق.يېنىمدا ئولتۇرغان بىر سەپەرداش مىنى ئويغۇتۇپ بىزنىڭ شاڭخەيگە يىتىپ كەلگىنىمىزنى،پويىزدىن چۈشىدىغانلىقىمىزنى ئېيىتتى.
     _رەھمەت سىزگە!
    ۋوگزالدىن چىقتىم دە قادىرغا تىلىفۇن قىلدىم.
   _ئەسسالامۇئەلەيكۇم  ئاداش! قانداق ئەھۋالىڭ؟
_ۋەئەلەيكۇم ئەسسالام.نەردە سەن؟ئۇنىڭ بىلەن تىزراق كۈرۈشۈپ ئىشنىڭ ھەققىقى ئەھۋالىنى بىلىش ئۈچۈن تۈزۈكمۇ سالاملاشماي قەيەردە  ئىكەنلىكىنى سۇرۇدۇم.
_مەكتەپتە ئاداش.نىمە بولدى ،  بىرەر ئىش بولمىغاندۇ؟
_ياق ھەي.سەن مەكتەپتىن بىر يەرگە كەتمەي تۇرۇپ تۇر .مەن سەل تۇرۇپ مەكتىپىڭگە بارىمەن.
_چاخچاق قىلمىغىنە ئاداش.سەن دىگەن گۇاڭجۇدا تۇرساڭ قانداقسىگە بىردەمدە شاڭخەيگە كىلىپ بۇلىسەن؟  ئۇ مىنى چاخچاق قىلىۋاتىدۇ دەپ ئىشەنمەي مىنى سورغۇلىغىلى تۇردى.
_مەن ھازىر شاڭخەيدە.
_نىمە ! نىمىگە كەلدىڭ بۇيەرگە؟ قاچان؟ئۇنىڭ تەلەپپۇزى خۇددى بىر دۈشمىنىگە سۆزلەۋاتقاندەك ئىدى.
_شۇ سىنى كۆرگىلى،ھازىر پويىزدىن چۈشۈشۈم.
_ھە....جېنىم ئادىشىم بۇ راسمۇ؟ئۇنىڭ تەلەپپۇزى ئەمدى بىر خىل ھەيرانلىق ۋە ھاياجانلىقنى بىلدۈرۋاتاتتى.
_شۇنداق. مەن قىسقىغىلا جاۋاب بەردىم.
_ئۇنداقتا تۇرغان جايىڭدا جىم تۇر،مەن سىنى ئالغىلى بارىمەن.
_بولدى سىنى ئاۋارە قىلماي،ئۈزۈم تېپىپ بارالايمەن،توردىن مىترو ۋە كۇچا ئاپتۇبۇس لىنىيەلىرىنى ئاختۇرۇپ بىلىۋالغان.
_شۇنداقتىمۇ جايىڭدا جىم تۇر.مەن ھازىرلا بارىمەن.دەپلا تىلىاۇننى قۇيۇۋەتتى.
مەنمۇ ئامالسىز ئۇنى ساقلاپ تۇرغاچ ئۇنىڭغا نىمە دىيىش توغرىسىدا ويلىنۋالدىم.ئۇزۇن ئۆتمەي بىرىنىڭ ئادىل دەپ ئىسمىمنى چاقىرغانلىقىنى سەزدىم دە ئارقامغا بۇرۇلۇپ قارىدىم .ئەسلىدە ئۇ بۇ يەرگە كىلىپ بوپتۇ.
مىدىرلىماي جايىمدا جىم تۇردۇم.ئۇ يىراقتىن يۈگۈرۈپ كىلىپ ماڭا  گىرە سېلىپ كۈرۈشتى،بىر قۇر ئەھۋال سۇراشقاندىن كىيىن
_ماۋۇ  دۆتلىكىمنى قارىمامدىغان ،،شۇنچە ئۇزاقتىن كەلسەڭ  بىرەر ئاشخانىغا باشلىماي بۇيەردە تۇرغۇنىمىز نىمىسى،خاپا بومما ئاداش،،يۈرە تاماق يەپ بۇلۇپ ئاندىىن مەكتەپكە كىتەيلى.
_بولدى ،قۇسقۇم توق.كەسكىنلا جاۋب بەردىم مەن._ئېيتە !سىنىڭ زادى گۈلنۇر بىلەن نىمە مۇناسىۋىتىڭ بار؟ئاچچىقىمدىن كۆزلىرىم چىڭقىلىپ كۆز چاناقلىرىمغا لىققىدە ياش تولدى.
