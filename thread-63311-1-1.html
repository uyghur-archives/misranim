            كۈزنىڭ ئاخىرقى كۈنلىرى ، ھاۋا بىرخىل سۇغۇق ،ئاجىز شامالمۇ ئادەمنىڭ يۈزىگە نەشتەردەك سانجىلاتتى، ،ئاسماندا قارا بولوتلار كۆپۈيۈپ يامغۇردىن دىرەك بىرىپ تۇراتتى ،كۆڭلۈمگە بىرخىل ۋەھىمە يۇشۇرۇنغان بۇلۇپ ، بۇنداق ھاۋا مىنى تېخىمۇ بىئارام قىلىپ تۇراتتى ، كەتپىياتىم بىرخىل سۇسلا، شۇنداق ھالەتتە
دادام بىلەن ئىككىمىز ئۆينىڭ كەينىدىكى قۇرۇق ئوت چۆپلەرنى باغلاۋاتاتتۇق !
-مەمەت ئاخۇن ، مەمەت ئاخۇن ،
تۇيۇقسىز بىرنىڭ توۋلىغان ئاۋازدىن چۆچۆپ كەتتىم ، داداممۇ ماڭا بىرخىل گۇمانى نەزەردە قاراپ قۇيۇپ ، قۇلىدىكى ئىشنى تاشلاپلا ئالدىرىغانچە ئۆيگە ماڭدى ، مەنمۇ دەقىقە ئىككىلىنىشتىن دادامغا ئەگشتىم ،
-:ئەسسالامۇ-ئەلەيكۇم ،
-ۋەئەلەيكۇم ئەسسالام ، ئاللاھ ھەممىمىزگە ئۇزۇن ئۆمۆر ئاتا قىلسۇن دېدى ئۇئادەم بىرخىل ئالدىراش كەيپىياتتا ؛:-ئەسقەرجان ئىنىمىز تۈگەپتۇ، سىلىگە دەپ قۇياي دەپ كىلىۋىدىم !
-نىمە ؟
دادام بىلەن ئىككىمىز تەڭلا شۇنداق دەپ دەققىقە قارىشىپ قالدۇق ،
-ۋاي خۇدايىم ، ۋاي جېنىم بالام .
دەۋەتتى دادام سەلدىن كىيىن ھۇشۇغا كەلگەن ئادەمدەك ۋە شۇھامان قۇيۇندەك تىز ئاكاملارنىڭ ئۆيىگە يۈگۈردى ، بۇ شۇم خەۋەردىن پۈتۈنلەي ھۇشۇمنى يۇقاتقان مەن ، ئورنۇمدىلا ئولتۇرۇپ قالدىم ، نىمىلەرنى خىيال قىلىغنىمىنى بىلمەيمەن ،خېلى ئۇزۇنغىچە شۇنداق ئولتۇرۇپتىمەن ، كىيىن ئۆيگە ئانام كىرىپ كەلدى ، ئانام منىڭ بۇ ھالىتىمنى كۆرۆپ چۆچۆپ كەينىگە بىر قەدەم داجىپ كەتتى :

-ئانا ، دىدىم مەن ئورنۇمدىن چاچراپ تۇرۇپ ئۇنىڭ باغرىغا ئۆزۆمنى ئاتقانچە :-ئەسقەر ئاكام ئۇ ....، ئاكام ئۇ ....ئەمدى يۇق! جىنىم ئانا ....
مەن شۇنداق دىگەنچە ئانامنى قۇچاقلاپ ئۈنسېلىپ يىغلاپ كەتتىم !
-نىمە ؟ نىمە دىدىڭ؟ ئەسقەر ....،ئەسقەر.....ئۇ؟
مەن شۇ ۋاقىتتا ئانامنىڭ كۆزلىرىدىن ئەزەلدىن كۆرۆپ باقىمىغان بىرخىل ئۈمىدسىزلىك نۇرىنى كۆردۈم !
-ۋاي مىنىڭ باغرىم بالام !! ئانام شۇنداق دەپ قولىنى چىشلىگەنچە يىغلاپ كەتتى،ھەم دەرھاللا رومىلىنى ئارتىپ ئۆيدىن چىقتى ،بۇ ۋاققىتتا مەنمۇ ئانامغا ئەگەشتىم ،بىز كەلگەندە ئۆي ئىچى لىق ئادەم بىلەن تولغان ، يىغا ئاۋازى ئۆينى بىر ئالغان ئىكەن . ئانام ھويلىغا كىرىپلا ھەدەم بىلەن قۇچاقلىشىپ قاتتىق يىغلىشىپ كەتتى .مەنمۇ ھويلىغا كىرىپلا ئۆزۆمنى يۇقۇتۇپ قۇيدۇم ،قايتىدىن ئىچىمگە ئاچچىق بىر نەرسە كىرىۋالغاندەك تولغۇنۇپ كەتتىم . يۈگۈرۈپ دىگۈدەك ئاكامنىڭ ھۇجرىسىغا كىردىم ،
-:جېنىم ئاكا ، سەن نىمە ئۈچۈن بىزدىن بۇنچە بۇرۇن ئايرىلىپ كىتىپ قالىسەن ؟ نىمە ئۈچۈن ؟ سىنىڭ ئورنىڭدا مەن بولوپ قالسامچۇ ! ئاكا ....،
مەن ئاكامنىڭ جەسىدىنى قۇچاقلاپ ئۇزۇنغىچە ئۆزۆمنى تۇتالماي يىغلاپ كەتتىم ،ئازدىن كىيىن ئاناممۇ كىردى ، ئانام ئادەتتە ئاكامغا زىيادە ئامراق ئىدى ، شۇڭا ئىشىكتىن كىرىپلا ئاكامغا تاشلىنىپ ،ئۇنى قۇچاقلىغانچە ھۆركىرەپ يىغلاپ كەتتى ،
:-ۋاخ جىنىم ،ئوغلۇم ، كۆزۈمنىڭ نۇرى بالام ......، ۋاي مەن سىنى ھېچكىمگە تەڭ قىلمايتىم بالام ، مەن چىدىمىدىمغۇ ئوغلۇم ، سەن كەتكۈچە مەن كەتسەم بولمامدۇ جېنىم بالام .........
ھەي !! مىڭ دىسىمۇ تەقدىرنىڭ ئالدىدا بەندىنىڭ بۇيسۇنۇشتىن باشقا چارىسى يۇق ئىكەن، بىز ھەرقانچە ئاللاھقا زارلىنىپ يىغلىساقمۇ ئۇنىڭ ئاشۇ دائىم كۈلكە ئۈزۈلمەيدىغان خۇش ئاۋازىنى قايتا ئاڭلىيالمىدۇق!ئاخىرى ئەنە شۇنداق يىغا زار ئىچىدە قەدىردان ئاكامدىن مەڭگۈلۈك ئۇزۇتۇپ قۇيدۇق ، مەن نەچچە كۈنگىچە خۇددى بىر نەرسەمنى يۇقۇتىپ قۇيغان ئادەمدەك خامۇش ھالەتتە يۈردۈم .كۈندە ئۇنىڭ قەبرىسىنى نەچچە رەت يۇقلاپ چىقاتتىم ،بىرگە خۇشال يۈرگەن كۈنلىرمىزۋە بىرگە ئىش قىلغان ھەربىر ئىشىمىز يادىمغا يەتسە مەن خۇددى ئوتقا چۈشكەن قىلدەك ئاچچىق تۇلغۇنۇپ كىتەتتىم . مانا بۈگۈن يەنەئاكامنىڭ قەبرىسىنى يۇقلاپ باردىم ، ئۆيگە قايتار چېغىمدا ئۇنىڭ ئاكامنىڭ ئۆيىنى قايتا يوقلاپ ئۆتمەكچى بولدۇم .
مانا بۇ ئاكام ياتقان ئۆي ،ئۆي ئچى خۇددى ئاكام باردەكلا شۇ پېتى تۇرغان ، ھەدەملەر يۇغۇشتۇرۇشقا ئۈلگۈرمىگەنمۇ ياكى مۇشۇنداق تۇرسۇن دىگەنمۇ ،ئەيتاۋۇر، ھەممە نەرسە ئۆز جايىدا ئىدى. جىمجىت ھەم سۇغۇق بۇ ھۇجرىنى كۈن چىقىش تەرەپتكى دېرىزە بىر خىللا يورۇتۇپ تۇراتتى .تامغا ھەرخىل ھۆسىنىخەتلەر قادالغان .دېرىزە يىنىغا قۇيۇلغان كىتاپ ئىشكابىغا بىر نەچچە خىل يېزىقتىكى رومان كىتاپلار قۇيۇلغان ، ...،ھەممىسى ئاكامنىڭ نەرسىلىرى .تۇنۇش ھىد ، تۇنۇش پۇراق ..،
ئۆي ئىچىدىكى ھەممە نەرسىگە تويماي قارايتىم ، ئاكامنىڭ گىتتىرى ، ئاكامنىڭ كومپيوتىرى، ئاكامنىڭ كىتاپلىرى ...، تۇساتتىن ئاكامنىڭ يىزىق ئۈستىلىگە قۇيۇلغان سىىرتقى مۇقاۋىسى قىزىل رەڭ بىلەن ئاددىلا يۈرەك نۇسخىسى نەقىشلەنگەن ئاق تاشلىق خاتىرە مىنى ئۆزىگە تارتتى ، مەن بىر خىل سېغىنىش ھېسياتى بىلەن خاتىرىنى قۇلۇمغا ئېلىپ ئاستا ۋاراقلاشقا باشلىدىم :
ھاياتىمغا ئېچىنمايمەن ، ئەمما ئۇھسىنىمەن !!!!

مۇھەببەت ئۇ ھاياتلىقتىكى مەڭگۈلۈك تىما .ئەمما ئۇنىڭ قەلىپ خانىڭىزگە ھەقىقى تۈردە سىڭىپ كىرىشى ئۇنچىلا ئاسان ئەمەس.ئەلۋەتتە ! بىر-بىرنى بىر قېتىم كۈرۈپلا ياخشى كۆرۈشۈپ قالىدىغانلارمۇ يوق ئەمەس ! لىكىن ،چىن_ھەقىقى مۇھەببەتكە ئېرىشىش بەسىي مۈشكۇلدۇر!!

مەن تەكلىماكاننىڭ جەنۇبىدىكى قىرغىقىدىكى يېزىدا ئۈسۈپ يىتىلگەن دىھقان ئوغلىمەن.كىچكىمدىنلا ئوقۇشقا بەك ھېرىسمەن ئىدىم،شۇڭلاشقىمۇ دادام ئائىلىمىزنىڭ نامرات بولىشىغا قارىماي مىنى ئالىي مەكتەپكىچە ئوقۇتتى.مەنمۇ ئائىلىمىزدىكىلەرنىڭ ئۇمىدىنى يەردە قويماي تىرىشىپ ئوقوپ . ئالىي مەكتەپكە كىرگەن .ئوقۇش پۈتتۇرشكە بىر يىل قالغاندا تەقدىرنىڭ ماڭا قىلغان رەھىمسىزلىكى بىلەن ئۆيگە يېنىپ كەلدىم .مانا بۈگۈن ناھىيىمىزدە بىر كىچىك كىتاپخانا ئچېپ ،ئۆمرۈمنىڭ ئاخىرىقى كۈنلىرىنى كىتاپ ۋە تور بىلەن ھەپىىشىپ ئۆتكۈزۈۋاتىمەن! مىنىڭ يەنە قانچىلىك ياشىيالايدىغىنىمنى بىلمەيمەن .بەلكىم بېشىندىن ئۆتكەن بۇ قىسمەتلەرنى خاتىرە قىلىپ يزىپ بولالماسلىقىممۇ مۇمكىن .لىكىن مەن ئۈزۈمنىڭ ھاياتىدىن ئۈمىتسىزلىنىپ باقمىدىم ! بىر كۈن ئۆتسە بىر كۈنگە ھىساپ! مانا بۇ مىنىڭ ھازىرقى كۈندىلىك خىتاپنامەم !مىنىڭمۇ گۈزەل ئارزۇ ئارمانلىرىم، ئولۇغۋار غايىلىرىم بولغان، ئەمما تەقدىر نىڭ چاخچىقى مىنى ئامالسىز قالدۇردى ، دۇنيادا ھچىقانداق نام-نىشانسىز كىتىشىتىن نۇمۇس ھېس قىلىمەن ،شۇڭا قانداقلا بولمىسۇن ، ئۆزۈمنىڭ بۇ قىسقىغىنا ھاياتىمدا بىشىمدىن ئۆتكەنلەرنى يېزىپ قالدۇرماقچى بولدۇم .

باشلانغۇچ مەكتەپنى پۇتتۇرگەن چىغىمدا دادام سىنىپ مەسئۇلىمىزنىڭ پىكىرى ،بىلەن مىنى ناھىيدىكى بىر ئوتتورا مەكتەپكە ئاپىرىپ بەردى .كىچىكىمدىنلا سەھرادا چوڭ بولغاچقا ،شەھەرگە كۆنۇپ بولغۇچىلىك ۋاقىتتا ئۈچ يىللىق ئۆمرۆم ئۆتۈپ كەتتى .جىمغۇر بولغاچقىمىكىن تولوقسىزدا بىرەر يىقىن دوس تاپالماي، زېرىككەندە كىتاپ رومان ئوقوپ يۈردۇم.ئانامىنىڭ كۆز يېشىغا چىدىماي تولوق ئوتتورىنى ئۈرۇمچىدە ئوقوش پۇرسىتىدىن ۋازكىچىپ شەھەرلىك تولوق ئوتتۇردا ئوقوپ قالدىم.بۇيەدىمۇ تولوقسىزئوتتورىغا ئوخشاش تىكەندەك يالغۇز قىلىشتىن قورقۇپ،ئۆزۈمنى باشقىلارغا شوخ،چىقىشقاق قىلىپ كۆرسىتىشكە تىرىشتىم.دەرىسنىمۇ ياخشى ئۆگۈنۇپ سىنىپتا ئەلاچىلاردىن بولدۇم،باشقلارغا ئاكتىپلىق بىلەن يىقىنلىشىپ، مۇزىكا ئوقوتقۇچۇمنىڭ كەينىگە كىرۋىلىپ گىتتار چىلشنى ئۆگەندىم.ئۇسۇل دىگەننى بۇرۇندىنلا خېلى ياخشى ئوينايتتىم .مەكتەپتە ئۆتكۈزگەن مەيلى زىھىن سىناش ياكى ئەدەبىيات -سەنـئەت پائالىيىتى بولسۇن ، ھەممە يەردە بار ئىدىم .شۇنداق قلىپ ھەرتەرەپتە باشقىلارنىڭ كۆزىگە كۆرۈنۈشكە باشلىدىم.
ئۇ كۈنى چۈشتە مەن چىملىقتا يالغۇز مۇئەللىم ئۆگەتكەن «پۇشايمان" ناملىق ناخشىنى چالا - بۇلا چېلىپ ئولتۇراتتىم .بىر چاغدا كەينىمدىن كۈلكە ئاۋازى ئاڭلاندى . قارىسام بىزنىڭ سىنىپتىكى ئككى قىز ماڭا قاراپ كۈلۈشۈپ تۇرۇپتۇ .ھازىرقى مەن ئىلگىرىكى مەن ئەمەستىم ،شۇڭا چاخچاق ئارىلاش ئۇلارغا قاراپ :
-قانداق ؟ بەك مۇڭ چېلىۋەتتىممۇ نىمە ؟ دېدىم مەن .
-ۋىيەي! نەدە ،مۇنداقلا قاراپ قويدۇق .دېدى ئارىدىن ئاينۇر ئسىملىك ساۋاقدىشىم يىنىدىكى رەيھانگۈلگە قاراپ .
-چىلىشنىغۇ يامان ئەمەس چالدىڭىز ،ئەمما ساۋاقداش بولغاندىكىن ، يالغۇز ئولتۇرۇپ پۇشايمان قىلغۇدەك بىرەر ئىش بولغان ئوخشايدۇ دەپ..
-ھاھاھا........
رەيھانگۈلنىڭ تەستىق سۆزىدىن ھەممىز كۈلۈشۈپ كەتتۇق! ئۇلارنىڭ يىنىمغا كىلىپ ئولتۇرىشى مىنى سەل تارتىندۇرۇپ قويدى .
بۇغداي ئۆڭ، قارا قاشلىرى ئاستىدىكى بىرجۈپ بۇلاقتەك شەھلا كۆزلۈك بۇ قىزنىڭ ئىسىمى رەيھانگۈل بۇلۇپ ،راستىمنى ئېيتسام مەن خېلىدىن بۇيان ئۇنىڭغا مەپتۇن بۇلۇپ قالغان ئىدىم.،ئۇنىڭ پات-پات مەن بىلەن تۇر مۇشتىن ئالغان تەسىراتلىرىمىز توغرۇسىدىكى ئۇزۇندىن -ئۇزۇنغا سوزىلغان مەنىلىك پاراڭلار، مىللەت ۋە كەلگۈسە ھەققىدىكى پەلسەپىلك مۇنازىرىلەر ........،ئشقىلىپ ئۇ ماڭا باشقا قىزلارغا پۈتۈنلەي ئوخشىمايدىغان تەسىر قالدۇرغان ئىدى.شۇ كۈنلەردە ئۇنى ياخشى كۆرۇپ قالغانلىقىمنى ،دائىم ئۇنىڭ بىلەن بىللە بۇلىشنى ئستەيدىغان بولۇپ قالغانلىقمىنى سەزگەن ئىدىم . ئۇنىڭغا بىر نەچچە قېتىم كۆڭلىمدىكىنى دىمەكچى بۇلدۇم -يۇ ،ئەمما قانداق دىيشىمنى بىلەلمەي يۇرىۋاتاتتىم.
شۇ كۈنى بىز ئۈچىمىز پاراڭلىشىپ ئولتۇرۇپ كەچلىك تاماق ۋاختى بولغىنىنىمۇ سەزمەپتۇق،بىر چاغدا ئاينۇرغا تىلفۇن، كىلىپ دادىسىنىڭ ئۇنى مەكتەپ ئالدىدا ساقلاپ تۇرغانلىقنى دىگەندە، ئاندىن كەچ كىرگىنى يادىمىزغا يەتتى.ئاينۇر كەتكەندىن كىيىن مەن رەيھانگۈل بىلەن يالغۇز قالدىم .
-كەچ بوپ كىتىپتۇ،مەن ياتاققا قايتاي ! دېدى ئۇمۇ مىڭىشقا تەمشىلىپ،
ئۇزۇندىن بىرى مۇۋاپىق پۇرسەت كۈتۈپ تۇرغان مەن ئەمدى يەنە مۇنداق پۇرسەتنىڭ كىلىشىنىڭ تەسلىگىنى ئوبدان بىلەتتىم،شۇڭا تۆمۈرنى قىزىقىدا سوقماقچى بولدۇم :
-بۈگۈن شەنبە ئىكەن،كەچتە مۇزاكىرەمۇ يوق، خالىسىڭىز بىللە تاماق يەپ كىلەيلى،
-كەچتە قىلىدىغان ئازراق ئىشىم باتتى ،باشقا كۇنى بولسۇن ،
-ۋاي-ۋوي ،خۇدا كۈن تولا ،ئەتە قىلىۋالسىڭىزمۇ بولار.يۈرۇڭە، بىرگە تاماق يەپ كىلىمىز !
ئۇ سەل ئككىلىنىپ تۇرغاندىن كىيىن ،
- ئەمسە سىز مىنى مەكتەپ دەرۋازىسى ئالدىدا ساقلاپ تۇرۇڭ ،مەن كىيىمىمنى يەڭگۈشلەپلا چىقاي ،
"ئاللاھقا مىڭ مەرتىۋە شۈكرى، مانا بۇ مىنىڭ بۈگۈن ئەڭ خۇشال قىلىدىغان گەپ "دېدىم ئىچىمدە ۋە:
- ماقۇل ئەمسە، تىزرەك بۇلۇڭ! دەپ ئۇنى ئالدىراتتىم
ئۇ تىزلا چىقتى ، بىز مەكتەپتىن ئۇنچە يىراق بولمىغان بىر ئاشخانىغا كىرىپ بىللە تاماق يىدۇق. تاماق يەۋاتقاندا مەن ئۇنىڭدىن كۆزۇمنى ئۇزەلمىدىم .قاپقارا قاشلار ، چوغدەك لەۋلەر،ئالمىدەك زىناقلىرى ئادەمنى شۇنداق ئۆزىگە تارتاتتى .......ئۇنىڭغا خېلى ئۇزۇن شۇنداق قاراپ تۇرۇپ قالغان بولسام ،ئۇ سىزىپ قالغان چېغى :
- نېمىنداق قاراۋېرىسىز،ئادەمنى خىجىل قىلىپ؟مىنى بۇرۇن كۆرمىگەنمىدىڭىز يا ؟
مەن دەرھال ئەسلىمگە كەلدىم ،ئەمما چاندۇرمايلا،ئوڭايسىزلانغان قىياپەت بىلەن :
-تاماق يىگەندىكى ھالىتىڭىز بەكمۇ چىرايلىقكەن،شۇڭا قاراپ قويدۇم ،بولماپتىمۇ؟ -
ئۇ ئۈندىمىدى. ئارىنى بىر پەس جىمجىتلىق قاپلىدى.بۇ جىمىتلىقنى يەنە ئۇ بۇزدى،
- بولدى قايتىپ كىتەيلى .مەن تويدۇم .قىسقا دۇئادىن كىيىن مەنمۇ ئۇنىڭغا ئەگەشتىم،پۇكەيگە كىلىپ پۇل تۆلىمەكچى بۇلوپ تۇر سام ،ئۇ مەندىن چاققان چىقىپ قالدى :
- بولدى بۈگۈن مەن مىھمان قىلاي؛ ، دەپلا مىنىڭ ياق دىيىشلىرىمگىمۇ ئۇنىماي مەندىن بۇرۇن پۇلنى پۇكەيگە قۇيۇپ ئالدىمدا چىقىپ كەتتى . مەن ئۇ بەر گەن پۇللارنى ياندۇرۋىلىپ،ئۇزۈم قايتىدىن تۆلەپلا ئۇنىڭ ئارقىسىدىن يۈگۈردىم. ئۇنىڭ كەينىدىن يىتىشىپ پۇلنى ئۇنىڭغا تەڭلىدىم .ئھتىمال ئۇ مىنى بۇنداق قىلىدۇ دەپ ئويلىمىغان بولسا كىرەك،شۇئان چىرايىنى تۈرۇپ:
-ساراڭمۇسىز؟ نمە قىلغىنڭىز بۇ ؟
مەن سەل ئوڭاي  سىزلىنىپ قالدىم ،شۇنداقتىمۇ مەن دىگەن بىر ئوغول بالا ، تۇنجى قىتىمدىلا قىزبالا پۇل تۆلەپ يۈرسە سەت ئەمەسمۇ !
-ھىم !ئىسىڭىزدىدۇ ، مەكتەپتە مەن مىھمان قىلىدىغان بولغانغۇ؟
-نەدە ؟ مەن مىھمان قىلساممۇ ئوخشاشقۇ! بۇنىڭدىن كىيىن مۇشۇنداق قىلسىڭىز سىز بىلەن ئىككىنچى بىر يەگە بارمايمەن !
-ھە ، ھە ماقۇل ،ماقۇل! مەن دىققەت قىلاي ،مەن دىققەت قىلاي ،
شۇنىڭ بىلەن مەن ئاستا پۇلنى يانچۇقۇمغا سالدىم،ئۇ ئۇدۇل مەكتەپكە قايتىپ كىتىشنى دەپ تۇرۋالدى ، مەن خېلى چىڭ تۇرۇپ ئاران دىگەندە ئۇنى بىردەم كوچا ئايلىنىپ ئاندىن قايتىپ كىتىشكە ماقۇل قىلدىم !شۇنىڭ بىلەن ئىككىمىز بىردەم كوچا ئايلىنىپ خەلىق مەيدانىغا كىلىپ خالىيراق بىر يەردە ئولتۇردۇق ،ئۇماڭا ئائىلىسى توغرۇلۇق جىق گەپلەرنى قىلىپ بەردى .
رەيھانگۇل مەن بىلەن بىر ناھىيىدىن بۇلۇپ، ئۇ ئائىلىسىنىڭ يالغۇز قىزى ئىكەن،ئانىسى كىسەلچان بۇلۇپ بىر يىلنىڭ يىرىمى ئۆرە يۈرسە ،يېرىمى كىسەل كارۋىتىدا ئۆتىدىكەن ، ئەمما دادىسى خۇددى دادامغا ئوخشاش ئۇنىڭ ئوقوشىنىڭ ياخشىلىقىغا پايلىماي ئۇنى ئوقۇتىۋېتىپتۇ . ئۇ شۇلار نى دەپ كلىپ يىغلاپ كەتتى .
مەن ئۇنىڭغا نىمە دەپ تەسەللى بىرشنى بىلمىدىم ،مەن سىز نى ياخشى كۆرىمەن ،قورۇقماڭ مەن باردەيمۇ ؟ئۇ خېلىغىچە شۇنداق يىغلاپ تۇردى ،بىر چاغدا مەن ئاستا غەيرەتكە كىلىپ قۇلۇم بىلەن ئۇنىڭ كۆز يېشىنى سۇرتتىم،ھەمدە ئاستا پەس ئاۋازدا :
-بولدى كۆڭلىڭىزنى يىرىم قىلماڭ ،ھەممە ئشلار ئۆتۈپ كىتىدۇ ،ئۈمدۋار بۇلۇڭ!
-رەھمەت سىز گە .
مەن ئۇنىڭ كۆڭلىنى ئاۋۇندۇراي دەپ بىر نەچچە يۇمۇر دەپ بەردىم ،ئۇنىڭ كەيپىياتىمۇ خلى كۆتۈرلىپ قالغاندەك قىلاتتى .
ئارىنى يەنە جىمجىتلىق باستى ،ئۇ :
-نىمە بولدىڭىز،جىمىپ كەتتىڭىزغۇ ؟- دەپ سورىدى .
مەن شۇتاپتا كۆڭلىمدىكىنى قانداق دىيىش توغرىسىدا قاتتىق ئويلىشىۋاتاتىم ،
-ھىچ، ھىچنىمە بولمۇدۇم،- ھاياجاندىن ئاۋازىم تىتىرەپ كىتىپ باراتتى، ئەنە شۇ چاغدىلا بىر سىگە كۆڭۈل ئىزھار قىلىشنىڭ قانچىلىك تەس ئىكەنلىكىنى ھىس قىلىپ يەتتىم !
ئۇ مىنىڭ جىددىيلىشىپ قېلىۋاتقىنىمنى سىزىپ :
- نىمىش بولسا دەۋىرىڭ،ئوغول بالا تۇرۇپ مۇنداق كىكەچلەپ يۈسىڭىز سىزگە قاملاشمايدىكەن ،
مەن ئاخىر جۇرئەتكە كەلدىم .ئۇنىڭ كۆزىگە تىكىلىپ تۇرۇپ دېدىم
- رەيھان ،مەن سىز گە ئۇزۇندىن بىرى گەپنى دىيەلمەي كەلگەن ئىدىم،
- مىنى دوستىڭىز قاتارىدا كۆرسىڭىز، نىمىش بولسا تارتنىماي دەۋىرىڭ مەن سەل جىمجىتلىقتىن كىيىن يەنە دېدىم :
-رەيھان سىزنى ياخشى كۆرىمەن ،سىز بىلەن ئاخىرغىچە بىللە ئۆتسەم دەيمەن !
بۇ گەپلەرنى دەپ بولغۇچە يۈرۈكۈم ھېلىلا قىپىدىن چىقىپ كىتىدىغاندەك قاتىتىق سوقوپ كەتتى .ئۇمۇ بۇ گىپىمنى ئاڭلاپ يەرگە قارىۋالدى .
-ماقۇل دەڭ، بىز چۇقۇم بەخىتلىك بۇلالايمىز ،ماڭا ئىشىنىڭ!
ئۇ سەل ئۇيلىنىپ تۇرىۋالغاندىن كىيىن :
-بۇ سەل تۇيۇقسىز بوپ قالدىمۇ قانداق ،مەن .....،ماڭا ئويلنىشقا پۇرسەت بىرىڭ،مەن بىر ياخشىراق ئويلىنىۋالاي،
مەن بەك ھاياجانلىنىپ كەتكەن ئىدىم ،
- بولىدۇ ،مەن ساقلايمەن ، قاچان ئويلىنىپ بولسىڭىز شۇچاغدا جاۋابىنى بىرىڭ!
ئۇ بۇلىدۇ دىگەن مەنىدە بىشىنى لىڭشىتىپ قويدى .
ئارىنى يەنە جىمجىتلىق باستى ،مەن ئەمدى نىمە دىيىشنى بىلەلمەي تۇرۇپ قالدىم ، دەل شۇ پەيتتتە ئۇ:
- قايتىپ كىتەيلى بەك كەچ بۇلۇپ كەتتى،دېدى .
ئارىدىن ئككى كۈن ئۆتكەندە مەن كىتابىمنىڭ ئارىسىدىن بىر پارچە كىچىك باغاقچە تىپىۋالدىم ، شۇ كۈنى كەچتىن باشلاپ مەن ئۇنى ئىزلەپ يۈرەتتىم ،شۇنداق قىلىپ ئككىمىز شىرىن مۇھەببەت بۆشىكىدە يەڭگىل تەرۋىنىشكە باشلىدۇق. بىللە دەرىس تەكرارلاپ، دەم ئىلىش كۈنلرى دەريا بويىغا،باغچىلارغا بىرىپ ،قايغۇدىمۇ شاتلىقتىمۇ بىرگە بولاتتۇق ، سىنىپتىكى باشقا ساۋاقداشلىرىمىزمۇ بىزگە ھەۋەستە قارايدىغان بولدى .

شۇنداق كۈنلەرنىڭ بىرىدە تۇيۇقسىز كەلگەن شۇم خەۋەر مىنىڭ تەقدىرىمنى ئۆزگەرتۋەتتى .

ئەتتىگەن بىرىنچى سائەتلىك دەرىستىن چۈشۈپلا سىنپ مۇدىرىمىز رەيھانگۈلنى سىرتقا توۋلاپ ئاچىقىپ كەتتى ،بىر ئازدىن كىيىن كارىدوردا بالىلارنىڭ ۋارقىراشقانلىرىنى ئاڭلاپ نىمە بولغىنىنى بىلمەك بولۇپ سىرتقا چىقسام ،رەيھان ھۇشىدىن كىتىپتۇ ،بىر نەچچە قىز ئۇنى يۆلەشتۇرۋاتقان ئىكەن ،مەن نىمە بولغىننى بىلەلمەي.دەقىقە تۇرۇپ قالدىم، ۋە تىزدىن ئىسىمنى يىغىپ ئۇنى كۆتۆرۆپلا مەكتەپ دوختوخانىسغا ئەكەلدىم .مۇئەللىممۇ مەن بىلەن بىللە كەلگەن ئىدى ،مەن مۇئەللىمدىن نىمىش بولغىنىنى سورىدىم .
- ئۇنىڭ ئانىسى تۈگەپ كىتىپتىكەن ،شۇخەۋەرنى دېۋىدىم ئۇ قۇبۇل قىلالمىدى بولغاي ،
ئۇنىڭغىچە مەكتەپ دوختۇرى ئۇنى ھوشىغا كەلتۇردى ،ئۇئاستا كۆزىنى ئچىپ ماڭا قارىدى ۋە ئۆزىنى قۇچىقىمغا ئاتقىنىچە ئۈن سېلىپ يېغلاپ كەتتى ،مەنمۇ ئۆزۇمنى تۇتالماي يىغلاپ كەتتىم ،مۇئەللىم بىلەن دوختۇر چىقىپ كەتتى ،ئۇ خىلى ئۇزۇن يىغلاپ ئاران پەس بولدى ،مەن ئۇنى يۆلەپ تۇرغۇزدىم.
- بىز دەرھال ئۆيىڭىزگە بارايلى دېدېم ،ئۇ باش لىڭشىتىپ ماقۇللۇق بىلدۈردى ،
مۇئەللم بىلەن خوشلىشىپ ئۇنىڭ يىزىغا بارىدىغان ماشىنىغا چىقىپلا ئۇنىڭ ئۆيىگە باردۇق.
ئۆي ئادەمگە لىق تولغان بۇلۇپ كىشىلەر جىنازىنى ئۆيدىن ئىلپ چىقىۋاتاتتى.ئۇ ماشىنىدىن چۈشۈپلا ئۆزىنى جىنازىغا --- ئانىسىغا ئاتتى ،
كىشىلەر جىنازىنى يەرگە قويدى ،ئۇ جىنازىنى قۇچاقلاپ شۇنداق ئۆكسۈپ -ئۆسۈپ يىغلايتتى ،ئۇنىڭغا قاراپ مىنىڭمۇ ئىچىم سېيرلىپ كەتتى،ئۇنىڭغىچە دادىسىمۇ ئۇنىڭ يىنىدا پەيدا بولدى ،ئۇ ئەمدى دادىسىغا ئسلىپ يىغلاشقا باشلىدى .ئاتا -بالا خىلى چاققىچە شۇنداق يىغلاشقاندىن كىيىن ،كىشىلەرتەسەللى بىرىپ ئۇلارنى ئاجىرىتىپ ،جىنازىنى يەردىن كۆتۈردى ،مەنمۇ دەرھال ئالدى تەرەپكە ئۆتۆپ جىنازىنى كۆتۈردۇم،بىز ئاستا ئاستا ئۇ ئۆيدىن يىراقلاشتۇق، رەيھاننىڭ ئاۋازىمۇ يىراقلاردا قالدى ،
مەن ئەتىسى مەكـتەپكە يىنىپ كەلدىم ،روھسىز ھالدا ئۇرنۇمدا ئۇلتۇرۇپ ئۇنىڭ ئورنىغا قارىدىم ،ئونىڭ ئورنى بوش ئىدى. شۇنڭدىن كىيىن مانا شۇنىڭدىن كىيىن مەن ئۇنى ئورنىدا قايتا كۆرەلمىدىم


ئۇ شۇقىتىم مەن بىلەن بىللە مەكتەپتىن كەتكەندىن كىيىن قايتا كەلمدى ، پەقەت ئۇنىڭ دادىسى مەكتەپكە كىلپ ئەمدى ئۇنى ئۇقۇتمايدىغىنىنى ئېيتپ ئۇنىڭ ياتىقىنى ئېلىپ كەتتى .

-بولدى تاماقنى قۇيغىن دېدى ھېلىق بۇرادەر ،ئويەردىمۇ قۇرۇق ئولتامايمىز ، قۇرسىقىڭدىن غەم قىلما ، ھەراست گىتتارمۇ بار ، غەيرەتلار سىرتتا ساقلاپ قالدى ، بىز يەنىلا تىزراق  ماڭايلى .
مەن نىجاتقا قارىدىم ، ئۇمۇ ئامالسىزلىق چىقىپ تۇرغان چىراي بىلەن مىنى بىرىشقا ئىما قىلدى ،مەن ئامالسىز ئۇنىڭغا ئەگشتىم .
مەكتەپ كۈتۈپخانىسىنىڭ كەينى تەرىپىدە بىر پارچە كەڭ سەينا بار ئىدى،ئۇ يەرلەگە ئەتراپىغا ئارچا تىكىپ ئىچىنى قۇرشاپ ،ئارىسىغا ئۇيەر- بۇيەرلەرگە تاشتىن ياسالغان ئۈستەل قۇيۇلغان بولوپ ئاستى سۈنئى چىملىق قىلىنغان ، بۇنداق سۇغۇق  كەچتە كەمدىن كەم ئادەم كىلىدىغان جاي ئىدى ،  ئاشخانىسدىن چىقىپ سۇغۇق چىرايلار بىلەن كۈتۈپ تۇرغان غەيرەت باشلىق بىر نەچچەيلەن بىلەن بۇيەرگە كەلدۇق ، ئەتراپ بىرخىل جىمجىت ھەم سۇغۇق ئىدى، قۇلنى چىقارماقمۇ ئادەم تەس كىلەتتى ، مەن ئۇلارنىڭ نىمە ئۈچۈن مىنى بۇيەرگە ئەكلىلدىغانىنى ،نىمە ئۈچۈن شۇنچە "كاتتا"  تۇغۇلغان كۈنىنى بۇنداق يەردە ئۆتكۈزۈدىغىنى پەقەت چۈشەنمىدىم ،
ئۇلار باشتا مىنى بۇيەردە يۇقتەكلا كۆرۆپ ئۆز پاراڭلىرىنى قىلىپ چىملىقنىڭ ئەڭ چېتىدىكى بىر جۇپ تاش ئۈستەلنىڭ يىنىغا كىلپ توختىدى، قۇلدىكى سۇلياۋ خالتىلارنى ئېچىپ ئىچىدىن ھاراق ۋە باشقانەرسىلەرنى ئۈستەلگە تىزدى ، شۇندىلا ئۇلارمىنى ئىسىگە ئالدىمۇ قانداق ،مىنى ئەڭ ئاۋال ئولتۇرۇشقا «تەكلىپ »قىلدى ، مەن تارتىنپ تۇرسام قاپاق تۈرۈپ سەل توڭ تەگدى ، نىمىلا دىمىسۇن كونا  ئوقۇغۇچى ھەم ياشتىمۇ مەندىن ئىككى ئۈچ ياش چوڭ بولغاچقا ئارتۇق گەپ قىلماي بىر ئۇرۇندۇقتا ئۇلتۇردۇم ،ئۇلار جەمئى بەش ئادەم ئىدى .  
ئەمدى ئولتۇرۇپ  بولوشىمىزغا ، ھاراق ئايلىنىشقا باشلىدى ، ئسسىتىلغان ھاراق بولسا كىرەك ،ئىستاكاندىن ھور ئۆرلەپ تۇراتتى ،بايام مىنى باشلاپ كەلگەن ئاداش بۇيەردە ساقىلىق قىلىپ بىر ئىستاكان ھاراقنى كۆتۆرۆپ دېدى:
-ھە ،ئاغىنىلەر ، بۈگۈن بۇرادىرمىز غەيرەتنىڭ تۇغۇلغان كۈنى،مەلۇم سەۋەپلەر تۈپەيلى.؛ دېدى ئۇ ماڭا بىرخىل سەت ئالىيپ قاراپ قۇيۇپ؛:- بۇرادىرىمىزنىڭ كۆڭلى يېرىم ، بۇ سۇرۇننى مەشە قۇرۇشتىكى مۇددىئامۇ شۇ ! مەن ئاۋال بۇرادىمىنىڭ تۇغۇلغان كۈننىنى قۇتلۇقلايمەن ، غەيرەت ئاغىە ، تۇغۇلغان كۈنىڭگە مۇبارەك بولسۇن !

ھەممەيلەن تەڭلا ئۇرنىدىن تۇرۇپ ئۇنىڭ بىلەن قايتا قول ئېلىشىپ كۆرۆشۈپ بىر ئېغىزدىن مۇبارەكلەشتى ، مەنمۇ ئۇرنۇمدىن تۇرۇپ قۇلۇمنى ئۇزۇتېۋىدىم ، ئۇزاتقان قۇلۇم ھاۋادا قالدى. كۆڭلۈم باشقىچە بىر شۇملۇقنى سەزگەندى،ئۆزەمچە شۈركىنىپ كەتتىم ئاستا ئورنۇمدا ئولتۇردۇم ، ھاراق ئايلىنىپ ماڭا كەلدى ، مەن ئۆزرە بىلەن ئىچمەيمەن دېدىم ،
-ھۇي ئۇكا ، بۇيەردە قاراپ ئولتاماي ئېىچپ قۇيسىڭىزا ،دېدى ساقى ماڭا تەنە ئارلاش .  
-مەن راسىت ئىچەلمەيمەن ، ھەم ئىچىپ باقىمىغان ،شۇڭا ......
-بولدى دېدى غەيرەت بىر خىل كەسكىن ھەم سۇغۇق تەلەپپۇزدا ؛-ئارتۇقى يوق بۇنىڭ ، ئىچمىسە مەيلى ، لىكىن بىزگە گىتتار چېلىپ ئولتۇرۇشۇپ بەرسۇن !
شۇنداق دەپ بولغىچە ساقى گىتتارنى قۇلۇمغا تۇتقۇزۇپ ئۈلگۈردى  ،مەن ئامالسىز  ئاستا چېلىشقا باشلىدىم ، شۇ تاپتا ئۆزۆمنى دۇنيادا ئەڭ بىچارە ئادەم ھېساپلايتىم ،بىر زامانلاردا ئاشىقلىق ، خۇشاللىق ئىلكىدە گىتتار چالغان مەن ئدىم ، كىين زىرىكىش ئىچىدە گىتتار چالغان مەن ئىدىم ، تەڭلىلىك ئىچدە گىتتار چالغان مەن ئىدىم ، مۇھەببەت ئۈچۈن گىتتار چالغانمۇ مەن ئىدىم ، لىكىن مەن بۈگۈن  .......  مەن نىمە قىلغىنىمغا بۇنداق خارلىنىمەن بۇيەردە ؟
كەچ كىرگەنسېرى سۇغۇق ھەدەپ ئەدەپ كىتىپ بارتتى ، گىتتار تۇتقان قۇلۇم بىردەمدىلا توڭلاپ -ئۇيۇشۇپ ،پەدىنى تۇتۇشقا كەلمىدى، لىكىن ئۇلارنىڭ  خىيالىدا يۇق ئۆزىنىڭ ئىشى،ئۆزىنىڭ گىپى بىلەن ئىدى ، قۇلۇم بەك توڭلاپ كەتكەنلىكتىنمۇ ياكى خىياللار مىنى ئەسىر قىلغانمۇ بىلىدىم . قۇلۇم توختاپ قاپتۇ ، ئۇلار بۇنى سېزىپلا ماڭا گەپ ئاتتى :- :
-نىمە توختاپ قالدىڭغۇ ؟
مەن تۇيۇقسقز ئىسىمگە كەلدىم ھەم قۇلۇمىنىڭ توڭلاپ كەتكىنىنى باھانىە قىلىپ دېدىم:
-قۇلۇم بەكلا توڭلاپ كەتتى ، چىدىمىدىم .
-ھۇھاھاھا !! ماگەپنى ، چىدىمىدىم دىگىن ، ئوغول بالا ئەمەس ئوخشايسەن ،شۇ ھالىڭغا كۆتۆڭنى قسىپ جىم يۈرسەڭچۇ بۇيەردە ئاتىكاچىلىك قىلپ يۈرگىچە .
مەن تىرىڭڭىدە بوپ قالدىم ، بۇ نىمە گەپلەر ئۆزى ،مەن نىمە قىپتىمەن ، ;
-سىنىزە دېدى يەنە بىرى ماڭا قۇلىنى چىنەپ قۇيۇپ:- بۇيەرگە ئەكەلسەك مىنى ھەممە ئادەم ئەزىزلەيدىغان بوپ كىتىپتۇ دەپ قالدىڭما؟ بىلىپ قۇي، سەندەك چۈپرەندىنى ئېلىۋەتكەن تىرنىقىمغا تەڭ قىلمايمەن مەن !
مىنىڭ راستىنىلا ئەجىلىم قالدىم  ،زەھەردىن ئاچچىق  بۇ گەپلەر نىرۋىلىرىمنى قالايمىقان قىلىۋەتتى ،ئاچچقىمدا گىتتارنى جۇزىغا قاتتىقراق قۇيۇپ ساپتىمەن ، بۇ ماۋۇ غەيرەت دىگەنگە بولوپ بەردى :
-ھەي ھارىمى ، ئۆزۆڭنى  نىمە چاغلاپ قالدىڭ ، ھىخ! مەنچۇ قارا سەندەك ماڭقىلار چالغان بۇ بىرنىمنى ؛-دەپ ئۈستەلگە قۇيۇلغان گىتتارنى بىر تىپىلا يەرگە چۈشۈرۋېتىپ ئۈستىگە كۈچەپ يانجىىپ :- پاچىغىم بىلەن چالىمەن ، جۇمۇ ! مىنڭ ئالدىمدا يۇغانلىق قىلپ مەن بىلەن ئۆزەڭگە سۇقۇشتۇرغىدەك بولدۇڭما ھەي تۈك!
    ئۇ شۇنداق دىگىنىچە بىگىز بارمىقى بىلەن بۇرنىمنى نۇقۇپ مىنى كەينىمگە بىر قەدەم داجىتتى ، مەن غەزەپ ئاچچىقىم بىلەن مۇشۇمنى تۈگدۈم ، ئەمما بۇلارنىڭ ئەلپازىدىن سەل قۇرقتۇم ، مۇش تۈگكۈنىمنى كۆرگەن غەيرەت تېخىمۇ چىچاڭشىپ كەتتى ، تۈز مۇش بىلەن مىنىڭ يۈزۈمگىلا بىرنى سالدى ، مەن تەڭپۇڭلىقىمنى يۇقىتىپ ،ئارقامدىكى ئۇرۇندۇققا پۇتلىشىپ يەرگە يقىلدىم، كەينىدىن ئۇيەنە مىنى بىر نەچچىنى تىپىۋەتتى،
-ھەي ھاراملىق ، ئاڭاڭ قارغاي بۇ مەكتەپنىڭ نوچىسى ، نەچچە يىلدىن بۇيان زۇلخۇمارنى دەپ تەنھا يۈرگەنمەن ،نەدە بىر تۈك مىنىڭ كۆڭلۈمدىكى ئادەمنى ئاشخنىغا باشلاپ ، يېشىنى سۈرتۈپ ! مەن باشتا سىنى بىلسە كۆتىنى قىسىدۇ دەپ گەپ قلمىدىم ، ئالدىنقى ھەپتە ئۇنى ئاشخنىنىڭ ئالدىدا سەتلەشتۇرۇپسەن ، شۇنداق قىلىدىغانغا سەن كىمتىڭ زادى ! گەپ قىلە ھەي ھارىمى ! تخى مۇشۇڭنى تۈگۈپ كىتىپسىنا !؟
مەن ھەممىنى چۈشەنگەندەك بولدۇم ، ئازراق چۈشەندۈرمەكچى بولوپ ئېغىزمىى ئۆمەللەپ تۇرشۇمغا ئۇيەنە بىر نى تەپتى ، مەن قۇرسىقىمنى تۇتۇپلا قالدىم ،-ھۇت ئاناڭنى سەن ھارىمى مىنىڭ يۈزۈمنى تۆكۆپ بولدۇڭ بىلەمسەن؟ ، شۇنداق دىگەنچە يۈزۈمگىلا يەنە  بىرنى تېپىۋەتتى ، بۇرنۇم قاناپ بېشىم قايغاندەك بولدى ، مەن قاننى كۆرۆپ ھۇشۇمدىن كىتىپتىمەن ، كىين نىمە ئىشلار بولغىنىنى بىلمىدىم ، قانچىلىك ۋاقىتتىنكىيىن ھۇشۇمنى تاپقىنىمنى بىلمەيمەن ، پۈتۈن بەدىنىم مۇز بولۇپ قاتقان ، ئەزالىرىم چانىۋەتكەندەك ئاغرىپ كەتكەن ئىدى ، مەن ئۆزۆمىنىڭ زادى قانچىلىك ياتقىنىمنى بىلەلمىدىم  ، ئورنۇمدىن تۇرۇشقا تەمشىلىپ باقتىم .ئەمما پۇتۇم ئۆزۆمنىڭ ئەمەس ، ئەتراپ بىرخىل سۇس يۇرۇق ، ھېچكىم يۇق بولوپ ، سۇنغان گىتتار بېشىمدىلا تۇراتتى ، ئۈستەلمۇ يۇغۇشتۇرۇلمىغان بولوپ قالايمىقان تۇراتتى .

مەن بىر ئاماللارنى قىلىپ يانچۇقۇمدىم تېلفۇنىمنى ئېلىپ نىجاتقا تېلفۇن قىلدىم ،تىزلا ئۇلاندى .
-ۋەي......، كۈتۈپخانىنىڭ كەينىدە، ئاران شۇنىلا دېيەلىدىم ،
توۋۋا ،دەيتتىم ئىچىمدە ، چىن دىلىمدىن سۆيگەن ئادەم مىنى تاشلاپ كەتسە ، ياخشى كۆرگەن ئادىمىم يىراقلاردا ، مەن ۋىسال ئۇتىدا پۇچىلىنىپ بۇيەرلەردە يۈرسەم ، ئەمدى بولغاندا باشقىلارنىڭ ماڭا تاڭغان مۇھەببىتننى دەپ مەن بۇيەردە كوچىدىكى لالما ئىشتلاردەك يەر باغرىلاپ ، باشقىلارنىڭ ئاياغ ئاستدا ئىزىىلپ يۈرسەم ، ئى ئولوغ ، كەرەملىك ئىگەم ، سەنمۇ منىڭ جېنىم بىلەن قىلىپ ماڭا ئېغىر كىسەلدىن ىرنى بەرسەڭ  ، يا پەرۋەردىگار، مەن ئالدىنقى ئۆمرۈمدە نىمە گۇناھ قىلغان بولغىيتىم ، نىمىمۇ قىلغان بولغىيتىم ، مىنىڭ پەقەت شۇ... پەقەت شۇ  بىرسىنى چىن كۆڭلۈمدىن ياخشى كۆرۆپ ئۇنىغا ساداقەت  كۆررسەتكەن گۇناھىم بارغۇ؟گىتتارچالغان گۇنايىم بارغۇ ؟نىمە ئۈچۈن ؟...جېنىم ئانا ،مەن نىمانچە شور پىشانىدۇرمەن ! ماڭا  بۇ نىمە كەلگۈلۈك؟ يۈزۈمنى بۇيلاپ چۈشكەن ئىككى تامچە ھەسرەتكە تولغان سۇغۇق كۆز يېشىم مەڭزىمنى بۇيلاپ ئۇدۇل يۈرۈگۈمنىڭ ئۈستىگە تامچىلىماقتا ئىدى ،
شۇنداق ئاللاغا نىدا قىلىپ تۇرسام نىجات ياتاقىكىلەرنى باشلاپ يىتىىپ كەلدى ،ئۇمىنىڭ ھالىمنى كۆرۆپ پاڭڭىدە يىغلىۋەتتى :
-ئاھ خۇدا ،بىچارە ئۇكام ساڭا نىمە بولدى ؟ مىنى كەچۈرگىن ، ئۇ شۇنداق دەپ مىنى باغرىغا بېسىپ بوقۇلداپ يىغلاپ كەتتى ،مەنمۇ بىر ھازاغىچە يىغلاپتىمەن .كىيىن باشقىلار مىنى يۈدۈپ ياتاققا ئەكەلدى ،بىز ياتاققا كەلگەندە سائەت ئاللىقچان كىچە ئىككىلەر بولوپ كەتكەن ئىكەن  ،دىمەك مەن شۇيەردە ساق 6 سائەت يېتىپتىمەن .
باشقلار ماڭا بىر ئىستاكان ئىسسىق سۇ بەردى ، مەن بىر يۇتۇم ئىچىپلا قاتتىق يۆتىلىپ كەتتىم ، باشقىلار مىنىڭ مەيدەمنى تۇتېۋىلىشتى، مەن شۇ يۆتەلگەنچە قان تۈكۈرۈپ توختدىم ، باشقىلار بۇنى كۆرۆپ قۇرقۇپ كېتىششتى، نىجات مىنى دوختۇرغا ئاپارماقچى بولدى، مەن ئۇنىماي تۇرۇۋالدىم ، سەۋەبى ئۆزۆمىنىڭ راستىنلا شۇنداق كىسەل ئىكەنلىكلىمنى «بىلگۈم» يوق ئىدى، مەن بەك توڭلاپ كەتكەچكە نىجات مەن بىلەن بىرگە ياتتى ، قانداق ئۇخلاپ قالغىنىمنى بىلمەيمەن ، ئۇيغانسام دوختۇرخانىدا يېتىپتىمەن ، ئۇلارنىڭ دىيىشچە مەن بەك قىززىپ كىتىپتىمەن ، قۇرقۇپ دوختۇرخانغا ئەكەلگەنىمىش ، كۈن ئاللىقاچان كەچ بوپ كىتىپتۇ، دۇختۇرلار مىنى بىر قۇز تەكشۈرگەندىن كىين مەكتەپنىڭ ئىسپاتىنى ئېلىپ كەلسەم «ب تىپلىق ئۇلتۇرا ئاۋاز ئۈسكىنسى »دە تەكشۈرەرمىش ، مەن قەتئى ئۇنىماي تۇرۋالدىم، مىڭ تەستە تەكشۈرۈشنى رەت قىلىپ ، دوختۇرخاندىن بىر يەشىك دۇرا ئىلىپ ياندىم ،
ياتاققا كىرىپ بىرنىچى بولوپ قىلغان ئىشىم ئۆزۆمنىڭ ئەڭ ئامراق گىتتىرىنى چېقىش بولدى ، باشقىلارمۇ ئارتۇق گەپ سۆز قىلمىدى، شۇنىڭدىن كىيىن مەن ھېلىلا ئۆلۆپ كىتىغاندەك تۇيغۇغا كىلىپ . ئۆزۆمنىڭ ھاياتى توغرۇلۇق تۇلا ئۇيلايدىغان بولوپ  قالدىم .
