                                                     يەنە كىرىپ قالدىم بۈگۈن،
                                                     مىسرانىم دىگەن مۇنبەرگە.
                                                     چىن قەلبىمدىن چىقىرىپ،
                                                     ئەسسالامۇئەلەيكۇم سىلەرگە.

                                                     ئاۋال باشلاي گېپىمنى،
                                                     شىۋاس كامنىڭ ئىسمىدىن.
                                                     كۆڭلۈمگە يېقىن تۇرىدۇ،
                                                     نىمىشقىكىن تاڭ بىلمىدىم.

                                                     ۋاي!شىۋاسكا_شىۋاسكا،
                                                     بەلەن ئوبدان تۇللىما.
                                                     سىزنى يەنە ئەسلىدىم،
                                                      ئۆزلىرى بۇنى بىللىما.   


                                         بىر چاغلاردا شىۋاسكام مىسرانىمدا يۈرەتتى،
                                         مىسرانىمنىڭ گۈلى بوپ پەخىرلىنىپ كۈلەتتى.
                                         ئەسەر يازماي گاھىدا خەخنىڭ پۇلىغا خىرىسقىپ،
                                         ياخشى دەپلا چىكىتىپ توت،بەش كوينى ئالاتتى.

                                        مەن كۆرۈپ قېلىپ بۇ ئىشنى شىۋاسكامنى پاشلىدىم،
                                        بىكارغا پۇل ئالامسەن دەپ شىۋاسكامغا قاقشىدىم.
                                        شىۋاسكامغا بۇ سۆزۈم ئېغىر كەتكەن ئوخشايدۇ،
                                        شۇندىن كېيىن شىۋاسكامنى مىسرانىمدا كۆرمىدىم.

                                        رۇستەم قارسۇ ئەسلىدە دەريادىكى ئېقىندى،
                                        ئارلىقىمىز يىراق بولسىمۇ كۆڭلىمىز بەك يېقىندى.
                                        بىرە قېتىم سېغىنىپ يازمايدۇ شىئىر ۋە چاقچاق،
                                        شۇنداق بولسىمۇ  يۈرۈگۈم رۇستەمجاننى سېغىندى.   

                                       ئەي! بۇراسىرىم رۇستەمجان تۇردۇڭلارمۇ سالامەت،
                                       ھېيت ھارپىسىدا قىلىڭلار ياخشى ئەمەل ئىبادەت.
                                       قايسى كۈنى كۆڭلىڭىزگە ئازار بېرىپلا ساپتىمەن،
                                       ھېيتنىڭ  قىلىپ يۈزىنى تۇتماڭ ماڭا ئاداۋەت.   

                                       قىز خۇمارمۇ بەك ئەسكى مۇشتى بەك يۇغان،
                                      گېپىگە ئۇنىمىغان قىزلارنى بەكمۇ سەت سالغان.
                                      قىزلارنى ئۇيۇنچۇق ئورنىدا كۆرگەچ ئۇ لاتا گۇي،
                                      ئاخىرىدا مىسرانىمدا قىز خۇماردەپ نام ئالغان.  

                                      مىسرانىمدا تۇخارجان يۈزىنى يا يۇمايدۇ،
                                      ئايدا يىلدا ئەسەر يوللاپ ئوزىنى تازا داڭلايدۇ.
                                      مايماق شەپكىنى كىيىۋېلىپ ئىچىمۇ يا پۇشمايدۇ،
                                      بۇشىئىرنى كۆرۈپ تۇخارجان 2256نى تىللايدۇ.

                                       ئەينەك خانىم بۇرۇنمۇ شىئىر يازىدىكەن،
                                       بىرە قېتىم شىئىر يازسا بىر يىل يوقايدىكەن.
                                       ئاخىر بىلدىم سەۋەبىنى بىركۈن ئويلۇنۇپ،
                                       ئەينەك خانىم ئەسلىدە ھاكاۋۇر ئىكەن.

                                                       ‹‹ زۈلايخاغا››

                                       ياقدىدىڭ يۇسۇف باغرىمغا ياقايمۇ دىسە،
                                       بۇ سۆزۈڭ بۇيرۇقمۇ ياكى رەت قىلىش.
                                       تاڭ دىدىڭ سۆزۈمگە جاۋاپ بەر دىسە،
                                       ئېيتقىنا جاۋابىڭ نىمانچە چىگىش.

                                       ئەي! زۈلەيخا ھاكاۋۇر بولمايلى ئۇنداق،
                                       يۇسۇفجاننى ئازاپتا قىيناملا بۇنداق.
                                       نىمە خىيال سۈرۈيلىكىن ھېچ بىلمىدىم،
                                       ئېيتسىلا قېنى مۇھەببەت بولامدۇ مۇشۇنداق.
.....................................
                                       ۋاي!خۇدايم كاپتالىس ئەسلىدە كاپىتان ئىكەن،
                                       باشقۇرىدىغان تۈمەنلەپ ئەسكىرى بار ئىكەن.
                                       ئەسكىرىم بار دەپ پو ئېتىپ كاپىتان غۇجام،
                                       مىسرانىمدىكى 2256نى پەقەت يوقلىمايدىكەن.

                                       ئەل بالىسى ئەسلىدە ئەلنىڭ بالىسىمىش،
                                       سۇت تاپالماي گاھىدا كالىنى ئىمەرمىش.
                                       شىئىرىغا قاتمىغان ھەربىر ئىنساننىڭ،
                                       باشقۇرغۇچىغا ئۈستۈدىن ئەرىز قىلارمىش.

                                       يۇچۇن ئادەم ئەسلىدە ئۇچۇق ئادەممىش،
                                       كۆڭلىدە يامانلىغى يوق دۇرۇس ئادەممىش.
                                       يۇچۇن ئادەمنى شىئىرىمغا قاتمايلا قويسام،
                                       مەندىن خاپا بوپ ئۇ مىنى تازا تىللايتمىش.

                                             ساييارە تېخىچە چوڭمۇ بولماپتۇ،
                                             شىئىرىمغا قاتمىسام ھوڭرەپ يىغلاپتۇ.
                                             يىغلىماڭ ئەي!ئۇماق سەييارە خېنىم،
                                             ئەمدى سىزنى2256  پەقەت ئۇنۇتماپتۇ.
