   يازمامغا‹‹ئۇقۇتقۇچى كىم؟›› دەپ تېما قويسام بەلكىم سەل مەۋھۇمدەك چۈشەنچە بېرىپ قۇيار.لىكىن ئۇقۇتقۇچىنىڭ كىملىكى ھەققىدە ئازراق بولسىمۇ ئىزدەنگۈم كىلىپ قالدى.100مېتىر يېراقلىقتىن سالام بېرىشكە ئەرزىيدىغان،ئىنسان روھىنىڭ ئىنجىنىرى بولغان،كەلگۈسى ۋە زامانىۋىلاشتۇرۇشنىڭ ئاچقۇچى بولغان،سۇيۇملۈك ئۇقۇتقۇچىلارنىڭ ،مىھرىبان باغۋەنلەرنىڭ،جاپاكەش ئۇستازلارنىڭ كىملىكىنى؛تەشۋىقات،ئېغىزدىكى قۇرۇق ماختاشلاردىن ئەمەس؛رىيال ئىجتىمائىي تۇرمۇشتىن ئىزدەنگۈم كەلدى.ئۇنداق بولسا بىر ئۇقۇتقۇچىنىڭ مەكتەپتىكى ئىجتىمائىي ئورنىغا قاراپ چىقايلى.
     بىرىنچى شەخس تىلى بىلەن بايان قىلاي:مېنىڭ ئۇقۇتقۇچىلىق سالاھىيەت كېنىشكامدا‹‹ئالى مەكتەپ››دەپ بار.بۇ مېنىڭ ئالى مەكتەپ ئۇقۇتقۇچىسى ئىكەنلىكىمنى بىلدۈرىدۇ،گەپنى مېنىڭ ئىشخانامدىن باشلاي:ئىشخانام بىز ئولتۇرىۋاتقان بىنانىڭ تۆتىنچى قەۋىتىدە يەنى شۇ بىنانىڭ ئەڭ ئۈستى ھەم ئەڭ چېتى،مەكتىپىمىزدىكى ئىشخانىلار ئىچىدە جايلىشىش ئورنى جەھەتتىن ئەڭ ناچىرى،قىزىقارلىق يېرى شۇكى يازدا ئىسسىق،قىشتا سوغوق.بالىلىق ئۆي بازا،بالىسىز ئۆي مازا دىگەن تەمسىل مەكتىپىمىزدىكى ئىشخانىلارغا  خېلە ماس كىلىپ قالىدۇ.بىزنىڭ ئىشخانىدا توققۇز نەپەر ئۇقۇتقۇچى بىر ئىشخانىدا ئىشلەيمىز.دەرسى يوق دەپ ھەرخىل سەۋەپلەر بىلەن تۆۋەنگە چۈشۈپ كەتكەن ،دەرۋازىۋەن،ياتاق باققۇچى بۇلۇپ كەتكەن ئۇقۇتقۇچىلار ، سالاھىتى ئەسلىگە كىلىپ قايتىپ كەلسە ئۇنىڭدىنمۇ كۆپىيىپ كېتىمىز،توققۇزىمىز ئىشخانا تىنچلىقى ئۈچۈن مىڭ تىرىشچانلىق كۆرسەتسەكمۇ مەغلۇپ بولىمىز ،بىزگە دەرس تەييارلاش،تاپشۇرۇق تەكشۈرۈش.....دىگەندەك خىزمەتلەرگە تىنچ مۇھىت لازىمتى تولىمۇ ئەپسۇس بۇ بىزدە يوق. مەكتىپىمىزدىكى ئۇقۇتقۇچى ئەمەسلەرنىڭ ئىشخانىسى ئۇنىڭ ئەكسىچە بىر ئىشخانىدا بىردىن،ئىككىدىن ئېشىپ كەتسە ئۈچتىن ئىشلەيدۇ،بەزىلىرى بىر جېنىغا ئىككى ئىشخانا ئېلىۋالغانلارمۇ خېلە بار.بونداق دىسەم بىز باشلىق بولغاندىكىن ،ھوقۇق جەھەتتە ئۈستۈن بولغاندىكىن شۇنداق بولىدۇ دىيىشى مۇمكىن،شۇنداق بولىشى ھەق دىگۈچىلەر ئاز ئەمەس. ،ئەمدى گەپنى خىزمەت ئۈستىلىمدىكى كومپىيوتىرغا يۆتكەي. سېپى ئۆزىدىن 90-يىللارنىڭ مەھسۇلاتى شىنجىياڭغا  ئېغىزلار بويىچە ياردەم بېرىش باشلانغاندا شاڭخەيلىك كەسىپداشلار ياردەم قىلغان.ئەگەر ئۇ كومپۇيوتىردا  ھازىر مەشغۇلات قىلسىڭىز بىرىنچى كەڭرى ۋاختىڭىز ئىككىنچى يۈكسەك سەۋرىچان بولىشىڭىز لازىم.ئۇنىڭ ئەكسىچە مەكتىپىمىزدىكى ئەڭ ئاددى مەمۇرنىڭ ئۈستىلىگە قارىسىڭىز كىرىستال ئېكرانلىق ئەڭ ئىلغار كومپىيېتىرلارنىڭ بىكار ئۇيۇن ئۈچۈن خوراۋاتقىننى كۆرىسىز.بۇنداق كىچىك تەپسىلاتلار بايان قىلىپ خېلە كۆپ ئەقىللىق ئۇقۇرمەننىڭ ۋاقتىنى ئىسراپ قىلغۇم يوق.پىكىرنى يۆتكەيكى،ئۈتۈپ كەتكەن بىرقانچە يىللاردا ئېلىمىزدىكى رەھبىرى كادىردىن تارتىپ ئەڭ تۆۋەن خىزمەتچىگە قەدەر ‹‹ئىلمىي تەرەققىيات قارشى››نى ئۈگىنىپ ئۆتتۇق،مەكتىپىمىزدىمۇ بىرقانچە رەت ئۈگىنىپ ،يۇقىرى ئۇرۇنلارنىڭ تەكشۈرۈپ ئۆتكۈزىۋېلىشىدىنمۇ مۇھاپىقىيەتلىك ئۆتتى.‹‹ئىلمىي تەرەققىيات قارشى››ھەممە ئۇرۇنلارنىڭ خىزمەتمەتلىرىدىكى يىتەكچى ئىدىيەسى بولىشى ،خىزمەتلىرىنى ئالغا سىلجىتىشى كېرەك ئىدى،مەيلى خىزمەت ۋە باشقا ئىشلاردا بولسۇن قانۇنيەتكە ھورمەت قىلىش،ئىلمىيلىكنى چىقىش قىلىش،تەرەققىياتنى ئاساس قىلىش مۇشۇ ‹‹ئىلمىي تەرەققىيات قارشى››نىڭ مۇھىم مەزمۇنى .يەنە كىچىك تەپسىلاتلاردىن گەپنى باشلىسام؛ئەگەردە بىزنىڭ مەكتەپتە ئۇقۇتقۇچىلار تەتىل قىلمايدۇ،مەكتەپتىكى مەمۇرى خادىملار تەتىل قىلىدۇ دىسەم ئىشىنەمسىز؟مەكتەپتىكى بارلىق قائىدە تۈزۈملەر ئۇقۇتقۇچىلار ئۈچۈنلا چىقىرىلغاندەك مەيلى مائاشنى تۇتۇش بولسۇن،چارە كۈرۈلۈپ جازالىنىۋاتقانلارنىڭ ئىچىدە ئۇقۇتقۇچىلار كۆپ دىسەم ئىشىنەمسىز؟مەن ئۆزەمنىڭ بىر قېتىملىق دەرس ئۈتۈشىمنى بايان قىلىپ بېرەي:ھېلىقى باشتا گېپىنى قىلىپ بەرگەن تۆتىنچى قەۋەتتىكى ئىشخانامدىن چىقىپ يەنە بىر بىنانىڭ تۆتىنچى قەۋىتىگە دەرس ئۆتكىلى ماڭىمەن،بەزىدە ئىككى سائەت بەزىدە تۆت سائەت،بەزىدە سەككىز سائەت(بىر كۈندە) دەرس ئۈتىمەن، دەرس ئارلىقى دەم ئېلىش ۋاقتىدا ئىشخانىغا قايتىپ كەلگىلى بولمايدۇ،ۋاقىت يەتمەيدۇ،ئۇقۇتقۇچىنىڭ دەم ئېلىش ئورنى يوق،دەرس سۆزلەپ ئېغىزلار شالدەك قۇرۇپ كەتكەن،ئىچەي دىسەك چاي ئورنىدا تازلىق ئۈيدىكى تۇربىدا سۇدىن باشقا نەرسە يوق،ئامال نەچچە ئۇقۇغۇچى يوق يەرگە بېرىپ ئاچچىق تاماكىدىن بىرنى چىكىمەن .يەنە قوڭغۇراق چېلىندۇ يەنە دەرسخانىغا ماڭىمەن.
    بىر ئۇقۇتقۇچى زادى مەكتەپتە كىم؟ئۇ مەكتەپنى مەكتەپ قىلىپ تۇرغۇچى،مەكتەپنىڭ چاقىنى ئالغا سىلجىتقۇچى كۇچ.مائارىپنى،مىللەتنى گۈللەندۈرگۈچى.شۇنداقلا ئۇمۇ ھېسياتلىق بىر ئادەم.ئۇمۇ مەكتەپتە ئۆز ئەجرىگە چۇشلۇق ئورۇن ۋە ئىمتىتازغا ئىگە بۇلۇش كېرەكمۇ ئەمەسمۇ؟خىيالىمغا ئىلىمىزنىڭ كىلاسسىك ئەسرى‹‹غەرىپكە ساياھەت››كىلىپ قالدى.بۇ ئەسەر ئىلىمىزدىكى ھەممە ئادەملەرگە تۇنۇشلۇق،تاڭ سىڭ،سۇن ۋۇكۇڭ،قۇ باجې،شا خېشاڭ،تاڭ سىڭنىڭ ئېتى بۇلار غەرىپكە بېرىپ نوم ئەكىلىشكە ماڭغان بىر كوللىكتىپ.بۇلارنىڭ كىملىكىگە قاراپ باقايلى؛تاڭ سىڭ _ئاددى تىل بىلەن ئىپادىلىسەك بۇ كوللىكتىپنىڭ باشلىقى.قالغانلار مۇشۇنىڭ ئۈچۈن خىزمەت قىلىدۇ.سۇن ۋۇكۇڭ_گوروپپىدىكى ئەڭ ئەقىللىق،ئەڭ باتۇر،ماھارەتتە ھەممىنى بېسىپ چۈشىدىغان پىرسۇناژ.جۇ باجې_يەپ تويمايدىغان ئاچكۆز،خۇتۇنپەرەس،شەخسىيەتچى،ئەتىگەندىن كەچكىچە يىسەم،ئىچسەم،ئوينىسام دەيدىغان بىر پىرسۇناژ.شا خېشاڭ_جاپاكەش،ئاق كۈڭۈل،سادىق بىر شەخس.تاڭ سىڭنىڭ ئېتى_ ئۇ ئەسەردە تاڭ سىڭ ئۈچۈن ئات ئورنىدا خىزمەت قىلىش بەلگىلەنگەن.ئەقىللىق ئۇقۇرمەن ئوبدان ئويلاپ باقساق،مۇشۇ ئەسەردىكى بۇ پىرسۇناژلار ئوبرازلىرى كۆز ئالدىمىزغا بىر جەمىيەتنى،جەمىيەتنىڭ تۈزۈلۈش قۇرۇلمىسىنى كۆز ئالدىمىزغا كەلتۈرىدۇ،ئەگەر مۇشۇ گوراپپا سۇن ۋۇكۇڭدىن ئايرىلسا ھىندىستانغا بېرىپ نومنى ئەكىلالىشى مۇمكىن ئەمەس،لىكىن ئەسەردە ئەڭ كۆپ جازالىندىغان،گوروپپىدىن قوغلاپ چىقىرىلدىغان ،بېشىغا چەمبىرەك سېلىپ باشقۇرۇلدىغان ،ئەڭ جاپاكەش يەنىلا سۇن ۋۇكۇڭ.
     خوش ئۆز گېپىمىزگە كېلەيلى؛مەيلى ئۇقۇتقۇچىنىڭ كىملىكى ،مەيلى باشقا كەسىپتىكى ھەرقانداق بىر ئىنساننىڭ كىملىكى بولسۇن،بۇلارنىڭ كىملىكى رىئالنى جەمىيەت تەرپىدىن ئېتراپ قىلىنىپ ئۆز ئىززەت ھورمىتىنى تېپىشى كېرەك!بۇنىڭ ئۆزى ئەسلى قانۇنيەت.ئەپسۇس بۇ قانۇنيەتكە بىزنىڭ بۇ جەمىيىتىمىز كىشىلىرى ھورمەت قىلمايۋاتىدۇ ياكى قىلمايۋاتىمىز،كىملىكنىڭ ھورمەت قىلىنىشى؛ئىنسانلارنىڭ ئۆزىنى ھۆرمەت قىلغانلىقى.
     ئاللا! تاس قاپتىمەن دەرسكە كېچىكىپ قالغىلى،مەن ئۇقۇتقۇچى بولغاندىكىن يەنىلا دەرسىمنى ئۆتەي.

مەكتىپىمىزدىكى ئەڭ ئاددى مەمۇرنىڭ ئۈستىلىگە قارىسىڭىز كىرىستال ئېكرانلىق ئەڭ ئىلغار كومپىيېتىرلارنىڭ بىكار ئۇيۇن ئۈچۈن خوراۋاتقىننى كۆرىسىز.

ما گەپنىڭ ھەقىقەتەن جېنى بار  مۇشۇ  يازلىق تەتىلدە مەن مەكتەپتىكى مۇئەللىمىمگە ياردەملىشىپ ئوقۇغۇچى قوبۇل قېلىشتا قالغان  ئۇ مۇئاللىمنىڭ ئالدىدىكى كومپىيۇتىرنى مەكتەپتىن  8500كويغا ئېلىپ بىرىپتۇ ئەمما ئۇ مۇئەللىمىمىز قارتا ئويناشتىن باشقىنى بىلمەيدىكەن. بىر كۇنى كومپىيۇتىرغا 金山卫士نى قاچىلاپ تۇرسام ئىكراندىكى بەلگىنى كۆرۈپ نىمە ئۇ ؟دەيدۇ .مەن كۈلۈپ كىتىپ : ئۇ بىر ۋىروسخۇر دىسەم :ھە مەن ئاڭلاپ باققان ئەمما كۆرمىگەنكەنمەن  ،قانداق ئىشلىتىدۇ ماڭا ئۈگۈتۈپ قويە دەيدۇ . ماقۇل دەپ كومپىيۇتىرنى تازلاپ قىلدۇق بۇرۇنقىدىن خېلىلا تىزلىشىپ قالغىنىنى كۆرۈپ :ۋاي بۇ خېلى بولدىغان نەرسىكەن ھە دەيدۇ .ھە ئە دەپ قويدۇم ئەمما ئارتۇق سۆزلىگىم كەلمىدى .( ئۇ ئوقۇتقۇچى مەلۇم ئىننىستىتوتنىڭ ئوقۇغۇچى قوبۇل قىلش ئىشخانسىدىكى ئىككنچى قول باشلق )
