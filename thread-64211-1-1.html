     مەن بۈگۈن بىر كىتاپ لازىم بولۇپ قىلىپ كونا ماتىرياللىرىمنى رەتلەۋىتىپ قولۇمغا بۇرۇنقى «شىنجاڭ ياشلىرى» ژورنىلى چىقىپ قالدى ھەم ئۆرۈپ-چۆرۈپ قاراپ بۇ ھىكايىغا كۆزۈم چۈشۈپ قالدى ، ئوقۇسام شۇ ۋاقىتلارغا نىسبەتەن خىلى ياخشى يىزىلغان ھىكايە ئىكەن ، شۇڭا مۇنبەرداشلارنىمۇ كۆرۈپ باقسۇن دەپ يوللاپ قويدۇم


                                                            خەت

                                          ئەسلى ئاپتورى   :   پەخىردىن  مۇسا

     ئايقىز بىلەن ئالىم ئىنتايىن ئىناق ياش ئەر-خوتۇنلار ئىدى ،ئالىم ئايقىزنى قەۋەتلا ياخشى كۆاەتتى ، ئايقىزمۇ ئالىمغا شۇنداق ئامراق ئىدى .
     بىر كۈنى ئالىم ئىدارىسىدىن ئورۇنلاشتۇرغان بىر پالىيەتكە قاتنىشىشى ئۈچۈن قىزىل گىلاستۇكىنى تاقىماقچى بولۇپ ، ئۆي ئىچىدىن ئىزلەپ تاپالماي ئايالىنىڭ چامادانىنى ئاختۇرۇپ بىقىشقا  مەجبۇر بولدى ، ئويلىمىغاندا چاماداننىڭ ئەڭ ئاستىدىن بىر كونۋىرىت چىقىپ قالدى ،ئالىم گۇمان بىلەن كانۋىرىتنى ئىچىپ ئىچىدىكى خەتنى ئالدى-دە ئوقۇشقا باشلىدى :
     « مىنىڭ سەۋرى قامەتلىك دىلبىرىم ، ياخشىمۇ سىز ؟ _ ئالىم خەتنى ئاتلاپ - ئاتلاپ ئوقۇشقا باشلىدى _ .........بىلەمسىز ، سىز ئۈرۈمچىدىن كەكەندىن كىيىن مەن سىزنى چەكسىز سىغىندىم ، سىز بىلەن بىرگە ئۆتكۈزگەن شىرىن كىچىلەرنى ئەسلەپ ياتالمايمەن ، سىزگە تەلپۈنىمەن ،
    سۆيۈملۈك دىلبىرىم ئايقىز ، بۇلتۇر مەن سىزنىڭ يۇرتىڭىزغا بىرىپ ئۆيىڭىزگە كىرگەندە سىز مىنى خۇددى ئايالىمدەك، تولۇق ھوقۇقۇلۇق ئايالىمدەك كۈتكەنتىڭىز ، كىيىن مەن يۇرتىڭىزدىكى بەزى دوسلىرىمغا گىپىڭىزنى قىلىپ بەرسەم ، ئۇلار يامان گىپىڭىزنى قىلغانتى ، لېكىن مەن ئۇنىڭغا قىلچە پەرۋا قىلمىدىم ،چۈنكى مەن سىزنى سۆيىمەن ، سىز ھازىر خىزمەت تەخسىماتى بىلەن ئۆز يۇرتىڭىزغا كەتكەن بولسىڭىزمۇ ، لېكىن مەن سىز ئۈچۈن ئۈرۈمچىدىن  خىزمەت ئىزدەۋاتىمەن ، بىز چوقۇم مۇرادىمىزغا يىتىمىز .................. »
      ئالىمنىڭ خەتنى تۇتقان قوللىرى تىترەپ كەتتى ،ئۇنىڭ خىيالى بۇنىڭدىن ئىككى يىل ئىلگىرى بىر قىتىم مۇشۇ خەتنى كۆرگەن چاغقا كەتتى .
      ئۇ چاغدا ئالىم بىلەن ئايقىز  تىخى ئەمدىلا خىزمەتكە چىقىپ ، يىڭىلا مۇھاببەت قاينىمىغا قەدەم تاشلىغان بىر جۈپلەر ئىدى ،لېكىن ئۇ چاغدا ئايقىز ئالىمنىڭ ئۆزىنى ئالداپ قويۇشىدىن ئەنسىرەيتتى ،شۇنىڭدىن كىيىن بىر كۈنى ئۇ ئالىمغا مەنىلىك قىلىپ مۇنداق دېگەن :
      _ ئالىم ، سىز « مۇھەببەت چۈشۈنۈش ۋە ئىشىنىشنىڭ ئاساسى »دىگەن گەپنى ئاڭلىغانمۇ  ؟
     _ ئاڭلىغان
     _ ئۇنداق بولسا ماڭا قەسەم بىرىڭ ،
     _ يۈرىكىم بىلەن قەسەم بىرمەنكى سىزگە مەڭگۈ ئىشىنىمەن _ دېدى ئالىم .
     _ ئۇنداق بولسا سىزگە بۇرۇن مەن بىلەن قىسقا ۋاقىت يۈرگەن بىر يىگىتنىڭ مۇھاببەت خېتىنى كۆرسىتىمەن ، _ دېدى ئايقىز ئالىمنىڭ كۆزلىرىگە سىرلىق تىكىلىپ . ئۇ بۇ ئارقىلىق ئالىمنى سىنىماقچى ، ئۇنىڭ ئۆزىنى ياخشى كۆردىغان- كۆرمەيدىغانلىغىنى بىلىپ باقماقچى ئىدى .
      ئەنە شۇ چاغدا ئالىم خەتنى كۆرۈپ بولۇپ ، ئايقىزغا چىچىلماقچى بولغان بولسىمۇ ، لېكىن ئۆز قەسىمىنى ئىسىگە ئىلىپ يەنىلا  ئۆزىنى تۇتىۋالغان  ئىدى .
     مانا بۈگۈن ئۇنىڭ بىلەن توي قىلغىلى ئىككى يىل بوپ قالغان بۈگۈنكى كۈندە ، بۇ خەت يەنە ئالىمنىڭ قولىغا چىقىپ قالدى . ئۆز ۋاختىدا ئايقىز بۇ خەتنىڭ ئىگىسىنى ئاللا بۇرۇن ئۇنتۇپ كەتكەنلىكىنى قەسەم قىلىپ تۇرۇپ ئېيتقانتى ، ئالىمنىڭ بۇ خەت ئۈچۈن چىچىلغىنىغا بولسا ئىچ-ئىچىدىن خۇشال بولغاندى ، ئەپسۇس ، بۇ خەت يەنىلا ئۇنىڭ چامادانىنىڭ چوڭقۇر يىرىدە ساقلانسا - ھە ؟........
     _ سىزگە نېمە بولدى ؟ _ سورىدى سىرتتىن كىرگەن ئايقىز . ئالىم گويا پىلىكى كۆيۈپ تۈگىگەن بومبىدەك بىراقلا پاتىلدى :
    _  نېمە بولاتتى ، ماۋۇ خەتنى تا ھازىرغىچە ساقلاپ كىلىۋىتىپسىنا ؟ ئېيتە  ، ئۇنى يەنە نېمە ئۈچۈن ساقلىدىڭ ؟
   ئايقىزنىڭ تاتارغان چىرايى ئەسلىگە كىلىپ ، قاقاھلاپ كۈلىۋەتتى_دە ، جوزىنىڭ ئۈستىدىن بىر ناننى ئىلىپ يەرگە تاشلىدى ۋە ئالىمغا مىختەك قادالغىنىچە دانىمۇ - دانە سۆزلەشكە باشلىدى .
     _ ئالىم ، مەن سىزنىڭ ئايالىڭىز ، سىز ماڭا ئىشەنسىڭىز بولاتتى ،ئەمدى ئىشەنمىگەندىكىن مۇشۇنداق قەسەم قىلىپ بىرەي ، لېكىن سىزگە شۇنى ئېيتايكى ، ئۆز ۋاختىدا مەن سىزنىڭ ماڭا كۆڭلىڭىز بار-يوقلىقىنى بىر سىناپ كۆرۈش ئۈچۈن مۇشۇ خەتنى قانداقتۇر بىر يىگىتنىڭ نامىدا يىزىپ ، ئۇنى يەنە بىر دوستۇمغا كۆچۈرتكۈزۈپ سىزگە كۆرسەتكەن ، مەن سىزنىڭ شۇ چاغدىكى ئەلپازىڭىزغا قاراپ مىنى ھەقىقى ياخشى كۆردىغانلىغىڭىزغا ئىشەنگەنتىم ،چۈنكى مىنىڭ نەزىرىمدىكى كۈنلەش - ياخشى كۆرۈشنىڭ ئىپادىسى ، بۇ سىرنى بۈگۈن يەشتىم ، ئەسلى بۇ خەتنىڭ ھەممىسى  ئويدۇرما ، مانا ئەمىسە ، ئايقىز شۇنداق دەپلا يەردىكى ناننى گاچ قىلىپ دەسسەپ قەسەم ئىچتى  ۋە :
      _ يالغان گەپ قىلغان بولسام نان دەسسىگەن مۇشۇ پۇتۇم ناكا بولغاي _ دېدى ، ئالىم بۇ ئىشلارغا چىدىماي بوش تەلەپپۇز بىلەن سورىدى :
     _ ئەمىسە نېمە ئۈچۈن بۇ خەتنى ساقلىدىڭىز ؟
     _ بۇ دېگەن بىزنىڭ مۇھاببىتىمىزنىڭ يالدامىسى ، ! _ دىدى ئايقىز .
    ئالىم نېمە قىلارىنى ۋە نىمە دىيىشىنى بىلەلمەي بىر پەس سۈكۈتتە تۇرغاندىن كىيىن ، سەل ئۆتۈنۈش ئاھاڭىدا سۆزلىدى :
    _ ئەمدى بۇ يالدامىنى ئۆز قولىڭىز بىلەن كۆيدۈرۋىتىڭ ، كىيىن يەنە مۇشۇنداق كۆڭۈلسىزلىك يۈز بىرىپ قالمىسۇن ،
     ئايقىز ھېلىقى خەتكە سەرەڭگە يىقىپ يىقىن ئاپاردى ، ئۇنىڭ سەرەڭگە تۇتقان قوللىرى سەل-پەل تىترىدى ، قەغەز ئاخىرى ئوت ئىلىپ كۆيۈپ كەتتى ، ئالىمنىڭ قايغۇلۇق چىرايىدا كۈلۈمسىرەش ئەكىس ئەتتى ، ئايقىزنىڭ كۆزىدىن بولسا نىمىشقىدۇر ئىختىيارسىز ياش تامچىلاشقا باشلىدى .
    ئاجايىپ يىرى شۇكى ، مۇشۇ ئىش بولۇپ دەل بىر ئايدىن كىيىن ئايقىزنىڭ بىر پۇتىغا ئىغىر جاراھەت چىقىپ راستىنلا ئاقساقلاپ ماڭىدىغان بولۇپ قالدى .
