    كورلا جامە مەسچىتى كورلىدىكى مەشھۇر يادىكارلىقلارنىڭ بىرى بۇلۇپلا قالماستىن،بەلكى ئۇ كورلىدىكى ئەڭ قەدىمىي مەسچىتلەرنىڭ بىرى.
تارىخى يازمىلارغا ئاسالانغاندا،مىلادىيە 10-ئەسىردىن بۇرۇنلا كورلىغا ئىسلام دىننىي كىرگەن.لېكىن،ئىسلام دىنىنىڭ مۇھىم ئىبادەت بوغان سۇرۇنى بولغان مەسچىت-مەدرىسەلەر پەقەت 10-ئەسىردىن كېيىن ھەر قايسى مازارلارنى مەركەز قىلىپ سېلىنغان.كېيىنچە ئاھالىلەر توپلىشىپ ئولتۇراقلاشقان كەنت-مەھەللىلەرگىچە كېڭەيگەن.پەقەت12-ئەسىردىن كېيىن شەھەر-بازارلارنى مەركەز قىلىپ سېلىنىشقا باشلىغان.شۇ دەۋرلەردە كورلا كونا بازار ئىچى،ھازىرقى جامە ئورنىدا ئىلگىرى-كېيىن مەسچىت-مەدرىسەلەر سېلىنىشقا باشلىغان.كورلىدىكى مەشھۇر دىننىي زات،دىننىي ئالىم ئېلى ھاجى داموللا ۋە ئۆز دەۋرىنىڭ يېتۈك شائىرى موللا كىچىك توختى قازى كالان(قۇسۇرى)نىڭ ئەۋلادلىرىغا قالدۇرغان تارىخىي پۈتۈكلەرگە قارىغاندا،كورلا جامە مەسچىتىنىڭ مۇشۇ ئورۇندا بىنا قىلىنغىنىغا 800يىلدىن ئاشقان بۇلۇپ،ئۇنىڭ ئالدى تەرىپىدە كورلا بويىنچە تارىخى ئەڭ ئۇزۇن بولغان قەدىمىي مازارلادىن ‹‹كورلا يىغىن زاراتلىقى››بار ئىدى.ئۇنىڭ يېنىدا بىنا قېلىنغا بۇ جامە مەسچىتىگە 14-ئەسىردىن تاكى مىلادىيە 1715-يىلىغىچە ھادى ئەلەم ئاخۇنۇم،ئەھمەد ئەلەم ئاخۇنۇم،ئاپپاق ئەلەم ئاخۇنۇم،يۇنۇس ئەلەم ئاخۇنۇم،ئاۋۇت ئەلەم ئاخۇنۇم،قادىر ئەلەم ئاخۇنۇم،موللا ئاسىم ئەلەم ئاخۇنۇملار ئىلگىرى-كېيىن بۇلۇپ ئىمام،خاتىپ،مۇدەررىس بۇلۇپ ئۆتكەن.مىلادىيە 1715-يىلى موللا ئۆمەر ئاخۇنۇمنىڭ تەشەببۇسى بىلەن كونا مەسچىت ئورنىغا بىر مەسچىت سېلىنىپ،ھويلىسىغا ئىككى تۈپ ئەرمۇدۇن قويغان(ھازىرقى چوڭ مەسچىت  ئالدىدىكى ئەرمۇدۇن شۇ).
      1715-يىلدىن1833-يىلىغىچە بۇ مەسچىتكە موللا ئۆمەر ئاخۇنۇم ۋە مۇھەممەت چۇمچۇق ئاخۇنۇم،ئۇنىڭ ئوغلى زايىت ئاخۇنلار مەزىن،يۈسۈپ ئەلەم ئاخۇنۇم،زەيدۇن ئەلەم ئاخۇنۇم،موللا ھەسەن مۇپتى ئاخۇنۇم،قاسىم ئەلەم ئاخۇنۇملار ئىمام،خاتىپ بولغان.چىڭ سۇلالىسى داۋگۇاڭنىڭ 14-يىلى(مىلادىيە1834-يىلى)بۇ مەسچىت كېڭەيتىلىپ،ئۇنىڭ غەرب تەرىپىگە تاش كېسەك قۇرۇلمىلىق بىر چوڭ مەسچىت سېلىنغان.
     ئۇمۇمىي يەر كۆلىمى 12مو بۇلۇپ مەسچىت قۇرۇلۇش كۆلىمى805كۋادرات مېتىر بولغان،بۇنىڭ ئالدى كەڭ سەينا بۇلۇپ،كورلىدىكى ئىككى ھېيت ۋە جۈمە نامىزى شۇ يەردە ئوقۇلغان.بۇنىڭ بىلەن ئۇنىڭ نامى ‹‹ئازنا مەسچىت››ياكى ‹‹ھېيتگاھ مەسچىتى››دەپ ئاتالغان.1867-يىلى ياقۇپ بەگ كورلىنى سېپىلگە ئالدۇرغاندا،مەسچىتنىڭ ئاساسىي قىسمى سېپىل سىرتىدا قالغان.مەسچىت قۇرۇلىشىمۇ بەك زەيلەپ كەتكەن.1895-يىلىغا كەلگەندە بۇخارادا بەش يىل ئىلىم تەھسىل قىلغان دىننىي ئالىم ئېلى ھاجى داموللا بىلەن دىننىي زات ئەمەت ئىشان ھەزرەتنىڭ تەشەببۇسى بىلەن مەسچىت يېڭىلاپ سېلىنغان.مىلادىيە 1833-1936يىلىغىچە بۇ جامەگە ئېلى ھاجى داموللا،ئېلى ئەلەم ئاخۇنۇم،ئابدۇقادىر ئەلەم ئاخۇنۇم،ياقۇپ ئەلەم ھاجى
،موللا كىچىك توختى قازى كالان،ئابدۇقادىر ئىشان ئەلەم قاتارلىقلار ئىمام،خاتىپ،مۇدەررس بولغان(ئىلاۋە:بۇ ھەقتىكى مەلۇماتلار شائىر موللا كىچىك توختى قازى كالان قۇسۇرىنىڭ ئوغلى ھىمىت مەخسۇم ھاجىمغا قالدۇرغان تارىخىي يازمىلىرى بىلەن ۋاقىتىنىڭ ئۆتۈشى بىلەن بۇ مەسچىتمۇ كونىراپ خەتەرلىك قۇرۇلۇشقا ئايلىنىپ قالغان،يەنە بىر تەرەپتىن سېپىل ئىچىدىكى ئاممىنىڭ ئىبادەت قىلىشىغا قولايسىز بۇلۇپ قالغان.مىلادىيە 1916-يىلى مەشھۇر دىننىي زات ئېلى ھاجى داموللامنىڭ تەشەببۇسى،پىلانلىشى بىلەن مەسچىت ئىچىگە(ھازىرقى ئورنىغا)يۆتكەپ سېلىنغان.قۇرۇلۇش  باشلانغاندا   ئېلى ھاجىم داموللام 500تۇياق قوي ئىئانە قىلغان.ئاممىدىن توپلانغان ئىئانە بىلەن مەسچىت 1918-يىلى پۈتكەن.مەسچىتنىڭ ئۇمۇمىي كۆلىمى 10مو ئەتراپىدا بۇلۇپ،قۇرۇلۇش كۆلىمى 2600كۋادرات مېتىر،مەسچىتنىڭ ئۇلى يەر يۈزىدىن 1.5مېتىر كۆتۈرۈلۈپ سېلىنغان.مەسچىتنىڭ ئىچى 84تۆۋرۈك بىلەن تۇرغۇزۇلغان.ئۈستى ۋاسا جۇپ،ئاستى تاختايلىق،تاملىرىغا قورئان-كەرىم ئايەتلىرى ئۇيۇلغان،مەسچىتنىڭ شەرق تەرىپىدە مەدرىسە بۇلۇپ،ئۇنىڭ 20نەچچە ئېغىزلىق ھۇجرىسىغا مۇدەررىسلەر،قارىيلار،خەتتاتلار،تالىپلار ئورۇنلاشقان.مەسچىتنىڭ چوڭ دەرۋازا ئۈستىگە 3قەۋەتلىك گۈمبەز ۋە پەشتاق چىقىرىلغان.تامچىلىقنى كورلىلىق ئىبراھىم ئۇستام،قادىر ئىماملار،ياغاچچىلىق،نەققاشلىقنى كورلا تەۋەسىدىكى بارلىق ياغاچچىلارنىڭ ئۇستىسى لەنگەر غۇڭرات جىڭ مەھەللىك تۇردى ئارزۇ بىلەن چامبۇللۇق مىجىت قاسىم ئۇستام شاگىرتلىرى بىلەن بىللە ئىشلەپ تۈگەتكەن.
1916  -يىلىدىن 1948-يىلىغىچە بۇ مەسچىتكە ئېلى ھاجى داموللام،مەخسۇت ئەلەم ھاجىم،تاھىر ئەلەم ھاجىم،ئابدۇقاھىر داموللام،ئېزىم داموللا ھاجىملار ئىمام ۋە خاتىب بولغان.بۇ كاتتا مەسچىت 1966-يىلى باشلانغان مەدەنىيەت زور ئىنقىلابى مەزگىلىدە بۇزۇۋىتىلگەن.1978-يىلى پارتىيە مەركىزىي كومىتىت 3-ئۇمۇمىي يىغىنى ئېچىلغاندىن كېيىن،دىننىي سىياسەتلەر ئەمەلىيلەشتۈرۈلۈپ،مەسچىت ئورنى ئاممىغا قايتۇرۇپ بېرىلگەن.1984-يىلى 9-ئاينىڭ 19-كۈنى مەسچىت سېلىندىغان ئورۇن بېكىتىلىپ،ئەزىم داموللا ھاجىم،ھىدايەت داموللا ھاجىم،كېرەم قارى ھاجىم،ئېلى قارى ھاجىم قاتارلىق 17كىشىدىن تەركىب تاپقان مەسچىت قۇرۇلۇشىغا يېتەكچىلىك قىلىش گۇرۇپپىسى قۇرۇلۇپ ئىش باشلانغان. قۇرۇلۇشقا ئوبلاستلىق بىناكارلىق شىركىتىدىن ئابدۇرېھىم ئۇستام مەسئۇل بولغان.مەسچىتنىڭ 1-قارارلىق قۇرۇلىشى 1984-يىلى باشلىنىپ 1988-يىلى پۈتكەن.پۈتۈن قۇرۇلۇش پولات-سىمىنۇت قۇرۇلمىلىق بۇلۇپ،مەسچىت،چىللىخانا،سۈننەتخانا،پەشتاقلار بىلەن قوشقاندا قۇرۇلۇش كۆلىمى 3190كۋادراتمېتىر،قۇرۇلۇش شەكلى جەھەتتە،مەسچىتنىڭ مېھراب تەرەپ بىر قەۋەت،ئوتتۇرسى ئۈچ قەۋەت قىلىپ سېلىنغان،مەسچىتنىڭ ئىچ-تېشى ئىسلام بىناكارلىق ئۇسلۇبى بويىنچە كاتتا بېزەلگەن.جامە مەسچىتنىڭ 2-قارارلىق قۇرۇلىشى 1993-يىلى باشلىنىپ1994-يىلى 10-ئايدا ئاخىرلاشقان.بۇنىڭدا ئاساسلىقى تاشقىرىقى مەسچىت بىلەن ئىچىدىكى مەسچىتنى تۇتاشتۇرۇش بىلەن پەشتاق قىسمىنىڭ كەم جايلىرىنى تۇلۇقلاش نۇقتا قىلىنغا،پۈتۈپ چىققاندىن كېيىنكى مەسچىت چوڭ دەرۋازىسى 6قەۋەت بۇلۇپ،دەرۋازا ئۈستىگە گۈمبەز چىقىرىلغان.ئىككى يېنىغا تۆت قۇببىلىك مۇنار چىقىرىلغان،دەرۋازا تاملىرىغا مېھراب شەكىللىك ئويما ۋە ھەرخىل نەقىش چىقىرىلىپ زىننەتلىنىپ،ھەيۋەتلىك تۈسكە كىرگۈزۈلگەن.كورلا جامەسىنىڭ ھازىرقى ئۇمۇمىي كۆلىمى5.6مو بۇلۇپ،1-،2-قۇرۇلۇشى قەۋەت قىلىنىپ سېلىنغاچقا قۇرۇلۇش كۆلىمى 5640.6كۋادراتمېتىرگە يېتىدۇ.3-قارارلىق قۇرۇلۇش 2009-يىلى 6-ئايدا باشلانغان بۇلۇپ،2010-يىلى 6-ئايدا تاماملاش پىلانلانغان.بۇ قۇرۇلۇش ئەسلىدىكى سېمۇنت قۇرۇلمىلىق شەكىلنى ناھايىتى زور دەرىجىدە ئىش قوشۇش ئارقىلىق پولات چىۋىق ۋە گەج بىلەن قەدىمقى ۋاسا جۈپ شەكلىدە ئۆزگەرتكەن.شۇنىڭدەك ئىچكى مەسچىت ئۈستىگە چوڭ گۈمبەز،ئىچكى مەسچىت بىلەن تاشقىرىقى مەسچىتنىڭ ئۇلانما كارىدۇرىغا 7گۈمبەز6ئەگمە،ئىچكى مەسچىت پېشايۋىنى ئالدىغا مىللىيچە ئەگمىلىك نالىن چىقىرىلىپ،زىننەتلەنگەن 3تۈۋرۈك بىلەن تۇتاشتۇرۇلغان.پۈتۈن مەسچىت تۈۋرىكلىرى سەللە شەكىللىك مۇرەككەپ گېئومېتىرىيىلىك قاپارتما نەقىشلەر بىلەن بېزەلگەن،مەسچىتنىڭ لىملىرى ۋە يېڭىدىن ئۆزگەرتىلگەن پىلاستىك پولاتلىق ئەگمە دېرىزىلەر ئالاھىدە بېزەلگەن.
   مەسچىتنىڭ 3-قارارلىق ئۆزگەرتىپ بېزەش لاھىھەسىنى ۋە ئۇنىڭدىكى ھەرخىل تام بېزەكلىرىنى قەشقەر مارالبېشى ناھىيەلىك مۇھەممەتئىمىن ئابدۇراخمان ئۇستام بىللەن قەشقەر شەھەرلىك ئابدۇللا ئابدىرھىم ھاجىم ئىشلىگەن.بۇنىڭغا كېتىدىغان راسخۇت 2مىليۇن يۈەن ئەتراپىدا بۇلۇپ،بۇ پۇلنى ئامما ئۆزلىرى تەييارلىغان،ھازىر مەسچىت ئىچىگە 600پارچىدىن ئارتۇق خوتەن گېلىمى ۋە سۈنئىي گېلەم،188پارچە ئەدىيال گېلەم سېلىنغان.ئۇنىڭدىن باشقا،يۇرۇتۇش،ئىسسىتىش،ھاۋا تەڭشەش ۋە رادىيۇ سايمانلىرى سەپلەنگەن بۇلۇپ،ئاممىنىڭ دىننىي پالىيەت قىلىشىغا تۇلۇق ئىمكانىيەت يارىتىلغان.بۇ جامە سېلىنىشتىن بۇرۇن يەنى 1948-يىلىدىن 1985-يىلىغىچە ھېزىم داموللا ھاجىم،ھۇشۇر قازى ئاخۇنۇم،ھىدايەت داموللا داموللا ھاجىملار بۇ جامەگە ئىمام،خاتىپ ۋە مۇدەررىس بولغان.1985-يىلى6-ئاينىڭ 16-كۈنىدىن 1991-يىلىغىچە ھىدايەت داموللا ھاجىم،ئابدۇغۇپۇر داموللا ھاجىملار يېڭىدىن سېلىنغان بۇ جامەنىڭ ئىمام،خاتىب ۋە مۇدەررىسلىكىنى ئۆتەپ كەلگەن.1991-يىلى ھىدايەت داموللا ھاجىم ۋاپاتىدىن كېيىن ھازىرغىچە ئابدۇغۇپۇر داموللا ھاجىم بۇ جامەنىڭ ئىمام ۋە مۇدەررىسلىكىنى،ئابدۇكېرىم تېيىپ قارى ھاجىم خاتىپلىقنى ئۆتەپ كەلمەكتە.
    ئۇمۇمەن،كورلا جامە مەسچىتى ئۆز تارىخىدا يوقۇرىقىدەك مەشھۇر دىننىي ئۆلىمالار قەدەم باشقان مۇھىم تارىخىي ئورۇن بۇلۇپلا قالماستىن بەلكى،قۇرۇلۇش كۆلىمى،ئۆزگىچە بېزىلىشى،لايىھەلىنىش ئۇسلۇبى جەھەتتە ئوبلاست بويىنچە ئالدىنقى ئورۇندا تۇرىدۇ.
(بۇ تارىخىي تەزكىرە 2009-يىلى جامەنىڭ باش ئىمامى مەشھۇر مۇدەررىس ئابدۇغۇپۇر داموللا ھاجىم تەرىپىدىن بېكىتىلگەن).
مەنبە:ـــ   ئابدۇمىجىت ھەسەن قۇربانىنىڭ‹‹باشئەگىم دىيارىدىكى تارىخىي مەدەنىيەت››دېگەن كىتابىدىن ئېلىندى.
