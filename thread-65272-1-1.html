ئۆزۇمنىڭ بىر نەچچە قېتىملىق يۇزتۇرانە ئىمتىھانلارغا كىرىپ ۋە باشقىلارنىڭ ياردىمىدە توپلاپ چىققان يۇزتۇرانە ئىمتىھاندا سورالغان سۇئاللارنى يېغىنچاقلاپ بىر تىما قىلىپ قويدۇم، ھەرقايسىڭلارغا پايدىسى بۇلۇپ قالار.

1-مۇۋەپپىقىيەت دىگەن نىمە؟
-2بەخىت دىگەن نىمە؟
-3سىز بۇ خىزمەتكە ئىرىشىپ قالسىڭىز قانچىلىك ۋاقىت ئىشلىمەكچى؟
-4سىز يېڭى بارغان خىزمەت ئورنىدا خىزمەتداشلىرىڭىز سىزنىڭ خىزمىتىڭىزنى قوللىمىسا سىز قانداق قىلىسىز ؟
-5سىزنىڭ خىزمەت ئورنىڭىزدا خىزمەت ۋاقتىدا باشلقىڭىز تۇيۇقسىز سىزنىڭ بىر خىزمەتدىشىڭىزنى ئەيىپلەپ كەتتى، سىز ئۇ خىزمەت دىشىڭىزنى ناھەق ئەيىپلىنىۋاتقانلىقىنى كۆز ئالدىڭىزدا باشقا خىزمەتداشلىرىڭىز بىلەن بىللە كۆرۈپ تۇرۇۋاتقان مۇشۇنداق پەيتتە قانداق قىلىسىىز ؟
6-سىز يېڭىلا بارغان ئورۇندا شۇ ئىدارىڭىز بىر قېتىملىق پائالىيەتكە تەشكىللىمەكچى بولۇپ سىزنى تەشۋىقاتقا مەسئۇل قىلدى ، ئەمدى سىز قانداق قىلىسىز ؟
7-سىزنىڭ بۈگۈنكى يۈز تۇرانە ئىمتىھاندىكى ئىپادىڭىزنى كۆردۇق، لىكىن سىز بىزگە ئالاھىدە بىلىنمىدىڭىز؟
8-ئاز سانلىق كۆپ سانلىققا بوي سۇنۇش دېگەن نېمە؟
9-ئىلمىي تەرەققىيات قارىشى دېگەن نېمە؟
10-سىز دىجۇرنىلىق قىلۋاتقان ۋاقىتتا باشلىقىڭىز سىزگە تىلغۇن بىرىپ سىزنى ئشقا بۇيرىدى لىكىن دىجۇرنىلىق ئورنىدا سىزدىن باشقا ئادەم يوق سىز باشقىلارنى تاپالمىدىڭىز بۇنداق ۋاقىتتا سىز باشلىق بۇيرىغان ئشقا بارامسىز ياكى دىجۇرنىلىكتە تۇرامسىز؟
11-سىز ساقچى سىز جىددىي بىر ئىش بىلەن ئوستەڭ بۇيىدا كىتىپ بارسىز تۇساتتىن قۇتقۇزۇڭلار دىگەن ئاۋاز ئاڭلاندى قارسىڭىز بىر بالا سۇدا قۇتقۇزۇڭلار دەپ ۋارقىراۋاتىدۇ لىكىن سىز سۇ ئۇزۇشنى بىلمەيسىز  باشقىلارنى ياردەمگە چاقىراي دىسىڭىز باشقىلار يوق بۇنداق ۋاقىتتا سىز قانداق قىلسىز؟
12-سىزنىمە ئۇچۇن بۇ خىزمەتنى تاللاپ قالدىڭىز؟
13-سىز نىمە ئۇچۇن بۇ خىزمەتتە مۇۋەپپىقىيەت قازىنىمەن دەپ قارايسىز؟
14-سىزنىڭ ئەڭ چوڭ ئارتۇقچىلىقىڭىز ۋە كەمچىلىكىڭىز نىمە؟ سىزنىڭ بۇ ئارتۇقچىلىقىڭىز ۋە كەمچىلىكىڭىزنىڭ كەسىپتە ئۇتۇق قازىنىشتا قانداق تەسىرى بار؟
15-سىزنى ئۇستازىڭىز ياكى باشلىقىڭىز ئوڭايسىز ئەھۋالغا چۇشۇرۇپ قويغان ياكى ئۆز ئىشەنچىڭىزنى يوقۇتۇپ قويىدىغان ئەھۋاللار بولغانمۇ؟ بۇ خىل ئەھۋالغا سىز قانداق ئىنكاس قايتۇرغان؟
16-سىز ئىلگىرى ئۆزىڭىزنىڭ مۆلچەرىدىن تۆۋەن نەتىيجىگە ئىرىشىپ باققانمۇ؟ ئەگەر ئىرىشىپ باققان بولسىڭىز سىز بۇ ئىشنى قانداق بىرتەرەپ قىلغان؟
17-خىزمەتتە ئۆسۇشكە توغرا كەلسە، سىز داۋاملىق ئۆز بىلىمىڭىزنى مۇكەممەللەشتۇرۇشنى ئويلامسىز؟
18- سىز ئىلگىرى قايسى خىل رىقابەت پائالىيەتلىرىگە قاتنىشىپ باققان؟ بۇ پائالىيەتكە قاتنىشىش سىزنىڭچە ئەرزىمدۇ؟نىمىلەرنى ھىس قىلغان؟
19- سىز باشقىلارنىڭ سىزنىڭ كۆز قارىشىڭىزنى قۇبۇل قېلىشىغا قانداق تەسىر كۆرسىتىسىز؟
20- ئېغىزچە ئىپادىلەشتە سىزنىڭ قانداق تەجىربىلىرىڭىز بار؟ سىز ئۆزىڭىزنىڭ نۇتۇق قابىلىيتىنى قانداق باھالايسىز؟
21- سىز ئۆزىڭىزنىڭ نۇتۇق قابىلىيتىڭىز ۋە يېزىقچىلىق قابىلىيتىڭىزنى قانداق سېلىشتۇرسىز؟
22-كەسپى ماقالە يازغاندا سىز ياقتۇرمايدىغان قانداق تەرەپلەر بار؟
23-كەسپىڭىزگە مۇناسىۋەتلىك سىز بىلىدىغان كەسپى ئەخلاق تۇزۇملىرىنى سۆزلەپ بېقىڭ؟
24-ئەخلاق بىلەن بىلىم مۇناسىۋىتىنى توغىرىسىدا چۇشەنچىڭىزنى سۆزلەڭ؟سىزنىڭچە قايسىسى مۇھىم؟
25- چىرىكلىككە قارشى تۇرغۇچى خادىملار چىرىكلەشسە قانداق قىلىش كىرەك، چۇشەنچىڭىزنى سۆزلەڭ؟

داۋامىنى يەنە بىر قېتىملىق يۇزتۇرانە ئىمتىھاندىن ئۆتەلمەي قايتىپ چىققاندا يازاي.......
