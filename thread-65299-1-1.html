مىنىڭ تۇر دۇنياسىدا يۈرىۋاتقىنىمغا  4يىلدىن ئىشىپتۇ شۇ جەرياندا توردىكى تالاش- تارىش ،بىر -بىرىگە ھاقارات قىلىش، بۇلۇپمۇ نازۇك مەسىلىلەردە ناھايتى كۆپ يۈز بىرىدۇ ،كۆپرەك يۈز بىرىدىغىنى دىننىي تىمىلاردا مەن سابىق سەلكىندىكى ۋاختىمدا بىر مۆجىزە توغۇرلۇق تىما يوللىنىپ قالدى بەزىلەر يالغان، دىسە بەزىلەر ھەيران قىلىپ ،توۋا يا پىرىم ، دىيشىپ كەتتى يالغان دىگەنلەرنىڭ، راستىنى ئىيتسام يالغان دىيىشكە پۇت تىرەپ تۇرغۇدەك  ئاساسى يوق، بىر نەرسىنى ئىنكار قىلىش ئۈچۈن چۇقۇم ئۇنىڭدا ئادەمنى قايىل قىلغۇدەك دەلىل ئىسپات بۇلىشى كىرەك بۇلۇپمۇ دىننىي مەسىلىدە شۇنداق ،چۈنكى مۇسۇلمان قىرىنداشلار بۇنى چۈشىنىدۇ(بۇ يەردە ئۇيغۇرلارنىڭ ھەممىسىنى مۇسۇلمان قىرىنداش دەۋاتىمەن)ئاللاھنىڭ قۇدىرىتىنىڭ ھەممىگە قادىر ئىكەنلىگىنى،شۇنداق بولغانكەن بۇنداق مۆجىزە يۈز بەرگەندە يالغان دەپ ئىيتىشقا ئادەمنىڭ تىلى بارمايدۇ ،بىرىسى ئۇنى يالغان دەپ سالسا ھەقىقەتەن مىنىڭمۇ ئاچچىغىم كىلىدۇ. چۈنكى يالغان بولغان تەقدىردىمۇ، ئاللاھنىڭ ئۇنچىلىك ئىشلارغا قۇدىرىتى يىتىدۇ، ئەتىراپىمىزدىمۇ ئاللاھنىڭ مۆجىزىسى ئاز ئەمەس. ئادەمنىڭ ئۆزىمۇ بىر مۆجىزە، بوشلۇقتا شۇنچىلىك چوڭ غايەت زور ئاسمان جىسىملىرىنىڭ لەيلەپ يۈرىشىمۇ بىر مۆجىزە،ئۆيىمىزدىكى ھەر بىر مۇسۇلماننڭ قۇلىدىكى قۇرئان كەرىممۇ بىر مۆجىزە،مۇشۇ يىلى رامزان ئىيىنىڭ ئوتتۇرىدا،يۇرتتا بىر جۈمە كۈنى، جۈمەدىن يانغاندا مەسچىتنىڭ ئىمامى كۆپچىلىكنى توختۇتۇپ ،بىرسىنىڭ ھويلىسىدا تىرىغان پەمىدۇرنىڭ (شوخلا) يۈزىدە ئاللاھ مۇھەممەت دىگەن خەت شەكىلىنىپ قالغانلىقىنى، جامائەتلەرنىڭ قىستاشماي رەتلىك نۆۋەت بىلەن كۈرۈپ بىقىشقا چاقىردى مەنمۇ نۆۋىتىم كىلىپ كۆردۈم ،ھەقىقەتەن شۇنداقكەن. بۇمۇ ئۆز كۈزۈم بىلەن كۆرگەن مۆجىزە ئۇندىن باشقا چىچىنىستاندا بىر كىچىك بالىنىڭ پۇتىغا (لا ئىلاھە ئىللەلاھ ) دىگەن خەتنىڭ ھەپتىدە بىر قىتىم يۇتۇسىدا كۈرىنىدىغانلىقىنى كۆرگەنتىم شۇنىڭغا ئوخشاش  كۆز ئالدىمىزدا شۇنچىلىك دەلىل ئىسپاتلار تۇرسا بۇنىڭغا نىمىشقا ئىشەنمىگۈلۈك، نىمە دەپ ئىنكار قىلغۇدەكمىز.  بەزىلەر شۇنداق توغرا كەپ قالغان. ئاۋاز پەردىسىنىڭ سىقىلىپ چىقىشى بىلەن ياكى ئۇنداق ياكى مۇنداق دىگەندەك. پەرەز بىلەن رەدىيە بىرىش خاتا دەپ قارايمەن. ئاللاھ شۇ شەيئىنى قانداق شەكىلدە ئاۋاز چىقىرىشقا ياكى شەكىلىنىشكە بۇيرۇق بىرىدۇ ھەم تەغدىر قىلىدۇ  دەپ قاراش كىرەك. تىمامغا كىلەيلى سەلكىندىكى چاغدا يوللىغان شۇ تىمىدىمۇ شۇنچىلىك تالاش- تارتىش بولغان ،مەنمۇ شۇ مۆجىزىلەرنى ئىتىراپ قىلغۇچىلارنىڭ بىرى ئىدىم. شۇ ئىنكاسلارنىڭ ئىچىدە  بىرسىنىڭ يازغان ئىنكاسنى ئۇقۇپ قالدىم  ئۇ ئىنكاسىدا دىن، دۈشمەنلىرىنىڭ بىزگە يالغاندىن مۆجىزە دەپ كىشىلەرنى قايمۇقتۇرۇپ ئىنتىرنىت تورىدا ھەرخل ۋىدىئو، شەكىللەرنى ئەۋەتىپ، كەينىدىن يالغانلىقىنى ئاشكارىلاپ ساددا مۇسۇلمان قىرىنداشلىرىمىزنى ئەخماق قىلىپ، بەزى ئىمانى ئاجىز كىشلەرنىڭ ئىمانىنى تىخمۇ ئاجىزلاشتۇرۇپ، دىندىن ككڭلىنى سۇۋۇتۇپ ياندۇرۇشقا،  ئارىدا پىتنە -پاسات تۇغدۇرۇپ جىدەل -ماجىراغا، سىلىپ ئۇيۇن كۈرۈش ئىكەن.دىمىسىمۇ شۇنداقتەك.  قارسام ئاساسى باردەك قىلىدۇ، شۇنىڭ بىلەن مۆجىزە توغۇرلۇق  ئىنتىرىنىت تورىدا ئىز دنىش ئىلىپ باردىم. بۇلۇپمۇ قۇرئان كەرىمنى.  بىر قىزنىڭ كۆيدۈرىۋىتىپ مايمۇنغا ئايلىنىپ قىلىشتەك. بىر مۆجىزە بۇلىدىغان، شۇ سۈرەتتە چىقارغان ھايۋاننى  بىر قىتىم توردا كانادادىكى بىر گىزىتتە سۈرىتىنى چىقىرىپ، شۇنداق بىر ھايۋاننىڭ بايقالغانلىقىنى يىزىپتۇ بۇنى كۆرۈپ ھەيران قالدىم. مەن قايسى كۈنى بۇنى توردا مۆجىزە شەكىلىدە كۆرگەنتىم بۇ قانداق بولغىنى دەپ قالدىم. يەنە بىر قىتىم بىرسى ماڭا چىچىدا ،ئاسماندا ناماز شام ۋاختىدەككەن. قارا بۇلۇتنىڭ شەكىللىنىشى  شۇنداق قارسا ھەقىقەتەن بىر كۆز گە ئوخشايدۇ. كۆزىدىن ياش ئاققان يانا شۇنداق غەزەپلەنگەن ھالاتتە تۇرغانلىقىنى كۆرسەتتى، بىرسى بۇنى خۇتەندە تاتىپتىكەن ،دىسە يەنە بىرسى يەنە باشقا شەھەر، دۆلەتنى دەپ بەكلا قايمۇقتىرىۋەتتى. قارىشىمچە   بۇنىڭ بەزىلىرى ھەقىقەتەن دىن دۈشمەنلىرىنىڭ ئۇيۇنىكەن دىگەن ئويغا كەلدىم ،لىكىن شۇنداقتىمۇ مەن ئاللاھنىڭ قۇدىرىتىنى بۈلگەچكە يەنىلا ئۇنى ئىنكار قىلىشقا تىلىم بارمىدى.ئاخىرى مۇنداق قارارغا كەلدىم مۆجىزىلەرنى كۆرسەممۇ ھەيران قىلىپ كەتمەيدىغان  ھەم توردا مۇنداق مۆجىزىلەرنى كۆردۈم دەپ داۋراڭ سالمايدىغان بولدۇم . ئاللاھ ھەممىگە قادىر دەپلا قۇيىمەن.مۇشۇ ئۇسۇل دۈشمەننىڭ ھىلىسىنى بەربات قىلىدۇ ، تىمىدا جدەل قىلىش ھەددىن ئارتۇق ماختاشلار دىن ، دىن دۈشمەنلىرى مۇرادىغا يىتىدۇ، شۇڭا ئۇلارنىڭ دىپىغا ئۇسۇل ئوينىماي ئىنكار قىلغۇچىمۇ ئىچىڭىزدە بۈلۈڭ ھەرگىزمۇ ئاشكارا قارىشلىشىپ جىدەل تىرىماڭ. بۇلۇپمۇ تىمىنى ئۇقۇغان ئىنكاسچىلار تىما يوللىغۇچىنىڭ ئىدىيە خاتالىغى ۋە ئۇقۇغۇچىنىڭ ئىدىيىسىدىن ئۆتمىگەن مەسىلىلەر بولسا شۇنى تۈزۈتۈپ ۋە تىمىسىغا ئاساسى بار رەدىيە بەرسە، باشقىلار چىچاڭشىپ كەتمەي، شۇ تىمىغا قارىتا پىكىرىڭىزنى يىزىڭ ياكى شۇ تىما تۇغۇرلۇق ئىلمىي مۇنازىرە قىلىڭ. بۇلۇپمۇ باشقۇرغۇچىلار مۇنبەرنىڭ تىنىچلىغى ئۈچۈن. ئىنكاسچىلارنىڭ بىر -بىرىنى ھاقارەت قىلىدىغان، ئاساسى بولمىغان رەدىيەلەرنى  بەرگىنىنى كۆرگەندە دەرھال ئۈچۈرۈش كىرەك. مۇنازىرە مۇنبىرى بولغاندىكىن مۇنزىرىنى ئادەم قايىل قىلغۇدەك، دەلىل بىلەن سۆزلەش،ئىنكاس يازغاندا ئەدەپلىك ،سىلىق، كەمتەرلىك بۇلۇش كىرەك ،ئىنكاستا سەن نىمىنى بۈلەتتىڭ ،سىنىڭ قۇلۇڭدىنى نىمە كىلەتتى ،قۇلۇڭدىن كەلگىنىنى قىل،سەن تىخى تۈنۈگۈنكى سويما،تور موللىسى ،ئىشان غۇجام ،قارىم دىگەندەك كىشىنى ھاقارەت قىلىدىغان زەھەرلىك يىلاندەك چاقىدىغان ،ئادەمنى  چۈشۈرىدىغان ئىبارىلەر بولسا دەرھال شۇ ئىنكاسنى ئۈچۈرۈش ھەم ئاگاھلاندۇرۇش كىرەك ،شۇنداق بولغاندا بۇ تور دۇنياسىدا پىتىشىپ ئۈتەلەيمىز ،
