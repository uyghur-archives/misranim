ئەسسالام ئەلەيكۇم ھۆرمەتلىك مۇنبەر زىيارەتچىلىرى ، كەچلىكىڭلار خەيىرلىك بولسۇن !
بۈگۈن 12 - ئاينىڭ 2 - كۈنى ئازنى . بۈگۈنكى خەۋەرلىرىمىزنى كۆرىشىڭلارنى قارشى ئالىمىز !

ئاۋۋال بىر ئىككى پەدە ئىلان كۆرۈڭلار :
ئۇستازىمىز مۇسلىنە 60 ياشقا كىرىپ قالغان تۇرۇقلۇق كۆزلىرى شۇنداق روشەن ۋە تىتىك ... سەۋەبى ئۇ ھەر كۈنى بەش ۋاخ قارىقۇش ماركىلىك كىرپىك لاكى ۋە يۇلغۇن كۆمىرى ئىشلەتكەنلكىدە .... نىمىگە قاراپ تۇرىسىلەر قىزلار ، تەكلىماكان يۇلغۇنلىرىنىڭ كۆمىرىدىن جەۋھەرلەنگەن قارىقۇش ماركىلىق قاش - كۆز لاكى سىزنى گۈزەللەشتۈرگۈسى !!!
پار - پار ماركىلىق ئاياغ مېيى ئايىغىڭىزنى پاقىرتىپ سىزنى تېخىمۇ سالاپەتلىك قىلىدۇ ، پار پارنى ئىشلىتىپ پاقىراپ كىتىڭ !  پار پار ...مۇنبەردىكى باش ۋاكالەتچى - ئىنتىلجان . قىسقا ئۇچۇر ياكى چ چ توپى ئارقىلىق ئالاقىلەشسىڭىز بولىدۇ .
قاۋۇل ماركىلىق ئەرلەر قەھۋەسى ۋە ئەۋرىشىم ماركىلىق ئاياللار قەھۋەسى سىزنى رومانتىك تۇيغۇلارغا باشلاپ كىرگۈسى ، جايلاردىن ۋاكالەتچى قىچىرىمىز  بىرنى ئالسىڭىز ئىككىنى سوۋغا قىلىمىز  ۋاكالىتەن ساتقۇچى نەي نەي .
مەن توي قىلاي دەيمەن ئەمما يۈزۈمدىكى بۇ سەپكۈن ، دانىخوررەكلەر بىلەن قېينىئانامنىڭ يۈزۈمنى ئاچقاندا يۈزۈمگە سۆيۈپ قويماسلىقىدىن ئەنسىرەۋاتىمەن ، كورىيەگە چىقىپ كۆرىنىپ كىرەيمىكىن يا .... ئەنسىرمەڭ شاتئاي ، نىمىگە ئەنسىرەيسىز . بۇنىڭ داۋاسى يۈزيورۇق ماركىلىق ئالتۇن سۈيى يۈز مېيى ، سىز بۇ ماينى تۆت باسقۇچ ئىشلەتسىڭىزلا يۈزىڭىزدىكى ھەرقانداق سەپكۈن ۋە دانىخوررەك ، تەمرەتكىلەرنى سۈپۈرۈپ تاشلايدۇ . يۈزيورۇق مايلىرى !!!!
ئاقسۇدىن تەكلىپ قىلىنغان مۇدىر ۋىراچ گۇناھمۇ مۇشۇ ئاينىڭ ئاخىرغىچە مۇنبەردە كىسەل كۆرىدۇ .  ئەر ئاياللاردا  بولىدىغان باش قېيىش ، كۆز قاراڭغۇلىشىش ، يىراقنى كۆرەلمەسلىك ، پۇر قوللىرى قولىشىپ قېلىش قاتارلىق كىسەللەرنى ھەقسىز كۆرۈپ قويىدۇ  ![/color

خەۋەرلىرىمىزنىڭ تەپسىلاتى تۆۋەندىكىچە :

مۇشۇ ئاينىڭ 2 - كۈنى چۈشتىن كىيىن مۇنبەر زۇڭتۇڭى مىسرانىم ،  باش ۋەزىرى ھۆكۈمران ، مۇنبەر ئىچكى ئىشلار مىنىستىرى ئەلتېكىن ، مۇنبەر ئەزالار ئۇيۇشمىسىنىڭ رەئىسى سەھرا ئوغلى ، كاتىپ توخارلار بىلەن بىرلىكتە مۇنبەرنى كەڭ كۆلەمدە تەكشۈرۈپ تەتقىق قىلدى .
مىسرانىم باشچىلىق تەكشۈرۈپ تەتقىق قىلىش ئۆمىكى ئەڭ ئاۋۋال مۇنبەرنىڭ تائام مەدىنىيتى سەھىپىسىگە كىرىپ سەرخىل تائاملاردىن ھوزۇرلانغاچ بۇ سەھىپىدىكى ئىشچى - خىزمەتچىلەر ۋە زىيارەتچىلەر بىلەن پىكىرلەشتى ۋە بۇندىن كىيىن يەنىمۇ سەرخىل تائاملارنى مەيدانغا كەلتۈرۈپ مۇنبەر ئەھلى ئۈچۈن تۇمۇشتا ئەمىلي نەپلەرنى ئېلىپ كىلىشىنى ئۈمۈد قىلدى .
تەكشۈرۈش ئۆمىكى بۇ سەھىپىدىن چىقىپ خەۋەر ئۇچۇرلار سەھىپىلىرىدىكى ئەڭ يىڭى خەۋەر ، خەلىقئارالىق خەۋەر ۋە مۇنبەر خەۋەرلىرىنىڭ يوللىنىش ئەھۋالى ، ئىگەللىگەن ۋاختى ۋە ئۇلارغا چۈشكەن ئىنكاسلاردىن كەڭ ئەزالار ۋە زىيارەتچىلەرنىڭ ياقتۇرغانلىقىدىن مەمنۇن بولۇشتى .
تەكشۈرۈش ئۆمىكى قايتىپ چىققاندىن كىيىن كومپيۇتېر ۋە تور بىلىملىرى ، كەسپى سەھىپە رايۇنىغا كىرىپ بۇ يەردىكى كەڭ جامائەتچىلىكنىڭ ھاجىتىدىن چىقىۋاتقان تىمىلار ۋە ئۇلارنى يوللىغۇچىلار بىلەن پىكىرلىشىپ بۇندىن كىيىن توختاپ قالماستىن مۇنبەر ئۈچۈن خەلققە پايدىلىق ئىشلارنى قىلىشنى تەۋسىيە قىلدى .
تەكشۈرۈش ئۆمىكى مۇنبەر ھەربىي ئىشلار قوماندانلىق رايۇنىغا كىلىپ بۇ يەردىكى كەڭ ئوفىسسىر ، جەڭچىلەر  بىلەن قىزغىن كۆرۈشۈپ ئۇلارنىڭ بۇندىن كىيىنكى مۇنبەرنىڭ ئامانلىقى ۋە مۇقۇملىقى ئۈچۈن ھەسسە قوشىشىنى ئۈمۈت قىلدى  ، ۋە شۇنىڭدەك  مۇنبەرنىڭ جاي - جايلىرىدىكى ساقچىلار ، ۋە چىگرا رايۇنلىرىدىكى قاراۋۇللارغا  بىلەن رەڭلىك ئۇچۇر ئارقىلىق سالام يەتكۈزۈپ ئۆزرىھالىق سورىدى .
تەكشۈرۈش ئۆمىكى مەدەنىيەت ۋە تارىخ سەھىپىسىگە كىرىپ  مۇنبەرنىڭ بۇرۇنقى تارىخى ۋە ھازىرقى ئەھۋالىغا قارىتا قىسقىچە بايانات ئالان قىلدى ، بۇ چاغدا نۇرغۇنلىغان مۇخپىرلار ۋە تىلىۋىزىيە خادىملىرى باياناتنى خاتىرلەشتى ، باياناتنىڭ قىسقىچە مەزمۇنى مۇنداق :
- ھۆرمەتلىك مۇنبەرداشلار ، كەڭ زىيارەتچىلەر مۇنبەرنىڭ بۈگۈنكىدەك ئاۋات ، ئەڭ قايناق مۇنبەر بولىشىدا ھەممىڭلارنىڭ تۆھپىسى بار ، بولۇپمۇ بىر قانچە يىلدىن بۇيان بىر قىسىم باشقۇرغۇچىلار ، ئالىي ئەزالار ۋە كونا ئەزالارنىڭ تىرىشچانلىقى نەتىجىسدە مۇنبەر نۇرغۇن بوران چاپقۇنلۇق كۈنلەرنى باشتىن كەچۈرۈپ ھالا بۈگۈنكى كۈنگە كەلدى .
كۆپچىلىكنىڭ بۇندىن كىيىن يەنىمۇ تىرىشقان ئاساستا مۇنبەرنى ئۇيغۇر تور - مۇنبەرچىلىكىدىكى ئەڭ نادىر مۇنبەر قىلىپ قۇرۇپ چىقىشىغا تىلەكداشمەن .
تەكشۈرۈش ئۆمىكى مۇنبەرنىڭ رەسىملەر دۇنياسىغا كەلگىنىدە  قەدىمىي ، قىممەتكە ئىگە رەسىملەر زالنىڭ ھەممە يېرىگە ئېسىلغان ئىدى .
رىنات پىشانىسىدىكى تەرىگە قارىماي رەسىملەرنى بىر بىرلەپ تاملارغا ئېسىۋاتاتتتى . تەكشۈرۈش ئۆمىكى زالدىكى تارىخى رەسىملەر ، كۈلكىلىك رەسىملەر ۋە قول سىزما رەسىملەردىن قانغۇچە ھوزۇرلىنىپ بۇندىن كىيىنكى رەسىملەر سەھىپىسىنى تېخىمۇ موم مەزمۇنلۇق رەسىملەر بىلەن تولدۇرۇشنى تەلەپ قىلىپ قايتىشتى .
تەكشۈرۈش ئۆمىكى جەمئىيەت ۋە تۇرمۇش سەھىپىسىگە قەدەم باستى ، بۇ يەردە ھەل قىلىنماي قېلىۋاتقان تۇرمۇش ۋە ئائىلە ئىشلىرى ، جەمئىيەتتىكى ئىشلار ۋە مۇنبەر خانىم قىزلىرىنىڭ دات پەريادىغا ھەمدەم بولۇپ ئۇلارنىڭ تەلىپىنى قاندۇرۇشقا كىپىللىك قىلىپ زىيارىتىنى داۋام ئەتتى .
تەكشۈرۈش ئۆمىكى ئۇنۋېرسال كۇلۇپقا كەگىنىدە بۇ يەر سەل قالايمىقان ئىدى . تاغدىن باغدىن بولىنىۋاتقان بەس مۇنازىرىلەر ۋە ئۇلارغا چۈشىۋاتقان ئىنكاسلار ئادەمگە تازا سەت كۆرىنەتتى .
تەكشۈرۈش ئۆمىكى بۇندىن كىيىن ئەدەپلىك ، ئەخلاقلىق ئىنكاس يېزىشقا رىغبەتلەندۈرۈپ تەكشۈرۈشنى داۋام ئەتتى .
تەكشۈرۈش ئۆمىكى ئەدەبىيات گۈلزارى سەھىپىسىگە كەلگىنىدە بۇ يەردە تۇرغان نادىر ھىكايە ، نادىر شىئىرلار ۋە باشقا ئەسەرلەر كۆزنى قاماشتۇراتتى ، ھېلىھەم مۇسلىنە ، يوچۇن ئادەم ... قاتارلىق يازغۇچى شائىرلار ھارماي بىر نەرسىلەرنى يېزىۋاتاتتى ، تەكشۈرۈش ئۆمىكى ئۇلاردىن ھال سورىغاندىن كىيىن يەنىمۇ نادىر ئەسەرلەرنى ۋۇجۇتقا چىقىرىشقا ئىلھاملاندۇرۇپ قايتىشتى .
تەكشۈرۈپ تەتقىق قىلىش ئۆمىكى ئاخىردا مۇنبەرنىڭ غەربىي شىمالىغا جايلاشقان كۆڭۈل ئېچىش كۇلۇبىغا كىلىپ ئىتوت كۈلدۈرگە ، ناخشا مۇزىكا ۋە ئاۋازلىق ئەسەرلەرنىڭ ئورۇنلاشتۇرلىشىنى كۆزدىن كۆچۈردى .
ئاخىردا تەكشۈرۈپ تەتقىق قىلىش ئۆمىكى مۇنبەرنىڭ بىر قىسىم دائىملىق ئەزالىرى ۋە قوللىغۇچىلىرى بىلەن بىرلىكتە چاي سۆھبەت يىغىنى ئۇيۇشتۇردى ،تەكشۈرۈپ تەتقىق قىلىش ئۆمىكىدىكىلەر ھەممىسى بىر قاتار بايانات ئىلان قىلدى :
مۇنبەرزۇڭتۇڭى مىسرانىم : يولداشلار ! مىسرانىم مۇنبىرى ھەممىمىزنىڭ ئىللىق ئائىلىسى ، بۇندىن كىيىن ھەممىڭلارنىڭ كۈچ قوشۇپ مۇنبىرىمىزنى ئۇيغۇر مۇنبەرچىلىكىدىكى ئىسىل مۇنبەر قاتارىدىن ئورۇن ئالدۇرۇشقا تىشىشىڭلارنى ئۈمۈت قىلىمەن .
مۇنبەر باش ۋەزىرى ھۆكۈمران : ھۆرمەتلىك مۇنبەر ئەھلى ، سىلەردىن كۈتىدىغان زور ئۈمۈدىمىز شۇكى مۇنبەر ئىتىپاقىغا ئالاھىدە ئەھمىيەت بىرىلسە ، ئەخلاقلىق ئادەم بولۇش ھەممىمىزنىڭ باش تارتىپ بولمايدىغان مەسئۇلىيتى !
مۇنبەر ئىچكى ئىشلار مىنىستىرى ئەلتېكىن : يۇقىردىكى تەكشۈرۈشتىن شۇنى ھىس قىلدىمكى مۇنبەردە ھازىر ئەڭ كەمچىل بولىۋاتقىنى مۇقۇملىق ۋە ئىتىپاقلىقتەك قىلىدۇ ، بۈگۈندىن باشلاپ مەن شۇنى جاكارلايمەنكى 2012 - يىلى يىل بېشىغىچە مۇنبەر ئەزالىرى تولۇق بىر ئاي سىياسى ئۈگۈنۈش قىلىدۇ .
مىسرانىم ئەزالار ئۇيۇشمىسىنىڭ رەئىسى سەھرا ئوغلى : كۆپچىلىك مۇنبەر تازىلقىغا ئەھمىيەت بىرەيلى ، قۇرۇق چىراي ئىپادىلىرى ۋە كۆچۈرۈلمە ئىنكاسلار كۆپىيىپ قاپتۇ ، بۇندىن كىيىن مۇشۇ ئىشلارغا ئەھمىيەت بەرسەڭلار !
كاتىپ توخار : ھە ، يۇقارقى زاتلارنىڭ سۆزىنى ھەممىڭلار ئاڭلىدىڭلار ! بۇ قېتىملىق تەكشۈرۈپ تەتقىق قىلىشتىن مەلۇم بولدىكى ، مۇنبەر ھازىر تەرەققىيات باسقۇچىدا تۇرىۋاتىدۇ ، بىر قىسىم چەتئەللىكلەر ۋە قېرىنداش مىللەتلەرمۇ كۆز يۈگۈرتىدىغان ھالەت شەكىللىنىپ قالدى . شۇڭا ھەر بىرىڭلار مۇنبەر ئىچىدە ئۆزۈڭلارغا ۋەكىللىك قىلساڭلار ، مۇنبەر سىرتىدىكىلەر ئۈچۈن مىسرانىمغا ، چوڭراق ئېيىتقاندا ئۇيغۇر تورىچىلىقىغا ۋەكىللىك قىلىسىلەر !
سۆھبەت ئاخىرىدا مىسرانىم باشچىلىق تەكشۈرۈپ ، تەتقىق قىلىش ئۆمىكى بىر ئىستاكاندىن كۆتۈرۈپ يىڭىچە مۇنبەر ، يىڭىچە قىياپەت شەكىللەندۈرۈشنى ئۈمىد قىلىشتى ، ۋە بۇندىن كىيىنكى ئىشلارغا ئاسايىشلىق تىلەشتى ...

ئۆز مۇخپىرىمىز سەزگۈرى خەۋىرى .


ئىلاندىن كىيىن باشقا پىروگراممىلىرىمىز بار تەخىر قىلىڭ !
ئېلىجان ، دۇكانغا چىقىپ بىر باك ماي ئېچىرىڭلا ئىرىم ، ئوخشىتىپ بىر لەغمەن ئىتىپ بىرەي ، ئاپا مانا ماي ... پاھ قەدىمىي سۆگەت مايلىرى ... قەدىمىي سۆگەت مايلىرى ئاشخانىڭىزدىن ئايرىلالمايدۇ ، ئاشخانىڭىزمۇ ھەم ئايرىلمىسۇن !
بودەك ماركىلىق كۈچلەندۈرۈش شەربىتىگە رايۇنىمىزدىن چىقىدىغان توشقان زەدىكى ، بۇغا مۈڭگىزى ، چار يىلان قاتارلىق قىممەت باھالىق ماتىرياللار قوشۇلغان بولۇپ ، تىنىڭىزگە قۇۋۋەت بەدىنىڭىزگە  ماغدۇر پەيدا قىلىدۇ . ئالاقىلاشقۇچى : رافىت .
خوشخەۋەر ، خوشخەۋەر ... ئەركەكلەر كۇلۇبىمىز داغدۇغىلىق تىجارەت باشلىدى ، كۇلۇبىمىزدا شاھمات ، ھوشۇق ،  قارتا ، بىليارت ۋە باشقا كۆڭۈل ئېچىش ئەسلىھەلىرى يۈرۈشلەشكەن بولۇپ كەڭ خېرىدارلارنىڭ قەدەم تەشرىپ قىلىشىنى ئۈمۈت قىلىمىز . باش دىرىكتور - جەمشىت ئەپەندىم مۇنبەردىكى ئىشقى بارلارغا سالام يوللايدۇ .
سىز كەتكەندىن بۇيان سىزنى غەزەپ سېغىندىم ، مىنى سېغىندىڭىزما ؟ سىزنىمۇ سېغىندىم رۇئىن بىراق شۈمىۋال ماركىلىق كەمپۈتنىڭ تەمىنى پەقەتلە ئۇنتىيالمىدىم . شۈمىۋال كەمپۇتلىرى ...
ھازىرقى ئاياللاردىكى بۈرە چاچ ، بوستان چاچ ، ۋە قىسقا چاچلارنى تاراشقا ، بوياشقا ماس كىلىدىغان ئەپەندىم شىركىتى ياساپ چىققان تۆمۈر تاغاقلىرى كەڭ خانىم قىزلارنىڭ ھاجىتىدىن چىققۇسى ! بۇ تاغاقنىڭ ئەڭ چوڭ ئالاھىدىلىكى شۇكى بەزى ئاياللار چېچىغا سالغاندىن كىيىن چاچلىرىنىڭ چۇۋۇقلىقىدىن چىشلىرىنىڭ ھېممىسى سۇنۇپ كىتىدۇ  .
مۇنبەردىكى توي قىلىش يېشىدىن ئارتىپ كەتكەن ، ئەمما  مەلۇم سەۋەپلەر تۈپەيلى توي قىلالمايۋاتقان قىز يىگىتلەرنىڭ بېشىنى ئوڭشايمىز ، مەيلى كىم بولسۇن سىز بۇ تۇمارنى ئېسىۋالغاندىن كىيىن خالىغان ئادىمىڭىزنى كەينىڭىزدىن يۈگۈرتەلەيسىز ! باش دۇئاخان قىزخۇمار قەشقىرى ھاجىم .

بۈگۈنكى خەۋەرلىرىمىز تۈگىدى ، كۆرگىنىڭلارغا رەھمەت ! خەير خوش !
