     بۇ تېخى 2 سائەت ئىلگىرى ئۆز كۆزۈم بىلەن كۆرگەن قىسقىغىنە بىر كۆرۈنۈش.
     مەن ئۈچ تاش ئۇزۇن يوللۇق قاتناش بېكىتىدىن بىر دوستۇمنى قەشقەرگە يولغا سېلىپ قۇيۇپ، قايتىپ كېتەي دەپ ‹311› يول ئاپتۇبۇسىغا چىقتىم، ئاپتۇبۇس جەنۇبى شىنخۇا يولى بىكىتىگە كەلگەندە،  ئاپتۇبۇسقا ئىككى نەپەر قازاق ۋە ئۈچ نەپەر ئۇيغۇر بۇلۇپ جەمئى بەش ئادەم چىقتى، ئۇلار غېرىق مەس ئىدى، بۇ قىلىق ئۇلارنىڭ يېشىغا قەتئى ماس كەلمەيتتى، ئۇلار ئەڭ ياش بولغاندىمۇ، 45 تىن تۈۋەن ئەمەس ئىدى،  ئۇلار ئاپتۇبۇسقا چىقىپلا ئالدىدىكى ئورۇندا ئولتۇرغانلارنىڭ يېنىغا كىلىپ: ‹ بىز ھازىر مەست، ئۆرە تۇرالمايمىز، بىزگە ئۇرۇن بۇشتۇپ بىرىڭلار!› دەپ باشقىلارنىڭ ئورنىنى مەجبۇرى تارتىۋالدى، ئورنىنى مەجبۇرى بۇشاتقانلارنىڭ  ئىككىسى  ياش قىزلار،(بەلكىم ئوقۇغۇچى بولۇشى مۇمكىن) بىرى ئۇتتۇرا ياش خەنزۇ يولداش، يەنە ئىككەيلەن  ياشانغان ئەرخۇتۇنلار ئىدى، بۇنى كۆرۈپ مىنىڭ ئاچچىقىم كەلدى ئەمما دەماللىققا گەپ قىلالماي، ھېلىقى ياشانغان ئىككەيلەننىڭ بىرسى ئولتۇرسۇن دەپ مەن ئورنۇمنى بۇشۇتۇپ بەردىم، ئارقىدىنلا يەنە بىرەيلەن  ئورنىنى بۇشۇتۇپ بەردى،  ئاپتۇبۇس ئىككى بىكەت ماڭا ماڭمايلا يولۇچىلار بىلەن لىق تۇشۇپ كەتتى، مەن ئارقىدىراق بولغاچقا نىمە بولغانلىغىنى بىلەلمىدىم، ئەمما ئالدىدا (ھېلىقى مەستلەر بار يەردە) ۋاراڭ-چۇرۇڭ بۇلۇپ كەتتى، مەن قىستىلىپ يۈرۈپ ئالدىغا باردىم، دەرۋەقە ھېلىقى مەستتىن بىرى بىر ئايالنى ھاقارەتلەۋاتاتتى، ‹خېنىم ئۆزىڭىزنى نىمە چاغلاپ قالدىڭىز؟ ئارقىڭىزدىن قاراپ، قىز بالامىكىن دەپ چاخچاق قىپتىمەن، سىزدەكلەرنى ئۈستەك بەرسىڭىزمۇ *  *  *  * .....› (گەپنىڭ ئاخىرىنى يېزىشقا پېتىنالمىدىم) دەۋاتاتتى، ھېلىقى ئايال:‹ ھايۋان›  دىدى-دە يىغلاپ تاشلىدى، بۇنىڭغا تېخىمۇ ئاچچىقىم كىلىپ تۇراتتى، ئاپتۇبۇسقا يەنە بىر قىز چىقتى،ھېلىقى مەسنىڭ ئاغزى يەنە ئېچىلدى :
_مانا مانداق قىزلار بولسا پۇل دىگەننىڭ كارى چاغلىق _ دىدى.  ھېلىقى قىز نىمە بولغىنىنى بىلەلمەي:
_نىمە؟_ دىدى.
_خانقىز ۋاقتىڭىز بولسا ئويناپ كىلەيلى پۇلدىن غەم قىلماڭ،...._  قىزنىڭ سەل ئاچچىقى كەلدى:
_ ئۈتەپ بارا ساراڭ،
  ھېلىقى مەست ئاستا ئورنىدىن تۇرۇپ قىزنىڭ يېنىغا بېرىپ پىچىرلىدى، لىكىن مەن ئاڭلىيالمىدىم، ئەمما قىزنىڭ زەردە بىلەن مۇنۇ گىپى ئاڭلاندى:
_قىزىڭغا ياكى ئاچاڭغا ئاپىرىپ بەر!
   قىزنىڭ چىرايى قىزىرىپ كەتكەن ئىدى، بۇ چاغدا ھېلىقى مەس قىزغا ھۆپەيدى:
_ئېغىزىڭنى چايقاپ گەپ قىل جۇمۇ! بىزدەكلەر بولمىساق مۇشۇنداق ياسىنىپ يۈرەلەمسەن! ......  
     قانداق بېرىپ قالغىنىمنى بىلمەيمەن، ئۇ مەستنىڭ مۈرىسىنى تارتىپ كاچىتىغا بىرنى ئۇردۇم، ئۇ سەنتۈرلۇپ  يىقىلىپ چۈشتى، ئەمما ئاپتۇبۇستىكىلەر قاچان بىرسى باشلاپ بىرەر دەپ تۇرغاندەك  بۇ مەستلەرگە تىل-ھاقارەت، تۈكۈرۈك دېگەننى ياغدۇرىۋەتتى. ئاڭغىچە ئاپتۇبۇس قىزىل تاغ بىكىتىدە توختىدى، ئۇلار ئاران دىگەندە چۈشۈپ قېلىشتى. ئۇلار چۈشۈپ قالغاندىن كىيىن ھەممەيلەن ئۇلارنى غۇدۇراپ تىللاشقا باشلىدى، ئەمما مىنىڭ ئىچىم ئاچچىق بۇلۇپ قالدى، خىيالىمدا ‹تۇۋا ماۋۇ ئاپتۇبۇستا نۇرغۇن ئادەم بار تۇرۇقلۇق مەن گەپ قىلمىغىچە ھىچكىم گەپ قىلمايدۇ ئەمدى    ئۆلگەندىن كىيىن قەھرىمان چىققاندەك، ھەممىسىنىڭ ئاچچىقى كەپتۇغۇ   قېنى بىزدىكى غۇرۇر، ۋىجدان دېگەنلەر..............› دەپ ئويلاپ قالدىم.

   مەنبە:   ئۆز كۆزۈم بىلەن كۆرگەن ھەقىقى ئىش.
