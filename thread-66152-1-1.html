يېقىندا مىسرانىم باياۋانىدا مەشرەپ بولىدىكەن دەپ ئاڭلاپ كىيىمىمنى مايماق كىيىپ،ساقال بۇرۇتنى مايماق ئېلىپ، مايماق ئىشىگىمنى مايماق مىنىپ،ئۇنىڭغا مايماق ھارۋامنى قېتىپ،ھارۋىكەش  ناخشىسىنى مايماق ئېيتىپ يولغا راۋان بولدۇم.يولدا مايماق ئىشىگىم ھاڭراپ ،مادا ئىشەكلەرنى
ئالداپ، چىرايلىق ئىشەكلەر ئېشىۋېلىپ ،سەتلىرى قېشىغا كىلىۋېلىپ سەپەردە ئانچە زېرىكمەيلا مەشرەپ سورۇنىغا كىلىۋالدىم.ئىشىگىمنى ئېشەكلەر سەھىپىسىگە باغلاپ،مادا ئىشەكلەرگە قاراپ ھاڭراپ مەشرەپكە تەسىر يەتكۈزمە دەپ قاتتىق تەربىيە قىلىپ،شۇنداق قىلماسلىققا ئۇنىڭدىن قەسەم ئېلىپ،مايماق دەسسەپ سورۇنغا كىرىپ كەلدىم.
قارىسام بىكەت باشلىقىنى بارلىق ئەزالار ئوتتۇرىسىغا ئېلىپ، ئاستىغا ئىككى قات كۆپە سېلىپ،ھۆرمىتىگە ناغرا سۇناي چېلىپ ئۇنى يىگىت بېشى قىلىپ سايلاپتۇ.باشقۇرغۇچىلارتوقماق كالتەك كۆتۈرۈپ ،بىلمىگەنلەرگە مەشرەپ تەرتىپىنى ئۆگىتىپ،تەرتىپنى ساقلاپ ، ھەقىقەتنى ياقلاپ تۇرۇپتۇ.ئىسكەتىمدىن
خىجىل بوپ،مايماق گەدىنىمنى مايماق تىرنىقىمدا قاشلاپ،سالام قىلىشنى تاشلاپ،ئىچىگە كىرىپ ،بىر بۇلۇڭنى تېپىپ ئولتۇرۇشۇپتىمەن.ئەمدى ئولتۇرۇشۇمغا قايسى بىر باشقۇرغۇچى شۇنداق دەپ توۋلىدى..
-دات!يىگىت بېشى!
يىگىت بېشى ئۇنىڭغا قاراپ بۇرۇتىنى تالاپ شۇنداق دىدى!
-تېخى مەشرەپ باشلانماي،مىرشاپ تېخى سايلانماي،نىمە دات؟
-ئاباي بىرى ئوتتۇز-يىگىت توققۇز قىزنى كۆزگە ئىلماي،ئۇيغۇر تۇرۇپ جامائەتكە سالام قىلماي،يەنە تېخى ئويۇن چىقىرىپ نازىركومغا دەسسەپ،
جىنايىتىدىن قورقۇپ ئۆزىنى بىر بۇلۇڭغا ئەپلەپ جىم ئولتۇرۇۋالدى!
-مۇنداق قائىدىسىزلارمۇ بارمىكەن تېخى ؟ دەرھال يۈزىگە سۇ چېچىپ ،ئاندىن تۇنۇر قىلىپ بەدىنىدە نان يېقىپ،ئاندىن پىشانىسىگە قائىدىسىز دەپ يېزىپ،ئاندىن ئىشەك قىلىپ ھارۋىغا قېتىپ ،مىسرانىمدىكى ئەڭ سېمىز قىزدىن بەشنى تاللاپ ھارۋىغا چىقىرىپ مەشرەپ سورۇنىنى بەش ئايلاندۇرغاي!!!
مەن جازادىن قورقۇپ ،ئىچىمدە مىڭ باھانە توقۇپ،بىلىدىغان ئايەتلىرىمنى ئوقۇپ ،ئۇلارنى تاپاممىسىكەن دەپ شۇنچە دۇئا قىلساممۇ ئۈنۈمى بولماي سورۇننىڭ ئوتتۇرىغا سۆرەپ كىلىندىم ۋە شۇنداق نالە قىلدىم !

مايماق ئۆيدە جىم ياتماي بالاغا قالدىم ئاخىرى،
سالام قىلىشنى ئۇنتىغاچ جازاغا قالدىم ئاخىرى.
گۇناھىمنى تونۇدۇم،يىگىت بېشى كەچۈرگەي،
گۇناھكارلار دەپتىرىدىن ،ئىسىمىمنى ئۆچۈرگەي.

ماڭا خۇدانىڭ رەھمىتى كىلىپ ،نالىلىرىم بىر باشقۇرغۇچىنىڭ باغرىنى تىلىپ،يىگىت بېشىنىڭ قۇلاق تۈۋىگە كىلىپ،ھۆرمەتلىك يىگىت بېشى بۇ گۇناھكارنىڭ پۇت قولى مايماق،كۆزلىرى چاپاق، بۇرنى پاناق ،ئىدىيىسى قالاق تۇرىدۇ...جازا كۆتۈرەلمىگىدەك شۇڭا ئۇ پۈتكۈل خالايىقنىڭ سالىمىنى
تولۇقلىسا ،ئۇنى كەچۈرۈم قىلساق دەپ پىچىرلىدى.يىگىت بېشى بۇگەپلەرنى توغرا كۆرۈپ :
-گۇناھكارنىڭ تېنى ئاجىز،يېراق يولدىن كەلگەنلىكىنى كۆزدە تۇتۇپ تەن جازاسى كەچۈرۈم قىلىنسۇن، ئەمما باشقۇرغۇچىلارنىڭ تەلىپى بويىچە سالام سائەت قائىدىلىرىنى قوشاق بىلەن تولۇق بىجا كەلتۈرگەي.قوشىقى كۆپچىلىككە يارىمىسا تەن جازاسى ئەسلىگە كەلتۈرۈلگەي!بۇنىڭغا ئەڭ ئەكتىپ تېمىلارغا تەلىپى ئەڭ قاتتىق باشقۇرغۇچى مەسئۇل بولغاي~~~~~
-قېنى ئاۋۋال سالامنى بىجا كەلتۈر،مەشرەپ ئەھلى سالامىڭنى ئىلىك ئالسا كىيىنكى قائىدىلەرنى بىجا كەلتۈرىسەن دىدى باشقۇرغۇچى~
مەن جازادىن قۇتۇلغانغا خوش بولۇپ ،يەنە جازاغا قېلىشتىن قورقۇپ،مىسرا يىپىدىن قوشاق رەختلىرىنى توقۇپ شۇنداق جاۋاپ بەردىم.

ئەسسالامۇ ئەلەيكۇم ئوتتۇز ئوغۇل توققۇز قىز،
كىچىككەن سالامىمنى، چوقۇم ئىلىك ئالغايسىز.
گەر بولمىدى دىسىڭىز،تەن جازاغا قويغايسىز،
شۇ جازادا جان بەرسەم،ئۇۋالىمغا قالغايسىز.

پۈتكۈل مەشرەپ ئەھلىدىن ‹‹ۋەئەلەيكۇم ئەسسلام›› دەپ جاۋاپ كەلدى.

-ئەمدى،مەشرەپ ئەھلىگە تىلەك تىلە،-دىدى باشقۇرغۇچى.مەن شۇنداق قوشاق قاتتىم.


ئەركەكلەر ساغلام  بولغاي ،گويا رامبۇ، خىمەندەك.
بوشاڭ بولماي شاش بولغاي، گويا‹‹يىگەن-ئىچكەندەك››.
قىز-چوكان گۈزەل تۇرغاي، قىزىل گۈلدەك ،چىمەندەك.
مۇلايىملىق يار بولغاي، مەجنۇنتالدەك، لىۋەندەك.


ئۆمرۈڭلار ئۇزۇن بولغاي،شىلانكىدەك لەغمەندەك،
پۇل مېلىڭلار كۆپ بولغاي،ۋارىقى كۆپ دەپتەردەك.
بىلىمنىڭ چوققىسىغا ،ئۇچقايسىلەر كەپتەردەك،
ئىمان ھەم كامىل بولغاي،ئىمام ھەمزە،جەئفەردەك.
