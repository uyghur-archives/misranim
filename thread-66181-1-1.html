     ئەسسالاممۇئەلايكوم مۇنبەر ئەھلى خەيىرلىك سەھەر ، ئۆز مۇخبىرىمىز       خەۋىرى
ھەممىمىز ئىنتىزارلىق بىلەن كۈتكەن ئىسپانىيە 16-نۆۋەتلىك دىربىسى 2011-يىلى 12-ئاينىڭ11-كۈنى سەھەر ئۈرۈمچى ۋاقتى سائەت 3تە باشلاندى.... خان جەمەدى مادىرىد كوماندىسى ئۆز مەيدانىدا بارسىلونا كوماندىسىغا3:1نەتجە بىلەن يېڭىلىپ قالدى.ئەمدى تەپسىلىي  مەزمۇنىنى كۆرگەيسىلەر:
   باش رېپىر بوربالەننىڭ پۇشتەك چېلىشى بىلەن مۇسابىقە باشلاندى...مۇسابىقە باشلىنىپ 23سېكىنۇت بولغاندا فىرانسىيىلىك ئالدى ھۇجۇمچى بېنزىما(Karim Benzema)نىڭ بىر توپى بىلەن نەتىجە 1_0بولدى .(يۈرگىم قارتلا قىلدى دېسە)  بۇ ھال بارسىلونا كوماندىكىسىلەرنى سەل تەنتېرتىپ قويغان بولسىمۇ  ئۇلار خېلى تەستە ئۆز ئىپادىسىنى تاپتى ، بۇ يەنە باش قوماندان مولىنىئاۋنىڭ تاكتىكىسىنىڭ ئۈنۈمى...
  لېكىن 30 مىنۇت بولغاندا مىسسى (پەتەك) قورچاۋنى بۈسۈپ چېقىپ سانچىزغا پاس بەردى ،ئۇتوپنى توختىتىپ ۋارتارنىڭ سول تەرىپىگە ئۇرۇپ نۇمۇرنى 1_1 قىلىپ تەڭلەشتۇردى.
   ئەمدى بارسىلونا كوماندىسىدىكىلەر ئۆز ئىپادىسىنى خېلىلا تاپتى ، پۇيىلنىڭ تەۋرەنمەس مۇداپىيىسى ،ۋە ئۇنىڭ ئوتتۇرا ئارقا مەيداننى ياخشى كونتىرول قىلىشى ۋە مىسسى ، فابرىگاس ، يىنىستا ، شاۋېي ،سانچىز قاتارلىقلارنىڭ ماسلىشىشى نەتىجىسىدە ئۇلار بۇلتۇرقى دىربىدىكىدەك ھالىتىگە تولۇق قايىتتى...
  كىيىنكى مەيدان مۇسابىقە باشلىنىپ 53 -مىنۇت بولغاندا شاۋېينىڭ ئۇرغان توپى قارشى تەرەپ مۇداپىيىچىسىگە تېگىپ ۋارتارغا كېرىپ كەتتى...نەتىجە2-1بولدى .    
   شۇ ۋاقىتتا مولىنىئاۋ ئوزىلنى كاكاغا ئالماشتۇرۇپ ۋەزىيەتنى ئۆزگەرتمەكچى بولسىمۇ ئامال بولمىدى.
   ئارقىدىدىن ئىككى مىنۇت ئۆتكەندە ئارۋېسنىڭ ئوڭدىن چىقارغان كاللا توپىنى فابرىگاس ئۈسۈپ گېگۈزىۋەتتى....پاھ ، نېمىدىگەن ھەيۋەتلىك كۆرىنىش ،ھە!  ئەمدى مۇسابىقە ئاياغلىشىپ بولغانتى....چۈنكى مادىرت كوماندىسىدىكىلەر كەيپىياتىنى تەڭشىيەلمىدى ، .ئۇلار ئۆزىنى ئوڭلاپ بولالمىدى.
  84مىنۇت بولغاندا ، سوغۇققان قاتىل كاكا سول تەرەپتىن قىستاپ ناھايىتتى خەتەرلىك توپتىن بىرنى ئۇردى ، لېكىن قاپقىچى ئۇنى ئۇرۇپ چىقىرۋەتتى(گەرچە بارسىلونانىڭ مەستانىسى بولساممۇ ،كاكانىڭ مۇشۇ توپىنى كېرگەن بولسىچۇ دەپ ئويلىدىم)
  شۇنداق قىلىپ بۇ مۇسابىقە 3-1 نەتىجە بىلەن ئاياغلاشتى ، بارسىلونا يەڭدى . راس گەپنى قىلسام مادىرىت بارسىلوناغا يانىلا سەي بولمىدى،  لوڭقا تەلۋىسى مولىنىئاۋنىڭ تاكتىكىلىرى شۇنچە ياخشى بولسىمۇ ، ئۇلار بارسىلونانىڭ شاھلىق ئورنىنى تەۋرىتىپ بولالمىدى...
[img]/http://g.[/img]

  ھۆرمەتلىك تورداشلار بۇ قېتىملىق خەۋىرىمىز دەل مۇشۇ يەردە ئاياغلاشتى(مادىرتنىڭ تېخىمۇ ياخشى مەشىق قېلىپ كېلىشىنى ئۈمىد قىلمىز)،ئىسيانكار  ئولتۇرۇپ تۇرۇپ ئاڭلاتتى...
بۇلبۇللار تۇتقىن بولدى يۇرتنى بۇلغىدى قاغا ،
كىم بولسا ۋەتەنپەرۋەر بىر كۈن ئىسىلۇر دارغا .

مۇنبەردىكى بارسىلونا  مەستانىلىرىنى تەبرىكلەيمەن.

چوڭ تەرەت قىلماق  تەس !

جاھاندا ھەممىدىن  چوڭ تەرەت قىلماق  تەس !
  
  جان كىرەك پۇل كىرەك جېنى  بارلارغا  ،
ئاش كىرەك نان كىرەك ، قېنى  بارلاغا .
ياخشچاق بولماققا ئەمەلدارلارغا  ،
داخىلسەن ئاخىرى ئاچچىق سورونغا .
يەپ بولوپ كەينىدىن تەرەت قىلماق تەس !

ياق دىيىش لوغەتتىن ئۆچكەن بولىدۇ .
ئۇنداقلار ھەم سىنى« قۇلى»  قىلىدۇ !
گەر خۇشى كەلگەندە   ھالۋا بىرىدۇ ،
تۇتسام ھەم بىر خۇيى ، قايتۇرۋالدۇ !
يىگەنى ئاغزىدىن تەرەت قىلماق تەس !

تەس ئەمەس باشقىنى ماختىماق ، سۆكمەك ،
سۇ بولسا تۇپىدا ئەنجان تام ئەتمەك .
ھەم يەنە جان بولسا جاھاندىن كەتمەك ،
ئەمما تەس بۇ جاننى راۋرۇس جان ئەتمەك .
ئەتمەككە يەپ بولوپ تەرەت قىلماق تەس !

تەس دىسە راستىنلا تەسمىكىن دىمەڭ .
بۇنداق گەپ قىلسام ھەم مەسمىكىن دىمەڭ !
ئاغزىمغا كەلدى دەپ بارچىىنى دىمەڭ!
ئالدىمغا كەلدى دەپ ھەممىنى  يىمەڭ !
يىسىڭىز ، كەينىدىن تەرەت قىلماق تەس !

مۇشۇنداق بىر گەپلەرنى تىما قىلىپ يىزىپ ۋاقتىڭىزنى ئىسراق قىلىپ يۈرمەڭ .نىمە قىلىسىز  سىزدىگەنگە ئۇنداقلار  توختاپ قالامدۇ؟ يازسا يازمامدۇ ،خاتامۇ ؟ ئۆزىنىڭ ئىچكۈسى يۇق ، بىراۋلار زورلىسا ، ئىچمىسە سىزدەك زوڭتۇڭ بولالمىسا ، ئىچسە تېخى گۇناھكار بولسا ، يازمامدۇ ئەمسە !شئېر دىگەن ئاشۇندا نەسە ، ئاچچقىنى كەلتۈرگەن ئىشلارنى   قامچىلايدىغان ، ياقتۇرغان ئىشلارنى  ئالقىشلايدىغان !بولدى سىزمۇ ئارتۇق ئاۋارە بولماڭ ئۇلارغا چۈشەندۈرۈپ ، ئۇلار ئۇنداق يازسا ، سىزگە قىلچىك زىينى بارمۇ ؟ يوق ، ئەمما نىمىشقا ئاچچىقلايمىز ، ئەدىپلىرىمىزنى دەپسەندە قىلدى شۇڭا ......

ھاھاھا.................،مەن جىق سۆزلمەي ،تەلەيلىك بولوڭ !
قىزلار ئۈچۈن چۈشكۈنلەشمەستىن تېخىمۇ قەيسەر بول!
