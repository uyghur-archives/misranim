             بۇنداق «نەپ يەتكۈزۈش»نى كۆرگەنمۇ؟  
   مەن ئادىتىم بويىچە بۈگۈنكى دەم ئېلىش ۋاقتىمدىن پايدىلىنىپ، گەرچە بىرەر نەرسە سېتىۋالمايدىغان بولساممۇ، لېكىن سودا سارايلىرىنى ئايلىنىپ كېتىپ باراتتىم.  زامانغا لايىق كۆركەم لاھىيىلىنىپ ناھىيە بازىرى مەركىزىگىلا سېلىنغان، خېلى ھەشەم-پەشەم بىلەن بېزەلگەن، تۇخۇ سۈتىدىن باشقا ھەممىسى دېگۈدەك تولۇقى بىلەن تېپىلىدىغان بۇ  ئاۋات سودا سارىيىدا  شۇنچە كۆپ كىشىلەر توپلىشىۋالغان  بىر جايغا بېرىپ  قىسىلىپ دېگۈدەك مىڭ بىر مۇشەققەتتە كىشىلەر توپى ئارىسىغا كىرىپلا قاتتىق ھەيرەتتە قالدىم. چۈنكى شۇنچە كەڭرى سودا سارىيىنىڭ خېلى بىر قىسىمنى ئىگىلىگەن  ئائىلە ئېلىكتېر سايمانلىرى سېتىلىدىغان  بۇ ئورۇنغا تىقما-تىقماق بولۇشۇپ، كونا تېلىۋىزورلارنى تۇتقىنىچە شۇنچە كۆپ كىشىلەرنىڭ دۆۋىلىنىپ كەتكەنلىكى مېنى قاتتىق ئەجەبلەندۈردى ۋە تولىمۇ قىزىقتۇردى! بۇ قايناق بازارغا بىردەم ھەۋەسلىنىپ قاراپ تۇرۇش ۋە بولىۋاتقان ئېلىم-بېرىم، مۇنداقچە ئېيتقاندا، «پايدىنى خېرىدارغا ئۆتۈنۈپ بېرىپ، ئۇلارغا كەڭ-كۆلەمدە نەپ يەتكۈزۈش» مەقسەت قىلىنغان بۇ «خالىسانە پىدائىيلىق بىلەن يۈكسەك ئىنسانپەرۋەرلىك»تە  تولىمۇ قىزغىن ئېلىپ بېرىلىۋاتقان بۇ پائالىيەتتىن، يۇقىرى ماھارەتتە لاھىلىنىپ خېلى نەپىس بېزىلىپ،  ئىنتايىن سېخىيلىق بىلەن ئارقا-ئارقىدىن تۈركۈملەپ تارقىتىلىۋاتقان  پالىيەت تەشۋىقات ماتېرىياللىرىنىڭ مەزمۇنى ئارقىلىق مۇنۇلارنى بىلدىم.
بۇ قېتىمقى بۇ «پايدىنى خېرىدارغا ئۆتۈنۈپ بېرىپ، ئۇلارغا كەڭ-كۆلەمدە نەپ يەتكۈزۈش» تۈپكى مەقسەت قىلىنغان بۇ قېتىمقى پائالىيەتنىڭ ئۆتكۈزۈلۈش ۋاقتى تولىمۇ كەمدىن-كەم ئۇچرايدىغان  ئالتۇندەك پۇرسەت بولۇپ، قىسقىلا ۋاقىت ئىچىدە (ئاران ئىككىلا كۈن) ئۆتكۈزىلىدىكەن. بۇ مەزگىلدە خېرىدارلار ئۆيىدىكى تېلىۋىزورلارنى ئېلىپ كەلسىلا، تېلىۋىزورنىڭ ماركىسى، چوڭ- كىچىكلىكى ۋە ئىشلىتىش يىل چەكلىمىسى قانچىلىك بولۇشتىن قەتئىي نەزەر، سۈرىتى، ئاۋازىنىڭ ياخشى- يامانلىقىغا قاراپ، ئەڭ تۆۋىنى 150 يۈۋەندىن 600 يۈۋەنگىچە باھاغا سۇندۇرۇلۇپ،داڭلىق ماركىلىق تېلېۋىزورغا ئالماشتۇرىلىدىغانلىقى... تەنتەنە بىلەن جاكارلىنىۋاتاتتى ۋە ئاكتىپ مۇلازىمەت قىلىنىۋاتاتتى....
مەن كۆز ئالدىمدىكى بۇ قايناق مەنزىرىنى كۆرۈپ ۋە ئاڭلاپ، ئائىلەمدىكى ئىشلەتكىلى تۇرغىلى بىرقانچە يىل بولغان «كونا» تېلىۋىزورۇمنى دەرھال مانا مۇشۇنداق «يېڭى ،ئېنىرگىيە تىجەيدىغان، ئالاھىدە باھادىكى كىرىستال ئىكرانلىق ئالىي تېلىۋىزورلار»دىن بىرىگە دەرھال ئالماشتۇرغۇم كېلىپ كەتتى. چۈنكى بۇ ماڭىمۇ نەپ يەتكۈزۈش پائالىتى ئەمەسمىدى!!؟؟ بىراق، بۇ «نەپ»كە ئېرىشىش ئۈچۈن ئۇزۇندىن-ئۇزۇن ئۇچىرەتتە تۇرىشىۋاتقانلارنىڭ ئاساسەن دېگۈدەك يېزا- قىشلاقلاردىن كەلگەن جاپاكەش دېھقانلىرىمىز ئىكەنلىكىنى كۆرۈپ، ئۇلارغا مېنىڭمۇ دەرھال ئىنسانپەرۋەرلىكىم كەلدى-دە، بۇ «پايدا»نىمۇ بولسىمۇ شۇلارغا ئۆتۈنۈپ بېرىش قارارىغا كېلىپ، دەرھال كەينىمگە ياندىم. ئەمما قەلب ئىكرانىمدا يەنە مۇنۇ سىماھلار، يېشىمىنى ئالدىراپ تاپقىلى بولمايدىغان تېگى يوق چوڭقۇر خىياللار ئەكىس ئېتىپ تۇراتتى: بۇ راستىنلا  ئەلگە نەپ يەتكۈزۈشمىدۇ؟ ھەممىمىزنىڭ دېگۈدەك تۇرمۇش سەۋىيىمىز خېلى ياخشىلانغاچقا، ئالدىن بېيىغان مانا مۇشۇنداق بىرقىسىم «ئىنسانپەرۋەر ساخاۋەتچىلەر» مەنىۋىيىتىمىزنى بېيىتىش ئۈچۈن ئۆزلىرى زىيان تارتسىمۇ، بىزگە ئەڭ كۆپ ساندىكىلەرنىڭ مەنپەئەتىنى كۆزلەپ نەپ- ئىللىقلىق يەتكۈزۈپ، بىزگە زىمىستاننى ھېس قىلدۇرماسلىق ئۈچۈن چېكىۋاتقان خالىسانە رىيازەتلىرىمىدۇ؟؟؟!!!
ئىقتىساد يەر شارىلىشىۋاتقان، جاھاندىكى كۇللىي مەۋجۇدىيەت پۇل-پايدا-مەنپەئەت ئۈچۈنلا دېگۈدەك تېىنىمسىز تىپىرلاۋاتقان مۇشۇ پەيتتىكى يالغۇز بۈگۈن مەن كۆرگەن جايدىلا ئەمەس، بەلكى باشقا ئورۇنلاردىمۇ خېلى ھەشەم-پەشەم بىلەن بەرىكەتلىك ھەم داغدۇغىلىق ئېلىپ بېرىلىۋاتقان «پايدىنى خېرىدارغا ئۆتۈنۈپ بېرىپ، ئۇلارغا كەڭ-كۆلەمدە نەپ يەتكۈزۈش»تىن ئىبارەت بۇ سېخىيلىقنى زادى قانداق چۈشەنسەك بولار!؟
مەن يۇقىرىقى بۇ ئىشنى كۆرۈپ، ئارىدىن ئانچە كۆپ ۋاقىت ئۆتمەي، مەلۇم بىر يان تېلېفون بازىرى ئالدىدىمۇ   نامى ۋە شەكلى «تېلېفون»لا بولسا، قارا-قوياق يۈز يۈۋەندىن ھېسابلاپ«يۇقىرى باھادا سېتىۋېلىپ»، ئۇنىڭ ئورنىغا خېرىدارلار ئۆزلىرى تاللىغان تېلېفوننى  «پايدىنى خېرىدارغا ئۆتۈنۈپ بېرىپ، ئۇلارغا كەڭ-كۆلەمدە نەپ يەتكۈزۈپ» سېتىشتىن ئىبارەت تەنتەنىلىك ئېلىپ بېرىلىۋاتقان پائالىيەتنى كۆرۈپ، خوجىدارلارنىڭ بۇ سېخىيلىقنى زادى قانداق چۈشىنىشىمنى بىلەلمەي گاڭگىرىغىنىمچە، ئۆيگە كىرىپ تېلىۋىزورنى ئاچسام، توۋا..... يەنە شۇ گەپ! يەنى، خېرىدارلار مەلۇم بىر ماركىدىكى( نامى ئاتالمىدى) كونا  DVD ئاپپاراتىنى ئەكەلسىلا، ئۇنى يۈز يۈۋەنگە ھېسابلاپ سېتىۋېلىپ، خېرىدار 180 يۈۋەن قوشسىلا، يېڭىسىنى بېرىدىغانلىقى ھەققىدىكى ئېلاننى كۆرۈپ، ئەتراپىممىزدا سېخىي ساخاۋەتچىلەرنىڭ شۇ قەدەر كۆپۈيۈپ كەتكەنلىكىدىن سۆيۈنسەم، يەنە نېمە ئۈچۈن بۇ ىوشاللىقلىرىمنى گادىرماش خىياللار چۇلغىۋېلىپ، ماڭا زادىلا ئاراملىق بەرمىدى:
ئەجىبا، ماقۇل، پايدىغۇ خېرىدارلارغا ئۆتۈنۈپ بېرىلسۇن، ئۇنداقتا ئەنە شۇ كونا مەھسۇلاتلار كېرەكسىز تۆمۈر-تەسەكلەر قاتارىدا جىڭلاپ سېتىۋېتىلەرمۇ ياكى قېپىلا پەردەزلىنىپ، « داڭلىق ماركىلىق يېڭى مەھسۇلات» بولۇپ، يەنە ئات ئايلىنىپ، ئوقۇرىنى تاپقاندەك ئۆز ئىگىسى بولمىش شۇ ئىستېمالچى ئىگىلىرىنى تاپارمۇ؟!  ئىستېمالچىلارنىڭ قانۇنلۇق ھوقۇق-مەنپەئەتى كاپالەندۈرىلىۋاتقان مۇشۇنداق ياخشى ۋەزىيەتتە مۇشۇنداق « نەپ يەتكۈشۈش»نى زادى قانداق چۈشىنىش كېرەك؟ مەن مۇنبەرداشلارنىڭ ئىلمىي ئاساستا بۇ ھەقتە ئاز-تولا مۇلاھىزە يۈرگۈزۈپ بېقىشىنى سەمىمىي ئۈمىد قىلىمەن!

     ( ئەسكەرتىش: بۇ پائالىيەتكە مۇناسىۋەتلىك شەخسلەر ۋە مەھسۇلاتلارنىڭ  نام –ئابرويىنى كۆزلەپ، كونكىرىت يەر- جاي ناملىرى ۋە مەھسۇلات ماركىسى قاتارلىقلارنىڭ ئىسىملىرى ۋە مۇناسىۋەتلىك سۈرەتلەر يوللانمىدى)

ئاپتور(ARKZAT) نىڭ تېمىسىغا

مەن بۇتىمىغا تەپسىلى جاۋاپ بىرەي؛ ھۈكۈمەتنىڭ يېزىلارغا چۈشۈرۈلگەن ئائىلە-ئىلىكتېر سايمانلىرىنى تۇلۇقلىما قانۇنى، 2009-يىلى يولغا قۇيۇلدى.2010-يىلى تۇلۇقلىمىنى نەخمەيداندا قالدۇرۇپ بىرىدىغانغا ئۆزگەردى.
تىلىۋىزو 3500 يۈئەنگىچە،  3500*13%= 455
توڭلاتقۇ 2500يۈئەنگىچە،  2500*13%=325
كىرئالغۇ2000يۈئەنگىچە، 2000*13%= 260
يۇقۇردىكى باھالارئەڭ يۇقۇرى ئۆلچەم، لىكىن يۇقۇردىكى باھادىن تۈۋەنلىرىگە 13% بۇيىچە قانچىلىك چىقسا شۇنچىلىك تۇلۇقلىما چىقىرىپ، ئەسلى باھاسىدىن قۇيۇپ بىرىدۇ.
يەنە 2011يىلىنىڭ بېشىدا، بۇ قانۇنغا يانداشتۇرۇپ، كونىنى يىڭىغا ئالماشتۇرۇش، قانۇنىنى يولغا قويدى، بۇنىڭدا  ئالماشتۇرۇش باھاسى؛
تىلىۋىزور 300يۈئەن،
توڭلاتقۇ 260 يۈئەن،
كىئالغۇ 200يۈئەن،
شۇ بۇيىچە ھىساپلاپ ئالىدۇ، لىكىن بۇنىڭ شەرتلىرى تۈۋەندىكىچە؛ سىز ئالماشتۇرماقچى بولغان مەھسۇلات چۇقۇم 2010-يىلى 10- ئايدىن كىيىن ئالغان بۇلىشىڭىز شەرت. قۇلىڭىزدا شۇنىڭغا مۇناسىۋەتلىك تالۇن بۇلىشى شەرت.( بۇ ھەرگىزمۇ شۇ كىشىنىڭ ساخاۋىتى ئەمەس، سىزگە قۇيۇپ بەرگەن شۇ پۇلنى ھۈكۈمەت مالىيەسىدىن ئاجرىتىپ بىرىدۇ)
لىكىن بۇنىڭغا سىزنىڭ نۇپۇس دەپتىرىڭىزنىڭ باش بېتى، سىزنىڭ ئىسمىڭىز چىقىدىغان بەت، سىزنىڭ كىملىكىڭىزنىڭ كۈپەيتىلگەن نۇسقىسى كىتىدۇ.
لىكىن ھازىر كىشىلەرنىڭ قۇلىقىغا تۇلۇقلىما دىگەن ئاڭ سىڭىپ قالغان بۇلۇپ، ئۇلارغا باھانى يۇقۇرى دەپ قۇيۇپ، باھادا 13% ئەمەس، 17% ھەتتا 20-25% قۇيۇپ بىرىدىغانلارمۇ بار. ئەمەلىيەتتە بۇ بىر ئالدامچىلىق. ساتقان مېلى ھەرگىزمۇ قانۇن ئىتراق قىلغان، تۇلۇقلىمىغا كىرىدىغان مال ئەمەس.
بولمىسا 13% تۇلۇقلىمىدىن زىيادە چىقارسا مەزكۇر ئۇرۇنغا زىيان چىقىدۇ.  شۇنىڭدەك ھەربىر تىلىۋىزورنى 300يۈئەندىن يۇقۇرى ئالماشتۇرسىمۇ شۇنداق.
تۇلۇقلىما بىلەن كونىنى يىڭىغا ئالماشتۇرۇشنىڭ بەلگىسىنى چىقىپ قۇياي دىگەنتىم چىقىرالمىدىم، ئەگەر شۇ بەلگە پايپاقتا بولسا چۇقۇم تۇلۇقلىما بار،
لىكىن ئۇ بەلگە يوق ھەرقانداق بىر نەرسىدىن سىزگە قانچىلىك تۇلۇقلىما چىقىرىپ بەرسۇن، بۇنىڭ ھەممىسى ساختا.
مەن قاچان ئەخمەق بۇلىدىكەنمە، قىزلارغا ئالدىنىپ قالىمە.قىزلار قاچان ئەقلىدىن ئازىدىكەن، ماڭا ئىشىنىپ قالىدۇ.
