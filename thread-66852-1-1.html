تىلفۇنۇمنىڭ قاتتىق سايرىشى شۇنچە شىرىن ئۇيقۇمنى بۇزۇۋەتتى....ھەي يەنە بەش مىنۇت ئۇخلىۋالغان بولسام نىمە دىگەن راھەت! ئىرەنچىكلىك بىلەن كۆزلىرىمنى ئاچماي تۇرۇپ تىلفۇنۇمنى ياستۇقۇمنىڭ ئاستىدىنن سىيپاشتۇرۇپ يۇرۇپ تىپىۋالدىم-دە قەيەرلىرىنىدۇر بىسىپ ئۇيقۇمغا دەخلى  
قىلىۋاتقان يىقىمسىز قوڭغۇراق ئاۋازىنى بىسىپ ئۆچۈرۋەتتىم...
ئەمدى چۈش كۆرۈشكە باشلىۋىدىم....يەنە شۇ يىقىمسىز ئاۋاز...ئۇف ئەجەپمۇ ئىچىمنى پۇشۇردى...تىلفۇننى قۇلۇمغا ئىلىپ بىر كۆزۈمنى ئىچىپ ۋاقىتتا
قارىۋىدىم سەھەر سائەت ئالتە يىرىمنى كۆرسىتىپ تۇراتتى...ئاپلا ....تۈنۈگۈن تاغام بىلەن سائەت يەتتە يىرىمدا مەن پىراكتىكا قىلماقچى بولغان مەكتەپنىڭ ئالدىدا كۆرۈشۈشكە ۋەدە بىرىپ قۇيغىنىم ئىسىمگە كەلدى.
دىرىڭڭىدە ئورنۇمدىن تۇرۇپ تۇرۇپ كەتتىم،شۇئان بىشىم پىررىدە قىيىپ كۆز ئالدىم قاپقاراڭغۇ بولۇپ كەتتى-دە جايىمدا موككىدە ئولتۇرۇپ قالدىم. خۇدايىم ساقلىسۇن مۇشۇنداق قىلىۋىرىپ مىڭەمگە قان چۈشۈپ قىلىشتىن...

ئالدىراشلا سۇخانىغا كىرىپ يۇزۇمنى يۇيۇشقا باشلىدىم. ئۆزۈم ئەڭ ياقتۇرۇپ كىيىدىغان  كاستىيۇم-شىمىمنى كىيىدىم. ئاق ياقىلىق مايكام بىلەن كاستىيۇم چاپىنىم مىنى بۆلەكچىلا روھلۇق كۆرسەتكەن ئىدى. نىمىلا دىگەن  بىلەن بۈگۈن بىرىنجى كۈن،چوقۇم ياخشى تەسىر قالدۇرۇشۇم كىرەك...چىچىمنى ئىگىز بىر بۇغۇپ سۇسقىنا گىرىم قىلىۋالدىم. سائەتكە قارىۋىدىم ،يەتتە بوپتۇ،دىمەك يىرىم سائەتكىچە مەكتەپكە بىرىپ بولۇشۇم كىرەك. لىكىن قايسى ئاياغنى كىيسەم بولار؟ ئاياغ ئىشكاۋىمغا قارىۋىدىم، كۆزۈم قارا رەڭلىك ئىگىز پاشنىلىق ئاياغقا چۈشتى،مۇشۇنى كىيەي. تازا ماسلاشقۇدەك. ئەينەككە قاراپ ئۆزۈمنى كەسپىي تەجىربىلىك ئوقۇتقۇچىدەك ھىس قىلدىم-دە مەمنۇن بۇلۇپ سومكامنى ئىلىپ سىرتقا قاراپ يۈگۈردۈم.

تاكسىدىن چۈشۈپلا مەكتەپكە كىرىپ كىتىۋاتقان ئوقۇغۇچىلار ئارىسىدا منى ساقلاپ تۇرغان تاغام بىلەن كۆرۈشتۈم. خۇددى ئىنتىھان مەيدانىغا كىرىشكە تەييارلىق قىلىۋاتقان ئوقۇغۇچىدەك نىمىگىدۇر بىر خىل جىددىيلىشىپ كىتىپ باراتتىم، تاغام سىزىپ قالدى بولغاي ماڭا قاراپ كۈلۈپ كەتتى:
__نازى قىزىم، جىددىيلەشمەڭ، سىز ئەمدى ئالىي مەكتەپ ئوقۇغۇچىسى ئەمەس، بەلكى ئوقۇغۇچىلارغا دەرس بىرىدىغان ئوقۇتقۇچى بولدىڭىز، سىز بۇنداق جىددىيلەشسىڭىز مەكتەپ مۇدىرى سىزنى ئىشلەتمەي قالمىسۇن يەنە...

قىزىرىپ كەتتىم:
__تاغا، شۇ جىددىيلەشمەي دىسەممۇ ئۆزلىكىمدىن جىددىيلىشىپ كىتىۋاتىمەن، ماقۇل مەن ئۆزۈمنى سالماق تۇتىۋالاي، خەق پىراكتىكىغا كەلگەن يىڭى ئۇقۇش پۈتتۈرگەن ئوقۇغۇچى ئەمەس، نەچچە يىللىق تەجىربىسى بار ئوقۇتقۇچى ئىكەن،دەپ بىلسۇن...

تاغام ئىككىمىز كۈلۈشكىنىمىزچە مۇدىر ئىشخانىسىغا كىردۇق. تاغام مۇدىر بىلەن قىزغىن كۆرۈشۈپ كەتتى. قارىغاندا ئۇلار كونا تۇنۇشتەكلا قىلاتتى.
__قادىر ئاغىنەم،كەلدىڭلارمۇ ئاغىنەم، قىنى قىنى، ئولتۇرۇڭلار ....
__ئۇنچە تەكەللۇپ قىلىپ كەتمىسەڭلارمۇ بولىدۇ پەرھات مۇدىر،__تاغام قول ئىلىشىپ كۆرۈشكىنىچە ماڭا قاراپ__قىنى نازى قىزىم،سىزمۇ ئولتۇرۇڭ__دىدى سافانى ئىما قىلغىنىچە.

ئۇنىڭغىچە ئىككسى ئەھۋاللىشىپ بىردەم پاراڭ سىلىشتى، مەن بىرچەتتە ئولتۇرغاچ مەكتەپ مۇدىرىغا سەپ سىلىشقا باشلىدىم، يىشى 45~50لەر ئەتراپىدا بولۇپ چىرايىدىن سالماقلىقى، ئاق كۆڭۈللۈكى خۇشخۇيلىقى كۆرۈنۈپلا تۇراتتى. جىددىيلىكىم ئاللىقاياقلارغىدۇر يوقاپ ئەمدى ئۆزۈمگە ئىشەنگەن ھالدا ئەركىن نەپەس ئىلىشقا باشلىدىم.

ئۆزۈمنىڭ بۇنىڭدىن كىيىن مۇشۇنداق ئاق كۆڭۈل،ھەم كىلىپ تاغامنىڭ يىقىن دوستىنىڭ باشقۇرۇشى ئاستىدا ئىشلەيدىغىنىمنى ئويلاپ بىر تەرەپتىن خۇش بولسام بىر تەرەپتىن خىزمەتداشلىرىم بىلەن چىقىشىپ ئۆتەلەرمەنمۇ؟ ئۇلار مىنى يىڭىيارچى كۆرۈپ كۆزگە ئىلماي قالارمۇ؟ دىگەنلەرنى ئويلاپ ئولتۇردۇم.
__ ھە نازى دىگەن جىيەنىڭلا مۇشۇ قىز ئىكەندە؟ _خىيالىم ئۈزۈلۈپ دىققىتىم مەكتەپ مۇدىرىغا يىغىلدى.دەرھال ئورنۇمدىن تۇرۇدۇم:
__ھە مەن شۇ پەرھات مۇئەللىم.
__سىزنىڭ تەرجىمالىڭىزنى تاغىڭىز ماڭا بەرگەن، بىر قۇر كۆرۈپ چىقتىم، ئوقۇغان كەسپىڭىزمۇ ئوقۇتقۇچىلىق ئىكەن، بىر ئاي پىراكتىكىدىن كىيىن رەسمىي ئوقۇتقۇچىلار قۇشۇنىنىڭ بىر ئەزاسى بولىسىز، بۇ جەرياندا كۆپرەك تەجىربە توپلاڭ، بىلمىگەنلىرىڭىزنى بىزنىڭ ئىلمىي مۇدىرىمىز دولقۇن مۇئەللىمدىن سورىسىڭىز بولىدۇ، سەل تۇرۇپ مەن سىزنى تۇنۇشتۇرۇپ قۇياي. تىرىشڭ قىزىم.
مەكتەپ مۇدىرىنىڭ ئىىلىق مۇئامىلىسىدىن تولىمۇ خۇرسەنلىك ھىس قىلدىم.
__رەھمەت سىزگە پەرھات مۇدىر، چوقۇم تىرىشىمەن،
__ياخشى تىرىشىپ نازى قىزىم ماۋۇ پەرھات مۇدىرنىڭ ۋەزىپىسىنى ئۈستىڭىزگە ئالغۇدەك بولۇڭ-ھە؟__دىدى تاغام چاقچاق ئارلاش.
__قادىر ئاغىنە، مەخسۇس مىنى يىقىتىش ئۈچۈن كەلدڭلارمۇ نىمە؟ قىنى بۇ قىزىمدا راستىنلا شۇنداق جاسارەت بولسا ئورنۇمنى شۇكۈنىلا بوشىتىپ بىرىمەن جۇمۇ؟
ئىككىسى كۈلۈشۈپ كەتتى.
ئۆزۈمنى خۇددى راستىنلا پەرھات مۇدىرنىڭ ئورنىنى ئىگەللىۋالغاندەك خىجىل بولۇپ كەتتىم.
__ئەمىسە پەرھات، نازى قىزىمنى ساڭا تاپشۇردۇم، قىزىمنى ياراملىق ئوقۇتقۇچى قىلىپ چىقىشىڭغا ئىشىنىمەن ئاداش. باشقا ئىشلار ساڭا قالدى-ئەمىسە.
__بولدى خاتىرجەم بول ، نازى مۇئەللىم ھەممىمىزنى خۇرسەن قىلغۇدەك ئىشلەپ كىتىدىغاندەك تۇرىدۇ،شۇنداقمۇ قىزىم_ھە؟
مەكتەپ مۇدىرىنىڭ نازى مۇئەللىم دىگىنىنى ئاڭلاپ شۇنچىلىق خۇش بولۇپ كەتتىم. ھەم مەسئۇلىيەت تۇيغۇسىنى ھىس قىلدىم.مۇئەللىم، ئىيتماققا ئاسان، ۋەزنى قانچىلىق ئىغىر سۆز-ھە؟!

تاغامنى ئۇزىتىپ قۇيۇپ پەرھات مۇدىر دولقۇن ئىسىملىق ئىلمىي مۇدىرغا مىنى تۇنۇشتۇرۇپ قويماقچى بولۇپ مۇئاۋىن مۇدىر ئىشخانىسىغا قاراپ ماڭدۇق. ئەمدىكى خىيالىم ماڭا يىتەكچى بولىدىغان دولقۇن ئىسىملىق ئوقۇتقۇچىنى يۇقۇرى گىرادۇسلۇق كۆز ئەينەك تاقاپ دەرس تەييارلاپ ئولتۇرغان سۈرلۈك ئوتتۇرا ياش بىر كىشىنى دەپ كۆز ئالدىمغا كەلتۈردۈم.

مانا ئىلمىي مۇدىر ئىشخانىسنىڭ ئالدىغىمۇ كىلىپ بولدۇق، ئىشىك يىرىم ئوچۇق ئىدى. پەرھات مۇدىر ئىشىكنى يىنىك چىكىۋىدى،ئىچىدىن :كىرىڭ دىگەن يىقىملىق ئاۋاز ئاڭلاندى.

ئىشىكتىن كىرىپلا كۆرگىنىم مۇدىرنى كۆرۈپلا ئورنىدىن لىككىدە تۇرغان ئىگىز بويلۇق، قاملاشقان 27~28ياشلار ئەتراپىدىكى بىر ياش ئوقۇتقۇچىنى كۆردۈم،كاللامدا ئىشخانىدا ئىمىي مۇدىر يوقكەندە؟ دىگەن خىيال لىپپىدە قىلىپ ئۆتتى.
__ دولقۇن مۇئەللىم، ئالدىراش ئوخشىمامسىز؟__دىدى مۇدىر كۈلۈمسىرەپ.

مۇدىر دىگەننىڭ سەل سۈرى بار دەپ ئاڭلىغان مەن، لىكىن بۇ مۇدىر ئازراقمۇ سۈرى يوق، كىشىشنىڭ ئامراقلىقىنى كەلتۈردىغان پىشقەدەم ئوقۇتقۇچى ئىدى،
__پەرھات مۇدىر، ياخشىمۇسىز،__دىدى،_كەينىدىن ماڭا قاراپ__ ياخشىمۇسىز؟__دەپ ئەھۋال سوراپ قويدى.
مەنمۇ جاۋابەن ياخشىمۇسىز دىگىنىمچە بىشىمنى لىڭشىتىپ قويدۇم.

__سىزگە بىر ئىشنى تاپشۇرغىلى كەلدىم بۇ قىزنىڭ ئىسمى نازى، مەكتىپىمىزگە يىڭى كەلدى، بۈگۈندىن باشلاپ بىر ئاي پىراكتىكا قىلىدۇ، كىيىن رەسمىي سىپىمىزغا قۇشۇلىدۇ__دولقۇن ئىسىملىق بۇ ئىلمىي مۇدىرنىڭ ئويلىغىنىمدىن دەل ئەكىسچە ئىكەنلىكىنى ھىس قىلىپ بايامقى تەسەۋۋۇرۇم كۆز ئالدىمغا كەلدىدە ئىچىمدە كۈلۈپ قويدۇم. مۇدىر گىپىنى داۋاملاشتۇرىۋاتاتتى:

___بۇنىڭدىن كىيىن نازى مۇئەللىمنىڭ بىلمىگەن يەرلىرىگە سىز يول كۆرسىتىڭ، بىر ئايدىن كىيىن سىناق ئالىمەن، لاياقەتلىك بولالمىسا سىزدىن كۆرىمەن-ھە؟__دىدى مۇدىر چاقچاق ئارلاش.

دولقۇن مۇئەللىم دەرھاللا ئىنكاس قايتۇردى:
__خاتىرجەم بولۇڭ مۇدىر، دىگىنىڭىزدەك قىلىمەن، سىز ئويلىغاندىنمۇ ياخشى يىتەكلاپ چىقىمەن. كىيىن مەكتىپىمىزنىڭ غوللۇق ئوقۇتقۇچىسى بولىشىغا ئىشىنىمەن.__دولقۇننىڭ شۇ گەپنى دەۋاتقان پەيتتە كۆزلىرىدىن بىرخىل ئۈمۈدكە تولغان نۇرنىڭ يالىت قىلىپ چاقنىغىنىنى كۆرگەندەك بولدۇميۇ دەرھال پەسكە قارىۋالدىم.

مۇدىر رازىمەنلىك بىلەن بىشىنى لىڭشىتىپ قويدى-دە :
__ئەمىسە شۇنداق بولسۇن، ياخشى ئىشلەڭ،_دىگىنىچە مىھرىبانلىق بىلەن كۈلۈمسىرەپ چىقىپ كەتتى.
__قىنى ئولتۇرۇڭ، ئاۋۇ سىزنىڭ خىزمەت ئۈستىكىڭىز.
__رەھمەت،__سەل قۇرۇنغىنىمچە بىرىپ تۇنجى ئۆزۈمگە تەۋە بولغان خىزمەت ئۈستىلىدە ئولتۇردۇم.

بۈگۈندىن باشلاپ مەن بىر ئوقۇتقۇچى،بۇنىڭدىن كىيىنمۇ ھەم شۇنداق،ئەمدى مەن تىرىشمىسام بولمايدۇ، دادام ئاپامنىڭ :‹‹يىنىپ كىلىپ بىزنىڭ يىنىمىزدا ئىشلىگىن،سىنىڭدىن ئەنسىرەپ قالىدىكەنمىز، ھىلىمۇ ئۈرۈمچىدە تۆت يىل ئوقۇدۇڭ،يەنە قىيمىغۇدەك نىمىسى بار...››دىگەندەك گەپلىرىگىمۇ لايىقىدا جاۋاب بىرىپ، تاغامغا دەپ يۇرۇپ مۇشۇ يەردە يەنى ئۈرۈمچىدە قىلىشنى تاللاپ قالغان ئىدىم، گەرچە بۇ يەردە تارتىشقىدەك ئالاھىدە سەۋەب بولمىسىمۇ تۇغۇلۇپ ئۆسكەن يۇرتۇمغا قارىغاندا ئاران تۆت يىل تۇرغان ئۈرۈمچى شەھىرىگە شۇنچە كۆنۈپ كەتكەن ئىدىم، خۇددى بۇ يەر مىنىڭ تۇغۇلۇپ ئۆسكەن ،كىندىك قىنىم تۆكۈلگەن زىمىندەك...

__ئىسمىڭىز نازىمۇ؟
تۇيۇقسىز سورالغان بۇ سۇئالدىن دەرھال ئىسىمگە كەلدىم،قارىسام دولقۇن مۇئەللىم ماڭا دوستانە قاراپ تۇرۇپتۇ.
__ھەئە،ئىسىم نازى،_دىدىم جاۋابەن.
__ھەقىقەتەن چىرايلىق ئىسىمكەن،ئىسىمىم دولقۇن، بۇنىڭدىن كىيىن ئىككىمىز خىزمەتداش، بىلەلمىگەن يەرلىرىڭىز بولسا،قانداق مەسىلىگە يۇلۇقسىز ئىككىللەنمەي مەندىن سوراڭ.

__رەھمەت سىزگە،__مەن بۇ خىزمەتدىشىمدىن تولىمۇ رازى بولغان ئىدىم،سىزگە پۈتۈنلەي ناتۇنۇش بولغان مۇھىتتا سىزگە ياردەم بىرىدىغان بىراۋنى ئۇچراتسىڭىز ھەقىقەتەنمۇ ئۆزىڭىزنى تەلەيلىك ھىس قىلىسىز،مىنىڭ ھازىرقى ھىسياتىم دەل شۇنداق ھىسىيات ئىدى.

ئەتىسىدىن باشلاپ رەسمىي ئالدىراش بولغىلى تۇردۇم، ئوقۇتقۇچى ئىيتماققا ئاسان،ئورۇندىماق تەس بولغان بىر ۋەزىپىكەنغۇ؟ ئۇھ...ھىلى دەرس پىلانى تۈزسەم ھىلى دەرسكە يۈگۈرۈيتتىم، كەچتە 40ئوقۇغۇچىنىڭ تاپشۇرىقىنى ئۆيگە ئىلىپ بىرىپ بىر تەرەپتىن تاماق يىگەچ بىر تەرەپتىن تاپشۇرۇق تەكشۈرۈيتتىم.

ئارلىقتا ئۆيدىكىلەر تىلفۇن ئۇرۇپتۇ. ئاپامنىڭ دىيىشىچە بۇ قىتىم تاغام خىلىلا كۈچىگەنمىش، شۇنىڭ ئۈچۈن تاغامنى پات-پات يوقلاپ تۇرىشىمنى، ئۆزۈمنى ئاسرىشىمنى، خىزمەت بىلەنلا بولۇپ كەتمەي ئەمدى سەل-پەل ئەتراپىمغا قاراپ قۇيۇشۇمنى جىكىلىدى،(بۇ دىگىنى تۇيۇڭغا تەييارلىق قىل دىگىنى ئىدى).

ئەمدى ئوقۇش پۈتتۈرۈپ ئاران دىگەندە ئىشقا چىقسام ماڭا ھازىر توي قىلىشنى كىم قۇيۇپتۇ؟ ئۇ ئىشلارنى تىخى ئويلاشقۇدەك بولمىغان ئىدىم، ئاپامغىغۇ ھە دەپ قۇتۇلمىسام بىر پاتمان ۋەز ئىيتاتتى.

ئارىدىن ئىككى ئايچە ۋاقىت ئۆتتى. مەنمۇ ھىسابتا كۆنۈپ قالدىم.
بۈگۈن جۈمە ،ئەتە ئۆگۈن ئىككى كۈن دەم ئىلىش كۈنىمىز بولغانلىقى ئۈچۈن ئۆزۈمنى بۆلەكچىلا يىنىك ھىس قىلدىم. تەكشۈرىدىغان تاپشۇرۇقلارمۇ ئۇنچە كۆپ ئەمەس ئىدى. ئۆزۈمچە شەنبە-يەكشەنبە ئىككى كۈننى قانداق ئۆتكۈزۈشۈم توغرىسىدا ئىشخانىدا ئولتۇرۇپ خىيال قىلىشقا باشلىدىم:

‹‹بۈگۈن ئۆيگە بىرىپ ھىچ ئىش قىلماي ياخشى تاماق ئىتىپ تىلۋىزۇر كۆرۈپ ئولتۇراي،دوسلىرىم بولغان بولسىغۇ ياخشى بولاتتى،لىكىن ھەممىسى ئوقۇش پۈتتۈرۈپ ھەرتەرەپلەرگە تاراپ كەتتى، قارىغاندا بۇ شەھەردە تاغامدىن باشقا يىقىنىم يوق ئوخشايدۇ، شەنبە قانغۇچە ئۇخلاپ ئاندىن تاغامنىڭ ئۆيىنى زىيارەت قىلىپ كىلەي،كەچتە قايتىپ كىلىپ تاپشۇرۇق تەكشۈرۈي،دەرس پىلانى تۈزەي...يەكشەنبە...››

__نازى،دەم ئىلىشنى قانداق ئۆتكۈزەي دەيسىز؟__قارىسام دولقۇن ماڭا سۇئال نەزىرىدە تىكىلىپ ئولتۇرۇپتۇ. خىياللىرىمنى بىلىۋالدىمۇ نىمە؟
__تىخى ئويلاشمىدىم،__يالغان سۆزلىدىم، ھازىر دەل شۇنى ئويلاۋاتقان ئىدىم،دىسەم قاملاشمىغۇدەك.
__خالىسىڭىز دەم ئىلىش كۈنى بىللە تاماق يىگەچ ئايلىنىپ كەلسەك، قارىسام سىزنىڭ ئالاقىلىشىدىغان ئادەملىرىڭىزمۇ كۆپ ئەمەستەك قىلىدۇ،_دىدى دولقۇن كۈلۈمسىرگىنىچە.

__ئالاقىلىشىدىغان ئادەملىرىمنىڭ ئازلىقىنى سىز قانداق بىلىسىز؟
__سىز بىلەن بىرگە ئىشلىگىلى ئىككى ئاي بولاي دىدى،يا تۈزۈك تىلفۇن كەلمەيدىكەن، كەلسىمۇ تۇققانلىرىڭىز،يىراقتىكى دوسلىرىڭىز ئىكەن.
__سىز قانداق بىلدىڭىز؟__ھەيرانلىقىمنى يۇشۇردۇم. شۇندىمۇ سەل خىجىل بولدۇم، دىمىسىمۇ ئۈرۈمچىدە ھازىر تۈزۈك ئارلاشقۇدەك دوسلىرىم يوق ئىدى، بولمىسا ئۆزۈممۇ دوست بولۇشقا يارايدىغان قىزلاردىن ئىدىم. ئىچىمدە ئوقۇش پۈتتۈرۈپ يىراقلارغا كەتكەن دوسلىرىمنى تازا تىللىۋالدىم، ئۇلار بولغان بولسا خەقمۇ بۇنداق ئويلىمايتتى.ئۆزۈممۇ يالغۇزلۇق ھىس قىلمايتتىم.

__نىمىشقا بىلمىگۈدەكمەن؟ بىر ئىشخانىدا ئولتۇرۇپ شۇنچىلىك ئىشنىمۇ بىلمىسەم سىزنىڭ يىتەكچىڭىز بولالمايتتىم.
مەنمۇ جاۋابەن كۈلۈپ قويدۇم.
__ئۇنداقتا ئەتە ئىشلىرىڭىز بولسا ئورۇنلاشتۇرىۋىلىپ،ئۆگۈنلۈككە يەكشەنبە كۆرۈشەيلى بولامدۇ؟__دولقۇننىڭ ئىلتىجالىق كۆزلىرى مەندىن جاۋاب كۈتمەكتە ئىدى، دىمىسىمۇ مىنىڭ باشقا ئىشلىرىم يوق،دەل يەكشەنبىنى قانداق ئۆتكۈزىشىمنىڭ خىيالىنى قالغاندا دولقۇن  بۇ تەكلىپنى قويدى، بۇمۇ بولغىنى،يالغۇز يۈرگەندىن كۆرە دولقۇن بىلەن ئايلىنىپ كىلەي،كۆڭلۈممۇ ئىچىلىپ قالار.

__مەيلى ئەمىسە.
__رەت قىلمىغىنىڭىزغا رەھمەت نازى.
__يىتەكچىمنىڭ بۇيرۇقىنى قانداقمۇ رەت قىلاي__تۇنجى قىتىم ئىككىمىز تەڭ ئىچىلىپ كۈلۈپتۇق،شۇندىلا ئوتتۇرىمىزدىكى بىزنى تۇسۇپ تۇرغان قىلىن بىر تامنىڭ غايىپ بولغانلىقىنى سۇس ھىس قىلدىم.

كەچتە ئۆيگە قايتتىم، يۇلدا ماڭغۇچە نىمىشقىدۇر يەكشەنبىلىك ئۇچرىشىشنى ئويلاپ بىرخىل بولۇپ قالدىم، ئويلاپ باقسام دولقۇن بىلەن بىر ئىشخانىدا ئولتۇرۇپ خىزمەت قىلغىنىمغا ئىككى ئاي بولاي دىگەن بولسىمۇ ئىككىمىز ئەزەلدىن تۇرمۇشىمىز توغرىسىدا پاراڭلىشىپ باقماپتۇق، دولقۇننىڭ قىز دوستى بارمۇ-يوق،بۇنى ئەزەلدىن سوراپ باقماپتىكەنمەن، ئەمدى ئويىسام...ۋاي خۇدايىم،خەق مىنى ئىككى ئايدىن بۇيان كۈزەتكەن بىلەن مەن ئەجەپ دىققەت قىلىپ باقماپتىكەنمىنا؟ ئەگەر ئۇنىڭ قىز دوستى بولسا مەن يەكشەنبە كۆرۈشۈپ يۈرسەم قاملىشامدۇ؟
توختا نازى،يەكشەنبە كۈنى چوقۇم سوراپ كۆر،بولمىسا ياخشى بولمايدۇ،گەرچە خىزمەتداش بولساقمۇ ئۇ دىگەن ئىشخانىدىلا قالىدۇ، سىرتقا چىقساق ئوخشىمايمىز....

مۇشۇنداق ئالجۇقا خىياللارنى قىلىپ ئۆيگىمۇ كىلىپ قاپتىمەن.
ئۆزۈمنىڭ پىلانىم بويىچە تاماق ئىتىپ بىر تەرەپتىن تاماق يىگەچ بىر تەرەپتىن ھۇزۇرلىنىپ كىنو كۆرۈشكە باشلىدىم، ئۆزىنى ئەركىن قۇيىۋەتسە نىمە دىگەن راھەت...
يانفۇنۇمغا ئۇچۇر كەلگەنلىك سىگىنالىنى ئاڭلاپ كۆزۈم تىلۋۇزۇردا ئالدىمدىكى ئۈستەلنى سىيپاپ تىلفۇننى قۇلۇمغا ئالدىم.
‹‹بىر ھەپتە ئىشلەپ ھىرىپمۇ قالغانسىز، ياخشى ئارام ئىلىپ، بالدۇر ئۇخلاڭ،خەيرلىك كەچ،يەكشەنبە ئۇچرىشىنى ئۇنتۇپ قالماڭ-ھە؟››
ئۇچۇر دولقۇندىن كەلگەن ئىدى،تۇيۇقسىز بۇنداق يىقىنچىلىققا كۆنەلمىگەن بولساممۇ لىكىن يامان كۆرمىدىم. خىلى ئۇزۇن بوپتىكەن بىراۋنىڭ ماڭا كۆيۈنۈپ بۇنداق ئۇچۇر يۇللىمىغىلى...
‹‹رەھمەت سىزگە›› ئۇچۇر قايتۇرۇپ بولۇپ خۇددى بىرى بۇيرۇق بەرگەندەك ياتاق ئۆيۈمگە ئۇخلىغىلى ماڭدىم.

تاغامنىڭ ئۆيىگە كەلسەم تاغامنىڭ ئوغلى ئەركىن ئوقۇش پۈتتۈرۈپ كانادادىن قايتىپ كەپتۇ، مىنى كۆرۈپ بۆلەكچىلا خۇش بولۇپ كەتتى:
__ۋاھ نازى،چوپچوڭلا بولۇپ كىتىپسەنغۇ؟ كۆرۈشمىگىلى خىلى يىللار بولۇپتۇ،بەكلا ئۆزگۈرۈپ كىتىپسەن جۇمۇ...
ئەركىننىڭ ئاغزى-ئاغزىغا تەگمەي سۆزلىشى مىنى چوڭلار ئالدىدا خىجىل قىلىپ قويدى،ئۆزىمۇ بۇرۇنقى سىرىق ياڭيۇ ھالىتىدىن 180گىرادۇس ئۆزگىرىپ باشقىچىلا ئىسكەت كىرىپ قالغان، بىر قارىسىلا يۇقۇرى مەلۇمات ئالغان كىشىدەك كۆزگە ئىسسىق كۆرىنەتتى.

__بالام،نازى قىزىم چوڭ بولدى، بۇرۇنقى ئۇششۇق،يىغلاپلا تۇرىدىغان نازى ئەمدى يوق، بۇرۇنقىدەك كىچىك بالا كۆرۈپ بۇنداق ماختاۋەرسەڭ خىجىل بولمامدۇ؟__ تاغامنىڭ ئايالى مىھرىبان ئاپامنىڭ شۇنداق دىيىشى بىلەن ئەركىنمۇ جىم بولدى، لىكىن كۆزلىرى تىخى ئويناپ تۇراتتى.

__ھە قىزىم،كۆنۈپ قالدىڭىزمۇ؟__تاغام مىھرىبانلىق بىلەن سورىدى.
__ھەئە تاغا،كۆنۈپ قالدىم،خىزمەتداشلىرىممۇ بەكلا ياخشى،ھەر جەھەتتىن كۆيۈنۈپ كۆپ ياردەملەرنى بەردى،ھازىر بىمالال ئۇلارنىڭ ياردىمىسىزمۇ خىزمەت قىلالايمەن،__نىمىشقىدۇر ياخشى،كۆيۈنۈپ،دىگەن گەپلەرنى قىلىۋاتقان ۋاقتىمدا خۇددى تاغامغا دولقۇننى ماختاپ بىرىۋاتقاندەك تۇيغۇدا بولۇپ قالدىمدە ئۆزۈمنىڭ قىلغان گەپلىرىمدىن ئۆزۈم خىجىل بولۇپ قالدىم. تاغام بولسا بۇ خىجىللىقىمنى سەزمىگەندەك گىپىنى داۋاماشتۇرماقتا ئىدى:

__ئالدىرماڭ قىزىم، سىز ئۈگۈندىغان بىلىملەر تىخى نۇرغۇن،ھەرگىز قانائەت قىلىپ توختاپ قالماڭ،ئالىي مەكتەپتە ئۈگەنگەنلىرىڭىز بىلەن ئەمەلىيەتتىكى خىزمەت مەشغۇلاتىدا كۆپ پەرق بولىدۇ، بۇرۇن نەزىرىيە ئۈگەنگەن بولسىڭىز ئەمدى شۇ ئۈگەنگەنلىرىڭىزنى ئەمەلىيەتتە باغلاپ تۇرۇپ خىزمەت قىلىش كىرەك،شۇندىلا ئۆزىڭىزنىڭ بىلىمىنى پۇختا ئىگەللىگەن بولىسىز.

تاغامنىڭ سۆزىدىن بايام ئۆزۈمنىڭ ‹‹ئۇلارنىڭ ياردىمىسىزمۇ›› دىگەن گەپنى ئوشۇق قىلىپ سالغانلىقىمنى ھىس قىلىپ قالدىم. ۋاي خۇدايىم،مىنىڭ مۇشۇ ئاغزىم-زە...
مىھرىبان ئاپامغا ياردەملىشىپ بىرەي دىگەننى بانا قىلىپ غىپپىدە تاغامنىڭ يىنىدىن ئايرىلدىم-دە ئاشخانا ئۆيگە كىرىۋالدىم.مىھرىبان ئاپام تۈجەبىلەپ بايا دۈملەپ قويغان پولۇغا سوغۇق سەي راسلاۋاتاتتى. مىنىڭ كىرگىنىمنى كۆرۈپ دەرھال ئىغىز ئاچتى:
__نازى بالام،مىڭىڭ،بىرىپ تاغىڭىز بىلەن،ئەركىن بىلەن پاراڭلىشىپ ئولتۇرۇڭ،سىز دىگەن مىھمان__دىگىنىچە سىرتتقا ئىتتىردى.
__مىھرىبان ئاپا، ياردەمىشەيچۇ؟  ھەرقىتىم كەلسەم مۇشۇنداق تەكەللۇپ قىلىۋالدىكەنسىز، خىجىل بولىدىكەنمەن،مىنى ئۆز كۆرمەمسىز-يا؟

مىھرىبان ئاپام گىپىمنى ئاڭلاپ كۈلۈپ كەتتى:
__ماۋۇ قىزنىڭ گىپىنى كۆرمەمدىغان، ئىشلەپ چارچاپ كەتتى،ئىشقا سالماي دىسەم قىلغان گىپىنى، ماقۇل ئەمىسە يىنىمدا تۇرۇپ قىلغانلىرىمنى كۆرۈڭ-ھە؟
تالىشىپ يۇرۇپ قىلغۇدەك ئىش بولسا ئۆزۈم قىلىۋەتتىم.
تاغام،مىھرىبان ئاپاملار ھەقىقەتەنمۇ ماڭا بەك كۆيۈنىدۇ،بۇلارنىڭ قىزى بولمىغاچقىمىكىن،مىنى ئۆز قىزىدەكلا كۆرەتتى. مۇشۇ قىتىم خىزمەت ئىشىغىمۇ تاغام خۇددى ئۆز بالىسىنىڭ ئىشىغا چاپقاندەك ياردەم قىلىپ كەتتى، شۇڭىمۇ ئاپاملار مەندىن خاتىرجەم ئىدى. پات-پات قايتىپ كىلىپ يىنىمىزدا خىزمەت قىلساڭ بولمامدۇ،دەپ قويىدىغىنىنى ھىسابقا ئالمىغاندا مەندىن ئانچە ئەنسىرەپ كەتمەيتتى.

تائاملار بىرەر قۇر يىيىلىپ بولغاندىن چىنە-قۇچىلارنى يۇيۇپ رەتلەۋىتىپ قۇنۇپ قىلىڭ،دەپ تۇتىۋالغىنىغا قارىماي ئۆيگە قايتماقچى بولدۇم، ئەركىن ئۇنىمىغىنىمغا قارىماي ئاپىرىپ قويىمەن دەپ تۇرىۋالدى، ئىككىمىزنىڭ يىشى ئۈچ ياش پەرقلەنسىمۇ(ئۇ مەندىن3ياش چوڭ ئىدى) ھازىرقى قىلىقى بەئەينى كىچىك بالىنىڭ ئۆزى ئىدى.
تاغام:
__بوپتۇ نازى بالام، ئەركىن ئاپىرىپ قويسۇن، بىخەتەر بوىدۇ،__دىدى تاغام،ئامالسىز بوپتۇ دىدىم.
ئۆيدىن چىقىپ كەتكەچ تاغام مىھرىبان ئاپاملارنىڭ كۆزىدىن ئىككىمىزغا مەسلىكى كەلگەندەك قاراپ تۇرىشىدىن بىرخىل شۈبىھلىنىپ قالدىم.

ئۆينىڭ ئالدىغا كەلگۈچە ئەركىن توختىماي گەپ قىلىپ كەلدى،ھىلى بۇنى دىسە ھىلى ئۇنى دەيتتى،بولۇپمۇ گەپنىڭ مەركىزى مىنىڭ ئۆزگىرىپ بۇرۇنقىغا ئوخشىماي قالغانلىقىمنى، ئۈرۈمچىدە قىلىپ ناھايتتى ياخشى بولغانلىقىنى،بۇنىڭدىن كىيىن داۋاملىق كۆرۈشۈپ تۇرالايدىغانلىقىمىز...دىگەندەك گەپلەر ئىدى.

خۇشلىشىپ ئۆيگە كىرىپ كەتتىم، كۆيۈمچان نەۋرە ئاكام،مىھرىبان تاغا-ھاممىلىرىم...مەن نىمىدىگەن بەخىتلىك!

ئۇچۇر قايتۇرمىدىم، دەرس پىلانىمنى تۈزۈپ بولۇپ ،ئىشلار بىرەر قۇر تۈگۈتۈپ سائەتكە قارىسام كەچ سائەت8نى كۆرسىتىپ تۇرۇپتۇ، بىردەم تورغا چىقىپ خەۋەر كۆردۈم، تىلفۇنۇمنىڭ سايرىغان ئاۋازىنى ئاڭلاپ ئىكرانغا قارىۋىدىم،دولقۇن تىلفۇن ئۇرۇپتۇ، تىلفۇننى ئالدىم:
__ئەسسالاممۇ ئەلەيكۇم،تىنىچلىقمۇ؟
__ۋەئەلەيكۇم ئەسسالام نازى،ياخشى تۇرۇپسىز.
__ياخشى دولقۇن مۇئەللىم.

__ماقۇل دولقۇن مۇئەللىم،ياق دولقۇن.
تىلفۇندا ئىككىلىمىزنىڭ كۈلگەن ئاۋازى بىر-بىرىمىزگە ئاڭلاندى.

__ھە،مەنمۇ بۈگۈن بىركۈن مۇشۇ ئىشلار بىلەن مەشغۇل بولدۇم،سىزنىڭ ئەھۋالىڭىزنى سوراي دەپ تىلفۇن ئۇرغان.

ئۆزۈمنىڭ دولقۇن بىلەن تۇنجى قىتىم  خىزمەت ئىشلىرىدىن تاشقىرى ئۆزئارا پاراڭلشىپ تۇرۇپمۇ خۇددى بۇرۇندىن تارتىپ مۇشۇنداق پاراڭلىشىپ كىلىۋاتقان ئادەمدەك تەبىئىيلىك ھىس قىلغىنىمدىن ھەيران قالدىم. بولدىلا قالغان گەپلەرنى ئەتە پاراڭلاشساقمۇ ئوخشاشقۇ...
__ھە،رەھمەت سىزگە،ئەمىسە ئەتە كۆرۈشەيلى.
__ھە ئالدىراش ئىشىڭىز بارمىتى؟ ئەمىسە دەخلى قىلماي ،__ئۆزۈمنىڭ پاراڭلىشىشنى رەت قىلىۋەتكەنلىكىمنى ھىس قىلدىم،شۇندىمۇ خۇش بولدۇم:

ئوھۇي....ئۇيقۇمنى ئۇخلاي،نىمە بولسا ئەتە بولسۇن،ئەتىكى ئىشلارغا بۈگۈن زىھنىمنى چىچىپ يۈرمەي.

يەنە شۇ ئۆلگۈر قوڭغۇراق ئاۋازى...ئۇيقۇمنى بۇزۇپ...ئادىتىم بويىچە كۆزۈمنى ئاچماستىن ياستۇقۇم ئاستىدىن تىلفۇنۇمنى سىيپاشتۇردۇم-دە خالىغان كۇنۇپكىلىرىنى باستىم،ئاۋاز ئۆچتى،ئۇيقۇمنى داۋاملاشتۇردۇم،قەيەردىندۇر غۇۋا:‹‹ۋەي...،نازى...،گەپ قىلىڭە›› دىگەندەك ئاۋاز ئاڭلىنىۋاتاتتى، قۇلاقلىرىم بۇ ئاۋازنىڭ چۈش كۆرگەندىكى ئاۋاز ئەمەسلىكىنى ئىسپاتلاپ تۇراتتى.

ئاپلا!قارىغاندا تىلفۇن ئۇلىنىپ كىتىپتۇ-دە؟!! كۆزلىرىمنى دەرھال يوغان ئىچىپ قۇلۇمغا تىلفۇننى ئالدىم،
دەرۋەقە بايامقى ئاۋاز دولقۇن تىلفۇن قىلغان ۋاقىتتىكى مۇزىكا بولۇپ مەن قوڭغۇراق ئاۋازى دەپ خاتا تۇنۇپ قىلىپ بىسىۋەتتىم،دەپ ئويلىغىنىمدا دەل تىلفۇننىڭ يىشىل كۇنۇپكىسىنى بىسىپتىكەنمەن. نىمە سەتچىلىك بۇ!

__ۋەي!
__ۋەي نازى، تىلفۇننى ئىلىپ گەپمۇ قىلمايسىزغۇ؟ سىزگە بىرەر ئىش بولمىغاندۇ-ھە؟
نىمە دەي؟ تىلفۇننى ئىلىپ قۇيۇپ ئۇخلاۋاتقان ئىدىم،دەمدىمەن؟
__بايا بىسىلىپ كتىپتۇ،ئاڭلىماي قاپتىمەن.
نىمە ئىش بار ئىدى،دىگەندەك ئۇنىڭ گەپ قىلىشىنى ساقلاپ تۇرۇدۇم.
__ھە ئەسلى مۇنداق ئىكەندە، سىزنى ئالغىلى باراي دىگەن،مىڭىشتىن بۇرۇن خەۋەرلەندۇرۇپ قوياي دەپ تىلفۇن ئۇرغان.
__مۇشۇنداق سەھەردىما؟
__ھا..ھا....__تىلفۇندىن كۈلگەن ئاۋاز ئاڭلاندى،__نازى ،ھازىر سائەت10دىن ئاشتى، تىخىچە ئۇخلاۋاتىپسىزدە؟ ئەمىسە سىز دەڭە، قاچان ئالغىلى باراي؟

تامدىكى سائەتكە قارىدىم،دەرۋەقە سائەت چۈشتىن بۇرۇن 10دىن25مىنۇتنى كۆرسىتىپ تۇراتتى، ئۆزۈمنىڭ مۇشۇ ۋاقىتقىچە ئۇخلىغىنىمنى بىلىپ دولقۇننىڭ ئالدىدا بەكمۇ خىجىللىق ھىس قىلدىم.
__ھە....ئەمىسە مەن سىزگە بىر سائەتتىن كىيىن تىلفۇن قىلاي،شۇ ۋاقىتتا كىلىڭ ئەمىسە.

شۇنداق دىيشىپ تىلفۇنى قويىۋەتتىم،مۇشۇنداقمۇ ئۇخلىغان بارمۇ؟ ھەقانچە دەم ئىلىش بولسىمۇ 10يىرىم بولغۇچە ئۇخلىغىنىمنى ئويلىسام ئۆز-ئۆزۈمگە كايىغۇم كىلەتتى،تىخى دولقۇننىڭ بىلىپ قالغىنىچۇ...

ئالدراش يۇزلىرىمنى يۇيۇپ يارىشىملىق كىيىملىرىمنى كىيىپ ئادىتىم بويىچە سۇسقىنە گىرىمنى قىلىپ تەق بولدۇم. مۇشۇنچىلىك ئىشلارنى قىلىپ بولغۇچىمۇ بىر سائەت ۋاقىت كەتكەن ئىدى، ھازىر ۋاقىتنىڭ پەقەتلا بەرىكىتى يوق ئىدى،بولۇپمۇ مەندەك بۈگۈن 10يىرىمغىچە ئۇخلىغانلارغا...

دولقۇن ئىككىمىز كۆرۈشكەندىن كىيىن يىڭىدىن تىجارەت باشلىغان باغۋەن تاماقخانىسىغا كىردۇق، تاماقلارنى بۇيرىتىۋەتكەندىن كىيىن ئىككىلىمىز نىمە دىيىشىمىزنى بىلەلمىدۇق، دولقۇننىڭ بىرنىمىلەرنى دىگۈسى كىلەتتىيۇ، دىيەلمەيتتى. ئۆزۈم نەچچە ۋاقىتتىن بىرى مۇشۇنداق بىرىنىڭ ھەمرا بولۇپ
بىرگە تاماق يەپ،بىرگە ئايلىنىپ يۇرۇشنى ئۈمۈد قىلغان بولساممۇ ،بۈگۈن بىرخىل كۆنەلمىگەندەك ھىسياتتا بولماقتا ئىدىم.

تاماق چىققۇچە ئۇنى-بۇنى دەپ يەنە خىزمەتنىڭ گىپىنى قىلىشىپ ئولتۇرغان بولدۇق، بىر چاغدا دولقۇن:
__نازى،ھەدىسىلا خىزمەتنىڭ گىپىنى قىلىۋەرمەيلى،ھىلىمۇ ئىككى ئايدىن بۇيان خىزمەتتىن باشقا گەپ قىلىشىپ باقماپتۇق، بۈگۈن ئىككىلىمىز خىزمەتداش ئەمەس، ئىككىلىمىز ئوقۇتقۇچى ئەمەس، مەنمۇ يىتەكچى ئەمەس، قانداق؟__دىدى دوستانە رەۋىشتە.

مەن كۈلۈپ كەتتىم،دىمىسىمۇ بۈگۈن تۇنجى قىتىم ئايرىم تاماق يىگىلى كىلىپمۇ ئىككى ئايدىن بۇيان كۈنۈپ قالغان ئادىتىمىز بويىچە مۇشۇ يەردىمۇ ئوقۇغۇچىلارنىڭ،ئوقۇتقۇچىلارنىڭ ،مەكتەپنىڭ گىپىنى قىلىشىپ ئولتۇرغان ئىدۇق.

__ماقۇل،دەڭە ئەمىسە، نىمە دىيىشىمىز؟
__ئۆزلىرىمىزنىڭ گىپىنى قىلىشايلى، مەسىلەن، بۇنىڭدىن كىيىنكى تۇرمۇشىمىزنى قانداق ئۆتكۈزۈش،قانداق ياشاش، تۇرمۇشنى قانداق ئورۇنلاشتۇرۇش،يەنە،يەنە بۇنىڭدىن كىيىنكى دەم ئىلىشلاردا نەلەردە تاماق يىيىش،...دىگەندەك...ھە راست نازى، سىزنىڭ ئوغۇل دوستىڭىز بارمۇ؟

ما گەپنى، ئەسلىدە بۈگۈن مەن ئاتايىن دولقۇندىن مۇشۇ سۇئالغا جاۋاب ئالماقچى بولغان ئىدىم،تىخى ئەگەر بار دەپ قالسا بۇنىڭدىن كىيىن بۇنداق ئايرىم تاماق يىسەك بولمايدىغانلىقىنىمۇ ئايرىم ئەسكەرتىپ قۇيۇشنى ئويلىغان، كىم بىلسۇن دولقۇننىڭ مەندىن چاققان چىقىپ بۇ سۇئالنى ئاۋال سورىۋالدىغىنىنى.

__ياق، مەن تىخى ئۇ ئىشلارنى ئويلاشقۇدەك بولمىدىم، خىزمەت ئىشىم ئەمدى ھەل بولدى،بۇنىڭدىن كىيىنكى ئىشلارنى ئەمدى ئويلىشارمەن.

__سىزنىڭچۇ؟ قىز دوستىڭىز بارمۇ؟

دولقۇن ئالدىراپ جاۋاب بەرمىدى، پىيالىدىكى قىززىق ھەم قىنىق دەملەنگەن بۆلجۈگەن تەملىك چاينى بىر ئوتلىدى،كەينىدىن ماڭا تىكىلىپ قارىدى، بۇ قارىشى شۇنچىلىك مەنىلىك ئىدىكى داۋاملىق قاراپ تۇرىۋىرىشقا جۈرەت قىلالمىدىم.

ئانچە خالاپ كەتمىسەممۇ ئەدەپ يۇزىسىدىن ماقۇل بولدۇم.
__بولىدۇ ئەمىسە،ئەتە سىنى ساقلايمەن.

ئۇنى-بۇنى دىيىشىپ تىفۇننى قۇيىۋەتتىم. پۈتۈن ئىسى-يادىم دولقۇندا قالغان ئىدى. باشتا رەنجىگەن بولسام ئەمدى ئەنسىرەشكە باشلىدىم. ئۇنىڭغا بىرەر ئىش بولۇپ قالمغاندۇ-ھە؟ ئاپام ئاغرىق دەۋاتاتتى، ئاپىسىغا بىرەر ئىش بولۇپ قالغانمىدۇ-يا؟

يەنە بىر تىلفۇن ئۇرۇپ بىقىشنى ئويلاشتىم. ئۇرىۋىدىم، ئالدىراش بولۇپ چىقتى،دىمەك تىلفۇن ئىچىلغان. خۇش بولۇپ يەنە قايتا ئۇردۇم. بۇ قىتىم تىلفۇن ئۇلاندى. نىمىشقىدۇر خىلى ئۇزۇندىن كىيىن دولقۇن تىلفۇننى ئالدى.
__ۋەي نازى،ياخشىمۇ؟
ئاۋازىن ئاڭلاپ ئالدىن تەييارلاپ قۇيغان گەپلىرىم ئاللىقاياقلارغىدۇر ئۇچۇپ كەتتى.
__ياخشى،باركەنسىز-ھە؟
__خاپا بولماڭ،جىددىي ئىچىم چىقىپ قىلىپ سىزنى خەۋەرلەندۇرۇپمۇ قۇيالمىدىم،قالغان گەپنى ئەتە ئىشخانىغا بارغاندا دىيىشەيلى،ئاپامنىڭ مىجەزى يوق، دوختۇرخانىدا ئاپامغا قاراۋاتقان.

ئويلىغىنىمدەك چىقتى. مەنمۇ ئوشۇق گەپ قىلىشنىڭ ئورنى يوقلىقىنى ھىس قىلدىم،ھەم خاتىرجەملىك ھىس قىلدىم.
__بوپتۇ ئەمىسە،ئاپىڭىزغا ياخشى قاراڭ، ئۆزىڭىزنىمۇ ياخشى ئارام ئىلىڭ، ئەتە كۆرىشەيلى.
__سىزمۇ ھەم شۇ نازى،سىزنى كۆرگۈم كەلدى.
__مىنىڭمۇ ھەم شۇ،ئەمىسە شۇنداق بولسۇن،خۇش،ئاللاغا ئامانەت.
دولقۇننى خىيال قىلغىنىمچە قاچانلاردا ئۇخلاپ قالغىنىمنى سەزمەپتىمەن.

ئەتىسى ئىشخانىغا بارسام،دولقۇن مەندىن بۇرۇن كىلىپ بوپتۇ،كۆزلىرى سەل ئولتۇرۇشقان بولۇپ چارچىغانلىقى مانامەن دەپ چىقىپلا تۇراتتى،كۈلۈمسىرىشىمۇ شۇنچە جانسىز. تۇرقىغا قاراپ ئىچىم سىرىلىپ كەتتى.
__دولقۇن ئاپىڭىزنىڭ ئەھۋالى قانداقراق؟
دولقۇننىڭ چىرايىدىن سەل ئۈمۈدسىزلىككە ئوخشايدىغان،لىكىن چاندۇرغۇسى يۇقتەك بىرخىل رەۋىشتە جاۋاب بەردى.
__يامان ئەمەس،خۇداغا شۈكرى.
مەنمۇ ئوشۇق گەپ قىلمىدىم،يەنە گەپ قىلسام ئىچىشقان يىرىگە تۇز سەپكەندەك ئىش قىلغان بولاتتىم.
__ھە،ياخشى بولۇپ قالغان بولسىلا بولدى، ئىشتىن چۈشۈپ ئاپىڭىزنى يوقلاپ كىلەيمۇ؟

دولقۇن چوقۇم خۇش بولۇپ ماقۇل دەيدۇ،دەپ ئويلىدىم. ئەپسۇس،دولقۇن دەرھاللا رەت قىلدى.
__بولدى،كىينچە بىرگەپ بولسۇن، ئاپامنىڭ ھازىر كۆپ ئادەم بىلەن كۆرۈشكىدەك ھالى يوق.كىيىن ئاپام ياخشىلىنىپ قالغاندا ئۆزۈم سىزنى ئالدىغا ئاپىرىمەن.

تۇنجى قىتىملىق ئاكتىپلىق بىلەن كۆرۈشۈش تەكلىپىنى قۇيۇپ رەت قىلىنىشقا ئۇچرىغىنىمدىن ئۇۋالچىلىق ھىس قىلغان بولساممۇ ھازىرقى ئەھۋالىنى بىلىپ چاندۇرمىدىم.
__مەيلى ئەمىسە.

دولقۇننىڭ ئوشۇق گەپ قىلغۇسى يوقتەك قىلاتتى. مەنمۇ كۆڭلى خاتىرجەم بولمىغاندىن كىيىن ئارامىدا قوياي دەپ ئويلىدىم-دە ئۆز ئىشىم بىلەن مەشغۇل بولدۇم.
بىر چاغدا سىرتتىن بىرنىڭ ئىشىكنى يىنىك چەككەنلىكىنى ئاڭلاپ بىرىپ ئىشىكنى ئاچتىم، قارىسام ئەركىن ئىكەن.
__قانداق،تىپىپ كىلىپتىمەنمۇ؟__ئەركىن خۇشخۇيلۇق بىلەن كۈلۈمسىرەپ ئىشخانىغا كىردى.
__بۇ يەرنى تىپىپ كىلەلمىسەڭ سىنىڭ چەتئەلدە تۇرۇپ ئوقۇغىنىڭغا ئىشەنمەيتتىمدە؟__ئەركىننى ئىشخانىغا باشلىغا كىرىسلوغا تىكلىپ قىلدىم،قىنى ئولتۇرە؟
__ياخشىمۇسىز؟__دولقۇن شۈبھە ئارلاش ئەركىنغا قاراپ ئەھۋاللاشتى.
__ياخشى تۇرۇپسىز؟__دىدى ئەركىنمۇ جاۋابەن. دولقۇننىڭ كىمۇ؟دىگەندەك قاراشلىرىغا قاراپ ئەركىننى تۇنۇشتۇرۇشقا باشلىدىم:
__بۇ نەۋرە ئاكام ئەركىن بولىدۇ،تاغامنىڭ ئوغلى.ئەركىن ئاكا،بۇ دولقۇن مۇئەللىم،مىنىڭ يىتەكچى ئوقۇتقۇچۇم.

__ھە،نازى گىپىڭىزنى داۋاملىق ماڭا قىلىپ بىرەتتى__دىدى دولقۇن كۈلۈمسىرەپ.
__ھە،نازىمۇ سىزنىڭ گىپىڭىزنى قىلىپ بەرگەن،__دىدى ئەركىن جاۋابەن.

لىكىن ئەركىن نىمىشقىدۇر بىرخىل ئىچىلىپ ئولتۇرالمىدى.دولقۇندىمۇ بۇ خىل ئەھۋال كۆرۈلگەندەك ھىس قىلدىم. بەلكىم خاتا سىزىمىم بولسا كىرەك.
قايناقسۇدىن بىر ئىستاكان ئەركىننىڭ ئالدىغا قويدۇم
__ئەركىن ئاكا،بىزنىڭ ئىشخانىدا قايناقسۇدىن باشقا ئىچىملىك يوق، مۇشۇنى ئىچكەچ تۇرغىن.
__ھىچقىسى يوق نازى، قانداق كۆنۈپمۇ قالغانسەن؟
__ھەئە،كۆنۈپمۇ قالدىم.
__بۈگۈن چۈشتىن بۇرۇن دەم ئىلىش ئىدى، سىنى يوقلاپ ئۆتەي دەپ كەلدىم، ھەراست مانى ساڭا ئالغاچ كەلدىم__دىگىنىچە قولىدىكى خالتىسىنى ماڭا تۇتقۇزۇپ قويدى.
__نىمە بۇ؟__ئىچىنى ئىچىۋىدىم،مەن ئەڭ ياقتۇرىدىغان شاكىلات ئىدى__رەھمەت ساڭا،ئاۋارە بولۇپ كىلىۋەرسەڭ بولمامتى؟
__سەن بۇرۇندىن ئامراق ئىدىڭغۇ؟ يىسۇن دەپ ئالغاچ كەلدىم.__دىدى سەمىمىيلىك بىلەن.
دولقۇنغا قارىۋىدىم،قاپاقلىرى نىمىشقىدۇر ساڭگىلاپ كىتىپتۇ. توۋا...ھىلىمۇ ئاكام دەپ تۇنۇشتۇرغىنىم.

ئەركىن بىردەم ئولتۇرۇپ قايتماقچى بولدى. نىمىشقىدۇر دولقۇنغا پات-پات قاراپ قۇياتتى.
__نازى،مەن ماڭاي، ئالدىمىزدىكى دەم ئىلىشتا ئاپام سەن ياقتۇرۇپ يەيدىغان پولۇ ئىتىپ قويماقچى،بار-ھە؟ بارمىساڭ ئۆيدىكىلەر خاپا بولۇپ قالىدۇ.
دولقۇننىڭ ئالدىدا جىنايەت ئۆتكۈزىۋاتقاندەك خىجىل بولۇپ كەتتىم:
__ھە ماقۇل ئەركىن ئاكا، بارىمەن.
__ئەمىسە ئۆزۈڭگە دىققەت قىل،كەچتە بالدۇرراق ئۆيگە قايتقىن__شۇنداق دىگىنىچە بىز بىلەن خۇشلىشىپ ماڭدى. دولقۇنمۇ ئەدەپ يۇزىسىدىن ئۇزۇتۇپ قويدى.

ئەركىن كەتكەندىن كىيىن دولقۇننىڭ چىرايى تۇتۇلۇپلا كەتتى.
__نەۋرە ئاكىڭىزما؟__دەپ سورىدى ئىشەنمىگەندەك قىلىپ.
__ھە؟ نەۋرە ئاكام؟ نىمە بولدى؟__دولقۇننىڭ ئىپادىسىدىن ھەيران قالدىم.ئادەتتە خىلى ئىش بولۇپ كەتسىمۇ ئىغىر-بىسىق يۇرۇيدىغان دولقۇن بۈگۈن چىدىماسلىق قىلىۋاتاتتى.

__ھىچنىمە بولمىدى، نەۋرە ئاكىڭىز سىزگە ئالاھىدە كۆيۈنىدىكەن،شۇ دەيمىنا؟
__چىدىماس، نەۋرە ئاكامدىنمۇ كۈنلەۋاتامسىز-نىمە؟__كۈلۈپ كەتتىم.
__ياقەي‹‹ئاكىڭىز››دىن كۈنىگىلى بولامتى نازى؟

دولقۇننىڭ تەلەپپۇزى ھىچ چاقچاق قىلغاندەك ئەمەس،تۇرۇپ ئاچچىقىم كەلدى،شۇندىمۇ دولقۇننىڭ بۈگۈن كەيپىياتىنىڭ ياخشى ئەمەسلىكىنى كۆزدە تۇتۇپ گەپ-سۆز قىلمىدىم.

دەم ئىلىش كۈنى دولقۇننىڭ تىلفۇنىنى ساقلىدىم،ھىچ تىلفۇن ئۇرىدىغاندەك ئەمەس، بۇرۇن بولسا ئاللىبۇرۇن سەھەردىلا تىلفۇن ئۇرۇپ مىنى ئويغىتىۋىتىدىغان. سائەت11بولدى،يەنە تىلفۇنى يوق، ئاچچىقىمنى ئىچىمگە يۇتۇپ تىلفۇن ئۇرۇدۇم.
__ۋەي دولقۇن ياخشىمۇ؟
__ياخشى نازى.
__نەدە سىز؟ تىلفۇن ئۇرمىدىڭىزغۇ؟
__دوختۇرخانىدا نازى،ئاپامغا قاراۋاتىمەن.
__ھە،شۇڭا تىلفۇن ئۇرماپتىكەنسىزدە؟
__ھەئە، سىزچۇ؟ نىمە قىلىۋاتىسىز؟
__مەن تىخى سىزنىڭ تىلفۇنىڭىزنى ساقلاپ ئولتۇرغان.
__بۈگۈنمۇ ئاپام سەل ئۆزگىرىپ قىلىپ دادام ئىككىمىز ئاپامغا قاراۋاتقان،__دولقۇننىڭ ئاۋازى تىترەپ چىقىۋاتاتتى.
__مەن بارايمۇ يە دولقۇن؟ ئاپىڭىز بىلەن كۆرۈشمىسەممۇ سىز بىلەن كۆرۈشۈپ كىلەي بولمىسا...
__مەيلى ئەمىسە.

شۇنداق قىلىپ دوختۇخانىغا قاراپ ماڭدىم. دوختۇرخانا ئالدىدىكى ئاشخانىدىن نارىن چۆپتىن ئۈچ كىشلىك ئىلىپ داباۋ قىلىۋالدىم.
تىلفۇن ئۇرسام دولقۇن كۆپ ئۆتمەيلا چىقتى، كۆزلىرى قىزىرىپ قالغاندەك بولۇپ يىغلىغاندەك قىلاتتى. نىمىشقىدۇر مىنىمۇ يىغا تۇتتى. چاندۇرمىدىم.
__مەڭ،ئۈچ كىشلىك نارىن ئىلىۋالدىم،ئىسسىقىدا يەۋىلىڭلار.
__نىمىمۇ قىلاتتىڭىز بۇنچە ئاۋارە بولۇپ،بايام تىخى تاماق يىگەن ئىدۇق.
__بولدى،تۈزۈتنى ئاز قىلىپ دەرھال يەۋىلىڭ،سۇۋۇپ قالسا يىگىلى بولمايدۇ.
قۇلۇمدىكى تاماقنى دولقۇنغا تۇتقۇزدۇم.

__مەيلى ئەمىسە، سىزچۇ؟بۈگۈن نىمە ئىش قىلىسىز؟

جاۋاب بىرىپ بولغۇچە تىلفۇنۇم سايرىدى. تىلفۇننى ئەركىن ئۇرغان ئىدى. قايسى كۈندىكى دولقۇنىڭ ئەركىننى كۆرۈپ كۈندەشلىق قىلغىنىنى ئويلاپ تىلفۇننى ئالماسلىقنى ئويلاشتىميۇ،تىخىمۇ خاتا چۈشەنمىسۇن،دەپ تىلفۇننى ئىلىۋەردىم:
__ۋەي ئەسسالاممۇ ئەلەيكۇم.
__ۋەئەلەيكۇم ئەسسالام، نازى،نەدە سەن؟
__مەن دوختۇرخانىنىڭ ئالدىدا.
__نىمە؟ ئاغرىپ قالدىڭما؟
__ياق،مەن ئاغرىپ قالمىدىم، بىر دوستۇمنىڭ ئاپىسى ئاغرىپ قاپتىكەن،يوقلىغىلى كەلگەن.
ئەركىن بىر پەس جىممىدە بولغاندىن كىيىن سۆزىنى داۋاملاشتۇردى:
__ئەمىسە قاچان كىلىسەن؟ ھەممىمىز سىنىڭ كىلىشىڭنى ساقلاپ ئولتۇردۇق، تاماقمۇ پىشىپ بولدى،تاكسىدىن بىرنى توسۇپ غاچچىدە كىلە.

تىلفۇننىڭ ئاۋازى سەل يۇقىرى بولغاچقا دولقۇن بىمالال ئاڭلىيالايتتى. دولقۇنغا قارىۋىدىم،دولقۇن ئىپادىسىز ماڭا قاراپ تۇراتتى. ھەئە ياكى ياق دىيىشىمنى بىلەلمەي سىكىنۇت ئىككىللىنىپ قالدىم.
__ۋەي گەپ قىلە؟ نىمانداق جىممىدە تۇرىسەن؟ بۈگۈنمۇ زۆرۈر ئىشىڭ بارمىدى-يە؟

__بىرىپ كىلىڭ نازى،باشقىلارنى ساقلاتماڭ__دولقۇن ئىپادىسىز كۈلۈمسىرىدى.
__ھە ماقۇل ئەمىسە.

تىلفۇننى قۇيىۋەتتىم.
__ئاپامنىڭ يىنىغا كىرىپ كىتەي،سىزمۇ مىڭىڭ ئەمىسە.
__مەيلى ئەمىسە،بىر ئىش بولسا تىلفۇن قىلارسىز.مەن ماڭاي.

دولقۇننىڭ جانسىز دوختۇرخانىغا مىڭىپ كىرىپ كىتىۋاتقانلىقىنى كۆرۈپ يۇرۇكۈم ئىچىشتى، مەنمۇ بىر بالىدە؟ دولقۇننىڭ ھازىرقى ھىسىياتىنى چۈشىنەتتىم. قىيمىغاندەك قاراپ قۇيۇپ تاكسىدىن بىرنى توسۇدۇم-دە تاغامنىڭكىگە قاراپ يۇرۇپ كەتتىم.

تاغامنىڭ ئۆيىگە كىرىشىمگە ئوخشىتىپ ئىتىلگەن پولۇنىڭ مىزىلىك پۇراقلىرى بۇرنۇمغا گۈپپىدە ئۇرۇلدى. مىھرىبان ئاپاملار كۈلۈمسىرەپ قارشى ئالدى.
__ئەركىن تىلفۇن ئۇرمىسا كەلمەيتىڭىز ھە قىزىم؟
__ياقەي،مەنمۇ ھازىر ماڭاي دەپ تۇرغان__خىجىل بولدۇم.
__نازىمۇ ماڭاي دەپ تۇرۇپتىكەن، مىنىڭ تىلفۇن ئۇرۇشۇپ توغرا كىلىپ قاپتۇ، ئۇرمىغان بولساممۇ كىلەتتىكەن،شۇنداققۇ نازى-ھە؟
ئەركىننىڭ مىنى ئاقلاپ دىگەن گىپىدىن تاغام ھوزۇرلىنىپ كۈلۈپ كەتتى:
__بالام، كۆردىڭىزمۇ ؟ يىقىندىن بۇيان ئەركىن سىزنى ئاقلاپلا تۇرىدىغۇ؟
__دادىسى، يوق گەپنى قىلمىڭە، نازى بالامنى خىجىل قىلىپ،قىنى بالام، يۇقۇرغا چىقىڭە...__مىھرىبان ئاپامنى گىپى بىلەن تاغاممۇ ئوشۇق گەپ قىلمىدى، ئەركىنمۇ ماڭا كۆزىنى قىسىپ قويدى.

تائاملار يىيىلىپ بولغاندىن كىيىن تاغام ئىشىم بار ئىدى،دەپ سىرتقا چىقىپ كەتتى. مىھرىبان ئاپام ئۇنىمىغىنىمغا قارىماي قازان-قۇمۇشلارنى ئۆزى يىغىشتۇردى، مەن يىنىدا پاراڭلاشقاچ تۇراتتىم.
__بالام، مەكتەپتە ئىشلىرىڭىز ياخشىدۇ-ھە؟__ دىدى مىھرىبان ئاپام قاچا يۇيغاچ.
__ھەئە،خۇداغا شۈكرى ،يامان ئەمەس.
__ھە، كۆنۈپ قالغان بولسىڭىزلا بولدى__مىھرىبان ئاپام كۈلۈمسىرەپ__قىزىم، ئۆيگە ئاز كىلىدىغان بولىۋالدىڭىز، خىزمىتىڭىز بەك ئالدىراشمۇ؟

__نازى، بىر ئوبدان ئىشلەۋاتاتتىڭ،نىمىشقا كىتىسەن؟

ئەركىنگە كۈلۈمسىرەپ قارىدىم، ئۇنىڭ ھازىرقى ھىسىياتىنى چۈشىنەتتىم.
__كىتەي،ئاپاملارنى كۆگۈم كەلدى،شۇلارنىڭ يىنىدا تۇرۇپ ئازراق بولسمۇخىزمىتىنى قىلىۋالاي.

__خۇدايىم بۇيرىسا يەنە كىلىسىز قىزىم، بۇمۇ سىزنىڭ ئۆيىڭىز تۇرسا ،شۇنداقمۇ بالام؟__مىھرىبان ئاپام كۆزلىرىنى ماڭا تىكتى.
__ھەئە ،خۇدايىم بۇيرىسا ئەلۋەتتە كىلىمەن.

ئەركىن بىشىنى سۇس چايقاپ قويدى.
__ئەمىسە نازى قىزىم، قارىڭىڭىزنى ئۆزگەرتمىسىڭىز،ئۇنداقتا مەن سىزنى تۇرپاندىكى مەكتەپكە يۆتكىلىپ كىتىشىڭىزنىڭ ئامالىنى قىلىپ بىرەي.
__رەھمەت تاغا،ھىلىمۇ سىزنى كۆپ ئاۋارە قىلىۋەتتىم،ئالدىڭىزدا بەك خىجىلمەن.

__ياقەي قىزىم، ئۇنداق دىمەڭ،سىزمۇ ئۆزۈمنىڭ بالىسى تۇرسىڭىز...

تاغاملارغا يەنە بىرقىتىم ئادەمگەرچىلىك قەرز بولدۇم،گەرچە بۇلار ئۇنداق ئويلىمىسىمۇ ئۆزۈم ئىچىمدە خۇدايىم بۇيرىسا كىيىن مۇشۇ قىلغانلىرىنى چوقۇم ھەسسىلىپ قايتۇرىمەن،دىگەننى كۆڭلۈمگە پۈكۈپ قويدۇم.

يۆتكىلىش ئىشىممۇ ھەل بولدى، بىكەتتە چامادانلىرىمنى كۆتۈرگىنىچە ئەركىن ماڭا ھەمرا بولۇپ مىڭىۋاتاتتى. بىرقەدەم-ئىككى قەدەم!
ئۈرۈمچىدىن ئەمدى ئايرىلىپ ئۆز يۇرتۇمغا قايتىدىغان بولدۇم، بۇ ھىسىياتىم بۇرۇنقىدەك قىيالماسلىق ئەمەس بەلكى ئۆزۈمنى بۇ يەردىن كىتىپ ناھايتتى توغرا قىلغاندەك ھىسىياتتا ئىدىم.
ماشىنا سىگىنال بىرىشكە باشلىدى، ئەركىن چامادانلىرىمنى ماشىنىغا ئورۇنلاشتۇرۇشقا باشلىدى.
ئەركىننىڭ ھازىرقى ھالىتى ماڭا ناھايتتى ئايان ئىدى.

__نازى،كىتىپلا يوقاپ كەتمەي يەنە كىلىپ تۇر-ھە؟ئارلىقتا پۇرسەت بولسىلا سىنى كۆرگىلى بارىمەن، كۆپ ئويلىما-ھە؟
__نىمىنى كۆپ ئويلىما دەيسەن ئەركىن ئاكا؟
__مىنى بىلمەيدۇ-دەپ قالما، ھىلىقى دولقۇن دىگەننىڭ كاساپىتىدىن كەتمەكچى بولىۋاتىسەنغۇ شۇ،ئۇمۇ ياخشى بولغىنى...نىمىلا دىمىگەن بىلەن بىر بىلىمگە ئىرىشتىڭ__ئەركىن ناھايتتى تەمكىنلىك بىلەن بۇرۇنقىدەكلا سەمىمىي گەپ قىلىۋاتاتتى. دولقۇن ئىسىمنىڭ چىقىشى بىلەنلا دولقۇنغا قەيەردىندۇر بىسىلىپ قالغان نەپرىتىم قوزغىلىپ قالدى.

شۇندىمۇ ھازىر ئەڭ ياخشى ھالىتىمنى ساقلىمىسام بولمايتتى. ماشىنا قوزغالدى،ئەركىن قولىنى پۇلاڭلىتىپ خۇش دىگەندەك ماشىنىنىڭ كىتىشىنى ساقلىدى. ماشىنا خىلى ئۇزىغان بولسىمۇ يىراقتىن ئەركىننىڭ سىماسىنى ئىلغا قىلغىلى بولاتتى، بارا-بارا ئەركىننىڭ سىماسى بىر چىكىتكە ئايلاندى،دە ئاخىرى كۆرگىلى بولمىدى.

‹‹كەچۇر ئەركىن، كۆڭلۈڭنى چۈشەنسەممۇ بىلمىگەنگە سالدىم،مىنى بىر سىڭلىڭ قاتارىدا ئەسلىسەڭلا بولدى...دولقۇن،ساڭىمۇ رەھمەت، ماڭا ياخشى بىر ساۋاقنى بەردىڭ، بۇنىڭدىن كىيىن بىرىگە ئالدىراپ كۆڭۈل بەرمەسلىكنى ئۈگەتتىڭ،ئەڭ مۇھىمى ئاتا-ئانامنىڭ يىنىغا قايتىشىمغا زور كۈچ چىقاردىڭ، سەن تۇرمۇشۇمدا پەيدا بولۇپ،مەندىن ئايرىلىشنى تەلەپ قىلمىغان بولساڭ ئىدى، ئۈرۈمچىدە يەنە قاچانغىچە تۇرۇشۇمنى بىلمەي يۇرەتتىم، بەلكىم مەڭگۈ قايتماسلىقىممۇ مۈمكىن ئىدى....››



ھەش-پەش دىگۈچە ئۆز يۇرتۇمغا كىلىپ ئىشقا چۈشكىنىمگىمۇ يىرىم يىل بولاي دىدى. ۋاقىت نىمە دىگەن تىز ئۆتكەن-ھە؟ ۋاقىت ھەقىقەتەنمۇ ھەممىنى ئۇنتۇلدۇرىدىكەن، ئۆز ۋاقتىدا مەڭگۈلۈك دەپ ئۆزىمىزگە ھەم ئۆزگىگە بەرگەن ۋەدىلەرمۇ ۋاقىتنىڭ ئۆتىشىگە ئەگىشىپ بىردەملىككە ئايلىنىدىكەن.

ئاپامنىڭ ئىشتىن چۈشۈپ تۇرۇشۇمغا ئۈلگۈرتۈپ ئەتكەن مىزىلىك تائاملىرى، ھەدەمنىڭ ئۇنىمىغىنىمغا قارىماي ئۆيىگە زورلاپ ئاپىرىپ قوندۇرۇۋىلىشلىرى...دەم ئىلىش ۋاقىتلىرىمدا يىزىدىكى تۇققانلارنىڭ ئۆيلىرىگە بىرىپ ناتۇنۇش بولۇپ كىتەي دەپ قالغان سەھرا مەنزىرلىرىدىن قانغۇچە زوق ئىلىشلىرىم....

ئىشقىلىپ ئۈرۈمچىنىڭ ھەرتەرەپتىن كىلىدىغان بىسىم ئىچىدە ياشاپمۇ ۋايسىماي شۇ يەردە نەچچە يىلدىن بۇيان ياشاپ كىلىۋاتقانلىقىمغا ھەيران بولدۇم.
ئۆز يۇرتىغا نىمە يىتىدۇ دەيسىز؟ كۈنلىرىم مانا مۇشۇنداق مەنىلىك ئۆتمەكتە.ئەلۋەتتە بۇنىڭغا ئەگىشىپ دولقۇننىڭ سىماسىمۇ ماڭا بارا-بارا يات تۇيۇلۇپ،ئەسلىمىلىرىم خاتىرىسىدىن يۇيۇلۇپ ماڭماقتا ئىدى.

ئىشتىن چۈشەي دەپ تۇرىشىمغا ھەدەم كەچتە ئۆزىنىڭ ئۆيىگە كىلىشىمنى ئىيتتى. سائىتىمغا قارىدىم ، ۋاقىتمۇ بىر يەرگە بىرىپ قاپتۇ.
ھەدەمنىڭ ئۆيىگە كىلىپ ئىشىكنى چەكسەم ئىشىكنى يىشى 30~35ياشلار چامىسىدىكى بىر ئايال ئاچتى ۋەز قىزغىنلىق بىلەن كۆرۈشۈپ كەتتى.

__ۋاي-ۋۇي ياخشىمۇسىز ئۇكام، قانداقراق ئەھۋالىڭىز؟ بايام تىخى سىزنىڭ گىپىڭىزنى قىلىشىپ ئولتۇرغان، قىنى قىنى، ھىرىپ كەتكەنسىز، يۇقۇرىغا چىقىڭە؟...ئاداش، سىڭلىڭ كەلدى!!__دەپ توۋلىدى ئاشخانا ئۆيگە قاراپ.

بۇ ئاچاشنىڭ قىزغىنلىقىدىن ئۆزۈمنى ھەدەمنىڭ ئۆيىگە ئەمەس بەلكى مۇنۇ ئالدىمدىكى ئاچاشنىڭ ئۆيىگە مىھمان بولۇپ كەلگەندەك ھىس قىلىپ قالدىم، ھەم بۇ خىل قىزغىنلىققا يىتىرقىغان ھالدا جاۋاب قايتۇردۇم:
__ھە،ياخشىمۇسىز ھەدە!

قىسقىغىنە كۆرۈشۈشتىن كىيىن ھەدەمنىڭ يىنىغا چىقتىم ،قارىسام ھەدەم پىشىپ بولغان چۆپلەرنى تەخسىلەرگە بۆلىۋاتقان ئىكەن، مىنىڭ كىرگىنىمنى كۆرۈپ كۈلۈپ قويدى:
__چاقىرمىغان بولسام كەلمەيتتىڭ-ھە ئۇكام!

مەنمۇ كۈلۈپ قويۇپ ھەدەمگە ياردەملەشكەچ ھەدەمدىن :
__ئاۋۇ ئاچاش كىم بولىدۇ؟__دەپ سورىدىم.
ھەدەم ماڭا قاراپمۇ قويماستىىن ئىشىنى داۋاملاشتۇردى:
__مىنىڭ يىقىن دوستۇم، ئۆزى مۇشۇ يەردىن، خىلى يىللار بولغان ئۈرۈمچىگە كىتىپ كۆرۈشەلمىگىنى، يىقىندا بۇ يەردىكى تۇققانلارنى يوقلىغىلى كەپتىكەن، شۇ قاتارىدا مەن بىلەن كۆرۈشكىلى كەپتۇ.

تاماقلارنى مىھمانخانا ئۆيگە تۇشۇشقا باشلىدىم، ھەدەمنىڭ بايىقى دوستى بولسا ئالدىمغا يۈگۈرۈپ كەلدى :
__ۋاي ئەكىلىڭە سىڭلىم مەن ئالاي،سىزمۇ ئىشلەپ چارچىغانسىز__شۇنداق دىگىنىچە قولۇمدىكى لەغمەننى ئالدى.

نىمانچە قىزغىن ئاكتىپ ئاچاش بولغۇيتتى-ھە؟ مەنمۇ ئانچە تۈزۈت قىلىپ كەتمىدىم. ئولتۇرۇپ تىلۋۇزۇر كۆرگەچ تاماق يىيىشكە باشلىدۇق.  
تىلۋۇزۇر كۆرگەچ كۆزۈمنىڭ قىرىدا ئۇ ئاچاشقا قاراپ سىلىۋىدىم، كۆزۈم مىنىڭ ئۈستۈمدىن توختىماي ئايلىنىۋاتقان ئۇ ئاچاشنىڭ كۆزى بىلەن ئۇچرىشىپ قالدى، ئىككىلىمىز بىربىرىمىزگە قاراپ كۈلۈمسىرەپ قويدۇق.

__ئاداش، بۇ سىڭلىڭنى مەن بۇرۇن كۆرمەپتىكەنمەن، بۇنداق چىرايلىق ئوماق سىڭلىڭنىڭ بارلىقىنى تىخى بىلمەيدىكەنمەن جۇمۇ؟__دىدى مەسلىكى كەلگەندەك ماڭا قاراپ.

ئۆزۈممۇ 24لەردىن ئىشىپ قالغان قىز بولۇپ قالغانلىقىم ئۈچۈنمۇ كىچىك قىزلارغا ماس كىلىدىغان بۇ خىل ماختاشتىن خىجىل بولۇپ قالدىم.

ئاچام يەۋاتقان تامىقىنى قۇيۇپ ماڭا مىھىر بىلەن قارىۋالدى،كەينىدىن دوستىغا:
__ئۆزىمۇ بەك ئوڭلۇق قىز بولدى بولمىسا، شۇ بىر يىرى گىپىمىزنى ئاڭلىمايدۇ.

يەۋاتقان تامىقىمنى قۇيۇپ ھەيران بولغىنىمچە ھەدەمگە تىكىلدىم!
__ھەدە،ماڭا ئۇۋال قىلما-ھە؟ مەن قانداق گىپىڭنى ئاڭلىماپتىمەن؟

__ھا..ھا...___ھەدەم ئىچىلىپ كۈلۈپ كەتتى___قارا تۇرقۇڭغا! باشقا گەپلىرىمىزنىغۇ بەك ئاڭلايسەن،شۇ تويۇڭنى قىلىۋال دىگەن گىپىمىزنى قىرىڭغا ئىلىپ قويمىدىڭ!!

ھەدەم بىلەن دوستى ئككىلىسى كۈلۈشۈپ كەتتى،ھەدەمنىڭ مەن تۇنىمايدىغان دوستى ئالدىدا مىنى خىجىل قىلىپ قويغىنىغا ئاچچىغىم كەلدى:
__ھەدە،كىيىن سەن بىلەن ئايرىم ھىسابلىشىمەن، دوستۇڭنىڭ يۇزى-ھە؟

ئۇلار تىخىچە كۈلىشىۋاتاتتى. دوستى بولسا ماڭا ھەۋەسلەنگەندەك قاراپ گىپىنى داۋام ئەتتى:
__ سەنمۇ قىززىق جۇمۇ ئاداش، نىمە بوپتۇ؟ ئالدىرماي قىلىدۇ شۇ توينى، خىزمەتكە چىققىلى ئەمدى بىريىل بولۇپتۇ، نىمە ئالدىراش دەيمەن؟ خۇدايىمنىڭ بىكىتىپ قويغىنى بار،ۋاقتى-سائىتى توشسا قىچىپمۇ قۇتۇلالمايدۇ،شۇنداقمۇ سىڭلىم-ھە؟

بۇ ئاچاشنىڭ تولىمۇ ئورۇنلۇق گەپ قىلغاندەك ھىس قىلىپ مىننەتدارلىقمنى بىلدۈرۈش يۇزىسىدىن توغرا دىگەندەك گىپىنى قوللاپ بىشىمنى لىڭشىتتىم...

مانا مۇنداق گەپ قىلمامدۇ دەيمەن؟ ھىلىمۇ كەلگەندىن بۇيان شۇ گەپ، ھىلى ئاپام ياندىن قىردىن گەپنى توي ئىشىغا بۇراپ گەپ قىلىۋاتقان، ھەدەممۇ ھەر كەلسەم مۇشۇ ئىشقا كۆڭلۈ بۆلۈپ قويۇشۇم توغرىسىدا نەسىھەت قىلىۋاتقان...قىز چوڭ بولسا نىمانداق تەستۇ؟ يا چىقىپ بىرسىنىڭ قولىنى تۇتۇپلا مانىڭغا تىگىمەن دىگىلى بولمىغان...

ئىش ئورنۇمدىنمۇ بىرنەچچە خىزمەتداشلىرىمنىڭ تەكلىۋىنىمۇ چىرايلىق رەت قىلدىم، باشقىلارنىڭ تۇنۇشتۇرىشىغا ئەسلا يىقىن كەلمىدىم. ھەممە ئادەمنىڭ كۆزى ماڭا تىكىلىپ تۇرغاندەك! مەن توي قىلىپ بولساملا ئاندىن بۇ كۆزلەر تەڭ يوقايدىغاندەك ھىس قىلدىم.

ئارىدىن بىرھەپتە ئۆيكەندىن كىيىن ھەدەمنىڭ ئۆيىدە تىلۋۇزۇر كۆرۈپ ئولتۇراتتىم،ھەدەم مىۋە ئاقلىغاچ ماڭا بىردەم قاراپ تۇردى،كەينىدىن :
__نازى؟ سىڭلىم،بىزنىڭ ئۆيدە توي قىلمىغاندىن سەنلا قالدىڭ، كۆڭلىڭدىكىنى دىگىنە؟

__ۋاي-ۋاي-ۋاي! يەنە باشلىدىڭغۇ ھەدە؟ شۇڭىمۇ كەلمەي دىگەن.

ھەدەمگە ئەركىلەپ ئاغزىنى توسىۋالدىم، ھەدەم قولۇمنى بىر تەرەپكە چۈشۈرۈپ قۇيۇپ سۆزىنى يەنە داۋام ئەتتى:
__ئۆزۈڭ بىلىسەن ،كىچىك يەرنىڭ كىچىكلىكى بار، بىكار گەپ-سۆزگە قالمايلى، ھازىر نىمەڭ كەم؟ خىزمىتىڭ دىسە خىزمىتىڭ بار، ينىڭدا ئاپام دادام،يەنە مەن بار، ئەمدى بىر ئۆينىڭ پىشىنى تۇتمىساڭ بولمايدۇ جۇمۇ؟ ئوبدانراق ئويلىشىپ قوي!

__سەنلا ماڭا تاپە بولدى، مەنمۇ يۇز كۆرۈشمەي. سەن بولىدىكەن،دىسەڭ شۇنىڭغىلا تىگەي!!__دەپ بولۇپ ۋىلىقلاپ كۈلۈپ كەتتىم.

ھەدەممۇ مۇشۇ گىپىمنى كۈتۈپ تۇرغاندەك كۆزى خوشاللىقتىن يوغىناپ كەتتى:
__ۋاھ راست-ھە؟ گىپىڭدە تۇرىسەن-ھە؟
قارىسام ھەدەم مۇنداقلا دەپ قويغان چاقچاق گىپىمگە ئىسىلۋاپتۇ.
__قانداق؟ ماڭا تىپىپ قويغىنىڭ بارمىدى؟__يەنە چاقچاق قىلدىم كۈلۈپ.

__شۇنداق ئۇكام، ساڭا راستىنلا بىرىنى ئويلىشىپ قالدۇق، ئۆتكەندە كەلگەن دوستۇمنى بىلىسەنغۇ؟ ئۆزىمۇ ناھايتتى ياخشى ئايال، ئۇنىڭ ئۈرۈمچىدە بىر نەۋرە ئىنىسى بار ئىكەن، ئۇلارنىڭ ئۆيدىكىلىرى مەسلىھەتلىشىپ تۇرپاندىن بىر قىز ئىلىپ بەرمەكچى بوپتۇ، ئۆزۈڭ بىلىسەن،ئۇلارنىڭ تىگى تۇرپاندىن ئەمەسمۇ؟ ئۇلارمۇ ئۆز يۇرتلىقىمىز باشقا يۇرتنىڭ قىزىغا قارىغاندا ياخشى دەپ ئويلىغان ئوخشايدۇ.
