بۇ نۆۋەت ساقال توغرىسىدا توختۇتۇلۇپ قالدۇق . بۇ ئىختىلابنىڭ بولىشىمۇ ۋە بۇ تېمىنىڭ يوللىنىش ، تەستىقلىنىشى  ھەم مۇنبەر يۈزىدە بۈگۈنگە تۇرىشى تېخىمۇ ئىنچىكە قەلەم تەۋرىتىشىمىزگە قىستىدى . شۇ مۇناسىۋەت بىلەن تېما ئىگىسىگە ۋە مۇنبەر باشلىقلىرىغا رەھمىتىمىزنى بىلدۈرىمىز .
كومپارتىيەنىڭ كىشىلەرنى 40 ياشتىن ھالقىغاندا قۇيۇشقا چاقىرىق قىلىشىدىكى سەۋەب نەدە ؟ ئۇلار نېمە ئۈچۈن بۇ خىل چاقىرىق قىلىشى بىز ئۇيغۇرلارنى ھەقىقەتتىن توسۇشمۇ ياكى ئۇيغۇرلارنى ھەقىقەتكە مايىللاشتۇرۇشمۇ ؟ دېگەندە ، سەۋەب شۇكى : بۇندىن ئىلگىرى بىزلەرنى توردا ۋە ياكى باشقا يەرلەردە سۆز قىلىشقا پۇرسەت بەرمەي ئۆزىنىڭكىنىلا راستقا چىقىرىپ ئەقىدىدە ئەرەبلەرگە ، تالىبانلارغا ئوخشاش ۋە ئۇلار قىلغاننى قىلىش ، ئۇلار كىيگەننى كىيىش ، ئۇلار يىگەننى يىيىش ئۈچۈن شۇلارغا ھەددىدىن زىيادە چوقۇنۇپ كەتكەنلىكتىن ئۆزىنىڭ كىملىكىنى ، تۇرىۋاتقان ئورنىنى ، ئەمەلىيەتتە ئىشلەشكە لايىق بولغان ئەمەللەرنى ئۇنتۇپ ئاتالمىش جىھاد دەپ ئاتىلىدىغان قاتىللىقلارغا بېرىلىپ كىتىش بىلەن ئۆزلىرىگە زۇلۇم قىلىشتىن توسۇش ئۈچۈن ۋە ھەقنىڭ نېمە ئىكەنلىكىنى ھەقىقىي چۈشىنىپ يەتكۈچە بارلىق مەۋقەلەردە ئالدىرىماسلىقنى تەكىتلەپ پەقەت بىز ئۈچۈن ياخشى بولىشىنى تىلەشتىن ئىبارەت ئىدى . كومپارتىيەنىڭ بۇ خىزمىتىگە رەھمەت ئېيتىمىز . جۇڭخۇا خەلىق جۇمھۇرىتىگە رەھمەت !
ئۇلارنىڭ ئويلىغىنى ئۇيغۇرلار ھەق - ناھەقنى تولۇق ئايرىغۇدەك بولغاندا ئاندىن دىنىي مەۋقەلەردە بولۇشنى تەكىتلىدى . شۇنداققۇ جۇمھۇرىيەت ؟
ئەمما كۆپلىگەن كىشىلەر قاتمال ئىدىيەدە دىننىڭ نېمە ئىكەنلىكى ، ئىسلامنىڭ بىزدىن نېمىنى كۈتىدىغانلىقىدەك ھەقىقىي مەلۇماتتىن خەۋەرسىز ھاللاردا نېمىنى كۆرسە شۇنى قىلىش ، نېمىنى توغرا كۆرسە شۇنىڭ ئۈچۈن جېنىنى قۇربان قىلىشتەك ئۆلۈك روھىيىتىگە قارشى تۇرۇشنى تىلەشتىن ئىبارەت . شۇ سەۋەبتىن جۇمھۇرىيەتكە رەھمىتىمىزنى ئېيتىمىز .
ماھىيەتتە ، گەرچە شۇ قاتمال ئىدىيىلىكلەر دىننىڭ قانداقلىقىنى بىلمىگەن ۋاقىتتىن ئىلگىرىلا كومپارتىيە دىننىڭ قانداق بولىدىغانلىقى ۋە ئۇنى قانداق قىلغاندا يوقۇتۇش ، قانداق قىلغاندا تېخىمۇ زورىيىشىشنىڭ مەنبەسىنى تېپىپ چىققان ۋە شۇ خىل ئاساستا ئاللاھنىڭ ئىزنى بىلەن بىزگە شۇ خىل توغرىلىقنى تىلىگەن ئىدى . كومپارتىيەگە رەھمەت !
ئۇلار ھەر قانداق فورمىنىڭ ئۆزىگە خاس شەكلى ، قىممىتى بارلىقىغا ئىشىنەتتى ۋە ئۆزگىچە خىزمەتچىلىرىگە ئۆزلىرىگە خاس فورمىلارنى تارقىتاتتى . ئۇلار ئىش بىجۈرگەندە شۇ كىيىملەرنى كىيەتتى . شۇ خىل فورمىنى كىيگەندە ئۇلارنىڭ خىزمەتتىكى مەجبۇرىيىتى تېخىمۇ ئاشاتتى . شۇنىڭدەك ئەھلى مۇسۇلمانلارنىڭ ھەم تىگىشلىق فورمىسى بار ئىدى . ئۇلار ئەنە شۇ فورمىنى كىيگەندىن كىييىن ، خۇددى جەمىيەتتە قانچىلىغان ج خ ئورۇنلىرى ۋە يوللاردا چارلىغۇچى ، ھەر ئون ئادەمنىڭ بىرى ساقچى بولۇشتەك كۆپلىككە يەتكەندە جەمىيەت تېنىچلىقتىن ئىبارەت ئىمتىيازغا كېلەتتى . ئەنە شۇ خىل ساقچىلارنىڭ بولماسلىقى بىلەنمۇ ئەسلى مۇسۇلمان ئەسلىلىرى ، مۇسۇلمان فورمىسىدىكىلەر تېنچلىقنى تىلەيدىغان ۋە ھەممە ئۇلاردىن ھېچ ئەنسىرمەيدىغان كىشىلەردىن تەركىب تاپىدىغان بولۇپ ، مۆئمىن ئەنە شۇنىڭدەك ئەڭ ئىشەنچىلىك كىشىلەردىن ئىدى . خۇددى ھەر خىل لېي فىڭدىن ئۆگىنىش چاقىرىقىغا ئاكتىپلىق بىلەن قاتنىشىپ ياخشى ئىش ئۈستىدە بولاتتى . بۇلارنى ھەم كومپارتىيە بىلەتتى . كومپارتىيەنىڭ كۈتكىنىمۇ بىزدىكى ئەنە شۇ خىل ئىشىنشىلىكتىن ئىبارەت مۇسۇلمان ئاممىسىنىڭ بارلىققا كېلىشىنى تىلىشى ئىدى . كومپارتىيەگە رەھمەت !
بۇ فورما شۇنداق فورمىكى ، فورما ئەينى ئۇ لىباسكى ، ئەنە شۇ لىباس بىلەن كىشىلەر ئۆز مەجبۇرىيىتىنى ئادا قىلغاي ۋە ھەر ۋاقىت تۇنىغاي ! بۇ شۇنداق فورمىكى ، بۇرۇندىن بۇرۇن ئادەم ئەلەيھىسسالامدىن تارتىپ ئىسلامنىڭ فورمىسى بولۇپ كەلگەن ۋە تاكى بۈگۈنگە قەدەر بىزگە يېتىپ كەلگەن فورمىكى ، بۇ فورما ھېچ بىر مىللەتنىڭ ، ھېچ بىر گوروھتىكىلەرنىڭ فورمىسى بولماستىن جىمى پەيغەمبەرلەر ئىشلىگەن ، كىيگەن ۋە تەۋسىيە قىلغان فورما بولۇپ ، كىملەركى شۇلارنىڭ ئىزىدىن مېڭىپ شۇنىڭدەك بىر ئىشەنچىلىك ئامما قاتارىغا كىرگەندە ياكى شۇ خىل ئۈستۈنلۈك قىممەتنى بويلاپ ماڭغاندىكى نىشانغا يەتمەككە تىز ئات سالماق ئۈچۈن شۇ خىل توننى كىيىش بىلەن ئۆزىنىڭ ياخشىلىقتىن ، ئەڭ ئىشەنچىلىك ئاللاھنىڭ يەر - يۈزىدىكى پەرىشتىلىرى بولۇشتەك ئىمتىيازنى قولغا كەلتۈرۈش ئۈچۈن كىيىدىغان فورما - تەقۋالىق لىباسى ئىدى . بۇ خىل لىباس بىرەر يەرنىڭ يەر شەكلىنىڭ قۇرغاق ياكى ھۆلچىلىكى بىلەن ، ياخشى - يامانلىقى بىلەن ھېچ ئالاقىسى بولمىغان لىباسكى ، كىم ھايا قىلىشتا ۋە ھايا تەرەپتىن كېلىدىغان نېمەتلەرگە يېتىش ئۈچۈن كىيگەن كىيىملەرنى ، ئەخلاقى پەللىگە يەتكەن ۋە شۇ خىل پەللىگە يەتمەكنى تىلىگەن كىشىلەر ساقال بولامدۇ ۋە ياكى تەقۋالىققا خاس ئوبرازلىق شەكلىگە ئورىنىشنى خالاپ شۇ خىل سۈرەتلىك لىباسلارغا ئىختىيار قىلىشى ، بۇ ئۇلارنىڭ ئاڭ ساپاسىنىڭ يۇقىرىلىقىغا يەتكەنلىكىنى بىلدۈرەتتى . ئەمما بۇنىڭسىز ئاڭغا تېخى يەتمەي تۇرۇپلا بىنلادىنغا ، تالىبانغا ، ئافغانغا ئوخشىۋېلىش ئۈچۈن شۇ خىل لىباسنى كىيىشنى قەستلەش بولسا بۇ پەقەت پەيغەمبەرگە چوقۇنۇش ئەمەس بەلكى قەھرىمانلارغا چوقۇنۇشنى ئىختىيار قىلغانلىقتىن ئىبارەت شەكىلۋازلىق بولۇپ ، بۇنىڭدەك ئىدىيەلەردىكى قاششاڭلىقتىكى قارغۇلاچە چوقۇنۇش بىردەك توسۇلىشى كېرەك ئىدى . چوقۇمكى ئاۋال ئۇلاردىكى ئىدىيە ساپلاشتۇرىلىشى كېرەك . كومپارتىيە ئەنە شۇنى بىزگە تىلىگەن ئىدى . ئەمما قاششاڭلىق ئىدىيەدە كۆپلىگەن كىشىلەر ئۇلاردىن ئاغىرىندى . كومپارتىيەگە رەھمەت ئېيتىمىزكى ، بىز تەرەپتىن توغرا بولغان ھەقىقەتنى قۇبۇل قىلماي شۇ خىل ياسالمىلىقنى تىلەشكە ئۆتكەنلەرنى بىردەك بىزلەردىكى ھەقىقەتكە ئاللاھنىڭ ئىزنى بىلەن قىستاشتا بولدى . بۇ ئەجىرىدىن ئېيتىمىزكى ، كومپارتىيەگە رەھمەت !

ئەھۋال شۇنىڭدەك ئىكەن ، ساقالنى تۈك ئورنىدا تۇنىۋېلىش ، 40 ياشتىن ئاشقاندا قۇيىۋېلىش ھېچ بىر ئىقلىمدا يوق ئىش . پەقەت ئەقىل يەتكەندە ۋە توغرا يولدا بولغاندا مەيلى ئۇ 20 ياش بولسۇن ، ئۇنىڭدىن چوڭ - كىچىك بولسۇن ھەممىگە ئورتاقتۇر . قايسى بىر مۇسۇلمان كىشىلەر ئارىسىدا ئىشەنچىلىك بولىدىكەن ۋە ئىشەنچىلىك بولۇشنى تىلەيدىكەن ، ئۇ شۇنىڭدەك لىباسنى كىيىشكە ھەقلىق . بىزمۇ ۋە ھەممە شۇنىڭدەك لىباستىكىلەردىن بىردەك خاتىرجەم بولىشىمىز كېرەك . ئەمما كىملەركى ، شۇنىڭدەك يامان مەۋقەلەردە بولىدىكەن ، ئۇنىڭدەك كىشىلەرنىڭ لىباسىنى سالدۇرىۋېتىش جىمى ئىقلىمدىكى قانۇننىڭ ئاساسىي قىسمىغا كىرگۈزىلىشى كېرەك . چۈنكى جىمى ئىنسان ۋە جىنلارنىڭ ئىسلامدىن نىسىۋىسى باردۇر . گەرچە ئىسلام دىينىدا بولمىسىمۇ ، ئىسلامدىن يېتىدىغان قىممەتتە ئورتاقتۇر . بۇنىڭغا قارشى پىكىرلەردە بولىدىغانلار بولسا پىكرىنى قويسا بولىدۇ .
تېمىدىكىدەك ، بىر ھازا ياخشىلىقنى سۆزلەپ ، ئاخىرىدا لامالىق تۈشى ئىچىدىكى دىننى تەشۋىق قىلىپ قويۇپ ، خۇددى ئۈزۈمتال ئېيتقاندەك 30 قېتىم تاھارەت ئېلىپ بىر قېتىم قەي قىلغاندەك ھەممىنى بىكار قىلىۋىتىشتەك ئاقىۋەت يۈز بېرىپ قالمىسۇن .
ئىسلام دىنى تۇڭگانلار ئويلىغاندەك ، پەقەت 40 ياشقىلا بارغاندا ‹‹ ئى خۇدايا مېنى ئازراق كەچۈرسەڭ ! ›› دەپ دەپ دىنغا كىرىدىغان ۋە ئىلگىركى يامانلىقىنى تۇتقا قىلغان ھاللاردا ئۆتىۋەرسە بولىۋېرىدىغان دىن ئەمەس . كىم قايسى ۋاقىتتا ئىسلامنىڭ نېمە ئىكەنلىكىنى تۇنىغان ۋە چۈشەنگەندىن تارتىپ شۇ ۋاقىتتىن باشلايدىغان دىندۇر . ياكى يەھۇدىي - ناسارالاردەك بىر نوقتىغىلا مەركەزلىشىۋېلىپ قانۇنلاشتۇرىۋالىدىغان دىن ئەمەس . دىننى ۋە ئىسلامنى چۈشىنىپ ۋە ئاندىن ئەدەبىياتقا كىرىشىپ بىر نىمىلەرنى يازساقمۇ كېچىكمەيمىز .
چوڭلارنىڭ ساقىلىنى تەشەببۇس قىلىپ ، شۇنىڭدەك چوڭ بولغاندا ، قېرىغاندا ئاندىن شۇنىڭدەك كىيىنىڭ دىيىش ، بۇ ھەقىقەتەن پىتنىلەردىندۇر .
مۇھەممەد سالىھ بىزنىڭ ئۈلگىمىز ئەمەس . ئۇ پەقەت بىر ئالىم . ئۇنى ۋە ئۇنىڭدىكى ئىلىمنى ھۆرمەتلەيمىز . ئەمما ئاللاھ بىزنى بۇيرىغاندەك ‹‹ ئاتاڭ سەن بىلمەيدىغان نەرسىگە ماڭا شېرىك كەلتۈرۈشكە زورلىسا ئۇنىڭغا ئىتائەت قىلما ! ›› دېگەنكى ئاللاھنىڭ قانۇنىغا چۈشىمىز .
شۇنىڭ ئۈچۈن ئۇ ئالىم قويمىسا ساڭا نىمە كەپتۇ دەپ پىتنە قىلىدىغانلار ، ئۇ ئالىمنى پەش قىلماي ، ساقال قويامدۇ - قويمامدۇ ، ئۇ ئۆزىنىڭ ئىشى .  پەقەت بىزدىكى خاتالىقنى تۇنۇتۇپ بەرسىلا بولىدۇ .
چۈشىنىشلىك بولدى ھە !

مۇنبەردىكى ئۇرۇنسىز تالىشىشلار تۈگىمەمدىغاندۇ؟ زادى. ھىچكىم ئىچى پۇشۇپ ياكى قىلغىلى ئىشى يوق مۇنبەرگە بىر نەرسىلەرنى يوللىمايدۇ. دىمىسىمۇ ھازىر مەدەنىيەت جەھەتتىكى چەكلىمىكلەر بەك ئېغىر. بۇلۇپمۇ تۆۋەندە شۇنداق. بۇرۇت قويساڭ بولمايدۇ.دوپپا كەيسەڭ بولمايدۇ. ياغلىق ئارتساڭ بولمايدۇ. ھەممىڭ «ئاكىمىز» كىيىنگەن بۇيىچە كىيىن. تاسلا قالىدۇ، ھەممىڭ «ئاكىمىز» يىگەننى يە دىگىلى. ھەممىمىز قانۇننىڭ ئىدارە قىلىشى بىلەن ياشايمىز، بىراق نىمىشقا قانۇن يول قويغان دائىرىدىكى مەنپەتىمىزنى ئاشۇ ئۇششاق باشلىقلىرىمزىنىڭ چەكلىشى بىلەن چەكلەپ قۇيىمىز. بىزنىڭ ئەنئەنىمىز ياكى باشقىلارنىڭ يېغىرىغا تىگىدىغان، غورورىنى دەپسەندە قىلىدىغان ئەنئەنىمىكەن.
   توغرا پىكىرنى قۇبۇل قىلايلى، ئۆز- ئارا تالىشىشىتىن ھىچقانداق نەتىجە چىقمايدۇ.
كىشىگە يامانلىق ئويلىغانلىق ئۆزىگە ئورا كولىغانلىق.
