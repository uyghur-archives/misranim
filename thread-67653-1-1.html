  بۈگۈن دەم ئېلىش كۈنى بولغاچقا ، تالاغا چىققىم كەلمەي ئۆيدە ئولتۇردۇم ، بىر چاغدا تېلفونۇم جىرىڭلىدى ، قارىسام دوستۇم زۇلفىيە ئىكەن ، ئۇ مۇ بۈگۈن دەم ئېلىش بولۇپ مەندىن ئەھۋال سوراپ تېلفون قىلغان ئىكەن ، مەن ئۇنىڭغا ئۆيۈمگە كەل ، ئۇزۇن بوپتۇ ئىككىمىز مۇڭداشمىغىلى دەپ ئۇنى ئۆيۈمگە تەكلىپ قىلدىم ، ئۇمۇ ئۆيدە ئىچى سىقىلىپ ئولتۇرغان ئىكەن ،بۇ گېپىگە قوشۇلۇپ ، كىلدىغان بولدى ، تېلفوننى قويۇپ يېرىم سائەتلەر ئۆتكەندە ئۇمۇ ئۆيگە كەلدى . ئىككىمىز تىچ ئامانلىق سورىشىىپ ، مەن ئۇنى ھۇجرامغا باشلاپ كىردىم ، ۋە شۇنىڭ بىلەن پاراڭغا چۈشۈپ كەتتۇق ، بىر كەملەر بولغاندا ئۇنىڭدىن :" ئاداش ، گۈلپىيە ھەدەم قانداقراق ....ئۇزۇن بوپتۇ ئۇ
ھەدەڭنى كۆرمىگىلى ....." دىيىشىمگە ئۇ ماڭا قاراپ تۇرۇپ كېتىپ ، بىردىنلا كۆزىدىن ياش قۇيۇلۇشقا باشلىدى ، مەن ئۇنىڭغا قاراپ ھەيران قالدىم ، ۋە ئۇنىڭدىن نىمە بولغانلىقىنى سورىدىم ، ئۇ ئۆزىنى تۇتۋالغاندىن كېيىن سۆزلەشكە باشلىدى .....
_ ئاداش ، ھەدەمنىڭ ئېسىل مىجەزلىك ، ئاددىي ساددا قىز ئىكەنلىكىگە . نىمىشقا بەختى چىقمايدىغاندۇ .....ھەدەمگە قاراپ ئاپام - دادام ئىچىدىن تۈگىشىپلا كېتىۋاتىدۇ ، بىلەمسەن جېنىم ئادىشىم ، يېزىنەم سىرتلىق بولغاچقا ، ئاتا - ئانام يېزىنەمنى شۇنچىلىك ۋاي دەيدۇ ، چۈنكى ئاتا - ئانىسى بۇ يەردە بولمىغاچقا يالغۇزلۇق تارتىپ قالمىسۇن شۇنچىلىك ھۆرمەتلەيدۇ ، ھەدەممۇ شۇنچىلىك ۋاي دەپ يېزىنەم ۋە بالىسىنىڭ ھالىدىن شۇنچىلىك ياخشى خەۋەر ئالىدۇ . ئۇ ھەدەمنىڭ تۇنجى مۇھەببىتى ، بۇنى بىلمەيسەن ھەقىچان ، لىكىن ئۇ نىمىشقا ئاچامنىڭ كۆزىدىن ياشلا ئاققۇزىدىغاندۇ ......بىز باشتا بىلمەپتىكەنمىز ، بىر كۈنى ھەدەم ئۆيگە كىردى ۋە گەپ سۆز قىلماي مېنىڭ كارۋىتىمغا ئۆزىنى تاشلاپ ياتتى ، بۇنىڭغا بىز پەرۋا قىلماپتۇق ، بىر چاغدا ئۇنىڭغا تېلفون كەلدى ، لىكىن ھەدەمنى تېلفون ئېكانىغا قاراپلا  تېلفوننى بېسىۋەتتى ، نەچچە قېتىملاپ جىرىڭلىسىمۇ
بېسىۋەتتى ، مەن تېلفوننى ئۇنىڭدىن تارتىۋىلىپ قارىسام يېزىنەمكەن ، ئاچامغا نىمە بولدۇڭ؟ بىرەر ئىش بولدىمۇ
دەپ سوراپ قويغىنىمنى بىلىمەن ، ئاداش!ئاچام ئېسەدەپلا كەتتى ، بىر چاغدا دادام بىلەن ئاپام كىرىپ نىمە بولدۇڭ قىزىم دەپ سورىشى ، ئىچى ئېچىشىپ پارتىلىيالماي تۇرغاندەك تېخىمۇ ئېسەدەپ يىغلاشقا باشلىدى ، ھەم بىر كەمدەيىغىسىنى پەسەيتىپ بىزگە بولغان ئىشلارنى بىر-بىرلەپ دىيىشكە باشلىدى ، بىز ئاڭلاپ ھەيران قالدۇق .....ئۇلارنىڭ ئوتتۇرسىدىكى جىدەل باشقا جىدەل ئەمەس بەلكى پۇل جىدىلىكەن، ئاڭلاپ ھەيرانلا قالدۇق
، ئۆزۈڭمۇ بىلىسەن ئاچامنىڭ ئىشلەيدىغان ئىش ئورنىمۇ ياخشى ، ئۆزىمۇ ئاددىي - ساددا ، ئىخدىسادچىل ، باشقا دوسلىرىغا ئوخشاش مودا قوغلاشمايدۇ ، لىكىن يېزىنەم ئاچامنى سەن بەك پۇل خەجلەپ كەتتىڭ دەپلا جىدەللىشىدىكەن ، ئۆزى پۇل خەجلىسە كارى يوقكەن ، لىكىن ئاچام پۇل خەجلىسە بولمايدىكەن ، ئۆزۈڭمۇ بىلىسەن ، يېزىنەم بەكلا ئويۇنچى ، ئاغىنەم دەپلا ئۆتىدۇ ، بەزىدە ئاغىنىسىنى دەپ خوتۇن بالىسىنىمۇ ئۇنتۇپ قالدىغان دەرىجىدە ، بەزىدە ئويلاپ قالىمەن ، ئاشۇ ئاغىنىلىرى بىشىغا قىيىنچىلىق ئۇچرىغاندا ئەسقاتامۇ دەپ
......ئۆزى پۇل خەجلىسە گەپ يوقكەن ، ھەدەم ئەمدى ئۆزىگە چۇشلۇق بىرەر نەرسە ئېلىشقىمۇ بولمايدىكەن ....نەچچە قېتىم ھەدەمنى سېسىق گەپلەر بىلەن ھاقارەتلەپتۇ ، بىز تېخى جىق ئىشلارنى بىلمەپتىمىز ....بۇ قېتىم ھەدەمگە ئايلىق كارىڭنى ۋەمۇكاپات پۇلى ئالدىغان  كارىڭنى ماڭا بېرىسەن دەپ جىدەللىشىپتۇ .....ۋە مۇشۇنداق قىلمىساڭ خېتىڭنى ئال دەپ ھەيۋە قىپتۇ ، ئادىشىم مۇشۇمۇ قىلاي دىسە گەپمۇ ...ھەدەمنى نىمىسىگە بۇنداق خارلايدۇ ، كۆپ قېتىم ھەدەم ئېغىرلىق قىلىپتۇ ، لىكىن ئۇ بارغانسىرى ھەددىدىن ئاشقىلى تۇرۇپتۇ ، ئۆزۈڭ بىلىسەن ئاتا- ئانام ھەدەمگە شۇنداق ئامراق ، مۇشۇ گەپلارنى ئاڭلىغاندا جېنىنىم دادامنىڭ چىرايى تاتىرىپ كەتتى ، ئاداش ھەدەمگە قاراپ مېنىڭ يۈرىكىم پۇچىلىنىپ كەتتى ، جېنىم ئاداش ، بالىسىنى بىز باقىمىز ، بىرەر قىيىنچىلىققا ئۇچرىسا ئاتا - ئانام ئوتتۇرغا چىقىپ ياردەم قىلىدۇ ، نىمىشقا يەنە ھەدەمنىىڭ كۆزىدىن ياش ئاققۇزىدۇ ....ھېلىمۇ ھەدەمنىڭ ئىش ئورنى يېزىنەمنىڭكىدىن ياخشى ، ئۇنداق بولمىغان بولسا ئىش بار ئوخشايدۇ ئۇنىڭغا.....لىكىن بىر يېرى يېزىنەمنىڭ ئاتا - ئانىسى ئېسىل ئادەملەر ، ھەدەمنىڭ ھۆمىتىنى شۇنچىلىك قىلىدۇ بۇنىڭغا كۆز يۇمسام بولمايدۇ ، لىكىن يېزىنەم نىمىشقا ئاتا - ئانىسىنىڭ مىجەزىنى تارتمىغاندۇ ..... نىمىشقا مۇشۇنداق قىلدىغاندۇ .......مۇ شۇ ئەر خەق نىمىشقىمۇ داۋخو دۇ .....ھەدەمگە ۋە ئەتراپىمدىكى كۆپىنچە كىشىلەرنىڭ نىكاھ تۇرمىشىغا قاراپ توي قىلىشتىنمۇ قورقىدىغان بولۇپ قالدىم ....لىكىن بىز قىزلار بەكمۇ ئاجىز كىلدىكەنمىز .... ئوغۇللار نىمانچە ئەسكىدۇ ، توي قىلماسنىڭ ئالدىدا ۋاي دەيدۇ ، توي قىلىپ بولۇپ ئۆزگىردۇدە خۇددى خامىليوندەك ..... ھەي ئاداش بۇ ئىشلارغا نىمە دەيلى ...ھەر قېتىم ئاچامغا قارىساملا ئىچىم ئېچىشىدۇ ..... ئۇ قايناپ سۆزلەۋاتاتتى ، ھەدىسىگە ئامراق بولغاچقىمۇ ئۇنىڭمۇ ئىچى ئېچىشىپلا تۇراتتى ،
   زۇلفىيەنىڭ گېپىنى ئاڭلاپ كۆڭلۈم ئاجايىپ بولۇپ قېلىۋاتاتتى ، دىمىسىمۇ گۈلپىيە ھەدەمنىڭ مىجەزىنىڭ  ئېسىللىقىنى بىزنىڭ بۇ يەردىكى ھەممىسى بىلىدۇ ، لىكىن ھازىرقىنى گەپلەرنى ئاڭلاپ گۈلپىيە ھەدەمنىڭ ئېرىگە بىردىنلا غەزىپىم قوزغالدى ،   ھەي !!! بۇ ئىشلارغا نىممۇ دىگۈلۈك ۋە نىممۇ دىيەلەيمىز ....
        شۇنداق دىگۈم كىلىدۇ ، ھەي يىگىت ، ئايالىڭنى خارلاشتىن بۇرۇن ئۆزۈڭنىڭ ئاچا - سىڭىللىرىڭنى ئويلاپ قوي ، چۈنكى سەن بۈگۈن ئايالىڭنى خالاپ ، ئۇنىڭ كۆزلىرىدىن ياش ئاققۇزغۇنۇڭ بىلەن ، كۈنلەرنىڭ بىرىدە ئاشۇ ئاچا - سىڭىللىرىڭمۇ مەلۇم كىشىلەر تەرپىدىن خارلىنىپ قېلىشى مۈمكىن ، چۈنكى ئۇنتۇمىغىن ، بۇ جاھان ھەممە كىشىگە ئۆتنە ،  ئۇلۇغ ئاللاھ  ھەممىنى تەڭشەپ تۇرغۇچىدۇر ...............
