جوڭنەنخەينى زىلزىلىگە سالغان جيەن بوزەننىڭ ئۆزىنى ئۆلتۈرۈۋېلىش ۋەقەسى


جاڭ چۈەنشى



ھەزرىتى ئەلى بارات تەرجىمىسى


«مەدەنىيەت ئىنقىلابى» مەزگىلىدە، ماۋزېدوڭ جيەن بوزەننى ھىمايە قىلغان بولسىمۇ، لېكىن لىن بياۋ، جياڭ چىڭ ۋە «مەركىزىي مەدەنىيەت ئىنقىلابى گورۇپپىسى» دىكى بىر ئوچۇم ۋەھشىي سۇيىقەستچىلەرنىڭ جيەن بوزەننى بوش قويۇۋەتكۈسى يوق ئىدى. جيەن بوزەن گومىنداڭ بىلەن كومپارتىيە 1936- يىلى ئاچقان «نەنجىڭ تىنچىلىق سۆھبىتى» گە قاتناشقان بولغاچقا، ئۇلار ئۇنى قىستاپ ئىقرار قىلدۇرۇپ، ليۇ شاۋچىنىڭ گومىنداڭنىڭ C C مەزھىپى ئىشپىيۇنلىرى بىلەن «تىل بىرىكتۈرۈپ»، «سوۋېت رايونىنى يوقىتىش، قىزىل ئارمىيەنىڭ ئەكسىلئىنقىلابىي ھەرىكەتلىرىنى يوقىتىش سۇيىقەستىدە بولۇش» قارا ماتېرىياللىرىغا ئېرىشمەكچى بولدى. جيەن بوزەننى قىستاپ ئىقرار قىلدۇرۇشقا مەركەزنىڭ مەخسۇس دېلولارنى تەكشۈرۈش گورۇپپىسى 1- ئىشخانىسى «ليۇ شاۋچى، ۋاڭ گۇاڭمېي مەخسۇس دېلو گورۇپپىسى» نىڭ مۇئاۋىن گورۇپپا باشلىقى ۋۇ جوڭ مەسئۇل بولغانىدى.

1968- يىلى 11- ئاينىڭ 22- كۈنى، ۋۇجوڭ مەخسۇس دېلو گورۇپپىسى ئەزالىرىنى باشلاپ بېيجىڭ ئۇنىۋېرسىتېتىغا باردى، بېيجىڭ ئۇنىۋېرسىتېتىدا تۇرۇشلۇق تەشۋىقات ئەترىتى قوماندانلىق بۆلۈمى بىلەن ئالاقىلەشمەيلا ئۇدۇل تارىخ فاكۇلتېتىغا كىردى ۋە بىر ئوقۇغۇچىنىڭ يول باشلىشى بىلەن جيەن بوزەننىڭ ئۆيىگە بېرىپ ئۇنى «سوراق قىلدى». ۋۇ جوڭ جيەن بوزەنگە 1936- يىلى ليۇ شاۋچى، جياڭ جيېشى بىلەن «تىل بىرىكتۈرۈپ» «سوۋېت رايونلىرىنى يوقىتىش سۇيىقەستىدە بولۇش» يىغىنىغا قاتناشتى، دەپ تۆھمەت چاپلىدى. سوراقتا جيەن بوزەندىن: «شۇ چاغدا قانداق يىغىنلارغا قاتناشتىڭ، نېمە ئىشلارنى قىلدىڭ، شېن شياۋلىڭ سەن بىلەن لۇ جېنيۈنى ئىزدەشنى قانداق مەسلىھەتلەشتى، ئۇ جۇ شياجوۋ، ۋۇ يىڭشەن بىلەن قاچان تونۇشقان؟» دېگەندەك سوئاللارنى سورىدى.

11- ئاينىڭ 28- كۈنى، ۋۇ جوڭ بىلەن مەخسۇس دېلو گورۇپپىسى ئەزالىرى بېيجىڭ ئۇنىۋېرسىتېتى تارىخ فاكۇلتېتىغا ئىككىنچى قېتىم كېلىپ، بېيجىڭ ئۇنىۋېرسىتېتىدا تۇرۇشلۇق تەشۋىقات ئەترىتى قوماندانلىق بۆلۈمى پىپەن قىلىپ ئۆزگەرتىش گورۇپپىسى ۋە تارىخ فاكۇلتېتىنىڭ ئەترىتىنىڭ مەسئۇللىرىغا «ليۇ شاۋچىنىڭ 1936- يىلى سكھبەت ئۆتكۈزۈشنى باھانە قىلىپ ئىشپىيۇنلار بىلەن تىل بىرىكتۈرگەنلىك ئەھۋالى ھەمدە جيەن بوزەننىڭمۇ شۇنىڭغا قاتناشقانلىق ئەھۋالىنى تونۇشتۇردى.» مەركىزىي مەخسۇس دېلو گورۇپپىسىنىڭ تەكلىپى بىلەن، تارىخ فاكۇلتېتى شۇ كۈنى كەچتە ئۈچ ئوقۇغۇچىدىن تەشكىللەنگەن «جەين بوزەن مەخسۇس دېلو گورۇپپىسى» قۇرۇپ، ئۇلارنىڭ جيەن بوزەننى سوراق قىلىشىغا ياردەملەشمەكچى بولدى.

3- دېكابىر چۈشتىن بۇرۇن سائەت 9:30 دىن 11:30 غىچە، تارىخ فاكۇلتېتى «مەخسۇس دېلو گورۇپپىسى» جيەن بوزەننى «سوراق قىلىپ»، ئۇنىڭ 1936- يىلىدىكى ئاتالمىش «گومىنداڭ- كومپارتىيە سۆھبىتىگە قاتناشقاندىكى جىنايى قىلمىشلىرى» نى تولۇق تاپشۇرۇشقا بۇيرىدى.

5- دېكابىر، ۋۇ جوڭ مەركىزىي مەخسۇس دېلو گورۇپپىسىدىكىلەر بىلەن بىللە بېيجىڭ ئۇنىۋېرسىتېتىغا يەنە كەلدى ۋە تارىخ فاكۇلتېتى «مەخسۇس دېلو گورۇپپىسى» دىكى بىر ئەر، بىر ئايالنىڭ ھەمراھلىقىدا جيەن بوزەننى ئىككىنچى قېتىم «سوراق قىلدى». جيەن بوزەن ئىقرار قىلمىدى. ۋۇ جوڭ ئۈستەلنى مۇشتلاپ تۇرۇپ: «بىزدە يېتەرلىك ئىسپات بار. ياۋاشلىقچە ئىقرار قىلساڭ قىلدىڭ، بولمىسا تۈرمىدە ياتىسەن» دەپ قاتتىق ۋارقىرىدى. جيەن بوزەن بۇنى ئاڭلاپ جىددىيلىشىپ كەتتى، ئەمما ناھايىتى كەسكىنلىك بىلەن: «مەندى ئىقرار قىلغۇدەك ھېچنېمە يوق، تۈرمىدە ياتسام ياتاي!» دېدى. ۋۇ جوڭ جيەن بوزەننى ئايالى بىلەن «ئايرىۋېتىپ»، جيەن بوزەنگە «ئىقرار ماتېرىيالى» يازدۇرۇشنى چىڭ تۇتۇشنى تەلەپ قىلدى.

7- دېكابىر، جيەن بوزەن كېسەل بولۇپ يېتىپ قالدى. تارىخ فاكۇلتېتى «مەخسۇس دېلو گورۇپپىسى» ۋۇ جوڭنىڭ بۇيرۇقى بويىچە جيەن بوزەندىن «ئىقرار ماتېرىيالى» ئالىدىغان بولدى.

9- دېكابىز، ۋۇ جوڭ مەخسۇس دېلو گورۇپپىسىدىكىلەرنى باشلاپ بېيجىڭ ئۇنىۋېرسىتېتىغا يەنە كەلدى. بىر تەرەپتىن جيەن بوزەننى «ئىقرار قىلىش» قا قىستىسا، يەنە بىر تەرەپتىن تارىخ فاكۇلتېتىدىكى «مەخسۇس دېلو گورۇپپىسى» نىڭ بارلىق ئەزالىرىغا «جيەن بوزەننىڭ ئەھۋالىنى تونۇشتۇردى». ئەمەلىيەتتە ئۇنىڭ قىلىۋاتقانلىرىنىڭ ھەممىسى قۇرۇق تۆھمەت ۋە باشقىلارنى كۈشكۈرتۈشتىنلا ئىبارەت ئىدى. كېيىن يەنە تەشۋىقات ئەترىتى قوماندانلىق بۆلۈمىگە كېلىپ، «بېيجىڭ ئۇنىۋېرسىتېتىدىن  جيەن بوزەننى ئايرىۋېتىش» نى تەلەپ قىلدى.

10- دېكابىردىن 16- دېكابىرغىچە تارىخ فاكۇلتېتى «مەخسۇس دېلو گورۇپپىسى» «جيەن بوزەننى تەكشۈرۈش» كە ئۈچ قېتىم ھەمكارلاشتى.

18- دېكابىر چۈشتىن كېيىن، ۋۇ جوڭ مەخسۇس دېلو گورۇپپىسىنىڭ بەش ئەزاسى بىلەن بېيجىڭ ئۇنىۋېرسىتېتىغا بېرىپ، جيەن بوزەننىڭ ئۆيىگە ئۇنى «سوراق قىلغىلى» ئادەم ئەۋەتتى. جيەن بوزەن كېسەل ئاسارىتىدە كارىۋېتىدا ياتاتتى. سوراق قىلغىلى كەلگەنلەر ئۇنى قوپاللىق بىلەن تۈرتۈپ تۇرۇپ: «تۇرە ئورنۇڭدىن، كۈن چۈش بولغۇچە يېتىۋېرەمسەن ئەبلەخ! سېنى ھازىر سوراق قىلىمىز» دەپ ۋارقىرىدى. سوراق قىلىشتىن ئىلگىرى ۋۇ جوڭ مەكتەپنىڭ ھەربىي تەشۋىقات ئەترىتىنىڭ باش قوماندانلىق بۆلۈمىگە كەتتى، باشقىلار سوراقنى داۋاملاشتۇردى. ئۇلار ھەدېسىلا جيەن بوزەننى قورقۇتۇپ، جيەن بوزەنگە ماۋ جۇشىنىڭ ئۆزى بىلەن دۈشمەن ھەققىدىكى «ئۈزۈندىلىرى» نى قايتا- قايتىلاپ ئوقۇتتى. «سوراق» ئالاھەزەل چۈشتىن كېيىن سائەت 5 كىچە داۋام قىلدى. شۇ ئەسنادا ۋۇ جوڭ سىرتتىن چىرايىدىن غەزەپ ياغدۇرۇپ كىرىپ كەلدى ۋە بىر ۋاراق قەغەزگە: «تېزراق سوراق قىلىڭلار، سىرتتا ماشىنا ساقلاپ تۇرىدۇ» دەپ يېزىپ ياڭ x كە بەردى. ئاندىن قەستەن جيەن بوزەنگە بۇرۇلۇپ: «سىرتتا ماشىنا بار، يەنە ئىقرار قىلمايدىغان بولساڭ، سېنى تۇتۇپ تۈرمىگە سولىۋېتىمىز! ئۆزۈڭچە تېخى خەلقئالەمنىڭ كۆز- قۇلىقىنى توسۇپ قالىمەن دەپ ئويلاۋاتامسەن؟» دېدى.

ئورنىدا جىممىدە ئولتۇرغان جيەن بوزەن بۇ گەپنى ئڭلاپ ئورنىدىن لىككىدە تۇرۇپ كەتتى، مەركىزىي مەخسۇس دېلو گورۇپپىسىدىكىلەرگە قاراپ قويۇپ ئۆي ئىچىدە ئۇياقتىن- بۇياققا بىردەم ماڭدى. ئاندىن غەزەپ بىلەن قولىنى ئېگىز كۆتۈرۈپ: «مۇشۇنچىلىك قولۇم بىلەن خەلقئالەمنىڭ كۆز- قۇلىقىنى قانداقمۇ توسۇۋالالايمەن؟» دېدى. گۇرۇپپا ئىچىدىكى يەنە بىر كىشى: «ئۆزۈڭچە تېخى ئۆلسەممۇ ھېچنېمىنى ئىقرار قىلمايمەن دەپ ئويلاۋاتامسەن؟» دەپ تىل سالدى. جيەن بوزەن ئىستىھزالىق بىلەن كۈلۈپ قويدى ۋە ۋۇ جوڭغا يۈزلىنىپ تۇرۇپ: «مەن سىلەرگە دەپ قوياي، ماڭا قاراڭغۇ ئىشلارنى خالىغانچە يېزىپ قويسام بولمايدۇ، مەن ھەقىقەتنى ئەمەلىيەتتىن ئىزدىگەن ئاساستا ئىش كۆرىمەن. تۈرمىدە يېتىشتىن قورقمايمەن. گومىنداڭ دەۋرىدە تۈرمىدە يېتىشتىن قورقمىغان ئادەمئەمدى قورقۇپ قالامتىم. ئۆلۈمدىن قورقمىغان ئادەم تۈرمىدىن قورقاتتىمۇ؟» دېدى. ۋۇ جوڭ قاتارلىقلار ماڭىدىغان چاغدا جيەن بوزەنگە يەنە تەھدىت سېلىپ: «ياخشى ئويلان، ساختىلىرىنى چىقىرىپ تاشلاپ راستىنى ئېيت، بىز ئەتە يەنە كېلىمىز» دېدى.

ۋۇ جوڭ جيەن بوزەنگە ئۆزىنى مەركىزىي مەخسۇس دېلو گورۇپپىسىدىكى مۇھىم شەخس دەپ كۆز- كۆز قىلىۋەرگەچكە جيەن بوزەنگە بولغان بېسىم بارغانسېرى ئارتىپ كەتتى. جيەن بوزەن تىز پۈكىمىگەن بولسىمۇ، لېكىن ىزيادە جىددىيلىشىپ كەتتى. ۋۇ جوڭ قاتارلىقلار كەتكەندىن كېيىن، «مەخسۇس دېلو گورۇپپىسى» دىكى ئوقۇغۇچىلار جيەن بوزەننى «ماتېرىيال يېزىش» قا قىستىدى. جيەن بوزەن: «بۇ دېگەن ليۇ شاۋچىغا بېرىپ تاقىلىدىغان ئىش، مەن قالايمىقان يازسام بولمايدۇ» دېدى ئوقۇغۇچىدىن: «مېنىڭ زادى نېمە قىلغان گۇناھىمغا بۇنداق قىلىسىلەر؟» دەپ سورىدى. ئوقۇغۇچى: «ۋۇ جوڭ دېدىغۇ؟ ئەگەر تولۇق تاپشۇرمايدىكەنسەن، تۈرمىدە ياتىسەن» دەپ جاۋاب بەردى. جيەن بوزەننىڭ ئايالى روھىي جەھەتتىن قاتتىق جىددىيلىشىپ ئۆز- ئۆزىگە توختىماستىن: «ھەي، ئۇلار يەنە كېلەرمۇ؟» دېيىشكە باشلىدى. كېچە سائەت ئالاھەزەل 12 لەر بولغاندا بۇ ئىككى ئەر- ئايال «ھەقىقەت» نى قوغداش، «ھەقىقەتنى ئەمەلىيەتتىن ئىزدەش» پرىنسىپىنى قوغداش، «ئۆز ئىززەت- غۇرۇرىنى قوغداش» يولىدا دورا ئىچىپ، 40 يىلدىن ئارتۇق بىللە كۈرەش قىلغان بۇ ئالەم بىلەن ۋاقىتسىز ۋىدالاشتى.

ئەتىسى چۈشتىن بۇرۇن سائەت 8 دە، دۇ شۈەن جيەن بوزەن ئەر- ئايالنىڭ ئىشىكىنىڭ ئېچىلمىغىنىنى كۆرۈپ، بىر نەچچە قېتىم چاقىردى. جاۋاب بولمىغاندىن كېيىن كۆڭلىگە گۇمان چۈشۈپ ئىشىكنى بۇزۇپ كىردى. جيەن بوزەن ئەر- ئايال رەتلىك كىيىنىپ ھەر ئىككىسى ئۆز كارىۋېتىدا يېتىپ بۇ ئالەم بىلەن جىممىدە خوشلاشقانىدى. جيەن بوزەننىڭ چاپىنىنىڭ ئوڭ يانچۇقىدا بىر پارچە باغاقچە تۇراتتى. باغاقچىگە: «مەندە ئىقرار قىلغۇدەك ھېچنېمە بولمىغاچقا، ئاخىرى مۇشۇ ھالاكەت يولىنى تاللىدىم. مېنىڭ بۇ ھالاكەت يولىنى تاللىغىنىمدىن دۇ ئۇستاز قىلچە خەۋەرسىز» دەپ يېزىلغانىدى. چاپىنىنىڭ سول يانچۇقىدىن يەنە بىر باغاقچە چىقتى. باغاقچىگە: «ماۋجۇشى ياشىسۇن! ماۋجۇشى ياشىسۇن! ماۋجۇشى ياشىسۇن!» دەپ يېزىلغانىدى.

جيەن بوزەن ئەر- ئايالنىڭ ئۆزىنى ئۆلتۈرۈۋېلىش ۋەقەسى بىلەن «باغاقچە» نىڭ ئەسلى نۇسخىسى ج خ ئىدارىسى تەرىپىدىن شۇ كۈنىلا ماۋزېدوڭ بىلەن جۇ ئېنلەيگە دوكلات قىلىندى. ئىككى رەھبەر: «ئۆلۈش سەۋەبى دەرھال تەكشۈرۈلسۇن» دەپ تەستىق يازدى. شۇ كۈنى كەچتە جيەن بوزەن ئەر- ئايالنىڭ جەسىتى بېيجىڭ مېدىتسىنا ئىنىستىتۇتىدا بېيجىڭ شەھەرلىك ج خ ئىدارىسىنىڭ قانۇن دوختۇرىنىڭ يېتەكچىلىكىدە ئوپراتسىيە قىلىنىپ تەكشۈرۈلدى. نەتىجىدە ئۇيقۇ دورىسىنى كۆپ ئىچىپ ئۆلۈۋالغانلىقى مەلۇم بولدى.

ئەتىسى مەركەزدىكى بىر مەسئۇل خادىم ماۋزېدوڭ بىلەن جۇئېنلەينىڭ يوليۇرۇقىنى تاپشۇرۇۋېلىپ، خەلق چوڭ سارىيىدا «ليۇ شاۋچى، ۋاڭ گۇاڭمېي مەخسۇس دېلو گۇرۇپپىسى ئەزالىرى يىغىنى» چاقىردى. يىغىندا ۋۇ جوڭ قاتارلىقلار قاتتىق تەنقىدلەندى. ئۇلارنىڭ جيەن بوزەننى قىستاپ ئىقرار قىلدۇرۇپ، جيەن بوزەن ئەر- ئايالنىڭ ئۆلۈشىنى كەلتۈرۈپ چىقىرىپ، مەركەزنىڭ سىياسىتىگە ئېغىر خىلاپلىق قىلغانلىقى ئەيبلەندى. ئۇلار«رەئىس ماۋزېدوڭ ھىمايە قىلىۋاتقان ئادەمنى سەنلەرنىڭ تۈرمىگە تاشلايدىغان نېمە سالاھىيىتىڭ بار؟ سەنلەر رەئىس ماۋزېدوڭغا ساداقەتسىزلىك قىلىشتىڭ!» دەپ قاتتىق تەنقىدلەندى.

ماۋزېدوڭ بىلەن جۇئېنىلەي ۋۇ جوڭنىڭ جيەن بوزەننىڭ ئۆلۈشىگە سەۋەبچى بولغانلىقىغا قارىتا: «خاراكتېرى ناھايىتى ئېغىر، قاتتىق بىر تەرەپ قىلىنسۇن!» دەپ تەستىق يازدى. ۋۇ جوڭ خىزمىتىدىن توختىتىلىپ تەكشۈرۈلدى، مەركىزىي مەخسۇس دېلو گورۇپپىسى «ليۇ شاۋچى، ۋاڭ گۇاڭمېي مەخسۇس دېلو گورۇپپىسى» دىكىلەرمۇ قاتتىق تەكشۈرۈلدى. بۇنىڭدىن باشقا، بېيجىڭ شەھەرلىك پارتكومنىڭ بىرىنچى قول شۇجىسى شيې فۇجىدىن تارتىپ مۇشۇ ۋەقەگە مۇناسىۋەتلىك بارلىق خادىملار، بېيجىڭ ئۇنىۋېرسىتېتىدا تۇرۇشلۇق تەشۋىقات ئەترىتى، بېيجىڭ ئۇنىۋېرسىتېتى تەشكىلىي گورۇپپىسى، قوغداش گورۇپپىسى، تارىخ فاكۇلتېتىدا تۇرۇشلۇق تەشۋىقات ئەترىتى قاتارلىق ئالاقىدار تەشكىللەرنىڭ ئاساسلىق مەسئۇللىرى ھەمدە تارىخ فاكۇلتېتى «جيەن بوزەن مەخسۇس دېلو گورۇپپىسى» نىڭ ئەزالىرى (ھەممىسى ئوقۇغۇچى) قاتارلىقلار ئىدىيىۋى جەھەتتە قايتا- قايتا تەكشۈرۈلدى.

جيەن بوزەننىڭ ئۆلۈمى مەركەز رەھبەرلىرىنىلا ئەمەس، بېيجىڭ ئۇنىۋېرسىتېتىنىڭ رەھبەرلىرىنىمۇ قاتتىق ساراسىمگە سېلىۋەتتى، ئەمما بۇ ئىش ئاممىغا ئاشكارىلانمىدى. بېيجىڭ ئۇنىۋېرسىتېتى رەھبەرلىرىنىڭ بۇيرۇقى بىلەن، لى دوڭخەي ئىسىملىك بىر تەشۋىقات خادىمى جيەن بوزەن ئەر- ئايالنىڭ جەسىتىنى بىر قارا ماشىنىغا سېلىپ بېيجىڭ ئۇنىۋېرسىتېتىدىن ئوغرىلىقچە ئەچىقىپ، باباۋشەن كۆيدۈرۈپ دەپنە قىلىش قەبرىستانلىقىدا كۆيدۈرۈپ دەپنە قىلىۋەتتى. «مەدەنىيەت ئىنقىلابى» ئاخىرلاشقاندىن كېيىن، مەزكۇر ماقالىنىڭ ئاپتورى جيەن بوزەننىڭ جەسەت كۈلىنى تېپىش ئۈچۈن ئەينى يىلى بېيجىڭ ئۇنىۋېرسىتېتىدا تۇرۇشلۇق ئازادلىق ئارمىيە، ئىشچىلار تەشۋىقات ئەترىتىنىڭ رەھبەرلىرىنى ئىزدەپ باردى. لېكىن ھەممىسى نېشىنى چايقاپ، «بۇ ئىشتىن خەۋىرىمىز يوق ئىكەن» دېيىشتى.

«يېڭى ئەقىل- پاراسەتكە دائىر ماقالىلەردىن تاللانما» ناملىق ژۇرنالنىڭ 2007- يىللىق 7- سانىدىن تەرجىمە قىلىندى
