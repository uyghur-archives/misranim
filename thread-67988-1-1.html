ئەسساللامۇ ئەلەيكۇم :مەنبەر ئەھلى مەن ھىكايەمنى باشلاتىن بۇرۇن ئاز تولا چۇشەندۇرۇش بىرىپ ئۆتەي .بۇھىكايىدىكى ۋەقەلىلىك دەرتمەن بىر يىگىتنىڭ بىشىدىن ئۆتكۇزگەن سەر گۈزەشتىسى مەن شۇنڭغا ئاساسەن بۇ ھىكاينى يازدىم .ئەمما ئەسىرىدىكى پىرسۇناژلارنىڭ ئسملىرى ئۆزگەر تىلدى .

قەھرىتان قىشنىڭ بىر كۇنى ئىدى .ئەمدىلائۆسمۈرلۇك ھاياتنى ئاخىرلاشتۇرۇپ ، ياشلىقنى گۈزەل قايناملىرىغا چوكۇش ئالدىدا تۇرۋاتقان  ئادىل بىلەن ئەلى .يولدا كىېتىپ باراتتى .ئۇلارنىڭ ئالدىنى ئىككى قىز كىلىۋاتاتتى .شوخ ،تەنتەك بو بالىلار بىر-بىرى بىلەن بەسلىشىپ ئۇقىزلار غا گەپ قلىماقچى بولدى .

ئادىل سول تەرەپتىكى قىزنى قەستەن سوقۇپ يەر گە يىقىتىۋەتتى .قىز تىرىكەن ھالدا،ساراڭمۇ ماۋۇ ،كۆزۇڭ يوقمۇ سىنڭ .قاراپ ماڭمامسەن .سوغۇقتا ئاران مېڭىۋاتسا ئادەم .دەپ غۇتۇلداپ كەتتى .ئادىل دەرھاللا ئۆزنىڭ قەستەن قىلغىنى چاندۇر ماسلىق ئۇچۇن .كەچۈرەڭ قىزچاق مەنغۇ قەستەن قىلمىغان ئەمما زە مۇشۇ ئايغىم بەك تىلىدۇ دەڭ .بولۇپمۇ سىزنى كۆرگەندە .قىز بۇ گەپنى ئاڭلا تېخىمۇ تىرىكتى .رەنجىپ سۆزلىگىنىچە كېتىپ قالدى .يىگىتلەر  ئۇلارنىڭ ئارقىسدىن قاراپ قالدى .

ئەتىسى ئادىل ئۇقىزنى يەنە ئىزلەپ كەلدى .ئەمما ئۇقىز كۆرۇنمەيتى .ئۇ تۇنۇگۈنكى جايدا ئۇقىزنى ساقلاپ تۇردى .راست دىگەندەك ئۇقىز بۇگۈن يالغۇز كېلىۋاتاتتى .

ئادىل قىزىرىپ تاتىرىپ دىگۈدەك ئۇنىڭ ئالدىغا كەلدى. قىز ئۇنڭغا قارىماي مڭىۋەرەردى .ئەمما ئادىل ئۇقىز غا يىتىشىۋالدى قىز دەرھاللا مەندىن يىراق مىڭىڭ ئايغىڭز تېيلىپ مىنى يەنە يېقىتىۋەتمىسۇن .دەپ كۇلۇپ كەتتى .ئۇلار شۇنداق قىلىپ چاقچاق قىلغاچ يوللا بىلە ماڭدى .ئۇقىزنىڭ ئسمى زوھرە بولۇپ ،ئۇقىز بۇيىل تولۇقسىزنىڭ 3.يىللىقىدا ئوقۇيتتى .ئۇ شۇ باھانە بىلەن ئۇنىڭ ئۆينىمۇ بىلىۋالغان ئىدى .

ئادىل ھەركۇنى دىگۈدەك زوھرەنى مەكتەپ ئالدىدا ساقلايتى .ئۇلار ھەركۇنى مۇشۇنداق بىلە مەكتەپكە باراتتى .بىللە مەكتەپتىن قايتاتتى .مەكتەپتىمۇ بىلە دەرس تەييارلايتى .شوخ ھەم ئۇماق قىز بىلەن تەنتەك بۇ يىگىتنىڭ مۇھەببىتى باشلانغان ئىدى .

كۇنلەر كەينىدىن كۇنلەر نىڭ ئۆتۇشى بىلەن ئۇلارنىڭ مەكتەپ پۇتۇرۇش ۋاقتىمۇ كەلدى .

بۇلارنىڭ بىشىنى ئاغرىتىۋاتقان مەسىلە بولسا قىز داۋاملىق ئۇقۇشىنى داۋاملاشتۇراتتى .ئەمما ئادىل مەلۇم سەۋەپلەر تۇپەيلى ئوقۇشتىن چېىكىنىشكە مەجبۇرى ئىكەنلىكىنى ئىتقاندا بۇلار قانداق قلىش كىرەكلكىنى بىلمىدى.

ئوقۇش  پۇتۇرۇش ۋاقتىمۇ كەلدى .باشقا ساۋاقداشلارنىڭ چىھرىدە خوشال كۈلكىلەر چىقىپ تۇراتتى .ئالدىرشچىلىق دەكە دۇكە ئىچىدە ئمتىھانلارمۇ بىرىلىپ بولدى .

يازلىق تەتىلمۇ كەلدى . ئۇلار كۇندە بولمىسمۇ كۇن ئارلاپ كۆرۇشۇپ تۇراتتى .تەتىلمۇ توشتى .زوھرە بولسائىلى تبي مەكتەپكە ئوقۇشقا ئۆتكەن ئىدى .   
ئۇياتاقتا يېىتپ ئۇقۇيتى .ھەپتىدە بىر ئويگەقايتاتتى.دەم ئېىلىش كۇنلىرى ئادىل ئۇقىز بىلەن كۆرۇشەتتى .ئۇلار باغچىلارغا ،دەريا بويلىرىغا باراتتى .ياتاقتىكى قىزلارئۇلارغا ھەۋەس قىلاتتى،ھەم ھەسەتمۇ قىلاتتى .بۇلارنىڭ مەكتىپىدە زوھرەنى ياخشى كۆردىغان بىر بالا بولۇپ ئۇنىڭ شۇنداق قوغلاشقان ئىدى ئەمما ،زوھرە ئادىل بولغانلىقى ئۇچۇن  ،ئۇنىڭ تەلىپىنى رەت قىلىپ تۇراتتى .شۇنداق كۇنلەرنىڭ بىرىدە ئۇلارنىڭ ئوتتۇرسىدا بىلىپ بىلمەي سوغۇقچىلىق پەيدا بولدى .ئۇلار كۇندە تېلفۇندا سوقۇشۇپ تۇراتتى .ئاخىرى بۇ تالاش- تارتىشلارنىڭ ئاخىرى ئايرىلىش بىلەن ئاياغلاشتى . `
بۇ پۇرسەتنى كۇتۇپ تۇرغان ئەركىن زوھرەگە تېخىمۇ يىقنلىشىقا باشلىدى .ئۇقىزئۇنىڭ مۇھەببتىنى قوبۇل قىلىدى .

ئەركىن ھەركۇنى دەررىستىن چۇشۇپ ،زوھرە بىلەن بىللە كەچلىك تەكراغا چىقاتتى ئۇلارنىڭ كۇنلىرى بىلە ئۇگۇنۇش قىلىش بىللە تاماق يىيشى بىلەن ئىدى .مەكتەپ ئىچىدە ھەممىسى ئۇلارغا ھەۋەس قىلاتتى .
بىركۇنى كەچلىك تەكرادىن چۇشۇپ زوھرە بىلەن ئەركىن پاراڭلىشپ ئولتۇراتتى.تۇيۇقسىز تىلفۇن سايراپ كەتتى.زوھرە تىلفۇن نى ئالغۇچە يۇرەكلىرى ئەنسىز سوقۇپ كەتتى .تىلفۇن ئىكرانىدا تۇنۇش نومۇر. كۆرۇندى .زوھرە جىدىليلىك ھالدا تىلفۇن نى ئالدى ئالمىسا ئەركىن خاتا چۇشۇنۇشى ئىنىق ئىدى.
ۋەي ،كىم؟دەپ سورىدى زوھرە
ۋەي ، بۇمەن ئادىل .قانداق ئەھۋالىڭ زوھرە ،؟
ياخشى كىمۇكەچۇرۇڭ  مەن سىزنى تونۇمايمەن .خاتا ئۇرۇپ قويۇپسىز .دەپلا بىسىۋەتتى .
تىلفۇن يەنە قايتا سايراپ كەتتى .قارىسا يەنە شۇنومۇرى .ئەركىن ھە ئالە ئالمايسەنغۇ ؟كىمكەن ئۇ راستىنلا خاتا ئۇرۇپتىمۇ ياكى خاتائۇرۇلۇپتىمۇ ؟ئالا دەيمەن .ئەكە بولمىسا مەن ئالىمەن .
ۋەي زوھرە چۇشەندۇرشۇمنى ئاڭلا بولامدۇ ؟سەندىن ئۆتۇنەي ئۇنداق قىلما ۋەي گەپ قىلە
ھە نىمە نىمنى چۇشەندۇر سەن ھە گىپىڭ بولسا ماڭا دەرە .
ئادىل كەچۇرگىن سەن كىمۇ مەن ئۇنڭ مۇھەببىتى دىگەندە ئەركىن سەنمۇ مەن مۇ مەن ئۇنڭ مۇھەببىتى .سەن قانداق مۇھەببىتى بولۇپ قالدىڭ ؟ھە سەن .......
تىلفۇن ئۇزۇلۇپ كەتتى .زوھرە شۇنچە چۇشەندۇرسىمۇ ئەمما ئەركىن ئۇنىڭ چۇشەنمىدى ئۇنى تاشلاپ قويۇپ كېتىپ قالدى.
ئەتىسى دەرسىخانىغا كىرگەندە زوھرە ئەركىندىن ئەھۋال سورىدى ئەمما       ئەركىن ئۇنى كۆرمىگەندەك گەپمۇ قىلىماي باشقا قىزلارغا چاقچاق قىلىپ ئولتۇرۋالدى.
ھەركۇنى زوھرە ئۇنىڭغا شۇنچە چۇشەندۇرسىمۇ ئەمما ئۇ ھىچ .ئىش بولمىغاندەك پەرۋاسىز ئىدى.كۇنلەر ئوتىۋەردى ، ئەركىن سىنىپىدىكى زوھرە نىڭ ئەڭ يىقىن دوستى بىلەن يقىنلىشىپ كەتتى .ئۇنىڭ بىلەن بىللە دەرسكە چىقتى بىللە ئايلاندى .زوھرە ئۆز ئۆزىگە ،ئادىلغا قىلغانلىرىغا ئىچ ئىچىدىن ئۆكۇندى .
ئادىلنىڭ تىلفۇنى سايراپ كەتتى .تىلفۇن ئىكرانىدىكى نومۇرلارنى كۆرۇپ يۇرۇكى ئويناپ كەتكەن ئىدى . ئۇ تىلفۇن نى ئىلىپلا ۋەي زوھرە ياخشىمۇ قانداق ئەھۋالىڭ ياخشى تۇردۇڭمۇ ؟ھەجەپ يوقاپ كەتتىڭ .
زوھرە ئۇنسىز يىغلاۋاتاتتى ،بۇنىڭ بىلىپ قالغان ئادىل . زوھرە . ھەي زوھرە ساڭا نىمە بولدى گەپقىلە يىغلاۋاتامسەن ؟ۋەي گەپ................ تىلفۇن ئۇزۇلۇپ كەتتى .\
زوھرە تىلفۇننى ئاۋاسزھالەتتە قىلىپ قويغان ئىدى .تىلفۇن كەلدى ئەمما ئۇ قاراپ قويۇپ ياستۇقى ئاسىغا قويۇپ قويۇپ ئۇنسىز يىغلاۋاتاتتى .   ئۇنڭغا نەچچە تىلفۇن نەچە ئۇچۇر كەلگىنىنىمۇ بىلمەي قويۇپ قويۇدى .ئۇقاچان ئۇخلاپ قالغىنىمۇ بىلمىدى .
ئەتىسى ئەتىگەندە ئۇنىڭ ئۇچۇرلىرىنى كۆرۇپ تىخىمۇ بەكىرەك يىغلاپ كەتتى .ئۇ دەرسكە چىقىشقا تەييارلىق قىلۋاتاتتى  .تىلفۇنى يەنە سايراپ كەتتى .ئۇ تىلفۇنى ئالدىدە يەنە يىغلاپ كەتتى ھەم شۇنداقلا ئادىل مىنى كەچۇر مىنى كەچۇر بولامدۇ ؟ساڭا يۇزكىلەمەيمەن .مىنى كەچۇر گىن بولامدۇ ؟
قارشى تەرەپتىن ساڭا نىمە بولدى زوھرە ؟ماڭا ئىنىق چۇشەندۇرۇپ قوي بولامدۇ ؟بىز كۆرشەمدۇق؟ ماڭا گىپىڭ بارمۇ ؟ بىز بىر پاراڭلىشپ باقايلى بولامدۇ؟
زوھرە بولدۇ مەن دەرستىن چۇشكەندە ساڭا تىلفۇن قىلاي .مەن ھارىز دەرىسكە ماڭدىم .ماقۇلمۇ ؟ بۇگەپنى ئاڭلىغان ئادىل خوش بولۇپ .ماقۇل تىلفۇنىڭ نى ساقلايمەن  . چوقۇم تىلفۇن قىل بولامدۇ ؟دەپ تىلفۇننى قويۇشتى .
ئادىل كەچ كىرىشىنى تۆت كۆزى بىلەن كۇتتى .كەچمۇ بولدى .ئادىلنىڭ تىلفۇنى سايراپ كەتتى .ئادىل خوشال بولغان ھالدا:ۋەي زوھرە دەرىستىن چۇشتۇڭ مۇ ؟مەكتەپنىڭ ئادىغا بارايمۇ؟ زوھرە بولدۇ سىنى مەكتەپنىڭ ئالدىدا ساقلايمەن .تىز بول ھە دەپ تىلفۇنى قويىۋەتتى .ئادىل تاكىس توسۇپ دەرھال تىبى مەكتەپكە قاراپ يول ئالدى .
مەكتەپ ئالدىدا زوھرە ئالدىلنى ساقلاپ تۇراغان ئىدى .ئۇلار خوشال ھالدا كۆرۇشتى بىللە تاماق يەپ كوچا ئايلاندى .
ئادىل زوھرە ساڭا نىمە بولدى ماڭا دە بولامدۇ ؟نىمە ئش بولغان بولسىمۇ ماڭا دە مەن ھەل قىلىپ بىرەي ماقۇلمۇ ؟
زوھرە نىمە دىيشىنى بىلمەي يىغلاپ كەتتى .ھەم بولغان ھەممە ئەھۋاللارنى چۇشۇندۇردى .
ئادىل بولارنى ئاڭلاپ بولدى .مەن با كوڭلۇڭنى يىرىم قىلما بولامدۇ ؟ بولدى ئۆتكەن ئشى ئۆتتى .مەن ھىچ كۆڭلۇمگە ئالمايمەن .بولدى خەير ماقۇلمۇ جىنىم ؟
ئۇلار يەنە ئەپىلشىپ قالدى ئەمما زوھرەنىڭ كۆڭلى تازا جايىدا  ئەمەس ئىدى .
ئۇلارنىڭ كۇنلىرى شۇنداق داۋاملاشتى .دەم ئىلىش كۇنلىر بىللە سەيلە قىلىش ،بىللە تاماق يىيىش ئىدى زوھرەنىڭمۇ كەيپىياتتى كۆتۇرلۇپ مۇ قالغان ئىدى .
زوھرەمەكتەپ ھاياتنى ئاخىرلاشتۇرۇپ خىزمەتكىمۇ ئۇرۇنلاشتى. ئۇ بىركۇنى كەچتە ئىشلىگەن ئىدى.ئۇنىڭ قورسىقى شۇنداق ئاشقان ئىدى .ئۇ سائەتكە قارىدى سىرتقا چىقاي دىسە يىرىم كىچە بولغان ئىدى .ئۇ ئادىلغا تىلفۇ ن قىلىپ ئادىل قوسقۇ م بەك ئاچتى .پەقەت بەرداشلىق بىرەلمەيۋاتىمەن .ماڭا تاماق ئەكەپ بىرە .دىگەن ئىدى .ئادىل بولىدۇجىنىم مىنى بىردەم ساقلاپ تۇر .مەن يلىىغىچە ئاپىرىپ بىرىمەن دەپ تىلفۇن نى قويۋەتتى . ئادىل سائەتكە قارىدى سائەت 12:30 ئۆتكەن ئىدى .
ئۇنىڭ بىشىدا مىڭ بىرغەم  بۇ كىچىدە نەدىن تاماق ئىزلەيمەن.بوپتۇ چقىىپ باقاي .دەپ سىرىتقا چىقتى .ئۇياققا ئايلىنىپ ،بۇياققا ئايلىنىپ بىر يەردە بىر توخۇ ساتىدىغان بىر كىشىنڭ يىنىغا بىرىپ توخۇدىن  بىرىنى ئىلىپ ئۇنىڭغا ئاپرىپ بەردى .
زوھرە شۇنداق خوش بولغان ئىدى .
ئادىل بىركۇنى ئۇنى تۇيۇقسىز خوش قىلۋەتمەكچ بولۇپ ،ئۇنىڭ يىنىغا باردى .ئۇ ئۇنىڭغا زوھرە ساڭا بىرگەپ باتتى .ساڭا قانداق دىسەم بولا
