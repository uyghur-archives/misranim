يوتقانغا قاراپ پۈت سۇنۇش
فىليەتون





ئىشتىن چۈشىدىغان ۋاقىت يېقىنلىشىشىغا ئەگىشىپ ئۆيىگە قايتىشقا ئالدىرۋاتقان خىزمەتداشلىرىم تاقەتسىزلەنگەنلكتىن سائەتلىرىگە قاراۋاتاتتى، ئەپسۇس ئۇلارنىڭ زارقىپ كۈتكەن شۇ ۋاقىتلىرىنىڭ كەلمەسلىكىنى مەڭگۇ كەلمەسلىكىنى تىلەپ چەكسىز خىيال دېڭىزىغا غەرىق بولماقتىمەن، قىسقىسى مەن ئىشتىن چۇشۇشنى، ئۆيگە قايتىشنى خالىمەيمەن.........
ئۇنداق دىسەم ئۇقۇرمەنلەرنىڭ مىنى ‹ خىزمەت سارىڭى،  خىزمەت مەستانىسى › دېيىشى تەبئىي، شۇنداق ئىلگىرى مىنىڭمۇ باشقىلاردەك كۇننىڭ بۇرۇنراق كەچ بۇلۇشىنى مەشغۇلىيەت ۋاقتىنىڭ توشۇشىشىنى بىتاقەتلىك ئىچىدە كۈتكەن ۋاقىتلىرىممۇ بولغان، ئۇ ۋاقىتلاردا مىنىڭ باي- باياشات بولمىسىمۇ ئەمما ئىللىق ئۆيۇم بار ئىدى، يىڭى ئۆيگە ئۆزەم ياخشى كۆرگەن مەھبۇبەم بىلەن كىرىپ ئۆزىمىزگە تەۋە ئىللىق ماكاننى قۇرغان ۋاقىتلار ئىدى... ئۇ ۋاقىتلاردىكى شىرىن بەخىت توغرىسىدا سۆزلەشنى ئارتۇقچە كۆرىمەن. ( چۇنكى دىمىسەممۇ تەسەۋۇر قىلالىشىڭىزغا ئىشەنچىم كامىل )
يۇرۋاتقان ۋاقىتلىرىمىزدا كەلگۇسىنى غايىۋانى تەسەۋۇر قىلاتتۇق، خۇددى كىيىنكى تۇرمۇشىمىز ئىككىمىز سىزغان لىنىيە بىلەن ماڭىدىغاندەك، خىيالىرىمىزنىڭ ھەممىسى روياپقا چىقىدىغاندەك... تاكى بىز مۇھاببەت دېڭىزىغا غەرىق بۇلۇپ نىكاھ كىمىسىگە چىقىۋالغانغا قەدەر ئىشلىرىمىز ناھايىتى ئوڭۇشلۇق بولدى، ئاتا- ئانام بىر تال ئوغلۇم دەپ يېزىغا سېلىپ بەرگەن ئۆيىمىز بىزنىڭ تۇنجى يىڭى ماكانىمىز بۇلۇپ قالغان ئىدى.... خىزمەت قىلىدىغان ئورنىمىزمۇ ئۆيىمىزگە بەكمۇ يېقىن بولغاچقا خىزمەتكە بىرىپ كېلىشىمىزگە قولايلىق، ئىككىلىمىز ئەتتىگەنلىك ناشتىنى ئۆيدە يەپ خىزمەتكە بىرگە باراتتۇق، مەن موتىسىكىلىت بىلەن ئۇنى خىزمەت ئورنىغا ئاپىرىپ قويغاندىن كىيىن ئۆزەمنىڭ خىزمىتىگە كىتەتتىم، كەچتە يەنە ئۇنى بىرگە قايتۇرۇپ ئەكىلەتتىم......تۇرمۇش، خىزمەت ئىشلىرىمىز بىر مەزگىل قىلىپقا چۇشۇپ قالغاندەك بىر خىل يۈنىلىشتە كىتىپ باراتتى...
جەمئىيەتنىڭ ئۈزلىكسىز تەرەققى قىلىشىغا ئەگىشىپ بىزنىڭمۇ تۇرمۇش ئىستىمالىمىزدا ئۆزگىرىش يۇز بىرىشكە باشلىدى، بارىغا قانائەت قىلىشتىن ھالقىپ مودا قوغلىشىش ‹ كىسىلى › بىزنىڭ ئائىلىنى ۋاباغا ئوخشاش يۇقۇملاندۇردى، بۇلۇپمۇ ئايالىمنىڭ ‹ ئىممىنۇت كۈچى › ئاجىزمۇ بىلمىدىم ئۇنىڭغا بۇ ‹ كىسەل › شۇنداق تىز يۇقۇپ ئېغىرلىشىپ كەتكەن ئىدى، دەسلىپىدە يېزىدا ئولتۇرۇشتىن زىرىكتى ۋە شەھەردىن بىنا ئۆي ئېلىش ‹ دورىسى › نى يىمىسەم بولمايدۇ دەپ ‹ كىسەل › كارۋىتىدا ئېڭىرغاچقا ئاتا- ئانامغا نىمە دىيىشنى بىلمەي ئاخىرى خىزمىتىمنى ئۆز يېزامدىن شەھەرگە يۆتكىدىم ۋە  ئۇلارغا ‹ بىرىپ كېلىشىم قولايسىز بۇلۇپ قالدى دىگەندەك سۇ كىرمەس باھانە سەۋەپنى كۆرسىتىپ ئۇلارنىڭ قوللىشىنى قولغا كەلتۇردۇم، ‹ دورا › قىممەت بولغاچقا ئاتا- ئانام ئۆز قولى بىلەن ئەھلىيا قىلىپ بەرگەن ۋە بىر مەزگىل بەخىت بۇشۇكىمىز ئىللىق ماكانىمىز بولغان ئۆينى ئەرزانلا سېتىۋەتتىم ، بانكىدىن قەرىز ئالدىم، كاملىغان قىسىملىرىنى ئاتا- ئانام پىيىنسىيە پۇلىسىدىن تىجەپ ياغقان پۇللىرى بىلەن تۇلۇقلاپ بەردى، شۇنداق قىلىپ ئاتا- بوۋامدىن تارتىپ سەھرادا ياشاپ كەلگەن مەن - ئەجداتلىرىمىنىڭ روھىنى‹ خوش ›  قىلىپ شەھەرلىك بۇلۇپ قالدىم... توۋا بۇ ‹ دورا › نىڭ ئۇنۇمىنى كۆرۇڭ، ئايالىم پۇتۇنلەي كىسىلىدىن ساقىيىپ قالغاندەك، بۇرۇنقى خوش – تەبەسسۇم كۇلكىلىرى ياڭراپ تۇراتتى... بۇ ‹ دورا › نىڭ ئۇنۇمى چۇقۇم ئۇزۇن مەزگىل بۇلىدۇ دەپ چوتنى خاتا سوقۇپتىمەن، بىر مەزگىللىك بەخىتلىك تۇرمۇشىمىزغا يەنە ئۈشۇك تەگدى، بۇ قېتىم ئايالىمنىڭ كېسلى بۇرۇنقىدىنمۇ ئېغىرلىشىپ ئەسەبىيلىشىش دەرىجىسىگە يەتتى، مانا ئەمدى ئۇنىڭ كىسىلىگە شىپالىق ئىزدەپ ‹ ماشىنا › دىگەن دورىنى ئېلىش ئۇچۇن ئىقتىساد توپلاۋاتىمەن....
يولدا كېتىۋاتىمەن، خۇددى بۇ شەھەردە مەنلا باردەك ئەتىراپىمدىكىلەرنىڭ مەۋجۇتلىقنى ئۇنتىغان ھالدا روھسىز كىتىپ بارىمەن... بىلىمەن ئۆيگە بارغاندىن كىيىن سۇيۇملۇك ئايالىمنىڭ ۋوجۇدىنى قاپلىۋالغان ‹ كىسەل › نىڭ ئازاپىدىن چىقىۋاتقان ئېڭىراشلىرىغا تەسەللى بەرمىسەم بولمايدۇ، لىكىن قانداق تەسەللى بىرىشنى بىلمەيمەن، چۇنكى ئۇ ئورۇنلاشتۇرغان ئىشلارنى بۇگۇن قىلالمىدىم....
ئىشكتىن كىرىشىمگە ئاشخانا ئۆيدىن يۇگىرىپ چىققان ئايالىم : قانداق بولدى ؟ بىرىپ تۇرىدىغان بولدىممۇ ؟ دەپ سورىدى ..
توۋا ئىلگىرى ئۆيگە قايتىپ كەلگىنىمدە ‹ ھارماڭ، سالامەت كېلىپسىز، چاي ئېچكەچ تۇرۇڭ، تاماق ھازىر تەييار بۇلىدۇ › دەيدىغان ئايالىم كىسەلجان بۇلۇپ قالغاندىن بىرى بۇلارنى تامامەن ئۇنتۇپ كەتكەن ئىدى...

- سىز ئۇنداق دىگەن بىلەن دوس يارانلىرىڭىز، مىنىڭ يېقىنلىرىمنىڭ ھەممىسى ئېلىپ بولدى، بىر يەرگە بارساق دائىم ئۇلارنىڭكىگە چىقىپ بارامدۇق ؟ سىزنىڭ قىزارمىغان بىلەن مىنىڭ بۇ يۇزۇمنى قويغىلى يەر تاپالمايمەن ... يېزىغا ئاتا- ئانىڭىزنى يوقلاش ئۇچۇنمۇ قانچە ماشىنىغا ئالمىشىپ بەك قېينىلىپ كەتتۇق..

- گەپ قىلمايسىزغۇ، ماشىنىنىڭ گېپى چىقسىلا سىز ئۇندىمەي ماڭا قارشىلىق بىلدۇرىسىز، بۇگۇن ئاخىرىقى چەك بىلىسىزغۇ، ھە يا ماشىنا ئېلىشقا تۇرۇڭ، يا مىنىڭ خېتىمنى بىرىۋىتىڭ.....
- ماقول، سىنىڭ دىگىنىڭىزچە بولسۇن !
شۇنداق قىلىپ بىر ئايدىن كىيىن 60 مىڭ يۇەنگە ماشىنىدىن بىرنى ئالدىم، ۋە ماي باھاسى يۇقىرى بولغاچقا 5 مىڭ يۇەنگە گاز قۇرۇلمىسى ئورنىتىپ يېقىلغۇ پۇلىنى تىجىدىم. ئايالىمنىڭ كەيپىياتىمۇ تۇراقلىشىپ ساقىيىپ قالدى... ئەتتىگەن ئايالىمنىڭ شوپۇرىدەك ئۇنى يېزىغا ئاپىرىپ قويۇپ ئاندىن شەھەرگە قايتىپ كىلىمەن، كەچتە ئۇنى ئالغىلى بارىمەن، ئاندىن كىيىن پويىزدەك سوزۇلۇپ كەتكەن رەتكە تىزىلىپ ماشىنىغا گاز قاچىلايمەن، نۆۋىتىم بەزىدە 2-3 سائەتتە، بەزىدە 5-6 سائەتتە كىلەتتى، گازنى قاچىلاپ بولغاندىن كىيىن قايتىپ كەلگىنىمدە ۋاقىت بىر يەرگە بىرىپ قالغاچقا ھېرىپ چارچاپ كەتكەن جىسمىمنى ئارانلا كارۋاتقا تاشلاپ ئۇخلاپ قالىمەن ۋە ئەتىسى روھسىز ھالدا ئورنۇمدىن تۇرۇپ شوپۇرلۇق مەجبۇرىيىتىمنى ئادا قىلىۋىتىپ ئىدارەمگە چاپىمەن....كۇنلەر شۇ تەرەققىدە ئۆتمەكتە، ئايالىم بىلەن بىرگە بۇلۇش ۋاقىتلىرىم گاز پونكىتىدا نۆۋەت كۈتۈش بىلەن ئۆتمەكتە، مۇناسىۋىتىمىزمۇ بۇرۇنقىدىن سۇسلىشىپ كەتتى، ئۆيدە ئازراق گەپ بولسىلا ھەر ئىككىلىمىز چېچىلىپ كېتىمىز، قىسقىسى ئۇنىڭ سۆزى ماڭا، مىنىڭ سۆزۇم ئۇنىڭغا ياقماس بۇلۇپ كەتكەن ئىدى.. بۇرۇن ماشىنا يوق ۋاقىتلاردا شەنبە،يەكشەنبە كۇنلىرى ئاتا- ئانىلىرىمىزنى يوقلايتتۇق، مانا ئەمدى ئايدا بىرەر قېتىم يوقلاپ قويغانلىرىمىزنى ئۆزىمىزچە چوڭ بىلىپ شەنبە، يەكشەنبە كۇنلىرى ئۇنى ماشىنغا چىقىرىپ شەھەر ئەتىراپىغا چىقىپ سەيلە ساياھەت قىلىمىز ۋە ئۇنىڭ ماشىنا ھەيدىشنى ئۇگۇنۇشىگە يىتەكچىلىك قىلىدىغان بولدۇم.....
بىر كۇنى گاز پونكىتى نۆۋىتىم كىلىپ ماشىنامنى قوزغىتاي دەپ تۇرسام، ئامانلىق ساقلىغۇچى كىلىپ – ماشىنىڭىزنىڭ گاز توڭ ئىسپاتى چىقىرىڭ
-        گاز توڭىنى قويغىلى ئۇزۇن بولمىدى، كىنىشكىسى تېخى چىقمىغان
-        ھازىر بىخەتەرلىككە سال قارىغىلى بولمايدۇ، شۇڭا كىنىشكە چىققاندا ئاندىن گاز قاچىلاڭ دەپ مىنى قايتىپ كېتىشىمنى بۇيرىدى.
4 سائەت نۆۋەت ساقلاپ بۇ مۇئامىلىگە ئۇچرىشىپ ئاچچىغىمنى كەلتۇرگەن بولسىمۇ ئەمما ئۇلار بىلەن تاكاللىشىشنىڭ مۇۋاپىق ئەمەسلىكىنى ھىس قىلدىم، چۇنكى بۇ ھەپتىدە ئىككى قېتىم ئۆلچەمسىز گاز توڭى پارتلاپ كېتىش ۋەقەسى بولغانلىقىنى ئاڭلىغاچقا ۋە ئۆزەم دۆلەت كادىرى بولغاچقا ئۇلارنىڭ بۇ ئىشىنى توغرا چۇشەندىم..
ئامالسىز ماي پونكىتىغا كىرىپ ماشىنغا ماي قاچىلاپ گاز قۇرۇلمىسى ئورناتقان زاۋۇتقا كەلدىم..بۇ يەردە مەندەق قىسمەتكە دۇچار بولغانلارنىڭ كۆپلىكىدىن يېرىم سائەتتىن كىيىن ماڭا گەپ قىلىش پۇرسىتى كەلدى، پوزۇر كېينگەن، ئۆزىنىڭ زەردار ئائىلىدىن ئىكەنلىكىنى ئىسپاتلاش ئۇچۇن 10 بارمىقىدا ئالتۇن ئۇزۇك تاقىۋالغان قاپىقى يامان قىز نىمە سەۋەپتىن كەلگەنلىكىمنى بىلگەندىن كىيىن
-        ئادەتتە گاز قۇرۇلمىسىنى ئورنىتىپ ئۇچ ئايدىن كىيىن ئاندىن كېنىشكىنى بىجىرىپ بىرىدۇ، شۇڭا ھازىر ئامال قىلالمايمەن.
-        ئەمىسە ئۇچ ئايغىچە ماشىنغا گاز قاچىلىغىلى بولمىسا قانداق قىلىمەن .
-        توۋا قارىسام كادىر سۇپەت تۇرىسىز، ماشىنا دىگەنگە مايمۇ قاچىلىغىلى بۇلىدۇ ئاكا، كېنىشكە چىقىپ بولغۇچە مايدا مېڭىپ تۇرۇڭ
-        ماي باھاسى يۇقىرى تۇرسا چىقىمى ئېشىپ كەتمەمدۇ
