    نېمىشقا مۇشۇ خىلدىكى تىمىلارنى يازىدىغانلىقىمنى ئۆزۈممۇ بىلمەيمەن ، باشقا بىرەر مەزمۇندا نەرسە يېزىپ باقاي دەپ كومپىيۇتېر ئالدىغا كىلىمەن يۇ لىكىن يازغانلىرىم  يەنىلا مۇشۇنداق باش ئاخىرى يوق پارچىلارغا بۆلۈنۈپ كىتىدۇ ،



       قىشنىڭ يېرىمى كەتتى ، يەنە يېرىمىغا بۇ جانلار چىدامۇ ؟  ئىسسىق ئۆيدىكىلەرگىغۇ ھىچ گەپ ئەمەس ئەمما غېرىبانە كەپىلەردە ئۈمىد قۇچاقلاپ ئولتۇرغان ، ئىشىك - دىرىزىلەرنىڭ يۇچۇقلىرىدىن كىرىۋاتقان ئاچچىق سوغۇقلارغا ھومىيىپ قاراپ تۇرغان بەندىلەر ئۈچۈن بۇ يېرىم قىش يېرىم يىلدەك تۇيۇلىدۇ . تازا قېلىن كىيىنىپ قار ئېتىشىپ ئويناۋاتقان چاغلىرىمىزدا مۇزلاپ  قىزىرىپ كەتكەن قولىنى يۇشۇرىدىغان يەر تاپالماي قىلىۋاتقان كىشلىرىمىز بىزگە زوقلىنىش نەزىرى بىلەن ئەمەس ھەۋىسى كىلىپ قارايدۇ . كوچىدا كىتىپ بېرىپ كىشلەرنىڭ ئەرمەك ھايۋانلىرىغا كىيىم كىيگۈزۈپ قويغىنىغا قاراپ كۈلۈپ قويىمەن ، قەھىرتان قىشتا كالتە يوپكا كىيىپ يوتىسىنى ئېچىپ قويغانلارغىمۇ كۈلۈپ قويىمەن ، ئەمما  قىشقا لايىق كىيىم كىيەلمىگەنلەرنى كۆرگەندە كۈلەلمەيمەن . ئىچىم ئاچچىق بولىدۇ ، قېرىنداشلىرىمىزنىڭ ئاچ- يالاڭاچ يۈرگەنلىرىكىنى كۆرۈپ تۇرۇپمۇ  مىھىرلىك قولىمىزنى سۇنمىغانلىقىمىزغا ئېچىنىمەن ، ئۆزۈمنىڭ گۇناھقا پېتىۋاتقانلىقىمنى ھىس قىلىمەن .



كۆكتە ئاق قار قانات قېقىپ ئوينايدۇ ئۇسسۇل ،

يەردە  مىھىر  قازا  تاپتى  ،  توڭلايدۇ   ئۇيغۇر.





      كونا يىلنىمۇ ئۆز يولى بىلەن ئۇزۇتۇپ قويدۇق ، كىملەرگە قەرىز بولدۇق ، كىملەرنى قەرىز قىلدۇق بۇنىڭ ھىسابىنى يىڭى يىلدا بىر بىرلەپ ئالىمىز ، يىڭى يىلغا قانچىمىز مەس كىردۇق ، قانچىمىز ساق بۇنىڭغا قۇرۇقدالغان بوتۇلكىلار بىلەن ئا يارەي! گە توۋلىغان تاڭلار جاۋاب بەرسۇن . ئەمەليەتتىغۇ يىل ئۆرۈلسە ھىچقانداق نەرسە ئۆزگەرمەيدىكەن ، قوياش يەنە شۇ شەرىقتىن چىقىدىكەن  ، كىچە بولىدىكەن ، قىش يەنىلا سوغۇق پىتى تۇرىدىكەن ، ماشىنىلارغا بېنزىن قۇيۇپ ماڭدۇرىدىكەن ، ئەمما.....  يىڭى يىل كەلسە كۆپ نەرسىلەر ئۆزگىرىپ كىتىدىكەن . يىڭى -يىڭى نىشانلار پەيدا بولىدىكەن .  يېشىمىزغا بىر ياش قېتىلىدىكەن  ، نۇرغۇن ياشلىرىمىز ئەمدى 18 ياش دەپ ئاتالمايدىكەن ، رېستۇران ، تانسىخانىدىكى جوزىلارنىڭ باھاسى بۇرۇنقىدىن قىممەتلىشىپ كىتىدىكەن راست. يەنە بەلكىم يىراق سەھرادىكى ئاپامنىڭ چىكىسىگە يەنە بىر نەچچە تال ئاق چاچ قېتىلغاندۇ ، دادامنىڭ پىشانىسىدىكى تۈرۈلمىلەر ئارسىغا يەنە بىر سىزىقچە پەيدا بولغاندۇ . بەلكىم...



ئون سەككىز ياش ، ئۆتكەن يىلدا قالدى ياش تۆكۈپ،

خەيىر ياشلىق ، ئۇخلاپ قاندىم ، كىتەي دەس تۇرۇپ.





  مەندە  مۇنبەر خۇمارى بار ، بەلكىم سىزدىمۇ باردۇ ، خىزمىتىم قانچە ئالدىراش بولسىمۇ مۇنبەرنى بىر ئېچىپ كۆرىۋالىمەن خۇددى مەن كۆرمىسەم بولمايدىغاندەك ، يېقىندىن ياقى نەگىلا قارىسا پىسخىك سوئاللار تاشلانغان تىمىلار  < ئەتىڭىزنى دەپ بىرىمەن > ، < سىزنى كىمنى ياخشى كۆرىسىز؟> .< ئۆگۈنلۈككە پۇل تېپىۋالامسىز؟> دىگەندەك تور پالچىلىرىنىڭ نەلەردىن تىرىپ كەلگەن ، كىشىگە بىر تىيىنلىك پايدىسى يوق جۆيلۈشلىرى ... چۈشىۋاتقان ئىنكاسلارنىڭ كۆپلىكى ھاي.. ھاي ... ساناپ بولغىلى بولمايدۇ . ئېغىزدا ئاللاھنىڭ بۇيرىقىسىز قىل تاۋرىمەيدۇ دەپ تۇرىدۇ يۇ شۇ قىلنى تەۋرەتمەكچى بولىشىدۇ ،  نىكاھ ئوقۇلىدىغان چاغدا قارىم كەلىمە شاھادەتنى ئوقۇپ بىرىڭ دىسە تىلىنى چايناپ تۇرغانلارنىمۇ كۆردۇق ،  شۇنى يوللىغاننىڭ ئورىنىغا  بىر نەچچە سۈرە يوللىغان بولسا نەچچە يۈز ئىنكاسنىڭ ئورىنىغا نەچچە يۈز كىشىنىڭ دۇئاسىنى ئالغان بولاتتى ئەمەسمۇ . كەلگۈسىنى بىلىشكە ئالدىرمايلى ، كەلگۈسىگە يۈز كىلىش ئۈچۈن تىرىشايلى . بىز كىيىم -كىچەك ساتىدىغان دۇكانلاردىكى جانسىز مودىل ئەمەس بەلكى ئەۋلادلارغا < كىيىم > تىكىدىغان تىككۈچى...



دىل ياش تۆكسىمۇ تۆكسۈن جايىدا ،

قۇلاق يېقىپ قويايلى مىللەت زارىغا .





        خىلى كۈنلەر بولۇپ قاپتىكەن ، غۇلجىدىكى خەنزۇ بازىرىنى ئايلىنىپ باقمىغىلى ، بۈگۈن ئاتايىتەن قەدىردان ، تىرىشچان كىشلىرىمىزنى كۆرگىلى باردىم ، 5-6 يىلنىڭ ئالدىدا ئۆزۈم كىچىك ھارۋىنى سۆرەپ يۈرۈپ سودا قىلغان جايغا كىلىپ توختىدىم ، بۇ يەردە بىر نەچچە ئاكىلىرىم بولىدىغان چىھرىدىن نۇر يېغىپ تۇرىدىغان ، بەزى چاغلاردا ماللىرىنى ماڭا قارىتىپ قويۇپ تاماقلانغىلى كىتىپ قالاتتى ،  ھاۋا سوغۇق بولغاچقىمۇ ياكى بۇ بازار رەتلەنگەچكىمۇ ئۇ نۇر يۈزلۈكلەرنى ئۇچىراتمىدىم ،  تىرىكچىلىك ئەمەسمۇ ياكى ئادەم كۆپىرەك ئولىشىدىغان يەرلەرگە ماللىرىنى سۆرىشىپ كىتىپ قالغاندۇ ،  ھايات بىردە ئىسسىق بىردە سوغۇق ،  تەر ئاققۇزسا بىر كويغا ئېچىملىك ئىچمەيدىغان ، توڭلاتسا نەچچە ئون سوم خەجلەپ ئىسسىق شورپا ئىچمەيدىغان شۇ بەندىلەر ئائىلىنى دەپ ياشاپ كىتىپ بارىدۇ ، خەنزۇ بازىرىدىن سەل يۇقىرغا  ئۆرلىسەك قايماقچى ئانىلىرىمىز قاتار ئولتۇرشۇپ كىتىدۇ ، مىنىڭ تۇتقان ئۆيۈم بولمىسىمۇ قايتاشىمدا  شۇ ئانىلارنىڭ قايماقلىرىدىن ئاز تولا ئېلىپ كەتتىم . بالدۇراق سېتىپ بولسا ئۆيگە بالدۇراق قايتارمىكىن دەپ ئويلىغاچ....  ئۆتكەن يىلدىمۇ مۇشۇ گەپنى قىلغان ئىدىم ،  شۇنداق پارچە -پۇرات ئوقەت قىلىدىغانلارنىڭ نەرسىلىرىنى ئېلىپ قويۇڭلا ، باھا تالاشماڭلا ، ئۇيغۇرغا قىلغان ياخشى ئىشىڭىز بولۇپ قالسۇن ......



كۈن ئۆتمەكتە ، ئاي ئۆتمەكتە ، يىلمۇ ئۆتمەكتە ،

ياشاش ئۈچۈن ئاقباش ئاتا   بىلەك  تۈرمەكتە .





بۇ يەرنىڭ سەھىرى جىمجىت دەپ بىر كىنۇ بولىدىغان ، ئىسمىنى كۆپ ئاڭلىغىنىم بىلەن مەزمۇنى ئېسىمدە يوق . قانداق يەردۇ ئۇ جىمجىت بولىدىغان ، ئەمما بىزنىڭ بۇ يەرنىڭ سەھىرى قالايمىقان . ئۇۋىسى چۇۋۇلغان بىنالاردىن ئالدىراش چىقىۋاتقان كىشلەر تاكسى تورىيالماي سۇغۇقتا تىتىرەپ تۇرۇپ كىتىدۇ ،    كىچىچە ئۆيگە كىرمىگەن ياش باللىرىمىز نەلەلەردىندۇ چىقىشىپ ئۆيلىرىگە ئالدىرىشىدۇ ، ئونىۋىرسىت ياتاق بىناسى ئالدىدا ئىشىكنى ئاچمامسە دەپ قېرى خەنزۇنى قاقشىتىدۇ ، كۆزلىرىدە لىق ئۇيقۇ  بەلكىم كىچىدىن ياقى ئوينىغان تور ئويىنىنىڭ سەۋەبى بولسا كىرەك .

بىزنىڭ بۇ يەرنىڭ سەھىرى قالايمىقان ،نەچچە كۈندىن ياقى يىڭى -يىل ئويناپ چارچاپ قالغان ئاتا-ئانىلار چالا تويغۇزۇپ ئۆيدىن چىقىرىپ قويغان بالىلار بىر قولىدا تەييار يىمەكلىك يەنە بىر قولىدا تىخى تولۇق كىيىپ بولالمىغان كىيىمى بىلەن بىلىم بۇلىقىغا ئالدىراش يۈگرىشىدۇ . بىنا ئۆينىڭ بالكونىدىن بېشىنى چىقىرىپ :" چۈشتە مەكتەپنىڭ ئالدىدىن تاماق يە ھە ، بىز ئۆيدە يوق " دەپ ۋاقىرغان ئانىنىڭ ئاۋازى پۇتۇن ئولتۇراق رايۇننى بىر ئالىدۇ ،

   بىزنىڭ بۇ يەرنىڭ سەھىرى قالايمىقان ، سۈت  ساتقۇچى بالىلار ئۈچ چاقلىق موتۇلىرىنى بىنا ئالدىغا توختىتىپ ۋاقىرايدۇ ،مىۋە -چىۋە توپ تارقىتىش بازارلىرىدا ماشىنىدىن يىڭى كەلگەن ماللارنى بۇلاپ -تالاپ دىگۈدەك چۈشۈرۈپ  ھارۋىلارغا بېسىشىدۇ ، بىر -بىرىنى تىللىشىپمۇ كىتىدۇ تىخى ، كىچىك كوچىلاردىكى ئاشخانىلاردىن ئوزۇقلانغان بويتاق ، يىرىم بويتاق ، ئۆزى بولىۋالغان بويتاقلار ئېغىز بۇنىلىرىنى ئېتىشىپ چىقىپ كىلىشىدۇ ، پارچە تىجارەت قىلىدىغانلار ئېغىر ھارۋىلىرىنى سۆرىشىپ بازارغا مېڭىشىدۇ ...

   شۇنداق بۇ يەرنىڭ سەھىرى قالايمىقان . بىراۋلار  ياش تۆكىدۇ ، بىراۋلار ئىسسىق يوتقانلاردىن ئېرىنچەكلىك بىلەن سۇغۇرلۇپ چىقدۇ ، بىراۋلار  چاڭ -توزانلار ئارسىدا قولىدا سۈپۈرگە تۇتقان ھالدا كۈندۈلۈك خىزمىتىگە كىرىشىپ كىتىدۇ ، يەنە  بىراۋلار ئالماي -تالمان ۋاقىراپ ئەتتىگەندىلا  قېقىۋالىدۇ .


تاڭ ئۆز نۆۋىتىدە ئېتىۋىرىدىكەن ،

ئىنسان ئۆمىرىنى سېتىۋىرىدىكەن...




شۇنداق ، يەرشارىنىڭ بىر بۈرجىگىدە بىر توپ كىشلەر كۈلىۋاتىدۇ ، يىغلاۋاتىدۇ  ، ئۇخلاۋاتىدۇ ، ياكى يۈگرەۋاتىدۇ ، ئىلىم تەھسىل قىلىۋاتىدۇ  ياكى ئۈگىتىۋاتىدۇ ، شۇنداق ، يەرشارىنىڭ مۇشۇ بۈرجىكىدە بىر قەۋىم ھاياتلىق ئۈچۈن كۆرەش قىلىۋاتىدۇ ، ئۇيغۇر ئەتىسى ئۈچۈن ، كەلگۈسى ئۈچۈن ، ئەجدادى ، ئەۋلادى ئۈچۈن بىلەكلىرنى تۈرۈپ نىشان ، ئارزۇ -ئارمانلىرىنى ئىگىز كۆتۈرۈپ تىرىشىۋاتىدۇ ، بىر بىرىگە مىھىر يەتكۈزىۋاتىدۇ .......

ھەيرانمەن


سائادەت ساتتار ئايتۇغ


ھەسرەت قايناپ تۇرسا ھەردەم يۈرەكتە،
زورلاپ ساختا كۈلگەنلەرگە ھەيرانمەن.
غەملىرىنى دەپنە قىلىپ تىلەككە،
يەنە مەغرۇر يۈرگەنلەرگە ھەيرانمەن.
ئەۋلاتلىرى قارچۇقلاردا نۇرلانسا،
ئۈمىتلەنمەي كۆنگەنلەرگە ھەيرانمەن.


مەسۈم كۆزدىن ياشلار ئاقسا تەلمۈرۈپ،
پەرۋا قىلماي تۇرغانلارغا ھەيرانمەن.
ئارزۇلارنى مەئىشەتكە تاپشۇرۇپ،
ئىسراپخورلۇق قىلغانلارغا ھەيرانمەن.
ھەقىقەتنى شۇئار توۋلاپ ئاغزىدا،
ئوڭ كۆزىنى يۇمغانلارغا ھەيرانمەن.

。

سورۇنلاردا ۋەدە قىلىپ،كېرىلىپ،
ئۆز لەۋزىدىن تانغانلارغا ھەيرانمەن.
ھاياتلىقتا نەپسىگىلا بېرىلىپ،
غورۇر،ۋېژدان....ساتقانلارغا ھەيرانمەن.
ماغدۇرى بار،جۈرئىتى يوق...ئىزىلىپ،
خىيال سۈرۈپ ياتقانلارغا ھەيرانمەن.

。

لاپ گەپ قىلىپ ئاسمان تىپىپ ئاغزىدا،
گەپلا ساتقان نوچىلارغا ھەيرانمەن.
ۋەتەن ئۈچۈن بىر تىيىنلىك ئەجرى يوق،
(شەھەر ئالغان)پوچىلارغا ھەيرانمەن.
چىقالماستىن ئوڭشاپقىنە شوتىغا...
ئۇچقانلارنى(كوچىلار)غا ھەيرانمەن.

。

كۈن ئۆتسىلا كەلگۈسىنى كۆزلىمەس،
تەلۋىلەرگە قەھرىم بىلەن ھەيرانمەن.
تىلى تۇرۇپ ھالاللىقنى سۆزلىمەس،
ناكەسلەرگە جەھلىم بىلەن ھەيرانمەن.
ئالىملىرىم باش چۆكۈرۈپ(مىللەت)دەپ-
ياشاۋاتسا تەلىم بىلەن ھەيرانمەن.

。

ھەيرانىمەن،ھەرقەدەمدە ھەيرانمەن،
ھەيرانىمەن ساقىيارمۇ بېشىم دەپ.
ھەيرانىمەن كۆزلىرىمگە ھەيرانمەن،
ھەيرانىمەن قاچان تۈگەر يېشىم دەپ.
ھەيرانىمەن جېنىم دوستلار ھەيرانمەن،
ئويلاڭلارچۇ بۇنى ئورتاق غېمىم دەپ؟
ھەيرانىمەن بولۇپ قالماي ھەيرانمەن،
كەتسەڭلەرگەر (مىنىڭ نىمە ئىشىم)دەپ.

.

ھەيرانلىقتا قاتتى ئويلاپ باشلارمۇ،
ئۆملۈك بىلەن پەخىرلىنىپ ياشايلى.
مەيۈسلۈكتە ئېقىۋەرمەي ياشلارمۇ،-
ئەجرىمىزدىن ئەسلىمىزنى تاپايلى.
نۇرلار چاچسۇن ئەجرىمىزدىن تاشلارمۇ،،
تۆھپىمىزدىن كۆكتە پەرۋاز قىلايلى
