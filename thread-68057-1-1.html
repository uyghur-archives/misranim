  ھىكىمەت ۋە مۆجىزە                                                            




            

                                                         
  
ئەسسالامۇ  ئەلەيكۇم :
    مەن ھەر دائىم كۆپۇنچە ھاللاردا نۇرغۇن كىشىلەرنىڭ ئادەمنى ھەيران قالدۇردىغان بىرەر پەلقۇۋاددە ئىش-ھەركەتلەرنى كۆرگەندە ياكى ئاڭلىغاندا شۇ ئىشقا قارىتا مۆجىزە !!! بولۇپتۇ جۇمۇ ئاداش .دىگەنلىرىنى ئاڭلاپ  قالىمەن .
  مەن نۇرغۇن يىللاردىن بۇيان مۇشۇ ھىكىمەت ۋە مۆجىزە مەسىلىسىنى ئوچۇق ۋە چۇشىنىشلىك بولغان بىرەر ياخشى نەتىجىگە ئىرىش ئۇچۇن ، ئىزدىنىش جەيرانىدا  يەكۇنلەپ چىققان ھىكىمەت ۋە مۆجىزە دىگەن تىما ئارقىلىق  تور ئىكىرانى ئالدىدىكى ھەر ساھە ،ھەر كەسىپ دوسلار بىلەن ئورتاقلىش ئۇچۇن، مىسرانىمغا ئەزا بولغاندىن بۇيان تۇنجى بولۇپ قەلەم تەۋىرتىپ ئولتۇرپتىمەن . يازما جەريانىدا ئانداق-مۇنداق خاتالىقلاردىن ساقلىنىپ بولالماسلىقىمغا ئىشەنچىم كامىل، شۇڭا كەڭ دوسلىرىمنىڭ يازما ئىچىدىكى چۇشۇنىكسىسز جۇملە،بايان ،تىما يىزىش قائىسىدىگە خىلاپ بولغان خاتالىقلارنى سەمىمىيلىك بىلەن قىسقا ئۇچۇر ،ياكى ئىنكاس ئارقىلىق مەلۇم قىلىشىنى سورايمەن .

  مەن بۇ ھىكىمەت ۋە مۆجىزە دىگەن ئىككى سۆزنىڭ ،ئىزدىمايى تۇرمۇشىمزدا قانداق ئوررۇن تۇتىدىغانلىقى ۋە بۇلارنىڭ بىز ئىنسانلار  ئوتتۇرسىدا ، قانداق مۇناسىۋەت ئۇسۇلى ئارقىلىق  ئۆزىنىڭ  مەۋجۇتلىقىنى ساقلايدىغالىقىنى چۇشۇنىپ يىتىش ئۇچۇن جاۋاپ ئىزدەپ ،مۇبارەك قۇرئان كەرىمنى ئوقۇش جەيرانىدا ، ھەر دائىم  بىر قانچە ئايەتتنى ئوقۇشنى داۋام قىلىپ بولغاندىن كىين ،<.....ئاللاھ تولىمۇ ھىكىمەت بىلەن ئىش كۆرگۇچىدۇر  > دىگەن ، بۇ قىسقا ئايەت بىلەن ، < ناگاھان ئۇلارغا ( كۇففارلارغا )  بىرەر  مۆجىزە كەلتۇرسەك ..........>  دىگەن ئايەتنىڭ، ئايرىم-ئايرىم ئورۇن ئالغانلىقىنى ھىس قىلىپ قالدىم  .

      مەن جەمىيەتتىكى كۆپ قىسىم كىشىلەرگە  تازا دىققەت قىلسام،  ئازىراقلا ساغلام ئەقلى بار  ھەر قانداق بىر مۇسۇلمان كىشىنىڭ ،  ھىكىمەت بىلەن مۆجىزنىڭ بىر-بىردىن ئىنتايىن زور دەرىجىدە پەرىق قىلدىغانلىقىنى ھىس قىلىپ يىتەلىسىمۇ، بىراق كۆپۇنچە كىشىلەر  كۆپ ھاللاردا ئاللاھ ياراتقان شەيئىلەرنىڭ ئۇستىدە چوڭقۇر ساغلام پىكىر قىلىپ باقمايلا ، بۇ ھىكىمەت ۋە مۆجىزە دىگەن ئىككى ئۇقۇمىنى ئاسەن بىر مەنا دائىرىسىدە  چۇشىنىۋالدىغانلىقىنى تولا بايقاپ قالدىم  .

   قۇرئادندىكى بايانلارغا ئاساسلانغاندا ،ھىكىمەت بىلەن مۆجىزە بىر-بىرى بىلەن تۇتقان يولى ئوقۇملىنىش شەرتى بىلەن زور پەرىقلىنىپ تۇرىدىكەن، ئۇنداق  بولسا ھىكىمەت دىگەن نىمە ؟ مۆجىزە دىگەن نىمە ؟   


    جاۋاپ -ھىكىمەت دىگىنىمىز، بىز كۆرۇۋاتقان ۋە ھەركىتىدىن شۇ شەيئىنىڭ بارلىقىنى ھىس قىلۋاتقان ( جانلىق-جانسىز ) شەيئىلەرنىڭ ، ۋۇجۇتقا كىلىش جەيرانى ۋە ھەركەت قىلىش جەيرانى، شۇ شەيئنىڭ ۋۇجۇتقا كىلىش ۋە ھەركەت قىلشتا ئۆزىگە لازىم بولىدىغان  ، ھەر خىل جانلىق-جانسىز  ماددىلارنىڭ بىر-بىرى بىلەن سەۋەپ-نەتىجە ئارقىلىق  بىر-بىرى بىلەن  مەلۇم بىر قانۇنىيەت ئاستىدا، تەرتىپلىك ھەركەت قىلشتىن ئىبارەت جەريانىدا، سىرىتتىن  ۋە شۇ جىسىمنىڭ ئىچكىرى قىسمىدىن  ئىشلەنگەن ،بارلىق تىرىشچانلىقلار   ھىكىمەت دەپ ئاتىلىدۇ .

  ئاللاھ  سۇرە زەمىر 63 ئايەتتە مۇنداق دەيدۇ .
...ئاسماننى ۋە ئۇلارنىڭ ئارىسىدىكى جانلىق-جانسىز شەيئىلەرنى بىكار ياراتمىدۇق ؟!  ئۇلار ( يەنى ھاماقاتلەر ) بۇلار ( ئاسمان-زىمىن ئارىسىدىكى جانلىق-جانسىز شەيئىلەرنى بىكار يارىتىلغان دەپ ئويلاش ) كاپىرلارنىڭ گۇمانىدۇر . كاپىرلارغا دوزاق ئوتىدىن ۋاي !!!  

   بۇ ئۇيغۇرچغا تەرجىمە قىلىنغان قۇرئادندىكى تەرجىمە بولۇپ پەقەت ئەسلى ئەرەپ تىلىنىڭ ئۆزىنىڭ تىپىك بولغان ھەرىپ مەنىسىنىلا، ئۇيغۇرچىدىمۇ زاغىرا تىل بىلەن  تەرجىمە قىلىنغان بولۇپ ، بۇ تەرجىمىمۇ ھەرگىزمۇ خاتا ئەمەس .بىراق بۇ ئايەتنىڭ زاغىرا تەرجىمىسىدىن قانداقكى بىر  مۇسۇلمان كىشى ئەلۋەتتە ،ئاسمان-زىمىن ئارىسىدىكى جانلىق-جانسىز شەيئىلەرنىڭ ئاللاھ تەرىپىدىن يارتىلغانلىقىغا شەك قىلمىسىمۇ ، بىراق بىرەر ئاستىرونومىيە ياكى بىئولوگىيە تەتقىقات ئورگانلىرى تەبىئەتتىكى بىرەر ھادىسلەر توغۇرسىدا بىرەر نەتىجىنى ئىلان قىلسا دەھاللا  نەتىجىنى قوبۇل قىلالماي، شۇ ئىشلار ئۇستىدىن چىقىرىلغان نەتىجىگە سوغۇق مۇئامىلە تۇتىۋالىدۇ .

      بۇلار ئاللاھنىڭ بۇ ئايەت ئارقىلىق، ئۆزلىرى ياشاۋاتقان يەر شارى ۋە  ئالەم بوشلۇقى ئىچىدىكى شەيئىلەرنڭ ھەركەت پىرىنسىپى، يۆتكلىش ،ئۆزگىرىش ھالىتى.......قاتارلىق ئۇسۇللىرىدىن ،ئۆزىگە پايدىلىق بولغان ئىلىملەرنى قىزىپ چىقىرىپ ،ئۆزىنىڭ زىمىندىكى بىردىن-بىر  ھوقۇقلۇق ،يەنى خەلىپە بولۇشتىن ئىبارەت شان-شەرەپ رامكىسىنى  تىخىمۇ تاكامۇللۇققا يۇكسەلدۇرسۇن دىگەن مەنادا نازىل قىلغان ئايەتنى ،پەقەت ھەر كۇنى كەچلەردە زىرىكىپ قالغاندا كوچا چىراغلىرىدىن ھوزۇر ئالغاندەك ،پەقەت قۇرئانى كۇنگە بولۇپ ئوقۇغان بەت سانىنىڭ، رەقەمىلىك ئۇنۇمدارلىقى يۇقىرى كۆتۇرۇش ئۇچۇن  ئوقۇپ قويۇشتن نىرىغا ئۆتەلمىدى .

      بىز ھەر ساھە ھەر كەسىپتىكى دوسلارنىڭ   چۇشىنىشىگە قويلايلىق بولۇشى ئۇچۇن ، ئەسلى ۋۇجۇتقا كىلىش مەنبىيى ئۇ ئالەمدىن بولغان بىر جانلىقنى ،  ۋە  ۋۇجۇتقا كىلىش مەنبىيي بۇ ئالەم :يەنى يەر شارىدا ياسالغان  بىر جىسىمنى ۋە شۇنىڭ بىلەن بىلەن بىرگە ،شەكلىنى كۆرگىلى  بولىدىغان، بىراق ھەركەت قىلىش پىرىنسىپىنى كۆرگىلى بولمايدىغان بىر جىسىمنى مىسال ئىلىش ئارقىلىق ، ھىكىمەتنىڭ مەناسىنى روشەنلەشتۇرۇپ باقايلى  .

مەسىلەن يۇقارقى ئۇچ شەيئىدىن ھەر بىرىگە بىردىن مىسال ئالساق  
1-ئىنسانلارنىڭ ئەۋلات قالدۇرشى .
2-ئايرۇپىلانىڭ ئۇچۇشى .
3-سىمىز تلىفۇنلار ئوتتۇرسىدىكى ئالاقە ...........................................................قاتارلىقلارغا ئوخشاش ئىشلار ھىكىمەت دائىرىسىگە كىرىدۇ دەپ پەرەز قىلايلى .

   ئەمدى بىز  1- مىسالنى يەنى ئىنسانلارنىڭ ئەۋلات قالدۇرشىنى يىغىنچاقلا  قىلىپ ئىلمى تەھلىل قىلىپ باقايلى  .

   ئەر-ئايال بىرگە بولۇش جەريانىدا ، ئەر-ئايالنىڭ  ئىسپىرمىسى بىر-بىرگە  قوشۇلىدۇ، بۇ قوشۇلغان ئىسپىرما ئايالنىڭ تۇخۇم توشۇش نەيچىسى ئارقىلىق  ئايالنىڭ بالىياتقۇسىغا ئورىنلىشىدۇ  ۋە بۇ ئىسپىرما ئايالنىڭ بالىياتقۇسىدا تۇرۇش جەيرانىدا ،بالىياتقۇغا تۇتاش بولغان نەيچىدىن، ئايالنىڭ ھەر قايسى ئەزالارىدىكى تومۇرلار ئارقىلىق ئىقىپ كەلگەن كالتېسى، تومۇر، ۋىنتامىن ،كىراخمال ،ئاقىسىل ........قاتارلىق نەچچە يۇزگە يىقىن ماددىللاردىن ئۆزىگە قۇۋەت ئىلش شەرتى ئاستىدا،  بۇ ئىسپىرما پەيدىن-پەي يىرىڭغا ( يەنى سىرىق قىزغۇش قان ئارلاش بەلغەمگە ھالىتىگە ) ، ئاندىن بوتقا  ھالەتتىكى قويۇقلۇق دەرىجىسى يۇقىرى بولغان قانغا ،ئاندىن لەش ھالىتىدىكى ئىپتىدايى ھامىلە شەكلىگە ........بارا-بارا بوۋاققا ئايلىنىپ، ئانىنىڭ بالىياتقۇسىدىكى مۇرەككەپ ۋە تولىمۇ نازۇك بولغان بىر مۇھىيتتىن ، ئانىسنىڭ بىر مەزگىل پەرۋىشىنى شەرت قىلغان ئاستىدا ،بۇ تەبىئەت دۇنياسىنىڭ مۇھىيتىغا ماسلاشقۇدەك ھالەتكە  كىلىپ م بۇ  زىمىنغا تۇغۇلۇش جەيرانى ،دەل بىز ماقالىنىڭ بىشدا بايان قىلغاندەك سەۋەپ-نەتىجە مۇناسىۋەت دائىرىسىگە كىرىدۇ .

      ئەمدى بىز  2- مىسالنى يەنى ئايرۇپىلاننىڭ ئۇچۇشىنى ئەقلى تەھلىل قىلىپ باقايلى  !
  
   ئايرۇپىلان بولسا ،ئىنسانلار ئاللاھ ياراتقان مۇنۇ تەبىئەتتىكى ئۇچار-قۇشلارنىڭ ھاۋادا ئۇچۇش ھەركىتىدىن ئىلھام ئىلىپ،  بۇ جەرياندا ئەقلى تەپەككۇر بىلەن، تۇنجى ئايرۇپىلانىڭ ئىپددايى خەرىتە شەكلىنى سىزىش ،ئاندىن خەرىتە  بويچە لازىملىق ماتىرياللارنى ئىشلىتىش ئارقىلىق ياسالغان نەرسە بولۇپ ، بۇ ئايرۇپىلانمۇ ياسىلىش جەيرانىدا ئەلا سۇپەتلىك ماتىريالدىن ئىشلەنگەن يۇقىرى بىىسىملىق ھاۋا شۈمۈرۇش ۋا ھاۋانى يۇقىرى بىسىلم بىلەن سىقىپ چىقىرىش خۇسىسىيتىگە ئىگە بولغان  ماتۇر ،  قانات ، ئەلا سۇپەتلىك داتلاشماس پولات،  تومۇر،  ۋولفىرام ، يىقىلغۇ ئۇچۇن لازىملىق  ھەر خىل ئىنىرگىيە ..........قاتارلىق ھەر خىل  ماتىريال ۋە ھەر خىل سۇيۇغلۇقلارنىڭ خەرىتە لايھەسىنى بوينچە ، بىر-بىرى بىلەن ماس قەدەملىك تەسىرلىنىش شەرتى بىلە  كۆكتە ئۇچىۋاتقان ئايرۇپىلاننى  ياساپ چىققان .
   بىز ئەقلى تەپەككۇر بىلەن ئايرۇپىلاننىڭ ياسىلىش جەرياننى. ئىنسانلارنىڭ ۋۇجۇتقا كىلش جەريانىغا تەقلىدى قىلساق ،گەرچە بىرسى تۆمۇر ،  بىرسى جانلىق بولسىمۇ ، ھامىيەتتە بۇ ئىككى شەيئنىڭ ۋۇجۇتقا كىلپ ھەركەت قىلشى ماھىيەتتە تولىمۇ ئوخشاش ئىكەنلىگىنى بايقايمز .

      بىز بۇ ھەقتە مۇنداق تەسەۋۋۇر قىلايلى ئايرۇپىلاننىڭ ياسىلىشىغا كىرەكلىك بولغان ھەر خىل پولات ،تومۇر،ۋولفىرام .......قاتارلىق ماددىلارنى ئەر-ئايالنىڭ ئىسپىرمسسى ( يەنى ئايرۇپىلان ياسىلىدىغان ماددىلار دەپ قىياس قىلايلى )، بۇ ماددىلار (ئىسپىرما )  ئايالنىڭ بالىياتقۇسى (يەنى ئايرۇپىلان ياسايدىغان زاۋۇتى ) دا ، ئايالنىڭ ھەر قايسى تومۇرلىرىدن ئىقىپ كەلگەن ھەر خىل قۇۋۋەتلىك  ماددىلار  ( يەنى زاۋۇتتكى ھەر قايسى ئىشچىلارنىڭ ئۆز-ئارا ھەمكارلىشىشى ئاستىدا ، قوراشتۇرۇش ، سەپلەش،كەپشەرلەش .ئورۇنلاشتۇرۇش ....جەريانلىرى )   بىلەن تەدىرىجى ئىپددايى يىرىڭ  ھامىلە شەكلىگە  (ئايرۇپىلان ) كىلىپ . بۇ ئىپدىدايى ھاملىە شەكلى  ( تولۇق پۇتمىگەن ئايرۇپىلان ) تەدىرجى ئانىنىڭ بالىياتقۇسىدىن  ( ئايرۇپىلان ئۆزى ياسىلىۋاتقان زاۋۇتتىن ) سىرىتقى تەبىئەت دۇنياسىغا ( ئايرۇپىلان زاۋۇت سىرتىدىكى مەيدانغا  ) تۇغىلىدۇ ( ئايرۇپىلان زاۋۇتتىن چىقىدۇ )، ۋە بۇ بوۋاق ( سىناقتىن تولۇق ئۆتمىگەن بۇ ئايرۇپىلان ) ئانىسىنىڭ ( تىخنىكا ئالىملىرىنىڭ ) مەلۇم مەزگىللىك پەرۋىشى بىلەن ( تىخنىك ئالىملارنىڭ زاۋۇتتىن يىڭى چىققان بۇ  ئايرۇپىلاننى قايتا-قايتا سىناق قىلش ئارقىلىق ،ئۆلچەمدىن ئۆتمگەن نامۇۋاپىق يەرلىرىنى تۇزەش بىلەن ،بۇ  بوۋاق تەدىرجى ( يەنى بۇ ئايرۇپىلان ) مۇكەممەل بولغان بىر ئادەمگە ئايلىنىدۇ .(  مۇكەممەل بولغان بىر ئايرۇپىلانغا ئايلىنىدۇ ).

    مانا يۇقارقى بۇ ئالەمدە ئىنسانلار قولى ئارقىلىق ياسالغان بۇ ئايرۇپىلاننىڭ مەيدانغا كىلىش جەريانىمۇ سەۋەپ-نەتىجە مۇناسىۋتى شەرتىنى ئاساسادامەيدانغا كەلگەن بولغاچقا ، بۇ جەيراندىكى قوللىنىلغان بارلىق تىرىشچانلىق ئىلىم  ھىكىمەت ئىلمىگە كىرىدۇ .

  بىز ئاخىردا 3- مىسال يەنى سىمسز قول تىلىفۇننى مىسال قىلساق . بىز بۇ قول تىلىفۇننىڭ شەكلنى كۆرىمىشىز  بىراق  قارشى تەرەپكە چىتىلغان ئىېلىكتىر سىمىنى كۆرمەي، پەقەت قارشى تەرەپتىن كەلگەن ئاۋازنى شۇنداق ئوچۇق ۋە جاراڭلىق ئاڭلايمىز، بىز  بۇ ئاۋازلار  ئارقىلىق بىز گەرچە ئىلىكتىر ماگېنت سىگنالىنى كۆرمىسەكمۇ . بىز سۆزلىشىۋاتقان سىمسىز قول تىلىفۇننىڭ ئېلىكتر ماگىنىت سىگىنالنىڭ ،بىز ۋە قارشى تەرەپنىڭ ئاۋازلىرىنى ئۆز-ئارا يەتكۇزۋاتقانلىقىنى ھىس قىلىپ يىتەلەيمىز ،
    دىمەك بۇ قول تىلىفۇنمۇ ،  يەر شارىنى ئايلىنىپ ھەركەت قىلىۋاتقان سۇنئى ھەمىرانىڭ تارقاتقان ئىلىكتىر سىگىنالى ۋە بىز  تەرەپتىكى سىمسىز تىلىفۇنىدىن چىقىۋاتقان  ئېلىكتىر سىگنالىنى قوبۇل قىلىپ، قارشى تەرەپكە يەتكۇزۇپ بىرىشتىن  ئىبارەت ئۇسۇللارنى سەۋەپ قىلىش ئارقىلىق ،ھەر خىل ئاۋازلارنى بىزگە ئاڭلىتىش جەيرانى سەۋەپنىڭ نەتىجىسى بولۇپ،  دەل بۇ ئىشلارمۇ  بىز يۇقۇردا بايا قىلغان ھىكىمەت قاتارىغا كىرىدۇ .

    ئۇنداقتا مۆجىزە دىگەن نىمە ؟
   جاۋاپ- مۆجىزە دىگىنىمىز ، بىرەر جانلىق-جانسىز شەيئىنىڭ  مەۋجۇتلىقىنى ۋە ھەركىتىنى كۆرگىلى بولىدىغان ،بىراق، شۇ شەيئىنىڭ  ۋۇجۇتقا كىلىش جەريانىنى ، ۋە ھەركەت قىلىش پىرىنسىپىنى ئەقلى يول بىلەنمۇ ، ئەقلى تەپەككۇر بىلەنمۇ . ھەتتا ئىنتايىن يۇقىرى بولغان پەن-تىخنىكا ئۇسكۇنلىرى بىلەنمۇ بىلگىلى بولمايدىغان ھادىسە ۋە ئىش-ھەركەتلەر  مۆجىزە دەپ ئاتىلىدۇ .

    سۇئال قانداق نەرسىلەر مۆجىزە تۇردىكى ئىشلارغا تەۋە بولىدۇ ؟
    جاۋاپ - ئۆلۇپ كەتكىنىگە نەچچە ئون يىللار بولغان بىر ئادەمنىڭ توساتتىن تىرىلىپ ،بىز بىلەن پاراڭلىشىش ياكى ھەركەت قىلشى مۆجىزە دەپ ئاتىلىدۇ.
  
    ئەسلى ئۆلۇپ كەتكەن بىرەر ئادەمنىڭ تۇيۇقسىز تىرىلىپ قىلشى ، شۇ ئادەمنىڭ توك سوقىۋىتىش ياكى يۇرەك كىسىلى قوزغىلىپ قىلىشى ئاستىدا  يۇرەك سوقۇشى ،نەپەس ئىلىشى مۇئەييەن بىر ۋاقىت توختاپ قىلىش سەۋەبى بىلەن ، دوختۇرلار بۇ ئادەمنى  ئۆلۇپ بولدى دەپ قاراپ، ھۆكۇم چىقىرىپ بولغاندىن  كىيىن . توساتتىن بۇ ئادەم ئەسلىگە كىلىپ  ھەركەت قىلدى دەپ پەرەز قىلساق ، بۇ ئادەمنىڭ مۇنداق ھالەتكە كىلىپ قىلشى، بۇ كىشىنىڭ بەدىنىدىكى ھەر قايسى رەئىس ئەزالار ۋە ئاساسى ھالقىلىق مۇھىم بولغان ئەزالار بىر-بىرى بىلەن بولغان ماسپ قەدەملىك زەنجىرسىمان تەسىرلىنىش تەرتىپى، تىخىغۇچە ، ئاللاھ بەلگۇلگەن ھاياتلىقنڭ نىسپى ساقلىنىش دەرىجە دائىرىسىدىن  چىقىپ كەتمىگەنلىكى ئۇچۇن،  دوختۇرلارنىڭ كۆپ تەرەپلىمە ئاناتومىيە داۋالاش ھەكەتلىرى  بويىچە قوللانغان ھەر خىل ئۇسۇللارنىڭ سەۋەبى بىلەن << تىرلىگەن >>  بولغاچ بۇ ئىش مۆجىزە قاتارىغا كىرمەيدۇ.
   بىراق ئۆلگۇنىگە ئون نەچچە يىلدىن ئىشىپ كەتكەن ئىنساننىڭ بۇ ئون نەچچە يىل  جەيرانىدا ، بەدىنىدىكى ھەر قايسى ئەزالار ئاللىقاچان چىرىپ ،بەدىندىكى تەبئىي  بىئولوگىيلىك ھالەت ئىغىر دەرىجىدە بۇزۇلۇپ بولغان ياكى ئۆزىنىڭ بەدىنىدكى قان، سۇ ،گۆش .........قاتارلىق  ھەر خىل سۇيۇغلۇقلارنىڭ،بىر-بىرى بىلەن مۇئەييەن قانۇنىيەت شەرتى ئاستىدا  تەسىرلىنشى بىلەن ،جەسەتنىڭ بەدىنىدە قۇرۇت ھاسىىل بولۇپ ،بۇ قۇرۇتلار ئۆزى ھاسىل بولغان جەسەتنى يەپ بولغان بىر ھالەتتە ،ئاللىقاچان ئىسكىلىتكە ئايلىنىپ بولغان ئۇ كىشى تۇيۇقسىز  ئورنىدىن دەس تۇرۇپ گەپ-سۆز قىلشى ،ۋە ھەركەت قىلىشى ، ھىكىمەتنڭ سەۋەپ قىلىش ئارقىلىق نەتىجە چىقىش شەرتگە خىلاپ ۋە تولىمۇ زىت بولغانلىقى ئۇچۇن ،بۇمۇ مۆجىزە ئىشلار قاتارىغا كىرىدۇ .

      -قۇرۇق پەتنۇسنى ئىچىگە سىلىپ يۆگەپ قويۇلغان داستىخاننى ئاچقىنىمزدا،  ئىچىدىن ھەر خىل يەيدىغان نازۇ-نىمەت ، ھەر خىل مىۋە-چىۋىلەرنىڭ چىقىشىغا ئوخشاش ئىشلارمۇ  مۆجىزە ھىساپلىنىدۇ .

     ئەسلىدە بىرەر ئىنسان تەرىپىدىن داستىخان ئىچىلىشتىن بۇرۇن ،داستىخىاننىڭ ئىچىگە يىمەك-ئىچمەك ۋە ھەر خىل مىۋە -چىۋە سىلىپ قويۇش شەرتى، بىراۋ تەرىپىدىن ئورۇندۇلۇپ بولغاندىن كىيىن، بىرەر ئىنسان شۇ داستىخاننى ئاچقاندا ،ئىچىدىن يىمەك-ئىچمەك ۋە ھەر خىل مىۋە-چىۋە چىقسا ،ئاللاھنىڭ بىزلەرگە چۇشۇرۇپ بەرگەن يەنى سەۋەپ قىلشنى شەرت قىلش جەيرانىدا ،كىينىك بىز كۆزلىگەن نەتىجىە چىقىدىغان،  ھىكىمەت ئۇسۇلىغا ئۇيغۇن كەلگەن بولاتتى .
  بىراق، قۇرۇق بىر داستىخاننى ئاچقىنىمىزدا، ئىچىدىن يىمەك-ئىچمەك ۋە ھەر خىل مىۋە -چىۋە چىقىشى، سەۋەپ قىلمىساقمۇ ،يىمەك-ئىچمەك ۋە ھەر خىل مىۋە -چىۋە چىقىشىتىن ئىبارەت، نەتىجىنىڭ يۇز بىرشى ، سەۋەپ-نەتىجە قانۇنىغا ئىنتايىن زىت ۋە بۇ قانۇننىڭ شەرتىگە قارشى نۇقتىدىن چىققانلىقى ئۇچۇن، مۇنداق ئىشلارمۇ مۆجىزە دەپ ئاتىلىدۇ .

    - بىر ئادەمنىڭ پىيادە ھالدا 10 نەچچە مىنۇت ئىچىدە ،قەشقەردىن ئۈرۇمچىگە بىرىپ-كىلىپ بولۇشى بۇمۇ مۆجىزە ئىشلارغا كىرىدۇ .
      دىمىسىمۇ تەخمىنەن 1470  كىلومىتىر ئۇزۇلۇقتىكى يولنى تاماملاشتا، ئاپتۇبۇستا بىلەن  20 سائەت ئەتىراپىدا، ئايرۇپىلاندا 3 سائەتكە يىقىن تامام بولىدىغان  مۇساپىنى، بىر ئادەم پىيادە 10 نەچچە مىنۇت ئىچىدە تامام قىلىش، ھازىرقى بىز ئىنسانلارنىڭ ئورتاق قىممەتكە ئىگە بولغان ،فىزولوگىيلىك بەدەن ساپايىمىز بىلەن تامام قىلىش ، سىپى ئۆزىدىن ساراڭنىڭ خىيالى دىسەكمۇ ئارتۇق كەتمەيدۇ . شۇڭا بۇ ئىشمۇ مۆجىزە ھساپلىنىدۇ .

- بىر ئالما كۆچەتنى بۇگۇنى تىكسەك، بۇ ئالما كۆچىتى  ئەتىسى سەھەردە چىچەك ئىچىلىپ ،20 مىنۇتتىن كىيىن چىچەكتىن ئالما غورىسى ئاشكارە بولۇپ ، چۇشتە بۇ ئالما غورسى قوش مۇشتىمىزىدەك چوڭلۇققا يىتىپ، كەچ تەرەپتە كۇن ئولتۇرۇشقا يىقىن ۋاقىتتا ، بۇ ئالما قىپقىزىل رەڭگە كىرىپ پىشىپ بولۇشىمۇ مۆجىزە ھىساپلىنىدۇ .
   
   چۇنكى ئەسلى  بۇ ئالما كۆچىتى بۇ يىل تىكىلسە ،تىكىلگەن شۇ يىلى ئۆزى تىكىلگەن يەرنىڭ جۇغراپىيلىك مۇھىيتىغا تەدىرجى ھالدا ماسلىشۋىلىپ،  كىلەر يىلى ئۆسۇس جەيرانىدا ئۆسكەن ،بۇ يىللىق ۋە شۇ يىللىق يۇمراق نوتللىرى يىلغا توشۇۋالغاندىن كىيىن،  ئاندىن ئۇچنجى يىلى مىۋە بىرىشكە باشلىسا ،ھىكىمەت ئىلمىگە توغىرا كىلەتتى.
   بىراق بۇ ئالما كۆچىتى ئۇچ يىللىق ( يەنى تەخمىنەن1000 كۇندىن ئارتۇق ۋاقىت كىتىدىغان ) جەرياندا ، شۇ ئالما كۆچىتىدە يۇز بىرىدىغان بىئولوگىيلىك  ھادىسلەر ،پەقەت بىر كۇن (يەنى  تەخمىنەن 8-7 سائەت ) ئىچىدە ،يەنى  ئۇچ يىللىق ۋاقىتنىڭ نەتىجىسى 8-7 سائەتنىڭ ئىچىدە  تاماملىنىپ بولۇشتىن  ئىبارەت ۋاقىت قىسقىراش جەيرانىنى  ،مەيلى ئەقلى يول بىلەن بولسۇن ، مەيلى ئەقلى تەپەككۇر بىلەن بولسۇن ۋە ياكى دۇنياۋى ئەڭ ئىلغار بولغان ئېلىكتىرۇنلۇق ئۇسكۇنلەرنىڭ ياردىمى بىلەن بولسۇن بىلگىلى بولمايدۇ ،مانا بۇلار مۆجىزە دەپ ئاتىلىدۇ .

  تۆۋەندە بىر قۇرئاندا ھەقىقى بايان قىلىنغان مۆجىزىلەردىن بىر قانچنى مىسال قىلايلى ، قۇرئاندىن چەتنەپ ئەپ قاچتى مۆجىزىلەردىن مىسال ئالساق،  ھەقىقى مۆجىزىنىڭ چىنلىقى بىزدە ئانچە جەلىپكارلىقىمىزنى قوزغىتالمايدۇ .

    قۇرئاننىڭ  مائىد سۈرىسى . ( تەرجىمىسى داستىخاننىڭ قىسىسى ) دە  بۇۋى مەيرەم ئانىمىزىنىڭ ،دادىسىز ئوغلى ئەيسا پەيغەمبەرنىڭ قىسسى بايان قىلىنىپ . ئاللاھنىڭ ئەيسا پەيغەمبەرگە ئاسماندىن مۆجىزە سۇپىتىدە بىر داستىخان چۇشۇرۇپ بەرگەنلىكى بايان قىلىنىدۇ.

       بۇ قىسىنىڭ مەزمۇنىدا ،مۇشرۇكلەرنڭ زۇلۇمىدىن قۇتۇلۇش ئۇچۇن ، پاناھلىق ئىزدەپ ئەگەشكۇچىللىرى بىلەن باشقا قەبىلىلەر رايۇنىغا كىتىۋاتقان ئەيسا پەيغەمبەر، تولىمۇ ئىسىق ۋە ھىچقانداق دەل-دەرەخمۇ بولمىغان بىر سايلىقتا ، شاگىرىتلىرى   ( ئەگەشكۇچىللىرى ) تەرىپىدىن ، بىرنجىدىن قوساقلىرىنى توقلىۋىلىشنى ئارزۇ قىلىشى، ئىككىنجىسىدىن ئاللاھنىڭ كامالى قۇدۇرتىنى ئۆز كۆزى بىلەن بىر قىتىم بولسىمۇ كۆرۇش ئارقىلىق ، شەيتان تەرىپىدىن داۋالغۇۋاتقان قەلبىدىكى تۇراقسىز ئىمانى ئۇچۇن خاتىرجەملىك تىپىۋىلىشنى نىيتىدە ، ئەيسا پەيغەمبەرنىڭ  ئاللاھقا دۇئا قىلىپ ،ئاسماندىن مۇشۇ ئىسسىق سايلىقتا ئۆزلىرىنىڭ يەپ-ئىچىشى ئۇچۇن، مۇزدەك سوغۇق ئىچىملىك بىلەن ،يىمەك-ئىچمەك چۇشۇرۇپ بىرىشنى قاتتىق تەلەپ قىلىپ تۇرىۋىلنىغان ، بۇ تەلەپتىن  ئەيسا پەيغەمبەر ئاللاھتىن قەلبىگە مۇۋاپىق ئىلھام جاۋاپلىرىنى سىلىشىنى ئۇمۇت قىلىپ خىجىلچىلىقتا، ئاسمانغا  تەكرار-تەكرا قارىغان ، بۇلارنى كۇزۇتۇپ تۇرغۇچى ئاللاھ ،ئەيسا پەيغەمبەرنىڭ دىلىغا  ئىلھام ئارقىلىق بىر قانچە سۆز تاشلىغاندىن كىيىن  ،ئەيسا پەيغەمبەرنىڭ دىلىغا خاتىرجەملىك سىلىش ئۇچۇن ، پەرىشتە جىبرائىل ئارقىلىق ئاسماندىن ئىچىدە مۇزدەك سوغۇك ئىچملىك ۋە مول يىمەكلىك  قاچىلانغان بىر داستىخان چۇشۇپ بەرگەن . ئەيسا پەيغەمبەرنىڭ ئەگەشكۇچىللىرى  بىر چاغدا كەينى تەرەپكە قارىغۇدەك بولسا ، ئاللىقاچان  يەردە بىر داستىخاننىڭ تۇرغىنى كۆرگەن ۋە  دەرھال يۇگۇرۇشۇپ بارغانچە داستىخاننى ئاچقانلىرىدا ، داستىخاندىن ھەقىقەتەن بايام ئۆزلىرى ئەيسا پەيغەمبەردىن تەلەپ قىلغاندەك ،مۇزدەك سوغۇك ئىچملىك ۋە نۇرغۇن يىمەكلىك چىققان، مانا بۇ قۇرئاندا مەشھۇر بولغان  مائىدە سۇرىسىدۇر .

     مۆجىزە بىلەن ھىكىمەت نەتىجىللىرى  ئىنسان تەرىپىدىن سەيرە قىلىنغادا ، بىرەر ئىنسان مۆجىزە كۆرمەي تۇرپ ئانچە-مۇنچە ئىسلامدا  خاتالىشىپ قالسا ،ئاللاھ جازالاشنى بىر مەزگىل كەينىگە تارتىپ شۇ ئىنساننىڭ ئۆزىدىن مەغپىرەت تىلىشنى ساقلايدۇ ، ئاللاھ مۇشۇنداق بەندىللىرىگە مىھرىباندۇر   .
    بىراق مۆجىزىنى كۆرگەن قادناقكى بىر ئىنسان بىرلا قىتىم دىن-ئىسلامنىڭ يۇزىنى تۆكىدىكەن ياكى ئىنتايىن ئېغىر بولغان جىنايەتتىن بىرنى سادىر قىلىدىكەن، ناھايتى تىزلا   جازالىنىدۇ ۋە بۇ جازالانغان كىشىگە جەننەت  ھارام قىلىنىدۇ  .بۇنداق كىشى دوزاقتا مەڭگۇ قالغۇچىلاردىن بولىدۇ .
  
     بۇنىڭ جانلىق مىسالى: ئەيسا پەيغەمبەرنىڭ شاگىرىتلىرى ( ئەگەشكۇچىللىرى ) ئىچىدىن شۇ ۋاقىتتىكى مۆجىزە سۇپىتىدە، ئاسماندىن چۇشۇرۇلگەن داستىخاندىكى تاماققا ئىغىز تەككەن بىر كىشى، كىينىكى ۋاقىتتا ئازىراق پۇلنىڭ ئىزىقتۇرشى بىلەن ، پەيغەمبىرى ئەيسا ئەلەيھىسسالام پاناھلىۋاتقان جايىنى  دۇشمەنلىرىگە ئىيىتىپ  قويغان سەۋەپلىك ئىككى ئالەملىك خار-زەبۇن بولۇپ كەتكەن ۋە ئاللاھنىڭ ئىزىقىتۇرشى بىلەن ئۆز ئادەملىرى  تەرىپىدىن ئىچىنشىلىق ئۆلتۇرلىگەن .
   مانا بۇ كىشى ،دۇنياغا داڭلىق بولغان ئەڭ ئاخىرىقى كەچىلىك غىزا  دىگەن مەشھۇر رەسىمدىكى ،ئەيسا پەيغەمبەر ئاسمانغا چىقىپ كىتىشتىن بۇرن ئەڭ ئاخىرقى ۋاقتىدا ،دوسلىرى بىلەن كۆڭلى پەرىشان ھالەتتە پاراڭلىشىۋاتقان  13 دوستىنىڭ  ئىچىدىكى خائىن دوستىنىڭ خارلىق ھالىتى  رەسىمنىڭ ئىچىدە رەسسام تەرىپىىدن يۇقىرى بەددئى ئۇسۇلدا  تولىمۇ جانلىق تولىمۇ ئوبرازلىق ئەكىس ئەتتۇرلۇپ دۇنياۋى كىشىلەرنىڭ قەلبىدە شەرمەندىلەرچە ياد ئىتىلدىغان بىر شەخىس سۇپىتىدە ،ئۆچمەس تەسىراتلارنى قالدۇردى  .
  
  ..... قۇرئاندا نۇرغۇن يەردە مۆجىزە كۆرسىتىلىپ بولغاندىن كىيىنمۇ ئىمانغا قايىتىمغان كىشىلەرنىڭ ھايال قىلمايلا جازاغا ئۇچۇرغانلىقى بايان قىلىنىدۇ .
    پەقەت يۇنۇس پەيغەمبەرنىڭ قەۋمىدىن بىرلا قىتىم ( يۇنۇس پەيغەمبەر قەۋمىگە ئاللاھنىڭ ئىزنى بىلەن مۆجىزە كۆرسەتكەندىن كىيىنمۇ قەۋمى ئىمانغا كىرمىگەن ، شۇنىڭ بىلەن يۇنۇس پەيغەمبەر  قەۋمىدىن رەنجىپ قەۋمىنى تاشلاپ چىقىپ كەتكەن ) پەرىشتىلەر جازانى يۇرگۇزىدىغان ۋاقىتتا ، ھەممسى قاتتق تۆۋبە  قىلىپ پەيغەمبىرى يۇنۇسنى ئىزدەپ سەكپارە بولۇپ كەتكەن .بۇلارنىڭ مۇشۇ بىر قىتىملىق كەچۇرىمىنى ئاللاھ ئەپۇ قىلغان.  

   ....قۇرئاندىكى يەنە بىر مۆجىزە بولسا  كەف سۇرىسىدىكى ئەسھابۇل غار كىشىللىرى ۋەقەسى.
،بۇ غاردىكى بىر نەچچە يىگىت ئاللاھ رىزاسى ئۇچۇن يۇرتىدىكى دىنسىزلارنىڭ زۇلۇمىدىن قىچىپ بىر غارغا كىرىۋالىدۇ ،ۋە بۇ غاردا ئاللاھنىڭ ئورۇنلاشتۇرشى بىلەن تەخمىنەن 300 يىلغا يىقىن ئۇخلاپ قالدۇ،كىيىن  ئويغانغاندا ئۆزلىرىنىڭ شۇنچە ئۇزۇن ئۇخلىغانلىقىنى ھىس قىلماي، بازارغا يمەكلىك سىتىۋىلىش ئۇچۇن بىرىپ ،شۇ يەرلىك كىشلەردىن ئۆزلىرىنىڭ 300 نەچچە يىلنىڭ ئالدىدىكى كىشلەر ئىكەنلىگىنى بىلىپ يىتىدۇ ۋە ئاخىرقى ئۆمۇرلىرىنى شۇ غاردا ئۆتكۇزىدۇ .

  قۇرئاندىكى يەنە بىر مۆجىزە بولسا ئۇزەيرە ئەلەيھىسسالام ۋەقەسى بولۇپ . ئۇزەيرە ئەلەيھىسسالاممۇ 90 يىلدىن ئارتۇق ئۇخلاپ ئويغىنىپ، ئۆز دەۋرىدىن 90 نەچچە يىل ئالدىنقى دەۋىرگە كىلىپ قالغانلىقى بايان قىلىنىدۇ  .

   يەنە بىر مۆجىزە بولسا ئىسلام دۇنياسىدىكى ئەڭ داڭلىق بولغان ۋەقە يەنى  قۇرئاندىكى   ئىسرا سۈرىسى بولۇپ ،
بۇ سۈرىدە   پەيغەمبىرىمىزنىڭ پەرىشتە جىبرائىل بىلەن مەلۇم بىر كىچىدە ،مەككىدىن شۇ ۋاقىتتا تۆگە بىلەن  ئىككى ئايدىن ئارتۇق ۋاقىتتا بىرىپ بولىدىغان ھازىرقى پەلەستىن دۆلىتىگە بىرىپ، شۇ يەردىكى قۇددۇسى مەسجىتىدە ئىككى رىكەت سەپەر نامىزىنى ئوقۇپ بولغاندىن،  بۇراققا مىنىپ ئاسمانغا ئۆرلەپ ئەرىشكە چىققان ، ۋە بۇ جەيراندا ھەر بىر ئاسماندا شۇ ئاسماننىڭ ئۆزىگە خاس بولغان ئاجايىبات تىلسىماتلىرىنى كۆزدىن كەچۇرگەچ ئاللاھ تەرىپىدىن ئەرىشتە قوبۇل قىلىنغان ....ئاللاھ بىلەن پاراڭلىشىش خاتىرسىنىڭ ھەممىسى  ھەدىسلەرنىڭ ئىچىدىكى ئەڭ مەشھۇر قۇددۇس ھەدىسلەر يەنى  ھەدىسلەرنىڭ گۇلتاجىسى بولۇپ، بۇ ھەدىسلەر ، رەسۇلۇللاھنىڭ ئاللاھ بىلەن بىۋاستە پاراڭلىشىشتىن جەيرانىدا ،ئاللاھ بىز مۇسۇلمانلارغا ئىنئام قىلغان سوۋغاتلىرىنى قوبۇل قىلىشقا نائىل بولغانلىقىدىن  ئىبارەت بىر قىسىم بايانلار  سۆزلىنىدۇ .

يەنە بىر مۆجىزە بولسا سالىھ ئەلەيھىسسالامنىڭ تۆگىسى بولۇپ ،ئاللاھ بۇ چىشى تۆگىنى تاشتىن چىقارغان بولۇپ،بۇ تۆگە تاشتىن چىققاندا  بوغاز ھالىتىدە بولۇپ ھايال ئۆتمەي نۇرغۇن كىشىلەرنىڭ كۆز ئالدىدا تۇققانلىقى قەيىت قىلىنىدۇ ..........  

    بۇ ھەقتە تەپسىلى دائىرىدە توختالمىساقمۇ ،كەڭ دوسلىرىمىز  بۇگۇنكى كۇندە بىر ئادەمنىڭ پىيادە سائىتىگە 30 كىلومىتىر يول مىڭىش ئارقىلىق ئەمەس . بەلكى ئالەمدىكى ھەركەت قىلۋاتقان سان-ساناقسىز پىلانىتلارنىڭ يەر شارى بىلەن بولغان ئارلىغىنى ئۆلچەشتە ئىشلىتىدىغان لازىر نۇرلۇق ئوپتىكا قورال ۋە ئۇسكۇنىلەر ،  يەنە بۇنىڭغا يانداشقان يەر شارىنىڭ قۇلىقى دەپ نام ئالغان ،ئاسمان جىسىملىرىنىڭ مەركىزى يادىرو قىسمىدىن كەلگەن مىكىرو نۇر زەرچىلەر ئاتۇمىنىڭ ، دەۋەرلىك مەلۇم بىر پەيىت ئىچىدە يىمىرلىش دەرىجىسىنى ئۆلچەش ئارقىلىق ،شۇ نۇر مەنبەسى بولغان ئاسمان جىسىملىرىنىڭ، بىز تۇرىۋاتقان يەر شارى بىلەن بولغان ئارلىقىنى ئۆلچەشتە ئىشلىتىدىغان شۇ پەن-تىخنىكا ئۇسكۇنلىرىدىن ،ئالەم بوشلۇقىغا قويۇپ بەرگەن نۇرنىڭ،ھەر بىر سىكۇنىتىغا تەخمىنەن 300 مىلىيۇن كىلومىتىر تىزلىك بىلەن ( 1998-يىلدىكى سانلىق سىتاستىكا بويىچە ھىساپلىغاندا )  280مىلىيۇن نۇر يىلى يىراقلىقتىكى ئالەم بوشلىقىغا يىترىپ بىرىپمۇ . بىرىنجى ئاسماننىڭ چىگىرسى بولغان مەۋھۇم ئوقۇم دائىرىسىگە يىتىپ بارالمىغان .

    ئىسلام ئالىملىرى بىرىنجى ئاسماننىڭ چەك -چىگىرىسنى قايسى ئۇسۇل بىلەن روشەنلەشتۇرۇپ چىقتى ؟؟؟؟؟؟

       ئىسلام ئالىملىرى تاكى يىقىنقى زامانغىچە ،بىرنجى ئاسمان ،ئىككىنجى ئاسمان ،ئۇچۇنجىى ئاسمان ......لارنىڭ ۋەكىلىگە ، قۇياش سىستىمىسىنىڭ رايۇن تەۋەلىگىدە  ھەركەت قىىلۋاتقان ۋىنىرا ،ساتىرۇن ،يۇپتىر ....... قاتارلىق   ھەركەت قىلىۋاتقان ئۇششاق پىلانىتلارنى ۋەكىل قىلىپ كىلۋاتقان ئىدى .

    بىراق ئويلىمىغان يەردىن قۇرئان كەرىم بۇ كۆز- قاراشلارنى ئاغدۇرۇپ تاشلىدى.
   يەنى قۇرئان ھىكىمەت ئىلمى بىلەن شۇغۇللىنىدىغان بىر قىسىم ئالىملارنىڭ مەيدانغا كىلشى بىلەن قۇرئان ئايەتلىرىنى تەرجىمە قىلشتا بۇرنقىدەك زاغىرا ۋە ئۆلۇك ئۇسۇلدا تەرجىمە قىلىش ، ئايەتلەر بايان قىلۋاتقان مەلۇم بىر شەيئىلەرنىڭ تۇزۇلىش پىرىنسىپى ئوقۇمىدا چوڭقۇر پىكىر يۇرگۇزمەسلىك سەۋەپلىرى تولا سادىر بولغان ئىدى  ، كۆپۇنچى ئەھلى كىتاپلار   قۇرئاننىڭ شەرىئەت قىسمى مۇھىم تۇتقا قىلىپ قۇرئاننىڭ ھىكىمەت قسىمىغا ئائىيىت ئايەتلەرنىڭ مەناسىغا سەل قاراپ كەلگەن سەۋەپلەر تۇپەيلىدىن نۇرغۇللىغان ئىـىتىلاپى مەسىللەر چىقىۋاتقان  ئىدى.

    شۇنداقتىمۇ بىر قىسىم ئىسلام سۆيەر زىيالىلار ،تارقاق ھالەتتە بولسىمۇ، قۇرئاننىڭ ھىكىمەت ئىلمى ئۇستىدە پىكىر يۇرگۇزۇشنى توختىپ قويماي ، دۇنيادىكى كەڭ  مۇسۇلمان جامائەتچىلىكىگە، قۇرئان ھىكىمەت ئىلمىدىن  زور بولغان يڭى بايقاش ۋە يىڭى نەتىجىلەرنى روياپقا چىقىرىپ  بەردى ، ۋە بۇ ئارقىلىق قۇرئاننىڭ ئۇستىدىن تۆھمەت قىلۋاتقان بىر قىسىم كۆڭلى قارا كىشلەرنىڭ قارا-قۇيۇق سەپسەتىللىرىگە دەل ۋاقدا رەدىيە بىرىپ، ئۆزىنىڭ ئىسلام دۇنياسىنىڭ شان-شۆھرىتىنى قوغداش يولىدا ، بىر كىشلىك مۇسۇلمانلىق مەجبۇريىتىنى ئادا قىلىپ كەلدى ۋە كىلۋاتىدۇ .
   مانا شۇ ھىكىمەت ئىلمى بىلەن شۇغۇللانغۇچى ئالىملارنىڭ ، كۆپ يىل ئىزدىنىش جەيرانلىرىدا ،قۇرئان ئىچىدىكى ئاسترونومىيەگە ئائىيىت بولغان ئايەتلەرنى رەتلەپ ،مەنىسنى تەتقىق قىلىش ئارقىلىق ،قۇرئاننىڭ ئىچىدىن .....بىز تۆۋەن ئاسماننى يۇلتۇزلار بىلەن بىزىدۇق .....دىگەن بىر جۇملە ئايەتنىڭ ئۆز ئىچىگە ئالغان مەناسىنى چوڭقۇر تەپەككۇر قىلىش .ھازىرقى زامان ئاستىرونومىيە تەتقىقات نەتىجىللىرىگە سىلىشتۇرۇش ،دەللەش ئارقىلىق ،ئەمىليىتتە يولتۇز ۋە سەييارىلەرنىڭ %90 تى پەقەت بىرنجى ئاسماندىلا بارلىقى دەللىلنىىپ ، بۇ ئارقىلىق بىرىنجى ئاسماننىڭ چىكى بىر قىسىم ئىسلام ئالىملىرى ئىيىتقاندەك قۇياش سىستىمىسى ياكى سامان يولى يۇلتۇزلار توپى  بولماستىن ،بەلكى نۇر ھەر بىر سىكۇنتىغا 300 مىلىيۇن كىلومىتىر سۇرئەتتە تەخمىنەن  250 مىڭ يىل نۇر يىلى يىراقلىققتىكى ۋاكۇم ئالەم بوشلۇقغا  يىتىپ بىرىپمۇ بىرنىجى ئاسماننىڭ مەۋھۇم بولغان چىگىرا ئوقۇمى دائىرىسىگە بارالمىغانلىقىنى دەللىلەپ بەردى .ۋە شۇ  ئورۇنلاردىكى  مىكىرو چاڭ توزاڭلارنىڭ تەركىبىدە يەنە نەچچە يۇز نۇر يىلى يىراقلىقتا، ئىنسانلار بايقاپ باقمىغان  سان-ساناقسىز يۇلتۇزلار تۇركۇمىنىڭ، ئۆز ئوققىدا پىلانلىق  ھەركەت قىلۋاتقان يۇلتۇزلارنىڭ مەركزىدىن چىققان قىسقا چاستۇتلۇق مىكىرۇ نۇر زەرچىللىرىنى بايقىغان .

  مانا مۇشۇنداق ئۇزۇن مۇساپىنى بىر كىچىنىڭ ئىچىدە تاماملاش ئىنسان ئەمەس  ھەتتا نۇر تىزلىكىدە ماڭىدىغان ئالەم ئايرۇپىلانى ياسالغان تەقدىردىمۇ ( ياساش مۇمكىنغۇ ئەمەس ) 10000 يىلدىمۇ  بىرىنجى ئاسماننىڭ مەۋھۇم بولغان چىگىرا ئوقۇم دائىرىسىگە بىرىش مۇمكىن بولمىغان بىر ئەھۋال ئاستىدا ،ئاللاھنىڭ قۇدىرتى بىلەن  رەسۇلۇللاھنىڭ نەچچە يۇز مىڭ نۇريىلى مۇساپىنى ،بىر كىچىدە تامام قىلىشى بىر  مۆجىزە ھىساپلىنىدۇ .

     چۇنكى ئاينىڭ يىرىلىپ كىتىشىگە باشقا بىر پىلانىتنىڭ كۇچلۇك  سوقۇلىشى سەۋەپ بولۇپ قالاتتى .بىراق رەسۇلۇللاھنىڭ ئاللاھقا قىلغان دۇئاسى بىلەن، ئاي شارى  ھىچقانداق بىر سەۋەپسىز ئوتتۇردىن يىرىلىپ ،ئىككى پارچە بولۇپ يەنە ئەسلىگە كەلگەن .
    مانا بۇگۇنكى ئامىرىكا ناسا ئالەم ئۇچقۇچىللىرى تەرىپىدىن ۋە روسىيە ئالەم قاتناش ئىدارىسنىڭ سانلىق سىستاستىكا مەلۇماتلىرىدىن بايقالغادندەك،ئاي شارىنىڭ يۇزىدە  چوڭقۇر بىر يىرىق ،يەنى بۇرۇن يىرلىغان بىر دەزنىڭ نىشانسىى، ئاي شارىنىڭ يۇزىدە بىر پاكىت سۇپىتىدە ساقلىنىپ قالغانلىقى مەتبۇئاتلاردا ئوتتۇرۇغا قويۇلىۋاتىدۇ  .

   .....ئەينى ۋاقىتتا  ئاي شارى ، رەسۇلۇللاھنىڭ دۇئاسى بىلەن ئەمەس، بەلكى  تاشقى پىلانىتلارنىڭ سوقۇلىش بىلەن ئىككى بولۇپ كەتكەن بولسا ئىدى ،ھەرگىزمۇ بۇ ئىككى پارچىسى ئۆزىنىڭ تەركىبىدىكى كۇچلۇك بولغان  ماگىنىت دولقۇنلىرىنىڭ تىپىشى  كۇچى بىلەن بىر-بىرگە چاپلىشىش بۇيانغا تۇرسۇن ، بەلكى بىر-بىرنىڭ ماگىنىت دولقۇننىڭ تىپىشى تەسىرى بىلەن بىر-بىرنى دەۋرى قىلىپ ئايلنىپ ،مانا بۇگۇنكى كۇندە يەر شارنىڭ بوشلۇقىدا ئىككى ئاي بولۇپ قالاتتى ۋە ئاستىرونومىيە ھادىسلىرىدە باشقىچە ئۆزگىرىش بولاتتى.

مۆجىزە ھەر بىر زاماندا شۇ زامانغا خاسلىقىغا مۇناسىپ بىر رەۋىشتە مەيدانغا كىلدۇ . مۆجىزە مەسىلىسدە سوغۇققانلىق بىلەن ساغلام بولغان ئەقلى يوللار بىلەن ھەقىقى تىگى مەنبىينى بىلىپ شۇ ئاساستا مۇئامىلە قىلىش ۋە شۇ ئىش-ھەركەتكە خاس مۇۋاپىق يول تۇتۇش كىرەك.



    مەن يۇقارقى مۆجىزە ۋە ھىكىمەت توغۇرسىدا تەپەككۇر قىلۋىتىپ شۇ نەرسنى ھىس قىلىپ يەتتىم،رەببىمىز ئاللاھنىڭ نەزىرىدە بىرەر شەيئىنى يارىتىش ئۇچۇن ، ھىكىمەت بىلەن مۆجىزە ئۇسۇلىدىن قايسى بىرىنى ۋاستە قىلىپ يارىتىش شەرىت ئەمەس ،بەلكى ،ئۆزى نىمنى يارىتىشنى خالىسا، شۇ نەرسنىڭ ئىسمىنى ئاتاپ بول ( يەنى كۇن دەيدۇ )شۇ نەرسە شۇ ۋاقتنىڭ ئىچىدە دەرھال ۋۇجۇتقا كىلىدۇ . ئۇ نەرسىنى يۇقاتماقنى ئىرادە قىلىدىكەن .شۇ نەرسىنى يوقال ( يەنى فەيەكۇن ) دەيدۇ شۇ نەرسە شۇ ۋاقىتنىڭ ئىچىدە دەرھال يوقىلىدۇ .
