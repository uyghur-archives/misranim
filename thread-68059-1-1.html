ئائىلىدە يالغۇز بالا بولغانلىقىم ئۈچۈنمىكىن كىچكىمدىن تولىمۇ ياۋايى چوڭ بولغانىدىم،ھەر قىتىم ئاتا ئانامنىڭ مەجبۇرى بىلەن كىرىپ قالمىسام ئارىلىقى ئۆيىمىز بىلەن 5 مىنۇتمۇ كەلمەيدىغان بوۋاملارنىڭ ئۆيىگىمۇ بىرىشقا ئىرىنەتتىم.
باشلانغۇچ ۋاقىتلىرىم ئىدى،تۆۋەن يىللىقتا ئوقۇۋاتقان كۈنلىرىمنىڭ بىرىدە ئاتا ئانام خوشنىلارنىڭ تويىغا بىرىش ئۈچۈن مىنى بوۋاملارنىڭ ئۆيىگە ئىلىپ كىرىپ،ئۆزلىرىنىڭ بەك كەچ قايتىپ كىلىدىغىنىنى،شۇڭا مىنىڭ بىر كىچە مومام بىلەن بىللە ئۇخلاپ تۇرۇشۇمنى ئىيتىپ كىتىپ قىلىشتى.ئىسىمنى بىلسەم شۇ كەمگىچە ئۆز ئۆيۈمدىن باشقا يەردە قونۇپمۇ باقمىغانىدىم.
بوۋاملار مىجەزىمنى بىلگەچ،دەسلىپىدە ئىختىيارىمغا باقتى،مەنمۇ خۇددى ئاپاملار ھىلىلا كىلىپ مىنى ئىلىپ كىتىدىغاندەك ئايىقىمنىمۇ سالغىلى ئۇنىماي ئىشىك تەرەپتىكى ياتاق ئۆينىڭ بوسۇغىسىدا ئولتۇرىۋالغانىدىم،خۇددى سائەت بىلىدىغاندەك تۇرۇپ تۇرۇپ سائەتكە قاراپ قوياتتىم،(ئۇ چاغلاردا سائەتكە قاراشنى بىلمەيتتىم)،كەچ بىر ۋاقىتلار بولغاندا،مومامنىڭ"10 دىن ئىشىپتۇ بالام،ئەمدى كىرىپ ئۇخلىمىسىڭىز ئەتە بامداتقا تۇرالمايمەن"دىگىنىنى ئاڭلاپ تولىمۇ مەيۈس بىر تۇيغۇلارغا كىلىپ قالغانىدىم،بالىلىقتە،خۇددى ئاپاملار ئەمدى مىنى مەڭگۈ ئالغىلى كەلمەيدىغاندەك،مەن ئەمدى بىر ئۆمۈر مۇشۇ ئۆيدە تۇرۇپ قالىدىغاندەك بار ئاۋازىمنى قويۇپ بىرىپ يىغلىغىلى تۇردۇم.
-يائاللا ما بالىنى!مەن سىزنى يىمەيمەن قازانغا سىلىپ!تازىمۇ ياۋايى چوڭ بولدى بۇ بالا!
نىمىشقىكىن مومامنىڭ ئاچچىقى سەل يامان ئىدى،ئۆزى قەشقەردىن بولغاچقىمۇ، تىگى بار گەپلەرنى تولا قىلاتتى،(بۇنىمۇ سەل چوڭ بولۇپ ئاندىن ھىس قىلدىم).جەمەتىمىزدە مومامدىن قورقمايدىغان بىرە ئادەمنىڭ بارلىقىنىمۇ بىلمەيمەن.
-نىمە قىلاسىلە بۇ بالىنى قورقۇتۇپ،بولدى،مەن بىردەم يىنىدا ئولتۇراي،سىلە ئۇخلاۋىرىڭلا...
ھەرقىتىم مومامنىڭ ئاچچىقى كەلگەن ھامان بوۋام ئوتتۇرىغا چىقاتتى،بوۋامنى بىزلا ئەمەس،پۈتۈن ئەل جامائەت بەكلا ھۆرمەت قىلاتتى.(رەھمەتلىك بوۋامنىڭ روھىنى ئاللا جەننەتتە قىلسۇن).شۇ تەرىقىدە ئەتىسى بامدات نامازغا تۇرىدىغان بوۋامنىمۇ ئۇخلاتماي،مەكتەپكە بارىدىغان ئۆزۈممۇ ئۇخلىماي دىگەندەك ئۆرە ئولتۇردۇق.جاھىللىقىمدىنمىكىن،كۆزلىرىم ئۇيقۇلۇقتىن يۇمۇلۇپ كىتىپ بارسىمۇ،بىر ئاماللارنى قىلىپ ئۇخلىماسلىققا تىرىشاتتىم.بىر كەملەردە ئىشىك بوش چىكىلدى،"ئاپام"دىگىنىمچە ئۆزۈمنى ئىشىككە ئاتتىم.ئاپاملار كىرىشىگىلا يىڭىلا ئويغانغان مومام كايىپ كەتتى:
-بۇ بالاڭ خۇددى بىركىم چىشلىۋالىدىغاندەك ئايىقىنىمۇ سالماي ئولتۇردى،سەل قاتتىقراق گەپ قىلسام يىغلىغىلى تۇرىۋىدى،بوۋىسى بىسىقتۇرۇپ بىللە ئولتۇرۇشقان،سەن كەچتە ئۇدۇل ئۆيگە كىتىمەن دىگەنىدىڭغۇ؟
-ھازىر قايتىپ كىلىشىمىز،ئىشىك ئالدىدىن ئۆتۈپ كىتىۋىتىپ دىرىزىگە قارىسام چىراق يىنىقكەن،بۇ بالىنىڭ مىجەزىنى بىلگەچكە ئۇخلىمىدى دەپ ئويلاپ كىرىشىمىز...-نىمىشقىكىن،ئاپامنىڭ چىرايى خاپا كۆرۈنەتتى.
كىيىنچە باشلانغۇچنىڭ يۇقىرى يىللىقىغا ئۆرلىگەنسىرى،نەۋرە ئىنى سىڭىللىرىم بىلەن باردى كەلدەم كۆپىيىپ كەتتى.ئاپام تۆت بالىنىڭ چوڭى ئىدى،شۇڭا مەنمۇ ئەركىلىكىمگە باقماي تۆت نەۋرىنىڭ چوڭى ئىدىم.دادامنىڭ دىيىشىچە،دادامنىڭ ئاتا ئانىسى بەك كىچىك ۋاقىتلىرىدا تۈگەپ كىتىپتىكەن،چىرايىنىمۇ ئىسىگە ئالالمايدىكەن،دادامنىڭ يۇرتتىكى دادا بىر ئانا باشقا بىر ئىنىسىدىن باشقا قىرىنداشلىرى بولمىغاچ،مەن پۈتۈن جەمەتىمىزدە ئەڭ چوڭ نەۋرە ھىساپلىناتتىم.
شۇ كۈنلەردە بوۋاملار مەھەللە ئىچىدىكى كىچىكرەك بىر دۇكاننى ئىجارىگە ئىلىشتى،بۇ دەل تەتىل مەزگىلىمىزگە توغرا كىلەتتى.دۇكاندا دەسلەپتە ئۇششاق چۈششەك يىمەكلىكلەرنى،تۇرمۇش بۇيۇملىرىنى،كىيىنچە،قوشۇمچە تاۋۇز چۈشۈرۈپ ساتاتتى،تۆت بالىنىڭ ھەممىسى خىزمەتتە بولغاچقا،دۇكاندا ئادەتتە بوۋاملار ئۆزلىرى ئولتۇراتتى.بىز تۆت نەۋرىمۇ تەتىلنى باھانە قىلىپ تولىسى بىزنىڭ مەھەللىگە يىغىلاتتۇق،بىر يەرگە كەلگەن ھامان بوۋاملارنىڭ دۇكىنىغا ھۇجۇم قىلاتتۇق.ئەمىلىيەتتە بۇمۇ بىزنىڭ يىغىلىشىمىزدىكى ئاساسلىق مەقسەت ئىدى.
ھەممىمىز مومامدىن قورققاچقىمۇ،مومام ئۆيگە قايتىپ چۈشلۈك ۋە كەچلىك تاماقلارغا تەييارلىق قىلىۋاتقان ۋاقىتلاردا دۇكاندا پەيدا بولاتتۇق ھەم ئۆزىمىز ياخشى كۆرگەن نەرسىلەرنى خالىغانچە ئىلىپ يەيتۇق.بوۋاممۇ خۇددى بۇ نەرسىلەرنى بىز ئۈچۈنلا تىزىپ قويغاندك،مەستلىكى كىلىپ بىر ھازا قاراپ ئولتۇرىۋەتكەندىن كىيىن كىتابىنى ئوقۇشقا كىرىشىپ كىتەتتى.بۇنداق چاغلاردا تولىسى نەۋرە ئىنىمنى مومامنى كۈزىتىشكە قوياتتۇق،ئۇ مومامدىن خەۋەر بەرگەن ھامان،ھەممىمىز دۇكان ئالدىدىكى گۈللۈك ئىچىگە مۆكۈۋالاتتۇق.تولا چاغلاردا نەۋرە ئىنىمغا ھىچنىمە تەگمەيتتى.كىيىنچە بوۋاملار نىمىشقىكىن دۇكاننى ئاچمايدىغان بولۇشتى،سەل چوڭ بولۇپ ئاپاملاردىن ئاڭلىشىمچە بوۋامنىڭ ئاشقازىنى ئاغرىيدىغان بولۇپ قالغانىكەن.
ئوتتۇرا مەكتەپكە يىڭى چىققان ۋاقىتلىرىم ئىدى،ئاپام دوختۇر بولغاچقا خىزمەت ئالدىراشلىقى بىلەن چۈشلىرى قايتىپ كەلمەيتتى،دادام خىزمەت قىلىدىغان نەشىرىيات بىزنىڭ مەھەللىگە يىقىنلا بولغاچقا ئاساسەن چۈشلۈك تاماقنى دادام بىلەن بەزىدە ئۆيدە،بەزىدە سىرتتىن ئىلىپ كىرىپ يەيتۇق.بىر چۈشلۈكى دادامنىڭ سىرتتا ئىشى بولغاچقا،بوۋاملارنىڭ ئۆيىدە تاماق يەيدىغان بولدۇم.
4-سائەتلىك دەرسنى تۈگىتىپ،دوستلىرىم بىلەن جاھاندا يوق پاراڭلارنى قىلىشىپ،ئۆيگە كىچىكىپرەك كىردىم،بوۋام يالغۇز ئولتۇرغانىكەن.كىرىشىم بىلەن بەكلا خۇشال بولۇپ كەتتى.نىمىشقىكىن بىلمەيمەن،ھەر قىتىم بوۋاملارنىڭ ئۆيىگە كىرگىنىمدە،بوۋام خۇددى بىركىم بەلگىلەپ قويغاندەك مىھمانخانا ئۆيدىكى تىلىۋىزۇرنىڭ يىنىدىكىلا ئورۇندا ئولتۇراتتى(سەۋەبىنى ھازىرغىچە بىلمەيمەن،بەلكىم ئۆزى ياخشى كۆرسە كىرەك).شۇ كۈنىمۇ بوۋام شۇ يەردە ئولتۇرغانىكەن:
-كىلىڭە بالام،سىزگە ئالما ئاقلاپ بىرەي،-بۇمۇ بوۋامنىڭ ئادىتى ئىدى،مەيلى ئۆيگە كىملا كەلمىسۇن،بوۋام مىھمانغا ئالما ئاقلاپ بىرىشكە بەكلا ئامراق ئىدى.مىۋە يىمەيدىغان ئادىتىم بولغاچقىمۇ يەنە جاھىللىق قىلىپ يىگىلى ئۇنىمىدىم.(ھەر قىتىم شۇنداق قىلاتتىم).
-ئەمسە تىلىۋىزۇر قويۇپ بىرەي بالام....-مەن بوۋاملارغا ھەيران ئىدىم،چۈنكى بۇ ئۆيدىكى تىلىۋىزۇر خۇددى مىھمان ئۈچۈن سىتىۋىلىنغاندەك،ئۆيگە بىرى كىرگەندىلا قويۇلاتتى.
تامىقىمنى تىزلا يەپ مەكتەپكە بىرىشنى باھانە قىلىپ چىقىپ كەتتىم،بوۋامنى يەنە يالغۇز قالدۇرۇپ چىقىپ كەتتىم...
تولۇقسىزنىڭ يۇقىرى يىللىقىغا چىققان ۋاقىتلىرىم ئىدى،نەۋرە ئىنىم ئىككىمىزنىڭ ئۆيى بىر مەھەللىدە،يەنە كىلىپ ئۇمۇ مەن بىلەن بىر مەكتەپتىكى تولۇقسىزدا ئوقۇغاچقىمىكىن،ئىككىمىز تولىسى بىللە ئوينايتتۇق.
سىنتەبىرنىڭ قاتتىق يامغۇر ياغقان بىر كۈنى،دەرستىن چۈشۈپ مەھەللىگە كىلىپ نەۋرە ئىنىمنى ئۇچرىتىپ قالدىم.ئىككىمىز تاپشۇرۇق،ئۆگىنىشنى ئۇنتۇپ يەنە ئويۇنغا چۈشۈپ كەتتۇق.خۇددى بىركىم ئورۇنلاشتۇرۇپ قويغاندەك،ھەر ئىككىمىزنىڭ ئاتا ئانىسى كەچ قايتىدىغان بولغاچقا،شۇ كۈنى ئۆينى بىشىمىزغا كىيىپ ئوينىدۇق،بىر چاغدا ئۆينىڭ تىلىفۇنى جىرىڭلاپ قالدى،ئىنىم مەندىن بۇرۇن يۈگۈرۈپ كىرىپ،نومۇرغا قارىدى.
-ئاچا،بوۋام تىلىفۇن قىپتۇ،مىنى يوق دەپ قويە!
مەنمۇ ئارتۇق ئويلىمايلا ماقۇل بولدۇم.دىگەندەك بوۋام ئىنىمنى سورىۋىدى،مەنمۇ ئۇنى كۆرمىگىنىمنى ئىيتتىم.
يەنە داۋاملىق ئويۇنىمىزنى ئويناۋەردۇق....بىر كەملەردە قورسىقىمىز ئىچىپ،سىرتتىن تاماق ئىلىپ كىرىش ئۈچۈن چىقتۇق،يامغۇرمۇ سەل پەسلەپ قالغانىدى.
مەھەللىمىزدىكى ناۋايخانا ئالدىغا كىلىپ كۈنلۈك كۆتۈرىۋالغان قوللىرى سوغۇقتىن توڭلاپ تىترەپ كەتكەن بوۋامنى ئۇچرىتىپ قالدۇق،گۆدەكلىك دىگەن شۇ بولسا كىرەك،يەنە ھىچ ئىش بولمىغاندەك سورىدىم:
-نەگە ماڭدىڭىز بوۋا؟
-ھە،خۇداغا شۈكرى،ئىنىڭىز سىز بىلەن بىللىكەن،ئاتا ئانىسىنىڭ ئىشى باركەن،كەچرەك كىلىدىكەن،شۇڭا مەكتەپتىن كەلسە سىلى ئەكىرىۋالسىلا دەپ تاغىڭىز تىلىفۇن قىپتىكەن،مەكتىپىگە بىرىپ تاپالمىدىم،شۇڭا ئىزدەپ چىققان...يۈرە بالام،كىتەيلى،توڭلىدىڭمۇ بالام...-بوۋام تىترەپ تۇرغان قوللىرى بىلەن ئىنىمنىڭ يۈزلىرىدىكى يامغۇر سۇلىرىنى سۈرتكەچ قولىدىكى كۈنلۈكنى ئىنىم تەرەپكە سۈرۈپ قويدى.
يەسلىدىن تارتىپ ئوتتۇرا مەكتەپنى پۈتتۈرگىچە ئۆيدىن ئايرىلىپ باقمىغانىدىم،ئالىي مەكتەپنى شەھىرىمىزدىكى نۇقتىلىق ئالىي مەكتەپتە ئوقۇدۇم،مەكتەپ بىلەن ئۆي ئارىلىقى سەل يىراق ئىدى،شۇڭا ياتاقتا يىتىپ ئوقۇيدىغان بولدۇم.ھىلىمۇ ئىسىمدە،نەۋرىلەرنىڭ ئىچىدە تۇنجى قىتىم ئالىي مەكتەپكە ماڭغىنى مەن بولغاچقىمۇ،كىيىم-كىچەك،تۇرمۇش بۇيۇملىرىمنى سالغان چامادانىمنى تاغام ماشىنىسىغا قاچىلاپ،گەرچە شەھىرىمىز ئىچىدىكى مەكتەپ بولسىمۇئاتا ئانام بىرلىكتە مىنى ئاپىرىپ قويۇشقانىدى،مەكتەپكە مىڭىش ئالدىدا ھەممىمىز بىرلىكتە بوۋاملارنىڭ ئۆيىدە تاماق يىدۇق،ماڭىدىغان ۋاقىتتا بوۋام ئىشىك ئالدىغىچە ئۇزىتىپ چىقتى،نىمىشقىكىن بوۋامنىڭ كۆزلىرىدە ياش لىغىرلاپ قالغانىدى.
كىچىكىمدىن تارتىپ ياۋايى مىجەزىم سەۋەپلىك بوۋاملارنىڭ ئۆيىىگە كەمدىن كەم كىرىدىغان ئادەم،ئالىي مەكتەپكە چىقىپ ياتاقتا ياتقاندىن كىيىن تىخىمۇ ئاز كىرىدىغان بولۇۋالدىم،دەم ئىلىش كۈنلىرىم ئۆيگە قايتقىنىمدا بوۋاملار تاماققا چاقىرسا،ئۇچرىشىشىم بارلىقىنى باھانە قىلىپ قىچىپ كىتەتتىم،ئالىي مەكتەپتىكى 3 يىل يىل جەريانىدا ھىيت ئايەم كۈنلىرى كىرگەندىن باشقا،بوۋاملارنى بەكلا ئاز يوقلىدىم،مەھەللىگە كەلگىنىمدە،يوللاردا دوستلىرىم بىلەن ماڭغانلىرىمدا،بوۋامنى بەزىدە يالغۇز، بەزىدە مومام بىلەن ئۇچرىتىپ قالاتتىم،بۇنداق چاغلاردا بوۋام پات پات كىرىپ تۇرۇشۇمنى تەكىتلەپ كىتەتتى،مەنمۇ ئاخىرى چىقماس قۇرۇق ۋەدىلەرنى بىرىپ،بوۋامنى نەۋرىسى كىرىپ قالىدىغاندەك ئىشىككە نەچچە قىتىملاپ تەلمۈرىتىپ،ئۆز سەنىمىمگە دەسسەپ يۈرىۋەردىم.
ئالىي مەكتەپنى 3 يىللا ئوقۇپ چەتكە چىقىش ئارزۇيۇم بويىچە مەكتەپتىن چىقىپ كەتتىم ھەم بىر تىل تەربىيلەش كۇرسىدا ئوقۇۋاتاتتىم،2010-يىلى،3-ماي،يۇرتقا توي مۇناسىۋىتى بىلەن كەتكەن بوۋام موماملار بوۋامنىڭ زىققا كىسىلى قوزغىلىپ قالغانلىقى سەۋەپلىك شۇ كۈنىلا ئايروپىلاندا قايتىپ كەلدى،ئۇرۇغ تۇغقانلار ھەممىمىز بوۋامدىن ئەنسىرەپ ئۆيگە يىغىلدۇق،6 يىل ئىلگىرى نەۋرىلەر سىپىگە قىتىلغان ئەڭ كىچىك نەۋرى ئىنىممۇ كەلگەنىدى.بوۋام ئىنىمنى كۆرۈپ جانلىنىپ قالغانىدى.
-كىلە بالام،ساڭا ئالما ئاقلاپ بىرەي...،-بوۋام يەنە بۇرۇنقى ئورنىدا ئولتۇرۇپ ھەممىمىزگە بىردىن ئالما ئاقلاپ بەردى...كەچكە يىقىن باشقا نەۋرىلەر بوۋامنىڭ زورلاشلىرى بىلەن ئاتا ئانىلىرى بىلەن ئۆيگە قايتىشتى،بوۋام ئەنە شۇنداق ئادەم،بىشىغا تاغدەك ئىش كەلسىمۇ ئۆزى ئەجىر قىلىپ بىقىپ چوڭ قىلغان بالىلىرىنى ئاۋارە قىلىشنى خالىمايتتى.سىرتتىكى قاتتىق بوران ۋە يامغۇر سەۋەبىدىنمۇ ياكى باشقا سەۋەپمۇ،كىچىك تاغام ماشىنا توسىيالمىغانلىىقىنى باھانە قىلىپ بىر ئائىلە بوۋاملارنىڭ ئۆيىدە قونۇپ قىلىشتى.بىزمۇ ئۆيگە چىقىپ كەتتۇق.
غەپلەت ئۇيقۇسى شۇنداق بولىدىكەن...كىچە بىر ۋاقىتلاردا تولىمۇ قورقۇنچلۇق بىر چۈشتىن چۆچۈپ ئويغىنىپ كەتتىم،ئۆيىمىزنىڭ ئىچى ئادەتتىن تاشقىرى جىمجىت ئىدى،كىچە بىرۋاقىتلارغىچە كىتاپ تەھرىرلەيدىغان دادامنىڭ ئۈستەل چىرىقىمۇ ئۆچۈك ئىدى،ئۆي ئىچىنى بىر ئايلىنىپ چىقىپ ئاپامغا تىلىفۇن قىلدىم.ئاپامنىڭ دىيىشىچە بوۋامغا ھەمرا بولۇپ تاغام،موماملار بىلەن دوختۇرخانىغا كەتكەنىكەن،يۈرىكىم ئەنسىز سوقۇپ كەتتى.بىر كەملەردە كىچىك تاغامنىڭ ئايالىدىن تىلىفۇن كىلىپ،مەن بوۋاملارنىڭ ئۆيىگە كىرىپ نەۋرە ئىنىمغا قاراپ تۇرىدىغان بولدۇم.ئۇزۇن ئۆتمەي ئىچكىرىدىكى ئالىي مەكتەپتە ئوقۇيدىغان نەۋرە ئىنىمدىن باشقا ھەممە نەۋرىلەر ئۆيگە يىغىلىشتۇق....
يىرىم كىچىدە ئەنسىز جىرىڭلىغان تىلىفۇننى ئىلىپ،سىكۇنت ئۆتمەي تورۇپكا قولۇمدىن چۈشۈپ كەتتى...نەۋرە ئىنى سىڭىللىرىمغا بوۋامنىڭ ۋاپاتىنى يەتكۈزگىنىمدە ئەمدىلا 6گە كىرگەن ئەڭ كىچىك ئىنىم مىنى قۇچاقلاپ يىغلاپ كەتكەنىدى:-ئاچا،ئەمدى كەلسەم كىم ماڭا ئالما ئاقلاپ بىرىدۇ...-ئۇنىڭ كىچىككىنە ئىغىزلىرىدىن چىقىۋاتقان سۆزلىرىدىن بىزمۇ ئۆزىمىزنى تۇتالمىدۇق...
شۇنداق قىلىپ بوۋام بىزنى بىر كىچىمۇ بىقىۋالغىلى قويماي باقىيغا سەپەر قىلدى،ئاللاغا يالۋۇردۇم،بىر كۈن بولسىمۇ بوۋامنى قايتۇرۇپ بەرسە،بۇرۇنقىدەك بوۋامنى تەلمۈرىتىپ يۈرمەيلا ئۆيگە كىلىپ قانغۇچە بىر مۇڭداشسام،يامغۇردا تىترەپ كەتكەن قوللىرىنى تۇتۇپ قويسام...لىكىن بوۋام كەلمىدى،ھەم مەڭگۈگە كەلمەسكە كەتتى...شۇ كۈندىن باشلاپ بوۋامنى ئويلىساملا ئۆتمۈشتىكى گۆدەكلىكلىرىم،بوۋامنىڭ يامغۇردا تىترەپ تۇرغان سىماسى كۆز ئالدىمغا كىلىۋىلىپ،شۇ بالىلىقتىكى گۆدەكلىكلىرىمگە ئەپسۇسلىنىمەن.
2012-يىل كىرىش ھارپىسىدا ۋەتەندىكى ئۇرۇغ تۇغقانلىرىم بىلەن توردا كۆرۈشۈپ،يىڭى يىللىق سالاملار يوللىغانىدىم...سىزنىڭمۇ يىڭى يىلىڭىزغا مۇبارەك بولسۇن بوۋا،ئاللا روھىڭىزنى مەڭگۈ جەننەتتە قىلسۇن...
