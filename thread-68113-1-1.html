تۇرمۇش بۇلاقلىرى توختىماي قوشۇلۇپ ئاقسۇن



بۇرايىن داسۇن (ئامىركا)

تۇرمۇش دەل بەش دانە شارچىنى بوشلۇققا چۆرۈپ تۇتۇۋېلىش ئويۇنىغا ئوخشايدۇ .بۇ بەش شارچە ئايرىم-ئايرىم ھالدا:خىزمەت،ئائىلە،سالامەتلىك،دوستلۇق ۋەروھتۇر .بەش دانە شارچىنى بوشلۇققا چۆرۈپ تۇتۇۋىلىش ھەقىقەتەنمۇ بىر سەنئەت .ھايال ئۆتمەيلا سىز پەقەت خىزمەتنىڭلا بىر رىزىنكە توپقا ئوخشايدىغانلىقىنى،يەرگە چۈشۈپ كەتسىمۇ يەنە ئۆرلەيدىغانلىقىنى ،ئەمما باشقا تۆت دانە توپنىڭ ئەينەك ئىكەنلىكىنى ،يەرگە چۈشۈپ كەتسە زىدە بولۇپ ،چاك ئالدىغانلىقىنى ياكى چىقىلىپ چۇل-چۇل بولۇپ كېتىدىغانلىقىنى،ئۇمومەن قىلىپ ئېيتقاندا يەنە قايتا ئەسلىگە كەلتۈرگىلى بولمايدىغانلىقىنى بايقايسىز .شۇنىڭ ئۆچۈن بىز تىرىشىپ ئۆز تەڭپۇڭلۇقىمىزنى ساقلىيالىغىنىمىزدىلا ئۇلارنىڭ ھەممىسىنى قۇلىمىزدا ياخشى ئوينىتالايمىز .

ئۇنداقتا قانداق قىلىش كېرەك ؟
ئۆزىڭىزنى داۋاملىق باشقىلار بىلەن سىلىشتۇرماڭ .بۇنىڭدا ئۆزىڭىزنى بارغانسىرى ھىچنىمىگە ئەرزىمەيدىغاندەك ھىس قىلىپ قالىسىز .خۇددى سىزنىڭ بارماق ئىزىڭىزغا ئوخشاش دۇنيادىكى ھەر بىر ئادەم پەقەت بىر دىنبىر بولىدۇ .
باشقىلار مۇھىم دەپ قارىغان نەرسىگە ئاساسەن ئۆزىڭىزگە غايە-نىشان بېكىتمەڭ .ئەكسىچە ئۆزىڭىز گۈزەل ھىس قىلغان نەرسىنى قولغا كەلتۈرۈش ئۈچۈن تىرىشىڭ.
ئۆز قەلبىڭىزگە يىقىن نەرسىلەرنى ئەزەلدىنلا بار ،خۇددى تۇرۇبا سۈيىگە ئوخشاش ئېلىشنى خالىغاندا ئالالايمەن دەپ قارىماڭ.ئۇلارنى ئۆز قارچۇقىڭىزنى ئاسرىغاندەك ئاسراڭ .ئۇلاردىن ئايرىلىپ قالسىڭىز ،سىزپەقەت يۈرىكى بار قەلبى يوق تىرىك موردىغا ئايلىنىپ قالىسىز .
بىر ئۆمۈر ئالدىراشچىلىقتا ئۆتۈپ ،ئۆزىڭىزنىڭ قەيەردىن كەلگەنلىكىڭىزنى ،نەگە بارىدىغانلىقىڭىزنى ئۇنتۇپ قالماڭ .ھايات بىر مەيدان سۈرئەت تالىشىپ يۈگۈرۈش مۇسابىقىسى ئەمەس ،بەلكى بىر قەدەم -بىرقەدەمدىن باسىدىغان مۇساپە .
تۈنۆگۈن ياكى ئەتە بىلەنلا بولۇپ كېتىپ ،بۈگۈننى بارماقلىرىڭىز ئارىسىدىن ئۆتكۈزۋەتمەڭ .ھەر بىركۈننى پەقەت ھەربىر كۈنى ئۆتكۈزگەندىلا سىز ھەممە كۈندىن بەھىرلىنەلەيسىز .ھايات سان -مىقدار بىلەن ئەمەس ،بەلكى سۈپىتى بىلەن ھىسابلىنىدۇ .
ئەگەر يەنە تەقدىم قىلىش ئىمكانىيىتىڭىز بولسا ،ئۇنداقتا ئاسانلا بولدى قىلماڭ .سىز تىرىشىشنى تۇختاتمىغانغا قەدەر ھىچنىمە ھەقىقى تۈردە ئاياغلاشمايدۇ .
ئۆزىڭىزنىڭ مۇكەممەل ئەمەسلىكىڭىزنى ئىقرار قىلىشتىن قورقماڭ .بىز دەل ئىنتايىن نازۇك ئىنچىكە يىپ ئارقىلىق
ئۆزئارا باغلانغان .
خىيمخەتەرگە يۈزلىنىشتىن قورقماڭ .بىز ھەممىمىز چېنىقىش داۋامىدا يۈرەكلىك بولۇشنى ئۈگەنگەن .
ئۆزىڭىزگە سۆيگۈ تاپالمىدىم دەپ ،بۇنىڭلىق بىلەن مەڭگۈلۈك بېكىنىۋالماڭ .سۆيگۈگە ئېرىشىشنىڭ ئەڭ تىز ئۇسولى بېرىشتۇر.سۆيگۈدىن ئايرىلىپ قېلىشنىڭ ئەڭ تىز ئۇسولى تەڭگە پۇلغا مۇئامىلە قىلغاندەك يانچۇقتا چىڭ ساقلاش،سۆيگۈنى ساقلاشنىڭ ئەڭ ياخشى ئۇسولى ئۇنى ئەركىنلىك بىلەن قاناتلاندۇرۇشتۇر .
ئۇنتۇپ قالماڭ ،بىر ئادەمنىڭ ھىسيات جەھەتتىكى ئەڭ چوڭ ئېھتىياجى باشقىلارنىڭ چۈشىنىشىگە ئېرىشىشتۇر .
يىڭى نەرسىلەرنى ئۆگۈنۈشتىن قورقماڭ .بىلىمنىڭ ۋەزنى يوق بولۇپ ،ئېلىپ يۈرۈشكە بولىدىغان خەزنىدۇر .ھىچكىم ئۇنىڭ ئېغىرلىقىدىن بېسىلىپ قالمايدۇ .ئەكسىچە قانچە كۆپ بولغانسىرى جىسمانى ،روھى جەھەتتىن بەردەم بولىسىز .
تۈنۆگۈن بولسا تارىخ ،ئەتە بولسا تېپىشماق ،بۈگۈن بولسا سوغا ،شۇڭا ئىنگىلىزچىدا بىز بۈگۈننى دەيمىز.


(ساۋۇت ھاجى تەرجىمىسى)
