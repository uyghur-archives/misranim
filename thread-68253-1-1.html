                                                           مۇھەببەت   

      مۇھەببەت بەزىلەر ئۈچۈن ئاجايىپ شاتلىقلارنى ئىلىپ كەلسىمۇ ، يەنە بەزىلەرنى تۈگىمەس  
ئازاپ دىڭىزغا  غەرىق  قىلدۇ . ئۇلار مۇھەببەتنى ئازاپنىڭ ئەلچىسى دەپ قاراپ ئۆزلىنىڭ  غازان  
بولغان مۇھەببىتنى ئەسكە ئىلىشىدۇ.لىكىن شۇ ئەسكە ئىلىشلاردا ھەممىدىن بەكراق ئويلايدىغىنى ،  
شىرىن تۇيغۇغا چۆمدۈردىغىنى يەنىلا دەسلەپكى چاغلاردىكى مۇھەببەتنىڭ شات-خۇرام ئەسلىمىسدىن  
ئىبارەت بۇلىدۇ . ئۇلار شۇ ۋاقىتنىڭ قايتىپ كىلشىنى ئۆمۈت قىلشىدۇ ...   

                                                    1

                                   مۇھەببەت _  
                                   سەھەردىكى چۈشتەكلا  گۇيا ...   
                                   ئەنە ئۇنىڭ شاخلىرىدا سۇلىشىپ ،  
                                   غازان گۈللەر تۇرىدۇ سارغىيىپ .   
                                  يوقتىشقا  جىلۋىسىنى  بىراقلا .
                                  ئەتراپىنى ئوراپ ئالغان تىكەنلەر ،  
                                  تۇرار بۇندا مەسىىرلىك قارىشىپ؛      

                                                 2   

                                   مۇھەببەت _  
                                   ئۆتمۈشتىكى خاتىرە پەقەت ...  
                                   ئەنە،ئۇنىڭ قۇرۇق قاخشال شېخىدا،
                                   ھۇقۇش ماتەم كۆيلىرىنى توۋلايدۇ .   
                                   گۆرىستاننى ئەسلىتىپ ھەم يىغلايدۇ .  
                                   ھەىسرەت ئېتى تېزگىن سىرىپ چاپچىغان .
                                   جۇدالىققا پەريات ئۇرۇپ يىغلايدۇ ؛

                                                  3

                                    مۇھەببەت _  
                                    بىلىنىدۇ ئازاپلىق ئۆتمۈش ...  
                                    باغلىرىدا مۇدھىش تۇمان ئەگىيدۇ ،
                                    غازان پەسلى ياپرىقىنى ئۈششۈتكەن .  
                                    بۇندا سۆيگۈ ماتىمىگە چۈمۈشكەن ،      
                                    نەغمىچى قۇش كىسىلدىمۇ تىللىرى .   
                                    مۇسىبەتلىك ماكانىدىن كىتىشكەن ؛   

                                                   4  

                                     مۇھەببەت _  
                                     جۇدالىقنىڭ كۈلەڭگىسى ئۇ ...  
                                     ھىجران قۇشى قەھ-قەھ ئۇرۇپ كۈلىدۇ ،  
                                     جىسمى ئۇنىڭ كۈلكىدىنلا پۈتكەندەك .   
                                     غازان پەسلى پىغانىسنى ئۆرلەتكەن ،  
                                     بۇندا ئاچچىق تولغىنىدۇ يۇپۇرماق .  
                                     ئۆتمۈشتىكى باھارىنى كۈتكەندەك ؛  


                                                                       _ياسىن ئاۋۇت .

جىنىمغا تەككەندە نەشتەر سۆزلىرىڭ،
چىقىلدى ئەينەكتەك چۇل-چۇل يۈرۈكۈم.
ئازاپقا ئايلاندى ۋىسال قاچىسى ،
كىملەرگە ئاياندۇر ئۆلۈك تىرىكىم.

قەھىرلىك كۆزۈڭنى تىكسەڭ كۆزۈمگە.
ئايىغىڭ ئاستىغا يىقىلدى غورۇر .
يىغلىسام ئىزىلىپ قۇمدەك چىچىلىپ،
سەن بەلكىم شۇنچىلىك سىزىسەن ھوزۇر.

مۇھەببەت سەن ئۈچۈن ھىچنىمە ئەمەس،
ئالدىڭدا مەن پەقەت ئاشىق قەلەندەر.
خالىساڭ مەھپۇستەك دارغا ئاسىسەن،
خالىساڭ يۈرەككە ئۇرىسەن خەنجەر.
ئوڭۇشسىزلىققا ئۇچرىغاندا بەل قويىۋىتىش،مەغلوبىيەتنىڭ ئىككى ھەسسىلىنىشىدىن باشقا نەرسە ئەمەس.

جىنىمغا تەككەندە نەشتەر سۆزلىرىڭ،
چىقىلدى ئەينەكتەك چ ...


               نۇرىيە سىزگە ئەگەشتىم .

مۇھەببەت سەن نىمە جاۋاپ بەرگىنە؟
كۆڭۈلنىڭ بىردەملىك خوشلۇقىمۇسەن؟
ياردەمدەك دىللارنى مەھلىيا قىلغان،
پەدەزخور سەتەڭنىىڭ قاشلىقىمۇ سەن؟


مۇھەببەت سەن نىمە جاۋاپ بەرگىنە؟
بىر ئەرنىڭ قىرىق قىز ئېلىشىمۇ سەن؟
ئاشىقلار يىتەلمەي مەشۇق ۋەسلىگە ،
دىللارنىڭ ھەسرەتتە قېلىشىمۇ سەن؟


مۇھەببەت سەن نىمە جاۋاپ بەرگىنە؟
پۇل پەرەس قىزلارنىڭ خۇيلىرىمۇ سەن؟
پەقەتلا پۇل ئۆچۈن نىكاھنى چاتقان،
تەكتى يات دىللارنىڭ سىھرىمۇسەن؟


مۇھەببەت سەن نىمە جاۋاپ بەرگىنە؟
پوشكالچى موللىنىڭ تۇمارىمۇ سەن؟
سۈڭگۈچنى ساقلاتقان قانچە يىگىتنىڭ،
يالغانچى جۇۋاننىڭ خۇمارىمۇ سەن؟


مۇھەببەت سەن زادى شۇنداقلا بولساڭ،
بولدى بەس يىراق تۇر سېنىڭسىز ئۆتەي.
باشقىچە مۇھەببەت بولسا ئەگەردە ،
ئېيتىڭلار دوستلۇرۇم مەن شۇنى كۈتەي.

مۇھەببەت ناھايىتى گۈزەل بىر چۈشتۇر ، سىز ئۇنىڭ ئىللىق باغرىدا ئەللەيلەنگىنىڭىزدە دۇنيانىڭ گۈزەللىكى ،ئىللىقلىقى ۋە يارقىنلىقىنى ھىس قىلالايسىز ...
ئۇ شۇنداق شىرىن بىر چۈش ...
ئەمما بۇ چۈشتىن ئويغانغىنىڭىزدا ، يۈرىكىڭىزنى چىرمىغان ھىجران، تۆكۈلگەن ياشلار سىزنى سەھرايى كەبىر چۆلىدە تىمىم سۇغا تەشنا بولغان جانلىقلاردەك ھايات-مامات ئارىسىدا جان تالىشىۋاتقاندەك تولغىنىسىز ،باغرىڭىز كۆيۈپ كۈل بولىدۇ ، جىسمىڭىز ئاستا ئاستا تەبىئەتنىڭ بىر قىسمىغا ئايلىنىدۇ ....
مانا بۇ سىزنىڭ مۇھەببىتىڭىز ....
ئوڭۇشسىزلىققا ئۇچرىغاندا بەل قويىۋىتىش،مەغلوبىيەتنىڭ ئىككى ھەسسىلىنىشىدىن باشقا نەرسە ئەمەس.
