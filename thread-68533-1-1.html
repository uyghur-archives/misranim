بەللىرى سەل مۈكچەيگەن ، كىيىنىشلىرى شۇنچىلىك ئاددى بۇ كىشى مېنىڭ مېھرىبان دادام .
قەھرىتان سوغۇقتا تۇمىقىنى چۆكۈرۈپ كىيىپ تۈگۈلۈپ مېڭۋاتقان ، كەينىگە يېنىش-يېنىشلاپ قاراپ < سوغۇقتا تۇرماي كىرىپ كەت ، سېنى ئاللاھقا تاپشۇردۇم > دەپ كېتۋاتقان شۇ ئاددى كىشى مېنىڭ دادام . كۆڭلۈم شۇنداق يېرىم بولدى .

مېنىڭ ئىشقا چۈشىدىغان خەۋىرىمنى ئاڭلاپ خۇشاللىقىدا ئىشىم بار ئىدى دېمەي ، سوغۇقنى سوغۇق دېمەي كەلگەن شۇ دادامنى بېكەتكە ئەچىقىپ يولغىمۇ سېلىپ قويالماي قېقىپ قويغان قوزۇقتەك تۇرۇپ قالدىم ، دادامنىڭ يىراقلاۋاتقان سىماسىغا قاراپ ، كۆز ياشلىرىم تۆكۈلدى ، دادامغا نېمە قىلىپ بېرەلەرمەن ھە ؟
بايا تېخى ئىشخانىدا ئولتۇرسام بىر ناتونۇش نومۇردىن تېلفۇن كەلدى ،
- ئەسسالامۇ - ئەلەيكۇم .
- ۋەئەلەيكۇم - ئەسسالام ، بالام ، مەن !
- دادا ؟
- ھە مەن ؟
- ھە ، سىز نەدە ؟
- مەن ئىدارەڭنىڭ ئالدىدا ، بالام ، بۈگۈن ماڭاي دېگەن ، شۇڭا ئۆينىڭ ئاچقۇچىنى ساڭا بەرگەچ خوشلىشاي دەپ چىقىشىم . 5 مىنۇت ۋاقتىڭنى چىقىرىپ چىقساڭ خىزمىتىڭگە تەسىر يېتەرمۇ ؟
- ياق ، ياق ، مەن ھازىر چىقاي .
چاپىمنىمنى كىيىپلا ئىدارىنىڭ ئالدىغا يۈگۈردۈم .
- ئاستا ماڭ ، ئاستا . ما بالىنى ، چوپچوڭ بولغاندىمۇ يۈگۈرۈپ يۈرەمسەن ؟
ھاسىرغىنىمچە دادامنىڭ ئالدىغا كېلىپ توختىدىم .
- نېمانداق تۇيۇقسىز كېتىدىغان بولۇپ قالدىڭىز ؟
- دۇككان بەك تاقىلىپ كەتتى بالام ، بۇ يەردە بىكار يۈرەمدىم ئىشى يوق ئادەمدەك ، سەنمۇ بىر چىرايلىق ئىشقا چۈشتۈڭ ، مەنمۇ خاتىرجەم بولدۇم .
- ئەمسە ساقلاپ تۇرۇڭ ، مەن رۇخسەت سوراپ چىقاي .
- ياق ، ياق ، ما بالىنى ، گەپ ئاڭلا ، قايسى كۈنى بىر مۇنچە تەربىيە قىلدىم ، ياىشى ئىشلە ، كۈن ئۆتكۈزمە ..........
- يەنە بىرەمدە چۈش بولىدۇ ، مەن بىلەن بىللە ئىشخانىغا كىرىپ ئولتۇرۇپ تۇرۇڭ ، چۈشلۈك تاماقنى بىللە يەپ ئاندىن مېڭىڭ .
- بولدى ، باشقلىڭنىڭ ئالدىدا سەت تۇرىدۇ ، مە ما ئاچقۇچنى ، ما پۇلنى مۇئاشىڭ چىققۇچە ئىشلىتىپ تۇر .
دادام ئۇنىمىغانغا ئۇنىماي پۇلنى قولۇمغا تۇتقۇزدى .
- ئەجەپبىر دادا ، يولدا نېمە يەيسىز ؟ سوغۇقتا چىقىپ يۈرمىسىڭىزمۇ بولاتتى ، يېقىن بولمىسا يە ئارلىقمۇ .
- سەن بىلەن خوشلىشىپ ماڭاي دېدىم . نان بىلەن چاي ئېلۋالدىم ، يولنىڭ تامىقىنى يېسەم زىيان قىلىدىكەن . دادام سومكىسىنى ئېچىپ ماڭا كۆرسەتتى .
دىماقلىرىم ئېچىشتى . ئاخشام مەن ئىچىپ تاشلاپ قويغان ئىچىملىكنىڭ قۇتىسىغا چاي قاچىلىۋاپتۇ . ئارتۇق گەپ قىلالمىدىم . دادام بۇ يەرگە ئىككى ئابتۇبوس ئالمىشىپ كېلىدۇ ، ئەگەر مەن بولسام كېلەمتىم ؟ تېخى ئاپتۇبۇستا . ئۈچ تاش بېكەتكە قانچىنجى يول ئاپتۇبۇسى بارىدىغانلىقى سورىۋېلىپ مېنى قايتا - قايتا ئاللاھقا تاپشۇرغانلىقنى جېكىلەپ كېتىپ بارغان دادام.....
كەينىمگە قايتتىم ، دادام ياشىنىپ قاپتۇ . بۇرۇن دادامنىڭ مېڭىشىغا يېتىشەلمەي پېشىگە ئېسىلىپ كېتىدىغان مەن ، مانا ھازىر دادامنىڭ ئېغىرلاشقان قەدەملىرى ، شۇنچە تېز ماڭسىمۇ ماڭا يېتىشەلمەي ھاسىراشلىرى ....
مېنىڭ دادامدەك پۇلۇم يوق ، يەنە ئولتۇرۇشۇم تاكسى .
مېنىڭ دادامدەك پۇلۇم يوق ، يەنە سەپەرگە ئېلىپ مېڭىشىم تەييار يېمەكلىك ۋە گازلىق ئىچىملىك .
مېنىڭ دادامدەك پۇلۇم يوق ، يەنە بىرەر قېتىم ئاممىۋى تېلفۇندىن ئۇرمايمەن .
مېنىڭ دادامدەك پۇلۇم يوق ، يەنە دوسلىرىم بىلەن تالىشىپ تۇرۇپ پۇل تۆلەيمەن .
مېنىڭ دادامدەك پۇلۇم يوق ، يەنە تېخى ئۇچى ئاقرىپ كەتكەن ، پاشنىسى ئۇپرىغان ئاياقنى ھەرگىز كەيمەيمەن .
مېنىڭ دادامدەك پۇلۇم يوق ، يەنە كۆكتاتلارنى يايمىدىن ئەمەس تاللا بازىرىدىن سېتۋالىمەن .
مېنىڭ دادامدەك پۇلۇم يوق ، يەنە كىيىم ئالسام باھاسىغا قارىماي ماركىسىغا قارايمەن .
مېنىڭ دادامدەك پۇلۇم يوق ، يەنە تېلفۇنۇمنىڭ سىرى چۈشۈپ كەتكىچە ئىشلەتمەيمەن .
مېنىڭ دادامدەك پۇلۇم يوق .........................

ئەجەپ ۋاپاسىز بالىكەنمەن ، مېنى دەپ كەلگەن دادامنى < يۈرە دادا > دەپ قولىدىن تۇتۇپ ئاپىرىپ بېلەتنى ئېلىپ يولغا سېلىپ قويالمىدىم .
ئەجەپ ۋاپاسىز بالىكەنمەن ، مېنى دەپ كەلگەن دادامنى < يۈرە دادا > دەپ قولىدىن تۇتۇپ ئاشخانىغا باشلاپ ئىسسىق بىر قاچا تاماق سۇنالمىدىم .
ئەجەپ ۋاپاسىز بالىكەنمەن ، مېنى دەپ كەلگەن دادامنى < يۈرە دادا > دەپ قولىدىن تۇتۇپ تاكسىغا سېلىپ قويالمىدىم .
بۇرۇن ، خىزمەتكە چىقسام دادام - ئاپامنى ئانداق باقىمەن ، مانداق باقىمەن دەپ ئويلاپ كېتەتتىم ، بىرىنىمۇ قىلالىمىدىم . دادامنىڭ كەينىگە يېنىش - يېنىشلاپ قاراپ < سوغۇقتا تۇرماي كىرىپ كەت ، سېنى ئاللاھقا تاپشۇردۇم > دېگەن ئاۋازى قۇلىقىمدىن كەتمىدى ، دادام مېنى دەپ شۇنچە يىراق يەردىن كېلەلىدى ، مەن نىمىشقا ئاپتۇبۇس بېكىتىگە بولسىمۇ بارمىدىم ؟
بۈگۈن دادامنىڭ سىماسى كۆز ئالدىمدىن كەتمەي شۇنداق بىئارام بولدۇم ، ئۆزەمنىڭ ۋاپاسىزلىقىدىن خىجىل بولغان ئوخشايمەن ، مېنى مۇشۇ ھەيۋەتلىك دەرۋازىنىڭ ئىچىدە ئىسسىق ئىشخانىدا ئولتۇرۇپ ئىشلەشكە ئېرىشتۈرگەن شۇ دادامنىڭ مېھنىتى ، دادامنىڭ پۇلى . لېكىن مەن بۇنى ئۇنتۇپ قېلۋاتامدىم قانداق ؟
ئېھ ئۇلۇغ ئاللاھ ، دادامنىڭ تېنىنى ئامان قىلغايسىز ، دادامنى بېقۋېلىشقا ، بالا باققانىڭ راھىتىنى كۆرسىتىشكە پۇرسەت بەرگەيسىز ، ئۇلۇغ ئاللاھ ، مېنىڭ دادامدەك پۇلۇم يوق ،لېكىن دادامنى ياخشى باققۇم بار .........
