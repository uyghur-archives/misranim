              كەشپىياتچى ئالىمجان ئاكاش بىلەن بولغان     
                                     تىلىفۇن  سۆھبىتىم                          

ئەسسالامۇ ئەلەيكوم قەدىرلىك ئۇيغۇر قىرىنداشلىرىم مەن تۈنۈگۈن( قۇناخۇنلاردىن )ھەزەر ئەيلەيلى تىمىسىدىكى ماقالنى ئۇقۇپ كۆڭلۈم شۇنچىلىك يىرىم بولدى،گەپنىڭ ئۇچىقىنى قىلسام چۈشىنەلمىدىم،ئىنكاسچىلارنىڭ كۆپىنچىسى چۈشىنەلمەپتۇ ،بىر ئۇقۇسام ئالىمجاننى ئاقلىغاندەك بىر ئۇقۇسام ئۇنىڭ كەشپىياتىنى ئاغدۇرغاندەك،  بۇگۈنكى كۈندىمۇ ئاق تاغلىق-قار تاغلىق بۇلۇپ بىر- بىرىنى يەۋەتكىدەك بۇلۇپ كىتىشى ئەينى ۋاختىدىكى ئىككى سۈلۈك يەنى ئىشقىيە ،ئىسھاقىيە سۈلىكىنى ئىسىمگە سەپ قويدى،ئەينى ۋاختىدىمۇ ئىككى سۈلۈكنىڭ ئۆزلىرىنى تەرغىپ قىلىدىغان ئىدىيىسى بۇلۇپ ئۆزلىرىگە كۈچ توپلاش ئۈچۈن نۇرغۇنلىغان يالغان ئەپسانىلەرنى تۇقۇپ خەلىقنى ئالداپ بىچچارە نادان خەلىقىنىڭ ئارىسىغا نىزا تەپرىقچىلىق پەيدا قىلىپ جەڭگە جىدەل چىقىرىپ زىمىنىنى سىتىپ خەلىقىنى باشقىلارغا قۇل قىلىشقا ئاساس سىلىپ بەرگەن، بۈگۈنگىچە شۇ نادانلىقىنىڭ دەدىنى تا ھازىرغا كەلگۈچە تارتىپ كىلىۋاتىمىز، بۈگۈنكى كۈندىكى سۈلۈكچىلەرنىڭ، بىرى ئالىمجان ئاكاشنىڭ كەشپىياتنى ماختاپ كۆككە كۈتۈرسە يەنە بىرى يەر بىلەن يەكسان قىلىپ ھەممىسىنى نۆلگە تەڭ قىلىپ ئەجىرىنى يوق قىلدى، شۇ مۇناسىۋەت بىلەن يىرىم سائەت ئالدىدا ئالىم جان ئاكاشقا تىلىفۇن قىلىپ ئىشنىڭ ھەقىقىتىنى سۈرۈشتۈردۈم،تۈۋەندە مەن ئالىمجان ئاكاش بىلەن بولغان قىسقىچە تىلىفۇن سۆھبىتىمىنى ھۇزۇرۇڭلارغا سۇنماقچى ھەر قانداق ئىش دەلىل ئسپات بىلەن بۇلىشى كىرەك.

ئەسسالامۇ ئەلەيكوم،ياخشىمۇ سىز ؟
ۋە ئەلەيكوم ئەسسالام ياخشى ئۈزىڭىزچۇ؟
سىز ئالىمجان ئاكاشمۇ؟
ھە شۇنداق.
سىز بىلەن بىر ئىش تۇغۇرلۇق سۆزلەشمەكچىتىم ئاۋەر قىلىدىغان بولدۇم.
سىز كىم بۇلىسىز؟
مىنىڭ ئىسىمىم ئابلىكىم شىركەت خىزمەتچىسى
ھەم مىسرانىم ئەزاسى تورىدىكى ئىسمىم خەلىفە
توردا سىزنىڭ كەشپىياتىڭىزنى 2009 -يىلدىن تارتىپ كۈرۈپ كىلىۋاتىمەن.
ھەم سىز بىلەن ئارلىقتا سۆھبەتلەشمەكچى بۇلۇپ تىلىفۇنمۇ قىلغان.سىزنىڭ ۋاكالەتچىڭىز بىر كۈنى ۋاقىت چىقىرىپ كۈرۈشىدىغانمۇ بولغان لىكىن شۇ چاغدا چەتئەلدىن مىھمانلار كىلىپ شۇلار بىلەن ئالدىراش بوپ كىتىپ كۈرۈشۈشكە ۋاقىت چىقماي قالدى.لىكىن تۈنۈگۈن توردا سىزنىڭ كەشپىياتىڭىز توغۇرلۇق بەزى گەپلەرنى ئاڭلىدىم شۇلارنى  ئۆز قۇلۇغۇم بىلەن ئاڭلاي ھەق ناھەقنى تور ئەھلىگە چۈشەندۈرەي دىگەن  مەخسەتتە سىز گە تىلىفۇن قىلىشىم،
ئالىمجان ئاكاش راس مەنمۇ يىقىندا توردا غۇل-غۇلا بولغانلىقىنى ئاڭلىدىم ئاڭلىشىمچە مىسىرانىم تور بىتىدە شۇنداقلا  باشقا تور بەتلەردە شۇنداق بوپتۇ دەپ ئاڭلىدىم ،بەزى تورداشلار مىنىڭ كەشپىياتىمنى ماختاپ كۆككە كۈتۈرۈپ ئاشۇرىۋەتكەنلەرمۇ باركەن،ئوتتۇردا نۇرغۇنلىغان ئۇقۇشماسلىق يۇز بىرىپتۇ مەن بەك ئالدىراش بولغاچقا توردا بەزى پىتنە پاساتلارغا چۈشەنچە بىرەلمىدىم تور بىكەتلىرىگە تىلىفۇن قىلىپ بەزى ئەھۋاللارنىمۇ دىگنتىم،بولسا سىز يانا مىنىڭ نامىمدىن ئۇلارغا تىلىفۇن قىلىپ شۇنداق تىمىلارنى ۋاختىنچە توختاتسا مەن ۋاقىت چىقىرىپ تورداشلارنى ماتىرىيال بىلەن تەمىنلىسەم ئاندىن تورغا يوللاپ ئىشنىڭ ھەقىقىتىنى بىر چۈشەندۈرسەم دىدى. كەينىدىن مەن سىزنىڭ پانتنىت ئالغان كەشپىياتىڭىز قايسىلار دەپ سورىدىم ؟
ئالىمجان ئاكاش ھاۋادىن پايدىلىنىپ ھەركەتلىنىدىغان ماتور،ئۇندىن باشقا بىر نەچچە تۈرلۈك كەشپىياتلار بار دۆلەت ئىتىراپ قىلغان تۈر بۇلاردىن مەن پانتىت ھۇقۇقىنك قۇلۇمغا ئەپ بولدۇم.
مەن: سىزنىڭ بۇ كەشپىياتىڭىز بىرەر نەماددىنى يىقىلغۇ قىلامدۇ؟
توردا ھاۋانى يىقىلغۇ قىلىغىلى بولمايدۇ ھاۋا كۈيمەيدىغان تۇرسا،ناۋادە ھاۋا كۈيىدىغان بولسا يەر شارىغا ئوت كەتمەدۇ دەپ ۋەھىمىلىك سۆزلەرنى قىلىپ سىزنىڭ كەشپىياتىڭىزنىڭ ئاساسى يۇقلىقىنى بەزى بىر فىزىكلار ئاغدۇرىۋەتتى بۇنى قانداق چۈشەندۈرىسىز؟
ئەلۋەتتە ھاۋا ئوتتا كۆيمەيدۇ مەن ماتۇرغا ھىچقاندا يىقىلغۇ دىن پايدىلانمىدىم پەقەت ھاۋانىڭ بىسىمى بىلەن ھەركەتلىنىدىغان ماتۇر كەشىپ قىلدىم بۇنى دۆلەت پانتىت ھۇقۇقىنى ماڭا بىرىپ بولدى بۇنىڭدا مۇنازىرلەشكۈدەك ھىچقانداق ئىش يوق،
قايىل بولمىغانلار توردا ئۈسەك سۆزنى قىلىپ باشقىلارنىڭ دىلىغا ئازار بەرمەي، كۆڭلى ئارام تاپىدىغان ئىش بولسا  دۆلەت پانتىت ھۇقۇقىنى بىرىش ئورنى بىلەن ئالاقىلىشىپ شۇ دەلىل ئىسپاتلىرىنى ئوتتۇرغا قۇيۇپ ماڭا بەرگەن بۇ پانتىتى ھۇقۇقىنى ئەمەلدىن  قالدۇرىۋەتسۇن.
يەنە بىر بىر ئىش مەن نەچچە تۈرلۈك كەشپىياتنى بازارغا سىلىش ئۈچۈن تەييارلىق قىلىۋاتقان جەمىيەتتە بۇنداق گەپ سۆز بۇلۇپ كەتسە بىز گە تەسىر ياخشى بولمايدۇ ھەم روھى بىسىم بوپ قالىدىكەن بۇنىڭدىن كىيىن تورداشلار قالايمىقان گەپ قىلماي  بۇنداق ئۆسەك گەپلەرنى توختاتسا دەيمەن.
مەن يىقىندا ئۈرۈمچىگە بارىمەن شۇ چاغدا سىز بىلەن تەپسىلى پاراڭلىشاي.
بۇلىدۇ مەندە سىزنىڭ نۇمرىڭىز بار مەنمۇ تىلفۇن قىلىمەن سىزنىڭ ئىشىڭىز توغرىسدا توردا بىر نەچچە ئىغىز ھەق گەپ قىلاي سىزنىڭ سۆزىڭىزنى يەتكۈزەي
خوش ئاللاغا ئامانەت كەلگەندە كۈرۈشەيلى،
رەھمەت سىزگە.
مانا بىزنىڭ ئوتتۇرىمىزدا بولغان تىلىفۇن  سۆھبىتىمىز سۆھبىتىمىز قىسقا بولسىمۇ  شۇنداق ياخشى سەمىمى رستچىلىق بىلەن ئۇنىڭ ھەقىقەتەن ھاۋادىن ھەركەتلىنىدىغان ماتۇرنىڭ كەشىپ قىلىپ پانتىت ھۇقۇنى ئالغانلىقىنى ئۆ ز ئاغزى بىلەن ئىسپاتلىدى.
بىز كىمنى مات قىلماقچى ؟
كىمنىڭ ئۇلىنى كولاۋاتىمىز ؟
كىمنى يىقىتماقچى بۇلىۋاتىمىز؟
بۇ ئىشنىڭ ھەقىقىتنى بۈلۈش شۇنچىلىك تەسمىكەن،ئالىمجان ئاكاش،يەرشارىنىڭ ئۇ تەرىپىدە ياكى بۇ تەرىپىدە ئەمەسكەن،ياكى ئانتىراكتىدا ئەمەسكەن ،قەشقەردىكەن ئۈرۈمچىدىكەن،كۆز ئالدىمىزدىكەن مەن كەشپىيات ياراتتىم دەپ بىرسىنىڭ پۇلىنى ياكى مىلىنى ئالداپ ئىلىۋالماپتۇ توردا سازايى قىلغۇدەك ،بىز قاچانغىچە بىر بىرىمىزنى غاجايمىز ،كۆرەلمەسلىكىنىڭ دەردىنى ئاز تارتتۇقمۇ؟
يەنە قاچانغىچە تارتىمىز؟
ھەممە ئادەملەرنىڭ ئىڭى ئىچىلغان مۇشۇ زاماندىمۇ تاشلىمامدۇق بۇ رەزىللىكنى.
بىشىمىزدىن نەچچە قىتىم تەجىربە ساۋاق ئۆتكەن تۇرۇپ يەنە ئۆز ئادىمىمىزنىڭ كەشپىيات قىلغان ھەقىقىتىغىمۇ ،گىزىت، ژورنال، تىلىۋزىيە زىيارىتىگىمۇ،قاتناشقان يەرمەنكىسىگىمۇ ئىشەنمىسەك بىز زادى نىمە قىلماقچى،
شۇ كەشپىياتچىمىزنى ئۆرىۋەتمىگىچە، ئۇنى ئەل جامائەتنىڭ ئالدىدا يۈزىنى تۆكىمىگىچە بولدى قىلمامدۇق. ئۆزىڭىزنىڭ قىرىندىشىكەنغۇ ئۆزىڭىزنىڭ دىندىشىكەنغۇ،ئۇنىڭدىمۇ  ئىقىۋاتقىنى ئۇيغۇرنىڭ قىنىكەنغۇ،توردا شۇنچىلىك تالاش تارتىش قىلىش كىتەمدۇ،شۇنچىۋالا فىزىكا ئىلىمدە يىتىشكەن تۇرۇقلۇق ،نەتىجىمۇ ياراتماڭ بىر نەرسىنىمۇ تەتقىق قىلماڭ تورغا چىقىپ ئالىمجاننىڭ كەشپىياتى توغۇرلۇق پۇتاق تىپىپ پىتنە تارقاتقۇچە، ئادىرىسىنى تىپىپ ئۆزى بىلەن مۇنازىرلىشىپ ئىشنىڭ ھەقىقىتىنى چۈشەنسىڭىز بولمامدۇ،شۇنىڭغىمۇ ئىشلىمىگەن كاللا قانداقمۇ ئۇنىڭ كەشپىياتىنى ئاغدرۇشقا ئىشلەيدۇ،قانداقلا بولمىسۇن ئالىمجان ئاكاش نەچچە تۈرلۈك كەشپىياتنىڭ پانتىت ھۇقۇقىنى ئاپتۇ سىز فىزىك شۇنچىلىك بىلىملىك تۇرۇپ نىمە قىلالىدىڭىز مىللىتىڭىزگە قانچىلىك ياخشى ئىش قىلىپ بەردىڭىز،سىلەرنى نىمە ئاشقۇن دەيمىز؟
ئىنكاسچى قىرىنداشلار سىلەرمۇ ئىنكاس يازغاندا ئۇياققا-بۇياققا ئۆتىۋالماي ھەقتە چىڭ تۇرۇڭلا،
بەزى ئىنكاسلارنى كۆرۈپ ھەيران قالدىم،ئاق قارىنى تۇلۇق ئايرىيلى

تۆۋەندە ئۈزەم ياغان بىر شىئر بارئىدى يىزىپ قۇياي بۇنىڭدىن 15 يىل بۇرۇن يازغانتىم،ئىسىمدە قالغىنى بىر كوبلىتكەن،ئۆزەمنىڭ ئھتىياتسىزلىغىمدىن ئۆزەم يازغان 10 نەچچە شىرىمنى ۋە قول يازمامنى يۈتتۈرۈپ قۇيۇپ قايتا توپلام قىلماي ئىسىمدىن كۆتۈرلۈپ كىتىپتۇ،ئىسىمدە ھەر بىر يازغان شئرىمنىڭ بىردىن كوبلىتى ئىسىمدە قاپتۇ. بۇتىمىغا مۇناسىۋەتلىك شىئىرىمدىن،كۆرەلمەس دىگەن شىئردىن ئاخىرقى كوبلىتىنى يىزىپ قۇياي دەپ قالدىم.
            
                         كۆرەلمەسلىكنىڭ دەردىنى ئاز تارتتۇقمۇ بىز،
                         كۆرەلمەسلىكنىڭ تامغىسى قالدۇرغانتۇغۇ ئىز.
                        چاشقان كەبىي ئۇر-ئۇر قىلايلى يۇقىتايلى تىز،
                         قاچانغىچە يۇقاتمايمىز بۇ ئىللەتنى بىز.  

بولسا باشقا تور بەتلەرگە يوللاپ قويساڭلار باشقىلارمۇ خەۋەردار بولپ قالسۇن يىقىندا ئۆزى ماتىرىيال تەييارلاپ تور
