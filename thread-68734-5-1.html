ئىشلىتىش قىممىتى بارمۇ يوخ بۇنى بىز بىلمەيمىز ، لكىن شۇنى بىلىمىز كى بۇ ماتۇر چىققىلى خېلە

ئاددىسىنى دىسەم مەنمۇ كىچىك بولسىمۇ بىر كەشپىياتچى ، لىكىن مەن پاتنىت ئېلىپ بولغىلى
خېلە ئۇزۇن بولغان بولسىمۇ سېتىۋالدىغان بېرەر شىركەت چىخماي ئويدە توپا بېسىپ يېتىدۇ ، مەن
دىمدكچى بېرەر پاتنىت يارىتىش ئۇچۇن ئۇنىڭ ئىختىسادى ئۈنۈمىنى ئالدىن ئويلىشىش كېرەك

ئۈستىدىكى ئىنكاسنى يازغان تورداشقا:
ئەگەر ئۇيغۇرچىنى  ...

جىنىم قىرىندىشىم مەن ئەسلى مۇشۇ ئىش ئىنقلانغۇچە تورغا چىقماي دىگەن ھىلىمۇ تۆت كۈن بولدى تورغا چىقمىغىلى بۈگۈن ئىچىپ باقاي بسىقىپ قالغاندۇ تالاش-تارتىشلار دەپ كىرگەن قارسام يانا باركەن ئاخىرى،مىنىڭ تىمام ۋە ئىنكاسىمدا ئۈچلا مەزمۇن بار
1-تىمىدا ۋە ئىنكاستا ھىچقانداق ماختاش مەۋجۇت ئەمەس
2-ئەمىلىيەتنى چىقىش قىلماي تۇرۇپ بىرەر نەرسىگە باھا بەرمەسلىك
3-شۇنداق تىما ۋە ئىنكاسلارنى يىزىپ ھىچنىمىدن خەۋىرى يوق بىر ئادەمگە ئازار بەرمەيلى
شۇ مەزمۇن بارغۇ كاللىسى ئۇچۇق ئادەم بىر ئۇقۇپ چۈشىنىدۇ ھەر قانچە گۇمان بولسىمۇ قارشى تەرەپنىڭ كۆڭلىگە ئازار كەلتۈرۈشكە سەۋەپ بۇلىدىغان تىمىنى ۋە ئىنكاسنى يازمىغان بۇلاتتى،ھەر قانچە ئەقلىگە سىغمايدىغان ئىش بولسىمۇ قاۋاپ بىرەي دىگەندىكىن بولدىغۇ قاراپ بقىڭ ئىنكاسقا نىمىلەرنى يىزىۋىتىپتۇ،بىزغۇ يىزىپ بۇلاپ ىاتىرجەم ئۇخلايمىز ھىچنىمىە بىلەن كارىمىز يوق،تىمىزمىزدىكى پىرشۇناژچۇ؟ توردىكى ئۆزى توغۇرلۇق كەمسىتىش يۇز ئابرۇيىنى چۈرۈش ،زىتىغا تىگىش،شۇنچە قىلغان ئەمگىكىنى بىر تىيىنغا ئەرزىمەس قىلىۋىتىش،مۇشۇ ئىنكاسنى ئۇقۇسا كۆڭلى يىرىم بولمامدۇ،خۇرلۇق ھىس قىلمامدۇ ،مەن چىغىمدا بۇ تىمىغا يىزىلغان ئىنكاسلارنى كۆرۈپ خىجىل بۇلىۋاتىمەن ،ئەمىلىيەتتە مەن شىمالدا ئۇ ئاكاش بولسا جەنۇپتا ھەم مەن كۆۈرۈپمۇ باقمىغان،نىمە دىگەن رەھىمسىزلىك بۇ كىشلەرنىڭ ئىززەت ئابروينى قۇغداش دىگەننى چۈشەنمەمسىلەر ،مەن باشتىن ئاياغ دەپ كەلگىنىم قالايمىقان گەپ قىلمايلى كىشىنىڭ ئىززەت ھكرمىتنى چۈشۈرىدىغان گەپ سۆزنى قىلمايلى،تالاش تارتىشنى توختىتايلى دىسەممۇ يەنىلا شۇ گەپ ،يوغان نەرسىنى بايقاپتۇ بۇ تىمىنى ئۇقۇپ شۇنداق بىلىملىك بوپ كەتتۇق قايتا ئۇقۇڭ دەپ ماڭا چۈشەنمىگەندەك بۇ تىمىنى تازا ئوبدان چۈشۈنۈشلۈك ،مىنىڭكىنى چۈشۈنۈڭ دىگۈچە مىنىڭكىنىمۇ چۈشۈنۈپ ئۇقۇسىڭىز بولمامدۇ؟ مەن نىمە مەنادا يىزىپتىمەن ،يا ئۇنى ماختىمىدىم ياكى چۈشۈرىۋەتمىدىم تالاش تارتىش قىلمايلى كىشلەرنى ھۆرمىتىنى ساقلايلى دىدىمغۇ.بۇ ئىنكاسىڭىزنى ماڭا دىمىگەن بولسىڭىز مۇ سىزگە يانا بىر دەپ قويدۇم

يوللىغان ۋاقتى 2012-3-25 03:50:23 PM |ھەممە قەۋەتنى كۆرۈش |
بۇ ماقالىنىڭ ئەمەلىي قايىل قىلىش كۈچى بار ئىكەن، كەتكەن ئېنېرگىيە بىلەن ئېرىشكەن ئېنېرگىيەنىڭ ئوتتۇرىسىدا چوڭ پەرق بولسا بۇ ئەلۋەتتە ئۈنۈمسىز ئىش بولماي نېمە ؟؟؟

يوق نەرسىلەرنى ئىنسانلار ھېس قىلغاندىن كېيىن قول سېلىپ ئىجاد قىلىدۇ، ئالىمجان ئاكىمىزمۇ دەل مۇشۇنداق قىلالىدى، ئەمدىكى ئەڭ مۇھىم مەسىلە شۇكى بۇنى ئەمەلىي ئىشلىتىلىش ئۈنۈمى بار قىلىپ بازارغا يۈزلەندۈرۈش مەسىلىسى، بۇ ئۇنداق ئاسان ئىش ئەمەس، توختىماي ئىزدىنىشكە، سىناق قىلىشقا ھەم بىر دۆۋە ئىلمىي تەتقىقات نەتىجىلىرىنىڭ سان - سىفىرلىق دوكلات نەتىجىسىگە ئىگە بولۇپ، باشقىلارنى تولۇق قايىل قىلالىغاندىلا ئاندىن بازار توغرىسىدا ئېغىز ئېچىشقا بولىدۇ ..... بۇنىڭغا مۇئەييەن جەريان شۇنداقلا ۋاقىت كېرەك .....

ئالىمجان ئاكىمىز كەلگۈسىدە باش قاتۇرۇپ، ئېرىشكەن ئېنېرگىيەنى سەرپ بولغان ئېنېرگىيە زور دەرىجىدە چوڭ قىلىپ ماتور ياسىيالىسا بۇ ھەممىمىزگە تولىمۇ خوش بولارلىق سوۋغات بولغان بولىدۇ ھەم ئالىمجان ئاكىمىزنىڭ شۇنداق قىلالايدىغانلىقىغا ئىشىنىمەن .... ئۈمىد غەلبىنىڭ ئاساسى ....

ئەنە شۇنداق كۈنلەر كېلىپ قالسا مەن بىرىنچى بولۇپ، قىسىلغان ھاۋا قاچىلايدىغان ھاۋا ئىسكىلاتىدىن بىرنى ئاچىمەن .... خۇددى گاز قاچىلاش پونكىتلىرىغا ئوخشاش ...
سىىز ماشىنىڭىزنى ھەيدەپ ھاۋا ئىسكىلاتىمغا كەلسىڭىز مەن خۇشخۇيلۇق بىلەن ئالدىڭىزغا چىقىپ ماشىنىڭىزنىڭ كەينىدىكى ھاۋا قاچىلايدىغان باكقا يۇقىرى بېسىم بىلەن سىقىلغان ھاۋادىن 10 يۈەنلىك قاچىلاپ بەرسەم سىز چاتاق يوق 100 كىلومېتىر يول يۈرەلەيسىز..
ئەل–يۇرت ئۈچۈن تارتقان ئازابنى، شاھلىق تەختىگە قىياس ئەتكۈلۈك. ئەل ئۈچۈن چەككەن بىر مىنۇت قايغۇ، مىڭ يىللىق راھەت ئۈچۈن يەتكۈلۈك.
