رىۋايەتچىلەردىن قۇرۇق گەپ سىتىپ شۇنداق رىۋايەت قىلدۇلەركى زامان زامان ئىلگىرى ،كەلمىگەن كۈننىڭ ئۆتكەن بىر يىلى جىجياڭ ئۆلكىسىدە لى فامىلىك بىر كومشا بايۋەتچە ئۆتكەنىكەن ، مىليونىرلارغا ،ئەمەلدارلارغا سەدىقىنى كۆپ بەرگەچكە ئەمىلى تىزلا ئۆسۈپ بازار باشلىقى، گىنرال لىسموس، سىتارشى لىيتنات قاتارلىق ئۇنۋانلارغىمۇ ئىرىشىپتۇ(ئارقا ئىشىككە راكىتا قاتنايدۇ،ئالدى ئىشىككە  باكتىرىيە پاتمايدۇ ئەمەسمۇ) قەلەندەر ئەمەلدارلارغا جىق سەدىقە بەرگەنلىكتىن خاتۇنى تۇغماي تۇرۋاپتۇ ،شۇنىڭ بىلەن لى فامىلىك بايۋەتچە راھىپ كالاننى چاقىرىپ ئۆيدە باققان ئەتىۋارلىق شىرنى بۇغۇزلاپ نەزىر قىپتۇ، راھىپ تازلاردىن شاراپتا تاھارەت ئىلىپ ئىشرەتخانىدا بۇدقا باش ئۇرۇپ بالا تىلەپ بىرىشنى ئۆتۈنۈپتۇ. راھىپ كالان بۇدقا باش ئۇرۇپ شىر گۆشى شورپىسىنى چايناپ تۇرۇپ مۇنداق دەپتۇ :
كىنەز جاناپلىرىنىڭ تىلىكى ئىجاۋەت بۇلۇش ئالدىدا تۇرۇپتۇ، بايا مەن بۇدخانىدىكى جىمى راھىپلارنى سەپەرۋەر قىلىپ بۇددىمىزدىن سىلىگە بالا تىلىدىم ، بۇددىمىزنىڭ ھاجىتى قىستاپ كىتىپ كىتىپ قالدى كىتىش ئالدىدا مەندىن تازلىق قەغىزى سورىدى تىخى !
لى بايۋەتچە: نەق گەپكە كىلەيلى ، بۇ ددىمىز بالا توغرىسىدا نىمە دىدى
راھىپ : بۇددىمز  <<بالا تاپماسلىق سىلەرنىڭ شەخسىي ئىشىڭلار ،مەندەك لايدىن ياسىغان بۇد بىلەن مۇناسىۋىتى يوق ، قانۇنىي جاۋابكارلىقنى ئۈستۈمگە ئالمايمەن،ئىككى تىزىڭ يېغىر بولغىچە تىرىشقىن >>  دىدى
لى بايۋەتچە:چۈشەندىم ، ماڭا يۇشۇرۇن بىشارەت بىرىپتۇ، بۇ تىرىشىپ ئىككى تىزىڭنى يېغىر قىل دىگىنىكەن .
كۈنلەرنىڭ بىركۈنى لى بايۋەتچىنىڭ خاتۇنى گاڭپەنگە خىمىر يۇغۇرۇپ ئولتۇرۇپ بىكارچىلىقتا جى گۇڭ نى تۇغۇپ سەپتۇ لى بايۋەتچە خوشاللىقىدا بۇدخانىدىكى پىركالانلارنى چاقىرىپ تۇرخۇنغا ياققان نان بىلەن مىھمان قىپتۇ،بۇدداماشايىخلىرىدىن ئانچە مۇنچە نوم ئوقۇپ بىرىشنى تەلەپ قىپتۇ، لاي بۇدقا قىلىنغان تاۋاپ تۇگىگەندىن كىيىن ماشايىخ بالىغا جىگۇڭ دەپ ئىسىم قويدى، بالىنىڭ قۇلىغا قەلەمنى  تۇتقۇزۇپ دىدى:  بالا چوڭ بولغاندا قۇلىغا قەلەم ئىلىپ بىكار تەلەتلەرنىڭ كۆزىگە تىقۋەتكەي
بالا قەلەمنى ئالغىلى ئۇنىمىدى ، دەرۋىش يەنە شىشىنى ئىلىپ : بالا چوڭ بولغاندا ئىچەرمەنلەر ساھەسىدە جىنىس دۇنيا رىكورتى ياراتقاي
بالا شىشىنىمۇ ئالماي ئاسمانغا تۈكۈرۈپ ئەتراپتىكىلەرگە بەش مىتىر ئالىيىپ قويۇپتۇ
ماشايىخ: ئۆزەڭ ئالە ئەمسە دەلتى شۇم
بالا قولىغا تەسۋىينى ئالدى
لى بايۋەتچە: بالام چوڭ بولسا سىنىڭ يولۇڭنى تۇتىدىغان ئوخشايدۇ جۇمۇ ،بالامغا ئەپسۇن ئوقۇپ جادۇ قىپسەن سىنى پايەنگە ئىيتىمەن ،
لى بايۋەتچە دەرغەزەپ بۇلۇپ<<پىلانىمىز پاش بوپ قاپتۇ>> دەپ كالىچىنى بىر ئىتىۋىدى چۈپرەندە تازلار تەسۋىي  تاك-تەكلىرىنى يىغىشتۇرۇپ تىكىۋىتىشتى
بالا ئاستا ئاستا چوڭ بولماقتا ئىدى.....
بىر كۈنى جىگۇڭ تاۋۇت ساتىدىغان  خوشنىسىنىڭ  تۇرخۇنىنى ماراپ ئۇنىڭ زىمىن مالائىكىسىگە باش قويۇپ تىلەك تىلەۋاتقانلىقىنى كۆردى
تاۋۇتچى : ئەي زىمىن مالائىكىسى ،جان دىگەن باكالىق نەرسە ، خالىغانچە ئال ،  بىرنىڭ ئورنىغا ئوننىڭ ،ئوننىڭ ئورنىغا ئوننىڭ كۋادراتىنىڭ جىنىنى ئال، ماتىماتىكا بىلسەڭ ئۆلۈكلەرنىڭ سانىنى  ھەرگىز يىلتىزدىن چىقىرىپ يۇرمە ،مەنلا ياشاي خەق ئۆلسۇن
جىگۇڭ خەپ دىگىنىچە ئۆگزىدىن چۈشۈپ توپاققا مىنىپ ماڭدى ، يىقىنلا يەردە سۈپكۆچ قىلىدىغان بۈۋۈم ئاسمان مالائىكىسىگە تاۋاپ قىلۋاتاتتى
بۈۋۈم : ئەي مالائىكە ، سەندىن ئۆتۈنەي، مەندىن باشقا مەھەللىدىكى ھەممە كىشىنى كىسەل قىلۋەتكىن ،ئۇلار راك ،ئەيدىز،  رىماتىزىم، رىماقۇلاق بوپ ئالدىمغا كەلسۇن،لىكىن ئىسىڭدە بولسۇن ، مەھەللىگە دوختۇر سىستىرادىگەندەك يا سۈپكۆچ قىلىشنى يا ئاشلاشنى بىلمەيدىغان   چالا ساۋاتلار  ھەرگىز يىقىن يوللىمىسۇن  جىگۇڭ دەرھال بۈۋۈمنىڭ يىنىغا كىلىپ ئەھۋال مەلۇم قىلدى
جىگۇڭ : تىز بولسىلا ،تاۋۇتچى ئاغرىپ قاپتۇ، چىشىنىڭ پىيى تارتىشىپ قولىقى ساراڭ بوپ قاپتۇدەك سىلىنىڭ مۇبارەك تۈكرۈكلىرى بىلەن ساقايغۇدەك
بۇنى ئاڭلاپ بۈۋۈم ئۇچقاندەك يۈگۈردى ... ئۇنىڭغىچە شۈپتۈر كىيىنگەن جىگۇڭ توپاققا تەتۇر مىنىپ يۇقىرى سۈرەتلىك توپىلاڭ يول بىلەن بىلەن تاۋۇتچىنىڭ  ئۆيىگە بىرىپ ئەھۋال مەلۇم قىلدى
جىگۇڭ: بۈۋۈمگە  قىممەت پۇللۇق تاۋۇتتىن بىرى لازىمكەن
تاۋۇتچى :كىم ئۆلۈپتۇ؟ ھەجەپ ياخشى بوپتۇ جۇمۇ
جىگۇڭ : بۈۋۈمنىڭ ئۆيىدىكى قىرى دىدەكنىڭ ئانىسىنىڭ بىر كىچىك سىڭلىسى بار ئىدى شۇنىڭ ئىرى ئۆزىنىڭ ئىنىسىنى تىللاپ ئۆلتۈرۈپ قويۇپتۇ
تاۋۇتچى : جەمئى ئالتە ئادەمنىڭ ئىسمىنى چىقاردىڭ ھەممىسىگە   تاۋۇت لازىممىكەن ؟
جىگۇڭ : ئىن
تاۋۇتچى ئۇياققا يۇگۈردى ، بۈۋۈم بۇياققا يۈگۈردى ، شۇنداق قىلىپ بۇلار ئىككىسى رەسۋا بۇلۇشتى .
جىگۇڭ بۇ رەسۋاچىلىقتا لىياڭشەنگە قىچىپ بىرىپ لۇجىشىن غا شاگىرتقا كىرىپ راھىپ بۇلىۋالدى ، نەچچەيىل تەربىيىلىنىپ باكلاۋۇرلۇق ئۇنۋانى ئىلىپ يۇرتىغا پىراكتىكا قىلىشقا كەلدى ، ئۆيگە كەلسە ئىش باشقىچە!!!!
ئۆينى خەق ئۆزىنىڭ قىلۋالغان ئىدى ئاچچىقى كەلگەن جى گۇڭ ئۆيگە بىنزىن چىچىپ ئوت قويۋەتتى ، ئاندىن ئوت قويۇش جىنايىتى بىلەن 5-باشقارما تەرىپىتىن قولغا ئىلىنىپ چىكاگوغا سۈرگۈن قىلىنىپ بەش يىلدىن كىيىن چىقتى ،تۈرمىدىن چىققاندىن كىيىنكى ئىشلارنى كىيىنكى قىسىمدا ئايرىم توختىلىمىز
