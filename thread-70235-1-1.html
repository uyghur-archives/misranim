               بالىلارنى باغچىلارغا باشلايلى!

                                          سۆز ئۇينى

                                                  (ب بىلەن باشلانغان )


بايرىمىنى بالىلار،

باغچىلاردا باشلىسۇن.

باغچىلارغا بېرىشقا،

بەلنى بەلەن باغلىسۇن.

  بالىلار بىزنىڭ باغرىمىز، بالىلار بىزنىڭ بەختىمىز، بالىلار بىزنىڭ بايلىقىمىز. بالىلار بولغاچقا بىز بەك بەختىيار، باياشات، بەختلىك. بىزنىڭ بارلىقىمىز بالىلارنىڭ بەختى بىلەن باغلانسۇن. بىز باغۋەنلەردە بالىلارنى باقىدىغان بەلگە بولسۇن. بالىلاردىمۇ بالىلىق بەلگە بار؛ بالىلار بەزىدە بەڭۋاش، بەزىدە بىپەرۋا، بەزىدە بىخىل، بەزىدە بەتخەج، بەرىبىر بالىدۇر- بالا.

  بىز بالىلارنى بۆشۈككە بۆلىۋالماي، بىلىدىن باغلىۋالماي، بۆشۈك بەلۋېغىدىن بۇشتۇپ،  باكار بولساقلا بۈگۈنكىدەك بالىلار بايراملىرىدا بالىلا بىلەن بىللە بۇلۇپ، باغچىلارغا باشلاپ بېرىپ باغچىلاردىكى بىلىم بايلىىقلىردىن بەھۇزۇر بەھىرلەندۈرەيلى!

  بالىلارنى بىزدە بار بولغان بالىلار باغچىلىرىغا، بوشلۇقتىكى بوستانلىققا، باغلارغا بۇلۇپمۇ بىزنىڭ بەشكىرەمگە باشلاپ بارايلى! بېلىڭلاركى بىز بالىلارنى باشلاپ بارماقچى بولغان باغچىلاردا، باغۇ-بوستانلىق بوشلۇقلاردا، بەشكىرەمنىڭ بوستانلىق باغلىرىدا بالىلار بەھىرلىندىغان بىلىم بايلىقلىرى بار. بۇ بىلىملەرنى بالىلار بەك بىلەلمەيدۇ، بىز بىلەلەيمىز. بىز بىلدىغان بىلىملەرنى بەركەتلىك بارمىقىمىزدا بالىلارغا بىۋاستە بىرەيلى.

  بالىلار بىلىم بەيگىسىدە بىلىمنى بالدۇرراق بىلىشكە باشلىسا، بىزنىڭ بالىلارنى بىقىشتىكى بېسىمىمىز،  بالىلارغا بىرىدىغان بەدېلىمىز بەكرەك بۇشايدۇ. بۇلۇپمۇ بالىلار بەشكىرەمگە بارسا بەشكىرەمدىكى بىھىشنىڭ بارلىق بەلگىلىرنى بىلۋالىدۇ.

  بۇرۇننىڭ بۇرنىسىدا بىھىشنى باشقۇردىغان باھادىرخان بىرىنچى بۇلۇپ بىھىشنىڭ باغۋەنچىلىك بۆلمىدىكى بوستان بۇۋاينى باشلاپ بەشكىرەمگە بېرىپتۇ. بوستان بۇۋاي بەشكىرەمدە بىھىشنىڭ بەركەتلىرنى بېخلاندۇرۇپ بارا-بارا باغ-باراڭلارنى بېيتىپتۇ. بۇ باغلارغا باھا بىرىپ بىرسىنى بايلا، بىرسىنى باغ ئېرىق، بىرسىنى بويلا، بىرسىنى بەش توغراق ... بولسا بۇلىدىغانلىقىنى بەلگىلەپتۇ.

  بەشكىرەمدىكى باغلارنىڭ بۇ بەلگىلىرنى بالىلار بايقاپ بىلەلىسە: بېيجىڭنىڭ باغچىلىرنى، بىرازىليەنىڭ بېلىقلىرنى،بومباينىڭ بايلىرنى، باغداتنىڭ بانكىسىنى بىلەلەيدىغان بۇلىدۇ.

  بۇقېتىمقى باھار بايرىمىدا بارلىق بالىلارغا بىزنىڭ بەشكىرەمگە بىرىشقا << بەخىت >> باغىقىنى بىرىمەن. بالىلار بالدۇرراق بەلنى باغلاپ بىر-بىرىڭلارنى باشلاپ بەشكىرەمگە بېرىپ باھار بايرىمىنى بىللە بەس-بەستە باشلايلى!

                               بۇرادەرلەرگە بېغىشلانغان بايانات
بىلىمسىزلىك ، بىكارچىلىق ،بىخەستلىك ، بۇزۇقچىلىق ، بىپەرۋالىقلار بىزنى بوزەك بولۇشقا باشلايدۇ .بىلىم ،بىرلىك ، باتۇرلۇق بولسا بىزنى بۈيۈكلۇككە ، باھارغا باشلايدۇ . بىز  بىلىمسىز ، بىكارچى ، بىۋاپا بولساق بالىلىرىمىزمۇ ، باشقىلارمۇ بىزدىن بىزار بولىدۇ .                  

     بالا بېقىشنى بۆرىلەرمۇ ، بۇغىلارمۇ بىلىدۇ .بىزمۇ  بالىلىرىمىزنى بۆرىلەردەك ، بۇغىلاردەك بېقىشنىلا بىلىپ، بىلىم بىلەن بەختكە ، بۈيۈكلۇككە باشلاشنى بىلمىسەك بولامدۇ ؟ بەك بەلەن باي  بولساڭمۇ  بىلىمىڭ  بولمىسا بىكارغۇ. بىلىمىڭ بولمىسا بايلىقىڭنى باشقۇرۇشنىمۇ بايلىقىڭدىن بەھوزۇر بەھرىمان بولۇشنىمۇ بىلمەيسەن . بىلىمدەك بازىسى بولمىغان بايلىقنىڭ بەركىتىمۇ بولمايدۇ . بايلىقىڭنى بىلىم بىلىشكە باشقىلارغىمۇ بىلىم بىلدۈرۈشكە بېغىشلىساڭ بايلىقىڭنىڭمۇ بەرىكىتى بولىدۇ. بىلىشىمىز بەرھەقكى ، بىلىمگە بېغىشلانغان بايلىق بىباھادۇر .بۈگۈنكى بۇ بازاردا بىلىم بەختنىڭ ، باياشاتلىقنىڭ بەلگىسىدۇر . بىلىمىمىز بولمىسا باققالچىلىق ،باقمىچىلىق بىلەنمۇ بالىلىرىمىزنى باقالمايمىز ، بىلىمىمىز بولمىسا باشقىلارغا بىكارلا بوزەك بولىمىز .بىزنىڭ بەدىنىمىزدە  بۆرىكىمىز  بولمىسا بولمىغاندەك بىلىمىمىز بولمىسىمۇ بولمايدۇ ! بۇرۇندىنلا بىلىمگە بېرىلگەنلەر بەختنى باغاشلىغان ، بىلىمسىزلەر بولسا بەئەينى بىھۇشلانغان بىماردەك باشقىلارغا بىچارىلارچە بوزەك بولغان .                                       

بۈگۈن بىزمۇ بىپەرۋالىق بىلەن بېزىرىپ بىلىمگە بېرىلمىسەك بولارمۇ ؟  بۇنداق بولغاندىكىن بىزمۇ بۈگۈندىن باشلاپ بىھۈدە بەزمىلەرگە بېرىلمەي ، بىر – بىرىمىز بىلەن بىرلىشىپ ، بەلنى باغلاپ بىلىمگە بېرىلىپ،  بارلىغىمىزنى باشقىدىن باشلىيالىساق ، بىزدىمۇ باياشاتلىق ،     بەختىيارلىق بەرپا بولماسمۇ ؟                                   

بىلەمسىز، بىلىم  بولمىسا بىناكارلىق ، باغۋەنچىلىك ، باقمىچىلىق ،  بانكىچىلىق ، بېزەكچىلىكنىمۇ بىمالال باشلىغىلى بولمايدۇ ! بىلىم بولمىسا بىخەتەرلىكىمىزمۇ بولمايدۇ !

بۈگۈن بىزگە باھانىنىڭ بولىشى بىھاجەت ! بۈگۈن بىزگە بەزمە بولمىسىمۇ بولىۋېرىدۇ ، بىراق بىلىم بولمىسا بولمايدۇ ! بۈگۈنكى بۇ بازاردا بىلىملىكنىڭ باھاسى بىباھا، بىلىمسىزنىڭ باھاسى بازاردىكى بورىچىلىكلا.    بىزمۇ بالىلىرىمىزغا بىلىم بېرىشنى بۆشىكىدىنلا باشلىيالىغان بولساق بەلەن بولاتتى . بىزنىڭ بالىلىرىمىزمۇ بېيجىڭ ، بېلگىيە،بىرازىلىيەلەرگە بىلىم بىلىشكە بارسا بولاتتى . باققاللاردىن باشلىقلارغىچە ، بوۋاقلاردىن  بوۋايلارغىچە بىلىمنى بىباھا بىلسەك ، بىلىمگە بېرىلسەك بىزدىمۇ بۆسۈش بولاتتى. بىراق ، بىزدە  بىلىملا بولۇپ ، بىرلىك ، باتۇرلۇق بولمىسىمۇ بولمايدۇ ، بىزدىكى بىلىم ، بايلىق ، بىرلىك ، باتۇرلۇق بىرلەشسە بۈيۈكلۈك بىزنىڭ  بولاتتى !!!     بىلەمسىزكىن ، بەزى بايلىرىمىز بايلىقىنى بىلىمگە بۆلۇشكە بېخىل بولۇپ ، بالىلىرىنىمۇ بىلىم بىلىشكە بەرمەيۋاتىدۇ . بەزى باشلىقلار بىلىمسىزلىكىدىن باشتىكىلەر بۆككە بۇيرىسا باشنى بېرىۋاتىدۇ ،  بۇلار بەكلا بىمەنىلىكقۇ .بەرھەقكى بۇلارنىڭ بارلىغى بىلىمسىزلىكنىڭ                       *                                 

بەلگىسىدۇر !                           

بىزدە بەزمىگە بېرىلىش ، بۆلىنىش ، بىكارچىلىق ، بىپەرۋالىق ، بۇزۇقچىلىقلارمۇ بولىۋاتىدۇ ، بىلىمدە بۆسۈشلەرمۇ بولىۋاتىدۇ ، بىزمۇ بارلىقىمىزنى بىلىم بىلەن بېزەشكە بېرىلىپ ، بىلىم بىلىشكە  بەل باغلىساق ، بوشلۇقلارغىمۇ بىمالال بارالىشىمىز بەرھەقتۇر !!            بۇرۇنقىلار بىزگە بىلدۇرگەن بىلدۈرگۈ : « بۆز بىلەن بېزىگىچە بەدىنىڭنى ، بىلىم بىلەن بېزىگىن بارلىقىڭنى ! »        



                      بىلىمگە بېرىلەيلى ، بەزمىگە بېرىلمەيلى !
