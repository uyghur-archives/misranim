پىروتوننىڭ تېزلىگى نۇر تېزلىگىگە يېقىنلاشقاندا  ،ئۇنىڭ ماسسىسى ئەسلىدىكىدىن 100ھەسسىگە يېقىن ئاشقان.  بۇ دىمەكلىك  تەجىربە نەتىجىسىدە پروتوننىڭ ھەجىمى بىلەن زىچلىغى ئەسلىدىكىدىن 100ھەسسىگە يېقىن ئاشقان بولىدۇ، لېكىن ماسسىغۇ ئاشقان،ئۇنىڭ زىچلىغىنىڭ ئاشقانلىغىنى قانداق دەلىللەپ چىققىلى بولىدۇ. زىچلىقنىڭ ئېشىشىنى مۇنداق چۇشەندۇرۇمىز بىر ئىستاكان قەھۋەگە قەھۋە بىلان شىكەرنى مۇۋاپىق سالىمىز،بۇنىڭدا بۇ مۇۋاپىق زىچلىققا ئىگە بولغان بولىدۇ،بۇ زىچلىقنى ئاشۇرۇش ئۇچۇن داۋاملىق قەھۋە بىلەن شىكەرنى يەنە قۇيىمىز،بۇنىڭ بىلەن سۇيۇقلۇقنىڭ زىچلىغى داۋاملىق ئاشىدۇ،لېكىن نۇر تېزلىگىگە يېقىن تېزلىكتىكى بۇ پروتونغا كىم يەنە پروتوننى قۇشۇپ،ئۇنىڭ زىچلىغىنى ئاشۇرىدۇ،بۇنى قايسى فىزىكا ئالىمى بىزگە چۇشەبدۇرۇپ بىرەلەيدىكىن؟
..................

فىزىكلار ئىجتىمايى ھەركەتنى فىزىكىلىق ھەركەتكە يىغىنچاقلاپ چۇشەندۇرۇدۇ،بۇ  ماركىسىزىملىق  لوگىكىغا خىلاپ  .   
ـــــــ
   «ماركىسىزىملىق  لوگىكـــا»دىگەن قاچان،قايسى نەشىرىيات نەشىر قىلغان كىتاپ؟،قايسى تىللاردا نەشىر قىلىنغان؟قەيەردە سىتىلىدۇ؟،قايسى يىللىقلارنىڭ دەرىسلىگى؟.بۇكىتاپتا ياكى دەرىسلىكتە ۋەياكى قورال كىتاپتا«فىزىكا»ھەققىدە قانچىلىك توختالغان؟.يەنە قايسى فىزىكا ئالىملىرىنىڭ تەلىماتلىرىنىڭ خاتالىغىنى ئۇچۇق كۇرسەتكەن؟ئۇنى كىملەر تۇزگەن؟.نىمە ئۇچۇن مائارىپ مىنىستىرلىگى شۇ بويىچە ئىجرا قىلمايدۇ ۋە خاتانى تۇزەتمەيدۇ؟ـــــــــــــــ جاۋاپ بىرىڭلار ئەپەندىلەر.
«لوگىكا»ـ ئىنسانىيەتكە ئورتاق تەئەللۇق ئىلىممۇ ياكى ھەر بىر پارتىيە -گۇرۇھ،مىللەت،شەخىسلەرنىڭ ئايرىم-ئايرىم لوگىكىسى بولامدۇ؟.«لوگىكا»دىگەن ئەسلى نىمە ئىدى، سىلەر ئەمدى ئۇنى نىمىگە ئايلاندۇرماقچى؟.ھازىر دۇنيادا «لوگىكا»نىڭ قانچە تۇرى بار؟.«ماركىسىزىملىق لوگىكا»ئۇلارنىڭ ئىچىدىكى قانچىنجىسى؟ ـــــ جاۋاپ بىرىڭلار.بىلىۋالايلى؟!.

سىز يوللىغان ئادرىستىكى ماقالە ئېچىلمايدىكەن ، يەنە  ...

سىز يوللىغان ئادرىستىكى ماقالە ئېچىلمايدىكەن ، يەنە كېلىپ توردا ئۇ ئىككى قانۇن بىر بىرىگە زىت ياكى زىت ئەمەس دەيدىغان مەزمۇندىكى يوللانمىلار ساماندەك تۇرسا ئۇنى ماڭا ئەۋەتىشنىڭ نېمە ئەھمىيىتى ؟سىز دائىم مۇشۇنداق باشقىلار تاغدىن گەپ قىلسا باغدىن جاۋاب قايتۇرۇپ ،يامان يېرىگە كەلسە بىز قىلچە قىزىقمايدىغان «داڭلىق»لارنى ئوتتۇرىغا تاشلاپ بىزگە «ھاقارەت »قىلىسز،بىزگە قانچىلىك ناچار تەسىر بەرگىنىڭىزنى بىلەمىسز ؟ بىزگە ئۇ داڭلىقلار تىزىملىكىنىڭ كېرىكى يوق .خۇددى سىز ئېيتقاندەك بىز شەخسكە ۋە نوپۇزغا چوقۇنمايمىز،بىز ئۆزىمىزنى باشقىلاردىن دۆت چاغلىمىغاچ ئۆزىمىزنىڭ كۆز قارىشىنى ئوتتۇرىغا قويىۋاتىمىز....بىزگە مۇنازىرە بىلەن قىلچە مۇناسىۋىتى بولمىغان شەخسلەر، تور بەت ،ژۇرناللار ئارقىلىق «سەن نېمىگە ھېساب بۇلار ئالدىدا!! »دەيدىغان كۆز قارىشڭىزنى ئىپادىلەپ بەرمەڭ ،،،،سىزنىڭ نيۇتۇن توغرىسدىكى سەپسەتىڭىز ئاللىقاچان  ئاغدۇرۇلۇپ بولۇپتۇ،مەن مۇنازىرىگە دىققەت قىلغۇچى ئۈچىنچى تەرەپ بولۇش سۈپتىم بىلەن شۇنداق قارايمەن ، لېكىن سىز ئېتىراپ قىلماپىسز ....--ئوماق لايلا،بەيدۇغا يوللاڭ بۇ ماقالىنى دەپسىز،بۇنى بەيدۇغا بۇرۇنلا يوللۇغان دەپ ئادىرسىنى بەردىم،نيوتون قانۇنلىرىنى بىر-بىرىگە زىت ئەمەس دەيدىغان ماقالىلەر ساماندەك تۇرسا---گەپ ماقالىلارنىڭ سانىغا قارىماڭ،ئاساسى مېغىزىغا قاراڭ،ماقالە قايسى ئەدەملەرگە چېتىلسا شۇ كىشىلەرنىڭ ئىسمى تىلغا ئېلىنىدۇ،بۇ سىزگە   قانداق قىلىپ  ھاقارەت قىلغانلىق بولىدۇ،
باشقىلارغا بەرگەن تەسىرنى دىسىڭئىز بۇ سىزنىڭ شەخسى قارىشىڭئىز،يەنە كېلىپ باشقىلار ماڭا قانداق قارايدىغاندۇ دەپ ياشاشنىڭ مەنىسى بولمايدۇ،مۇستەقىل ئەدەم بولغاندىن كېيىن مۇستەقىل قارىشىڭئىز بولسۇن،‹سۇنىڭ ئېقىشىغا ،توكنىڭ يېتىشىغا›بېقىپ ياشاش ئاشۇ دەۋەردە توغرا بولىدۇ،لېكىن ھەق ،بىلەن ناھەق ئايرىلغاندا بولسا،ئۇنداق بولمايدۇ،نيوتون توغۇرسىدا تېخى بىرەرى بۇ نەزىريەنى ئاغدۇرۇپ باقمىدى، راستىنلا ئاغدۇردى دەپ قارىسىڭئىز دۇلىتىمىز ياكى خەلقئارادا نەشىر قىلىنىدىغان ژورنالدا ئېلان قىلىڭ،بىزمۇ شۇنداق قىلغان،مۇنازىرە بىزنىڭ مۇشۇ توردىلا بولىۋاتىدۇ دىسىڭئىز بولمايدۇ،پۇتۇن دۇنيا دىگۇدەك بۇ مەسىلىنى مۇنازىرە قىلىۋاتىدۇ،بۇ نورمال ئەھۋال،ئادەتتە بىز دىگەن سۇزىڭئىز كوپەرەك كىشىگە ۋەكىللىك قىلسا بولاتتى،
شەپەرەڭ ياخشى كۇرمەيدۇ دەپلا قۇياشقا قارشى چىقىش مۇمكىنمۇ؟

سىزگە يەنە تەۋسىيەرىم ماقالىڭىزنى بەيدۇغا يوللاڭ ،پەننىي جەھەتتە شۇ يەردە جاۋاب ئالالايسىز ،بىز بۇ يەردە سىزگە  مىڭ چۈشەندۈرسەكمۇ بەرىبىر سىلەردەك  «ئالىملار »توپى ئالدىدا بىز بىر توپ  «ساۋاتسز»لار ، ئۈستىلەپ بۇ يەردە پەقەت «شەپەرەڭ»لەرلا بار،چۈنكى دىنىي جەھەتتىن  99 پىرسەنت ئۇيغۇر تورچىلىرىمىز دىنغا ئېتىقاد قىلىدۇ ،ئاللاھنى ئېتىراپ قىلىدۇ،  شۇڭا سىزدەك ماركىسىزمچى  بىلەن مەڭگۈ پۇرچىقىمىز پىشمايدۇ ، سىز بىر ماقالە ئارقىلىق بىزنى مىڭ يىلدىن بېرى مېڭىۋاتقان ئىدىيىگە زىت ھالدا ماركىسىزملىق قاراشنى ئېتىراپ قىل دېسڭىز چۆچۈرىنى خام ساناپ قاپسىز !!!يەنە كېلىپ مۇشۇنداق چەك باسمايدىغان ماقالىلەر ،تايىنى يوق ئىنكاسلار ئارقىلىق!!! ،،،،،--بۇ سۇزىڭئىز تامامەن دىگۇدەك چەك باسماپتۇ،قاراڭ،نيوتون 1684-يىلىغا كەلگەندە ، خىرىستىئان دىنىدىكى تەڭرى بۇ ئالەمنى يارىتىپ بولۇپ ،ھەركەت قىلدۇرۇشنى ئۇنتۇپ قاپتۇ دەپ قاراپ،تەڭرىنى چاقىرغان،ئەمىليەتتە بولسا نيوتون  بۇ تەڭرىنى مۇشۇنچىلىك ئاددى ساۋاتنى بىلمەيدۇ دەپ قارىغانلىغىدىن بولغان ئىش،سىزنىڭ  ۋە سىز داۋاتقان 99پىرسەن ئويغۇر مۇشۇلمانلىرىمىزمۇ شۇنداق قارايدۇ دىسىڭئىز بولمايدۇ،پۇتۇن دۇنيانىڭ 26پىرسەن ئاھالىسى خىرىستىئان دىنىغا ئېتىقات قىلىدۇ،بۇ يەردىكى مەسىلە خىرىستىئان دىنى مەسىلىسى،شۇنىڭ ئۇچۇنمۇ ماقالىنىڭ تىمىسى‹ كېمبىرىج ئۇنىۋېرىستىتوتى خىرىسىتئان دىنى ئىنىستىتوتى دىققەت قىلمىغان نيوتوننىڭ لوگىكىلىق خاتالىغى›دىگەن تىمىدا ئېلان قىلىنغان،ئىسلام دىنى دىگەن سۇز يوق،يەنە كېلىپ ئىسلام دىنىدا تەڭرى ھەممىنى بىلىدۇ،قانداق ئىشقا چاقىرسا ‹لەببەي› دىمەيدۇ،ئاساسى گەپ مۇشۇ شۇ،
شەپەرەڭ ياخشى كۇرمەيدۇ دەپلا قۇياشقا قارشى چىقىش مۇمكىنمۇ؟

بۇ ماقالىگە بۇنچە ۋاقىت ئىسراپچىلىق قىلىشنىڭ ھاجىتى بارمۇ؟
   مۇشۇ ماقالىنى مۇنبەرگە يېزىشقا ئىسراپ قىلغان ۋاقىتنى،باشقا ئىزدىنىشلەرگە قارىتىپ ،ئۆزۈڭلار نىڭ ۋاقتىنى چىڭ تۇتۇپ ئىزدەنسەڭلار،ھەم باشقىلارنىڭ مۇشۇ قالايمىقىنىڭلار ئۈچۈن ۋاقىت .ئىسراپچىلىقى قىلماسلىق ئۈچۈن ، بىر ياخشى ئىش قىلىپ مۇشۇ ماقالەڭلارنى ،مۇكەممەل قىلىپ نەتىجە قازانغاندا ئېلان قىلساڭلار بولمامدا؟--- گېپىڭئىزگە قارىغاندا بۇنى نەتىجە قازاندى دەپ ئېيتىشقا بولمامدۇ نىمە،پۇتۇن دۇنيا ئۇقۇپ بولغان ماقالىنىڭ ئۇيغۇر تىلىدىكىسىنى  تورغا يوللاپ قويدۇق،يەنە كېلىپ بۇ خەنزۇ تىلىدا ئېلان قىلىنغاندىن كېيىن نۇرغۇن كىشىلەر بولەكلەرگە پارچىلاپ ،كۇچۇرۇپ ئۇزىنىڭ قىلىۋېلىشتى بۇ ماقالىنى،بەزى تور بىكەتلەت ئاپتورلارنى ئۇچۇرۇپ تاشلاپلا بۇ ماقالىنى ئۇزىنىڭ قىلىۋالدى،ئاگاھلاندۇرۇشمۇ كار قىلمىدى،يەنە كېلىپ تورداشلار قىزىقۇرمۇ بۇ تىمىغا دەپلا يوللۇغان،مەجبۇرلانغاندەكلا ھىس قىلىپ قاپسىز،ئۇنداق ئەمەس ئەپەندىم،بەزىدە بىر مەسىلىگە ئۇزىمىزنى بەك ئۇرىۋەتسەك شۇنداق تۇيغۇغا كېلىپ قالىمىز،بىر مەزگىلدىن كېيىن ‹ نيوتون پۇتكۇل ئىنسانىيەتنى ئالداپتۇ›دىگەن ئىدىيەگە كېلىپ قالىسىز،بۇنىڭغا سەل ۋاقىت كېرەك،ئالدىرىمايلى،
شەپەرەڭ ياخشى كۇرمەيدۇ دەپلا قۇياشقا قارشى چىقىش مۇمكىنمۇ؟

جاۋاب :بۇ سۆزىڭىزگە خەزنىچى شۇنداق ئىېنىق جاۋاب بېرىوپ بولغان ،خىرىستيان دىنى توغرىسدا ئاددىي ساۋات بىلمەيدىغان ىسزنىڭ سۆزىڭىز ماڭا نىسبەتەن ئۆزىڭىزنىڭنى راستقا چىقىرىشتىن باشقا ئىش ئەمەس.. نيۇتۇن ئەينى ۋاقىتتا نېمە ئويلىدى ،ئۆزىڭىزنىڭ بۇ ماركىسزمچە كاللىڭىزدىن مەڭگۈ ئۆتكۈزەلمەيسز
،سىزنىڭ  ۋە سىز داۋاتقان 99پىرسەن ئويغۇر مۇشۇلمانلىرىمىزمۇ شۇنداق قارايدۇ دىسىڭئىز بولمايدۇ،پۇتۇن دۇنيانىڭ 26پىرسەن ئاھالىسى خىرىستىئان دىنىغا ئېتىقات قىلىدۇ،بۇ يەردىكى مەسىلە خىرىستىئان دىنى مەسىلىسى،شۇنىڭ
