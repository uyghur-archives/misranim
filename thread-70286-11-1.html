گىتلىر يەھۇدىيلارنىڭ ۋە ناتسىىتلارنىڭ زىيانكەشلىكىگە ئۇچرىغان باشقا مىللەت كىشىلىرىنىڭ  نەزىرىدە بىر قانخور ئالۋاستى ، جاللات .
نېمىس مىللەتچىلىرىنىڭ نەزىرىدە ئۇ گىرمانلارنىڭ دۇنيا دىكى ئەڭ مۇنەۋۋەر مىللەت بولۇش سۈپىتى بىلەن باشقا پەس ئىرقتىكىلەرگە  ھۆكۈمران بولۇش غايىسى  ئۈچۈن بىر ئۆمۈر كۈرەش قىلغان مىللىي قەھرىمان ۋە داھىي .
ئەمدى گىتلىرنىڭ زىيانكەشلىكىگە ئۇچرىمىغان ئەكسىچە ئۇ يېڭىپ چىققان بولسا پايدىسىنى كۆرۈش ئېھتىمالى بولغان  بەزى قەۋملەر ئۈچۈن (مەسىلەن : ئۆزلىرىنى  گېتلىر  مەستانە بولغان ساپ ئارىيان  نەسلىگە تەۋە  ھىسابلىشىدىغان مىللەتلەرئۈچۈن ) ئۇنىڭ قىلغانلىرىنىڭ ياخشى -يامانلىقى مۇھىم ئەمەس ، مۇھىمى ئۇ شۇ چاغدا يىڭىپ چىققان بولسا ئۆزلىرىمۇ ئۇنىڭ سايىسىدە قانداق مەنپەئەتكە ئىگە بولدىغانلىقى توغرىسىداشېرىن چۈشلەرنى كۆرۈش خالاس .
ئەگەر شۇچاغدا بۇ تەلۋە سوۋىتنى يەر-بىلەن يەكسان قىلىپ ،  بۈيۈك بىرتانىيە بىلەن ئامرىكىنى تىز پۈكۈشكە مەجبۇرلىغان بولسا ، دۇنيادا ناتسىتلارنىڭ مارشى ھازىرغىچە ياڭراپ ، بايرىقى ھەممىلا يەردەلەپىلدەپ  تۇرغان بولسا (ئۇلۇغ ئاللا ئىنسانىيەتنى ئۇنداق قارا كۈنلەردىن ساقلىغاي ) زامانىمىزدىكى سىياسئونلار،تارىخچىلار ئۇنى ھازىرقىدەك جاللات دەپ تىللارمىدى ؟تارىخ كىتابلىرى ھازىرقىدەك گىتلىرنى سۆكۈدىغان مەزمۇنلار بىلەن توشارمىدى ؟
شىڭ شىسەيمۇ خۇددى شۇنداق بىر شەخس . ھەركىىم ئۇنىڭغا باھا بەرگەندە ئۆزىنىڭ مەنپەئەتىنى  چىقىش قىلىپ باھا بېرىدۇ.
ئەينى چاغدا ئەگەر يەنە پەيغەمبەر كېلىدىغان بولسا ، جانابى شىڭ دۇبەن جاناپلرى شەكسىز بۇ شەرەپكە نائىل بولاتتى دەپ مەدھىيلىگەنمۇ بىزدىن چىققان ئۆلىما ئىدى .  
قېرىنداشلار، تىللاشقا توغرا كەلسە ئەشۇ ئۆلىمادەك كىشىلەرنى تىللىساق ئەڭ توغرا بولىدۇ . شىڭ شىسەي  قولى ئەجداتلىرىمىزنىڭ قېنى بىلەن بويالغان بولسىمۇ ، ئۇنىڭ قىلغانلىرى ئۆزى ئىتقاد قىلىدىغان مەسلەككە ، ئۇ ۋەكىللىك قىلىدىغان گورۇھنىڭ مەنپەئەتىگە ئويغۇن ئىدى . شۇڭا مېنىڭچە ئۇ ئۆمرىنىڭ ئاخىرغىچە ئۆزىنىڭ قىلغانلىرىدىن ئۆكۈنمىگەن بەلىكى ئۆزۈم قىلىشقا تېگىشلىك بۇرچۇمنى ئادا قىلدىم دەپ پەخىرلىنىپ ئۆلگەندۇ....
