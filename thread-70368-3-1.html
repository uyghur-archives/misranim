يەنە قايتىپ بېرىپ تىجارەتكە چۈشۈپ كەتتىم،يەنىلا ئوخشاش 3 ۋاق تاماق يېسەم 5 ۋاق تاياق يەيتىم،خۇداغا نەچچە زارلىنىپ يىغلاپ كەتتىم،لېكىن خۇدايىمنىڭ بەرگەن جازاسى شۇكەن،ئۆيگە نەچچە يەردىن ئەلچى كىردى بىراق يېشىم توشمايتى ھەم توي قىلىشتىن قورقاتتى،چۈنكى مەن بىر نابۇت بولغان ئادەم،توي قىلسام ئۆيدىكىلەرنىڭ يۈزى يەنە تۆكۈلەتتى،بۇ ئىشنى سىزگە دەۋەتتىم ھەدە مېنى ئاپام ھازىرغا قەدەر پاك دەپ ئويلايدۇ،لېكىن مەن ئۇلارغا يۈز كېلەلمەيمەن،بەزىدە دادامنىڭ شۇ قاتتىق قوللىرىدىن ئاغرىنىمەن،مېنى بىتەلەي قىپ قويغىنى دادامنىڭ شۇ تاياقلىرى دەيمەن لېكىن ھەي بولدىلا..........
كۈندە تاياق يەپ يامان گەپ ئاڭلاپ يۈرىۋەرگەچكىمۇ كاللام ئانچە ساق بولمىدى،قويغان-تۇقىنىمنى بىلمەيتىم،بەزىدە پۇل قايتۇرۋېتىپ خەقنىڭ نەچچە پۇل بەرگىنى ئېسىمدىن چىقىپ قېلىپ قايتا سورايتىم،بۇرۇن ئەستە تۇتۇش قابىلىيتىم بەك ياخشى ئىدى،بىر نەرسىنى بىر قېتىملا ئوقۇسام يېرىمى دېگۈدەك ئېسىمدە قالاتتى،لېكىن ھازىر نېرۋامدىن كېتەيلا دېگەن ئىدىم،بىر قانچە قېتىم سودىدا ئېزىپ كېتىپ دادامدىن شۇ ئادەم جىق بازاردىلا تاياق يەپ كەتتىم،بۇنداق تۇرىۋەرسەم ئۆزەمنىڭ ئېلىشىپ قالدىغانلىقىمنى بىلىپ يەتتىم،مېنى تېپىۋالغان چاغدا 15 ياش ئىدىم،1 نەچچە يىللىق تاياقنىڭ زەربىسىدىن كاللامدىن چاتاق چىقايلا دېگەن ئىدى،بۇنداق ئۆتىۋەرسەم بولمايدۇ دەپ يەنە قاچتىم،بۇ قېتىم ئاپامنىڭ ئالتۇن ھالىقىسىنى ئېلىپ چىققان ئىدىم،مەندىمۇ ئامال قالمىغان ئىدى،يەنە ئۈرۈمچىگە كەلدىم،بىر رېستۇراندا ئىشلەيدىغانغا ئورۇنلاشتىم،2 ئاي ئىشلەش جەريانىدا خېلە ئەسلىگە كەلگىلى تۇرغان ئىدىم،ئالدىراپ بىر نەرسىنى ئۇنتۇپمۇ قالمايدىغان بولغان ئىدىم،لېكىن 8-ئىيۇن ۋەقەسى يەنە ھەممە ئىشنى بۇزىۋەتتى،بۇ رېستۇرانمۇ چېقىلىدىغان بولۇپ يەنە قايتىپ كەلدۇق،يۇرتۇمغا كېلىپ يەنە بىر رېستۇراندا ئىشلىدىم،بۇ يەردە چىن كۆڭلۈمدىن بىرىنى ياخشى كۆرۈپ قالغان ئىدىم،ئۇمۇ ماڭا كۆڭلى بارلىقىنى ئېيىتقان لېكىن مەن ئۇ بالىغا لايىق ئەمەس چۈنكى مەن بىر ناپاك قىز ئىدىم،ئالدىمدا نەچچە يىغلاپ كەتتى،نېمە دەي ماقۇل دېدىم،ئوغۇل بالىدىن شۇنچىلىك سەسكىنىپ كەتكەن ئىدىم،يەنە كۆڭۈل شۇ ياخشى كۆرۈش يولىغا باشلاپ قويدى،مەن ئۇنىڭغا ھەممىنى دېدىم لېكىن ناپاك ئىكەنلىكمنى يۇشۇرۋالغان ئىدىم،مەن بىلەن توي قىلىدىغانلىقى ئېيتىپ مېنى ئۆيگە بېرىشقا ئۈندىدى،گېپىگە ھە دەيتىم لېكىن ئاپىسى مەن بىلەن ھەرگىز توي قىلغىلى قويمايتتى،بۇنىسى ماڭا ئايان چۈنكى ئۇلار كەلپىندىن ئىدى،كەلپىننىڭ قائىدىسى ئوغۇللىرىنى ئاساسەن ئۆز يۇرتىدىن ئۆيلەش،توي قىلسام بىلىپ قالارمۇ دەپ قورقمۇدۇم،چۈنكى توي قىلالىشىم مۇمكىن ئەمەس شۇنداقتىمۇ ئۇ  مەن بىلەن بىللە ئۆيۈمگە باردى،ئۆيدىكىلەرمۇ ئارلىشىشىمىزغا رازىلىق بەردى،لېكىن مېنى ئۇ ناھىيىدىن قايتۇرۇپ ئەكىلىۋالىدىغان بولدى،ئۇمۇ يۇرتىدىن چىقىپ مەن ئىشلەۋاتقان يەردە ئىشلەۋاتقان ئىكەن،شۇنداق يالۋۇرۇپتىمۇ ئۆيدىكىلەر قايتۇرۇپ ئەكىلىۋالىمىز ئەگەر توي قىلىمەن دېگەن نىيتىڭىز بولسا،قائىدىسىنى قىلىپ ئېلىپ كېتىڭ دېدى،بۇ مەۋلاننىڭ 19ياش مېنىڭ 17ياش ۋاقتىم ئىدى.
بىلىپ سۆيگىن كېيىن ئېچىنما، بىلىپ تاشلا كېيىن سېغىنما.

شۇنداق قىلىپ مەۋلانمۇ كەتتى،مەن ئۆيگە قايتىپ كەلدىم،لېكىن كۈنلۈرۈم بۇرۇنقىدەك تاياق بىلەنلا ئۆتمەيتى،داداممۇ ئايىدى بولغاي،خېلە يامان ئەمەس ئۆتىۋاتاتتىم،مەۋلانمۇ كۈندە تېلىفۇن قىلىپ تۇراتتى،5 ئايچە ئارلاشتۇق لېكىن ماڭا مەۋلان سەل ئۆزگۈرۈپ قالغاندەك تۇيۇلىۋاتاتتى،بەزىدە تېلىفۇن قىلسام بېسىۋېتەتتى،ئالدىراشتۇ دەپ ئويلۇدۇم،لېكىن ئىش مەن ئويلۇغاندەك ئۇنچىلا ئاددىي ئەمەسكەن،بۇ 2- قېتىم بىرسىگە ئىشەنگىنىمنىڭ جازاسى ئىدى،مەن بۇ يەردە مەۋلاننى ياد ئېتىپ تۇرغان بىلەن مەۋلان ئۇ يەردە باشقا بىرى ئارلىشىپ يۈرۈپتۇ،توي قىلالمايدىغانلىقىمنى بىلىپ تۇرۇقلۇق يەنىلا بۇ ئازاپلارغا بەرداشلىق بېرەلمىدىم،ماڭا ۋاي ئۇنداق ۋاي مانداق دېگەن بىلەن ئەسلى ئۇ بىر ئالدامچىكەنتۇق،بۇلارنى بىلگەندە بولسا يۈرۈگۈم پىچاق بىلەن تىلغاندەك ئاغرىۋاتاتتى،بۇ ئىشلار بولۇندى،مەۋلانمۇ مېنى تاشلىۋەتتى،ئۆيدىكىلەرنىڭ پوزىتسىيسى يەنە ئۆزگەردى،بىر كىيىم-كېچەك دۇكىنىدا ئىشلىدىم،چۈنكى كۈندە دادامنىڭ كۆزىگە كۆرۈنۈپ يۈرۈپ كۈندە تىل ئاڭلىغىم يوق ئىدى،نېمىشقىكىن بۇرۇن بەك ئىتتىك يىغلىۋىتەتتىم،ھازىر بولسا كۆزۈمنىڭ يېشى قۇرۇپ كەتتىمىكىن تاڭ،ئۇنچە ئىتتىك كۆزۈمدىن ياش چىقمايدىغان بولۇپ قالدى،دەسلىۋىدە دادام ئۇنىماي تۇرىۋالدى،يەنە يوقاپ كېتىدۇ دەپ،مىڭ تەستە ماقۇل كەلتۈرۇپ ئىشقا كىرىشىپ كەتتىم،دەسلەپكى ئايلىق مۇئاشىمنى ئالغۇچە دادام پۇل بەردى،ئايلىقىمنى ئالغاندىن كېيىن دادامغا بەرگەن ئىدىم،700 يۈەندىن دېيىشىپ ئىشلىگەن ئىدىم،دادام دەسلىپىدە 100 يۈەن ئايرىپ بېرىپ قالغىنىنى لازىم بولسا بېرەي دېگەن ئىدى،مەنمۇ ماقۇل دېدىم،كۈندىلىك ئاپتوبۇس كىراسى،چۈشتە بىر ۋاق تاماق دېگەنلەرگە بۇ 100 يۈەن نەچچە كۈنلۈكلا بولىدىكەن،15 كۈن ئۆتتى،داداممۇ 100 يۈەندىن باشقا پۇل بەرمىدى،ئاپام كۆكتات ئالغىلى ئەپ چىققان پۇلدىن ئاشقان پۇللارنى يىغىپ ماڭا بېرەتتى،چۈشتە بىر ۋاق تاماق بىر تال نانغا ئۆزگەردىن شۇنداق قىلىپ ئۈچ ئاي ئىشلىدىم،ھە بوپتۇ مەنمۇ نەچچە يۈز ئەمەس نەچچە مىڭلاپ يوقاتتىم،دادام خوش بولسىلا ئاچ قالساممۇ مەيلى دېدىم بىراق دادام باشقىچىلا بولۇپ قالدى،خۇددى مېنىڭ قولۇمغا قاراپ قالغاندەكلا 3-ئايلىق مۇئاشىمغا 800 يۈەن بەرگەن ئىدى،ئۆستۈرگەندىن كېيىن مەنمۇ دادامغا پۇلنى بەرسەم،ئاران 800 بەردىما دەيدۇ،ھە بۇ ئايلىققا يۈز يۈەن قوشۇپتۇ دېسەم ھە دەپ،يەنە 100 يۈەن بەردى،ھەممىنى چۈشەندۈردۈم ھازىر چۈشلىكى نان يەۋاتىمەن،كىرا دېگەندەكلەرگە بىر ئايدا 100 يۈەن يەتمەيدىكەن دادا،200 بەرسىڭىز دېسەم شۇ يەردىلا پۇلنى يۈزۈمگە ئېتىۋەتتى،تۇرۇپلا قالدىم.........
بۇ بىر كېچە بولسا 100 يۈەن ئارتۇق بېرىڭ دېگەننىڭ ئورنىغا نەچچە يۈز قېتىملىق تاياقنى ئالدىم،كۆزۈممۇ كۆكىرىپ كەتكەن ئىدى،سىرىتقىمۇ چىقالمىدىم،شۇنچىلىك خورلۇق ھېس قىلدىم،سىزمۇ ئويلاپ بېقىڭ ھەدە ئۆيۈم بولسىمۇ ئەتىدىن كەچكىچە سىرىتتا يۈرۈيدىغان ئادەمگە 100 يۈەن نېمە بولىدۇ،چىرايلىق گەپ قىلسىمۇ بولىدىغۇ نېمە ئۈچۈن شۇنچىلىك خورلۇققا قالىمەن،قاراڭ قولۇمغا مانا بۇ دادامنىڭ يالدامىسى.(قىزنىڭ سول بېلىكىدە جەينەكتىن باشلاپ تەخمىنەن بىر غېرىچ كەلگۈدەك يوغانلا تارتۇق بار ئىدى.)قانچىلىك تەستە ساقايغانلىقىنى بىلمەيمەن،كۈندە قولۇمغا قاراپ يىغلاپ كېتەتتىم،قىش ۋاقتى بولغاچ ھەم دادام بىلەن بازارمۇ بازار ماڭغاچ قولۇم ئىششىۋالغان كېيىن يىرىڭداپمۇ قالغان تېخى ياز ۋاقتى كالتە يەڭ نەرسىمۇ كىيەلمەيمەن،ھەر قېتىم قولۇمغا قارىسام دادامغا شۇنچىلىك ئۆچ بوپ كېتىمەن،لېكىن مېنىڭ ئۇلارغا قىلىپ قويغىنىممۇ ئاز ئەمەس،چاۋاك بىر قولدىن چىقمايدۇ بۇنىسى ئايان،لېكىن مەن بەك كۆپ خورلۇقلارنى تارتىپ كەتتىم،نەچچە كۈنگۈچە ئورنۇمدىن تۇرالمىدىم،قورسىقىمغا بەكلا دەسسىۋەتكەن ئىدى،چوڭراقمۇ تىنالمىغان ئىدىم،تاماققۇ ئەسلا ئىككى كۈن يېتىپتىمەن،ئىشقىلىپ تىللا ئاڭلىدىم،ئادەم بولغانغا بەكلا تويغان ئىدىم،سەل مىدىرلىغىدەك بولغاندىكىن شۇ كۈنى ئايرىپ تاشلاپ قويغان 100 يۈەننى ئېلىپ يەنە ئۆيدىن چىقىپ پۇتۇم باشلىغان يەرگە ماڭدىم،20 يۈەنلىك كىيىم دۇكىنىدىن بىر قۇرلا كىيىم ئېلىۋالدىم،رېستۇرانغا بېرىپ ئىشلىسەم،ياتاق-تاماق ھەممىنى ئورۇنلاشتۇراتتى،بۇنداق دېسەم قالايمىقان يەردە يۈرۈپ كەتكەن قىز بالىكەن دەپ ئويلاپ قالماڭ ھەدە رېستۇران دەپ ئېتى يامان ئاتىلىپ قېلىشى مۇمكىن لېكىن ئۇ يەر پەقەت شۇنداق قالايمىقان يۈرگەنلەر ئۈچۈن قالايمىقان،ئەمما ئۇ يەردە ئىشلىگەن قىزلار بەزىلىرى ئائىلىسىدىكى قىيىنچىلىق سەۋەبىدىن ياتاق-تاماق ئورۇنلاشتۇرۇلغاندىن كېيىن پۇلنى كۆپرەك تېجىيەلەيمەن دەپ ئىشلەپمۇ قالىدۇ،ئۆزىڭىزنى باشقۇرۇپ يۈرسىڭىز بولدى،مېھمانلار قالايمىقان گەپ قىلسا دېرىكتور چىقاتتى،ئەمەلىيەتتە ئىشلەيدىغان قىزلار بەك ياخشىئ.
بۇ يەردە 3 ئاي ئىشلىگەن بىر كۈنۈم ئاپامنىڭ بىر سىڭلىسى دوسلىرى بىلەن قاتار چاي ئوينىغىلى كىرىپ كۆرۈپ قالدى،كۆرۈپلا ھەيران قالدى مەنمۇ گەپ قىلماي ئۆز ئىشىمغا مېڭىۋەردىم،كېيىن شۇ يەرگە مەسئۇل قىز بىرەيلەننىڭ مېنى چاقىرغانلىقىنى دېدى،مەنمۇ چوقۇم شۇ ئاچامغۇ دەپلا كىردىم،راسلا شۇ كەن دوسلىرى كەتكەندىن كېيىن مېنى چاقىرتىپتىكەن،ئىشلىرىمنى يەنە بىر قىزغا تاپىلاپ قويۇپ ئاچامنىڭ قېشىغا كىردىم،چاچلىرىمنى سىلاپ يىغلاپ كەتتى،نېمىشقا يىغلايسىز ئۆيدىكىلەرگە بىرە ئىش بولمىغاندۇ دېسەم ئۇنداق ئەمەس بەكلا ئەنسىرەتتىڭىز،سىز ئۆيدىن چىقىپ كەتكەن كۈنى ئاپىڭىز تېلىفۇن قىلىپتۇ،ئاخشام مۇشۇنداق ئۇرىۋەتتى،ئۆيگە سولىۋالغانقا مەنمۇ ئامال قىلالمىدىم،كېيىن قارىسام كۆك چۈشمىگەن يېرى قالماپتۇ،بىچارە ئاچچىقىدا بىرە ئىشقا يولۇقۇپ قالمىسا بولاتتى دەپ بەك يىغلاپ كەتتى..................گەپ بەك نۇرغۇن شۇنداق قىلىپ ئاچام بىلەن خوشلاشتۇق،ۋاقتى يەتسىلا كېلىپ تۇرىدىغانلىقىنى ئېيىتتى،يەنىلا تۇققاندە باشقىچلا،ئاچام كېتىپ نەچچە كۈن بولغاندا شۇ رېستۇراننىڭ ئالدىدا دادام بىلەن ئۇكامنى كۆرۈپ قالدىم،چوقۇم مېنى ئىزدەپ كەلدىغۇ دەپ ئويلۇدۇم،دېگەندەكلا مېنى ئىزدەپ كەپتىكەن،ئاپامنىڭ شۇ يىغىلىرىنى دېمىگەن بولسام داداممنىڭ ئالدىغا ھەرگىز چىقمايتىم،قىز بالىغا ئاتا بولغان لاتا بولا دەپ يەنە يۈزۈمگە سېلىپ تۇرىۋالمىسۇن دەپ دادامنىڭ ئالدىغا چىقتىم،خوجايىندىن 15 كۈنلۈك ئىش ھەققىمنىمۇ ئالماي يەنە دادام بىلەن قايتىپ كەلدىم.

داۋامى بار.
لېكىن بۇ قىزغا نېمە دېيىشنى پەقەتلا بىلمىدىم،قىزغا ئىچ ئاغرىتاي دېسەم ئەسلى،ئاتا-ئانىسىغا ئىچ ئاغرىتاي دېسەم ئەسلى قىزنىڭ بەدەنلىرىدە قايتا-قايتا كۆك چۈشكەن يەرلە كۆپ بولغاچ داغ بولۇپ قالغان ئىدى،قالغىنىنى ئاخىرىنى ئوقۇپ ئاندىن ئويلۇنۇپ باقساڭلار.
بىلىپ سۆيگىن كېيىن ئېچىنما، بىلىپ تاشلا كېيىن سېغىنما.

قايتىپ كەلگەندىن كېيىن دادام ئەمدى ئۆيدىن چىقىپ كەتمەسلىكىمنى كېيىنلىكىمنى ئويلۇشۇشىمنى،يامان يولغا مېڭىپ قېلىشىمنى خالىمايدىغانلىقىنى ھەم مېنى ھۈنەرگە ئەكىرىپ بېرىدىغانلىقىنى ئېيىتتى،كۆڭلۈم شۇنچىلىك خوش بولدى،بۇرۇنقى بەتخەجلىكىمنى تۈگەتتىم دادام كۈندە 10 يۈەن بېرەتتى،پۇل تېجەش ئۈچۈن مېڭىپ بېرىپ كېلەتتىم،شۇ بەرگەن پۇللاردىن ئاشقىنىنى يىغىپ قويۇپ دادامدىن پۇل سورىماي كىيىم دېگەندەك لازىمەتلىكلەرنى ئالاتتىم،داداممۇ خېلىلا خوشال ئىدى،لېكىن بىر ئۆتكۈزگەن خاتالىقنى ئەسلىگە كەلتۈرىۋالغىلى بولمايدىكەن،يەنە ئازراقلا بىرە ئىش بولۇپ چېچىلىپ قالسا ياكى بىرە نەرسىگە ئازراق پۇل كاملاپ قالسا مېنىڭ يەنىلا تۈگەشكەن ۋاقتىم ئىدى،شۇڭا بىر كۈنمۇ دەم ئالمايتىم،ئەتىگىنىسى ئەتىگەندە تۇرۇپ نامازلىرىمنى ئۆتەپ بولۇپ،ئاپامنىڭ تاماق ئېتىشىگە ياردەملىشەتتىم،تاماقنى  يەپلە دۇكانغا ماڭاتتىم،ئۇستام ماڭا بەك ئىشىنەتتى،مەنمۇ ئۇستامنىڭ ئىشەنچىسىنى يەردە قويمايتىم،1 يىل دۇكاندا ھۈنەر ئۆگۈنۈپ نەچچە يۈزلەپ سودا قىلىپمۇ ئۇستامغا پۇلنى جايىدا تاپشۇراتتىم،مەن قىلىپ قويغان بىلەن ئۈستۈمدە خۇدايىم باردە،شۇ بىر قېتىملىق ساۋاق بەكلا ئاچچىق بولغان ئىدى،بەزىدە شۇ پۇلنى تۇتۇپ ئولتۇرۇپ كۈلۈپ كېتىمەن،مۇشۇ پۇل بولمىغان بولسا،مېنىڭ مۇشۇ بالايى نەپسىم بولمىغان بولسا ئاتا-ئانامنىڭ ئالدىدا قارا يۈز بولۇپ قالمايتىم،مەنمۇ باشقا سەبىيلەردە مەكتەپتە تۇللۇق ئوقۇيالايتىم،ھەتتا ئالىي مەكتەپتىمۇ ئوقۇيتۇم،ھەي............ھازىر قانداق بولۇپ قالدىم،توۋا دەيمەن،باشقىلارنىڭ شۇنچىلىك تاتلىق بالىلىق ئەسلىملىرى بولىدۇ،لېكىن مېنىڭ ئۆتمۈشۈم ۋە ئەسلىمىلىرىمنى سورىسىڭىز مانا مۇشۇنداق مۇرەككەپ ھەم قورقۇنۇچلۇق تاكى 13 ياش ۋاقتىمدىن 18 ياشقا كىرگۈچە كۆرمىگەن ئىشلارنى كۆرۈپتىمەن،يېمىگەن تاياقلارنى يەپتىمەن،ئەڭ پۇشايمان قىلىدىغىنىم دوستۇم دەپ يۈرگەن ئادەم مېنى ئالداپ ئاپىرىپ خەققە تۇتۇپ بەرگىنى بولدى،شۇڭا ھازىر بىرە ئوغۇل بالا گەپ قىلسىمۇ شۇنداق يىغۇ باسىدۇ،مەنمۇ خەققە ئوخشاش نورمال ئادەمتىمغۇ نېمىشقا مۇشۇ كۈنلەرگە قالىمەن........
شۇنداق قىلىپ 1 يىل ئىچىدە ئۇستامنىڭ ھۈنىرىنى تۇللۇق قولۇمغا ئېلىپ بولغان ئىدىم،چۈنكى كاللام بۇرۇنقىدىن خېلىلا ساق ئىدى.مەن دۇئا ئېلىپ قايتىپ چىققاندىن كېيىن دادام ئىشلىمەي بىزگە يېقىن تۇرىسەن دەپ تۇرىۋالدى،ئۇنداق ئادەم نېمىگە ئۆگەتكەن بۇ ھۈنەرنى،ھۈنەر دېگەن قىلسا ئۈنىدىغان،قىلمىسا ئۆلىدىغان نەرسە بۇ........................دادام بىلەن يەنە جىدىلىمىز باشلاندى........يەنە تاياق يېدىم بوپتۇ،يېمىگەن تاياق بولمىغاندىنكىن ئۆلسەممۇ دادام قولىدا ئۆلەمەن،مېنى باققىنى ئۆزىنىڭ ئاچچىقىنى چىقرىىشتۇ دەپلا ئويلۇدۇم.نەچچە كۈن ئۆتكەندە دۇكاننىڭ بىر بۈرجىكىنى توسۇپ مېنىڭ ئىشلىرىمغا چۇشلۇق بىر نەرسە قىلىپ بېرىشنى دېدىم،بىر دۇكان بولغاندىكىن داداممۇ ماقۇل بولدى شۇنداق قىلىپ تىجارىتىم باشلىنىپ كەتتى،دەسلەپ سودا ئانچە ياخشى بولمىدى،كېيىن يېتەلمەيدىغان بولۇپ كەتتىم،دۇكان سايمانلىرىمنىمۇ تۇللۇق ئېلىپ بولدۇم،دەسمىينى دادام سالغاندىن كېيىن پۇل يىغىلىپ 1000 يۈەن بولغاندا دادامغا بېرەتتى،بىر قېتىم فۇيىنجى ئېلىش ئۈچۈن 4300 يۈەن پۇلنى دادامغا بەرمەي يىغىپ كەلگەن ئىدىم،بىر كۈنى تىللاپ كەتتى،بەكلا ئاچچىقىم كەلدى،مەن خەنزۇ بىلەن كۆرۈشۈپ ئەڭ ئەرزان بولغاندا 4300 يۈەنگە بېرىدىغانلىقى بىلگەندىن كېيىن پۇلمۇ خەجلىمەي فۇيىنجىغا پۇل يىغىۋاتاتتىم،بەلكىم دادام كۆڭلىدە بۇ پۇلنىمۇ كۆتۈرۈپ قېچىپ كېتەرمىكىن دەپ ئويلىدى بولغاي،ھە دەپ ئۆتكۈزىۋەتتىم،پۇلنىمۇ بەرمىدىم.شۇنداق قىلىپ پۇلنى شۇنىڭغا چۇشلۇق يىغىپ بولغاندىكىن دادامغا ئېيتىپ بىللە ئېلىشىپ بېرىشىنى ئۆتۈندۈم،داداممۇ بوپتۇ دەپ ماقۇل بولدى،ئىشلىرىمى تېخىمۇ ئاسانلاشتى.لېكىن دادامنىڭ كۆزىگە سىغمايلا قېلىۋاتاتتىم،ھە دېسىلا باشقىلار بىلەن كېلىشكەن باھانى بىر پو بىلەنلا يوق قىلىۋەتتى،ئەڭ ئاددىيسى خەقنىڭ ئالدىدا يۈزۈمنى قويمايتى،تىللاۋىرەتتى.ھەي ئىشقىلىپ ھەدە بۇ گەپلەرنى دەپ بەرسەم بۇ قىز يالغان سۆزلەۋاتامدىكىن دەپ ئويلاپ قالغانسىز،لېكىن ھەممىسى ئەمەلىيەت مەن 5 يىلنىڭ ياقى كۆرگەن كۈنلەرنى سىز 50 يىلدىمۇ كۆرمەيسىز ھەم بۇنداق كۈنلەرنى كۆرۈشتىن خۇدايىم ساقلسىۇن،خۇدايىم بەختىڭىزنى بەرسۇن.
-ئەمىسە دۇكىنىڭىزچۇ؟قانداق بولۇپ ئۈرۈمچىگە چىقىپ قالدىڭىز؟قىزدىن ئاخىرقى مىنۇتلىرىنى سورىدۇم.
