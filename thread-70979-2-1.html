مۇشۇ كىنونىڭ ئاخىرسىدىكى ئويغۇرچە ئېيتىلغان ئاشۇ ناخشىنىڭ نەرى بولمىغاندۇ؟
مۇشۇنداق بىر چوڭ تەسىر قوزغىغان كىنودا ئويغۇرچە تىلنىڭ چىققىنى بىزنىڭ تىلىمىزغا پايدىسى بولسا بۇلىدىكى ھەرگىزمۇ زىيىنى يوق......
بۇندىن بۇرۇن بىزنىڭ ئويغۇر تىلىمىز قايسى بىر چوڭ سەھنىدە كۆرۈنۈپ باققان؟
قايسى بىر سەھنىدە ئوتتۇرغا چىقىپ باققان؟
مانا ئەمدى مۇشۇنداق بىر تەسىرى چوڭ بولغان كىنودا بىزنىڭ تىلىمىز مىليارد خەلىققە ئاڭلىنىدىغان،بىرسى ئاڭلىسا:
بۇ قايسى مىللەتنىڭ تىلى؟ دەپ سورىشى مومكىن ھەمدە مۇشۇ سەۋەپ بىلەن بىزنىڭ تىلىمىزنىڭ تەسىر كۈچى ئاشىدۇ ئەمەسمۇ؟
كىچىككىنە بىر ئىشنى چوڭ دىن نۇقتىسىغا كۈتۈرۋاپسىلەر..............
مۇنبەردىكى بىر تۈركۈم ساختا مۇللىلار،مىللەت ئۈچۈن پۈتۈن بىر ئۈمرىدە دانچىلىك بىر ئەمەلى ئىش قىلىپ بېرەلمەيدىغان بىر تۈركۈم ساختا مىللەت سۆيەر شۇ بىر قىسىم كىشىلەردىن سورايدىغىنىم:
سىز ھازىرغىچە مىللەت ئۈچۈن نىمە قىلىپ بەردىڭىز؟
قۇلىڭىزدىن پەقەت كومپىيوتىرنىڭ ئالدىدا ياكى كۇچىدا ياكى چوڭ-چوڭ سورۇنلاردا باشقىلارنى سۈكۈشنىڭ بەدىلىگە ئۆزىڭىزنى مىللەت سۆيەر كىشى قىلىپ كۆرسىتىشتىن باشقا نىمە قىلىپ بېرەلەيسىز؟
بۇنىڭسىزمۇ ھازىر بەزى ياشلىرمىز ھەتتا نۇرغۇن ئويغۇرلىرمىز ‹‹ۋاي ئاللاھ››دەپ دىبالاردا ھەرخىل ھاراق سورۇنلىردا ئېيتىپ ئۇسۇل ئويناپ نەقرات كەلتۈرۈپ يۈرگەنلەر كۈرمىڭغۇ؟ھەجەپ شۇلارنى بىرەرسىمىزمۇ چىشلەپ تارتمىدۇققۇ؟
بەلكىم سىزمۇ مۇشۇنداق ئالاھىدە ناملار تىلغا نەقرەت كەلتۈرۈلۈپ ئېيتىلغان ئويغۇرچە ناخشىلارغا ئۇسۇلنى تازا ئوينىغانسىز..........
