ئەشۇنىڭغا كەتكەن ئىقتىسادىڭىزنى ساۋابى دەل تېگىدىغان ئىشلارغا ئىشلەتسىڭىز بولمامدۇ دىسىڭىز، بۇ دىگەن بۇرۇندىن تارتىپ بۇلۇپ كەلگەن ئىشكەن، ئاتا-بوۋىلىرىمىزمۇ شۇنداق قىپتىكەن ، ياكى بولمىسا ئىشىڭىزنى قىلىڭە ، سىزنىڭ نىمە كارىڭىز دەيدىكەن، كىچىك بولغاندىكىن ئېغىز غېرىچلىماي دەپ!
ئەمدى تېمىغا كەلسەك، ئىنساننىڭ دۇنياغا كىلىپ كەتمىكى بەكلا ئاددى ئەھمىيەتسىز بوپ قالغىدەك دىگەندەك گەپلەرنى قىپسىز، ئۇنداق بولسا ئالىمىمىز مەھمۇت قەشقەرىدەك مەشھۇر ئالىم بولۇڭ ياكى ئاكا - ئۇكا مۇسا بايۇفلاردەك مەشھۇر بايلاردىن بولۇپ خەلىق ئۈچۈن ياخشى ئىشلارنى قىلىپ دۇنياغا ياخشى دىگەن نامنى قالدۇرۇپ كىتىڭ، ئاندىن سىزگە شۇ ئۈلۈمنىڭ ئەھمىيىتى بولىدۇ! ئۇ كىشىلىرىمىزدەك مەشھۇرغۇ بولماڭ ماقۇل، ئاتا - ئانا، خۇلۇم-خوشنا، ئۇرۇق-تۇققانلار ماۋۇ كىشىمىز دۇنيادا ياخشى ئەمەللەرنى قىلىپ كەتكەن ، ياخشى ئىشلارنى قىلاتتى دىگەندەك دۇئالارنى قىلسۇن، شۇندىلا سىزنىڭ ۋە بىزنىڭ دۇنياغا كىلىپ كەتمىكىمىز ئاددىي ھەم ئەھمىيەتسىز بولمايدۇ ، شۇنداقلا قانداقتۇر ياغ پۇرىتىش دىگەندەك بىھۇدە ئىشلارنىمۇ قىلىشنىڭ ھاجىتى يوققۇ دەيمەن؟ گەپلىرىم ئېغىر كەتكەن بولسا ئىگىلىپ تۇرۇپ ئەپۇ سورايمەن! لىكىن  مۇنازىرە قىلغۇدەك تېمىدىن بىرنى ئوتتۇرغا قويۇپسىز!

تەڭ ئىدۇق تۇغۇلغاندا تەڭرى ئالدىدا،
ۋەلىكىن تەڭ ئەمەس دەۋىر ئالدىدا.
كۆرەڭلەپ كەتمىگىن نەتىجىلىرىڭدىن،
تەڭ بولارمىز ئاخىر بىركۈن قەبرە ئالدىدا.
