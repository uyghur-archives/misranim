يقىندا مىنىڭ بىردوستۇم ئۇيعۇر تىبابەت دوحتۇرلۇق سالاھىيەت كىنىشكىسى (2009-يىلى ئاتا مىراس كەسپى بۇيىچە شاگىرتلىققا كىرگەن سالاھىيەتتە)بىجىرگەن ئىكەن، بۇ كىنىشكىنى توردىن چىقارمۇ -چىقماسمۇ، راستمۇ-يالعانمۇ بۇنى بىلەلمەي مەندىن مەسلىھەت سۇراپ تلىفۇن ئاپتۇ ،مەنمۇ تەپسىلى ئەھۋالنى بىلمىگەچكە تۇرداشلاردىن مەسلىھەت سۇراي دەپ .....................
بۇ كىنىشكىنى باشقا كىشىلەرنىڭ ۋاستىسى بىلەن بىجىرگەن ئىكەن ، ئەگەر توردىن چىقمىسا قايسى ئۇسۇلدا تەكشۇرسە بۇلار؟
باشقىلار دەسلەپتە ئەنگە ئالدۇرۇپ بىجىرگەن سەھىيە ئىدارىسىدىن سۇرۇشتۇرسە ئارحىپى چىقسا بۇلىدۇ دەپتىمىش، بىراق قايسى سەھىيە ئىدارىسى ئارقىلىق بىجىرگىنىنى بىلگىلى بولمايدۇ-دە:
يەنە بەزىلەر ھەممە كىنىشكىلەر(تىببى ئونۋىرسىتلارنى ئۇقۇپ ئىمتىھان بىرىپ ئالعانلارنىڭمۇ) تۇردىن چىقمايدۇ، بىراق كىنىشكە چىققاندىن كىيىن بىرەر دۇحتۇرحانا نامىعا ،شۇدوحتۇرحانىنى باشقۇردىعان سەھىيە ئىدارىسى ئارقىلىق ئەنگە ئالدۇرعاندىن كىيىن ئاندىن توردىن چىقىدۇ دەپتىمىش :
زادى  قانداق بۇ ئىشلارنى تەپسىلى بىلسەڭلار بىر مەسلىھەت بەرگەن بولساڭلار...............
