مەن قايسى كۈنى ئۈرۈمچى كەچلىك گېزىتىدىن مۇشۇ تېما توغرلۇق بىر ماقالىنى ئۇقۇپ قالدىم . ھەم بۇنى يازغان ئادەمنىڭ قۇلىغا دەرىت بەرمىسۇن دەپ ، ئۇلۇغ ئاللاھتىن تىللىدىم . ھەقىقەتەن بۇنداق ئەبجەش گەپلەر تۇرمۇشىمىزدا بار . بۇ ئەبجەش گەپلەر زادى قانداق گەپلەر ؟ تۈۋەندىكىگە قاراڭ :
   « خىزمەتچىلەر : ئىدارىدە يىغىن ئاچتى ، يىغىندا يۇقۇردىن خۇڭ توۋ ۋىنجەن چۈشۈپتىكەن ، شۇنى ئۈگىنىدىكەنمىز . جىبەرلىكتە تۇرۇۋېرىپ  بىكارچىلىقتا دۇدىجۇۋ ئوينىدۇق ، بىر ۋاقىتتا فۇ جۇرىن بىزنى چاقىرىپ قالدى ، نىمە ئىش باردۇ دەپ كېرسەك ، ئەتە بىزنى شىياپەي قىلىپ تۈۋەنگە چۈشۈردىكەن ، ۋۇسەنفىينىمۇ بېرىدىكەن ..............
   ياش بالىلار : بۈگۈن بىر نەچچە ئاغىنلەر بىلەن ۋاڭباغا كېردۇق،جىزا يوق، تۇرۇپ كەتتۇق ، ۋاڭگۈەندىن بىكار جىزا بارمۇ دىسەك يوق دىدى . دىيەنناۋرىمغا شىيادۇ رۇنجەن قاچىلسام بولمىدى ، قايتىدىن زۇ شىتۇڭ قىلدىم . بىركەمدە باللا بىلەن بەنتاۋ ئىتىشىپ كەتتۇق ، مەن بى سى سەن ئەپ ۋەھشى ئىتىپتىمەن . باللا بىلەن بۈگۈن دىباغا كېردۇق . كېرىپ بىر خالتا بو مىخۇا ئاپتۇق ، بىر تىزا پىجۇ ، بىر توتۇلكا بەيجۇ ئىككى پوتۇلكا ۋسكىي سەت ئىچىپتۇق . ئالىم شۇۋ جىسىنى چىقىرىپ قىزلارغا تېلفۇن قىلغانتى ، ئۈچ قىز بالا كەلدى ، كاساپەتلەر ئالامەت چىرايلىكەن پىگور دىگەن ئالامەتكەن . ئىچىپ شالاقىشىپ كەتتۇق . بىر ۋاقىتتا چۇزۇ توساپ يولغا چىقسام تېجىڭلار تۇتۇۋالدى ، شىڭفىڭجىڭ تەكشۈرۈپ بۇپ قۇيۇپ بەردى ......

شوپۇرلار :   ماشىنا ھەيدەپ كىتىپ بارسام جوتۇڭ تۇتۇۋالدى . نىم بولدى دىسەم چاۋسۇ بوپ كىتىپتىمەن 200كوي فاكۈەن يىدىم . بىر چاغدا ماشىنام غار غار قىلىدۇ ، نىم بولدىكىن دىسەم پەچىگۈەندىن چاتاق چىقىپتۇ . ئۇنى ئوڭشاپ بولسام ، دىيەنپىڭدىغا توك ئۈتۈشمىدى ، لىخېچىنى دەسسەم  خوتقا چۈشمىدى . ئاچچىقىم كەپ داۋداڭغا چۈشۈرۈپ ماينى بەرسەم ماشىنام ئوڭشالدى . ماشىننىڭ شىڭ چى جىڭسى يوق ھەيدەۋېرىپتىمەن . ئەتە ماشىنىنى داشۇۋغا بېرىمەن ......
ئوقۇغۇچىلار : مەكتەپكە بارسام ، گۇاڭبۇدا بىر نەچچە بالىنى توڭ باۋ قىلدى . ئۇلارنىڭ ئىككىسىگە جىڭ گاۋ چۇفىڭ بېرىپ ، ئۈچىسىنى كەي چۇ قېپتۇ . بىرسىگە لۇۋشياۋ ساكەن بېرىپتۇ . شوجاڭ گۇاڭبودا تاماكا چەكسەڭلەر مەكتەپتىن چارە كۆرىمىز دەيدۇ . مۇئەللىم جاڭيى تەكشۈرۈپ بىر قىز بالىنى تىللاپ كەتتى . ماتىماتىكا مۇئەللىم بۇندىن بىجىبىر تۇتىسلەر دەيدۇ .......
پۇتبۇلچىلار : تۆنۈگۈن جاۋلەن چاقىرىپ ، شۈنلىيەننى ياخشى قىلمىساڭلار . مۇسابىقىدە ئۇتتۇرۇپ قۇيىمىز دىدى . بۈگۈن شاڭبەنچاڭ ئويناپ ، خۇاڭ پار يەپ كەتتىم . تاس قالدىم شىيابەنچىياڭدا خۇڭ پار يەپ كەتكىلى . بۈگۈن مەشىقتە ۋۇ گۇڭلى يۈگۈرەپتۇق . بەك يېرىپ كەتتىم . پۇتتا جان دىگەن نەرسە يوق . ئەمدى ئوف دىسەم ، جاۋلىيەن شۇاڭگاڭ تارتىسلەر دەپ ، شۇاڭگاڭ تارتقۇزدى ئاندىن يى بەي مىدا
بىسەيلەشتۇق . مەن دى يى مىڭ بۇپتىمەن . بىركەمدە يازىبۇ مىڭىپ ياتاققا كېرىپ كەتتۇق . .........

تىجارەتچىلەر : بۈگۈن پىفادىن مال ئەپ كىلىپ بەك يېرىپ كەتتىم ، دۇكاننى ئىچىپ ئەمدى ئولتۇرسام ، گۇڭ شاڭ جۈنىڭ كەپ ، ماللارنى تەكشۈردى ، ئۇلار كىتىپ ئۇف دەپ ئولتۇرسام ، جىفاجۈنىڭ كەپ يىڭ يە جاۋلىرىمنى تەكشۈرۈپ كەتتى . فاڭ يى زەننىڭ كەپ ئۇلار بىر تەكشۈردى . كەچتە دۇكاننى تاقاپ ماڭاي دىسەم ، داسانىڭ باۋ ۋىي كې سى كەپ ئەتە دۇكاننىڭ تازلىقىنى ياخشى قىلىڭلار ، ئەتە شەھەردىن باشلىقلار كىلىدىكەن . دەپ تاپىلاپ قويدى .
ئاشخانىدىكىلەر : ۋۇي باللار مىھمان جوخو . ھە ئاكا نىمە يەيسىز ، سەي دەننى ئاچسىڭىز ھەممە تاماق بار . گۈيرۇ لەغمەن ، سۈيرۇ لەغمەن ، دىڭ-دىڭ سومىيەن ، ئەرجەزى سۇمىيەن ، گەنبىيەن سومىيەن ،  خۈيمىيەن ، سۇ پولو ، گۆش پولۇ ، راڭپىزا ، لەڭپۇڭ ، كاۋاپ چاڭزا ئۈچەي ئشىقىلىپ ھەممىسى بار .......
قىزلار : ۋىيەي ئاداش ، بۈگۈن باغۇ ، شىياشىمىرغا بېرىپ ، ماۋۇ يۈرۈڭفۇنى ئاپتىمەن ، قانداق چىرايلىقمىكەن ئاندىن ماۋۇ يەندەينى ئاپتىمەن . ماۋۇ مايكا خىلى گاۋداڭ چىرايكەن دىياۋدارلىق ، كەيسەك سىللىق تۇردىكەن .
ئاداش تۈنۈگۈن ئالىيەگە تېلفۇن قىلاي دىسەم تېلفۇنۇم چەن فىي تىڭجى بوپ قاپتۇ ، خۇاباغا كېرىپ خۇافىي قاچىلاپ ، تېلفۇن قىلسام ، ئالىيەنىڭ تېلفۇنى ۋۇ فا جىيە تۇڭ ، ساڭا تېلفۇن قىلسام ، تۇڭ خۇا جۇڭ . ئەتتىگەن چىچىمنى كورىيە پوسۇندا جۈيۇۋ قىلدۇردۇم ، ماۋۇ كوخۇڭنىڭ رەڭگى بەك سەتكەن ، بىر خىل مەتو چىراي قىپ قويدى ............
مۇزىكانتلار : بۈگۈن دىيەنزىچىڭغا ئۇيغۇرچە پاس قاچىلدىم ، مىللىيچە جىيە زۇ دىگىنە ، گىتتاردا سولو ئوينىساڭ دەھشەت پەيزى . دىيەنزى گىتتاردا ئالامەت چىقىدۇ . بىر خىل چەتئەل نۇۋزەي پوسۇندا دىگىنە . كەي چاڭ قىلغاندا ئاۋۋال ئۇسۇلنى ، ئاندىن تاڭگۇ چالىمىز جۇمۇ . يىڭشاڭنىڭ زايىڭسى بەك يۇقۇركەن ، تاۋيىڭتەيدە تەڭشەيلى ............
   تورداشلار،قېرىنداشلار، يۇقارقى گەپلەردىن سىز نىمىنى ھېس قېلدىڭىز؟ بىلسەم بىلمىسەم يوقارقى بۇ گەپلەرنى ئوقۇپ كۈلگىڭىز كەلدى ياكى چوڭقۇر خىياللارغا بەنىت بولدىڭىز . جەمئىيتىمىزدە ، تۇرمۇشىمىزدا بۇنداق ئەبجەش گەپلەر ناھايتتى نۇرغۇن ھەتتا قۇلاق تۈۋىدىن كەتمەيدۇ. بىز بۇلارنى ئۆز تىلىمىزدا تەلەپپۇز قىلساق ، مەسىلەن بىڭ شىياڭنى توڭلاتقۇ دەپ ، سىزنىڭچە بۇ زاماننىڭ كەينىدە قالغانلىق ھىساپلىنامدۇ ؟ قېرىنداش مىللەتلەرنىڭ تىلىنى ئۈگىنىش ، شۇ مىللەتنى چۈشۈنۈش ئۇلار بىلەن پىكىر قىلىش ئارقىلىق دوستلۇقنى كۈچەيتىش ، ئۇلار بىلەن ئۆز-ئارا مەدەنىيەت ئۆتكۈزۈش بۇلىدۇ . لىكىن تىل ئۈگەندۇق دەپ ئۆز ئانا تىلىمىزنى تاشلىۋەتسەك بولمايدۇ . ئۇ نىمىلا بولمىسۇن بىزنىڭ ئانا تىلىمىز ، دىلىمىزغا سىڭگەن تىل . بىز قېرىنداش مىللەتنىڭ تىلىنىڭ چىرايلىق ئاڭلىندىغان ياكى دەبدەبىلىك ئاڭلىندىغان سۆزلىرىنى ئۆز ئانا تىلىمىزغا سىڭدۈرۋالساق سىزنىڭچە بىز بەك بىلىملىك ياكى ئوقۇمۇشلۇق كۆرۈنەمدۇق ؟ بۇنداق قىلىش ئەخمىقانىلىق ، بۇنداق سۆزلىسەك ، ئۆزىمىزنىڭ تىلىنى ئەبجەش قىلغاندىن سېرىت قېرىنداش مىللەتلەرنىڭمۇ تېلىنى بۇزغانلىق ياكى ئۇنىڭغا ھۆرمەتسىزلىك بىلدۈرگەنلىك ، ھەممىمىز ئويلىنىپ باقايلى . بۇنداق ئەبجەش سۆزلىسەك سىزنىڭچە توغرا بۇلامدۇ ؟ بىزنىڭ ئىدارىدىكى خىزمەتچىمىز ئەنۋەر بىر قېتىم ئىشخانىدا ئىش بىجىرگۈچىلەرگە ئەستايىدىل ، ئالدىرماي ساپ تىلىمىز بىلەن مۇلازىتمىزنى چۈشەندۈردى ، لىكىن سالىھ دىگەن خىزمەتچىدىشم بولسا ، ھىجيىپ قېشىمغا كەلدى مەن ئۇنى پول تېپىۋالدىمۇ نىمانداق خوشال بۇلۇپ كەتتى دەپ ئويلىسام ، ئۇ قېشىمغا كىلىپ : ئاداش سەت كۈلگۈم كىلىۋاتىدۇ . ئاۋۇ ئەنۋەرنىڭ گىپىنى ئاڭلاپ دىدى . مەن سالىھغا : نىمە بولدى ؟ ئۇ ساڭا بىرەر قىززىق لەتىپە ئېيتىپ بەردىما ؟ دىسەم ، ئۇ : ياق ئاداش ، ئانىڭ گىپىنىڭ بىرخىل سەھرا چىراي چىققىنىغا كۈلگۈم كىلىۋاتىدۇ . ئۆزىگە قارسا بىرخىل شەھەر سۈپەت گاۋداڭ بالا ، گىپى دەھشەت ....ھى ھى ھى . ئۇنىڭ بۇ گىپىنى ئاڭلاپ ئاچچىقىم دەررۇ كەلدى ھەم ئۇنىڭغا قاراپ : ئاداش .. سۆزلەۋاتقىنى ئۇيغۇرچىدۇ ؟ سەن سۆزلىيەلەمسەن ؟ قارا گپىنى، ئۈگەنسەك بۇلىدۇ شۇنداق سۆزلەشنى . سەن بىز سۆزلىيەلمەيمىز ، خەق سۆزلەپ قويسا زاڭلىق قىلامدۇق ؟ ئۇنىڭ سۆزلىگنى ئاتام زاماندىكى سۆز ئەمەس ھەققىي ئۇيغۇر تىلى جۇمۇ بىلىپ قوي ، ساڭا ئوخشاش بالاڭزا چەيدو پەدىسىدە گەپ قىلىسا بۇلىدىكەن ھە . دەپ ئۇنىڭغا كايىپ كەتتىم . ئۇ ماڭا قاراپ ھۇمىيىپ : ھوي دەپ قويسام سۆزلەپلا كەتتىڭغۇ، خو جۇمۇ . ماقا ئەمسە دەپ ، جالداقشىپ چىقىپ كەتتى . مەن ئۇنىڭغا غەزىپىم تولماستىن بەلكى مەندە بىرخىل ئوھسىنىش بولدى .
     قېرىنداشلار ، ئانا تىلنى قوغداش ، شالغۇت تىلنى تۈزۈش ھەممىمىزنىڭ مەسئولىيتى .  دىمەكچى بولغىنىم : گازىر گازىردەك بولسۇن، ئالما ئالمىدەك بولسۇن . (ئۇيغۇرتىلى ئۇيغۇرتىلىدەك بولسۇن، خەنزۇتىلى، خەنزۇتىلىدەك بولسۇن ، قازاق تىلى قازاق تىلىدەك بولسۇن . ئارلاشتۇرۇپ، سۆزلەش نۇمۇس بولسۇن ، بۇ كۆڭۈل غەشلىك مودىدىن قالسۇن . )

بىلىملىكنىڭ ، يېڭىلىقنىڭ ،  نوچىلىقنىڭ ،ئىپادىسى بوپ قالغىلى تۇردى .
بۇلۇپمۇ شەھەرلەردە بەكرەك كۆپ  .   يۇقۇرلاپ ئۇقىۋاتقانلارغۇ  بەزى ئۇيغۇرچە گەپنى ئۇنتۇلۇپ قالغىلى تۇردى .... تېلفۇن نۇمۇرنى  ئۇيغۇرچە ئاتاشنى بىلمىەيدىغانلار  ياكى قىينىلدىغانلار
خېلى كۆپ    .   راس دەيسىز  تېلفۇن نۇ مۇرنى ، ۋە شۇنىڭدەك بەزى سۆزلەرنى    ئۇيغۇرچە ئاتاش    مەتولۇقنىڭ  بەلگىسى دەپ چۈشىندىغان  بىلىملىكلەر . ئاز ئەمەس  .  
بارلىق يېڭىلىقلار  مەدەنيەتلەر شەھەردىن  يېزلارغا قاراپ تارقىىلدۇ  . بىلىم مەرپەتمۇ شۇنداق

بىلىملىكلەر شەھەرگە كۆپ جايلاشقان  . ئالى مەكتەپ ۋە شۇنىڭدەك  تۇلۇقسىزدىن يۇقىرى دەرجىلىك مەكتەپلەر  نىڭ ھەممىسى شەھەردە .  شۇ مەكتەپلەردە  .ئۇقىۋاتقانلارنىڭ
ھەممىسى دىگۈدەك  خەنسوچىنى  قېتىپ سۆزلەيدۇ ،    بۇنداق ئەھۋال شۇ ئالى مەلۇماتلىق
   بىلىملىك كىشلىرمىزدە  بەك ئېغىر  . شۇكىشلەر مىللەت  ئانا تىل كىملىك توغۇرلۇق بىلدىغانلىرى بىز سەھرالقىلاردىن يۇقىرراق دەپ ئويلايمەن    ھەي........................
  
بۇمۇ بىر مەدەنيەت  بۇتېخى باشلىنىشى  ئۇنىڭ تەرققى قىىلدىغان   يوقىلدىغان ۋاقىتلىرى بار  

ۋاڭباخانغا كىرىپ ۋاڭبا ئويناپ چىقىلى باللا  بۇ ھەقتە زىجى  ئويلانماي  باشقىلامۇ ئوي:qiray13لۇنۇپ باقسۇ ن قانداق دىدىم باللا...   بوممىسا ئۈزەم  ‹‹ زىجى ››  كىرىپ ئويناپ چىقىمەن  جۇمۇ بىكا .
ئۆز مىللىتى ئۈچۈن جېنىنى پىدا قىللايدىغان مەرت باھادىرلار چىقمىسا بۇنداق مىللەت تىزلا ھالەك بۇلىدۇ .
