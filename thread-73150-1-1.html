                                                                     كىپەن ئوغرىسى
                                                 ھەي،
                                                                جان بىقىشنىڭ كۇيىدا دۇستۇم،
                                                               كىشى نىمە قىلمايدۇ دەيسىز.
                                                               بەزىلەر ھالالدا ياشايدىكەن لىك،
                                                               مەن باشلىق،
                                                               بەزىلەر ھارام يەركەنمىز.
                                                              مەن ئىدىم بىر نەپەر كىپەن ئوغرىسى،
                                                             (لىكىن دەپ قويماڭ بىشىم ساق قالسۇن)
                                                             بۇۋامنىڭ ئەجرىنى خەقلەرگە ساتقان،
                                                            لىكىن ھەق مەندەككە رەھمەت قىلمىسۇن.

                                                             كىرگەنتىم بۇۋامنىڭ قەبرىسىگە مەن،
                                                             پۇرىدى جەسەتنىڭ قاڭسىق ھىدلىرى.
                                                            توۋا،
                                                           شۈكۈر دىدىم ئاللاغا شۇئان،
                                                          كۆرۈپ داڭ قالدىم بىر مۆجىزىنى،
                                                           جەسەتتىن تارايتتى سىسىق پۇراق لىك،
                                                          ئەتراپتا ئۈسۈپتۇ چىچەك_گۈللىرى،
                                                              
                                                          چىرايلىق چىچەكلەر كۆرۈنمەس لىكىن،
                                                         كەلدى كۆز ئالدىمغا ئاجايىپ روشەن.
                                                        بىر قىتىم ئۆلگەنتى ئۇلار ئۆمرىدە،
                                                        بوينىدا يوق ئىدى ئېغىر بىر كىشەن.
                                                        ئۇچىسىدا تۇراتتى پاكلىق سىمۋولى،
                                                        ئاقكۈڭۈل قەلىبتەك ئاپئاق بىر كىپەن.

                                                       چىرىپ كەتكەن جەسەتنىڭ ھىدى،
                                                       قىززىق ئىش،
                                                       پۇرىدى بۇرنۇمغا شۇنداق خۇش پۇراق.
                                                      ئۇ پۇراق نەلەردىن كەلگەن ئىكەنكىن؟
                                                      بىلمىدىم مەنبەسى يوقكەنغۇ بىراق.
                                                         
                                                       تۇيۇقسىز پۇرىدى مەندىنمۇ خۇش ھىد،
                                                      ۋە كىيىن ئۆزگەردى جەسەت ھىدىغا.
                                                      سەپكەنتىم ئۆيدە مەن خۇش پۇراق ئەتىر،
                                                      نىمىشقىدۇ؟
                                                      خۇش پۇراقلار كەتتى قاياققا؟
                                                      قورقتۇم.
                                                      لىكىن مەن تىرىك ئىدىم.
                                                      تۇيۇقسىز كەپ قالدى كاللامغا بىر ئوي،
                                                      مەن زادى قانچىلاپ ئۈلۈپ تىرىلدىم؟

                                                    ئۇ تىرىك ياتاتتى جىمجىت ئۇيقۇدا،
                                                    مەن ئۈلۈك تۇراتتىم تىپىرلاپ بۇندا.

                                                   مانا مەن قەبرىدىن كەڭ سرتقا چىقتىم،
                                                   مەۋجۇدات ھايات بىر قەبرىگە كىردىم.


                                                                                       _   ئىسمائىل ئابلىتىپ ئىنتىل
