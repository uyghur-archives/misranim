ئەسسالامۇ ئەلەيكۇم مىسرانىم تور بېكىتى مىسرانىم مۇنبىرىنىڭ ئەزىزانە ئىشتىراكچىلىرى...
مۇنبىرىمىز شىنجاڭ بويىنچە تۇنجى قەدەمدە يېڭى بىر مۇنبەر ھالىتىنى شەكىللەندۈرگەن قىسقىغىنە 2 ئاي ۋاقىت ئىچىدە نۇرغۇن تورداش دوستلارنىڭ قوللاپ قۇۋەتلىشى ھەم تەكلىپ پىكرى ئاستىدا مۇنبىرمىز ھەقىقىي مۇكەممەللىككە قاراپ تەرەققىي قىلماقتا، بۇ ھال بىزنى چەكسىز خوشاللىققا چۆمدۈرگەن بولسا ،يەنە بىر تەرەپتىن زىممىمىزدىكى مەسئۇلىيەتنىڭ تولىمۇ ئېغىرلىقى بىزنى تېخىمۇ مەسئۇلىيەتچان بولۇشقا ئۈندىمەكتە..
بۇنىڭ ئۈچۈن يالغۇز مەن بىر ئەزانىڭ كۈچىشى بىلەن ئىشلار ۋۇجۇدقا چىقمايدۇ، مۇنبىرىمىز ئۇيغۇر تور دۇنياسىدىكى چاقنىغان يۇلتۇز بولالمايدۇ ئەلۋەتتە.، بۇنىڭ ئۈچۈن مۇنبىرىمىزدىكى قەلىمى ئۆتكۈر پىكىرلىك كىشىلىرىمىزنىڭ تىما ئىنكاسىدىن ئايرىپ قارىيالمايمىز ئەلۋەتتە..

يېقىندىن بۇيانقى سەپەر ئىشلىرى ،شەخسىي بىر قىسىم ئىشلار تۈپەيلى ،مۇنبىرىمىزگە يوللانغان كۆپلىگەن تورداشلارنىڭ سۇئاللىرى جاۋابسىز قالدى، سىلەرگە دەل ۋاقىتىدا مەسھىلەت بېرەلمىگىنىمدىن تولىمۇ خىجىلمەن،

قارىسام مۇنبەر بىرخىل چېچىلاڭغۇ ھالەتتە،توغرا 5000 ئەزانىڭ 5000 خىلچە خۇيى بولىدۇ، بەزىلىرىمىز ئاتا ئانىمىزنىڭ ئەركىن بالىسى، ،بەزىلىرى شۇ بالىلارنى قاتارغا قوشقان ئاتا ئانىلار...

بۇ يەردە ئائىلىدىكى ،ياكى جەمئىيەتتىكى پوزىتسىيە ئۇسۇللارنى ئشقا سالساق بولمايدۇ، بۇ ھەممىمىز توردىن خوشاللىق ،بىلىم ئىزدەپ كىرىدىغان مەنىۋىي دۇنيادۇر..
بۇ يەردە نە رەنجىش،نە قېيىداش، نە پۇتاقچىلىق بىلەن ھېچقانداق ئىشنى قىلغىلى بولمايدۇ، ياكى سىزگە ياكى بىزگە قىلچىلىك پايدىسىمۇ بولمايدۇ،
ئىنسان شۇنداق بارىدا قەدرىگە يەتمەيدىغان. قولدىن كەتكەندىن كېيىن ئاھ ئۇردىغان ،كىيىن پۇشايمان قىلىدىغان مەخلۇق ئىكەن..

مۇنبىرىمىز ئىسسىق سوغۇقلارنى باشتىن كەچۈرۈپ ئەزا تەستىقلاش، ئىنكاس تەستىقلاشتىن تەستىقلىماسلىققا، دىنىي تىمىلارنى تەستىقلىماسلىقتىن قۇرئان ھەدىستىكى تىمىلارنى قۇلۇبلاپ  تەستىقلاشقا ئۆتتۇق ، بارا بارا قىسقا ئۇچۇر قاتارلىق ئالىي ھوقوقلارنىمۇ ئېچىۋىتىمىز ئەلۋەتتە...

بۇنىڭغىمۇ شۈكىرى قىلماي يەنە ئۇنىڭ بۇنىڭدىن پۇتاق تېپىپ باقىمىز، توغرا چۈشەنمەيمىز..
بىرى تىما يازسا دەرھال شۇنىڭغا قارشى تۇرۇپ ئىككىنجى قەلەم ئالماس قىلۋىتىمىز، سىزگە شۇنداق ئىشلارنى قىلىپ نېمە پايدا؟  كۆڭلىڭىزنى قانچىلىك شاد ئېتەلەيسىز؟

تېخنىكىلىق مەسىلىلەر بولسا قولۇمدىن كېلىشىچە كۈچەيمەن..
كۆڭۈل توردىشىمىز دېگەندەك تېخنىكىلىق مەسىللەر بولسا سەۋەبىنى چۈشەندۈرۈشكە تەييارمەن...
ئوخشاش ئىسىمنىڭ شەكللىنىپ قېلىش سەۋەبى
<نازى>   <نازى >   ئەمەلىيەتتە ئىسىم ئوخشاشتەك تۇرىدۇ..  بىراق ئىسىم ئارقىسىدا بىر بوشلۇق بارلىقىنى ھېچكىم ھېس قىلمايمىز. شۇڭا بوشلۇق قويۇپ ئىسىم تىزىملىتىشنى چەكلىدۇق، بىراق ئۇيغۇرچە پىنيىنچە ئىسىم تىزىملىتىشقا چەك قويالمايمىز.ئىسىملار ھامان چەكلىكلىككە ئىگە، بۇنىڭ ئۈچۈن مۇنبەردە بىزگە تونۇشلۇق ئەزالىرىمىز چوقۇم باش سۈرەت بىرخىللىقىنى قوغدىسا بولىدۇ، يەنى ئىشلەتكەن باش سۈرەتنى باشتىن ئاخىرى بىرلا قىلىپ ئىشلىتىشكە ياكى كۆپرەك مەزگىل شۇ باش سۈرەتنى ئىشلىتىشكە ئادەتلىنشى كېرەك..


يەنە شۇنىڭغا ئوخشاش سۇئاللار بولسا تارتىنماي سورىغايسىلەر،،قەستەن ئۇرۇش چىقىرىپ ،بىرىنىڭ شەخسىي ئىززەت ھۆرمىتىگە تەسىرى يېتىدىغان تىما ئىنكاس يازغان ھەرقانداق كىشىگە ئاگاھلاندۇرۇش ،مۇنبەر تۇرمىسىگە قاماش ئۇسۇللىرى بويىچە چارە كۆرىمىز..

سىزنىڭ مۇشۇ مەۋھۇم دۇنيادىمۇ پېتىشماي ئۇنىڭدىن بۇ بۇنىڭدىن ئۇ قۇسۇر ئىزدەپ ئولتۇرشىڭىزنى خالىمايمەن.

مەۋھۇم دۇنيادا جىدەل ئۇرۇش چىقىرىپ ،تورداشلار ئىززەت ھۆرمىتىگە تەسىر يەتكۈزۈپ ھېچقانداق خوشاللىققا ئېرىشەلمەيسىز، قانداق گەپ سۆزىڭىز بولسا بۇنىڭدىن كىين ئۇدۇل مېنى ئىزدەڭ ،ماڭا دەڭ ،ئەزالاردا ھېچقانداق گۇناھ يوق، بىر بىر گەپتە ئېزىپ بولغۇچە ئۇنىڭغا نەشتىرىمىزنى سانجىشقا تەييار تۇرمايلى، مۇشۇ تور دۇنياسىدىمۇ باشقىلار ئىززەك ھۆرمىتىگە تېگىپ ،باشقىلارنى ھاقارەتلىسىڭىز  سىزنىڭ رىئال تۇرمۇشتكى ئىجتىمائىي مۇناسىۋىتىڭىزدىن گۇمانللانماي قالمايمىز ئەلۋەتتە....

ئەمسە ھەرقانداق پىكىر تەلەپ بولسا مۇشۇ تىمىغا قالدۇرغايسىز، مەن ئىنكاس سۈپىتىدە 24 سائەتكىچە سىزگە قانائەتلىنەرلىك جاۋاب بېرىشكە تەييار.. بۇنىڭدىن كىيىن ھەرقانداق مۇنبەر قائىدىسىنى ،مۇنبەر تەرتىپىنى بۇزۇشقا ئۇرۇنغانلارنىڭ ھەممىسىگە قاتتىق چارە قوللىنىشقا مەجبۇرمىز.

قېنى دوستلار، ئۇيغۇر تور تەرەققىياتى  ،شۇنداقلا مىسرانىم ئۈچۈن كۈچ چىقىرىپ بىرلىكتە ئالغا باسايلى . ئۆزىمىزنى ئۆزىمىز قوللايلى ...

ھەجەپ دانا پىكىر بوپتۇ. پىكىرىڭىزنى قوللايمەن.
مەنمۇ ئازىراق پىكىر بېرىپ قۇياي. ئىنكاس جەھەتتە : ياخشى يىزىپسىز ، ياخشىكەن  دەپ يېزىپ قويغان پۈچەك  ئىنكاس ئاپتۇماتىك ئىنكاس رايۇنىغا چىقمىغان بولسا بەك ياخشى بولاتتى.
باش سۆرەت جەھەتتە: مۇنبىرىمىزدىكى شاتلان ھەممەيلەنگە باش سۆرەت لاھىيلەپ بەرگەن . بولسا بارلىق ئەزالارنىڭ مۇنبەر باش سۆرىتى بىرلىككە كەلگەن بولسا.
تېما يوللاش جەھەتتە: مەنبەسى ئەسكەرتىلمىگەن ھەر قانداق ئەسەر تەستىقلانمىغان بولسا.

ئالىي ئەزالارنىڭ بارلىق ھۇقۇقىنى ئەسلىگە كەلتۈرۈپ بەرگەن بولساڭلار...
ھە راسىت ، مىسرانىم ئەپەندىم ئەتىگەننىڭ ياقى دىققەت قىلىۋاتىمەن .ئاخشام مۇنبەردە تۇرۇش ۋاقتى 636 سائەت ئىدى. بۈگۈن كەچكىچە مۇنبەردە يۈردۈم . مۇنبەردە تۇرۇش ۋاقتى يەنىلا 636 سائەت ئۆرلىمەپتۇ. سېستىمىدا ئازىراق خاتالىق بارمۇ -ياكى؟؟؟... بىر تەكشۈرۈپ باقسىڭىز بولغۇدەك.
ئەزالارغا

مىسرانىم مۇنبىرى ئېچىلغاندىن بۇيان  توردىكى مەنىۋىي تۇرمۇشىمىز بېيىپ ، بىزنىڭ ھېلىقى زىرىكىشلىك ، مۇنبەرسىز كۈنلىرمىز ئاياغلاشقان ئىدى. شۇڭا مىسرانىم مۇنبىرىنى كۆز قارچۇقىمىزدەك ئاسراپ ، ھەممەيلەننىڭ ئورتاق كۈچ چىقىرىشىنى ئۈمىد قىلىمەن. ئىسلامى ، دىنغا ئائىت تېمىلارنىڭ تەستىقلىنىپ روياپقا چىقىشى ئاسان ئەمەس . شۇڭا مۇشۇنداق تېمىلارغا ئىنكاس يازغاندا سەل دىققەت قىلىپ ، تېمىدىن چەتنەپ كەتمەسلىڭلارنى ئۈمىد قىلىمەن. بۇنداق تېمىغا چەك قۇيۇشنى توغرا چۈشەنگەن بولساڭلار. بولسا ساۋاپ ئىزدەپ ، باشقىلارنىڭ پايدىلىنىش ئۈچۈن يوللاشنى ئەڭ ئاساسىي مەقسەت قىلساقلا كۇپايە.
ئىنكاس يازغاندا باشقىلارنىڭ زىتىغا تىگىدىغان ئىنكاسلارنى قەتئىي يىزىشتىن ساقلىنىڭ. تېمىنى ئۇقۇماي ، ئىنكاسنى ئۇقۇپ ئىنكاس يېزىشتىن ساقلىنىڭ . پۈچەك ئىنكاس يېزىپ مىسرانىم مۇنبىرىنىڭ كۇرسىنى چۈشۈرمەڭ!
مىسرانىمنى ئاسراش ھەمىمىزنىڭ بۇرچى... شۇڭا ھەمىمىز مىسرانىم ئۈچۈن كۈچ چىقىرايلى! ئۇيغۇر تورچىلىقى ئۈچۈن ھەسسە قۇشايلى !!!

Re:قانداق پىكرىڭلار بولسا مەرھەمەت..

تەكلىۋېىمنى ئولتۇرغا قۇياي ئارتۇق كەتكەن بولسا ئىچىڭلارغا ئالماي چۈشەنچە بېرىڭلار ھە

1.باشقۇرغۇچىلارنى تۇلا ئالماشتۇرشتىن ساقلانسا
2 .بۇرندىن ھۆرمىتمىزگە سازاۋار بۇلۇپ كېلۋاتقان ئەزاللار نوختۇلۇق باشقۇرغۇچىلىققا قۇيۇلسا (نەدە بىر ئۈز ئىسمىنى جايلاپ يازالمايدىغان ،ياكى گەپنى بايقاپ قىلمايدىغان ،ئۈزۈمنىڭلا راس دەپ تۇرۋالدىغان ئەزالارنىڭ باشقۇرغۇچى بۇلۇپ قېلشىدىن ساقلانسا )
3 .باشقۇرغۇچىلارنىڭ ئىسملكى مۇنبەردە ئىلان قىلنىپ شۇ باشقۇرغۇچىلار ئىچىدە قەلەم قابليىتى يۇقۇرى باشقۇرغۇچى ،مۇنبەر قۇرغۇچى ،باشقۇرغۇچى ،ئەزاللارنى زىيارەت قىلىپ ئۇلارغا بولغان چۈشنىشمىزنى چوڭقۇرلاشتۇرسا
4 . تىما يولىغۇچىغا ئۇنى  يولا بۇنى يولا دەپ مۇقۇم قېلىپقا كېرۋېلىش شەرتىنى قويمىسا
5 .ئالىي ئەزاللارنىڭ ھوقۇقىنى ئەسلىگە كەلتۇرسە (يا تىمىىمىزنى ئۈزىمىز تەھىرليەلمىسەك ،بىر قۇرۇق قالپاق بۇلۇپ قالدى ھازىر ئالىي ئەزا دىگەن نام )
6 .ئەزاللارنىڭ قىزغىنلىقىغا سوغۇق سۇ سېپىدىغان ئاكتىپ پىكىر قىلمايدىغان باشقۇرغۇچىلار ئەمەلدىن قالدۇرلۇپ ئورنىغا مۇسىلمە ،توغرۇل ،ئارشاڭ ،ئەزىزان ،كۈڭۈل ،سۈزۈك ئايلار تاللاپ باشقۇرغۇچىلۇققا قۇيۇلسا
7 . باشقۇرغۇچىلار ئسىمىغا ئالاھدە ئىتبار بەرسە ئىنگلىسچە ،خەنزۇچە ئىسملىرنى تەھىرلەپ مۇنبەرداشلارنىڭ ئىسىدە قالدىغان ،ئالاھدە تەسىر قىلدىغان ،مەنسنى چوڭقۇر ئويغۇرچە ئىسملارنى قۇيۇپ ئەزاللارغا ئۈلگە بولسا بۇ ئۇيغۇرچە مۇنبەر بولغاندىن كىيىن
ھازىر ئىسىمگە كەلگىنى مۇشىلار ،قالغىنى سەلتۇرۇپ دەيمەن
سۈزۈلۈپ يۇرۇيدۇ سۈزۈك ھەر زامان تېرىكەن دېسىمۇ ئۆلۈك ئۇ ھامان چاپلىشىپ قان شۇراش ئادىتى ئۇنىڭ قانخورغا يوقتۇر ھىچ تىنىچ ماكان

مەنمۇ ئازىراق پىكىر بېرىپ قۇياي. ئىنكاس جەھەتتە : ياخشى يىزىپسىز ، ياخشىكەن  دەپ يېزىپ قويغان پۈچەك  ئىنكاس ئاپتۇماتىك ئىنكاس رايۇنىغا چىقمىغان بولسا بەك ياخشى بولاتتى.
باش سۆرەت جەھەتتە: مۇنبىرىمىزدىكى شاتلان ھەممەيلەنگە باش سۆرەت لاھىيلەپ بەرگەن . بولسا بارلىق ئەزالارنىڭ مۇنبەر باش سۆرىتى بىرلىككە كەلگەن بولسا.

تېمىغا كەلسەم مەن باشقا يەردە دەپ بولغان بۇيەردە دىسەم بولماس ،تۇرمىگە كىرىپ قالىدىغاندەك .خەيىر !
