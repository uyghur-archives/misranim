مىنىڭ باشقا پىكىرىم يوق پەقەت مۇشۇ مۇنبەر ئۈچۈن جان كۆيدۈرۈپ جاپالىق ئىشلەۋاتقان باشقۇرغۇچىلارنىڭ تېمىلارغا پات - پات كىرىپ ئىنكاس يېزىپ تەكلىپ پىكىر بىرىپ تۇرشىنى ئۈمۈت قىلىمەن ،بەزى باشقۇرغۇچىلارنىڭ مۇنبەردە قارسىنىمۇ كۆرگىلى بولمايدىكەن ،يەنە بەزىدە بەزى تېمىلارغا باشچىلاپ كىرىپ كىتىدىكەن ،بىلىمىز باشقۇرغۇچىلارمۇ ئارقا سەپتە ئالدىراش ئىشلەۋاتىدۇ ،بۇنى مۇنبەردىكى ھەمىمىز ئوبدان بىلىمىز ،شۇنداق بولسىمۇ مۇنبەرنىڭ تەرەقىياتى ئۈچۈن ۋاقىت ئىچىدىن ۋاقىت چىقىرىپ تېمىغا مۇناسىۋەتلىك پىكىرلەرنى بىرىپ ئىنكاس شەكىلدە يوللاپ تۇرسىكەن دەيمەن .
بۇ گەپلىرىمدىن باشقۇرغۇچىلار خاپا بۇلۇپ كەتمەڭلار ھە ... بۇ مىنىڭ پەقەت مۇنبەر ئۈچۈن ئويلىغانلىرىم خالاس .....

يەنە بىر گەپ بەزى يوللانغان تېمىلارنى ئۇقۇغان تورداشلارنىڭ تېمىنى قاراتملىق ھالدا چۈشىنىۋالماي ،كەڭ دائىرىدە چۈشۈنۈپ باشقىلارنىڭ زىتىغا تىگىدىغان ئىنكاسلارنى يازماسلىقىنى ئۈمۈت قىلىمەن .

ئاخىرىدا مۇنبەرنىڭ كەلگۈسى تەرەقىياتى ئۈچۈن بارلىق مۇنبەرداشلارنىڭ قولنى - قولغا تۇتۇپ بىرلىكتە ئالغا ئىلگىرلىشىگە تىلەكتاشمەن .
ئانا تىلغا كەلگەندە خىرىس ، ئۆزگە تىلغا بولغاندا ھىرىس  ، شۇنچە يىللىق ئىسىل ئەنئەنە ، داۋام ئىتىپ بولغاندا چىرىش . ئاھ ! تىلىم دەپ يۈرەك يىغلايدۇ !؟

ئەسسىلامۇ ئەلەيكۇم مىسران ئەپەندى ياخشۇمۇسىز ؟ پىقىر مۇشۇ پۇرسەتتە بىر نەچچە سۇئال سوراپ بىلمىگەننى بىلۋالسام .
1- بەزى تېمىلاردا نۇقسانلارنى ئۇچراتساق ، شۇ نۇقسانلارغا سۈكۈت قىلىش كىرەكمۇ ياكى پىكىر بىرىز كېرەكمۇ ؟ شۇ نۇقسانلارغا پىكىر بىرىپ ئىنكاس يازساق قۇسۇر ئىزدىگەنلىك بولامدۇ ؟
2- مۇنازىرە دېگەندە قارمۇ - قارشى پىكىر بولىدۇ . ئەمما بەزىلەر قارشى پىكىر بەرگۈچىلەرنى "دۈشمەن " كۆرۈپ ، دوست تارتىپ ، يۇرت تارتىپ ،يەنى بىرنىمە تارتىپ ، گورۈھلىشىپ تىللاپ ، ھاقارەت قىلىپ كىتىدىكەن . ھەق گەپ قىلغان ئادەم زەربىگە ئۇچراپ ئۇۋالچىلىققا قالدىكەن . مىسرانىم ھەق گەپ قىلغۇچىلارنى قوللامدۇ ياكى قوغلامدۇ ؟
3- بەزى ئىجادىي تېما يازغۇچى - شائىرلارنىڭ ئەسىرگە تەنقىت ياكى تەكلىپ بەرسە ، شۇ يازغۇچى - شائىر ،ئىجادىيتىنى تەنقىت - تەكلىپ بەرگۈچىگە قارىتىپ ، قالتىس ماھارەت بىلەن تېما يازدىكەن . بۇنى باشقۇرغۇچىلار تەستىق سىلىپ مۇنبەرگە چېقىرىۋېردىكەن . ئەجىبا ئاشۇ بىر قىسىم باشقۇرغۇچىلار ئىتتىپاقسىزلىق تىرىيدىغان مۇنداق تېمىلارنى نېمىشقا تەستىقلايدۇ ؟
4- پېقىر "قايسىنى خوتۇنلۇققا تاللايسىز ؟ نېمە ئۈچۈن ؟" دەپ بىر تېما يازسام شۇنداق ياخشى ئىنكاسلار چۈشكەن ئىدى . بىردەمدىن كېيىنلا بىر توردۇشىمىز ، مەزمۇنى مېنىڭ تېمام بىلەن ئوخشاش ، ئەمما شەكلىنى ئۆزگەرتىپلا " سىز قايسى ئەرگە ياتلىق بولىسىز ؟ نېمە ئۈچۈن ؟" دەپ بىر تېما يوللىسا ، دەرھال مېنىڭ تېمامنى ھىچقانداق بىر سەۋەبسىز ، ماڭىمۇ سەۋەبى توۈرلۇق ئۇچۇر قىلماي ئۆچىرۋېتىپتۇ ؟ بۇ قانداق ئىش ؟
5- تېمامنىڭ ئىز - دىرەكسىز يوقاپ كەتكىندىن ئازراق نارازى بولۇپ شۇ ھامان بىر تېما يوللىسام ، بىر باشقۇرغۇچى تېمىڭىز ئۆچۈرۈلدى دەپ ئۇچۇر بىرىپتۇ . ئەمما ھىچقانداق بىر چۈشەنچە بەرمەپتۇ . مىڭ مۇشەققەتتە ئالغان تۆھپىلىرىمنى بىكار قىلۋېتىپتۇ . بۇ باشقۇرغۇچى نېمە سەۋەپتىن شۇنداق قىلىدۇ ؟
