ئەسسالامۇ ئەلەيكۇم ،مىسرانىم مۇنبىرىدىكى ئەزىز قېرىنداشلار ياخشىمۇ سىلەر؟؟؟ مەن پەقەت بىر بىر نەرسە يېزىپ باقمىغان بۇ تۇنجى قىتىملىغى، تورداشلارنىڭ مىنى ئېلھاملاندۇرۇپ ،قۇللاپ قۇيىشىنى تەنقىت تەربىيەڭلنى، ئىنكاسىڭلارنى ئايىماي بىرىشڭلارنى تۆۋەنچىلىك بىلەن سورايمەن.  بۇ ئەمىليەتتە بولغان ئىش ئككى كۈننىڭ ئالدىدا يولدىشىم ھەيداۋاتقان ماشىنىسىنى ساتقۇسى كىلىپ ،بازارغا ساتقىلى ئەكىرىپ كەتكەن. ئۇ كەچقۇرۇن قايتىپ كەلدى. مەن ئۇنىڭغا- ھەي ماشىنىنى سېتىۋەتتىم دىگەنتىڭىز ، ھەيدەپ كەپسىزغۇ؟؟ -دەپ سورىدىم ئۇ ؛ - ئۆيگە كىرگەندە دەپ بىرەي دىدى.  كەينىدىن مەنمۇ كىردىمدە ئىككى ياستۇقنى تاشلاپ بەردىم. ئاندىن ئۇنىڭغا ؛ - ئۇسسۇزلۇق ئىچەمسىز ياكى تاماق يەمسىز دىسەم ئۇ ئىككىلىسىنى دىدى، مەن ئۇسسۇزلىقىنى ئىچېقىپ بىرىپ تاماقنى تەييارلاپ كىردىم. تاماقنى يەپ بولغاندىن كىيىن- ھە ئەمدى سۆزلەڭ نىمە ئىشلار بولدى . دىسەم ئۇ بىردەم تۇرۋىلىپ ؛ - ھەي تۇۋا بۇ جاھاننىڭ ئىشلىرىغا بىر نىمە دىمەك ، مەن بۇنداق ئادەمنى كۆرمەپتىكەنمەن. قارا مەن ماشىنىنى بازارغا ئەكىرىپ تۇراتىم، ئانچە مۇنچە خېرىدارلار ئايلىنىپ كورۋاتاتى. ئارىدىن بىر ياش يىگىت، چامامچە 36-37 ياشلار چامىسىدا بولسا كىرەك، ئاكا ماشىناڭنى راست ساتامسە ؟ ماشىناڭغۇ ماڭا يارىدى ، ھە باھاسىدا كىلىشەيلىمۇ دىدى . ئەتىراپتا قاراپ تۇرغانلار - ھە بىزمو بار ئارتۇق كىمى بولسا بىز كىلىشتۇرۇپ قويىمىز ھە ئۇكا دىتىڭنى دىگىنە ، سودىنى قىزىتقىلى تۇردى . ئۇنى دەپ بۇنى دەپ، ماشىنىنىڭ باھاسىمۇ پىشتى، ئۇ يىگىت ؟- پۇلنى مەن ئۆيدىن ئىلىپ بىرەي جۇرۈڭ ئاكا دىدى . بىز ئۇنىڭ ئۇيىگە كەلدۇق. بۇ پۇتكەن سودا بۇلغاندىن كىيىن پۇلنى ئىلىپ چىقىڭ دەپ بىز سىرتتا تۇرۇپ تۇردۇق.  ئالاھەزەلدىن كىيىن بۇ يىگىت ئايالى بىلەن ئۆيدىن چىقتى ، - بۇ ئايال نىمە قىلىدىغاندۇ ئەرلەرنىڭ ئىشىدا دەپ ئويلىدىم ئىچىمدە ، بىلسەم ئەسلى ئۆينىڭ چوڭى مۇشۇكەن ئەمەسمۇ ، ئايال ئېرىگە ؛ كىم سىنى ماشىنا ئال دىدى. خەقنىڭ ماشىنىسىنى ھەيدە مەن ساڭا 60 مىڭ كويلوق ماشىنا .ئىلىپ بىرەمتىم؟؟ دىدى تەرىنى سەت قىلىپ. ئەر بوينىنى قىسقان ھالدا خوتۇنىغا يالۋۇرۇپ ؛ - جىق ئەمەس 60 مىڭ كويغۇ ؟ دىۋىدى. خوتۇنى تىخىمۇ تەرىنى تۇرۇپ ؛ - خۇدايىمغا ئامانەت ئاناڭ ئوغۇل دەپ تۇغقان بولسا پۇل تېپىپ ئال تاپقان پۇلىڭنىڭ تايىنى يوق نىمە دەيسە ؟؟؟ .......................مەن خىزمەت قىلىپ تەستە تاپقان پۇل بۇ . بۇ ئۆيدە مىنىڭ دىگىنىم دىگەن . سەن نىمەڭگە مەندىن سورىماي ماشىنا بازىرىغا بىرىپ ماشىنا سودىلاشتىڭ ؟؟  مەن بۇ ئۆيگە ھەر ئايدا ئايلىق مائاشم بىلەن ئۆي تۇتىۋاتىمەن، سەنچۇ ؟؟؟ تاپقان پۇلۇڭنىڭ تايىنى يوق......................... بۇ ئايالنىڭ ئەپسانە بۇزۇق تىلىرى ئالدىدا بۇ بىچارە ئەر سالپىيىپ تۇراتى.  ئەي خوتۇن مۇشۇمۇ ئەر بولدىمۇ ؟؟؟ خوتۇنى ئايلىق ئالىدىغان ئەرلەرنىڭ ئىشى تەسكەن جۇمۇ؟؟ دىدى ھەسىرەت بىلەن. - ھەي ئاخىرى قانداق بولدى ؟؟؟ ماشىنىنى ساتالىدىڭىزما؟؟  -ئۇ ئەرنىڭ ئەھۋالى ئۇ تۇرسا قانداق ساتقىلى بولىدۇ ؟؟ قايتىپ كەلدىم . -ئىسىت ئۇ ئەرنىڭ ئەر بولۇپ قالغىنىغا !!!  مانا بۇ ئايلىقچى خوتۇن ، مانا بۇ پۇلى بار خوتۇن ، ھەي بۇ ئەرنىڭ غۇرۇرىنىڭ دەپسەندە بولغىنىغا ئەجەپ ئېچىندىم . پۇل ھەممىگە قادىرمۇ ؟
  ئەسكەرتىش:ئايلىقچى خوتۇنلار ئۆزەڭلىگە ئەپ كەتمەڭلار، ئايلىقچى خوتۇنلارنىڭ ، ھەممىسى  ئۇنداقمۇ ئەمەس .
