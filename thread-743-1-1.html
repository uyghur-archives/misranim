                       چاقچاق - چاقچاق ئۈچىنجى قىسىم

                            ئىزباسارغا

                    بولدى يازماڭ دەپلا سىز ،
                    ئاچىقلاپسىز ئىزباسار .
                    چېچىلاڭغۇ ئادەمنىڭ ،
                    ئىزىدىنمۇ كىم ماڭار .

                           كۆڭۈلگە

                    كۆڭۈللەرنىڭ خوشلىقى ،
                    كۈلكە چاقچاق قۇشاقتۇر.
                    يازاي سىزگە كۆڭۈل قىز ،
                    چاقچىقىمىز تۇشاقتۇر .

                          ئەمچىگە

                    قابائەتلىك چۈشنىمۇ ،
                   يېزىۋىتىپسىز ئۆزگىچە .
                   ئەمچى دوستۇم تۇرمىدە ،
                   ياتامسىزكىن ئۆلگۈچە .

                           تۇمارغا

                   يامان كۆزگە قالقاندۇر ،
                   ئېسىۋالغان تۇمارىم .
                   چاقچاق قىلسام دوسلارغا،
                   بېسىقمايدۇ خۇمارىم .

                          ئوت يۈرەككە

                    داڭلىق رولچى شاكامۇ ،
                    ئاتالغانتى ئوت يۈرەك.
                    دوراپ سىزمۇ ئۇلارنى ،
                    بۇلىۋالماڭ قاپ يۈرەك.

                          دەلىتكە

                   دوستىمىزنىڭ ئىسمىمۇ ،
                   بەك غەلىتە دەلتىكەن .
                   چۆپ تاشلىسا قويلىرىغا،
                   ئۆچىكىلىرى يەيدىكەن .

                          نازۇك قىزغا

                   مۇنبىرىمىزدە بىرسى بار ،
                   ئىسمى ئۇنىڭ نازۇك قىز .
                   ئاچىقلاشنىڭ دەردىدە ،
                   بۇلۇپ قالماڭ تاتۇق قىز .

                          كەلگۈسىگە

                    كەلگۈسىنىڭ ئالدىدا ،
                    بۇرچىمىزمۇ بەك ئېغىر .
                    ھاشارلارنىڭ دەستىدىن،
                    قول پۇتىمىز بەك يېغىر.

                          شىرنەگە

                    ھەرىلەرنىڭ ئۇزۇقى ،
                    ھەسەل بىلەن شىرنىدۇر.
                    يامغۇر ياغسا توختىماي،
                    ياتار جايى زەي يەردۇر .

                          يامغۇرلۇققا

                    سىرلىق كۈنلۈك ئاستىدا ،
                    مۇڭلىنىدۇ جۈپ گەۋدە .
                    بوران چىقىپ قالسا گەر،
                    پاناھ تىلەر قەيەردە .

                              كامرانغا

                     كامران دوستۇم مايكولغا ،
                     بەك چۇقىنىدىغان ئوخشايدۇ.
                     لىنكىن ئاكىسى كەينىدە ،
                     ھەسرەت بىلەن ياش تۈكىدۇ.

                              دىل سىرىگە

                      ئىسمىڭىزمۇ بەك تاتلىق ،
                      يارىشىپتۇ دىل سىرى .
                      سىز يولىغان تېمىنىڭ ،
                      بەك راۋاندۇر تىللىرى .

                             ياسىنجانغا

                      چاقچاق يېزىپ ھەممىگە ،
                      ئايرىپسىزغۇ ياسىنجان .
                      تاماق ھەرگىز ئوخشىماش ،
                      تازلانمىسا داش -قازان .

دوسلار قاملاشمىغان بولسا خاپا بولماڭلار ھە ؟ ئىغىر كەلسە كۈلۈپ قۈيۈپ ئۆتكۈزىۋىتىڭلار .ئىسمى چىقمىغانلارمۇ
ھەم شۇنداق ھەمىسىگە يېزىپ بولغىلى بولمايدىكەن .
                  مەنبە  ---توردىشىڭلار (تاڭنۇر )
مەندە كىچىكىنە يۈرەك بار ، لىكىن ئۇنىڭغا چەكسىز مىھىر مۇھەببەت قاچىلايمەن
