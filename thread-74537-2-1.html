ئۇ ئادەمنىڭ قەبرىسى ئۈرۈمچى ئۇلانبايدىكى ئىنقىلابىي قۇربانلار قەبرىستانلىقىدا،ئەپسۇس،ئۇ ئادەم بېيجىڭدا ئاغرىپ يېتىپ قالغان چاغلاردا،ئاپتۇنۇم رايۇنلۇق ئۇيغۇر تىبابەت دوختۇرخانىسىدىكى يۇقۇرى سەۋىيىلىك دوختۇرلار داۋالاش ئېلىپ بارغان،تۈگەپ كېتىشنىڭ ئالدىدا‹‹مېنى تۇغۇلغان يۇرتۇم ئاتۇشقا قۇيۇڭلار...››دەپ ۋەسىيەت قىپتىكەن.
لېكىن،دۆلەت ۋە ئاپتۇنۇم رايۇن رەھبەرلىرى ۋەسىيىتىگە قۇشۇلماي،تۈگەپ كېتىپ،4كۈندىن كېيىن،يەنى 28-چىسلاغۇ دەيمەن،نامىزىنى چۈشۈرۈپ بولغاندىن كېيىن،ئۈرۈمچى ئۇلانباي ئىنقىلابىي قۇربانلار قەبرىستانلىقىغا دەپنە قىلدى.
ئۇلانباي ئىنقىلابىي قۇربانلار قەبرىستانلىقىدا ئۈچ ۋىلايەت ئىنقىلابىغا قاتناشقان جەڭچىلەر،ئاپتۇنۇم رايۇن رەھبەرلىرىنىڭ كۆپ قىسمىنىڭ قەبرىسى شۇ يەردە.
