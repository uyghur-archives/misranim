ئەسسىلامۇ ئەلەيكۇم تويلاردا نىكاھ قىلغاندا قۇرئاندىكى شۇ نىكاھ توغرىسىدىكى ئايەتلەرنى ئۇقۇپ بىردەم پەتىۋا سۆزلەپ نىكاھ قىلىپ بۇنىڭ ئۈچۈن ھەق ئالىدىغۇ ، يەنى بەش ئالتە قارىيلار بىرگە بۇلۇپ تەكلىپكە ئاساسەن ئۈلۈپ كەتكەنلەرنىڭ قەۋرە بېشىدا، بالا قازالارنىڭ دەپى بۇلۇشى ئۈچۈن  ئويلەرگە بېرىپ قۇرئاننى ئۇقۇپ بىرىپ بۇنىڭ ئۈچۈن ھەر بىرى 50يۇۋەندىن تارتىپ 100يۇۋەنگىچە ھەق ئېلىۋاتمامدۇ ؟ ھەي بۇ ئىنساننىڭ بالا بۇ نەپسى بۇنىڭغا نىمە دىگۈلىك ئەمدى ؟ ھەي ئاللاھ نىڭ بۇ ھەق سۆزلىرنى ئەگەردە سېتىۋېلىشقا توغرا كەلسە نەچچە مىلىياتقا سېتىۋالساقمۇ يەنى ئارزان بۇلۇپ قالمامدۇ ؟ ھەي مۇنبەرداشلار ! ئېيتقانلىرىڭلار توغرا! شۇنداق بەزى كىشىلەر بۇ فانى دۇنيانىڭ ئەرزىمەس ئەرزان نەرسىلىرنى دەپ  باقى دۇنيانىڭ قىممەتلىك نەرسىلىردىن مەھرۇم قېلىۋاتىدۇ . قارىماققا بۇ ئىشلار بىزگە يۈزەكى تۇيۇلغىنى بىلەن . ئۇ ئۇنداق يۈزەكى ئىش ئەمەس . چۈنكى   ئىش دىگەن شۇنداق يۈزەكىلىكتىن ئاندىن چوڭقۇرلىشىدۇ . دىمەك بىز يۈزەكى دەپ قارىغان شۇ ئىشلار شۇنداق چوڭقۇرلاشقانسىرى ئاقىۋەت بىزنى دۇزاخقا سۆرىيدۇ . سىلەر ئاللاھ نىڭ تۈۋەندىكى تەۋسىيەسىنى ئوبدانراق پىكىر قىلىپ بېقىڭلار .
يەنى[اﷲ كىتابتا (يەنى تەۋراتتا) نازىل قىلغان نەرسىلەرنى يوشۇردىغان ۋە ئۇنى ئازغىنا پۇلغا ساتىدىغانلارنىڭ قارنىغا يېگىنى پەقەت ئوتتۇر(چۈنكى ئۇلارنىڭ يېگەن شۇ ھارام مېلى ئۇلارنى دوزاخقا ئىلىپ بارىدۇ)، قىيامەت كۈنى اﷲ ئۇلارغا (ئۇلارنى خوشال قىلىدىغان) سۆز قىلمايدۇ، ئۇلارنى گۇناھلىرىدىن) پاكلىمايدۇ، ئۇلار قاتتىق ئازابقا دۇچار بولىدۇ]2-سۈرە 174-ئايەت.
   دىمەك يۇقارقى ئايەتنىڭ مەزمۇنىدىنمۇ شۇ ھەقىيقەتلەر چىقىپ تۇرۇپتىكى ، جانابى ئاللاھ ئايەتلىرنى ئازغىنا پۇلغا (يەنى يۇقاردا ئەسكەرتىپ ئۆتكۈنىمىزدەك 50 كويغا بىرەر يۈز كويغا ساتقانلارنىڭ قانىغا يىگىننى پەقەت ئوتقا ئوخشۇتۇش بىلەن بىرگە  ،  شۇ يىگەن ھارام مېلى ئۇلارنى بۇ دۇنيادىن دۇزاخقا سۆرەيدىغانلىقىنى ، قىيامەت كۈنىمۇ ئاللاھ ئۇلارغا :ئۇلارخوش بولغۇدەك سۆزمۇ قىلماستىن ئۇلارنى دۇزاختىن مەڭگۈ چىقارمايدىغانلىقىنى بىلىۋالغىلى بۇلىدۇ . بۇ يەردىكى ئازان ساتماسلىق بولسا  ئاللاھ نىڭ قۇرئان كەرىمدىكى ئايەتلىرگە قارتىلغان بۇلۇپ . ئەگەردە ئاللاھ نىڭ يۇقارقىدەك ھەق تەۋسىيەلىرنى قۇبۇل قىلماستىن نەپسىگە ئەگىشىپ شۇنداق ساتقۇچىلار ئاللاھ نىڭ ۋەدىسكە ئاساسەن قىيامەت كۈنى يۇقارقى ئازاپلارغا دۇچار بۇلىدۇ خالاس . چۈنكى ئاللاھ نىڭ يۇقارقىدەك ۋەدىسى ھەقتۇركى ،ئاللاھ نىڭ ۋەدىسىدە ئاللاھ نىڭ قانۇن پىرىنسىپلىردىن ھالقىپ كەتكۈچىلەر ئۆزگۈرۈش كۆرمەيدۇ .

ۋەئەلەيكۇم ئەسسىلام خان ئاچا ! زۇۋان سۈرىشىم ھاجەتسىز بوپتىكەن ئىدۇق ، شۇ كىشىنىڭ چۈشەنچىسىنى قوللاپ . ئەمما سىز مېنىڭ ئاغزىمنى تاتىلاپ يامان قىلدىڭىز . مەن مۇشۇ خوتۇن خەقلەر بىلەن مۇنازىر قىلىشنى قەتئىيلا خۇش كۆرمەس بولۇپ قالدىم . سىزنىمۇ خېلى گاڭگۇڭ ئايالمىكىن دېسەم ، بەزىدە ئەدەمنى چىشلەپ قويىدىكەنسىز . ھېلىقى تېمىڭىزنىڭ ئەخلەتخانىغا تىقىلىپ كېتىشىنىڭ بىر سەۋەبىي مەن بولۇپ قاپتىمەن . مېنىڭ شۇ تېمىدا پەلسەپەنى تولا سۆزلەپ كەتكىنىم باشقۇرغۇچىلارنى سىزنىڭ تېمىڭىزنى ئۆچۈرىشىگە تۈرتكە بولغاندەك چۈشىنىپ قاپسىز . بەلكىم بۇ ئىشتا سىز قارىي باشقىچە گۇماندا بولۇپ قاپسىز دەيدىغانسىز . ئەمما سىزنىڭ ماڭا قىلغان ھېلىقى ئىنكاسىڭىزدىن شۇنداق چۈشەندىم . مېنىڭ چۈشەنچىم مەنچە تۇرىۋەرسۇن . سىز بىلەن بۇ توغرىسىدا توختالمايمەن . چۈنكى سىز خېلى مەغرۇر ئايال . مەن مەغرۇرلار بىلەن راستلا چىقىشالمايمەن ۋە پىكرىم بىردەك كەلمەيدۇ . بۇ توغرىسىدا سىزمۇ ئۆز چۈشەنچىڭىزدە قېلىڭ . چۈنكى مەن بۇ ئاللاھنىڭ ئايەتلىرىنى سېتىشنىڭ ماھىيەتلىرى ئۈستىدىكى چۈشەنچىدە سىزنى قايىل قىلالماسلىقىم مۇمكىن .  
رۇسۇللاھ بىزنى چاقىرغان يەرگە بېرىشنى ئېيتقان . ئۇ يەردە مەيلى قۇرئان ئوقۇيدىغان ئىش بارمۇ ياكى باشقا نەزىر لەر بارمۇ بارغاندىن كىيىن بىلىمىز . ئاللاھ بىزنى ئۆزىگە چوقۇنۇشقا بۇيرىغانلىقتىن ، شۇ خىل سۇرۇنغا داخىل بولۇشنى ئاللاھ بىزگە شۇنى ئىرادە قىلدى دەپ چۈشىنىمىز . كىيىن سۇرۇندىن قايتقاندا بەزى كىشىلەر شۇ سۇرۇندىكىلەرنىڭ ئالدىغا 50 سوم 100 سوم سېلىنغان كانۋېرت ۋە ياكى بىر پارچە جەينامازلارنى قويۇپ تۇرىۋالىدۇ . بۇ خىل پۇلنى ئالساقمۇ بولىدۇ ۋە ئالمىساقمۇ بولىۋېرىدۇ . مانداق دېسەم كۆزنەك قارىي بۇ پۇلغا بەكلا ئامراق ئوخشايدۇ  دەپ قالماڭلار ، مەن تېخى بۇ خىل پۇلنى گەرچا ئەل - ئاغىينىلەرنىڭ ئىشىدىمۇ ۋە ياكى تۇنىمايدىغان كىشىلەر تەرەپتىنمۇ ھېچ يەپ باقمىدىم . ئەمما مەن ھەر ئىشتا ئادىل تۇرىشىم ، سىزگە ئوخشاش ھىسىياتقا بېرىلمەسلىكىم كېرەك . بىر ئىشنى ئادىل بىر تەرەپ قىلىمەن دېگەن كىشى چوقۇمكى ھىسىياتتىن نېرى تۇرمىقى لازىمدۇر .
سىلەر كىشىلەرنىڭ پۇل قويغان ۋە ئالغانلىقىنى كۆرىسىلەر . شۇنىڭ بىلەن ئۇلارنىڭ ھەممىنى بىر تاياقتا قاغايسىلەر ۋە توختىماي قاۋايسىلەر . ئەمما بىزلەرنىڭ پەرۋايىمىزغا كەلمەيدۇ . چۈنكى ئاللاھنىڭ ئايەتلىرىنى سېتىش دېگەن ، ئاشكارا سودىلىشىشنى ۋە شۇ ئوقۇغان كىشىنىڭ سېتىشنى ئىختىيار قىلغانلىقى بىلەن بولىدۇ . باشقىلارنىڭ سېتىۋالدىم دەپ چۈشەنچىسى بىلەن بولمايدۇ . شۇڭا ئاللاھ ئايىتىدە ساتماڭلار دېدى . ھېچ سېتىۋېلىڭلار دىگىنى يوق . كىشىلەرنىڭ مەيلى خۇددى سېتىۋالغاندەك ھىسىيەتقە كېلىشى ئاللاھنىڭ ئايەتلىرىنى سېتىلغانلىقىنى بىلدۈرمەيدۇ . بۇ يەردىكى سېتىۋېلىش بولسا شۇ كىشىلەرنىڭ شۇ قويۇلغان پۇل شۇلارنىڭ ۋاقتىغا قويۇلغان پۇلدۇر . بۇ ۋاقىتقا قارىتا قويۇلغان پۇلنى ئالسا - ئالمىسا ئۇ ئۆزىنىڭ ئىشى . ئالسا ئۇنى ساتتى دېيىش سىلەرنىڭ ئىشىڭلار .
ئەمەلىيەتتە بىزلەرنىڭ شۇ سۇرۇندا ئوقۇغان بىرەر ئايەتنىڭ ساۋابىنى سېتىۋېلىش ئەمەس ھەتتە بىزلەرنىڭ شۇ سۇرۇنغا قارىتا ماڭغان ھەر بىر قەدىمىمىزنىڭ بىرىگە 100 سومدىن قويسىمۇ ، شۇ قەدىمىمىزدىكى ساۋابقا تەڭ بولالمايدۇ . بىز بۇلارنى ھېچ ساتقىنىمىز ھەم يوق . پەقەت كۆپىنچە كىشىلەر ۋاقىت تونىلىغىلا قاراپ ئېلىپ سالىدۇ . بۇ ئايەتلەرنى ساتقانلىق ئەمەس ۋاقىتنى تىگىشكەنلىك بولىدۇ خالاس !
ئەمما مەن مانچىلىك ئوقۇيمەن ، مۇنچىلىك پۇل بېرىسەن دەپ ياكى مۇنتىزىم بىر باھا قىلىۋېلىپ شۇنىڭغا چىقىشسا ئوقۇيدىغان بولمىسا رەنجىيدىغان كىشىلەر دىندىكىلەردىن ئەمەس . ئۇلارنى ئاللاھنىڭ ئايەتلىرىنى ساتتى دېمەيمىز ، دىننى سۈيئىستىمال قىلىۋېلىپ دىننى ئۇيۇن قىلىۋالدى . دىن ئارقىلىق جان باقماقنى تىلىۋالدى دەيمىز . بۇنىڭدەك كىشىلەر ھەم جەمىيىتىمىزدە نۇرغۇن .
يەنە بىرى قۇرئان ئايەتلىرى يېزىلغان رەسىملەرنى ، كىتابلارنى  ساتقاننى ئايەت ساتتى دېمەيمىز . بەلكى ۋاقىت ئاجىراتقان ئەجىرىنى تاپتى دەيمىز . بۇ خۇددى قۇرئان ئوقۇغان مۆئمىنلەرنىڭ ۋاقتىغا بېرىلگەندەك ئەجىردۇر خالاس . ئوقۇتقۇچى پۇل ئالسا ئۇنىمۇ ساتتى دېمەيمىز ، پەقەت ۋاقىتتىكى ئەجىرىنى ئالدى دەيمىز .
ئىمام - مەزىنلەر ھۆكۈمەتتىن پۇل ئېلىپ قالسا ، ساتتى دېمەيمىز ، بەلكى شۇلارنىڭ ئالدىدا ئۆزلىرىنىڭ غورورىنى ، ئەركىنلىكىنى ساتتى دەيمىز . بۇ يەردىكى بىز مىللەتتىكى ئۆز موللا - قارىيلىرىنى باقماي تاشلىۋېتىپ ، ئاللاھدىنلا تەلەپ قىلمامدۇ - دەپ ئۆزلىرى پىنسىيە مائاشى ، خىزمەت مائاشى ، باللىرىنىڭ مائاشىغا قارىتا يولىدا جان بېقىۋېلىپ ، باشقا ھېچ تاپاۋىتى يوق قارىيلارنىڭ ، موللىلارنىڭ مۇشۇ خىل ۋاقتىغا قارىتا ھېچ ياردەم قىلماي ‹‹ ئانداق قىلىپ كەتتىيەي ، مانداق قىلىپ كەتتىيەي ›› دەپ داۋراڭ سالىدىغانلارنىڭ ئاۋازى ھەردەم چوڭ چىقىۋاتىدۇ . توختا ، قارىيلىقنى ساقلاپ قالماق ئانچە ئاسان ئەمەس . بىز بۇلارنى دۇنيالىقىنى بىر تەرەپ قىلايلى . بۇلارنى دۇنيالىقنى زىننىتىدىن قالدۇرايلى . ئاندىن بۇلار بىز ئۈچۈن ئاللاھنىڭ كالامىنى قىرائەت ، تىلاۋەت قىلىش بىلەن خىزمەت قىلسۇن دېيىش يوق ، موچەن بەرمەي مەجبۇرى غورورىنى سۇندۇرۇشقا قويۇپ ھۆكۈمەتنىڭ ئايدا 2 - 3 يۈز سوم مائاشىنى ئېلىشقا قىستاپ قويۇپ ‹‹ بۇنداق ئادەمنىڭ ئارقىدا ناماز ئوقۇسا قۇبۇل بولمايدۇيەي ، بۇلار دىنىنى ساتقانلار يەي ›› دېيىشىپ غەۋغا قىلىشتىن باشقىنى بىلمەيدىغان ئەبگالار !
ئاللاھ ھەر كىشىگە ھېمكەت بىلەن رىزىق بەرگۈچىدۇر . كۆپىنچە جېدەلخورلار ئىلگىرىدىن يامان ئىشنى ، غورورى بىلەن ئويلاشمىغان ھاللاردا تاپاۋەتنى تېپىپ بولۇپ ، بالىلىرىنى خىزمەتكە ئۇرۇنلاشتۇرىۋېلىپ ئۇلارغا قاراشلىق بولىۋىش بىلەن موللا - بۈۋى بولىۋېلىپ ، جەمىيەتتە بۇرۇندىن ئاجىز ۋە ھاياتىنى قۇرئان يادلاش بىلەن ئۆتكۈزگەن ياخشى مۆئمىنلەرنىڭ ئارقىدىن ھەر نېمە دىيىشىپ ، ياكى بولدى سەن ئۇنداق قىلما ، بىز نەچچىمىز سېنىڭ ھۆددەڭدىن چىقايلى دىيىشەلمەي شۇ خىل ئۆزلىرى يامان كۆرىدىغان قىسمەتلەرگە قىستاپ قويۇپ غاجاشقا چۈشكەن .
ھېلىما ئۇيغۇر مىللىتىنىڭ مەۋقەسى كۈچلۈك ئىكەن . بولمىسا شۇلارنىڭ ھەممىسى دىگۈدەك پاساتقا قالاتتىكەن .
مەنمۇ كىيىنكى ھاياتىمدا سىلەرگە قاراشلىق بولمايمەن . ئاللاھ ماڭا يىتەرلىك نېمەت ئاتا قىلسا يەنە داۋاملىق ئىلىم تەلەپ قىلشتا ۋە ئىلىم تەشۋىق قىلىشتا بولىمەن . بولمىسا مەنمۇ جان باقمىسام بولمايدۇ . ھاياتىمنىڭ كىيىنلىكتىكى قىسمەن تەرەپلەرنى پۇل تېپىشقا قويمىسام بولمايدۇ . بولمىسا يەنىلا سىلەردەك كىشىلەر ئارقامدىن غاجاپ ‹‹ دۇنيالىقنى تەرك ئەتتىيەي ، سوپىي بولىۋالدىيەي ›› دېيىشىپ يەنىلا غاجاشقا ئۆتىدۇ . ئەگەر مەن ئاللاھقا ۋەدە بەرمىگەن بولسام بۇنچە دىشۋارچىلىقلارغا قالماس بولغىيتىم . ئەمما مېنىڭمۇ ۋەزىپەممۇ تۈگەيدۇ . مېنىڭ دىنىي ۋە ئىسلامدىن ئالغانلىرىم ئۆزەمگە يىتەرلىك بولدى . قالغىنى ئەمدى سىلەرنىڭ ئاللاھدىن نېمىلەرنى تەلەپ قىلىشىڭلارغا باغلىق بولدى .
سىلەرنى غاجاپ قويدۇممۇ ھە ؟ غاجىشىمنى مۇشۇ يەردە توختىتاي . بۇ سۆزلەرنىڭ ھەممىسىنى ئۆزىڭىزگىلا ئېلىۋالماڭ خان ئاچا . بۇرۇنمۇ بەزى تەرەپلىك مۇشۇ خۇسۇستا ئىنكاس بەرگەن ئىدىم . ئەمما سىز يەنىلا تۇنىشىڭىزدا تۇرىۋالدىڭىز . سىز يەنىلا مۇشۇ تۇنىشىڭيىزدا تۇرىلىۋەرسىڭىز كىيىنچە يەنە كۈچلۈكراقى ، قاتتىقراقى چىقىدۇ شۇ .

ئەسسىلامۇ ئەلەيكۇم :

ھۆرمەتلىك مۇنبەرداشلار! سىلەرنى پايدىنىلارمىكىن دىگەن مەخسەتتە (تۈۋەندە شۇڭا ئۇ تېخى مۇسۇلم ان بۇلالمىدى )دىگەن ئەسىرىمنى ئىنكاس قاتارىدا يوللاپ قويدۇم . پىكىر قىلىپ كۆرگەيسلەر . ئىسىڭلاردا بولسۇنكى ، ئەگەردە ئىنكاس قايتۇرماقچى بولساڭلا ھەقنى دەلىللەپ تۇرۇپ قايتۇراسىلەر . ھەقنى دەلىللىيەلمىسەڭلار قۇرۇق پەتىۋايىڭلار بىلەن نەپسى خايىشىڭلارغا ئەگىشىپ ۋاقىت ئىسىراپ قىلىپ ئىنكاس تازىمەن دەپ يۈرگۈچە سۈكۈتتە تۇرۋالغىنىڭلار ياخشى . بولمىسا مۇنبەرنىڭ تىنىچ ئىناق كەيپىياتىنى بۇزۇپ قۇيىسلەر .
ناھايىتى شەپقەتلىك ۋە مىھرىبان ئاللاھ نىڭ ئىسمى بىلەن باشلايمەن .
ئاللاھ دىن ئۇنىڭغا ۋەز نەسىھەت قىلنىپ يەنى [ئۇنىڭغا (ۋەز نەسىھەت قىلىنىپ يامان سۆز – ھەرىكەتلىرىڭدىن قايتىپ )ئاللاھ دىن قورققىن دىيىلسە، غۇرۇرى ئۇنىڭغا گۇناھ يۈكلەيدۇ (يەنى ھەقتىن تەكەببۇرلۇق بىلەن باش تارتىپ، پىتنە-پاساتتا تېخىمۇ ئەزۋەيلەيدۇ)، ئۇنىڭغا جەھەننەم يىتەرلىكتۇر، جەھەننەم ناھايىتى يامان جايدۇر]2-206-ئايەتتە دىيىلگەن؛
ئاللاھ نىڭ ھەق تەۋسىيەلىرگە قۇلاق سالماستىن ئۇنىڭ ئەكسىچە ئۇ ؛
            بىرىنجىدىن - ئاللاھ  ۋە ئاللاھ نىڭ پەيغەمبەرلىرى تەرىپىدىن ۋەز نەسىھەت قىلىنغا يەنى  يامان سۆز- ھەركەتلەردىن يېنىپ ، قورقۇشتا يالغۇز بىر ئاللاھ دىنلا قورقاشقا ، قۇرقۇشتا ئاللاھقا  باشقا مەخلۇقاتلارنى  شىرىك قىلىۋالماسلىققا تەۋسىيە قىلىنغان تۇرسا ؛ بىراق ئۇ ئۇنىڭ ئەكسىچە قورقۇشتا يالغۇز بىر ئاللاھ دىنلا  قورقماستىن  مەخلۇقاتلاردىنمۇ قورقۇپ ، قورقۇشتا ئاللاھقا ئىنسانلارنى شىرىك قىلىپ نەتىيجىدە ئۆزلىرى سەزمەستىنلا  ئاللاھ نىڭ بىرلىكىنى ۋە بەيغەمبىرىمىزنىڭ بەرھەقلىقىنى ئىنكار قىلدى .  شۇڭلاشقا ئۇ تېخى  مۇسۇلمان بۇلالمىدى .  
          ئىككىنجىدىن-  يەنى ئۇنىڭغا  ئاللاھ ۋە ئاللاھ نىڭ ئەلچىلىرى  تەرىپىدىن  چوڭ  بىلىپ  ئۇلۇغلاش ، ياخشى كۆرۈپ دۇئالىرنى قارىتىشتىن ئىبارەت بارلىق ئىبادەتلىرنى يالغۇز ئاللاھ قىلا قىلىشقا ، ئىبادەتلەرنى قارىتىشقا يالغۇز ئاللاھ دىن باشقا ھېچ قانداق مەبۇت بەرھەقنىڭ يوقلىقىغا تەۋسىيە قىلنىپ ؛ بۇ ھەقتە ئاللاھ مۇنداق دىگەن تۇرسا :   
يەنى[ئاسمانلاردىكى ۋە زېمىندىكى ئۇلۇغلۇق ئاللاھ غا خاستۇر، ئاللاھ غالىبتۇر، ھېكمەت بىلەن ئىش قىلغۇچىدۇر]45-37. دىگەن تۇرسا .  يەنى [ ھەقىقى قىلىنغان دوئا ئاللاھ غا  قارىتىلىدۇ، (يەنى ئىخلاس بىلەن قىلىنغان دوئانى ئاللاھ ئىجابەت قىلىدۇ). كۇففارلارنىڭ ئاللاھ نى قويۇپ، دوئالىرىنى قاراتقان مەبۇدلىرى كۇففارلارنىڭ ھىچقانداق دوئاسىنى ئىجابەت قىلمايدۇ. ئۇلار (يەنى كاپىرلار) شۇنداق بىر ئادەمگە ئوخشايدۇكى، ئۇ ئاغزىمىزغا سۇ چۈشسۇن دەپ (نېرىدىن) ئىككى ئالقىنىنى سۇغا قارىتىپ ئېچىپ تۇرىدۇ، (ئەمىلىيەتتە سۇئاڭلىمايدىغان، ھىس- تۇيغۇسىز نەرسە بولغاچقا) سۇ ھەرگىز ئۇنىڭ ئاغزىغا چۈشمەيدۇ، كاپىرلارنىڭ مەبۇدلىرىغا قىلغان دۇئاسى پۈتۈنلەي بىكاردۇر]13-سۈرە (14).
ئۇ ھەقىيقەتلەرنىڭ شۇنداق بۇلىشىغا قارىماي شۇنداق ئىبادەتلىرنى ئاللاھ غىلا قىلىشنىڭ ئورنىغا ئۇ بۇ سەۋەپلەر بىلەن ئىبادەتلىردە ئاللاھ غا غەيرىلەرنى شىرىك قىلىۋالدى . ئاللاھ نىڭ ھەق تەۋسىيەلىرگە يەنى[كاپىرلار مېنى قويۇپ بەندىلىرىمنى مەبۇد قىلىۋىلىشقا بولىدۇ دەپ ئويلامدۇ؟ بىز جەھەننەمنى كاپىرلارغا ھەقىقەتەن مەنزىلگاھ قىلىپ تەييارلىدۇق](18-سۈرە 102) دىگەن ھەق تەۋسىيەلىرگە قۇلاق سالماستىن؛ (مەبۇد قىلىش دىگەنلىك ياخشى كۆرىش دىگەنلىك بۇلۇپ ) ئۇ ئىبادەتلىرنى ئاللاھ دىن غەيرىلەرگە قارىتىپ ئاللاھ نى قۇيۇپ  بىرنى ئالىم ، ئۈلىمادى ، يەنى بىرنى ئاتا-ئانامدى دەپ ئىبادەتلىرنى ھېچ نەرسىنى يارىتالمايدىغان بەلكى ئۆزلىرى يارىتىلغان بەندىلەرگە  قارىتىپ ، ئىبادەتلىردە بەندىلەرنى ئاللاھ غا شىرىك قىلىۋالدى . شۇڭلاشقا ئۇ تېخى مۇسۇلمان بۇلالمىدى .
          ئۈچۈنجىدىن - ئاللاھ قۇرئان كەرىمدە  ئۇنىڭغا تەۋسىيە قىلىپ يەنى[مەن نازىل قىلغان تەۋراتنى تەستىق قبلغۇچى قۇرئانغا ئىمان كەلتۈرۈڭلار، قۇرئاننى بىرىنچى بولۇپ ئىنكار قىلغۇچى بولماڭلار، ئايەتلىرىمنى ئەرزان باھادا سېتىۋەتمەڭلار، ماڭىلا تەقۋادارلىق قىلىڭلار] 2-سۈرە 41-ئايەت. دىيىلسە ئۇ شۇنداق قىلىشنىڭ ئورنىغا نەپسى خاھىشلىرغا ئەگىشىپ ، ئاللاھ نازىل قىلغان قۇرئانغا ئىمان كەلدۈرۈپ ، تەقۋادارلىقنى ئاللاھ غىلا قىلماستىن ، ئۇنىڭ ئەكسىچە ئىنسانلار تەرىپىدىن يېزىلغان كىتاپلارغا ئىشىنىپ ، ئۆزلىرى سەزمەستىنلا قۇرئاندىكى ئاللاھ نىڭ  ھەق سۆزلىرنى ئىنكار قىلغانلىقتىن  شۇڭا تېخى ئۇ مۇسۇلمان  بۇلالمىدى .
ئۇنىڭ ئەڭ چوڭ كەتكۈزۈپ قويغان پاجەسى شۇكى ئۇ نەپسى خاھىشلىرىغا ئەگەشكەنلىكتىن قۇرئاندا  ئاللاھ نىڭ  يۇقارقىدەك يەنى[ كىتابتا (يەنى تەۋراتتا) نازىل قىلغان نەرسىلەرنى يوشۇردىغان ۋە ئۇنى ئازغىنا پۇلغا ساتىدىغانلارنىڭ قارنىغا يېگىنى پەقەت ئوتتۇر(چۈنكى ئۇلارنىڭ يېگەن شۇ ھارام مېلى ئۇلارنى دوزاخقا ئىلىپ بارىدۇ)، قىيامەت كۈنى ئاللاھ ئۇلارغا (ئۇلارنى خوشال قىلىدىغان) سۆز قىلمايدۇ، ئۇلارنى گۇناھلىرىدىن) پاكلىمايدۇ، ئۇلار قاتتىق ئازابقا دۇچار بولىدۇ](2-174) دىگەن ھەق سۆزلىرگە پەرۋا قىلماستىن  قۇرئان كەرىمنى  ئۆيلەردە  ۋە ياكى مازارلاردا  ئۇقۇپ قۇرئان كەرىمنى  تىجارەت قىلىشنىڭ ۋاستىسى ، جان بېقىشنىڭ قۇرالى قىلىپ ، ئاللاھ نىڭ ئايەتلىرنى ئەزان باھادا سېتىۋەتكەنلىكتىن شۇڭا ئۇ تېخى مۇسۇلمان بۇلالمىدى .  
دىمەك ئاللاھ  قۇرئاننى ئۈلۈكلەرگە  دىلنىڭ ئۈلۈكلىكىدە تىرىك تۇرۇپ ئۈلۈكلەرگە ئوخشاپ  قېلىۋاتقان  دىلى ئۈلۈك كاپىرلارغا  ئاڭلىتالمايسەن دىسە  ئۇ ئۇنىڭ ئەكسىچە، قۇرئان كەرىمنى  ئۈلۈپ كەتكەنلەر ئۈچۈن ئۇقۇيدىغان ، قۇرئان ئايەتلىرى بىلەن ئۈلۈكلەرگە دۇئالىرنى قارىتىدىغان بۇلىۋالدى.  شۇڭلاشقا ئۇ تېخى  مۇسۇلمان بۇلالمىدى .

مەنبە: قۇرئان كەرىمنى پىكىرقىلىپ تۇرۇپ ئۆز تەپەككۇرۇمغا تاينىپ يېزىلدى .

ناھايىتى شەپقەتلىك ۋە مىھرىبان ئاللاھ نىڭ ئىسمى بىلەن باشلايمەن .
ئاللاھ دىن ئۇنىڭغا ۋەز نەسىھەت قىلنىپ يەنى [ئۇنىڭغا (ۋەز نەسىھەت قىلىنىپ يامان سۆز – ھەرىكەتلىرىڭدىن قايتىپ )ئاللاھ دىن قورققىن دىيىلسە، غۇرۇرى ئۇنىڭغا گۇناھ يۈكلەيدۇ (يەنى ھەقتىن تەكەببۇرلۇق بىلەن باش تارتىپ، پىتنە-پاساتتا تېخىمۇ ئەزۋەيلەيدۇ)، ئۇنىڭغا جەھەننەم يىتەرلىكتۇر، جەھەننەم ناھايىتى يامان جايدۇر]2-206-ئايەتتە دىيىلگەن؛
ئاللاھ نىڭ ھەق تەۋسىيەلىرگە قۇلاق سالماستىن ئۇنىڭ ئەكسىچە ئۇ ؛
            بىرىنجىدىن - ئاللاھ  ۋە ئاللاھ نىڭ پەيغەمبەرلىرى تەرىپىدىن ۋەز نەسىھەت قىلىنغا يەنى  يامان سۆز- ھەركەتلەردىن يېنىپ ، قورقۇشتا يالغۇز بىر ئاللاھ دىنلا قورقاشقا ، قۇرقۇشتا ئاللاھقا  باشقا مەخلۇقاتلارنى  شىرىك قىلىۋالماسلىققا تەۋسىيە قىلىنغان تۇرسا ؛ بىراق ئۇ ئۇنىڭ ئەكسىچە قورقۇشتا يالغۇز بىر ئاللاھ دىنلا  قورقماستىن  مەخلۇقاتلاردىنمۇ قورقۇپ ، قورقۇشتا ئاللاھقا ئىنسانلارنى شىرىك قىلىپ نەتىجىدە ئۆزلىرى سەزمەستىنلا  ئاللاھ نىڭ بىرلىكىنى ۋە بەيغەمبىرىمىزنىڭ بەرھەقلىقىنى ئىنكار قىلدى .  شۇڭلاشقا ئۇ تېخى  مۇسۇلمان بۇلالمىدى .
كۈچى يەتمىگەن ئىشلارنى  قىلىشتىن قورقسا، زومىگەر ۋە زالىملاردىن مېلىغا، باللىرىغا خەيىم خەتەر يېتىش سەۋەبلىك زومىگەرلىكىدىن قورقۇپ ئۇلار بىلەن تەڭ تۇرمىسا، بىرەر مەخلۇق ياكى ھايۋانلاردىن قورقسا ئۇلارنى بىلىپ بىلمەي شېرىك قىلىۋېلىش سەۋەبلىك ئاللاھنىڭ بىرلىكىنى ۋە پەيغەمبىرىنى ئىنكار قىلغانلىق بولامدۇ؟ كىچىكىمدىن تا ھازىرغىچە پاقىنىڭ ئالدىمغا سەكرەپ كېلىشىدىن قورقاتتىم . ئايالىم بولسا دۆ، قوڭغۇزلاردىن قورقىدۇ. سىزنىڭ تەھلىلىڭىز بويىچە بىزمۇ پاقا بىلەن دۆ نى مەبۇت قىلىۋالدۇقمۇ؟
