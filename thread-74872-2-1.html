ئەسسىلامۇ ئەلەيكۇم مۇنبەرداشلارنىڭ ئېتقانلىرى توغرا بۇ تىما مۇشۇ مۇنبەرگە نىسىپىتەن كۆچۈرۈلمە . چۈنكى مەن بۇ مۇنبەردىنمۇ باشقا يەنى ئوننەچچە مۇنبەرگە كىرىدىغان بولغاچقا ، ئەگەردە بىر مۇنبەرگە تىما يازسام شۇ تىمىنى ئىمكانىيەتنىڭ بارىچە بىراقلا  مەن كىرىدىغان ھەممە مۇنبەرلەرگە بىرۋاتىمەن . بولمىسا بۇ ئۆزۈمنىڭ ئەسىرى دىمەك مەن  تىما يازسام ئۆزۈم قۇرئان كەرىمنى دەلىللەپ تۇرۇپ بىر ۋاستە مۇنبەر يۈزۈگىلا يازىمەن . بىراق ئەپسۇس كۆچۈرۈش جەريانىدا شۇنداق ئىملا خاتالىقى يۈز بىرىپ قالدى . ئېيىپكە بۇيرىمىغايسىلەر . چۈنكى مەن بىر 50ياشلارغا كىرىپ قالغان پىنسىيدىكى ئايال. ئاللاھ نىڭ مەدەتكار بۇلۇشى بىلەن مۇشۇنداق ھەقنى باشقىلارغا تەۋسىيە قىلىش مەخسىتىدە خەت ئۇرۇشنى ئۆزۈم يىڭىدا ئۈگەنگەنتىم . ئۇنىڭ ئۈستىگە تىل گىرامماتكىسىغا مەن دۈگۈدەك پىشىق ئەمەس ، ئاللاھ خالىسا بۇنى ئۆزگەرتىشكە  بۇنىڭدىن كىيىن تىرىشىمەن . ئەسىرىمنى  كىچىكىپ بولسىمۇ تەھىرلەپ قويدۇم مەزمۇنىنى پىكىر قىلىپ كۆرگەيسىلەر .

مەن خەنزۇ مەكتەپتە ئوقۇغان ، لىكىن ھازىر نامازنى تولل ...

بۇيەردە ھېچكىم ئەرەپكە ئىتقاد قىلمايدۇ ....   بىر اللەقا ئىتقاد قىلىدۇ .

بۇيەردە بىرەرسى  بىرەرسىنى نەچچە ۋاق ناماز ئۇقۇيسىز دەپ سورىغىنى يوق .
ئۇقۇسىڭىز دەرىجىسىنى ئۆزىڭىز كۆرىسىز ، ھەم ناماز ئۇقۇشتىن باشقىمۇ مانا  يۇقىردىكى  دوستىمىزغا ئوخشاش باشقىلارنىمۇ ياخشىلىققا ئۈندىشىمىز ، كىرەكلىكىنى مەندىن ياخشىراق بىلەرسىز  ھەقىچان .

                     بۇ يازمىدا بىرەرسى ئۇيغۇر تىلىنى بۇزۇپ  بىر نەرسە  يازمىغان بۇ پەقەت خەتنىڭ نورمال كۆرۈنمىگىنى سىستىمىنىڭ ئىشى ...... بىرەر سى مەجبۇرى بۇزۇپ  چېچېپ كەتكىنى يوق .
