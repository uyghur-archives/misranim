تېما كىشىنى قايىل قىلغۇدەك دەلىل-ئىسپاتلار يوق ھالدا بەك يۈزەكى بولۇپ قاپتۇ. مەركىزىي ئىدىيەگە مۇناسىۋەتلىك مۇھىم سۆزلەر بىر-بىرسىگە قارمۇ-قارشىغۇ....
ئاپتور  باشتا ‹ئىسلام پەن تەتقىقاتقا توسقۇنلۇق قىلمايدۇ.بۇ بىر ھەقىقەت ،شەك-شۆبىھسىزكى ئۈزەممۇ بۇنىڭغا مۇتلەق ئىشىنمەن› دەپتۇ. ئاخىرىدا ‹ئەمما دىيارىمىزدىكى «ئىسلام»چۈشەنچىسى  ئۇيغۇرلارنىڭ پەن تەتقىقات ئىشلىرغا ئېغىر توسقۇنلۇق قىلماقتا.› دېگەن سۆزىگە ئەمەلىي پاكىت يوق.
دىيارىمىزنىڭ ۋە مىللىتىمىزنىڭ تېخنىكا سەۋىيىسىنىڭ ھازىرقى ئەھۋالى ۋە تەرەققىياتى توغرىسىدا تەتقىقات ئېلىپ بېرىشنى ھەممىمىز قارشى ئالىمىز. لېكىن بۇ تېمىدا بۇنىڭ سەۋەبىنى ئىسلامغا باغلىۋاپتۇيۇ، ئەمەلىي پاكىت ئوتتۇرىغا قويۇلمىغان.
ئاپتور ‹    ئەمما ئارىمىزدىكى بىرقىسىم ئادەملىرمىز ئاپپاق خوجا دەۋىردىن قېلىپ قالغان دىنغا قارغۇلارچە ئىشنىش ،دوگماتىزىملىق قىلىش ۋە ياكى ئەسلىگە ئەقىل كۈزى بىلەن ۋارىسلىق قىلمايدىغان سۇپىۋىزىملىق ئىدىينى ھازىرمۇ بار كۈچى بىلەن تەشۋىق قىلىپ ئىتقادىمىزغا تىغ تەككۈزۈشكە بولمايدۇ دەپ ئۈزىمىزدىكى بار مەسىللەرنى تەتقىق قىلىشقا ،ئىلمى باھا بىرىشكە توسقۇنلۇق قىلماقتا.ھەدىسە ھەدىسلەردىن نەقىل كەلتۈرۈپ بۇ ھەقتە ئىلمى ئىجادى ئەمگەكلەرنى قىلغانلارنى جەدىت كۇففۇرلۇققا چىقىرۋەتمەكتە.  › دەپ يازغاندەك بۇنداق ئەھۋاللارمۇ يوق ئەمەس، بۇنى ھەممىمىز بىلىمىز، بۇ ئىنتايىن ئازساندىكى كىشىلەرنىڭ كۆز-قارىشىغا ۋەكىللىك قىلىدۇ.
بۇرۇن قانداق بىلمىدىم، مەن ھازىرغا قەدەر دىيارىمىزنىڭ نۇرغۇن يەرلىرىگە باردىم، جۈمە كۈنلىرى شۇ يەرنىڭ مەسچىتلىرىدە جۈمە نامىزىنى ئوقۇدۇم، كۆپ ساندىكى ھەتتا يېزىلاردىكى مەسچىتلەردىمۇ كۆپ ساندىكى ئىلىم ئەھلىلىرىمىز  پەننى ئىلىمنى چوقۇم ئىگەللەش كېرەكلىكىنى خەلقىمىزگە يەتكۈزۈۋاتىدۇ.
يەنە بىر نۇقتىدىن پارتىيە ۋە ھۆكۈمەتمۇ دىننىي ئالىملاردىن مۇشۇ جەھەتتە تەشۋىق قىلىشنى ھەرۋاقىت تەلەپ قىلىۋاتىدۇ، جۈمە كۈنلىرى قۇرامىغا يەتمىگەن مەكتەپ باللىرىمۇ مەسجىتكە كىرەلمەيدۇ.    مانا مۇشۇنداق شارائىتتا بىز ياشاۋاتقىلى خېلى زامان بولدى. بىز ئۇيغۇرلارنىڭ پەننى بىلىم ئۆگىنىش مۇھىتىمىزنى تەرەققىي تاپقان ئەللەرگە سېلىشتۇرالمىساقمۇ، لېكىن خېلى خېلى كۆپ دۆلەتلەردىن ياخشى .  ئاپتور ئۆزى ئېيتقاندەك رايونىمىزدىمۇ كۆپ ساندىكىلىرىمىز دىننىڭ پەننى ئۆگىنىشكە قارشى تۇرمايدىغانلىقىنى بىلىمىز.
ئۇنداقتا پەن تېخنىكا جەھەتتتىكى بۇ خىل ناچار ئەھۋالىمىزنى ئاپتوردەك  مەلۇم ئىنتايىن ئازساندىكى كىشىلەرنىڭ كۆز قارىشىنى ئومۇمىي دىنغا ئىشىنىدىغان ئاممىنىڭ كۆز قارىشىغا ۋەكىل قىلىپ  ئېلىپ،  قارىغۇلارچە ئاتالمىش ‹تەتقىقات› ماقالىسى يېزىپ، دىنغا ئارتىپ قويساقلا بولامدۇ؟
بۇنىڭ ھەقىقىي باشقا سەۋەپلىرىنىمۇ تەتقىق قىلىپ باقايلى!  
ھەربىر تىنىقىڭنى ئەھمىيەتسىز ئۆتكۈزمە!

ئىسلام ئۇيغۇرلارنىڭ  پەن-تەتقىقات ئىشلىرىغا تودقۇنلۇق قىلىدمۇ؟قىلمىدى چۈنكى ئۇيغۇرلاردا ئىسلام دىنى ئۇنچىلىك كەڭ تەشۋىقاتتا ئەمەس ،ھەر كىم باللىرىنى پەننىي مەكتەپتە ئۇقۇتۇپ تاكى ئاي مەكتەپكە بارغۇچە جاپارغا چىداپ ئۇتۇۋاتىدۇ ئۇلارنڭ يۇلىنى ھەر قانداق ئىمام مۇئەزىن،دىگەندەك دىننىي يول باشچىلار توسىغىنى يوق ھەم ئۇنداق بىلىم ئالسا ياكى پەن تەتقىقات بىلەن شۇغۇللانسا كافىر ياكى مۇناپىق دىگەن پەتىۋانى بەرمىدى، بۇ ئۇچۇقتىن ئۇچۇق بەىزلەرگە ۋە ئىتقاتقا تۆھمەت قىلغانلىق،توردا مۇنبەردە ئانچە مۇنچە دىننىي تيىمىلاردا،بۇلۇپمۇ ئىران ئىراق ئافغانىستان ،ئامرىكا ئىسرائىلىيە دىگەندەك تىمىلار يوللانسا قۇللاش ۋە قوللىماسلىق تالاش تارتىشى بار ،لىكىن ئىلىم پەننىي چەكلەيدىغان، سۇپىزىمنى تەرغىپ قىلىدىغان،ئاپپا  غۇجا ئىدىيىسىنى تەشۋىق قىلىدىغان بۇنداق خۇراپىي سۆزلەر مەۋجۇت ئەمەس،بۇنى يىزىش ئارقىلىق ئوتتتۇرغا زىدىيەت سىلىش ئۇيى بارمۇ ئۇخمىدىم ئىش قىلىپ تىما يازغۇچى كۈڭلىڭىزگە كەلمىسۇن بۇلۇپمۇ ئۇيغۇر جەمىيىتىدە بۇنداق ئاشكارە ياكى يۇشۇرۇن بولسۇن بۇنداق تۇسالغۇ مەۋجۇت ئەمەس چۈنكى99% باللار پەننىي مەكتەپتە پارتىيىنىڭ تەلىمىنى قۇبۇل قىلىم بىلىم ئىگىلىۋاتىدۇ ،شۇنداق بولغانكەن سىزنىڭ بۇنداق دىيىشىڭىزنىڭ ئاساسى بارمۇ؟
