بۇنداق نومۇر قوشۇپ بېرىش ياكى نومۇر چېكىنى تۆۋەنلىتىپ بېرىش ئىشكىردىكى ئالى مەكتەپلەردە تېخىمۇ ئېغىر.  بولۇپمۇ ئېمتىھانلاردا ئالغان نومۇرىنى يىلتىزدىن چىقىرىپ 10 غا  كۆپەيتىدۇ. . مەسىلەن:  36 نى يىلتىزدىن چىقارسا 6     .6*10= 60. شۇڭا شۇنچە يوقۇرى نومۇر بىلەن كەلگەن نى نى ياخشى بالىلارنىڭمۇ  مەنزىلى ئاران36دىن يوقۇرى بولۇپ قالدى.  ئالدىنقى قېتىم چوڭ سىنىپتا خەنلەر بىلەن ئوخشاش ئىمتىھان بېرەپ 90دىن يوقۇرى ئالغۇدەك بېرەپ چىقسام ئاران 73 قۇيۇپ قويۇپتۇ. ئاچچىغىمدا چىقىپ مۇئەللمنى ئىزدىدىم. ئۇنۇڭغا ئېمتىھاننى قايتا تەكشۈر. مەن 90 دىن يوقۇرى ئالاتتىم دېسەم ياراتمىغاندەك كۈلۈپ قويۇپ تۇرىدۇ. سىنىڭ ئادەتتىكى ئۆگۈنۈش ئەھۋالىڭ(平时成绩)  ياخشى ئەمەسكەن دەيدۇ.تىزىملىكلىرىنى قايتا تەكشۈرتۈپ ،  پۇتاق تاپالمىغاندا مۇناسىۋەتلىك كىشىلەرنى ئىزدەيمەن دەپ قوقاق سېلىپ يۈرۈپ ئاخىرى ئۆزگەرتتىم. يوقۇرى يىللىقتىكىلەر مەكتەپتە شىنجاڭلىق باللانىڭ ئىناۋېتىنى قويماپتىكەن . كاساپىتىگە بىز كېتىپ بارىمىز ....
     قايسى كۈنى تېلفۇنۇمنى ئۇيغۇرچىلاشتۇرۇپ بۇلۇپ بىر نەچچە ئۇيغۇرقىز ساۋاقدىشىمغا ھاياجان
بىلەن كۆرسەتسەم ئېلىپ كۈرەپ بېقىپ تېلفۇننى ئۇيغۇرچە قىلىپمۇ ئىشمۇ دەيدۇ.شۇنداق ئېسىل تۇرمامدۇ دېسەم،<< خەنزۇلار كۆرسە زاڭلىق قىلمامدۇ ، ماۋۇ تېخى خەنزۇچىنى بىلمەيدىغان ئوخشايدۇ ،نىمدا ئىرماش چىرماش خەت بۇ دەپ>> دەپ كۈلۈشۈپ كەتتى .  ئاچچىغىم غۇژژىدە توتۇپ سەت تىللىۋېتىپتىمەن. ئۆز ئانا تىل _ يېزىقىدىن نۇمۇس قىلغىنىغا  قارىغاندا بىزدە كىملىك تۇيغۇسى تۆۋەنلەپ كەتكەن ئوخشايدۇ!!
        يەنە تېخى ماۋۇ خەنلەر بىزنى سىلەرمۇ دۆلەتلىك بىرتۇتاش ئالى مەكتەپ ئىمتىھانىغا قاتناشقانمۇ؟ ئاڭلىساق  50نومۇر قوشۇپ بېرىدىكەن. بىزمۇ شىنجاڭلىق بولساق بىزگىمۇ قوشۇپ بېرەمدۇ دەپ سورايدۇ. شۇنچىلىك   سەمىمىيلىك بىلەن چۈشەندۈرۈپ كېتىمەن. مىللى ئوقۇغۇچىلار دېسە مەيلىتى ،لېكىن تېخى (民族特招生) دەپ تەنتەربىيەدە ئالاھىدىلىكى بار ،بىۋاستە قۇبۇل قىلىنغان ئوقۇغۇچىلار بىلەن بىر ئورۇنغا قۇيۇۋالىدۇ ، مۇشۇنداق گەپنى ئاڭلىسام ۋىجدانىم قايناپ كېتىدىكەن .  مۇشۇ ئازسانلىق مىللەتلەرگە بولغان ئىمتىھان تۈزۈمىنى ئەمدى ئىسلاھ قىلىدىغان يەرگە يەتتى جۇمۇ لېكىن .
