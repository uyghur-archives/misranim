قومۇلدا ئۈتكۈزۈلگەن نۇرۇز سەنئەت كىچىلىكىدىكى شىڭشڭشا ناملىق ناخشا مۇزىكىنىڭ كىلىپ چىقىش تارىغى توغۇرسىدا چۈشەنچە
كەڭ تورداشلارنىڭ مەلۇم چۈشەنچە ھاسىل قىلىشىغا ياردىمى بولىشى ئۈچۈن سۇندۇم ،  


1907-يىلى قومۇل دېھقانلىرى شاھ مەخسوتنىڭ زۇلىمىغا قارشى قوزغىلاڭ كۈتىرىدۇ ، بۇ قوزغىلاڭغا ( ھازىرقى)   
شەھەرئىچى يېزا سۇقۇق مەھەلىسىدىكى تورپاقلار جەمەتىدىن كۈپ كىشى باشلامچىلىق قىلغاچقا تارىختا تورپاقلار قوزغىلىڭى دەپ ئاتىلىدۇ ، ، بىراق قوزغىلاڭ چىڭ خانلىقنىڭ بارىكۈلدە تۇرۇشلۇق ئەمەلدارى يى شىفۇنىڭ قولى قانلىق باستۇرۇلۇپ قوزغىلاڭ يىتەكچىلىرى قەتلى قىلىندۇ ، بەزىلىرى زىندانغا تاشلىندۇ ، بەزىلىرى ياقا يۇرتلارغا سۈرگۈن قىلىندۇ ، شۇ قېتىمقى قوزغىلاڭ قاتناشقۇچىلىرىدىن راھەتباغ شامالچارباغ مەھەلىسىدىن چىققان يارتا ئابدۇلباقى ئاتىسى يارتا خوجىئەھمەتكە ئەگىشىپ قوزغىلاڭغا قاتنىشىدۇ ، تورپاقلار قوزغىلىڭى باستۇرۇلغاندىن كىيىن يارتا غوجىئەمەت تۇرپانغا سۈرگۈن قىلىندۇ ، يارتا ئابدۇلباقى بولسا تۈگىچى بۇلاق بىلەن  شىڭشىڭشانىڭ ئارلىقىدا  3يىل ھارۋا ھەيدەشكە سۈرگۈن قىلىندۇ ، شڭىشڭشادا ۋاڭنىڭ 400ئاتلىق ئەسكىرىي بولۇپ ئۇلارغا كىرەك بولغان يىمەك ، كىيم ، ئوت بوغۇز قاتارلىق نەرسىلەر 30ھارۋىدا توشىلىدۇ ، ئابدۇلباقى 3يىل ئەمەس بەلكى تاكى شاھ مەخسوت ئۈلگەنگە قەدەر ھارۋاكەشلىك قىلىدۇ ، ئۇنىڭ تۈگىچى بۇلاقتىن قومۇل تەرەپكە بىر قەدەم ئاياغ بېسىشىغىمۇ يول قويۇلمايدۇ ، بۇ جەر ياندا قارا قىش ، تومۇزنىڭ ئىسسىقىدا ، بىر تال گىياھ ئۈنمەيدىغان قاقاس سايدا ئابدۇلباقىنىڭ تاتمىغان جاپاسى قالمايدۇ .
چىدۇغۇسىز غەزەپلەنگەن ئۈزىنىڭ دەرت ئەلىمىنى ، غەزەپ نەپرىتنى ئىچىگە سىغدۇرالمىغان ئابدۇلباقى قەپەزدىكى بۇلبۇلدەك سايراپ شىڭشىڭشا دىگەن ناخشىنى ئىجاد قىلىپ ئادالەتسىز زامانغا بولغان نەپرىتىنى ئىپادىلەيدۇ ، پەقەت 1911-يىلى قومۇلدا تۈمۈر خەلپە قوزغىلىڭى پارتىلىغاندا پۇرسەتتىن پايدىنلىنىپ قومۇلغا قايتىپ كىلىىدۇ ۋە 80نەچچەيىل ئۈمۈر كۈرۈپ ۋاپات بولىدۇ ، لېكىن ئۇنىڭ ئىجاد قىلغان ناخشا قوشاقلىرىنى قومۇل خەلقى قانلىق تارىخنىڭ گۇۋاھچىسى سۈپىتىدە ھازىرغىچە ئېيتىپ كەلمەكتە ، ئازاتلىقتىن كىيىن بۇ ناخشا سەھنىلەشتۈرۈلدى ،  پىشقەدەم ، قومۇل مۇقاملىرىنىڭ پىشۋالىرىدىن بىرى بولغان مەرھوم نەمەت بوسۇقنىڭ ( لەيلىغان) ئورۇنلىشىدا ناخشا قىلىپ ئورۇنلاندى ، كىيىن بۇ ناخشا ئاپتۇنۇم  رايون، مەركەز سەھنىلەردە ئېيتىلىپ ئاممىدا چوڭقۇر تەسىر پەيدا قىلدى ، ھازىر بۇ ناخشا دۇئا مۇقامىنڭ 10-چۈشۈرگىسىدە ئېيتلماقتا ،

                         ئىككى تاغنىڭ ئاراسى
                          قومۇل يولىغا قاراپ،
                            ئاقتى كۈزنىڭ قاراسى

                      شاھمەخسوتنىڭ زۇلمىتى
                      شىڭشىشانى كۈرسەتتى
                       ئىككى تال قامچا بېرىپ
                     دىڭگىرمانى ھەيدەتتى
   
                  شىڭشىڭشاغا ماڭغاندا ،
                  ئەل يۇرتۇمدىن ئايرىلدىم .
                  ماڭماي دىسەم قورقىمەن ،
                 خۇدايىمغا يىغلىدىم  .

               شىڭشىشا ئۇزۇن گوبى ،
               تاغ ئىچىدىن يول ئاچقان .
                 گاڭنىڭ ئالۋان دەستىدىن
            ھارۋىنى تاشلاپ قاچقان ،
      
                 شىڭشىڭشانىڭ يولىدا ،
                گوبى بىلەن تاغلار بار .
                بۇ يولدا ئازاپ چەكتۇق .
              دىلدا  ئاچچىق داغلار بار .

              ئۈچ يىل ھەيدەپ ھارۋىنى ،
              ئىككى قامچە چۈشۈردۈم
              ئەلتۈمۈردە ياتقاندا
             ئاتا ئانامنى چۈش كۈردۈم

             شىڭشىڭشاغا ماڭغاندا ،
             يولدا ئۈلدى ئاتلىرىم .
            ئاياغىمغا تاش پاتتى ،
           كۈككە يەتتى ئاھلىرىم .
            
            ئات ھارۋىلارنى بىز ھەيدەپ ،
               كۈشەيگە باردۇ ق كەچ بولدى .
           ئاچچىقسۇنى بىز ئىچىپ .
          تاڭ ئاتقۇزماق تەس بولدى ،
            
         كۈشەينىڭ سۈي ئاچچىق ،
          مەنزىلى ئاندىن قاتتىق .
           گەنچەيلەرنى ھەيدەتكەن ،
           شاڭيو دىگەن مۇناپىق ،

          شىڭشىڭشانىڭ ئۈزىدە
           ئىسسىق كاڭلاردا
          شاڭيولار ئۈزى يېتىپ
           بىچارىلەر تالادا
  
         شاھ مەخسوتنىڭ دەۋرىدە
            ھالىمىز ئېغىر بولدى
          كەيجۇلارنى كۆتۈرۈپ
          دۈمبىمىز يېغىر بولدى
      
          ئىگىز تاغنىڭ باغرىدا
          ئاقتى يامغۇر تامچىسى
            ئاتا ئانىدىن ئايرىغان
           زالىملارنىڭ قامچىسى

          قومۇل دىگەن شەر ئىكەن
