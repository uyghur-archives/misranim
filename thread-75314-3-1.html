مانا مۇشۇنداق«داۋراڭ»سالغاچقا بۇ تەتقىقاتقا قاتنىشىدىغان ئادەملەر كۈپەيدى. بەزىلىرى تەجىرىبە ئىشلىدى. يىقىندا بەزىلىرىنىڭ بۇ نەزەريىنىڭ ماتىماتىكىلىق مۇدىلىنى دەسلەپكى قەدەمدە ئىشلەپ چىققانلىغىدىن خەۋەر تاپتۇق.ئۇنىمۇ توردا ئېلان قىلىپ «داۋراڭ»سالىمىز.مەقسەت كەسىپ بىلىدىغانلارنىڭ قىممەتلىك پىكىر-تەكلىپلىرىنى ئاڭلاش. كەمچىلىكلىرىنى تولۇقلاش.پىشىپ يىتىلگەن ۋە مەسىلە يوق دەپ قارالغاندىن كىيىن ئاندىن تەرجىمە قىلىپ مەملىكەت ئىچى ،سىرتىدا ئىلان قىلىش.بۇ نۇرغۇن ئادەملەر ھەمكارلاشقاندىلا غەلبىلىك تاماملىغىلى بولىدىغان بىر ئەمگەك.شۇڭا بۇنىڭدىن كىيىن«قاسىم سىدىق» ياكى«توقماق»دىگەنلەربىلەن خۇشى يوقلار بۇ تېمىلارغا كىرىپ ۋاقتىنى زايا قىلمىغاي.تېمىغا ئىلمى جەھەتتىن تەنقىت بەرمەكچى بولسا بۇمۇ كاتتا ئىش،دەلىل-ئىسپاتلار بىلەن تەنقىت قىلغاي.بىز«داۋراڭ»غا مۇھتاج،داۋراڭ سالمىغاچقا 20يىلدىن ئارتۇق ۋاقىت بىھۇدە ئۈتۈپ كەتتى،بىز 20نەچچە يىل كىچىكتوق.ئەمدى يەنە كىچىككىمىز يوق.بىزدە ئۈزىمىزگە بولغان يۈكسەك ئىشەنىچ بار،شۇڭا بىز قورىقماي داۋراڭ سالالايمىز.بىز شەكىل نەزەريىسىنى ئىلمى تەنقىت ئارقىلىق ئاغدۇرىۋىتەلەيدىغان ھەرقانداق بىر ئادەمنى قىزغىن قارشى ئالىمىز.شەكىل نەزەريىسىنى ئاساسسىز سىستماقچى بولغان ھەرقانداق كىشىنى قىلچە مەنسىتمەيمىز!. بۇ يازمىنى ئاخىرىدا   توقماق تەھرىرلىگەن. ۋاقتى  2012-4-1 03:19 PM

بىڭسىڭىز بولسا داڭلىق فىزىكلار ئارىسىدا دارۋاڭ قىلىڭ، مۇنبەردىكىلەرنىڭ كاللىسىنى قايمۇقتۇرماڭ، مىسرانىم سىزنىڭ نەزىرىيىڭىزنى تەشۋىق قىلىدىغان قورال بولۇپ قالسا بولمايدۇ.
ئەگەر قولۇمغا قەلەم ئېلىپ قالسام سىزگە شۇنداق بىر ھۇجۇم قىلىمەنكى، ئاخىرىنى تەسەۋۋۇر قىلىشقا ئۆزەممۇ ئامالسىز.
ئامال بار كۆپرەك ئۆگىنىڭ بولامدۇ؟
================
نەپرەت تورداش، ئېنكاسڭىزنىڭ ئاخىرىنى كۆرۈپ يۇقارقى تېتىقسىز پاراڭلارنى سىزنىڭ قىلغانلىغىڭىزنى بىلدىم.
شەكىل نەزەرىيەسى ھەققىدە بولۇنغان ئەڭ بېمەنە  ئاساسسىز  گەپ -سۆزلەر ، پىتنە-ئېغۋالار، تەلۋىلەرچە ھاقارەت -تەھتىدلەر، فېزىكا تەجىربىخانىلىرىدا ئەمەس، ئۇيغۇر مۇنبەرلىرىدە بولۇندى. ئىشەنمىسىڭىز سۈرۈشتۈرۈپ كۆرۈڭ. سىزچە ئۇيغۇر مۇنبەرلىرى  نېمىنىڭ قورالى بولۇپ  يەنە نېمىلەرنى قىلىش كېرەك ..؟!  كاللىسى ئەزەلدىن قايمۇققان، ئەقلى-پىكرى نورمالسىز بىر توپ جىدەلخورلار، ھەر يەردە ئاغزىغا كەلگەننى دەپ، سىزدەك بالىلارنى قايمۇقتۇرغاچ يۇقارقىدەك سۆزلەرنى قىلىۋاتىسىز بىلىۋىلىڭ..!!
قولىدا تۆمۈرنىڭ سۇنىقى يوق بالىغا ھۇجۇم قىلىشقا تەمشەلگەن ئاۋۇ ئەسكەرگە قەۋەتلا زوقلىنىدىغاندەك تۇرىسىز. قاسىمكام ۋە ئۇنىڭ قوللىغۇچىلىرىنىڭ ئېغىر -بېسىق، ئەپۇچانلىغىنى كۆرۈپ تايىنىڭىز يېتىپ قېلىۋاتامدا..؟ ئەڭ ياخشىسى ئۇكام، سۆزۈمنى ئاڭلاپ، ئاخىرىنى تەسەۋۋۇر قىلالمىغان ھۇجۇمنى ھەرگىز قىلماڭ. .!!

بۇ تىمىلار بەك قايتا يوللىنىپ كەتتى ، ئاز ئەمما ساز بولسۇن . سىلەر بۇ تىئورمىلارنىڭ ماتىماتىكىلىق مودىلىنى تۇرغۇزۇشتا قانداق پەرەزلەرنى قىلىۋاتىسىلەر ، ئىششلىرىڭلار قانداق بولىۋاتىدۇ ، نەدىن نەگىچە؟  بىرەر فورمىلا  تېخى يىراق گەپ . شۇ تەجرىبىنى ھەر خىل قىلىپ ئىشلاۋاتامسىلەر ، تەجرىبىدە ھەر خىل سانلىق قىممەت تەكرارلانغاندا ، ئاندىن ماتىماتىكىغا سالسا ئاندىن فورمىلا پەيدا بولىدۇ .  شۇلار قىنى ؟؟؟؟؟؟؟؟؟؟؟؟؟؟؟؟  ئاندىن ماۋۇ يەر مۇنداق بولىۋاتىدۇ ، پىكىر قىلىشىپ بىرىڭلار ؟ دىسەڭلار مۇنداق بولسا بولامىكىن دەپ پىكىر قىلمامدۇق! نىمە ھاياجانلاندۇرۇش بۇ؟ لىق نەزىرىيەلەر بىلەن توشقىزىۋىتىپ توغرىما دىسەڭلار ، بىز ماتىماتىكا ئارقىلىق خاتالىقىنى چۈشەندۈرۈپ بىرەمدۇق؟  مۇشۇندا قىلىۋەرسەڭلار ‹‹شەكىل نەزەرىسىگە›› نى چۈشەنمىسمۇ قارشى تۇرىدىغانلار چىقمامدۇ؟ سىلەر ئۆزەڭلارنى خەقنىڭ ئورنىغا قويۇپ ئويلاپ باقالمامسىلەر؟؟؟؟؟؟؟؟؟؟؟؟    قىنى ئىچىڭلاردا بىرەرسى قانداق ئۇرۇنۇپ باقتى ؟ بىزنىڭ بىلمەكچى بولغىنىمىز دەل شۇ ، بىشىنى بولسىمۇ كۆرسەتسەڭلار ، بىز پىكىر قاتناشتۇرۇپ باقمامدۇق؟ بۇنداق بولىۋەرسە بۇ نەزەرىيەلەڭلار قوللاڭلار دىگەن خەلىقتىن ئايرىلىپ قالمامدۇ؟  

  يەنە بىر گەپ قىلاي سىلەر بۇ نەزەرىيەنىڭ بىرەر تىئورمىسىنى  ئىسپاتلاش ئۈچۈن قىلغان پەرىزىڭلار؟ ياكى ئويلىغىنىڭلار بولسا تىما قىلىپ يوللىشىڭلارنى ئۈمىد قىلىمەن ! شۇچاغلاردا باشقىلار بىلەن ھەمكارلىشالايسىلەر.
بىر مىللەت ئۇرۇشتا ئەمەس ،بەلكى سۈكۈتتە يوقىلىدۇ .

esyankar75  ئەپەندىم،تەجىربىنى ئىككى خىل لايىھە بۇيىچە قايتا ئىشلەشكە مەلۈم ئالى مەكتەپ تەييارلىق قىلىۋاتىدۇ.مەنچە بۇ تەجىربىنى باشقا ئورۇنلارمۇ قايتا ئىشلەپ باقسا بۇلاتتى.مەن سىزنىڭ ئۈتكەنكى پىكرىڭىزنى (ھەر خىل تىزلىكلەر بۇيىچە سىناپ بېقىش تەكلىۋىڭىزنى)يەتكۈزۈپ قويدۇم. بۇ قىتىمقى تەجىرىبدىن كىيىن ئىشلەپ كۈرمەكچى.
ماتىماتىكىلىق مۇدىلىنىڭ دەسلەپكى ئارگىنالى پۈتۈپتۆ.ئاڭلىشىمچە ئىككى پارچە ماقالىكەن. يوللاپ بەرمەكچى،توردا ئېلان قىلىپ پىكر ئالماقچىمىز.
سىز خالىسىڭىز چ چ نومىرىڭىزنى قالدۇرۇڭ. تەجىربە ئىشلىگۈچىلەر ۋە ماتىماتكىلىق مودىلىنى ئىشلەۋاتقانلار بىلەن بىۋاستە ئالاقىلىشىپ باقسىڭىزمۇ بولىدۇ.مەن ئالاقىلىشىش ئادىرىسى بىلەن تەمىنلەيمەن.
