  ئەسسىلامۇ ئەلەيكۇم ئابلىكىم ئاكا! بۈگۈن مىسرانىم تورىدىن ، سىز بىلەن ئۆتكۈزگەن زىيارەت خاتىرىسىنى چەكسىز ھاياجانلىق تۇيغۇسى ئىچىدە كۆرۈپ چىقتىم . ئاپىرىن ئاكا ! ئۇلۇغ ئاللا ئىشلىرىڭىزنى ئوڭۇشلۇق قىلسۇن ، مىللەت ئۈچۈن كۆيىۋاتقان يۈرىكىڭىزگە ئىنساپ ئىمان ئاتا قىلسۇن ، ئائىلىڭىزگە خاتىرجەملىك بەرسۇن <<ئامىن>>
   ئەسلى مەخسەتكە كەلسەم ، ئابلىكىم ئاكا سىزنىڭ ئوچۇق مىجەز ، توغرا سۆزلۈك ئادەم ئىكەنلىكىڭىز مىنىلا ئەمەس بەلكىم بۇ زىيارەت خاتىرىسىنى كۆرگەن ھەممە ئادەمنى قايىل قىلغان بۇلىشى مۇمكىن . رەڭلىك سورۇن توغرىلىق سۆزلىگەن سۆزىڭىز مىنى بەكلا تەسىرلەندۈردى . چۈنكى تېخى مىللىتىمىز بېشىدا كۆتۈرۈپ چوڭ بىلگەن سەنئەتكارلىرىمىزدىن بىرەرسىمۇ ، سىزدەك ھەقىقى رەۋىشتە ئوچۇقتىن - ئوچۇق جەمىيىتىمىزدىكى مۇشۇنداق بۇزۇقچىلىقلار توغرىلىق توختىلىپ ، مۇشۇنداق بۇزۇقچىلىقلارغا ئۆزى <<مىللىتىمىز چوڭلىرى>>نىڭمۇ يۈرىكىنىڭ ئېچىشىدىغانلىقىنى ، مۇشۇنداق ناپاكلىقلارنى تۈزەلسىكەن دەپ ئويلايدىغانلىقلىرىنى دىيەلمىگەن. لىكىن سىز دىيەلىدىڭىز .
   لىكىن بۇ زىيارەت خاتىرىسىدە سىزدىن ئائىلە تەربىيەسى توغرىلىق سۇئال سۇرالمىغىنىدىن تولىمۇ كۆڭلۈم يېرىم بولدى . چۈنكى مىنىڭچە  جەمىيىتىمىزنىڭ جۈملىدىن مىللىتىمىزنىڭ بۇزۇقچىلىق پاتقىقىغا پىتىپ قېلىشىمىزنىڭ ئاساسلىق سەۋەبى ئائىلە تەربىيەسى بولسا كىرەك.  شۇڭا ئابلىكىم ئاكا  ئەگەر بۇ خىتىمنى كۆرسىڭىز ئائىلە تەربىيەسى توغرىلىق ئۆز كۆز قارىشىڭىزنى ئوتۇرغا قويۇپ باققان بولسىڭىز ؟ سىزنىڭ ، بىرقىسىم خوتۇنىنى باشقىلارنىڭ قۇچىقىغا چىقىرىپ قۇيىدىغان ، يېرىم يالىڭاچ يۈرىشىگە كۆزىنى يۇمىۋالىدىغان ، قىزلىرىغا شەرمى -ھايالىق ، ئەدەپ - ئەخلاقلىق بولۇش توغرىلىق نەسىھەت قىلمايدىغان بىرقىسىم ئىپلاس كىشىلىرىمىز توغرىلىق ، ئۆز كۆز قارىشىڭىزنى دادىل ئوتۇرغا قۇيالايدىغانلىقىڭىزغا ئىشىنىمەن .

يۇرتدېشىڭىز : ئابدۇۋەلى تۇرسۇن .

ئاياللارنىڭ بۇزۇلۇپ كېتىشىگە ناكەس ئەرلەرنىڭ مەسئۇلىيتى ئېغىر، يېرىم يالىڭاچ خوتۇنلارغا شالدىمىنى ئېقىتىدىغان، مۇشۇنداقلارغا ئامراق ئەرلەر بولمىسا، مۇشۇ بىر قىسىم ئەقىلسىز، ئىمانسىز، شەرم- ھاياسىزلار يەنە يېرىم يالىڭاچ بولىۋالارمىدى، بازاردا خېرىدار بولغاچقا مۇشۇنداق مال بولىدۇ، ھېچكىم ئالمايدىغان مالنى بازارغا كىم سالىدۇ؟
    مۇھىمى ئىمان، شەھۋەت، ھاۋايى- ھەۋەس شەيتىنى ئەرلارنىڭ قېنىدا، بىراق ئىمان بۇ شەيتاننى، بۇنىلا ئەمەس پۈتكۈل شەيتانلارنى ئىمان چەكلەپ، توسۇپ تۇرىدۇ، ئەرلەر ھەم ئىمانى سۇس، ئىمانسىز بولسا ئامان يوق. ئىسلام ئائىلە تەربىيسىنىڭ مۇھىم بىر تەركىبىگە ئايلانسا، جىق ئىشلار تۈزىلىدۇ، ئۆزلىرىنىڭ بۇ دۇنياغا نېمە ئۈچۈن يارىتىلغانلىقىنى بىلمەي، ياكى ئۇنتۇپ، ئويۇن- تاماشا، پۇلنىلا تونۇيدىغان چوڭ بولسا، ئاتا- ئانىلار ھەم گۇناھكار بولىدۇ.
