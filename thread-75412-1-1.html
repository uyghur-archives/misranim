10 كويلۇق قــــــىـــــز

.....بالام بۇغدايلارنى خامانغا ئەكەپ قويغىلى ئۇزۇن بولدى،پاتراق تىپىتەيلى،بولمىسا يامغۇردا قالسا بىر يىللىق يەيدىغىنىمىز تۈگەيدۇ ، ئەمما بۇرۇنقىدەك جاپا تارتساقمۇ تۆمۈر كوڭكىلارغا سالماي يۇمشۇتۇپ،خۇدانىڭ شامىلىدا سورۇپ يەيلى ، زىيانسىز كۈچلۈك بولىدۇ ،،،،، --
-ماقۇل-دادا يىڭى بىر ئۇسۇل تاپتىم،10كويدىن پۇل بىرىپ ماۋۇ دىبادا ئويناۋاتقانلارنى دەسسەتسەك بۇغدايلار تىز يۇمشايىتتى،..   
--ياق بالام،پاسىقىنىڭ مەينەتنىڭ ئايقى خامانغا تەگسە بەركەت قاچىدۇ،خىزىر نەزەر قىلمايدۇ،بۇنداق ئاشنى يىگىلى بولمايدۇ،،،،، --دادا دىگىنىڭ توغىرا،،،،،،،،،،،،،،،،،،،،،،  
******************************************************  
6-ئاينىڭ 27-كۈنى ئەزىزانە قەشقەرگە كەلدىم،28-چىسلا كەسپى سالاھىيەت ئىمتىھانى بىرىدىغان ئىش مۇناسىۋىتى بىلەن ، قەشقەرگە 20نەپەر ئىمتىھانغا قاتناشقۇچى بىللە كەلگەن ئىدۇق،ياتاققا جايلاشتۇق،ئاندىن تاماق يەپ بولۇپ سەگدىگىلى چىقتۇق ، جىق ئادەم باريەردە گەپمۇ جىق بولىدىكەن،قەشقەرلىق بىر باشلىقمىز بولىدىغان،يەكەنگەكەلگەندە خىلى چىقىشىپ قالغان ئىدۇق،ئۇ كۈتىۋاكىمەن دەپ ئۇنىمىساق ئۇنىماي 6-7مىزنى باشلاپ ماڭدى،ئەگەشتۇق،چۈنكى (بىز مۇھىم ئىشمىز بولسىمۇ ناخشا-مۇزىكا-ئۇسۇل دىسە جاننى ئاتايمىز)مىڭىپ (ئوۋچى)دىباغا كىردۇق.(سىلەر مىنى بۇنداق يەردە پوق يەمدىغاندۇ دەپ قالغانسىلەر ، بەلكىم ، تەكشۈرمىگەن كىشىنىڭ پىكىر بىرىش ھوقۇقى يوق،شۇڭا كۆرۈپ باقاي دەپ)بۇ كۈنى كىچىلىك پائالىيەتكەن،ئىشك ئالدىغا كەلسەك  8-10 قىز تۇرۇپتۇ،شۇنداق چىرايلىق،كىچىك قىزلار ئىكەن، بىز مىڭىشىمىزغا قىزلاردىىن بىرى مىنى تۈيۈقسىزلا تارتتى،----ئاكا،سىزمۇ دىباغا كىرەمسىز؟--
-ھە .. ئەمىسە بىزنىمۇ ئەكىرسىڭىزچۇ؟  
نىمە دىيىشمىنى بىلەلمەي،ھەم ئۇشتۇمتۇت بۇنداق گەپنى ئاڭلىغىنمغا ياخوش بولۇشنى ياخاپا بولۇشنى بىلمەي تۇرسام ئۇ قىز يەنە گەپ باشلىدى، -پەقەت 10 كويلا بەلەت ئالىدۇ،چىداڭە،بىز نى ئەكىرىپ قويسىڭىز سىز بىلەن بىللە تانسا-ئۇسۇل ئوينىشىپ بىرىمىز،زىرىكىپ قالمايمىز ، بۈگۈن كىچىچە ئاچىدۇ ، تازا ئوينايمىز ، بولۇڭە باشلاپ بولدى مۇزىكىنى ، ئاران چوڭلاردىن سوراپ چىققاندا ، ۋاقىتنى ھارام قىلمايلى ، ماقۇل دەڭە ئاكا، مەن ماقۇل بولدۇم ، بىز كىردۇق،بىللە كەلگەنلەر باشقا قىزلارنى ئەكىرىپ بوپتىكەن،جايلاشتۇق،دىمىسىمۇ شوخ قىزلاركەن مۇزىكا چىقىپ بولغىچە ئولتۇرالماي ئەرلەرنى تەشەببۇسكارلىق بىلەن تارتىپ سورۇننى جانلاندۇردى،باشقا گەپنى دىمەيلا قوياي،ئوينىغان ئۇسۇلىدىن،مىللى پۇراقنىڭ ئەسىرىمۇ يوق،خۇددى بۇرۇنلاردا خامانغا قاتىدىغان خامان كالىسىنىڭ ئۆزى،<دىسكو>!!!!!!!!!!!!!دىيىشى بىلەنلا ئىش باشقىچە،قىزلارنىڭ بىلىدە سۆڭەك يوقمىكىن دەپ قالدىم،خۇددى كالاڭ بولۇپ قالغان ئىشەك ھارۋىسىنىڭ چاقى ، بىرەر  سائەتتىن  كىين يىنىپ ياتاققا كەلدىم  ،ئۇيقۇم قاچتى ، نۇزۇگۇم ماڭا كۆرۈندى ، مەن ئۇنىڭغا قارىيالمىدىم،،،   
***************************************************************
-بالام سەن ئۇستا يانچۇقچىدىن بىرنى تىىپىپ كەلسەڭ، --نىمە قىلىسىز دادا-ھەممە ئادەم ئۇلاردىن بىزار تۇرسا سىز ھەجەپ ئۇلارنى ئىزلەپ قاپسىزغۇ؟ --مۇنداق ئىش بالام ،ياز بولمىسا دەيمەن،ياز بولسا ئاياللارنىڭ يوتىسى ،كۆكسى ئىچىلىپ قالىدىكەن،(قىسمەنلەرنىڭ)قاراپ سالىدىكەنمىز،ئاللا ئالدىدا گۇناھكار بوپ قالىمىز،تەربىيە قىلساقمۇ پايدىسى بولمىدى،گەپ بىز چوڭلاردىمۇ بار،شۇڭا يانچۇقچىغا پۇل بىرىپ ئاشۇ ھاياسىز يوتىلارنى-مەيدىلەرنى ئىتىك تىغ بىلەن تىلىپ بۇزىۋەتسەك ،ئۇلار ئۆزىنى ئوڭشارمىكى دەيمەن،   
--ياق دادا،بۇ قانۇنغا خىلاپ،بولمايدۇ،  
*************************************************************** ئەتىسى ئىمتىھاندىن چىقتۇق،تاماق يەپ بولۇپ قەشقەرنىڭ گۈزەل يەرلىرىنى ئايلانماقچى بولغان ئىدى، مۇمكىن بولمىدى،ئاخشام يىنىپ كەلگىنىمنىڭ جازاسى ئۈچۈن،ئۇلارنى مىھمان قىلىدىغان بوپ قالدىم، --مۇنداق قىلايلى-دىدى-باشلىق-مەن 4-5قىز تاپاي ،ماۋۇ (ۋىناس)دىباغا كىرەيلى،چۈشتىن كىيىن سائەت 2دە ئاچىدۇ،مۇزىكىسى بەك ياخشى،چولپانلار كەپتۇمىش، بىز كىردۇق،يەنە شۇ 10كوي تاپسا دىباغا كىرىدىغان كىچىك قانجۇقلاركەن،ئاقبالداق،قىزىل چاي،قىزىل كالا دىگەندەك نەرسىلەرنى بۇيرۇشتى،يەنە شۇ قۇلاق مىڭىنى يەيدىغان مۇزىكا،مۇزىكا توختىدى،بىر قىزغا تىلفۇن كەلدى،مۇڭدىشىپ بوپلا قىشىمىزغا كىلىپ؛-ئاپلا بەك سەت بولدى،ئاچامنىڭ ئۆپكىسىدە سۇ بار ئىدى،ھازىرى ئوپىراتسىيە قىلدىكەن،مەن ماڭاي -دەپ تۇرىشىغىلا ئىلانچى بىز مەيدان (ۋالىس)دەپ مۇزىكىنى باشلىدى،-ۋاي-ۋۇي،ۋالىسقا مەن بەك ئامىراق،بىر مەيدان ئوينىۋىلىپلا ماڭاي،  
ۋالىس تۈگىدى،كەينىدىن يەنە شۇ (دىسكو)!!!!!!!!!!!!!!!!!!!!!!!!!قىز كەتمىدى،مەن ھىساپنى قىلىپ چىقىپ كەتتتىم، ********************************************************** دادام بىلەن بولغان گەپلەرنى قەشقەردە كۆرگىنمنى دەپ بولدۇم،ئەمدى ئۆزۈمنىڭكىنى دەيمەن،ياق پۈتۈن ئۇيغۇرۇمنىڭ يۈرەك سۆزىنى ئىپادىلىيەلمىسەممۇ دەپ باقىمەن،بىز ھازىرقى ھالىتىمىزنى ساقلاپ قىلشنىڭ ئۆزىمۇ بىر گەپ بوپ قالدى،ئابدۇللا ئابدۇرىھىم ناماز ئوقۇيدىكەن دىۋىدى،بۇ سۆزقىززىق ھەم غەلىتە ئاڭلىنىپ باشقىلار بەس-بەستە مۇلاھىزە قىلغىلى تۇردى،،ئەمما ماڭا قىزىزىق بىلىنمىدى،،ئەگەر چىڭلۇڭ،ياۋمىڭ،لىلىيەنجى ناماز ئوقۇپتۇ دىسە قانداق بولاركىن تاڭ،چۈنكى ئۇيغۇر دىگەن ئىسلام دىنىغا ئىتقات قىلىمىز،ناماز ئوقۇش دىگەن پەرز تۇرسا،شۇڭا ئۇيغۇر چولپانلىرىمىزنىڭمۇ ناماز ئوقۇشى مۇقەررەر،بىز ھەيران قالىدىغان ئىشلار بەك تولا،كاللىمىز بەك ئۆتكۈرلىشىپ ئاخىرىدا ئىشلىمەس بوپ قالغان،ئىمان سۇس يەردە،مەينەتچىلىك بولىدۇ،بىزدەل مەينەتچىلىكنىڭ ئارىسىدا تۇرۇپمۇ يەنە مەينەتچىلىك ئىزلەۋاتىمز،،مەن ئەسلىدىن بىر تومپاي،،،،ئىدىم،،تەتۈر ئىدىم،داشقال ئىدىم،،،،تىل ئىششتىم..ئەمما كۆزقارىشىمنى دەپ  بولدۇم قۇرۇق نامىڭلار قالىدۇ خالاس،،بولدى خاتالاشقىم يوق مىنىڭ..ھازىر ناخشا مەشىق قىلىۋاتاتتىم،،ھە-راس ئارتىس قوبۇل قىلىمىز ،ئازقالدى،،،ئوينىمىغان ئۇسۇلدىن 3ى قالدى،،سەۋەپ ئادەم ئاز كەپقالدى،،ياردەملىشەرسىلەر،،مۇشۇ قىزلار نى ئۇرايمۇ دىدىم ئەمما مەن قورقىمەن، چۈنكى؛ ئۇيغۇرنىڭ مەنىسى-ئويۇشماق.بەزىدە ئۇرۇشقاق،قىلدەك ئىش ئۈچۈن قان ئىچكىلى تاس قالىمىز، پىلدەك ئىشنى پاشىدەك كۆرمەي ھە-بوپتۇلا دەيمىز،بۇرۇن 2ئوغلاق بارئىكەن،بىر تار كۆۋرۈككە تەڭلا كەپتۇ،بىر-بىرىگە يول بەرمەي،ئاخىرى موللاق ئىتىپتۇ،دىمەك ئۆلۈپتۇ،پىتشماسلىق،ھەسەتخورلۇق،مەنسەپ،بىز خەخنى نىمە كۈنگە قويمىدى،كىملەرنىڭ قول ئاستىدا ئىزىلدۇق؟ئىزىلىۋاتىمىز،لىكىن بىلمىدۇق،چۈنكى بىزدە بىر ئاڭ باركى ئۇ بولسىمۇ تەڭ تۇرۇش،يانماسلىق،ئەمما باشقىلارغا ئەمەس،ئاۋال ئۇلۇغ ئىگەم ھەممىزگە تىنىچلىق ،ئاسايىشلىق،بەختيىار كۈنلەرنى ئاتا قىلغاي،
،ئەپسۇس سۈرگەن  دەۋرانى يەنە شۇ ئۆزى تەرىپىدىن ھالاكەت بولدى،سەۋەپ--ھەسەتخورلۇق،پىتنىخورلۇق،چىدىماسلىق،مەنسەپ تالىشىپ .......دىگەندەك سەۋەپ جىق.مانا ئەمدى ئەجداتلىرىمىزنىڭ ياخشى تەرەپلىرى،   
ئۇنتۇلدى.ئەمما ئۇلاردىن قالغان پىتىشماسلىق  تىخى تۈگىگىنى يوق.ھەي ئۇيغۇرۇم،.ئىسىت!!!!!!!!!!!!!!!!!!!!!!!!!گەپ قىلىڭ ئۇيغۇرۇم، ھەممىىنى بلىمىز، ھەممىنى قىلالايمىز، لىكىن  .  . ئەينى ۋاقىتتا مۇسا ئەلاھسالام ئۆز قەۋىمىنى ئۆزگەرتىش ئۈچۈن ئۇلارنى باشلاپ 40 يىل چۆلدە ياشىغان.قەۋىمى ئۆزگەرگەن ھازىرقى ئۇيغۇر قەۋىمىنى سىز باشلىيالامسىز؟ مەن ئەگىشەلەمدىم؟ مانا بىز ئىتى ئۇلۇغ سۈپۈرىسى قۇرۇق يا ئىگىسى يوق، ياكى بىشى يوق قۇرۇق سۆڭەك ئىسكىلىتلا خالاس.    خۇدا ئىشىڭىزنى ئوڭشىغاي.نەچچە يىل دەۋران سۈرگەن يەكەن خانلقى ئاپاق غوجا تەرىپىدىن يوقالدى،(جاللات خىنىمنىڭ بىشىنى ھىلەك دىگەن مۇناپىق كەستى) 2008-يىلى يىزىپ ساقلاپ قويغان تىمام ئىدى
