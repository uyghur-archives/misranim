                                  يايرىماس قايسى ئىنسان يېنىدا مۇڭداش دىلدارى تۇرسا،
                                   ئېچىلماس قايسى ئىنسان يېنىدا شوخ گۈلنىگارى تۇرسا.
                                    ۋە لېكىن ئۇ بەخىتلەر مەندىن نى يىراق تۇرسا،
                                   قانداق شوخ ناۋا ئەيلەي ۋىسالىم بى دىدار تۇرسا.

                          ئىزلەپ يار ۋىسالىنى چەكتىم مىڭ مۇشەققەتنى ،
                         چەكسەم مەيلى مۇشەققەت لېكىن تېگى راھەت تۇرسا.
                          يارسىز تەنھا يۈرۈش ئۆلۈمدىنمۇ ئارتۇق ماڭا،
                            چۈنكى ئۇ يۈرەك پارەم مېنىڭ دىلغۇبارىم تۇرسا.

                           ئىشقىڭدا بولۇپ گۈلخان سەن ئۈچۈن ئۇرساممۇ خەندان،
                           يۈردىڭ سەن بىخەۋەر بۇ كۆڭلۈم ئىنتىزار تۇرسا.
                           كۆيۈپ باغرىم كۈل بولدى بىلمىدىڭ بۇ ئازابىمنى ،
                            بەرمىدىڭ يارامغا دورا -ۋۇجۇدۇم بىمار تۇرسا.

                             كۆيۈپمەن ئىشقىڭدا يىقىلسام بېرەلمەي بەرداش ،
                             يۈرۈپسەن ئۆزگىنى ئىزدەپ تېخى جېنىم ھايات تۇرسا.
                             دەردىنى تارتىسەن بىر كۈن قىلمىشىڭ چۈن تۆلەپ بەدەل،
                              چۈنكى بۇ ئۆتنە ئالەم ھەر ئىشنىڭ سورىقى تۇرسا.

                             دەپسەن كۆرەڭلەپ:نىگار مەنسىز ياشالماس،
                             چۈنكى مەن يۈرەك پارى زىنھار بىر دىلدارى تۇرسا.
                             ۋە لېكىن قۇلاق سالغىن كۆرەلمەس مەنمەنچى قىزچاق،
                            نەزەر سالغىن جاھانغا قىزلار كۆپ بازار تۇرسا !!

بەزىلەر

                                          ھۆرمەت قىلسا ،قوغلاشسا،
                                          قاچارئىكەن بەزىلەر.  
                                          مەنسىتمىسە ئۆزىنى،
                                          چاپار ئىكەن بەزىلەر.

                                          كىچىككىنەنەپ تەگسە،
                                          چاپار ئىكەن بەزىلەر.  
                                           قىلچەزىيان تارتماستا،
                                           قاچار ئىكەن بەزىلەر.  

                                          پايدا مەنپەتنى كۆڭلىگە،
                                          پۈتەر ئىكەن بەزىلەر،
                                          ئېرىشەلمىسە ھېچبىر نەپ ،
                                          مۈكەر ئىكەن بەزىلەر.

                                           پايدامەنپەت كۆرسەم دەپ،
                                           مەنلا دەۋران سۈرسەم دەپ،
                                           خام خىياللار دېڭىزىغا،
                                           چۆكەر ئىكەن بەزىلەر.

                                           ھوزۇر ھالاۋەت پۇل ئۈچۈن،
                                           سەپ ئالدىدا بەزىلەر.
                                           ھارام بولسا ھالال دەپ،
                                           نە پ كويىدا بەزىلەر.

                                           باشلىق كۆرسە ۋاي خوش دەپ،
                                           پاقلان توپاق سوياي دەپ،
                                           گاداي نامرات يېتىمنى،
                                           كۆزگە ئىلماس بەزىلەر.

                                                                            ئاپتۇرى:ۋاسىتجان
