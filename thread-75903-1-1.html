  شۇ تەرەقىدە 10 كۈن ئۆتتى،شۇكۈنى يەنە ھىلىقى نومۇردىن تىلفۇن كەلدى. بۇقىتىم قىزىم سوفىيە ئۆزى ئۇرۇپ تۇ- تىشلىقمۇ ئاپا ،دادام سالامەتمۇ دەپ گەپ باشلىغان ئىدى ،مەن ئۇنىڭ گىپىگە قۇلاق سالماستىن سۆزلەپلا كەتتىم.بىز سىنى شۇنداق جاۋاپ قايتۇرسۇن دەپ بىقىپ چوڭ قىلىپ ئالى مەكتەپلەردە ئوقۇتتۇقمۇ؟  سەن بىر خەنزۇنى دەپ ئاتا ئاناڭدىنمۇ ،قىرىنداشلىرىڭدىنمۇ يۈز ئۆرۈدۈڭما ؟سەن نىمىدىگەن سەخسىيەتچى؟ئۆزىڭدىن باشقىلارنى ئويلاپمۇ قويمايدىغان ،بىز ئەمدى دوسىت-دۈشمەنلەر ئالدىدا قانداق باش كۆتۈرۈپ يۈرىمىز؟
دىگەندەك نۇرغۇنلىغان ئاچچىق سۆزلەرنى قىلىپ كەتتىم.ئۇ بىر ئىغىزمۇ گەپ قىلماي يا تىلفۇننى قويىۋەتمەي تۇرىۋاتاتتى،بۇنداق قاتتىق گەپ قىلىشىم ئۇنى بىقىپ 23 ياشقا كىرگۈچە قىلغان تۇنجى قىتىملىقى ئىدى، ئەمما ئۇ ئۈندىمىدى.شۇ ئاچچىقىمدا يىنىمدا بولسا ئىدى نەچچە تەستەك سىلىۋەتكەنمۇ بولار ئىدىم.ئەمما ئۇ يەنىلا جىملا يۇرىۋالدى. مەن تازا سۆزلەپ ئىچىممۇ ئازراق بوشاپ قالغاندەك بولدى ،شۇ قاتتىق گەپلەردىن كىيىن ئۇ  زۇۋانغا كەلدى،ئاپا ھەجەپمۇ گۆش يىگۈم كىلىپ كەتتى،بۇيەردە مەن يەيدىغان گۆش يوقكەن،بۇ يەردىكى مۇسۇلمانچە گۆشلەرنىڭ تەمى يۇرتىمىزنىڭ گۆشلىرىنىڭ تەمىگە ئوخشىمايدىكەن،-ئەمىسە نىمە يىدىڭ؟دەپ سورىدىم مەن،ئۇ ئاساسلىقى توخۇ  بىلىق سۈت تۇخۇم گۆشىنى يىگۈم كەلمىدى دىدى.ئۇنىڭ بۇ گىپىنى ئاڭلاپ ئىچىم سىرىلىپلا كەتتى.تۇغۇتىڭنى كىم بىقىۋاتىدۇ؟-ئۇنىڭ ئاچىسى ھەم دادىسى،ئۇ ئۆزى ئىشلەپ يۈرىدۇ ،ئۇنىڭ ئاپىسى يوق.ئۆزى ئىشقا كەتسە شۇلار قارايدۇ-دىدى.ئوپراتسىيە قىلدىمۇ؟-ياق ،ئۆزۈم دوختۇرخانىدا توغدۇم...تۇغۇتتا ياخشى بىقىلمىساڭ كىيىنچە كىسەلچان بولۇپ قالىسەن دىدىم...شۇنىڭدىن كىيىن ئۇ كۈندە دىگۈدەك تىلىفۇن ئۇرۇپ تۇرىدىغان بولدى.مەن چوڭ قىزىمغا ئۇنىڭ گۆش يىگۈم كەلدى دىگىنىنى ئىيىتسام ئۇ دەرھاللا  -ئەمىسە بىزگە ئادىرىسىنى دەپ بەرسۇن بىز گۆش قورۇپ،توقاچ ياقتۇرۇپ قۇۋۋەت تالقانلىرىدىن ئىلىپ بىر يەشىك قىلىپ پوچتىدىن ئىۋەتىپ بىرەيلى دىگەن مەسلىھەتنى بەردى.شۇنداق قىلىپ بىر ئۇنىڭغا پوچتىدىن دائىملا شۇنداق نەرسىلەرنى ئىۋەتىپ تۇرىدىغان بولدۇق.كۈنلەر ئەنە شۇنداق ئۆتىۋاتاتتى،مىنىڭ ئىچىمنى بىرنىمە يەپ پەقەتلا ئارامىمدا قويمايىتتى.كۈندە دىگۈدەك ئۇنىڭ بىلەن-ئاپا ئانداق   ئاپا مۇنداق..ئاپا بالام يىغلىدى ..كۈلدى دىگەندەك پاراڭلارنى قىلىشاتتۇق.ئۇلار تۇرغان يەر   كۆمۈركان رايۇنلىرىنىڭ بىرى بولۇپ ئۇ بالىسى بىر ياشقا كىرگەندە شۇ تاغدىكى كۆمۈر كاننىڭ ئادەم تۇرمايدىغان قۇدۇق بىشىغا ئورۇنلىشىپتۇ.يولدىشى كانغا ئاستىدا ئىشلەيدىكەن،ئۆزى ئۈستىدە بالىسىنى باققاچ توكيوللىرىنى باشقۇرۇپ ئىشلەيدىكەن ،بۇمۇ ئۇلارنىڭ تىرىكچىلىكى ئىكەن،يولدىشىنىڭ دادىسى شۇ ئۆلكىنىڭ يىراق يىزىسىدىن ئىكەن ئۇلانىڭ ھال كۈنىمۇ ئارانلا بولۇپ ئۇ .ئادەم ياشىنىپ قالغان 70 ياشتىن ئاشقان ئادەم ئىكەن ،بىر ئاكىسى بىر ،بىر ئاچىسى بار ئىكەن.دادىسى ئاساسەن ئاچىسى بىلەن تۇرىدىكەن ،ئۇلارمۇ ئۆز تۇرمۇشىدا ئارانلا ئىكەن.ئۇلار ئايدا بىرەر قىتىم بازارغا كىرىپ سودىلىق قىلىپ كىتىدىكەن .بولمىسا ئىش ئورنىدە نەچچە كۈندە بىرەر قىتىم باشلىقلىرى ئۇ ئورۇننى تەكشۈرۈپ كەلمىسە ئادەممۇ كۆرمەيدىكەن ...توۋا  مىنىڭ بۇ ئەقىللىق قىزىم مۇھەببەت ئۈچۈن شۇ كۆڭۈلنىڭ كەينىگە كىرىپ شۇ تۇرمۇشقا كۆنۈپ ياشايدىكەن ،ئۇلاربىلەن ئانچە -مۇنچە توردىمۇ كۆرۈشۈپ تۇردۇق .توۋا...ھەيرانمەن بۇ كۆڭۈل دىگەن ساراڭغا قىزىم  شۇنچە قىلىپ كەينىدىن قىچىپ كەتكۈدەك  نە ئۇنىڭ يولدىشىنىڭ قاراپ قالغۇدەك چىرايى ،يا ئۇنىڭ ئۆز ئالدىغا ئىگىلىكى ،يا خىزمىتى ياكى ئۆز ئالدىغا ئادەملەرنى قىزىقتۇرغۇدەك  ئۆي -زىمىنلىرى بولسا كاشكى...توۋا...كۆڭۈل ياتقان يەردە ئېت ياتمايدۇ دىگەن شۇ كەن ،ھەرقىتىم كۆرۈشكىنىمدە ئۇنىڭ گەپ سۆزىدىن ئۇنىڭ ئۇيەردە پەقەتلا خورلۇق ھىس قىلمايدىغانلىقىنى ئۇنىڭ شۇنداق بەخىتلىك ئىكەنلىكىنى ھىس قىلىمەن.ئەمما بىرەر قىتىممۇ سىلەرنى سىغىندىم دەپ باققىنىنى ئاڭلاپ باقمىدىم.توۋا....شۇ جەرياندا ئۇلار شۇ تاغ ئۆڭكۈرىدە ئادەمسىز ياشاپ  بالىسىنى ئۈچ ياش قىلدى،بىز بولساق دادىسى بىلەن بىرىمىز چىقساق بىرىمىز كىرىپ بالنىستىن چىقماي يىتىپ قىلىۋاتاتتۇق.ئاستا-ئاستا ئۇرۇق تۇققانلىرىمىز كىچىك قىزىمنىڭ ئىشىدىن خەرەر تاپتى بولغاي ،ھىچكىم ئۇنىڭ گىپىنى بىزنىڭ قىشىمىز دا دىمەيدىغان  ھەتتا ئۇنى قېنى يوققۇ   دەپ سوراپمۇ قويمايدىغان بولدى،ئەمما خۇلۇم-خوشنىلار  دوسلار بىلمەيىتتى.ئۇلارغا بىر شەخسى شىركەتتە ئىشلەۋاتىدۇ دەپلا ئۆتتۇق...بىر كۈنى ئۇ قىزىم يەنە تىلفۇن قىلدى - ئاپا بىز شىنجاڭغا كىتىدىغان بولدۇق دىدى.مەن بۇ گىپىدىن بىر ئويلاپ شۇنداق خوش بولدۇم يەنە بىر ئويلاپ شۇنداق ئەندىشىدە قالدىم ،ئۇزۇن ئۆتمەي ئۇلار قايتىپ  كەلدى،قىزىم كۆزۈمگە چىرايلىرى ئۆزگۈرۈپ قالغاندەكلا كۆرۈندى،لىكىن يولدىشى كۆزۈمگە پەقەتلا سىغمايىتتى.ئۇنىڭ قىزى شۇنداق چىرايلىق ،شۇنداق ئەقىللىق ھەم ئەدەپلىك بولۇپ ھەيران قىلدىم،ئەمما ئۇ مەن ئىتراپ قىلمايۋاتقان نەۋرەم ئىدى.ئۇلار ئۇدۇللا ئۆيگە چۈشكەچكە ئۆيۈمگە بىرەركىم كىرىپ قالارمۇ  بۇخەنزۇ كىرىپ چىقىپ يۈرسە نىمە دەپ قالار دەپ شۇنداق قورۇنۇپ يۈرگەن كۈنلىرىمنىڭ بىرىدە ئۇلار  داۋەن تەرەپتىن ئۆي ئىجارىگە ئىلىپ چىقىپ كەتتى.مەن مۇ ئۇنىڭغا ئاتاپ تەييارلىغان ئۆي سەەرەمجانلىرىمنى بىرەر قۇر بەردىم.ئەمما ئۇنىڭ يولدىشىنى پەقەتلا قوبۇل قىلالمىدىم.ئۇنىڭ مۇسۇلمان بولغىنىغا بىر يەرلىرىمدىن ئىشەنگۈم كەلمەيىتتى،ئەمما ئامال قانچە؟...كۈنلەر ئەنە شۇنداق ئۆتىۋاتاتتى،قىزىم ئۆيدە بالىسىنى بىقىشبىلەن ئانچە-مۇنچە كىرقات يۇيۇشتىن باشقا ھىچ قانداق ئىش قىلمايىتتى ،ئاددىسى تاماقمۇ ئەتمەيىتتى.تۇرمۇشتىكى ھىچنەرسىنىڭ غىمىنى قىلمايىتتى.بۇ شەھەردە بىرى ئىشلەپ ئۆي ئىجارىگە  ئولتۇرۇپ ئۈچ جاننى بىقىشمۇ ئاسان ئەمەس ئەلۋەتتە.ئۇنى ئىچكىردىن قايتىپ كەلدى ،كۆزئالدىمدا بولدى دەپ بىر يىرىمىدىن خوش بولغىنىم بىلەن ئىچىمنى بىرنىمە يەپ تۈگىشىپلا كىتىۋاتاتتىم.يەنىلا بۇسىرىمنى ئۇرۇق-تۇققان دوس-بۇرادەرلىرىمگە ئۆز ئاغزىم بىلەن ئاشكارە قىلمايىتتىم ،بەلكىم بىلگەنلىرىمۇ بىزنى ئاياپ ئالدىمىزدا ئۇنىڭ ىپىنى قىلمايىتتى...پەقەت ۋە پەقەت <<ئۇلۇغ ئاللاھ>>دىن ئۇلارغا ئىنساپ تىلەيتتىم.بىر كۈنى چوڭ قىزىم كىرىپ يولدىشىنىڭ كۈندىن كۈنگە خۇمىرى ئىغىرلاپ تۈگىشىپلا كىتىۋاتقانلىقىنى  ،قايتا-قايتا زەھەر تاشلاش ئورنىغا ئەكىرىپ بىرىپمۇ يەنىلا زەھەردىن قول ئۈزەلمەيۋاتقىنىنى ئۆزىنىڭمۇ بۇ تۇرمۇش بىسىمى ھەم روھى بىسىمىنىڭ ئىغىر لا ۋاتقىنىغا بەرداشلىق بىرەلمەيۋاتقانلىقىنى ،ئۇنى تاشلاپ ئاجىرىشىپ كىتەي دىسە بىرتال قىزىنىڭ دادىسىز بولۇپ قىلىشىدىن ھەمدە يولدىشىنىڭمۇ ئۇلاردىن ئايرىلىپ قالسا تىخىمۇ تىزلا تۈگىشىپ كىتىدىغانلىقىنى ئۆزى ياخشى كۆرۈپ تەگكەن يولدىشىغا يەنىلا ئىچ ئاغىرىتىدغانلىقىنى يىغلاپ تۇرۇپ سۆزلەپ كەتتى،مەن بۇ ئىچمىجەز قىزىمنىڭ ئەزەلدىن تۇرمۇش ئىشىنى سوراپ ئۇنىڭ يۈرىكىنى تاتلىغىم كەلمەيىتتى.مەن نىمە قىلىپ بىرەلەيمەن؟پەقەت قىزىمغىلا ئىچىم ئاغرىسا ئاقىرشىپ كەت دىيىشىم مومكىن ،لىكىن ئۇلارنىڭمۇ ياخشى كۈنلىرى بولغان،ئۇنىڭغا ھەرگىزمۇ كۆز يۇمالمايمەن.مەن ئۇنىڭ ئەھۋالىنى پەقەت باشقىلاردىنلا بىلىپ تۇراتتىم.ئۇ ئۆزىمۇ ئەزەلدىن ئۆيگە كىلىپ بىزگە خوشلۇقى ياكى خاپىلىقى توغۇرلۇق گەپ قىلمايىتتى.بۇگۈن ئۇنىڭ سەۋرى قاچىسىمۇ توشۇپ كەتكەن بولسا كىرەك، ئىچ-ئىچىدىن يىغلاپ تۇرۇپ سۆزلەيىتتى.ئۇ ئاتا-ئانام بىقىۋاغان ئىنىمنىڭ ئىككى بالىسى ياۋرۇپاغا ئوقۇشقا چىقىپ كىتىپ قايتىپ كەلمىگەنلىكىنى ،ئۇلارنىڭ شۇيەردە ھال كۈنىنىڭ ياخشىلىقىنى ،ھەمدە يولدىشىنىڭمۇ موھىت ئالماشتۇرۇپ باقسا پايدىسى بوپقالار دىگەن ئۈمۈدلەر بىلەن ئۇنىڭغىمۇ شۇ يولنى مىڭىۋاتقانلىقىنى پات ئارىدا پاسپورتى چىقىدىغانلىقىنى ئۇنىڭغىمۇ نۇرغۇن پۇل كىتىدىغانلىقىنى ،ئۆينى سىتىپ بولسىمۇ ئۇنى يولغا سالىدىغىنىنى تەپسىلى قىلىپ سۆزلىدى.مەن تۇرۇپ كىتىپ ئۇنىڭ ھالى بىلەن قالماي ئۆزۈمنىڭ ھالى بىلەن قالدىم.ئۇنىڭغا كىچىك قىزىمنىڭ يولدىىنىمۇ ئۇنىڭ بىلەن قوشۇپ بىللە يولغا سالساق ئۇ  قارا باشتىن قۇتۇلمامتۇق دەپ ئىشلارنى ئۆز خاھىشىم بىلەن ئاددىلا ئويلاپتىمەن.ھەم چوڭ قىزىمغا ئۇلارغىمۇ مەسلھەت سىلىپ بىقىشىنى ،ئەگەر پۇلى بولمىسا بىزنىڭ ياردەم قىلىدىغانلىقىمىز نى دەپ ئۇنىڭمۇ ماقۇل كۆرۈشىنى كۈتۈپ ئولتۇردۇم.ئەمما ئۇ كىچىك قىزىمنىڭ قاراباش يولدىشىمۇ ئىنتايىن تىرىشچان بالا بولۇپ بىردەم بىكار ئولتۇرمايىتتى.قىزىمنىڭ ئاسان ئۆرۈلسە تىگىدە مانتا يەپ ئولتۇرىدىغان غەم يىمەس مىجەزى ئۇنىڭ بىلەن ھەجەپمۇ ماسلاشقان ئىكەن.شۇنداق قىلىپ ئۇلارمۇ بۇ ئىشقا قىزىقىپلا قالدى .2010-يىلى 8-ئايدا ئۇلارنىڭ رەسمىيەتلىرى پۈتۈپ يولغا چىققان ئىدى.مەن شۇنداق خوش ئىدىم.خۇددى يەلكەمدىن ئىغىر بىر تاشنى ئىلىۋەتكەندەكلا بولدۇم.بۇ دۆلەتتىن بىۋاستە ياۋرۇپاغا بىرىش ئۇنداق ئاسان ئەمەس بولغاچقا ئالدى بىلەن تاشقى مۇڭغۇلىيگە چىقىپ ئاندىن ئۇ دۆلەت ئارقىلىق بارماقچى بولغان دۆلەتكە بارىمىز دەپ مىڭىشتى،شۇ ۋاقىتتا خوشلاشقاندىكى كۆرۈنۈش پەقەتلا ئىسىمدىن چىقمايدۇ.توۋا ...چوڭ كۈيئوغلۇم ئۇ كۈنىمۇ تازا نورمال ئەمەسمۇ  خۇددى ئۈرۈمچىدىن قەشقەرگە ماڭغاندەك كۈلۈپلا خوشلاشتى،ئەمما كىچىك قاراباش كۈيئوغلۇم بولسا تۆت ياشلىق قىزىنى چىڭ قۇچاقلاپ خۇددى كەلمەسكە كىتىدىغاندەك ھۆڭگىرەپ يىغلايتتى،كىچىك قىزىم بىر تەرەپتە يىغلاپ كۆزىنى قىزارتىپ تۇراتتى.ئۇلار بىلەن خوشلىشىپ ئۆيگە قايتىپ كىرىپ ئىچ-ئىچىمدىن خوش بولدۇممۇ قىن-قىنىمغا پاتمايىتتىم.شۇنچىلىك يىنىك ھىسقىلاتتىم.ھەم سىرىمنى دەپ ئىچىمنى بوشىتىدىغان ئىككى دوستۇمغا تىلفۇن قىلىپ كۆڭلۈمدە زىخمۇ كاۋاپمۇ كۆيمەيدىغان بۇ يوللار بىلەن  ئائىلىمىزگە چۈشكەن بۇقۇرۇتلاردىن كۇتۇلدۇق دەپ ئويلىدىممۇ ئۇلارغا سۆزلەپ كەتكىنىم شۇنداق ئىسىمدە،ئەمما بۇخوشلۇقۇم ئارانلا 10كۈن بولدى.ئۇلار مىڭىپ 10-كۈنى سەھەردىلا ئىشىك چىكىلدى ،ئىشىكنى ئاچسام ئۇ ئىككىسى ئۆيلىرىگىمۇ بارماي ئۇدۇللا ئىشىك ئالدىمدا تۇراتتى،مەن چۈچۈپ كىتىپ ھەيران بولۇپ ئۆيگە باشلىدىم.ھەم نىمەئىش بولغىنىنى سورىدىم.ئۇلارمۇ روھى چۈشكەن ھالدا باشقىلار تەرىپىدىن قويۇلغانلىقىنى ،پەقەتلا يىنىدىكى پۇلى بىلەنلا قالغانلقىنى ،ماڭغان يولنىڭ بىكار بولغانلىقىنى بەس-بەس بىلەن سۆزلىشىپ بۇ يولداكەتكەن پۇلى ئۈچۈن ئىچلىرى ئىچىشىپ كەتكەن ئىدى.بىردەمدىلا قىزلىرىم كىلىپ ئەھۋالنى ئۇققاندىن كىيىن ئۇلار بىرگە ئۆيلىرىگە كىتىشتى.ئەمما تىپىپ شۇ يەردە بىرەر ئىش-ئوقەت قىلىدىغانلىقىنى دىيىشتى،ئەمما ئۇنىڭغا پۇل بەكمۇ جىق كىتىدىكەن.قىزىم قەتئى ئامال قىلىپ يولدىشىنى يولغا سالىدىغانلىقىنى دەپ چىقىپ كەتتى،ئارىدىن ئىككى ئاي ئۆتتى شۇنىڭغىچە چوڭ كۈيئوغلۇمنى ئىككى قىتىملا كۆردۈم.شۇ يىلى 11-ھينىڭ17-كۈنى چوڭ قىزىممۇ ئۆيدە بار ئىدى،ئۆيگە تىفلۇن كەلدى.ئۇ كۈيئوغلۇم ئىدى،بۈگۈن توي قىلغانلىقىمىزنىڭ 15يىللىقى ،مەن گۈلمىرەنىڭ قىشىدا بۈگۈن
