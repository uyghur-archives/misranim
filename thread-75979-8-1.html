ئەسسالامۇ ئەلەيكۇم؟ يۇقىرىدا نۇرغۇن ئىنكاسلارنى كۆردۇم، ھەممەيلەن سىزگە كۆڭۇل بولۇپ ھەر نىمىلەرنى دەپتۇ. مىنىڭ سىزگە بىرىدىغان تەكلىپىم كېسەلنى مۇنتىزىم تەكشۇرتۇپ، مۇنتىزىم داۋالىتىڭ! نۇرغۇن ئىنكاسلاردا ئاللادىن تىلەش توغرىسىدا گەپلەر بوپتۇ، ئاللا چۇقۇم تىلىگەنگە بىرىدۇ، ئەمما ھەممە ئىشنىڭ شەرتى بار، ئۇلۇغ پەيغەمبەر ئەلەيھىسسالام ئاللادىنلا تىلەپ ئولتۇرمىغان سەۋەبىنى قىلغان، ھەمدە ھەر قانداق ئاللا سالغان كېسەللىكنىڭ دورىسى بار، دەپ ئۇلۇغ سەھىھ ھەدىس كىتاپلىرىدا قەيت قىلغان. ئىسلام دىنى ئاجايىپ ئۇلۇغ، ھەقىقى توغرا، ئىلمىي دىن، خۇراپاتلىق يوق، خۇراپاتلىقنى بىز ئۆزىمىز دىنغا كىرگۇزۇۋالغان. شۇڭا سىز ئۇلۇغ پەيغەمبەرنىڭ ھاياتىغا قاراپ باقسىڭىز ھەممىنى چۇشىنىسىز. بىر ئۇلاغنى باغلاپ قويۇپ ئاندىن ئاللا تەۋەككۇل قىلىپ، ئاللا تاپشۇرۇپ ئارام ئالىسىز بولمىسا ئات باشقا يەرگە قىچىپ كىتىدۇ، ئاللاغا تاپشۇرمىسىڭىز ئوغرى يالغان ئېلىپ كىتىشى، بوشۇنۇپ كىتىشى مۇمىكىن. ئاللاغا تەۋەككۈل قىلىپ خوتەنگە كىلىپ « ئاتا بولاق ئۇيغۇر تىباتچىلىك دوختۇرخانىسى» غا كىلىپ مەتتۇرسۇن ھاجىمنى ئىزدەڭ، چۇنكى مەتتۇرسۇن ھاجىم داڭلىق تىۋىپ تۇردىمۇھەممەت ئاخۇنۇمنىڭ ئوغلى، تۇردىمۇھەممەت ئاخۇنۇم ئۇيغۇر تىبابەتچىلىكى بۇيىچە يىڭى جۇڭگۇدا تۇنجى قېتىم مۇدىر ۋىراچ ئۇنۋانى بىرىلگەن ئالتە تىۋىپنىڭ بىرى، ئۈدۈمنى ئوغلى مەتتۇرسۇن ھاجىمغا قالغان، دىئاگنوز قويغان كېسەللىكلىرى ئۈرۈمچى ياكى بېيىجىڭدا تەكشۇرۇتسىمۇ ئاساسەن ئوخشاش چىقىپ قالىدۇ، «شىنجاڭ ئۇيغۇر تىبابەتچىلىكى ئالىي تىخنىكومىغا بىۋاستە قاراشلىق دوختۇرخانا»غا بىرىپ ئەنۋەر مۇئەللىمنى ئىزدەڭ، ئەنۋەر مۇئەللىم ئىچكىرىدىكى بىر داڭلىق تىببى ئۇنىۋىرستىنىڭ جۇڭگۇ تىبابىتى كەسپىدە ئۇقۇغان، يىڭنە سانچىپ، ئۇۋۇلاپ داۋالاشقا ئۇستا، خوتەن  كىلىپ ئۇزۇنراق تۇرۇپ داۋالىنىشقا تەييارلىق قىلىپ كىلىڭ، سىزنىڭ كىسىلىڭىز ئاستا خاراكتىرلىك كېسەل. خوتەندە تۇرۇپ بىر مەزگىل ئۆزىڭىزگە داۋالاش لايىھەسى تۇرغۇزىۋالغاندىن كىيىن، تىۋىپلارنىڭ ئالاقىلىشىش تىلىفۇنىنى يېزىۋىلىپ ئاندىن قايتىپ كىتىپ، دورىڭىز تۈگىگەندە، سىزدە باشقا ئالامەت بولغاندا تېلىفۇندا ئالاقىلاشسىڭىزمۇ بولىدۇ. چۇقۇم ئاللادىن تىلەش بىلەن دورا بىلەن داۋالاشنى بىرلەشتۇرۇڭ، ئاللا ئىنساننى روھ بىلەن ماددىنىڭ بىرىكىشىدىن ياراتقان، بىرسى يەنە بىرىنىڭ ئورنىنى تولدۇرالمايدۇ، سىزگە ئاللاھ شىپالىق ئاتا قىلسۇن!
