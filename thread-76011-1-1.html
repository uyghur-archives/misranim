ئۇچۇنچى شەخىس مەجبۇرى سۇغۇرتىسى ۋە ئۇچۇنچى شەخىس سۇغۇرتىسنىڭ پەرقى

قاتناش ھادىسىدىكى مەسئۇليەت مەسىلىرى بارغانسىرى كۇپلىگەن كىشىلەر كۇڭۇل بۇلىدىغان مەسىلگە ئايلانماقتا .يول قاتنىشى بىخەتەرلىك قاتناش قانۇنىدا ھەر قانداق ماشىنا ماتۇرلۇق قاتناش ۋاستىللىرى مەجبۇرى سۇغۇرتىسغا قاتنىشىشى كىرەكلىگى بەلگىلەنگەن.بىراق مەجبۇرى سۇغۇرتىنىڭ تۆلەش مەزمۇنى ۋە ئۇسۇلى ھەققىدە دۆلەت تىخى ھازىرغىچە ئىنىق بىر قانۇن ياكى ئىزاھات چىقارمىدى .ھازىرى ماتۇرۇلۇق قاتناش ۋاستىللىرى ئۇچۇنچى شەخىس مەجبۇرى سۇغۇرتىسىنى مەجبۇرى سىىتۋالىدۇ، لىكىن ئۇچۇنچى شەخىس سۇغۇرتىسىنى ئىختىيارى سىتىۋالىدۇ.

ئەمىلليەتتە ئۇچۇنچى شەخىس مەجبۇرى سۇغۇرتىسى ۋە  ئۇچۇنچى شەخىس سۇغۇرتىسى ئوتتورسىدا پەرىق كۆپ. ئۇچۇنچى شەخىس سۇغۇرتسى ئاەتتە سودا خارەكتىرلىك ئۇچۇنچى شەخىس سۇغۇرتىسى دىيلىدۇ. ماشىنا سۇغۇرتىسىنى سىتىۋالغۇچى رۇخسەت قىلغان لاياقەتلىك شۇپۇر ماشىنا ھەيدەپ ،قاتناش ھادىسى يۇز بىرىپ جىسمانى زىيان ۋە مال مۇلۇك زىيىنى يۇز بەرسە سۇغۇرتا شىركىتى ئۇچۇنچى شەخىس مەجبۇرى جاۋاپكارلىق دائىرسدە تۆلەم جاۋاپكارلقنى ئۇستىگە ئالىدۇ.ئۇچۇنچى شەخىس مەجبۇرى سۇغۇرتىسدا تۇلەيدىغان دائىرە ئۇچۇنچى شە خىس سۇغۇرتىسنىڭكىدىن كۇپ .يول قاتنىشى بىخەتەرلىك قانۇنىنڭ 76- ماددىسغا ئاساسلانغاندا -  ماتۇرلۇق قاتناش ۋاستىللىرى قاتناش ھادىسى يۇز بىرىپ جىسانى زىيان ۋە مال مۇلۇك زىيىنى يۇز بەرسە ،سۇغۇرتا شىركىتى ئۇچۇنچى شەخىس مەجبۇرى سۇغۇرتىسى تۆلەم  جاۋاپكارلىقى دائىرسىدە تۇلەپ بىرىشى كىرەكلىگى كۇرسىتىلگەن .

ئىككى سۇغۇرتىنىىڭ ئەڭ چۇڭ پەرقى شۇكى ئۇچۇنچى شەخىس مەجبۇرى سۇغۇرتىسىنىڭ تۆلەم دائىرسى كۇپ ،پەقەت قانتاش ھادىسى يۇز بىرىپ جىسمانى زىيان مال مۇلۇك زىيىنى يۇز بەرسە سۇغۇرتا شىركىتى ئاۋال مەجبۇر سۇغۇرتا جاۋاپكارلىقى دائىرسىدە تۆلەيدۇ.ئىشىپ قالغان قىسمىنى شۇ ھادىسە تۇغرىسىدىكى ئىتىراپنامىدە مەسئۇليىتى بار دەپ كۇرسىتىلگەن كىشى مەسئۇليەت دائىرسدە تۆلەيدۇ.دىمەك ۋەقە چىققان ماشىندا مەسئوليەت بۇلسۇن بۇلمىسۇن سۇغۇرتا شىركىتى تۇلىشى كىرەك .ئۇچۇنچى شەخىس سۇغۇرتىسدا قاتناش ھادىسىدىكى ماشىنىدا مەسئۇليە ت بۇلغان چاغدا سۇغۇرتا شىركىتى شۇ مەسئۇليە بۇيىنچە تۇلەيدۇ،  قاتنا ش ھادىسىدىكى زىيانغا ئۇچرىغۇچى ئۇچۇنچى شەخىس مەجبۇرى سۇغۇرتىسىنڭ ئەڭ چۇڭ مەنپەتلەنگۇچىسى ھىساپلىنىدۇ.بۇ يۇل قاتنىشى بىخەتەرلىگى قانۇنىنىڭ ئادەمنى ئاساس قىلىش پىرىنسىپىنى چىقىش قىلغانلىقىنى كۆرسىتىدۇ. قاتناش ھادىسى يۇز بەرگەن ماشىنىدا مەسئۇليەت بۇلمىغان چاغدا سۇغۇرتا شىركەتلىرى دائىم مەجبۇرى سۇغۇرتا جاۋاپكارلىقى نىزامىدىكى ئالاقىدار بەلگىلىمەلەرنى ئاساس قىلىپ تۆلەم بىرىشنى رەت قىلىدۇ .قانۇنغا ئاساسلانغاندا نىزامدىن قانۇن يۇقۇرى ئۇرۇندا تۇرىدۇ، شۇڭا كۇپلىگەن سوت مەھكىمىللىرى قاتناش ۋەقەسى يۇز بەرگەنلا بۇلسا مەسئۇليەت سۇرۇشتۇرمەي سۇغۇرتا شىركىتىنى ئۇچۇنچى شەخىس مەجبۇرى سۇغۇرتىسى دائىرسىدە تۇلەشكە ھۇكۇم قىلىۋاتىدۇ، لىكىن بەزى سوت مەھكىمللىرى ئەكىسچە ھۇكۇم قىلىۋاتىدۇ.بۇ دۇلىتىمىزنىڭ ئالاقىدار قانۇنلىرغا خىلاپ .مەجبۇرى سۇغۇرتىنىڭ ئەڭ يۇقۇرى تۇلەش چىكى 122000 يۇەن بۇ بۇلۇپ پۇتۇن مەملىكەت بۇيىنچە بىرلىككە كەلگەن ئۇلچەم ھىساپلىنىدۇ. ئۇچۇنچى شەخىس سۇغۇرتىسنىڭ مەملىكەت بۇيىنچە بىرلىككە كەلگەن ئۇلچىمى ھەر خىل بۇلۇپ سۇغۇرتا سىتىۋالغۇچىلار خالىغىنى ئىختىيارى تاللاپ سىتۋالسا بۇلىدۇ ،سىتىۋالمىسمۇ بۇلىدۇ. ئۇچۇنچى شەخىس مەجبۇرى سۇغۇرتسىنى مەجبۇرى سىتىۋالىدۇ.ماشىنا ئىگىللىردە ھىچ قانداق تاللاش ھۇقۇقى يۇق ،ئەگەر سىتىۋالمىسا بۇ ماشىنىا يۇلغا چىقالمايدۇ.قىسقىسى ئۇچۇنچى شەخسى  مەجبۇرى سۇغۇرتىسى  مەسئۇليەت بۇلسۇن بۇلمىسۇن چۇقۇم تۇلىشى كىرەك .ئۇچۇنچى شەخىس سۇغۇرتسى مەسئۇليەت بۇلغان چاغدا تۇلەيدۇ.ھازىر ئالى خەلق سوت مەھكىمسى  قاتناش ھادىسى ۋەقەلەلىرىگە ئائىت ئىزاھات تۇزۇۋاتقان بۇلۇپ بۇ قاتناش ھادسىدىكى سۇغۇرتا مەسىللىرنى ھەل قىلىشىمىزغا ياخشى بىر قانۇنى ئاساس بۇلغۇسى .
