    بىلىك مۇنبىرىنىڭ خەيىر ساخاۋەت سەھىپىسىنى كۆرىۋىتىپ تاساددىبىيلىق ئىچىدە 2006-يىلى 14-يانۋار كۈنى يازغان "بىر يۈەن  پىلانى تەستىقىڭلارنى كۈتمەكتە "دېگەن ،بۇندىن نەق ئالتە يىل بۇرۇن يازغان تېمامنى كۆرۈپ شۇنچە خۇشھال بولۇپ،يېزىلغان ئىنكاسلارنى بىر بىرلەپ كۆرۈپ چىقتىم ،بۇ تېما يېزىلغانغا بۈگۈن مۇشۇ دەقىقىلەردە نەق ئالتە يىل بولۇپتۇ ، تېمىنى جەمىئىي 1مىلىيون 300مىڭدىن ئارتۇق تورداش كۆرۈپتۇ ..كىچىك چاغلاردا ل مۇتەللىپنىڭ "يىللارغا جاۋاب" دېگەن شئېرىدىكى "قىزلارغا قورۇق، ئەرلەرگە ساقال" دېگەن مىسرالارنى ئوقۇپ كۈلۈپ كېتىشكەن ئىدۇق ،2000-يىلى زامان ئاخىر بولىدىكەن دەپ 11-سىنىپقا چىقىشىمىزغىلا ھەممە تۈگەيدۇ دەپ بالىلار بىلەن تۈگىمەس خىياللارنى سۈرەتتۇق ،دەرھەقىقەت ۋاقىت ھەقىقەتەنمۇ تېز ئۆتۈپ مانا 30نىڭ بوسۇغىسىغا ئەكىلىپ قويدى پېقىرنى !
ۋاقىت ئۆز قانۇنىيىتى بىلەن ئۆتىۋەردى ،تالايلار ئارىمىزدىن كەتتى ،تالايلار بۇ قوشۇنغا قېتىلدى ،بالا ئىدۇق بالىلىق بولدۇق ،تور دېگەن نېمە بولۇپ كەتتى دەپ يۈرگەن ئىدۇق بىر چاغلار ،سائىتىنى 4 سومدىن تورغا چىقىپ، ماكانىم تورىدا ئارىلاپ ،مانا ئەمدى ئالدى 10 يىللىق تورخۇمارلاردىن بولۇپ قاپتۇق ،تورغا ئىلىنىپتۇق ،توردا ئۆتكەن ۋاقىتىمىزدا شئېر قوشاق ئەبجەش چۈشلەرنى كۆرۈپ يېزىپتىكەنمىز ،مانا ئالتە يەتتە يىل بۇرۇنقى تېمىلارنى كۆرۈپ شۇ چاغلاردىكى دۇنيا قاراش ۋەھاكازالارنى ئويلاپ ئۆزىمىزنىڭ چوڭ بولغانلىقىمىزغا ،يىللارنىڭ بىزنى قېرىتىۋاتقانلىقىغا شەكسىز ئىشىنىدىغان بولۇپ تىمسىقلاپ يۈرۈپتۇق يەنە شۇ توردا ...
"بىر يۈەن پىلانى "توغرىسىدىكى تەسەۋۇرنى بىلىك تورىغا ئېلان قىلغاندىن كىيىن ئوت يۈرەكلەر قوللاپ قۇۋۋەتلەپ ،چوقۇم قىلىشىمىز لازىملىقى توغرىسىدا ئىنكاسلار قالدۇرغان ئىكەن ،مانا ئەمدى 6يىللىق يىغىلغان پۇلنى ساناپ باقسام "بىر سوم پىلانى "پائالىيىتىگە 6يىلدا يىغىلغان پۇل 56مىڭ 288 سوم 91تىيىنغا يېتىپتۇ .
بۇ قۇرلارنى يازغانچە ھاياجانلىنىپ تارتمامنى ئېلىپ بۇ پىلاننىڭ ئىئانە تىزىم جەدىۋىلى ،ئىسپاتلار چاپلانغان خاتىرە ،رەھمەت خەتلىرى ۋە نەق مەيدان ئىسپات رەسىملەر سېلىنغا ن ئالبومنى ئېلىپ كۆرۈپ چىقتىم ،سەنەۋەر ،ئابلەت ئاقسۇ ،روسى، ھاماقەت ،مۇرادىل، ھەۋەسكار ،مەلىكىزات ،سەيپىدىن شاڭخەي ،شەمشىدىن شىنجىن ......تالاي تالايلىغان ساخاۋەتچىلەرنىڭ نامى ئالتە يىلغا يېزىلغان ئالتە دانە ئىئانە پۇلى تىزىم دەپتىرىگە يېزىلىپ تۈپلىنىپتۇ ،2012-يىللىق تىزىملاش جەدىۋىلىمۇ ،مىننەتسىز ،ئەلسۆيەر ساخاۋەتچىلەرنىڭ قىلغان 1 سوم ،20،سوم 500 سومدىن قىلغان پۇللىرى بىلەن باشلىنىپ كېتىپتۇ ..ھەق گەپ شۇكى سىز،مەن  ۋە ئۇنىڭدەك تالايلارنىڭ بۇ ئىشنى كىچىك كۆرمەي ،تېرىكمەي  بانكىلاردا ئۆچرەت تۇرۇپ ،تور بانكىلىرىدا ياكى پۇل بەرگۈلەردە ئېرىنمەي پۇل يوللاپ بېرىشلىرى بىلەن 6يىلدا بىر سوم بىر سومدىن يىغقان پۇلىمىز 56مىڭ سومغا يېتىپ ئۇيغۇر ،مۇڭغۇل،خەنزۇ ،قازاق مىللىتىدىن بولغان تالايلىغان ياردەمگە مۇھتاج ئوقۇغۇچىلارغا ،بىمارلارغا ،يوقسۇللارغا....مېھىر -مۇھەببەت يەتكۈزۈپتۇق ،خاتىرە دەپتىرىگە قارىدىم،مۇراجەتنامىلىرىگە پۇل يىغىپ بېرىلگەن كۆپ ساندىكى بىمارلار ئاللىقاچان بىزدىن ئايرىلدى ،لىكىن ئۇلارنىڭ ئۇرۇق تۇغقانلىرىنىڭ مەدىتى ،بىلگەنلەرنىڭ دۇئاسى ھېلىھەم سىز ۋە مېنىڭ ئەتراپىمدا چۆرگىلەپ يۈرەر ئەتىمالىم،بىز بۇ ئېغىر بىمارلارغا پۈتۈنلەي ساقايغۇدەك ياردەم قىلالمىدۇق ،لىكىن ئۇلارغا ئاز بولسىمۇ  ئۈمۈت ،ئارزۇ ئاتا قىلالىدۇق ،سۆيۈندۈرەلىدۇق ،كۆڭلىنى ئاۋۇندۇرالىدۇق دەپ ئويلايمەن ..
6 يىللار بۇرۇن ماگىستىرلىق ئوقۇشىغا كىرگىنىدە ئازغىنە ياردەملەرنى قىلغان ئىدۇق بىر قىزغا ،ئۇ مۇشۇ تاپتا بېيجىڭدا دوكتورلۇققا  تەييارلىقتا ئوقۇپ بىزگە بىز ياردەم قىلغاندىن نەچچە ھەسسە ئوشۇقلاپ پۇل ئەۋەتمەكتە ،
-11سىنىپتا مەكتەپ بويىچە 1-بولۇپ ئالىي مەكتەپتە ئۆتكەن بىر قىزنىڭ ئالىي مەكتەپ ئىمتاھان پۇلىنى تۆلەپ بەرگەن ئىدۇق ،بېيجىڭدا ئوقۇۋاتىدۇ ھازىر ،ھەر ئايلىقى 300 سوم  تۇرمۇش پۇلى بېرىۋاتىمىز ،پات ئارىدا ئوقۇشىمۇ پۈتىدۇ ....
چاپچالدىكى مەلۇ م بىر يېزا مەكتىپىگە كىيىم كېچەك ۋە پۇل ئېلىپ بارىدىغانلىقىمىزنى بىلگەن ئۈرۈمچىدىكى بىر قىسىم سىتۇدىنىتلار كەچلىك ئاپتوبوسلاردىن ئۆزلىرىنىڭ كىيىم-كېچەكلىرى ،كىتاپلىرىنى ئەۋەتىپ بېرىشكەن ئىدى 6يىللار مۇقەددەم ،يۇيۇپ تارالغان ،ئامالنىڭ بارىچە قايتا پاكىزلانغان شۇ كىيىملەرنى كىيگەن بالىلار بىزنى مەڭگۈ ئۇنۇتمايدۇ ئەتىمالىم، بورتالا شەھرىدىكى يىراق بىر يېزا باشلانغۇچ مەكتىپىگە كىتاپ ئىئانە قىلىشىپ بالىلار ئوقۇشلۇقى ۋە دەرسلىك پايدىلانما كىتاپلىرىدىن 400 پارچىدىن ئارتۇق راسلاپ كىچىك بىر قىرائەتخانا قۇرۇپ بېرىۋىدۇق ،باش بەتلىرىگە "بىر سوم مېھرىبانلىقى "ئىئانىسى دېگەن خەتلەر يېزىلغان ئەشۇ كىتاپلارنى ئوقۇغان دېھقان بالىلىرى بىزنى چوقۇم ئۇنۇتمايدۇ ۋە پات يېقىندا بۇ ساخاۋەت سېپىگە قوشۇلغۇسى  ....
قايسى بىر كۈنى ئىشتىن ياناشىمدا تاكسى ساقلاپ خېلى ئۇزۇن ساقلاپ كەتكەن ئىدىم ،ئارىدا بىر پىكاپ كېلىپ توختاپ چىقىڭ دېدى ،پاراڭلىشىپ ماڭدۇق ،ئۆيۈمنىڭ ئالدىغا كېلىپ قارا تاكسى دەپ بىلىپ پۇل تەڭلىسەم،مەن سىزنى بىلىمەن ،سىزدىن قانداقمۇ پۇل ئالاي ،ئەسلى كىرا قىلمايتىم ،سىزنى تونۇپ ماشىنىنى توختاتتىم ،بولدى چۈشۈپ كېتىڭ دەپ كېتىپ قالدى ..قانچىلىك ھاياجانلانغانلىقىمنى ئۆزۈم بىلىمەن ،ۋە شۇ دەقىقىلەردە "بىر سوم پىلانىنى "بىر ئۆزۈڭ بولسىمۇ داۋام ئەمىسەڭ ھەرگىز بولمايدۇ دەپ خىتاپ قىلىپ كەتتىم ....
ئىناق جەمىئەت بەرپا قىلىش ،مىللەتلەر ئىتتىپاقلىقىنى ئىلگىرى سۈرۈش ۋە ئاسراش ئۈچۈن قىلغان بۇ مىننەتسىز ئىئانىلەر،كارتىدا ئۇزۇنمۇ تۇرماستىن بارىدىغان يېرىگە بارماقتا ،پۇل ئاز بولسىمۇ لىكىن بۇ ئىشنى توختىتىپ قويمايلى تورداشلىرىم ،6يىل بۇرۇن بۇ مېھرىبانلىق پائالىيىتىنى باشلار ۋاقىتتا مۇراجەت قىلغىنىمدەك "بىرىمىزگە كۆپ بىلىنمەيدىغان 1 سوم ياكى 10 سوم پۇل 10مىزنىڭ يانچۇقىدىن بىرلا ۋاقىتتا چىقسا كۈنلىرى 3ۋاق ناق يەپ تۇرۇق يەنە ئاجايىپ تىرىشىپ ئوقۇۋاتقان سىتۇدىنىتلىرىمىزنىڭ بىر ئايلىق تاماق پۇلىنى ھەل قىلىپ بېرەلەيمىز ،ھەر بىر سورۇنلىرىمىزدا ئىككى تەخسە قورۇمىنىڭ پۇلىدىن كېچىپ يىغىپ قويساق ھەر ئوقۇش مەۋسۇملىرى نەچچە ئون سوم پۇلنى تۆلەشكە قىينىلىۋاتقان ئونلىغان  باشلانغۇچ مەكتەپ بالىلىرىنىڭ پۈكلەنگەن قەددىنى رۇسلىيالايمىز ،كېسەل كارۋاتلىرىدا ئىڭراپ ،قېنى ئىنسانلار ماڭا قىلىدىغان ياردىمىڭلار بارمۇ دەپ ھاياتقا بولغان ئوتتەك ئۈمۈت ئىستىكىدە خىرە يىلىنجاۋاتقان مىسكىن قەلىبلەرگە يالقۇن بېرەلمىسەكمۇ ئىللىق سېزىم بېرەلەيمىز .
سىز ۋە مەن مىڭ سوملاپ كېيىملەردىن يىلىغا  نەچچىنى يىرتىپ،قېپى 10 سوملۇق تاماكىلاردىن كۈنىگە قاپلاپ چېكىپ 200 گىرامى 30-40- سوملۇق  ھاراقلارنى ئىچىپ يۈرگەنلىرىمىزدە ئۇنۇتمايلىكى ئەتراپىمىزدا ئېيىغا 200 گىرام گۆش يېيەلمەي ئۆتىۋاتقان يوقسۇل قوشنا قۇلۇملىرىمىزنى ،ئۆيلىرىدىن ئايلىقىغا 50 سوملاردىن كەلگەن تۇرمۇش پۇلىغا 3ۋاق نان بىلەن تىرىشىپ ئاجايىپ نەتىجىلەر يارىتىۋاتقان ئۈمۈتۋار سىتۇدىنىتلىرىمىزنى ،مەۋسۈملۈكى 100يۈەنگىمۇ يەتمەيدىغان ئوقۇش ۋە پارچە خىراجەتلەرنى تۆلەلمەي سىنىپلاردا بويۇن قىسىپ قېلىۋاتقان ناتىۋان بالىلىرىمىزنى .........
