مانا مەن كەلدىم دەپ ،مۇنبەر ئەھلىگە شۇنداق سالام قىلاي دەپ تۇرۇشۇمغا ،ماۋۇ ئەلتىكىن شارپامنىڭ ئارقىدىن تارتىپ يۈرۈيدۇ. ۋۇي نىمە ئىش بولدى؟ باشتا سالام ئاندىن كالام دىسەم رەنجىپلا كەتتى ،نىمىشقا دەپ سوراۋاتامسىلەر؟ نىمىشقا بۇلاتتى....مىنى كالىغا ئوخشاتمىسىلا كۈڭۈلخان !نەچچە ۋاقىت ياقى مۇنبەردە يوقاپلا كەتتىڭىز ،ئەمدىلىكتە ماۋۇ مۇنبەرنىڭ ئىشىكى ئالدىدىلا ئايرىم پاراڭلىشاي دىسەم،ئەجەپ مىنى  كالىغا ئوخشىتىۋاتىسىزغۇ دەيدۇ.قانداق قىلاي؟ مۇشۇنداق ئورۇنسىز رەنجىپ يۈرۈشلەردىن جاق تويغان چاغدا،مۇشۇ ئەلتىكىن بىكاردىن ئۆزىنى كالىغا ئوخشىتىۋالغىنى قىززىق بولدى دىسە.ئامال يوق ئۇنىڭغا بىرمۇنچە چۈشەندۈردۈم ،چۈشەندۈرۈش جەريانىدا ئۇنىڭدىن بىرمۇنچە مۇش-پەشۋالارنى ئويناپ كەتتۇق دىسە.ئاخىرى

ئۇ، بۇ سۆزۈمنىڭ نىڭ بىر تەمسىل ئىكەنلىكىنى بىلىپ ،ئاندىن  يەرگە سەكسەن گىرادۇس يىقىلىپ تۇرۇپ مەندىن  كەچۈرۈم سورىدى.ئۇف.........بايامنىڭ ياقى مانا مۇشۇ ئىش بىلەن بۇلۇپ، سىلەردىن ئەھۋالنى كىچىكىپ سورايدىغان بولدۇم .

ئەسسالامۇ -ئەلەيكۇم !مۇنبەردىكى كۈلكە چىراي -خۇشچاقچاق خانىم قىزلار،چاخچاققا تەڭكەش بولغۇچى يۇمۇرىستىك يىگىت-ئەپەندىلەر  ھەممىڭلارنىڭ كەچلىگى خەيرىلىك بولسۇن! تىمامنى ۋاقتىدا ئۇقۇيالماي كۈندۈزى كۆرۈشكەمۇيەسسەر بولغان دوستلارنىڭ كۈندۈزلىگى خەيرىلىك بولسۇن!

مەن ئەزەلدىن كۈلكە خۇمار مۇنبەرنىڭ بىر ئەزاسى ئىدىم، مۇشۇ تاڭنۇر ماڭا بىرقىتىم ناھايىتى ئاچچىقلىغان ھالدا :كۆڭۈلخان ، مۇنبەردە  باشقىلارغا تۇلا چاخچاق قىلىۋىرىپ ،مۇنبەرنىڭ قىز- ئوغۇللىرىنىڭ قاتتىق  نارازىلىقىغا ئۇچىراپ كەتلە ،شۇڭا  ئەمدى دەرھال توختاپ قالسىلا دەيدۇ. ياق دىسەم ،كومپىيۇتۇرنىڭ ئەيىنىگىدىن بىگىز قۇلىنى تىقىپ، قاڭشارلىق بۇرنۇمنى نۇقۇشلاپ پاناق قىلىپ قويىدىغاندەك قىلىدۇ، شۇنىڭ بىلەن بىرمەزگىل قايناقسۇغا سوغۇق سۇنى قۇيغاندەك تىمتاس بۇلىۋالدىم.كىيىن مالچى دوستۇم نەسىرۇللادىن ئاڭلىسام ماڭا تاڭنۇر دىگەن سەت تاز ،شۇنداق قوقاق سىلىپ قۇيۇپ ئۆزى مۇنبەرنى ئىگەللەپ كىگىز -كۆرپىسىنى، مۇنبەرنىڭ قاق ئوتتۇرسىغا سىلىۋىلىپ مۇنبەردىكىلەرنى قايماقتۇرۇپ ئولتۇرۇپتۇ.

شۇنىڭ بىلەن دەر غەزەپ بۇلۇپ مۇشتۇمنى تۈگۈپ ،ئۇنىڭ بىلەن بىر سۇقۇشىۋىتىپ مايماق ئېغىزىغا ئارتۇق چىقىۋالغان چىشىنى چىقىۋىتىپ مۇنبەرگە كىرىشىم.ھەم ئەلتىكىن بىلەن تۇلا مۇنازىرلىشىپ

يۈرۈپ ۋاقتىدا كىلەلمىدىم كەچۈرۈڭلار!

تۆنۈگۈن،  مۇنبەرگە كىرىش ئۈچۈن بەك ئۇزاق تەييارلىق قىلىپ ئولتۇرسام ،زامانى ھەسىرەپ ھۈمىدەپ مۇنبەرنىڭ ئايرىمخانىسىغا پالاقشىپ يۈگۈرۈپ  كىرىپتۇ.ۋاي كۆڭۈل........ ئۆزۋاقتىدا مىنى ئىشىك قىسىۋىلىپ كۆكۈرۈپ قالغان دەپ مۇنبەرگە يايغانتىڭىز، ئەمدى بۇ قىتىم  مىنى ئۇنداق دىمىسىڭىز..... چۈنكى مۇنبەردىكى قىزلار مىنى ياراتماي قالمىسۇن.

كىلەر قىتىم مىسىران دىگەن پىخسىق، يەنە بىلەت تارقاتسا ھىچكىم مەن بىلەن ئۇيۇن كۆرمەي قالمىسۇن دەيدۇ.

ئۇنىڭ ھەسىرەت بىلەن شۆلگىيى  ئېقىپ پۇتىغا چۈشكىچە ،يالۋۇرۇپ سۆزلەۋاتقان سۆزىدىن مەن ئەمەس

تام بولسىمۇ تەسىرلىنەر شۇنىڭ بىلەن ماقۇللا دەۋەتتىمغۇ.

بۈگۈن سىلەرنى بۇيەرگە يىغىشىمدىكى مەخسەت ، مۇنبەرنىڭ يان تەرىپىگە يامان ئېسىل ئاشخانا ئاچماقچى بۇلىۋاتىمىز ، شۇڭا ئىشسىز دوستلارنىڭ كىرىپ بىز بىلەن بىرگە خىزمەت قىلىشىنى قارشى ئالىمىز!

بىرچاغلاردا مۇشۇ مۇنبەردە شاقۇر ، نازۇك بىرقانچىمىز ئاشخانا ئاچقان ئىدۇق.نازۇك شۇ ۋاقىتتا ئامباردىكى ئىككى تۇڭ ماي بىلەن ، بىر كىلو گۆشنى ئوغۇرلاپ قىچىپ  مۇنبەرگە ئاياغ باسماس بولغان ئىدى. شۇنىڭ بىلەن <كۆكنۇر>شاقۇر ئىككىمىز ،ئاشخانىغا ئىشەنچىلىك ئادەم ئىلىش ئۈچۈن نەسىرۇللانى چاقىرتقان ئىدۇق.

ئارلىقتا، مىسىران بىزنىڭ شۇنچە بىيىپ كەتكەنلىكىمىزنى كۆرۈپ،ئىچىگە بىرسىقىم  لازا قۇيىۋەتكەندەك ئىچىشىپ ئىجارىنى ئۆرلىتىۋىتىپ ئاشخانىمىز تاقالغان.بىز بىرمەزگىل يۇقالغان ياقى ،باشقا دۇكانلاردىن پايدا ئالالماي جىددي تىلگىرامما يوللاپتۇ، تىزدىن كىلىپ ئاشخاناڭلارنى ئىچىڭلار! ئەرزان ئىجارە بىرىمەن دەپ، شۇنىڭ بىلەن ئامالسىز قايتىپ كەلدۇق.

بۈگۈن مۇشۇ مۇناسىۋەت بىلەن نازۇك ئوغرىدىن باشقا، مىسىرانىم - مۇنبىرىدە يىقىندا ئىچىلماقچى بۇلىۋاتقان  ئاشخانىمىزغا خىزمىتى يوق دوستلارنىڭ كىلىپ ئىشلىشىنى ئۈمۈت قىلىمىز.

شاتئاي نەچچە كۈن بۇرۇن تىزىملاتقان ئىدى،ئەمما  بۇيى دەششەت پاكاركەن شۇڭا  ئويلىشىۋاتىمىز.

كۆزۈمنىڭ بىر كۆزى بىردەم ئالدىغا- بىردەم ئارقىغا قاراپ قالىدىكەن ئۇنىمۇ ئويلىشىۋاتىمىز. بۇ ئىككىسى ئەڭ بولمىغاندا، ئوچاق بىشىنى تازلاشقا  بولسىمۇ ئەسقىتىپ قالار.

مارشال ھازىرچە ئاشخانا مۇلازىمەتچىلىكىگە ئاران ئۆتتى ، ئۆزى سىمىز بولغان بىلەن ،

سول پۇتى بىلەن ئوڭ قۇلى خىلىلا ئۇزۇنكەن.قاراپ باقساق  مۇلازىمەتچىلىكنى بىر تاقلاپلا  قاملاشتۇرغۇدەك

دەپ ئويلاپ قالدۇق.

مۇشۇ ھۆكۈمىران دىگەن توڭ قاپاقنى، ئاشخانا ئىشىگىدە ئۆتكەن كەچكەننى  بوم ئاۋازى بىلەن  توۋلاپ چاقىرىپ تۇرۇشقا ئالدۇق.

گۈلبىكەنى بولسا ،مەخسۇس  بەلشىر  ۋە چۈچۈرە  مانتا  تۈگۈشكە ئالدۇق . ئەسلى ئالمايلى دىگەن، چۈنكى بۇ قىز  ئولتۇرسا-قوپسا بەك تۇلا نەرسە يەيدىكەن.شۇڭا مۇشۇ  خام قىيمىنى بولسىمۇ  يىيەلمەس دەپ شۇنىڭغا ئورۇنلاشتۇردۇق. بۇنداق قىلىشىمىز ،ئاشخانىمىزنىڭ تىخمۇ  تەرەققىي قىلىشى  مىسىرانىمغا مەبلەغ  توپلاش ئۈچۈندۈر! مۇسلىنە مۇمامنى  بولسا مەخسۇس لەغمەن تارتىشقا تەيىنلىدۇق، مۇسلىنە مۇماي خىلى شوخ قىرىغان مۇماي بولغاچقا لەغمەن دىگەننى ئويناپ تۇرۇپ تارتىدىغان بولسا، بىر ئۇچىنى مۇنبەرنىڭ ئۇ بىشىدىن- بۇ بىشىغىچە تارتىپ خىرىدارلارنى قاتتىق جەلىپ قىلامدىكىن دەپ ئويلاشتۇق.

دەتمەننى بولسا ، ئاشخانىدىكى چىنىلەرنى يۇيۇشقا قۇيۇپ بۇلۇپلا  ئۆزگەرتىۋەتتۇق.نىمىشقا دەمسىز؟ئۇنىڭ يۈرىكىدىلا ئەمەس جىگىرىدىمۇ دەردى باكەن، شۇڭا ئاشخانىنىڭ ئارقا ھويلىسىنى تازلاشقا بىكىتتۇق. ئۆل بىلەن نۆلنى ئاشخانىنىڭ ئايرىمخانىسىدا، خېرىدارلارغا <<خەجلەۋاتقان پۇلىڭىزنىڭ كەينىگە نۆلدىن جىقىراق چاپلاپ>>  تاماق يىمىسەڭلار، چۇقۇم ئاشخانىدىن چىقماي تۇرۇپلا ئۆلىسىلەر

دىگەن لەپەرنى ئورۇنلاپ ،خىرىدارلارنى ئەلا مۇلازىمەت بىلەن كۈتۈشكە بىكىتىپ بولدۇق.

بۇلاقنى، مەخسۇس ئاشخانىغا سۇ تۇشۇپ بىرىشكە  ئالماقچى بۇلىۋاتىمىز.

چەشمە زۇلال بىلەن جەسۇرمەننى،زادى  قايسى ئىشقا قۇيۇشنى پەقەت  بىلەلمەيۋاتىمىز. چۈنكى بۇلار، كەلگەن خېرىدارلانىڭ ھەممىسىگە تۇلا قارىۋىلىپ ،كاچاتقا تەتۈر شاپىلاق يەپ كىتەرمىكىن دەپ ئەنسىرەپ قالدۇق.

كاۋا ئاشخانىدا مەخسۇس كۈتكۈچىلىك قىلاي دەپ تۇرىۋىلىپلا بارىدۇ، ئالدىدا دىيەلمىدۇق كەينىدە بولسىمۇ دەۋالاي  بەك مىس...مىس...مىس يۈرۈدىكەن قاملاشتۇرالارمۇ؟ئۇنىڭ ئورنىغا گۈلسۈمنى قۇياي دىسەك ئۇ تىخىمۇ بولمايدىكەن،ھەربىلىكتە تۇرۇپ كەلدىم دەپ ھەممە ئادەمگە مۇش كۈتۈرۈپ چىقارمۇ دەيمەن. ئوتتۇرلۇقتا  مەن خەقنى ئاجىرتىمەن دەپ، گالاچ-گۇلۇچ  يەپ كەتسەم پەشۋا دىگەننى ئاۋاق بەدىنىم تىشىلىپ كىتەر ھەقاچان!

دوپپىچى ،ئاكاش ۋە ئاپتاپلار  دۇكاندا ئوچاققا  مەخسۇس، ئوتان يېرىش ۋە قالاش ۋەزىپىسىنى ئۆتەيمىز-

دەپ بىرسىنى بىرسى ئۇرۇپ يۈرۈپ ،بۇ خىزمەتنى تالىشىپ تۇرىۋىلىپ ئامال يوق ئۈچىلىسىنى ئالدۇق.


ئەمدى مۇشۇ ئاشخانىدا ئىشلەشنى خالايدىغانلار بولسا ،ۋاقتىنچە تىزىملىتىشقا ئويغان ئوغۇل بىلەن شاتئاينى مەسئۇل قىلىپ قويدۇق. مۇشۇ ھويلىغا كىرىپ تىزىملىتارسىلەر! ئاچچىقى زەھەردەك يامان،قۇلىدىن ئىش كەلمەيدىغانلار تىزىملاتمىساڭلارمۇ بۇلىدۇ. تىزىملىتىش جەريانىدا ئويغان ئۇغۇلنىڭ  ئىڭىكىنىڭ سۈڭىكى يىرىلىپ شاتئاينىڭ بۇرنى چۇۋۇلۇپ  كىتىشنى قەتئي خالىمايمىز.

ئۇنىڭ ئۈستىگە مەن مۇنبەردىكىلەرنى بەك تۇلا چۈشەپ قالىمەن ، شۇڭا ئاشۇنداق ئاچچىقى يامانلار چۈشۈمگە

كىرىپ قالسا ئويغىنالماي قىلىشىمدىن ئەنسىرەيمەن.<ھى....ھى...ھى>


شۇنىڭ بىلەن، بۈگۈنكى مىسىرانىم ئاشخانىسىغا ئادەم ئېلىش ئۇقتۇرۇشمنىڭ مەزمونى تۈگىدى.ئاخىرىدا مۇنبەر ئاشخانىسىنى قايسى خىل ئۇسۇلدا باشقۇرۇش تۈزىمىنى قويساق بۇلار ؟ سىلەر پىكىر بىرىپ باقامسىلەر يا.
