                                  مۇھەببەتمۇ شۇنداق بولامدۇ؟   

                                        (*1*)
                        چىن سۆيگۈنىڭ پراقى باركەن،
                        يېقىن كۈننىڭ  يىراقى باركەن.
                         ئاشىقلارغا ئاخىرقى ھېساپ،
                         ئۇيقۇسىز تۈن تەنھالا  ياركەن.
   
                                        (*2*)
                         سۆيگەنىدىم بىر قىزنى دىلدىن،
                         ئازاپ يىراققا قاچقان ھەم بىزدىن.
                         شۇنداق كۈلۈپ ياشىدۇق مەزگىل،
                         ئارمان يوقتى سۆيۈشنى ئۇندىن.

                                         (*3*)
                         بەخىت تىلەپ كەتكەن بەزىلەر،
                         ئىككى دۇنيا بەخىتلىك بول دەپ.
                         ھەم قىزغىنىپ قالغان بەزىلەر،
                         كوچىلاردا پۇتلىشىپ ئۆل دەپ.

                                         (*4*)
                         قارغىساڭمۇ پەرۋايىم پەلەك،
                         سۆيگەن قىزىلا  ماڭا يار_يۆلەك.
                         ئىت قارغىسا پىتمۇ ئۆلمەپتۇ،
                         سىنىڭ ئۇ گىپىڭ پۇتۇمغا پېتەك.

                                        (*5*)
                         دەپمۇ قۇيۇپ  كۈلەتتىم بەزى،
                         ئاچچىقىمدىن ئۈلەتتىم بەزى.
                         ھەم ئۇرۇشۇپ مۇشت پەشۋا ئېتىپ،
                          تاياقمۇ يەپ يۈرەتتىم بەزى.

                                         (*6*)
                          ئۆتۈپ كەتتى كۈنلەر ھەم يىللار،
                          ئاشىق باھارىن ئەمدى ياز چىللار.
                          تاتلىق كۈنلەر بىر_بىرلەپ يوقاپ،
                          يازمۇ ئەمدى كۈز پەسلى چىللار.

                                          (*7*)
                          ئاشۇ تاتلىق كۈلكىلەر ئازلاپ،
                          ئاشىق_مەشۇق بىر بىرىن تىللاپ.
                          خۇشال كۆزلەر ھىجرانلىق يىغلاپ،
                          ھەتتا كۆرۈشمەي بىر_بىرىن ئايلاپ.

                                          (*8*)
                           يوقالغاندەك يۈرەكتىن شاتلىق،
                           ھەم يوقالغان چىن سۆيگۈ پاكلىق.
                           كەلدى كۈنلەر بىر_بىرىن ئاچچىق،
                           شۇنداق يوقالدى قەلىپدىن ئاقلىق.

                                          (*9*)
                           قىزمۇ  قاقشار ئۆكسۈپمۇ يىغلاپ،
                           ئاھ ئۇرار ھەم تەڭرىگە تىزداپ.
                           مەن غېرىپمۇ ئۆزۈمنى تىللاپ،
                            تۈن قوينىدا ئۈن سېلىپ يىغلاپ.

                                          (*10*)
                             ئۆتۈپمۇ كەتتى مەنىسىز كۈنلەر،
                             ئۇيقۇسىزلا ئۆتمەكتە تۈنلەر.
                             ھەم قۇرۇماس يامغۇرلۇق كۆزلەر،
                             دوست يارەنلەر بىزلەرنى بەزلەر.

                                           (*11*)
                             بۇ تەقدىرنى قىزدىن كۆرەيمۇ؟؟،
                             ياكى بەڭۋاش مەندىن كۆرسۇنمۇ؟؟.
                             مەن پىراقتا ئاھ دەپ ئۆلەيمۇ؟؟،
                              قىز ئازاپتا  جان دەن ئۆلسۇنمۇ؟؟.
  
                                           (*12*)
                              بۇ تەقدىرنى كىمدىن كۆرىمەن؟؟،
                              بۇ قەرىزنى كىمدىن ئالىمەن؟؟.
                              تەقدىر شۇنداق بولغاچقا مەنمۇ،
                                 يالغۇزلۇقتا  تەنھا قالىمەن.

                                           (*13*)
                                ياشلىقتا ھەر باشقا كىلەر مۇھەببەت،
                                ئەمما ياخشىلىق كەلمىكى تەس ئاقىۋەت.
                                ئەگەر بىراۋ تاشلىۋەتسە سەندىن ۋاز كىچىپ،
                                سەنمۇ ئەگەشمە ئىككىلەنمەي تاشلىۋەت.

                                          (*14*)
                              سۆيگۈ دىگەن ئېرىشىش ئەمەس،
                              ۋە ياكى جاندىن كېچىشمۇ ئەمەس.
                              ئۇ دىگەن ياشلىقتا بىرخىل ئىشتىياق،
                              ھەر باشتا باربىرخىل تەبىئىي ھەۋەس.

                                           (*15*)
                              يازايمۇ يازمايمۇ يەنە بىر كۇبلىت،
                              ياز دەيدۇ يېنىمدا قېرى مويسىپىت.
                              ياق بولدى يازمىغىن دەيدۇ بۇلۇڭدا،
                              چاقى ھەم رولى يوق مايماق ۋېلسىپىت.
