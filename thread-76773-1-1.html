يوللىغان ۋاقتى 2012-4-21 10:19:39 PM |ھەممە قەۋەتنى كۆرۈش |
مىلادى 1864-يىللىرى مانجۇلارغا قارشى خەلق قوزغىلاڭلىرى كەينى-كەينىدىن يۈز بەردى،ئەڭ ئاۋال كۇچادا راشىدىن خوجا باشچىلقىدا،كىيىن قەشقەردە سىدىق تۆرە باشچىلقىدا (بۇ كىشى قىپچاق بولۇپ مانجۇلارنىڭ قول ئاستىدىكى ئەمەلدار ئىدى.قىپچاقلار قىرغىزلار بىلەن ئىنتايىن يىقىن بولۇپ تۈركىي مىللەتتۇر!) ئاندىن خوتەندە ھاجى پادىشاھ باشچىلقىدا،ئىلىدا ئابدۇرۇسۇل شاغىبەگ،مەزەمزات تەيجى،سادىر پالۋان باشچىلقىدا،چۆچەكتە تۇڭگان ۋە ئۇيغۇرلار،ئۈرۈمچىدە داۋات خەلىپە(تۇڭگان،ئىسمى تۇدىلىن) باشچىلقىدا شۇ دەۋىردەياشىغان ئافاق خوجا ئەۋلادلىرىدىن بىرى بۇزرۇك خوجا توغۇرسىدا توختىلىشتىن بۇرۇن كىيىن بۇزرۇك خوجىدىن ھوقۇق تارتىۋىلىپ تەخىتكە چىققان ياقۇببەگ توغىرسىدا توختىلىمىز:-قەشقەردىكى قوزغىلاڭلار ئارقىلىق مانجۇلار يوقۇتۇلۇپ بەزىسى تەسلىم بولغاندىن كىيىن قەشقەردە بىر مەيدان ھوقۇق تالىشىش كۆرەشلىرى باشلىنىش ئالدىدا تۇراتتى،يەنى سىدىقبەگ قاتارلىقلار يۇرت چوڭلىرى تەرىپىدىن مەسخىرە قىلىنىپ يۇرت سوراشقا سالاھىيتى يوق دەپ كەمسىتىلدۇ!(چۈنكى سىدىق بەگ نەسەبى خوجىلاردىن ئەمەس ئىدى،ھەم قىپچاق بولغىنى ئۈچۈنمۇ بولسا كىرەك)شۇنىڭ بىلەن ھوقۇق تالىشىش كۆرىشىنىڭ ئالدىنى ئىلىش ئۈچۈن قوقەنت(ھازىرقى ئۆزبىكىستان،قىرغىزىستان ئەتراپىدىكى خانلىق)تا تۇرۋاتقان خوجا ئەۋلادى يەنى جاھانگىر خوجىنىڭ ئوغلى بۇزرۇك خوجىنى(جاھانگىر خوجا ئافاق خوجا ئەۋلادىدىن بولۇپ شۇ دەۋىرلەردە خوجىلار ئۇيغۇرلار ئارىسىدىكى ئەڭ يۇقىرى مەرتىۋىگە ئىگە كىشىلەر ئىدى،جاھانگىر خوجا يەتتە قىتىم قەشقەرگە بىسىپ كىرگەن،بىر قىتىم قەشقەرنى ئىشغال قىلغاندا قوقەندىتكى ئوغلى بۇزرۇك خوجا تۇغۇلغان شۇنىڭ بىلەن جاھانگىر خوجا:_قەشقەرنى <بۇزرۇك>(ئۇلۇغ،كاتتا دىگەن مەنىدە)دەيدۇ ئوغلۇمنىڭ ئىسمى بۇزرۇك بولسۇن دەپ ئاشۇ ئىسىمنى قويغان ئىكەن.بۇ 1826-يىللارغا توغرا كىلىدىغان بولۇپ مەشھۇر ئۇيغۇر قىزى نۇزۇگۇم مۇشۇ قوزغىلاڭغا قاتناشقان ئىكەن،كىيىن جاھانگىر خوجا تۇتۇلۇپ قىلىپ بىيجىڭدا كاللىسى ئىلىنغان) قەشقەردە ھاكىمىيەت ئىشلىرىنى بىللە(ئەمەلىيەتتە خان بولۇشقا) باشقۇرۇشقا تەكلىپ قىلىدۇ.شۇ چاغدا قوقەند ھۆكۈمدارى خۇدايارخان يىقىنى ياقۇببەگنى باشقىلارنىڭ زىيانكەشلىك قىلىشىدىن ساقلاش ئۈچۈن(ياقۇببەگ سىر دەرياسىنى سۈيى بىلەن قوشۇپ روسلارغا قىرىق مىڭ تىللاغا سىتىپ ھاكىملىق ۋەزىپىسىدىن ئىلىپ تاشلانغان) بۇزرۇك خوجىغا قۇشۇپ قەشقەرگە ئەۋەتىدۇ،يىرىم يولغا بارغاندا موللا ئالىمقۇل دىگەن كىشى بۇزرۇك خوجىغا ياقۇببەگدىن ئىھتىيات قىلىشىنى تاپىلايدۇ،شۇنىڭ بىلەن ياقۇببەگ بىر يىل ئۆتكەندىن كىيىن بۇزرۇك خوجىنى قورچاق قىلىپ قويۇپ ھاكىمىيەت رەقىبلىرىنىڭ ھەممىسىنى يوقاتقاندىن كىيىن(خوتەن قوزغىلاڭچىللىرىنىڭ رەھبىرى ھاجى پادىشاھ <ئۆز ئىسمى مۇپتى ھەبىبۇللا> نى قەستلەپ ھىيلە بىلەن ئۆلتۈرىدۇ،ۋە ئۆزىگە قارشى ھىققان خوتەن خەلقىدىن 50 مىڭ كىشىنى قەتلىئام قىلىدۇ،راشىدىن خوجا،سىدىقبەگ قاتارلىقلارنىڭمۇ قەستلەپ ھىيلە بىلەن يوقۇتىدۇ،شۇنداق قىلىپ قوزغىلاڭ مۇشۇ ياقۇببەگنىڭ تىزگىنىدە قالىدۇ.(ئىلىنى ئىشغال قىلىش ئۈچۈن ئەسكەر ئەۋەتكەندە روسلار مەنچىڭ خاندانلىقىنىڭ ھاۋالىسى بويىچە ئىلىنى بىسىۋالىدۇ)بۇزرۇك خوجىنى ئاقساراي دىگەن جايغا نەزەربەنىد قىلىدۇ.(ئۇنىڭدىن نۇرغۇن يىللار ئىلگىرى ئافاق خوجا سەئىدىيە خانلىقىنىڭ ئاخىرقى ھۆكۈمدارى ئىسمائىلخاننى ئاشۇ جايغا نەزەربەند قىلغان ئىكەن!)بۇزرۇك خوجىنىڭ مۇشۇ ۋاقىتتىكى نەزەربەنىد ئەھۋالىنى چۈشىنىش ئۈچۈن <<بەدۆۋلەت>> رومانىنىڭ شۇ قۇرلىرىغا نەزەر سالغايمىز:_تۇيۇقسىزدىن ھاجى پادىشاھنىڭ نۇرانە سىماسى پەيدا بولۇپ ئۇنىڭ كۆز ئالدىدا تۇرۋىلىپ،ئۇزاققىچە كەتمىدى.بۇ لەنىتى ئەنجانلىق ھاجى پادىشاھنى ئۆزىنى بايراق قىلىپ تۇرۇپ يوقاتقانىدى.ئىھتىمال.شۇ رەھمەتلىكنىڭ قان قىساسى تۇتقان بولسا كىرەك...ئەگەر شۇنداق بولغان بولسا.ئەل قىساسى مىنەلھەق!
بۇزرۇك خوجا ئۆز تەقدىرى ئۈستىدە،قەشقەردە بولىۋاتقان بۇ ئۇرۇش_جىدەل ھەققىدە ئويلىنىۋىتىپ.نىمە ئۈچۈندۇر بۇزرۇكۋار بوۋىسى ئافاق خوجىنى ئەسلەپ قالدى.ئۆزىنىڭ مۇشۇ زەي دىندان ئىچىدە مەھبۇس بولۇپ يىتىشى.تاجۇ-تەخىت ئۈچۈن ئاق تاغلىق،قارا تاغلىق بولۇپ كىتىشكەن ئاشۇ ئەجدادلىرىدىن مىراس قالغان يىغا-بالالار ئۈچۈن بڭلغان بولسا ئەجەب ئەمەس،تەخىت تەمەسىنى كۇلاھۇ جەندىللىرى ئىچىگە يوشۇرۇپ كەلگەن ئاشۇ نەمەنگانلىق بوۋىللىرى(ئەڭ دەسلەپكى تەرىقەتچىلەرنىڭ بىشى نەمەنگانلىق مەخدۇم ئەزەم،ئافاق خوجىنىڭ نەسەبى مۇشۇنىڭغا تۇتىشىدىغۇ دەيمەن.) مۇشۇ زىمىنغا تەپرىقىچىلىك ئۇرۇقىنى تىرۋەتكەنلىكى ئۈچۈن بولغان بولسا ئۇمۇ ئەجەب ئەمەس،چۈنكى ئەل قىساسى مىنەلھەق!
يەكى ئۇنى بۇزرۇكۋار بوۋىسى ئافاق خوجا تەرپىدىن مۇشۇ سارايغا بەند قىلنغان،سەئىدىيە خانلىرىنىڭ ئاخىرقىللىرىدىن بولغان ئىسمائىل خاننىڭ قىساسى تۇتتىمىكىن؟شۇنداق بولغان بولسا ئەجەپ ئەمەس،چۈنكى ئەل قىساسى مىنەلھەق!
بۇزرۇك خان تۆرەمنىڭ ئافاق خوجا دەپ مەشھۇر بوۋىسى ھىدايىتۇللا ئىشان خوجا بۇ ئەلنى قارا تاغلىقلارنىڭ قولىدىن زادىلا تارتىۋالالمىدى،ئاخىر،ئۇ تىبەتكە بىرىپ،لىخاسادىكى بۇتپەرەسلەرنىڭ چوڭى دالاي لامانىڭ ئايىغىنى سۆيدى،دالاي لاما ئۇنى ئىلى بالىقتىكى قالماقلارنىڭ خانى غالدانغا تونۇشتۇرۇپ،قولىغا خەت يىزىپ بەردى.ئافاق خوجا ئاتمىش مىڭ خام(ماتا پاختىدىن توقۇلغان يىرىك رەخت) تۆلەش بەدىلىگە،ئاتمىش مىڭ قالماق لەشكىرىنى باشلاپ كىلىپ،ئاران دىگەندە قەشقەرنى قارا تاغلىقلاردىن تارتىۋالدى.شۇندىن باشلاپ،بۇ ئەلنىڭ ئەرلىرى قۇل،خوتۇن-قىزلىرى دىدەك بولۇپ،مەزلۇم خەلىقنىڭ كۆز يىشى قۇرۇماي كەلدى،ئاخىرى بىرىپ ئۇلارنىڭ تەقدىرى كىملەرنىڭ قولىغا چۈشۈپ قالدى -ھە؟شۇ خەلىقنىڭ قىساسى تۇتىۋاتامدۇ؟نۇرغۇن ساھىبجامال قەشقەر قىزلىرى ئۆزلىرىنىڭ ئىپپەت-نۇمۇسىنى ساقلاپ قىلىش ئۈچۈن،<<يانتاقنىڭ كۈلى بولۇشقا رازىمىزكى،قالماقنىڭ گۈلى بولمايمىز>> دەپ،ئۆزلىرىنى ئوتقا ئىتىپ كۆيۈپ ئۆلۈشتى.شۇ ئاجىزەلەرنىڭ قىساسى تۇتتىمىكىن؟.....ئەگەر شۇنداق بولغان بولسا.ئەل قىساسى مىنەلھەق!
بۇ تۆرەمنىڭ ئاتا-بوۋىللىرى بۇ ئەلنىڭ قىنىنى دەريا_دەريالاپ تۆكۈشمىگەنمىدى؟بۇ ئۇلاردىن يىتىم قالغان نارسىدىلەرنىڭ كۆز ياشلىرى،پەرزەنتلىردىن ئايرىلغان ئانىلارنىڭ ئاھۇ_پەريادلىرى ئۈچۈن ئۆزىگە ياندۇرۇلغان جاۋاب ئەمەسمۇ؟شۇنداق بولغان بولسا،ئەل قىساسى مىنەلھەق!....
تۈنۈگۈن ئۇ ياساۋۇل ئارقىلىق ئاتالىق غازىغا(ياقۇببەگگە بىرىلگەن نام) ئىككى پارچە خەت يوللاپ،ئۆزىنىڭ سۆھبەتلىشىش ئارزۇسىدا ئىكەنلىكىنى بىلدۈردى،ئۇ قەشقەر سەلتەنەتىنىڭ تاج-تەختىدىن ئۆز مەيلى ۋە ئۆز ئىختىيارلىقى بىلەن ۋاز كىچىش قارارىغا كەلگەنلىكىنى ئۇنىڭغا ئىيتماقچى ۋە ئۇنىڭدىن بەيتۇلھەرەمگە بىرىپ ئاللاھنىڭ مۇبارەك ئۆيىنى تاۋاپ قىلىپ كىلىپ،قالغان ئۆمرىدە ئىستىقامەت قىلىپ ئۆتۈشكە رۇخسەت بىرىشىنى سورىماقچى ئىدى.
ئىشىك غىچىرلاپ يەر ئاستى ئۆيىگە بىرى كىردى بولغاي،بۇزرۇك خان تۆرە كۆزىنى ئاچتى.پەلەمپەيدە پەيدا بولغان ياساۋۇل كەينىگە يىنىپ،سىمىز،تۇنۇش بىر باكاۋۇلغا(پادىشاھ سۇلتانلارنىڭ يامىقىنى باشقۇرغۇچى ئەمەلدار)يول بەردى،باكاۋۇلنىڭ قولىدىكى يوغان پەتنۇستا قىزرىپ پىشقان ئىككى شىرمان نان،بىر تاۋاق ئاش ۋە بىر چەينەك چاي بار ئىدى،باكاۋۇل باش ئىگىپ تۆرەمگە تەزىم قىلدى-دە،پەتنۇسنى شىرەگە قويدى ۋە ئالدىرماي-تىنىمەي،سوۋۇپ قالغان تاماقنى يىغىشتۇردى،كىيىن ئۈندىمەي ئارقىسىغا يانغاچ نانغا قاراپ قويدى-دە،گەپ قىلماستىن چىقىپ كەتتى.
تۆرەم ھىچنىمىنى چۈشىنەلمەستىن بىر نانغا،بىر چىقىپ كىتىۋاتقان باكاۋۇلغا قاراپ تۇرۇپ قالدى.
يىڭى پىشقان ناننىڭ مەزىلىك پۇرىقى يەر ئاستى ئۆيىنى بىر ئالدى.شۇندىلا تۆرەم ئۆزىنىڭ تىخى ناشتا قىلمىغانلىقىنى ئىسىگە ئالدى-دە،<< نىمە بولسام بولارمەن>> دەپ ناننى ئوشتۇدى-دە،بىر بۇردا ئاغزىغا سالدى.تۇيۇقسىز تىلىغا غەلىتە بىر نەرسىنىڭ چاپلاشقانلىقىنى سەزدى.ئۇ ئاغزىدىكى ياخشى چاينالمىغان ناننى قولىغا تۈكۈرۈپ قارىۋىدى،نان پارچىللىرى ئىچىدە تۇرغان ھىمىرىپ كىچىك نۇقۇتتەك ھالەتكە كەلتۈرۈلگەن بىر پارچە قەغەزنى كۆردى.ئۇ دەرھال تۈزلەپ قارىسا خەت يىزىلغان بىر باغاقچە ئىكەن.
خەتتە يىزلىشىچە،ئوردا ئىچىدە قىيامەت باشلانغان.تۈنۈگۈن ھىيتگاھ جامەسىدە ئوقۇلغان جۈمە خۇتبىسى ئاتالىق غازىنىڭ نامىدا ئوقۇلغان.ئۇ <<بەدۆلەت>>(دۆلەت ساھىبى دىگەن مەنىدە) دەپ جاكارلانغان ۋە ئۇنىڭ نامىدا پۇل چىقىرىلدىغانلىقى ئىلان قىلىنغان.بۇزرۇك خان تۆرىنىڭ ھەج پەرزىنى ئادا قىلىش ئۈچۈن مەككىگە قاراپ يولغا چىقىدىغانلىقى جاكارلانغان.بەدۆلەت،تۆرەمنىڭ نۇرغۇن يىقىنلىرىنى تۇتۇپ زىندانغا تاشلىغان،قارشىلىق كۆرسەتكەنلەرنى بولسا،قەتلى قىلدۇرغان...

بۇزرۇك تۆرە يولغا چىقىشتىن بىر كۈن ئىلگىرى ھىيتگاھ جامەسىدە پىشىن نامىزىغا ئىماملىق قىلدى.بۇ-بەدۆلەتنىڭ ئۇنىڭغا بەرگەن ئەڭ ئاخىرقى بىر قىتىملىق ئىجازىتى ئىدى.
_دەۋران ئۆزگەردى.زىمىننىڭ ئىگىللىرى ئالماشتى.-دىدى ئۇ نامازنى ئوقۇپ بولغاندىن كىيىن تەلقىن سۆزلەۋىتىپ،-بىراق،خاقانىي چىننىڭ بۇ زىمنغا ئىگىدارلىق قىلىش ئىرادىسى ھەرگىز ئۆزگەرمەيدۇ،ئۇزاققا قالماي،خاقانىي چىن ئەۋەتكەن قارا چىرىكلەر رويى-زىمىننى چىكەتكىدەك قاپلاپ،مۈلكىي قەشقەرنى ئۆز تەسەررۇپىغا ئالۇر.چۈنكى.ھەدىستە ئاللا:<< مەن ئۇلارنى قەھىر-غەزىپىمدىن ياراتتىم>> دىگەن.بالا-قازانىڭ ئۇرۇقى بۇ زىمىنغا چىچىلغانىكەن،ئەجەل ھوسۇلىنى يىغماقتىن باشقا نىمە چارە بولسۇن!...
بۇزرۇك خان تۆرەمنىڭ بۇ چىدىماسلىق بىلەن ئىيقانلىرى ئون بىر يىلدىن كىيىن دەل شۇنداق بولۇپ چىقتى. بۇزرۇك خان تۆرەم مۇشۇ ماڭغىنىچە باغداتقىچە بىرىپ،يول ئىچىلماي،ھەج قىلىش شەرىپىگە ئىگە بولالمىدى،ئاخىر پەرغانىگە قايتىپ كىلىپ،قاراۋۇلتۆپە دىگەن جاينىڭ قاشتۈگمەن يىزىسىغا يىقىن كونا ئىغىز دىگەن كەنتتە ماكانلاشتى ۋە شۇ يەردە ۋاپات بولدى.
تەسىرات:-ئىنسان بىر-بىرىگە سۇيقەست قىلىش،ئورا كولاش بىلەن ئۆمرى ئۆتىدۇ،ئىنسان ئۈچ نەرسىگە نەپسى قانمايدىكەن:-ھايات-يەنى ئۆمۈرگە،پۇل-ھوقۇققا،پەقەت قانائەت قىلمايدىكەن،يەنە بىرى ھاۋايى-ھەۋەس،>>.بىز تارىخنى شۇنچىۋىلا ئوقۇپ،كىمگە نەپرەتلىنىپ كىمنى ياد ئىتەيلى تارىخ يەنىلا تارىختۇر،ئۆتۈپ كىتىپ بولغان ئىش ئۆزگەرمەيدۇ،شۇڭا ئۆتكەنلەرنى تىللاش دانالارنىڭ ئىشى ئەمەستۇر!نەسىردىن ئەپەندىمنىڭ بىر لەتىپىسى بار:-بىر كۈنى ئەپەندىم ئوغلىنى بازارغا بىر نەرسە ئەكىرىشكە بۇيرۇپ پۇل بىرىپ،بىر شاپىلاق قويۇپ:-ماڭ پۇلنى يۈتتۈرۋەتمەي دىگەن نەرسىنى ئەكىر!
دەپتۇ،بۇنى كۆرۈپ ئايالى ئاچچىغلاپ،:-يۈتمىگەن نەرسە ئۈچۈن بالىنى نىمىشقا ئۇرىسىز!ساراڭ بولدىڭىزمۇ؟:- دەپتۇ.
-يۈتكەندىن كىيىن ئۇرغاننىڭ نىمە پايدىسى؟دىققەت قىلىپ يۈتتۈرۋەتمەي ئەكىرسۇن!-دەپتۇ.
شۇنىڭدەك ئۆتكەن ئىش ئۈچۈن باشقىلارنى ئەيىبلىمەي،ھەممە ئىشنىڭ ئالدىنى ئىلىڭ،باشقىلارنى ئورۇنسىز ئەيىبلىمەڭ،قىلىۋالدىم-ئىتىۋالدىم دەپ خوش بولۇپ كەتمەڭ،بۇرۇنقىلاردىن ئىبرەت ئىلىڭ!ئافاق خوجا شۇنچە قانلارنى تۆكۈپتىكەن،ئاخىرىدا ئىشەنگەن ئايالى زەھەرلەپ ئۆلتۈردىغۇ؟ئەۋلادلىرى بوۋىسىنىڭ قىلغان يامانلىقلىرىنىڭ جازاسىنى تارتمىدىمۇ؟تارىختا <<جۇڭغارلارنى باشلاپ كىرىپ،سەئىدىيەنى يىقىتقان مەشھۇر مۇناپىق!>> دەپ نامى قالمىدىمۇ؟ياقۇببەگمۇ ئاخىرىدا ئۆزى ئەڭ ئىشەنگەن نىياز ھىكىمنىڭ قولىدا ئۆلۈپ جەسىدى چۆللەردە قالمىدىمۇ؟
خۇلاسە:-سىز بىر قىزسىز،كەلگۈسىددە ئانا بولىسىز،نىمىگە نىمە يەتمەي ئەرلەرنى ئالداپ باشقىلارنىڭ ئائىلسىنى بۇزىسىز؟سىزگە نىمە يەتمىدى؟نىمىشقا قىسقا يوپكىلارنى كىيۋالدىڭىز؟نىمىشقا بىشىڭىزنى ئوچۇق قويدىڭىز؟نىمىگە پىنھانلاردا قاتراپ يۈرىسىز؟كەلگۈسىدىكى ئىرىڭىزنى،ئىرىڭىز ھەم سىزنىڭ ئۈمىدىڭىز بولغان بالىڭىز ئالدىڭىزدا نىمىگە ھىساب؟
سىزچۇ؟ئەي ئەركەك!نىمىگە نىمىڭىز يەتمەي زىنا قىلىپ پاسىقلارچە ياشاپ،باشقىلارنىڭ بەختىگە ئولتۇرىسىز؟نىمىگە چوڭچىلىق قىلىلپ ئاجىزلارنى بوزەك قىلىسىز؟نىمىشقا مەس بولۇپ كوچىلاردا يىتىپ قالىسىز؟سىزدە نىمىشقا ئازراقمۇ غورۇر يوق؟سىز قىلغان يامانلىقلارنىڭ جازاسىنى سىز تارتماي كەلگۈسىىكى قىزلىرىڭىز،ياكى ئاچا-سىڭىللىرىڭىز تارتسۇنمۇ؟ياق؟ئەلۋەتتە ياق دەيسىزدە،سىزدە زەررىچە ئىمان بولسا ئەلۋەتتە يامانلىقلاردىن ئۆزىڭىزنى تارتىسىز!
قىرىندىشىم ئۆزىڭىزنى تارتىڭ،بۇ ئىنتايىن قىسقا مۇددەتلىك ھاياتىڭىزدا ئاددىيسى ئىنساندەك ياشاڭ!ئۆزىڭىزنى ئۆزىڭىز تۈگەشتۈرمەڭ!سىز مەيلى قانداق ئىش قىلىڭ چوقۇم ھىساب بىرىسىز،ھەرگىز مەغرۇرلانماڭ،يەر يۈزىدە غادىيىپ ماڭماڭ.سىز زىمىندا غادىيىپ مىڭىش بىلەن ئاسمانغا تەڭلىشەلمەيسىز،قاتتىق دەسسەپ مىڭىپ يەرنى تىشىۋىتەلمەيسىز.بىلىڭكى سىز بەك ئاجىز مەغرۇرلۇق سىزنى ھالاكەتكە باشلايدۇ،ھاكاۋۇرلۇقنى تاشلاڭ،باشقىلاردىن بىلىملىك كۆرىنىمەن دەپ پەسلەشمەڭ،چوڭ سۆزلىمەڭ!بىلىڭكى شەيتانمۇ پەرىشتىلەرگە دەرس ئۆتىدىغان ئالىم ئىدى.تەكەببۇرلۇق ئۇنى تۇيۇق يولغا باشلىدى!سىز شەيتاننىڭ يولىدىن ماڭماڭ!چۈنكى ئۇ ھەقىقەتەن سىزگە ئاشكارا دۈشمەندۇر!
خاتىمە:_بۇ پەقەت مىنىڭ شەخسىي كۆز قارىشىم،باشقىلارنىڭ خاتا چۈشىنىشنى خالىمەيمەن،مەن ھىچكىمنى ئەيىبلىمەيمەن،پەقەت ياخشىلارنىڭ يامانلارغا ھەۋەس قىلىپ بىر يولدا مىڭىشىدىن ئەنسىرەيمەن!
مەنبە:-ئۆزۈم
