                                           ئەي گۈلۈم كۆردۈممەن سېنى چۈشۈمدە،

                                           يېنىشلاپ باغرىڭغا باستىڭ سەن مېنى،
                                           قانمىغاندەك شۇ سېنىڭ بىغۇبار قەلبىڭ.

                                           ھەقىقەت ئىككىمىز ئارلاشتۇق ئۇزاق،
                                           ۋە لېكىن ۋەسلىمىز سۆيگۈدىن يىراق،
                                           بىر رەت سۆيگۈڭدىن ئالمىغاچ ھوزۇر،
                                           قەلبىمنى باساتتى دەرت-ئەلەم پىراق.

                                           ئاڭلىسام بولماقچى ئىكەنسەن ياتلىق،
                                           بىلىمەن يولدىشىڭ تومپاي بىر تاغلىق.
                                           كۆردۈم،سېنىڭمۇ ھالىڭ چاغلىقكەن،
                                           چىن سۆيگۈ ئۇدىگەن ۋىجدانغا باغلىق.

                                           ئەلۋىدا ئەي رەزىل باغرى تاش جانان،
                                           ئاخىرى قىلدىڭ قەلبىڭنى ئايان.  
                                           مەڭگۈلۈك بىرگە بولىمىز دەپ قويۇپ،

                                          بەرگەن شۇ ۋەدىللىرىڭ قېپقالدى قايان!

دادا

     دادا -ئۆينىڭ تۈۋرىكى،شۇمېھرىبان ،كۆيۈمچان ،ئىشچان دادىمىز بولغاچقىلا بىز ھېچنىمىدىن غېمىمىز يوق خۇشال -خۇرام كۈنلەرنى كەچۈرەلەيمىز.تۆۋەندىكى شىئېرىمنى بىز ئۈچۈن ئۈن-تۈنسىز ياشلىق ھاياتىنى بېغىشلاۋاتقان دادىلاررغا بېغىشلايمەن.

                                          ئاھ كۆيۈمچان مېھرىبان،
                                          شەپقەتچىم دادا.
                                          مېنى دەپ بارىڭنى،
                                          قىلدىڭسەن جۇدا.

                                          تاپساڭ بىر كەمپۈت،
                                          گېلىڭدىن ئۆتمەس.
                                          بالام دەپ ھەرۋاقىت،
                                          غېمىڭ تۈگىمەس.

                                          كىيمەي كىيگۈزدۈڭ،
                                          يىمەي يىگۈزۈپ.
                                          قارچۇقىڭدىن ئەتىۋارلاپ،
                                          كۆككە كۆتۈرۈپ.

                                          مېنى دەپ تارتتىڭ،
                                          مىڭ مۈشكۈل-جاپا.
                                          تارتقان شۇ ئازابىڭغا،
                                          قىلمىدىڭ تاپا.

                                          ئاقىرىپتۇ قار كەبى،
                                          قۇندۇز چاچلىرىڭ.
                                          سۇسلىشىپتۇھەم سېنىڭ،
                                          نۇرلۇق كۆزلىرىڭ.

                                          خاتىرجەم بول ئەزىزەم،
                                          شاھبانۇم دادا.
                                          مەن ئوغۇللۇق بۇرچۇمنى،
                                          قىلىمەن ئادا.

                                          مەن ئىزىڭنى باسىمەن،
                                          ھامان بىر كۈنى.
                                          مەيدەڭگە گۈل تاقايمەن،
                                          ھامان بىر كۈنى.

                                                                         ئاپتۇرى:ۋاسىتجان
