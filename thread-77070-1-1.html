ئەسكەرتىش:فىيجۇلىيۇ توغرىسىدا ئىلگىرى قىسقا بىر يازما يازغان لېكىن بۇنچە تەپسىلى يازمىغان .باشقۇرغۇچىلار بولسا تەستىقلاپ قويساڭلار .
ئەسسالامۇ ئەلەيكۇم ھۆرمەتلىك ئاتا ئانىلار .
يېقىندا مەن ئوتتۇرا مەكتەپتىكى ئۆسمۈلىرىمىزنى كىرەكتىن چىقىرىۋاتقان شەيتاننىڭ قۇرالى بولغان <سېرىق فىلىم> كۆرۈش توغرىسىدا جەمىيەت تەكشۈرەش مەلۇماتىمنى بەرگەن ئىدىم.ئاڭلىسام جەميەتتە خېلى غۇل غۇلا قوزغاپتۇ .ئۆزىنىڭ جان جىگەر بالىسىنىڭ سېرىق فىلىمنىڭ قۇلى بولۇپ قېلىشىدىن نۇرغۇن ئاتا -ئانىلار بالىسىنىڭ كومپىتوردا نىمە قىلىۋاتقانلىقىغا كۆز قۇلاق بولىدىغان بوپتۇ .
مەن ھازىر بايان قىلىدىغان ھادىسە بالىلارنىڭ روھىيىتىگە زور تەسىرلەرنى پەيدا قىلىدىغان مەدەنىيەت ئەخلىتى بولغان فىيجۇلىيۇ توغرىسىدا توختالماقچىمەن .
فىيجۇلىيۇنى بەزىلەر پاھىشىۋى قىياپەت دەپ تەرجىمە قىلىپتۇ .ئاتا ئانىلارنىڭ تۇرمۇش ھەلەكچىلىك تۈپەيلى جۈملىدىن ئۆزىنىڭ ياش قاتلىمىدا بۇ ئۇچۇرلار بىلەن ئۇچىرشىش ۋە چۈشىنىش ئىمكانىيىتىنىڭ كەمچىللىكى تۈپەيلى بۇ مەدەنىيەت ئەخلىتىدىن خەۋەرسىز يۈرىشى ۋە بالىسىنىڭ روھىي ھالىتىنىڭ بۇ مەدەنىيەت ئەخلىتىنىڭ تەسىرىدە غەيرى ۋە بىنورمال،جاھىل پىسخىكىغا كېلىپ قېلىۋاتقانلىقىنى بىلەلمەسلىكى تەبىئىي .
ئالدى بىلەن تۆۋەندىكى بىر نەچچە تور بەتنى ۋە رەسىملەرنى كۆزدىن كەچۇرۇڭ  .

خوش بۇ رەسىملەرنى كۆرۈپ قانداق ھىسياتقا كەلدىڭىز ؟بەلكىم يۈزەكى شۈشەنچىگە كېلىشىڭىز مۇمكىن .ئەمدى مەن فىيجۇلىيۇ توغرىسىدا قىسقىچە مەلۇمات بىرەي .
1.بۇ ئەخلەتنىڭ بىر ئۇچى ياپۇنىيەدە يەنە بىر ئۇچى ئامىرىكىدا ئىكەن .ياپۇنىيەدىكىسى   
بولسا 70-80 يىللىرىدا ياپۇن جەمىيتىنىڭ تۆۋەن قاتلىمىدا جەمىيەتكە قارشى ئېقىم سۈپىتىدە  
مەيدانغا كەلگەن .ياپۇنىيەدە سېرىق فىلىم سانائىتى باغچىسى بولۇپ بۇ باغچىدىكى ئەر ئاياللار دەل فىيجۇلىيۇ ئۇسلۇبىدا ياسىنىدۇ .تەيۋەنگە ۋە چوڭ قۇرۇقلۇققا تاقالغىنى دەل مۇشۇ پاھىشىلەرنىڭ ياسىنىش ئۇسولىدۇر .ئۇنىڭ ئىپادىسى بولسا ھەرخىل ئەخلاق ۋە گۇزەللىككە خىلاپ قىياپەتتە   
رەسىمگە چۈشۈش ۋە ياسىنىش ،گىرىم قىلىش .بۇ <بەدنىيەتلىك>نى كەڭ ئومۇملاشتۇرىدىغانلار   
ئاسەسەن پاھىشە ۋە سېرىق فىلىم ئىشلەيدىغانلار ياكى جىنسى جەھەتتە ئەركىن ياشلار بولغان .
كېيىن بۇ <بەدنىيەتلىك >تەيۋەنگە تارقالغان.90-يىللارنىڭ ئاخىرىدا چوڭ قۇرۇقلۇققا تارقىلىپ
جەمىيەتكە ،ئەخلاققا، ئېغىر  زىيانلارنى سالىدىغان مەنىۋى ئەخلەتلارنىڭ بىرى بولۇپ  
قالغان .يەنە بىر ئۇچى بولسا ئامىرىكىدىكى مانسونچىلار ئېقىمىدىكىلەر .



بۇ ئېقىمدىكىلەر چېكىدىن ئاشقان ئەخلاقسىزلىقنى ئوموملاشتۇرۇدىغان روك موزىكا ساراڭلىرى بولۇپ  
موزىكا ۋە بەدەن ئارقىلىق ئوچۇقتىن ئوچۇق خىرىستيان گادىغان(خىرىستيان خۇداسىغا ) قارشى تۇرۇدۇ .ئۇلارنىڭ موزىكا ئېقىمى ۋە ئىدىيەسى ھازىر بەزى ئۇيغۇر ياشلىرىنىڭ روھى دۇنياسىغا
بۆسۈپ كىرىۋاتىدۇ . دېسكو موزىكىسى دەل فىيجۇلىيۇنىڭ يەنە بىر شاخچىسىدور
ئۇيغۇر ئوتتۇرا مەكتەپ ياشلىرىدا فىيجۇلىيۇنىڭ تەسىرى ئىككى خىل ھالەتتە ئىپادىلىنىۋاتىدۇ .
1.يۇشۇرۇن ھالەت .بۇ ئاساسەن تورلاردا شۇنداق ئۇچۇرلارنى كۆپ كۆرگەنلىكى ياكى دورامچىللىق سەۋەبىدىن شۇ خىل مەدەنىيەت ھادىسىگە ئەگىشىدىغان بۇنى چچ تور بوشلۇقىنى لاھىلەش ياكى 33591 دەك تور ساھاسى ئەخلەت دەپ قارايدىغان بىلوگ سۇپىلىرىنى فىيجۇلىيۇ ئۇسلۇبىدا بىزەيدىغان شەكىللەردە ئىپادىلىنىۋاتىدۇ .33591 ھازىر ئۇيغۇر ئۆسمىرلىرىنى بۇزىدىغان بىر مەيدان بولۇپ قاپتۇ .

2.ئاشكارە ھالەت .بۇ ھالەت چاچ پاسونىدا ۋە گىرىم قىلىشتا ئۆز ئەكسىنى تاپتى .ئوتتۇرا مەكتەپ قىزلىرىدا ۋە ئوغۇللىرىدا فىيجۇلىيۇ ئېقىمىغا ئوخشاپ كېتىدىغان چاچ پاسونلىرى مەيدانغا كەلدى .مەسىلەن :ئوغۇللار چېچىنى  ئوتتۇرسىنى پۆپۇك چىقىرىش،ئىككى چىكىسىنى قىرقىۋېتىش .قالايمىقان بىر ھالەتتە ياسىتىش .
قىزلار غەيرى ھالەتتە چېچىنى ياسىۋېلىش .
ئىنگىلىز ۋە خەنزۇچە تور بەتلەردىن فىيجۇلىيۇنىڭ ئۆسمۈرلەرگە بولغان زىيىنى توغرىسىدا مەخسوس تەتقىقاتلار ئېلىپ بېرىلغان ھەم مەخسوس بۇ ئەخلەت مەدەنىيەتكە قارشى تور بىكەتلەر قۇرۇلغان .http://365fanfei.5d6d.com/

فىيجۇلىيۇ يەنى پاھىشىۋى قىياپەتنىڭ ئۆسمۈرلەرگە بولغان زىيىنى زور .فىيجۇلىيۇغا ئەگەشكەن ياشلار غەيرى رەسىم ۋە دېسكو مىتال موزىكىلارنى كۆپ كۆرگەچكە، ئالدى بىلەن پىسخىكا جەھەتتىن گەپ ئاڭلىمايدىغان ۋە جاھىللىق پىسخىكىسى شەكىللىنىدىكەن .ئاندىن ياسىنىشتا غەيرىلىككە ئىنتىلىدىغان ،يوشۇرۇن مۇھەببەتلىشىشكە بىرىلىدىغان، ھەتتا سېرىق فىلىم كۆرىدىغان ھادىسىلەرنى مەيدانغا كەلتۈردىكەن .ئۆسمۈرلەرنىڭ ئۆز ئۆزىنى قوغدايدىغان ئەخلاق تېمى نىپىز بولغاچقا بۇ خىل مىڭىنى غىدىقلۇغۇچى غەيرى رەسىم ۋە موزىكىلار ئۆسمۈرلەرنى يېشىغا ماس كەلمىگەن ھالدىكى بالدۇر يىتىلىشنى كەلتۈرۈپ چىقىدىكەن .تەتقىقاتلاردا بايقىلىشىچە فىيجۇلىيۇ ئېقىمىغا ھىسداشلىق قىلىدىغان ئۆسمۈرلەرنىڭ بۇ خىل بىنورمال ھالىتى ئاخىرى بىنورمال جىنسىي تۇرمۇشقا بالدۇر مېڭىپ قېلىشنى كەلتۈرۈپ چىقىرىدىكەن .
ھازىر خەنزۇ مەكتەپلەردە بۇ ئەخلەت مەدەنىيەتكە نىسبەتەن لىكسىيەلەر ۋە تەربىيەلەر قىلىنىۋاتىدۇ .لېكىن مەسىلە ئوڭشالماقتا يوق كۈندىن كۈنگە ئېغىرلىشىپ كېتىۋاتىدۇ .خەنزۇ مەكتەپلەردە ئوقىۋاتقان ئۇيغۇر بالىلىرى ئارسىدا بۇ خىل ئەخلەت مەدەنىيەتكە بىرىلىش ۋە ياخشى كۆرۈش ھادىسىسى ئېغىرلىشىپ كېتىۋاتىدۇ .ھەمدە ئۇيغۇر مەكتەپلەردىمۇ بەلگۈلىك تەسىرلەرنى كۆرسىتىۋاتىدۇ .
سىز بالىڭىزنىڭ چچ تور بوشلۇقى ،ۋە 33591 دىكى بېتىنىڭ قانداق بىزىلىشىغا دىققەت قىلىپ باقتىڭىزمۇ؟بالىڭىزنىڭ قانداق رەسىملەرنى باش رەسىم قىلىۋالغانلىقىغا دىققەت قىلىپ باقتىڭىزمۇ؟بالىڭىزنىڭ ياسىنىشىغا دىققەت قىلىپ باقتىڭىزمۇ ؟
سۇكەلگۈچە توغان تۇت دىگەن تەمسىل بار .ھازىر فىيجۇلىيۇنىڭ زىيىنى چوڭ شەھەرلەردىكى ئۇيغۇر ئۆسمۈرلىرى ئىچىدە بەكرەك تەرەققىي قىلدى لېكىن كىچىك شەھەرلەرگە تېخى ئەمدى تەسىر كۆرسىتىۋاتىدۇ .شۇڭا ئۆز پەرزەنتىڭىزنىڭ بۇ مەدەنىيەت ئەخلىتىنىڭ زىيان سالماسلىقى ئۈچۈن پەرزەنتىڭىزگە دىققەت قىلىپ قويۇڭ.
بۇ ھەقتە كۆپچىلىك بىلەن پىكىرلىشىپ باقماقچىمەن .

تولىمۇ مۇھىم بىر ئەھۋالدىن خەۋەردار قىلىپسىز،ئەجرىڭىز بىكارغا كەتمىسۇن.ئەمما يىغىپ ئېيتقاندا بۇ مەسىلىدە ئائىلە تەربىيىسىنىڭ قانداق بولىشى بەك مۇھىم.نۇرغۇن ئاتا-ئانىلار بالىلىرىنى چوڭ بولىۋېلىپلا گېپىمىزنى ئاڭلىماس بولىۋالدى،قانداق قىلارمىز دەپ ۋايسايدۇ،شۇنى ئېتىراپ قىلمايدىكى،لوڭڭىدە چوڭ بولۇپ قالغان بالىسىنى تۇغۇلىشىدىن باشلاپ مەخسۇس تەربىيەلىمىگەن.بوۋاق ۋاختىدا ھىچنىمە ئۇقمايدۇ دەيمىز،بالا(مەكتەپكە كىرىشتىن بۇرۇن) ۋاختىدا كىچىك ئەمەسمۇ،ئوينىۋالسۇن دەيمىز،ئۆسمۈر ۋاختىدا(باشلانغۇچ،تولۇقسىز)بىز ئاللىقاچان كېچىككەن بولىمىز،ئوتتۇرا مەكتەپكە قەدەم قويغاندا باشقۇرىمەن،قۇلىغىنى يۇلىمەن دېمەي،ئۆزىمىزنى كاچاتلىساق بولىدۇ،ئامال يوق،جەمئىيەت،مۇھىت شۇنداق رەھىمسىز.بىر تەتقىقات ماقالىسدا بالىنىڭ مىجەز-خاراكتېرى ئەمەلىيەتتە 3ياش ئەتراپىدا ئاساسەن مۇقىملىشىپ بولىدىكەن،3ياشتىن بۇرۇنقى تەربىيىنى چىڭ تۇتمىساق،3ياشتىن كېيىن ئوڭشىماق تەسكە چۈشىدىكەن،لېكىن ئاتا-ئانىلار بالىغا كۆڭۈل بۆلسە ياش نوتىنى ئاستا-ئاستا ئەيۋەشكە كەلتۈرگىلى بولىدىكەن،بالا 7ياشتىن ئاشقاندا ئۇنىڭدىكى شەخسىيەتچىلىكنى ئۆزگەرتىش بەسى مۈشكۈلكەن...دېمەك بۇ توغرىدا كۆپ سۆزلەشنىڭ ئورنى يوق،ئاتا-ئانىلار بىر-بىرىدىن بىلىملىك.لېكىن بالىنى تېپىپ قويۇپلا تەربىيەنى ئۇنتۇپ قالساق ئاقىۋەت مۇشۇنداق بولىدۇ.مەن تېخى ئالدىنقى كۈنى خەلق يولى ئۆتۈشمە يولىدىن ئۆتۈپ كېتىۋېتىپ،شۇ يەردىكى جىرلا ناملىق(قايسى ئىمانسىز ئاچقانكىن)دېسكوخانىدىن شۇنداق خىلدا ياسىنىۋالغان بىر توپ بالىلارنىڭ(ئۇيغۇرمۇ،قازاقمۇ ئۇقالمىدىم)چىقىۋاتقانلىغىنى،قىز ئوغۇللارنىڭ بىر-بىرگە ئېسىلىشىۋالغانلىغىنى كۆرۈپ توۋا دەپ ئۆتۈپ كەتكەنتىم،مۇشۇنداق سورۇنلارنى ھازىرلاپ ئەۋلاتلىرىمىزنى ھاڭغا ئىىتىرىۋاتقانلارنىڭ بالىلىرى ،ئاچا-سىڭىللىرى يوقمىدۇ؟دەپ ئويلىدىم كۆڭلۈمدە،ئاشۇنداق ھارامدىن تاپقان پۇللىرى قانداق سىڭەر دەيمەندە.
بىلمىگەننى بىلدىم ، كۆرمىگەننى كۆردۇم، ئاڭلىمىغاننى ئاڭلىدىم دېمە!
