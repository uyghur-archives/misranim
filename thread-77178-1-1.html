مەكتەپتە ئىش ئۈستىدە يارىلانغان ئوقۇتقۇچىلار قانداق قىلىپ مەنپەئىتىنى قوغدىشى كېرەك ؟
مەن بىلەن بىر مەكتەپتە ئىشلەيدىغان بىر ئوقۇتقۇچى بار بولۇپ بىر ئايچە ئىلگىرى مەكتەپتە تەنتەربىيە دەرسى بېرىدىغان ئوقۇتقۇچى رۇخسەت سورىغانلىقى سەۋەبتىن ،ئۆزى مەسئۇل بولغان سىنىپ ئوقۇغۇچىلىرىغا تەنتەربىيە دەرس ۋاقتىدا دەرس ئۆتكەن .دەرس ۋاقتىدا بالىلار بىلەن توپ ئويناۋېتىپ گارنىڭ ئالدىدىكى سېمونتى چىقىپ ،كاتەك بولۇپ قالغان جايغا ئېھتىياتسىزلىقتىن توپ ئېلىپ سەكرەپ چۈشكەندە بىر پۇتى چۈشۈپ كىچىك بارماقنىڭ سۆڭىكى ئوشۇققا يېقىن جايدىن سۇنۇپ كەتكەن .سۇنۇپ كەتكەن كۈنى كەچتە مەكتەپتە نۆۋەتچى بولغاچقا ،بىر كېچە نۆۋەتچىلىك قىلىپ ئاندىن ئەتىسى دوختۇرخانىنىڭ ئىسپاتىنى تاشلاپ بېرىپ ،يۇقىرى دەرىجىلىك دوختۇرخانىلار دا تەكشۈرگەندە ،ھەقىقەتەن سۇنۇپ كەتكەنلىكى ئىسپاتلانغان .ئاندىن گىپىسلاپ تاڭدۇرۇپ ،3-ئاينىڭ 19-كۈنى مەكتەپكە ئۆزى ئېلىپ چىققاندا ،مەكتەپ تەرەپ بىر ھەپتە رۇخسەت بەرگەن.لېكىن سۇنۇق دېگەن بىر ھەپتىدە ساقىيامدۇ.ئامال يوق ،بىر ھەپتە رۇخسەت بەرگىنىگە خۇش بولۇپ قايتىپ دەم ئالغان ،بىر ھەپتە توشقاندا يەنە ياخشى بولالمىغاچقا تېلىفۇن ئۇرۇپ يەنە بىر ھەپتە رۇخسەت سورىغان .بۇ رۇخسەت ۋاقتى توشۇپ قالغاندا مەكتەپ مۇدىرى تېلفۇن ئۇرۇپ :«تەكشۈرۈش باركەن ،مەكتەپتە ئادەم ئاز ،چىقمىسڭىز بولمايدۇ»دېگەچكە 3-ئاينىڭ 31-كۈنى شەنبە بولسىمۇ مەكتەپتىكىلەر ئىشلىگەچكە ،ئىش ئورنىغا چىققان .18كۈندە سۇنۇپ كەتكەن جاي قانداقمۇ ئەسلىگە كەلسۇن .لېكىن شۇنداق بولسىمۇ مەكتەپ رەھبەرلىكنىڭ قىيىنچىلىقىنى كۆزدە تۇتۇپ مەكتەپتە ئىشقا چۈشكەن ،ئۇ قوش ھاسا تاياققا تايىنىپ ھەرىكەت قىلىپ خىزمەت قىلغان .دوختۇرنىڭ« 40 كۈن ئورنۇڭدىن تۇرماي دەم ئال!بولمىسا سۇنغان جاي قايتا سۇنۇپ كېتىدۇ » دېگەن گەپنىمۇ قايرىپ قويۇپ خىزمەت قىلغان .مەكتەپ بۇ ئوقۇتقۇچى ئىشلەۋەرسە گەپمۇ قىلماستىن ئوبدان ئىشلەيدىكەنسەن دەپ ئىشلىتىۋەرگەن .نەتىجىدە بۇ ئوقۇتقۇچى تاھازىرغا قەدەر ساقىيالماي،ھاسا تاياقنى تاشلاپ ماڭالمىغان .دوختۇرلار«بۇ جاي ياللۇغلىنىپ قاپتۇ،يېتىپ ئارام ئالساڭ بولاتتى»دېگەن .بۇ جەرياندا مەكتەپ رەھبەرلىكى بۇ ئوقۇتقۇچىنى مەكتەپتە 24 سائەتلىك نۆۋەتچىلىككە ئورۇنلاشتۇرغان بولۇپ ،بۇ ئوقۇتقۇچىغا ئۇقتۇرۇلمىغان ،پەقەت شۇ كۈنى مەسئۇل نۆۋەتچى كەچ سائەت 8دىن ئاشقاندا ئاندىن تېلفۇن ئارقىلىق ئۇقتۇرغان .بۇ ئوقۇتقۇچى چارچاپ قالغاچقا بالدۇرلا يېتىپ ئارام ئالغان بولۇپ كىيىملىرىنى كىيىپ تەييار بولۇپ بارغۇچە يېزىدىن چارلاشقا چىقىپ مەكتەپتە 9ئادەم نۆۋەتچى تولۇق يوقكەن دەپ 200 يۈەن جەرىمانە قويۇپ ئەتىسى تۆلەشنى بۇيرىغان .مەكتەپنىڭ رەھبەرلىرىگە شۇنداق دېگەن بولسىمۇ ،ئۇلار پىسەنت قىلمىغان .نەتىجىدە ئەتىسى بۇ ئوقۇتقۇچىنىڭ ھاسا تاياققا تايىنىپ ماڭىدىغان ھالىتىنى كۆرگەن يېزا رەھبەرلىرى ئەھۋالنى نەزەرگە ئالماستىن 200 يۈەن جەرىمانىنى ئالمىسا بولمايدىغانلىقىنى ئېيتىپ ،بۇ ئەھۋالنى مەكتەپ رەھبەرلىرىگە ئېيت ،بۇ شۇلارنىڭ ئىشى بىز جەرىمانىنى ئالغىنىمىز ئالغان دەپلا ،جەرىمانىنى ئالغان ،مەكتەپنىڭ شۇجىسىمۇ گەپ قىلىپ ئاقتۇرالمىغان .ئاخىرى بۇ ئوقۇتقۇچىنىڭ قورسىقى كۆپۈپ بۇ ئىشلارنى قانۇن بىلىدىغانلار بىلەن مەسلىھەتلىشىپ ئاندىن بىر تەرەپ قىلىش ئويى بارلىقىنى ئېيتقان بولغاچقا ،تورغا يوللىدىم .تورداشلارنىڭ ياخشى مەسلىھەت بېرىشىنى ئۈمىد قىلىمەن .
1) بۇنداق ئەھۋالدا بۇ ئوقۇتقۇچىنىڭ يارىلىنىشى ئىش ئۈستىدە يارىلىنىشقا كىرەمدۇ؟
2)ئەگەردە ئىش ئۈستىدە يارىلىنىشقا كىرسە مەكتەپ قانداق مەسئۇليەتنى ئۆز ئۈستىگە ئالىدۇ؟
3)مەكتەپنىڭ مەكتەپتە ئوقۇتقۇچى قىس بولىشىنى باھانە قىلىپ بۇنداق مۇئامىلە قىلىشى توغرا بولامدۇ؟
4)بۇنداق ئەھۋالدا زادى قانچە كۈن دەم ئېلىش رۇخسىتى بېرىشى كېرەك؟(دوختۇرخانا 3ئايلىق دەم ئېلىش يېزىپ بەرگەن )
5)بۇ ئوقۇتقۇچىنى نۆۋەتچىلىككە(جىبەرلىككە)ئورۇنلاشتۇرىشى ،ئاندىن جەرىمانە قويىشى توغرىمۇ؟
6)بۇ ئوقۇتقۇچى قانداق يوللار ئارقىلىق قانۇنىي ھوقۇق مەنپەئەتىنى قوغدىشى كېرەك ؟
7)بۇ ئوقۇتقۇچىنىڭ مەكتەپ رەھبەرلىكىنىڭ مەكتەپتە ئوقۇتقۇچى قىس بولۇش توغرىسىدىكى قىيىنچىلىقى كۆزدە تۇتۇپ ئىشقا چىقىشى خاتامۇ؟
8)بۇ ئوقۇتقۇچىنىڭ قوش ھاسا تاياققا تايىنىپ ئىشلىگىنى كۆرگەن رەھبەرلەر سېنىڭ قانداق قىيىنچىلىقىڭ بار دەپمۇ سوراپ قويماسلىقىنى قانداق چۈشىنىش مۇمكىن؟
تورداشلارنىڭ ياخشى مەسلىھەت بېرىشىنى ئۈمىد قىلىمەن .

جۇڭخۇا خەلق جۇمھۇرىيىتى دۆلەت ئىشلىرى مەھكىمىسى (گوۋۇيۇەن) 375- نومۇرلۇق پەرمانى «ئىش ئۈستىدە يارلىنىش نىزامى» (2003- يىلى 4- ئاينىڭ 27- كۈنى ئېلان قىلىنىپ، 2004- يىلى 1- ئاينىڭ 1- كۈنىدىن باشلاپ يولغا قويۇلغان) ۋە شىنجاڭ ئۇيغۇر ئاپتونوم رايۇنىنىڭ كەسپى ئورۇنلاردىكى خىزمەتچى خادىملارنىڭ ئىش ئۈستىدە ئۆلۇش (يارلىنىش) نى بىر تەرەپ قىلىش ۋاقىتلىق چارىسى (بۇ چارە 2000- يىلى 10- ئاينىڭ 30- كۈنى ئېلان قىلىنىپ يولغا قويۇلغان) غا ئاساسەن، بىر تەرەپ قىلىنسا بولىدۇ.
ئاپتونوم رايۇننىڭ «چارىسى» نىڭ 1- ماددىسىدا : «كەسپى ئورۇنلاردىكى خىزمەتچى خادىملارنىڭ يارلىنىش، مېيىپ بولۇش، ئۆلۇشىنى بېكىتىش ۋە ئاخىرەتلىك ئىشلىرىنى بىر تەرەپ قىلىش ئۈچۈن، دۆلەتنىڭ ئالاقىدار بەلگىلىمىلىرىگە ئاساسەن، كەسپى ئورۇنلارنىڭ ئەمەلىيىتىگە بىرلەشتۈرۈپ، مەزكۈر چارە تۈزۈپ چىقىلدى» دىيىلگەن. مەكتەپ كەسپى ئورۇن، شۇنىڭ ئۈچۈن بۇ چارە مەكتەپكىمۇ تەبىقلىنىدۇ.  شۇنىڭ بىلەن بىر ۋاقىتتا، بىخەتەر ئىشلەپچىقىرىش قانۇنى ۋە ھوقوققا چېقىلىش قانۇنى قاتارلىق قانۇنلارغا ئاساسلانغاندا، مەكتەپنىڭ ۋاستكىبول مەيدانىدا سىمونىتنىڭ چىقىلىپ، كاتەك بولۇپ قالغانلىقى مەكتەپنىڭ پائالىيەت مەيدانىدا خەتەر خەۋىپىنىڭ ساقلانغانلىقىنى، مەكتەپنىڭ خەتەرنى بايقىمىغانلىقىنى ۋە ئۇنى بىر تەرەپ قىىلىش تەدبىرىنى قوللانمىغانلىقىنى بىلدۇرىدۇ.
بۇ ٪100 ئىش ئۇستىدە يارلىنىش بولۇپ، ئوقۇتقۇچى مۇناسىۋەتلىك دەلىل ئىسپاتلارنى توپلاپ (مەسىلەن: تەنتەربىيە دەرسى ئۆتكەن چاغدىكى ئوقۇغۇچىلارنى ئىسپاتلىقى، دەرس ۋاقتىدا يىقىلىپ كەتكەنلىكى، دوختۇرنىڭ تەكشۈرۈش ئىسپاتى) قاتارلىقلارنى ئېلىپ، ئۆزى تۇرۇشلۇق ئادەم كۈچى ۋە ئىجتىمائى كاپالەت ئىدارىسىغا بېرىپ، «شىنجاڭ ئۇيغۇر ئاپتونوم رايۇنىدىكى كەسپى ئورۇنلاردىكى خىزمەتچى خادىملارنىڭ ئومومنىڭ ئىشى (ھۆكۈمەت ئىشى) يارلىنىش، مېيىپلىنىش، ئۆلۇش ھادىسلىرىنى بېكىتىشنى تەستىقلاش جەدۋىلى» نى ئېلىپ، بىر نۇسخىدا ئۈچ پارچە تولدۇرۇپ، ئىلتىماس قىلسا بولىدۇ. بۇ يەردە ئېيتىلىدىغان بىر گەپ ئۆزى ئىشلەيدىغان مەكتەپنىڭ پىكرى دىگەن يېرى بار، بەلكىم مەكتەپ تامغا بېسىپ بېرىشكە قوشۇلماسلىقى مۇمكىن، ئەگەر شۇنداق بولۇپ قالسا،  شۇ جايدىكى مائارىپ ئىدارىسىگە ياكى سوت مەھكىمىسىگە ئىلتىماس قىلسا ۋە ياكى بىۋاستە ئادەم كۈچى ۋە ئىجتىمائى كاپالەت ئىدارىىسغا ئەھۋالنى ئىنكاس قىلىپ بېكىتىپ بېرىشنى دىسە بولىدۇ.

كۆپچىلىككە ئاگاھلاندۇرۇش: ئەگەر ھەقىقەتەن ئىش ئۇستىدە يىقىلىپ زەخمىلەنسەڭلار، ھەرگىزمۇ خىجىل بولماستىن، شۇ ۋاقىتنىڭ ئۆزىدە دوختۇرخانىغا بېرىپ تەكشۇرتۇڭلار، ھەرگىزمۇ نۆۋەتچى ئىدىم، دەرسىم بار ئىدى دەپ ئولتۇرماڭلار، دەل ۋاقىتنىڭ ئۆزىدە بارغاندا ئەڭ ياخشى ئۇنۇمنى قولغا كەلتۇرۇپ، ئۆزەڭلارنى قانۇنى ھوقوقىنى قوغدىيالايسىلەر.
بايا مەن ئېيتقان «جەدۋەل» نى تەستىقلاتساڭلار بولىدۇ. ئەگەر ھەقىقەتەن ئىش ئۇستىدە يارلانغان دەپ قارالسا، داۋالىنىش ۋە ئارام مەزگىلىدىكى تەمىناتى ئىش ئۇستىدىكى تەمىناتى بىلەن ئوخشاش بولىدۇ، ھىچقانداق ئورۇننىڭ دەخلى قىلىشىغا، تۇتۇپ قېلىشىغا بولمايدۇ، ھەمدە يىللىق باھالاشقىمۇ تەسىر يەتمەيدۇ.

سىز ئالدى بىلەن تور http://ishare.iask.sina.com.cn/f/24184076.html دىن 工伤认定申请表 دېگەن جەدۋەلنى چۈشۈرۈپ، مۇناسىبەتلىك جايلىرىنى تولدۇرۇپ، مەكتىپىڭىزنىڭ، مائارىپ ئىدارىسىنىڭ تامغۇسى بېسىلىدىغان يەرلەرگە تامغۇ باستۇرۇپ(ئىدارىنىڭ تامغۇسىنى باستۇرۇپ بولغاندىن كېيىن  بۇ جەدۋەلنى بىرنۇسخا كۆپەيتىۋىلىشنى ھەرگىز ئۇنتۇلۇپ قالماڭ)، دوختۇرخانىنىڭ دىئاگنوز قەغىزى ۋە دوختۇرخانىغا تۆلىگەن پۇلنىڭ ھۆججىتىنى كۆپەيتىپ ۋە باشقا يارىدارلانغانلىقىڭىزغا ئىسپات بولىدىغان ماتىرىياللارنى تەييارلاپ ئەمگەك كادىرلار ئىدرىسىغا ئېلىپ بارسىڭىز، ئۇلار يەنە تولۇقلاشقا  تىگىشلىك ماتىرىياللارنى دەپ بېرىدۇ، دېگىنى بويىچە تولۇقلىسىڭىز، بىرقانچە كۈندىن كېيىن مەكتىپىڭىزگە تەكۈرۈشكە ئادەم ئىبەرتىپ، يارىلىنىش ئەھۋالىڭىز ۋە ھازىرقى ئەمگەك ئىمگەك ئىقتىدارىڭىز توغرىسىدا ئىسپات ئالىدۇ، ئاندىن  ئىشتا يارىلانغانلىقىڭىز توغرىسىدا ئىسپات چىقىرىپ بېرىدۇ. (بۇ ئىسپاتنى ياخشى ساقلاڭ ھەم بىرقانچە نۇسخا كۆپەيتىۋىلىڭ).
يەنە داۋاملىق ئىلگىرلىگەن ھالدا  ئەمگەك كادىرلار ئىدارىسىدىكى خادىملار بىلەن ئالاقىلىشىپ، يارىلىنىش دەرىجىڭىزنى بىكىتىش توغرىسىدىكى جەدۋەلنى تەلەپ قىلسىڭىز، سىزگە ئىككى خىل جەدۋەل بېرىدۇ، بۇ جەدۋەلنىڭ ھەربىرىنى 3 نۇسخادىن تولدۇرۇپ، بۇ 6 جەدۋەل(2دىن 3 نۇسخا)گە يەنە مەكتەپنىڭ تامغۇسىنى باستۇرۇپ، ئالدىنقى قېتىم كۆپەيتىۋالغان جەدۋەل بىلەن بۇقېتىمقى 6 جەدۋەلنى مائارىپ ئىدارىسىغا ئېلىپ بېرىپ، ئىدارىنىڭ تامغۇسىنى باستۇرىسىز(ئالدىنقى قېتىم تامغۇ باستۇرغاندا ھېلىقى كۆپەيتىۋالغان جەدۋەلنى كۆرسەتسىڭىز، ئاسانراق تامغۇ باستۇرالايسىز).  بۇ ئىشلارنى تاماملاپ بولغاندىن كېيىن ئەمگەك كادىرلار ئىدارىسىنىڭ ئەڭ چوڭ باشلىقىغا قول قويغۇزۇپ، ئەمگەك كادىرلار ئىدارىسىنىڭ تامغۇسىنى باستۇرۇپ، ۋىلايەتلىك ئەمگەك كادىرلار ئىدارىسىغا ئېلىپ بارىسىز،  ئۇلار جەدۋەلنى كۆرۈپ، سىزنىڭ 300 سوم ھەق تاپشۇرىشىڭىزنى تەلەپ قىلىدۇ، ھەقنى تاپشۇرۇپ بولغاندىن كېيىن ۋىلايەتلىك دوختۇرخانىلاردىن  بىرىگە دەرىجە تۇرۇغۇزۇشقا خەت قىلىپ بېرىدۇ، خەتنى كۆتۈرۈپ بەلگىلەنگەن دوختۇرخانىغا بارسىڭىز يارىلىنىش دەرىجىسىنى بىكىتىپ بېرىدۇ، دەرىجىڭىز 4-گە چۈشسە، سىزنى ئىش ھەققىڭىزنى %100 بويىچە پىنسىيىگە چىقىشىڭىزنى تەستىقلايدۇ. 5- دەرىجىگە چۈشسە تەستىقلىمايدۇ، قانچىنجى دەرىجىدە بولماڭ سىزگە مانچە ئايلىق ئىش ھەققى بېرىدۇ ھەم مەكتەپ تۇتۇپ قالغان بارلىق پۇللىرىڭىزنى ئەمگەك كادىرلار ئىدارىسى مەكتەپ مۇدىرىنى سارغايتىپ تۇرۇپ ئېلىپ بېرىدۇ. دەرىجىنى بىكىتىش ئىشىدا كىسەللىك پىنسىيىسىگە چىققان يېقىن تونۇشلىرىڭىزنىڭ مەسلىھەتىنى ئېلىشنى ئۇنتۇلۇپ قالماڭ.
