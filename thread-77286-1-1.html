     ئەسسالامۇ ئەلەيكۇم تورداشلار ياخشىمۇسىلەر؟..مەندىن سىلەرگە ئوتلۇق سالام.
  مەن سىلەر بىلەن يەنە قايتا كۆرىشەلىگەنلىكىمگە ئۇلۇغ ئاللاھقا مىڭلاپ شۈكرى قىلىمەن...ئاللاھ بىزلەرنى بۇ دۇنيانىڭ ، گۆرنىڭ،قىيامەتنىڭ،دەۋزەخنىڭ ئازابىدىن يىراق قىلسۇن.نىجىس لەنىتى شەيتاندىن يىراق قىلسۇن...........

          مەن بۈگۈن توردا ئىشلىتىپ كىلىۋاتقان ئىككى ئىسمىمنىڭ كۆڭلىمدىكى مەنىسىنى يىشىش بىلەن بىرلىكتە ئاشۇ ئامراق ئانامنى مەدھىيلەپ كىچىككىنى قابىليىتىمنى ئىشقا سېلىپ بىر قانچە ئۆزۈم شېئىر دەپ قارىغان تۆت مىسرالىق قۇرۇق پاراڭنى بىرگە ئېلىپ كەلدىم .خالىساڭلار قاراپ قويغان بىلەن بىرلىكتە تەكلىپ پىكىرلىرىڭلارنى يېزىپ قۇيۇشىڭلارنى ئۈمۈت قىلىمەن.

                                       تەنھا
                      ( بۇ مىنىڭ چ چ دىكى ئىسمىم)

                     قاراڭغۇ كېچىدە يۈرىمەن تەنھا،
                     كۆزدە ياش ساراڭدەك كۈلىمەن تەنھا.
                    سۆيگۈ جەڭگاھىدا ئەلەملىك دادلاپ،
                    يۈرەكتىن قان تامچىپ تۇرىمەن تەنھا.
        
                    تۇنجى رەت باغچىدا كۆرگەنتىم تەنھا ،
                    مەن سىنى ھاياتتا سۆيگەنتىم تەنھا.
                    تەڭ كۈلۈپ تەڭ يىغلاپ ئۆتەيمىكىن دەپ،
                    جانىم سېنى مەھبۇبەم بىلگەنتىم تەنھا.  
         
                    شۇ ناخشاڭ ياڭرايدۇ قۇلاقتا تەنھا،
                    ئېيتقان سۆز ئۆچمەيدۇ يۈرەكتە تەنھا.
                    شۇ ناخشاڭ ئىزناسى ئاڭلىنىپ تۇرار،
                    مەن قۇرغان بىركىشلىك ئۆمەكتە تەنھا.

                    سەن باسقان شۇ ئىزلار ئۆچمەكتە تەنھا،
                    بۇ كۆزلەر ئاچچىق ياش تۆكمەكتە تەنھا.
                    يۈرەككە سەن سالغان تاتلىق ئەسلىمە،
                    قەلبىمدىن كارۋان بوپ كۆچمەكتە تەنھا.

                    بىر ئاشىق مەشۇقنى كۈتمەكتە تەنھا،
                    كۈن تۈنلەر كۈتۈپلا ئۆتمەكتە تەنھا.
                    ئاي بىلەن قۇياشمۇ ئېچىنسىمۇ گەر،
                    ئۇلار كىتىپ ئۇيەنە قالماقتا تەنھا.  

           
                              بىر شائىر يازغان شېئىر

                   تۈن قوينىدا توخۇلار سايرىغاندا قونداقتا،
                   مەن ئىدىمكى بۇ يەردە تاتلىق ئۇيقۇ ئۇخلىغان.

                  كاززاپ كەنمەن ئاشىقلىق داۋرىڭى بىلەن قەسەمكى،
                  ئاشىق بولسام بۇلاتتىم توخۇلاردىن قالمىغان.

                 دەۋاپتىمەن رەببىمگە ئىشىقدارمەن ئۆزۈمچە،
                 ھايۋان يىغلىغان يەردە ئۇنىڭدەك يىغلىيالمىغان.

  تۆۋەندىكى شېئىرىمنى ۋە ئىسمىمنى مۇشۇ شېئىردىن ئىلھام ئېلىپ يازغان.

                                    ئويغان ئوغۇل
                            (بۇ مۇنبەردىكى ئىسمىم)  

                    نە قۇدرەتتىن قانداق تۇغۇلدۇڭ؟،
                    نە سەۋەبتىن قانداق چوڭ بولدۇڭ؟.
                    نىمە قىلىپ نەلەردە يۈردۈڭ؟؟،
                    ئويلانغىن ھەم ئويغانغىن ئوغۇل.

                    ئاتاڭ ساڭا نىمىلەر قىلدى؟؟،
                    ئاناڭ ساڭا نىمىلەر قىلدى؟؟.
                    سەن ئۇلارغا نىمىلەر قىلدىڭ؟،
                    ئويلانغىن ھەم ئويغانغىن ئوغۇل.

                    رەببىمىز بىزنى قانداق ياراتتى؟؟،
                    بىزنى ئۆزىگە قانداق قاراتتى؟؟.
                    ئەھلى_ئىلىمنى قانداق تاراتتى؟،
                    ئويلانغىن ھەم ئويغانغىن ئوغۇل.

                    ئەرزىمەس سۇئىدىڭغۇ ئەسلىي،
                    كەپ قالغاچقا يارىلىش پەسلىي.
                    بولدىڭغۇ مانا ئاتىنىڭ نەسلىي،
                    ئويلانغىن ھەم ئويغانغىن ئوغۇل.

                    ئاتاڭ يالغۇز غەزەپلىك قاقشار،
                    ئاناڭ يالغۇز ئەلەملىك يىغلار.
                    ئاچچىق پىغان يۈرەكنى غاجىلار،
                    ئويلانغىن سەن ئويغانغىن ئوغۇل.

                    ئاناڭ گەپ قىلسا مازاق قىپ كۈلدۈڭ،
                    ئاتاڭ گەپ قىلسا قاپاقنى تۈردۈڭ.
                    ئېيتقىن ئۆزۈڭ نەلەردە يۈردۈڭ؟؟؟،
                    ئويلانغىن ھەم ئويغانغىن ئوغۇل.

                    ئەسلىڭگە قايت غادىيىپ تۇرما،
                    ئەمدى گەپنى باشقىغا بۇرىما.
                    ھەم مۆشۈكتەك كۆزۈڭنى يۇمما،
                    ئويلانغىن ھەم ئويغانغىن ئوغۇل.

                    تۆۋە قىلغىن سەن گۇناھىڭنى بىل،
                    ئەمدى ھەر جايدا ئاتاڭ ئىشىنى قىل.
                    بىكار ئۆتمىسۇن كۈن،ئاي،ياكى يىل،
                    ئويلانغىن ھەم ئويغانغىن ئوغۇل.

                                     سېغىندىم ئانا
                     بۇ شېئىرنى جاپاكەش ئانىلارغا يازدىم

                   ھەجەپ يېقىملىق ئەللەي ناخشىسى،
                   ئېيتاتتى ئانام ھەر كۈن ئاخشىمى.
                   لىكىن ئەمدى يوق ئۇيقۇسىز تۈنلەر،
                   ھەجەپمۇ سىنى سېغىندىم ئانا....!.

                   ئۇيقىسىز تۈنلەر يۈرەكلەر يارا،
                   گەر ئىشەنمىسەڭ يېقىن كەپ قارا.
                   كەلگىن كۆرۈپ باق يۈرەكىم مانا،
                   ھەجەپمۇ سېنى سېغىندىم ئانا...!.
           
                   قاچان قارىسام يېنىمدا بارتىڭ،
                   بالىلىقىمدا بىر سەنلا يارتىڭ.
                  مانا ئەمدى يوق ھىچ قەيەردە سەن،
                  ھەجەپمۇ سېنى سېغىندىم ئانا...!.

                  بالىلىقىمداھېچ  غېمىم يوقتى،
                  ياندا بولغاچ سەن بۇ كۆڭۈل توقتى.
                 لىكىن ئايرىلىپ قاپتىمەن ئەمدى،
                 ھەجەپمۇ سېنى سېغىندىم ئانا...!.

                 بىز ئۈچۈن ئەتكەن تاتلىق ئېشىڭنى،
                 بىز ئۈچۈن ياققان ئىسسىق نېنىڭنى.
                 مىننەتسىز بەرگەن ئىللىق مېھرىڭنى،
                 ھەجەپمۇ سېنى سېغىندىم ئانا...!.

                 ئېتىزدىن كەلگۈچە كۆزۈڭ يۇلۇمدا،
                 ئولتۇرۇپ بولغۇچە چېيىڭ قۇلۇمدا.
                 قانچە كۆيگەنسەن؟ ئانا ئوتۇمدا،
                 ھەجەپمۇ سېنى سېغىندىم ئانا...!.

                 ئەگەر ئاغرىسام ئاغرىيتتى سېنىڭ،
                 ئەگەر يىغلىسام چىقاتتى جېنىڭ.
                 ساپساق ئىدىغۇ؟ جان ئانا تېنىڭ،
                 ھەجەپمۇ سېنى سېغىندىم ئانا...!.

                 ۋاقىتسىز سەندىن ئايرىلدىم ئەجەپ،
                 تەقدىرگە كەلدى شۇنداق چىڭ غەززەپ.
                 لىكىن ئامال يوق قالدىمغۇ پەسلەپ،
                 ھەجەپمۇ سېنى سېغىندىم ئانا...!.

                 ياتقان يېرىڭىز جەننەتتە بولسۇن،
                 بارچە دۈشمەنلەر ھەيرەتتە قالسۇن.
                 سىزگە ئاللاھنىڭ شەپقىتى ياغسۇن،
                 سىزنى ھەجەپ چىڭ سېغىندىم ئانا...!.

   ئەمدى قالغىنىنى كىيىن يازاي دوسلار... قاملاشسىمۇ قاملاشمىسىمۇ مۇشۇنچىلىك يازالىدىم، قابىلىيەت شۇنچىلىك تەلىم بىرەرسىلەر ھە...!
