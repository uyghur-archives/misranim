بۈگۈنمۇ يەنە ئىچىپ كەلمەڭ،قوشنىلارنىڭ ئالدىدا گەپ قىلغۇچىلىك يېرىمىز قالسۇن.كۈندە مەست كىرىپ مەن بىلەن ئۇرۇشقىنىڭىزنى ئاز دەپ بىنادا قالايمىغان ۋارقىراپ-جار قىراپ قولۇم –قوشنىلارنىمۇ بىزار قىلىۋەتتىڭىز،بىچارە ئوغلىمىزنىڭ ئۇيقۇسىمۇ ئېچىلىپ كېتىدىكەن، يا ئۆزىڭىزگە تۈزۈك ئارام يوق يا بالا بىلەن ماڭا بالىمىزغا ئىچىڭىز ئاغرىسۇن ،شۇ بالىمىز بولمىغان بولسا مەن ئاللاقاچان ئۆينى سىزگە ئوڭچە تاشلاپ بېرىپ چىقىپ كەتكەن بولاتتىم 。ھېچ بولمىغاندا بالىمىزنى بولسىمۇئويلاڭ ئەتە شەنبە بولغاندىن كېيىن ئۇنى باغچىغا ئاپىرىپ ئوينۇتۇپ كېلەيلى بولامدۇ؟شاھىدە سۆزنى تۈگىتىپ كۆز چاناقلىرىنى قولىنىڭ دۈمبىسى بىلەن يىنىك سۈرتتى.ئالىم پۇشۇلدىغىنىچە ئولتۇارتتى. بىر ئازدىن كېيىن شاھىدە ئۇنىڭغا كۆزىنىڭ قۇيرۇقىدا سەپ سالدى.ئۇ سەل بئارام بولغاندەك قىلاتتى لېكىن ئۈندىمىدى ئورنىدىن تۇرۇپ چايدىن بىر يۇتۇم ئوتلىۋالغاندىن كېيىن:

-     ئىشقا بېرىپ كېلەي دىگەن پېتى سىرتقا قاراپ ماڭدى.

-     ھەي،مەن دىگەن گەپلەرنى ئۇنتۇپ قالماڭ. ماڭا ئىچىڭىز ئاغرىمىسىمۇ ھېچ بولمىغاندا بالىغا ئىچىڭىز ئاغرىسۇن بولامدۇ؟

ئالىم قاانداقتۇر خىياللار ئىلىكىدە ئۈزۈشكە باشلىدى .شۇنۇڭ بىلەن بېكەتتىن ئۆتۈپ كەتكەنلىكىنىمۇ سەزمەيلا قالدى. ئۇ ئاپتوبۇسقا ئولتۇرماستىن ئىدارىغا پىيادە كېتىپ باراتتى ھەي دىمىسىمۇ مەن نىمە بولۇپ قالدىم ھاراق ئىچسەملا قالايمىقان ۋارقىرايدىغان بولۇپمۇ بىناغا كىرگەندىن كېيىن شۇنداق قىلىدىغان ،ئۆيگە كىرىپ ئايالىم بىلەن ئۇرۇشىدىغان بولۇپ قالدىم توۋا ئالدىنقى كۈنى كەچتە ئۈچىنچى قەۋەتتىكى قوشنىمىزنىڭ ئىشىكىنى ئۇرغۇنۇمنى "ئۆلمەكنىڭ ئۈستىگە ،تەپمەك "دېگەندەك تېخى سەن مېنىڭ ئۆيۈمدە نىمىش قىلىسەن چىقە ئۆيدىن دەپ ئۆيدىن قوغلىغىنىمنى كۆرمەمدىغان ھەي ئاشۇ نېجىس ھاراقنى ئەمدى ئاغزىمغا ھەرگىزمۇ ئالمايمەن ئويلىغانسىرى ئۇنىڭ يۈزى چىمىلداشقا باشلىدى چىگىش خىياللار بىلەن ئىدارىگە كېلىپ قالغانلىقىنى سەزمەيلا قالدى...

ئالىم ئىشتىن چۈشۈپ ئەمدىلا ئىشخانىدىن چىقىپ تۇرۇشىغا تېلىفونى سايراپ كەتتى ئۇ سەل ئىككىلەنگەن ھالدا ئۇنى ئالدى.

_ئەسسالامۇ ئەلەيكۇم ئالىمجان ئاداش بارمۇ سەن ؟يوقاپلا كەتتىڭغۇ .

_ۋە ئەلەيكۇم ئەسسالام ياخشىمۇسەن ۋەلى قانداقراق ئەھۋالىڭ بالا جاقىلىرىڭ تېچلىقمۇ؟ئۆزۈڭ ياخشى تۇرۇۋاتامسەن دىدى جاۋابەن ئالىم.

_ھەر ھالدا ياخشى تۇردۇم، تېخى تۈنۈگۈنلا ئىچكىرىدىن كەلدىم ،ئاداش بۈگۈن كەچ قۇرۇن مەشرەپ رېستۇرانىدا مېھمانلىرىم بار ئىدى ساۋاقداشلاردىن سېنىلا چاقىردىم خاپا بولماي بىر كەلسەڭ.قانداق قىلسام بۇلار؟بارمىسام خاپا بولىدۇ ،بارسام يەنە شۇ ھدراق ئىچىدىغان گەپ دىگەنلەرنى چاقاماق تېزلىكىدە خىيال قىلىپ ئۆتتى ئۇ،تۇيۇقسىز ئۇنىڭ كاللىسىغا ئىچمىسەم ئاغزىمغا قۇيمىغاندىن كېيىن بېرىپ ئولتۇرۇپ كېلەي .ۋەلى بىلەن كۆرۈشمىگىنىمىزگىمۇ خېلى بولدى دىدار غەنىمەت دىگەنلەر كەلدى-دە،بولىدۇ دىيەلىدى ئارانلا ...

ئالىم رېستۇرانغا كەلگەندە باشقا مېھمانلار ئاساسەن كېلىپ بولغاندى ۋەلى ئالىمنىڭ كەلگەنلىكىنى كۆرۈپ قەۋەتلا خۇش بولۇپ كەتتى ...

سورۇن باشلاندى ۋەلى ئىككى رومكىغا تولدۇرۇپ قۇيغاندىن كېيىن بىرىنى ئالىمغا ئۇزارتتى.ساۋاقدېشىنىڭ يۈزى چوڭ كەلدى بولغاي ئالىم ياق دىيەلمىدى ،ۋەلى رومكىنى قولىغا ئالدى ۋەسورۇندىكىلەرنى بىر باشتىن تونۇشتۇرشقا باشلىدى.بۇياق م جۈيجاڭ،بۇياق س كېجاڭ،بۇياق ب جۈيجاڭ ،بۇ خانقىزنىڭ ئىسىمى ئالىيە،بۇياقنىڭ سوفىيە،بۇياقنىڭ مەريەم،بۇياقنىڭ ئىسىمى بولسا گۈلناز ،بۇياق ساۋاقدېشىم ئالىمجان بولىدۇ مانا شۇنداق قىلىپ تونۇشتۇرۇشلار ئاخىرلاشتى سورۇندا رومكىلار ئۇياقتىن بۇياققا ئۆتۈشكە باشلىدى بىر دەمنىڭ ئىچىدىلا بىر نەچچە بوتۇلكا قۇرۇغدىلىپ بولدى ئالىم ئىچمەيمەن دىسەم بولماس ئاياللار جېنىدا ئىچىۋاتسا شۇنداقتىمۇ مەن مەست بولماسلىقىم كېرەك يەنە مەست بولۇپ بىنانى بېشىمغا كېيىپ قۇلۇم –قوشنىلارنىڭ ئارامىنى بۇزسسام بولمايدۇ شۇنىڭ ئۈچۈن چوقۇم مەست بولماسلىق كېرەك دىگەنلەرنى خىيالىغا كەلتۈردى. سورۇندا يۇمۇر-لەتىپە دىگەنلەر ئېيتىلىشقا باشلىدى ئالىم بۈگۈن تولىمۇ ئېچىلىپ كەتكەندى شۇڭلاشقىمىكىن ئۇ يۇمۇر دىگەنلەرنى تۆكۈۋەتتى ھەممە ئۇنى بىر ئېغىزندىن ماختاشقا باشلىدى ئۇ يەنىلا ئۆزىنى تۇتۇۋېلىشقا تىرىشاتتى ...

ئاق پىشماق ،قوشما قاپاق،قارا قاش،كىرپىكلىرى تۈكۈلۈپ تۇرىدىغان ،قىرقىما چاچ،مانتا ياقىلىق كۆينىكىدىن كۆكسى مانا مەن دەپ چىقىپ تۇرغان چوكان ئۇنى ئاغزى-ئاغزىغا تەگمەي ماختاشقا باشلىدى ئالىم مەغرۇر ئولتۇراتتى بىر ھازادىن كېيىن ھېلىقى ئايال ئىككى رومكا ھاراقنى كۆتەرگىنىچە ئۇنىڭ يېنىغا كەلدى ۋە:

-قېنى ئالىمجان ئاكا تونۇشۇپ قالغانلىقىمىز ئۈچۈن بىر قەدەھتىن ئىچىۋېتەيلى مەن سىزدەك قىززىقچى،خۇش-چاقچاق ئادەمگە ئامىراق دىگىنىچە ئۇنىڭغا قەدەھ تۇتتى ئالىم ئۇنىڭ قولىنى قايتۇرمىدى ...

تارقىمايدىغان سورۇن يوق،شامال ئۆتمەيدىغان تام دىگەندەك سورۇنمۇ تارقىدى ئۇلار بىر –بىرىنى يۆلەشكىنىچە ھەرقايسى-ھەر تەرەپلەرگە يول ئېلىشتى.  

ئاخشامقىنىڭ كۈچىدىنمۇ ئىشقىلىپ ئۇنىڭ بېشى چىڭقىلىپ ئاغرىيتى ،ئۇ بىر نەچچە قېتىم ئورنىدىن تۇرۇشقا تەمشەلدى بىراق يەنىلا بېشىنى كۆتۈرەلمىدى.مانا كۈنمۇ نەيزە بويى كۆتۈرۈلدى ئۇ ئاخىرى بېشىنى كۆتۈردى شۇ ئان ئۇنىڭ كۆزى كارىۋات يېنىغا قويۇلغان ئورۇندۇق ئۈستىدىكى بىر پىيالە قايناق سۇ بىلەن بىر قاپ دورىغا چۈشتى ئۇ دەررۇ دورىغا قول ئۇزارتتى تۇيۇقسىز دورا قېپىنىڭ ئاستىغا باستۇرۇلغان بىر ۋاراق قەغەز ئۇنى مەپتۇن قىلىۋالدى ،ئۇ قەغەزنى قولىغا ئالدى ئۇنۇڭغا.

سۆيۈملۈكۈم تاماق تەييارلاپ قويدۇم دورىنى ئىچىۋېلىپ ئىسسىقىدا يەۋېلىڭ،كۆينەك بىلەن ئىشتاننىمۇ دەزماللاپ قويدۇم ئالماشتۇرۇۋېلىڭ ،مۇرتىزا تېخى ئۇيغانمىدى دەم ئېلىش بولغاندىن كېيىن ئۇخلىۋالسۇن مەن ئەتىگەنلىك بازاردىن كۆكتات ئالغىلى چىقىپ كەتتىم چۈشتە ئوخشۇتۇپ لەڭمەن ئېتپ برىمەن دىگەن خەتلە يزىلغاندى. ئۇ ئۆي ئىچىگە بىر قۇر كۆز يۈگۈرتكەندىن كېيىن ئىڭىكىنى چىمداپ قويدى ۋە توۋا ئوڭۇممۇ ؟چۈشۈممۇ بۇ؟بۇ خوتۇنغا نېمە بولغاندۇ؟ ،ئەزەلدىن قىلىپ باقمايدىغان قىلىقلارنى قىلىپ كېتىپتۇغۇ؟دىگەنلەرنى خىيال قىلغاچ ئاشخانا ئۆيگە قاراپ ماڭدى بۈگۈن ئۆي تولىمۇ پاكىزە تازلانغاندى ...ئاشخانا تەرەپتىن تاماقنىڭ مىزلىك ھىددى كېلىۋاتاتتى ئۇ ھەيرانلىق ئىلىكىدە ئاشخانىغا كىردى،ئەمدىلا 5ياشنىڭ قارىسىنى ئالغان مۇرتىزا تاماق ئۈستىلىدە تاماق يەۋاتاتتى ئۇ دادىسىنى كۆردى-دە:

-دادا!دادا ساقايدىڭمۇ دىگنىچە ئۇنىڭغا ئېتىلدى .

-مەن ئاغرىپ قالمىسام ساقايدىڭمۇ دەيسەنغۇ ؟سورىدى ئۇ بالىدىن.

-كېچە سەن بەك جېق ھاراق ئىچىۋاپسەن ئاپام بىلەن مەن سېنى بەك تەستە ئۆيۈڭگە ئەكىرىپ قويدۇق .مەن قورقۇپ يېغلىدىم.

_نىمىدىن قورقتۇڭ؟

- سەندىن قورقتۇم .

-نىمىشقا قورقىسەن؟

-سەن دىگەن يولۋاستەك بولۇپ كېتىپسەن ئۇدۇلدىكى داداشمۇ قورقتى دىدى جاۋابەن بالا.

-يەنە نىمە ئىشلار بولدى؟

-سەن زالدا قوشنىلارنى تىللاپ ئۇرۇمەن دەپ تۇرۇۋالدىڭ.ئۇدۇلدىكى داداش .

-قانداق ئاداش بۇ ھاراق دىگەننى ئىچەلمىگەندىن كېيىن ئىچمىسەڭ بولمامدۇ؟يا بىزگە يابالىلىرىڭغا ئارام يوق كۈندە دىگۈدەك شۇنداق قىلامسەن؟ دەپ تىللىغانتى سەن ئۇنى ئۇرىمەن دەپ تۇرۇۋالدىڭ.ئاپام ئۇنىڭغا ۋەباشقىلارغا.

-خاپا بولماڭلار دىدى شۇنىڭ بىلەن ئۇلار خاپا بولماي سېنى ئۆيگە ئەكىرىشىپ بەردى ئاپام پىيالىگە چاي قۇيۇپ بەرسە ئۇنى يەرگە ئېتىپ چېقىۋەتتىڭ.تېخى ئۇنى ئۇردۇڭ.مەن قورقۇپ يېغلاپ كەتتىم.ئاپام ئىككىمىز سېنى ياتىقىڭغا ئەكىرىپ قويدۇق كىيمىڭ بەك مەينەت بولۇپ كېتىپتىكەندۇق ئاپام سالدۇرۇپ قويۇۋاتسا سەن ئۇنى:

-يوقال قانجۇق مەن دىگەن خوتۇن -بالام بار ئادەم،يوقال دىگەندىن كېىيىن يوقال بولمىسا كۆرگۈلۈكىڭنى كۆرسىتىمەن.ھوي ئىپلاس سەن مېنى كىم كۆرۈپ قالدىڭ  مەن ئۆيگە كېتىمەن دەپ تىللاپ كەتتىڭ دىدى بالا جاۋابەن...

ئۇ بېشىنى ئىككى قولىنىڭ ئارىسىغا ئالدى-دە،ھەي، مەستلىك دىگىنىچە   تېگى يوق خىياللارغا بەند بولۇپ كەتتى...
دۇنيادا ئەۋلادلارنىڭ مۇھتاجلىققا قېلىشىدىن قورقۇنۇچلۇق ئىش يوق،شۇ سەۋەبلىك پەرزەنتلەرنى ئىلىم-پەن بىلەن قوراللاندۇرۇش ھەممىدىن مۇھىم...
