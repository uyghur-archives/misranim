     ______ ئادەمدىكى يۇشۇرۇن ئىقتىدارنى كۆمۈپ ياتقان يەنىلا ئادەم ئۆزىدىكى پاسسىپچانلىق.

ئادەم ئەسلىدىنلا ھەممە جەھەتتە تەلتۈكۈس يارتىلغان. ئىنساننىڭ ۋۇجۇدىدا ئادەمدە بولوشقا تىگىشلىك ھەممە ھەممە نەرسە بار، پەقەتلا ئۇ ئۆ‍زمىڭىمىز تەرىپىدىن قىزىپ چىقىرىلىپ تونۇپ يىتىلمىگەنلىكى ئۈچۈنلا «كۆمۈلۈپ قالغان».بىز ھاياتىمىزدا دائىم كۆمۈلۈپ قالغان ئالتۇنكەنسىز،كۆمۈلۈپو قالغان تالانىت ئىگسىكەنسىز دىگەندەك سۆ‍زلەرنى ئۇچرىتىپ قالمىز ياكى ئۆزىمزدىكى بىرەر ئىقتىدار شۇنچىكىملا نامايەن بولوپ قالسا باشقىلارنىڭ شۇنداق دىگەنلىكىنى ئاڭلايمىز،شۇندىلا بىز ئىتىراپ قىلامىساقمۇ،باشقىلار تەرىپىدىن ئىتىراپ قىلىنىشنىڭ شىرىن لەززىتىدىن بەھرىمەن بولىمىز،ئاغزىمىزدا دىمىگەن بىلەن سىزدىكى پاسسىپ  روھى ھالەت ئاكتىپ ھالەتكە ئۆزگەرگەنلىكىنى سىز ھىس قىلىپ يىتىسىز.مانا بۇ خىل «كۆمۈلۈپ ياتقان»ئىقتىدارلار ھەر بىر كىشىدە بار.لىكىن ئۇنىڭ قانداق ئەھۋالدا «كۆمۈلۈپ قالدىغان»لىقىنى سىزگە مەن ئۆز بىشىمدىن ئۆتكەن بىر مىسال ئارقىلىق كۆرسىتىپ بىرەيمۇ؟سىز،بىز ھەممىمىز بىرلىكتە تەرەققى قىلىمىز دەيدىكەنمىز چۇقۇم ئۆز ئارا چۈشنىشنى مۇھىم ھالقا، خىزمەتداشلار ئارا مۇناسۋەت يىقىملىق،ھىچقانداق شەخسىيەتچىلىك (شەخسى ياخشى كۆرۈش ياكى ياقتۇرماسلىق،سىلىشتۇرۇش ياكى تۇخۇمدىن تۈك ئۈندۈرۈش) قاتارلىق ھالقىلاردا دىققەت قىلماسلىق ئەھۋاللىرى كۆرۈلمەسلىكى كېرەك ئىدى. ئەمما خارەكتىرمىزدىكى ئاجىزلىقلارنى ھەم شۇ كۈندىكى ناچار پاسسىپ ھالەتنى خىزمەتكە سۆ‍ررەپ كىرىش، مۇكەممەل بولمىغان رەھبەرلىك بەنزىسىنىڭ بولىشى،خىزمەت تەقسىماتى توغرا بولماي خىزمەتداشلار ئارا توقۇنۇشنىڭ كۆپ بولوپ،ھەتتا ئۇ توقۇنۇشنىڭ خىرىدار بار يوق دىمەي ئەۋىج ئىلىشى،باشقىلارنى بىكارچىلا قىلىپ قويۇپ ئۆ‍زىنى باشقا مىللەت ئىسىللىرىغا ئوخشاش نامايەن قىلىشقا ئالدىراپ كىتىشلەركۆپ،بۇلار مىللىلار ئىچىدە ئەمدىلا باش كۆتۈرۈۋاتقان  مىللىتىم ئۈچۈن خىزمەت قىلىپ ھەم ئۆزەمنىڭ جەمئىيەتتە بىر كىشلىك ئىناۋىتىمنى تىكلەيمەن ،ھەم ئۆزۈمگە ئىقتىسادى قىممەت يارىتىمەن دىگەن گۈزەل غايە ھەم ئارزۇ ئارمانلارغا تولوپ تۇرۇپ ئەمدىلا تەرەققى قىلماقچى بولۋاتقان، خەلىقىمىز ئەمدى ئۆزلىشىۋاتقان مەزگىللەردە قوپوپلا تۇيۇقسىزلا تاقىلىپ قىلىشقا يۈز لىشىش قۇرغۇچىلىرمىزدا پاسسىپ بولغان بىر روھى ھالەتنى شەكىللەندۈرگەن بولسا ،ئىشلىگۈچىلەردە نىمە پاسسىپ بولغان روھى ھالەتنى پەيدا قىلدى؟مۇئاشنىڭ تۆۋەن بولىشىمۇ؟ياق،ئۇلار جەمئىيتىمىزدە بىر تۈركۈم شۇ قۇرغۇچىلارغا ئوخشاش ئەركىن كەسىپ تەرەپدارلىرى ،خىزمەت ھەم جەمئەيەتتە بىر كىشلىك ئوچۇن ئۈچۈن زار بولۋاتقان، ئۆزىدىمۇ شۇ خىزمەتنىڭ ھۆددىسىدىن چىقىپ كىتەلەيمەن دىگەن ئاكتىپ روھى ھالەتنىڭ ئەمدىلا باش كۆتۈرۈشى تۈرتكىسىدە سىز بىلەن ھەمكارلىشىۋاتقان، بىر تۈركۈم شۇنداق «كۆمۈلۈپ قالغان»تالانتىنى نامايەن قىلىش ئۈچۈن سىزنىڭ ئورنىڭىزنى تاللىغان، ھايات يولىداتۇرمۇش دوقمۇشلىرىدا تىڭرقاپ سىزنىڭ ئورنىڭىزنى ئۆزىنىڭ تاللاش نۇقتىسى قىلغان ياشلار، سىز ھەرگىز ئاللىقاچان ئىديەسى كاردىن چىققان،سىزگە ھىچقانداق يىڭلىق ئىلىپ كىلەلمەيدىغان ياشانغان بوۋاي مومايلارنى تاللاپ ئىشلەتكىنىڭىز يوق،بىر تۈركۈم ھاياتى كۈچى ئۇرغۇپ تۇرغان ياشلارنى تاللىدىڭىز،ئۇلارنىڭ پاسسىپ روھى ھالەتتە بولىشىغا نىمە سەۋەپ بولدى دىيىشىڭىز مۈمكىن، ئىنىق بولمىغان ئىش تەقسىماتى ھەم سىزنىڭ ئىلمى بولمىغان باشقۇرۇش ئۇسۇلىڭىز ئۇلارنىڭ سىزنىڭ سودىڭىز ئۈچۈن جان تىكىپ پىداكارلىق بىلەن خىزمەت قىلىپ بىرىش كەيپىياتىنى ئاللىقاچان ئۇچۇرۋەتكەن،شۇڭا سىزنىڭ سودىڭىز يۈرۈشمەيۋاتقاندەك،ئۇلارنىڭ قايسىدۇر بىرى سىزگە ھەم سىزنىڭ پۇللىرىڭىزغا ئاسيىلىق قىلىۋاتقاندەك تۇيغۇ بىرىدۇ.سىزدىكى بۇ خىل تۇيغۇ ئۇلاردا بۇ جەمئىيەتتە سەن تىخى ئۆ‍ۈڭگە تەۋە بىر كىشلىك ئورۇن تاپالمىدىڭ دىگەن بىر خىل پاسسىپ ھىسياتنى سوۋغا قىلىدۇ. سىزدە شۇنچىلىك رەھبەرلىك ئىستىداتى يوق تۇرۇپ،چىرايلىق سۆ‍ز ، ئىشەنىچلىرىڭىز بىلەن ئىشچى خىزمەتچىلىرىڭىزنىڭ ئىستىداتىدىن پايدىلىنالىغىدەك سەۋىيە يوق تۇرۇپ جەمئىيەتتىن ئادەم يىغىپ نىمە قىلسىز؟شۇڭا تەكلىپىم،بىرىنجىسى ئۆ‍زڭىزنىڭ ھەم سىز قۇرغان شۇ شىركەتنىڭ ئىھتىياجى نىمە؟قانداق ئادەمگە ئىھتىياجىلىق،بىرسىنىڭ قىلغان ئىشىنى يەنە بىرى ئۈستىگە ئىلىپ كىتەلەمدۇ؟ئادەتتە ئىش تەقسىماتى ئىنىقمۇ؟مانا بۇلارنى ئويلىشىسىز،بىرسىنىڭ ئىشىنى يەنە بىرسى گەپ سۆ‍زسىز ئۈستىگە ئىلىپ كىتەلىسە سىز يەنە بىر كىشنى قىسقارتىشىڭىز كىرەكمۇ يا باشقۇرۇش تۈزۈلمىڭىزنى مۇكەممەللەشتۈرۈشڭىز كىرەكمۇ؟ بۇلارنى قىلالىغىدەك  زىھنىڭىز بولسا زىھنىڭىزنى شۇلارغا چىچىڭ،ئۇششاق  ئشىلار ئۈچۈن كاللىڭىزنى ئۇپۇرتۇپ يۇرمىگىدەك ئىشنى قىلمامسىز؟شۇندىلا سىز ناھايتى مۇۋاپپىقىيەت قازانغاندەك تۇيغۇدابولغان ھەم مۇۋەپپىقىيەت قازنىش ئۈچۈن خاتىرجەم زىھنىڭىزنى سەرىپ قىىپ يەنىمۇ مۇكەممەل پىلانلارنى تۈزەلىگەن بولاتتىڭىز،  سىزنىڭمۇ «كۆمۈلۈپ قالغان»زىھنى تالانتىڭىز جارى بولوپ كىتىپ ئۆ‍ز كەلگۈسىڭىز ھەم كەلگۈسى ئەۋلادلىرىڭىزنىڭ كەلگۈسى ئۈچۈن تىخىمۇ مۇكەمەل بولغان لايھەلەرنى تۈزۈپ چىقىپ تەرەققى قىلغان بولاتتىڭىز.ئەپسۇس بىزلەرىدكى مۇكەممەل بولمىغان خىزمەت شارائىتى،ئىستىداتلىق بولمغان رەھبەرلىك بەنزىسى ،ئىنىق بولمىغان خىزمەت تەقسىماتى شۇنداقلا پاسسىپ بولغان روھى ھالەتنى باشقىلارغابىلدۈرۈپ قويۇشلار،ئىشچى خىزمەتچىلەرنىڭ بەزى بىر قىىسم شەخسى قىزىقىشلىرىغا ئىلمى بولمىغان ئۇسۇلدا ئارلىشۋىلىش،يىڭى كەلگەن خىزمەتچىلەرگە مۇئامىلە قىلىشتا بىر تەرەپلىمە كۆز قاراشنىڭ تەسىرىدە ئۇنىڭدىن ھەر خىل ئىۋەنلەرنى تىپىش ھەم ئۇنىڭدىن مەلۇم مەزگىلدىلا نۇرغۇنلىغان نۇقسانلارنى تىپىپ،زىھنىڭىزنى خۇددى كىنولاردىكىدەك تولوپ تاشقان تالانىتلارنى ئۆزىگە مۇجەسسەملىگەن يەنە بولسا تىخى ئالى تەلىماتلارنى ئالغان كىشلەرنىڭ كىلىشنى ھەم ئۇلارنى تولىمۇ ئەتىۋارلاپ ئىشلىتىش شىرىن چۈشىنى كۆرۈپ،ئۆزىڭىزنىڭ بۇ ئورنىڭىزنىڭ تىخى ئۇ دەرىجىدە ئىسىل تاللاش ئوبيىكتى ئەمەسلىكىنى تەن ئالماي،ھە دىسىلا خىيالى خادىم ئىلىش ھەم ئالماشتۇرۇشتىلا بولىدىغان بۇ خىل شىركەتنىڭ ئىستىقبالى ھەم تەرەققىياتىدىن سۆ‍ز ئاچقىلى بولامدۇ؟يەنە ھىس قىلغىنىم مىللى شىركەتلەر ئارا بىردەكلىكنىڭ بولماسلىقى مىنى قاتتىق ئەپسۇسلاندۇردى.ھەمكارلىشىش ئەمەس بىرىنىڭ ئۇلىنى يەنە بىرى كولاش،بىرىنى بىرى يامان قىلىشتىن قورىقماستىن يامان تەرىپىنى خىرىدارلارغا ئېيتىپ ئۆز ئىناۋىتىنى ھەم باشقىلارنىڭ ئىناۋىتىنى تۆكۈش،خىرىدارنى   ئۆزىگە كۆندۈرۈش ئۈچۈن بازارنى قارغۇلارچە قالايمىقان قىلىپ بىردەك بولمىغان باھا پەرقىنى كەلتۈرۈپ چىقىرىش مەن ھىس قىلغان  ئىككىنجى مەسىلە. مەن ئىشلىگەن مۇشۇ ئۇيغۇرچە ماتىرىياللارنى بىسىپ چىقىرىش ھەم توختاملارنى بىسىشتا توختام بىسىش باھاسىنىڭ ئوخشاش بولماسلىقى يەنە كىلپ تولىمۇ مۇكەممەل بولغان،ھەتتا ئادۋوكاتمۇ ئارانلا شۇنچىلىك يىزىپ بىرەلىگۈدەك توختاملارنى تولىمۇ ئەرزان باھادا يىزىش يەنە تىخى ئۇرۇپ چىقىرىپ بىرىشى مىنى تولىمۇ ئەپسۇسلاندۇردى.ئەسلى قىلىدىغان خىزمىتىمىز پەقەتلا باشقىلارنىڭ يازمىچە يىزىپ كەلگەن نەرسىسىنى ئۇرۇپ چىقىرىپ بىرىش ئىدى. پەقەت شۇنداق بولغاندىلا بىز ئاندىن شۇ تۆۋەن باھانى ئالساق مۇۋاپىق بولاتتى.ئەمما ھازىرچۇ؟بۇرۇن مەن دەۋا دەستۇردا يۈرگەن مەزگىللىرىم، مەن ئۇرۇپ چىقىرىش باھاسىنى تۆۋەنىرەك بولىشى ئۈچۈن ئەرىزنامىنى ئۆزۈم يىزىپ كىلىپ ئۇرغۇزدىغان بولغان،چۈنكى مەن دەسلىپىدىلا ئادۋوكاتتىن بۇ دەۋا ئۈچۈن مەسلىھەت سوراپ مەسلىھەت سوراش ھەققىنى ئايرىم تۆلىگەندىن سىرىت،ئۇلارغا ئۆلچەملىك بولغان بىر ئەرىزنامىنى يازغۇزۇش ئۈچۈن ئاز بولغاندا 50 يۈەن خەجلىگەن،كىيىن خەت ئۇرۇش ئورۇنلىرىدا «بىز ئادۋوكات ئەمەس،بىز پەقەت ئۇرۇپ چىقىرىپ بىرىمىز،يىزىپ كەلگەن نەرسىڭىزنى ئوقۇپ بىرىڭ ئۇرۇپ چىقىرىپ بېرىمىز»دىگەندەك سۆزلىرىنى تولا ئاڭلىغان،ئەگەر ئۆزىمىز ئويلاپ يىزىپ بىرىدىغان بولساق باھاسى يۇقىرىراق بولىدۇ دىگەننى ئاڭلاپ،ئەجەپكەن توۋا دىگەن خيالدا بولغان.لىكىن كىيىن ھىس قىلسام ھەممىسى شۇ ئۆزىدىكى شۇ «كۆمۈلۈپ تۇرغان»ئىقتىدارلىردىن پايدىلىنىپ بازاردا بىر كىشلىك قەدىر  قىممىتىنى ھەم ئىقتىسادى قىممىتىنى ياراتماقچى بولغان،ئەلۋەتتە ئۇلارنى ئىزدەپ كەلگۈچىلەرمۇ ئىتىراپ قىلىدۇ.ئۇلار بىلىم دەسمايە قىلىپ بۇ يەرگە سالغان مەبلىغىنى قايتۇرۇپ كەلمىسە،ئۇلاردا يەنە بۇ يەرگە داۋاملىق دەسمايە قىلىپ سالغۇدەك نەدە شۇنداق ئارتۇق مەبلەغلىرى بار؟قورسقىمىدىكى بىلىمىمىزنىڭ  تەننەرخى شۇنچە تۆۋەنمۇ؟ئۇيغۇرچىنى خەنزۇچىغا تەرجىمە قىلدۇرۇپ توختاملارنى يازدۇرۇپ چىقىپ كىتىدىغانلارمۇ بار،ئۇنداقلار ئەگەر خەنزۇچىنى ئۇيغۇرچىغا تەرجىمە قىلالىسا بىزنى ھەرگىز ئىزدەپ كەلمەيدۇ.سىز بىزنىڭ شۇنچىلىك ئىقتىدارىمىز بولغانلىقى ئۈچۈن شۇنداق قىلىدۇ.خەنزۇچە خەت بىسىش ئورۇنلىرىدا ئاران 3-4 يۈەنلا تۇرسا،بىز ئۇنىڭدىن يۇقىرى دىسەك تىخى مۇشۇ باھانىمۇ قىممەت كۆرۈپ چىقىپ كىتىدۇ دىگەن باھانىنى ئاڭلاپ ھەيران قالدىم.توغرا دەيسىز،ئۇلارنىڭكىدە شۇنداق ئەرزان ئىكەن چۈنكى بۇلارنىڭ بىسىش ئورۇنلىرى ھەممىلا يەردە بار،شۇنداق باھا بولمىسا چەت ياقا جايلاردىكىلىرى تاقىلىپ قىلىش مۈمكىن،لىكىن يەنە بىر سەۋەپ ئۇلارنىڭ ئۇرۇپ چىقىرىپ بىرىدىغىنى باشقىلارنىڭ مەن يۇقىرىدا ئېيتىپ ئۆتكىنىمدەك تەييار يىزىپ ئىلىپ كىرگەنلەرنى ئۇرۇپ چىقىرىپ بىرىشى شۇنداق. باشقا دۇككان ئىگىللىرى ھەرگىز ئۇلارنىڭ دۇككىنىدا يىغىن ئاچقاندەك ئولتۇرۋىلىپ مەسلىھەتلىشىپ بىر خەتنى بىر يازغۇزۇپ ئۇلارنىڭ ئاغزىغا قاراپ ئولتۇرۇپ ۋاقىت ھەم زىھىن سەرەپ قىلىشغا يول قويمايدۇ،شۇنداقراق ئىش بولسىلا چىقىپ  سەن ھەممىنى ئىنىق قىلۋىلىپ يىزىپ ئىلىپ كىر بىز ئۇرۇپ چىقىرىپ بىرىمىز دەيدۇ،2-3 يۈەن ئۈچۈن ئۆزىنىڭ كىچككىنە باسمىخانىسنى قالايمىقان قىلىپ يۈرمەيدۇ.مانا بۇنىڭدىن ئۇلارنىڭ باھانى شۇنداق قىلىشنى چۈشەنگىلى بولىدۇ.ئۇلارغا قولايلىق بولغان تەرىپى ئۇلار پۈتۈنلەي خەنزۇرچە ئۇرۇپ چىقىرىپ بىرىدۇ،سەل قانۇنلۇق قانۇنى نەرسىلەر سوتقا يوللايدىغان نەرسىلەر،توختام دىگەندەك نەرسىلەر بولسا ئۇلارمۇ يازغان نەرسىنىڭ مۇھىم ياكى ئەمەسلىكىگە قاراپ باھا ئالىدۇ.چۈنكى ئۇلار بىلىدۇبىر قىتىم نەرسە باستۇرغان ئادەم ئىككىنجى قىتىم ئۇ يەرگە قايتا كەلسە كىلىدۇ بولمىسا يوق،بۇلاردا خىرىدار كۆندۈرۈش دىگەن خام خىيال يوق.مىللى كارخانلىرمىزچۇ؟ئىقتىسادى قىممەتنى تۆۋەنلىتىش ئارقىلىق باشقىلارغا نەپ يەتكۈزۈش ئارقىلىق سودىسىنى راۋاجلاندۇرۇشنى خيال قىلىدۇ.ئۇ بىر ئەخمىقانىلىق.ھازىر كىشىلەرنىڭ ئىڭى ئېچىلىپ كەتتى. ئۇلار باھا ئەمەس قىلغان ئىشڭىزنىڭ سۈپىتىگە قارايدىكەن،بولوپمۇ بىر مۇھىم توختامنى ھەممە كىشى مۇكەممەل تۈزۈپ،ئۆزىنىڭ ھوقوقىنى ھەم كىيىنكى كۆرلىدىغان مەسلىلەرنىڭ يۈز بىرىشنىڭ ئالدىنى ئىلىشنى مەقسەت قىلىدۇ.شۇڭا سىزنىڭ قىلغان ئىشڭىزنىڭ ئەتراپلىق بولىشى ھەم ئۇلارنىڭ ھەر بىر ئېغىز سۆزلىرىنى تولىمۇ ئىلمى بولغان ئۇسۇلدا يىزىپ چىقىرىپ ئۇلارنى رازى ھەم خاتىرجەم قىلغىنىڭىز ئۈچۈن ئاندىن سىزنى باشقىلار كۆپىرەك ئىزدەپ كىلىدكەن.مەن  بىر نەچچەيلەننىڭ "سىلەرنىڭ يازغان توختامىڭلاركەنتۇق،بەك مۇكەممەل بوپتىكەن شۇڭا شۇنداق بىر توختام يازغۇزايلى دىدۇق"  دىگەن تەرىپلىرنى ئاڭلاپ ئۆز بىلىمىم ئارقىلىق ئاز تولا بولسىمۇ قىممەت ياراتقىنىمدىن خوش بولدۇم،ئەلۋەتتە ئۇنداق دەپ كەلگەنلەرگە باھا جەھەتتە چۈشۈرۈشنىڭ ئەمەس بەلكى مۇكەممەل بولغان بىر توختامنىڭ لازىم ئىكەنلىكىنى چۈشنۈپ يەتكىلى بولىدۇ،  مانا بۇ بازار ئىھتىياجى.ئۇلار ھەم سىزنىڭ شۇ باھاغا بولغان ئھتىياجىڭىزنى چۈشنىدۇ.چۈنكى سىز بىلىم ھەم سەۋىيە ئشلىتىسىزشۇنىڭغا لايىق ھەق ئالسىز.ئەمما خىرىدارلاردا  نارازلىق پەيدا قىلىۋاتقان ئامىل نىمە؟ھەرگىز ئۇلارنىڭ ئۆز‍ىدىكى ئىش ئەمەس،بازاردىكى  قالايمىقانچىلىق،شۇ تۆت تاللا ئۇيغۇر ئاچقان دۇكانلاردا باھانىڭ بىردەك بولماسلىقى،بىرسنى يەنە بىرسى خىرىدارىمنى تارتىۋالدى دىگەن خىيالدا سۈپەتنى يۇقىرى كۆتۈرمەي،باھانى چۈشۈرۈش ئارقىلىق ئۆزىنىڭ كۇرسىنى چۈشۈرۋاتقانلىقىنى كۆرۈپ تۇرۋاتىمىز.ئەپسۇس  بۇلاردا كام بولىۋاتقىنى نىمە؟نەق مەن يۇقىرىدا ئېيتىپ ئۆتكەندەك ئىلمى بولمىغان قىممەت قارىشى.بۇلارنىڭ ھەممىسى ئۆز ئارا ھەمكارلىق ئەمەس، بەلكى ئۆز ئارا رىقابەت تۇيغۇسىدا بولىشى.يەنە بىرى ئۆزىنىڭ ئالدىغا كىرگە خىرىدارغا  يەنە بىر رىقابەتچىسىنىڭ يامان تەرىپىنى سۆ‍زلەپ يۈرىشى بىزلەردىكى ئەجەللىك بولغان رىقابەت ئىڭى.ئەگەر ئۆز ئارا ھەمكارلىشىش ئىڭى بولغان بولسا چۇقۇم ھەر بىرىنىڭ ئۆز ئالدىغا خىرىدارلىرى بولغان بولاتتى.ئەمما بۇ خىل ناچار بولغان كەيپىياتلارنىڭ بارلىق مىللى كارخانلىرىمىزغا تەسىر كۆرسىتىۋاتقانلىقىنى مەن بىر نەچچە ئاي ئىشلىگەن خىزمەت جەرىيانىدا ھىس قىلىپ يىتىپ نىمە ئۈچۈن ئەندىلا باش كۆتۈرگەن مىللى كاخانىلىرمىزنىڭ بوۋاق ھالىتىدىلا بۆشىكىدە تۇنجۇقۇپ قالدىغانلىقىنى ھىس قىلىپ يەتتىم.ئەپسۇس تولىمۇ ئەپسۇسمەن بۇۇرنلاردا كىرگەن يەرلەردىكى خۇددى ماشىنا ئادەملەردەكلا بىر خىل تەبەسسۇم بىلەن،بىر خىللا مۇئامىلىدە،تولىمۇ يالغان ئەمما ئىلمى بولغان مۇئامىلىلەرنىڭ،يەنە يادلىۋالغاندەكلا بىر ئېغىز سۆزنىڭ يەنى «ھەممىلا يەردە ئوخشاش مۇشۇنداق" دىگەن سۆزنىڭ كەينىگە تولىمۇ ئىلمى بولغان بىردەكلىكنىڭ يۇشۇرۇنغانلىقىنى مۇشۇ بىر ئىككى ئاي جەرىيانىدا چوڭقۇر ھىس قىلىپ يەتتىم.ئۆز ئارا خىزمەتداشلار ئارسىدىكى ئورۇنسىز پاراڭ ھەم  يىقىنچلىقلارنىڭ باشقا خىرىدارلارنىڭ ئالدىدىلا تارتىنماستىن قىلىنىۋىرىشنىڭ، يەنە كۆڭۈلسىز كەيپىياتلىرنىڭمۇ باشقىلارغا بىرىدىغان تەسىرىنىڭ سودىدا ھەممەيلەنگە ئىلىپ كىلىدىغان ناچار پاسسىپ تەسىرىنى قاتتىق چوڭقۇر تونۇپ يەتتىم.خىزمەت ئورىنىدا ماشىنا ئادەم،سىرتقا چققاندىلا ئاندىن يىقىن دوس بۇرادەر،تونۇش بىلىش بولوشنىڭ مەلۇم بىر شىركەت ياكى كارخانىغا نىسبەتەن تولىمۇ مۇھىم ئىكەنلىكىنى.بۇنىڭ ئۈچۈن يەنىلا يۇقىرىدا ئېيتىپ ئۆتكىنىمزدەك ساپالىق رەھبەرلىك بەنزىسى ھەم ئىناق ئىتتىپاق بولغان،بىردەك بولغان سودا سىتىق كەيپىياتنىنىڭ بولىشى،يەنە ئىلمى بولغان خىزمەت كۆز قارىشىنىڭ بولىشى،مۇقىم بولغان ئىش تەقسىماتىنىڭ بولىشنىڭ مۇھملىقىنى تۇنۇپ يەتكۈدەك ئاڭلىق بولغان بىر باشقۇرغۇچىغا ئىھتىياجلىق ئىكەنلىكىنى،يەنە مۇشۇ نۇقتىنى تونوپ يىىتپ ئۇيغۇر بازارلىردا ئۆز قەددىنى تىكلەپ كىلىۋاتقان بەزىبىر كارخانلىرىمىزنىڭ شۇنداق باشقۇرغۇچىلارنى تەكلىپ قىلىپ ئىشلەتمەكچى بولۋاتقانلىقىنى بىرمۇ بىر سىلىشتۇرۇپ كۆردۈم.تەرەققى قىلماقچىمۇ سىز؟مەن يۇقىرىدا ئېيتىپ ئۆتكەندەك ئەھۋاللار سىزنىڭ ئورنىڭزدا مەۋجۇتمۇ ئەمەسمۇ؟سىز بۇ مەسلىلەرنى تۈزەپ كىتىشكە ئىشەنچىڭىز بارمۇ؟مىنىڭچە سىزدە پەقەت مۇشۇ نۇقتىلارنى تۇنۇپ يەتكىنىڭزدىلا بىشڭىزنى ئايلاندۇرۇپ تۇرغان قانداق مەسلىلەر بولسا ئۇلارنىڭ ھەممىسى بەك تېز ھەل قىلىنىپ كىتىدۇ.1.سىز خىزمەت ئورنىڭزنىڭ ئىھتىياجىنى تۇنۇپ،جەمئىيەتتىن خادىملارنى تاللاپ ئىشلىتىڭ ھەم ئۇلارنىڭ ئارتۇقچىلىقىنى نامايەن قىلدۇرشىغا سەھنە ھازىرلاپ بىرىڭ.2.خزمەت تەقسىماتىنى ئىنىق قىلىڭ،سىزنىڭ ئورنىڭدىكى بارلىق خادىملىرڭىزنىڭ (مەيلى ئاز بولسۇن كۆپ بولسۇن)ئۆز ئالدىغا ئىنىق ئىش تەقسىماتى بولسۇن،يەنە بىرنىڭ ئىشىغا ئارلاشمايدىغان بولسۇن ھەم ئۆز خىزمەت دائىرسىدىكى ئىشنى مۇكەممەل بىر تەرەپ قىلىش ئۈستىدىلا ئىزدەنسۇن.3.خىزمەتداشلار قوشۇنى ئىناق ئىتپاق بولسۇن،سىزنىڭ ئۇلارغا بولغان ئىشەنچىڭىز ئۇلارنىڭ ئاكتىپ روھى ھالىتىنى يۇقىرى كۆتۈرسە كۆتۈرسۈنكى سىزدىكى پاسسىپ روھى ھالەت ئۇلارنىڭ ئاكتىپ روھى ھالىتىنى پاسسىوپلاشتۇرۋەتمىسۇن.سىز ئىلاجى بار ئۇلاردا بار بولغان «كۆمۈلۈپ ياتقان»ئىقتىدارلاردىن پايدلىنىشنى ئۆگۈنۈڭ.4.باشقا ئوخشاش كەسىپتىكىلەر بىلەن مۇقىم بولغان ھەمكارلىشىش ئىڭىنى تۇرغۇزۇڭ،بازارنى بىردەك قىلىپ ھەممىمىز ئۆزىمىزنى يەنىمۇ مۇكەممەلشتۈرۈشكە ئۇلار بىلەن پىكىر بىردەكىلىكىنى قولغا كەلتۈرۈڭ.يەنى ئۇلا بىلەن سۆھبەتلىشڭ ئۇلارنى رىقابەتچىڭىز قاتارىدا كۆرۈپ ئەمەس ھەمكارلاشقۇچلىرىڭىز قاتارىدا كۆرۈپ سۆھبەتلىىشڭ ھەمبىر خىل باھانى بىسىپ ئۇرۇپ تارقىتىپ ھەممە يەردە باھانىڭ بىردەك بولىشنى قوغلىشىڭ  بۇندىن كىيىنكى ئىشلىرڭىزدا يەنىمۇ چوڭقۇر ىىيالالرنى قىلىپ ھەم ئەمەلگە ئاشۇرۇپ كىتىشڭىز ئۈچۈن بىلىم ساپايىڭىزنى يەنىمۇ يۇقىرى كۆتۈرۈشكە تىرىششڭىزنى،ئىلمى بولمىغان ئۇسۇلدا باشقىلارنى ئۆز سودىڭىز ئۈچۈن پايدىلنىش قىممىتى ياراتقۇزىمەن دىمەي،ئۆز ئىقتىدارىڭىزنى يەنى پىسخىكا ساپايىڭىزنى ھەم باشقۇرۇش ساپايىڭىزنى يۇقىرى كۆتۈرۈپ ئاندىن تىجارەت دائىرىڭىزنى كېڭەيتىڭ،ئاغزاكى پىلاننى تۈزۈپ سودا دائىرىسنى كىڭەيتىش بەك ئاسان،لىكىن شۇنى تەرەققى قىلدۇرۇش ئىھتىياجى يەنىمۇ يۇقىرى.بۇنىڭ ئۈچۈن سىز قارغۇلارچە ئادەم ئىشلىتىش ھەم بوشتىشقا ئىسراپ قىلغان زىھنىڭىزنى قول ئاستىڭىزغا سىز ئۈچۈن جان كۆيدۈرۈپ ئىشلەيدىغان بىر تۈركۈم قابىل ياردەمچىڭىزنى كۆپەيتىشكەسەرىپ قىلىڭ،شۇندىلا مۇشۇ رىقابەتكە تولغان سودا دۇنياسىدا ئۆ‍ز ئورنىڭىزنى ساقلاپلا قالماي،يەنىمۇ مۇستەھكەملىيەلەيسىز.
