مەن بۈگۇن بىر كۈن ئىشلىدىم ،بەكمۇ ھىرىپ كەتكەن ئىدىم ،ئىشتىن چۇشەي دەپ تۇراتتىم بىر دوستۇمدىن تىلفون كەلدى، ئۇ ماڭا دەيدىعان  گەپ بار كۆرۇشەيلى دىدى.مەنمۇ ماقۇل بولدۇم ،ھەم ئۇنىڭ بىلەن كۇرۇشۇش ئۇچۇن كىرا ماشىنىسىعا ئولتۇرۇپ حۇتەن ئالىي پىداكوكىكا تىحنىكومى  ئالدىدا چۇشتۇم، ئۇ دوستۇمعا تىلفۇن قىلسام بىر ئاز ۋاقىتىن كىيىن چىقاي ساقلاپ تۇرعىن دىدى.مەن ساقلاپ تۇردۇم. ئەتراپقا قاراپ تۇراتتىم شۇ مەكتەپ ئىچىدىن ئىككى ئوقۇعۇچى قىز چىقىپ كەلدى.ئۇ قىزلارعا قاراپ ھەيران قالدىم ،ئۇلار ئىككىلىسىنىڭ كاللىسىدا ياعلىقمۇ يوق ،چاچلىرىنى چۇۋۇپ دۇمبىسىگىچە چۇشرۇۋالعان.يۇزلىرىدىكى ئۇپا ئەڭلىكنىڭ كۆپلىكىدىن كىم ئىكەنلىكىنىمۇ تۇنۇعۇلى بولمايدىعان،كىرپىكلىرى {كۆك ئەرۋاھ}نىڭكىدەك ئۇزۇن ،كۆزلىرى قاپقارا بۇلاق كۆزلۇك،بىرسى كۆكسى ئىچىلىپ تۇرىدىعان ئاق رەڭلىك كۆينەك،پايپاق ئىشتان ئۇستىگە قىسقا تار يوپكا كىيگەن ئىدى،پۇتىدا ئۆتۇك بار ئىدى،يەنە بىرسى تار بەدىنىنى چىڭ سىقىپ تۇرىدىعان كۆكسى ئۇچۇق سىرىق رەڭلىك كۆينەك كىيگەن ئىدى پۇتىعا ئۆتۇك ،پايپاك ئىشتان ئۇستىگەتار يوپكا كىيگەن ئىدى،ئۇلار مەكتەپتىن چىقىپ يولدىن ئۆتتى،ئۇلارنى كۆرۇپ بىر ئىسىل ماشىنىدىن بىر ئەر چۇشتى ،مەن يىراقتا بولعاچقا ئۇ ئەرنى ئۇلارنىڭ بىرسىنىڭ دادىسى ياكى ئاكىسى دەپ پەرەز قىلدىم،شۇلارعا قاراپ تۇراتتىمدوستۇم چىقىپ مىنىڭ دىققىتىمنى چاچتى،دوستۇم بىلەن كۇرۇشۇپ بىرگە تاماق يىيىش ئۇچۇن يولدىن ئۆتۇپ ئاشحانىعا ماڭدۇق،ئۇقىزلارنىڭ يىنىدىن ئۆتتۇق،ئۇلاردىن كۇچلۇك ئەتتىرە پۇرىقى كىلىۋاتاتتى،مەن كۆڭلۇمدە ئۇ ئەر بىرەرسىنىڭ دادىسى ياكى ئاكىسى بولسا ئۇلارنىڭ بۇنداق چاكىنا يۇرىشىگە يول قويامدىعاندۇ دەپ ئويلىدىم،ئۇلارنىڭ گىپىگە قۇلاق سىلىپ ماڭدىم،ئۇلار قەشقەر تەلەپپۇزىدا گەپ قىلىۋاتاتتى،نىمە دىيىشكەنلىكىنى ئۇقالمىدىم。تاماق يەۋىتىپ يەنە حىيالىم شۇلاردا ئىدى.قەلبىمدە ئۇلارعا سۇئاللار بەردىم.ئۇلار بۇنى كورسە ئۆزىنىڭ ئوقۇعۇچىلىق ھاياتىعا قايتىپ قالارمىكى دەپ ئويلايمەن،

ئۇلارعا سۇئال:ئوماق قىزلار سىزلەر بۇ مەكتەپنىڭ ئوقۇعۇچىسىمۇ ياكى مودىللىرىمۇ؟ھازىر سىزلەرگە قاراپ كىممۇ سىزلەرنى ئوقۇعۇچى دىسۇن؟حۇددى جەمىيەتتە ئويناپ دىبادىن چىقمايدىعان قىزلارعا ئوحشاپ قاپسىزلەر؟سىزلەر ئوقۇعۇچى بولعاندىكىن جەمىيەتتىكى قىزلاردىن پەرىقلىق بولساڭلار بۇلاتتى.سىزلەردە ئىلىم-مەرىپەت ،ئەدەپ ئەحلاق ،شەرمىي –ھايا گۇللىرى پۇرەكلەپ تۇرسا بۇلاتتى،سىزلەر  بەلكى بىرەر فاكۇلتىتنىڭ ئەلەچى ئوقۇعۇچىلىرى بۇلىشىڭلار مۇمكىن،ئەمما ھازىر مودىللاردەك ئەگرى دەسسەپ دەلدەڭشىپ يۇرۇيسىزلەر؟سىزلەرنى ئاتا-ئاناڭلار مۇشۇنداق يۇرسۇن دەپ ئوقۇشقا يولعا سالعانمۇ ؟ياكى ياحشى ئوقۇپ كەلگۇسىدەياحشى حىزمەتكە چىقىپ كەلگۇسىنىڭ عۇنچىلىرىنى ياحشى تەربىيلىسۇن دەپ مۇ؟سىزلەر مۇشۇنداقلا ئوقۇش پۇتتۇرۇپ قالسىڭىز،سىز تەربىيلەيدىعان عۇنچىلار سىزدەك مودىل بۇلىدۇ ئەلۋەتتە، نىمىگە ئاساسەن ئۇنداق دەيسىز دىدىڭىزما؟ مەن كۆردۇم،مەن ئولتۇرىدىعان ئۆي ئەتراپىدا بىر باشلانعۇچ مەكتەپ بار.ئۇ مەكتەپتە يىڭىدىن ئوقۇش پۇتتۇرگەن ئىككى حانىمچاق بار، ئۇلار ئۆزى ئوقۇتقۇچى بۇلۇپ چاكىنىراق يۇرۇيدۇ .شۇنچە تار ئىشتان،بەدەننى سىقىپ تۇرىدىعان كىيىملەرنى كىيىدۇ،چىرايى ئۇن تارتقاندەك.كۆزلىرىدىكى سۇرمىلەر ھەر حىل، ئۇلۇع ئالاھ ئاتا قىلعان شۇنچە چىرايلىق كىرپىكلىرىنى قانداق قىلعىنىكى بىلمىدىم بەك سەتلەشتۇرۇپ،ئۇلار بىر توۋەن يىللىققا دەرىس بىرىدىكەن،بىر ئوعۇل ئوقۇعۇچىسىدىن سورىسام دەرىس ۋاقتقدا ئوعۇل ئوقۇعۇچىلار شۇ حانىمچاقلارنىڭ كىيىم كىچىكىگە،ئۇلارنىڭ بەدەن قۇرلۇشىعاقاراپ ئولتۇرۇپ،ۋىسال دەرياسىعا شۇڭعۇپ  كەتكەنلەرمۇ بارمىش.يەنە بۇ يىلقى قۇربان ھىيتتا ئۇلارنىڭ بىر قىز ئوتۇعۇچىسى ماڭا بىزنىڭ حانىمنىڭكىگە ئوحشاش كىيىم ئىلىپ بىرىسەن دەپ ئاتا ئانىسىعا جار سىلىپ كۆڭلىنى ئاعرىتقان ئىكەن،ئاتا-ئانىسى؛ قىزىم ئۇنداق كىيىم كىيسەڭ بولمايدۇ دىسە؟قىز تۇرۇپ؛ئەينا بىزنىڭ ئوتۇتقۇچىمىزمۇ كىيىدۇ دەپتۇ،.مانا بۇ دەلىلمىكەن.

ئوماق قىزلار ؛ئوقۇش پۇرسىتى ھەممە ئادەمگە كىلىۋەرمەيدۇ،مەن بىر ئوعۇل بالا .مىنىڭ كىچىكىمدىكى ئارزۇيۇم؛ تەنتەربىيە ئوقۇتقۇچىسى ياكى ساقچى بۇلۇش ئىدى.مەن چوڭ بۇلعاندىن كىيىنمۇ شۇ ئارزۇ بۇيىچە ئوقۇدۇم.ئوقۇشۇم حىلى يامان ئەمەس ئىدى.مەن ئارزۇيۇمعا يىتىدىعان بولدۇم دەپ حۇشال بۇلاتتىم .لىكىن بۇ حۇشاللىق ئۇزۇنعا بارمىدى.مەن تۇلۇقسىزنى پۇتتۇرۇپ يۇقۇرى ئۆرلەش ئىمتاھانىعا قاتناشماقچى بولدۇم。ئەمما بۇنى ئويدىكىلەرگە ئىيتىشىم كىرەك ئىدى.ئۆيدىكىلەر قوللىعاندا نەتىجە قازانعىلى بۇلاتتى،،لىكىن ئۇلار مىنى قوللاش تۇگۇل ئىمتاھان بەرمەيسەن دىدى.نىمە ئامال ئۇلار نىڭ گىپىگە بويي سۇنۇش كىرەك ئىدى.شۇندىن كىيىن مەندىكى گۇزەل ئارزۇلار يىراقلارعا ئۇچتى.شۇندىن بۇيان مەن جەمىيەتتە يۇردۇم.نۇرعۇنلىعان ئىشلارنى كۆردۇم،نۇرعۇنلىعان ئىشلارنى قىلدىم .لىكىن مەن ھىچ نىمىدىن پايدىلىنالمىدىم.مەن بىلەن بىرگە ئوقۇعانلار حىلى نەتىجە قازاندى .ھازىر ئۇلەردىن بىرسى شىنجاڭ سىفەن داشۆدە،بىرسى شىنجاڭ يىزا ئىگىلىك ئۇنۋىرسىتىدا،يەنە بىرقىز ساۋاقدىشىم حوتەن تىببىي مەكتەپتە،بىر دوستۇم سىزلەر ئوقۇعان مۇشۇ مەكتەپتە ،يەنى حۇتەن ئالىي پىداگوكىدا ئوقۇيدۇ.مەن ئۇلارعا بەك ھەۋەس قىلىمەن.ئۇلار بىلەن ھەر قىتىم كۇرۇشكەندە ئۇلار مەكتەپتىكى ئىشلارنى دەپ بىرىدۇ.مەنمۇ كۆڭلۇمدە ‹مەنمۇ ئوقۇعان بولسام قانچە ياحشى بۇلاتتى دەپ قالىمەن›لىكىن ئەمدى ئامال يوق .مىنىڭ  سىزلەرگە دەيدىعىنىم؛سىزلەر ھازىر ئوقۇعۇچى سىزلەر،بۇنداق ياسىنىپ نىمە قىلىسىلەر؟سىزلەر ھازىر ئىلىم مەرىپەت  ئۆگۇنىدىعان ۋاقتىڭلار بۇ ،بۇ مەكتەپمۇ قانداق مەكتەپكىن بىلمىدىم؟ئوقۇعۇچىلارنىڭ يۇرۇش تۇرۇش،كىيىم –كىچەك دىققەت قىلمايدىعان ئوحشايدۇ،ئاتا ئانىلار مەكتەپتە مۇئەللىملەر ياحشى تەربىيلەيدۇ دەپ قىزلىرىنى ئەۋەتسە بۇ مەكتەپنىڭ نىمە قىلعىنىكى بۇ بىلمىدىم .ئاڭلىسام قىزلارنىڭ ياعلىقىنى ئىلىۋىتىپ يۇرۇشنى يولعا قۇيۇپتۇمىش،مەكتەپ ئوقۇعۇچىلارنى ئەقلىي،ئەحلاقى،گۇزەللىك.ئەمگەك شەرمىي-ھايا،جەھەتلەردە ياحشى تەربىيلىشى كىرەك ئىدى.مائارىپ ھەرگىز چاكىنىلىقنى تەشەببۇس قىلمايدۇ.بۇ مەكتەپتە سىرتلاردا ،ھەرحىل ياتاقلاردا قۇنۇپ قالىدىعان قىز ئوقۇعۇچىلارنىمۇ ئاڭلىدۇق،ئوعۇللار بىلەن ناشايىن ئىشلارنى قىلىپ لاعايلاپ يۇرگەنلەرنىمۇ ئاڭلىدۇق ،،بۇ مەكتەپتىلا ئەمەس،ھەر قايسى ياتاقلىق مەكتەپلەردە مۇشۇ ئەھۋاللار بارمىش ،لىكىن مىنڭ ئىشەنگۇم كەلمىدى،بۇ تىما باشقا تەرەپكە يۆتكىلىپ كەتتىعۇ ئەمدى،ھەر قايسى مەكتەپتىكى ئوعۇل-قىز  ئوقۇعۇچىلىرىمىزعا دەيدىعىنىم بۇ ئوقۇش پۇرسىتى ھەممە ئادەمگە كىلىۋەرمەيدۇ.ئوقۇش،حىزمەت ،ناچار ئەحلاق بىزنىڭ ۋىجدانىمىزنى تۇسۇۋالمىسۇن .ۋىجدان بىلەن ياشايلى ،


بىشىنى يىزىپ بۇلۇپ ئاحىرىنى قىسقارتىۋەتتىم،يىزىۋەرسەم بۇ گەپلەر تۇگۇمەيدىعان ئوحشايدۇ،چۇقۇم كىيىنچە ئاحىرىنى يازىمەن، كومپىيۇتىردىكى حاتالىق تۇپەيلى بەزى حەتلەر توعرا بولماي قالدى تۇزۇتۇپ ئوقۇعايسىزلەر.
