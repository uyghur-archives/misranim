                                                         جىگدىلەر پىچىرلايدۇ


                                                                 ( ھىكايە )


                                                             زوردۇن    سابىر



                                                                             
                                                                  سىزنىڭ مىنگەن ئىتىڭىزنىڭ ،
                                                                  بوينى ئالا قاشقا .
                                                                   بىز ئىككىمىز بىرگە بولساق ،
                                                                   ھىچنىم لازىم ئەمەس
                                                                    بىر قاپاق سۇدىن باشقا .

                                                                                       ( مەكىت خەلىق قوشاقلىرى )



                                                             1


       بۇ خەۋەرنى ئاڭلاپ ئۇنىڭ ۋۇجۇدى خۇددى توك ئۇرۋەتكەندەك سىلكىندى ، كۆڭلى خۇددى چىۋىن يۇتىۋالغاندەك ئەسكى بولدى ، ئۇنىڭ دايىم كۈلكە يىغىپ تۇردىغان ، تاڭ سەھەردە ياقا يىرىتقان قىزىل گۈل غۇنچىسىدەك ئوماق لەۋلىرى كۆكەردى ، ياشلىق غۇرۇرى ، بەخىت لەززىتى بىلەن ئويناپ ، يىگىتلەر يۈرۈكىگە ئوت چىچىپ ، نەيزىدەك قادىلىدىغان يوغان، قاپقارا خۇمار كۆزلىرىدىن يىپى ئۈزۈلگەن ئۈنچىلەردەك مەرۋايىت تامچىلار تۆكۈلدى ، ئۇ زىلۋا بويىغا ياراشقان توم ئىككى ئۆرۈم چىچىنى چۇۋۇپ - ئۆرۈپ تۇرۇپ ئەسەبىلىك بىلەن : << دادامنى ھەقىقەتچىلەرنىڭ ھەقىقەتچىسى دەپ بىلەتتىم ، ئانامنىڭ مىھرىنى قۇياشقا ئوخشىتاتتىم ، خاتالاشقان ئىكەنمەن ، مىنىڭ مەيلىمنى بىلمەيلا ، بىر ئىغىز پىكىر ئالمايلا مىنى خۇددى قۇشتەك ساتماقچى ، كونا كىيىمدەك بېرۋەتمەكچى بوپتۇ ، ئالى مەكتەپكە بارالمىغىنىمنىڭ جازاسىمۇ بۇ ياكى ئەتىۋارلىق قىز بولغىنىمنىڭ مۇكاپاتىمۇ ؟ >>
     ئۇ كەچكى تاماقنىمۇ يىمەستىن ئۆز خانىسىنىڭ ئىشىكىنى ئىچىدىن ئىلىۋالدى - دە ئۆزىنى ئازاپلىغان بۇ خەۋەر ئىسكەنجىسىدىن قۇتۇلۇش ئۈچۈن يول ئىزدىدى ، << ئۆلۈش كىرەكمۇ ؟ ئەلۋەتتە بۇ ئاخىرقى چارە ، كۆڭلۈم تارتمىغان بىرسىنىڭ ئايالى بولۇشتىن ئۆلۈش ئەۋزەل ، لىكىن مەن نۇرغۇن گۈزەل نەرسىلەرنى ئويلىغانتىمغۇ ؟ ئۇلارمۇ مەن بىلەن بىللە ئۆلەمدۇ ؟ >>
     قىز ئۆرە ئەينەككە قارىدى ، ئۇ ئاۋال دادىسى ئۈرۈمچىگە يىغىنغا بارغاندا ئىلىپ كەلگەن ئاق ياقۇت مارجاننى كۆردى ، مارجان ئۇنى ياشاشقا ، باشقىلار ئالدىدا ئۆز ئۈستۈنلىكى بىلەن ماختىنىشقا ئۈندەيتتى ، ئۇ ئانىسى ئەتىۋارلاپ تىكتۈرۈپ بەرگەن ئىسىل كوڭلىكىدىن ئالتۇن بۇلاپكىسىغا قارىدى ، بۇ يالتىراق زىننەت بويۇمى ئۇنىڭغا غەمكىن كۆرۈندى ، ئۆزىنىڭ رامكىغا ئىلىنغان رەڭلىك سۈرىتىگە قارىدى ، سۈرەتتىكى قىز كۈلۈپ تۇراتتى ، قىزنىڭ ئاپپاق چىشلىرى ئاۋۇ بوينىدىكى مارجاندەك گۈزەل ، ئۆلسە ئۇنىڭ گۈزەل جىسمى بىر سىقىم توپىغا ئايلىنىدۇ ،
     << ياشاش كىرەكمۇ ؟ دادىسىغا قارشى ئىسيان كۆتەرسىچۇ ؟ مەسلەن : باش ئىلىپ قىچىپ كەتسىچۇ ؟ ياكى ناھىيىلىك پارتىكوم سىكرىتارىغا بېرىپ دادىسىنى چاقسىچۇ ؟ ئۇ چاغدا دادىسى ئەيىپلىنىدۇ ، دادىسىنىڭ يۈزىگە ئۆز قىزى قارا سۇۋىغان ، ئۇنىڭ مەنسىپىنى كۆزلەپ يۈرگەنلەرگە ياردەم قىلغان بولىدۇ .......... ئاتا - ئانىسىنىڭ مەسلىھەتىگە قۇلاق سالسىچۇ ؟ تەۋەككۇل قىلىپ كۆڭلى چۈشمىگەن بولسىمۇ ئەشۇ يىگىتكە ماقۇل بولسىچۇ ؟ ئەلچىنىڭ سۆزى ئۇنىڭ قۇلىقىدا : يىگىت بۇلارنىڭ تۇققىنى ، يىگىرمە تۆت ياش ، چىرايلىق ، ئىغىر - بىسىق ، ئوقۇغان ، ھەممە قىزلانىڭ كۆزى ئۇنىڭدا ، ئەڭ مۇھىمى دادۇي پارتىيە ياچايكا سىكرىتارىنىڭ يالغۇز ئوغلى ، سىكرىتار شۇنچىلىك بايكى ، ئوغلىغا ئاتاپ ئىككى قەۋەتلىك ئۆي ، باغ ياساتقان ، گىلەملەر زاكاز ، تىلىۋىزور رەڭلىك ، ئۆي جابدۇقلىرى دوب ياغىچىدىن ، ئۆيى ئەتىر پۇرايدۇ ، ئاشۇ يىگىتنىڭ ئايالى ، ئاشۇ ئۆينىڭ خوجايىنى بولسىلا ھەرقانداق قىز ئۆزىنى بەخىتلىك ھىساپلايدۇ >>
     قىز ئويلاپ شۇ يەرگە كەلگەندە ئۇنىڭ يادىغا ساغۇچ كوڭلەك ، ھاۋا رەڭ شىم كىيگەن ، زىلۋا ، چىرايلىق كەلگەن  باشقا بىر يىگىت كەلدى - دە ، قىز يۈزىنى تۇتۇپ ڭىغلىۋەتتى ،
     _ پەرھات _ دىدى قىز دېرىزىدىكى گۈللەرگە يۈزىنى يىقىپ _ نىمەڭگە قىزىققاندىمەن ؟ چىرايىڭ ئۇنىڭغا يەتمەيدۇ ، بىلىمدە تىخى يوق ، بايلىق سىنىڭ ئۈچۈن قول يەتمەيدىغان يۇلتۇز ، قىلىدىغان ئىشىڭچۇ ؟ تۆمۈرچىنىڭ شاگىرتى ، كۈيىگە مىلەنگىنىڭ مىلەنگەن ، ئاناڭ تۇل خوتۇن ، ئۆيۈڭگە كىرسە ئادەمنىڭ بىشى ئايلىنىدۇ ، قىزلار قىزىققۇدەك ھىچنىمە تىپىلمايدىغان بىر ساياق بالا تۇرۇپ نىمىشقا مىنى ئۇسۇلغا تارىتتىڭ ؟ ئۇسۇل ئويناۋىتىپ نىمىشقا سىنىڭ كۆزلۈرۈڭگە قاردىم ؟  سىنىڭ راۋابىڭ بىلەن ناخشاڭ نىمىشقا يالغۇز مىنىڭلا ئۇيقۇمنى قاچۇردۇ ؟ شۇ كۈنى نىمىشقا ئەتەي ئىشتىن كەچ قايتتىم ، ؟ ھېلىقى جىگدىلىككە كەلگەندە قولۇمنى تۇتساڭ نىمىشقا قارشىلىق قىلمىدىم ، ؟
     قىز قىزارغان يۈزىنى ئالقانلىرى بىلەن سىلىدى ، ئۇ خۇددى يىگىت لىۋى تۇنجى قىتىم تەككەن جايلاردىكى سۆيگۈ ئىزناسىنى ئۆچۈرۋەتمەكچى بولغاندەك ياغلىقى بىلەن سۈرتتى ،
      قىز ئۇخلىيالمىدى ، تولغاندى ، ئويلاندى ، يىغلىدى .......يىرىم كىچىدە ئىچ كوڭلەك بىلەن ئولتۇرۋىلىپ خەت يازدى ،
      << ئۆگۈنلۈككە تويۇم بولىدۇ ، بۈگۈن بىزنىڭ نىكاھ قەغىزىمىزنى ئىلىپ كەپتۇ ، سىز بىلىسىز ، مىنىڭ دادام ھەممە ئىشنى باشقىلارغا قىلدۇرىدۇ ، نىكاھ قەغىزىمىزنىمۇ باشقىلار ئىلىپ كەپتۇ ، كۆڭلۈم تارتمىغان بىرىنىڭ قانۇنلۇق ئايالى بوپتىمەن ، لىكىن مەن ئېرىمنى كۆرۈشتىن بۇرۇن ئۆلمەكچىمەن ، نىمىشقا ؟ سەۋەبى مەن خورلاندىم ، خورلۇقتىن ئۆلۈم ياخشى ، ئۆلۈم ئاخىرقى خورلۇقتىن قۇتۇلۇش يولى ، ئۆلمەسلىك ياخشى ئىدى ، مەن ئۆلۈشنى خالىمايتتىم ، ياشاشنى ، ئىشلەشنى . بەخىتلىك بولۇشنى ئەجەپمۇ ئارزۇ قىلاتتىم ، لىكىن بىز قىزلار ئىرادىسىزكەنمىز ، بەخىتسىزلىكتىن قۇتۇلۇشنىڭ يولىنى بىلمەيدىكەنمىز ، ماڭا يول كۆرسىتىدىغان ، قولۇمنى يىتىلەپ ماڭىدىغان ئادەم تولىمۇ لازىم بولىۋاتىدۇ ، ناۋادا سىز ئىنسان بولسىڭىز ھېلىقى جىگدىلىكتە قىلغان گىپىڭىزنى ئۇنۇتماڭ ، سىز مەن ئۈچۈن ھىچنەرسىڭىزنى ئايىمايدىغان بولغان ، مىنى ھەقىقى ياخشى كۆرسىڭىز ، جىنىڭىزنى قولىڭىزغا ئىلىڭ ، ھېلىقى جىگدىلىككە كىلىڭ ، قارار قىلالمىسىڭىز كەلمەڭ ، مەن سىزنى كۈندىكى جايدا يەنە شۇ چاغدا يىرىم سائەت كۈتىمەن ............ >>
    قىز دادىسىغا يەنە بىر پارچە خەت يازدى :
     << ھۆرمەتلىك دادا ، مىنىڭ رازىلىقىم سىزنى قانچىلىق خۇشال قىلىشنى بىلىمەن ، سىز مىنىڭ رازىلىقىم بىلەن كۆپچىلىك ئىچىدە شۆھرەت، نام ، ھەتتا ئورۇنغا ئىگە بولالايسىز ، سەۋەبى كۆپچىلىك سىزنى شۈنچە چوڭ كادىر تۇرۇپ قىزىنى دىخانغا ياتلىق قىلدى دەپ ماختىشىدۇ ، ئەمگەكچىلەرگە راستىن مۇھاببىتى باركەن دىيىشىدۇ ، دىھقاننى قوللاش تازا بازىرى بار ئىش بولغانلىقى ئۈچۈن سىزنى بەلكىم بۇ قىتىم ھاكىملىققا سايلىشار ........ مەن سىزگە شۈنچىلىك ياردەم قىلالغىنىم ئۈچۈن خۇشال ، بالا ئۆز خۇشاللىقىنى ئاتا _ ئانىسى ئۈچۈن قۇربان قىلالىشى بولۇپمۇ ئاتىسىنىڭ مەرتىۋىسى ئۈچۈن ئۆز بەختىنى تەقدىم قىلالىسا ، بالا بالىلىققا يارىغان بولار - ھە ؟ كىلەر يىلى ئالى مەكتەپكە بارالايتتىم ، كەلگۈسىدە بىلىملىك ئادەم بولالايتتىم  ، لىكىن ئۇ چاغدا سىزنى ھازىرقىدەك رازى قىلالىشىم ناتايىن ئىدى ، بازار تاپقان سودىگەرلەردەك خۇشال بولۇڭ ، سالامەتلىك ، شۆھرەت ، مەنسەپ ، ئورۇن سىزگە يار بولسۇن ......... >>
