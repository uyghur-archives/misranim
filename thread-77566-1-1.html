    ئالدىنقى قېتىملىق بىر يازمىغا يازغان ئىنكاستىكى ۋەدەمگە ئاساسەن يېزىلدى .
       ئالدىنقى ئەسىرنىڭ ئاخىرقى يىللىرىنىڭ بىرىدە ئاللاھنىڭ يۇلىشى ، تىرىشچانلىقىم ،شۇنداقلا ئەتراپىمدىكىلەرنىڭ قوللىشى بىلەن نەچچە يىل ئۇقۇماي يۈرۈپ ، بىر ئىمتىھان بىلەنلا شىنجاڭ يېزا ئىگىلىك ئۇنۋىرسىتىتى تەييارلىق 971 سىنىپ ئۇقۇغۇچىسى بۇلۇپ قالدىم . جەمىئەتنىڭ قايناملىرىدا ئوڭ - تەتۈر ئېقىپ ، تالاي ئىسسىق - سۇغۇقلارنى بېشىمدىن ئۆتكۇزگەچكىمۇ  باشقا ساۋاقداشلىرىمغا قارىغاندا ئېغىر - بېسىق ، يۈرەكلىك ھەم دادىل ئىدىم . دەستلەپ ئۇرۇمچىگە بارغاندا كۆپلىگەن ئوقۇغۇچىلارغا ئوخشاش مەنمۇ يېڭىلىق بىلىنگەن ئىشلارنىڭ باشچىسى بۇلۇپ ماڭدىم . سىنىپتىمۇ ھەققانىيەتچىلىگىم ، ھىچكىمدىن تەپ تارتمايدىغانلىقىم بىلەن ناھايىتى تىزلا ئۈزۈمگە چۇشلۇق ئورنۇمنى تاپتىم . ھەتتا ئاخىرىدا سىنىپ مەسئۇلىمىز  ( خەن ) مۇ ماڭا يالۋۇرمىسا سىنىپتا ھىچ ئىش ئاقمايدىغان ھالەتكە چۈشۈپ قالدى .  ئۇنىڭ پارتىيىگە كىرگۈزۈش ، تەييارلىق پۈتكەندىن كىيىن ئىچكىرىدىكى ئەڭ ياخشى مەكتەپكە ماڭدۇرۇش .... .....دىگەندەك گەپلىرىمۇ مىنىڭ توغرا دەپ قارىغىنىمنى قىلىدىغان ، ساۋاقداشلىرىمنىڭ مەنپەئەتى ئۈچۈن  بارنى بار ، يوقنى يوق دەيدىغان ، زۈرۈر تېپىلسا چاپاننى سېلىپلا چىقىدىغان خاراكتىرىمنى ئۆزگەرتەلمىدى . خاركتىرىم سەل ئۆزگىچە بولغاچقىمۇ  مىنى ياقتۇرۇپ قالغان قىزلارمۇ كۆپىيىشكە باشلىدى ، ھەتتا مىنى ئۇدا نەچچە ھەپتە تانسىغا تەكلىپ قىلىپ ساقلايدىغان قىزلارمۇ ئوتتۇرىغا چىقتى . لېكىن مەن ئۇ قىزلار بىلەن تانسىغا چىقىپ باقمىدىم . لېكىن باشقىلار بىلەن ئاندا - مۇندا چىقىپ قۇياتتىم .
      بىر يىل ئەنە شۇ يۇسۇندا ئۈتۈپ كەتتى ..............! ئىككىنچى يىلى ئۇقۇش باشلانغاندا دەرس بېسىمى ھەم نۇرغۇن ياشلارغا ئوخشاش ھىسسىيات تەرەپتىكى ئوڭوشسىزلىق ئېغىر چارچىتىۋەتكەن بولسا كىرەك ، نېرۋا چارچاش كىسىلى بۇلۇپ قالدىم . مىجەزىم ئوساللىشىپ ، كەيپىياتىم تازا نۇرمال بولمايدىغان بۇلۇپ قالدىم .
      شۇ كۇنلەرنىڭ بىر جۇمە كۈنىدە مەكتىۋىمىزگە ئەكبەر قەھرىمان ناخشا ئېيتقىلى كىلىپ قالدى . ئەينى ۋاقىتتا ئەكبەر قەھرىماننىڭ تەسىرىنى ھەقىقەتەن سەل چاغلىغىلى بولمايتتى .  گەرچە ئەكبەر قەھرىمان دىسە ئۈلۈپ بەرمىسىمۇ كىسەل بۇلۇپ بىرىدىغان ، ھەتتا ئۇنىڭ يامان گىپىنى قىلغانلار بىلەن ئۇرۇشۇپ ، چاچلىشىدىغان قىزلارمۇ كۆپ ئىدى . شۇ دەۋرلەردە ئۇقۇغان مۇنبەرداشلار بىلىدۇكى بىزنىڭ مەكتەپ ھەقىقەتەن بەك چىرايلىق ھەم باشقا مەكتەپ ئوقۇغۇچىللىرى كىلىشكە ئامراق جاي ئىدى . جۇمە كۇنى بولغاچ بىزنىڭ مەكتەپنىڭ مىھمانلىرىمۇ كۆپ ئىدى . ئېنىقكى بۇ كىچىلىك چۇقۇم قىززىيتى . پوسكاملىق ئاغىينەمنىڭ زورلىشى بىلەن كۆڭلۇم تارتمىسىمۇ ئاخىرى بۇ تانسا كىچىلىكىگە چىقتىم .  ئۈزۈممۇ ئەكبەر قەھرىماننىڭ ناخشىللىرىغا ئامراق بولغاچ ئەڭ بولمىغاندا ناخشا ئاڭلىمايمەنمۇ دەپ ئويلىدىم .
   تانسىخانا قىلىنغان تىك -تاك توپ كۇلۇبىدا ھەقىقەتەن ئادەم جىق ئىدى . ھەتتا ئولتۇرغىدەك ئورۇنمۇ ئاز ئىدى .  مىنى كۈرۈپ خىجىل بۇلۇپ مۇكۇپ يۇرگەن قىز ساۋاقداشلىرىمغا قارىماستىن ، مىڭ تەستە بىكار ئورۇندىن بىرنى تېپىپ ئولتۇردۇق . تانسىمۇ باشلاندى .  دەستلەپتە بىز باشقىلارنىڭ ئوينىغىنىغا قاراپ ئولتۇردۇق ، كىيىن ئاغىينەمنىڭ تانسا ئويناۋاتقان پۇرسىتىدىن پايدىلىنىپ ئەكبەر قەھرىماننىڭ  قېشىغا بېرىپ يىنىدا بىردەم ئولتۇرۇپ قايتىپ كەلدىم  . كەلسەم ساۋاقدىشىم مىنى ئىزدەپ ھالى قالماپتۇ . شۇ پەيتتە يەنە مۇزىكا باشلاندى . ئاغىينەم ئۇنىمىغىنىمغا قويماي قۇلۇمدىن تۇتۇۋېلىپ ئىككى قىزنىڭ ئالدىغا باشلاپ باردى ھەم ئۇلارنى تانسىغا تەكلىپ قىلدى ، تەبىئىكى يەنە بىرقىزمۇ ئورنىدىن تۇرۇپ ماڭا قاراپ تۇردى . مەنمۇ ئارتۇق نازلىنىپ تۇرمايلا ئۇقىزنىڭ قۇلى بىلەن بىلىدىن ئالدىم دە ، ئويناشنى باشلاپ كەتتىم . .............................  .
    مۇڭلۇق ناخشىنىڭ ( توغرىسى شەيتاننىڭ ) تەسىر قىلىشىدىنمۇ ياكى ھېلىقى ھىسسىيات تەرەپتىكى كاللامدىن ئۆتمەيدىغان ئىشلار سەۋەبىدىنمۇ ۋە ياكى قىزلار بىلەن مۇنداق يىقىن تۇرۇپ باقمىغاچقىمۇ ئەيتاۋۇر مۇشۇلارنىڭ ھەممىسى بىرلەشكەنمۇ ئىشقىلىپ  مىنى ئەقلىمدىن بىھۇش قىلىۋەتكەن ئىدى . ئاشۇنداق ھىسلارنىڭ يەڭگىل چايقىشى بىلەن قانچىلىك ۋاقىت ئۆتكىنىنى بىلمەيمەن بىر چاغدا ئىسىمگە كەلدىم ............. كەلدىميۇ چۈچۈپ كەتتىم .................!
    ( كەچۈرۈڭلار ، مۇشۇنداق يازماي بولمىدى )  قىز پۈتۈن پۈتۈن ئەس -ھۇشىنى يۇقاتقان ھالدا ماڭا ئېسىلغان ، مەنمۇ ئۇنى چىڭ قۇچاقلىۋالغان ھالەتتە  شۇنچىۋالا كىشلەر ئوتتۇرىسىدا خۇدىمىزنى بىلمىگەن ھالەتتە چىڭ قۇچاقلىشىپ تۇراتتۇق . پۇتلىرىمىز بىلىنەر - بىلىنمەس مىدىراپ تانسا ئويناۋاتقانلىقىمىزنى بىلدۈرۈپ تۇراتتى . گەرچە پۇرسەت ياراتقان گۇڭگا چىراق نۇرىدا بەكرەك قارىمىسا بىلىنمىسىمۇ لېكىن ئەتراپتىكىلەردىن بەزىللىرى بىزگە قاراپلا قالغان ئىدى . شۇ چاغدىلا ئەمەس تا ھازىرغىچە ئۇ قىزنىڭ كىم ،نەلىك ئىكەنلىكىنى بىلمەيمەن .
   ئىسىمگە كىلىپلا كۈچلۈك نۇمۇس تۇيغۇسىدا نېرۋىللىرىم ( تۇيۇچلىرىم ) قاتتىق سىلكىنىپ ئىسىمگە كەلدىم دە ، قىزنىڭ چاپلىشىپ كەتكەن بەدىنىنى  ئاستا نېرىغا ئىتتەردىم . قىز يېرىم يۇمۇلۇپ كەتكەن كۆزلىرى بىلەن ماڭا سۇئال نەزىرى بىلەن تىكىلدى . جۇرئەتسىز كۆزلىرىم بىلەن ئۇنىڭغا قارىيالمىدىم . ئەيىبدار ھالەتتە داۋاملىق ئۇنى كەينىگە ئىتتەردىم . قىزمۇ ئاستا ئىسىگە كەلدى بولغاي ئۈزىنى دەرھال ئوڭشىدى . مۇزىكا تۇگە - تۈگىمەيلا قىزنى قۇيۇۋەتتىمدە باشقىلارغا ئۇرۇلۇپ - سۇقۇلۇپ يۈگۈرگىنىمچە ئۇدۇللا سىرتقا چىقىپ كەتتىم . سىرتنىڭ سالقىن ھاۋاسى تەسىرىدە قالغان سەزگۈلىرىم ئەسلىگە كەلدى . ئېيتىپ بولغۇسىز ئەيىبدارلىق تۇيغۇسى ئىچىدە ئىچ - ئىچىمدىن ئۈكۈنۈپ كەتتىم ......................             !
  شۇنىڭدىن بىرى ئاساسەن تانسا ئويناش تۈگۈل تانسىخانىغىمۇ چىقماس بۇلۇپ ئۇقۇش پۈتتۈردۈم .


  ھازىر ئويلىسام ئىشلار شۇ پېتى بۇلۇلىۋەرگەن بولسا  ..........ھېلىقى خۈشياققان سەزگۈلەرنى ئىزدەپ يۈرگەن بولسام ئەلۋەتتە ھازىرقى ئەلباش باشقىلا بىر ئادەم بۇلۇپ چىقاتتى . ھازىرقى ئەلباش مۇنبەرچى ئەمەس چچ چى ئەلباش بۇلۇپ ئەزا نامىمۇ باشقىچە مەسىلەن :  ............. بۇلغان بۇلاتتى !


مۇنازىرە قىلساڭلار تۈۋەنچىلىك بىلەن سورايمەن : شەخسىيىتىمگە تىگىدىغان سۆزلەر بۇلۇنمىسا !
