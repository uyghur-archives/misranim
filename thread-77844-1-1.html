                                                    ئەجەپمۇ ئۆچ بولدۇم ساڭا



                                     كۆرسەڭمۇ كۆرمەسكە سىلىپ،
                                          ئۇرۇلۇپ ئۆتكەندە ماڭا ،
                                          يۈرۈكۈم تىلىنىپ كەتتى،
                                          ئەجەپمۇ ئۆش بولدۇم ساڭا.




                                         نىمە قىلدىمكى بىلمىدىم،
                                         بىرىپسەن گۈلۈڭنى ئاڭا.
                                         ئۆرۈلۇپلا ئەسكى دەپسەن،
                                         ئەجەپمۇ ئۆش بولدۇم ساڭا.




                                        تەنە قىلغاندا ئۆزگىلەر،
                                        نە ھالىڭ دىمىدىڭ يانا.
                                        يالغۇز قالدىم بۇ كۇنلەردە،
                                        ئەجەپمۇ ئۆش بولدۇم ساڭا.




                                       بىلدىمكى مىھرى ئۆزگەرمەس،
                                       پەقەتلا شۇ ئاتا _ ئانا،
                                       ئاخىرەت دەيمەنكى خەير،
                                       ئەجەپمۇ ئۆش بولدۇم ساڭا.






                            سىنى شۇنداق ياخشى كۆرۇپ قاپتىمەن


                            تۇنجى كۆرۇپ سىنىڭ قاراشلىرىڭنى،
                                    ئۇچراشقاندا سالام قىلىشلىرىڭنى.
                                    يۈرەكنى ئويغاتقان شۇ مىنۇتلارنى،
                                    سىنى شۇنداق ياخشى كۆرۇپ قاپتىمەن.


                                   
                                   ماڭا لاپپىدە كۆز سىلىشلىرىڭنى،
                                   يىراقتىن شۇنداق كۆزىتىشلىرىڭنى.
                                   بايقاپ قالسام كۆز ئەپقىچىشلىرىڭنى،
                                   سىنى شۇنداق ياخشى كۆرۇپ قاپتىمەن.




                                  ئۇچرىشىپ قالساق قىزىرىشلىرىڭنى،
                                  ھودۇقۇپ تەمتىرەپ كىتىشلىرىڭنى.
                                  گەپ تاپالماي كىتىپ قىلىشلىرىڭنى،
                                  سىنى شۇنداق ياخشى كۆرۇپ قاپتىمەن.




                                 يول ماڭغاندا قول سىلىشلىرىڭنى ،
                                 يىگىتلەرگە خاس شۇ قامەتلىرىڭنى،
                                مەنىگە تولغان شۇ جۈپ كۆزلىرىڭنى،
                                سىنى شۇنداق ياخشى كۆرۇپ قاپتىمەن.




                               ناز قىلسا گۈزەل ،رەت قىلىشلىرىڭنى،
                               ئۆزگىگە پەرۋاسىز يۇرۇشلىرىڭنى.
                               شۇندا بىلىپ سىنىڭ ئەخلاقلىقىڭنى،
                              سىنى شۇنداق ياخشى كۆرۇپ قاپتىمەن.




                              دۇشمەن ئازار قىلسا توسۇشلىرىڭنى،
                              ھار كىلىپ يىقىلسام يۆلەشلىرىڭنى.
                              مەن ئۈچۇن قىلغان شۇ مەرتلىكلىكىڭنى،
                              سىنى شۇنداق ياخشى كۆرۇپ قاپتىمەن .




                              ئاخىر بىلدىم مانا ئوغلانلىقىڭنى،
                              يار ئۈچۈن كۆرسەتكەن ئەركەكلىكىڭنى.
                              تەقدىم ئىتەي مەندىن كۈتكەنلىرىڭنى،
                              سىنى شۇنداق ياخشى كۆرۇپ قاپتىمەن.
