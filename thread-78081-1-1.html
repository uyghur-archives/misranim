    يېقىندا بىر تونۇشۇمدىن مۇنداق بىر ۋەقەنى ئاڭلاپ تولىمۇ ئۆكۈندۈم.
   ئۇلارنىڭ مەھەللىسىدە بىر جۈپ ئەر-ئايال بولۇپ، ئىككى بالىسى بار ئىكەن. بۇ بەك باياشات ئائىلىلەردىن بولمىسىمۇ، ھەر ھالدا ئەر - خۇتۇن ئىككىسى كۈڭۈللۈك، خاتىرجەم ياشايدىكەن. ئايالنىڭ تەقى تۇرقى خېلى يامان ئەمەس بولغاچقا، تەڭ - تۈشلىرى ئارىسىد خېلىلا پەرقلىنىپ تۇرىدىكەن. شۇ ۋەجىدىنمۇ ئەيتاۋۇر، شۇ مەھەللىدىكى بىرى ئۇنىڭ پېيىگە چۈشۈپتۇ. ھېلىقى ئەر دائىم دېگۈدەك ئۇ ئايالنىڭ ئىرى يوق ۋاقىتتىن پايدىلىنىپ ئۇلارنىڭ ئۆيىگە كەتمەن، گۈرجەك دېگەندەك نەرسىلارنى سوراپ كىرىدىغان بولۇپ قاپتۇ.دەسلەپتە ھېچنىمىنى سەزمىگەن ئايال بار بولسا بىرىپ، يوق بۇلسا يوق دەپ يولغا ساپتۇ. لېكىن بۇنداق ئىش تەكرارلىنىۋەرگەندىن كېيىن ئايالنىڭ كۆڭلىگە شەك چۈشۈپ ئۆز بىخەتەرلىگىنى قوغداش خىيالىغا كەپتۇ.
راس دىگەندەك ھېلىقى ئەر بىركۈنى ئايالغا ئۆزىنىڭ كۆڭلىنى ئىزھار قىپتۇ. ئايال ئۇنىڭ تەلەپلىرىنى چىرايلىق رەت قىلىپ، ئۇنى باشقىچە خىياكدا بۇلماي ئۆيىدىكى خوتۇن، بالىلىرىغا ياخشى قاراشقا دەۋەەت قىپتۇ. لېكى ئۇ ئەر يەنىلا نىيتىدىن يانماي، پەيت تاپسىلا ئايالنىڭ يۇلىنى تۇسۇپ، ئىشىكىنى چىكىپ ئاۋارە قىلىشنى داۋاملاشتۇرۇپتۇ. ئايالنىڭ ئىرى ھۈنەرۋەن بۇلۇپ، كۈپىنچە ۋاقىتلاردا سىرتلاردا ھۈنەر قىلىپ يۈرگەچكە ئۆيدە كۆپ تۇرالمايدىكەن.شۇڭا دائىم ئۆيدە يالغۇز قالىدىغان بۇ ئايال مۇشۇنداق كىتىۋەرسە، مەھەللىدە ئۆزى توغۇرلۇق ئۆسەك سۆز تارقىلىپ ئىرىنىڭ قۇلىقىغا يىتىشتىن ۋە ئاخىرىدا بەخىتلىك ئائىلىسىنىڭ بۇزۇ لىشىدىن ئەنسىرەپ بۇ ئىشلارنى ئىرىگە دىيىش قارارىغا كاپتۇ. چۈنكى ئۇ:<< ھەممە گەپنى ئەينەن دىسەم، مەندە گۇناھ بۇلمىغاندىن كىيىن ئۇ ماڭا ئىشىنەر، ئۇنىڭ ئۈستىگە ئۇ مىنى ئىرىم بۇلغاندىكىن مىنى قوغداش مەجبۇرىتىمۇ بار.باشتا دىمەي كىيىن باشقىلاردىن ئاڭلىسا بالانىڭ چۇڭى چىقمىسۇن يەنە. نەچچە يىل ئۆي تۇتۇپتۇق، مىنىڭ قانداقلىقىمنى ھەرقانچە بۇلسىمۇ بىلەر. ھېلىقى ئەرنى قۇرقىتىپ قۇيسا، .ئۇمۇ ئۆز يۇلىغا ماڭسا،مەنمۇ دەككە-دۈككىدىن قۇتۇلۇپ خاتىرجەم بالىلىرىمنى باقارمەن>> دەپ ئويلاپ ئېرىغا بۇلغان ئەھۋاللارنى دەپتۇ. ئەمما ئايالىدىن بۇ گەپلەرنى ئاڭلىغان ئەر:<< ئۆزەڭنىڭ بىر چاتىقىڭ بۇلمىسا ئۇ ئەبلەخ نىمە ئۈچۈن باشقا ئاياللارنىڭ ئارقىسىغا كىرۋالماي سىنىڭ ئارقاڭغا كىرىۋالىدۇ؟ ئېيتە بۇزۇق، يەنە نىمە ئىشلارنى قىلدىڭ؟>> دەپ ئايالىنى سوراق قىلىشقا چۈشۈپ كىتىپتۇ. بارا - ابارا سوراق مۇشت-تىپىككە ئۆزگىرىپتۇ. ئايال مۇشت - تىپىك ئىچىدە ھۇشىدىن كەتكەندىلا ئىرى ئۇرۇشتىن توختاپتۇ. جىدەلنى ئايرىش ئۈچۈن ئۆيگە قۇلۇم-قۇشنا، مەھەللە-كوي يىغىلىپتۇ، ئېرى:
-بۇ بۇزۇق مەن يۇق چاغدا خەق بىلەن ئىچ-پەش تاتىشىپ يۈرۈپتۇ. مەن ئەمدى بۇ يۈزۈمنى قانداق كۈتۈرۈپ يۈرۈيمەن؟-دەپ جىدەلنىڭ سەۋەبىنى ھەممە ئادەمگە ئاشكارلىۋېتىپتۇ. ھەر كىم ھەر نىمىلەرنى دەپتۇ.
-شۇنداق ئىشنىمۇ ئىرىگە دىگەن بارمۇ؟
- ئىرىگە ئۆز ئاغزى بىلەن دىگەندىكىن بۇ تاياق ھالال بۇپتۇ.
- ھەرقانچە ئىش بۇلسىمۇ ئەر كىشىگە بۇنداق ئىشنى ئۇقتۇرمىغان ياخشى.
تاياق زەربىسىدىن يۈز كۆزلىرى ئىششىپ كەتكەن ئايالغا ھېچكىم ھىسداشلىق قىلماپتۇ. ئۆز نامىغا داغ تىگىپ قېلىشتىن ئەنسىرەپ ئېرىغا سەمىمىيلىك بىلەن راس گەپ قىلىپ ئىرىدىن پاناھلىق ئىزدىگەن بۇ ئايال ئاخىرى ئۆزى قورققان قالپاقنى كىيىپ بالىلىرى بىلەن بىللە ئۆيدىن ھەيدەپ چىقىرلىپتۇ......
يۇقۇردىكى ۋەقە ھەرقانداق كاللىسى سەگەك ئادەمنىڭ يۈرىكىنى ئەزمەي قالمايدۇ، ئەلۋەتتە. ئەرقانداق بىر ئايال خەۋىپكە ئۇچچۇرغاندا بىر ئەركەكنىڭ قانىتى ئاستىدا پاناھلىنىشنى ئويلايدۇ.ئۇنىڭ يېنىدا ۋاپادار ھىمايىچىسى بۇلسا ھەرقانداق بوران ئۇنى لىڭشىتالمايدۇ. لېكىن شۇنى كەسكىنلىك بىلەن دىمەي بۇلمايدۇكى ، بۈگۈنكى كۈندە بىر قىسىم غۇرۇرلۇق ، ئىپپەتلىك ئاياللىرىمىز ھەقىقەتەنمۇ يۈلەنچۈكسىز، قوغدىغۇچىسىز قېلىۋاتىدۇ. ئۇلار بىرەر خەۋىپكە يۇلۇقسىمۇ، ئۆزىنى قوغدىيالمايدىغان ھالەتكە كىلىپ قالسىمۇ، ئىچىگە تىنىپ يۈرۈشكە مەجبۇر بۇلماقتا. چۈنكى بەزى ئەرلەر بىر بۇلسا ئۆزىدىن باشقىنى ئويلىمايدىغان، ئاياللىرىنى ، قزلىرىنىڭ، ئاچا-سىڭىللىرىنىڭ نىمە بۇلىشى، قايسى كۈننى كۈرۈشى بىلەن كارى بۇلمايدىغانلامزەللىلىككە كۈنىۋالدى، بىر بۇلسا يۇقۇردىكى ۋەقەدىكىدەك << چىېىن چاقسا غىڭشىپ، كالتەك تەگسە قىم تۇرىدىغان>> تەتۈرلۈك، مەسئۇلىيەتسىزلىك، تۇتۇرۇقسىزلىقنى ئادەت قىلىۋالدى.
ھازىر ئەرلەنىڭ ئاغزىدىن<< ھازىرقى قىزلار ئۇنداق، ھازىرقى قىزلار مۇنداق>> دىگەندەك گەپلەر چۈشمەيدىغان بۇلۇپ كەتتى. شۇنىمۇ ئەستىن چىقارماسلىق كىرەككى ، شۇ قىزلارنى قۇلىدىن يىتىلەپ شۇنداق يەرلەرگە باشلاپ بېرىۋاتقان ،<< بىر قېتىمغا ھېچنىمە بۇلمايدۇ>> لار بىلەن نۇرغۇن ئىشلارنى قىللىشقا مەجبۇرلاۋاتقان ، كۆندۈرۋاتقان يەنىلا بىزنىڭ ئەركەكلىرىمىز. ئاچا - سىىللىرى، ئاياللىرى كىچە-كىچىلەپ رېستۇران، ئولتۇرۇشلاردا يات ئەرلەنىڭ قۇللىرىنى مىجىقلاپ، بەللىرىنى تۇتۇپ تانسا ئويناپ كىلىپ:<< بىر دوستۇمنىڭ تۇغۇلغان كۈنى ئىدى، بارماي دىسەم ئادەمگەرچىلىكتىن قاچقاندەك بۇلمىسۇن، دەپ بارغان>> ياكى << بىر خىزمەتدىشىمىز پالىنى يەرگەئوقۇشقا ماڭماقچى ئىدى،ئىدارىدىكى تۆت-بەش ئايال بىرلىشىپ ئۇزۇتۇپ قۇيغىلى باردۇق>> دىگەنگە ئوخشاش يالغان گەپلەر بىلەن ئالدىسا ،<< ھەراست>> دەپ ئىشىنىپ،<< تۇغرا چۈشىنىپ>>ھاردۇق چېيى قۇيۇپ بىرىۋاتقانلارمۇ بىزنىڭ ئەركەكلىرىمىز.
مەن بىر كىتاپتىن:<< ئەرلەرنىڭ چۈشكىنلەشكەن ۋاقتى دەل ئاياللارنىڭ پەردىشەپتىن چەتنىگەن ۋاقتىدۇر>> دىگەن بىر جۈملىنى ئوقۇغان ئىدىم. شۇڭا ئەركەكلىرىمىزمۇ سەمىمىيەت بىلەن خۇپسەنلىكنى ئايرىشنى بىلىۋالسا، ئاياللىرىنى يات كۆزلەردىن تۇسۇشنى ھەقىقىي مەسۇلىيەت، مەجبۇرىيەت دەپ چۈشەنسە، قارغۇلارچە كۈندەشلىك قىلمىسا؛ ئىپپەتلىك ئاياللىرىمىزمۇ تۈرلۈك خەۋىپلەر ئىچىدە يار-يۈلەكسىز ئىچىگە تىنىپ يۈرمىسە، ئەر - خۇتۇنلار ئوتتۇرسىدىكى ئىشەنچ غېرىپ سىنىپ قالمىسا؛ بالىللار ئۆز ئاتا-ئانىسىنىڭ قۇللىرىنى تۇتۇپ مېڭىپ ، بېشىنى كۈتۈرۈپ چوڭ بۇلسا؛ يىتىم - يىسىرلارنىڭ سانى ئازايسا دەيمەن. مىنىڭچە، راست گەپ ھامان راست بۇلىشى، يالغان گەپ ھامان يالغان بۇلىشى مۇمكىن. شۇڭا يالغان گەپ قىلىپ كۈلدۈرگەندىن، راس گەپ قىلىپ يىغلاتقان ئەۋزەل.

مەنبە شىنجاڭ ئاياللىرى ژورنىلىدىن ياقتۇرۇپ باشقىلارنىڭمۇ تەڭ پايدىلىنىشى ئۈچۈن مۇنبەرگە يوللىدىم.
