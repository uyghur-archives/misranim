    جۇڭخۇا خەلق جۇمھۇرىيىتىنىڭ باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىش ئۈچۈن بېرىلىدىغان تۆلەم تولۇقلىما ھەققىدىكى ۋاقىتلىق چارىسى


2011-يىلى 3-ئاينىڭ 1-كۈنىدىن باشلاپ ئىجرا قىلىنىدۇ.

ئىقتىسادىي تەرەققىياتنىڭ تەلىپىگە ئۇيغۇنلىشىش، خەلق ئۈچۈن تېخىمۇ خاتىرجەم ماكان بەرپا قىلىش، قاتناشنى قۇلايلاشتۇرۇش، زامانىۋى قاتناشنىڭ قۇلايلىقىنى تېخىمۇ ياخشى جارىي قىلدۇرۇپ، ۋاقىت ئارقىلىق چەت يىراق رايۇنلار ئارلىقىنى قىسقارتىش، شەھەر يېزىلارنىڭ بىر گەۋدىلىشىشىنى ئىلگىرى سۈرۈش ئۈچۈن مەزكۇر ۋاقىتلىق چارە تۈزۈپ چىقىلدى.

بىرىنچى، يەر ئۈچۈن بېرىلىدىغان  تولۇقلىما

1.تېرىلغۇ يەرگە بېرلىدىغان تولۇقلىما ئۆلچىمى

قۇرغاق يەرنىڭ ھەر مولىقى ئۈچۈن 53000يۈەن

سۇلۇق ئېتىزنىڭ ھەر مولىقى ئۈچۈن 90000يۈەن

كۆكتات ئېتىزلىرىنىڭ ھەر مولىقى ئۈچۈن 150000يۈەن

2.ئادەتتىكى يەرگە بېرىلىدىغان تولۇقلىما ئۆلچىمى

قۇرغاق يەرنىڭ ھەر مولىقى ئۈچۈن 58000يۈەن

سۇلۇق ئېتىزنىڭ ھەر مولىقى ئۈچۈن 99000يۈەن

كۆكتات ئېتىزلىرىنىڭ ھەر مولىقى ئۈچۈن 156000يۈەن

3.ئورمانلىق ۋە باشقا يېزا ئىگىلىك ئۈچۈن ئىشلىتىلىدىغان يەرلەرنىڭ ھەر مولىقى ئۈچۈن 138000يۈەن بېرىلىدۇ.

4.كان قۇرۇلۇش ئورۇنلىرى دېھقانلار تۇرالغۇسى، يول قاتارلىق كوللىكتىپ ئىشلىتىدىغان يەرلەرنىڭ ھەر مولىقى ئۈچۈن 136000يۈەن بېرىلىدۇ.

5.بىكار يەر، قاقاسلىق، بىنام يەر، ساسلىق ۋە ئىشلىتىلمىگەن يەرلەر ئۈچۈن ھەر موسىغا 21000يۈەندىن بېرىلىدۇ.

ئىككىنچى، باشقا تۈرلۈك باجلار

1.تېرىلغۇ يەر بېجى، ھەر كۇۋادىرات مىتىرى ئۈچۈن 2يۈەندىن ھىسابلىنىدۇ.

2.تاۋار يەر (سەي ئېتىزى) ئېچىش فوندى ھەر موسى ئۈچۈن 10000يۈەن بويىچە ھېسابلىنىدۇ.

3.يەرنى يەر ئىگىلىرىدىن ئاجرتىپ ۋاقىتلىق باشقۇرۇش ھەققى ئومۇمىي تۆلەم پۇلىنىڭ %3بويىچە ھېسابلىنىدۇ، دۆلەت زىمىن باشقۇرۇش ئورۇنلىرى مۇناسىۋەتلىك بەلگىلىمىلەر بويىچە قاتتىق ئىجرا قىلىدۇ.

4.ئەسلىدىكى تېرىلغۇ يەرلەرنى تۈزلەپ يېڭىدىن ئېچىش، ھەر موسى ئۈچۈن 4000يۈەن، بىر تۇتاش پىلانلىنىپ تەڭشىلىدۇ. ئۆلكىلىك زىمىن بايلىق نازارەتلىرى نازارەت قىلىشقا مەسئۇل بولىدۇ.

ئۈچىنچى، باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىش خىزمىتى تەرتىپى

1. باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىش ئەھۋالىنى ئۇقتۇرۇش. باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىش تەستىقلىنىپ ئىجرا قىلىنىشتىن بۇرۇن، شۇ جايدىكى يەر باشقۇرۇش ئورگانلىرى باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىش مەقسىتى، ئورنى، تۆلەم ئۆلچىمى، ئورۇنلاشتۇرۇش يولى قاتارلىقلارنى يازما شەكىلدە دېھقانلارغا، كوپراتىپلارغا، يېزا ئىگىلىك كوللىكتىپ ئىقتىسادىي تەشكىلاتلىرىغا ئۇقتۇرىشى كېرەك. ئۇقتۇرۇش قىلىنىپ بولغاندىن كېيىن دېھقانلار ياكى ئىقتىسادىي كوللىكتىپ كارخانىلار ئۆز مەيلىچە پۇرسەتپەرەسلىك قىلىپ تېرىسا، كۆچەت قويسا ياكى كۆچۈرسە، تولۇقلىما بېرىش ۋاقتىدا بىردەك ئىتراپ قىلىنمايدۇ.

2. باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىشنى تەكشۈرۈش نەتىجىسىنى جەزىملەشتۇرۇش. دۆلەت زىمىن بايلىق ئىدارىلىرى قاتناش، ئورمانچىلىق ئىدارىلىرى بىلەن بىرگە ئىشلىتىلىدىغان يەرنىڭ ئىگىدارلىقى، تۈرى، كۆلىمى ۋە يەردىكى باشقا قوشۇمچە مۈلۈكلەرنىڭ ئىگىدارلىقى، تۈرى، سانى قاتارلىقلارنى تەكشۈرىشى، تەكشۈرۈش نەتىجىسىنى يېزا كوللىكتىپ ئىقتىسادى گەۋدىسى، دېھقانلار ۋە يەر ئىگىلىرى، يەردىكى مۈلۈكلەر ئىگىدارلىرى ۋە شەھەرلىك چېقىش يۆتكەش خىزمىتى ئىشخانلىرى بىلەن بىرگە جەزىملىشى كېرەك.

3. باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىش ئىسپات ئاڭلاش يىغىنىنى تەشكىللەش.

باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىش تەستىقلىنىپ ئىجرا قىلىنىشتىن بۇرۇن، شۇ جايدىكى يەر باشقۇرۇش ئورگانلىرى باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىش مەقسىتى، ئورنى، تۆلەم ئۆلچىمى، ئورۇنلاشتۇرۇش يولى قاتارلىقلارنى يازما شەكىلدە دېھقانلارغا، كوپراتىپلارغا، يېزا ئىگىلىك كوللىكتىپ ئىقتىسادىي تەشكىلاتلىرىغا ئۇقتۇرىشى كېرەك. مۇناسىۋەتلىك شەخسلەرنى ئىسپات ئاڭلاشقا ئىلتىماس قىلىش ھوقۇقى بولىدۇ. ئەگەر ئىسپات ئاڭلاشقا ئىلتىماس قىلغۇچىلار بولسا، «دۆلەت زېمىن بايلىقى ئىسپات ئاڭلاش بەلگىلىمىسى» بويىچە ئىسپات ئاڭلاشقا تەشكىللەش كېرەك.

4.باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىش تۆلەم كېلىشىمى ئىمزالاش.

دۆلەت زېمىن بايلىقى ئىدارىلىرى مۇشۇ لاھىيىدە بېكتىلگەن باھا ئۆلچىمى بويىچە دېھقانلار ياكى يېزا ئىگىلىك ئىقتىسادىي تەشكىللىرى بىلەن تۆلەم كېلىشىمى ئىمزالايدۇ. ھەمدە كېلىشىمنى باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىش دوكلاتى بىلەن بىرگە يۇقىرىغا يوللايدۇ.

5. باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىش ئىشلىرىنى تەستىقلاش ئىش ھالقىلىرىنى ئاشكارىلاش.

قانۇن بويىچە باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىش تەستىقلانغان يەرلەر دۆلەت مەخپىيىتىگە چېتىلىدىغان ئالاھىدە ئەھۋاللاردىن باشقىلىرىنى ئۆلكە، ئاپتونۇم رايۇنلۇق دۆلەت زېمىن بايلىق ئورۇنلىرى جەمئىيەتكە ئاخبارات ۋاستىسى ئارقىلىق ئۇقتۇرۇشى كېرەك. ناھىيىلىك زېمىن باشقۇرۇش تارماقلىرى « باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىش ئەھۋالىنى ئېلان قىلىش چارىسى» بويىچە ئىش قىلىشى، ئىشلىتىلگەن يەردىكى كەنىت،  گۇرۇپپىلارغا يەر ئىشلىتىش ھەققىدىكى تەپسىلاتلارنى ئۇقتۇرۇشى كېرەك.

6. باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىش تۆلەم تولۇقلىمىسىنى تاپشۇرۇش.

باشقىلار ئىگىدارلىقىدىكى يەرلەرنى ئۆتكۈزىۋېلىپ ئىشلىتىش لاھىيىسى شەھەرلىك، ناھىيىلىك ھۆكۈمەت تەرىپىدىن تەستىقلانغاندىن كېيىن مۇناسىۋەتلىك تەرەپلەرگە ئۆز ۋاقتىدا تۆلەم بېرىلىشى كېرەك. دۆلەت زېمىن بايلىق ئورگانلىرى يېزا ئىگىلىك، خەلق ئىشلىرى ۋە باشقا مۇناسىۋەتلىك تارماقلار بىلەن بىرگە يىرى ئىشلىتىلگەن كوللىكتىپنىڭ ئىچكى قىسمىدىكى تۆلەم پۇل تەقسىماتىغا نازارەتچىلىك قىلىشى كېرەك.

تۆلەم كويفىنسىنتى = ۋاقىتلىك ئىگىلەنگەن يەر (ئىككى يىلدىن يۇقىرى ئىشلىتىلگەن) ھەر يىللىقى ئۈچۈن 2.5-5ھەسسە. مەڭگۈلۈك يەر (3يىلدىن يۇقىرى، تۆۋەندىمۇ ئوخشاش) 10-25ھەسسىگىچە.

2.دۆلەت ئىگىلىكىدىكى باشقا ئورمانلار تۆلىمىنى ھېسابلاش (مايسىلىقنى ئۆز ئىچىگە ئالمايدۇ)

باشقا ئورمان تۆلىمى = يېزا ئېتىزلىرىنىڭ ئالدىنقى ئۈچ يىللىق ئوتتۇرىچە كىرىمى * ئورمان كۆلەم ۋە تۆلەم كويفىنسىنتى

3.كوللىكتىپ ئىگىلىكىدىكى باشقا مەڭگۈلۈك ئورمانلار تۆلىمىنى ھېسابلاش (مايسىلىقنى ئۆز ئىچىگە ئالمايدۇ)

كوللىكتىپ ئىگىلىكىدىكى ئورمان تۆلىمى = يېزا ئېتىزلىرىنىڭ ئالدىنقى ئۈچ يىللىق ئوتتۇرىچە كىرىمى * ئورمان كۆلەم ۋە تۆلەم كويفىنسىنتى (6-10)

4. كوللىكتىپ ئىگىلىكىدىكى باشقا ۋاقىتلىق ئورمانلار تۆلىمىنى ھېسابلاش (مايسىلىقنى ئۆز ئىچىگە ئالمايدۇ)

كوللىكتىپ ئىگىلىكىدىكى ۋاقىتلىق ئورمان تۆلىمى = يېزا ئېتىزلىرىنىڭ ئالدىنقى ئۈچ يىللىق ئوتتۇرىچە كىرىمى * ئورمان كۆلەم ۋە تۆلەم كويفىنسىنتى (ئىشلىتىش ۋاقتى بىر يىل بولسا 1.5-3 ھەسسە، ئىككى يىل بولسا 5ھەسسە)

ئون ئالتىنچى، ئۆيى چېقىلغان ئالاھىدە قىيىنچىلىقى بارلار، مىيىپلار، ۋە باشقا كىرىمى بولمىغا مەجرۇھلار ئۈچۈن تۆۋەندىكىدەك تۆلەم ۋە ئىتبار بېرىلىدۇ:

1.«تۆۋەن تۇرمۇش كاپالەت كېنىشكىسى» بار ئائىلىلەرنىڭ ئۆيلىرى چېقىلسا، تۆلەم پۇلى 55000يۈەندىن تۆۋەن بولسا 55000يۈەن بويىچە ھېسابلىنىپ بېرىلىدۇ.

2.«تۆۋەن تۇرمۇش كاپالەت كېنىشكىسى» بار مىيىپ كىشىلەرنىڭ ئۆيلىرى چېقىلسا، «جۇڭخۇا خەلق جۇمھۇرىيتى پۇقرالار مېيىپلىك كېنىشكىسى» دىكى ئىزاھاتقا ئاساسەن، بىرىنچى ۋە ئىككىنچى دەرىجىلىك ئاڭلاش، سۆزلەش ۋە باشقا ئەزالار جەھەتتە مېيىپ بولسا، ياكى كۆرۈش، روھى كېسەل قاتارلىق كېسەللەر بولسا، بىرىنچى تۈردىكى تۆلەمگە يەنە 10000يۈەن قوشۇپ بېرىلىدۇ.

ئون يەتتىنچى، ئالاھىدە ئەھۋاللاردا مۇناسىۋەتلىك بەلگىلىمىلەرنى ئاساس قىلىپ شۇ جايدىكى مال باھا ئۆلچىمى بويىچە، شەھەرلىك باشقۇرغۇچى ئورگانلار ئىسپات ئاڭلاش يىغىنلىرىنى ئېچىپ، مۈلۈك ئىگىلىرى بىلەن كېڭىشىپ بىر تەرەپ قىلسا بولىدۇ.

ئون سەككىزىنچى، ئىقتىسادى قۇرۇلۇش خەلق تۇرمۇشى قۇرۇلۇشىدۇر. يەرلەرنى ئېلىۋېلىپ ئىشلىتىش پۇقرالارنىڭ قوللىشىغا ئىرىشىشى، پۈتۈن خەلقنىڭ يۇرىت ماكان قۇرۇش كەيپىياتىنى بەرپا قىلىش، زورلۇق كۈچ بىلەن چېقىش ھەركەتلىرنىڭ ئالدىنى ئېلىش، ئەگەر پۇقرالارنى قورقۇتۇش، تەھدىت سېلىش ئەھۋاللىرى كۆرۈلسە بىۋاستە مەسئۇل رەھبەرنىڭ مەسئۇلىيىتى سۈرۈشتە قىلىندۇ.

مەزكۇر پىرىنسىپ 2011-يىلى 3-ئاينىڭ 1-كۈنىدىن باشلاپ ئىجرا قىلىنىدۇ.
