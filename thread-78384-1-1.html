ئوتتۇرا ياشلىقلاردا نىمە تۈگىمىگەن ھاراق جىدىلى؟




ئەسسالامۇ ئەلەيكۇم ھەممىڭلارغا ئوتلۇق سالام .
  پارتىيەنىڭ توغرا رەھبەرلىكىدە تۇرمۇش ساپارىمىزنىڭ ئۆسۈشى ،مىللەت سۆيەر ،ئەخلاق سۆيەر كەڭ خەلىق ئاممىسىنىڭ مەدەت بىرىشى ،مۇھىمنىڭ موھىمى يىراقنى كۆزلىگەن،ھاراق مەدىنىيىتىگە ھۆ دىگەن ،رېستۇران خۇجايىنلىرىمىزنىڭ ئاۋاز قوشىشى ئارقىسىدا، شىنجاڭ تەۋەسىدە ھاراق قۇيۇشقا يول قويمايدىغان رېستۇرانلار كۈندىن- كۈنگە كۆپەيدى .دۇنياغا قارايدىغان بولساق قايسى رايون نامرات ۋە قالاق بولسا شۇ رايوندا ئىچەرمەنلىك  ئەۋىج ئالغانلىقىنى بايقايمىز ،قايسى رايون تەرەققىي قىلسا شۇ رايوندىكى كىشىلەر سۈپەتلىك تۇرمۇشقا شۇنچە ئېتىبار بىرىپ،ھاراقنى ئاز ئىچىۋاتقانلىقىنى بايقايمىز .مانا تۇرمىشىمىز ياخشىلىنىپ ھاراققا ياق دەيدىغان ھالەتكە يەتتۇق .ئەسلىدىمىغۇ دىنى ئەنئەنىمىز ۋە مىللى ئۆرۈپ ئادىتىمىز بويىچە ھىندىئانلارنى تۈگەشتۈرگەن  بۇ ھەلەققە يېقىن كەلمەسلىكىمىز كىرەكتى .ھاراق ئۇيغۇرغا ئاز بولمىغان دەرت ئەلەمنى ،ئەخلاقى پاساتچىلىقنى،روھىي يوقىتىشلارنى ۋە مىللەت ئىچىدىكى زىديەتلەرگە سەۋەپ بولمىدى  ،شۇڭا بۇ ھالەتنى ئۆزگەرتىش ئۈچۈن ھەممىمىز يىللاردىن بۇيان ئويلاندۇق ،ئىزدەندۇق ،مانا ھازىرئۇيغاندۇق ،ھاراق تاشلاش ھازىر ياشلىرىمىزدا ئاساسىي ئېقىمغا ئايلانغىلى تۇردى .
مىسالەن :مەن تۇرىۋاتقان قومۇل رايونىدا يېقىندا ،بارلىق توي رېستۇرانلاردا ھاراق قۇيۇلماسلىقتەك ياخشى كەيپىيات شەكىللەندى .بۇ ھادىسە ھەرگىز مەلۇم بىر كىشىلەرنىڭ پىلانلىق قىلغان سۈنئىي ھادىسە ئەمەسلىكى ئەقىللىك كىشىلەرگە ئايان .بۇ ھادىسە شۇنداق تەبىئىي پەيدا بولدى .چۈنكى مەيلى مەرىكە  ئىگىسى بولغاي ،مەيلى رېستۇران خۇجايىنلىرى بولغاي ھاراقتىن زىرىكتى .كۆپچىلىكنىڭ كۆڭلىدىكىچە بولسا ھاراقنى مەرىكىلىرىگە  قەتئىي يولاتماسلىقنى ئويلاپ تۇرۋاتاتتى .لېكىن ئاز بىر قىسىم روھىي مەغلۇب كىشىلەرنىڭ قۇتراتقۇلىقى ،باتنىشى ،سېسق گەپلەر بىلەن كەلتۈرۈشلىرى ئارقىسىدا كۆپچىلىكنىڭ بۇ ئارزۇيى يىللاردىن بۇيان ئەمەلگە ئاشماي كېلىۋاتاتتى .مانا ھازىر قومۇلدىكى بارلىق رېستۇرانلىرىدا تەبىئىيلار ھاراق ئىچىلمەسلىك ھادىسىسى شەكىللەندى .بۇ ھادىسە ھەرگىز رېستۇران خۇجايىنلىرى مەسلىھەتلىشىۋالغان ئىش ئەمەسلىكى بىزگە ئايان بولىۋاتىدۇ .بەلكى يوقىردا دىيىلگەندەك تۇرمۇش ساپارىمىزنىڭ ئۆسىشى ۋە ئەخلاق قارىشىمىزنىڭ ئەسلىگە كېلىشى بىلەن تەبىئىي شەكىللەندى .ئاللاھنىڭ بىزگە بۇ نىئمەتنى ئاتا قىلغانلىقىغا شۇكرى ھەمدۇسانە ئېيتايلى .
لېكىن مانا مۇشۇ تەستە كەلگەن ۋەزىيەتنى كۆزگە ئىلىمايدىغان بەزى كىشىلەر ،ئۆز شەيتانى نەپسىنى دەپ، رېستۇران ئىگىلىرىنىڭ دىلىغا ئازار بىرىدىغان ،ھاراق بەرمىگەنلىكى ئۈچۈن ئۇلارنى كەمسىتدىغان، ھەتتا  مىنىرال سۇ بوتولكىسىغا ھاراق قۇيۇپ ئەكىرىپ ،بىر چىرايلىق پاكىز مەرىكىنى بۇلغايدىغان ئىشلار دائىم يۈز بېرىۋاتىدۇ .شۇنى بايقايمىزكى بۇ ھادىسىلەرنى پەيدا قىلىۋاتقانلارنىڭ ئالدىدا ئوتتۇرا ياشلىق ئىچەرمەنلەر كۆپساندا ئىكەنلىكىنى بايقايمىز .
يېقىندا مەن بىر ساۋاقدىشىمنىڭ تويىغا قاتنىشىپ قالدىم .بۇ توي ئىۋىرغول رېستۇرانىدا ئېلىپ بېرىلدى. ئېۋىرغول رېستۇرانى قومۇلدا ھەممىدىن بالدۇر ھاراققا چەك قويغان پاكىز رېستۇران بولۇپ قالغانلىقىنى ھەممىمىز بىلىمىز .شۇ كۈنكى تويدىمۇ توي ئىگىسى ھاراق يولاتمىدى .مېنىڭ بىلەن جوزىداشلار ئاساسەن تونۇش كىشىلەر بولغاچقا، ئۆز- ئارا گۇڭۇر- مۇڭۇر پاراڭلىشىپ ئولتۇرغان ئىدۇق .پاراڭ تېمىسى رېستۇرانلاردىكى بۇ ياخشىلىنىش ئۈستىدە بولىۋاتاتتى .بىردەمدىن كېيىن مەدەنىيەت ساھەسىدە خېلى نام ئابرويى بار ،قومۇل ناخشا -ئۇسسولى ،رەسساملىق،خەتتاتلىق  سەنئىتىگە تۆھپە قوشىۋاتقان مەلۇم بىر ئىسىل زادىمىز كىرىپ كەلدى .ھەممىمىز بۇ كىشىنى تونۇغاچقا ئۇنىڭغا ھۆرمەت سالاملىرىمىزنى بىلدۈردۇق .
مەن بۇ كىشىنى سىرتتىن  تونۇپ، ئۇنىڭ قىلىۋاتقان ئىشلىرىغا ئاپىرىن ئېيتقان ئىدىم .لېكىن بۈگۈنكى غەشلىككە بۇ كىشىنىڭ سەۋەپ بولۇپ قالىدىغانلىقىنى ئەسلا ئويلىمىغان ئىدىم ھەم بۇ كىشى بىلەن بۇنداق يېقىن ئارلىقتا سۆھبەتلىشىپمۇ باقمىغان ئىدىم .
تاماقلار تارتىلىپ ھەممە مەززە بىلەن غىزالىنىۋاتقان بەھۇزور كەيپىياتنى بۇ كىشى بۇزدى .
--چۈشتىلا دىدىم بۇ گۇيغا ،ھاراق قۇي، ھاراق قۇي دەپ .ھاراق قۇيمىسا توينىڭ كۆركى نەدە قالىدۇ .ئەتە مانا شىكەر شەنبە ،بۈگۈن كەچ بىر پەيزە قىلساق،قېنىپ ئۇخلىساق ئەتە ۋاخچە تۇرۇپ ئۇيقىنى قاندۇرساق بەلەن بولمامدۇ .شۇنداق ئەمەسمۇ ھە ئاداش .
دەپ ئۇددۇلىدىكى بىر ئەمەلدارغا سۆز تاشلىدى .ئۇ ئەمەلدار زورىغا كۈلۈمسىرەپ،كۆڭلى يەردە قالمىسۇن دىدىمۇ -ھە شۇ ئەمەسمۇ -دەپلا قويدى .
لېكىن سورۇنكى باشقا كىشىلىرىمىزدىن ئۇنىڭ ئاغرىنىشلىرىغا ئاۋاز قوشقۇدەك سادا چىقمىدى .
بۇ ھالەتتىن نارازى بولدىمۇ ،ئۇنىڭ تولا سورۇن ئويناپ خۇنۇكلەشكەن كۆزلىرى تېخىمۇ خۇنۇكلىشىپ كەتتى ،يۈزى قارىداپ ،بۇجغۇر چاچلىرى ئارسىدىن تەرلىرى چىقىپ ،سىمىز بەدىنى تىترەپ كەتتى .ئۇ دەرھال ئورنىدىن تۇرۇپ باشقىلارنى كۈتىۋاتقان توي ئىگىسىنىڭ ئاكىسىنى يېنىغا چوڭ ئاۋازدا چاقىردى .
--ھەي ماياققا كىلە ئاداش ،چۈشتىلا دىدىمغۇ مەن ساڭا ،قېنى ھاراق ؟قۇيمامسەن ھاراقنى دەپ ۋاقىرىدى .ئۇنىڭ ئاۋازىنى ئاڭلىغان باشقا جوزىدىكىلەرنىڭ سوغۇق نەزىرى ئۇنىڭغا ئاغدۇرۇلدى .سورۇن ئىگىسى ئازراق ئوڭايسىزلاندىمۇ جىق گەپ قىلغۇسى كەلمەي پەقەتلا .
-ھە ماقۇلا ،جىم ئولتا ئورنىڭدا ئاداش دەپلا كېتىپ قالدى .لېكىن شۇ كەتكىنىچە بۇ جۇزىغا يېقىن يولىمىدى .
بۈگۈنكى مەرىكىنىڭ ھاراق جىدەلخورى بولغان مىللىتىمىزنىڭ بايلىقى بولمىش،  بۇ ئەتىۋارىمىز خۇددى  لالما كۈچۈكتەك  ئىچىدە غۇدۇڭشىپ ،بىرنىمىلەرنى دەپ   جىمىىپ قالدى .
مانا بۇ مەن يولۇققان ھادىسىلەردىن كىچىك بىر مىسال .ئەمەلىيەتتە مۇشۇنداق ھاراق جىدىلى قىلۋاتقانلارنىڭ كۆپ ساندىكىلىرى 38 ياشتىن يوقىرى بولغان ئوتتۇرا ياشتىكىلەر بولىۋاتىدۇ .بىلمىدىم بۇ زادىي نىمە ئۈچۈنكى ؟ مانا تۆنۈگۈنلا قاتناشقان بىر تويدا يەنە يوقىرقىدەك ھادىسىنى ئۇچىراتتىم .كۆپچىلىك مەززە قىلىپ غىزالىنىۋاتاتتى .توساتتىن بۇرنىمىزغا ھاراقنىڭ پۇرىقى كەلدى .قارىسام بىر ئىشەكباش مىنىرال سۇ بوتولكىسىغا ھاراقنى قاچىلاپ ئەكىرىپ ،چاي چىنىسىغا ھاراقنى قۇيۇپ ،يېنىدىكى ھەمرايىغا تەڭلىگىلى تۇرۇپتۇ .ئۇ بىزنىڭ بىلىۋالغانلىقىمىزنى بايقاپ، ھەييارلىق بىلەن بىزگە سەت ھىجىيىپ -قېنى ئىچىدىغانلار بارمۇ ؟ئاز پاز ئويناپ قويمامدۇق دەيما .دەپ بېزىرىپ تۇردى .
لېكىن بۇ سورۇندا ئۇنىڭغا ئۈلپەدداش بولىدىغاندىن پەقەت ئىككى نەپەر تونىشى زورىغا ئىچىپ قويدى .مەن بىر خىل مەنسىتمەسلىك ۋە غەزەپ بىلەن ئۇنىڭغا قاراۋاتقان ئىدىم.ئۇنىڭ كۆزى ماڭا چۈشتى ۋە ئازراق ئوڭايسىزلاندىمۇ ئېيتاۋۇر بوتولكىنى ئېلىپ ،باشقا سورۇندىكى تونۇشلىرىنىڭ يېنىغا كەتتى .مانا بۇ بۇزغۇنچىمۇ دەل ئوتتۇرا ياشلىق .
مىسالغا ئالسام بۇنداق مىساللار بەك تولا .شاراپسىز سورۇنلارنى ئاساسەن ئوتتۇرا ياشلىق ئىچەرمەنلەر بۇزماقتا .سورۇن ئىگىسىنىڭ مەقسىتى پاكىز بىر مەرىكە قىلىش بولسا، بۇ ساراڭلارنىڭ مەقسىتى سورۇن ئىگىسىنى بۇ مەقسىتىگە يەتكۈزمەسلىك بولدى .ئۇلار نىمىدىگەن شەخسىيەتچى ،ئىچى تار ھە!!
كۆپچىلىك سىلەرمۇ شۇنداق ھادىسىگە ئۇچىرىدىڭلارمۇ؟ئوتتۇرا ياشلىقلاردىكى بۇ ھادىسىنىڭ سەۋەبى  نىمە زادى؟

ھاراق ئىسلام دىندا ھارام قىلىپ  قۇرئاندا مۇنداق بايانقىلدى ئاللاھ قۇرئان كەرىمدە مۇنداق دەيدۇ:« ئى


مۆئمىنلەر! ھاراق ئىچىش،قىمار ئويناش،بۇتلار(يەنە چۇقۇنۇش ئۈچۈن تىكلەنگەن تاشلار)غا چۇقۇنۇش،پال ئوقلىرى بىلەن پال سېلىش شەيتاننىڭ ئىشى،پاسكىنا قىلىقلاردۇر، بەختكە ئىرىشىشڭلار ئۈچۈن شەيتاننىڭ ئىشىدىن يىراق بولۇڭلار‹90›، شەيتان ھاراق،قىمار ئارقىلىق ئاراڭلاردا دۈشمەنلىك ،ئاداۋەت تۇغدۇرماقچى،سىلەرنى نامازدىن ۋە ئاللاھنى ياد ئىتىشتىن توسماقچى بولىدۇ.سىلەر ئەمدى (ھاراقتىن،قىماردىن) يانمامسىلەر‹91›»؟ « سۈرە مائىدە 90-91-ئايەتلەر»
بۇ  توغرىسىدا  پەيغەمبەر ئەلەيھىسسالام مۇنداق  دىگەن جەيشان دىگەن يەردىن بىر ئادەم كىلىپ پەيغەمبەر ئەلەيھىسسالامدىن ئۆزلىرىنىڭ زىمىندا تېرىقتىن ياسىلىدىغان مىزر دەپ ئاتىلىدىغان ئىچىملىك توغرىسىدا سورىغانىدى، پەيغەمبەر ئەلەيھىسسالام :ئۇ نەرسە مەست قىلامدۇ؟ دەپ سورىدى، ئۇ ئادەم ھەئە دەپ جاۋاپ بەردى. پەيغەمبەر ئەلەيھىسسالام : ھەرقانداق مەست قىلغۇچى نەرسە ھارامدۇر، ئۇلۇغ ئاللاھتائالا مەست قىلغۇچى نەرسىنى ئىچكەن ئادەمنى «تېنەتى خەبال» دىگەن نەرسە بىلەن سۇغىرىشقا ئەھدى قىلغان، دېدى. ساھابىلار، ئى رەسۇلۇللا: «تېنەتى خەبال» دىگەن نىمە ئۇ دەپ سوراشتى. رەسۇلۇللا: ئۇ دوزىخىلاردىن چىققان تەر ياكى دوزىخىلاردىن سىقىپ چىقىرىلغان نەرسە دەپ جاۋاپ بەردى. (بۇنى مۇسلىم بىلەن نەسەئى رىۋايەت قىلغان)

ھاراقنىڭ يەنە ئىنساننىڭ جېنىغا  بولغان تەھتىتى بەك  ئېغىر ..جىگەر گە بولغان  زىينى　جىگەر كېسلىگەگىرىپتار بولغانلار ھاراق ئىچسە، جىگەر كېسىلى ئاسان قوزغىلىپ، جىگىرى تېخىمۇ ئاجىزلاپ، ھەتتا راكقا ئايلىنىپ كېتىدۇ. شۇڭا مەيلى ئىسپىرىت تەركىبى يۇقىرى ئاق ھاراق ياكى ئىسپىرىت تەركىبى تۆۋەن پىۋا ۋە ياكى ساقلىق ساقلاش رولىغا ئىگە قىزىل ھاراق بولسۇن، ھېچقايسىسىنى ئىچىشكە بولمايدۇ. B تىپلىق جىگەر كېسىلى ۋە جىگەر قېتىشىش كېسىلىگە گىرىپتار بولغانلار دائىم ئىنچىكە ئوسۇڭ، بامبۇك نوتىسى ئىستېمال قىلىشى كېرەك، بۇنىڭ جىگەرنى قوغداش ۋە جىگەر راكىنىڭ ئالدىنى ئېلىش رولى بولۇپلا
بۇ توغرىسىدا ئامىركىلىق تەتقىقاتچىلارنىڭ   سۇزى ئامېرىكا كېسەللىكلەرنى كونترول قىلىش ۋە ئالدىنى ئېلىش مەركىزى ھاراقكەشلىك پىلانىنىڭ مەسئۇلى دوكتور روبېرت برېۋېر يېقىندا مۇنداق دېدى: «بۇلتۇر 450 مىڭ ئاد
ەمنى تېلېفوندا زىيارەت قىلىش ئارقىلىق، ئامېرىكا ياشلىرىنىڭ ھاراق ئىچكەندە، بىر قېتىمدا ئاز بولغاندىمۇ توققۇز ئىستاكان پىۋىنى ئىچىۋېتىدىغانلىقىنى، ھەر يىلى ئوتتۇرا ھېساب بىلەن تۆت قېتىم غەرق مەست بولغىچە ھاراق ئىچىدىغانلىقىنى، ھەر يىلى 40 نەچچە مىڭ ئادەمنىڭ ھاراق ئىچىپ ئۆلىدىغانلىقىنى بىلدۇق».




دوكتور برېۋېر يەنە مۇنداق دېدى: «بۇ ھۆكۈمنىڭ كىشىلەرنى چۆچۈتىدىغانلىقى ماڭا ئايان. بىراق بىز بۇ ھۆكۈمنى ھەقىقىي ئەھۋالدىن خېلىلا تۆۋەن مۆلچەر ئاساسىدا چىقاردۇق. ئامېرىكىنىڭ ھاراق سودىسى دوكلاتىغا قارىغاندا، كىشىلەر ئەمەلىي سېتىۋالىدىغان ھاراقنىڭ مىقدارى بىزنىڭ مۆلچەرىمىزدىكىدىن خېلىلا كۆپ ئىكەن. ھەر يىلى ھاراق 40 نەچچە مىڭ ئامېرىكىلىقنىڭ بېشىغا چىققاندىن باشقا، يەنە قانلىق ۋەقەلەرنىمۇ پەيدا قىلىدىكەن. ھاراق ئىچكەندىن كېيىن، ئاپتوموبىل ھەيدەپ قاتناش ۋەقەسى پەيدا قىلىدىغانلارنىڭ سانى كۆپەيگەن، ھاراق ئىچىپ راك قاتارلىق جانغا زامىن كېسەللەرگە گىرىپتار بولغانلارمۇ ئاز ئەمەس ئىكەن. نۇرغۇن ئادەم ھاراقنىڭ كاساپىتىدىن يۈرەك كېسىلى ۋە جىگەر ئىقتىدارى ئاجىزلاش كېسىلىگە مۇپتىلا بولىدىكەن».

بۇيەردە  دەپ ئۆتمەكچى بولغان  قىسقا  سۈزىمىز بولسا ھارقنى ئاللاھ  قۇرئان دا ئۇچۇق  قىلىپ  چەكلىدى .يەنە  پەغەمبىرىمىزمۇ چەكلىدى  ..تەرەققى قىلغان ئەللەردە  بۇنىڭ  زىيىنى  بىلىپ بۇتوغرىسىدا  ماقالىلەرنى ئىلان قىلماقتا لىكىن  بىزلەر تېخى  ھىس قىلالمايۋاتىمىز .. بىز مۇسۇلمانلىق  سۇپىتىمىز بىلەن  ھاراق ئىچسەك ئۇزۇمىزنىڭ ئىمانىمىزغا  ھاقارەت قىلغان  بۇلىمىز ،،يەنە ھاراق  ئىچسەك ئۈزىمىزنىڭ ئۇيغۇرلۇقىمىزغىمۇ  ھاقارەت قىغان بۇلىمىز ..يەنە ھاراق ئىچسەك ئۈزىمىزنىڭ  مۇبارەك  تەنلىرىمىزگەئىكەك  سالغان  بۇلىمىز ..يەنە گۇزەل  ئايالىمىز  ئوماق  پەرزەنتلىرىمىزنىڭ  خوشال  تۇرمۇشىغىمۇ چاڭ  سالىمىز ،،ئاتا ئانىمىزنىڭ  قاغىشىغا خەلىقنىڭ  نەپرىتىگە  ئۇچرايمىز .سىز  ھاراقئىچىدىغان ئادەم بولسىڭىز  مۇنداق قىلىپ  بېقىڭ سىز شۇمەس  بولمىغان (ھاراق ئىشمەيتۇرۇپ) شۇ  سىز دائىم ھاراق ئىچىدىغان  بويكىلار.ياكى  قاۋاقلارنىڭ  ئالدىدا  ئودا  5كۈن تۇرۇڭ  بۇچاغدا ،ئۈزىڭىزنىڭ  ئەينىنى كۈرىسىز  بۇچاغدا ئۈزىڭىزدىن نۇمۇسقىلىسىز  ،بۇيەرلەردە  تۇمشۇقۇنى  ژۇندا ئورىكىگە چىلاپ  ياتقان  بايقۇشلارنىڭ  .يەنە  ئىشتاللىرىغا سىيىپ  تۇرغان لىقىنى كۈرۈپ بەلكىم ئىككىنچى بۇنداق  يەرلەرگە كەلمەسلىكىڭىز  مۇمكىن چۇنكى سىزمۇ  بىر  چاغلاردا قۇسۇقلىرىڭىزنى  يالاپ  يولقىرغىدىكى  لىڭدەيدە يېتىپ  ھاراقنىڭ كەيپىنى سۈرۈۋاتقان ۋاختىڭىزدىمۇ  نۇرغۇن  ئادەملەر سىزگە  قاراپ كۈلگەن  نۇرغۇنلىرى  ياقىسىنى تۇتۇپ  توۋا دەپ  كىتىپقالغان  ،بىچەرە  ئايالىڭىز  سىزنى يۇلەپ  ئۆيگە  ئەكىرسە ئۇنى راس ئۇرغان  ..شۇ كۈنلىرىڭىز  يادىڭىزغا يىتىدۇ  دە ئىككىنچى  ئاياق باسماسلىقىڭىزمۇمكىن  ..

بۇرادەرلەر  ھاراق  ئىنساننىڭ بۇدۇنيادىمۇ  ئۇ  دۇنيادىمۇ  خار قىلىدىغان  بىر شەيتاننىڭ  سۇيدۇكىدۇر . ھاراق ئىچىش نىرۋائاجىلىق .جىنسى ئاجىزلىق ،ئاشقازان  راكى  ،جىگە  راكى  قاتارلىق قورقۇنۇشلۇق كېسەلنى سىزگە  سوۋغا قىلىدۇ ..ھاراق  سىزگە  ھاياسىزلىقنى ئۆگۇتۈدۇ ....قېرىنداشلار  ھاراقتىن  يىراق  تۇرايلى ..ھاراق  بىزنىڭ  مىللىتىمىزنىڭ  ئەشەددى  دۇشمىنى ..خوش  ماشەگىچە يازاي .ھەممىڭلارغا ئاللاھنىڭ  رەھمىتىبولسۇن  بارلىق مۇنبەر ئەھلىگە ئاللاھ ئىمان  ئاتاقىلسۇن
