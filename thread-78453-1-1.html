     ئەسسالامۇ ئەلەيكۇم  تورداشلار ، بۈگۈنكى قىلغان ئىشلىرىڭلار كۆڭۈللۈك بولغاي !!...
    مەن كەڭ تورداش دوستلىرىمغا << ھۆكۈم قېلىڭ!!(مۇھەببەت خېتى) ، داۋامى2 >> نى يازغاندىن كىين خېلى كۆپ دوسىتلار قىزىقىپ پىكىر قاتناشتۇرۇپتۇ ، بۇنىڭغا  كۆپ رەھمەت ئېيتىمەن ، تورداشلىرىمىزنىڭ ئىنكاسنى تۇلۇق كۈرۈپ چىقىتىم ، كۆپ قېسىم تورداشلىرىمىز ناھايتى ئەمىليەتچانلىق بىلەن جاۋاب بىرىپتۇ ، بەزى ئەھۋالدا ئادەم بۇنداق ئىشلاردا ئېنىق بىر نەرسە دەپ بولغىلى بولمايدۇ ،  ئاتا - ئانا ھەقىقەتەن ئۇلۇغ ، ئەمما ئاتا - ئانىنىڭمۇ خاتالىشىپ قالىدىغان ۋاقتى بۇلىدۇ ، چۈنكى كونىلاردا << ئەقىل ياشتا ئەمەس ، بەلكى باشتا >> دىگەن گەپ بار ئەمەسمۇ ؟...شۇڭا بىر قىسىم ئىشلاردا خۇدا دەپ ياخشى تەرەپنى ئويلاپ ئىش قىلمىساق بېكار . ئەمدى بۇ يەردە تورداشلارنىڭ قايتۇرغان ئىنكاسىغا نىسبەتەن دەپ ئۆتمەكچى بولغېنىم ئاتا - ئانىنى رازى قىلىش ھەممىدىن مۇھىم ، بىراق بەزى ئەھۋالدا ئاتا - ئانىنىڭ ئۆز پەرزەنتلىرىگە نىمنى دەپ شۇنداق تەلەپ قويغانلىغىنىمۇ بىلىپ  بېقىش زۈرۈر . مۇھەببەت ۋە تۇرمۇش پۇلدىن ئايرىلالمايدۇ ، ئەمما پۇلنى بەك يۇقىرى ئۇرۇنغا قۇيىۋېلىشقىمۇ ھەم بولمايدۇ ، بۇنىڭدا  چۇقۇم ئۇغۇل ۋە قىزلاردا ئۈزىگە  خاس ۋىجدان ۋە ئەخلاقنىڭ  بۇلىشى   ھەممىدىن بەك مۇھىم ، شۇڭا بۇ يەردە مەن دىمەكچى  مۇھەببەت ۋە ياخشى كۈرۈشتە خاتالىق يوق . مۇھەببەت بولسا ئۇلۇغ ئاللانىڭ ھەربىر بەندىسگە ئاتا قىلىغان مەنىۋى بايلىقى ، ئەمما بەش قول ئوخشاش بولمىغاندەك تۇرمۇشتا مۇھەببەت ۋە كەسىپ جەھەتتىن تەڭلا ئۇتۇق قازىنىپ كېتىشىمىز ناتايىن ، ئەمما نىيتىمىز دۇرۇسلا بولسا ، بۇ كۆنلەرمۇ  كۆزدىن ئانچە يېراق ئەمەس ،  شۇنداق بولغانىكەن مەيلى بىز ياكى دوسىتلىرىمىز ، يېقىن ئۇرۇق - تۇغقانلىرىمىز ۋە ياكى  ئاتا - ئانىمىز  بولسىۇن تۇرمۇشىمىزدا بۇلارنىڭ  ھەممىسنى بىرلا ۋاقىتتا  قولغا كەلتۈرۈش ئويىدا بۇلۇش مەلۇم جەھەتلەردىن ئېلىپ ئېيتىقاندا بۇ بىر ئاچكۆزلۈكنىڭ ئىپادىسى .
      بۇ خەتتە دېيىلگەندەك قىز ۋە ئۇغۇلنىڭ مۇھەببتىدە مەنچە خاتالىق يوق . ئۇ ئوغۇل بالىمۇ كۆڭلىدىكى گەپنى ئەمىليەتكە ماسلاشتۇرۇپ دەپتۇ ، قىز بالىنىڭ كۆڭلىگىمۇ خېلى ھۆرمەت قىلىپتۇ ۋە قىزنىڭ ئاتا - ئانىسنىمۇ ئويلاپتۇ ، بىراق بۇيەردە ئۇ قىز بالىنىڭ ئاتا - ئانسى بولسا ئىزچىل ئۆز بالىسنىڭلا مەنپەتىنى ئويلاپتۇ ، تۇرمۇشتا بىرقىسىم ئىشلارغا ۋاقىت كېرەكلىكنى ئېسىدىن چىقىرىپ قۇيۇپتۇ ، شۇنداق بولغان ئىكەن ، بۇ قىز بالىنىڭ ئاتا - ئانىسنىڭ قىزى ئۈچۈن ئويلىغىنىنى تازا توغرا دەپ كەتكىلى بولمايدۇ ، ئەلۋەتتە بۇ يىگىتمۇ كىينچە توي قىلىپ قالسا قاراپ ئولتۇرماستى  .
     بۇ خەتتىن قارىغاندا بۇ ئۇغۇل بالىنىڭ ئۈزىگە ئىشەنچىسى بار ئىكەن ، بۇ قىز بالىنمۇ ھەم شۇنداق ياخشى كۈرىدىكەن ، قىز بالىمۇ  بۇ ئۇغۇل بالىنى ئانچە - مۇنچە ياخشى كۈرۈپمۇ قالغان ئوخشايدۇ ،  شۇڭلاشقا  قىز بالىمۇ بۇ ئىشتا  ئۆز ئاتا - ئانىسى ۋە بۇ يىگىت ئارلىقىدا  تېڭىرقاش ئىچىدە قالغاندەك كۈرۈنىدۇ ،شۇڭا  بۇنى توغرا چۈشنىشكە بۇلىدۇ .
    ئەمدى بۇ خەتتە بۇ ئۇغۇل دىگەندەك  ئادەمنىڭ  ئۆزىنى بىلىشى ئىنتايىن مۇھىم ، ئادەم ئۈزىنى بىلگەندىلا ، ئاندىن باشقىلارنى چۈشەنگىلى ۋە شۇ ئارقىلىق ئۆز تۇرمۇش يۇلىدا توغرا يۇلىنى تېپىپ ماڭغىلى بۇلىدۇ ، بۇنداق ئادەمگە يىتەكلەش ۋە باشقۇرۇش  كەتمەيدۇ . بۇنىڭغا كونىلاردىكى بىر ھېكمەت << ئۆزىنىمۇ بىلسە ، ئۆزگىنىمۇ بىلسە ھەممە ئىشلاردا غەلبە قىلغىلى بۇلىدۇ>> دىگەن تەمسىل ياخشى مىسال بۇلالايدۇ . ھازىر بۇ  ئىشلارغا  يۈز بەرگىلى ئاللىقاچان ئىككى يىلدىن ئاشىتى ، ئەمما بۇ ھېكاينىڭ ئاخىرى ئەملىيەتتىمۇ تېخىچە  تۈگىگىنى يوق . ئۇلۇغ ئاللا بۇ بىر جۈپلەرنى بەلكىم مۇشۇنداق بىر سېناپ بېقىشنى ئويلاپ قالغاندۇ ؟... ياكى بۇ بىر جۈپلەر ئۇزىمۇ بىلىپ - بىلمىگەن ھالدا بۇ سناقنىڭ ئىچىگە  كىرىپ قالغاندۇ ؟...شۇنداق بولغاندا ھازىر تورداشلىرىمىز بۇ قىزنىڭ يىگىتىنىڭ بۇ خېتىگە نىمە دەپ جاۋاپ بەرگەنلىكىنى بەلكىم ھېس قىلپ بولدىغۇ ؟ - دەيمەن . بولمىسا بۈگۈنكى كۈندىمۇ بۇ يەردە بۇ ئىشلارنى تېلغا ئېلىپ يۈرگەننىڭ نىمە لەززىتى بولسۇن . ئەمما مەن بولسام بىر قەلەمكەش . مېنىڭ بۇ ئىشلارغا بۇنىچىۋالا كۈچ سەرپ قىلىپ كېتىشىم مەن بۇ ئىشلارنىڭ بىر قسىم نەتىجىسنى ئۆز كۆزۈم بىلەن كۆرگەن ، ئارىلىقتا يۈزبەرگەن بىر قىسىم كۆڭۈلىسىز ئىشلار كىشنى چۈچۈتكىدەك دەرىجىدە بولغان ، ئەمما پۇشايمان قىلغان بىلەن ئورنىغا ئەكىلىشمۇ ئۇنچىلىك  ئاسان ئەمەستە !!... شۇنداق ۋاقىتتا ھەممە كىشى بۇ ئىشلارنى باشتا  باشلىغان ۋاقىتتىلا  ئازراق دىققەت بىلەن ئىش قىىلىشنىڭ تولىمۇ  زۈرۈرلىكىنى  ھېس قىلىدۇ . شۇڭا بىزگە ئوخشاش كەڭ ياشلار ۋە مېھرىبان ئاتا - ئانىلىرىمىزنىڭ ئۆز باللىرىنىڭ كەلگۈسى ئۈچۈن بولسىمۇ رىئال  تۇرمۇشتىكى مۇھەببەت ۋە خىزمەت مەسلىسىنى ئېنىق ئايرىۋېلىش ۋە مۈمكىن بولسا بۇنىڭغا ئوخشاش ئىشلاردىن  ياخشى نەتىجىگە ئېرىشىش مەقسىتىدە كەڭ تورداشلارغا بۇلارنىڭ ھېكايسىنى تېىما سۈپتىدە يوللاش ئويىغا كېلىپ قالدىم ، شۇڭا توغرا چۈشەنگەيسىلەر .
   بىراق بۇ ئىشلارنىڭ كەينى تېخىمۇ قىزىقارلىق ، قىزىقساڭلار داۋامىنى كۈڭۈل قۇيۇپ تېڭشىغايسىلەر .
   شۇنداق قىلىپ بۇ ئۇغۇل خەتنى قىزغا تاپشۇرغاندىن كىين كەچ تەرەپتە قىزغا تېلفۇن قىلىپ ، بۇ خەتكە قانداق قارايدىغانلىغىنى سورايدۇ ، قىز بۇلىسا بۇ ئۇغۇلنىڭ جاۋابىغا سۈكۈت بىلەن جاۋاب بىرىدۇ ، ئەگەر قىز << دېگىنىڭىز توغرا ، شۇنداق بولسۇن >> دىگەن بولسا بۇ ھېكايە ئېھتىمال  شۇ يەردىلا ئاخىرلاشقان بۇلاتتى ، ئەمما قىز سۈكۈت قىلىپ ھېچىنىمە دېمىدى ، تېلفۇننىمۇ ھەم قويمىدى ،شۇنداق قىلىپ بىر ھازاغىچە ئىككى تېلفۇندىن سادا كەلمىدى ، شۇنىڭ بىلەن ئۇغۇل قىزنىڭ كۆڭلىنى دەسلەپكى قەدەمدە چۈشنىپ يەتتى ۋە قىزغا : << گەرچە ئاتا - ئانىڭىز قۇشۇلمىغان بولىسمۇ ، ئەمما بىز يەنىلا ئۈمىدۋارلىق بىلەن بىر - بىرىمىزنى ياخشى كۈرۈپ ئۈتەيلى ، ئەگەر خۇدايىم نېسىپ قىلغان بولسا بىر كۈنى توي قىلىپ قالارمىز !!..>> دىگەن گەپنى ئېيتتتى . قىز بۇ جاۋابنى ئاڭلاپ ناھايتى خوشاللىق بىلەن يېگىتنىڭ سۈزىگە قۇشۇلدى .
     شۇنىڭ بىلەن بۇ ھېكايىنىڭ دەسلەپكى مۇقەددىمسى بۇ بىر پارچە ئاددىي خەت بىلەن مۇشۇنداق باشلاندى . كىين  ئۈچ ئايچە ئۈتۈپ قىز بۇ يىگىتكە  ئۆز  ئاتا - ئانىسنىڭ ئالدىغا بېرىپ بىر كۈرۈشۈپ بېقىش تەلپىنى قويدى . ئوغۇل دەسلەپ ئىككىلەنگەن بولسىمۇ ، ئەمما يەنىلا ماقۇل كۆردى ، ھەم  ئۇ قىزنىڭ  ئۈيىگە يالغۇز باردى ، خۇدا نىسىپ قىلىپ دەسلەپكى قەدەمدە بۇ ئۇغۇل قىزنىڭ ئاتا - ئانسىغا ياراپ قالدى ، شۇنىڭ بىلەن بۇ ئىشلار بارغانسىرى ياخشىلىنىش تەرىپگە قاراپ ماڭدى ، بۇئۇغۇل بۇ قىزنى بەك ياخشى كۆرگەچكە كينچە ئۇنىڭغا گۈل سوۋغا قىلىشى ئۈزىگە ئادەت قىلىۋالغان ئىدى ، كۈرۈشكەن كۈنلەردە بىر كۈنى بولسا ئىككى تال ، بىر كۈنى بولسا بەش تال ، بىر كۈنى بولسا بىر تال ، بىر كۈنى بولسا بىر قۇچاق گۈل .... شۇنداق قىلىپ بۇلارنىڭ كۈنلىرى باشقىچە خوشاللىقتا خېلى ئۇزۇنغىچە مۇشۇنداق داۋام قىلىدى .
   بىر كۈنى بۇ يىـگىت تور ئارىلاۋېتىپ مۇنداق بىر نەرسىنى ئۇچىرىتپ قالدى . ئۇ ھەر قېتىم مەشۇقىغا ئوخشمىغان ساندا ھەر خىل ئەتتىر گۈللەر ۋە باشقا خىل  گۈللەرنى سوۋغا قىلاتتى ، بىراق بۇنىڭ مەنسنى ئانچە چۈشنىپ كەتمەيتى ، ئەمما بۇقېتىم توردا ئۇچراتقان بۇ نەرسە ئۇنىڭ دىققىتىنى تارتتى ، ھەم بۇ نەرسە ئارقىلىق بۇ قىزنى يەنە بىر خوشال قىلىۋەتمەكچى بولۇپتۇ  ، ۋە ئۈزى تاللىغان ساندىكى ئەتىرگۈلنى ئۈزۈپ سوۋغا قىلماقچى بولدى ، ھەم  شۇنداقلا بۇ نەرسىنىڭ تېمىسىغا  << مېنىڭ قۇياشىم ۋە ئەتتىرگۈل ) دەپ ماۋزۇ قويدى . ( بۇ ئۇغۇل قىزغا يېڭى تۇنۇشقاندا  << مېنىڭ قۇياشىم >> دىگەن ناخشىنى ئاڭلاپ بېقىشنى دىگەن ئىدى ، بۇ قىزمۇ بۇ ناخشىنى بەك ياقتۇرۇپ قالغان بۇلۇپ  كىينچە بۇ قىز يېگىتىنىڭ تېلفۇن نۇمۇرىنى مۇشۇنداق ئىسىم بىلەن ساقىلىۋالغان ئىكەن ) .
   ئۇنداقتا تورداشلار ، تېپىپ بېقىڭلارچۇ ؟... بۇ زادى نىمە نەرسە ؟.... ئەمسىە تۆۋەندىكى قۇرلارغا نەزەر سالغايسىلەر .
