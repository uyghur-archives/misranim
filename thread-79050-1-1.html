ئەسسالاممۇ ئەلەيكۇم مىسرانىم مۇنبىرىدىكى ئەزىزانە تورداشلار كەمىنە قۇندۇزدىن سىزلەرگە ئوتلۇق سالام!
بەزىلەر مۇھەببەتنى ئۇلۇغ دەپلا ئويلايدۇيۇ مۇھەببەتنى چۈشەنمەيدۇ .  
يىقىندا بىر دوستۇم بىلەن مۇھەببەت توغۇرلۇق پىكىرلىشىپ ئۇ ئۆز بىشىدىن ئۆتكەن ئاچچىق مۇھەببەت سەرگۈزەشتىلىرى بايان قىلىپ مۇھەببەتكە بولغان كۆز قارىشىنى قىسقىچىلا بىر كۆبلىت شېئېر بىلەن بايان قىلدى .

                                        مۇھەببەت گوياكى مىسلى بىر ئاپتاپ
                                       چۈشكەندە دىللارغا لال قىلار شۇنداق
                                       نە ئۈچۈن  ئۆزگەردى  ئۇنىڭ  مەناسى
                                       بەردىڭ سەن دىلىمغا  ئالەمچە  ئازاپ

ئۇنىڭ بۇ ھىكايىسىنى ئاڭلاپ ھەيران بولدۇم شۇ سەۋەپ تورداشلار بىلەن ئورتاقلىشىپ باققۇم كەلدى.



                                              مۇھەببەت ئەمەس ئاسىيلىق

ئاپتۇبۇس بىكىتىگە كىرشىم بىلەنلا مەن چۈشمەكچى بولغان ماشىنا قوزغالغان ئىكەن . شۇنچە ئالدىراپ كىلىپمۇ ئارانلا ئۈلگۈرگەن ئىدىم .
ئابتۇبۇس بىكەتتىن چىقىپ بىر رىتىمدا مەنزىلگە قاراپ يۈرۈپ كەتتى . دەرىزىدىن ئىلىنىڭ گۈزەل مەنزىلىرى كۆزگە چىلىقاتتى .
شۇ مەنزىلەرگە قاراپ ئولتۇرغان بولساممۇ كۆز ئالدىمدىن ئۇنىڭ سىماسى كەتمەيتتى . ئۇنىڭ بىلەن دەل مۇشۇ ماشىنىدا دەل مۇشۇ ئورۇندا تونۇشقان ئىدىم شۇ قىتىممۇ ماشىنىغا ئارانلا ئۈلگۈرگەن ئىدىم .

شۇنداق قىلىپ ئۇنىڭ بىلەن ئۇنى- بۇنى دىيىشىپ يۈرۈپ ئۇنىڭ مۇشۇ ناھيەدە ئاياللار كىيم كىچەك دۇكىنى ئاچىدىغىنىنى شەھەرگە مال ئىلىش ئۈچۈن ماڭغىننى بىلدىم . ئۇنىڭ ھەربىر ئىغىز گىپى ھەم قىسقا ھەم ئورۇنلۇق ئىدى .شۇنداق پاراڭلىشىپ مىڭىپ بۇ ئۇزۇن يولمۇ بىردەمدىلا تۈگەپ قالغان ئىدى .
تاكى ئۇنىڭ بىلەن خوشلاشقۇچە ئۇنىڭ چىرايىنىمۇ كۆرمىدىم ئۇ شۇ قەدەر ھايالىق ئىدى .
بىكەتتە ئۇنىڭ قارىسى يۈتكىچە كەينىدىن قاراپ تۇردۇم شۇ چاغدا خۇددى بىر نەرسەم يۈتۈپ كەتكەندەك بولۇپ قالدىم .
ئۆيگە قايتىپ كىلىپمۇ ئانچە خوشال بولالمىدىم خىيالىمدا ئۇنىڭ دۇنياسىنى قىياس قىلاتتىم دەپ ئىلىش ۋاقتىمنىڭ چاققانراق توشۇپ قايتىشنى ھەم ئۇنىڭ تىپىشنى ئۈمۈد قىلاتتىم .
ئاخرى دەم ئىلىش ۋاقتىممۇ توشتى ، مەن قايتىپ كىلىپلا ئۇنىڭ ئىتىپ بەرگىنى بىلەن ئۇنىڭ دۇكىنىغا ئۇنى ئىزدەپ باردىم .

ئاخرى ئۇنىڭغا ئۆز خىيالىمنى ئىتىپ بىقىش قارارىغا كىلىپ ئۇنىڭغا تىلفۇن قىلدىم .

ئۇ ھىچقانداق ئىنكاس قايتۇرمىدى كۆڭلۈمدە ئالدىراقسانلىق قىلىپ ئىشنى بۇزۇپ قويغان ئوخشايمەن دەپ ئويلاپ مەنمۇ ئارتۇق گەپ قىلماي تاماقنى يەپ قايتىشتۇق ، بىركۈنۈم خۇددى چىۋىن يەپ قالغاندەك غەشلىك ئىچىدە ئۆتتى .
كەشتە ئۇنىڭدىن كەلگەن قىسقا ئۇچۇرنى تاپشۇرۇپ ئالدىم ئۇنى ئوقۇپ كۆزلىرىمگە ئىشەنمەي يىنىش يىنىشلاپ ئوقۇشقا باشلىدىم.

                                          مۇھەببەت ئىزھارى
ئەسسالاممۇ ئەلەيكۇم  ئاق كۆڭۈل ساپدىل دوستۇم ئىلھام سىزنىڭ بۈگۈنكى گەپلىرىڭىزنى ئاڭلاپ تولىمۇ خۇرسەن بولدۇم . مەنمۇ تۇنجى كۆرۈشكەندىن تارتىپ سىزنى يىقىن كۆرۈپ كەلدىم ئۆز ئارا پىكىرلىشىش ئارقىلىق ئۆزۈممۇ تۇيماي سىزگە ماھىل بولدۇم بۈگۈن ئۆز قەلبىمدە ساقلىنىپ كەلگەن سىرلارنى سىزنىڭ ئاغزىڭىزدىن ئاڭلاپ تولىمۇ خوشال بولدۇم رەھمەت سىزگە مەن بۇ ئىلتىماسنى قوبۇل قىلدىم

بۇلارنى ئوقۇپ تولىمۇ شادلاندىم بىر كىچىنى شىرىن ئارزۇلار بىلەن كىرپىك قاقماي ئۆتكۈزۈپ ئەتىسى ئۇنى ئىزدەپ باردىم . شۇكۈندىن ئىتبارەن بىزنىڭ مۇھەببەت ھاياتىمىز باشلاندى ئۆزۈمنى تولىمۇ بەختلىك ھىس قىلاتتىم ئامال بار ئۇنى خوشال قىلىشقا تىرىشاتتىم .
ئۇمۇ مىنى ئۆزىنىڭ بىرقىسمىدەك بىلەتتى .
شۇنداق قىلىپ بىرنەچچە ئاي كۈنلۈرۈم بەخت ئىچىدە ئوتتى كۈندە ئۇنى كۆرگىلى دۇكىنىغا باراتتىم مەن بارالماي قالسام ئۇ ئىش ئورنۇمغا ئىزدەپ كىلەتتى . بىر بىرىمىزگە تولىمۇ ئىچەكەشىپ كەتكەن ئىدۇق .

                                         بەربات بولغان مۇھەببەت
بۇگۈنمۇ ئۇنى كۆرۈش ئۈچۈن دۇكىنىغا باردىم لىكىن دۇكان تاقالغان ئىكەن . ئۇنىڭغا تىلىفۇن قىلدىم تىلفۇنىمۇ ئىتىۋىتىلىپتۇ .
يۈرۈگۈم خۇددى بىر شۇملۇقنى سەزگەندەك ئەنسىز سوقۋاتاتتى . ئۇنىڭغا توختىماي تىلفۇن قىلدىم . ئاخىرى تىلفۇن ئۇلاندى ،

ئۇنىڭ بىلەن  سۆھبەتلىشىش ئارقىلىق ئۇنىڭ ساداقەت بىلەن توي قىلغىنىغا تۆت يىل بولغانلىقىنى بىر ئوغلىنىڭ بارلىقىنى بىلدىم . بۇلارنى ئويلاپ ئۆزۈم شۇنچە ھايالىق پاك قىز دەپ بىلىپ چىن قەلبىمدىن سۆيۈپ بارلىق ئۈمۈدۈمنى ئاتىۋەتكەن ئۇ رەسۋاغا شۇنداق ئاچچىقىم كىلىۋاتاتىتى ھەم شۇ يولدىشىغا ھىسداشلىقىممۇ كىلىۋاتاتتى
جەميىتىمىزدىكى شەرمى ھايالىق قىزلىرىمىزنىڭ نامىغا داغ تەگكۈۋۈتقان ئۇ رەسۋانى تىتىپ ئاچچىقۇمنى چىقىرىۋالغۇم كەلدىيۇ ئۇنداق تىگى پەس بىلەن تەڭ بولۇپ ئاۋارە بولۇشنى خالىمىدىم .
ياتاققا قايتىپ ئۆتكەن ئىشلارنى ئەسلەشكە باشلىدىم ئۆزۈمنىڭ ئەقىلسىزلىقىمگە كۈلگۈم كەلسە ئۇنىڭ پەسكەشلىكىگە نەپرىتىم قوزغىلاتتى
ئاخىرى ئۇنىڭ ئەسلىمىلىرى بىلەن تولغان بۇ يۇرتتىن كېتىش قايتا ئاياق باسماسلىق قارارىغا كەلدىم .


                                            خاتىمە
بىر نەچچە كۈن قولۇمدىكى ئىشلارنى تۈگۈتۈپ ياتاقلارنى يىغىشتۇرۇپ قايتىشنىڭ تەييارلىقىنى قىلىۋاتقان ئىدىم ، كاللامدا پەقەتلا بىر پەسكەش ئىنسانغا بەرگەن چىن سۆيگۈمنىڭ خارلانغانلىقىدىن ئىبارەت ئاچچىق رىئاللىق داۋالغۇيتتى  ، تۇيۇقسىز تىلفۇنۇمنىڭ سايرىسى خىيالىمنى بۆلدى ، تىلفۇننى ئىلىپ كەلگەن خەتكە قارىدىم ئۇ ئەۋەتكەن ئىدى .

ئەسسالاممۇ ئەلەيكۇم يۈرگۈمدىكى ئۆچمەس ئوتقا ئىگە بولغان قەدىرلىك ئىلھام مەن بۇ خەتنى سىزدىن كەچۈرۈم سوراش ئۈچۈن ياكى ئۆزۈمنى چۈشەندۈرۈش ئۈچۈن يازمىدىم ئۈمۈدۈم پەقەتلا سىزگە بولغان مۇھەببىتىمنىڭ چىنلىقىنى بىلىپ قالسىكەن دىگەن .
مەن كىچىك تۇرۇپلا ئاتا -ئانامنىڭ زورلىشى بىلەن ئۆزۈممۇ چۈشەنمەيدىغان بىر كىشىگە ياتلىق بولدۇم ھەم ئۇنىڭ بىلەن تۆت يىل ۋاقىتنى خۇددى قەپەسدىكى قۇشتەك ئۆتكۈزدۈم كاشكى ئۇنىڭ ماڭا كۆڭلى بولغان بولسا بۇنچە ئازاپ تارتماس ئىدىم . ئەپسۇس ئۇنىڭمۇ ھەم مەنبىلەن پەرۋايى يوق .
شۇنداق ئازاپ ئىچىدە دەردىمنى ئىچىمگە يۇتۇپ يۈرگەن كۈنلەرنىڭ بىرىدە سىز بىلەن تونۇشۇپ قالدىم سىز ماڭا ئىللىقلىقنى ھىس قىلدۇردىڭز مۇھەببەتنى تونۇتتىڭىز ئۆزۈممۇ ئۇخماي سىزگە مەھلىيا بولدۇم . سىزنى يوقۇتۇپ قويغۇم بۇ ئازغىنە خوشاللىقتىن مەھرۇم بولغۇم كەلمىدى شۇ سەۋەپ ئۆزۈممۇ سەزمىگەن ھالدا بۇ يوللارنى باستىم  گەرچە بۇ قىلغانلىرىم ئەخمىقانىلىق  بولسىمۇ سىزگە بولغان مۇھەببەت ھەقىقى مۇھەببەت .مەن بۇ مۇھەببىتىمدىن ئۆكۈنمەيمەن ھەم نۇمۇس ھىس قىلمايمەن باشقىلارنى ياخشى كۆرۈش ھەرگىزمۇ خاتالىق ئەمەس .
مەن سىزنىڭ بۇئۈمرۈمدە پەقەت بىر سىزنىلا سۆيىمەن خەير ئامان بولۇڭ سىز خوشال بولسىڭىزلا مەن نىمىلا بولاي مەنمۇ ئوخشاشلا خوشال  ھەرجايدا خوشال خورام ئۆتىشىڭىزنى ئۇلۇغ اللادىن تىلەيمەن ئازغۇن ئاشىقىڭىز ساداقەت
