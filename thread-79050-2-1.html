قىز بالا دىگەن مۇشۇنداق كۆپىنچە ئىشلاردا ئالدىراڭغۇ كىلىدۇ ، كىين بولسا پۇشايمان قىلىشقا باشلايدۇ ، ئۇلۇغ خۇدامۇ ھەممە ئادەمگە شۇ ئادەمنىڭ نىيتىگە مۇناسىپ كىشنى ئۇچراشتۇرىدۇ ، شۇڭا ئۇماق قىزلارغا دەيدىغىم سىلەر ھېسىياتنىڭ كەينىگە كىرىپ ئەمەس ، بەلكى ئەقىل بىلەن ئىش قىلىڭلار ، ئۈزۈڭلەرگە ئۇچرىغان يىگىت ئەدەبلىك ، ئەخلاقلىق ۋە ۋىجدانلىق بولسلا ئۇنىڭ باشقا تەرەپلىرى  ( باي ياكى نامرات ) تەرەپلىرى بىلەن ھېسابلىشىپ كەتمەي ئۇنىڭ مۇھەببتىگە سادىق بۇلۇڭلار ، چۈنكى ئۇلۇغ خۇدا ئادەمگە ئارزۇلىغان ھەممە نەرسىسنى بىراقلا بەرمەيدۇ ، ئاز - ئازىدىن بىرىدۇ ، بىزدىكى گەپ بولسا بۇنى چۈشنىپ يىتپ شۈكرى قىلىشنى بىلىشتىن ئىبارەت . شۇندىلا سىلەر ئارزۇ قىلغان ياخشى تۇرمۇشمۇ سىلەرنىڭ سەۋىر قىلىشڭلار بىلەن رىئاللىققا ئايلىندۇ ، ئۇ كەمدە سىلەر مۇھەببەتتىنمۇ ۋە ياخشى تۇرمۇشتىنمۇ زىيان تارتىپ قالمايسلەر ، ئۇغۇلنىڭ بېشىغا ھەر قىسما دەرت كىلىدۇ ، بۇنى سىلەرگە ئوخشاش يېغلاپ يۈرۈپ ئۆتكۈزگىلى بولمايدۇ ، بۇنىڭ ئامالى بۇنداق ئەھۋالىدىمۇ بىزنى چۈشندىغان ھەم يەنىلا قوللاپ ئۆتەلەيدىغان ئايالنىڭ بۇلىشى بىز ئۈچۈن ئەڭ مۇھىم ، چۈنكى بىز ئۇغۇللار نىمىلا  ئىش قىلايلى ، بىز ياخشى كۆرگەن ئۇقىز بىزنىمۇ چۈشنىپ كەڭ قورساق بولالىسىلا  ، شۇندىلا بىزمۇ خۇدا نىسىپ قىلىپ تۇنۇشتۇرغان ھەم كۈيۈپ ئالغان  شۇخۇتۇننى بەختىلىك قىلىشنىڭ كويىدىلا يۈرىمىز . ئۇ قىزنىڭ بىزگە بۇنچلىك كەڭ قورساق بولغىنىنى مەڭگۈ ئۇنتۇپ قالمايمىز ، شۇ سەۋەپتىن ئۇ قىز قەلبىمىزنىڭ ئەڭ چۇڭقۇر يېرىدىن ئۇرۇن ئالالايدۇ ،  ھەم شۇنداقلا خۇتۇنىمىزنىڭ خوشاللىق ۋە جاپادىمۇ بىز بىلەن بىللە بۇلۇشنى ئۈمىد قىلىمىز ، ھەرگىزمۇ خوشال ۋاقىتلاردا كۈلۈپ يۈرۈپ ، زىيان چىققاندا تاشلاپ قاچىدىغان يۈزسىزلەرنى ياقتۇرمايمىز ، مەيلى ئۇ قىز قانچلىك گۈزەل بولسىۇن ، ۋىجدان ۋە ھايا  بولمىغانلىغى ئۈچۈن ئۇ بىز ئۈچۈن ئەرزىمەس بىر نەرسىگە ئايلىنىپ قالىدۇ ، شۇڭا بۇ ھېكايىگە نىسبەتەن شەخسەن دىمەكچى بولغىنىم بۇ قىز تېخى نىكاھدا تۇرۇقلۇق تالاغا قاراپ قاپتۇ ، بۇ ئۇغۇل ھەرقانچە ئەسكى بولىىسىمۇ ، قىزنىڭ بۇنداق قىلىشى ئۇقۇغان نامىزىغا ۋە كىيگەن كىيمىگە ماس كەلمەيدۇ ، بۇ ھەرگىزمۇ توغرا ئەمەس . شۇڭا بۇ ئۇغۇلنىڭ بۇ قىزنىڭ ئەھۋالىنى بىلگەندىن كىيىن ئاخىرىدا بۇنداق ئويلاپ قالىغىنىنى قىزلار توغرا چۈشىنشى كېرەك - دەپ ئويلايمەن .   ( دوسىت يېغلىتىپ ئېيتار ، دۈشمەن كۈلدۈرۈپ ) بۇ يازمىنى ئاخىرىدا   yollugh تەھرىرلىگەن. ۋاقتى  2012-5-23 11:56 PM
