قومۇل شەھىرى شەھەر ئايلانما جەنۇبى يولىدا يېڭىدىن قۇرۇلىدىغان مېزىلىك تائاملار كوچىسى،مېھمانخانا،سودا بازىرى ۋە ئائىلىكلەر رايۇنىنىڭ يەر نامى توغرىسىدا پىكىر ئېلىش ئۇقتۇرىشى

شەھەر قۇرۇلىشى بىر تۇتاش پىلانىغا ئاساسەن،قومۇل شەھىرى شەھەر ئىچى يېزىسىدىكى شەھەر ئايلانما يولىنىڭ جەنۇبى تەرىپىدە، شەھەر ئىچى يېزىلىق مەگەش كەنت ئاھالىلەر كومېتېتى ۋە غەربى ئۆستەڭ رايۇنى غەربى كۆكتاتچىلىق كەنت ئائىلىكلەر كومېتېتى ئەتراپىدىكى يەكشەنبە بازارنى مەركەز قىلىپ،مېزىلىك تائاملار كوچىسى،مېھمانخانا،سودا بازىرى ۋە كاپالەت ئولتۇراق رايۇن بىر گەۋدە قىلىنغان،قويۇق مىللى پۇراققا ئىگە رايۇن قۇرۇلىدۇ،جەمىيەتتىكى ھەر قايسى ساھەدىكىلەرنىڭ پىكرى-تەكلىپلىرىگە ئەستايىدىل قۇلاق سېلىش ئۈچۈن،«شىنجاڭ ئۇيغۇر ئاپتۇنۇم رايۇنىنىڭ يەر ناملىرىنى باشقۇرۇش چارىسى»،«قومۇل شەھرىنىڭ يەر ناملىرىنى باشقۇرۇش ۋاقىتلىق چارىسى»دىكى مۇناسىۋەتلىك بەلگىلىمىلەرگە ئاساسەن،«قومۇل شەھىرى شەھەر ئايلانما جەنۇبى يولىدىكى مېزىلىك تائاملار كوچىسى،مېھمانخانا،سودا بازىرى ۋە ئولتۇراق رايۇن بىر گەۋدە  قىلىنغان رايۇننىڭ ئۆلچەملىك يەر نامى»توغرىسىدا جەمىيەتتىن ئوچۇق-ئاشكارا ھالدا، يېڭىدىن يەر نامى قوبۇل قىلىنىدۇ.شۇڭلاشقا مۇناسىۋەتلىك ئىشلار تۆۋەندىكىدەك ئوقتۇرىلىدۇ:

1.يەر نامى قوبۇل قىلىشتىكى شەرتلەر:

1)  يەر نامى ئۆزگىچە ،ھەم يەرلىكنىڭ ھەقىقى ئەھۋالىنى ھەم دەۋر ئالاھىدىلىگىنى گەۋدىلەندۇرگەن بولۇشى كېرەك.

2) مىللى پۇراق كۈچلۈك،قومۇلنىڭ تەبىئى مەنزىرىسى،تارىخى بىرلەشتۈرۈلگەن ۋە مەدىنىيىتىگە ۋارىسلىق  قىلغان بولىشى كېرەك.

3) خەتلىرى ئۆلچەملىك،تەلەپپۇزى راۋان،چۈشىنىشلىك،ئەستە ساقلاشقا ئاسان بولىشى كېرەك.

4) دۆلەت ئىچى ۋە سىرتىدىكى قۇرۇلۇشلارنىڭ نامى قايتا ئىشلىتىلمىگەن بولىشى كېرەك.

5) كىشى ئىسىملىرى قوللانمىغان بولىشى كېرەك.

2.پىكىر ئېلىش 2012-يىل 6–ئاينىڭ 15-ئاياقلىشىدۇ،ماقالە ئەۋەتكۈچى يەر نامى قويۇشتىكى ئاساسىنى ۋە مەزمۇنىنى،ئۆزىنىڭ ئىسمىنى،خىزمەت ئورنىنى ۋە كەسپىنى ۋە تېلفۇن نومۇرىنى ئەسكەرتىشى كېرەك.

3.خەلق ئىشلىرى تارماقلىرى مۇتەخەسىسلەرنى تەشكىللەپ،تاللىنىپ چىققان ئۆلچەملىك يەر نامىغا 2000يۈەن نەق پۇل بېرىپ مۇكاپاتلايدۇ.

قومۇل شەھەرلىك خەلق ئىشلار ئىدارىسى

2012-يىل5-ئاينىڭ29-كۈنى

مەنبە: http://www.xjhm.gov.cn/10038/10038/00006/10000/2012/139535.htm


ئەسسالامۇ ئەلەيكۇم مۆھتىرەم قېرىنداشلار:

بۇ ئۇقتۇرۇشنى شەھەرلىك ھۆكۈمەت تۆنۈگۈن ھۆكۈمەت تورىدا تارقىتىپتىكەن،بۇگۈن كۆرۈپ دەرھال تەرجىمە قىلىپ تورغا يوللۇدۇم.مەن ئۆزەم كىچىكىمدىن مىللى مائارىپ تەربىيىسى ئالغان،قوش تىللىق مائارىپ تەربىيىسى ئالمىغانلىقىم ئۈچۈن تەرجىمە جەريانىدا خاتالىقلار بولىشى مۇمكىن تۈزىتىپ ئوقارسىلەر،لېكىن مەزمۇنىنىڭ قىلچىلىك ئۆزگىرىپ كەتمىگەنلىگىگە كاپالەت بېرىمەن.

بۇ ئۇقتۇرۇشنى شەھەرلىك ھۆكۈمەت ناھايىتى ياخشى يېزىپتۇ،ماڭا ئەڭ يارىغان يېرى بولسا ھەر ساھادىكىلەرنىڭ پىكىر- تەلەپلىرىگە قۇلاق سېلىش ئۈچۈن دەپتۇ.شۇڭلاشقا مەنمۇ بىر گىراژدان، قومۇل شەھىرىنىڭ شەھەر ئىچى يېزىسىنىڭ توپىسىدا ئۈنۈپ،توپىسىدا چوڭ بولغان بىر قانۇنلۇق ئاھالىسى بولۇش سۈپىتىم بىلەن ئۆزەمنىڭ پىكىر-تەلەپلىرىمنى ئوتتۇرىغان قويۇپ باقاي:

قومۇل شەھىرى شەھەرئىچى يېزىسى قومۇل شەھىرىنىڭ جەنۇپ تەرىپىدە،شەھەر مەركىزىدىن پەقەت 2 كىلومېتېر كېلىدۇ.شەھەر ئىچى يېزىسىنىڭ تارىخى ئۇزۇن بولۇپ،تارىختا قومۇل ئۇيغۇرلىرىنىڭ مەدىنىيەت،ئىقتىساد،ھاكىمىيەت مەركىزى بولغان.ئەينى يىللاردا ئۇيغۇر پادىشاھنىڭ ئوردىسى شەھەرئىچى يېزىسىدا بولغان،قومۇلدىكى بەش شەھەرنىڭ ئەڭ ئاۋاتلىرىدىن بىرى بولغان.

بۇگۈنكى كۈندىمۇ قومۇل شەھەر ئىچى يېزىسى قومۇل ئۇيغۇرلىرىنىڭ مەدىنىيەت مەركەزلىرىدىن بىرى،قومۇلغا كەلگەن ھەر قانداق بىر ساياھەتچى شەھەر ئىچى يېزىسىدىكى قومۇل ئۇيغۇر پادىشاھلىرىنىڭ مەقبەرىسىنى،شەرقى شىنجاڭدىكى ئەڭ چوڭ جامىئە قومۇل ھېيىتگاھ جامىئەسىنى،شاھ ئوردىسىنى(كېيىن ياسالغان)،مۇقام ۋارىسلىق مەركىزىنى،قومۇل مۇزىيىنى،شىنجاڭدىكى ئەڭ زامانىۋىي،ئەڭ چوڭ مۇسۇلمانلار قەبرىستانلىقىنى كۆرمەي كەتمەيدۇ.شەھەر ئىچى يېزىسىنىڭ ئالتۇنلۇق كەنتى ب د ت تەرىپىدىن قوغدىلىنىدىغان كەنت نام-شەرىپىگە ئېرىشكىنىنىمۇ ئاڭلىغان.بايىقى ئۇقتۇرۇشتا قەيت قىلىنغان يەكشەنبە بازار ھازىرقى كۈندە ھەر جۈمئە ۋە شەنبە-يەكشەنبە كۈنلىرى راسا قايناپ كېتىدىغان،ئىشىك-دەرىزىدىن تارتىپ يامداق-سۇپۇرگىكىچە،توخو-تۇخۇم،كەپتەردىن قوي-قوزىغىچە،موتۇتسىكىت-ۋېلسىپىتتىن سۆرىمە كەشكىچە،تېلۋىزور-ئۇنئالغۇدىن ئاددى تىڭشىغۇچقىچە،سەۋزە-تۇرۇپتىن ياڭاق-چىلان كۆچەتلىرىگىچە ھەممە نەرسە تېپىلىدىغان ئاجايىپ قايناق بازار.بۇ يەرگە بارسىڭىز يۇزلىرى قورۇق باسقان دېھقان دادىنىڭ سەي-كۆكتاتلىرىدىن تارتىپ خوتەنلىك تىۋېپ-ھەكىم قېرىنداشلىرىمىزنىڭ دورا-دەرمەكلىرىگىچە ھەممە نەرسىنى تاپالايسىز.مەنمۇ شەنبە-يەكشەنبە كۈنلىرى بىكار ۋاقتىمدا بېرىپ ئاۋاتچىلىقنى كۆرۈپ كېلىدىغان،ئۆيگە كېرەكلىك نەرسىلىرىمنىڭ سودىلىقىنىمۇ قىلىپ قويىدىغان ئادىتىم بار.لېكىن بۇ بازارنىڭ ئازراق كەمچىلىكى بار،بازار يول ياقىسىدا بولغاچقا يول بويىغا توختىتىپ قويۇلغان ماشنىلار قاتناشنىڭ راۋانلىقىغان،شەھەر تازىلىقىغا بەلگىلىك تەسىر كورسىتەتتى.شۇڭلاشقىمىكىن ھۆكۈمەت بۇ بازار ئەتراپىنى ئۆزگەرتىپ كۈچلۈك مىللى پۇراق شەرتى ئاستىدا مېزىلىك تائاملار كوچىسى،مېھمانخانا،سودا بازىرى،كاپالەت ئولتۇراق رايۇنى  قۇرۇشنى پىلانلاپتۇ.بۇ پىلاننى مەن شەخسەن ئەلۋەتتە قوللاپ قۇۋۋەتلەيمەن. بۇ يەر نامى توغرىسىدا خەلقتىن پىكىر ئالغىنى ناھايىتى توغرا بوپتۇ،شۇڭلاشقا مەن تۆۋەندىكىدەك پىكىر-تەكلىپ بېرىمەن.ھۆكۈمەت تەرەپنىڭ ئەستايىدىل ئويلىنىپ بېقىشىنى سورايمەن.

1.بۇ رايۇنغا نام قويۇش مەسلىسىدە:شەھەر ئىچى يېزىسىنىڭ ھەر قانداق بىر يېرىنىڭ ئۆز نامى بار،يەنە كېلىپ ئۇيغۇرچە.بۇ يەكشەنبە بازىرى بولىدىغان نەق ئورۇن ياۋاغ كۆۋرۈك دەپ ئاتىلىدۇ ۋە بۇ بازارنىڭ جەنۇپ تەرىپى سۇقوق(سۇ دەرۋازا)تەرەپكە قاراپ تارقىلىدۇ.شەرقى تەرىپى بولسا كۆتەمە،دورىلىق دەپ ئاتىلىدۇ،شىمال تەرىپى چاقسا،غەربى شەھەر ئىچى يېزىسى مەركىزىگىچە،ھەر قانداق قۇمۇللۇق ياۋاغ كۆۋرۈك دىسە دەرھال بىلىدۇ،بىلمەيدىغان ئادەم يوق.

ياۋاغ ئاتالغۇسى توغرىسىدا بۇگۈن 85ياشقا كىرگەن مومام بىلەن سۆھبەتلەشتىم.مومامنىڭ دىيىشىچە كىچىكىدە ئۆزىنى بىلگەندىن تارتىپلا بۇ يەرنىڭ نامى ياۋاغ ئىكەنلىكىنى ئېيتتى.شۇڭلاشقا ياۋاغ ئاتالغۇسىغا 100يىلدىن ۋە ياكى ئۇنىڭدىنمۇ ئۇزۇن بولغانلىغى ئېنىق تۇرۇپتۇ.ياۋاغ دىگەن ئاتالغۇ يارباغ دىگەن گەپ بولۇپ چاھارباغ،يار تۈگمەن مەھەلىلىرىدىكى ئىگىز يارنى ھەممىمىز بىلىمىز،شۇ يار قېشىدىكى باغ دىگەن مەنىدە بولۇپ،قومۇل ۋاڭلىرىنىڭ باغچىسى ئىكەن.شۇ يىللاردا بۇ كۆۋرۈكنىڭ ئاستىدىن شىلدىر-شىلدىر سۈزۈك سۇلار ئېقىپ تۇراتتىكەن،بۇگۈنكى كۈندىمۇ شۇنچىلىك يېشىللىق،بۇ يەردە« ئۇزۇمزار»،«ئۇيغۇرىيە» قاتارلىق داڭلىق ئارامگاھلار بار.شۇڭلاشقا بۇ يەرگە قۇرىلىدىغان ھەر قانداق بىر قۇرۇلۇش بۇ نامدىن ئايرىلسا بولمايدۇ.ھەمدە ھۆكۈمەتنىڭ تارىخى مەدىنىيەت بىرلەشتۈرۈلگەن دىگەن شەرتىگىمۇ،تەلەپپۇزى راۋان،چۈشىنىشلىك دىگەن شەرتىگىمۇ (ياۋاكې دەپ)دەپ-دەل چۈشىدۇ. مەسىلەن ياۋاغ سودا بازىرى،ياۋاغ مېزىلىك تائاملار كوچىسى ۋە ياۋاغ ئولتۇراق رايۇنى دىگەندەك..لېكىن ياۋاغ مېھمانخانىسى دىيىشكە تىلىم بارمىدى. ئەزىزانە قەشقەردىمۇ ياۋاغ دەيدىغان يەرنىڭ بارلىقى كۆرگەن،ھىچقىسى يوق بۇ ئىسىمنى قويساق(داۋاملىق ئىشلەتسەك) قەشقەرلىك قېرىنداشلار بىز بىلەن سېنىڭ مېنىڭ دەپ ھەرگىز تالاشمايدۇ.

2.بۇ يەردە قۇرۇلماقچى بولغان مېزىلىك تائاملار كوچىسى چوقۇم مىللى پۇراق قويۇق گەۋدىلەندۈرگەن بولىشى كېرەك.ئۇيغۇر مۇسۇلمانلىرىنىڭ يىمەك-ئىچمەك مەدىنىيىتىگە ھۆرمەت قىلىنىش كېرەك.بۇ كوچىغا ھەر قانداق غەيرىي مۇسۇلمان تائاملىرىنى ياكى ساختىلىق بىلەن «مۇسۇلمانچە»تاختىسىنى ئېلىۋالغان ئاشخانىلارنى ئاچقۇزماسلىق كېرەك.

3.بۇ رايۇندا ھازىرمۇ مىللى قۇرۇلۇشلار ئاز ئەمەس،بۇ قۇرۇلۇشلارنى ئىمكان بار ساقلاپ قېلىش ئاساسىدا ئۆزگەرتىلىشى كېرەك.بۇ رايۇندا يېشىللىق كۆپ،يېشىللىقنىمۇ ئىمكان بار قوغداش كېرەك..مۇھىت ئاسراشقا ئالاھىدە ئەھمىيەت بېرىش كېرەك.مۇھىتنى بۇزۇش بەدىلىگە كەلگەن ھەر قانداق زامانىۋىي بازار ياكى قۇرۇلۇش چەكلىنىشى كېرەك.

4.بۇ يەرگە قۇرۇلماقچى بولغان مېھمانخانا ھۆكۈمەت قارمىقىدىمۇ ياكى مەبلەغ سالغۇچىلارنىڭمۇ؟بۇ مېھمانخانىنىڭ ئاستىغىمۇ ھەر خىل KTV،تانسىخانا،مەيخانا،مۇنچىخانا قۇرۇلامدۇ؟ئەگەر شۇنداق بولىدىغان بولسا،ئەڭ ياخشىسى بۇ مېھمانخانىنى بۇ يەرگە قۇرمىساق،چۇنكى ئۇنداق بولغاندا كۈچلۈك مىللى پۇراق گەۋدىلەندۈرۈلمىگەن بولۇپ قالىدۇ ھەمدە بۇ رايۇننىڭ ئوبرازىغىمۇ تەسىر يەتكۈزىدۇ.شۇڭا بۇ مېھمانخانىغا نام قويۇشقا مەن ئاجىزلىق قىلىدىكەنمەن.

   مەن پەقەت مۇشۇنچىلىكلا پىكىر بېرەي،باشقىسىنى باشقا قېرىنداشلار بەرسۇن!قېرىنداشلارنىڭ پىكىر بەرگەندە ئىلمىي،سوغۇققان بولۇپ ھۆكۈمەتكە ھەمدەم بولىشىنى ئۈمىد قىلىمەن!ھەمدە يوقۇرقى ئالاقىلىشىش ئۇسۇللىرى ئارقىلىق ھۆكۈمەتكە ئىنكاس قىلىشىنى ئۈمىد قىلىمەن.

  مەن قومۇل ۋىلايەتلىك ئۇيغۇر تىبابەتتە ئۇيغۇر تىبابەت خادىمى،كەسپىم دوختۇر،شەھەر ئاھالىسى بولۇش بۇرچۇم بىلەن خالىس پىكىر بەردىم.مۇكاپاتلىمىساڭلارمۇ بولىدۇ.شۇڭلاشقا ئىسمىمنى،تېلفۇن نومۇرۇمنى قالدۇرمۇدۇم.
