بالىلار بايرىمى كىلىش ئالدىدا نۇرغۇنلىرىمىز ئارزۇلۇق ئوماق بالىلىرىمىزنىڭ يىلدا بىر كىلىدىغان بايرىمىنى ئەھمىيەتلىك ئۆتكۈزۈپ بىرىش ئۈچۈن باش قاتۇرۇشنى باشلىۋەتكەن بولىشىمىز مۇمكىن، مەنمۇ شۇنداق، قىزىمغا بەرگەن ۋەدە بويىچە خىزمەتنى قويۇپ شۇ كۈنىگە ئۈلگۈرۈپ ئۆيگە قايتماقچى. بارلىق سەبى بالىلارنىڭ بايرىمىنى مۇبارەكلەيمەن.
لىكىن شۇنىڭ بىلەن كۈلپەتلىك ئاچچىق تەغدىر شور پىشانىسىگە تۇغۇلۇشىدىلا پۈتۈلۈپ كەتكەن بىزنىڭ بالىلىرىمىزغا ئوخشاشلا ئوماق سەبى بىگۇنا بالىلار ئىسىمگە كىلىپ يۈرۈگۈم ئىزىلدى.
ئۇلار بولسىمۇ ئاتا ئانىسى ئەيدىز كىسىلى بىلەن كەينى كەينىدىن ۋاپات بولۇپ كىچىك تۇرۇپلا يىتىم قىلىۋاتقان، ياكى چوڭلارنىڭ سەۋەنلىگى تۈپەيلى تۇغۇلىشىدىنلا ئەيدىز ۋىروسى بىلەن يۇقۇملانغان بالىلاردۇر. تىخى كىچىك تۇرۇپ مۇدھىش كېسەلنىڭ ياكى يىتىملىكنىڭ ئازاۋىنى يەتكىچە تارتىۋاتقان، ئۇنى ئاز دىگەندەك قورقۇنۇچلۇق ۋابادەك مۇئامىلە قىلىنىپ جەمىيەت، خوشنا-خولۇم، مەكتەپ ھەتتا ئۇرۇق تۇققانلىرى تەرىپىدىنمۇ كەمسىتىلىپ رەھىمسىزلەرچە چەتكە قېقىلىۋاتقان، كىچىكلىگىدە كىچىك دوستلىرى بىلەن بىرەر قىتىم قىيغىتىپ كۈلۈپ ئويناپ باقمىغان، كىچىككىنە كېسەللەر بىلەن بىر كۈنلەردە تۇيۇقسزلا يىقىلىپ ئۆزلىرى ئۈچۈن يورۇق ئەمما مۇدھىش، سۆيۈملۈك ئەمما قورقۇنۇشلۇق،بىپايان ئەمما ناھايىتى تار دۇنيادىن تۇيۇقسىز خوشلىشىۋاتقان نارىسىدە بالىلاردۇر. سىلەرنىڭمۇ بالىلار بايرىمىڭلارغا مۇبارەك بولسۇن بالىلىرىم، ياراتقان ئىگەم ئارىمىزدىن ۋاقىتسىز كىتىپ قالغىنىڭلارنىڭ ئۇ دۇنيالىغىڭلارنى ئايىماي بەرگەي.
دۆلىتىمىزدە ئالدىنقى ئەسىرنىڭ 90-يىللىرىدىن باشلاپ ئەيدىز ۋىروسى بىلەن يۇقۇملانغۇچىلارنىڭ كىسەللىك باسقۇچىغا كىرىشى بىلەن ئەيدىز يىتىم بالىلىرىدىن ئىبارەت كىشىلەر توپى پەيدا بولۇپ ھازىرغىچە ئۈزلۈكسىز كۆپۈيۈپ 2010-يىلى 260 مىڭغا يىتىشى مۆلچەرلەنگەن. ئەيدىز كىسىلىنىڭ تەسىرىگە ئۇچرىغان بۇ بالىلار ئىنتايىن ئاجىز بولغان توپ بولۇپ يىتىم بولۇپلا قالماي (ئاتا-ئانىسى ياكى ئاتا ئانىسىنىڭ بىرى ئەيدىز كىسىلى بىلەن ئۆلگەن بالىلار) ئاتا-ئانىسىنىڭ ھىمايىسى كۈتۈشىدىن ئايرىلىش بىلەن تۇرمۇشتا يار يۆلەسىز قالغان، تولاراق يالغۇز ياشايدىغان، مەكتەپتىن چىكىندۈرۈلگەن ياكى مەكتەپ تەرىپىدىن رەت قىلىنغان، ئۆزى بىلەن تەڭ دىمەتلىك بالىلار بىلەن ئارىلىشىشى، بىللە ئوينىشى چەكلەنگەن، پىسخىكا تەرەپتىن كۆتۈرگۈسىز ئېغىر روھى ئازاپ ۋە كۆڭۈل يېرىمچىلىغىغا دۇچار بولغان، تۇرمۇش سۈپىتى ۋە بەخت كۆرسەتكۈچى ئىنتايىن تۆۋەن، جىسمانى ۋە روھى تەرەپلەردىكى يىتىلىشى زور دەرىجىدە توسقۇنلۇققا ئۇچراۋاتقان بالىلاردۇر. بۇلارنىڭ ئىچىدىكى تېخىمۇ بەختسىزلىرى يەنىلا تۇغۇلۇشىدا ئەيدىز ۋىروسى بىلەن يۇقۇملىنىپ تۇغۇلغانلاردۇر.
ھەر ھالدا شۇ گۇناھسىز نارىسىدىلەرنىڭ بەختىگە يىقىنقى يىللاردىن بۇيان پارتىيە- ھۆكۈمەت ئارقا-ئارقىدىن مۇناسىۋەتلىك سىياسەت- تەدبىرلەرنى تۈزۈپ چىقىپ پۈتۈن جەمىيەتنى بۇ بالىلارغا كۆيۈنۈش، كۆڭۈل بۆلۈشكە چاقىرىۋاتىدۇ ھەم بۇ توغرىلىق دۆلەت كۆپلەپ مەبلەغ ئاجرىتىپ ساناتورىيە قۇرۇش ۋە سۈپىتىنى ياخشىلاش،بالىلارنىڭ نورمال تەربىيىلىنىشى ئۈچۈن مەكتەپ تەربىيەسى ئېلىشىغا كاپالەتلىك قىلىش،ئەيدىز ۋىروسى بىلەن يۇقۇملانغان بالىلارغا قارىتا ھەقسىز داۋالاش ئىلىپ بېرىش قاتارلىق نۇرغۇن خىزمەتلەر ئىشلەنمەكتە.

گەپنى شىنجاڭ ئۇيغۇر ئاپتونۇم رايونىغا يۆتكىسەك، رايونىمىز بۇ يىلقى ئەيدىز تەكشۈرۈش دوكلاتىدا ئوتتۇرىغا قويۇلغان 67 مىڭ ئەيدىز بىمارى ۋە ئەيدىز ۋىروسىدىن يۇقۇملانغۇچى بىلەن دۆلىتىمىز بويىچە خېنەن ئۆلكىسىدىن قالسىلا ئىككىنچى ئورۇنغا ئوتتى. لىكىن ئاھالە نىسبىتىدىن ئېيتقاندا 25 مىليون نوپۇسقا ئىگە شىنجاڭ بۇ نىسبەتتە جۇڭگو بويىچە بىرىنچى ئورۇندا! يەنى جۇڭگودا ھازىر ئەيدىز كېسىلى ۋە ئەيدىز ۋىروسىدىن يۇقۇملانغۇچى700 مىڭ ئەتراپىدا بولۇپ دۆلەت ئومۇمى نوپۇسىنىڭ 0.05 پىرسەنتىنى ئىگىلەيدۇ دىسەك، شىنجاڭ ئۇيغۇر  ئاپتونۇم رايونىدا بۇ نىسبەت 0.27 پىرسەنت بولۇپ، دۆلىتىمىز ئۇتتۇرىچە سەۋىيەسىدىن 5-6 ھەسسە يۇقۇرى! يەنە بىر ئويلىنىشقا ئەرزىيدىغىنى شىنجاڭ رايونىدىكى ئەيدىز بىمارلىرىنىڭ مەنچە ئاز دىگەندە 80% ئاز سانلىق مىللەرلەر بولۇپ ئۇ چاغدا 10 مىليونچە ئاز سانلىق مىللەتكە توغرا كەلگەن تېخىمۇ چوڭ نىسبەت ئادەمنى چۆچۈتمەي قالمايدۇ.

مەسىلەن ئۈرۈمچىنى ئالايلى، 2011 -يىللىق ئۈرۈمچى ئەيدىز دوكلاتى مەلۇماتىدىن قارىغاندا بۇلتۇر31-دېكابىرغىچە، ئۈرۈمچى شەھرىنىڭ جەمئىي ئىستاتىستىكا قىلغان مەلۇماتىدىن مەلۇم بولىشىچە ئەيدىز ۋىرۇسىدىن يۇقۇملانغۇچىلار جەمئىي  10536 نەپەر، يىڭىدىن قېتىلغانلىرى 1312 نەپەر، ئېشىش نىسبىتى 12.45 پىرسەنت بولغان! قازا قىلغانلارنىڭ سانى 997 نەپەرئىكەن. يۇقۇملانغانلارنىڭ سانى شىنجاڭ بويىچە ئىككىنچى ئورۇندا تۇرىدىكەن. زەھەر چېكىش، جىنسىي مۇناسىۋەتتىن يۇقۇملىنىش تەڭلا مەۋجۇت بولۇپ، ئاساسلىق تارقىلىش يولى بولۇپ قالغان.
http://xj.people.com.cn/n/2012/0515/c186332-17038311.html

ئادەم دىگەن ئۆزىنى چاغلىۋالسا بولىدۇ.بولمىسا ھىلىغۇ ئون يىلدا بايقىلىدىغان ساقايماس ئەيدىز ئىكەن،ئەتىلا ئۆلتۈردىغان يۇقۇملۇق كىسەلگىمۇ دۇچار بولىدۇ كىشى.توۋا دەيلى دە.

ئۇيغۇر ئاپتۇنۇم رايۇنىمىزدا ئەيدىزنىڭ كىڭىيىش سۈرىتى چوڭ شەھەرلەردىن ناھىيە يىزا بازار كەنىتلەرگىچە قەدەم قۇيماقتا.
ئۇيغۇر ئاپتۇنۇم رايۇنىمىزدا ئەيدىزنىڭ يۇقۇش ئۇسسۇلىدا ئادەتتە زەھەرلىك چىكىملىك چىكىپ يۇقۇشتىن كۆرە جىنسى مۇناسىۋەتتىن يۇقۇش ئالاھىدە ئۇرۇندا تۇرىدۇ مىنىڭچە(.چۈنكى زەھەرلىك چىكىملىكنىڭ ئاپتۇنۇم رايۇنىمىزغا كىرىش يوللىرى ئىلگىرى خەلىقئارادا نامى داڭ كەتكەن ئالتۇن ھىلال ئاي رايۇنىنى(ئافىغانىستان پاكىستان چىگرىسى) مەركەز قىلغان ئافىغانىستان كۆكنارلىقىدىن ئافىغانسىتان پاكىستان چىگراسى ئارقىلىق پاكىستانغا،پاكىستاندىن جۇڭگۇنىڭ قەشقەر چىگرىسى ئارقىلىق ياكى خوتەن چىگراسى ئارقىلىق ئۇيغۇر ياكى باشقا مىللەت شوپۇرلىرىنىڭ سودىنى باش تىما قىلغان <<ئىكىسپورىت>>يوللىرى ئارقىلىق ئاپتۇنۇم رايۇنىمىزغا ،ئاندىن جۇڭگۇ چۇڭ قۇرۇقلۇقىغا كىرەتتى.كىيىن چىگرا تامۇجنىسى ساقچىلىرى بۇ ئىشنى قاتتىق تۇسقانلىقى ۋە زەربە بىرىشنىڭ كۈچلۈكلىكىدىن بۇ يوللار ئاساسلىق ئىتىلگەن،كىيىن يۈننەن بىرما چىگرىسىدىن ئالتۇن ئۈچ بۇرجەك رايۇنىدىكى زەھەرلىك چىكىملىك ياساش ئۇۋىسىدىن يەنە شۇ <<باتۇر،ئەقىللىق>>ئۇيغۇر ۋە باشقا مىللەتلەر <<چىگرا سودىسى>>نى جانلاندۇرۇپ يۈردى.كىيىن بۇ يوللارمۇ نۇرغۇن ۋاستىلەر ئارقىلىق ئىتىۋىتىلدى.لىكىن جەنۇبى شىنجاڭ رايۇنلىرىدىكى بەزى قىسمەن جايلاردا نىشە ،كەندىر قاتارلىق بۇيۇملارنىڭ ئۆستۈرلىۋاتقانلىقىنى ،قانۇننىڭ قازاسىغا ئۇچىراۋاتقانلىقىنى ئاڭلاپ تۇرىمىز. ھازىر تارقىلىۋاتقان ئەيدىزنىڭ شۇ زامانلاردىكى قالدۇقلىرى مانا بىر كۆرىۋاتقان ۋە ئاڭلاۋاتقان نىجىس كىسەلنىڭ ئاسارىتى.

ھازىرقى يىڭىدىن كۆپەيگەن بىمارلارنىڭ يۇقۇم ئەھۋالى ئاساسەن جىنسى يۇقۇمدىن داۋام قىلىۋاتىدۇ.كىشىلەرنىڭ ئاڭ سەۋىيەسىنىڭ ئۆسىشى سەۋەبلىك يىڭىچە كۆڭۈل ئىچىش يوللىرى كىڭەيمەكتە.مەيلى كارائوك خانىلاردىن مەيخانىلارغىچە،كوچىلاردا ئىپپىتىنى ساتقۇچىلاردىن ھەر قايسى ئورۇنلاردىكى خىزمەتچىلەرگىچە،دىھقان ئەمگەك كۈچلىرىدىن كوچىلاردىكى بىكارچىلارغىچە بۇ خىل كۆڭۈل ئىچىش ھىلىھەم داۋام قىلماقتا.قانۇن قانچە چىڭ بولغانچە ئۇيۇنچىلارنىڭ ئۇيۇنلىرىنىڭمۇ شەكلى شۇنچە ئۆزگەرمەكتە.

شۇ ئەيىپكارلار نىجىس كىسەلنى ساپمۇ ساق ساغلام كىشىگە، باشقىلارغا توچەي قىلغۇچە ئۆزىگە ئوخشاش كىشىلەرگە يۇقتۇرۇپ بۇ جاھاندىن بىراقلا قۇتۇلسا بولمامدىغاندۇ.تىخى شۇنداق كىشىلەرنىڭ قانۇنى ساپاسى يۇقىرى،دىنى ئىلىمى بىزدىنمۇ كۈچلۈكرەك كىلىدۇ.
