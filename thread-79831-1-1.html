تۈنۈگۈن ۋاقىت مۇناسىۋىتى بىلەن تىمامنى ئاياغلاشتۇرۇشقا مەجبۇر بولدۇم.تىمامغا نىسبەتەن نۇرغۇن تورداشلار ئۆز  لىرىنىڭ  قاراشلىرىنى ،مۇناسىۋەتلىك  ئۇچۇرلار نى ئايىماي دوستانىلىق بىلەن ئىنكاس شەكلىدە قالدۇرۇپتۇ.مەنمۇ   شۇ ئىنكاسلارنى كۆرۈش جەريانىدا رىئال تۇرمۇشتا بولۇۋاتقان ئىشلارنىڭ مېنىڭ ئويلىغىنىمدىنمۇ ئېغىر ھەم چىگىش ئىكەنلىكىنى ھىس قىلدىم.بەزى تورداشلار"  ئۇزۇن ئۆمۈر كۆرۈش بىلەن ھارام نەرسە ئىستىمال قىلىشنىڭ بىۋاستە  مۇناسىۋىتى بارمۇ؟"دېگەندەك ئىنكاسلارنى قالدۇرۇپتۇ.ئەمما مېنىڭ بۇ تىمىدا سۆزلىمەكچى بولغىنىم ئۇزۇن ئۆمۈر كۆرۈشنى تەرغىپ قىلىش ئەمەس .ھەربىر ئادەمنىڭ ئۆمرىنىڭ قانچىلىك بولىشى شۇ ئادەمنىڭ رىزىقى بىلەن مۇناسىۋەتلىك.مېنىڭ ئويلىغىنىم سالامەت ياشاش."ساغلام تەندە ساپ ئەقىل"دىگەندەك ماقال -  تەمسىللەرمۇ ساغلام ياشاشنىڭ مۇھىملىقىنى كۆرسىتىپ بەرگەن.چۈنكى ئادەم سالامەت بولغاندا ئاندىن نۇرمال ئىبادەت ئېلىپ    بارالايدۇ،قۇربان ھېيىتتەك بايراملارمۇ بايرامدەك ئۆتىدۇ.بۇلار ئەنە ئاشۇنداق ئەڭ ئەقەللىي قائىدىلەر .ئەمما سىز   بالىلارنىڭ ئاشۇ ئەرزان باھالىق ئاچچىقلارنى يىگىنىدە،قىز-چوكانلارنىڭ مالاچۈەر يىيىش ئۈچۈن ،راڭپىزا يىيىش  ئۈچۈن ئۆچرەت ساقلاپ تۇرغانلىقىنى كۆرگىنىڭىزدە مۇشۇلار ئېسىڭىزگە كىلەرمۇ؟ھازىر جەمىيەتتە سېنىڭ بىلەن مېنىڭ مېنىڭ نىمە ئېشىم دەيدىغاندەك ئىشلار ئەۋجىگە چىقتى.ئەلۋەتتە سىز ،مەلۇم ئادەملەرنى كۆرۈپ نەسىھەت قىلغۇڭىز كەلسىمۇ شۇنىڭ زىيانلىق ئىكەنلىكنى سەمىمىيلىك بىلەن ئەسكەرتىپ قويغىڭىز كەلسىمۇ ،يەنىلا شۇنداق قىلالمايسىز .چۈنكى سەۋەبى پەقەتلا بىر دەشنام يەپ قېلىشتىن ئەنسىرەيسىز ھەم مۇشۇ كىچىك  ئىشلار ئۈچۈن ئۇلار  بىلەن سۈركىلىشىپ قېلىشىمۇ ئوشۇق ئىش دەپ بىلىسىز.توغرا ،بۇ بىرئادەم ئۆزگەرتەلەيدىغان مەسىلە ئەمەس.بۇ بىر جەمىيەتنىڭ يېڭى دەۋىردىكى قانۇنىيىتى خالاس.ئەمما "قۇلغا ئات مىنسە چاپمىغان دۆڭى قالماپتۇ،دىدەك مۇنچاق ئاسسا كىرمىگەن ئۆيى" دىگەندەك ھەر قانداق ئىشنىڭ ئەسلىگە قايتىش ۋاقتى ھامان بولىدۇ.بۇيەردە بىر    ئىشنى مىسال ئالسام:ئىسلاھات ئىشىكنى ئېچىۋىتىش سىياسىتى يولغا قويۇلغاندىن كىيىن ئىقتىسادتا زور دەرىجىدە  تەرەققىيات بارلىققا كەلدى.كىشىلەرنىڭ مەدەنىيەت تۇرمۇشىمۇ موللىشىپ چەتئەلنىڭ مەدەنىيەتلىرىگە قىزىقىش     دولقۇنى كۆتۈرۈلدى.ئېنىقى توي تۆكۈننىلا مىسالغا ئالساق ئۇيغۇرلارنىڭ توي-تۆكۈن ئادەتلىرىدە ئۆزگىرىش بولدى.ئۇيغۇرچە ئۇسۇلنىڭ  ئورنىغا دىسكو ئويناش.دوپپا كەيمەي شىلەپە كىيىش،چوڭ زاللاردا ئۆتكۈزۈش دىگەندەك ئىشلار بۇرۇنقى ئادەتلەرنىڭ ئورنىنى ئالدى ،ئەمما ھازىرغا قارايدىغان بولساق ھەممىسى ئەسلگە قايىتتى.ئەمما يەنە يېڭى ئىشلار قېتىلغىلى تۇردى.مەسىلەن:تويدا ھاراق ئىچىش ،تاماق ئىسراپ قىلىش ،سۆلەتۋازلىق قىلىش دىگەنگە ئوخشاش.ئەمما ئۇ ئىشلارمۇ ۋاقىتنىڭ ئۆتىشگە ئەگىشىپ يەنىلا ئۆز ئوقىغا قايتىپ كىلىدۇ ئەلۋەتتە . بۇنىڭدا پەقەت ۋاقىت ئىلگىرى كىيىنلىك مۇناسىۋىتىلا مەۋجۇت.ھەقاچان سىزمۇ بىلىسىز،ھازىر شىنجاڭدىكى زەنجىرسىمان دۇككان ئارقىلىق تەرەققىي قىلىۋاتقان كەسىپنىڭ راڭپىزىچىلىق ئىكەنلىكىنى.ھازىر بەزى چەت يىزىلارغا بارسىڭىزمۇ مەلۇپ ماركىلىق راڭپىزىنىڭ 500-تارماق دۇكىنى دېگەندەك ۋېۋىسكا ئېسىقلىق تۇرغان دۇككانلارنىڭ بارلىقىنى كۆرىسىز.ئەمما بىرەر ئۇيغۇر تائام ئاشىخانىسىنىڭ ئۇنچىۋالا كۆپ تارماق قۇرغانلىقىنى ئەلمىساقتىن تارتىپ بىلمەيمىز.بۇنى ياخشى بولغان نۇقتىدىن كۆزەتسەك  ئۇلار جەمىيەتتە ئۆزلىكىدىن ئاتلىنىپ ئىگىلىك تىكلەپ مەيلى ئۆزىگە بولسۇن ياكى يەنە نۇرغۇن ئىشسىز ،ئىش     كۈتۈپ تۇرغانلارغىمۇ ئىش ئورنى يارىتىپ بېرىپ ئۇلارنىڭ نامراتلىقتىن قۇتۇلىشىغا ياردەم بەردى.ھەم زەنجىرسىمان دۇككان قۇرۇشتا چەتئەلنىڭ 50-60يىل بولغان ماركىلىرىغا جەڭ ئىلان قىلدى.ئۇلارنىڭ ياخشىلىقلىرىنى تىل بىلەن  ئېيتىپ تۈگەتكىلى بولمايدۇ.ئەمما بىر قارشى نۇقتىدىن تۇرۇپ تەھلىل قىلساق راڭپىزا-بۇ بىر تەبىئىتى سوغۇق ،ئەل- مىساقتىن تارتىپ ئۇيغۇر تائام غەزىنىسىدە يوق بولغان يىمەكلىكنىڭ تىز سۈئەتتە تەرەققىي قىلىش بوشلۇقىغا ئېرىشىشى مەلۇم خىل تەڭپۇڭسىزلىقلارنىمۇ كەلتۈرۈپ چىقىرارمۇ؟دىگەندەك ئويلارغا سالىدۇ.بەزى ئادەملەرنىڭ قا-رىشىچە شۇ راڭپىزا قاندا ماي ئېشىپ كەتكەنگە پايدا قىلارمىش.ئەگەردە ئىلمىي نۇقتىدىن تەھلىل قىلساق ماي-     مەيلى ئادەم بەدىنىدىكى ماي بولسۇن ياكى ئۆسۈملۈك مىيى بولسۇن،يۇقىرى تىمپۇراتۇرىدا ئاندىن پارچىلىنىپ       سەرىپ بولىدۇ.تۆۋەن تىمپۇراتۇرىدا ماينىڭ يىپىشقاقلىقى كۈچىيىپ تېخىمۇ ئۇيىۋالىدۇ.قاندا ماي ئېشىپ كەتكەنگە  بەلكىم راڭپىزىنىڭ ناھايىتى ئازدەرىجىدە پايدىسى  بولىشى مۇمكىن.(ئەلۋەتتە مەن بۇنى تەجرىبە تەتقىق قىلىپ باقمىغان)ئەمما بىزدەك تەبىئىيتى ئىسسىقلىقنى خالايدىغان قوۋم ئۈچۈن ئېيىتقاندا سوغۇقلۇق قىلىش سالامەتلىككە بەلگىلىك  يامان تەسىرلەرنى ئېلىپ كېلىدۇ دەپ قارايمەن. (ئەگەردە زىيان ئېلىپ كەلمەيدۇ دىسىڭىز نىمە ئۈچۈن بەدەنگە قاپارىتما چىقىدۇ؟يەنىلا ئاددىي ئۇقۇم)
  يۇقىرىقى بايانلار ئارقىلىق مېنىڭ مەركىزىي ئىدىيەمنى ھىس قىلغانسىلەر ،ھىس قىلالمىغان بولساڭلارمۇ مەيلى       تۆۋەندىكى مەزمۇنلارنى ئوقۇساڭلارمۇ ئوخشاشلا مەركىزىي ئىدىيىنى ھىس قىلالايسىلەر .
  باشتىكى تىمامغا بېرىلگەن ئىنكاسلار ئىچىدىن يەنە بىر مەزمۇن دىققىتىمنى تارتتى.مەلۇم بىر ناۋايخانىنىڭ ئادەتتىكى تۇزنى قىممەت كۆرۈپ سانائەت تۇزى ئىشلەتكەنلىكى.بەلكىم بۇ ئىككىلىسى "تۇز"بولغاندىكىن قايسىسىنى ئىشلەتسە بولۇۋەرمەمدۇ دىيىشىڭىز مۇمكىن.مېنىڭچە شۇ ناۋايمۇ خەقلەر بىر ئاق رەڭلىق كىرىستالنى      تۇز دەپ ساتسا ئەرزانكەنغۇ دەپلا سېتىۋالغاندۇ.ئەمما بۇ ئىككى تۇزنىڭ ئاسمان زىمىن پەرقى بار .سانائەت تۇزى   
ئەگەردە بەلگىلىك ئۆلچەمدىن ئېشىپ ئىستىمال قىلىنىپ قالسا ئادەمنىڭ ھاياتىغا خەۋىپ ئېلىپ كېلىدۇ.چۈن-
كى بۇنى مانا http://baike.baidu.com/view/77489.htm    ۋەhttp://baike.baidu.com/view/415733.htm
بۇ ئىككى يەردىن كۆرۈپ بېقىڭ.سانائەت تۇزى ساپ خىمىيە سانائىتىگە ئىشلىتىلىدىغان نەرسە.مېنىڭچە ئۇيغۇر       نانلىرىنىڭ يەنىلا شۇ  پۇل ئۈچۈن بۇلغۇنۇپ كېتىشى بىز ھەممەيلەنگە نىسبەتەن بىرخىل كاللىمىزنى سەگەتمىسەك  
بولمايدىغان، يىمەكلىككە بولغان قارىشىمىزنى ئۆزگەرىتمىسەك بولمايدىغان باسقۇچقا يېتىپ كەلگەنلىكىمىزنىڭ  بىشارىتى. .....
(بۈگۈنمۇ ۋاقىت مۇناسىۋىتى بىلەن تىمامنى ئاياغلاشتۇرۇشقا مەجبۇر بولۇپ قالدىم.قىزىققۇچىلار بولسا ئەتىكى مۇنازىرىمىزگە دىققەت قىلغايسىلەر. ھەم كەڭ ئىشتىراكچىلارنىڭ ئۆزىنىڭ قاراشلىرىنى  ئېغىر كۆرمەي قالدۇرىشىنى  ئۈمىد قىلىمەن)
