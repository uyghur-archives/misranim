مۇھەببەت دېگەن نېمە؟ نېمە ئۈچۈن مەن مۇھەببتىن شۇقەدەر قورقىدىغاندىمەن؟ نېمە ئۈچۈن مۇھەببەت ماڭا چىدىغۇسىز ئازابتەك تۇيىلىدۇ؟


سىز مەندىن مۇھەببەت دېگەن نېمە؟ دەپ سورايسىز.


مۇھەببەت شۇنداق نەرسىكى، ئۇ ئادەمنى بىر پۈتۈنلۈككە قوشۇلۇشقا دالالەت قىلىدىغان بىر خىل چوڭقۇر ئىچىكى كۈچتۇر. بۇ خىل ئىچكى كۈچ سىز بىلەن مېنى بىر گەۋدە قىلىپ يۇغۇرىۋېتىدۇ. مۇھەببەت ئۆزىمىز بىلەن ئەسلىمىزنىڭ ئايرىلىشى، بۇ خىل ئايرىلىشتىن ئاشۇ بىر پۈتۈنلۈككە قايتىش ئىستىكى تۇغىلىدۇ – دە، ئىستەك ئۇنىڭ بىلەن بىر گەۋدە بولۇپ يۇغۇرۇلۇپ كېتىدۇ.


مۇبادا سىز بىر تۈپ دەرەخنى يىلتىزى بىلەن قومۇرۇپ تاشلىسىڭىز، ئۇنىڭدا، يىلتىزىم تۇپراققا قايتىپ بارسىكەن، دېگەن كۈچلۈك بىر ئىستەك پەيدا بولىدۇ، چۈنكى ئۇنىڭ ھەقىقىي ھايات مەنبەسى شۇ يەردە، تۇپراقتىن ئايرىلسا، ئۇ مەۋجۇت بولۇپ تۇرالمايدۇ، ئۆلىدۇ. ئۇ چوقۇم توپراقتا مەۋجۇت بولۇپ تۇرىشى، ئۇنىڭ قوينىدا ياشىشى كېرەك، مۇھەببەت دېگەن ئەنە شۇ.



سىزدىكى ئۆزلۈك سىز بىلەن سىزنىڭ تۇپرىقىڭىزدىن ئىبارەت بىر پۈتۈنلۈك ئوتتۇرىسىدا بىر توسالغۇغا ئايلىنىدۇ، ئادەم بوغۇلغان چاغدا نەپەسمۇ ئالالمايدۇ، دەرەخ يىلتىزىدىن ئايرىلىپ قالغاندا، ئوزۇقلىنىشىمۇ ئۈزۈلۈپ قالىدۇ، مۇھەببەت دېمەك، ئوزۇقلۇققا بولغان تەلپۈنۈش ھەم مەۋجۇتلۇق ئىچىدە يىلتىزغا ئېرىشىش دېمەكتۇر.



مۇبادا سىز قارمۇ قارشى قۇتۇپقا تاشلانسىڭىز، بۇ بىر خىل نورماللىقتۇر، ئۇ، ئەرنىڭ ئايالنى جەلىپ قىلىشىدىنلا ئىبارەت بولىدۇ. ئەرلەر ئاياللار ئارقىلىق ئۆزىنىڭ تۇپرىقىنى بايقايدۇ. ئاياللار ئارقىلىق تۇپراقلىشىدۇ. ئاياللارمۇ ئەرلەر ئارقىلىق تۇپراقلىشىدۇ، ئۇلار بىر – بىرىنى تولۇقلايدۇ. يىگانە ئەرنىڭ ئۆزى پەقەت يېرىم نەرسە، ئۇ بىر پۈتۈنلۇككە ئايلىنىشنى ئىنتايىن ئارزۇ قىلىدۇ، يېگانە ئايالمۇ پەقەت يېرىم ھېسابلىنىدۇ. بۇ ئىككى يېرىملار بىرلەشكەندىلا، ئادەم تۇنجى قېتىم يىلتىزىغا جان كىرگەنلىكىنى، يىلتىزىنىڭ تۇپراققا چوڭقۇرلاپ كىرگەنلىكىنى ھېس قىلىدۇ، قەلبىدىن ئالەمشۇمۇل خوشاللىق بالقىپ چىقىدۇ.
بۇ پەقەت ئاياللارغا يىلتىز سۈرۈش بولۇپلا قالماستىن، بەلكى ئاياللار ئارقىلىق ئىلاھقا يىلتىزلىنىش بولۇپ ھىسابلىنىدۇ. ئاياللار پەقەت ئىشىكنىڭ بىر قانىتى، ئەرلەر بىلەن ئاياللار ئىلاھلىققا ئۆتىدىغان ئىشىك، مۇھەببەتكە تەلپۈنۈش خۇداغا تەلپۈنۈش دېمەكتۇر. سىز ئۇنى چۈشىنىشىڭىزمۇ، چۈشەنمەسلىكىڭىزمۇ مۇمكىن، لېكىن مۇھەببەتكە بولغان تەلپۈنۈش ھەقىقەتەن خۇدانىڭ مەۋجۇت ئىكەنلىكىنى ئىسپاتلاپ بىرەلەيدۇ. چۈنكى، كىشىلەرنىڭ قەلبىدە مۇھەببەت بولغانلىقى ئۈچۈنلا ، خۇدا مەۋجۇت بولغان، كىشىلەر مۇھەببەتسىز ياشىيالمايدۇ، شۇڭا خۇدا مەۋجۇتتۇر.
مۇھەببەتنىڭ قۇدرىتى شۇنىڭدىن ئىبارەتكى، بىز يېگانە قالغان چاغلىرىمىزدا ئازابلىنىمىز، ھالاك بولىمىز، بىرگە بولغان چاغلىرىمىزدا يېتىلىمىز، كۈچ – قۇۋۋەتكە تولىمىز، ئۆزىمىزدە مۇكەممەللىك ۋە قانائەت ھاسىل قىلىمىز.



سىز مەندىن «مۇھەببەت دېگەن نېمە؟ مەن نېمىشقا مۇھەببەتتىن شۇ قەدەر قورقىدىغاندىمەن؟» دەپ سورايسىز.
چۈنكى سىز ئاياللارغا ئىچكىرلەپ كىرگەن شۇ دەقىقىدە، ئۆزىڭىزنى يوقىتىپ قويىسىز، ئاياللارمۇ ئەرلەرگە ئىچكىرلەپ كىرگەندە ئۆزىنى يوقىتىپ قويىدۇ.



ھازىر چۈشىنىۋىلىش لازىم بولغان مۇنداق بىر نۇقتا بار: سىز پەقەت ئۆزىڭىزنى يوقاتقان چاغدىلا ، بىر پۈتۈنلۈككە يىلتىز سۈرۈپ كىرەلەيسىز، سىزنى بىر پۈتۈنلۈككە مەھلىيا قىلىدىغان بۇنىڭدىن باشقا ئۇسۇل يوق. سىز ئۆزىڭىزدە كۈچ – قۇدرەتنىڭ يوقلىقىنى ھېس قىلغىنىڭىز ئۈچۈن، بىر پۈتۈنلۈك ئىچىدە غايىپ بولغان شۇ دەقىقتە قاتتىق قورقۇشقا باشلايسىز، سىز ئۆزىڭىزنى يوقىتىپ قويىسىز – دە، قاتتىق ۋەھىمە باسىدۇ، سىز ئۆزىڭىزنى تارتماقچى بولسىڭىز، مانا بۇ ئادەمنى دىلىغۇل قىلىپ قويىدىغان بىر پەللە، ھەر بىر ئادەم ئۇنى كۈتىۋېلىشى، ئۇنى بېشىدىن كەچۈرىشى، ئۇنى چۈشىنىشى ۋە ئۇنىڭدىن ھالقىپ ئۆتىشى لازىم.
سىز چوقۇم ئاشۇ ئىككى ئىشنىڭ ئوخشاش بىر ئىشتىن كېلىپ چىقىدىغانلىقىنى بىلىشىڭىز كېرەك، سىز ئۆزىڭىزنىڭ غايىپ بولىشىنى چوقۇم ئىنتايىن راھەت ھېس قىلىسىز، سىزدە غەم قىلىش، تىت-تىت بولۇش ۋە مەسئۇلىيەت بولمايدۇ، سىز خۇددى كاۋچۇك دەرىخى ۋە يۇلتۇزلارغا ئوخشاش ئالەمنىڭ بىر قىسمىغا ئايلىنىپ كېتىسىز، بۇ نەقەدەر قالتىس ئىش – ھە! ئۇ بىر ئىشكنى، سىزنىڭ مەۋجۇتلۇقىڭىزغا ئىچكىرلەپ كىرىدىغان سىرلىق ئىشىكنى ئاچىدۇ، ئۇ ئېسىل شېئىرلارنى بارلىققا كەلتۈرىدۇ، بۇ ئاجايىپ رومانتىكا بولماي نېمە! لېكىن، ئەمەلىيەتتە سىز ئۇنىڭغا ئىچكىرلەپ كىرگەندە، سىزنى ۋەھىمە باسىدۇ، سىز «مەن ئەمدى غايىپ بولىمەن، كېيىن قانداق ئىشلار يۈز بېرەر؟» دېگەن ئەندىشىدە بولىسىز.



بىر دەريا قۇملۇققا ئىچكىرلەپ كىرىپ، قۇملۇقنىڭ پىچىرلاشلىرىنى ئاڭلىدى، دەيلى...شۇ ئەسنادا ئۇ يەنە ئىككىلىنىدۇ، ئۇ قۇملۇقتىن ھالىقىپ ئۆتۈپ، دېڭىزنى ئىزدەپ بارماقچى بولىدۇ، ئۇنىڭ قەلبىدە بىر خىل ئارزۇ، بىر خىل سىرلىق ھەم كەسكىن تۇيغۇ پەيدا بولىدۇ – دە، «مېنىڭ تەقدىرىم ھالقىشتىن ئىبارەت» دېگەن ئىشەنچ باش كۆتۈرىدۇ. گەرچە ئۇنىڭغا يېتەرلىك ئاساس كەمچىل بولسىمۇ، لېكىن «بۇ مېنىڭ ئاخىرقى نىشانىم ئەمەس، مەن چوقۇم تېخىمۇ بۈيۈك نەرسىلەرنى ئىزدىشىم كېرەك» دېگەن بىر قۇدرەتلىك ئىچكى كۈچ ئۇنىڭ قەلب چوڭقۇرلىقىدا : «سەن دادىللىق بىلەن سىناپ كۆر، بۇ قۇملۇقتىن ھالقىپ ئۆت!» دەپ چۇقان سالىدۇ.



قۇملۇق ئۇنىڭغا: «سۈزۈمگە قۇلاق سال. بىردىنبىر چارە شۇكى، سەن پارغا ئايلىنىپ شاماللارغا قوشۇلۇپ كەت، شاماللار سېنى ئېلىپ قۇملۇقلاردىن ئۇچۇپ ئۆتىدۇ» دەيدۇ. دەريا قۇملۇقلاردىن ھالقىپ ئۆتۈش خىيالى بولسىمۇ، لېكىن ئىنتايىن تەبئىي بولغان مۇنداق بىر مەسلىنى ئويلانماي قالمايدۇ: «شامالنىڭ مېنى يەنە قايتىدىن دەريا قىلىپ شەكىللەندۈرەلىشىگە كىم كاپالەت بېرەلەيدۇ؟ مۇبادا مەن غايىپ بولسام، ئۆزۈممۇ بىر نېمە دېيەلمەيمەن، مەن يەنە ئەسلىدىكى شەكلىم، ئەسلىدىكى ئىسمىم ۋە ئەسلىدىكى تېنىم بىلەن دەريا ھالىتىگە قايتالارمەنمۇ؟ كىم بىلىدۇ دەيسىز؟ مەن شامالغا قوشۇلۇپ كەتسەم، ئۇلار مېنىڭ يەنە ئۆزىدىن ئايرىلىشىمنى راۋا كۆرەرمۇ؟» مانا بۇ دەل مۇھەببەت ۋەسۋەسىسى دېمەكتۇر.



ئېسىڭىزدە بولسۇن ھەم ئىشىنىڭكى، مۇھەببەت بولمايدىكەن خۇشاللىق بولمايدۇ، مۇھەببەت بولمايدىكەن ھاياتلىقمۇ بولمايدۇ، مۇھەببەت بولمايدىكەن سىز نامەلۇم بىر نەرسىگە ئىچىرقاش بىلەن تەلپۈنىسىز، نېمىنىدۇر بىر نەرسىنىڭ كەملىك قىلىۋاتقىنىنى، روھىڭىزنىڭ بوش قالغانلىقىنى ھېس قىلىسىز، ئۆزىڭىزدە ھېچ نەرسە قالمىغاندەك قۇپقۇرۇق ھەم بوش ھېس قىلىسىز، مەنىسىزلىك ۋە ئازابلىنىش سىزنى چىرمىۋالىدۇ، يەنە تۇرۇپ ئۆزىڭىزنى قانائەتكە ئىرىشتۈرىدىغان نۇرغۇن ئۇسۇللارنىڭ بارلىقىغىمۇ ئىشىنىسىز.
لېكىن سىز مۇھەببەتكە يېقىنلىشىدىغان چاغدا بولسا غايەت زور ۋەھىمە ۋە گۇمان سىزنى قىيناشقا باشلايدۇ، مۇبادا سىز ئۆزىڭىزنى بىمالال تۇتۇپ، مۇھەببەت قوينىغا كىرەلىسىڭىز، ئۇنداقتا سىز يەنە ئەسلىڭىزگە قايتالامسىز؟ ئۆز سالاھىيىتىڭىزنى ۋە خاراكتېرىڭىزنى قوغداپ قالالامسىز؟ مۇشۇنداق تەۋەككۇل قىلىشقا ئەرزىمدۇ؟ ئىدراكىڭىز بۇنداق تەۋەككۇلچىلىقنى قىلما، دەيدۇ، چۈنكى سىز ئوزۇقلىنىش كەمچىل بولغان، ئوزۇقى بولمىغان، ئېچىرقىغان، ئازابلانغان بىر مەۋجۇت مەخلۇق. ئەگەر سىز قانداقتۇر بىر مۇھەببەت ئىچىدە غايىپ بولسىڭىز، سىزنى ھېچكىم تەسەۋۋۇر قىلالمايدۇ، سىز غايىپ بولغان يەردە خوشاللىق، بەخت ۋە خۇدانىڭ بولىدىغانلىقىغا ھېچكىم كاپالەتلىك قىلالمايدۇ.



بىر تال ئۇرۇقتىمۇ ئوخشاشلا شۇنداق ۋەھىمە پەيدا بولىدۇ، ئۇ ئۆلۈپ تۇپراق ئاستىغا كۆمۈلگەن چاغدا، ئاشۇ ھاياتلىقنىڭ ئۆلۈمدىن كېلىدىغانلىقىنى ئۇرۇق ئۆزى ھەرگىزمۇ تەسەۋۋۇر قىلالمايدۇ.
مۇھەببەت _ ھاياتىڭىزنىڭ نازاكەتلىك ئۇسۇلى.
ھايات بىر خىل پۇرسەت، ئۇ مۇھەببەتنىڭ قىزىل گۈلىنى ئېچىلدۇرىدىغان تۇپراق.
مۇھەببەتنىڭ ئۆزى ئاجايىپ قىممەتكە ئىگە، ئۇنىڭدا مەقسەت بولمايدۇ، غەرەز بولمايدۇ، ئۇنىڭ ئاجايىپ خۇشاللىقى ۋە لەززىتى بولىدۇ، لېكىن بۇ ئۇنىڭ مۇددىئاسى ئەمەس.
مۇھەببەت ھامان مەلۇم بىر خىل ئەسەبىيلىك بولىدۇ.
ئەسەبىيلىك دېگەن نېمە؟ سىز ئۆزىڭىزنىڭ مەلۇم نەرسىنى نېمە ئۈچۈن ياخشى كۆرگەنلىكىڭىزنى ئىسپاتلاپ بېرەلمەيسىز، مانا بۇ ئەسەبىيلىكتۇر.



كۈندىلىك تۇرمۇشىڭىزدىكى ھەرقانداق بىر ئىش مەلۇم مەقسەت ئاستىدا بولىدۇكى، مۇھەببەت ئۈچۈن بىرەر سەۋەب ئويلاپ تاپالمايسىز، سىز پەقەت «بىلمەيمەن، مەن پەقەت كىشىلىك ھاياتتىكى ئەڭ گۈزەل دەقىقىلەرنى بېشىمدىن كەچۈردۈم» دېيەلەيسىز. لېكىن ئۇنى ھەرگىزمۇ مۇددىئا دېگىلى بولمايدۇ.
مۇھەببەت ھاياتىڭىزنىڭ نازاكەتلىك ئۇسۇلىدۇر.
شۇڭا، مۇھەببەتنىڭ نېمىلىكىنى چۈشۈنمىگەن كىشىلەر ھاياتىدىكى نازاكەتلىك ئۇسسۇللارنى قولدىن بېرىپ قويدى، قىزىلگۈلنى يېتىلدۈرىدىغان پۇرسەتلەردىن مەھرۇم بولدى. كۆرۈپ تۇرۇپتىمىزكى، ئادەتتىكىچە تەپەككۇر قىلىدىغان، قىلنى قىرىققا يېرىپ ھېسابات قىلىدىغان، ھېسابلاش ماشىنىسىغا ئوخشاش ئويلايدىغان كىشىلەر، شۇنداقلا، ماتېماتىكلار، ئىقتىسادشۇناسلار ۋە سىياسىيونلار ئۈچۈن ئېيتقاندا، مۇھەببەت بىر خىل تەلۋىلىكتەك بىلىنىدۇ.



لېكىن، مۇھەببەتنى چۈشىنىدىغان كىشىلەر ئۈچۈن بولسا ئۇ بىردىنبىر ئاقىلانىلىكتۇر، قەلبىدە مۇھەببەت بولمىغان بىر كىشى بەلكىم باي بولۇشى ۋە نام – ئابرۇيلۇق بولۇشى مۇمكىن، لېكىن ئۇنىڭ ھېسسىيات ۋە ئىدراكىنى مۇكەممەل دېگىلى بولمايدۇ. چۈنكى ئۇ ھەرقانداق بىر ئىشنىڭ ئىچكى قىممىتىنى بىلىپ يېتەلمەيدۇ. ئاقىلانىلىك سىزنىڭ قەلبىڭىزدە ئېچىلغان قىزىلگۈلنىڭ خۇشبۇي پۇرىقىدىن ئىبارەتكى، ئۇنىڭدىن باشقا ھېچقانداق نەرسە ئەمەس.



بىر-بىرىنى سۆيگەن كىشىلەرنى روھىي جەھەتتىن داۋالاش ھاجەتسىز، ئەمەلىيەتتە مۇھەببەت كىشىلەرگە ساغلاملىق ئاتا قىلىدىغان ئەڭ ئۇلۇغ كۈچتۇر. مۇھەببەتتىن مەھرۇم بولغان كىشىلەر نېمىلەردۇر كەملىك قېلىۋاتقاندەك بىمەنىلىك ئىچىدە ياشايدۇ. ئادەتتىكى ئەسەبىيلىكتە ئەقلىيلىك بولمايدۇ، بىراق، مۇھەببەت ئەسەبىيلىكىگە مەلۇم بىر خىل ئەقلىيلىك يوشۇرۇنغان بولىدۇ. ئەقلىيلىك دېگەن نېمە؟ ئۇ ھاياتىڭىزنى ناخشىغا ئايلاندۇرىدىغان، سىزگە خوشاللىق ۋە زور ھىممەتلەرنى ئېلىپ كېلىدىغان نەرسە.
سىز كىشىلەرنى كۆزىتىپ باققان بولغىيتىڭىز؟ مەلۇم بىر ئادەم مۇھەببەت سەۋدالىقىغا چۈشكەندە، ئۇ بۇنى جاكارلاپ ئولتۇرمىسىمۇ، سىز ئۇنىڭ كۆزلىرىدىن بۇرۇنقىغا ئوخشىمايدىغان بىر خىل چوڭقۇرلۇقنى، چىرايىدىن يېڭىچە ئىپادە ۋە گۈزەللىكنى بايقىۋالالايسىز. ئۇنىڭ يول مېڭىشلىرىدىنمۇ باشقىچە سېھرىي يەڭگىللىك چىقىپ تۇرىدۇ. ئۇ ھەم ئەسلىدىكى ئۆزىگە ئوخشايدۇ، ھەم ئەسلىدىكى ئۆزى ئەمەس، چۇنكى مۇھەببەت ئۇنىڭ ھاياتىغا كىرىپ كەلگەندە، قەلبىنى باھار نۇرلىرى ئىللىتىدۇ، كۆڭۈل بېغىدا چىرايلىق گۈللەر ئېچىلىدۇ.
مۇھەببەت ئادەمنى قاسراق تاشلىتىپ قايتىدىن يارىتىدۇ.



مۇھەببەتنى بىلمىگەن كىشىلەرنى ئاقىل كىشىلەر دېگىلى بولمايدۇ، ئۇلارنىڭ ھاياتى تىراگېدىيە بىلەنلا ئۆتىدۇكى، ھەرگىزمۇ لاتاپەت ۋە گۈزەللىككە يار بولالمايدۇ.



بارلىق دىنىي ئۇستازلارنى بىزنى مۇنداق ئاگاھلاندۇرىدۇ:«سېنىڭ ھاياتىڭ خۇددى كۆپۈككە ئوخشاش بىھۇدە نەرسىدىن ئىبارەت، ئۇ ھازىرلا كۆزۈڭڭە كۆرۈنۈپ تۇرسا، ئەتىسىلا غايىپ بولىدۇ، ئۇ، بۇ دۇنيادا سېنىڭ ۋۇجۇدۇڭدا ھېچقانداق قىممەتكە ئىگە ئەمەس. چۈنكى ئۇ ۋاقتىنچە ساڭا مەنسۇپ، ئۇنىڭ بىردىنبىر رولى شۇكى، سەن ئۇنى ئىنكار قىلىپ، ئۇنىڭدىن ۋاز كېچەلەيسەن – دە، خۇدانىڭ نەزىرىدىكى كەرەملىك ئادەم بولالايسەن.»



بۇ ھەقىقەتەن كىشىنى ئەجەبلەندۇرىدىغان بىر خىل قاراش بولسىمۇ، لېكىن ئۇ نەچچە ئەسىر كىشىلەرنىڭ ئىددىيىسىگە ھۆكۈمرانلىق قىلدى، ھېچكىم ئۇنىڭغا قارشى پىكىر قىلالمىدى، بولۇپمۇ شەرقلىقلەر دۇنيا بىر خىيالىي نەرسە دەپ قارىدى. نېمىشقا چۈنكى، دۇنيا ئۆزگىرىپ تۇرىدۇ، ئۆزگىرىپ تۇرىدىغان ھەرقانداق ئىشنىڭ ئەھمىيىتى، قىممىتى بولمايدۇ، باشتىن- ئاخىر ئۆزگەرمەيدىغان، مەڭگۈلۈك بولغان نەرسىلەرلا ئەڭ مۇھىم نەرسە بولالايدۇ، سىز بۇ دۇنيادا مەڭگۈ ئۆزگەرمەس بىر خىل تۇرىدىغان نەرسىنى تاپالمايسىز، دەپ قارىدى.



ئۇلارنىڭ مەقسىتى يەنىلا، بۇ دۇنيا خىيالىي نەرسە، مەڭگۈلۈك ئەمەس، دېگەننى تەكىتلەشتىن ئىبارەت. ئەبەدىي بولمىغان نەرسىنى چۆرۈپ تاشلاش، بۇ دۇنيادىكى بارلىق دىننىڭ پوزىتسىيىسى بولسا كېرەك.
ئۆزگىرىش مەڭگۈ ئۆزگەرمەيدۇ، ئۇنىڭدىن باشقا ھەرقانداق ئىش ئۆزگىرىپ تۇرىدۇ، پەقەت بىز ئۆز خىيالىمىز بويىچە خۇدا بولۇپ ئۆزگىرەلمەيمىز، خالاس. چۈنكى ئۇ دۇنيادىكى بىردىنبىر

مەڭگۈلۈكتۇر. بىز خۇدادىن باشقا مەڭگۈلۈكتىن بىشارەت بېرىدىغان ھېچقانداق بىر نەرسىنى تاپالمايمىز.



ھېراكلىت:«ھايات خۇددى ئېقىۋاتقان دەرياغا ئوخشايدۇ، ئېسىڭىزدە بولسۇنكى، سىز بىر دەريادىن ھەرگىزمۇ ئىككى قېتىم كېچىپ ئۆتەلمەيسىز» دېگەنىدى.
ئۆزگىرىۋاتقان تۇرمۇشقا ئەڭ يېقىن تۇرغان كىشىلەرلا بەختنى ئەڭ كۆپ ھېس قىلالايدىغان كىشىلەردۇر. ئۇلار ھەتتا سوپۇن ماغزاپلىرىنىڭ قۇياش نۇرىدا ھەسەن – ھۈسەندەك چاقناشلىرىنىمۇ ياخشى كۆرەلەيدۇ. ئۇنىڭدىن بەخت ھېس قىلالايدۇ.



سىزنىڭ ئاشۇ ئەۋلىيالىرىڭىز ئازاب ۋە كۈلپەتتىن باشقىنى بىلمەيدۇ، سىز ئۇلارنىڭ قېتىپ، تاشقا ئايلىنىپ كەتكەن چىرايى ۋە ۋۇجۇدىدىن ھاياتلىق ئىزنالىرىنى كۆرەلمەيسىز، ئۇلاردا ھېچقانداق ئۆزگىرىش بولمايدۇ، ئۇلارنىڭ ھاياتى بىر خىل رەسىم – مۇراسىمدەكلا ئۆتىدۇ، ئۇلار ئۆزگىرىۋاتقان بارلىق شەيئىلەرنىڭ كۈشەندىسى بولىدۇ.



نېمە ئۈچۈن خوشاللىق ئەيبلىنىدۇ؟ نېمە ئۈچۈن مۇھەببەتمۇ ئەيبلىنىدۇ؟ چۈنكى ئۇلار ئۆزگىرىپ تۇرىدۇ. نېمە ئۈچۈن بۇ دىنلار مۇھەببەت دۇنياسىدا نىكاھ دېگەن نەرسىنى پەيدا قىلدى، دەمسىز؟ ئۇلار قانۇن ئارقىلىق، كېلىشىم ئارقىلىق، جەمئىيەت ئارقىلىق، شۇنداقلا بالىلىرىمىز نېمە ئىش قىلىپ قويار دېگەن ئەندىشە ئارقىلىق نىكاھقا ھېچ بولمىغاندا خىيالىي بىر ئەبەدىيلىكنى ئاتا قىلماقچى، ئۇنى مەڭگۈلۈك بىر نەرسىگە ئايلاندۇرماقچى بولۇشتى. ئەپسۇسكى، نىكاھ ئۆزىنىڭ ئۆزگىرىدىغانلىقىنى، ھەرگىزمۇ مەڭگۈلۈك نەرسە ئەمەس ئىكەنلىكىنى نامايەن قىلغاندىن كېيىن بارلىق كونىچە دىنلار ئۇنىڭغا قارشى چىقتى.
نەچچە مىڭ يىللاردىن بۇيان ناھايتى كىچىك بالىلارنىڭمۇ تويىنى قىلىپ قويىدىغان، ھەتتا تېخى تۇغۇلمىغان، ئانىسىنىڭ قورسىقىدىكى بالىلارنىمۇ، بىرى ئوغۇل، بىرى قىز بولۇپ قالسا، تويىنى قىلىپ قويىمىز، دەپ تۇغۇلماستىنلا بېشىنى چېتىپ قويىدىغان ئىشلار بولۇپ كەلدى.



ھازىر ھىندىستاندا يەتتە – سەككىز ياشلىق بالىلارنىڭ تويىنى قىلىپ قويىدۇ، بۇ ئىش گەرچە قانۇنغا خىلاپ بولسىمۇ، لېكىن ئۆرپ – ئادەتكە مۇخالىپ كەلمەيدۇ. نېمە ئۈچۈن توي قىلىشنىڭ نېمىلىكىنى چۈشەنمەيدىغان، قانداق ئىش يۈزبەرگەنلىكىنىمۇ بالىلارنىڭ شۇنچە ئالدىراپ تويىنى قىلىدۇ؟ چۈنكى ئۇلار، بالىلار ئادەم بولۇپ يېتىلىشتىن بۇرۇن، قەلبىدە مۇھەببەت پەيدا بولۇشتىن بۇرۇن تويىنى قېلىۋىتىش كېرەك، دەپ ھېسابلايدۇ. شۇنداق قىلىپ، بالىلارنىڭ قەلبىدە مۇھەببەت يېتىلگەن چاغدا بولسا، بىرى ئەرنىڭ ئىلكىدە، بىرى خوتۇنلۇق بولۇپ بولغان بولىدۇ. دېمەك، يېتىلمىگەن بالىلارنىڭ تويىنى قىلىپ قويىدىغان بۇ ئىللەتنىڭ پۈتۈن دۇنيادا يامرىشى ئاخىرقى ھېسابتا مۇھەببەتنى گۇمران قىلىدۇ.



دۇنيادا نىكاھ كەلتۈرۈپ چىقارغان ئازاب – ئوقىبەت ھەرقانداق بىر ئىش كەلتۈرۈپ چىقارغان ئازاب – ئوقىبەتتىن كۆپ بولسا كېرەك. بۇنى ھەرگىزمۇ توغرا كېلىپ قىلىش دېگىلى بولمايدۇ، چۈنكى ئۇ بەختنىڭ بىردىنبىر مۇمكىنچىلىكىنى ۋەيران قىلدى، كىشىلەر مۇھەببەتنىڭ پەيدا بولىشىنى يوققا چىقاردى، كىشىلەر مۇھەببەتنىڭ نېمىلىكىنى چۈشەنمەي، قەلبىدىن تەنتەنە ئۇسۇللىرىغا ئورۇن بىرەلمەي، شۇنداقلا ياشاپ ئۆلۈپ كېتىدۇ...



سوپۇن كۆپۈكلىرىگە، كىپىنەكلەرگە، مەيىن شامالدا لەرزان تەۋرىنىپ ئۇسسۇل ئويناۋاتقان قىزىلگۈل غۇنچىلىرىغا نەزەر سالسىڭىز، ھاياجاندا ۋۇجۇدىڭىز لەرزىگە كېلىدۇ، خۇشاللىق ياشلىرىڭىز تۆكۈلىدۇ، كۈيلەپ ناخشا ئېيتقىڭىز كېلىدۇ، ھاياتىڭىز شۇقەدەر جۇش ئۇرىۋاتقاندەك بىلىنىدۇ. لېكىن ئۇ ئەبەدىي داۋاملاشمايدۇ، چۈنكى، پەقەت ئۆلگەن نەرسىلەرلا ئەبەدىي بولىشى مۇمكىن...



سىزنىڭ ئىلاھىڭىز ئۇسۇل ئويناشنى بىلەمدۇ؟ مۇھەببەتنى بىلەمدۇ؟ ناخشا ئېيتىشنى بىلەمدۇ؟ سىزنىڭ ئىلاھىڭىز كېپىنەكلەرنى قوغلاپ، ياۋا چېچەكلەرنى تېرىپ كېلەلەمدۇ؟ ئۇلاردىن ھوزۇرلىنالامدۇ؟ كۆز يېشى قىلالامدۇ؟ ناخشا ئېيتالامدۇ؟ پەقەت مۇشۇ ئىشلارغا قادىر بولالىغان ئىلاھلا ھاياتلىقنىڭ ھەقىقىي ۋەكىلى، ھاياتلىقنىڭ ھەقىقىي ئۆزى بولالايدۇ، خالاس.
