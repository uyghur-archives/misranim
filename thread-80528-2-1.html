مەن ئىنكاس يازمايلا جاۋابىنى كۆرىۋالدىمغۇ ؟؟؟؟؟؟  يا  مەن يۇشۇرۇن تىمىلارنى كۆرەلەيدىغان دەرىجىگە ئۆستۇممۇ ؟؟؟


       
zokra، ئەگەر بۇ تىمىدىكى يوشۇرۇن مەزمۇننى كۆرمەكچى بولسىڭىزئىنكاس يېزىڭ

كىشلەر نەرسە كېرەك سىتۋالغاندا كۆپ ھاللاردا ئىنىق نىشانى بۇلىدۇ،ئەگەر سىز ناھايتى زېرىككەن ۋاقتىڭىزدا بازارغا چىقىپ ئايلانسىڭىز ئۆيەگە قايتقىچەقۈرۇق قول قايتىشنى بىيئەپ ھىس قىلىپ بىر نەرسە سىتۋىلىشنى قارار قىلسىڭىز، ئۇنداقتا  تۇۋەندىكىلەردىن قايسسىنى سىتۋالسىز؟؟

ئا؛كىتاپ خانىغا بېرىپ كىتاپ سىتۋىلىش، بۇنۇڭدا زېرىكىشلىك ۋاقىتنى ئۈتكىزگىلى بۇلىدۇ،

ب؛چىرايلىق كىيىم سىتۋىلىش ،بۇنۇڭدا ئەمىلىي ئىشلەتكىلى بۇلىدۇ،

س؛مېيۋە تەبىئىيلا ئەڭ ياخشى تاللاش، بۇنۇڭدا ئۆيدە تۇگەپ قالغاندا يەنە بىر قىتىم مىڭىشنىڭ ئالدىنى ئىلىشقا بۇلىدۇ،

د؛غەرىپچە بولكىلاردىن ئازراق سىتۋالغان ياخشى ،ھەم كۆركەم ھەم تەملىك،بىر ۋاخلىق تاماق ئىتىشتىن قۇتۇلۇپ قالسىڭىز ئەجەپ ئەمەس.




ئا؛مۇھەببەتكە بولغان تەلىۋىڭىز ناھايتى يۇ قىرى ئىكەن،ئەگەر قارشى تەرەپنىڭ جەلىپ قىلىش كۇچى بولمىسا ھەمدە سىز ئارزو قىلغاندىك رومانتىك تۇرمۇشنى ھازىرلاپ بېرەلمىسە،ئۇنداقتا رىشتىڭلار قۇشۇلسىمۇ رىسقىڭلار قۇشۇلماسلىق ئېھتىماللىقى كۆپرەك،سىز ئالغان ما ئارىپ تەر بىيىسى بىر قەدەر يۇقىرى،شۇڭا تۇر مۇش سۇپىتىگە بولغان تەلىپىڭىزمۇ ئادەتتىكى ئادەملەرنىڭكىگە ئوخشىمايدۇ،مەنىلىك بولغاندىن سىرتئىسىل نەپىس بۇلىشى كېرەك،شۇڭا بۇنى ئورىندىيا لايدىغان كىشلەر كۆپ بولماسلىقى مۇمكىن،ئىسىڭىزدە بولسۇن،بەك قۇسۇرچى بولغاندا نۇرغۇن پۇر سەتلەردىن مەھروم قىلىشىڭىز مۇمكىن، ۋاقىت ئۈتۈپ كىتىۋېرىدۇ شۇڭا سەل ئەمىلىيرەك بولغان ياخشى.



ب؛مۇھەببەت دىڭىزىدا ئۇزىۋاتقان سىزنىڭ دائىم ئىككىلنىپ يۇرۇيدىغانلىقىڭىزنى ياخشى ھەمراھنىڭ زادى قەيەردىلىكىنى ئۇقالماي قالىدىغانلىقىڭىزنى بىلدۇرىدۇ،سۆيگەندە ھىچنىمىگە قارماي سۈيىسىز ھەتتا ئۇچۇنچى شەخىس دىگەن ئاتاققا قىلىشمۇ سىز ئۈچۈن بەرىبىر،لىكىن ئەپسوس لىنارلىقى شۇكى بۇنداق قىزغىنلىق ئۇزۇن داۋام قىلمايدۇ ،ئۇزۇن ئۇتمەيلا ئۆزتاللىشىڭىزنى خاتادەك ھىس قىلىسىز , كەينىڭىزگەيىنىپ يەنە باشقىنى ئىزلەيسىز،مۇھەببەتتە ئالا كۇڭۇل بولغان سىز گەرچە ئۆز سەزگۈڭىزگە ئەھمىيەت بەر سىڭىزمۇ لىكىن دائىم ئۇنى ئاڭقىرالماي قالىسىز،شۇڭا دائىم مۇقۇم بىرسىگە كۇڭۇل قۇيالمايسىز ،مىنىڭ سىزگە بېر دىغان تەكلىۋىم؛يەنىلا جىمجىت ئولتۇرۇپ ئوبدان ئويلۇنۇپ كۆرۈپ نىشانىڭىزنى ئايدىڭلاشتۇرۇپ پۇتۇن كۈچىڭىز بىلەن ھۇجۇمغا ئۇتسىڭىز ئاندىن كۇڭۇلدىكىدەك مۇھەببەتكە ئېرىشەلەيسىز.

zokra، ئەگەر بۇ تىمىدىكى يوشۇرۇن مەزمۇننى كۆرمەكچى بولسىڭىزئىنكاس يېزىڭ

كىشلەر نەرسە كېرەك سىتۋالغاندا كۆپ ھاللاردا ئىنىق نىشانى بۇلىدۇ،ئەگەر سىز ناھايتى زېرىككەن ۋاقتىڭىزدا بازارغا چىقىپ ئايلانسىڭىز ئۆيەگە قايتقىچەقۈرۇق قول قايتىشنى بىيئەپ ھىس قىلىپ بىر نەرسە سىتۋىلىشنى قارار قىلسىڭىز، ئۇنداقتا  تۇۋەندىكىلەردىن قايسسىنى سىتۋالسىز؟؟

ئا؛كىتاپ خانىغا بېرىپ كىتاپ سىتۋىلىش، بۇنۇڭدا زېرىكىشلىك ۋاقىتنى ئۈتكىزگىلى بۇلىدۇ،

ب؛چىرايلىق كىيىم سىتۋىلىش ،بۇنۇڭدا ئەمىلىي ئىشلەتكىلى بۇلىدۇ،

س؛مېيۋە تەبىئىيلا ئەڭ ياخشى تاللاش، بۇنۇڭدا ئۆيدە تۇگەپ قالغاندا يەنە بىر قىتىم مىڭىشنىڭ ئالدىنى ئىلىشقا بۇلىدۇ،

د؛غەرىپچە بولكىلاردىن ئازراق سىتۋالغان ياخشى ،ھەم كۆركەم ھەم تەملىك،بىر ۋاخلىق تاماق ئىتىشتىن قۇتۇلۇپ قالسىڭىز ئەجەپ ئەمەس.




ئا؛مۇھەببەتكە بولغان تەلىۋىڭىز ناھايتى يۇ قىرى ئىكەن،ئەگەر قارشى تەرەپنىڭ جەلىپ قىلىش كۇچى بولمىسا ھەمدە سىز ئارزو قىلغاندىك رومانتىك تۇرمۇشنى ھازىرلاپ بېرەلمىسە،ئۇنداقتا رىشتىڭلار قۇشۇلسىمۇ رىسقىڭلار قۇشۇلماسلىق ئېھتىماللىقى كۆپرەك،سىز ئالغان ما ئارىپ تەر بىيىسى بىر قەدەر يۇقىرى،شۇڭا تۇر مۇش سۇپىتىگە بولغان تەلىپىڭىزمۇ ئادەتتىكى ئادەملەرنىڭكىگە ئوخشىمايدۇ،مەنىلىك بولغاندىن سىرتئىسىل نەپىس بۇلىشى كېرەك،شۇڭا بۇنى ئورىندىيا لايدىغان كىشلەر كۆپ بولماسلىقى مۇمكىن،ئىسىڭىزدە بولسۇن،بەك قۇسۇرچى بولغاندا نۇرغۇن پۇر سەتلەردىن مەھروم قىلىشىڭىز مۇمكىن، ۋاقىت ئۈتۈپ كىتىۋېرىدۇ شۇڭا سەل ئەمىلىيرەك بولغان ياخشى.



ب؛مۇھەببەت دىڭىزىدا ئۇزىۋاتقان سىزنىڭ دائىم ئىككىلنىپ يۇرۇيدىغانلىقىڭىزنى ياخشى ھەمراھنىڭ زادى قەيەردىلىكىنى ئۇقالماي قالىدىغانلىقىڭىزنى بىلدۇرىدۇ،سۆيگەندە ھىچنىمىگە قارماي سۈيىسىز ھەتتا ئۇچۇنچى شەخىس دىگەن ئاتاققا قىلىشمۇ سىز ئۈچۈن بەرىبىر،لىكىن ئەپسوس لىنارلىقى شۇكى بۇنداق قىزغىنلىق ئۇزۇن داۋام قىلمايدۇ ،ئۇزۇن ئۇتمەيلا ئۆزتاللىشىڭىزنى خاتادەك ھىس قىلىسىز , كەينىڭىزگەيىنىپ يەنە باشقىنى ئىزلەيسىز،مۇھەببەتتە ئالا كۇڭۇل بولغان سىز گەرچە ئۆز سەزگۈڭىزگە ئەھمىيەت بەر سىڭىزمۇ لىكىن دائىم ئۇنى ئاڭقىرالماي قالىسىز،شۇڭا دائىم مۇقۇم بىرسىگە كۇڭۇل قۇيالمايسىز ،مىنىڭ سىزگە بېر دىغان تەكلىۋىم؛يەنىلا جىمجىت ئولتۇرۇپ ئوبدان ئويلۇنۇپ كۆرۈپ نىشانىڭىزنى ئايدىڭلاشتۇرۇپ پۇتۇن كۈچىڭىز بىلەن ھۇجۇمغا ئۇتسىڭىز ئاندىن كۇڭۇلدىكىدەك مۇھەببەتكە ئېرىشەلەيسىز.
