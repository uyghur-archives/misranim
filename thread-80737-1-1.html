تۆت تادۇنىڭ تۇپراق ئوت سۇ ھاۋا دىن ئىبارەت ماددىي دۇنيانى تۇزگەن تۆت ئاناسۇر، تۆت ئاساسىي تۇزگۇچى ئىكەنلىكىنى كۆپۇنچىمىز بىلىمىز.جانلىقنىڭ ئۆسىشى ئۇچۇن ئوزۇقلۇق تۇپراق، ئىسىقلىق ئوت، نەملىك سۇ، ئوكسىگىن ھاۋا ناھايتى زۇرۇل.بىرى كام بولسا جانلىقنىڭ ھاياتى كاپالەتكە ئىگە بولالمايدۇ. ئادەتتە تىلىمىزدا تۆت پەسىل،شەرىق غەرىپ تۆت تەرەپ،قازاننىڭ قۇلىقى تۆت دىگەندەك سۆزلەر دائىم ئۇچىراپ تۇردۇ، بىردىن سانىساق ئۇچنىڭ كەينىدىن  تۆت كىلىدۇ.  ئاندىن، تادۇ دىگەنگە كەلسەك؛ ماددىلارنى ۋە جانلىقلارنى تۇزگەن تۆت چوڭ ئانا سۇر ماددا تادۇ دەپ ئاتىلىشتىن سىرىت ، قىلىچ داتلاشسا كەسمەس،سارىت تاتلاشسا ئەشمەس‹ماخمۇد قەشقىرى›بۇيىچە، تۆمۇرنىڭ دىتى تۇپراق ئەسلى ماددىغا قايتىشىنى ياكى ئەسلى خۇسيىتىنى يوقۇتۇپ تاتلاپ قىلىشىنى،شۇنداقلا قۇتادغۇ بىلىگىدىكى تاد بۇيىچە، تۆت تادۇغا سىلىپ تەپەككۇر بىلەن تاتىلاشنى تادۇ دەپ قىياس قىلىشقىمۇ بۇلىدۇ.كۇندىلىك ئىستىمالىمىزدا تات ئۆلچەش، تارتىش. تاتلىق تىتىش، تاپ تىپىش،تۇت دىگەندەك ئىبارىلەرمۇ ئۇچراپ تۇرىدۇ.

تۆت تادۇ، ئۇيغۇر تىبابىتىنىڭ نەزىريەۋىي ئاساسىي بۇلۇپ، ئۇزاق تارىختىن بىرى ئەجداتلىرىمىز دۇنيانى بىلىش سەپرىدە ئىزچىل ئىشلىتىپ،ئەۋلاتتىن ئەۋلاتقا ئۇلاپ،دۇنيا مەدىنىيەت خەزىنىسىگە ئۆچمەس تۆھپىلەرنى قوشقان. ئۇيغۇر تىبابىتى بۇيىچە ،ئىنسانلار سەۋدا سەپرا بەلغەم خۇن دەپ تۆت خىلىتتىن تۇزلىدۇ. خىلىتلارنىڭ تەڭپۇڭسىزلىقى كىسەل مەنبىيى دەپ قارىلىپ، خىلىتلار تەڭپۇڭلاشتۇرۇش ئارقىلىق كىسەللەرگە داۋا قىلىنىدۇ. ئادەتتە چوڭلاردىن ياكى دىھقانلار ئارسىدا ئىسسىق سۇغۇق، ئوتنى سۇ بىلەن ئۆچەرگۇلۇك، تۇپراقنى قاماللاپ تۇتقىلى بولغان بىلەن، سۇنى سىقسا ساقىيدۇدەپ ئاڭلاپ قالىمىز.بەزىلىرىمىز ئىش پائالىيەتلىرىمىزدە شەرىئەت ئىستىسەڭ ئۆلتۇر يىلاننى دەپ، قىلنى قىرىق يىرىشقا ، كەلگەن يەردە كىسىپ مىڭىشقا ئادەتلەنگەن.بەزىلىرىمىز تەرىقەت ئىستىسەڭ قىينىما جاننى بۇيىچە، قۇلىمىزدىن كىلىشىچە باشقىلارغا ياخشىلىق قىلىمىز.ئادىللىق بۇيىچىمۇ مەخسەتكە يەتكىلى بۇلىدۇ، ياخشىلىق قىلىپمۇ كۆڭۇللەرنى باغلىغىلى مەنزىلگە يەتكىلى بۇلىدۇ.دىمەك ئادەمدىكى تۆت خىل مىجاز خۇي پەيىللەرنىڭ،ياشاش سەپىرى پائالىيتى ھەتتا تەغدىرگىچە باغلىنىشى بۇلىدۇ.

قۇتادغۇبىلىگدە؛ ئايتولدى  بەخىت سائادەت بىلىشنىڭ ئاچقۇچى سۇپەتتە تۇپراق خۇسيەتتىكى سەۋدانى ناھايتى ياخشى سۇرەتلىگەن بۇلۇپ،توپتەك دومىلايدۇ،ئۆزگۇرۇشچان،يىڭىلىق ئىستەيدۇ.دىمىسىمۇ بۇ خىل كىشلەر ئاسان تىتىت بۇلىدۇ،بىر ئىشنى تۇتۇپ قىلمايدۇ،غىلدىغۇراز تۆت تامدا يورغىلايدۇ،چاتاق تىرىپلا يۇرىيدۇ.لىكىن ئەقىللىق كىلىدۇ.كۇنتۇغدى ئاچچىقى يامان ئوت خۇسيەتتىكى سەپرا ، ھۇقۇق  قانۇن  ئادىللىقنى دەبدەبە قىلىدۇ، ئوگدۇلمىش ئاداش ئايماق تىرىپ  ۋەزىر سۇپەتتە ياخشىلىقنى ئىگىرى سۇرىدۇ، سۇ خۇسيەتتە بەلغەم. ئودغۇرمىش دۇنيانى تەرىك ئىتىپ، تائەت ئىبادىتى بىلەن ھاۋا قان خىلىتتىكى تەقۋادارلىقنى ئىپادىلەيدۇ.ئەمىلىيەتتە بۇيەردە دىيىلگىنى تار دائىردىكى بىر شەخىسنىلا كۆرسەتمەستىن،بەلكى بىر جەمىيەت كارتىنىسىنى، ياكى تۆت خىل خۇي پەيلىدىكى ئىنسانلار توپىنى،شۇ توپلارنىڭ رولى شەكلى خۇسيەت ماھىيەتلىرىنىمۇ سۇرەتلەيدۇ. قۇتادغۇ بىلىگ ئىنسان روھىيتىدىكى تۆت تادۇنىڭ زىدىيەت مۇناسىۋىتى بۇيىچە ،تادۇچە تەپەككۇردا يىزىلغان كىتاپ بۇلۇپ، پىكىر يۇكسەكلىكى جەھەتتەيىتۇك قىممەتكە ئىگە.‹قىرىق ياشقا كىرگۇچە تادۇ بىلمىگەنلەرنى ئادەم سانىمىغان›. ئىزدەنسەك، روھىيەتنى تۇنۇش داۋالاش باغلاش يىغىنچاقلاش جەھەتلەردە دەرىتلەرگە داۋا تاپقىلى بۇلىدۇ.  

تادۇ بۇيىچە باشتىن باشلىساق،بىز ياشاۋاتقان دۇنيا ماكان زامان ئىنسان مەدىنىيەتتىن ئىبارەت تۆت تادۇنىڭ بايانى. ماكان ماددا بۇيىچە تۇپراق ئوت سۇ ھاۋادىن تۇزلىدۇ. جانلىق بۇيىچە ئىنسان جىن  ھايۋان ئۆسۇملۇك دەپ تۆت تادۇغا ئايرىلىدۇ.بىرى جىسمى بولسا بىرى روھى.زامان خۇددى تۆت پەسىلدەك ماڭىدۇ. بەزىدە كىتاپلاردا ئۇچۇرتۇپ قالىمىز. ئىپتىدايى جەمىيەت، قۇللۇق جەمىيەت ، ياكى تاش قۇراللار، مىس قۇراللار دىگەندەك، ئىپتىدايى  زامانىۋىي دىگەنلىرى ماكان تۇپراق بۇيىچە زامانغا ئىسىم قۇيغىنى بولسا، تاش قۇرال ئۇچۇر دەۋرى دىگەنلىك زامان ئوت بۇيىچە شۇ خىل خۇي پەيلىدىكىلەرنىڭ قويىۋالغانلىرى. ئىنسان سۇ بۇيىچە قويساق ،باھار سۇبھىگە ئوخشاش بىلىش دەۋرى‹ ئىنساندىن نوھقىچە›. ياز ياشلىققا سىما ئادىللىق دەۋرى ‹نوھتىن ئىبرايىمغىچە›.ئاندىن كۇز قىرانلىققا ۋەكىل ياخشىلىق دەۋرى ‹ئىبرايىمدىن رەسۇللىلا غىچە›. ‹رەسۇللىلادىن زامان ئاخىرغىچە› قىش قىرىلىق تەقۋادارلىق دەۋرى دەپ تۆت تادۇغا ئايرىلىدۇ.ئىنسانغا كەلسەك ، مەرھۇم ئابدۇشۇشكۇر مۇھەممەد ئىمىننىڭ نەقلىسى ئۆزلۇك بۇيىچە ئۆزىمىزنى ئەسلەشكە توغرا كىلىدۇ. ھۇقۇقدار شىر، مەپتۇنكار توز، ئوقۇر ئاشناسى كالا سۇپەتتە جىسمىمىز ئۇستىدە ئىزدىنىپلا قالماستىن،بەلكى ئىنسانىي سۇپەتتە روھىيتىمىز ئۇستىدە ئىزدىنىش ئىنسانلىقنىڭ مۇھىم قىسمى. ئىنسان روھىيتى بىلىش ئادىللىق ياخشىلىق تەقۋادارلىق دەپ تۆت تادۇغا ئايرىلىدۇ.بىلىشتە پەن- ئىلىم مۇھىم قىممەتكە ئىگە.ئادىللىقتا قۇرال قانۇن ئەتىۋا، ياخشىلىقتا ئىنسانچىلىق ئادىمىيلىك مۇھىم. تەقۋادارلىقتا ئادەت كۇنەگ ئاساس. مەدىنىيەت ئام ئاتالغۇ بۇلۇپ، دۇنيادىن تاشقىرلىققا ئىگە. خۇي پەيىللەرنىڭ ئوخشىماسلىقى، بۇنىڭغا سىڭگەن مەدىنىيەت ھازىرقى دەۋىرگە كەلگەندە يەھۇدىي دىنى ۋەبىلىش پەلسەپىسىنى چۆرىدىگەن يەھۇدىي مەدىنىيتى. ئادىللىق سۇپەتتىكى خىرىستىيان مەدىنىيتى، ياخشىلىق سۇپەتتىكى ئىسلام مەدىنىيتى،تەقۋادارلىق خۇسيەتتىكى بۇددا مەدىنىيتى دەپ تۆت خىل مەدىنىيەت ئەندىزىسىگە يىغىنچاقلاندى. يەھۇدىيلار ماكاننى چىقىش قىلىدۇ،ياكى شۇنىڭغا تۇغما مايىل.نىسارالار زاماننى ئاساس قىلىدۇ،مۇسۇلمانلار ئىنساننى چۆرىدەيدۇ، بۇددىسىتلار مەدىنىيىتىنى چىڭ تۇتىدۇ.

ماكانغا تۆت تادۇ بۇيىچە  قارساق، ئاي تۇپراق، كۇن ئوت، زىمىن سۇ ، باشقا يۇلتۇزلار ئۆز ئىچىگە ئالغان ئاتمىسبىرا ھاۋاغا قىياس قىلىنىدۇ. سامان يۇلى مۇشۇ تۆت تادۇنىڭ مۇناسىۋەت زىدىيەت باغلىنىشىدا ھەركەتلىنىپ مۇساپىلىنىدۇ.قەدىمدە نۇرغۇن ئۇلىمالىرىمىز يۇلتۇزلار ئىلمىدە يىل ئاي كۇن ھىساپلاش، زاماننىڭ ئۆلچىمىنى قىياسلاش قاتارلىق ئىلىملەر بىلەن شۇغۇللانغان، تەدىرجى ھازىرقى ماتىماتىكا ئىلمىگە ئاساس سالغان.  زىمىن سۇ، شۇ ۋەجىدىن زىمىنمۇ تۇپراق ئوت سۇ ھاۋادىن تۇزۇلگەن.تۇپراق قۇرۇق سوغۇق ئۇندۇردۇ، ئوت قۇرۇق ئىسسىق ئىسسىتىدۇ،سۇ  ھۆل سۇغۇق سوغۇتۇپ، ھاۋاھۆل ئىسسىق تۇتۇپ تۇرىدۇ. جانلىقلىرىغا كەلسەك ئىنسان  تۇپراقتىن ئاپرىدە قىلىنغان. جىن ئوتتىن، ھايۋان سۇدىن، ئۆسۇملۇك ھاۋادىن.  قۇرۇق ئىسسىق شەرىقتە كونىلارنىڭ رىۋايىتىگە يانداش كويقاپ بارمىش ، ھۆل سوغۇق غەرىپتە سۇ چاھار پاي ھايۋانات كۆپرەك ، قۇرۇق سوغۇق شىمالدا تۇپراق ئىنسان زىچراق ، ھۆل ئىسسىق جەنۇپتا  ھاۋا ئۆسۇملۇك ئورمان، مۇشۇنداق تەڭشەلگەن.ھازىرقى  ئىلمىنىتلار جەدىۋىلىگە قارساقمۇ 106 ئىلمىنىت قەيىت قىلىنغان،شۇلارنىمۇ قىينالماي تۆت تادۇغا ئايرىش مۇمكىن. مىتاللويىدلار يەنى كاربۇن ‹تاشنىڭ تەركىۋى› كالتىسىي .. بۇلار تۇپراق. ئۇراندىن باشلاپ قۇرال ياسايدىغان مىتاللار ئوت.ئاندىن سۇيۇقلۇق كىسلاتا ئىشقارلار سۇ. ئىنىرىت گازلار ، گازلار  ھاۋا.بالىمىز جىجاۋاتقان تۆت ئەمەلمۇ تۆت تادۇ.تادلىغىلى تۇرساق تۆت تادۇغا چۇشمەيدىغان نەرسە يوق.

زامانغا كەلسەك ،تۆپىدىرەك ۋاقىت زاماننىڭمۇ تۆت پەسىل تۆت تادۇ بۇيىچە ماڭىدىغانلىقىنى قەيىت قىلىندى.ھازىر 4-تەقۋادارلىق زامانى تەدىرجى ھەرەڭ سەرەڭدە تاسقايدۇ،جىن سەكرىسە سەكرىسۇن مەن سەكرىمەي ئوقۇدۇم دىگەنلەر پايدا ئالىدۇ. بۇزاماننىڭ تۇنجى بىلىش قەدىمىنى ئەرەپلەر باشلىغان بۇلۇپ، ئىككىنچى ئادىللىق قەدىمىنى ياۋرۇپا مەدىنىيەت سەنئەت گۇللىنىشى باشلىدى. تەدىرجى ئادىللىققا سىموۋۇل قۇرال ،مىتال، ئىككى قىتىملىق دۇنيا ئۇرشى ،ئاندىن ماشىنا تىلىۋۇزۇر ئايرۇپىلان، زەرەت ئۇچۇر گەۋدىلەندى.ئەمدى ياخشىلىق تەقۋادارلىق دەۋىرلىرى مۇشۇنداق دەۋرىيلىنىدۇ.زامان جەھەتتىن،تۆت تادۇنىڭ ئۇيغۇرلارنىڭ ۋارىسىي ‹ئىرسىي› تەپەككۇر ئۇسۇلى ئىكەنلىكىنى پاكىتلىساق، ماكان پاكىتى؛ ئىجتىمائىي ئەمىليتىمىزدە بىزدە ئىلمگە چۇشلۇق ،خەقنىڭ ئالدىغا قۇيغىلى بۇلدىغان نەرسىدىن مۇشۇ ئۇيغۇر تىبابىتى بار.ھىلى ھەم تىخى نۇرغۇن ئىلغار تىخنىكىلار ئىچىدىن ئەسقىتىۋاتقىنى يەنىلاشۇ بولماقتا. بۇندىن كىيىنمۇ يەنە شۇ يولنى بويلا. چۇنكى؛ جەنۇپتىكى دوختۇرخانلارنىڭ كۇلىمى ،دوختۇرلۇق ساھاسىدە يىتىلىۋاتقانلارنىڭ كۇلىمى ئاتا بوۋىمىز دوختۇر تۇغۇپ قويغاندەك خىلى زور سالماقنى ئىگەنلەيدۇ. كۇندىلىك تۇرمىشىمىزدىمۇ ساۋاتسىزلارغىچە ھەممىسى ئىسسىق سۇغۇق ئايرىيدۇ.ئىرسىيتىمىزدە بۇ خىل ئىلمىنىت يەتكۇچە بار.خۇي پەيلىمىز مۇرەككەپ ،مەدىنىيەت قاتلىمىدىكى مۇرەككەپلىك تەدىرجىي تۆت تەرەپ تۆت تامنى يىغىنچاقلاشقا تەقەززاھ قىلىدۇ.زامان پاكىتى؛ ئۇيغۇرلارنىڭ دۇنيا مەدىنىيەت خەزىنىسىگە قوشقان تۆھپىسى ئىچىدە تۆت تادۇ ۋە سودا مۇھىم سالماقنى ئىگەنلەيدۇ‹ئابدۇشۇشكۇر مۇھەممەد ئىمىن›.ھازىرغىچىلىك ياۋرۇپا نىڭ تۆت تادۇنى بىزدىن ئۇگەنگەنلىكى قۇۋەتلەنمەكتە.ئىھرامدىن باشلىنىپ، سەددىچىن ئاندىن مايالار قۇرلۇشلىرىغىچە بۇبىر سىزىق تۇركىي قەۋىملەر نىڭ پائالىيەت ئىزىغا جايلاشقان. ئىھرام تۆت تەرەپلىك، چۇققىدىكى بىرى بىلەن تامام گەۋدىسى شەكلى ماھىيەتتە ئوخشاش، بۇ بىلىش.ئىھرام تاشتىن ياسالغان ياكى تاشتەك قاتتىق‹تاش قاپاقنى يارىدۇ› بۇ ئادىللىق. تۆت قىر تۆت تەرەپلىك‹تۆت تادۇ› يۇلەپ تۇرىدۇ، بۇياخشىلىق.بىرنى بىرنىڭ ئۇستىگە قۇيۇپ بوي سۇندۇرۇلغان، بۇ تەقۋادارلىق. يەنە فارابىي  ئىبىنسىنا يۇسۇپلەر تىبابەتچىلىك تۆت تادۇ پەلسەپىسى بۇيىچە چوقا نوختىلارغا مۇيەسسەر بولغان، ئىسلام پەلسەپىسىنىڭ ئاساسىنى تۇزگەن.شۇنداقلا،قەدىمقى پۇتۇن دۇنيا مىڭە قاتلىمىسىغا ئومۇملاشقان. ئىنسان پاكىتى؛ ئۇيغۇر تىبابىتى بىلەنمۇ ئادەمنى شەرھىلىگىلى بۇلىدۇ،بۇئىنسان پاكىتىغا يىتىپ ئاشىدۇ.يەنە تادۇ بۇيىچە تۇركىي قەۋىملەر سۇ ياخشىلىققا ۋەكىللىك قىلىدۇ.تەپەككۇرىمىز تادۇچە، دىلىمىز ياخشىلىققا مايىل،تىلىمىز تۆت كوپىلىتلىق شىئىر،خۇيىمىز بەلغەم. قۇرۇق ئىسسىق كىلاماتتا ھۆل سۇغۇق بەدەن بولمىسا ئۆزلەشمىكى تەس.ۋە ئىزچىل بىزدە ساقلانغان.مەدىنىيەت پاكىتى؛ مىللەت قەۋىم ئىككىسى بىر گەپ بۇلۇپ، بۇنىڭغا مەدىنىيەت قىتىلسا، مىللىي مەدىنىيەت بۇلىدۇ.بۇنىڭ تەڭپۇڭلۇق باغلىنىشى ناھايتى مۇھىم.ئافاق دەۋرىدە مەدىنىيەت تەكىتلىنىپ،مىللىيەت چەتكە قىقلغان بولسا يىقىنقى زاماندا ،مىللىيەت رولچى بۇلۇپ،مەدىنىيەت سۇس قارالغان.بۇلارنىڭ تارىخىي قىسمىتى ، سائادىتى ،ئۆز ئوقىدا پىقىراشتىن باشقا ئىلگىرلىك بىلەن نەتىجىلەنمىدى. ھازىر بولسا، پارتىيىمىزنىڭ توغرا رەھبەرلىكى ئاستىدا، مىللەت ۋە مەدىنىيەتكە تەڭ ئىتىبار بىرىلىپ ،بۇجەھەتتە تەرەقىقىيات تەڭ تۇتۇلماقتا. بىزلەر ، ئەلگە كىرسەڭ ئىلىڭچە، سۇغا كىرسەڭ بىلىڭچە دەپ ئاسان خۇي ئالىدىغان بولغاچ، ماسلاشتۇرۇش، باغلاش رولىمىز گەۋدىلىكرەك.ئاد قەۋىمى ئازاپ بۇلۇتىنى، يامغۇربۇلۇتى خىيال قىلغاندەك،ئىنىقلىماي دادىل چايناشقا ئالدىرايمىز.ئەمىلىيەتتەجەمىيتىمىزنى مىللەت بىلەن مەدىنىيەت گىرەلەشكەن جانلىق ئۆرنەك،ياكى ھەرخىل ئىقىملارنىڭ يۇرۇش تۇرۇش نەتىجە ھاسىلاتلىرىنىڭ تەجىربە سورىنى دىسەك ،ھەرگىزمۇ ئىشىپ كەتمەيدۇ.يۇسۇپ، ماخمۇد  زامانلىرى ئىسلام بىلەن ياخشىلىق تىپىدىكى مىللەتنىڭ مەدىنىيەت بىلەن گىرەلەشكەن ھالىتىنى قىياسلىسا.تەسەۋۋۇپ تەقۋادارلىق تىپىدىكىنىڭ قۇشۇلغان ھالىتىنى.ھازىرقى زامان بولسا بىلىش تىپىدىكىسىنى ئىپادىلەيدۇ.مىللەت بىلەن مەدىنىيەتنى بىرىكتۇرۇش جەھەتتە قىممەتلىك جەريانى بار.دىمەك ئۇيغۇرلار مۇشۇلارنى كەڭ مەنىدىكى ماسلاشتۇرۇش باغلاش يۇغۇرۇش جەھەتتە ئىرسىيتىدىكى تادۇچە تەپەككۇر بىلەن ياخشى ئۆرنەك بۇلالايدۇ.

ئىنساننى تادلىساق،ئىنسان ماھىيتى تادۇ بۇيىچە ياخشىلىق. ئىنسان جىسىم ۋە روھىيەتتىن تۇزىلىدۇ. جىسمى سەۋدا‹تۇپراق› سەپرا‹ئوت› بەلغەم‹ سۇ› قان‹ھاۋا› دەپ تۆت خىل مىجازغا ئايرىلىدۇ.بۇ ئادەم بەدىنىدە مۇشۇ تۆت خىلىت ھەممىسى بار،بۇنىڭ ئىچىدە بىرسى يىتەكچى بۇلىدۇدىگەنلىك.سەۋدا مىجازدىكىلەرنىڭ بەدىنى قۇرۇق سۇغۇق بۇلىدۇ.تۇمۇرى ئىنچىكە ئۇزۇن ئاستا سالىدۇ.ئۇرۇق كىلىدۇ.سەپرادا قۇرۇق ئىسسىق بۇلىدۇ،تۇمۇرى ئىنچىكە قىسقا تىز سالىدۇ،ئاچچىقى يامان. بەلغەم ھۆل سۇغۇق بۇلىدۇ،توم ئۇزۇن ئاستا سالىدۇ.قان ھۆل ئىسسىق،  توم قىسقا تىز بۇلىدۇ.بەزىلەردە بىر خىل خىلىت رۇشەن ئەكىس ئىتىدۇ.بەزىلەردە ياكى كۆپۇنچىلەردە خۇنىي سەپرا، بەلغەمىي سەۋدا دىگەندەك ئىككى خىلىت ئالامىتى بىلەن بۇلىدۇ.روھىيتى جەھەتتە بىلىش ئادىللىق ياخشىلىق تەقۋادارلىق دەپ تۆت تادۇغا ئايرىلىدۇ.ھەمدە سەۋدا سەپرا بەلغەم خۇن خىلىتلارغا باغلىنىدۇ.تۇپراق خۇسيەتتىكى ئىنسان بىلىشى ،تەپەككۇرنى ئۆز ئىچىگە ئالغان ھالدا ھۇش ئەقىل پەزىلەت بىلىم دەپ تۆت تادۇغا ئايرىلىدۇ.ھۇش ،سەزگۇرلۇك ئىجادى كۇچ بايقاشنى كۆرسەتسە، ئەقىل تەدبىر بۇلىدۇ. پەزىلەت  گۇزەل ئەخلاق، بىلىم توپلانغان تەجىربە.  ئادىللىقنىڭ تادۇسى ئوت بۇلۇپ،دىلدا ئەكىس ئىتىپ ئاساس رول ئوينايدۇ.ئىنسان دىلى ھاۋايى- ھەۋەس، غەزەپ -خۇشلۇق ، مۇھەببەت- نەپرەت،ۋىجدان ساداقەت دەپ تۆت تادۇغا ئايرىلىدۇ.ياخشىلىق سۇ خۇسيەتتىكى تىل بىلەن مۇناسىۋەتلىك ،تىلمۇ مەنا تاۋۇش گىرامماتىكىسى يىزىق دەپ تۆتكە بۆلىنىدۇ.تەقۋادارلىق خۇي ئادەتتىن ئۇنۇپ ھاۋا بۇلىدۇ.قەۋىملەرنىڭ خۇيدا پەرىق بار. مەسلەن؛ ھىندىلار تەقۋادارلىققا مەنسۇپ بولغاچ ،بىلىم ۋىجدان ساداقەت يىزىق ئادەت قارشى ئۇستۇن بۇلىدۇ.يەھۇدىيلار دا بولسا، ھۇش بايقاش ھاۋايى ھەۋەس مەنا يىڭىلىق ئىستىكى يۇقرى بۇلىدۇ.نىسارالار ئەقىل تەدبىر غەزەپ خوشلۇق تاۋۇش قۇرال سۇپىتىگە ئىگە.خۇلاسىلىغاندا خۇينى يىڭىلىق قۇرال پەزىلەت ئادەت دەپ ياكى بىلىش ئادىللىق ياخشىلىق تەقۋادارلىق تۆت تادۇغا ئايرىشقا بۇلىدۇ.قۇتادغۇبىلىگ خۇي بۇيىچە يىزىلغان كىتاپ.

ئاددىيراق دىسەك؛  ئىنسان جۇملىدىن ئىنسانلارنى بىرتۇپ دەرەخ دىسەك، دەرەخنىڭ بۇكلىكى ئۇچۇن ئوزۇقلۇق كىتىدۇ،بىلىش، يەنى تەپەككۇرقىلىپ تۇنۇش ئۇگۇنۇش ئىزدىنىش قاتارلىقلار ئوزۇقلۇغى‹تۇپراق› ھىساپلىنىدۇ.ئاندىن يەنە مۇئەييەن ئىسسىقلىق لازىم بۇلىدۇ،بەش پەرھىز ئەھدە قانۇن ئادىللىق‹ئوت› بولمىسىمۇ بولمايدۇ.يەنە دەرەخكە سۇ باشلاش لازىم، باشقىلارغا ياخشىلىق قىلىش،يىتىم يىسىرلارنى يۇلەش،پۇل- پىچەك كىيىم- كىچەك تارقىتىش بۇلار دەرەخكە‹ سۇ› باشلىغانلىق. يەنە ئازدۇر كۆپتۇر تاڭنىڭ ئالدىدا پىنھاندا زىكرى تەسبىھ ئوقۇش، ئىتىكاپ قىلىش تەقۋادارلىق ‹ھاۋا› بۇلىدۇ.ياكى ھەرخىل ھەرياڭزا پائالىيەتلەرگە ئادەتلىنىشكە توغرا كىلىدۇ. مۇشۇ تۆت شارائىتنىڭ بىرى كام بۇلۇپ قالسا ئىنسان روھىيتىنىڭ تۇز تەمى جايىدا بولمايدۇ.ئۇندىن باشقا بىلىشنىڭ بەلگىسى بولغان بۇرۇننىڭ تۇشىگى ئىككى، بىرى سامرىدەك زىبۇ زىننەتتە كالا  تەرەققىيات ياسىسا،بىرى مۇسا ئەلەيھىدەك روھىيەتكە باشلايدۇ.بىلىش سۇپەتتىكى يەھۇدىي سەئۇدىيلارنى قىياس قىلىشقا بۇلىدۇ. كۆز دىن نۇر چاقنايدۇ ، ئادىللىق بۇيىچە قىلىچ ئوقيالىرى بار.ئىغىز كالپۇك چىش تىل بۇغۇز دىن ئىبارەت تۆت تادۇ بىلەن چاينايدۇ.ياخشىلىق بۇلىدۇ.قۇلاق قاتقات چاچلار ئارسىدىن چىقىپ،تەقۋادارلىق بىلەن ھاۋادا تىتىرەپ گەپ ئاڭلايدۇ.خۇلاسىلىغاندا ئىنسانلىق ماھىيەتنى ئىستەش ، تۆت تادۇ مۇشۇ خىل پىكىر يۇلىنىشى بىلەن زىچ مۇناسىۋەتلىك.

مەدىنىيەت جەھەتتە، ئىسلام دىنى تۇركىي قەۋىمىگە ئۇمۇملاشقاندىن كىيىن، تۇرۇكلەر يەنى فارابىي ، ئىبىنسىنا،يۇسۇپ خاس ھاجىپ قاتارلىقلار ئىسلام مەدىنىيتىنى پەلسەپە بىلەن بىزىگەن.ھەممىمىزگە مەلۇملۇق، ئىسلام دىندا قۇرئان ئاساسى يوق ھەرقانداق ئىش سۆز ھەركەتلەر قارشى ئىلىنمايدۇ.قۇرئان كىرىمدە ، ئىنساننىڭ يارتىلىش جەريانى تۆت تادۇ بۇيىچە بايان قىلىنغان. ئىنسان تۇپراقتىن،جىن ئوتتىن،ھايۋان سۇدىن ئاپرىدە قىلىنغانلىقى قەيىت قىلىنغان. تۆت كىتاپ، تۆت ساھابە،تۆت چارىقۇل، يەنە سۇرە رەئىدتە ئىنسان روھىيتى بىلىش، ئادىللىق ياخشىلىق،تەقۋادارلىقتىن ئىبارەت تۆت تادۇدىن تۇزۇلىدىغانلىقى توغرىسىدا ئازراق پەلسەپە قاتساق بىشارەتلەر بار.قافتا تۆت خىل قەلىپ سۆزلىنىدۇ.ئۇنىڭدىن باشقا قەدىمقى باغداتتا دۇنيا يۇزدىكى ئىلىم كىتاپلارنىڭ ھەممىسىنىڭ يىغىلىپ تەھسىل قىلىنىشىغا تۇركىي قەۋىملەر يىتەكچىلىك رول ئوينىغان.دىمەك مەدىنىيەت جەھەتتە تۆت تادۇ پەلسەپىسى ئىسلام پەلسەپىسىنىڭ ئاساسى.

تۆت تادۇ پەلسەپىسى ئۇيغۇرمىللىي مەدىنىيتىنىڭ ئۆزتەگ پىكىر ئۇسۇلى پەلسەپىسى بۇلۇپ،بۇنى قانداقتۇر كونىلىق، ۋاختى ئۆتتى دەپ ساناپ، پەرۋىش قىلماسلىق ، ماكان زامان ئىنسان مەدىنىيەت ئالدىدىكى مەسھۇلىيەتسىزلىكتىن باشقا نەرسە ئەمەس.   

  مەنبە ؛قۇتادغۇبىلگ،ئۇيغۇر تىبابىتى قامۇسى.

4تادۇ توغرىسىدا چۈشەنچە بەرگنىڭىزگە رەخمەت.ئەمما بەزى تورداشلار سىزنىڭ مۇشۇ 4 تادۇ توغرىسادا قىلغان يەكۈنلىرىڭىزدىن تازا رازى بولمايدىكەن(بۇرۇن بىرنەچچە ئىنكاسلاردا شۇنداقتەك بايقاپ قالدىم).بەلكىم بۇ بىر نازۇك مەسىلىدۇ.سىز 4 تادۇنى ئىسلام دىنىغىمۇ باغلاپ تەھلىل قىپسىز،بۇ سىزنىڭ شەخسى مۇلاھىزىڭىزمۇ ياكى ئىسلام ئۆلىمالىرى ياكى ئىسلام تەتقىقاتىدا مۇشۇنداق 4 تادۇنىڭ ئىسلام دىنىدا بارلىغىنى ئېتىراپ قىلامدۇ؟ئەگەر مەيلى مىللى مەدىنيەت،تەپەككۈر،ئىقتىساد بولسۇن ھەممىسىدە مۇشۇنداق بىر قانۇنىيەتنىڭ بارلىغىنى دەپ ئۆتۈپسىز،ئۇندقتا مۇشۇنداق بىر  مۇقەددەس قانۇنىيەتنى ھەممە ساھەدە بار دەپ قارىساق كىشىلەرنىڭ ئېڭىدا  مۇشۇ مەسىلىگە نىسبەتەن سەل ئابىستراكىتلىق چۈشەنچە پەيدا بولۇپ قېلىپ ئېتىقاد بىلەن سەل توقۇنۇشۇپ قالامدۇ قانداق؟ يەنە ئەتراپلىق چۈشەنچە بەرسىڭىز تېمىڭىز تېخىمۇ مۇكەممەللىشپ كەتكەن بولاتتى.بولمىسا يەنە ئايىغى چىقماس تالاش-تارتىشنىڭ بولىشىنى،قارشى پىكىرلەرنى چۈشمەيدۇ دېگىلى بولمايدۇ.چۈنكى كۆپىنچە كىشىلەر بۇ ھەقتە تەكشۈرۈپ تەتقىق قىلمىغان.بۇنى سىز بىزدىن بەكرەك ياخشى بىلىسىز.

زامان بۇيىچە تارىختا بەلكىم ئىنسانىيەت مەدىنىيتىنىڭ نۇرغۇن تەپەككۇر مۇساپىسى تۆت تادۇ بۇيىچە بولغان.تادۇچە ئۇلىما پەيلاسۇپلىرىمىز  فارابىي  تەپەككۇر، تەسەۋۋۇپ دىل،ماخمۇت قاشقىرى تىل، يۇسۇپ خاس ھاجىپ خۇي بۇيىچە ئىسلام ئىممەتلىرىنىڭ ئىچىدە ھىلى ھەم داڭلىق. باشقا پەنچىلەرنى ياكى پەلسەپىچىلەرنى بۇلارغا ھازىرمۇ تەڭ قىلغىلى بولمايدۇ.
