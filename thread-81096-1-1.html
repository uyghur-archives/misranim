          ھەي ،،،،نىمىسىنى دەي،ئاران بىر دەم ئالغاندا بىر قانچە قۇر يازلىق كىيىم سېتىۋېلىش مەقسىتىدە بازارغا چىقتىم.يىڭى ئۇزۇنكەن دەپ يۈگىرەپ بارسام ،دۈمبىسى ئوچۇق....دۈمبىسى يۆگەلگەننىڭ مەيدىسى ئوچۇق،،،كۇسارلار ئوشۇقتا،،،،ئىشتانلار تىزدا.................كىيىم ساتقۇچىدا چاتاق بارمىكىن دېسەم،باشقا خانىم-قىزلار ھەدەپ ئېلىۋاتىدۇ،،،،چاتاق ئۆزەمدە ئوخشايدۇ،،،بۇ يىڭى كىيىملەرنى ئالغاندىن يەنىلا ئۆيدىكى كونا كىيىملىرىمنى كىيەي،،،بۇنداق كىيىملەرنى كىيگەندىن ،،مەقسەت ئېنىق ئېچىپ قويۇش بولغاندىكىن كېيمەيلا ماڭاي.........
              مانا مۇشۇ خىياللار بىلەن چوڭ بازار سودا-سارىيدىن چىقىپ ماروژنا يىيىشكە ئولتۇردۇم،،،ئاران بىر بازارغا چىققاندا سودىلىقىمنى قىلالمىغىنمغا ئىچىم سىقىلدى،،ئۆز-ئۈزەمگە غۇدۇراپ تۇيۇقسىز يېقىنلايەردىكى تۆگىنىڭ ھەيكىلىگە كۆزۈم چۈشتى،،،،نىمىنى خىيال قىلدىم دىمەيسىز،،،تۆگىنى كۆرۈپ مۇنبەردىكى باش سۆرىتىم كۆز ئالدىمغا كەلدى...........................
            بېشىمدا تۇماق،،،ئۈستىمدە تون،،يېنىمدا قېلىچ،،،تۆگىگە مىنىپ مېڭىپ كەتسەم:
  -ھەي،،ھەي،،ئۆرۈپ ئادىتىمىزدە بۇنداق يۈرۈش چەكلەنگەن،،كېلىڭلار.....ماۋۇ خۇدادىن قورقمىغان ،بەندىدىن ھايا قىلمىغان ئەخلاقسىز خوتۇنلارنى ئاپىرىپ زىندانغا تاشلاڭلار،-دەپ قامچامنى شىلتىپ تۇرۇپ بۇيرۇقلارنى قىپ كەتسەم.........
               ****                       ******                         ******                                 ****
    -تارت قولۇڭنى،-دەپ ۋارقىرغان ئاۋازدىن كۈپكۈندۈزدە ئۇخلىماي تۇرۇپ كۆرىۋاتقان چۈشۈمدىن ئويغىنىپ كەتتىم،،تەمبەل،،قارامتۇل،،چىرايىدىن قەتئىيلىك چىقىپ تۇرىدىغان 34~35ياشلار چامىسىدىكى چوكان چوڭ بازار ئامانلىق ساقلاش خادىمىنى ئىتتىرۋاتاتتى،،، ئامانلىق ساقلاش خادىمى ئوتتۇرا بوي،،تولىمۇ زىلۋا،،بېشىدىكى شەپكىسىمۇ چوڭ كىلىپ قۇلىقىغا چۈشۇپ كۆزىنى توسىۋالغان بالا بولۇپ ، قارىماققا تولىمۇ ئاجىز كۆرىنەتتى،،،،شۇنداقلىقىغا باقماي مائاشى بەك يۇقىرىمۇ ياكى خىزمىتىنى بەك سۆيەمدۇ بىلمىدىم ، يانماي ئۇرۇشىۋاتاتتى.........
          -بولدى،بولدى قىلىشسىلا،،مەن ساتماي،،سېۋەتنىمۇ ئالماي،،جېنىم ئۇكام ،جىدەل قىلمايلى،-دەپ 55ياشلار چامىسىدىكى بۇ بوۋاي ئايالنى تارتماقتا ئىدى......
              -ھە ،مانا ماۋۇ ئىش بولدى،،قايتىڭلار،،قايتىڭلار،-ئۇرۇشتا ئۇتقان ئامانلىق ساقلىغۇچى ئەتراپتىكىلەرنى تارقاقلاشتۇرۇپ گېنرالدەك قىياپەتتە سېۋەتنى كۆتۈرۈپ مېڭىشقا تەمشەلدى...بۇ چاغدا مەن ئاللىبۇرۇن ئولىشىۋالغانلار توپىنىڭ ئەڭ ئالدىدىن ئورۇن ئېلىپ بولغانىدىم.......
                  -ئاكا،،ئۆيىڭىز نەدە،،قارىسام بۇ يەرلىك ئەمەستەك قىلىسىز؟-سورىدى ئايال بوۋايدىن.
          -مەن بۇ يەرگە موماينىڭ كىسىلىنى داۋالىتىش ئۈچۈن كەلگەن،،ئۇ ھازىر دوختۇرخانىدا،،كېسلى ئېغىر ،،،بىز پەرزەنت يۈزى كۆرمىگەچكە قېرىغاندا جاپادا قالدۇق،،ئۇرۇق-تۇققانلىرىمىزمۇ كۆپ ياردەم قىلدى لېكىن دوختۇرخانىنىڭ چىقىمى يامانكەن،،موماينىڭ كىسىلى ياخشىلىنىپ بولغىچە تۇرمۇش راسخوتىنى بولسىمۇ تاپاي دەپ بوش ۋاقىتلىرىمدا بۇ ئەتراپتا دەسلەپ غورا ھازىر ئۆرۈك سېتېپ كۈن ئۆتكۈزىۋاتقان،،قالغان ئىشلار بايا سىز كۆرگەندەك  ئۇكام،،بۇ خەلقئارالىق چوڭ بازارمىش،،دوقمۇش-دوقمۇشتا تىلەمچىلەر ئولتۇرۇپ تىلەمچىلىك قىلسا كارى يوق،،،مەن سېۋەتتە كۆتۈرۈپ يۈرۈپ ئۆرۈك ساتسام بازار تەرتىپى قالايمىقان بولارمىش،،،نەچچە كۈندىن بېرى قوغلاپ ئارام بەرمىگەن،بۈگۈن مانا ھەممە نەرسەمنى تارتىۋالدى،-ساقاللىرى تىترەپ سۆزلەۋاتقان بوۋاي گىپى تۈگىمەيلا پاڭڭىدە يىغلىۋەتتى ..................
                    80كىلودىن كەم چىقمايدىغان چوكان غەزەپ بىلەن يۈگۈرۈپ بىردەمدىلا ئامانلىق ساقلىغۇچىنىڭ ئالدىنى توستى:
              -ئۇكام بۈگۈن بۇ سېۋەتنى سەندىن قايتۇرۋالغىنىم قايتۇرۋالغان،،ھەممىمىز ئۇيغۇر بولغاندىكىن بۇنداق قىلما،،سىۋەتنى قايتۇرۇپ بەر....................
             -ئوھوي،،تازىلىق ئىشچىسىمۇ مىللەتتىن چۈشىۋاتىدۇ دىسە،،سەت تەرەققىي قىپتۇق جۇمۇ،، بۇ قېرىغا بەك ئىچىڭ ئاغرىغان بولسا ئۈزەڭمۇ تۇل خوتۇن بولغاندىكىن تىگىپلا باقمامسەن بولمىسا،-ئامانلىق ساقلىغۇچىنىڭ بۇ گېپى تېىخى ئاخىرلاشماي تۇرۇپ گۇم قىلىپ تەگكەن مۇشتىن يەرگە يىقىلدى،،،سېۋەتمۇ يەرگە چۈشۈپ ئۆرۈك چېچىلدى.ئايال بولدى قىلمىدى،،ئۇنى تۇرغۇزۇپ كاچىتىغا ،،،كاسىسىغا تازا قويدى،،،،شاپىلاق ،پەشۋا دىگەندەكلەر بىلەن ئائىلىدە ،،مەكتەپتە ئۆگىنىپ بولالمىغان ئەدەپ-ئەخلاقنى ئىسىگە سېلىپ قويدى،،قاتتىق سىلكىپ بوينىدىن تۇتۇپ بوۋايغا قاراتتى:
            -قارا،،بۇ سىنىڭ ئاتاڭ،،مەن سىنىڭ ئاچاڭ،،،بۇ چوڭ بازار بىز ئۇيغۇر خەلقى ئۇزۇن يىللاردىن بۇيان تىرىكچىلىك قىلىپ كىلىۋاتقان جاي،،،دەرۋازىغا قاراش ئەسلى ئىتنىڭ ۋەزىپسى،،ئىتنىڭ قىممىتى ئۆسۈپ ئۆيدە پەپىلىنىپ بېقىلىۋاتقاچقا ئىشىككە قاراش سەندەكلەرگە قالدى،،شۇڭابەك ھەددىڭدىن ئاشماي ئۈەڭنى بىلىپ يۈر ۋۇ ئىككى پۇتلۇق ئىت،-دىگىنىچە ئۇنى ئىتتىرۋەتتى.توپنىڭ تېشدا باياتىن بېرى جېدەلگە ئارلىشىشقا پېتىنالماي قاراپ تۇرغان ئىككى نەپەر خىزمەتدېشى ئۇنى يۆلەپ كىتىپ قالدى.......
                ئەتراپتىن گۈلدۈرقاراس بارىكاللا ساداسى ياڭرىدى،،مەنمۇ بۇ ھەدىمىزگە قول قويماي تۇرالمىدىم،،تېخى باياتىن ھېلىقى نا ئەھلى ئامانلىق ساقلىغۇچىنى يۆلەپ ماڭغاندا  كەينىدىن تۈكۈرۈپمۇ ئۈلگۈردۈم،،،قىز بالا بولساقمۇ يوق دىگەندە ئىككى كىلو غورور بادە بىزدىمۇ..................
                   ھەممەيلەن ئالدىراش ئۈرۈك تېرىشكە باشلىدۇق.......................
             - يەنە مۇشۇ خوتۇنكەنغۇ ،،ئۆتكەن ھەپتە بىر ئامانلىق ساقلىغۇچىمىزنى نارسىدە قىزنىڭ پۇتىنى سىلىدى دەپ ئۇرۇپ قولىنى سۇندۇرۋەتكىنىمۇ مۇشۇ خوتۇن،،،نۇرغۇن ئادەم تەرىپىنى ئېلىپ تۇرۋېلىپ ساقچىخانىمۇ قويىۋەتتى...بۈگۈن بۇ ئىش بوپتۇ،،ئوغۇل بالا بولغاندىكىن سەنمۇ ئۇرساڭ بولمامتى لەقۋا،،،،بۇندىن كىيىن بۇ خوتۇن بىلەن ھەپىلەشمەڭلار،،،،تازىمۇ بىر بالا بولدىغۇ بۇ،، دۆڭكۆۋرۈكتە ئايال لۆكچەك پەيدا بولۇپ،-باياتىنقى يالاقچىلار چېقىشتۇرۇپ ئېلىپ كەلگەن باشلىقى كەلگىنىگە پۇشايمان قىلغاندەك بۇ گەپلەرنى دەپلا كىتىپ قالدى...........
             رەخمەت،،بارىكاللا ئېيتىۋاتقانلارنى بېسىقتۇرۇپ،  سورلىۋاتقان سوئاللارغا كۈلۈمسىرەش بىلەن جاۋاپ قايتۇرۇپ ،يوغان سۈپۈرگىسىنى كۆتۈرۈپ ماڭغان ئايالنىڭ كەينىدىن بەزىلىرىمىز توۋۋا دىسەك،،بەزىلىرىمىز قالتىس ئايالكەن دىگىنىمىزچە قاراپ قېلىشتۇق...........................
                      *****                                 ******                                     *****
           دۆڭكۆۋرۈكتىكى ئايال لۆكچەك........پاھ،،كىيىم سېتىۋالالمىغان بىلەن مىسرانىمغا تېما خۇرۇچى سېتىۋالدىم مانا،،،، ئۆرۈكلەرنى تېرىۋاتقان،، تەرەپ-تەرەپتىن چاندۇرماسقا سېلىپ بوۋاينىڭ يانچۇقىغا كۆڭۈللىرىنى ئىپادىلەپ پۇللارنى تىقىپ قويىۋاتقان قوللارغا،،ئۇيغۇرنىڭ غورورى ھەم مىھرى-مۇھەببىتى بىلەن تولغان دۆڭكۆۋرۈككە  قاراپ كۆزلىرىمگە ياش ئالغىنمچە ئۆيگە،،كومپىيوتىرىم ئالدىغا قاراپ يۈگۈردۈم.............
