بۇ ئالەمدە ئۆزىنىڭ پۈتكۈل ئۆمرىنى خۇشال-خۇراملىق ۋە بەخىت ئىچىدىلا ئۆتكۈزگەن  بىرمۇ  ئىنسان بولمىسا كېرەك، ھاياتلىق ئادەمگە بەزىدە شادلىق شارابى تەڭلىسە،بەزىدە قايغۇ خەنجىرى تەڭلەيدۇ،بىز مەڭگۈ چاپچىپ تۇرغان تايچاق بۇلۇشنى خالىساقمۇ،ئەپسۇس تۇرمۇش  بىز ئۈچۈن مەڭگۈ سۇزۇلۇپ ياتقان يايلاق ئەمەس،بىزنى ھەر خىل قىسمەتلەر ۋە خىرىسلار كۈتۈپ تۇرىدۇ.بۇلۇپمۇ ھازىرقى جەمىيەتتە يېرىم دۇنيا دەپ ئاتالغان ئاياللار دۇچ كىلىدىغان بېسىملار ۋە قىيىنچىلىقلار ھەقىقەتەن كۆپ.

      ئەي ئايال،سىزنىڭ مۇشۇ تۈگىمەس تۇرمۇش بېسىملىرىنى ئۇڭۇشلۇق  يېڭىپ،نىكاھ تۇرمۇشىڭىزدا بىر پارلاق سەھىپە ئاچقۇڭىز بارمۇ؟ئۇنداقتا سىز مەڭڭۈ ئۈزىڭىزنى تۆۋەن كۆرمەسلىكىڭىز،ئۆزىڭىزنى يۇقىتىپ قويماسلىقىڭىز،ئۆزىڭىزدىن ۋاز كەچمەسلىكىڭىز كېرەك،سىز ھاياتلىقنىڭ مۈشكۈل تاللاشلىرىغا دۇچ كەلگەن چېغىڭىزدا ،تۇرمۇشتىن ئۈمۈدسىزلىك ھىس قىلىپ ئۈزىڭىزنى يۇقىتىپ قويغان چېغىڭىزدا،ھەتتا بېشىڭىزغا كەلگەن كۈنلەرنىڭ قاتتىقلىقىدىن ھاياتىڭىزدىن ۋاز كەچكەن چېغىڭىزدا....سىزنى قۇتقۇزالايدىغان ئۈزىڭىزدىن باشقا ئادەم چىقمايدۇ،سىز بۇنىڭ ئۈچۈن چۇقۇم قەيسەرلىك بىلەن تۇرمۇشقا يۈزلىنىشىڭىز ،ئۈزىڭىز ئۈچۈن يول ئېچىشىڭىز كېرەك!

ئاياللار قانداق ۋاقىتتا ئەڭ گۈزەل ۋە ئەڭ بەخىتلىك بۇلىدۇ؟

   نىكاھ ئۇقۇلۇپ بولغاندىن كىيىن،سىز چىرايلىق توي كۆينىكى كىيىپ،يۈزىڭىزنى پەرىجىلەردە يېپىپ،ئاتا-ئانىڭىزنىڭ ،يۇرىت ئەھلىنىڭ دۇئاسىنى ئېلىپ ،رەڭدار بىزەلگەن توي ماشىنىسىغا چىقىپ  كەلگۈسى ھەمرايىڭىزنىڭ ئۆيىگە قەدەم باسقان ئاشۇ مىنۇت سىزنىڭ ھاياتىڭىزدىكى ئەڭ  گۈزەل،ئەڭ بەخىتلىك  مىنۇتلار  بۇلىدۇ،ئەنە شۇ  چىرايلىق كۆينەكلەر، چىرايلىق بىزەلگەن  ھۇجىرا ئۆينىڭ ھەر  بىر جايلىرىغا بەخىت  يۇشۇرۇنغان بۇلىدۇ،بۇ يەردە قىزىرىپ،كۈلۈمسىرەپ  ئولتۇرغان سىز خۇددى رىۋايەتلەردىكى ھۆر-پەرىلەردەك گۈزەل،خۇددى يېڭى ئېچىلغان گۈل بەرگىللىرىدىكى شەبنەم تامچىسىدەك پاك...

     سىز بىر قىزدىن ھەققى بىر ئايالغا ئايلانغان ئاشۇ مىنۇتتا،بەخىت ھەقىقەتەن سىزنى ساقلاپ تۇرارمۇ؟بۇ ۋاقىتتا سىز پۈتۈن ئۆمرىڭىز خۇددى مۇشۇنداق بەخىت ئىچىدىلا ئۆتىدىغاندەك ھىس قىلىسىز،مۇشۇ ۋاقىتتا سىز مەيلى نىكاھىڭىزدىكى بىر ئەتىۋارلىق گۆھەرگە ئايلىنىڭ ياكى نىكاھىڭىزنىڭ قۇلى بۇلۇڭ،پارچە پۇرات تۇرمۇش سىزگە ھاياتىڭىزنىڭ يەنە بىر بېتىنى ئېچىپ بېرىدۇ،  مۇشۇ ۋاقىتتا سىز ھەيرانلىق ئىلكىدە ھەققى تۇرمۇشنىڭ ئەسلىدە مۇنداق بۇلىدىغانلىقىنى بايقايسىز،تويدىن ئىلگىركى تەسەۋۇرىڭىزدىكى گۈزەل مۇھاببەت ئاخرلىشىدۇ،سىز ئەسلىدە تۇرمۇش  خۇددى ئەتىرگۈلگە ئوخشاش گۈزەل ھەم رۇمانتىك دەپ ئويلايىتتىڭىز،تويدىن كىيىن سىز تويدىن ئىلگىرى پۈتۈن مۇھاببىتىڭىزنى بېغىشلىغان، نوقسانسىز،كىلىشكەن  شاھزادىڭىزنىڭمۇ باشقىلارغا  ئوخشاشلا تاماق يەيدىغان،ھاجەتخانىغا بارىدىغان بىر ئادەتتىكى ئادەم ئىكەنلىكىنى ھىس قىلىسىز،ئۇنىڭ كىيىملىرىمۇ مەڭڭۈ ئۇنداق قاتۇرۇلغان،پاكىز ھالەتتە تۇرىۋەرمەيدۇ،ئۇنىڭ چاچلىرىمۇ مەڭڭۈ ئۇنداق پارقىراق،رەتلىك تارالغان ھالەتتە بۇلىۋەرمەيدۇ.ئۇنىڭمۇ كىرلەشكەن كىيىملىرىنى،سېسىق پۇرايدىغان پايپاقلىرىنى ،مايلاشقان چاچلىرىنى،چاپاق باسقان،ئۇيقۇدىن ئىششىپ     كەتكەن كۆزلىرىنى،مەسىت ھالەتتىكى قورقۇنۇشلۇق ھالىتىنى،قۇسۇقلىرىنى،غەزەپلىك چىرايىنى،،،،،،،،،كۆرىسىز،مانا مۇشۇلارنىڭ ھەممىسى سىزنىڭ ئاكتىپلىق بىلەن قىلىشىڭىزنى،تۈزىتىشىڭىزنى كۈتۈپ تۇرىدىغان ئىشلار.

سىز ئەمدى ئانىڭىزنىڭ قۇچىقىدا يېتىپ ئەركىلەيدىغان كىچىك قىزچاق ئەمەس،سىز ئاللىبۇرۇن بىر ئايالغا ئايلاندىڭىز،سىز ئەمدى تاماق ئىتىشىڭىز كېرەك،كىر قاتلارنى يۇيىشىڭىز كېرەك، ئىرىڭىزنىڭ ،بالىڭىزنىڭ،قېيىن-ئاتا،قېيىن-ئاپىللىرىڭىزنىڭ  ھالىدىن خەۋەر ئېلىشىڭىز كېرەك. قازان-قۇمۇچ،قاچا-قۇچىلار سىزنىڭ تۇرمۇشىڭىزنىڭ بىر قىسمىغا ئايلىنىدۇ،كۈندە يۈز بىرىۋاتقان چۈشىنىشمەسلىك،سۈركىلىشلەر تۇرمۇشىڭىزغا سىڭىپ كىرىپ،سىزنى پاراكەندە قىلىدۇ،بۇرۇنقى شىرىن-تاتلىق پاراڭلارنىڭ ئورنىنى ھەر خىل تالاش-تارتىشلار،پىلانلار ئىگەللەيدۇ،  بۇرۇنقى  يول قۇيۇشلار،مەسلىھەت سېلىشلار ئەمدى بۇيرۇققا ئايلىنىدۇ،تويدىن بۇرۇنقى پۈتۈن ھاياتىڭىز  بىر كىتاب بولسا،تويدىن كىيىنكى ھەر بىر كۈنىڭىز بىر كىتاب بۇلىدۇ،بۇ تۇرمۇش بىر مەكتەپكە ئوخشايدۇ،بۇ يەردىكى سىناقلار بىر-بىرىدىن قاتتىق،ئۈتەلمىسىڭىز قايتا ئىمتىھان بىرىش پۇرسىتى يوق،سىز بۇيەردە رۇخسەت سورىيالمايسىز،خالىغانچە دەرىس تاشلىيالمايسىز،كۆڭلىڭىز خالىىمىسىمۇ، ھەتتا ئاغرىپ قالسىڭىزمۇ چۇقۇم بۇ دەرىسكە قاتنىشىشىڭىز كېرەك.

    تويدىن ئىلگىرىكى تۇرمۇشنى ئەرلەرنىڭ بىر مۈشكۈل كۈرىشىگە ئوخشاتساق،تويدىن كىيىنكى تۇرمۇش ئاياللارنىڭ قۇللۇق ھاياتىنىڭ باشلىنىشى،تويدىن بۇرۇن مۇھەببەت ئۈچۈن ياشىغان بولسىڭىز تويدىن كىيىن ياشاش ئۈچۈن ياشايسىز،ئەمىلىيەتتە نۇرغۇن كىشىلەر شۇنداق  بىر مۇھىم  نوقتىنى ئۇنتۇپ قېلىشىدۇ،ئەمىلىيەتتە نىكاھنى ھەقىقى تۇتۇپ تۇرىدىغىنى يەنىلا مۇھەببەت.

    تويدىن ئىلگىرى سىز ئۆزىڭىزنىڭ ئەڭ گۈزەل ھالىتىڭىزنى يىگىتىڭىزگە كۆرسىتىش ئۈچۈن ئۈزىڭىزنى شۇنداق چىرايلىق ياسايسىز،تويدىن كىيىن پەقەت سىرىتقا چىقىشنىڭ ئالدىدىلا ئانچە-مۇنچە ئۈزىڭىز بىلەن ھەپىلىشىپ قۇيىسىز.سىز نىمە ئۈچۈن ئۆيىڭىزگە قايىتقاندىن كىيىن دەرھال  ئۈزىڭىزنىڭ ئەڭ چىرايلىق كىيىملىرىڭىزنى سېلىۋىتىپ،ئەڭ سەت،ئەڭ ئادەتتىكىچە كىيىملىرىڭىزنى كىيىپ،چاچلىرىڭىزنى پاخپايىتقان ھالدا ئۆيىڭىزدە مېڭىپ يۈرىسىز؟بەلكىم بۇنداق قىلسام ئىش قىلىش قۇلايلىق،قىممەت باھالىق كىيىملەرنى ئۆيدە كىيسەم كونىراپ كىتىدۇ دەپ ئويلىشىڭىز مۇمكىن.ئىرىڭىز سىزنىڭ بۇ ئۆمرىڭىزدىكى ئەڭ يېقىن ئادىمىڭىز تۇرسا ،سىز ئۈزىڭىزنىڭ ئەڭ  سەت ھالىتىڭىزنى ئۇنىڭغا قالدۇرۇپ قۇيامسىز؟شۇنى ئۇنتۇپ قالماسلىقىڭىز كېرەككى تويدىن   كىيىنكى ئەرلەرمۇ ئوخشاشلا سىزنىڭ گۈزەللىكىڭىزگە مۇھتاج،سىزنىڭ گۈزەللىكىڭىز يۇقالسا ئۇ   بەلكىم باشقا جايلاردىن گۈزەللىك ئىزدەپ قېلىشى مۇمكىن.گۈزەل ئاياللارنىڭ ئالدىدا ئەرلەر  مەڭگۈ  بىر ‹‹مۈشۈك››.ئەرلەرنى ئەسكى دىمەڭ،‹‹بەز يىيىش››مۈشۈكنىڭ تەبئى خۇسۇسىيىتى.

    نىكاھ خۇددى بىر چەكسىز دېڭىزغا ئوخشايدۇ،سىزنىڭ ئائىلىڭىز مۇشۇ دېڭىزدا لەيلەپ يۈرگەن بىر كىچىك قولۋاق،سىز مۇشۇ قولۋاقنىڭ كۆمتۈرۈلۈپ كەتمەسلىكى ئۈچۈن  كىچە-كۈندۈز تىرىشىسىز،بەدەل تۈلەيسىز،لىكىن سىز كۆپ ھاللاردا ئۈزىڭىزنىڭ مەۋجۇتلىقىنى ئۇنتۇپ قالىسىز،    ئىرىڭىزنىڭ ئۆيدىكى چاغدا كۆرىدىغىنى سىزنىڭ ئەتىگەندە ئورنىڭىزدىن تۇرغاندىن كىيىنكى   چاچلىرىڭىز پاخپايغان ،كۆزلىرىڭىز ئىششىق ھالىتىڭىز،يۈكۈنۈپ ئولتۇرۇپ پول سۈرتىۋاتقان ھالىتىڭىز،چاچلىرىڭىز ،يۈزلىرىڭىز مايلاشقان،پارقىرىغان ھالەتتە ئاشخانىدىن چىققان ھالىتىڭىز،بالىڭىز  ئۆينى قالايمىقان قىلىۋەتكەندە ئىككى قۇلىڭىزنى بىلىڭىزگە تىرەپ تۇرۇپ بالىڭىزغا تۈكۈرۈكلىرىڭىزنى چاچرىتىپ تۇرۇپ چېچېىلىۋاتقان ‹‹چىشى يولۋاس››تەك قورقۇنۇشلۇق ھالىتىڭىز،،،، ،   بۇلىدۇ.ئەرلەر نىكاھنى بىر خىل مەسئۇلىيەت دەپ ئويلىغاچقا ،ئۇ نۇرغۇن ئىشلارنى چىشىنى  چىشلەپ تۇرۇپ چىداپ ئۆتكۈزىۋىتىدۇ.دەل مۇشۇنداق پەيىتتە يەنە بىر كىچىك قولۋاق سىزنىڭ قولۋىقىڭىزغا ئاستا يېقىنلىشىپ كىلىدۇ،ئۇ قولۋاق شۇنچىلىك ئاددى،لىكىن ئۇ ئىنتايىن يېقىملىق،ئىنتايىن گۈزەل،ئۇ ئۈزىنىڭ مۇلايىملىقى بىلەن ئىرىڭىزنى ئاستا-ئاستا ئۈزىگە مەھلىيا قىلىۋالىدۇ،شۇنىڭ بىلەن ئىرىڭىزدىكى تەڭپۇڭلۇق بۇزۇلۇپ مۇشۇ ئىككى قولۋاق  ئارىسىدا  ئىرغاڭلاشقا   باشلايدۇ. بارا-بارا ئۇ ئۈزىنى تۇتىۋالالمايدىغان ھالغا يىتىپ قالىدۇ،بۇ چاغدا سىز ئىرىڭىزنىڭ ۋىجدانسىزلىقىدىن ئاغرىنىسىز،قارشى تەرەپنى  ھارغۇچە تىللايسىز،قارغايسىز،يۈرىكىڭىز  پىچاق  بىلەن تىلغاندەك  ئاغرىيدۇ،پۈتۈن دۇنيا سىز ئۈچۈن قاراڭغۇ تۇيىلىدۇ،كۆڭلىڭىزدە ‹‹مەن ھەتتا ئۈزۈمنىمۇ ئۇنتۇپ ،پۈتۈن ۋۇجۇدۇم بىلەن ئۆي تۇتىۋاتسام،بارلىقىمنى ئۇنىڭغا،باللىرىمغا ئاتاۋاتسام، ئۈزۈم يىمەي، ئۈزۈم كىيمەي ئۇنىڭغا ،باللىرىمغا يىگۈزىۋاتسام،كەيگۈزىۋاتسام،ئۇ نىمە ئۈچۈن ماڭا بۇنداق  ۋىجدانسىزلىق قىلىدۇ...››دەپ ئويلايسىز،لىكىن سىز ئىرىڭىزنىڭ سىرىتقا قارىشىنىڭ ھەقىقى  سەۋەبىنىڭ ئۈزىڭىزدە ئىكەنلىكىنى ئويلاپمۇ قويمايسىز،سىزنىڭ  دەل ئۈزىڭىزنى   ئۇنتۇپ پۈتۈن ۋۇجۇدىڭىز بىلەن ئۆي تۇتقىنىڭىز دەل ئۇنىڭ سىرىتقا قارىشىنىڭ سەۋەبى ،ئەگەر سىز ئۈزىڭىزنى يۇقۇتۇپ قويماي ،خاسلىقىڭىزنى ساقلاپ ،چىرايلىق كىيىنىپ بۇرۇنقىدەك گۈزەل ھالىتىڭىزنى ساقلىسىڭىز،بۇرۇنقىدەك مۇلايىم بولسىڭىز ئۇ سىرىتتىكى ‹‹بەز››لەرگە قاراپ قۇيارمىدى؟توي قىلغاندىن كىيىن نەدىمۇ ئۇنداق جىق ۋاقىت بولسۇن؟ دەپ ئويلىشىڭىز  مۇمكىن، بۇ ئىشلار ئۈزىڭىزنىڭ تەبئىتىگە باغلىق،كۈندە پەقەت ئۈزىڭىز ئۈچۈنلا 20مىنۇت ۋاقىت ئاجرىتىش ئىرىڭىزنى يۇقۇتۇپ قويغاندىن كىيىنكى ئازابنىڭ دەردىنى تارتىشتىن كۆپ ئاسان ھەم كۈڭۈللۈك.

     نىكاھ تۇرمۇشىڭىزدا سىز چۇقۇم ئۆز ئورنىڭىزنى تېپىۋېلىشىڭىز كېرەك،ئۈزىڭىزنىڭ  مۇقۇم خىزمىتى،تۇرمۇشقا بولغان كۆز قارىشى،ئۈزىڭىزگە تەۋە دوسىتلىرىڭىز ۋە ئۈزىڭىزنىڭ بىر تۇرمۇش رامكىڭىز بۇلىشى كېرەك،سىزمۇ بىر مۇستەقىل ئادەم بۇلۇش سۈپتىڭىز بىلەن ئۈزىڭىزنىڭ نىكاھتىكى  سىھرى كۈچىڭىزنى نامايان قىلىڭ، سىز مەڭڭۈ باشقىلارغا تاينىپ ياشاپ بولالمايسىز،نىكاھ پەقەت سىزنىڭ تۇرمۇشىڭىزنىڭ بىر قىسمى بولىشى كېرەك.ئەلۋەتتە، ئۈزىڭىز بۇلۇش بىلەن بىر ۋاقىتتا چۇقۇم ئىرىڭىزگە ئەڭ يېقىملىق كۈلكىڭىزنى ئاتا قىلىشىڭىز كېرەك، ئۇ ھامان سىزنىڭ تۇرمۇشىڭىزدىكى ئەڭ مۇھىم ئادەم.

    ئالدى بىلەن سىز ئۇنىڭ مۇلايىم ،كۈيۈنۈشنى بىلىدىغان ئايالى بۇلۇڭ،ئۇنىڭ كىيىملىرىنى پاكىزە يۇيۇپ رەتلەپ قۇيۇڭ،ئوخشىتىپ ئەتكەن تاماقلىرىڭىز بىلەن ئۇنى كۈتىۋېلىڭ،ئۆيلەرنى پاكىز-رەتلىك تۇتۇڭ،ئۇنىڭغا ئۆينىڭ ئىللىقلىقىنى ھىس قىلدۇرۇڭ،لىكىن ئىسىڭىزدە بولسۇنكى ئۇنىمۇ ئۆي ئىشلىرىغا ئاز-تۇلا يارىدەم قىلدۇرۇڭ،مۇشۇنداق قىلسىڭىز ئۇ سىزنىڭ مۇشۇ ئائىلە ئۈچۈن خېلى زور بەدەل تۈلەۋاتقانلىقىزنى بىلەلەيدۇ،ھەم سىزنى قەدىرلىشى كېرەكلىكىنى بىلىدۇ.

    نىكاھ تۇرمۇشىڭىزدا سىزنىڭ يەنە بىر رۇلىڭىز بار،ئۇ بولسىمۇ ئىرىڭىزنىڭ سىرداش دوستى بۇلۇش،ئۇ ئۈزىنىڭ بارلىق يۈرەك سۆزلىرىنى سىزگە دىيەلەيدىغان بولسۇن،بۇنىڭ ئۈچۈن سىز ئالدى بىلەن ئۇنىڭغا سەمىمى مۇئامىلە قىلىڭ،ئۇنىڭغا يالغان سۆزلىمەڭ،ئۇنىڭ ئىشەنچىسىگە ئىرىشىڭ،ئۇنى قانداق ۋاقىت بۇلىشىدىن قەتئىينەزەر دوسىتقا مۇھتاج بولسىلا بىرىنچى بۇلۇپ سىزنى ئويلايدىغان ھالەتكە كەلتۈرۈڭ،ئۈزىڭىزنىڭ ئۇنى مەڭڭۈ ئۇنى قوللايدىغانلىقىڭىزنى بىلدۈرۈڭ،ئۇنىڭ گەپلىرىنى ئەستايىدىللىق بىلەن ئاڭلاڭ،ئۇنى رىغبەتلەندۈرۈڭ.شۇنداق بىر كۈنلەر كىلىپ سىلەرنىڭ دىيىشكۈدەك ھېچقانداق گىپىڭلار قالمىغان چاغدا،سىلەرنىڭ نىكاھىڭلار زاۋاللىققا يۈزلەنگەن  بۇلىدۇ.

سىز يەنە ئۇنىڭ كىچىك قىزى بۇلۇڭ،ئازراق يولسىزلىق قىلسىڭىز،ئازراق بىشەملىك قىلسىڭىز ھېچقىسى يوق،لىكىن چەكتىن ئاشۇرىۋەتمەڭ،شارائىتقا،ئۇنىڭ كەيپىياتىغا قاراپ ئىش تۇتۇڭ.

ئۇنىڭ قۇچىقىدا خالىغانچە ئەركىلەڭ،ئەر سىزنىڭ،ئۇنىڭغا قانچىلىك ئەركىلىيەلىسىڭىز،نازلىنالىسىڭىز،ياخشى خۇلقىڭىز بىلەن ئۇنى ئۈزىڭىزگە مايىل قىلالىسىڭىز ،مانا بۇ سىزنىڭ قابىلىيىتىڭىزگە باغلىق،ئىشەنمىسىڭىز سىناپ كۈرۈڭ،سىز مۇشۇنداق قىلىشقا جۈرئەت قىلالىسىڭىزلا،ئازراق تاقەت قىلالىسىڭىزلا چۇقۇم ياخشى ئۈنۈمگە ئىرىشەلەيسىز.

    سىز يەنە ئۇنىڭ ئانسى سۈپتىدە،ئۇ قىيىنچىلىقلارغا دۇچ كەلگەن چاغلاردا ئۇ تەرەپتە تۇرۇپ قىيىنچىلىقلارنى بىللە يېڭىڭ،ئۇنىڭ بىلەن قىيىن ئۆتكەللەردىن بىللە ئۈتۈڭ،ئۇنىڭغا پۈتۈن ئالەم ئۇنىڭدىن ۋاز كەچسىمۇ،ئۈزىڭىزنىڭ ئۇنىڭدىن مەڭڭۈ ۋاز كەچمەيدىغانلىقىڭىزنى،ئۇنى بىر ئۈمۈر   قوللايدىغانلىقىڭىزنى بىلدۈرۈڭ.

  ھەي ئايال! ئورنىڭىزدىن دەس تۇرۇڭ،ھەرگىزمۇ نىكاھ كارۋىتىدا ھۇرۇنلۇق بىلەن يېتىپ   قالماڭ!

   ھەي ئايال! ئورنىڭىزدىن دەس تۇرۇڭ،ھەرگىزمۇ يېنىڭىزدىكى بەخىتنى باشقىلارغا تارىتقۇزۇپ  قويماڭ!
