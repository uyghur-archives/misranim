                              ئېيتقۇم كىلۋاتىدۇ

     بىر ئوغۇل بالىنىڭ ئانىسى بەكلا تەلەپچان بۇلۇپ، ھەمىشە ئۇنى قائىدە يۇسۇنلۇق بۇلۇشقا ئۈندەيدىكەن. ئەمما بالا تېخى 3 ياشتا ئىكەن.
بىر كۈنى بۇ ئائىلىدىكىلەر كەچلىك زىياپەتكە قاتنىشىپتۇ. بىر چاغدا ھېلىقى بالا«ئانا سىيگۈم كەپ كەتتى»دەپ ۋارقىراپ، ھەممەيلەننىڭ دىققىتنى تارتىپتۇ. ئانىسى بالىنىڭ بۇ قىلقىدىن بەكلا ئوڭايسىزلىنىپتۇ. ئۆيگە قايتقاندىن كىيىن ئانىسى بالىنىڭ ھەر قانداق چاغدا سىيگۈسى كەلسە «ناخشا ئېيتقۇم كىلۋاتىدۇ» دىيىش لازىملىقىنى تاپىلاپتۇ.
بىر نەچچە كۈن ئۆتكەندىن كىيىن، بالىنىڭ چوڭ دادىسى ئۇلارنى يوقلاپ كەپتۇ. بالا چوڭ دادسىغا بەك ئامراق ئىكەن. كەچتە بالا چوڭ دادسىدىن چۆچەك ئاڭلاپ ئۇخلاپ قاپتۇ. بىر نەچچە سائەتتىن كىيىن ئويغىنىپ:«بوۋا، ناخشا ئېيتقۇم كىلۋاتىدۇ»دەپتۇ. بوۋاي، بۇ كىچىدە باشقىلارنى ئويغىتۋەتمىسۇن دەپ نەۋرىسىگە:«ئوبدان بالام، ناخشا ئېيتقۇڭ كەلگەن بولسا مىنىڭ قۇلىقىمغا ئېيتقىن»دەپ پىچىرلاپتۇ.

                                              يۈزى قېلىن تىلەمچى

     تىلەمچى بىر ئۆينىڭ دەرۋازىسىنى قېقىپتۇ، ئۆيدىن بىر موماي چىقىشىغا سەدىقە سوراپتۇ:
-ماڭا ئىچىڭىز ئاغرىسۇن، يېنىمدا بىر تىيىن پۇلۇممۇ يوق، قورسىقىممۇ ئاچ، ماڭا ئازراق سەدىقە بەرگەن بولسىڭىز...؟موما، دەپتۇ تىلەمچى.
-ۋاي بىچارە، دەپتۇ موماي ، مەندە پۇلمۇ يوق، يەيدىغان نەرسىمۇ يوق. ناۋادا كىيىملىرىڭنى ياماتماقچى بولساڭ، بۇنىڭغا ياردەم قىلالايمەن دىسە،
     تىلەمچى يانچۇقىدىن بىر دانە تۈگمىنى چىقىرىپ:
-ئۇنداق بولسا ناھايىتى ياخشى گەپ ئىكەن، ماڭا مۇشۇ تۈگمىنى قاداپ بىر چاپان تىكىپ بېرىڭ- دىگۈدەك..

                                                  ھەربىي مانىۋېر
  
      بىر تۈركۈم يېڭى ئەسكەرلەرنى زەمبىرەك ئېتىش مەشىقى قىلدۇرۇش ئۈچۈن بىر ئېتىزغا ئاپىرىپتۇ. بىرىنچى ئەسكەر زەمبىرەكنى شۇنداق ياخشى دەل جايىغا ئېتىپتۇ. ئىككىنچى ئەسكەر بەكلا ھۇدۇقۇپ كىتىپتۇ-دە، نىشاننى توغىرلىمايلا ئېتۋېتىپتۇ. زەمبىرەك ئۇقى ئۇدۇل بېرىپلا، يېقىن جايدىكى قۇناقلىققا چۈشۈپ پارتىلاپتۇ. قاراپ تۇرسا قۇناقلىقتىن كىيىملىرى كۈيۈپ جۇل-جۇل بۇلۇپ كەتكەن بىرسى يۈگۈرۈپ چىقىپ:
-توۋا قىلدىم خۇدايىم، تۆت باش قۇناق ئوغىرلىغانغىمۇ زەمبىرەك ئاتامدۇ،-دىگۈدەك.

                                      ئىشەكنىڭ گىپىنى قىلساڭلارچۇ

      بىر نامرات ئەر-خۇتۇن بۇلۇپ، بۇلارنىڭ بالسىنىڭ يېشى چوڭىيىپ توي قىلدىغان ۋاقتى بۇلۇپ قاپتۇ. بىر كۈنى كەچتە بۇ بالا كۆزىنى يۇمۇپ ياتسا بۇ ئەر-خۇتۇن بالسىنى ئۇخلاپ بولدى دەپ ئويلاپ، بالىمىز چوڭىيىپ قالدى ئۆيلەپ قويساق بۇلاتتى، ھازىر ئىقتىسادى ئەھۋالىمىز تازا ياخشى ئەمەس، شۇڭا ئېغىلدىكى ئېشەكنى سېتىپ پۇلىغا تۇيىنى قىلىپ بېرەيلىمىكىن دىيشىپتىكەن.
     بۇ بالا خوشاللىقتا ئورنىدىن تۇرۇپ كىتىپتۇ، شۇنىڭ بىلەن بۇ ئەر خۇتۇن بالسىنى خىجىل بۇلۇپ قالمىسۇن دەپ باشقا پاراڭلارنى قىلىپتىكەن. بۇ بالا دادا-ئاپا باشقا گەپنى قۇيۇپ بايىقى ئىشەكنىڭ گىپىنى قىلساڭلارچۇ،-دەپتۇ.

                                                         دوختۇر

      بىر ئادەم ئۇقۇشماستىن بىر بۇتۇلكا بېنزىننى ئىچىپ ساپتۇ. ئۇ ناھايىتى جىددىيلىشىپ دوختۇرخانىغا بېرىپتۇ.
-دوختۇر، ھىچقىسى يوق، جىددىيلىشىپ كەتمەڭ، پەقەت بىر ھەپتىگىچە تاماكا چەكمىسىڭىزلا بۇلىدۇ، -دەپتۇ.
