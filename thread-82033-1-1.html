ساڭا بولغان نەپرەت مىڭ گەز ئۆرلىسىمۇ ساڭا بولغان سېغىنىش گەزدىنمۇ ئۈستۇن تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
سەن مەندىن ساراڭلارچە بىزار بولساڭ مەن تەلۋىلەرچە ساڭا ئىنتىزار تۇرسام سىنى قانداق ئۇنتۇيمەن جېنىم .
سەن رەھىمسىزلەرچە مىنى تاشلاپ قويساڭمۇ مۇھاببەت بىلەن يۇغۇرۇلغان قەلبىم ساڭا تارتىشىپ تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
سەن ئايرىلىشنى تەلەپ قىلغىنىڭدا ئۈلۈم كۈزۈمگە كۈرۈنگىنىدە سەن مازاق قىلغاندا بىزار بولغاندا سىنىڭمۇ بىر ۋاقىتلاردا يۈرىكىم ئاغرىپ كىتىۋاتىدۇ ،ماڭا كىلىشمەسلىك كىلىپ قالسا دوستۇم ئارقىلىق ساڭا تىگىشلىك نەرسىلەرنى تاپشۇرۋال دەپ ئېلتىجا قىلغىنىڭ يادىمدا تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
سەن بىر ماكانغا يۇشۇرنىۋېلىپ بارلىق ئالاقەڭنى ئۈزۈپ مىنى خۇددى قوقاستا كۆيگەندەك ھالەتكە قويغىنىڭدا دەرۋازام ئالدىدا كېچە كۈندۈز مىنى كۈتۈشلىرىڭ،ئېنتىرنىت ئالدىدا ماڭا زارىقىشىڭ ئېنىق ئىسىمدە تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
مىنىڭ ئاۋازىمدىن شۇنچە بىزار بولغىنىڭدا ئەينى چاغدا ئاۋازىمنى بىر ئاڭلاش ئۈچۈن كېچە كۈندۇز ئۇقۇمىنى ئۇنتۇغانلىقىڭ ھېلىمۇ يادىمدا تۇرساسىنى قانداق ئۇنتۇيمەن جېنىم .
ماڭا قاراشقىمۇ كۈزۈڭ يوق بۈ كۈنلەردە سائەتلەپ جامالىمدىن كۆز ئۈزمىگىنىڭ دىلىمدا تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
يىغلاپ يېلىنسام نەپرەت بىلەن ئالايغىنىڭدا يىغلىما جېنىم سەن يىغلىساڭ مىنىڭ يۈرىكىم يېرىلىپ كېتىدۇ دىگەنلىرىڭ دىلىمدا تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
بارلىق ئاللا بەرگەن ئەيىپلىرىم بىلەن ھاقارەت قىلغىنىڭدا ئەينى چاغدا بۇلار نىمە ئىدى،ماڭا سىنىڭ كەلگۈسۈڭ كېرەك دىگەن ۋەدىللىرىڭ دىلىمدا تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
مەن نامرات دىھقان قىزى دىسەم مەن ئۈچۈن سىنىڭ ھەر كۈنى ئۆيدە كۈتۈشۇڭ موھىم ،مەن سىنىڭ كۈتۈشىڭگە ئېھتىياجلىق دەپ يېلىنىشلىرىڭ دىلىمدا تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
مىنى كۈرۈشكىمۇ رايىڭ يوق بۇ كۈنلەردە مىنى كۈرۈش ئۈچۈن يىراق يەرلەردىن ئۇچۇپ كىلىشلىرىڭ دىلىمدا تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
بىر بىرىمىزگە پىچىرلاپ بەرگەن ۋەدىلەر بۈگۈنكى كۈندە سەن ئۈچۈن بىر پوڭغىمۇ ئەرزىمىگەن بىلەن ئەينى چاغدىكى ۋەدىلەر بىلەن، قەسەملەر بىلەن مىنى كۆمگىنىڭ ئەمىليەت تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
ۋىجدانسىزلىغىڭ ئامىلى سەۋەپلىك ئۈلۈم پۇراپ بالنىستا ياتقىنىمدا سەن گەرچە شۇنداق مۇز ھالەتتە كەمسىتىش نەزىرى بىلەن ماڭا باققان بولساڭمۇ ئەينى ۋاقىتتىكى يۆتىلىمدىنمۇ ئەنسىرەپ كىرپىك قاقمىغان چاغلىرىڭ كۆز ئالدىمدا تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
سەن مىنىڭدىن سىنى قۇيىۋىتىشىڭنى ئۈتۈنۈپ تۇرساڭمۇ ساڭا يۈرەك شۇنچە كۆنگەن تۇرسا خىيالىمدىن قوغلىساممۇ سىمايىڭ كۆز ئالدىمدا تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
يالغۇز قالدۇرۇپ ،كېچە كۈندۇز ئازاپلاپ ئۇيقۇلاردىن  بىدار قىلغان بولساڭمۇ ،يۈرەكلىرىم تارتىشىپ نەپەسكىمۇ كۈچۈم يەتمىگەن بولسىمۇ ،سەن ياخشى كۆرىدىغان سەن قاراپ تەلمۇرىدىغان قارا كۆزلىرىمدىن ياش قۇرىماي جامالىم سۇلۇپ نادامەتتە قويغان بولساڭمۇ قەلبىم تىنىمسىز سىنى ئىزدەپ تۇرسا سىنى قانداقمۇ ئۇنتۇيمەن جېنىم.

سەن باشقا بىر ماكاننى قۇرۇشنى شۇنچە ئارزۇ قىلىۋاتقان بولساڭمۇ بىز قۇرغان ماكان مۇڭلىنىپ تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
يۈرەكنىڭ رىشتىللىرى تاس تاس قىلپ ئۈزۈلمەكتە كۆزنىڭ يېشى تاراملاپ قۇيۇلماقتا،كىشىلەر مەسخىرىلىك ماڭا قاراپ كۈلمەكتە،دىلىم سەندىن رەنجىمەكتە ،ۋۇجۇدۇم قىلمىشلىرىڭدىن سەسكەنمەكتە ،لىكىن قەلبىمدە ساڭا بولغان مۇھاببەت ناھايىتىمۇ بىزەڭلەرچە سەندىن كەچمىسە سىنى قانداق ئۇنتۇيمەن جېنىم .
قايسى پەسلەر ساڭا قانداق قەسلەرنى قىلدى بىلمىدىم ،سىنى ئازدۇرغانلار قان يۈتسۈن ،سىنىڭ ئەركەكلەرگە خاس بولمىغان ۋىجدانسىزلىغىڭ ،مەسئۇليەتسىزلىكىڭ ،شۇنچىلىك ياخشى كۆرۈشكە ئەرزىمەيدىغان ئەركەك ئىكەنلىكىڭنى نامايەن قىلىپ تۇرسىمۇ مۇنۇ مەندىكى پەس كۈڭۈل سەندەك ئادەمدىن ھىچ ۋاز كېچەلمىسە  سىنى قانداق ئۇنتۇيمەن جېنىم .
ئەمدى سەن يېنىمغا قايتىپ كەلمەيسەن قايتىپ كەلگەندىمۇ سىنى قۇبۇل قىلمايدىغىنىم ئېنىق ،ئازلا قالدى سىنىڭ مەندىن ئازات بولىدىغان كۈنلىرىڭگە ،لىكىن ئۈزۈلمەي قالغان يەنە ئازراق رىشتىم بار تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
سىنىڭ كۆرسەتمىگەن ھاقارەتلىك باھانىللىرىڭگە چىدىغۇچىلىكىم قالمىغان بولسىمۇ ھەر قېتىم يېنىمدىن ئايرىلغىنىڭدا قان زار يىغلىساممۇ قايرىلمىغىنىڭ ماڭا ئازار بولسىمۇ لىكىن سىنىڭ ئەينى ۋاقىتتا ماڭا شۇنچە بىر يېلىنىش ،ئىنتىزارلىق بىلەن قايرىلىپ بىر قاراشلىرىڭ مىنىڭ تەقدىرىمنى ئۆزگەرتكەن تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
سىنى پەسلەر قەس قىلدى دەپ مۇھاببىتىمىزنى قۇتقۇزماقچى بولدۇم لىكىن سەن تەسىرلەنمىدىڭ ،قان زار يىغلىدىم تەسىرلەنمىدىڭ ،ئايىقىڭغا يىقىلدىم تەسىرلەنمىدىڭ ،شىرىن ۋاقىتلارنى ئەسلەتتىم تەسىرلەنمىدىڭ،ۋەدە قەسەملەرنى ئەسلەتتىم تەسىرلەنمىدىڭ ،ئەكسىنچە منى دوس دۈشمەن ئۇرۇق تۇققانلىرىم ئالدىدا شەرمەندە قىلىپ باش كۆتەرگۈسىز قىلدىڭ ،نىمە دىگەن پەس كۈڭۈل مەندىكى يانمىغان ،ئەي ئىنسان مۇشۇ شەرمەندىچىلىككە سالغان ھەر قانداق بىر ئەركەك قىلغانلىرى ئۈچۈن مەسئۇليەتنى ئۈستىگە ئالىدۇ ،لىكىن سەن مىنى مەڭگۈلۈك نادامەتتە قۇيۇپ ماكانىمىزدىن بوسۇغا ئاتلاپ چىقىپ كەتتىڭ،بۇ رەھىمسىزلىكىڭ كۆز ئالدىمدا تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
بىز كۈلۈشمىگەنمىدۇق،سۈيۈشمىگەنمىدۇق،ۋەدىلەشمىگەنمىدۇق،جاپانى بىللە تارتىپ بىللە ھۇزۇر كۆرمىگەنمىدۇق ،رەھىمسىزلىكىڭنى ئۈستۈمگە تاغ قىلىپ تاشلىغان تۇرساڭ سىنى قانداق ئۇنتۇيمەن جېنىم .
سەندىن ساڭا ئۆمۇرلۇك ھەمرا بۇلۇپ ،بەختىڭنى تىلەپ سىنى ئاياللىق بۇرچۇم بىلەن كۈتۈشنى ،بەخىت بىلەن سىنى قوينۇڭغا ئېلىشىڭنىلا تىلىگەن ئەمەسمۇ ،كۆز يۇمساممۇ ،چۈشۈمدىمۇ ،بىزنىڭ ئاشۇ ئىللىق ھازىرقى مۇڭ پۇراپ تۇرىدىغان ماكانىمىزدۇ سىنىڭ بۇرۇنقى ئوماق كۈلكەڭ،ئوتلۇق نەپەسلىرىڭ ،ئەقىل پۇراپ تۇرىدىغان سۆزلىرىڭ ،ھاياتقا بولغان ئارمىنىڭ ،ماڭا بولغان مۇھاببىتىڭ ،مەن ئۈچۈن ھەممىدىن ۋاز كىچىپ ئەلنى تەسىرلەندۈرگەن جەسۈرلۇغۇڭ تەسۋېرلىگۇسىز ھالدا ماڭا نىمە ئۈچۈن دىگەن تۈمەنمىڭ جاۋاپسىز سۇئاللارنى قالدۇرغان تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
كۈنلىرىم شۇنداق تەستە ئۆتمەكتە،ئازاپلىنىش،ئۈمۈتسىزلىنىش ،نەپرەتلىنىش،جىددىلىشىش ،قېسىلىش ،يىغلاش ،روھسىزلىنىش،مادارسىزلىنىشتەك بارلىق ئامىللار مىنى شۇنچە قىيناپ تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
سىنى قانداق ئۆتمۈشكە قايتۇرۇپ كىلىشنى بىلمەيمەن ،خۇددىى مىنى نىمىگە تاشلىغىنىڭ ئۈزەڭگە نامەلۈم بولغىنىدەك،سىنى چوڭقۇر سۈيۈشۈم سىنى سەسكىندۇردى بولغاي،لىكىن سەن ئارمان قاناتلىرىمنى سۇندۇردۇڭ،ئارزۇ ھەۋەسلىرىمنى بەربات قىلدىڭ ،ياشلىقىمنىڭ ئاخىرقى پەسىللىرىگە قانلىق تارىخلارنى يېزىپ ئۆمۈرلۈك نادامەتتە ،پۇشايماندا قالدۇردۇڭ،ھالىم شۇ تۇرساسىنى قانداق ئۇنتۇيمەن جېنىم .
مەيلى نەپرەتلىنەي ،مەيلى سۆيەي،سەن قەلبىمگە يېزىلغان تارىخ،سىنىڭ نامىڭنىڭ ئۆزى مەن ئۈچۈن مەڭگۈلۈك ئەلەم تۇرسا سىنى قانداق ئۇنتۇيمەن جېنىم .
ئويغانغىن جېنىم ،قەلبىمنىڭ سەن ئۈچۈن يېنىپ تۇرغان ئاخىرقى ئۈمىدىمنىڭ نۇرىدا يېنىمغا قايتىپ كەلگىن ،قەلىب دەرۋازام ساڭا يېپىلىپ قايتۇرۇپ كەلگۈسىز پۇشايمانغا قېلىشتىن بۇرۇن يېنىمغا قايتىپ كەلگىن .

قەلبىمدىكى مۇھاببەت ئازاپتىمۇ ئېڭراپ تۇرسا مىنى مۇھاببەتكە ئۆز قۇلۇڭ بىلەن مەجبۇرى سۆرەپ كىرىپ ئەمدى ساڭا مۇھاببەت ئويغانغان چاغدا ئەرىزىمەس ئەخلەتتەك تاشلاشلىرىڭنى يەنە بىر قتىم ئويلانغىن،چۈنكى قەلبىم نۇمۇسنى قايرىپ تۇرۇپ سەندىن تىلەمچىلەردەك مۇھاببەت سوراۋاتىدۇ،ساڭا بولغان مۇھاببىتىم تۇنجۇققۇچە يېنىمغا قايتىپ كەل جېنىم.
ئەيىپلىمە ،بىزار بولما  سىنى قانداق ئۇنتۇيمەن جېنىم .
