              ۋېيشىن دەپ بىر نەرسە چىقىپتۇ،خەنزۇچە ئاتالغۇمۇ ناھايتى تىز قوبۇل قىلىنىپتۇ،شۇنىڭغا قارىغاندا بۇ خېلى دەۋىر سۈرۈپ قالغۇدەك.شۇڭا مەن تىل بايلىقىمىزنىڭ نامايەندىسى سۈپىتىدە بۈگۈن مۇشۇ نەق مەيداندا بىر ئۇيغۇرچە ئىسىم قويۇپ بېرەي،بىكاغا......
         微信-ئابدىكىرەم ئابلىزچە تەرجىمىسى -سۈنئىي ھەمراھ ئۇچۇرى.
         كەسپىي تەرجىمىدە-مىكرو ئۇچۇر دىسەك بولغىدەك.
                             ئۇيغۇر تىلىمىزدا بىز بۇنى تېخىمۇ يېقىملىق ئاتىيالايمىز،ئاڭلاڭ ھە،مەن بۇنى ئاممىبابلاشتۇرۇپ -ئۈن دىدار-دەپ ئاتاشنى مۇۋاپىق كۆردۈم،قېرىندىشىم ئەمدى ئۆتۈنەي ۋېيشىن دىمەڭ!
        *******                      ********                  ********                      ***********
                    خىزمەتداش قىزچاق ئۈيۈمگە ئۈسۈپلا كىرىپ كەلدى،خوشلۇقۇمدا ئولتۇرۇشقىمۇ تەكلىپ قىلماي ئۈن دىدارنىڭ ئىسمى ھەم ئويلىغانلىرىمنى سۆزلەپ كىتىپتىمەن،ئۈن دىدار تىلغا ئېلىنىشى بىلەن قىزچاق بار ئۈنىنى قويىۋېتىپ يىغلاشقا باشلىدى،ھەي ،،بىز ئىسمىنى ئەمدى قويغان بىلەن چىقارغان چاتىقى بوش ئەمەسكەن بۇنىڭ،قىزچاق تېلفونىدىكى ئۈن دىدار ئۇچۇر ساندۇقىنى ئېچىپ بىرتەرەپتىن ئۇچۇرلارنى كۆرسەتكەچ،بىر تەرەپتىن بۇ ئىشنى ھېچكىمگە دىمەسلىكىمنى ئۆتۈنۈپ بۇ كىچىككىنە ئالاقە ۋاستىسىنىڭ تەغدىرىدە پەيدا قىلغان زور بۇرلىشىنى سۆزلەشكە باشلىدى............
                    ئاقىللار <ئىككى ئادەمنىڭ بىلگىنى سىر ئەمەس> دەپ بەكمۇ توغرا ئېيتىپتىكەن.گەپنىڭ ئاخىرى تۈگىمەيلا يازغۇچىلىق كىسىلىم قوزغىلىشقا باشلىدى،ئاندىن يەنە مۇشۇ مۇنبەرگە ئەكىلىپ تاشلىدى،بۇ يەردەخەقنىڭ سىرىنى يايماقچى ئەمەسمەن،بىراق ھازىر مودا بولۇۋاتقان ئۈن دىدار ئۇچۇر سىستىمىسىنىڭ ئامىتىنى ھەم ئاپىتىنى كۆپچىلىك ئاڭلاپ توغرا ئىشلىتىشنى ئۈگىنىۋالار دىگەن ئۈمىدتە قەلەم تەۋرەتتىم،كۆپچىلىكنىڭمۇ خىزمەتدىشمنىڭ سىرىنى ياخشى ساقلىشىنى ،ئوقۇپ بولۇپ ھېچكىمگە دىمەسلىكىنى ئۈمىد قىلىمەن!
                      ******                                    *******                                     ******
                             كەسپ ۋە تۇرمۇشتا ئۇتقان- مۇكەممەل قىز

            بۇ خىزمەتدېشىمغا قاراپ ھەيران قالىمەن،ئەمىليەتتە ئۇ27-28ياشلارنىڭ قارىسىنى ئەمدى ئالغان بىر بالالېكىن بىر رەھبەر،ئۇ بىر سەنئەتكار لېكىن بىر تالانت ئىگىسى.كىچىدىمۇ ئۇخلىمامدىكىن دەيمەن بۇ قىزنى،كىچە-كىچىلەپ چىرىغى ئۆچمەيدۇ،بىز خوشنا بولغاچقا سىنچىلاپ قارىسا ئۇنىڭ كومپىيوتىرنىڭ ئالدىدا دۈمچەكتەك ئولتۇرۇپ خەت بېسىۋاتقىنىنى كۆرگىلى بولىدۇ.ئىشقىلىپ ھەممىلا ئىشقا ئارلىشىدۇ،ئۇنىڭ بۇ مەكتەپكە كەلگەندىن بۇيان تەنتەربىيە دەرىسى ئۆتكىنىنى كۆرمىدىم،باشقا ھەممە ئىش قولىدىن كىلىدۇ،مائارىپتا قوش قىرلىق بولۇڭلار دەپ بولغۇچە يەتتە-سەككىز قىرلىق بولۇۋالغان بۇ ياشلار ئادەمنى چارچىتىپ.سەنئەت ئوقۇتقۇچىسى بولغىنى بىلەن قاپىقى ئاجايىپ يامان ،چىرايىنىڭ سۈرلىكلىكىدىن يىراقتىن يەرنىڭ تاناۋىنى تارتىپ قاسىلداپ مېڭىپ كەلسە،ئولتۇرۋاتقان ئورنىڭىزدىن ئىختىيارسىز تۇرۇپ كىتىسىز...گۇرۇپپا باشلىقى بولغۇچە مەكتەپ مۇدىرى بولسا تازا يارىشاتتى دەيمەن بەزىدە.
                  قابىلىيەتلىكلىكىگە يارىشىپ باينىڭ قىزى،بىر تۇققان قېرىنداشلىرىنىمۇ تىنغىلى قويمايدۇ ئۈگىنىڭلار،،تىرىشىڭلار دەپ،ئائىلىدىكىلىرى ھەممىسى يۇقىرى دەرىجىلىك ئورۇنلاردا خىزمەت قىلىدۇ،ئۆزىگە ھەم ئائىلىگە بولغان ئىشەنچ بۇ قىزدا شۇنداق بىر سالاپەتنى پەيدا قىلغانكى ھاي-ھاي....قەددىنى رۇسلاپ مېڭىشلىرىغا،ئازراقمۇ پۈكۈلمەي تىك ئولتۇرۇپ ماشىنا ھەيدەشلىرىگە دادىسىنىڭ زوقلىنىپ قارىغانلىرىنى كۆرۈپ قالىمەن،مەن ئالاي دەمسىز،ماۋۇ گەپنى ئۇ ھەرگىزمۇ ئاڭلاپ قالمىسۇن.چەتئەلدە ئوقۇۋاتقان چاي ئىچكۈزۈپ قويغان لايىقى بار،ئۇنىڭ دىيىشىچە ھەممە نەرسىنىڭ خىلى بولارمىش..............

                      مانا بۇ مىنىڭ ئەسىرىمدىكى باش پىرسوناژ ئوبرازى،ئەمدى مەن ئۇنىڭ تۆۋەنكى بايانلىرىنى <مەن>نىڭ تىلىدا يېزىپ دۋاملاشتۇرماقچى،مەن دەپ يازسام مىنىمىكى دەپ قايناپ نەسىھەت قىلماڭ،ئەسەر تۈگىگەندە ئاندىن پىكىر قىلىڭ،بۇ تىمىدا مىللەت مەسىلىسى مەۋجۈت ئەمەس،ئاشۇرما بايانلارنى،بەزى تەسۋىرلەرنى شۇ پېتى قوبۇل قىلىۋالماڭ،ئەمەلىيەت نۇقتىسىدىن ئەمەس،ئەدەبىيات نۇقتىسىدىن ئەسەرگە باھا بېرىڭ،چۈشەنمىگەنلەر ئىنكاس يازمىسىڭىزمۇ كۆڭلۈمگە ئالمايمەن،باشقىلارغا بىر تامچە سۇ بېرىش ئۈچۈن ئۆزىڭىزدە بىر چىلەك سۇ بولىشى كېرەك،ئەدەبىي چۈشەنچىسى يۈزەكى قېرىنداشلار باھا بەرمىسىڭىزمۇ بولىدۇ،پىكىرىڭىز بولسا ئوچۇق ئوتتۇرغا قويۇڭ،بولمىسا ئۆزىڭىزنى بېسىۋېلىڭ...ماسلاشقىنىڭىزغا رەخمەت!

                                             ئۈن دىداردا ئۆزىنى  تونۇغان قىز

              نامازىمنى ئوقۇپ بولۇپ كۈندىلىك ئادىتىم بويىچە مەھەللە كوينى سۈپۈرۈپ كىرگىنىمدە ۋاقىت بىر يەرگە بېرىپ قالغانىدى،تۇيۇقسىز يىغىن بارلىقى يادىمغا يىتىپ ناشتا قىلماي مەكتەپكە چاپتىم،يېرىم كۈنلبك ئالدىراش خىزمىتىمنى تاماملاپ ئىشخانا ئالدىغا كەلگىنىمدە ئىشخانىدىن ئاڭلانغان ۋاراڭ-چۇرۇڭ كۈلكە ئاۋازىدىن تېخىمۇ پەيلىم بۇزۇلۇپ ئىشىكنى ئىتتىرىپلا كىردىم،كۈلكە ئاۋازى قايناۋاتقان قازانغا سۇ قۇيغاندەك بېسىقىپلا كەتتى،قوسىقىمنىڭ ئاچلىقىدىن گەپمۇ خوشياقمىدى.
           -بىر يىڭى سېستىما چىقىپتىكەن،ئادەتتىكى تېلغونلار قوللىمايدىكەن،سىزنىڭ ئالما بولغاندىكىن قوللايدۇ چۈشۈرۋېلىڭە-دىدى پاتىگۈل مۇئەللىم ئالدىمغا كىلىپ.قوسىقىمنىڭ ئاچقىنىنى بانا قىلىپ قاچاي دىگۈچە، ئۆيگە بېرىپ بولغۇچە تېخىمۇ ئېچىپ ئۆتۈپ كىتىشىنى باھانە قىلىپ بۇلار مىنى تۇتىۋالدى،ئادەتتە ئادەم كۆپ جايلاردا تاماق يىيىشنى ياقتۇرمايتتىم،ئاشخانىلارنىڭ تامىقىنى تېخىمۇ ياقتۇرمايتتىم،بۈگۈن مىنى قارا بېسىپتۇ،تېخى ماۋۇ خوتۇن ماڭا بىر نىمە چۈشۈرۈپ بېرىمەن دەپ تېلفونۇم بىلەن ھەپىلىشىپ چىشىمغا تەگكىنى ھەي....
                         ھارغىنلىقتا بىركۈننى تولىمۇ تەستە ئۆتكۈزدۈم ،ئۆيدىكى ئىشلىرىمنى تىزلا تۈگىتىپ كارۋىتىمغا ئۆزەمنى ئاتتىم،ئۇھ ...ئوڭدا يېتىپ تېلفونۇمدىكى كەلگەن تېلفون ۋە ئۇچۇرلارنى كۆرۈپ چىقتىم،،تۇيۇقسىز يىڭى چۈشۈرگەن سىستىمىدىن بىرىنىڭ ئەسسالامۇ-ئەلەيكۇم دىگەن ئۇچۇرىنى تاشۇرۋالدىم،بىلەرمەن خىزمەتداشلىرىم تاماق يىگۈچە بولغان ئارلىقتا سىستېمىنى چۈشۈرۈپ،سەنئەت ساھەسىدىن دوسلارنى قاچىلاپمۇ ئۈلگۈرگەنىدى،چىرايلىق بېرىلگەن سالامنى ئىلىك ئالماي بولمىدى،ئىختىيارسىز پاراڭغا چۈشۈپ كەتتۇق ،كەسپ توغرىسىداپاراڭلىشىۋېتىپ قارشى تەرەپنىڭ بەزى يۇمۇرلۇق ئوخشىتىشلىرىغا كۈلمەي تۇرالمىدىم،،تونۇشمىساقمۇ ناخشىلارنى ئېيتىپ بېرىشلىرى تېخى ،،،،توۋۋا مەنمۇ كۈلىدىكەنمەنغۇ مانا،ماۋۇ ئۈن دىدارمۇ خېلى بولىدىغان نەرسىمۇ قانداق دىگەن ئويلار بىلەن تېلفوننى چىڭ تۇتۇپ ئۇيقۇغا كىتىپتىمەن......
