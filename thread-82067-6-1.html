                                              ئۈن دىداردىن ئۆز دىدارنى ئىزدەپ........

            ئۈن دىداردىكى ئۈزۈلمەس سۆھبەت بىزنى تېخىمۇ ئىنتىزار قىلدى،كۆرۈشمەكچى بولدۇق.يۈرەكلىرىم قېپىدىن چىقىپ كەتكۈدەك سوقسىمۇ يەنىلا ئۇنىڭ بىلەن كۆرۈشىشنى خالايتتى.ھېلىمۇ تەتىلگە ئاز قالغان مەۋسۈم ئاخىرى بولۇپ قالغاچقا خىزمەتتىكى بوشاڭلىقىم ئانچە چېنىپ كەتمىدى،ھازىرقى ئۇنىڭغا بولغان قىززىقىشىم تۇرمۇشىمدىكى ھەممە نەرسىنى كۆزۈمگە كۆرۈنمەس قىلىۋەتكەن ئىدى،ئۈزەمنىڭ كىملىكىنى ،جەمئىيەتتىكى ئورنىنى،باشقا بىرىنىڭ ئامانىتى ئىكەنلىكىمنى پۈتۈنلەي ئۇنتۇغان ئىدىم ھەتتا قايسى كۈنى ئۇنىڭ مۇھەببىتىڭىز بارمۇ ،دىگەن سوئالىغا يۈزۈمنى داپتەك قىلىپ <يوق>دەپ جاۋاپ بەردىم...........
                  كۆرۈشمەكچى بولدۇق،ئۇنىڭ ئىشى چىقىپ قالغاچقا سىرتتا ساقلىماقچى بولدۇم.توپتوغرا ئالتە سائەت ئازراقمۇ ئۈمىدسىزلەنمەي ساقلىدىم،خەۋەر يوق.قانداق ماڭغىنىمنى بىلمەيمەن كىچە سائەت12بىلەن ئۆيگە قايتىپ كەلدىم.ئۈزەمنى تۇتىۋېلىپ تۇرۇپ ئۇنىڭغا <ئالدامچى>دەپ ئۇچۇر ئەۋەتتىم،دەرھال جاۋاپ كەلدى،ئۇ مىنىڭ سىرتتا شۇنچە ئۇزۇن ساقلىغىنىمنى بىلىپ ھەيران قالدى ،چۈشەندۈردى ،كەچۈرۈم سورىدى...ئاخىرى راست گەپنى قىلدى،ھاراق ئىچكەنلىكىنى ،مەس بولۇپ كەتكىنىنى ،ساقلاش جەريانىدا بىر ئۇچۇر ئەۋەتىۋەتكەن بولسام ئىشلارنىڭ بۇنداق بولۇپ كەتمەيدىغانلىقىنى،ھازىر قەيەردىلىكىمنى ،يېنىمغاكىلىدىغىنىنى ئېيتتى.....ئۇ ھاراق ئىچكەن ئادەمگە ئۆچلىكىمنى بىلەتتى،ئەلۋەتتە يېرىم كىچىدە مەس بىلەن كۆرۈشكىلى چىقىدىغان ئەقىلسىزلاردىن ئەمەسمەن،مۇشۇ ئورۇنسىز ئاۋارىچىلىقلارمۇ يىتەر،تېلفوننى ئىتپلا ئۇخلۇدۇم............
                       ئەتىسى تېلفوننى ئاچتىم،كىچىچە ئۇچۇر دىگەننى يېزىپ قويۇپتۇ.....بەك ھاكاۋۇر بۇ يىگىت،بىلىمىمۇ بار تەققى تۇرقىمۇ ناھايتى جايىدا ھال تارتسىمۇ يارىشىدۇ لېكىن بۇنداق تەكرار ئۇچۇر يېزىش ئادىتى پەقەتلا يوق ئىدى،تۇرۇپلا ئەنسىردىم،بەڭۋاش كۆڭلۈم ئۇنىڭ ساقلىقىنى بىلىشكە ئالدىرىدى،ئۇچۇر ئەۋەتتىم:
         -ساقمۇسىز؟-ئازراق ۋاقىت ئۆتكەندىن كىيىن جاۋاپ كەلدى:
-بەك مىجەزىم يوق،ھەممە يېرىم ئاغرىپ كىتىۋاتىدۇ،-بۇ گەپنى ئاڭلاپ تىپىرلاپ كەتتىم،نەدىلىكىنى سورىدىم.ئۇ ئاخشام بەك مەس بولۇپ كەتكەچكە ئۆيگە قايتالمىغىنىنى،ئاغىنىسىنىڭ بىر ياتاققا ئورۇنلاشتۇرۇپ قويغىنىنى،ئورنىدىن تۇرالمايۋاتقانلىقىنى،بولسا مىنىڭ مۇزدەك سۇ بىلەن ناشتىلىق ئەكىلىپ بېرىشىمنى ئېيتتى.ئاھ خۇدا،،نىمە دىگەن تەس ئىش بۇ،،،قانداق قىلاي مەن ئەمدى..............سائەتكە قارىدىم،12بولغانىدى.چۈشلۈك دەم ئېلىش ۋاقتى بولغاچقا مەكتەپتىن ئوقتەك چىقتىم،ئىچىملىك،گۆشنان،لۆڭگە،پايپاق،ئۇششاق-چۈششەك يىمەكلىك،مىۋە-چىۋە ئىشقىلپ تاللا بازىرىدىن ئۇنىڭغا مۇناسىۋەتلىك ھەممە نەرسىنى ئالدىم،ئۇدەپ بەرگەن ئادرسقا قاراپ توختاپ قالدىم، بارىدىغىنىڭ قانداق جاي،سەن كىم ،ئۈزەڭنى بېسىۋال خىياللىرىم مىنى توختىتىۋالدى.ئەقلىم ئوتتەك ھىسياتىمدىن غالپ كىلىپ بارماسلىق قارارىغا كەلدىم لېكىن ئۇنىڭ ھازىرقى ئەھۋالىنى كۆز ئالدىمغا كەلتۈرۈپ چىدىمىدىم،تېلفوننى قولۇمغا ئېلىپ نۇمۇس ھەم غۇرورۇمنى قايرىپ قويۇپ دوستۇمغا تېلفون قىلدىم ،بىر دوستۇم بىلەن ئۇرۇشۇپ قالغىنىمنى ،كۆرۈشكۈم يوقلىقىنى بىراق ھازىر ياردەم قىلمىسام بولمايدىغانلىقىنى دەپ ئۇنىڭ قولۇمدىكى نەرسىلەرنى ئاپىرىپ بېرىشىنى ئېيتتىم.....
       -ھازىرلا بارىمەن،يەنە نىمە لازىم؟-ئۇنىڭغا ئۇچۇر يازدىم،مىنى ئوبدانلا چۈشىنەتتى ،خاراكتېرىمدىن مىنىڭ بارمايدىغانلىقىمنى ئاللىبۇرۇنلا بىلىپ بولغانىدى:
-ئۆزىڭىز كىلەمسىز،-دەپ سورىدى.يالغان گەپ قىلىشنىڭ ئورنى يوق،ئۈزەمنىڭ ئۇنداق يەرلەرگە ھەرگىزمۇ دەسسىمەيدىغانلىقىمنى،ئۇنىڭغا كېرەكلىك ھەممە نەرسىنى ئېلىپ دوستۇمدىن ئېۋەتكىنىمنى،ھازىرلا يىتىپ بارىدىغانلىقىنى دىدىم،بەش مىنۇت ئۆتمەيلا ئۇنىڭدىن خەت كەلدى:
      -رەخمەت ،سىزنى ئاۋارە قىلدىم،دوستىڭىز كەلمىسۇن، مەن سىرتقا چىقىپ كەتتىم!

             شۇنىڭدىن باشلاپ سەل يىرىگلىشىپ قالغان بولساقمۇ ئىزدىشىپ تۇردۇق ئەلۋەتتە،ئۈن دىداردا.بىر كۈنى ئۇ تۇيۇقسىز تېلفون نومۇرۇمنى سورىدى ئۇھ،ئاخىرى سورىدى ،سورىمىسىمۇ دەپ بېرىشنىڭ ئۇرنى يوق يا ئۈزەم سورىيالماي بەك ئىچىم سىقىلغان،دەرھال ئەۋەتىپ بەردىم.كەچ بولغاندا ماڭا تېلفون قىلدى،ئىككى سائەت پاراڭلاشتۇق،ئۆمرۈمدە ھېچكىم بىلەن بۇنچە ئۇزۇن پاراڭلىشىپ باقمىغان......گەپتىن-گەپ چىقىپ كىنو كۆرۈشنىڭ گىپى بولدى،مەن ئادەتتە تىلۋوزۇرنىمۇ ئانچە كۆرۈپ كەتمەيدىغىنىمنى،باشلانغۇچ مەكتەپتە كوللىكتىپ مەجبۇرىي كىنوغا ئاپارغاندىن باشقا چوڭ بولۇپ كىنوخانىغا كىرىپ باقمىغىنىمنى ئېيتتىم،ئۇ كۈلۈپ كەتتى...ئەتە مىنى كىنو كۆرۈپ كىلىشكە تەكلىپ قىلدى ،مەن دەرھال قوشۇلدۇم.......
              بىر كۈن ۋاقتىمنى ياسىنىشقا سەرپ قىلدىم،چاچلىرىمنى ئاتۇرۇپ،يۈزلىرىمنى قاتۇرۇپ دىگەندەك قىلىدىغان قېلىق بولسا ھەممىنى قىلىپ تەييار بولدۇم،ئىشىكتىن چىقاي دەپ تۇرشۇمغا ئىكرامنىڭ ئاتا-ئانىسى كىرىپ كەلدى....ئۇلار رەسمىيەتلىرىمنى بىجىرىپ بولغاچقا مىنى تۇيۇقسىز خوش قىلىپ مۇبارەكلىگىلى كىرگەنىدى،ئايلىنىپ كەتكىلى تاسلا قالدىم،ئۇ سىرتتا ساقلاۋاتىدۇ،،قۇدىلار پاراڭلىشىۋاتىدۇ،،سائەت 12دىن ئاشقاندا مىھمانلار چىقىپ كەتتى،تېلفونغا قارىدىم:
         -قايسى كۈندىكى ئۆچىڭىزنى ئېلىۋالغان بولسىڭىز كېرەك،-دەپ يېزىلغان ئۇچۇرى تۇراتتى.ئەتىسى بىر كۈن جىمجىتلا ئۆتتى،بۈگۈن ئىكرام ئىككىمىزنىڭ ئىشىدىن خەۋىرى بار دوسلۇرۇم ئۇنىمىغىنىمغا قارىماي يىغىلىشتى.شۇ يەردە ئولتۇرۇپ ئۇنىڭغا خەت يازدىم:
               -بىز كۆرىشەيلى،-جاۋاپ كەلدى:
                     -قاچان،-مەن ھازىرلا كۆرۈشكۈم بارلىقىنى ئېيتتىم،ئۇ بىر مىھماندارچىلىق ئۈستىدە ئىكەنلىكىنى ،ئىككى كۈندىن كىيىن قەشقەرگە ماڭىدىغانلىقىنى،بۈگۈن چوقۇم ساقلىشىمنى ئېيتتى.ھازىرمۇ سائەت11يېرىمدىن ئاشقان ئىدى،ساقلىسام ساقلىدىم بۈگۈن كۆرۈشكىنىم كۆرۈشكەن..............داشىمىندىكى خۇڭفېڭ قەھۋەخانىسدا كۆرۈشىدىغان بولدۇق،بۇيەرگە كىلىپ ساقلىغىنىمغىمۇ يېرىم سائەت بولۇپ قالغاندا ئۇچۇر كەلدى:
            -مەن كەلدىم ،قايسى ئورۇندا؟-ئۇنىڭغا خەت قايتۇرۋېتىپ سائىتىمگە قارىدىم....
                  بۇۋاقىت دەل 2012-يىلى6-ئاينىڭ 12-كۈنى سائەت1:10ئۆتكەن چاغ ئىدى...........ئاھ،قىسمەت بۇنىڭدىن بىر ئاي بۇرۇن سۈنئىي ھەمراھ ۋاستىسى بىلەن مىنى ئۇنىڭغايولۇقتۇرۇپ مۇھەببەتسىز ياشىغان تۇرمۇشىدا مۇھەببەتنى بار قىلدىڭ،ئەمدى بۈگۈن يەنە يۈزتىرانە ئۇچراشتۇرۇپ ئوتلىرىدا كۆيدۈرۈپ خارقىلارسەنمۇ....كۆڭلۈمنى بۇنىڭغا بەرگۈزۈپ،ئۈزەمنى باشقا بىرىگە يار قىلارسەنمۇ.....ئۇنى قانداق تېز ئەكەلگەن بولساڭ ،شۇنچە تېز ئىلىپ كىتىپ ماڭا بۇ ئالەمنى تار قىلارسەنمۇ ...................؟

                                             ئۈن دىدارنىڭ سوۋغىسى-تۇنجى مۇھەببەت

                ھۆرمەتلىك ئوقۇرمەن،بەلكىم بۇ قۇرلارنى ئوقۇۋاتقىنىڭىزدا مۇھەببەتنى بەك يۈكسەكلىككە كۆتۈرۋېتىپتۇ دەپ ئويلىشىڭىز  مۇمكىن،لېكىن مۇشۇ قۇرلاردىكى سۆزلىرىمنىڭ راستلىقىغا چوقۇم ئىشىنىڭ......
             ئۈزەممۇ ھەيران ،ئۇ ئىشىكتىن كىردى،چوقۇم شۇ دىدىم،بىلەمسىزيۈرۈكۈم ئۇنىڭ ئاياغ تىۋىشىنى تونىۋالدى.ئىككىنچى قەۋەتكە چىقتى،يەرگە شۇنداق قاتتىق دەسسەپ ماڭىدىكەن خۇددى مەندەك.ئىككى تەرىپىگە قاراپمۇ قويماي ئۇدۇل يېنىمغا كەلدى،ئۆز سەزگۈسىگە شۇنچىلىك ئىشىنىدىكەن، خۇددى مەندەك.شۇنداق سالاپەتلىك ئىكەن، خۇددى مەندەك.ئوتتۇر بوي ،كىلىشكەن،تەمكىن،جۇشقۇن ئىدى ،،،خۇددى مەندەكلا..ئىۋەنسىز ئىدى...بىر ماركىنىڭ ئۇ ئەرەنچە ،مەن ئايالچە لېكىن باشقا جايلىرى تۈپتىن ئۇخشاش مەسھۇلاتىغا ئوخشايتتۇق...................
           -بۇ يەردە ئولتۇرايلى،-ئۇ پەردىلەر بىلەن ئورالغان خالى ئورۇننى كۆرسەتتى،مەنمۇ تۈزۈت قىلمايلا يۆتكەلدىم،يىگىت مەس ئىدى،بىرەر قۇر ئەھۋاللاشقاندىن كىيىن كىچىككىنى ئۈچۈن ئۆزرە ئېيتىپ گەپ باشلىدى:
   -ھە،،ئۈزەمنى كۆرۈپ جىمىپلا كەتتىڭىزغۇ،سىز مىنىڭ ھەممىدىن قايسى يېرىمنى ياقتۇردىڭىز دىگەن سوئالىمغا،چىرايىڭىزنى دەپ جاۋاپ بەرگىنىڭىز ئىسىڭىزدىمۇ،ئەمدى دەڭە راسلا شۇنچە چىرايلىقمىكەنمەن؟-دەپ سورىدى.نىمىسىنى دەي ئومىقىم،ئادەم بالىسىنىڭ سىزدەك چىرايلىقىنى،سىزدەك مىھرى ئىللىقىنى ئۇچراتمىغانىدىم دەيمۇ،ياق،مەن بىر قىز بالا.دەرھال ئۆزەمگە قايتتىم-دە،كۈلۈمسىرەپ تۇرۇپ:
        -ئادەتتىكىدەك،-دەپ جاۋاپ بەردىم.جاۋابىمدىن قاقاقلاپ كۈلۈپ كەتتى،قاراڭغۇ قەھۋەخانا ئۇنىڭ كۈلكۈسىدىن يورۇپ كەتتى،توۋۋا،،بىر كىنودا سىز ناخشا ئېيتسىڭىز ،پۈتۈن جاھان سىزگە جۆر بولۇپ بىللە ئېيتىدۇ دىگەن جايلىرىنى كۆرۈپ،نىمانچە ئاشۇرۋېتىدىغاندۇ دەپ غۇدراپ قويغانىدىم،مانا ھازىر ئۇ كۈلىۋىدى ئۇنىڭغا جۆر بولۇپ پۈتۈن كائىناتنىڭ كۈلىۋاتقىنىنى كۆردۈم.يەنە گەپ باشلىدى،ئارلىقتا مىنى ماختاشنىمۇ ئۇنتۇپ قالمىدى.ئۇنى-بۇنى بۇيرۇپ ئۈستەلنى لىق توشقۇزۇپ،توسىغىنىمغا ئۇنىماي ھاراق ئىچىشكە باشلىدى،خىزمەت مۇھىتىدىن تارتىپ ئائىلە ئەھۋالىغىچە ،جەمئىيەتتىكى ئورنىدىن تارتىپ دوسلىرى ئىچىدىكى تەسىرىگىچە تولۇق سۆزلىدى.گەپلىرىدىن ئاتا-ئانىسىدىن پەخىرلىنىش تۇيغۇسىنىڭ تولىمۇ كۈچلۈكلىكى،باشقىلار ئالدىدىكى ئوبرازىغا بەكلا دىققەت قىلىدىغىنى چىقىپ تۇراتتى،ماڭا ئوپمۇئوخشاش بۇ مىجەزلىرىڭىزنى قانداق قىلاي دىدىم ئىچىمدە.ئۇ گىپىنى تۈگەتكەندىن كىيىن سوئال نەزىرى بىلەن ماڭا تىكىلدى،ئەمدى سۆزگە چىقىش نۆۋىتى ماڭا كەلگەنىدى............................
                 كەسپىم توغرىسىدائازراق،خاراكتىرىم توغرىسىدا ئازراق سۆزلەپ ئۇزۇندىن بېرى چەتئەلگە چىقىش يولىنى مېڭىۋاتقىنىمنى،بۈگۈن رەسمىيەتلىرىمنىڭ پۈتكەنلكىنى،ئۇزۇن ئۆتمەي ئوقۇشقا بارىدىغىنىمنى ،بۇ يەردىكى ئازغىنە ۋاقتىمنى خوشال ئۆتكۈزۈش ئارزۇسىدا ئىكەنلىكىمنى دەپلا گەپنى تاماملىدىم.ئۇ تۇيۇقسىزلا يېنىمغا ئۆتۈپ ئولتۇردى،ھاراقنىمۇ ئىچىپ بولغانىدى،بىئارام بولىۋاتقىنىنى ئېيتىپ مۆرەمگە بېشىنى قويۇپ كۆزلىرىنى چىڭ يۇمدى...........شۇ ئارىدا ئىشتىن چۈشىمىز دەپ كىرگەن مۇلازىمنىمۇ دەخلى قىلماسلىقىنى ئۆتۈنۈپ چىقىرۋەتتىم،مانا ئۇ قۇچىقىمدا يىنىك تىنىپ ئۇخلاۋاتىدۇ،قوللىرىمىز قاچانلاردا تۇتىشىۋالدى بىلمىدىم ،مەندە ھېچقاچان بۇنداق تۇيغۇلار بولۇپ باقمىغان،ھېچقاچان بۇنداق تەلپۈنۈشلەر بولۇپ باقمىغان،بۇ تۇنجى قېتىمى،تۇنجى قېتىم ................
                قۇچىقىمدا ئاي ئۇخلاۋاتىدۇ،يۇلتۇزلار ئۈستىمىزدە بىزگە قاراۋاتىدۇ،ئۇنىڭ قولىنى تۇتقان قوللىرىم تىتىرەپ پۈتۈن جىسمىم شامدەك ئېرىپ كىتىۋاتىدۇ،كۆزىمدىن مۇھەببەت ئۈچۈن تۇنجى قېتىم ياشلىرىم تۆكىلىۋاتىدۇ،يۈرىكىم نالە قىلىۋاتىدۇ:
                -ئاھ،شۇنچە گۈزەل تۇيغۇلارنى،تۇنجى مۇھەببەتنىڭ مىھرىنى ماڭا ئەمدى 28يېشىمدا ئاتا قىلغان قۇدرەتلىك ئىگەم،،،ئۆتۈنەي،،،بىزنى ئايرىمىغىن.....................!
