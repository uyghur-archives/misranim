ئەسسالامۇئەلەيكۇم مىسرانىم مۇنبىرىنىڭ قىزغىن ئىشتىراكچىلىرى . ھازىر ئالدىنقى  تىمامنىڭ داۋامى ھىكايىسى بولغان  مۇھەببەت ھەممىگە قادىر ئەمەسنى ھوزۇرۇڭلارغا سۇنماقچى .



ئالدىنقى تىما ؛   مۇھەببەت قۇربانى


مۇھەببەت ھەممىگە  قادىر ئەمەس .



مەرھابا بىلەن دىلفۇرە  يىقىن دوسلاردىن بولۇپ ، بۇلار ھەر ئىككىلسىنىڭ ئائىلىسى ئىسلامنى چۈشەنگەن ئائىلىلەردىن ئىدى . ئائىلىدە  ئەرنىڭ دىگىنى يۇقىرى ئورۇننى ئىگەللەيتتى . ئايالى ئىتائەتچان  ئائىلىدىن ئىدى . يۇرۇش تۈرىشىمۇ  ناھايىتى پاكىز ھەم  ئىسلامىي ئىدى . ئۇلار تولۇقسىزدىن كىيىن چىقىپ كىتىپ بىرسى دىلفۇرە تىككۈچىلىكتە ئوقىدى . مەرھابا بىر مەزگىل ئوقۇغاندىن كىيىن ئىشلەشنى توغرا تىپىپ  بىر  سودا ساراينىڭ دۇككىنىدا  كىيىم-كىچەك سىتىشتا ئىشلىدى . مۇئاشى خىلى ياخشى ئىدى . بۇلار ئىككىلىسى  10 نەچچە يىللىق  دوسلاردىن بولۇپ ، خىلى يىقىن ئۆتەتتى . ئۇلار  ئىشتىن چۈشكەندە  بىللە ئۆيگە  قايتاتتى . ھەم بىللە  ئۈگۈنۈشكە  ئىشقا بىللە  چىقاتتى . مەرھابانىڭ  رۇخسارى ھەم  كىيىشى ناھايىتى يارىشىملىق بولغاچقا  ، ئۆزىگە بىر قەدەر ئاۋارىچىلىق كەلتۇرەتتى .. يەنى ئوغۇللار كەينىگە  كىرىۋالاتتى . لىكىن  رەت قىلىپ كەلگەن ئىدى ....
كىيىن  مەرھابا  بىر نەچچە قىتىم ئىش ئورنىنى يۆتكىدى . ھەممىسى  دادىسىنىڭ بۇيرۇقى بويىچە بولاتتى . چۈنكى  دادىسىنىڭ دىگىنى دىگەن ئىدى . ئانىسى بولسا  ئىتائەتچان ئايال بولغاچقا  يان تاياق بولدىغان كىشىمۇ يوق ئىدى . شۇڭا ئامالىمۇ يوق ئىدى . لىكىن بۇمۇ قارشى تۇرغىدەك ئىش ئەمەس ئىدى . چۈنكى بۇ ئورۇنلاردا ئۆزىنى ئاۋارە قىلىدىغانلار  يوق ئىدى ،
دىلفۇرە بولسا  ئەمدى 18نىڭ  قارىسىنى ئالغان بولۇپ ، بىر قىتىم توي قىلىپ ئاجراشقان دىسەكمۇ بولدۇ . ئۇ ئىسلامىي ئائىلىدە تۇغۇلغان بولسىمۇ  ئىسلامنى كىشى كاللىسىدا  قانداق چۈشەنسە  ئىشقا قارتا  پوزىتىيەسى شۇنداق بولدىغاندەك  تويغا  ئۆزىنىڭ ئۇنى ياخشى كۆرمەيدىغانلىقىنى كۆرۈپمۇ باقمىسام تونۇشمىسام دىگەن باھانىلار بىلەن قارشى تۇرسىمۇ ئامال بولمىدى . چۈنكى  ئۇنىڭ كۆڭلىدە باشقا بىرسى بار ئىدى . ئۇنىڭ ياخشى كۆرىدىغىنىنى باشقا بىرسى بولسىمۇ ئۇنى ياخشى كۆرىۋەردى . ھەم  تويغا قارتا  قىزغىنلىقىنى قوزغاتمىدى ،،.ئاجرىشىشنى نىيەت قىلىپ توي قىلغان ئىدى .
تۇنجى توي قىلغان ۋاقىتلاردا ئۇ  ئىرىگە  ياخشى مۇئامىلىدە بولمىدى . ھەم سوغۇق مۇئامىلىدە بولدى ، يولدىشىمۇ شۇنداق  ياخشى مۇئامىلىدە بولسا  تەتۇرلۇق قىلىپ تۇرۋالدى . كۆڭلىدە ئاشۇ باشقا بىرسى بار ئىدى . كىيىن ئۇنىڭ توي قىلغانلىق خەۋرىنى ئاڭلاپ  ناھايىتى  ئازاپلاندى . لىكىن ئورنىغا كەلمەيتتى . چۈنكى ياخشى كۆرۈش  بىر ئىشنى ۋۇجۇدقا  ئەپ چىقالمايدىغانلىقىنى بايقىغان چىقىغدا  ناھايىتى كىچىككەن ئىدى . ياخشى كۆرۈشنى دەپ كۆز ئالدىدىكى بەختنى قولدىن كەتكۈزۈپ قويدى . كىيىن يولدىشىمۇ كىچە كىرمەيدىغان . ئۈنىڭ بىلەن كارى بولمايدىغان بولۇپ قالدى . ئامال يوق بىشىدا  ئىززىتىنى قىلسا  بىلمىگەن ئەمدى بىلگەندە  ئىززەتلىگۈچى يوق . ئامال يوق بىر نىكاھ مۇشۇنداق  ئاياغلاشتى . كىچىك تۇرۇپ  تۇل قالدى دىسەك بولىدۇ . شۇنداق قىلىپ تۇنجى خاتا تاللىشى ئۇنى بىر قىتىم توي قىلغان يامان ئاتاققا قالدۇرۇپ قويدى . شۇندىن كىيىن قانداقلا بولسۇن يىگىت بولغان ئىكەن بىر قىزنى ئۆز ئەمرىگە ئىلىشنى خالايدۇ . يالغانچىنىڭ قۇيرۇقى بىر تۇتام . ھامىنى ئاشكارلىنىدۇ . شۇنداق قىلىپ  ئۇ بىر نەچچىسى بىلەن كۆرۈشۈپ يىنىشىپ كەتتى .
شۇندىن كىيىن ئۇنىڭ كاللىسىدا ئاجراشساممۇ بىر بالام بولسا  ئۆزۇم ئىشلەپ ئۆزۇم خان يۇرۇپ ، بالامنى بىقىپ ياشاش دىگەندەك  خاتا  قاراش  ئىڭىدە پەيدا بولدى .

دىلفۇرەنىڭ ھاياتىنى ھىچ نەرسىگە  ئەس قاتمايدىغان  ياخشى كۆرۈش بىر تەرەپ قىلىپ بولغان ئىدى .
ئەمدى  مەرھابانىڭ ھاياتىغا  كەلسەك ، ئۇ شۇنداق يۇرۇش  جەريانىدا  يىقىن بىر يەردە ئىشلەيدىغان  ئەركىن ئىسىملىك بىر بالا كۆزىگە ئىسسىق كۆرىنىدىغان بولۇپ قالدى . بىر نەچچە قىتىم ئاشخانىدا  ئۇچرۇشۇپ قالاتتى .. ئۇنىڭ قىلىقلىرى قىزىقارلىق ھەم  خۇشچاقچاق بالا بولۇپ بىر قىتىم بىر بىرسىگە كۆزى چۈشۈپ قىلىشتى . قىز ھەم  خىجىل بولغاندەك يەرگە  قارىۋالدى . ئۇمۇ پات-پاي قاراپ قوياتتى . شۇنىڭ بىلەن يۇرەكلەرگە  ئوت تۇتاشتى . كىيىن قانداقتۇر  بىر بىرسىنىڭ تىلفۇن نومرىنى قولغا  كەلتۇرۇشتى . ھەم  كۆرۈشۈپ قالدى . كىم بىلدۇ شۇنداقلا  توغرا كەپ   ئىككەيلەن توردا  پاراڭلىشىپ قىلىپ كىيىن  بىر بىرسىنىڭ  ئاشخانىدا كۆرۈشكەن ھىلىقى قىز ۋە ئوغۇللىقىنى بىلگەندە يۇرەكلىرى قىن-قىنىغا  پاتماي قالدى . يىگىتمۇ شۇنداق سالاپەتلىك ھەم  5 ۋاخ نامازنى تاشلىمايدىغان ئۆز ئىشى بىلەن دوسلىرى بىلەن ناھايىتى ئىجىل ئۆتىدىغان بالا ئىكەنلىكى  ئۇنىڭغا ئايان بولدى . قىزنىڭمۇ شۇنداق ياخشى قىز ئىكەنلىكى ھەم ناھايىتى ئەدەپلىك قىز ئىكەنلىكىنى بىلدى . ھەم بىر بىرسىنى ناھايىتى ياخشى كۆرۈشۈپ قالدى .كۆرۈشكەن ۋاقتى ئۇزاققا بارمىسىمۇ  خۇددى ئۇزاق يىللىق ئاشىق-مەشۇقتەك  يۇرۇپ كەتتى .
كىيىن قىزنىڭ دادىسى قىزنىڭ ماقۇللىقىسىز  بىرسىگە  قىزنى بىرىش توغرىسىدا  دىيىشىپ قويغان ئىدى . قىز ئاڭلاپ بىشىغا بىر چۆگۈن قايناقسۇ قۇيغاندەك بولۇپ قالدى . چۈنكى ئۇ يىگىتى ئەركىننى ناھايىتى ياخشى كۆرەتتى . لىكىن  دادىسىنىڭ بۇيرىقى بۇيرۇق . ئۆزى قارشى تۇرۇپ بولالمايتتى . ياخشى كۆرىدىغىنى بار دىيىشتىن  ئۇيالدى . نىمە قىلىشى كىرەك بىلمەيتتى .
كىيىن ئەركىنگە  بۇنى قانداق  دىيىش بىر تەسكە چۈشتى ، شۇنداق  تەستە دىگەندە ئەركىننىڭ چىرايىنىڭ ئۆزگىرىشى ئۇنى تىخىمۇ قورقۇتۇپ ئايلاندۇرۇپ قويدى .لىكىن يەنىلا ئۆز  خۇدىنى بىلەتتى .  ئەركىن بولسا  ئۇنى يۆلەپ تۇرغۇزۇپ ئۆيى ئالدىغا ئاپىرىپ قويدى . ھەم ئۆزىنى چەتكە ئالدى . ئائىلىسىدىكىلەرگە  تىلفۇن قىلغۇزۇپ  ئىلىپ چىقىپ كىتىشىتى . كىيىن كۆرۈشكەندە  يىگىت  مەرھابانى بەكلا ياخشى كۆرىدىغانلىقىنى ، ئۆيگە باردىغانلىقى توغرىسىدا سۆز ئاشتى . لىكىن توي ۋاقتى  ھەممە نەرسىسى بىكىتىلىپ بولغان ئىدى . يىگىت شۇنداق  سۆزلەر ئارىسىدا  كىيىن بولسىمۇ سىنى ساقلايمەن . مەن سنى  ياخشى كۆرىمەن دىدى . ھەم چىرايلىقچە  خوشلاشتى . كىيىن مەرھابانىڭ تويى بولدى . يىگىتنى تويغا كىلىشىنى ئىيتتى . لىكىن  يىگىت يىراقتىنلا ئۇنىڭ  توي ماشىنسىدىن چۈشكەنلىكىنى كۆردى . يۇرىكى دەرتكە تولدى .، ھەم دوستىنىڭ  ياردىمىدە سوۋغىسىنى ئۇنىڭ دوستىغا يەتكۇزدى . سوۋغىسى بولسا  بىر قۇرئان بىلەن تەسۋى ئىدى .

كىيىن ئۇلار توردا بىر نەچچە قىتىم خەت يىزىشىپ قالدى . پاراڭلىشىش جەريانىدا  قىز بىلەن  قىيئانا  مۇناسىۋىتى ياخشى ئەمەسلىكىنى ، ئۇرۇشۇپ قالدىغانلىقىنى بىلدى . يىگىت بولسا  بۇنداق ئىشلاردا ناھايىتى  ئىغىر  سالماق بولىشىنى ،، ئاچچىقنىڭ كەينىگە كىرمەسلىكىنى  قايتا قايتا  تەكىتلىدى . ھەم  ياخشى ئۆتۈپ كىتىشىنى  ئەسكەرتتى ،
قىزدا بولسا  قاچان بولسۇن مەن سىنى ساقلايمەن سىنى ياخشى كۆرىمەن دىگەن سۆز ئۇ توي قىلىپ ئاجراشقاندىن كىيىنمۇ ئۇنى قوبۇپ قىلىدۇ دەپ چۈشەندى .
ھەم  بىر كۈنى  يىگىتكە  ، ئەگەر مەن ئاجرىشىپ كەتسەم مىنى قوبۇل قىلامسىز ؟ قوسقۇمدا بار بولسىچۇ دىگەن دىمەككە ئاسان  لىكىن  ۋەزنى  ئىغىر بۇ سۇئالنى سورىدى . يىگىت  داڭ قىتىپ قالدى .
قانداق قىلىشى كىرەك  ، زادى قانداق قىلىشى كىرەك .

بىلىپ قويۇڭكى ھەقىقى مۇھەببەت  نىكاھتىن كىيىن بولىدۇ . ھەرگىز نىكاھتىن بۇرۇنقى مۇھەببەت  ئادەمنى شۇنداق  ھەر كويغا  ھەر ئۇسۇلغان سالىدۇ . شۇڭا  مۇھەببەت دىگەن بۇ   چوڭقۇر ھاڭ ئۇستىدىكى پولات سىمدا مىڭىپ ئۆتۈشنى خالىسىڭىز ھەقىقى مۇھەببىتىڭىزنى  يولدىشىڭىزغا بىرىڭ . ئاندىنلا بەخىت ئامەت سىزگە قۇچاق ئاچىدۇ .


سەمىمىي پىكىر تەكلىۋىڭلارنى قوبۇل قىلىمەن . قاينىمىغا ئاساستا مۇنازىرە قىلىشنى خالايمەن .
