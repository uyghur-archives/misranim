ئەسسالامۇ ئەلەيكۇم مىسرانىم تور بېكىتىنىڭ ئەزىز قوللىغۇچىلىرى ،ئەڭ ئالدى بىلەن ھەممىڭلارنىڭ ئائىلىسىنىڭ تىنىچ - ئامان، ئىشلىرىنىڭ ئۇتۇقلۇق بولىشىنى ئۇلۇغ ئاللاھـتىن تېلەيمەن!

مەلۇم بىر جەمئىيەتنىڭ نورمال تەرەققىي قىلىشى ئۈچۈن چوقۇم ياخشى بولغان قانۇن تەرتىپ- شۇنداقلا ئىقتىدارلىق باشقۇرغۇچىلار قوشۇنى بولغىنىدەك ،مۇنبىرىمىزنىڭمۇ بىخەتەر،ساغلام، ھەممە ياقتۇرىدىغان مۇنبەر بولىشى ئۈچۈن بىز يەنىلا مۇنبەردىكى تور بېكىتىمىزنىڭ قوللايدىغان،ھەقىقىي چىن مەنىسى بىلەن ئۇيغۇر تورچىلىقى ئۈچۈن بىر كىشىلىك ھەسسە قوشالايدىغان ئىقتىدارلىق ،مۇئەييەن مۇنازىرە تەشكىللەش،سىياسىي ئېڭى يۇقىرى بولغان باشقۇرغۇچىلارغا ھەر ۋاقىت مۇھتاجمىز.

توربەتنىڭ تەرەققىي قىلىشىغا ئەگىشىپ كۈندىلىك يازمىلار تېخىمۇ كۆپىيىپ مېڭىپ ،ھەر قاتلامدىكى ھەرخىل كىشىلەرنى مۇنبىرىمىزگە جەم قىلىۋاتىدۇ.ئۈن -تىنسىز ئۇيغۇر تورچىلىقىمىز،ئۇيغۇر ئانا تىلىمىزنى راۋاجلاندۇرۇش ئۈچۈن كۈچەۋەتقان مۇشۇنداق پەيىتتە،مەلۇم بىر كىشى  سەۋەبلىك بىز نەچچە يىلدا بەرپا قىلغان بۇ مېۋىلىك باغ ۋەيران بولسا ھەممىمىزنىڭ يۈرىكى ئېچىشىدۇ ئەلۋەتتە.

يېقىنقى بىر قانچە ئايدىن بۇيان بىر قىسىم ئەخلاقسىز، ئەدەپسىز،تور ئەخلاقىنىڭ نېمىلىكى چۈشەنمەيدىغان كىشىلىرىمىز مۇنبەردىكى توغرا تېما يۆنىلىشىنى قەستەن بۇراپ،جىدە چىقىرىشقا ئۇرۇنۇپ ،ئەزالار بىلەن سوقۇشۇپ مۇنبەرنىڭ ئىللىقلىقىغا ،ئىناقلىقىغا ئانچە - مۇنچە تەسىر كۆرسەتكەنمۇ بولدى، بىز ئۆز نۆۋىتىدە ئۇلارغا نىسبەتەن چەكلەش، پۈتۈن ئارخىبىنى يۇيۇش جازالىرىنى ئىشلەپ بولدۇق.

بىز مۇنبىرىمىزنىڭ ئۇزۇن مۇددەتلىك تەرەققىياتى ئۈچۈن يەنە بىر تۈركۈم مىسرانىم ئۈچۈن كۈچ قوشۇشنى ئويلىغان خالىسانە مېھنەتكەشلىرىمىزنى باشقۇرغۇچىلىق سېپىمىزغا قوشۇشنى قارار قىلدۇق.

باشقۇرغۇچىلىق شەرتلىرىگە ئارتۇق شەرت ۋە تەلەپلەرنى قويمايمىز ،ئەخلاقى ئالاھىدە ياخشى بولىشى،ئانچە مۇنچە يېزىقچىلىقتىن خەۋىرى بولىشى ،مۇنبىرىمىز ئۈچۈن خالىس كۈچ قوشۇشقا ئۆز رازىلىقى بىلەن قوشۇلسا، مۇنازىرىنى توغرا تەرەپتىن تەھلىل قىلىپ نورمال ئېلىپ ماڭالىسا، كۈندە بەلگىلىك تورغا چىقىش ۋاقتى بولسا ،سىياسىي تېماۋە ئىنكاسلارغا نىسبەتەن سەزگۈر بولسا بىز ئەھۋالغا قاراپ باشقۇرغۇچىلىق ۋەزىپىسىنى بېرىمىز.(باشقۇرۇش بېتى پۈتۈنلەي ئۇيغۇرچە ،ئالاھىدە كومپىيۇتىر سەۋىيسى تەلەپ قىلمايمىز).

باشقۇرغۇچىلىق خىزمىتى جاپالىق ئەمما شەرەپلىك خىزمەت.قېنى بىز بىلەن بۇ يولدا ئاخىرغىچە مېڭىشقا جۈرئەت قىلالايدىغان تورداشلار بىرلىكتە مىسرانىمنىڭ كەلگۈسى ،ۋە ئەتىسى ئۈچۈن بىرلىكتە تەڭ ئىلگىرلەيلى!

باشقۇرغۇچىلىققا ئىلتىماس قىلىشنى مەزكۇر تېمىغا ئىنكاس سۈپىتىدە قالدۇرۇپ قويۇڭ ياكىQQ:364103 گە سۆز قالدۇرۇپ قويۇڭ!

راسىت گەپنى قىلسام باشقۇرغۇچى بولماق ئۇنداق ئاسان ئىش ئەمەس . ئۇلارنىڭ تارتىۋاتقان جاپاىسىغا بولسا سىلەرنىڭ ئىقتىساتىڭلار يار بەرسە قولۇڭلاردىن كەلسە مائاش تەستىقلاپ بەرسەڭلەرمۇ ياكى ھېچ بولمىغاندا كۆڭلۈڭلارنى بىلدۈرسەڭلەرمۇ ئازلىق قىلىدۇ .
لېكىن ئۇلار ئۈن - تىنسىز جاپا چېكىپ ئېشلەۋاتىدۇ .بىرسىگە ياخشى بولۇپ بىرسىگە يامان بولۇپ دىگەندەك ، كىم ئۈچۈن ؟ يەنىلا شۇ بىزنىڭ ئۇيغۇر مۇنبەرچىلىكىمىز ئۈچۈن .

شۇنداقلا ئۆزىنى سىناپ باقىدىغانلار ئۆزۈڭلارنى نامزات كۆرسەتسەڭلارمۇ بولىدۇ .
