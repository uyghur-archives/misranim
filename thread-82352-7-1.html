مىنىڭچە باشقۇرغۇچى بۇلۇش ئۇنداق ئۇڭاي ئىش ئەمەس دەپ قارايمەن . ياخشى بىر باشقۇرغۇچى بولۇش ئۈچۈن ئالدى بىلەن ئۈزى باشقۇرۇۋاتقان مۇنبەرگە مەسئۇل بۇلالايدىغان، سىياسىي ئىڭى يۇقىرى بولۇش، بەلگىلىك رەھبەرلىك ئىقتىدارى بولۇش، بەلگىلىك قەلەم قۇۋۋىتى بولۇش، نۆۋىتى كەلگەندە مۇنبەردە توغرا بولغان ئېقىمنى يىتەكلىيەلەيدىغان ،ئىغىر-بىسىق ئادەم بولۇش ھەدىسىلائىنكاس يازغۇچىلار بىلەنتالاشمايدىغان ۋە مۇنبەرنى چۈشىنىش ئىختىدارى بولغان بىر قاتار شەرىتلەر بۇلۇش  تەلەپ قىلىنىدۇ. ئۇنىڭدىن باشقا قاتتىق قول بولۇشقا تىگىشلىك يەردە  قاتتىق قول بولالايدىغان، يۇمشاق قول بولىدىغان يەردە ئانىنىڭ بالىغا قىلغان مۇئامىلىسىدەك ئاقكۆڭۈل بولالايدىغان بولۇشى  كىرەك مۇنبەرنى ئەزالارنى چۈشىنىدىغان باشقۇرغۇچى ھەرقانداق بىر زىل تېمىنى تەستىقلاشتىن بۇرۇن، شۇ تىمىغا چۈشىدىغان ئىنكاسلارنى ئالدىن پەرەز قىلىشى شۇ ئارقىلىق تەدبىرنى توغرا بەلگۈلۈيەلەيدىغان بۇلۇشى كىرەك .ھەمدە بەزىبىر توقۇنۇشلارنىڭ ئالدىنى ئالالايدىغان بۇلۇشى كىرەك، بىرقىسىم باشقۇرغۇچىلىرىمىز بار باشقا تېمىلارغىغۇ دىققەت قىلمىسۇن، ئۆزى تەستىقلىغان تېمىغا چۈشۈۋاتقان ئىنكاسلارغا قاراپمۇ قويمايدىغان،قانداق ئىنكاس پىكىرلە چۈشۈۋاتقانلىغىغىمۇ قارىمايدىغان، تورقانۇنىغا خىلاپ گەپ-سۆزلەر ياكى جىدەل چىقىپ شۇنچە سەت گەپلەر بولۇنۇپ كەتسىمۇ قارايدىغان ئادەم يوق بۇلۇش ئەھۋاللىرى مەۋجۇت بۇلۇۋاتىدۇ. يەنە كۈپۈنچە ھاللاردا ئىناقسىزلىق،جىدەل ماجرا زىددىيەتدىنى تۈس ئالغان تىمىلاردىن چىقىۋاتىدۇ .شۇڭا دىنىتىمىلارغا قارىتا ياىشى بولغان دەدبىر بىكىتىلىشى كىرەك.
       ئەسلىدە ھەر بىر باشقۇرغۇچى ئاۋۋال ئۆزى تەستىقلىغان زىلراق تېمىلارغا باشتىن ئاخىرى ئىگە بولۇشى كىرەك ،مىنىڭ بۇيەردە  مۇنداق دېگىنىم باشقا تېمىلار بىلەن كارى بولمىسۇن دىگەنلىك ئەمەس، ئۆزى مەسئۇل بولغان تىمىغا بەكرەك كۈڭۈل بولسۇن ،ھەمدە ئۆز ى  تەستىقلىغان تىمىسىغا ئۈزى نازارەتنى كۈچەيىتسۇن دىمەكچىمەن.
تېمىلاردىكى ئىملا خاتالىقى مەسىلىسىنى باشقۇرغۇچىغا ئارتىپ قويۇۋەرسەك قاملاشمايدۇ، بۇنىڭ ئۈچۈن ئىملا خاتالىغى كۆپ تىمىلارنى قەتئىتەستىقلىماسلىغى كىرەك دەپ قارايمەن.بەلكىم ھازىرقى مۇنبەر سىسىتىمىسىدا ئارقا بەتتە تېما تەھرىرلىگىلى بولمايدۇ، شۇڭلاشقا ئەسلى تېما قانداق بولسا شۇ پىتى تەستىقلىنىپ قالىدۇ، باشقۇرغۇچىلارنىڭمۇ ئۆزىگە چۇشلۇق ئالدىراش ئىشى بار، بەزى ۋاقىتلاردا تەستىقلىغان تېمىسىنى قايتا تەھرىرلەشنى ئۇنتۇپ قالىدۇ، شۇڭلاشقا ئىملا مەسىلىسىگە تېما ئىگىلىرى بەكرەك كۈچىسە دىگەن پىكىردىمەن. ئەمدى خالىس باشقۇرغۇچى بۇلۇش مەسىلىسىگە كەلسەك ئابابەكرى مۇختەر ماڭا ئىشەنسە ، كەڭ مىسرانىم تور ئەھلى ماڭا ئىشەنسە بىركىشىلىك ھەسسە قۇشۇشقا تەييارمەن
