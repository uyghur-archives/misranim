ئەسسىلامۇ -ئەلەيكۇم قېرىنداشلار؟ماڭا بىر مەسلەھەت بەرسەڭلار ؟؟؟؟؟
      مىنىڭ بىر دوستۇم ئىنتايىن نازۇك بىر مەسىلىگە دۇچ كىلىپ قېلىپ مەندىن ياردەم سورىغان ئىدى،مىنىڭمۇ بېشىم قېتىپ دوسلاردىن ياردەم سوراپ ئولتۇرۇپتىمەن.
  ئەسلى ئىش مۇنداق .دوستۇم بىر قىز بىلەن يۇرىۋاتقىلى ئىككى ئايدەك بوپ قالغان.ئۇ قىز شۇنچىلىك شەرمى -ھايالىق ئىسىل قىز ئىدى .ئۈلۈشكۈن دوستۇم مىنى جىددى مەسلەھەتلىشىدىغان ئىش بار دەپ چاقىرتىپتىكەن،مەن ئۇنۇڭ بىلەن كۈرشۈپ،دىگەن گەپلىرىنى ئاڭلاپ بەكلا ھەيران قالدىم.چۈنكى ئۇ يۈرىۋاتقان قىز بىر قېتىم توي قىلىپ ئاجىرشىپ كەتكەن قىز ئىكەن.(لىكىن دوستۇم تېخى نىكاھلانماغلىق).
  ئەسلى قىز دوستۇم بىلەن تۇنجى كۈرۈشكەن كۈنىلا بۇ ئىشلارنى ئۇنۇڭغا دىگەن ئىكەن،لىكىن دوستۇم ئۇنى چاق-چاق قىلىۋاتىدۇ دەپ ئانچە ئىرەن قىلىپ كەتمەپتۇ.(چۈنكى قىزنىڭ تاشقى قىياپىتىدىن توي قىلغان قىزدەك كۈرۈنمەيتى).كىيىن قىزنىڭ ئۆز گېپىدە چىڭ تۇرۋېلىشى،دوستۇمنى ئويلاندۇرۇپ قويۇپتۇ ۋە قىزنىڭ مەھەللىسىگە بېرىپ سۈرۈشتە قىپتۇ. قىزنىڭ دىگەن گېپى راس چىقىپ، دوستۇم نىمە قىلارىنى بىلمەي قاپتۇ.دوستۇم بۇنداقلا بولدى قىلىۋىتىشنى لايىق تاپماي ،قىزنىڭ ھەقىقى سەرگۈزەشتىسىنى ئاڭلاپ بېقىشنى لايىق تېپىپ قىز بىلەن كۈرۈشۈپتۇ.
بىچارە قىزنىڭ سەرگۈزەشتىسى:
قىز تۇلۇقسىز ئوتتۇرا مەكتەپنى ئەمدىلا پۇتتىرىشىگىلا،ئاتا -ئانىسىنىڭ زورلىشى بىلەن ئۆزىدىن خېلىلا چوڭ بىر ئادەم بىلەن توي قىلىپ قۇيۇپتۇ.توي بولغان كۈنگىچە قىز ئۇ ئادەمنى پەقەتلا كۆرۈپ باقماپتۇ.شۇنداق قىلىپ ئۆزى خالىماي قىلغان تويدىن قىزنىڭ كۆڭلى سۇ ئىچمەپتۇ.شۇنداق بولسىمۇ تەقدىرگە تەن بىرىپ،ئۆيىنى ياخشى تۇتۇشنى ئويلاپتۇ.لىكىن ئۇ ئادەم قىزنىڭ قەدەرگە يەتمەي،ئۇنى يوق ئىشلەرنى باھانە قىلىپ ئۇرۇپ -خارلاپتۇ.قىز چىداپ تۇرالماي توي قىلىپ 15 كۈن بولغاندا يامانلاپ يېنىپ كەلگىنىچە،ئىنتايىن تەستە ئاجىرشىپ كىتىپتۇ.ئۇ نىجىس ئادەم ئاجىراشقىلى ئۇنۇماي تۇرۇۋېلىپ،قىزغا تويلۇققا ئەكەگەن ۋە تويغا كەتكەن چىقىملىرىنى تۆلەپ بىرشنى پەسكەشلىك بىلەن دەپ تۇرىۋېلىپتۇ.قىزنىڭ ئاتا-ئانىسى نائىلاج،پۇشايمان ئىچىدە ئۇ ئادەمگە خېلى كۆپ پۇل بىرىپ قىزىنى مىڭ جاپادا ئاجىراشتۇرۋېلىپتۇ.
  دوستۇم قىزنىڭ يېغلاپ تۇرۇپ دىگەن گەپلىرىنى ئاڭلاپ خېلى ئويلۇنۇپتۇ،ئۇنى ئۇنتۇپ كىتىشنى،بۇندىن كىيىن كۈرەشمەسلىكنى توغرا تېپىپ قىزدىن ئاررىلىپتۇ.ئەپسۇس دوستۇم ئۆزىنىڭ كىچىككەنلىكىنى ئۇنى ئۇنتۇشنىڭ مۇمكىن ئەمەسلىكىنى ھىس قىپتۇ،لىكىن شۇنداق بولسىمۇ يەنە چىداپتۇ، شۇ كۈنلەردە دوستۇمنىڭ گېلىدىن تاماق ئۆتمەيدىغان ،ھېچقانداق ئىش خۇش ياقمايدىغان بۇلۇپ قاپتۇ.(قىزمۇ شۇنداق بۇلۇپ ئاغرىقچان بۇلۇپ قاپتۇ)ئۇ ئاخىرى چىداپ بۇلاماي ئىنتايىن كەسكىن بىر قارار چىقرىپتۇ.ئۇ قىزبىلەن توي قىلىشنى، مەڭگۈ بىللە بۇلۇشنى، قىزنى ئۆمىرىدە كۈرۈپ باقمىغان بەخىتكە ئىرىشتۈرۈشنى ئويلاپتۇ.لىكىن دوستۇم ئاتا-ئانا ئۇرۇق تۇققانلىرىنى ئويلاي بېشى قېتىپتۇ.چۈنكى ئۇلارنىڭ بۇ ئىشقا قوشۇلمايدىشانلىقىنى ئۇنىڭغا بەش قولدەك ئايان ئىكەن.شۇڭا دوستۇم مۇنداق قىلىشنى ئويلاپتۇ،ئۆيىدىكىلەرنىڭ تېپىپ بىرىشى بىلەن باشقا بىر قىز بىلەن توي قىلىدىكەن،ئاندىن ئۇ قىز بىلەن قانداق قىلسا قىلىپ ئاجىرشىپ،ھازىر يۇرۇۋاتقان قىز بىلەن توي قىلىدىكەن.شۇنداق قىلسا ھەممە ئىشنى ياخشى بىر تەرەپ قىلغىلى بۇلۇدىكەن.لىكىن مەن بۇ پىكىرگە قوشۇلمۇدۇم.(سىلەرچە قانداق دوسلار شۇنداق قىلسا بۇلارمۇ؟؟؟؟بۇنۇڭغا دىنىمىز نىمە دەيدۇ؟؟؟؟بۇ ئىشتا دوستۇم قانداق قىلسا مەسىلىنى مۇۋاپىق بىر تەرەپ قىلىلايدۇ؟؟؟؟

ئاللاھ خالىسا دوستۇمنىڭ ئۇ قىز بىلەنلا نىكاھلانغۇسى بار ئىكەن.لىكىن ئادەمنىڭ دوس -دۈشمىنى بۇلۇدۇ.ئۇلار ئالدىدا نىكاھلانمىغان يىگىتنىڭ ،توي قىلىپ ئاجىراشقان (سىكىلەك)بىلەن توي قىپتۇ دەپ سۆز-چۆچەككە قېلىشىدىن بەكلا قورقىدىكەن.دىمىسىمۇ بۇ ئويلۇنۇشقا تىگىشلىك مەسىلە .لىكىن دوستۇم نىمە بولسا مەيلى مەن شۇ قىز بىلەن توي قىلىمەن دەپ تۇرۋېلىۋاتىدۇ.......ئاساسلىق مەسىلە دوستۇمنىڭ ئاتا-ئانىسى ۋە ئۇرۇق تۇققانلىرىدا،ئۇلار يۇرۇتتا خېلى يۈزلۈك ئادەملەر بولغاچقا،بۇئىشقا قوشۇلۇشى تەس.دوستۇمغا شۇنداق بولغاندىكىن بولدى قىل،ئۇ قىزنى ئۇنتۇپ كەت دەپ كۆپ نەسىھەت قىلىپ كەتتىم لىكىن پايدىسى بولمۇغۇدەك......قېرىنداشلار سىلەر بىىر دەپ بېقىڭلارچۇ ،دوستۇم قانداق قىلسا ئاتا-ئانسى ۋە ئۇرۇق -تۇققانلىرىنى قايىل قىلالايدۇ؟؟بۇ ئىشقا قانداق يول تاپىدۇ؟؟ياخشىراق يول كۆرسەتىكەن بولساڭلار؟؟؟مىنىڭ يەنە بىر قانچە سۇئالىم بار شۇنىڭغا بىر جاۋاپ بەرگن بولساڭلار.

سۇئال:
توي قىلمىغان يىگىت،توي قىلىپ ئاجىرشىپ كەتكەن قىزنى نىكەھىغا ئېلىش دۇرۇسمۇ؟؟؟بۇنۇڭغا دىنىمىز  نىمە  دەيدۇ؟؟؟

    مەنبە:ئويلۇنۇپ ساراڭ بۇلۇپ قالاي دىگەن كاللام.

ئەسسالامۇ ئەلەيكۇم بۇرادەر تىچ ئامانچىلىقمۇ ئۇزۇن بولدى كۆرۈنمىدىڭىز يا مەن كۆرمىدىممۇ ؟
ئەمدى ما تېمىدىكى ئىشلارغا ھەم دوستىڭىزنىڭ بېشىغا كەلگەن كۈنگە كەلسەك دوستىڭىز ئۇ قىزنى ياخشى كۆرگەن بولسا ئۇ قىزمۇ دوستىڭىزنى ھەقىقىي ياخشى كۆرگەن بولسا ،دوستىڭىز ئاتا-ئانىسىغا ئېرىنمەي خىزمەت ئىشلىسۇن(اللا ئاسان قىلغاي)مىنىڭچە چوڭلارنىڭ رازى بولماسلىقى مۇمكىن ئەمەس،ئۇقۇمۇشلۇق كىشىلەر بولسىلا.پۇرسەت كۈتسۇن ئالدىراپ بەل قۇيىۋەتمىسۇن.قىز ئىمانلىق ئەدەپ-ئەخلاقلىق بولسا دوستىڭىزنىڭ ئاتا-ئانىسى چۇقۇم ياقتۇرۇپ قالىدۇ،ھازىر مۇشۇ ھالىتىدە ئۇ دوستىڭىز قىزنى تاشلىۋەتسە بىچارە قىز بۇرۇنقى كۆڭۈل ئاغرىقلىرى قۇشۇلۇپ تۈگىشىپ كىتىشى مۇمكىن.بىر ئادەمنى ياخشى كۆرمەك ئۇنچە ئاسان بولمىغىنىدەك ئايرىلماقمۇ ئۇنچە ئاسان ئەمەس.توي قىلىپ ئاجراشسا نىمە بولۇپتۇ؟بەخىتلىك ئۆتۈپ كەتسىلا ھەقىقىي ھايات دىگەن شۇ.ئادەمنىڭ ئىچىگە كىرىپ چىققىلى بولمىغاندىن كىيىن ھەرخىل مىجەزلىك ئادەم بىلەن تۇرمۇش قۇرۇپ قالىدىغان گەپ،ئاشۇنداق كوچىدىكى لۈكچەكلەرنى كۆرسە قۇيرۇقىنى چۆۋىسىگە تىقىپ تىكۋىتىدىغان ئۆيىگە كەلسە ئايالىنى دومبالايدىغان  لاتا غىلاپ ئەرلەرگە تىگىپ قالسا تۇرمۇشتا ئاخىرغىچە بەرداشلىق بىرەلمەيدىغان گەپ.دوستىڭىزغا مىنىڭ گېپىمنى يەتكۈزۈپ قۇيۇڭ ئوغۇل بالا قەدەمنى ئالدىمۇ ئالدىراپ يېنىشقا بولمايدۇ، ئىلگىرلەشكە بولىدۇ.توي قىلىپ ئاجراشسىمۇ ئۇمۇ اللاھنىڭ بەندىسى.
