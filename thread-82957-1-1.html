ۋىنجاباۋ: جۇڭگۇدا سىياسى تۈزۈلمە ئىسلاھاتى مۇۋەپپەقىيەتلىك بولمىسا، مەدەنىيەت زورئىنقىلابىنىڭ تىراگىدىيەسى قايتا يۈزبىرىشى مۇمكىن



جۇڭگۇ تور تىلىۋىزىيەسى خەۋرى: 11-نۆۋەتلىك  مەملىكەتلىك خەلق قۇرۇلتىيى 5-قېتىملىق يىغىنى يېپىلدى، زۇڭلى ۋىنجاباۋ خەلق سارىيىدا جۇڭگۇ ۋە چەتئەل مۇخبىرلىرى بىلەن يۈزكۈرۈشتى ۋە سۇئاللىرىغا جاۋاپ بەردى.

سىنگاپۇر" بىرلەشمەئەتىگەنلىك گېزىت" مۇخبىرى سۇئالى: زۇڭلى ياخشىمۇسىز، سىزنىڭ ئەڭ ئاخرقى بىرقېتىملىق ئىككى يىغىندىن كىيىنكى مۇخبىرلارنى كۈتۈۋېلىش يىغىنىدا سۇئال سۇرىيالىغانلىقىمدىن خۇشالمەن. مېنىڭ سۇئال، يېقىنقى نەچچە يىلدىن بىرى سىز  ئوخشىمىغان سورۇنلاردا سىياسى تۈزۈلمە ئىسلاھاتى ھەققىدەتەكىتلىدىڭىز ۋە زور دىققەت قوزغىدىڭىز. سىزنىڭ سىياسى تۈزۈلمە ئسىلاھاتىنى قايتا-قايتا تەكىتلىشىڭىزدىكى سەۋەپ نېمە؟ جۇڭگۇدا سىياسى تۈزۈلمە ئىسلاھاتى ئېلىپ بېرىشتىكى ئەڭ قىيىن بۇلىۋاتقان ئىش نېمە؟ رەھمەت.

ۋىنجاباۋ:  شۇنداق، نەچچە يىلدىن بىرى مەن سياسى تۈزۈلمە ئىسلاھاتى ھەققىدە كۆپ قېتىم توختالدىم، مەندىن نېمەئۈچۈن بۇمەسلىگە بەك دىققەت قىلغىنىمىنى سورىسىڭىز، مەسئۇلىيەت تۈيغۇسى نوقتىسىدىن دەپ جاۋاپ بىرىمەن. " تۆت كىشلىك گوروھ" تارمار قلىنىغاندىن كىيىن ، پارتىيمىز بىرنەچچە تارىخى مەسلە ھەققىدە قارار چىقارغان ۋە ئىسلاھات ئېلىپ بارغان بولسىمۇ، ئەمما " مەدەنىيەت زور ئىنقىلابى" نىڭ خاتالىقى ۋە فىئودالىزىمنىڭ تەسىرىنى تۇلۇق تازىلىيالمىدۇق. ئىقتىسادنىڭ تەرققىياتىغا ئەگىشىپ،  تەقسىماتنىڭ ئادىل بولماسلىقى، ئىشەنچىنىڭ كەمچىل بۇلىشى، پارخورلق- چىرىكلىك  قاتارلىق مەسىللەر كىلىپ چىقتى، بۇمەسىللەرنى ھەل قىلىشتا، ئىقتىسادى تۈزۈلمەئىسلاھاتى ئېلىپ بىرىپلا قالماستىن، بەلكى يەنە سىياسى تۈزۈلمە ئىسلاھاتى، بۇلۇپمۇ پارتىيىمىز ۋە دۆۋلەت رەھبەرلىرى تۈزۈمى ئىسلاھاتى ئېلىپ بېرىش لازىمىلىقىنى چۇڭقۇر بىلەتتىم. نۆۋەتتە ئسلاھات ھۇجۇم باسقۇچىغا كىردى، سىياسى تۈزۈلمە ئىسلاھاتىنىڭ مۇۋەپپىقىيىتى بولمايدىكەن، ئىقتىسادى تۈزۈلمە ئىسلاھاتىنى ئاخىرغىچە ئېلىپ بارغىلى بولمايدۇ، ھازىرغىچە قولغا كەلگەن نەتىجىلەرنىڭ يەنە قولدىن كىتىش ئىھتىمالىمۇ بار، جەمىيەتتىكى يېڭى مەسىللەرمۇ نىگىزىدىن ھەل بولماسلىقى مۇمكىن، مەدەنىيەت زورئىنقىلابىدىن ئىبارەت بۇ تارىخى تىراگىدىيە يەنە قايتىدىن يۈزبىرىشى مۇمكىن. ھەربىر مەسئۇلىيەت تۇيغۇسىغا ئىگە پارتىيىلىك كادىرلاردا تەخىرسىزلىك تۇيغۇسى بۇلىشى كېرەك. ئەلبەتتە، مەن ئىسلاھاتنىڭ قىيىنلىقىنى بىلىمەن، ئاساسلىقى ھەرقانداق بىر ئىسلاھاتقا قارتا، خەلقنىڭ ئويغۇنىشى، قوللىشى، ئاكتىپچانلىقى ۋە ئىجاتچانلىق رۇھى بۇلىشى كېرەك. جۇڭگۇغا ئوخشاش 13مىلىيارتتىن ئارتۇق نۇپۇسقا ئىگە چوڭ ئەلدە يەنە دۆلەت ئەھۋالىنى چىقىش قىلىش، تەدرىجى ھالدا سوتسىيالىستىك دىمۇگىراتتىك سىياسىنى ئورنۇتۇش لازىم.بۇئاسان ئىش ئەمەس، ئەمما ئىسلاھات ئالغا ئىلگىرلىشى كېرەككى ھەرگىزمۇ توختاپ قالماسلىقى، ھەتتا ئارقىغا يېنىپ كەتمەسلىكى لازىم، توختاپ قېلىش ۋە ئارقىغا چىكىنىشنىڭ چىقىش يولى يوق. ماڭا ئايانكى، كىشلەر مېنىڭ نېمەدېگەنلىكىمگە، مېنىڭ غايەئىتىقاتىمغا قارايدۇ، شۇنداقلا ئۈزۈمنىڭ تىرىشچانلىقىم ئارقىلىق قانداق نىشاننى ئەمەلگە ئاشۇرالايدىغانلىقىمغا قارايدۇ. كۆپچلىككە شۇنداق دىيلەيمەنكى، جۇڭڭۇنىڭ ئىسلاھات ئېچىۋىتىش ئىشلىرى ئۈچۈن، مەن ئاخىرقى بىرتىنىقىم قالغىچە كۈرەش قىلىمەن. رەھمەت!

ۋىنجاباۋنىڭ مۇخبىرلارنى كۈتۈۋېلىش يىغىنىدىكى قىياپىتى ھەققىدىكى تەھلىل:

. نۆۋەتتە ئسلاھات ھۇجۇم باسقۇچىغا كىردى، سىياسى تۈزۈلمەئىسلاھاتىنىڭ مۇۋەپپىقىيىتى بولمايدىكەن، ئىقتىسادى تۈزۈلمە ئىسلاھاتىنى ئاخىرغىچە ئېلىپ بارغىلى بولمايدۇ، ھازىرغىچە قولغا كەلگەن نەتىجىلەرنىڭ يەنە قولدىن كىتىش ئىھتىماللىمۇ بار، جەمىيەتتىكى يېڭى مەسىللەرمۇ نىگىزىدىن ھەل بولماسلىقى مۇمكىن، مەدەنىيەت زورئىنقىلابىدىن ئىبارەت بۇ تارىخى تىراگىدىيە يەنە قايتىدىن يۈزبىرىشى مۇمكىن. ھەربىر مەسئۇلىيەت تۇيغۇسىغا ئىگە پارتىيىلىك كادىرلاردا تەخىرسىزلىك تۇيغۇسى بۇلىشى كېرەك.( زۇڭلى ۋىنجاباۋنىڭ 11-نۆۋەتلىك خەلق قۇرۇلتىيى5-قېتىملىق يىغىنى مۇخبىرلارنى كۈتۈۋېلىش يىغىنىدىكى سۈزى)

ۋىنجاباۋنىڭ بۇسۈزىدە بەش تۈرلۈك مۇھىم مەزمۇن بار:

بىرىنجىسى ئىسالاھات ھۇجۇم باسقۇچىغا كىردى. بۇيەردىكى " ھۇجۇم" سۈزى " سىياسى تۈزۈلمە ئىسلاھات" دىن ئىبارەتتۇر.

ئىككىنجىسى ناۋادا سىياسى تۈزۈلمە ئىسلاھاتىدىن ئىبارەت بۇ قورغان ئېلىنمايدىكەن، سىياسى تۈزۈلمەئىسلاھاتىدىن ئىبارەت بۇ ئىسلاھاتنىڭ مۇۋەپپىقىيىتىگە ئىرىشكىلى بولمايدۇ، ئىقتىسادى تۈزۈلمەئىسلاھاتىمۇ ئاخىرغىچە ئېلىپ بېرىلماسلىقى مۇمكىن.

ئۈچۈنجسى مەدەنيەت زورئىنقىلابى تارىخى تىراگىدىيەدۇر.

تۆتىنجىسى سىياسى تۈزۈلمەئىسلاھاتىنىڭ مۇۋەپپىقىيىتى بولمايدىكەن، قولغا كەلگەن نەتىجىلەرنىڭ يەنە قولدىن كىتىشى مۇمكىن، تارىخ ئارقىسىغا چىكىنىپ، مەدەنىيەت زورئىنقىلابىدىن ئىبارەت تارىخى تىراگىدىيەنىڭ يەنە قايتا يۈزبىرىشى مۇمكىن.

بەشىنجىسى ھەربىر پارتىيەئەزاسى ۋە رەھبەردە تەخىرسىزلىك تۇيغۇسى بۇلىشى كېرەك.

بۈگۈنكى جۇڭگۇدا، پارتىيە-ھۈكۈمەت كادىرلىرى ھازىرقى ئەھۋالدىن نارازى. سوداسانائەتچىلەرمۇشۇنداق، ئۇقۇغۇچىلارمۇ شۇنداق، دېھقانلار مۇ شۇنداق، ئىشچى-دېھقان-تىجارەتچى- ئۇقۇغۇچى، ھەممىسى شۇنداق بۇلىشى مۇمكىن. بۇھالنى " ھەممە تەرەپنى كىرزىس قاپلاپ كەتتى" دېگەن سۆزبىلەن تەسۋىرلىسەك ھەرگىزمۇ مۇبالىغە بولماس. مۇشۇنداق ئەھۋال ئاستىدا، مەيلى قارامۈشۈك ياكى قىزىل مۈشۈك بولسۇن، ھەممىسى پىچاقلىرى بىلەپ تەييارتۇرماقتا، ناۋادا پۇرسەت پشىپ يىتىلىسىلا ھەممىسى پىچاقلىرىنى قۇلىغا ئالىدۇ...

تارىخقا نەزەرسالغىنىمىزدا، ھەربىر زور مۇھىم ۋەقەنىڭ ئازگىلىغا تولدۇرۇلغىنى خەلقنىڭ قان ۋە گۈشى، جۇڭگۇ تارىخىدىكى ھەربىر فىئودال پادىشاھلىقنىڭ ئالمىشىشىدىمۇ، كومىنىستىك پارتىيىسى بىلەن گومىنداڭ ئوتتۇرىسدىكى ئىچكى ئۇرۇشتىمۇ،  ئون يىلغا يېقىن داۋام قىلغان مەدەنىيەت زور ئىنقىلابىدىمۇ،ھەممىسىدە ئەلەملىك كۈرەشلەردە ئۈلۈپ كەتكىنى يەنىلا خان- قەسىرىدىكى ئەمەلدار-تۈرىلەر بولماستىن، بەلكى ئاددى پۇقرالاردۇر. ئۆتمۈشنى ئەسلەش تۇلۇمۇ ئازاپلىقتۇر، يېقىنقى يۈزيىلدىن بىرى، مىللىتىمىز (ت: خەنزۇمىللىتىنى دېمەكچى)  نۇرغۇن قانلارنى ئاققۇزدۇق، گەرچە ئالغا بېسىش يۇلىدا بىز ھەرخىل سىناقلارغا بەرداشلىق بەرگەن بولساقمۇ، ئەمما بىز قان تۈكۈشتىن بىزاربولدۇق، قېرىنداشلارئارا قىرغىنچىلقتىن  ۋە رەھىمسىزلەرچەئېلىپ بېرىلىدىغان كۈرەشتىن بىزاربولدۇق. ئەگەر بىرلائادەمنىڭ قېنى بىلەن پۈتۈن شەرىقنى بۇياپ قىزارتقىلى بولسا ئىدى، مەن كۈزۈمنى يۇمۇپ تۇرۇپ باشقىلارنى قىلچنى ماڭا چېپىشقا دەۋەت قىلغان بۇلاتتىم، ئەمما بۇبىرخام خىيال خالاس، باشقىلار شەخسى مەنپەتى ياكى كىچىك گورھ مەنپەتى ئۈچۈن رەھىمسىزلەرچە كۈرەش قىلىۋاتقاندا، زىيان تارتىيدىغىنى يەنىلا سانسىزلىغان پۇقرالارئىدى.

سىياسى تۈزۈلمە ئىسلاھاتى مۇۋەپپىيەتلىك بولمايدىكەن، " مەدەنىيەت زورئىنقىلابى" دىن ئىبارەت تىراگىدىيە قايتا يۈزبىرىشى مۇمكىن دېگەن بۇسۆزدە ئاگاھلاندۇرۇلغىنى ھەربىر پارتىيەئەزاسى ۋە رەھبىرى كادىرلا بولماستىن، بەلكى ھەربىر ۋىجدانى بار جۇڭگۇ خەلىقىدۇر.

سىياسىى تۈزۈلمەئىسلاھاتى مەغلۇپ بولسا قانداق ئاقىۋەتلەر كىلىپ چىقىدۇ؟

بۇنى ۋىنجاباۋ ئۇچۇق تۇنۇپ يەتكەن بۇلۇپ، مەدەنىيەت زورئىنقىلابىدىن ئىبارەت تارىخى تىراگىدىيە قايتا كۈرۈلىشى مۇمكىن، دېدى. مەدەنىيەت زورئىنقىلابى تىلغا ئېلىنسا، كىشىلەر يولداش ماۋزىدۇڭنىڭ :" جاھان چوڭ مالىمان بولغاندىلا ئاندىن جاھاننى چوڭ تۈزۈگىلى بۇلىدۇ، شۇڭا يەتتە- سەككىز يىلدا بىرقېتىم مۇشۇنداق ھەركەتلەرنى ئېلىپ بېرىش كېرەك." دېگەن سۈزىنى ئىسىگە ئالىدۇ. " يەتتە- سەككىز يىلدا بىرقېتىم ئېلىپ بېرىش كېرەك" دېگەن سۆز ھەممە ئادەمنى قورقىتۋىتىدۇ، بۇلۇپمۇ ئەمەلدارلار تېخىمۇ شۇنداق، ئويلاپ بېقىڭچۇ، يەتتە- سەككىز يىلدا بىرقېتىم بىرتۈركۈم ئەمەلدار كۈرەش قىلىنسا... يائاللا.

ئەمما ھازىر مەدەنىيەت زورئىنقىلابىدىن بەكلا يىراقتا، ھەتتا ئۇنتۇلۇپ كەتتۇق. فىئوداللىق ئىدىيەنىڭ يىلتىزى جۇڭگۇلىقلارنىڭ مىڭىسىدە بەكلا چۇڭقۇر، بۇقاراش سىياسى تۈزۈلمە ئىسلاھاتىغا ۋە دىمۇگىراتتىيە، ئەركىنلىك، كىشىلىك ھۇقۇق قاتارلىقلارغا توسقۇنلىق قىلىدۇ. ئىككىنجى ئەۋلات بايلار، ئىككىنجى ئەۋلات ئەمەلدارلارنىڭ ناچار تەسىرى ۋە جەمىيەت تەقسىماتىدىكى ئادىل بولماسلىق كىشىلەردە ئېغىر بولغان ئادالەتسىزلىك تۇيغۇرسى پەيدا قىلماقتا. چىرىكلىك يامراپ كەتكەن پارتىيە ۋە ھۈكۈمەت ئەمەلدارلىرىنىڭ خاتا ھاكمىيەت يۈزگۈزىشى، بايلار ۋە ئەمەلدارلارغا ئۆچمەنلىك تۇيغسى پەيدا قىلىدۇ. ھۈكۈمەت ئەمەلدارلىرىنىڭ ئىسلتىدىكى قۇپاللىق، شەكىلۋازلىق، بىئوگىراتلىق  خەلقنىڭ ھۈكۈمەتكە بولغان زۇراۋانلىق ۋاستىسى بولمىغان ۋاستىلار بىلەن قارشى تۇرىشىنى كەلتۈرۈپ چىقىرىدۇ. ۋەھالەنكى غەرىپنىڭ قىممەت قارىشىنىڭ كېڭىيىشى دۆۋلەت ئىچى-سىرتىدىكى دۈشمەن كۈچلەرنىڭ غەرىپلەشتۈرۈش پارچىلاش ئاجىزلاشتۇرۇش سۈيقەستى ئاستىدا، سوتسىيالىستىك قىممەت قاراش بۇزغۇنچلىققا ئۇچۇراپ، ئاساسى ئېقىم ئاخپارات ۋاستىللىرى ۋە ئەنئەنىۋى مائارىپنىڭ سوتسىيالىزىمنىڭ ئەۋزەللىكى ھەققىدىكى تەشۋىقاتلارنى بۇزغۇنچىلققا ئۇچرۇتىدۇ.

ئەگەر سىياسى تۈزۈلمە ئىسلاھاتى مۇۋەپپىقەتلىك بولمايدىكەن، خەلق ئاممىسىنىڭ دىمۇگىراتتىيە ۋەئەركىنلىككە تەلپۈنۈشى ۋە يېڭى ئىجتىمائى قاتلام ئىقتىسادىنىڭ قەت كۈتۈرۈشىدىن كىيىن، سىياسى تەلەپلەرنىڭ قاندۇرۇلماسلىقى قالايمىقانچلىق كەلتۈرۈپ چىقىرىشى مۇقەررەردۇر. ئەگەر ئەمەلدارلىرىمىزنىڭ ئەمەل-دەرىجىسىنىڭ ئۆرلىشىدە بىرىنجى قولنىڭلا دېگىنى ھىساپ بۇلۇش، ۋەزىپىگە يۇقىرىدىنلا تەيىنلەش، كوللىكتىپ رەھبەرلىك تەكىتلەنسىمۇ، ئەمما ماۋزىدۇڭ دەۋرىدىكىگە ئوخشاش شەخىسلەر بىر-بىرىگە تايىنىش بوي سۇنۇش مۇناسىۋىتى شەكىللىنىدۇ.

مۇشۇنداق بولغاندا، ھۇقۇق تالىشىش كۈرىشى يۈزبىرىپ، ئەمەل سېتىۋېلىش-سېتىش قىلمىشىنى توسقىلى بولمايدۇ، ئەمەلدارلار ھازىردىكىگە ئوخشاش، پەقەت خىيانەت قىلىشقا جۈرئەت قىلسىلا مىلىيۇنىرغا، ھەتتا ئونمىلىيۇن، يۈزمىلىيۇن پۇلغا ئىگە بايلارغا ئايلىنىۋىرىدۇ. ئەمما سىياسى تۈزۈلمەئىسلاھاتى ئېنىقكى مانابۇ ئەمەلدارلارنىڭ پارخورلىق قىلىش مەنپەتىگە بېرىپ تاقىلىدۇ، شۇنىڭ ئۈچۈن پارخورلار ئامال بار سىياسى تۈزۈلمە ئىسلاھاتىغا قارشى تۇرىدۇ ياكى ئۇنى ئارقىغا سۆرەيدۇ.

شۇنىڭ ئۈچۈن،يولداش دىڭشاۋپىڭنىڭ ئىسلاھات ئېلىپ بارماسلىق ھالاكەت يۇلىغا ماڭغانلىق دېگەن سۈزىنى ئىسىمىزدە چىڭ ساقلاپ، ئىسلاھات بايرىقىنى ئىگىز كۈتۈرۈپ،تۈرلۈك ئىسلاھاتلارنى تېخىمۇ چۇڭقۇرلاشتۇرۇپ، خەلق ئاممىسىنى ئورتاق قۇرۇشتىن بەھىرقىلىشىمىز لازىم.

ئىچكىرىدە قانداق مەن بىلمەيمەن ، بىلمىگەندىكىن سۆزلىمەيمەن. بىراق مۇشۇ يۇرتىمىزنى ئېلىپ ئېيتايلى ، قايسى بىر ۋالىي ، نازىر ،ھاكىملارنىڭ  مىليونلاپ پۇلى ، نەچچە يەردە ئۆي-زىمىنلىرى يوق ؟ بۇلارنىڭ مۇئاشى بەلكى ئادەتتىكى كادىرلاردىن 2-3ھەسسە يۇقىرىدۇ، 10ھەسسە يۇقىرى بولغان تەقدىردىمۇ شۇنچىۋالا مۈلككە ئىگە بولالامدۇ؟ ئەجەبا ئىسلاھاتنىڭ غەلبىسىنى شۇلار يالغۇز قولغا كەلتۈرگەندەك، ۋاستەتاللىماي بايلىق توپلىسا، پارا ئالسا ،مەنسەپ سودىسى قىلسا ، گورۇھ توپلاپ قارشى پىكىردىكىلەرنى باسسا نېمىشقا سۈرۈشتۈرۈلمەيدۇ؟
پارىخورلۇققا ئىسپات يوق دىگەنگە كىچىك بالىمۇ ئىشەنمەيدۇ ، ئادەمنىڭ خىيالىغا كەلمەيدىغان بۇلۇڭ- پۇشقاقلارغا يوشۇرنىۋالغان ئۇنسۇرلارنى نەچچە كۈندىلا تېپىپ چىقالايدىغان بىخەتەرلىك تارماقلىرى ، ئەجەبا بۇ پارىخورلارنىڭ يۇشۇرۇن مال -مۈلكىنى پەقەت يەشارىدا بولسىلا تېپىپ چىقىشقا تامامەن قادىر .
مۇھىمى بۇ ئەمەلدارلار ئۆزئارا تىىل بىرىكتۈۋالغان ، توردەك مۇناسىۋەت چەمبىركىنى قۇرۇۋاالغان، شۇڭا بىر-بىرىگە چاپان ياپىدۇ، بولمىغاندا بىرەر شورپىشانە پارىخورنى قۇربان قىلىپلا ، خەلقنىڭ كۆزىنى بويايدۇ خالاس.
ۋىنجاباۋ ئەپەندى توغرا دەيدۇ، بىر جەمئىيەتتە توغرا ،ئادىل بولغان بايلىق تەقسىماتى ، باراۋەر رىقابەت پۇرسىتى ، ھەركىىم ئۆز پىكرىنى ئېيتالايدىغان كىشىلىك ھوقۇق بولمايدىكەن، ئەڭ مۇھمى ئەمەلدارلارنى تىزگىنلىيەلەيدىغان، ھۆكۈمەتنى خەلىق نازارەت قىلالايدىغان سىياسىي تۈزۈم بولمايدىكەن ،ئىقتىسادىي مۇۋەپپقيەتلەر   ۋاقىتلىق بولۇپ قالىدۇ، خەلق نارازى بولسا سۈكۈت قىلىپ تۇرىۋەرمەيدۇ. جەمئىيەت مۇقىملىقىنى  تەشۋىقات ۋە بېسىم بىلەنلا ئىشقا ئاشۇرغىلى بولىدۇ دەپ قاراش ،سوتسىيالىىستىك ئەلدىكى مەمۇرىي باشقۇرغۇچىلارنىڭ ئەمەس، فىئوداللىق جەمئىيەتتىكى چىرىك ،زوراۋان ئەمەلدارلارنىڭ قارىشى.
مۇقىملىقنى، ئىناقلىقنى  پەقەت خەلىقنىڭ كۆڭلىنى ئېلىش بىلەن ، خەلقنىڭ ئىشەنچىسىنى قولغا كەلتۈرۈش بىلەنلا ئىشقا ئاشۇرغىلى بولىدۇ،

مەسلەكداشلىق مىللەتداشلىقتىن ئەلا!
