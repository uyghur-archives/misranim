بۈگۈن شىنجاڭ كوماندىسى ئۆز مەيدانىدا شاڭخەي جۇڭباڭ كوماندىسىنىڭ كۈتىۋالدى، مۇسابىقە باشلىنىش بىلەنلا شىنجاڭ كوماندىسى بىر خىل پاسسىپ ھالەتكە چۈشۈپ قالدى. ئەزالار ئۆزىنىڭ ئەڭ ياخشى ھالىتىنى تاپالمايۋاتقاندەك بولۇپ قالدى، بۇنىڭغا ئەگىشىپ پۇتبول مەستانىلىرىنىڭ قىزغىنلىقىمۇ يۇقىرى كۆتۈرەلمىدى، بۇ پۇرسەتنى چىڭ تۇتقان شاڭخەي جۇڭباڭ كوماندىسىنىڭ ئەزالىرى مۇسابىقە باشلىنىپ 18 مىنۇت بولغاندا، بىر توپ كىرگۈزۈپ 0 گە قارشى 1 نەتىجە بىلەن شىنجاڭ كوماندىسىنى كەينىدە قويدى. بۇنىڭ بىلەن شىنجاڭ كوماندىسىنىڭ ئەزالىرى سەل ئويغانغاندەك بولۇپ، ئۆز ھالىتىگە كىرىشكە باشلىدى.
ئالدىنقى مەيدان مۇسابىقە داۋاملىشىپ 45 مىنۇت بولغاندا، قارشى تەرەپنىڭ بىر ئەزاسى قائىدىگە خىلاپلىق قىلغانلىقتىن، قىزىل كارتا بىلەن ئاگاھلاندۇرۇلۇپ، مەيداندىن چۈشۈرۋېتىلدى. شىنجاڭ كوماندىسىمۇ قارشى تەرەپنىڭ بۇ ۋاقىتتىكى قالايمىقانلىقىدىن پايدىلىنىپ، شۇنداق چىرايلىق بىر كاللا توپنى قارشى تەرەپ ۋارتاسىغا كىرگۈزىۋەتتى. شىنجاڭ كوماندىسى 4-نومۇرلۇق ئەزاسىنىڭ بۇ بىر كاللا توپى بىلەن نەتىجىنى 1:1 قىلىپ تەڭلاشتۇرۋىلىپ يېرىم مەيدان مۇسابىقىنى ئاخىرلاشتۇردى. مەيداندىكى توپچىلارمۇ، كۆرۈش سۇپىسىدىكى مەستانىلەرمۇ قايتىدىن جانلىنىپ كەتتى، بولۇپمۇ مەستانىلارنىڭ قىزغىنلىقى بىراقلا ئۆرلەپ كەتكەن بولۇپ، ھۇررا ۋە ئالقىش سادالىرى مەيداننى بىر ئالدى.
كېيىنكى يېرىم مەيدان مۇسابىقىدە شىنجاڭ كوماندىسى قارشى تەرەپتىن بىر ئەزاسى ئارتۇق بولۇش ئەۋزەللىكىدىن پايدىلىنىپ، قايتىدىن كۈچلىنىپ قارشى تەرەپنى بېسىپ، قارشى تەرەپنىڭ مەيدانىدا توپ تېپىشكە باشلىدى، قارشى تەرەپ ۋارتاسىغىمۇ نۇرغۇن چىرايلىق توپلار تېپىلىپ تۇردى، ئەپسۇس بۇ توپلار شۇنچە چىرايلىق ئۇچۇپ قارشى تەرەپ ۋارتاسىغا يېقىنلاشقان بىلەن ۋارتا سىزىقىدىن ئۆتۈش نېسىپ بولمايلا، 45 مىنۇت ۋاقىت توشۇپ كەتتى. شۇنداق قىلىپ شىنجاڭ كوماندىسى ئۆز مەيدانىدا 1:1 نەتىجە بىلەن شاڭخەي جۇڭباڭ كوماندىسى بىلەن تەڭلىشىپ قالدى.

كوماندىمىز مۇشۇ پىتىلا ئوينىسا،ھىلغۇ B دەرىجىلىكەر تۇرۇپ تۇرسۇن،شىمالى رايوندىنمۇ چىقالماسلىقى مۇمكىن.بۇ پەسىللىك كوماندىمىزنىڭ بەك بولسا 4-5 مەيدان مۇسابىقىسىى قالدى..ھازىر كوماندىمىز تۇراقسىز ھالەتتە 5-6ئورۇندا پىقىراپ يۈرىدۇ..
شىمالى رايوندىن چىقىش ئۈچۈن چوقۇم ئالدىنقى تۆتنىڭ ئىچىگە كىرىشى كىرەك.ئاندىن كىيىن شىمالى رايوندىن چىققان ئالدىنقى تۆت كوماندابىلەن جەنۇبى رايوندىن چىققان ئالدىنقى تۆت كوماندا قايچىلاشتۇرۇلۇپ مۇسابىقە ئويناپ ،ئەڭ ئاخىردا ئۈزۈپ چىققان ئىككى كوماندا A دەرىجىلىكلەر مۇسابىقىسنى ئويناش سالاھىيتىگە ئىرىشىپ،سەۋىيە جەھەتتىن سەل يۇقىراق مۇسابىقە پەسلىگە كىرىدۇ....
A دەرىجىلىكەردىمۇ ئوخشاشلا شىمالى ۋە جەنۇبى رايونغا ئايرىلىدۇ،قالغان جەريان ئىلگىركى ئۇسۇل بىلەن ئوينىلىپ ئەڭ ئاخىردىكى ئۈزۈپ چىققان ئىككى كوماندا رەسمى ھالدا، خاڭجۇ دىكى بارى ،شەندۇڭدىكى مىر ئەخمەت،گۇاڭجوۋ خېڭدادىكى شاۋكەتلەر ئويناۋاتقان،سەۋىيە جەھەتتىن  يۇقىرى تەلەپ قويدىغان ،جۇڭگو دەرىجىدىن تاشقىرى مۇسابىقە پەسلىگە كىرىدۇ...بۇ  مۇسابىقە پەسىلدە توپ تىپىۋاتقان نۇرغۇنلىغان خەلىقئارالىق توپچىلار بار، شاڭخەي كوماندىسىدا توپ تىپىۋاتقان،بۇرۇنقى چېلسى كوماندىسىدىكى دېروبا بىلەن ئانېركا ئەڭ تىپىك ۋەكىلەرنىڭ بىرى....

كوماندىمىزنىڭ ھازىرقى ئەھۋالىدىن ئېلىپ ئىيتقاندا Bدەرىجىلىكەردىنمۇ چىقىشى سەل تەسكە توختايدىغان ئوخشايدۇ..لىكىن ھىچ گەپ يوق ! بۇ يىل چىقالمىسا كىلەر يىل بار.بۇلۇپمۇ ئوغلانلىرىمىزنىڭ ئارقىسىدا نەچچە ئون مىڭلىغان ئارقا سەپ سادىق قوللىغۇچىلىرى بار،كوماندىمىزنىڭ تۇنجى پەسىللىك مۇسابىقىنى ئويناپ مۇشۇ نەتىجىلەرگە ئېرىشىشىمۇ كىشىنىڭ قايىللىقىنى قوزغىماي تۇرالمايدۇ!،...گەرچە كوماندىنىڭ مەيدان كۆرۈشى ،مۇسابىقىگە ماسلىشىشى،پاس جەھەتتىن ئاجىزلىق ،شەخسىيەتچىلىك،قالايمىقانچىلىق كۆرۈلگەندەك بولسىمۇ،لىكىن ئېشىنىمىزكى ئەزىمەتلىرىمىز بۇخىل مەسىىلەرنى چوقۇم تۈزىتىپ كىتەلەيدۇ!

ئەتە ئۆگۈن ئاللا بۇيرۇسا جېنىمىزلا بولدىكەن،ئوغلانلىرىمىزنىڭ تالاش تارتىشلاردا،يۇقىرى باھالاردا ياۋرۇپادىكى داڭلىق كوماندىلارغا ماڭغىنىنى چوقۇم كۆرىمىز..! چۈنكى بىز  تارىختىكى جاھاننى تىتىرەتكەن ئوغۇزخاننىڭ ئەۋلادى ئەمەسمۇ...!
ھۇررا ھۇررا...تاشئالما!
