بۇلارغا قاراپ كۈلۈشنى ياكى يىغلاشنى بىلمىدىم . ئاتا -بوۋىلىرىمىزدىن مىراس قالغان بۇ ھۈنەرلەردىن پەخىرلىنەيمۇ ؟ ياكى زامانىۋى بۇيۇملارغا سېلىشتۇرغاندا خۇددى كونا بۇيۇمدەك بوينىنى قىسىپ تۇرغان بۇ مەخسۇلاتلىرىمىزدىن نومۇس قىلايمۇ ؟ بىزنىڭ يېنىك سانائىتىمىزنىڭ زامانىۋىلىشاماسلىغىنىڭ تۈپ سەۋەبى نەدە ؟ تارىخقا نەزەر سالساق ،ئىنساننىڭ كۈندۈلۈك تۇرمۇشى بىلەن مۇناسىۋەتلىك بولغان  تۇرمۇش بۇيۇملىرىمىز ھەرقانداق بىر مىللەتنىڭ ئىجات قىلغىنىدىن كۆپ ئىكەن،ياراتقان مەنىۋى مەدەنىيىتىمىزمۇ ھەرقانداق بىر مىللەتتىن قىلىشمايدىكەن ؟ قانداق قىلىپ بىزنىڭ  بۇ ھۈنەرلىرىمىز يىرىك -خام ھالەتتە تۇرۇپ قالدى ؟ ئۇ قانداق قىلىپ پەن-تېخنىكىدىن ئايرىلىپ قالغان ؟ بىز  بۈگۈنكى كۈنگىچە يېمەك يولى مەدەنىيىتىمىزگە ۋارىسلىق قىلىپ باشقىلار ياسىغان نەرسىلەرنى(شەرىقنىڭكىنى غەرىپكە ،غەرىپنىڭكىنى شەرىققە سېتىپ )  ئېلىپ -سېتىپ جاھاندارچىلىق قىلىۋاتىمىز (سودىگەرچىلىك ،چىگرا سودىسى---ئەمىلىيەتتە شەكلى ئۆزگەرگەن يىپەك يولى سودىسىدۇر ) .ئۆزىمىز نېمىشقا بىرەر  نەرسىنى ئەڭ بۇرۇن ،ئەڭ ئىلغار ،ئەڭ بازارلىق قىلىپ ئىشلەپ باقمايمىز ؟؟؟؟؟؟؟؟؟ ئۆزىمىزنىڭ ماللىرىنى (ياسىغان بۇيۇملىرىنى) دۇنيانىڭ بۇلۇڭ پۇشقاقلىرىغىچە ئاپىرىپ ساتىدىغان ،ھەممە ئەل خەلىقلىرى باشقىلارنىڭكىنى ئالمايمەن ئۇيغۇرنىڭكىنى ئالىمەن دەپ تالىشىپ سېتىۋالىدىغان كۈنلەرمۇ كېلەرمۇ ؟؟؟؟؟؟؟؟؟
