ئەسسالاممۇ ئەلەيكۇم ئەزىز قېرىنداشلىرىم بۇ تېمىنى ۋە ئىنكاسلارنى بىرمۇ بىر كۆرۈپ چىقىپ بەزى ئىنكاسلارنى كۆرۈپ سۈيۈندۈم بەزى ئىنكاسلارنى كۆرۈپ كۆڭلۈم ناھايىتى يېرىم بولدى . بىرمىزنىڭ تىلىنى بىرمىز چۈشەنمىسەك زادى نىمىنى شۇنچىۋالا تالىشىمىز ؟؟؟؟؟؟؟؟ باشقىلار ساياھەتكە چىقسا ئۇنىڭ نەرى خاتا خالىسىڭىز سىزمۇ چىقىڭ چەتئەللەرگە چىقىپ ساياھەت قىلغانناڭ نەرى خاتا ؟نىمىگە شۇنداق تالىشىپ بىر _بىرڭلارنىڭ دىلىغا ئازار بىرسىلەر  نىمىنى شۈنچىۋالا تالىشىسىلەر . ئۆزەڭنىڭ قەدىرنى ئۆزەڭ قىلمىساڭ خەق قىلىپ بەرمەيدۇ .باشقىلار نۇرغۇن جاپا مۇشاققەتلەرنى يىڭىپ ئۆز ئەجىرنىڭ مېۋىسىنى كۆرسە ئۇنىڭغا ئەسلى چىدىمايمىز . ئۆزىمىز قىلالمايمىز ،باشقىلار قىلسا كۆرەلمەيمىز. ئابدىكىرەم ئابلىزنى مۇنبىرمىز دە بەزى كۆرەلمەس قېرىداشلىرمىزنىڭ ئۇنى بۇنى دەپ كەينىدىن گەپ تاپقىنى راس، گەپنىڭ قىسقىسى نادانلىق توۋا ئەسلى مىللىتىمىز ئارسىدىن چىققان شۇنداق تالانت ئىگىلىرنى قەدىرلىشىمىز ئۇلارنى تېخىمۇ ھۆرمەتلىشىمىز كىرەك ، مىللىتىمىزدىككى بەزى ناچار ئىللەت نادانلىق ئىشقىلىپ تۈزىتىشكە تىگىشلىك كىبىرلەرنى بىزگە ئەينەك ئورنىدا كۆرسىتىپ بېرۋاتقان كىملەر. خەق ئەمەس يەنە ئۆزىمىزنىڭ قېرىنداشلىرى بولغان يازغۇچىلار  ناخشىچىلار  ئابدىكىرەم ئابلىزغا ئوخشاش تالانت ئىگىرى . توۋا ھەيرانمەن . سۆزلەڭ دىسە ئاغىزدا شەھەر ئالىدىغان . مىللەتنىغۇ قۇيۇپ تۇراي ئائىلىسى ئۈچۈن بىرەر ياخشى ئىش قىلىپ بىرەلمەيۋاتقان قېرىنداشلىرىم سىلەر گەپ تېپىپ ئۇنى بۇنى دەۋاتقان  كىشلەر بىزنىڭ قېرىنداشلىرمىز، ياندا بولسا يىگىلى ياخشى دەپ تىكەن ،توردا خەق ئازراق سۆزلەپ قويسا ئەقىل يۈرگۈزۈپ ئويلاپ  باقمايلا خەقنى تىللايدىغانغا نىمە ھەققىمىز بار .  يىرىتقۇچ ھايۋان جېنىدا ئۆز قېرىنداشلىرنى يىمەيدۇ . بىز ئەمدى شۇ تىلسىز ھايۋاندىنمۇ دۆتمۇ. مۇنبەر دىگەن مۇنازىرە سورۇنى ، غەيۋەت قىلدىغان ئۆز گۆشىنى ئۆزى يەيدىغان جاي ئەمەس ئەگەر شۇنداق تالانتىڭىز بولسا نامايەن قىلىڭ .سىزنىڭمۇ مەيدىڭىزگە قىزىل گۈللەر تاقاپ چاۋاكلار بىلەن قارشى ئالىمىز مەدەت بىرمىز .چۈنكى بىز قېرىنداش ھەممىمىزنىڭ ئەجدادى بىر سۈرۈشتۈرسەك ھەممىمىز قېرىنداش،ئىناق ئۆتسەك ھەمىمىزگە ياخشى  .مەدىنىيەت سەۋىيەم تۆۋەن ئوقۇش تارىخىم تولىمۇقىسقا بىلىدىغانلىرىم ناھايىتى ئاز بولغانلىقى ئۈچۈن ئىنكاسىمنىڭ قايىل قىلىش كۈچى ناھايتى تۆۋەن دىمەكچى بولغىنىم قېرىنداشلىرىم ئادەم ئەقلى بىلەن ياشىشى كېرەك. ئۆزەمنى ئېلىپ ئېيتسام مەن كىتاپ ئوقۇشنى ياخشى كۆرىمەن شۇڭا ئۇيغۇر يازغۇچىلىرمىزدىن پەخىرلىنىمەن .ئەگەر ئۇلاربولمىسا ئىدى بىز خەقىنىڭ تەرجىمە ئەسەرلىنى ئۇقۇپ يۈرگەن بۇلاتتۇق. ناخشا ئاڭلاشنىمۇ ياخشى كۆرىمەن ، ھەم ناخشىغا چوڭقۇر ئىشتىياقىم بار ، بەزىدە دوراپ ئېيتپمۇ قويىمەن.ئەگەر ئۇيغۇر سەنئەتكارلىرمىز بولمىغان بولسا بىز يەنە خەقنىڭ كىنى ئاڭلىغان بولاتتۇق. بۇلۇپمۇ ئابدۇللاھ ئابدىريىم ،تاشمۇھەممەت باتۇر  ، ئابلاجان ئاۋۇت ئايۇپ دەيمىز ، يەنە ئەلى كىرورەن ،ئىشقىلىپ سەنئەتكارلىرمىز خېلى كۆپقۇ ئۇلارنىڭ تىرشىپ يېڭى ناخشلارنى ھۇجۇتقا كەلتۇرشى كىم ئۈچۈن يەنىلا بىز ئۈچۈن . بىزنىڭ ئانا تىلىمىزنى دۇنياغا تۇنۇلدۇرۋاتقان كىملەر يەنە شۇ ساناقلىقلا كىشلەر ،سىز  بىلەن بىز توردا ئۆزەمنىڭ راس دەپ دەت تالاش قىلىپ ھېچ ئىشنى باشقا ئەچىقالمايمىز. بولسا ئۇلارنى قوللايلى ،ئەگەر بىرەر خاتالىقنى بايقىساق قايىل قىلارلىق قىلىپ سۆزلىسەك ئۇلارغىمۇ پايدىلىق ئۇلارمۇ ئادەم .خاتالىق ئۆتكۈزمەيدىغان ئىنسان يوق . پەقەت ئىنساننى ياراتقۇچى جانابى ئاللاھ خاتالىقتىن كەمچىلىكتىن يىراق ، ئەسلى بۇ يېرىم كىچىدە بۇرۇنقى كونا مۇنبەردىككى كونا دوستۇمنى مۇشۇ مۇنبەدە بار كەن دەپ ئاڭلاپ قاراپ باقاي دەپ كىرگەن ئىدىم ، ئۆزھالىمغا باقماي ئاتىكاچىلىق قىلىپ كۆپ سۆزلەپ قويدۇم، ھەممىڭلاردىن كەچۈرۈم سورايمەن . ھەراس مۆمىنجاننى ئۇنتۇپ قاپتىمەن مۆمىنجانمۇ ئىسمىنى ئاتاپ تۇرۇپ پەخىرلىنشكە ئەرزىيدىغان ياشلىرمىزنىڭ بىرى مىللىتىمزنىڭ تەرەققىياتى ئۈچۈن ھارماي ئىزدىنۋاتقان ياش قېرىنداشلىرمىزنىڭ بىرى بۇ يازمىنى ئاخىرىدا   heslirim تەھرىرلىگەن. ۋاقتى  2012-10-17 05:21 AM
