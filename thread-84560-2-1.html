  مىنىڭچە بۇنىڭغا قاينىمىسىڭىزمۇ بۇلىدۇ ......... يۇقىرىدا تىلغا ئالغان ئىنكاسلاردىكى 6- ئىنكاستەك ئىشلارنى مەن كۆرمىدىم ، 7- ئىنكاس سەل ئېشىپ كىتىپتۇ ، 9- ئىنكاسمۇ تازا جىڭ باسمايدۇ . قالغان ئىنكاسلارنى لايىقىدا قۇبۇل قىلىشقا بۇلىدۇ . بىز نىمىشقا بارنى يوق دىگۈدەكمىز ؟ نىمىشقا ئۆزىمىزنى قاچۇرىمىز ؟ شۇلارنىڭ ئالدىدا دىسۇن دەپسىز ، ئۇقۇتقۇچى بولغان ئادەم ئانچە - مۇنچە تورغا چىقىدىغاندۇ ؟ مۇشۇنداق مۇنبەرلەرنى كۆرمىگەن ئۇقۇتقۇچى چۇقۇم ياخشى ئۇقۇتقۇچى ئەمەس !
  مەنمۇ ئۇقۇش پۈتتۈرۈپ كىلىپ ئالتۇندەك ياشلىقىمنى ، ئانا قۇشتەك قىزغىنلىقىمنى ، ساپ مۇھەببىتىمنى ، چۈمۈلىدەك تىرىشچانلىقىمنى ......................  سەرپ قىلغۇدەك ھەممە نەرسەمنى بۇ كەسىپكە ئاتىغان ئىدىم . ئەپسۇس ! ھازىر ئويلىسام ئوقۇغۇچىللىرىمدىن باشقا كۆڭۈل خۇشلۇغۇم بولماپتىكەن ، 10 - ئىنكاسنى مەن قىستۇرۇپ قۇياي : ناۋادا مەكتەپتە ھەققانى بىرەيلەن كۆپچىلىك ئۈچۈن كۆكرەك كېرىپ تۆت ئېغىز گەپ قىلىپ كۆپنىڭ مەنپەئەتىنى قوغدىغان بولسا ، ۋاقتى كىلىپ مۇدىر شۇ سەۋەپلىك ئۆچ ئالماقچى بولۇپ كىچىككىنە مەنپەئەتنى بىرىپ ، ھېلىقى ئەسلىدىكى مەنپەئەتى قوغدالغانلارنى  « ئالە ، كوش ، كوش !»  دەپ قويسا كۆپىنچىسى ئاۋۇ ئۆزلىرىنىڭ ھوقۇقىنى قوغدىغانلارنى خام تالاش قىلىۋىتىدۇ .  ئېتىراپ قىلىش كىرەككى ، ئوقۇتقۇچىلاردا ساپا ھەقىقەتەن تۆۋەن ، بۇلۇپمۇ ئەخلاق ساپاسى !!! گۇرۇھۋازلىق ، كۈرەلمەسلىك ، پۇتلاش ، شەكىلۋازلىق ، ئەخلاقسىزلىق ، يۇرتۋازلىق ، تەخسىكەشلىك ................  خېلى چوڭ سالماقتا سىڭىپ كەتكەن .
   شۇكرى .........! 10 يىل ئىشلەپ ئارزۇ قىلمىغان لېكىن ئىشلىگەندىن كىيىن ياقتۇرۇپ قالغان بۇ كەسىپتىن خوشلاشتىم ، نۇرغۇن ئەسلىمىلىرىم قالدى ، نۇرغۇن ئارزۇلىرىم قالدى ، «مۇئەللىم» دىيىش بىلەن ماڭا بولغان ھىسسىياتىنى دەل ئىپادىلەپ بۇلالمىغاندەك ھىس قىلىپ « دادا» دەپ چاقىرىدىغان ، « ئەلباش» مۇئەللىم دەپ ئاتايدىغان ئوقۇغۇچىلىرىم قالدى ، ھەق ئۈچۈن مىقتەك تۇرالايدىغان نامىم قالدى ....................... ئەمما يەتتە يىل ئىمتىھان بىرىپ داۋاملىق بىرىنچى بولساممۇ باھالاشتا نۇمۇر بەرمەي ئوتتۇرا دەرىجىنى ئالالمىدىم ، مۇدىر ئىپلاس بىلەن بارنى بار ، يوقنى يوق دىيىشكەنلىكىم ئۈچۈن ، ئىمتىھاندىكى كۆز بويامچىلىقلارغا قارشى تۇرغانلىقىم ئۈچۈن تۈرلۈك باھانىلەر بىلەن  ئايلاپ مائاشىم تۇتۇلدى ، جوزىنى بىرنى ئۇرسا مەن ئىككىنى ئۇرۇپ مۇنازىرىلەشتىم ، مۇدىر ئىپلاس تۈزۈمنى چىڭىتىش ئۈچۈن  يامانراق مەن قاتارلىق بىر قانچىمىزنىڭ مائاشىنى تۇتقانلىقىنى ، كىيىن قايتۇرۇپ بىرىدىغانلىقىنى دەپ ۋەدە - قەسەملەرنى قىلىپ تۇرۇۋالدى ، ئەمما ھازىرغىچە بىرىمەن دىدى بەرمىدى . ( 4-5 يىل ئىلگىرى تۇتۇلغان مائاشىمنى قايتۇرىۋېلىش ئۈچۈن ،مۇھىمى ھەققانى تۇيغۇم ئۈچۈن ،يول قۇيىلىۋەرسەم ئاۋۇ ئىپلاسنىڭ ئۆزىنى بىلمىگەنلىكى ئۈچۈن ئىنتىزام تەكشۈرۈش ئورۇنلىرى بىلەن ئالاقىلىشىپ قويدۇم .)
   ئوقۇتقۇچىلاردىن نۇرغۇن دوستلىرىم بار . مەكتەپكىمۇ چوڭقۇر مىھرىم بار . مىنىڭچە بولسا بۇيەردە ئانداق دىمەڭلار ، مۇنداق دىمەڭلار دىگەندىن كۈرە ئەتراپىڭلاردىكى ئاشۇنداق ئىشلىرى بار ئىككى - ئۈچ ئوقۇتقۇچىغا شۇنداق تېمىنى ئېچىپ بىرىڭلار ، ئىبرەت ئالسۇن ، مۇنبەردە دوڭ سۈنرۈيدەك ، خۇاڭ جىگۇاڭدەك ، چىيۈ شاۋيۈندەك ئوقۇتقۇچىلارنىڭ يۈز - ئابرويى ئۈچۈن مەيدە كېرىپ چىققان بىلەن ئاۋۇ بىرقىسىملار نامىمىزنى « پوق قىلىۋەتكەن » ئۇنىڭغا ئامال يوق ، ئاجايىپ ئىسىل ئوقۇتقۇچىلىرىمىزمۇ بار ، كۆڭلۈڭلارغا ئالماسسىلەر . بىزنىڭ بۇ يەردە « بارلىقىنى ئۇيغۇر مائارىپىغا بېغىشلىۋېتىپ » توي قىلاي دىسە ئىرىگە بېغىشلايدىغان بىر نەرسە قالماي يۈرۈپ كەتكەن نەچچە « پىداكار»، « قىز » ئوقۇتقۇچىلارمۇ بار .........  تەڭرى كەچۈرسۈن ، ئىنساپ بەرسۈن ، سۈيۈملۈك ئۇيغۇر پەرزەنلىرىنىڭ بۇنداق ناكەسلەرگە قېلىشىنى كۆرگۈم يوق ! نىمىلا بولسۇن ، ياخشى ئۇقۇتقۇچىلار بىزنىڭ ئۈمىدىمىز ، پەخرىمىز ناچارلىرى مۇشۇنداق دۇمبىقىمىز ..................كەچۈرۈڭ ! كەچۈرۈڭلار ! بۇ يازمىنى ئاخىرىدا   albax تەھرىرلىگەن. ۋاقتى  2012-8-6 12:31 PM

   مەن ئوقۇتقۇچى لىكىن ھاراقكەش ئەمەس،مەن ئوقۇتقۇچى لىكىن پاھىشىۋازئەمەس،مەن ئوقۇتقۇچى ئوغرى-قاراقچى ئەمەس.....
بىز زادى قانداق خەق ؟بىردەم ھازىرقى قىزلىرىمىز چەتئەللىك بىلەن قاچتى ،ختاينىڭ قوينىغا ئۆزىنى ئاتتى ،ئوغۇللار غىتمەكلەشتى،قىزلارنى بۇزىۋەتتى،ئانىلار بۇنداق ،ئاتىلار بۇنداق ،ساقچىلار بۇزۇق ،باشلىق بولسا پارىخۇر؟؟زادى ياخىش ئادەم قالمىدىمۇ؟ۋەياكى بىزلەرگەپارتىيەنىڭ غەمخۇرلىقىدا مۇنبەرلەرگە ئولىشىپ غەيۋەت قىلىشتن مىللەتنى سۆكۈشتىن باشقا ئىش قالمىدىمۇ؟؟
مۇنداق ئىككى ئىشنىڭ ئوقۇتقۇچىدىكى خاتالىق ئىكەنلىكىنى ئىتراپ قىلماي بولمايدۇ
1.ئەدەبىيات ئوقۇتقۇچىڭىز سىز گە موبالىغە ،ئوخششتىش سۈپەتلەشنى توغرا ئۈگەتمەپتۇ.
2.سىزگە چوڭلارچە تەپەككۈر قىلىشنى ئۈگەتمەپتۇ.
3،سىزگە تۇز كورلۇق قىلماسلىقنى ئۈگەتمەپتۇ.
4.سىزگە خەت يىزىشنى .ئوقۇشىنى ئۈگىتىپ بەك يامان قىپتۇ.

بەلىكم ئاشۇ سىز كىچىك ۋاقتىڭىزدا ،سىزگە ياخشى بىلەن ياماننى ،«ئا»بىلەن«ب»نى ئۈگەتكەن ،ئۇستازلىرىڭىز ،سىزنى بۈگۈنكى كۈنگە كەلگەندە ئۆزىنى بىر تېيىنگە ئەرزىمەس ئادەم قىلىپ سۈپەتلەش ئۈچۈن ئۈگەتكەن بولغۇيمىتى؟؟؟
مەن ئوقۇغۇچىلىرىغا باسقۇنچىلىق قىلغان ئوقۇتقۇچى دەپ ئاڭلىغان ،لىكىن مىللتىنى ئۇيغۇر دىمەپتىكەن.مەن ئەخلاقىسز ئوقۇتقۇچىلارنىمۇ كۆرگەن،لىكىن بىر ياكى ئىككى.....
ئاخىردا پەخىرلىنپ شۇنى دەيمەنكى مەن ئوقۇتقۇچى ،دەرس مۇنبىرى مىنىڭ بارلىقىم!
