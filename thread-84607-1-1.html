ئۈمۇرۋايەت دىگەن نىمە؟ مەڭگۈلىك دىگەن يەنە نىمە ؟ھەقىقى مۇھەببەت دىگەن زادى نىمە؟ كىملەر مۇھەببەتنى دەپ ھەممىدىن ۋاز كىچدىغاندۇ ؟ بىر ناخشى بارئىكەن ئىسمى مەڭگۈلىك ،يەنە بىر ناخشا بارئىكەن ئىسمى ھەقىيىقى مۇھەببەتنىڭ كۇچ قۇدۇرتى چەكسىز''،،،،،،،،،،،،،،،،،يەنە نۇرغۇن نۇرغۇن ناخشىلار بار ئىكەن ،بۇرۇن بۇلارنى ئاڭلايتتىم ئاڭلىغان سېرى تېخىمۇ تەسىر لىنەتتىم،،،دۇنيادا پاك مۇھەببەتنى بارلىقىغا تېخىمۇ ئىشنەتتىم،بىراق ھازىر ئىشنىشكە ئامالسىز مەن ،چۇنكى مۇھەببىتىم جاراھەتلەنگەن بىرتال كىپىنەكتەك ئۇچۇپ كىتىپ قالدى،،،،،،،،
ئۇنىڭ بىلەن ناھايتى تاساددىبىى ئۇچرىتىپ قالدىم،لىكىن رۇمانتىك بولغان مەن بۇنى تەقدىر دەپ بىلەتتىم،بىزنىڭ گىپىمىز بىر يەردىن چىقاتتى، ئۇ ناھايتى ئەقىللىق قىزىقچى ئىدى،گەپ سۆزلىرىدىن چىچەنلىكى چىقىپ تۇراتتى بىر خىل بىلىملىك بالىلاردىن ئىدى،بىر قىتىم پاراڭلىشىشتىن كىيىنلا ئىلىكتىرونلۇق پۇچتا يوللانمىسى قىلىشتۇق،ئۇنىڭ يوللانمىلىرىنى كۆر گەندە مەندەك ئىش كۆرمىگەن بىر قىزنىڭ يۇرەكلىرى ئويناپ كىتەتتى،
كىيىن تىلفۇن ئالاقىسى قىلىشتۇق ئۇنىڭ ئاۋازى ناھايتى يىقىملىق،ناھايتى ياش ،ناھايتى ساددا ئاڭلىناتتى،ئۇ مەزگىللەردە ئۇنىڭ تېلفۇنىنى ساقلايتتىم،ئۇھەر كۈنى دەل ۋاقتىدا تېلفۇن قىلاتتى،كۇندە ئۇياندىن بۇياندىن پاراڭلىشاتتۇق ،بۇ مەزگىل مەن ئۇچۇن شاتلىققا چۇمگەن ۋاقىتلار ئىدى،ئەگەر بىرەر كۈن ئۇنىڭ ئاۋازىنى ئاڭلىمىسام تۇلىمۇ مەنسىز ھىس قىلاتتىم،بۇخىل ھىسىياتنى بايان قىلىش تەس،بۇ كۇتۇشتىكى بىرخىل تەقەززالىق سېغىنىش ئارلىشىپ كەتكەن ھىسىيات ئىدى،
ئۆزەمنىڭ ئۇنى ياخشى كۆرۇپ قالغانلىقىمنى ھىس قىلدىم.شېرىن ئۈمۇت خۇددى پۇرەكلەپ ئىچىلغان گۇللەردەك بەرىق ئۇرغان ئىدى،لىكىن،،،،،،،تېلفۇن ئازلاپ كەتتى دائىم ھەرخىل سەۋەپلەرنى كۆرسىتەتتى،ھەرقىتىم ئۇنىڭ سۆزىگە ئىشىنەتتىم،ھەرقىتىم ئۈمۇتسىزلىنەتتىم،كىيىن ئۇنىڭ سەۋەپلىرىگە ئىشىنىشكە زادىلا ئامالىم قالمىدى.بىلەتتىم بۇ پەقەت بىر چۇش................
ئۇنىڭ بىلەن تاسادىبى توردا ئۇچرىشىپ قالغاندا مەندىن ئەپۇ سورايتتى،ئۆزەمنى ئۆزەم ئالداپ ئۇناھايتى ئالدىراش................. دەيتتىم.خۇددى ئەخمەقلەر پورتىدىكى ئەخمەققە ئۇخشاش گومىشلارچە ئۇنى ساقلايتتىم،ساقلىغانسېرى كۆڭلۇم سۇۋىيتتى ،بىراق ئۇ يەنە ماڭا خەت يېزىپ ئۆزىنىڭ مىنى ياخشى كۆردىغانلىقىنى دەيتتى،ئۇ سىزنى ساقلايمەن ھەتتا بىر ئۆمۇر دەيتتى،مىنىڭ بىلەن ئالاقىلىشىشقا ۋاقتى يەتمىگەنلىكىنى ئۆزىنى كەچۇرىشىمنى سورايتتى،ئۇنىڭغا ئىشىنەتتىم،گەرچە ئۇنىڭ ۋەدە قىلىشىنى ئەزەلدىن ئويلىمىغان بولساممۇ،ئەڭ ئاخىرىدا ئۇنىڭ بىلەن بارغانسېرى يىراقلىشىپ كەتتۇق،بىر نەچچە پەسىل بولدىغۇ دەيمەن ئۇنىڭ ئىكراندىن ئەۋەتكەن خېتىنى كۆرسەممۇ ماڭا قىلچىلىك تەسىر قىلماس بولدى,ھەر بىر جۇملىسىنى كىچە ئاسمىندىكى گۇزەل پوجاڭزىغا ئوخشىتاتىتم،ئۆزەمنى ئۆزەم ئالداۋاتاتىتم،لىكىن قەلبىمدىكى بوشلۇقنى ھەرقانچە قىلغان بىلەنمۇ تولدۇرۋالالمايتتىم،بۇرۇن ئۇماڭا تىلفۇن قىلمىسا ئۇنىڭغا ھەرگىزمۇ قارىمايمەن دەيتتىم،لىكىن ئۇ بىر ئېغىز گەپ قىلسىلا خۇددى دورا ياۋالغان ئادەمدەك ئىختىيارسىز ئۇنىڭ سۆزىنى ئاڭلايتتىم،بۇيىل يازدا نىمە بولدىكىن يامغۇر تۇختىماي ياغدى،مەن بولسام تىلفۇنىمنى تۇتۇپ گومۇشلارچە ئۇنىڭدىن تېلفۇن كىلىشىنى ساقلايتتىم
بىر تامچە بىر تامچىدىن تامچىلاۋاتقان يامغۇر خۇددى يۈرۇكىمگەتامچىلاۋاتقاندەك،يۈرىكىمنى ھۆل قىلۋەتكەندەك ھىس قىلاتتىم،يۈركۇم يىغلاۋاتقاندەك،گەرچە كۆزىمدىن ياش چىقمىغان بولسىمۇ يۈرۇكىمدىكى توغان ئىلىپ كەتكەندەك بىلىنەتتى،ئاخىرى ھۇشۇمنى تاپتىم،بۇزۇلماس ئەھدە دىگەن نىمە ئۇ .ھەممىسى ئۆزەمنىڭ ئەخمەقلىقىمدىن بۇلۇپتۇ ,ئۆزەمنى ئۆزەم ئىشىنىشكە مەجبور قىلىپتىمەن،شۇنىڭ بىلەن ئۇنىڭغا قايتا تېلفۇن قىلمىدىم،ئۇنىڭ تېلفۇنىنىمۇ ساقلاپ يۇرمىدىم،ئۇنىڭ بىلەن قايتا ئۇچرىشىپ قالماسلىق  ئۇچۇن توردىكى ئىسمىمنى ئۆزگەرتۋەتتىم،،،،
