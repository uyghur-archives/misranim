پەيغەمبەر ئەلەيھىسالام ساھابىلەرگە بەنى ئىسرائىلدا بىر كىشىنىڭ ئاللاھ يۇلىدا 83يىل كۇرەش قىلغانلىق ھىكايىسىنى سۆزلەپ بىرىپتۇ ساھابىلەردىمۇ 83 يىل كۇرەش قىلىش ياكى ئاللاھقا 83يىل سالىھ ئەمەل قىلۋېلىش ئشتىياقى پەيدا بوپتۇ لىكىن ئۇلار ئۆزلىرىنىڭ ئۆمرىنىڭ ئۇنچىلىك ئۇزۇن ئەمەسلىكىدىن چۇشكۇنلۇشۇپتۇ شۇ ئارادا ئاللاھ تەئالا قەدرى سۇرىسىنى نازىل قىپتۇ [بىز ھەقىقەتەن قۇرئاننى شەبى قەدىردە نازىل قىلدۇق،شەبى قەدىرنىڭ نىمە ئىكەنلىكىنى بىلەمسەن، شەبى قەدىر شەرەپ ۋە پەزىلەتتە مىڭ ئايدىن ئارتۇقتۇر،پەرىشتىلەر ۋە جىبرىلئاللاھنىڭ ئەمرى بىلەن زىمىنغا ئاللاھ تەقدىر قىلغان بارلىق ئىش ئۇچۇن چۇشىدۇ،شۇ كىچە تاڭ يۇرۇغانغا قەدەر پۇتۇنلەي تىنچ-ئامانلىقتىن ئىبارەتتۇر]قەدر سۇرىسى بۇسۇرە نازىل قىلىنغاندىن كىيىن شۇ ۋاقىتتىكى مۇسۇلمانلار ئىنتايىن خۇش بولغان چۇنكى پەيغەمبەر ئەلەيھى سالام شۇ كىچىنى شەرھىيلەپ قەدىر كىچىسى رامزان ئېيىنىڭ ئاخىرقى ئونىدا ئاخىرقى ئونىنىڭمۇ تاق كىچىلىرىدە دەپ بىشارەت بەرگەن.
قېرىنداشلار ئەمدى تاق كىچىنى قانداق پەرق قىلىمىز رامزان ئېيىدا بۇلۇپمۇ تۇنجى ئاخشام رامزانغا تەييارلەق قىلۋاتقان كىچىسى باشلىنىدۇ شۇ سەۋەپتىن كىچە ئىلگىرى بۇلىدۇ مەسلەن رامزان تۇتقان كۇنىمىز 15 بولغان بولسا كىچە 16 بۇلىدۇ مۇجتەھىد ئىماملىرىمىز قەدرى كىچىسىنى 27- كىچىسى دەپ كۆرسەتكەنلەر كۆپپراق يەنەبەزى ئىماملىرىمىز تاق كىچىدە بۇلىدۇ لىكىن شەبى قەدىرنىڭ قايسى كىچىدە ئۇتۇدىغانلىقى نامەلۇم دىگەنلەرمۇ بار.
قېرىنداشلار مۇشۇنداق كاتتا كۇنلەر يىتىپ كەلدى بىزنىڭ 83يىل ئۆمۇر كۇرشمىز بەك تەس ئۇنداق ئۆمۇر كۆرسەكمۇ ھەممىسىدە ئىبادەت قىلالمايمىز ئىماملىرىمىز ئەگەر شەبى قەدەر ئۆتكەن كىچىدە ئۇخلىماي ئىبادەت قىلغان كىشى شەبى قەدىرنى ئەينەن كۆرمىسىمۇ شەبى قەدىرنى كۆرگەننىڭ ئەجرىنى تاپىدۇ دىگەنگە چوڭ ئۆلىمالىرىمىز بىرلىككە كەلگەن.
شەبى قەدرنىڭ كۇرۇنىشى پۇتۇنلەي نۇردۇر قاراڭغۇ كىچىدە كۇرۇنسىمۇ شۇ كىچىنى يۇرۇتىۋاتىدۇ دىيىلگەن ئەشۇ نۇرمۇ پەقەت ئاللاھتىن ساۋاپ ئۇمۇت قىلىپ كىچىلەردە ئىبادەت قىلغان كىشىگىلا كۇرۇنىدىكەن ئۇممۇلموئىمىنۇن ئائىشە رەزىيەللاھۇ ئەنھا يا رەسۇلىلىلاھ شەبى قەدىر كۆرسەم نىمە دەيمەن دەپ سورىغاندا رەسۇلىللاھ [ئاللاھۇممە ئەفۇۋۇن تۇھىببۇن ئەفۇۋە فەئفۇ ئەننا ]دەڭ دىگەن قېرىنداشلار كاتتا كۇنلەر يىتىپ كەلدى شۇڭا بارلىق تىرىشچانلىقىمزنى كۆرسۇتۇپ شۇ كاتتا شەرەپكە نائىل بۇلىۋالساق دىگەن ئۇمۇتتە بۇتىمىنى يوللۇدۇم سەۋىيەم چەكلىك خاتا كەتكەن يەرلىرى بولسا تۇزۇتۇپ پايدىلانغاچ  كەمىنىگە قىممەتلىك پىكىر-تەكلىپىڭلارنى ئايىماسلىقىڭلارنى ئۇمۇت قىلىمەن

ئاللاھ تائالا بۇ ئۇلۇغ كېچىنى رامىزان ئېيىدا قىلغان بولۇپ، بۇ ئۇلۇغ كېچىدە قىلغان ئىبادەت قۇرئاننىڭ ئېنىق كۆرسەتمىسى بويىچە مىڭ ئاي ئىبادەت قىلغاندىنمۇ ئەۋزەلرەكتۇر، ئاللاھ تائالا بۇنى قەدر سۈرىسىدە ئېنىق بايان قىلغاندۇر. پەيغەمبەر ساللەللاھۇ ئەلەيھى ۋەسەللەم قەدىر كېچىسىنى رامزاننىڭ ئاخىرقى ئون كۈنى ئىچىدە بولىدىغانلىقىنى بايان قىلغان بولۇپ قايسى كېچىدە بولىشى ئېنىق ئەمەس، شۇڭا ئومۇمەن پۈتۈن رامىزان ئېيىدا خۇسۇسەن ئاخىرقى ئون كۈندە پۇرسەتنى غەنىيمەت بىلىش زۆرۈر.
قۇرئان ۋە ھەدىسلەردە ئىسپاتلاندىكى : بىر ئادەم ئۆلسە ، ئۇ بۇ دۇنيادا قىلغان ياخشى بولسۇن يامان بولسۇن ، بارلىق چوڭ – كىچىك ئىشلىرى ئۈچۈن ھىساب – كىتاب قىلنىدۇ   . ياخشى ئىش مۇكاپاتلىنىدۇ ، ئەسكى ئىش جازالىنىدۇ ، ھىساۋاتنىڭ بىرىنچى سەھنىسى قەبرىدە بولۇدۇ  .

قەدىر كېچىسى ئەرەپ تىلىدا "ليلە القدر" دېگەن ئىزاقەتلىك  بىرىكمە بىلەن ئىپادىلىنىدۇ. بۇ بىرىكمىدىكى "ليلە'' نىڭ مەنىسى :كېچە،تۇن" دىگەندىن ئىبارەت.   "القدر "نىڭ  مەنىسى بولسا قەدىر -قىممەت، شەرەپ ،يۇكسەك، مۇقەددەس. دىگەنلەردىن ئىبارەت. بۇ ئىككى سۇز بىرىكىپ ئىزاقەتلىك  بىىكمە ھاسىل قىلغان.
قەدىر كېچىسى رامزان ئېيىنىڭ ئاخىرقى ئون كۇنىنىڭ بىر كىچىسى ،يەنى رامزان ئېيىنىڭ 21-كۈنىدىن 30-كىچىسىگىچە بولغا 10 كۈنىنىڭ بىر  كىچىسى دېگەن قاراش.  ئائىشە رەزىيەللاھۇ ئەنھادىن رىۋايەت قىلىدۇركى ،<ئەگەر رامىزان ئېيىنىڭ ئاخەرقى ئون كۈنى كىرسە ،پەيغەمبەر ئەلەيھسسالام ئۇخلىمايتى ، بالا-چاقىلىرنى  ئويغىتاتتى ، ئاياللىرى بىلەن يېقىنچىلىق قىلمايتى >.بۇخارى ،مۇسلىم رىۋايەت قىلغان.
يەنە بىر رىۋايەتتە  ئەبى سەئىد  خۇردەردىن  رىۋايەت قىلىنىدۇركى ،پەيغەمبەر ئەلەيھىسسالام رامزان ئېيىنىڭ 20-كۈنى سەھەردە ساھابىلىرىنىڭ ئارسىغا كەلدى ۋە ئۇلارغا  سۆز قىلدى ، ئۇ  مەن چۇشۇمدە  قەدىر كېچىسىنڭ ۋاقتىنى كۆرگەن،كېيىن ئۇنتۇپ قالدىم.سىلەر رامىزان ئېيىنىڭ ئاخىرقى ئون كۇنىدىن ئېز دەڭلار ، بۇلۇپمۇ تاق كۈنلىردىن . دېدى(بۇخارى ۋەمۇسلىم رىۋايەت قىلغان).
دېمەك رامىزان ئېيىنىڭ 21-23-25-27-29-كۇنلىرىنىڭ كىچىسى قەدىر كىچىسىنى تىپىشقا ئەۋزەل ھەم ساۋاپتۇر.
يەنە بىر رىۋايەتتە ئىبنى ئۆمەر رەزىيەللاھۇ  ئەنھۇدىن  رىۋايەت قىلىنىدۇركى ،پەيغەمبەر ئەلەيھىسسالام  "كىمكى قەدىر كېچىسىنى ئىزدىگۈچى  بولىدىكەن ، ئۇنى رامىزان ئېيىنىڭ 27-كۇنى كېچىسىدىن ئىزدىسۇن ''(مۇسلىم رىۋايەت قىلغان)
ئولىمالار نۇرغۇن ئىزدىنىشلەرنى ئېلىپ بارغان بولسىمۇ ،قەدىر كېچىسىنىڭ  زادى قايسى كىچىدە بۇلىدىغانلىقى توغرىسىدا ئېنىق يەكۇنگە ئىرشەلمىگەن.شۇنداق بولسىمۇ نۇرغۇن ئولىمالار يەنىلا قەدر كىچىسى رامىزان ئېيىنىڭ 27- كۈنى كىچىسى بۇلىدۇ دېگەن قاراشنى توغرا تاپقان.  ھەمدە بۇ قاراشنى ئىبنى ئابباس رەزىيەللاھۇ ئەنھۇدىن " <قۇرئان كەرىم > دىكى <قەدر> سۈرسى ئوتتۇز  ئەرەبچە سۆزدىن تەركىپ تاپقان. ئۇنىڭدىكى 27-سۆز بولسا ھى (ئۇ) دۇر .دېگەن سۆز ئارقىلق ئىسپاتلانغان .يەنى ئاللاھ تائالا <قەدر>سۈرسىدە  قەدر كېچىسىنى بۇ سۈرىنىڭ 27- سۆزىگە ئۇدىللاپ ئۇنى ئالماش"ھى"(ئۇ) بىلەن ئىپادىلگەن دەپ قارىغان.
   

ئاللاھ تائالا قەدىر كېچىسىنىڭ ئورنىنى ناھايىتى  يۇقىرى ئورونغا قويغان. بۇ كىچە بۇلسا بەرىكەتلىك ،پەزىلەتلىك  ئۇلۇغ  كېچىدۇر. شۇنىڭ ئۇچۇن ئاللاھ تائالا <قۇرئان كەرىم >دە بۇ كېچە توغرىسىدا مۇنداق دېگەن:
شەبى قەدرە(شەرەپ ۋە پەزىلەتتە ) مىڭ ئايدىن ئارتۇقتۇر.
  بۇ كېچە بۇلسا  پەرىشتىلەر ئاللاھنىڭ رەھمەت ،تىنىچلىق ،سالام ۋە بەرىكەتلىرنى ئىلىپ چۇشىدىغان كېچە. بۇ كېچىدە تىنىچلىق ، ئامانلىق تاڭ ئاتقۇچە داۋام قىلىدۇ.
   
ئەبۇ ھۈرەيرە  رەزىيەللاھۇ ئەنھۇدىن رىۋايەت قىلىنىدۇركى .پەيغەمبەر ئەلەيھىسسالام  مۇنداق دېگەن < سىلەرگە مۇبارەك رامىزان ئېيى كەلدى .ئاللاھ تائالا سىلەرگە ئۇنىڭ  روزسنى تۇتۇشنى پەرز قىلىندى . بۇ ئايدا جەننەتنىڭ ئېشكلىرى ئىچىلىدۇ . دوزاقنىڭ ئىشكلىرى تاقىلىدۇ.جىن -شەيتانلارنىڭ پۇت-قۇللىرغاكويزا- كىشىەنلەر سىلىنىدۇ. بۇ ئايدا بىر كىچە بار بۇلۇپ، بۇ كېچە مىڭ ئايدىن ئارتۇقتۇر .كىمكى شۇ كېچىنىڭ ياخشىلىقىدىن مەھرۇم قىلىنسا بارلىق ياخشىلىقلاردىن مەھرۇم قىلىنغان بۇلىدۇ. >(ئىمام ئەھمەد رىۋايەت قىلغان.)

ھازىرغىچە 2 ئادەم باھالىدىمۇنبەر پۇلى يىغىش سەۋەبى
ئىرشاد + 50 ھەقىقەتەن كۈچ بەردى
uzumtal + 100 ماختاشقا تېگىشلىك

ھەممە باھا نومۇرى : مۇنبەر پۇلى + 150   باھا خاتىرىسى

قۇرئان ۋە ھەدىسلەردە ئىسپاتلاندىكى : بىر ئادەم ئۆلسە ، ئۇ بۇ دۇنيادا قىلغان ياخشى بولسۇن يامان بولسۇن ، بارلىق چوڭ – كىچىك ئىشلىرى ئۈچۈن ھىساب – كىتاب قىلنىدۇ   . ياخشى ئىش مۇكاپاتلىنىدۇ ، ئەسكى ئىش جازالىنىدۇ ، ھىساۋاتنىڭ بىرىنچى سەھنىسى قەبرىدە بولۇدۇ  .
