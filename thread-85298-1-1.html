ئەسسالامۇ ئەلەيكۇم دوستلار ياخشىمۇ سىلەر ، كەمىنە سۆزۈمنى باشلىدىم ئەمسە ، مەن ئۆزۈم جىننىڭ ھېكايىسىنى ئوقۇشقا ھەم  ئاڭلاشقا بەك ئامراق ، شۇڭا ئۇيغۇرچە توربەرتلەر بولسۇن ياكى خەنزۇچە توربەتلەر بولسۇن ئىزدەپ يۈرۈپ ئوقۇيمەن ، شۇڭا ماڭا ئوخشاش قىزىقىپ ئوقۇيدىغانلار  ئۈچۈن تۆۋەندىكى ھېكايىلەرنى خەنزۇچە مۇنبەرلەردىن تەرجىمە قىلىپ يوللاپ قويدۇم ، ئەمما بۇ ۋەقەلىكلەر  توقۇلما بولماستىن ھەقىقىي بولغان ئېشلار ئىكەن . سىلەرمۇ ئۆزۈڭلارنىڭ بېشىدىن ئۆتكەن ئېشلارنى دىسەڭلار  قىزىققۇچىلەر قورقۇپ تۇرۇپ تەڭ ئوقىساق ..ئەمما ئەسكەرتىپ قوياي يۈرىكى ئاجىزلار ھامىلدار ئاياللار ھەرگىز ئوقىماڭ ، ئوقۇپ قۇيۇپ چاتاق چىقسا ئۆزىڭىز مەسئۇل ئاقىۋىتىگە .

1. كىچىك ۋاقىتلىرىم ئىدى ، ئىسىمدە قىلىشىچە 4 ، 5 ياش ۋاقىتلىرىم ئىدىغۇ دەيمەن ، بىز ئىنتايىن كونا بىر ئۆيگە كۆچۈپ كىردۇق ، يېڭى كۆچۈپ كىرگەن دەسلەپكى مەزگىللەردە مەن ھەر كۈنى يېرىم كېچىلەردە ئويغۇنۇپ كېتەتتىم ( ئۆزۈم يالغۇز ئايرىم ھۇجرىدا ياتىدىكەنمەنتۇقۇم ) ، بىر كۈنى ئويغۇنۇپ قارىغىنىمدا تامنىڭ بۇلۇڭىدا بىر ئۇزۇن چاچلىق بىر ئايال ماڭا قاراپ تۇراتتى ، شۇ ھامان  قورققىنىمدىن چىررىدە قىلىپ يىغلاپ كەتتىم ، يىغلىغان ئاۋازىمنى ئاڭلاپ ئاتا ئانام يۈرگۈپ يىنىمغا كىرى ، مەن ئاتا- ئانامغا يىغلاپ تۇرۇپ ئاۋۇ يەردە بىر ئادەم باركەن ،  دەپ تامنىڭ بۇلىڭىنى كۆرسەتتىم ، ئەمما ئاتا-ئانام ئۇ يەرگە قاراپ ھىچنىمىنى كۆرمىدى ئاندىن مىنى كۆتۈرۈپ ئۆزلىرىنىڭ ھۇجرىسىغا ئەپچىقىپ كەتتى ،  ئىككىنجى كۈنى بۇ ئىش  يەنە داۋاملاشتى مەن يەنە يىغلاپ تامنىڭ بۇلۇڭىنى كۆرسەتتىم ئەمما ئۆيدىكىلەر ئۇ ئايالنى كۆرەلمىدى ، لىكىن ئۇلارنىمۇ ۋەھىمە باستى شۇنىڭ بىلەن ئەتىسى ئانام بىر قاچىغا گۈرۈچنى توشقۇچە تۆكۈپ ئاندىن ئۈستىنى  يەنە بىر قاچا  بىلەن يېپىپ مىنى كارۋاتقا ياتقۇزۇپ قۇيۇپ ھېلىقى گۈرۈچ تولدۇرۇلغان قاچىنى كۆتۈرۈپ  xx قايتىپ كەل ،  xx قايتىپ كەل دەپ توۋلاپ ئۆينىڭ ئىچىنى بىر ئايلاندى ،،شۇنىڭ بىلەن بۇندان ئىش قايتا تەكرارلانمىدى .... بۇ ئىشلار كاللامغا بەك تەسىر قىلغاچقا ھازىر ئويلىساممۇ  بەك قورقىمەن.......
2. مىنىڭ ئۆيۈم يېزىدا ئىدى ، مەھەللەمنىڭ سىرتى پۈتۈنلەي جاڭگاللىق ئىدى، بىر كۈنى كەچتە قىزىمنى ۋېلىسپىتنىڭ كەينىڭ ئولتۇغۇزۇپ ئۆيگە قايتىۋاتاتتىم ، ،ئەتىراپ قاراڭغۇ يولدا ئادەم يوق ، يېرىم يولغا كەلگەندە قىزىمغا قارىسام قىزىم كەينىگە  قاراپ تۇرۇپتۇ ، بىرىنجى قېتىم ئانچە ئىرەن قىلىپ كەتمەپتىمەن بىر ئازدىن كىيىن قارىسام يەنە كەينىگە قاراپ تۇرۇپتۇ ، شۇنىڭ بىلەن قىزىمدىن نىمىشقا كەينىڭگە قاراۋىرسەن دىسەم ، قىزىم نىمە دەيدۇ دىمەمسىلە " ئاپا ، كەينىمىزدە بىر ئايال بىزگە ئەگىشىپ يۈگرەۋاتىدۇ '' دەيدۇ ، ئەمما مەن قارىسام ھىچكىم يوق ، قورقىنىمدىن ۋەلىسپىتنىڭ پىدالىنى تېخىمۇ تېز تەپتىم ،، پۈتۈن بەدەنلىرىم چىلىق چىلىق تەر ئىدى ، مەھەللىگە كىلەي دىگەندە قىزىمدىن ھېلىقى ئايالنىڭ يەنە بار يوقلىقىنى سورىسام قىزىم ،ۇ ئاچاش بەك تېز يۈگرەيدىكەن ئابايتىن ياقى بىزگە ئەگىشىپلا كەلدى دەيدۇ ، ئەمما مەن قارىسام ھىچكىم يوق  ، ئۇ ئايالنى بىررەر يەرگە مۈكىۋالدىمىكەن دىسەم ئەتىراپ پۈتۈنلەي تۈپتۈز جاڭگاللىق ئىدى مۆكۈنىۋالغۇدەك شارائىت يوق ئىدى ،،شۇنداق قورقۇپ كەتتىم ھاياتىمدا بۇنداق ئېشقا ئەزەلدىن يۇلۇقۇپ باقماپتىكەنمەن ،،شۇنداق بولسىمۇ نىمە ئامال قورقۇپ يۈرۈپ مەھەللە ئىچىگە ساق سالامەت كىرىۋالدىم ،،شۇنىڭدىن ياقى كېچىسى تولا سىرتقا چىقماس بولدۇم .

3. بىزنى مەكتەپتىن تەتىل قويۇپ بەرگەندە ياتاق بىناسىدا ياتاققا مەسئۇل ئادەم ۋە ياتاقدىشىم ئۈچىمىدىن بىلەن ھىچكىم يوق ئىدى ، ئوقۇغۇچىلارنىڭ ھەممىسى نەرسە كېرەكلىرىنى يىغىشتۇرۇپ ئۆيگە قايتقان ئىدى ، شۇنداق قىلىپ پۈتۈن ياتاق بىناسىدا ئۈچىمىزلا قالدۇق ، ياتاق بىنارىمىز ئىككى  قەۋەتلىق ، بىز  ئىككىنجى قەۋەتتە ياتاتتۇق، بىرىنجى كۈنى كەچتە ھىچ ئىش بولمىدى ئەمما ئىككىنجى كۈنى كەچتە ياتاقدىشىم ئىككىمىز چىراق ئۆچۈك ياتساق تۇيۇقسىز ياتاقنىڭ تورىسىدا بىر كىشىنىڭ گەپ قىلىۋاتقان ئاۋازى ئاڭلاندى ، قۇلىقىمغا ئىشەنمەي قايتا ئاڭلىسىام راست بىرەيلەن چۈشىنىكسىز بىر تىلدا گەپ قىلىۋىتىپتۇ ، قوققىنىمدىن يوتقاننىڭ ئىچىگە چىڭ تۈگۈلۈپ يېتىۋالدىم ، ياتاقدىشىمنى چاقىراي دىسەممۇ ئاۋاز چىقىرىشتىم قوقتۇم ، بىر دەمدىن كىيىن ئۇ ئاۋاز يوقىدى ، ئەمما يۈرۈكۈم توختىماستىن قاتتىق سوقىۋاتىدۇ ، ھۇشۇمدىن كەتكىلى ئاز قىلىۋاتاتتىم ، تاڭنىڭ تېز ئېتىشىنى ساقلىدىم ، ئەمما مۇشۇنداق ۋاقىتتا ئىنتايىن ئاستا ئۆتىدىكەن ، بىر ۋاقىتلاردا ئۇخلاپ قاپتىمەن ، ياتىقىمىز ئالتە كىشىلىك بۇلۇپ ياتاقدىشىم ئىككىمىز ئۈستىدىكى كارۋاتتا ياتقان ، ئەتىسى تاڭ ئاتقاندا كارۋىتىمدىن پەسكە چۈشمەيتۇرۇپلا بولغان ئەھۋالنى دوستۇمغا دىسەم ئۇمۇ بىرەيلەننىڭ گەپ قىلىۋاتقانلىقىنى ئاڭلاپتۇ ، شۇنىڭ بىلەن كۈندۈزى بولسىمۇ ھەر ئىككىلىمىزنى يەنە بىر قېتىم قورقۇنۇچ باستى دە كىيىملىرىمىزنى كىيىپ ئاستىنقى كارۋاتقا چۈشسەك پەستىكى ئىككى كارۋاتنىڭ ئۈستىدە كىىچىك بالىلارنىڭ بىرمۇنچە ئاياغ ئىزى پەيدا بۇلۇپ قاپتۇ ، ياتاقدىشىم ئىككىمىز بىر بىرىمىزگە قاراپ ھەيران قالدۇق ، تېخى ئاخشاملا يوق ئاياغ ئىزى ئەمدىلىكتە نەدىن پەيدا بۇلۇپ قالدى ، كىچىك بالىلارنىڭ يېرىم كېچە ياتاققا كىرىشى مۇمكىن ئەمەس ، يەنە بىرسىگە ياتاقنىڭ ئىشىگى ئىچىدىن قۇلۇپلاقلىق ،، ئەمدى بۇ يەردى ئىككىمزنىڭ ئۇخلىشىغا كۆزىمىز يەتمەي شۇ كۈنىلا نەرسە كېرەكلىرىمىزنى يىغىشتۇرۇپ بىزمۇ ئۆيگە قايتىپ كەتتۇق ،، كىيىن مەكتەپكە يىغىلغاندا بۇنداق ئاۋاز قايتا ئاڭلانمىدى ،

بۈگۈن مۇشۇنچىلىك يوللاپ تۇراي ، كىيىن داۋاملىق يوللاپ تۇرىمەن ، ئەمدى سىلەرگە كەلدى ئەمسە ، ئىنكاس شەكلىدە يوللىۋىتىڭلار ھې ھې ھې ..

ئانچە قورقۇنۇشلۇق ئەمەس ئىكەن

   مەلۇم دوختۇرخانىنىڭ يۇقىرى دەرىجىلىك تاشقى كېسەللىكلەر دوختۇرى بىر قېتىملىق جىددى قۇتقۇزۇشتىن چىقىپ ئۆيگە قايتىشقا تەمشىلىۋاتاتتى. بۇ ۋاقىت ئاللىقاچان يېرىم كېچە بولغان بولۇپ، دوختۇر لېفىتقا چىقىش ئۈچۈن لېفىت ئالدىغا كەلدى ۋە بىر سىستېرا قىزنىڭمۇ لېفىت ساقلاۋاتقانلىقىنى كۆرۈپ، ئىككىيلەن بىللە لېفىتقا چىقتى. لېكىن لېفىت بىرىنچى قەۋەتتىمۇ توختىماي ئۇدۇللا يەرئاستى ئۈچىنچى قەۋەتكە كېلىپ توختىدى. ئىشىك ئالدىدا بىر كىچىك قىز باشلىرى تۆۋەنگە ساڭگىلىغان، چاچلىرى چۇۋۇلغان ھالدا: مەن لېفىتقا چىقىۋالاي- دېگىنىچە تۇراتتى. دوختۇر ئىشنىڭ چاتاقلىغىنى بايقاپ لېفىتنىڭ ئىشىكىنى شۇ ھامان تاقىۋالدى. سېسترا قىز ئەجەبلەنگەن ھالدا سورىدى:
- نىمدەپ ئۇ قىزنى چىقىرىۋالمايسىز؟
دوختۇر: – يەر ئاستى ئۈچىنچى قەۋەت  دوختۇرخانىنىڭ جەسەت ساقلاش ئۆيى، ئادەتتە دوختۇرخانا ھەر بىر جەسەتنىڭ ئوڭ قول بېغىشىغا بىر دانە قىزىل لېنتا تاقاپ قويىدۇ، ئاۋۇ قىزنىڭ قولىدا…ئاۋۇ قىزنىڭ قولىغىمۇ شۇنداق لېنتا باغلاقلىقكەن….. سېسترا قىز دوختۇرنىڭ گېپىنى ئاڭلىغاندىن كېيىن، ئالدىرىماي ئوڭ قولىنى چىقاردى ۋە كۈلۈپ تۇرۇپ دوختۇردىن سورىدى:
- مۇشۇنداق لېنتىمۇ؟
