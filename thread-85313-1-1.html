                                         قۇياشنىڭ قولدىشى قەشقەر
                                (ق ھەرپى بىلەن باشلانغان سۆز ئويىنى)
   
                                             قۇياشتەك قەددى قامەتلىك،
                                       قۇياشنىڭ قوشمىقى قەشقەر.
                                       قارا  قۇندۇز  قارا  قاشلىق،
                                       قۇياشنىڭ قولدىشى قەشقەر.
       قەدىرلىك قېرىنداشلار،قىممەتلىك قەلىپداشلار،قەيسەر قولداشلار؛  قەشقەرنىڭ قىممەتلىك قەدەملىرىگە، قامەتلىك قىياپەتلىرىگە قاش-قاپاقلىرىڭلارنى قىمىرلاتماي قاراڭلار قېنى!
     قەشقەر قەلئەسىدىكى قەدىردانلار قالدۇرغان قىممەتلىك قامۇسلاردا قەشقەرنىڭ قوياشقا قولداش قۇدرەتلىك قىياپىتى قەيت قىلىنغان.قەشقەرنىڭ قەدىر-قىممىتى قەشقەردىكى قەيسەر قەدىردانلارنىڭ قەدەم قالدۇقلىرى،قەسىر قايناملىرىدىكى قۇتلۇق قولۋاقلىرىنىڭ قاتمۇ-قات قۇرۇلمىلىرى-قاش قىرىنى قەشقەر قىلىپ قاققان قائىدىلىك، قىممەتلىك قوزغالماس قوزۇقلىرى قېرىداشلارنى قايىل قىلالايدۇ.
       قەشقەرنىڭ قىممەتلىك قائىدىللىرىنى قۇتلۇقلايدىغان قارى-قۇرئان قارىملار،قۇدرەتلىك قانۇنلارنى قوللاپ قۇۋەتلەپ قوغدايدىغان قانۇنشۇناس قەيسەر، قابىل قازىلار قول قويغان قامۇسلاردىكى قاش قىرىغا قېقىلغان قوزغالماس قوزۇقلار- قېرىنداشلارنى قايرىپ قويۇپ، قالماقلارنى،قارۇغۇلارنى قىچىرساقمۇ قايىل قىلالايدۇ.
     قەشقەر-قۇياش قەسىرىدىكى قامەتلىك،قۇدرەتلىك،قىممەتلىك،قەيسەر قەلئە.قەشقەر قەلئەسى قوياش قەسىرىگە قاراشلىق قەلئە قۇرۇلمىسىنىڭ قوزغالماس قوۋۇرغىسى.
      قەشقەرنى قەشقەر قىلغان قىممەتلىك قاشقا قەشقەرنى-قاشقىرى قىلىپ قېقىلغان قوزغالماس قوزۇق. قەشقەر -قاش قىرىدىكى قېرىماس قەلئە.
      قاش قىرىغا قېقىلغان قەشقەر قاشقىسىنىڭ قوزىقىنى قوزغىماسلىققا قەشقەرنىڭ قۇدرەتلىك قازىللىرى، قىممەتلىك قارىملىرى،قەدىرلىك قۇلۇم-قوشنىللىرى قەسەم قىلىپ قول قويغان.
     قەشقەردىكى قەيسەر،قەدىردان،قورقماس قېرىنداشلار قەشقەرنىڭ قەددى-قامىتىنى،قەدىر- قىممىتىنى،قاملاشقان قەدەملىرىنى قۇۋەتلەندۈرۈپ قۇياشقا قاراپ قوزغىلىۋاتقاندا،قاچانلاردا قاتاردىن قالغان قاراڭغۇ قازناقلاردىن قوزغالمىغان؛قارا قالپاقنى قاڭشارلىرىغا قادىۋىلىپ قوياشقا قارىيالماي قەلبى قارىداپ قالغان؛ قۇرۇق قۇدۇقنىڭ قاغجىراپ قالغان قايناملىرىغا قاپسىلىپ قېلىپ قايمۇقۇپ قالغان قارا قوساقلار ؛-قېرىنداشلىرىنى قاقشىتىپ قاغىشىغا قالغان  قاپاقباشلارنى قارىسىغا قالايمىقان قىمىرلىتىپ قېرىنداشلىرىنىڭ قوياشقا قاراپ قوزغىلىۋاتقان قائىدىلىك قەدەملىرىگە قارىغۇلارچە قائىدىسىز قوشۇلىۋىلىپ، قارا قۇيۇن قۇمنى قوچىغاندەك قېرىنداشلىرىنىڭ قائىدىلىك،قىممەتلىك قىلمىشلىرىنى قوچىۋەتتى.قەشقەر قەۋمىنىڭ قەدىرلىك، قىممەتلىك قائىدىللىرىگە قاملاشمايدىغان قىلىقلارنى قوشۇپ قويۇپ قەشقەرنىڭ قۇرۇلۇش قىلغۇچىلارنى قايمۇقتۇرىۋەتتى.
    قېرىنداشلار،قارا قوساقلارنىڭ قاملاشمىغان قىلىقلىرىدىن قىلدەك، قلتىرىقتەك قىسمىنى قىستۇرۇپ قويدۇم.
     قاراڭلار قارا قوساقلارنىڭ قاملاشمىغان قىلىقلىرىغا!
     قۇمباستىنىڭ قۇملىرىنى قاسقاندا قاينىتىپ ،قەغەزدەك قىش قاتۇرۇپ،قۇدۇقتا قۇرۇتۇپ،قەۋەت-قەۋەت قۇرۇلۇش قىلىشتا قوزۇق قىلىپ قېقىپ،قوتازنى قوشۇقتا قوماچ قىلىپ قورۇپ،قوزۇقتىكى قۇرىتتەك قىمىرلاپ،قۇملۇقتىكى قومۇشتەك قاغجىراپ، قاغىدەك قاقىلداپ قوش قاتمىغان قوتازدەك قوغلۇشۇپ قېرىنداشلىرىنى قان قۇستۇرماقتا.
     قاملاشمىغان قىلىقلارنى قىلىۋاتقان قاقۋاش قاۋانلار قېرىداشلىرىڭلارنىڭ قاتارىغا قايتىپ قائىدىلىك قىلىقلارنى قىلىڭلار، قايتمىساڭلار قانۇن -قائىدىلەرنىڭ قارارىدىن قۇتۇلالمايسىلەن!
      ئاپتۇرى مەخەت ئابدۇرېھىم

  قەشقەرنى قەشقەر قىلغان قىممەتلىك قاشقا قەشقەرنى-قاشقىرى قىلىپ قېقىلغان قوزغالماس قوزۇق. قەشقەر -قاش قىرىدىكى قېرىماس قەلئە.
