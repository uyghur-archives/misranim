بورھان شەھىدىنىڭ مەكىت ناھىيىسىنىڭ ھاكىمىنى سايلاش يىغىنىدا سۆزلىگەن سۆزى


«11 بېتىم» ئىمزالانغان دەسلەپكى مەزگىللەردە ، ئۇنىڭ ئاساسلىق مەزمۇنى بويىچە ، پۈتكۈل شىنجاڭ دائىرىسىدە دېمكراتىك سايلام ئېلىپ بېرىش پائالىيىتى باشلىنىپ ، ئۆلكىلىك ھۆكۈمەت تەركىۋىدىكى ئاساسلىق رەھبىرى يولداشلار مەسئۇللىقىدا كىچىك گورۇپپىلار تەشكىللىنىپ ، ھەرقايسى ۋىلايەتلەرگە ئىبەرتىلدى . شۇ قاتاردا ، 1946 – يىل 12 – ئاينىڭ 1 – كۈنى بورھان شەھىدى باشچىلىقىدىكى گورۇپپا مەكىت ناھىيەسىگە يىتىپ كىلىپ ، مەكىت ناھىيىسىنىڭ ھاكىمىنى سايلاشقا رىياسەتچىلىك قىلغان . سايلام باشلىنىشنىڭ ئالدىدا ، ناھىيە مەركىزىدە چوڭ كۆلەملىك يىغىن ئېچىلىپ ، يىغىندا بورھان شەھىدى مۇنداق سۆز قىلغان .

      «قەدىرلىك قېرىنداشلار ، يۇرتداشلار !

     ئۆلكىلىك ھۆكۈمەتنىڭ رەئىسى جاڭ جىجۇڭ ، مۇئاۋىن رەئىسى ئەخمەت ئەپەندى ۋە ھۆكۈمەتنىڭ بارلىق ئەزالىرى سىلەرگە سالام يوللىدى .

بۈگۈن ھەممىڭلار بىلەن يۈز كۈرۈشۈپ ، مۇشۇنداق سۆزلەش پۇرسىتىگە ئىگە بولغانلىقىم ئۈچۈن ئۆزۈمنى ناھايىتى بەختلىك ھىس قىلماقتىمەن ، شۇنداقلا ناھايىتى ھاياجانلانماقتىمەن . بىزنىڭ بۇ يەرگە كېلىشىمىزدە ئىككى مەقسەت بار : بىرى ، بۇ يەردىكى سايلام خىزمىتىنى نازارەت قىلىش ؛ يەنە بىرى ، ئاممىنىڭ ئەھۋالىنى ئۇقۇش ، ئەگەر قانۇنغا خىلاپ ئىشلار ، قىيىنچىلىقلار بولسا ئۇنى ھەل قىلىش ، ئاممىغا ھازىرقى دىمگىراتتىك سىياسەتنى چۈشەندۈرۈش ، ئاممىنىڭ پىكىرلىرىنى ئاڭلاش .

      سايلام مەسىلىسى توغرىسىدا ، ئەڭ ئاۋال ، نىمە ئۈچۈن سايلام ئۆتكۈزىدىغانلىقىمىز ئۈستىدە توختىلىپ ئۈتىمەن . ھەممىمىزگە مەلۇم ، ئۆتمۈشتە ئادالەتسىز ئىشلار بەك كۆپ بولدى . خەلقىمىز نۇرغۇن ئازاپ - ئوقۇبەت تارتىپ ، نۇرغۇن ئىزىلىشكە ئۇچرىدى ، ئەمما دەرت ئەھۋالىنى ئېيتىدىغان يەر تاپالمىدى . كونا ھۆكۈمەتنىڭ باشلىقلىرى تۈرلۈك زۇلۇم ۋاستىلىرىنى ئىشقا سېلىپ خەلقنىڭ قېنىنى شوراپ ، خەلقنى چىدىغۇسىز ھەلەتكە كەلتۈرۈپ قويدى . شۇنىڭ بىلەن جايلاردا قوزغىلاڭلار كۆتۈرۈلدى . بۇ قوزغىلاڭلارنىڭ كۆلىمى ئەڭ زور بولغىنى ئىلى ، تارباغاتاي ۋە ئالتاي ۋىلايەتلىرى بولدى . ئۈچ ۋىلايەتتىكى قېرىنداشلار كۆتۈرگەن قوزغىلاڭ مۇستەبىتچىلەرگە قارشى تۇرۇپ ، پۈتۈن ئۆلكىدىكى ئىزىلگەن 4 مىلىيۇن 500 مىڭ خەلقنى ئازاتلىققا چىقىرىش يولىدىكى بىرخىل ھەركەتتۇر . ئۈچ ۋىلايەتنىڭ خەلق ۋەكىللىرى مەركىزى خەلق ھۆكۈمىتىنىڭ ۋەكىلى يەنى ھازىرقى ئۆلكىلىك ھۆكۈمەتنىڭ رەئىسى جاڭ جىجۇڭ بىلەن سەككىز ئاي مۇھاكىمە ئۆتكۈزۈپ ، «11 ماددىلىق تېنچلىق بېتىمى» نى ئىمزالىدى . خەلقنىڭ نۇرغۇن يىللاردىن بىرى ئارزۇ قىلىپ كەلگەن ۋە تەلەپ قىلىپ كەلگەن ھوقۇقى ئاشۇ بېتىمنىڭ ئىچىگە تولۇق كىرگۈزۈلدى .

ھازىر مۇشۇ تىنچلىق بىتىمىنىڭ 1 – ماددىسىغا ئاساسەن ، خەلققە ئۈزى ئىشىنىدىغان ئادىل كىشىلەردىن كېڭەش تەشكىللەپ چىقىش ۋە ناھىيىنىڭ ھاكىمى ، مۇئاۋىن ھاكىمىنى سايلاپ چىقىش ھوقۇقى بىرىلدى . ئىچكى ئۆلكىلەردە خەلققە ناھىلىك كېڭەشنىڭ ئەزالىرىنى سايلاپ چىقىش ھوقۇقى بىرىلگەن بولسىمۇ ، ناھىيىنىڭ ھاكىمىنى سايلاش ھوقۇقى بىرىلمىگەن  . ۋاھالەنكى ، ئۆلكىمىزدە بۇنداق ھوقۇق ھەممىدىن ئاۋال يولغا قويۇلماقتا . بۇ تىنچلىق بىتىمىدە بەلگىلەنگەن ھوقۇق . ئۇ ئاسان قولغا كەلمىدى . بەلكى مىڭلىغان - تۈمەنلىگەن ئادەمنىڭ قان ئاققۇزۇپ قۇربان بىرىشى بەدلىگە كەلدى . شۇنىڭ ئۈچۈن بىز ئۇنى قەدىرلىشىمىز ، ئاممىغا كۆڭۈل بۈلىدىغان ، ئامما ئىشىنىدىغان ، ئامما ياخشى كۈرىدىغان كىشىلەرنى خىزمەتنى باشقۇرۇشقا ئەستايىدىل سايلاپ چىقىشىمىز كېرەك .

      بىز بارغانلا يەردە ئاممىدىن نەچچە يۈزلىگەن تەلەپنامىلەرنى تاپشۇرۇپ ئالدۇق ، بۇ تەلەپنەمىلەردە : ئىزىلىشكە ئۇچراپ ، ئېغىر ئازاپ تارتقانلىقىدىن شىكايەت قىلىنغان ، ئۇنى بىر ئامال قىلىپ ھەل قىلىپ بىرىش ، بۇنىڭدىن قۇتۇلدۇرۇش تەلەپ قىلىنغان . بۇنىڭدىن كىيىن ، بەختكە ئىرىشىش ھوقۇقى ئۆز قولۇڭلاردا . ئەگەر سىلەر ئادىل ، ۋىجدانلىق ، ئەخلاقلىق ، بىلىملىك ۋە خەلقنىڭ مەنپەتى ئۈچۈن ئىشلەيدىغان ئادەملەرنى سايلاپ چىقىدىغان بولساڭلار ، سىلەرنىڭ مەنپەئىتىڭلار كاپارلەتكە ئىگە بولىدۇ . ئەگەر سىلەر نامۇۋاپىق ئادەملەرنى ، مەسىلەن ، خىيانەتچى ، قىمارۋاز ۋە ۋىژدانسىز ئادەملەرنى سايلاپ قويىدىغان بولساڭلار ، ئۇ ھالدا ، بۇنىڭدىن كېلىپ چىقىدىغان بالايىئاپەتنىڭ دەردىنى ئۆزۈڭلار تارتىسىلەر ، ئەلۋەتتە . خەلقتە يامان ئادەملەرنى خىزمىتىدىن ئېلىپ تاشلاش ھوقۇقىمۇ بولىدۇ . خەلق ئاشۇنداق يامان ئادەملەرنى خىزمىتىدىن ئېلىپ تاشلاپ ، باشقىدىن سايلىسا بولىدۇ . ئەمما كونىلار : «بېگىڭگە يەتكۈچە بېلىڭ سۇنۇپتۇ» دەپ ياخشى ئېيتقان . سەن ئۇنى خىزمىتىدىن ئېلىپ تاشلىغۇچە ، ئۇ قىلىدىغان ئەسكىلىكلىرىنى قىلىپ بولىدۇ ، شۇڭا خەلق ئاممىسىغا غەمخورلۇق قىلىدىغان ياخشى ئادەمنى سايلاش كېرەك .

      بىز قەشقەر ، يەكەنلەردە تەكشۈرۈش ئېلىپ باردۇق . پوسكام ناھىيىسىنىڭ كاتتىۋاشلىرى «مۇئاۋىن رەئىس كېلىدىكەن» دىگەن نام بىلەن خەلقتىن نۇرغۇن توخۇ ۋە تۇخۇم يىغىۋاپتۇ ، بۇنى ئاڭلاپ ناھايىتى خاپا بولدۇم . چۈنكى بىز بۇ يەرگە خەلقنىڭ يۈكىنى ئېغىرلىتىش ئۈچۈن ئەمەس ، بەلكى خەلقنىڭ ئۈستىدىكى سېلىقلارنى تۈگىتىش ئۈچۈن كەلدۇق . مەن پوسكام ناھىيىسىدە ئېچىلغان ئاممىۋى يىغىندا سۆزلىگەن سۆزۈمدە : سىلەر توخۇ ۋە تۇخۇملىرىڭلارنى ئېلپ كىتىڭلار ! دەپ جاكارلىدىم . ئۇنىڭدىن قالسا ، تەكشۈرۈشلەرگە قارىغاندا ، بەزىلەر جايلاردا تۇرۇشلۇق ھەربىيلەرگە كېرەكلىك قۇرۇق ئوت – چۆپ ، يەم -خەشەكلەرنى ئارتۇق ئالىدىكەن . 100 باغ بىدە تاپشۇرۇشقا تىگىشلىك بولسا ، 150 باغ بىدە ئېلىنىكەن . بىر دادەن كۆممە قوناق ئېلىشقا تىگىشلىك بولسا بىر يېرىم دادەن كۆممە قوناق ئېلىنىپ ، ئارتۇق ئېلىنغان كۆممە قوناق ، يەم – خەشەكلەرنى ئۆز چۆنتىكىگە سالىدىكەن . ھۆكۈمەت بەرگەن پۇل مال ئىگىسىنىڭ قولىغا تەگمەيلا باشقىلار خىيانەت قىلىپ كىتىدىكەن .

يەنە بەزى جايلاردىكى كاتتىۋاشلار ھۆكۈمەتكە كېرەكلىك نەرسىلەرنى ئەزەلدىن ئۆزلىرىنىڭ ئۇرۇق تۇغقانلىرىدىن ئالمايدىكەن . بۇلار قانۇنغا خىلاپ ، شۇنداقلا دىنىي ئەقىدىلەرگىمۇ مۇۋاپىق كەلمەيدۇ . ئىسلام دىنىنىڭ دىنىي ئەقىدىلىرىدە ئاممىغا زۇلۇم سېلىشقا ، ئاممىنى ئىكىسپىلاتاتسىيە قىلىشقا يول قويۇلمايدۇ . مەسىلەن : مىنىڭ ئۇرۇق تۇغقانلىرىم مەن مۇئاۋىن رەئىس بولغانلىقىم ئۈچۈن تاپشۇرۇشقا تىگىشلىك ئالۋان ياساقنى تاپشۇرمىسا ، بۇ توغرا بولامدۇ ؟ ئەلۋەتتە ، توغرا بولمايدۇ . مىنىڭ ئۇرۇق تۇغقانلىرىم مۇئاۋىن رەئىسكە تۇغقان ئەمەس ، بەلكى بۇرھانغا تۇغقان . مۇئاۋىن رەئىس بىرنەچچە ئادەمنىڭ مۇئاۋىن رەئىسى ئەمەس ، بەلكى خەلقنىڭ مۇئاۋىن رەئىسى . شۇنىڭغا ئوخشاشلا ، ھاكىم ، شاڭجاڭ ، باۋجاڭلارنىڭ ئۇرۇق – تۇغقانلىرى تۆلەشكە تىگىشلىك ئالۋان – ياساقلارنى باشقىلارنىڭ زىممىسىگە ئارتىپ قويسا ، ئۇنداق ئادەم ھۆكۈمەت ئالدىدا ، ئامما ئالدىدا ۋە خۇدا ئالدىدا گۇناھكار بولىدۇ . ھۆكۈمەت بۇنىڭدىن كىيىن مۇشۇنداق ئىشلارنىڭ يۈز بىرىشىگە ھەرگىز يول قويمايدۇ . ئەگەر تەكشۈرۈپ تېپىلىپ قالىدىغان بولسا ، ئۇنداقلار قانۇن بويىچە جازالىنىدۇ .

     بۇرۇن ئۆلكىلىك ھۆكۈمەتنىڭ بىر رەئىس بىلەن قوشۇلۇپ 13 ھەيئەت ئەزاسى بار ئىدى . ھازىر «11 ماددلىق تىنچلىق بىتىم» نىڭ 9 – ماددىسىدىكى بەلگىلىمە بويىچە ، ھۆكۈمەت ھەيئەت ئەزالىرى 25 بولىدۇ . بۇنىڭ ئونىنى مەركەز تەيىنلەيدۇ ، ئون بەشى يەرلىك زاتلاردىن تەيىنلىدۇ . ھازىر بىر رەئىس ، ئىككى مۇئاۋىن رەئىس بار . جاڭ جىجۇڭ رەئىس ، ئەخمەت ئەپەندى بىلەن ئىككىمىز مۇئاۋىن رەئىس . ئۆلكىلىك ھۆكۈمەتنىڭ ھەيئەت ئەزالىرى ئۆلكىمىزدىكى 4 مىلىيۇن 500 مىڭ خەلققە ۋەكىللىك قىلالايدۇ ، شۇنداقلا ئاممىنىڭ مەنپەئىتىگە كۆڭۈل بۆلەلەيدۇ . بۇ يىل 7 – ئاينىڭ 1- كۈنى ، بىز خىزمەتكە ئولتۇرغان ۋاقتىمىزدا ، خەلق ئۈچۈن ئادىل خىزمەت قىلىمىز ، خەلقنىڭ خاتىرجەم ، بەختلىك تۇرمۇش كەچۈرىشى ئۈچۈن ساداقەتلىك بىلەن ئىشلەيمىز ، پۈتۈن ئۆلكىمىزدىكى 4 مىلىيۇن 500 مىڭ خەلقنىڭ تاپشۇرىقىنى يەردە قويمايمىز ، دەپ قەسەم بەردۇق .

      ئۆلكىلىك ھۆكۈمەت تىنچلىق بىتىمىنىڭ روھىغا ئاساسەن «سىياسى پروگرامما» نى تۈزۈپ چىقتى . بىز مۇشۇ پىروگرامما بويىچە ئىشلەيمىز . ھازىر خەلققە سۆز ، مەتبۇئات ، تەشكىلاتلارغا ئويۇشۇش ئەركىنلىكى بىرىلدى ، بۇنىڭدىن كىيىن ، بىز ئۆز پىكىرلىرىمىزنى ئاشكارا بېرىپ ، كۆپچىلىكنىڭ تەلەپ ۋە پىكىرلىرىنى گېزىتلەردە ئېلان قىلالايمىز . مەسىلەن : جايلاردىكى مەسئۇل كىشىلەر ئەگەر خىيانەتچىلىك قىلسا ياكى قىمار ئوينىسا ، ئەپيۈن ، نەشە چەكسە ، سىلەر ئۇلارنىڭ قانۇنسىز ھەركەتلىرىنى پاش قىلىش ھەمدە گېزىتلەردە ئېلان قىلىشقا ھوقۇقلۇق ، بۇنى ھەرقاندا كىشىنىڭ توسۇش ھەققى يوق . ئەمدى كىشىلەرنى يولسىز ، قانۇنسىز سولاپ قويۇشقا ، قانۇنسىز جەرىمانە قويۇشقا يول قويۇلمايدۇ . ئەمدى سىلەر ئۆزۈڭلارنىڭ نەقەدەر چوڭ ھوقۇققا ئىگە بولغانلىقىڭلارنى بىلىشىڭلار كېرەك . ئالايلۇق ، سىلەرنىڭ ئالدىڭلارغا يوغان لىگەندە پولو كەلتۈرۈلسە ، ئەگەر سىلەر شۇ پولونى يىسەڭلار قورسىقىڭلار تويىدۇ ، يىمىسەڭلار ئاچ قالىسىلەر . ھازىر سىلەرنىڭ ئالدىڭلارغا پولو كەلتۈرۈلدى ، ئەمدى سىلەر ئۇنى يەپ قورسىقىڭلارنى تويغۇزالايسىلەر .

      بىز مەكىتتە ئون كۈندەك تۇرۇپ ، سايلام خىزمىتى ئاخىرلاشقاندىن كىيىن قايتىپ كىتىمىز . مۇشۇ ئون كۈن ئىچىدە تەشۋىق خىزمىتىنى تولۇق ئىشلەپ ، خەلقنىڭ كېڭەش ئەزالىقىغا ۋە ناھىيىنىڭ ھاكىملىقىغا ياخشى ئادەملەرنى سايلىشىغا ئوڭايلىق تۇغدۇرۇپ بىرىش ئۈچۈن ، بەش كىشىلىك سايلام ھەيئىتى سايلاپ چىقتۇق ، ھەر بىر يېزىدا ئۈچ كىشىلىك شۆبە ھەيئەت تەشكىل قىلىنىدۇ .شۇڭا سىلەرنىڭ سايلام ھوقۇقۇڭلاردىن تولۇق پايدىلىنىپ ۋە ئۇنى ئىشلىتىپ ، كەڭ خەلق ئاممىسى ئۈچۈن تىرىشىپ خىزمەت قىلىدىغان ئادەملەرنى سايلاپ چىقىشىڭلارنى ئۈمۈت قىلىمەن» ①.

① بۇرھان شەھىدى ، «شىنجاڭدىكى 50 يىلىم» ، مىللەتلەر نەشىرىياتى ، 1986 – يىل 5 – ئاي نەشىرى . 646 ~ 650 – بەتلەر .


    1946 – يىلى 12 – ئاينىڭ 10 – كۈنى مەكىت ناھىيىسىدە سايلام ئۆتكۈزۈلۈپ ، مۇھەممەت ھاجى ، قادىر خەلىپە قاتارلىقلار كېڭەش ئەزاسى ، مويدىن نىياز ھاكىم بولۇپ سايلانغان . بىراق ئۆلكىدىن كەلگەن سايلام گورۇپپىسىدىكىلەر قايتىپ كىتىپ ئۇزاق ئۆتمەي ، يەكەننىڭ ۋالىيسى جۇ فاڭگاڭ زورلۇق بىلەن مويدىن نىيازنى ھاكىملىقتىن ئېلىپ تاشلاپ ، ئۇنىڭ ئورنىغا ئۆزىنىڭ يېقىن كىشىسىنى ھاكىم قىلىپ تەيىنلىگەن ① .

① «مەكىت ناھىيىسى تەزكىرىسى» ، شىنجاڭ ئۇنۋېرسىتىتى نەشىرىياتى ، 1994 – يىل 5 – ئاي ، خەنزۇچە نەشرى ، 10 – بەت .


مەنبە : «مەكىت تارىخ - مەدەنىيىتى» ، شىنجاڭ ئۇنۋېرسىتىتى نەشىرىياتى ، 2006 - يىلى4 - ئاي نەشىرى ، 161~167 – بەتلەر .
