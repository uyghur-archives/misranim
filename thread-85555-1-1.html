                                             قاتىلنى قوغلاۋاتقان قىلىچ
                                      (سۆزئويۇنى)

         قاسىم قىمارۋاز  قىمار قەسىرىنىڭ قەرز قاينىمىغا قاپسىلىپ قالغاندا،قىماۋازلار قەرزنى قايتۇرۇشقا قايتا-قايتا  قاتتىق قىستىغان. قاسىم قىمارۋاز قىمارنىڭ قەرزىنى قايتۇرالمىغاچقا، قانخور قىمارۋازلار، قاسىم قىمارۋازنىڭ قانىنى قاتىللىق قۇرالى قىلىچتا قوچۇپ قان قۇستۇرۋىتىشنى قارار قىلىپ قولىدىكى قىلىچنى قاسىم قىمارۋازنىڭ قولتۇقىغا قاداشقا قاڭقىتىۋاتقاندا، قاسىم قىمارۋازنىڭ قېرىداشلىرى قېرىداشلىقنى قەدىرلەپ، قانخور قىمارۋازلارغا قوتاندىكى قوي-قوتازلىرىنى، قىشلاقتىكى قوغۇن-قاپاقلىرىنى، قازناقتىكى قىممەتلىك قوناقلىرىنى، قازاندىكى قۇيماقلىرىنى، قاچا-قۇچۇلاردىكى قىمىز-قايماقلىرىنى،قەپەزدىكى قاغا-قۇزغۇن قۇچقاچلىرىنى قەرزنىڭ قىممىتىگە قەتئى قايغۇرماستىن قايتۇرۇپ  قېرىداشلىرىنى  قەھرىمانلارچە قۇتۇلدۇرۋالغان. قاسىم قىمارۋازمۇ قەرزدىن قۇتۇلۇپ قېرىداشلىرىنىڭ قېرىداشلىرىنىڭ قاتارىغا قوشۇلغانلىغىنى قەدىرلەپ،قىماردەك قاملاشمىغان قىلىقنى قىلماسلىققا قەسەم قىلىپ قەغەزگە قالدۇرۇلغان قەسەمنامىغا قول قويغان.
     قاسىم قىمارۋاز قىماردەك قارا قويمىچىنىڭ قىلتىقىدىن قۇتۇلغالىقىنى    قەدىرلەپ،قېرىداشلىرىنىڭ،قۇلۇم-قوشنىللىرىنىڭ قوللىشىغا قوشۇلۇپ قورونىڭ قارشى قىرغىقىدىكى قوشۇقتەك قولتۇقتا قاسساپلىق قىلدى.
       قاسىم قاسساپ  قاسساپلىق قىلىشتا  قوي قوتازنىڭ قارنىنى قىڭراقتا قوچۇپ ،قوينىڭ  قۇيرۇقىنى قوتازنىڭ قاڭشىرىغا قاداپ قالتىس قاملاشتۇردى،
         قاسىم قاسساپ قوي-قوتازنىڭ قارنىنى قوچۇۋەرگەچكە قولى قاندەك قىزىرىپ،قىياپىتى  قارىداپ،قەلبى قاراملىشىپ،قاتىللىقنىڭ قۇراللىرىنى قولمۇ-قول قۇراشتۇرۇپ،قاتىللىقا قەدەم قويدى. قاتىللىق قىلغان قىلىقلىرىغا قارايلى، قورودىكىلەر قوروغا قاراڭغۇدا قايتقاندا قارشىلىشىپمۇ،قايرىلىپمۇ قارايدىغان،قارىغاندىمۇ قاش-قاپاقلىرىنى قىمىرلاتماي قولتۇقتىكى قاپقۇچقا قارايدىغان قالتىس قائىدىنى قوللاندى. قورودىكىلەر قاسىم قاسساپتىن قورقۇپ قاچسا قوغلىدى.
      قاسىم قاسساپنىڭ قاملاشمىغان قىلىقى ؛-قاراڭغۇدا قوروغا قايتقان قېرىندىشىنى قورقاتقان، قېرىندىشى، قاسىم قاسساپنىڭ قولىدىكى قاتىللىق قۇرالى قىڭراقتىن قورقۇپ قاچقان،قاچسا قوغلىغان.قوغلاپ، قوغلاپ قاچقانلاردىن قەدىمى قسقارغان،قامىتى قاۋۇل، قېرىپ قالغىننى قۇچاقلىۋىلىپ، قوينىدىكى قارا قاپقۇچنى قايتۇرۋىلىشنى قارار قىلىپ ، قولىدىكى قاتلىما قىڭراقتا قارنىنى قوچىۋىتىپ  قارىسا قان قۇسقىنى قېرىندىشى، قوينىدىكى قاپقۇچ قۇرۇق.قاپقۇچنىڭ قۇرۇقلىغى قاپقۇچتىكى قىزىل قالدۇقلارنى قىمار قەرزىگە قايتۇرغاندا قۇرۇغدالغان.
      قاسىم قاتىل قىلغان قىلىقلىرىدىن قورقۇپ قېچېشقا قايرىلىۋاتقاندا قورونىڭ قاراۋۇللىرى قولىدىكى قانۇنلۇق قىلىچلىرىغا قوغلۇتۇپ قورشىۋالدى،قاراۋۇللار قولىدىكى قىلىچنى قاسىم قاتىلنىڭ قوڭىغا قاداپ قويۇپ، قاتىلنىڭ قولىنى قاتلاپ قۇلۇپلىدى.
       قاسىم قاتىلنىڭ قېدىرلىك قېرىندىشى قازا قىلدى،قوينىدىكى قىزىل قىممەتنى قوغىدايدىغان قاپقۇچ قۇرۇغدالدى.
       قاسىم قىماۋازنىڭ قىماردىكى قەرزىنى قېرىنداشلىرى قايتۇرغان.قاتىللىق قىلمىشىغا قانۇن قانداق قارار قىلار؟! قېرىنداشلىرى قانداق قارا؟
