_ ھايامدىن بوگان _ ئىككى يېرىم ياشلىق قىزىمنىڭ ئېغىزىدىن چىققان بۇگەپ بىلەن قولىڭىز مىنى ئۇرۇشتىن توختىدى ، قىزىمغا شۇنداق بىر ئالىيىۋەتتىڭىز-دە سىرتقا چىقىپ كەتتىڭىز.
چىقىپ كېتىشىڭىزگە ئىككىلا بالام ماڭا ئېسىلىپ يىغلاپ كەتتى .
يىغلاۋاتىمەن ، يىغلىشىم سىزنىڭ ھېيتنىڭ تۇنجى كۈنىدىلا نامازغا بالدۇر ئويغاتقانلىقىم سەۋەبلىك مىنى قاتتىق ئۇرىشىڭىز ، ۋە ياكى ئېغىزىڭىزدىن چىققان ئاشۇ تىلغا ئالغۇسىز تىللىرىڭىز سەۋەبىدىن ئەمەس، بەلكى ئاشۇ كىچىك قىزىمنىڭ سىزنى، ئۆز دادىسىنى سىز مىنى تىللىغان ئاشۇ سەت ئاتالغۇ بىلەن تىللىشى ئىدى .
مەن سىز بىلەن مۇشۇنداق تىل - تاياقلىرىڭىز ئاستىدا چىشىمنى چىشلەپ ئۈچ يىل بىللە ئۆتىشىم ئاشۇ نارەسىدە قىزىم ئۈچۈن ئىدى . ئاشۇ قىزىمنى قانداق قىلىپ بولمىسۇن ياخشى تەربىيەلەش ئۈچۈن دادا مىھرىگە تەشنا بولىدىغانلىقىنى ئويلاپ تىللىسىڭىزمۇ ،ئۇرسىڭىزمۇ چىداپ كەلگەنىدىم لىكىن بۈگۈن ...
بىز توي قىلغىلى ئۈچ يىل بولدى ، نىكاھىمىز ئوقۇلىۋاتقان شۇ ۋاقىتتا ئۆز - ئۆزۈمگە ئاياللىق بۇرچۇمنى ياخشى ئادا قىلىشقا ۋەدە بەرگەنىدىم ، ۋەدەمدە تۇرۇش ئۈچۈن تېرىشتىم ، توختىماي تېرىشتىم . سىز ياقتۇرمايدىغان ئىشلارنى ئامالنىڭ بارىچە قىلمىدىم ، ئانامنىڭ ئۆيىگىمۇ بىرىشقا رۇخسەت قىلمىسىڭىز بارمىدىم ، ھەتتا ئاي - ئايلاپ ئۆيدىن سىرتقا چىقماي ، ئۆيدىكىلەرنىڭ خىزمىتىنى قىلدىم- يۇ ،لىكىن سىزنىڭ مەندىن كىچىككىنە رازى بولغىنىڭىزنى ھېس قىلالمىدىم .
توي قىلىپ بىر ئاي بولغاندا ئېغىر بوي بولغانلىقىمنى بىلگىنىمدە كۆڭلۈم شۇنچىلىك يېرىم بولدى ، چۈنكى سىز بىلەن بىللە ياشىغان بۇ بىر ئاي جەريانىدا سىزنىڭ بەزى مىجەزىڭىزنى چۈشىنىپ سىز بىلەن مەڭگۈ بىللە ئۆتۈشكە كۆزۈم يەتمىگەن ئېدى ، چۈنكى سىز بەكمۇ شەخسىيەتچى ، مەنمەنچى ئىدىڭىز . ئۆزىڭىزدىن باشقىلارنى ھەتتا ئۆز ئاتا - ئانىڭىزنىمۇ  ھەرگىز ياراتمايتىڭىز .
لىكىن مەن يەنىلا بۇ بالىنى تۇغۇشنى ئويلىدىم ، بىر ئانا تۇرۇپ ئۆز بالامنى ئۆلتۈرۈشكە(ئالدۇرىۋىتىشكە) پەقەتلا كۆڭلۈم ئۇنىمىدى . شۇنداق قىلىپ بۇ بالا سەۋەبلىك ئائىلىمىز ساقلىنىپ قالدى . سىزنىڭ "كىمنىڭ بالىسى بۇ ؟ توي قىلغىلى ئۇزۇن بولمىغان تۇرسا " دىگەندەك گەپلىرىڭىزگىمۇ چىداپ ساق - سالامەت يەڭگىۋالدىم ،ماڭا ئاشۇ ھاقارەتلىك سۆزلەرنى قىلغىنىڭىزغا ، قىزىمىز قۇيۇپ قويغاندەك سىزگىلا ئوخشايتتى .
كۈنلىرىمىز بالىمىزنى بېقىش بىلەن ئۆتتى ، 3 ئايدىن كىيىن ئاللا بىزگە يەنە بىر پەرزەنت ئاتا قىلدى . سىزنىڭ ماڭا قىلىدىغان ئۇۋالچىلىقلىرىڭىزمۇ شۇنىڭ بىلەن ئېشىپ باردى .
تۇغۇلغىنىغا ئەمدى ئۈچ ئاي بولغان قىزىمنى سۈتتىن ئايرىش، مىنىڭ سىزىكلىك مەزگىلىم بىللە كەلدى . سىز ئەنە شۇ چاغدىن باشلاپ ، ئۆزىڭىزنىڭ تېخىمۇ يامان مىجەزىڭىزنى ئاشكارىلىدىڭىز. يوقىلاڭ سەۋەبلەر بىلەن مىنى ئۇراتتىڭىز ، قورسىقىمدىكى بالىڭىز خىيالىڭىزغا كىرىپ چىقمايتتى ، كەچلىرى يېرىم كىچىنى قىلماي ئۆيگە كىرمەيتتىكىز ، ئۆيگە كىرسىڭىزلا يەنە شۇ ئۇرۇش ... سىز ئىشىكتىن چىقىپ كەتكەندىلا ئۇف... دەپ تىنىۋالاتتىم ، سىزدىن شۇ دەرىجىدە قورققانىدىم ، ئاللادىن بۇ بالامنىڭ دۇنياغا كۆز ئاچماسلىقىنى تىلەپ ئۈنسىز ياش تۆكەتتىم . چۈنكى بۇ بالامنىڭمۇ دۇنياغا كۆز ئېچىپلا جەڭگى - جىدەل ئارىسىدا بوينىنى قىسىپ ياشىشىنى خالىمايتتىم .
ئاللانىڭ مەلۇم بىر ھىكمىتى بولسا كىرەك بۇ بالىمۇ دۇنياغا كۆز ئاچتى ، يەنە بىر ئوغۇل پەرزەنتكە ئانا بولدۇم ، قەلبىمدە خۇشاللىق بىلەن بىرگە ئېيتىپ تۈگەتكۈسىز قايغۇمۇ بار ئىدى ، سەۋەبى سىزنىڭ يىقىندىن بۇيان بۇرۇنقىدىنمۇ بەك ئېشىپ كەتكەن ئۇرۇش - تىللاشلىرىڭىز ئىدى .
ئەنسىرىشىمنىڭ توغرىلىقىنى بارا - بارا ھىس قىلدىم ، سىزگە تىكىلىپ قارىغانلىقىم ، بالىلارنىڭ يىغىسى سەۋەبلىك ئويغۇنۇپ كەتكەنلىكىڭىز ، سىزگە بىرەر ئېغىز ھەق گەپ قىلغانلىقىم ئۈچۈنمۇ تاياق يىدىم .  ئۇرۇپ قولىڭىز ئاغرىسا قولىڭىزغا نىمە چىقسا شۇنىڭ بىلەن ئۇراتتىڭىز. قىيىن ئاتا - قىيىن ئانامنىڭمۇ ماڭا ئىچى ئاغرىپ كىتەتتى . بىر نەچچە قېتىم" بولدىلا "دەپ ئانامنىڭ ئۆيىگە كەتتىم ، لىكىن يەنىلا شۇ بالا ئۈچۈن .... ئانا دىگەن بالىسى ئۈچۈنلا ياشايدىغان گەپكەن دادام شۇنچە ئامراق ئارزۇلاپ باققان قىزىنىڭ ئۇيەر - بۇ يەرلىرىنىڭ كۆك ، ئىششىق ئىكەنلىكىنى كۆرىۋىرىپ چىدىماي ئاجرىشىپ كىتىشىمنى تەلەپ قىلغان بولسىمۇ ، قايغۇلىرىمنى ئىچىمگە يۈتۈپ يەنىلا يىنىڭىزغا قايتىپ كەلدىم ،شۇ سەۋەبلىك داداممۇ ماڭا قارىماس بولدى  . لىكىن يەنىلا بالىلىرىم ئۈچۈن ، ھەممىسى بالىلىرىم ئۈچۈن قۇربانلىق بىرىشكە رازى بولدۇم . لىكىن سىز بۇنى مىنىڭ ئاجىزلىقىم سۈپىتىدە چىڭ تۇتىۋىلىپ مىنى يەنە قىيناشنى داۋاملاشتۇردىڭىز. مەن بىر ئېغىزمۇ گەپلىشىپ باقمىغان تۇققانلىرىڭىزدىن كۈنلەپمۇ ئۇردىڭىز، مىنىڭ ئۆيۈمدىكىلەرنى پەس - گاداي دەپ تىللىدىڭىز ، مىنى بولسا ئىرىگە خىيانەت قىلىدىغان يامان ئايال قاتارىغا چىقىرىۋەتتىڭىز .
ھېلىمۇ ياخشى قىيىن ئانام يەنى ئاپىڭىز ياخشى بولغاچقا ماڭا تەسەللى بەردى ، ئۆزىنىڭ شۇنداق بالىنى تۇغقانلىقى ، تەربىيىلىگەنلىكىدىن ئۇنىڭمۇ كۆڭلى يېرىم ئىدى. ۋاقىتمۇ ئاستا - ئاستا ئۆتىۋاتاتتى ...
مانا بۈگۈن قۇتلۇق ئايەم كۈنلىرىدە قىزىمنىڭ سىزنى شۇنداق سەت گەپ بىلەن تىللىشى مىنى ئويغا سېلىپ قويدى . قىزىمنى ئەخلاقلىق، ئىمانلىق قىلىپ تەربىيەلەشنى ئارزۇ قىلغانىدىم ، سىزنىڭ مىنى توختىماستىن تىللىشىڭىز ، ئۇرىشىڭىز بالىغا يامان تەسىر پەيدا قىپتۇ ، قىزىم ئاشۇ چىرايلىق ئاغزىدىن شۇنداق سەت گەپنى چىقىرىپ سىزنى تىللىدى . بۇ سىزنىڭ ئۇرۇشقاقلىقىڭىزنىڭ قىزىمغا بەرگەن تەسىرى ، قېنى ئېيتىڭە _ سىزگە كىرىكى مۇشۇنداق نەتىجىمۇ ؟

كۆڭلىڭىزنى يېرىم قىلماڭ. ئاللاھ يولدىشىڭىزغا ئىنساپ بېرەر. قەيسەر بولۇڭ. بالىلار يېتىم بولمىسۇن.
ئۇنداق ئەركىشىلەر كۈچلۈكتەك كۆرۈنگەن بىلەن، ئەمەلىيەتتە ئەڭ ئاجىز بايقۇش. ئۆيدە سىزنى ئۇرالىغان بىلەن، بەلكىم سىرتتا باشقىلارنىڭ ئالدىدا ئەڭ بىچارە بولۇشى مۇمكىن.
ئايال كىشى يولدىشىدىن يۈز ئۆرۈپ، خىيانەت قىلىپ ئىرىگە يىشىل قالپاق كەيدۈرۈپلا قويمىسا، باشقا ئىشلار ئۈچۈن ئايال كىشىنى ئۇرۇشقا قەتئىي بولمايدۇ، بۇ، مېنىڭ شەخسى كۆز قارىشىم.
ئائىلىدە بىلىپ بىلمەي بالىلارغا يامان تەسىر بېرىپ قويىمىز. مەنمۇ بالامنى يېزىدىكى تۇغقانلارنىڭ قارىشىغا بەرگەن ئىدىم. بىر چاغدا ئوغلۇم(تېخى 1 يېرىم ياشقا كىرمىگەن ۋاقىت) ئاناڭكەي دىگەن گەپنى ئۈگىنىۋاپتۇ. مەن قاتتىق نارازى بولدۇم، ۋە ۋاقتىدا تۈزىتىش بېرىپ، كىچىك بالا بار يەردە ئۇنداق گەپلەرنى قىلماسلىقنى ۋە يەنە «خوتۇن ئېلىپ بېرەيمۇ» دىگەندەك گەپلەرنى قەتئىي دىمەسلىكنى ئەسكەرتتىم، ھازىر ئوغلۇم ئۇنداق گەپلەرنى دىمەيدۇ. بىر كۈنى ئايالىم بىلەن چاقچاقلىشىپ، مېنىڭ مۇئاش كارتامنى تالاشتۇق. ئوغلۇم بىزنى ئۇرۇشىۋاتقان چېغى دەپ ئويلاپ يىغلاپ كەتتى. كىيىن دىققەت قىلساق، ئىككىمىز قىزىق گەپلەرنى قىلىشىپ كۈلۈشسەك، گەرچە ئوغلۇم دىيىلگەن گەپلەرنى چۈشەنمىسىمۇ، لېكىن قوشۇلۇپ كۈلىدەكەن. شۇڭا، ئويلاپ قالدىم، بالا گەپ سۆزنى بىلمىسىمۇ، لېكىن ئاتا ئانىنىڭ  كۈلگەن كۈلكىسىنى بىلەلەيدىكەن، قوشۇلۇپ كۈلىدىكەن.
كىچىك بولسىمۇ، بالىلارنىڭ ئالدىدا ناھايىتى چىرايلىق سۆزلەرنى ئىشلەتكىنىمىز ياخشى.

ئاق چاشقان، قارا چاشقان، مۆشۈككە تۇتۇلۇپ قالمىغان ياخشى چاشقان.
