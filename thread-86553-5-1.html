بۇ تېمىنى ئوقۇپ خەنزۇلارنىڭ نەيزە،قالقان ساتقۇچى تەمسىلى ئېسىمگە كەلدى.نەيزە-قالقان ساتقۇچى بازاردا نەيزىسىنى ماختاپ:بۇ نەيزە شۇنداق ئۆتكۈركى ھەرقانداق قالقاننى تېشىۋېتەلەيدۇ دەپتۇ،ئارقىدىنلا قالقىنىنى ماختاپ بۇ قالقان شۇنداق پۇختىكى دۇنيادىكى ھەرقانداق نەيزە تېشىپ ئۆتەلمەيدۇ دەپتۇ.بۇنى ئاڭلىغان بىر دانىشمەن:ھەي بۇرادەر سەن ئۆزۈڭنىڭ نەيزىسى بىلەن ئۆزۈڭنىڭ قالقىنىنى تېشىپ باقە دەپتۇ.خەنزۇچىدىكى ئۆز-ئۆزىگە زىت 矛盾دېگەن سۆز شۇنىڭدىن كەلگەنىكەن.
مۇسۇلمان نامى ھەرگىزمۇ نام-ئەمەل،ياكى خالىغانچە قويىۋالىدىغان نام-ئاتاق ئەمەس ۋەياكى ئاتا-بوۋىمىزدىن مىراس قالىدىغان مىللەت نامى ئەمەس.مۇسۇلمان بولۇشنىڭ شەرتى(قۇرئان ۋە ھەدىسلەردە ئېنىق كۆرسىتىلگەن)نى ھازىرلىغان ئادەم مۇسۇلمان بولالايدۇ،ئەگەر ھازىرلىمىغان بولسا ئۇ ئىنسان ئۆزىنى مۇسلمان ئاتىسۇنمۇ؟!ئۇنداقتا مەزكۇر تېمىدا دېيىلگەن 10 خاتالىقنى سادىر قىلىپ يۈرگەنلەر مۇسۇلمان ھىساپلىنامدۇ؟!
ئاتا-ئانىمىز مۇسلمان،بىزمۇ مۇسۇلمان ئائىلىسدە تۇغۇلۇپ چوڭ بولدۇق،ئۆرپ-ئادەت جەھەتتە مۇسۇلمانغا مايىلمىز،لېكىن ھاراق-شاراپ ئىچمىز،قىمار ئوينايمىز،ئوغۇرلۇق قىلىمىز،غەيۋەت-شىكايەتنى توننىسى بىلەن قىلىمىز،ئىسلام دىنى،مۇسۇلماندارچىلىق توغىرلىق 2بەت كىتاپ ئوقۇپ باقمايمىز،ھارامدىن قورقمايمىز......شۇنداق كىشىلەرمۇ مۇسۇلمانمۇ؟!
ئۇنداقتا مېنڭچە تېمىنىڭ نامى ئۆز-ئۆزىگە زىت! ئەگەر تېمىدا دېيىلگەندەك مۇسۇلمانلار(بۇنداق ئاتاش پۈتۈن مۇسۇلمانلارنى كۆرسىتىدۇ،ئەگەر ئۇيغۇر مۇسلمانلار ياكى مەلۇم توپ كۆزدە تۇتۇلغان بولسا ئۇ باشقا گەپ)يۇقارقى خاتالىقنى سادىر قىلىپ يۈرگەن بولسا ئۇنداقتا دۇنيادا مۇسۇلمان يوقكەن!
بىلمىگەننى بىلدىم ، كۆرمىگەننى كۆردۇم، ئاڭلىمىغاننى ئاڭلىدىم دېمە!

بۇ تېمىنى ئوقۇپ خەنزۇلارنىڭ نەيزە،قالقان ساتقۇچى تەمس ...

ئەسسالامۇ ئەلەيكۇم !  ئەزىز قېرىندىشىم، قىممەتلىك ۋاقتىڭىزنى چىقىرىپ مەزكۇر تېمىغا ئىنكاس ياغانلىقىڭىزغا رەھمەت.

سىز مەزكۇر تېمىنى ئۇقۇشتىن بۇرۇن تېمىنىڭ بېشىدىكى  بىرىنجقى قۇرنى ئۇقىدىڭىزمۇ ياكى ئۇقىمايلا ئۆتۈپكەتىڭىزمۇ ؟  ئۇ يەردە يېزىلغىنى   “بىز مۇسۇلمانلار سادىر قىلىۋاتقان بىر قىسىم خاتالىقلار“  دەپ مەۋزۇ قۇيۇلغان.
بۇ يەردە يېزىلغىنى، ھەرگىزمۇ بارلىق مۇسۇلمانلار سادىر قىلىۋاتقان بارلىق خاتالىقلار دىيىلمىگەن.
يەنە مەزكۇر ىاتالىقلارنى سادىر قىلغان مۇسۇلمانلارنى ھىچقانداق ئالىم-ئۆلىمالار كۇپرىغا ھۈكۈم قىلمىغان. خاتالىق باشقا - كاپىر بۇلۇش باشقا.  شۇڭا بۇ ھەقتە (قانداق كىشى كاپىر بۇلىدۇ) چۇڭقۇرراق ئىزدىنىپ كۆرەرسىز.

ئەسسالامۇئەلەيكۇم !ئەلبەتتە ئوقۇدۇم،ئوقۇماي ئۆتۈك كەتسەم ئىنكاسنى ھەرگىزمۇ ئاشۇ سىز دېگەن ئوقۇماي ئۆتۈپ كەتكەن قۇرلار توغرىسدا يازماس ئىدىم.ئەمما مەنمۇ تۇخۇمدىن تۈك ئۈندۈرمەكچى ئەمەسمەن،پەقەت ئۆزۈمچە تازا مۇۋاپىق كۆرمەي ياكى تازا ياخشى چۈشىنەلمەي .ىنكاس قايتۇردۇم.لېكىن سىزنىڭ بۇ تېمىڭىزغا 《مۇسۇلمانلاردىكى خاتالىقلار》 دەپ  ماۋزۇ قويۇلغان.مېنىڭ چۈشەنچەم بويىچە مۇسۇلمانلار دېگىنىمىز ھەرگىزمۇ يولدىن ئازغان بىرقىسىملىرىنى ئەمەس بارلىق مۇسۇلمانلارنى كۆرسىتىدۇ.ھەم شۇنداقلا بىز مۇسۇلمانلار دېگىنىڭىزمۇ ئوخشاشلا  مۇسۇلمانلارنى كۆرسىتىدۇ،ئايرىملىقنى ئىپادىلىمەيدۇ.يەنە كېلىپ مەن ئىنكاسىىمدا كاپىر دېگەن سۆزنى ئىشلەتمىدىم،مۇتلەقلەشتۈرمىدىم،سوئال تەرىقىسدە يازدىم،چۈنكى مەن ئۆلىما ئەمەس،دىننى بىلىمىم ئىنتايىن چەكلىك.ئىزدىنىشكە ئۈندىگىنىڭىز تامامەن توغرا،مەن بۇ ھەقتە ئىزدىنىشكە تىرىشىمەن.
قوشۇمچە قىلاي:مەسىلەن ئامېرىكا ياكى مەلۇم غەيرى مۇسۇلمان دۆلەت ھۆكۈمىتى:دۇنياغا ئاخپارات ئېلان قىلىپ مۇسۇلمانلار .......خاتالىق ئۆتكۈزىۋاتىدۇ،دۇنيا تەرتىۋىنى قالايمىقان قىلىۋاتىدۇ!دەپ بىلجىرلىسا،ئۇلارنىڭ كۆزدە تۇتقان مۇسۇلمانلىرى پۈتۈن دۇنيادىكى مۇسۇلمانلاردۇ؟ئۇنداقتا بارلىق مۇسۇلمانلاردا  قانداق ئىنكاس پەيدا بولىدۇ؟ئۇنىڭغا جاۋابەن،مۇسۇلمان دۇنياسىدىكى نوپۇزلۇق ئۆلىما ۋەياكى مەلۇم ئورگان:بۇ غەرپنىڭ بىز مۇسۇلمانلارغا قىلغان ھاقارىتى، بىز مۇسۇلمانلار ھىچقاچان ......خاتالىق ئۆتكۈزمىدۇق دەپ تېگىشلىك جاۋاپ قايتۇرۇپ ئۇلارنىڭ ئاغزىنى تۇۋاقلىسا،بۇ جاۋاپ بارلىق مۇسۇلمانلارغا ۋەكىللىك قىلامدۇ؟دېمەك مۇسۇلمانلار بىلەن بىز مۇسۇلمانلار ماھىيەت جەھەتتە ئوخشاش.مۇشۇ تېمىغا:مۇسۇلمانلار سادىر قىلىشقا بولمايدىغان خاتالىقلار دېگەن بولساق تولىمۇ جايىغا چۈشكەن بولاتتى.
مېنىڭچە بۇ مەسىلىدە ياخشىراق ئويلىنىپ كۆرگىنىمىز تۈزۈك.بۇ پەقەت مېنىڭ شەخسى كۆزقارىشىم،زادى مېنىڭ راست دەپ سىز بىلەن دە تالاش قىلماقچى ۋە ياكى كۆڭۈل ئازارچىلىغى تېرىماقچى ئەمەسمەن.ھەربىر مۇسۇلمان ئۆزىگە پۇختا بولۇپ يۇقۇرقى 10 خاتالىقنى سادىر قىلماسلىغى كېرەك.
بىلمىگەننى بىلدىم ، كۆرمىگەننى كۆردۇم، ئاڭلىمىغاننى ئاڭلىدىم دېمە!
