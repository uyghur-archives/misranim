قاشتېشى بايرىمى ئۆتكۈزدۇق دىدى ھە دىدۇق،بايرام كۈنى ھەممە يوللارنى قامال قىلپ يەرمەنكە رايونىغا يېقىنمۇ كەلتۈرمىدى،شۇڭا بۇ بايرامدا نىمە پائالىيەتلەر بولدى ، نىمىلەر يەرمەنكىگە قويۇلدى مانا مۇشۇ رەسىمدىن كۆرۈۋاتىمەن.
زورمۇ-زور ئۇ بايرىمى ،بۇ يەرمەنكىسى دەپ ئېلىپ كېلىدىغان ئىقتىسادىي پايدىسدىن چىقىمى جىق يەرمەنكىلەرنى ئۆتكۈزگەننىڭ ئەھمىيتى نىمە؟
قاشتىشى بايرىمى ئۆتكۈزىمىزدەپ بىرمۇنچە چىقم قىلىپ ئارتىسلارنى تەكىلپ قىلىدۇ، ئىچكىردىكى يولداشلارنى تەكلىپنامە يېزىپ زورمۇ-زور ئەكىلدۇ،ئۇلارنى بىكارغا مېھمان قىلىپ ،سوۋغا بېرىشىدۇ. شۇنىڭ بىلەن يىگەن ئېغىز ئۇيىلىپتۇ دىگەندەك :«يۇرتۇڭلار ھەقىقەتەن قاشتىشى،ئەتلەس ،گىلەم ماكانىكەن»  دەپ ماختاپ قويىشىدۇ .
شۇلارنىڭ بۇيەرگە قانچىلىك مەبلەغ سالغىنى ،يەرلىكنىڭ ئقىتىسادىي تەرەققىياتى ۋە ئىشقا ئورۇنلاشتۇرۇش ئۈچۈن قانچىلىك نەپ ئېلىپ كىلدىغنى بىزگە قاراڭغۇ...
بىراق بىزنىڭ تاش كولاپ ئاز-تولا بېيىپ قېلىپ قىلغىلى ئىش تاپالماي قالغان بەزى ئەخمەق كىشلىرىمىزنىڭ يانچۇقىنى قۇرۇقداش ئۈچۈن ئويۇنچۇقىخانا نامىدىكى قىمارخانىلار، مېھمانساراي ،ھورداقىخانا ناملىرىدىكى پاھىشىخانىلار تۈركۈملەپ ئېچىلىۋاتقىنى كۆرىۋاتىمىز.
بىزگە تايلانىدچە ئاتالمىش ساياھەت ئىگىلكى ئەمەس ،بەلكى شاڭخەيدەك كۈچلۈك سانائەت ئىگىلكى كېرەك!
  بىرەر ئالتۇن كېنى بارلىقى نامەلۇم بولغان دۇبەيمۇ ھازىر دۇنيا ئالتۇن ،ئۈنچە -مارجان سودىسىنڭ مەركىزى بولۇپ قالدى. خوتەنمۇ دۇنيادىكى مەشھۇر قاشتشى ماكانى شۇڭا نېمىشقا ئېسىل گۆھەر تاش سودىسىنىڭ مەركىزى بولالمايدىكەن؟
شۇڭا يەرلىك ھۆكىمەت ئالدى بىلەن تەدبىر قوللىنىپ قاشتىشى كولاش ،سېتىش يوللىرىنى قېلىپلاشتۇرشى ، تاش بازىرىدىكى قالايمىقانچىلىق ،ئالدامچىلىق قىلمىشلىرىنى ۋاقتىدا تۈزەپ ،خوتەن قاشتېشنىڭ ئوبرازىنى ،نامىنى قوغدىشى  كېرەك . بازار رىقابىتىدە ئىشەنچ-ئىناۋەت بولمىسا قانداقمۇ ئۇتۇپ چىققىلى بولىدۇ؟
