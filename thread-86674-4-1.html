تاپشۇرۇقنى قانچە كۆپ ئىشلىگەن بالىنىڭ ئۆگۇنۇشىنىڭ ياخشى بولۇپ كېتىشى ناتايىن! بەزى بالىلار، بېشىنى كۆتۇرنەي تاپشۇرۇق ئىشلەيدىكەن، ئۇگۇنۇش قىلىدىكەن، لېكىن ئىمتىھاندا، ئالغان نومۇرى چاغلىق ئىكەن. بەزى بالىلار تاپشۇرۇق ئىشلىمەيدىكەن،لېكىن ئىمتىھانلاردا يۇقۇرى نومۇر ئالىدىكەن.... بۇنىڭغا قاراپ شۇنداق يەكۇن چىقاردىم: بالىلار دەرس ۋاختىدا دەرسلەرنى چۇشۇنۇپ ماڭالىسا، شۇ سائەتلىك دەرسلەرنى ئۇنۇملۇك ئۆزلەشتۇرگەن بولىدۇ.شۇ سائەتلىك دەرسنى ئۆزلەشتۇرەلمىسە چۇشىنەلمىسە، ئۆيدە قارغۇلارچە كۆچۇرۇپ يىغلاپ ئولتۇرۇپ تاپشۇرۇق ئىشلىگەننىڭ نىمە پايدىسى؟ ئوقۇتقۇچىلار، دەرسخانىدا، دەرسنى بالىلارغا چۇشۇنۇشلۇك قىزىقارلىق قىلىپ ئۆتسە،  بالىلار دەرسنى ۋاختىدا چۇشۇنۇپ ماڭسا، بالىلارغا شۇنچە كۆپ تاپشۇرۇقلارنى يۇكلەپ مەسئۇلىيەتتىن ئۆزىنى تارتىشنىڭ ئورنى بارمۇ؟ بەزى ئوقۇتقۇچىلار، دەرس ئۆتىۋاتقاندا، تېلفۇن كېلىپ قالسا، دەرسنى توختۇتۇپ قويۇپ ، سىرتقا چىقىپ سائەتلەپ سۆزلۇشۇپ كېتىدىكەن.... سائەت توشۇپ قالسا، ئۆتۇپ بولالمىغان دەرسلەرنى بالىلارغا ئائىلە تاپشۇرۇغى قىلىپ بېرىدىكەن! پۇتۇن ئۆتۇپ بولۇنمىغان دەرسنىڭ تاپشۇرۇغى ،سىلەرچە بالىلارغا چۇشۇنۇشلۇكمىدۇ؟

  ‹‹يازما تاپشۇرۇق بېرىلمەيدىغان سىنىپ››نىڭ چىقىش يولى نەدە؟

خۇبېي ئۆلكىسىنىڭ ۋۇچاڭ شەھەرلىك تەجرىبە باشلانغۇچ مەكتىپى ئوقۇغۇچىلارنىڭ يۈكىنى يېنىكلىتىش ئۈچۈن، بۇلتۇر يېڭى ئوقۇش مەۋسۇمىدىن باشلاپ، يازما تاپشۇرۇق بېرىلمەيدىغان سىنىپ ‹‹ئۆمۈرلۈك تەرەققىيات تەجرىبە سىنىپى›› ئېچىش ئۈستىدە ئىزدەنگەن. بۇ مەكتەپ بارلىق يازما تاپشۇرۇقلارنى ئەمەلدىن قالدۇرۇپ، ئەمگەك قىلىش، كىتاب ئوقۇش قاتارلىق ئالتە تۈرلۈك تاپشۇرۇقنى كۆپەيتكەن. بىر يىلدىن كېيىن، ئىككى ‹‹يازما تاپشۇرۇق بېرىلمەيدىغانسىنىپ›› ئوقۇغۇچىلىرىنىڭ نەتىجىسى ئوخشاش يىللىقتىكى باشقا سىنىپلارنىڭ نەتىجىسى بىلەن ئوخشاش چىققان. بۇ مەكتەپ بۇ يىل ئوقۇغۇچى قوبۇل قىلغاندا، بۇنداق سىنىپنى ئىلگىرىكى ئىككى سىنىپتىن ئۈچ سىنىپقا كۆپەيتكەن.

   بۇ ھەقتە مۇخبىر جېجياڭ ئۆلكىسىنىڭ شاڭيۈ شەھەرلىك ئوقۇتۇش تەتقىقات ئىشخانىسىنىڭ مۇئاۋىن مۇدىرى فۇ جىخۇڭ، شېنجېن شەھەرلىك ‹‹يۈسەي›› 4 – باشلانغۇچ مەكتىپىنىڭ مۇدىرى سۈي شۆخۇڭ، جياڭسۇ ئۆلكىسىنىڭ دافىڭ شەھەرلىك 5 – باشلانغۇچ مەكتەپنىڭ مۇدىرى گاۋ جىخۇانى زىيارەت قىلدى.

   ئوقۇتۇش سۈپىتى يۇقىرى بولسا، يازما تاپشۇرۇق بېرىلمىسىمۇ بولىدۇ

   مۇخبىر: ‹‹يازما تاپشۇرۇق بېرىلمەيدىغان سىنىپ››قا كىرىش قىزغىنلىقىنىڭ كۈچلۈك بولۇۋاتقانلىقىدىن ئاتا – ئانىلارنىڭ پەرزەنتلىرىنىڭ زىيادە ئۆگىنىش بېسىمىنى يېنىكلىتىشكە بولغان كۈچلۈك ئارزۇسىنى كۆرۈۋالغىلى بولىدۇ. بۇنداق سىنىپلارنى يەنىمۇ كۆپەيتكىلى بولامدۇ؟ ئوقۇغۇچىلار يۇقىرى يىللىققا ئۆرلىگەندە ‹‹بۇنداق ئوقۇتۇش ئۇسۇلى››نى داۋاملاشتۇرغىلى بولامدۇ؟

   سۈي شۆخۇڭ: مائارىپ مىنىستىرلىقىنىڭ بەلگىلىمىسىگە ئاساسەن، باشلانغۇچ مەكتەپلەرنىڭ 1 –، 2 – يىللىقلىرىدا يازما تاپشۇرۇق بېرىلمەسلىكى كېرەك. ئەمەلىيەت مائارىپ مىنىستىرلىقىنىڭ بۇ بەلگىلىمىسىنىڭ توغرىلىقىنى ئىسپاتلىغان ئىكەن، بۇنداق سىنىپلارنى ئەلۋەتتە كۆپەيتىشكە بولىدۇ. ئوقۇغۇچىلارنىڭ يۇقىرى يىللىققا ئۆرلىشىگە ئەگىشىپ، ئۇلارغا مۇۋاپىق تاپشۇرۇق ئورۇنلاشتۇرۇشقىمۇ بولىدۇ، بۇمۇ مائارىپ مىنىستىرلىقىنىڭ مۇناسىۋەتلىك بەلگىلىمىلىرىگە ئۇيغۇن. دەرسخانىدىكى ئوقۇتۇش ئۈنۈمى يۇقىرى بولسا، باشلانغۇچ مەكتەپ باسقۇچىدا يازما تاپشۇرۇق ئورۇنلاشتۇرماسلىقنى سىناپ بېقىشقا بولىدۇ.

   فۇ جىخۇڭ: بۇيەردە تىلغا ئېلىنىۋاتقان ‹‹يازما تاپشۇرۇق بېرىلمەيدىغان سىنىپ››، بىرتۇتاش ئورۇنلاشتۇرۇلىدىغان يازما تاپشۇرۇقنى ئازايتىپ، ئوقۇغۇچىلارنىڭ ساپاسىنى ئۆستۈرىدىغان تاپشۇرۇقنى كۆپەيتكەن. مېنىڭچە، بۇ سىناق ناھايىتى ئەھمىيەتلىك بولغان. ئوقۇغۇچىلار ئۆزلىرىنىڭ ئۆسۈپ يېتىلىش قانۇنىيىتىگە ئۇيغۇن، ئومۇميۈزلۈك تەرەققىياتىنى ئىلگىرى سۈرىدىغان ئالتە تۈرلۈك ساپا تاپشۇرۇقىنى ئورۇنداش جەريانىدا ياخشى ئادەتلەرنى مۇقەررەر يېتىلدۈرىدۇ.

   گاۋ جىخۇا: بۇ سىناق سىنىپىنى ‹‹يازما تاپشۇرۇق بېرىلمەيدىغان سىنىپ›› دەپ چۈشىنىشكە بولىدۇ. ئوقۇغۇچىلار كىتاب ئوقۇش داۋامىدا، بەزى نۇقتىلارنى كۆچۈرۈۋېلىش ئارقىلىق كىتاب ئوقۇش تەسىراتلىرىنى يېزىپ چىقىدۇ. ئەگەر بۇ سىنىپنىڭ يازما تاپشۇرۇق بېرىلمەيدىغان ئالاھىدىلىكىنى گەۋدىلەندۈرمەكچى بولغاندا، ئوقۇغۇچىلارنىڭ كىتاب ئوقۇش داۋامىدا كىتاب ئوقۇش خاتىرىسى يېزىش ئادىتىنى يېتىلدۈرۈشكە سەل قارالسا، بۇنداق ئوقۇتۇشنىڭ تازا ئۈنۈمى بولماي قالىدۇ.

   يازما تاپشۇرۇقنى ماھىرلىق بىلەن لايىھەلىگەندە، يۈكنى يېنىكلىتىش رېئاللىققا ئايلىنىدۇ

   مۇخبىر: يازما تاپشۇرۇق ئورۇنلاشتۇرمايمۇ ئوقۇغۇچىلارنىڭ ئوقۇش سۈپىتىنى ئۆستۈرگىلى بولىدىغانلىقى بىر قىسىم تاپشۇرۇقلارنىڭ ھەقىقەتەن ئارتۇقچە يۈك ئىكەنلىكىنى ئىسپاتلايدۇ. يازما تاپشۇرۇق ئورۇنلاشتۇرۇش كېرەكمۇ – يوق؟ قانچىلىك ئورۇنلاشتۇرۇشتا ئىلمىي ئاساس بولۇشى كېرەكمۇ؟ ياكى ئوقۇتقۇچىنىڭ ئورۇنلاشتۇرۇشىغا قاراش كېرەكمۇ؟

   فۇ جىخۇڭ: باشلانغۇچ مەكتەپ باسقۇچىدىكى يادرولۇق ئوقۇتۇش ۋەزىپىسى ئوقۇغۇچىلاردا ياخشى ئادەت يېتىلدۈرۈش بىلەن بىللە، ئۇلارنى ئۆگىنىشكە قىزىقتۇرۇش ۋە ئۇلارنىڭ ئۆزىگە بولغان ئىشەنچىنى تۇرغۇزۇشتۇر. بۇ يەردىكى مەسىلە يازما تاپشۇرۇقنى ئورۇنلاشتۇرۇش ياكى ئورۇنلاشتۇرماسلىق مەسىلىسى بولماستىن، بەلكى قانچىلىك ئورۇنلاشتۇرۇش، قانداق ئورۇنلاشتۇرۇش مەسىلىسىدۇر.

   گاۋ جىخۇا: نوقۇل ھالدا تاپشۇرۇق‹‹بېرىش›› ياكى ‹‹بەرمەسلىك›› مەسىلىسى بىلەنلا ئوقۇغۇچىلارنىڭ ئۆگىنىش ئەھۋالىغا مېخانىكىلىق باھا بەرگىلى بولمايدۇ. ئوقۇغۇچىلارنىڭ مۇۋاپىق يازما تاپشۇرۇق ئىشلىشىنىڭ ئۆزئالدىغا توغرا تەرىپى بار، ئوقۇتقۇچىلار دەرسخانىدىكى ئەھۋالغا قاراپ، تاپشۇرۇقنىڭ مىقدارىنى مۇۋاپىقلاشتۇرغاندا، بۇنداق تاپشۇرۇق ئوقۇغۇچىلانىڭ ئۆگىنىشىنى پۇختىلاش رولىنى ئوينايدۇ. مېنىڭچە، ئوقۇغۇچىلارنىڭ يۈكىنى يېنىكلىتىشنىڭ تۈپ مەنبەسى تاپشۇرۇق بولماستىن، بەلكى ئوقۇتقۇچىنىڭ ئوقۇتۇش قارىشى ۋە دەرسخانىدىكى ئوقۇتۇش ئۈنۈمى بىلەن مۇناسىۋەتلىك.

   سۈي شۆخۇڭ: ئوقۇتقۇچى ئورۇنلاشتۇرغان يازما تاپشۇرۇق ئوقۇتۇش مەزمۇنى ۋە ئوقۇغۇچىنىڭ ئەھۋالى بىلەن زىچ مۇناسىۋەتلىك بولۇشى، ئىلمىي ئورۇنلاشتۇرۇلۇشى كېرەك. يازما تاپشۇرۇقنى خالىغانچە ئورۇنلاشتۇرۇش ئوقۇغۇچىنىڭ غۇرۇرىغا ھۆرمەت قىلمىغانلىق ۋە ھاياتلىقنى قەدىرلىمىگەنلىك بولىدۇ، يازما تاپشۇرۇقنى كۆپ ئورۇنلاشتۇرۇش ئوقۇتقۇچىنىڭ ئۆزىنىڭ دەرسخانىدىكى ئوقۇتۇشىغا ئىشەنمەسلىكى بولۇپ ھېسابلىنىدۇ. بەزى ئوقۇتقۇچىلار دەرسخانىدا ئۆتمەكچى بولغان دەرسىنى كۆڭۈل قويۇپ پىلانلىمىغان، دەرس ئۆتۈش ئۇسۇلى جانسىز، ئوقۇتۇش ئۈنۈمى تۆۋەن بولغاچقا، كۆپلەپ تاپشۇرۇق بېرىش ئارقىلىق ئۆزىدىكى يېتەرسىزلىكنى تولۇقلىماقچى بولىدۇ. مەلۇم مەنىدىن ئېيتقاندا ئوقۇغۇچىلارنىڭ تاپشۇرۇق يۈكىنىڭ قانچىلىك بولۇشى ئوقۇتقۇچىنىڭ مەسئۇلىيەتچانلىقى ۋە ئوقۇتۇش سەۋىيەسىنى ئەكس ئەتتۈرۈپ بېرىدۇ.

   يازما تاپشۇرۇق بېرىش – بەرمەسلىكتە يەنىلا ‹‹يوتقانغا قاراپ پۇت سۇنغان››تۈزۈك

   مۇخبىر: بەزى ئاتا – ئانىلارنىڭ يازما تاپشۇرۇق ئورۇنلاشتۇرماسلىققا گۇمانىي نەزەر بىلەن قارىشىنى چۈشىنىشكە بولىدۇ. پەرزەنتلەر كەلگۈسىدىكى كۈچلۈك رىقابەتكە ئۆگەنگىنىنىڭ كۆپلۈكى، بىلىم ئاساسىنىڭ پۇختىلىقى بىلەن تاقابىل تۇرىدۇ. ئۇنداقتا، ‹‹يازما تاپشۇرۇق بېرىلمەيدىغان سىنىپ››نى قانداق يولغا قويغاندا ۋە داۋاملاشتۇرغاندا، بۇ سىنىپنى تاللىغان ئاتا – ئانىلارنىڭ كۆڭلىنى تىنجىتقىلى بولىدۇ؟

   گاۋ جىخۇا: ‹‹يازما تاپشۇرۇق بېرىلمەيدىغان سىنىپ›› ئوقۇغۇچىلارنىڭ ئۆگىنىش ئەھۋالىنىڭ تەكشى بولۇشىنى تەلەپ قىلىدۇ. ئەگەر بەزى ئوقۇغۇچىلار دەرسكە يېتىشەلمىسە، دەرستىن كېيىن تاپشۇرۇق ئورۇنلاشتۇرۇلمىسا، بۇ سىنىپ ئاتا – ئانىلارنى خاتىرجەم قىلالمايدۇ. شۇڭلاشقا ‹‹يازما تاپشۇرۇق بېرىلمەيدىغان سىنىپ››قا كىرگەن ئوقۇغۇچىلار مۇنداق ئۈچ تۈرلۈك شەرتنى ھازىرلىشى كېرەك: بىرىنچى، دەرسخانىدا ئۆتۈلگەن دەرس مەزمۇنىنى ئاساسەن ئىگىلىيەلىشى، ئىككىنچى، ئائىلىدە كۆپرەك يېتەكلىنىشى، ئوقۇغۇچىنىڭ دەرستىن كېيىنكى قىزىقىشىنى يېتىلدۈرۈشكە ئەھمىيەت بېرىلىشى، ئۇزاق مۇددەت داۋاملاشتۇرۇشقا كاپالەتلىك قىلىنىشى، قىيىنچىلىققا يولۇققاندا قانداق ھەل قىلىش قاتارلىق مەسىلىلەردە ئاتا – ئانىلار كۆپرەك يېتەكلىشى، ئوقۇغۇچىدا مۇستەقىللىقنى يېتىلدۈرۈشكە ياردەم بېرىلىشى كېرەك. ئۈچىنچى، ئائىلىدە پەرزەنتكە ئەمەلىيەتتە كۆرسىتىشىگە ۋە پائالىيەت قىلىشىغا كۆپرەك ۋاقىت ئاجرىتىلىپ بېرىلىشى كېرەك.

   سۈي شۆخۇڭ: ‹‹يۈكنى يېنىكلىتىش، ئەمما سۈپەتكە كاپالەتلىك قىلىش››نىشانىغا يەتكەندە ئاتا – ئانىلارنىڭ چۈشىنىشى ۋە قوللىشىغا ئېرىشكىلى بولىدۇ، بۇنىڭدا دەرسخانا ئوقۇتۇش ئۈنۈمىنى تىرىشىپ ئۆستۈرۈش ئاچقۇچ ھېسابلىنىدۇ. دەرسخانا ئوقۇتۇش سۈپىتىنى ئۆستۈرگەندىلا، تاپشۇرۇق ئورۇنلاشتۇرماسلىققا ياكى ئاز ئورۇنلاشتۇرۇشقا بولىدۇ. شۇڭا، بىز يۈكنى يېنىكلىتىش مەسىلىسىنى ئەخلاق نۇقتىسىدىن تونۇشىمىز، ئوقۇتۇش تەتقىقاتى نۇقتىسىدىن تۇرۇپ يۈكنى يېنىكلىتىش مەسىلىسىنى ھەل قىلىشىمىز، شۇ ئارقىلىق ئوقۇغۇچىنىڭ دەرسخانىدىكى ئۆگىنىش ئۈنۈمىنى ۋە دەرستىن سىرتقى پائالىيىتىنىڭ سۈپىتىنى ئۆستۈرۈشىمىز كېرەك.
