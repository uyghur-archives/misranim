                                       1. مەخپىيەتلىكىڭىز بارمۇ ؟


      مىسرانىم ئەپەندىم مۇنبەرگە بىر نەچچە مۇنبەر باشقۇرغۇچىلىرىنى تاللاپ بولغاندىن كىيىن ، ئۆزىگە بىر شوپۇر ئالماقچى بولۇپ sazguri  , aptap104 , BAHADIERI  دىن ئۈچتىن بىرىنى تاللىماقچى بوپتۇ - دە ، بۇ ئۈچىنى تاماققا تەكلىپ قىپتۇ ، ھەم ھاراق قۇيۇپ بىرىپتۇ ،  
          aptap104 : << شوپۇر دىگەن ھاراق ئىچسە بولمايدۇ  ، بۇ دىگەن يۇقۇرنىڭ بۇيرىقى ، شۇڭا مەن ھەرگىز ئىچمەيمەن >> دەپ ئويلاپ ھاراق ئىچمەپتۇ ،     BAHADIERI : << باشلىق ئىچ دىگەندىكىن چوقۇم ئىچىش كىرەك >> دەپ ئويلاپ بىر تىنىقتىلا كۆتۈرۋىتىپتۇ ، sazguri : << نىمە گەپ بولسا ئىچىپ بولغاندىن كىيىن دىمەيمۇ >> دەپ ئىچىۋىتىپتۇ  ۋە بىر دەمدىلا تەڭشىلىپ قاپتۇ ، شۇنىڭ بىلەن شوپۇر تاللاشنىڭ 1-باسقۇچى ئاخىرلىشىپتۇ ، 2-باسقۇچتا مىسرانىم ئەپەندىم ئۇلاردىن :
       _ سىزنىڭ مەخپىيەتلىكىڭىز بارمۇ ؟ _ دىگەن سۇئالنى سوراپتۇ ،
      _ دوكلات باشلىق ، مىنىڭ مەخپىيەتلىگىم يوق _ دەپ جاۋاپ بىرىپتۇ  aptap104 .
     _ نىمە ياسالمىلىق بۇ ، مىنىڭ مەخپىيەتلىگىم با ر باشلىق _ دەپتۇ  BAHADIERI دەلدەڭشىگەن پىتى .
   sazguri مەسىتلىكتىن ھىچنىمە دىيەلمەپتۇ ، شۇنىڭ بىلەن مىسرانىم ئەپەندىم << ئۆلگۈدەك مەس بولسىمۇ ھىچنىمە دىمەسلىك شوپۇرلۇقنىڭ ئەڭ يۇقۇرى پەللىسى >> دەپ sazguri نى شوپۇرلۇققا تاللاپتۇ ، ئەتىسى sazguri مۇنبەر ئىشخانىسىغا خۇشال ھالدا كىرىپتۇ - دە :
      _ ئىمتاھان كۈنى ئاتىدارچىلىق قىلغىنىڭىزغا كۆپ رەخمەت ، ئۇ كۈنى ھاراقنى كۆپ ئىچىپ ، يۇتقۇنچاق ياللۇغۇم قوزغىلىپ قالغاچقا سۇئالىڭىزغا جاۋاپ بىرەلمىگەن ئىدىم ، ئەمدى جاۋاپ بىرەي ، مىنىڭ نۇرغۇن مەخپىيەتلىگىم بار ، باشقىلارنىڭمۇ نۇرغۇن مەخپىيەتلىگىنى بىلىمەن ، بۇلانى سىزگە كىيىنچە ئالدىرماي ئولتۇرۇپ دەپبىرەي _ دەپتۇ ، مىسرانىم ئەپەندىم كۆزلىرىنى يوغان ئاچقان پىتى قاراپ قاپتۇ ،


                                               2 .   چىقىمنى تىجەش

     كارۋان ئايالىنىڭ تۇققانلىقىنى ئاڭلاپ ، جىددىلەشكەن ھالدا دوختۇرخانىنىڭ تۇغۇت بولۈمىگە كەپتۇ ، ئۇنىڭغا دوختۇر شاتئاي ئۇچراپتۇ - دە :
        _ ھەي ، بۇ دىگەن تۇغۇت بولۈمى ، ئەرلەر كىرىشكە بولمايدۇ _ دەپتۇ ،
        _ بۇنى بىلىمەن _ دەپتۇ كارۋان _ مەن بەش يىلدىن بىرى ئاران پەرزەنىتلىك بولدۇم ، ئايالىمنىڭ تۇققان خەۋرىنى قىينى ئانام تىلىفۇندا دىگەن ئىدى ، لىكىن قىز - ئوغۇللىقىنى دىمەيلا تىلىفۇننى قويۋەتتى شۇڭا ئالدىراپ كىلىشىم ، بۇنى ئاڭلىغان دختۇر شاتئاي كارۋاننىڭ جىددىلىشكەن چىرايىغا تازا بىر قارۋەتكەندىن كىيىن :
       _ سىزدە يۈرەك كىسىلى ، قان بىسىمى دىگەندەك كىسەەللىك ئالامەتلىرى بامۇ ؟ _ دەپ سوراپتۇ ،
       _ بار _ دەپتۇ كارۋان شاتئايغا _ بىرە ئىشقا ھاياجانلانسام ھەتتا ھۇشۇمدىن كىتىمەن ،
       _ ھىم ، ئۇنداق بولسا دوختۇرخانىنىڭ سىرتىغا چىقىپ پاراڭلىشايلى _ دەپتۇ شاتئاي كارۋاننى سىرىتقا باشلاپ چىقىپ _ ئايالىڭىز ئوغۇل تۇغدى ، بالا بەكمۇ ساغلام _ دەپتۇ ،
          _ بۇ خوش - خەۋەرنى ئاتايتەن سىرىتقا ئەپ چىقىپ دىگۈچە ، بايام دوختۇرخانىدىلا دىسىڭىز بولمامدۇ ؟ _ دەپتۇ كارۋان ھەيران بولغان ھالدا شاتئايغا قاراپ
           _ بۇنداق قىلىشىمدىكى سەۋەپ ، ئەگەر سىز ھاياجانلىنىپ ھۇشىڭىزدىن كىتىپ قالسىڭىز ،سەۋەپ - نەتىجە مۇناسىۋىتى شەكىللىنىپ ، تۆلەم مەسلىسى ئوتتۇرغا چىقىدۇ ، سىرىتتا ھۇشىڭىزدىن كەتسىڭىز ، بۇ مەسىلە كۆرۈلمەيدۇ ، چۈشەندىڭىزمۇ ؟ مانا بۇ دوختۇرخانىمىزنىڭ چىقىمىنى ئازايتىشنىڭ بىر خىل تەدبىرى _ دەپتۇ شاتئاي كارۋانغا .


                                                3 .    ئىقرار قىلىش


          بىر كۈنى nazurgum بىلەن 国民党  ئەر - ئايال ئىككىسى  بىر ئولتۇرۇشتىن قايتىشىدا كەچ قاپتۇ ، يولدا چارلاشقا چىققان  قاتناش ساقچىسى ئۇلارنى توسۇپتۇ - دە :
        _ نىمىشقا ماشىنىنىڭ چىرىغىنى ياقماي يول يۈرۈيسىز _ دەپتۇ ،
        _ كەچۈرۈڭ باشلىق ، ماشىنام تۈنۈگۈن سوقۇلۇپ كىتىپ ئالدى چىرىغى بۇزۇلۇپ كەتكەن _ دەپ جاۋاپ بىرىپتۇ 国民党 ساقچىغا ،
        _ پىراۋىڭىزنى چىقىرىڭ _ دەپتۇ ساقچى يەنە ،
        _ پىراۋام تىخى چىقمىغان _ دەپتۇ 国民党
        _ ياخشىغۇ ، بىراقلا ئىككى تۈرلۈك قاتناش قائىدىسىگە خىلاپلىق قىپسىز ، سىزنى قاماپ قويسام بولغۇدەك ، بۇنى ئاڭلىغان 国民党 نىڭ ئايالى nazurgum دەرھاللا :
        _ باشلىق ، باشلىق ، توختاپ تۇرۇڭ ، ئۇنىڭ گەپلىرىگە ھەرگىز ئىشەنمەڭ ، بايا ئۇ كۆپرەك ئىچىپ قويغان ، شۇڭا قالايمىقان سۆزلەۋاتىدۇ _ دىگۈدەك .


                                                   4 .  مىھمان دوسىت ئەر - خوتۇن

       بىر كۈنى كەچتە  hokumran  بىلەن musilina    ئەر - ئايال ئەمدى ياتاي دەپ تۇرشىغا يولدىن ئىزىپ قالغان بىر يولۇچى بىر كىچە پانالىۋىلىشنى ئىيتىپتۇ ، hokumran  بىلەن musilina  بەكمۇ  مىھمان دوسىت بولغاچقا بۇ ناتونۇش مىھماننى قىزغىن كۈتۈپتۇ ،  بۇ ئەر - ئاياللارنىڭ 6 بالىسى بار ئىكەن ،  musilina ئاۋال كىچىك ئىككى بالىسىنى كارۋاتتا ئۇخلىتىپتۇ ، ئۇلار ئۇخلاپ بولغاندىن كىيىن ، ئۇلارنى كارۋاتتىن كۆتۈرۈپ ئىلىپ ، پولدا ياتقۇزۇپتۇ -دە يەنە ئىككى بالىسىنى كارۋاتتا ئۇخلۇتۇپتۇ ، مۇشۇ يوسۇندا 6 بالىسىنى ئۇخلۇتۇپ بولغاندىن كىيىن ، مىھماننى كارۋاتتا يىتىپ ئارام ئىلىشقا تەكلىپ قىپتۇ ، مىھمان بۇلارنىڭ قىزغىن مۇئامىلىسىدىن ئىنتايىن خۇرسەن بوپتۇ  ھەم كارۋاتقا چىقىپ ئارام ئاپتۇ ، ئەپسۇس ، يولۇچى ئەتتىگىنى ئويغۇنۇپ ئوزىنىڭمۇ بالىلار بىلەن پولدا ئۇخلىغىنىنى بىلىپتۇ ، كارۋاتتا بولسا ئەر - ئايال ئىككىسى تاتلىق ئۇخلاۋاتقان ئىكەن .


                                            5 .  لۇغىتىمدە يوق

     بىر كۈنى ئوقۇتقۇچى دەرىس ۋاختىدا ئوقۇغۇچىلارغا كايىپ :
      _ ساۋاقداشلار ، چوقۇم نەتىجە قازىنىشىڭلار كىرەك ، مىنىڭ لوغىتىمدە  << مەغلۇپ بولۇش >> دىگەن سوز يوق _ دىيىشىگە   ئوقۇغۇچى rizwangul0995   دەرھال :
        _  مۇئەللىم ، مىنىڭ لوغىتىمنى ئىشلىتىپ تۇرۇڭ ئەمىسە _ دىگۈدەك

                                      6 .    قورقۇنۇشلۇق  خەۋەر

       uzumtal  ئەتتىگەندە ئىشخانىسىغا كىرىپ تۇرشىغا ئادۇۋكاتى كىرىپ دەپتۇ :
         _ سىز ئالدى بىلەن يامان خەۋەر ئاڭلامسىز ياكى قورقۇنۇشلۇق خەۋەرمۇ ؟
         _ ئالدى بىلەن يامان خەۋەر ئاڭلاي _ دەپتۇ uzumtal ئادۇۋكاتىغا قاراپ
         _ ئايالىڭىز قىممىتى نەچچە  ئونمىڭ يۈئەنلىك رەسىم تىپىۋاپتۇ ،
          _ قورقۇنۇشلۇق خەۋەرچۇ ؟
          _ ئايالىڭىز تىپىۋالغان ئۇ رەسىم سىزنىڭ كاتىۋىڭىز بىلەن ئوينىغىلى چىققاندا چۈشكەن رەسىم ئىكەن .
