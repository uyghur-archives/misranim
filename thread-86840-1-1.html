مەمۇرلۇق ئىمتىھانىنىڭ «بايان ۋە مۇھاكىمە» قىسمىغا تەييارلىق قىلىش ئۇسۇلى




بايان ۋە مۇھاكىمە ئىمتىھان بەرگۈچىنىڭ يالغۇز يېزىقچىلىق ئىقتىدارى ياكى سىياسەت ئېقىم مەسىلىلىرىنى چۈشىنىش ئىقتىدارىنى سىناش بىلەنلا چەكلەنمەيدۇ. ئۇ مەمۇرلۇق ئىمتىھانى بەرگۈچىلەرنىڭ يەتتە خىل ئىقتىدارىنى سىنايدۇ: ئوقۇپ چۈشىنىش ئىقتىدارى، تەھلىل قىلىش ئىقتىدارى، مەسىلىنى ئوتتۇرىغا قويۇش ۋە ھەل قىلىش ئىقتىدارى، تىل بىلەن ئىپادىلەش ئىقتىدارى، يېزىقچىلىق ئىقتىدارى، ئېقىم مەسىلىلىرىنى قوللىنىش ئىقتىدارى ۋە مەمۇرىي باشقۇرۇش ئىقتىدارىدىن ئىبارەت.
     بايان ۋە مۇھاكىمە سوئاللىرىنىڭ مۇقىم بېكىتىلگەن جاۋابى يوق. كۆپىنچە ھاللاردا سىياسىي، ئىقتىساد، قانۇن، ۋە جەمئىيەت باشقۇرۇشقا ئائىت مەسىلىلەر ئارقا كۆرۈنۈش قىلىنغان بولۇپ، بەزىلىرىگە ھۆكۈم چىقىرىلىپ بولۇنغان، بەزىلىرى بولسا مۇھاكىمە قىلىنىش باسقۇچىدا تۇرۇۋاتقان بولۇپ، ئىمتىھان بەرگۈچىلەردىن ھەق قىلىش لايىھىسىنى ئوتتۇرىغا قويۇش تەلەپ قىلىنىدۇ.
    ئۆزۈمنىڭ ۋە ئەتراپىمدىكى ئىلگىرى ئىمتىھانغا قاتنىشىپ باققان كىشىلەرنىڭ تەجرىبىلىرىگە ئاساسلانغاندا بىلىم قۇرۇلمىسى قانچىكى مۇكەممەل، ئۇنىۋېرسال بىلىمى قانچىكى كۆپ بولسا، ئىمتىھانغا تاقابىل تۇرۇش شۇنچە ئاسان بولىدۇ. ئالىي مەكتەپتە ئۆزىنىڭ كەسپىنى ناھايىتى پۇختا ئىگىلىگەن، باشقا جەھەتتىكى بىلىملەرنى ئىگىلەشكە سەل قارىغان بولسىڭىز ئۇنداقتا بايان ۋە مۇھاكىمە ئىمتىھان سوئالىنى ياخشى ئىشلەپ كېتەلىشىڭىز تەسكە توختايدۇ. بەزىلەر ئۆز كەسپىنى ئانچە پۇختا ئۆگەنمىگەن بولسىمۇ ئەمما ئانچە مۇنچە گېزىت ئوقۇپ تۇرىدۇ، خەۋەر كۆرۈپ تۇرىدۇ، ئەتراپىدا يۈز بېرىۋاتقان قىزىق نۇقتا مەسىلىلىرىگە نىسبەتەن مۇلاھىزە ئېلىپ بېرىشنى ياخشى كۆرىدۇ. بۇنداق كىشلەر ئىمتىھانغا ياخشى تاقابىل تۇرالايدۇ دەپ ئېيىتىشقا بولىدۇ. ئالاھىدە ئىش ئورنى ئوقۇتقۇچىلىقى ئىمتىھانى بىلەن بىلەن مەمۇرلۇق ئىمتىھانىنىڭ پەرقى دەل مۇشۇ يەردە. كەسپكە قانچە پىششىق بولسىڭىز ئوقۇتقۇچىلىق ئىمتىھانىدىن ئۆتۈش نىسبىتىڭىز يۇقىرى بولىدۇ، ئۇنىۋېرسال بىلىمىڭىز قانچە مول بولسا مەمۇرلۇق ئىمتىھانىغا تاقابىل تۇرۇش ئىقتىدارىڭىز شۇنچە كۈچلۈك بولىدۇ.
     يۇقىرىدا دېگەنلىرىمىز ھەرگىزمۇ مۇتلەقلەشتۈرۈپ ئېيتىلغان ئەمەس. كەسپىي بىلىمىمىز ياكى ئۇنىۋېرسال بىلىمىمىزنىڭ قانچىلىك بولۇشىدىن قەتئىينەزەر ئىمتىھان ئالدىدىكى تەييارلىقمۇ ئىنتايىن مۇھىم رول ئوينايدۇ. ئىمتىھان ۋاقتى يېتىپ كېلىشتىن بۇرۇن تۆۋەندىكى تەييارلىقلارنى ياخشى ياخشى ئىشلەش كېرەك.
      ئالدى بىلەن پايدىلىنىش ماتېرىيالىدىكى ماقالە يېزىش قىسمىنى پىششىق ئوقۇش، ئاندىن كېيىن بۇ ئۇنى ئىككى ئۈچ قېتىم كۆچۈرۈپ يېزىش كېرەك. بۇنداق قىلىشتىكى مەقسەت ماقالىنى قانداق يازسا بولىدىغانلىقى توغرىسىدا دەسلەپكى چۈشەنچىگە ئىگە بولۇشتىن ئىبارەت.
     ئىمتىھان بېرىشتىن بىر ئاي بۇرۇن ھەركۈنى بىر قېتىم مۇھاكىمە ۋە بايان تەقلىدىي ئىمتىھان سوئالىنى تولۇق بىر قېتىم ئىشلەپ چىقىش كېرەك. ھېچ بولمىغاندا ئىككى كۈندە بىر قېتىم بولسىمۇ بولىدۇ. بۇ ئارقىلىق ئۆزىنىڭ قانچىلىك ئۆگەنگەنلىكىنى، يېزىقچىلىق ئىقتىدارىنىڭ قانچىلىك ئىكەنلىكىگە قارىتا كۆڭلىدە سان تۇرغۇزغىلى بولىدۇ.
ئىمتىھاندىن بۇرۇن يېقىنقى مەزگىلدىكى شىنجاڭ گېزىتى ۋە ئۈرۈمچى كەچلىك گېزىتىنى بىر قېتىم كۆرۈپ چىقىش ئاقىلىق نۆۋەتتە يۈز بېرىۋاتقان قىزىق نۇقتا مەسىلىلىرى، ھۆكۈمەت تەرەپ دائىم ئىشلىتىدىغان سۆز-جۈملىلەرنى بىر قۇر ئىگىلەش كېرەك. چۈنكى ماقالە يازغاندا ئىشلىتىلگەن سۆز جۈملىلەر قانچىكى ئۆلچەملىك، ھۆكۈمەت تەرەپ تىلىغا قانچە يىقىن بولسا، يۇقىرى نومۇر ئېلىش نىسبىتى شۇنچە يۇقىرى بولىدۇ.
     توردىن يېقىنقى مەزگىلدىكى قىزىق نۇقتا، قىيىن نۇقتا مەسىلىلىرىنى تېپىپ بىر قېتىم كۆرۈپ چىقىش كېرەك. بىر قانچە يىلنىڭ ئالدىدا ئۆي باھاسى، دېھقانلارنىڭ يەردىن ئايرىلىپ قېلىش مەسىلىسى، مائارىپ ئادىللىقى، يېمەكلىك بىخەتەرلىكى، ئىشقا ئورۇنلىشىش قاتارلىقلار بىر قەدەر قىزىق نۇقتىلار ئىدى. يېىقىدىن بۇيان قاتناش تىرانسىپورت بىخەتەرلىكى جۈملىدىن مەكتەپ ئاپتوموبىللىرى بىخەتەرلىك مەسىلىسى، داۋالىنىش باھاسى، ئىجتىمائىي كاپالەت، مال باھاسىنىڭ ئۆسۈشى، ئىستېمالچىلارنىڭ ھوقۇق مەنپەئىتىنى قوغداش قاتارلىقلار نىسبەتەن ئاممىنىڭ دىققەت ئېتىبارىنى قوزغىدى. ئىشقىلىپ بۇ مەسىلىلەر سىزگە قانچىكى تونۇش بولسا، بايان ۋە مۇھاكىمە سوئالىنى ئىشلىشىڭىز شۇنچە ئاسان بولىدۇ.
      قەغەزدىكى بوشلۇقنى توغرا مۆلچەرلەش كېرەك. ئۇيغۇرچە ئىمتىھاندا خەت سانىغا ئەمەس بەلكى جاۋابنىڭ قەغەز يۈزىدە  ئىگىلىگەن ئورنىغا تەلەپ قويۇلىدۇ. شۇڭا ماقالىنىڭ ھەر قايسى بۆلەكلىرىگە نىسبەتەن كۆڭۈلدە سان تۇرغۇزۇپ، ئاندىن قەلەم تەۋرىتىش كېرەك. بەك كۆپ ئورۇن ئېشىپ قالسا ئىمتىھان تەلىپىگە يەتكىلى بولمايدۇ. ماقالە ئاخىرلىشىش قىسمىدا ئورۇن يېتىشمەسلىك سەۋەبىدىن چالا يېزىلسىمۇ ئوخشاشلا يۇقىرى نومۇر ئالغىلى بولمايدۇ.
ماقالىدە ئوتتۇرىغا قويۇلغان كۆز قاراش ۋە پىكىرلەرنى زىيادە مۇتلەقلەشتۈرۈۋېتىشكە بولمايدۇ. دىئالېكتىكىلىق ئۇسۇل بويىچە ئۆز پىكىرىنى ئوتتۇرىغا قويۇش كېرەك. مەسىلەن، 2010-يىلدىكى بايان ۋە مۇھاكىمە سوئالىدا «قانۇن ئىجىرا قىلىشتا قاتتىق بولۇش كېرەكمۇ ياكى يۇمشاق قول بولۇش كېرەكمۇ» دېگەن سوئالغا قارىتا نوقۇل قاتتىق قول بولۇش كېرەك ياكى نوقۇل ھالدا يۇمشاق قول بولۇش كېرەك دەپ ئوتتۇرىغا قويۇلسا، مېنىڭ شەخسىي پىكىرىم شۇكى ، تازا مۇۋاپىق ئەمەس. قانۇن ئىجىرا قىلىشتا قاتتىق قول بولۇش زۆرۈر، شۇنىڭ بىلەن بىرگە يۇمشاق قوللۇق قىلىشنى قوشۇمچە قىلىش كېرەك دېيىلسە بىر قەدەر مۇۋاپىق.(دىققەت: بۇ مېنىڭ شەخسىي كۆز قارىشىم، پۈتۈنلەي توغرا دېيەلمەيمەن.)  
خەتنى ئىمكانقەدەر چىرايلىق، تىنىش بەلگىلىرى تولۇق ھەم مۇۋاپىق، ئابزاس بۆلەكلەرگە ئايرىلىشى ئېنىق بولۇش كېرەك. ئىمتىھاندا خەتنىڭ چىرايلىق بولۇش بولماسلىقىغا باھا بېرىلىدىغان ياكى بېرىلمەيدىغانلىقىنى بىلمەيمەن.  ئەگەر خەت زىيادە سەت بولۇپ، نېمە مەزمۇن ئىپادىلىگەنلىكىنى بىلگىلى بولمىسا ئېنىقلا نومۇر تارتىلىدۇ. مەيلى خەنزۇچە بولسۇن ياكى ئۇيغۇرچە بولسۇن. ھەممە ئادەم گۈزەللىككە ئىنتىلىدۇ، چىرايلىق خەتمۇ كىشىنىڭ زوقىنى قوزغاپ كۆرگۈچىدە بۇ خەتنى يازغۇچىغا نىسبەتەن بىر خىل گۈزەللىك سېزىمى ھاسىل قىلىدۇ. مېنىڭ پەرىزىمچە، بايان ۋە مۇھاكىمە سوئالىنى بىر قانچە تەكشۈرگۈچى ئايرىم-ئايرىم تەكشۈرۈپ باھالاپ ئاخىرىدا ئۇلار قويغان نومۇردىن ئوتتۇرىچە نومۇرنى ھېسابلاب چىقىرىدۇ. ئەگەر ئۇلارنىڭ ھەممىسىدىلا بۇ ئىمتىھان بەرگۈچىنىڭ خېتىگە قاراپ بىر ئىللىق سېزىم ھاسىل بولسا، بۇنىڭ ئىمتىھان بارگۈچى ئۈچۈن پايدىسى باركى زىيىنى يوق.   
    يۇقارقىلار مېنىڭ پەقەتلا شەخسىي كۆز قارىشىم ۋە ئۆزەمنىڭ باشتىن ئۆتكۈزگەن تەجرىبىلىرىم، پۈتۈنلەي توغرا دەپ ئېيتالمايمەن، خاتا بولسا كۆپچىلىكنىڭ پىكىر بېرىپ تۈزىتىشىنى قارشى ئالىمەن.
     بايان ۋە مۇھاكىمە ئىمتىھان سوئالىدا پايدىلىنىش ئۆرنىكىنى ئايرىم يوللىدىم. ۋاقىت يار بەرسە مەمۇرىي ئىقتىدارنى سىناش ئىمتىھانىغا تەييارلىق قىلىش ئۇسۇللىرىنى يوللاپ كۆپچىلىكنىڭ پايدىلىنىشغا سۇنىمەن.
