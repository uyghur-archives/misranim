مانا ھەش پەش دىگۈچە سىنىڭ بۇ دۇنيادىن كىتىپ قالغىنىڭغا نەچچە يىلار بولدى.  
سەن كەتتىڭ. مىنىڭ ئەڭ مۇھىم بولغان سىنى سۆيدىغان يۈرۈكۈمنىمۇ بىللە ئېلىپ كەتتىڭ.                                
گەرچە مەن تىرىك بولساممۇ ئاللا بۇرۇن ئۆلگەن ئادەم. چۇنكى مەندە پەقەت مۇشۇ قۇرۇق ئۇستىخانلا قالدى چۇنكى مىنىڭ بۇ قۇرۇق بەدىنىم جىسمىم بىر كۈنى تۇپراقنىڭ ئاستىدا توپىغا ئايلىنىدۇ.                                      

ھەركۈنى ئاخشىمى دېرىزە تۈۋىدە مۇڭلانغان ھالدا سەن بىلەن ئۆتكەن شۇ كۇنلەرنى ئەسلەيمەن.                    
سىنى ئويلىمىغان ھەر بىر كۈن ماڭا ھارام. چۈشلۈرۈمدە پەقەت سىنى ئويلايمەن.                                       
مۇھەببىتىمزنىڭ مازىرىنى يوقلاش ماڭا ئادەت بۇلۇپ قالدى.  
ئىسڭدە بارمۇ بىز تۇنجى قېتىم ئۇچراشقان ۋاقىتتا سەن ماڭا قارىغان ۋاقتىتا مەن قىزىرىپ كەتكەنتىم

نىمىشقا سەن قارىساڭ قىزىرىپ كىتىدىغانلىقىمنىڭ سەۋەبىنى بەكمۇ كىچىكىپ ھىس قىلدىم.ھەرقېتىم كۈزلىرىمىز ئۇچراشقاندا ئىككىلىمىز قىزىرىپ كىتاتتۇق. شۇ چاغلارنى ئويلىسام ئۆپكەم ئۆرۈلۈپ يىغلىغۇم كىلىدۇ. سەن ھازىر مىنىڭ قايسى ھالەتتە چۈشۈپ قالغانلىقىمنى كۈرۈپ تۇرۋاتامسەن؟ بىر چاغلاردىكى كۆزلىرى پاقىراپ تۇردىغان چىرايى نۇرلۇق مۇھەممەت يوق. ئۇنىڭ ئورنىغا كۆزلىرى سوغوق غەمكىن بىچارە بىر ئادەم ئالماشتى.



سەن بىلەن قىلىشقان ھەربىر پاراڭلىرىمىز ھەربىر ۋەدىلىرىمىز ھەربىر ئېغىز سۆزۈڭ مىنىڭ قۇلاق تۈۋۈمدە. بىلەمسەن بىرقېتىم تۇيۇقسىز توردا ساڭا شۇنچىلىك ئوخشايدىغان بىرقىزنىڭ سۈرۈتىنى كۈرۈپ ئۆزەمنى تۇتالمدىىم. خۇددى سەن تۇيۇقسىزلا قايتىپ كىلدىغاندەك شۇنداق تۇيغۇدا بۇلۇپ قالدىم. لىكىن سىنىڭ قايتىپ كىلىشىڭ مۇمكىنمۇ؟ سەن ئاشۇ قارا تۇپراق ئاستىدا بەلكىم مىنىڭ يۈرەك سۆزلىرىمنى تىڭشاپ تۇرۋاتقانسەن.



مەن مەڭگۇ مۇشۇنداق ئۆتىمەن. مەڭگۇ سىنى ئارمان قىلىپ ئۆتىمەن.
ئىشقىڭدا مەجنۇن بۇلۇپ مازارلاردا ياتىمەن. پەقەت ئەنە ئاشۇنداق يەرلەردىلا سەن بىلەن ئۇچۇرشالىشىم مۇمكىن.
چۇنكى يۈرۈكۈم ھەر بىرقېتىم سوققاندا سىنىڭ ئىسمىڭنى زىكرى قىلىدۇ گۈلۈم.
ساڭا قىلغان ھەربىر ئېىغىز قاتتىق گىپىم ئۇچۇن ئۆزەمگە ھىساپ بېرۋاتىمەن.





بىزگە بۇ ئالەمدە مۇھەببەت ۋىسالى نىسىپ بولماپتۇ گۈلۈم.

مەن دۇنياغا مىڭ قېتىم كەلسەممۇ مۇھەببىتىم پەقەت ساڭىلا مەنسۇپ.
بۇ يالغان دۇنيادا كۆزۇم پەقەت سىنىلا يۈرۈكۈم پەقەت سىنى سۆيدى گۈلۈم.


كۈزۈمنىڭ نۇرى ئىدىڭ. ئەمدى سەن يوق. كۈزۈمگە قاراڭغۇلۇق تىقىلدى.
سىنى سۆيىمەن دىگەن گىپىڭنى ئاڭلىغان ۋاقىتتا شۇنچىلىك خوشال بولدۇم.
چۇنكى بۇ يالغان دۇنيادا يۈزۈم ھىچقاچان شۇ كۇندىكىدەك كۈلۈپ باقمىدى..

           ياخشى كۆرۈش بىلەن سۆيۈشنىڭ پەرقى


ئەگەر بىرسىنى ياخشى كۆرسىڭىز، ئۇنىڭ بىلەن بىرگە بولغاندا ئىنتايىن خۇشال بولىسىز.
ئەگەر بىرسىنى سۆيسىڭىز، ئۇنىڭ بىلەن بىرگە بولغاندا قەلبىڭىز چۈشۈنۈكسىز بىر ھىسسىياتقا ئەسىر بولىدۇ.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، سىز كەلگۈسىڭلار ئۈستىدە ئويلانمايسىز.
ئەگەر بىرسىنى سۆيسىڭىز، سىلەر دائىم بىرلىكتە ئەتەڭلارگە تەلپۈنىسىلەر.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، سىلەر بىرگە بولغاندا مەڭگۈ خۇشال بولىسىلەر.
ئەگەر بىرسىنى سۆيسىڭىز، سىز دائىم ياش تۆكىسىز.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، سىلەر ئۇزۇنغىچە كۆرۈشمىسەڭلار، سىز توساتتىن ئۇنى ئەسكە ئالىسىز.
ئەگەر بىرسىنى سۆيسىڭىز، سىلەر ئۇزۇنغىچە كۆرۈشمىسەڭلارمۇ، سىز كۈندە ئۇنى ياد ئېتىسىز.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، سىز ئۇنى ئەسلىگەندە، كۈلۈمسىرەپ قويىسىز.
ئەگەر بىرسىنى سۆيسىڭىز، سىز ئۇنى ئەسلىگەندە، كۆككە بېقىپ تۇرۇپ قالىسىز.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز،كۆپچىلىكنىڭ ھەممىسىنىڭ خۇشال بولىشىنى ئۈمىد قىلىسىز.
ئەگەر بىرسىنى سۆيسىڭىز، ئۇنىڭ تېخىمۇ خۇشال بولىشىنى ئۈمىد قىلىسىز.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، بۈگۈنگىلا ئېرىشىشنى ئويلايسىز.
ئەگەر بىرسىنى سۆيسىڭىز، ئۇنىڭ بىلەن مەڭگۈ ئايرىلماسلىقنى ئويلايسىز.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، ئۇنىڭ ئارتۇقچىلىقىنىلا كۆرىسىز.
ئەگەر بىرسىنى سۆيسىڭىز، ئۇنىڭ كەمچىلىكلىرىنىمۇ كەچۈرۋېتىسىز.
ياخشى كۆرۈش- بىر خىل روھى ھالەت، بىر خىل بىۋاستە سېزىم.
سۆيۈش- بىر خىل تۇيغۇ.
ياخشى كۆرۈشنى توختاتقىلى بولىدۇ.
سۆيۈش ھەرگىزمۇ توختاپ قالمايدۇ.
بىرسىنى ياخشى كۆرۈش تولىمۇ تەبىئى بولىدۇ.
بىرسىنى سۆيسىڭىز ئۆزىڭىزنى تولىمۇ خاتىرجەم ھىس قىلىسىز.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، بەزىدە ئۇنىڭ بىلەن بىرگە بولۇشنى ئارزۇ قىلىسىز.
ئەگەر بىرسىنى سۆيسىڭىز، بەزىدە ئۇنىڭ بىلەن بىرگە بولۇشتىن قورقۇپ قالىسىز.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، توختىماستىن ئۇنىڭ بىلەن تالاش- تارتىش قىلىسىز.
ئەگەر بىرسىنى سۆيسىڭىز،توختىماستىن ئۇنىڭ ئۈچۈن قۇربانلىق بېرىسىز، دائىم ئۇنى تېپىۋېلىشنى ئارزۇ قىلىسىز.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز،ھەمىشە ئۇنىڭ ئۈچۈن كۈلىسىز.
ئەگەر بىرسىنى سۆيسىڭىز، ھەمىشە ئۇنىڭ ئۈچۈن يىغلايسىز.
ياخشى كۆرۈش تولىمۇ ئاددى، ياخشى كۆرۈشكە ئەرزىيدۇ دەپ قارىسىڭىزلا، ياخشى كۆرىۋېرىسىز. بىراق سۆيۈش بىرخىل مۇرەككەپ ھىسسىيات، ئۇ سىزنى ياخشى كۆرۈشى مۈمكىن، بىراق سىزنى سۆيۈشى ناتايىن. بىرسىنى سۆيۈش ئۈچۈن چوقۇم ئۇنى  ئىنتايىن ياخشى كۆرگەن بولۇشى شەرت، ياخشى كۆرۈش بىلەن سۆيۈش ئوتتۇرسىدا پەقەت بىر قەدەملا ئارلىق بار، گەپ سىزنىڭ بۇ قەدەمنى بېسىشنى خالاش،  خالىماسلىقىڭىزدا.
ئەگەر بىرسىنى سۆيسىڭىز، ئۇ بار چاغدا، كۆز- ئالدىڭىزدا شۇ بىرلا ئادەم باردەك ھىس قىلىسىز، ئۇ كېتىپ قالغاندا بولسا، ئۇنىڭ سىماسى ھەر دائىم كۆز ئالدىڭىزدىن كەتمەيدۇ.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، يېرىم كېچىدە كىتاب ئوقۇۋېتىپ، تۇيۇقسىز ئۇنى ئەسكە ئېلىپ، ئۇ ھازىر نېمە ئىش قىلىۋاتقاندۇ دەپ ئويلايسىز، كۆڭلىڭىز بىر ئىللىققىنە سېزىمغا ئەسىر بولىدۇ، بىراق ئۇنىڭغا تەشەببۇسكارلىق بىلەن تېلىفۇن قىلمايسىز. بىر نەچچە مىنۇتتىن كېيىنلا ھەممىنى ئۇنتۇپ، كىتابتىكى ۋەقەلىك ئىچىگە كېرىپ كىتىسىز.
ئەگەر بىرسىنى سۆيسىڭىز، يالغۇز كېچىلەردە، قەلبىڭىز سېغىنىشقا تولغان ھالدا قولىڭىزغا كىتاب ئالىسىز، بىراق قانچە قىلىپمۇ دىققىتىڭىزنى يىغالمايسىز، ئۇ ھازىرمۇ سېمىنا قوشۇپ ئىشلەۋاتقانمىدۇ، كەچلىك تاماق يىگەنمىدۇ، مەن ئۇنى سېغىنغاندەك ئۇمۇ مېنى سېغىنغانمىدۇ دىگەن ئويلار كاللىڭىزنى چىرمىۋالغان بولىدۇ.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، مەسىللەرنى مۇزاكىرە قىلغاندا ئۇنىڭ بىلەن ھارغىچە تالىشىپمۇ، ئۇنىڭغا يول قويمايسىز، ئۇنىڭ ئالدىدا خۇددى كىرپىگە ئوخشاش ئۇتتۇرغىنىڭىزغا پەقەت تەن بەرمەيسىز، بىراق كۆڭلىڭىزدە ئۇنىڭ قابىليىتى ۋە ئىقتىدارىغا قايىل بولىسىز.
ئەگەر بىرسىنى سۆيسىڭىز، ھەر قانداق ئىشتا ئوي –خىياللىرىڭلارنىڭ بىردەك بولۇشىنى ئۈمىد قىلىسىز، ئۇنىڭ ئويلانمايلا قىلغان بىر جۈملە چاخچىقى تۈپەيلىدىن كۆڭلىڭىز يېرىم بولىدۇ ھەتتا ياش تۆكىسىز. ئۇنىڭ ئالدىدا، ساختىلىق قىلىشنى خالىمايسىز.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، سىرتقا چىققاندا ئۇنىڭغا ئۇچۇر يوللاپ، بۇ يەرنىڭ ھاۋاسىنىڭ ناھايىتى ياخشى ئىكەنلىگىنى ئېيتىپ قويۇپلا تېلىفۇننى ئېتىۋىتىسىز، سېرتتا بىر ھەپتە قانغۇچە ئويناپ، ئۇنىڭ ئالدىدا تۇيۇقسىز پەيدا بولۇپ ئۇنى ھەيران قالدۇرىسىز.
ئەگەر بىرسىنى سۆيسىڭىز، ھەردائىم، ھەرقانداق چاغدا ئۇنىڭ سىزگە ھەمراھ بولۇشىنى ئۈمىد قىلىسىز. ئۆزىڭىز يالغۇز سېرتلارغا چىقىپ، دېڭىز بويىغا بارغىنىڭىزدا، دېڭىز بويىدا تۇرۇپ ئۇنىڭغا تېلىفۇن قىلىپ، ئۇنىڭغا دېڭىز شاۋقۇنىنى ئاڭلىتىسىز، ياقا يۇرتلاردا كوچىدا كېتىۋېتىپ، توساتتىن ئۇنىڭغا ئوخشايدىغان بىرەرسىنىڭ سىماسىنى كۆرۈپ قالسىڭىز، ئورنىڭىزدا مىدىرلىماستىن ئۇزاققىچە تۇرۇپ قالىسىز.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، ئۇ كاماندىروپكىغا چىقماقچى بولغاندا ئۇنىڭغا ئاددىلا بىر ئېغىز «ئاق يول بولسۇن» دەپ ئۇزۇتۇپ قويىسىز، ئۇنىڭ يىراقلىشىۋاتقان سىماسىغا قاراپ، قەلبىڭىزنى بىر ئاز قىيالماسلىق ھىسسىياتى قاپلايدۇ، بىراق ھىچنىمە دىمەيسىز، پەقەت ئۈن-تىنسىز ھالدا ئۇنىڭ قايتىپ كېلىش خەۋېرىنى كۈتىسىز.
ئەگەر بىرسىنى سۆيسىڭىز، ئۇ كاماندىروپكىغا چىقماقچى بولغاندا، ئۇنىڭغا توختىماستىن نېمىلەرگە دىققەت قىلىش توغرىسىدىكى ئىشلارنى تاپىلايسىز، ئۇنىڭ سومكىسىغا كېيىم- كېچەك ۋە يىمەكلىكلەرنى ئۆز قولىڭىز بىلەن قاچىلايسىز، بىكەتتە ماشىنا قوزغالغۇچە ساقلاپ، ئاندىن ئۇ جايدىن ئايرىلىسىز. ئۇ كەتكەن شۇ كۈنلەردە كۆڭلىڭىز ئەنسىزلىككە تولىدۇ، ئاللاھدىن ئۇنىڭ ئامان-ئېسەن قايتىپ كېلىشىنى تىلەيسىز.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، ئاغرىپ قالغان چاغلىرىڭىزدا، ئۇنىڭ سىزنىڭ بۇ زەئىپ ھالىتىڭىزنى كۆرۈشىنى خالىمايسىز، كۆز ياشلىرىڭىزنى سۈرتىۋېتىپ، ئۇنىڭغا پەقەت قەيسەر ۋە خۇشال ھالىتىڭىزنى كۆرسىتىشكە تىرىشىسىز.
ئەگەر بىرسىنى سۆيسىڭىز، ئۇۋالچىلىققا ئۇچرىغاندا، ئۇنىڭ باغرىغا ئۆزىڭىزنى ئېتىپ، ھىچنىمېنى ئويلىماستىن، يىغلاپ تۇرۇپ بارلىق ئازابلىرىڭىزنى ئۇنىڭغا بايان قىلىسىز ھەم ئۇنىڭ باغرىدىن ئۆزىڭىزگە تەسەللى تاپىسىز.
ئەگەر ئۇنى ياخشى كۆرسىڭىز، ھەپتە ئاخىرىدا بازار ئايلىنىپ چارچىغاندا سېرتتا بىرگە تاماق يەيسىز؛ سوغۇق قىش كۈنلىرىدە بىر ئىستاكان قىززىق قەھۋەنى ئۇنىڭ بىلەن تالىشىپ ئىچىسىز؛ كوچىلاردا ئۇنىڭ بىلەن ماڭغىنىڭىزدا، يېرىم مېتىر ئارلىق قالدۇرۇپ ماڭىسىز؛ئۇنىڭ  بىلەن بىلەن بىرلىكتە كومپىيۇتېردا ئويۇن ئويناپ، خۇشاللىقىڭلاردا خۇددى كىچىك بالىغا ئوخشاپ قالىسىلەر.
ئەگەر ئۇنى سۆيسىڭىز، يېرىم كۈن ۋاقىت ئاجرىتىپ، ئۆز قولىڭىز بىلەن ئۇ ئەڭ ياقتۇرىدىغان تاماقلارنى ئېتىپ بېرىسىز؛ سوغۇق قىش كۈنلىرىدە، ئۇنىڭغا قىززىققىنە چاي دەملەپ بېرىسىز؛ كوچىلاردا ماڭغاندا ئۇنىڭغا چاپلىشىپ، قوللىرىنى تۇتۇپ ماڭىسىز؛ ئۇنىڭ يېنىدا جىمجىتقىنە ئولتۇرۇپ، ئۇنىڭ كومپىيۇتېر ئالدىدا ئەستايىدىل خىزمەت قىلىۋاتقان ھالىتىنى بەختىيارلىق ئىلكىدە تاماششا قىلىسىز.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، ئۇنىڭ بالىلىق دەۋرىدىكى قىزىقارلىق ئىشلىرىنى ئاڭلىغىنىڭىزدا، قاقاھلاپ كۈلىسىز، قەلبىڭىزدە چۈشۈنۈكسىز بىر خىل تەسىرلىنىش تۇيغۇسى پەيدا بولىدۇ.
ئەگەر بىرسىنى سۆيسىڭىز، ئۇنىڭ بالىلىق دەۋرىدىكى قىزىقارلىق ئىشلىرىنى ئاڭلىغىنىڭىزدا، كۈلۈمسىرەپ قويىسىز، قەلبىڭىزدە كۆز ئالدىڭىزدىكى بۇ پىشقان ھەم ئوماقلىقىنى يوقاتمىغان ئەرنى تېخىمۇ قەدىرلەش ھىسسىياتى تېخىمۇ كۈچىيىدۇ.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز،كارىدوردا ئۇنى ئۇچرىتىپ قالسىڭىز، قىزغىنلىق بىلەن ئۇنىڭ بىلەن سالاملىشىسىز، بىر نەچچە ئېغىز پاراڭلىشىپ قويىسىز، يېنىڭىزدىن ئۆتۈپ كەتكەن ۋاقىتتا، دەرىزە سىرتىدىكى يېقىملىق قۇياش نۇرىغا قاراپ، قەلبىڭىز ئىللىق بىر سېزىمغا ئەسىر بولىدۇ.
ئەگەر بىرسىنى سۆيسىڭىز، ئۇنىڭ ئۈچۈن بارلىقىڭىزنى سەرىپ قىلغاندىن كېيىن، قېلىپ قالىدىغانى قەلبىڭىزنىڭ چوڭقۇر قېتىدىكى يارا بولىدۇ.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، ئۇنى ئۆزىڭىزنىڭ قىلىۋېلىشنى ئويلايسىز، شۇڭا نۇرغۇن كىشىلەرنى ياخشى كۆرىسىز، نۇرغۇن كىشىلەرنى ئۆزىڭىزنىڭ قىلىۋالغۇڭىز كېلىدۇ.
ئەگەر بىرسىنى سۆيسىڭىز، ئۇنىڭدىن ئايرىلالمايدىغانلىقىڭىزنى ئېنىق بىلىپ تۇرۇپمۇ، ئەكسىچە ئۇنىڭدىن ئايرىلىشنى تاللايسىز، چۈنكى ئۇ ئېرىشمەكچى بولغان بەخىتنى مەن بېرەلمەسلىكىم مۈمكىن دەپ ئويلايسىز. ئۇنى ئىگەللىۋېلىشقا جۈرئەت قىلالمايسىز، ئۇنىڭ ئۆز بەختىنى تاپقانلىقىنى كۆرۈشنى ئۈمىد قىلىسىز، گەرچە بەخىتتىدىن تەڭ بەھىر ئېلىش پۇرسىتى سىزگە تالىق بولمىسىمۇ.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، يالغۇز قالغان، زېرىككەن، ئازابلانغان چاغلىرىڭىزدا ئۇنىڭ بىلەن پاراڭلاشقۇڭىز كېلىدۇ.
ئەگەر بىرسىنى سۆيسىڭىز، ھەر قانداق چاغدا، خۇشاللىق ھەم خاپىلىقلىرىڭىزدىن ئۇنىڭ بىلەن  ئورتاقلىشىشنى خالايسىز، خۇشال بولغان چاغلىرىڭىزدا، ھەتتا بۇ خۇشاللىقلارنىڭ ھەممىسىنى ئۇنىڭغا بېرىۋەتكۈڭىز كېلىدۇ.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، ئۇنىڭ بىلەن ئۇزۇنغىچە ئالاقىلاشماي، توساتتىن ئۇنىڭ تېلىفۇنىنى ئالغىنىڭىزدا، كۈلۈپ تۇرۇپ ئۇنىڭ سۆزلىرىنى ئاڭلايسىز.
ئەگەر ئۇنى سۆيسىڭىز، بىر نەچچە كۈن ئالاقىلىشالماي قالغان چاغلىرىڭىزدا، جىددىيلىشىپ ئۇنىڭغا تېلىغۇن قىلىسىز، ھەم تۆكۈلۈشكە تەييار تۇرغان ياشلىرىڭىزنى زورىغا بېسىپ كۈلۈمسىرەيسىز.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، ئۇنىڭ بىلەن بىرگە بولغان چاغلاردىلا، ئۇنىڭدىن ئەنسىرەيسىز.
ئەگەر بىرسىنى سۆيسىڭىز، مەيلى ئۇنىڭ بىلەن بىرگە بولغان ۋاقىت ياكى ئۇ بولمىغان ۋاقىتلاردا بولسۇن، ھەرمىنۇت ھەر سىكىنۇت ئۇنى سېغىنىسىز، ئەسلەيسىز، ياد ئېتىسىز.
ئەگەر بىرسىنى ياخشى كۆرسىڭىز، كۆپرەك ئادەم بىللە بولغاندا، ئۆزىڭىزنى تولىمۇ كۆڭۈللۈك ھىس قىلىسىز.
ئەگەر بىرسىنى سۆيسىڭىز، پەقەت ئىككىمىزلا بىللە بولساق دەپ ئويلايسىز، ئىككىڭلارنىڭ دۇنياسىغا ھىچكىمنى سىغدۇرغىڭىز كەلمەيدۇ.
قېنى ئەستايىدىل ئويلۇنۇپ بېقىڭ، سىز ئۇنى ياخشى كۆرەمسىز ياكى سۆيەمسىز؟
