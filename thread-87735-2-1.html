دېپلۇم دېگەن بىر ماركا ،ئۇ كىشىنىڭ ھەقىقىي ئۇنۋانىنى ئىپادىلەپ بېرەلمەيدۇ.
قىزغىز يازغۇچى ئابدۇراھمان ئەزىز ئوغلى ھەر قېتىم ‹‹شىنجاڭ مەدەنىيىتى››ژۇرنىلىغا ئەسەر ئەۋەتىدۇ،ئۇ دېھقان يازغۇچى ئاكىمىزنىڭ يازغان ئەسەرلىرىنىڭ قاراتمىلىقى كۈچلۈك،قۇيۇق يېزا تۇرمۇش كارتىنىسىغا يۇغۇرۇلغان.نەزەرىيە يوق دېيەرلىك.ئەمەلىي تۇرمۇش پىرۇتىپىدىن ئېلىنغان ئېسىل يازما دېيىشكە ھەقلىقمىز.يازما يازغانغىلا ھېساب ئەمەس،ئۇنى خەلق ئېتىراپ قىلىشى ھەمدە ئۇنىڭدىن مەنىۋى زوقلارنى ئالالىشى كېرەك.
نۆۋەتتە،ئۇيغۇر ئاپتۇنۇم رايۇنىمىزدىكى ھەر قايسى گېزىت-ژۇرناللاردا ھەر قايسى تۈردىكى ماقالىلەر ئېلان قىلىنماقتا،بىز ئۇچراۋاتقان كىرىزىس توغۇرلۇق يېزىلغان يازمىلار ئىنتايىن ئاز ئۇچرايدۇ،كۆپ كىشىلەر قەلەم ھەققى،ئۇنۋان ئېلىش ئۈچۈن پايدىلىنىش قىممىتى تۆۋەن بولغان يازمىلارنى چىقىرىپ تۇرماقتا.
قۇرۇق نەزەرىيە،ئەمەلىيەتتىن چەتنەپ كەتكەن شۇ يازمىلارنىڭ ئاپتۇرلىرى ئۆز ئىقتىسادىي مەنبەسىنى يۇقۇرى كۆتۈرۈش،ئۆزىگە نۇپۇز تىكلەش،ئۇنۋان ئېلىش ئۈچۈن ھەرخىل يوللار بىلەن مەتبۇئات كوچىسىنى قالايمىقان قىلدىغان ئەھۋاللار جەمىيتىمىزدە خېلى كۆپ.
بۇ ئىشنى تۈزەش ئۈچۈن بەلكىم نۇرغۇن ۋاقىتلار كېتىشى مۇمكىن،تۇلۇق كۇرس دېپلۇمىنى ئۆزلىكىدىن ئۆگىنىپ ئالغان كۆپ ياشلىرىمىز ئىلمىي ماقالىنى زورىغا يېزىپ قويىدۇ،لېكىن،ئۇلارنىڭ يازغان ئاشۇ يازمىلىرىنىڭ قانچىلىك قىممىتى بار؟بۇنى بىلىش ئۈچۈن شۇ كىشىنىڭ جەمىيەت رىقابىتىدە قانداق تاقابىل تۇرغانلىقىنى،ئەمەلىي خىزمەت جەريانىدىن بىلىشكە توغرا كېلىدۇ.
تۇرمۇش مۇرەككەپ ھادىسە،جەمىيەتتىكى ناتوغرا بولغان بەزى تەخىرسىز مەسىللەرنى ئۆزگەرتىپ چىقىش ئۈچۈن نۇرغۇن ۋاقىت ۋە جەريانلارنى،ئۇزۇن بولغان بىر مۇساپىنى باشتىن كەچۈرىدۇ.بۇنى تۈزەش ئۈچۈن مائارىپ نازارىتى ئالاقىدار تۈزۈملەرنى قايتىدىن ئىسلاھ قىلىش كېرەك.
چۈنكى،ئۇنۋان ئۈچۈن ماقالە يازىدىغان كىشىلەرنىڭ كۆپى مائارىپ ساھاسىدە خىزمەت قىلىدىغان كىشىلەر،تۈزۈلمىدە ئۆزگىرىش،ئىسلاھ قىلىش  بولمايدىكەن،بۇ ئىش يەنىلا تۈگمەيدۇ.

مۇنداق مەكتەپتە ئىشلەشنى خالامسىز؟
ئالدى بىلەن ئەھۋالنى قىسقىچە بىر بايان قىلاي،ھەقىقى ئەھۋاللار...
بىزنىڭ مەكتەپ بىر يىغىۋېتىلگەن مەكتەپ شۇنداقلا باشلىقلىرىمىز (مەكتەپ باشلىقلىرىمىز)ئۆزى خالىغانچە ،ئۆزىنىڭ كىچىك پادىشاھلىق دۆلىتىدەكلا باشقۇردىغان مەكتەپ،نىمىشقا شۇنداق بولىدۇ،چۈنكى چەترەك،ئۆزى خالىغانچە باشقۇرۇپ ھەرخىل ئۇسۇل چارىلىرى بىلەن شۇنىڭغا بىزلەرنى كۆندۈرۈپ بولغان مەكتەپ ،كۆنمەيمۇ نىمە ئامال،چۈنكى سەھرالاردا ئىشلەۋەرسە ئادەممۇ بىر قىسىملىرى  قويدەك،بىر قىسىملىرى بۆرىدەك بولۇپ كىتىدىغان ئوخشايدۇ.
بۇ مەكتىۋىمىزنىڭ يەنە بىر ئارتۇقچىلىقى شۇكى،ئۇزۇن ۋاقىتلاردىن بۇيان بىر خىللا دىگۈدەك،مائاشى يۇقىرىلار ئارامخۇدا ،مائاشى ئادەتتىكىدەكلەر ماشىنا ئادەمدەك ئىشلەپمۇ يەنە شۇ ئىسىل مۇدىرلىرىمىزنىڭ سېسىق گىپىنى پات-پات ئاڭلاپلا تۇرىمىز،چۈنكى ‹‹ئىشەك قانچە بولسا يېغىرى شۇنچە ››دىگەندەك ئىشى جىقلارنىڭ ئەلۋەتتە ‹پۇقى›مۇ جىقراق بولىدۇ دە.
بۇ مەكتىۋىمىزدە كونكىرىت تەقسىماتمۇ مۇنداق،تۇلۇقسىز 3سىنىپ ۋە 120 دەك ئوقۇغۇچى بار،بۇنىڭغا دەرىسكە كىردىغان ئوقۇتقۇچىلاردىن 20چە بار،بەزىلىرى ھەپتىسىگە تۆت سائەتلەر ئەتراپىدا دەرىسمۇ ئۆتمەسلىكى ،ھەتتا دەرسىنىمۇ تەييارلاپ يۈرمەسلىكى مۈمكىن،
ئەمما باشلانغۇچتا 8سىنىپ ۋە 300چە ئوقۇغۇچى بار، بۇنىڭغا دەرىسكە كىردىغان ئوقۇتقۇچىلاردىن بۇنىڭغىمۇ 20چە بار،مانا مەن پىقىر ئالتە دەرىس بىلەن ئالتە تەييارلىقنى يازىمەن،خانىنىڭ خاتىرسىنى يازىمەن،جەدىۋەللىرىنى ئىشلەيمەن،تەجىرىبە دوكلاتلىرىنى يازىمەن،شۇ ئالتە پەننىڭ تاپشۇرقىنى تەكشۈرىىمەن،ھەر بىر سائەتلىك دەرسنىمۇ شىنجاڭ مائارىپى ژورنىلىنىڭ چوڭلۇقىدىكى دەپتەرگە 4 بەتتىن ھەر كۈندە 16 بەتتىن يازىمەن 4سائەت دەرىس ئۆتىمەن،4سائەت ساراڭدەك تەييارلىق يازىمەن،ئاخشىمى شۇ پەننىڭ تاپشۇرۇقلىرى ئۈستىدە ئۈگدەپ يېتىپ ئۇخلاپ قالىمەن ،يەنە بىر مۇنچە ئېشىپ قالىدۇ ،ئەمما ،مۇناسىۋەتكە ئۇستا باشلىقلارنىڭ نەزىرى چۈشكەن بەزىلەر مەن قىلغاننىڭ يېرىمنى ھەتتا ئۇنىڭمۇ يېرىمىنى قىلىشى مۈمكىن،باھالاشتا شۇلار خو!مەن پو!
ئەمدى تۇرمۇش مەسلىسىگە كىلەيلى،ئوقۇغۇچىلار زامانىۋى ياتاقلاردا ياتىدۇ،ئوقۇتقۇچىلار دېھقانلارنىڭ ئارتۇقراق سېلىۋالغان ئىجارە ئۆيلىرىدە ئايدا 70يۈئەندىن 100 يۈئەنگىچە تۆلەپ ئولتۇرىمىز.
ئوقۇغۇچىلار غا مەكتەپتە بىكارلىق تاماق بار شۇنداق ياخشى ،ئارلاپ پولو ۋە خوردەك،
ئەمما ئوقۇتقۇچىلار سرىتتىن ،دادۇنيالىقتىكى بىر ئىككى ئاشخانىلاردىن قورساق توقلاپ يۈرىشىدۇ.
ئەمدى ئايلىق تەمىناتلىرىغا قاراڭ،قانچە كۆپ ئىشلىىگەنلىرىنىڭ مائاشى ئادەتتىكىچە ،ئەمما ئادەتتىكچە ئىشلەپ قويىدىغانلىرىنىڭ مائاشى بىرەر مىڭ يۈئەندىن ئارتۇقلا قاتلاپ قويىدۇ.
ھېلىمۇ ياخشى يېرى ئۇلار بىزلەرنى زاڭلىق قىلىپ يۈرمەيدۇ ،بولمىسا ئاچچىقنى قانچىلىك يۈتىۋالار بولغىيدۇق دەيمەن ،مۇدىرلىرىمىزنىڭ دەشناملىرىنى قوشقاندا.....
مۇدىرلىرىمىزخۇددى ھۆكۈمەتنىڭ مائاشىنى ئۆزى بىرىۋاتقاندەك قىلىپ كىتىدۇ،ھېلىمۇ بانكىدىن بىرىلىۋاتقىنى... ئىشقىلىپ ،نىمىلا دىمەيلى ‹‹زامان زورنىڭ ،تاماشا كورنىڭ ›دىگەن ماقال-تەمسىللەر جەنۇبى شىنجاڭ يېزىلىرىدا ئەمەلىيەت بۇلۇپ ،كۈچىنى كۆرسەتسە كېرەك.
ئەرىز قىلساڭ،پىكىر بەرسەڭ بولمامدۇ دىيىشىڭ مۈمكىن ،ئەمما سۇ بېشىدىن لاي،چىڭغا چىققاندا يەنە شۇلارنىڭ ھىساپ بولىدىغىنى كۆپ ئادەملەر بىلىدۇ،شۇنىڭ بىلەن ‹سەۋىر قانائەت ھەممىدىن ئۇلۇغ›دەپ يۈرىۋېرىمىز ،ئەمما ھەق-ناھەقنى باللارغا ئېنىق بىلدۈرۈشكە ئادەتلىنىپ  ئۆگىنىپ قالغان يۈرەك تىپىرلاپ توختىمايدىكەن.
ئۇنىڭ ئۈستىگە پىكىر بەرسەكمۇ ،بىلىپلا قالدىمۇ بولدى،ئارقىچىلاپ سۈرۈشتۈرۈپ ،خىلمۇ-خىل ئاماللار بىلەن تېتپ دات پەندىياد،ئىككىنچى ئۇنداق قىلمايمەن ياكى بۇ خىزمەتنى قىلمايمەن دىگۈزمىسە ،مىڭ توۋا قىلدۇرمىسا ،ياكى يۆتكىلىپ كىتىپ قۇتۇلۇپ كەتمىسە ،قەتتىي باشقا ئامال يوق.
خىزمەتلەرگە پايدىلىق بولغان تەكشۈرۈشلەرنىمۇ كەلمەي كەتسىدى دەپ تىلەيدىغانلا بوپ كەتتۇق،چۈنكى ئۇلارنىڭ باشلىقلارغا ئېلىپ كەلگەن پايدىسى بولغان بىلەن بىزلەرگە كەلتۈردىغان دەرىت –ئەلەملىرىنى خېلى بىر كەملەرگىچىمۇ ئېيتىپ تۈگەتكىلى بولماس،خۇددى ئۇلارمۇ بىزنىڭ دەردىمىزنى ئاۋۇتقىلى ۋە كىچىككىنە سەۋەبلەر بىلەن بىزنى مۇدىر ۋە باشىقلىرىمىزغا تازا تالاتقىلىلا كىلىدىغاندەك ھىس قىلىدىغان بۇلۇپ قالدۇق.
ھوي مالىم قانچىلىك دەردىڭ بار،قانچىلىك دەردىڭ بار،ئىلمىي مۇدىر ئالە خىزمەت تەقسىماتىڭنى دىمەي،ئاۋۇ پەنىڭنى ئەكەل،ھوي مۇنۇ نىڭ كىچىككىنە يېرى بولماپتۇ دەپلا كىتىپ قالىدۇكى ،ئۇلارنىڭ بىزنىڭ تازا تالىشىغا بىرمۇنچە يىتەرلىك سەۋەبلەرنى تېپىپ بىرىپ كىتىدۇ.
يا ئۇلارنىڭ كۆزى كور،قۇلىقى پاڭ،ئايىقى چىقمىغان كىشىلەرمىكىن ،باشلىقلىرىمىز باشلىغان يولغا كىرىپ،ئۇلار دىگەننى ئاڭلاپ،جاپالىق ئىشلىگەنلەرنىڭ كىچىككىنە نوقسانلىرى ئۈچۈن خېلى ئايلارغىچە تارتۋالغۇدەك دەرىت ۋە ئەلەملەرنى قالدۇرۇپ كىتىشىدۇ.
ئۇلار ھەر گىز ئۇنداق ئىشلارنى ئەمەلىي تەكشۈرۈپ،ھوي ئىلمىي مۇدىر ،مۇنۇ مۇئەللىم ياشىنىپ قاپتۇ،ھوي مۇنۇ مۇئەللىم بۇ ئىشنى قانداق قىلىپ بولىدۇ دەپ ئىككى ئېغىز لىللا سۆزلىمەيدۇ ،ئەجەبا ئۇلار ھەقىقەتنى سۆزلەشتىنمۇ قاچامدىغاندۇ،سۆزلىيەلمەس،ھىس قىلالماس بوپ قالغانمىدۇ؟!!!
مەنغۇ خېلى ئۇزۇن بولدى ،بۇنداق ناھەقچىلىكلەرنىڭ دەستىدىن خىزمەتتىن جاق تويغىلى ،ئەمما نىمىشقىدۇر ئىشلەۋاتىمەن ،ئەمما مۇشۇنداق كىتىۋەسە ئۆز ئەجىلىمدە كىتىشكە ،خاتېرجەم كىتىشكە ،ياكى پىنسىيىگە چىقىشقا زادىلا كۈزۈم يەتمەيدۇ.
تاشلاي دىسەم ئون نەچچە يىل بوپتۇ،ئوقۇغان كەسپىممۇ شۇ،ئۇنىڭ ئۈستىگە بۇ ئىشمۇ ئاشۇنداق ناھەقچىلىكىنى چىقىرۋەتكەندە شەرەپلىك خىزمەت،ھەم كۈڭۈللۈك خىزمەت ئىدى،ئەمما .....
ئەمدى تورداشلار سىلەرمۇ بىر ئويلىغىنىڭلارنى دەپ بېقىڭلار مۇشۇنداق خىزمەتكە سىلەر ئىرىشىپ قالساڭلار ،مۇشۇنداق ئىشلارغا دۈچ كىلىۋەرسەڭلەر،لۈشۈننىڭ گىپىچە‹تۆت ئەتراپنىڭ ھەممىسى تام تۇرسا ئۈسىۋالمامدۇ كىشى› دىگەندەك بولسا سىلەر قانداق قىلاتتىڭلار؟
مەنغۇ ھازىر شۇنداق تۇيۇپ ،كىكىرىپ كىتىپ بارىمەن بۇ خىزمىتىمگە ،ئۆتكەندە ئادۇۋكات بىرىنى ئىزدەپ باردىم،دىدىم:‹‹ ھۈكۈمىتىمىزنىڭ قانۇنلىرىنى ئوبدان بىلىسىز،مەن ئىشلەۋاتقىلى ئون نەچچە يىل بولدى ،ئەمدى چىداپ بولالماي كىتىپ بارىمەن دەڭا قانداق قىلسام ،ئىشسىزلىق ،داۋالىنىش سۇغۇرتا ۋە ئۆي پۇللىرىمنىڭ بىر تەرەپ قىلىدىغىىنىنى قىلىپ ،راۋانلاشتۇرۇپ ،مىنىڭ ھۆكۈمەتتە، ھۆكۈمەتنىڭ مىنىڭدە ئېلىشى قالماي ،ئىشلىگەن ئەجرىم يەردە قالماي ،مۇۋاپىق بىر يولنى تېپىۋالالايمەن›› دىسەم،ئۇمۇ بىر نىمە دىيەلمەي تۇرۇپ قالدى،مەنمۇ شۇ ئاچچىق يۇتۇپلا ئىشلەپ كىلىۋاتىمەن.
سىلەرچە قانداق قىلسام بولار،مىنڭچە ئىشلىگەنلەرنىڭ دەردىنى يەنىلا ئىشلىگەنلەر بىلىدۇ....
شۇڭا بىنادىن ئۆيمۇ ئالمىدىم،ماشىنىمۇ ئالمىدىم ،ھۆكۈمەتتىن تاپقىنىمنى خىلمۇ-خىل يوللار ۋە تەربىيلەشلەر ئارقىلىق ئۆزىگە كىتىپ بارىدۇ،بىر قىسىمدا تۇرمۇشۇمنى قامداپ كىتىپ بارىمەن..
مۇۋاپىق بىر يۇلى چىقىپ قالسا ،بىر ئارام ئېلىپ ئۇھ دىسەم بولاتتى، دۇنياغا داڭلىق باي بىل گەيتىسمۇ‹‹ئادەمنىڭ بىر نانغا قورسىقى تويىدۇ›› دىگەن ،مەنمۇ ھەم شۇ قاراشتا...
