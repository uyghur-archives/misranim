بۇنداق دىيىشىم جوڭگونىڭ %70 نېفىت مەنبەسى شۇ ئىراندىن  ...

  تەھلىلىڭىزغۇ ناھايىتى ئورۇنلۇق .بىراق جۇڭگۇ سىياسىتى ناھايىتى قىل ئۈستىدە كىتىۋاتقان سىياسەت . بۇنى دۆلىتىمىزنىڭ سىياسەت بەلگۈلەش خادىملىرىمۇ ناھايتى ياخشى چۈشىنىدۇ . شۇڭلاشقا بەزىدە بىز قايناپ قالغان بىر قىسىم مەسىلەەرگە قارىتا دۆلىتىمىز سۈكۈت قىلىش ئامال بولمىغاندا . ئىشنى يىنىكلىك بىلەن بىر تەرەپ قىلىش پوزىستىيەسى ئارقىلىق زىل مەسىلىنى جىدەلنىڭ ھىددىن قاچۇرۇپ ئىمكان بار ، ھەر قانداق بىر مەسىلىنى تىنىچلىك رامكىسى ئىچىدە ھەل قىلىپ كىلىۋاتىدۇ ۋە بۇ ئۇسۇلنى ئاقلانە چارە دەپ قاراپ كىلىۋاتىدۇ . مەسىلەن پارىخورلۇق .  كان-كارخانا . تاشقى سودا قاتارلىق تىجارەت ئېشلىرىدا مۇنپۇل قلىۋىلىش ، چىگرا مەسىلىسى ، خەلىقئارادىكى پايدا-مەنپەت مەيدانىدا دۆلىتىمىز ئىزچىل ئۆز خادىملىرىنىڭ يۇقارقى كەسىپلەر بىلەن شوغۇللىنىپ ،يۇقارقىدىكىدەك قانۇنغا خىلاپ قىلمىشلارغا  چىتىلىپ قالغان خادىملارغا نىسپەتەن تۆگىدەك ئىشنى  تۈگمىدەك قىلىش يەنە چڭ ئىشنى كىچىكلىتىش ئۇسۇلى بويۇنچە سەل قىلىپ پارتىيەگە كۆپلەپ دۈشمەن تىپىۋالماسلىق يولىنى تۇتتۇپ كىلىۋاتىدۇ .

دۆلىتىمىزنىڭ ھەربىي ئېشلار تېخنىكىسى تەرەقى قىلمىدى ئەمەس ناھايتى ياخشى تەرەقى قىلدى . بىراق ئۇرۇش يۈز بەرگەندە نۇقۇل بىر ئۆزۇنىڭلا ھەربىي ئىشلاردىكى ئىقتىدارىغا  تايىنىپلا مەسىلىنى ھەل قىلغىلى بولمايدۇ. ئۇرۇش قىلىش ئارقىلىق ئۇرۇش مەيدانىنى قوغداش دەپ بىر ھەربىي ئىشلار تاكتىتكىسى بار بۇ تاكتىتكىنى قوللىنىش ئۇچۈن مۇقەرەر بىسىپ ئوتۇشكە تېگىشلىك يول ۋە شەرتلەردە ئامېرىكا باشچىلىغىدىكى ئوق بىشى دۆلەتلەردىن باشغقا دۆلەتلەر بىسىپ باقمىدى، بىسىپ مىڭشنى ئويلىغان ۋە قەدەم باسقان دۆلەتنى  يۇقارقى ئوق بىشى دۆلەتلەر بۆشۈگىدىلا  تۇنجۇقتۇرۋەتتى . بىراق ئامېرىكا باشچىلىغىدكى دۆلەتلەر ئامېرىكىنىڭ كۈنلىگى ئاستىغا جەم بولۇپ بۇ پىلاننى تۈزۈش بىلەن ئىشقا ئاشۇرۇپ بولدى . ۋە ئىشقا  ئاشۇرۇشتىمۇ پۇختا ،مۇستەھكەم ئىشقا ئاشۇرۇپ بولدى . قارسىام ئەقىدلا تازا تولۇپسىز . ئامېرىكا دۆلىتىمىزنىڭ نەچچە ۋاقىتتتىكى جان-جىگەر دۆلەتلىردىن روسىيە . ئافغانسىتان ،پاكىستان . كورىيە .......قاتارلىق دۆلەتلەرنى جۇڭگۇغا قارىتا سوغۇق مۇئامىلە قىلىش دەرىجىسىگە ئەكىلپ بولدى . ئۇنىڭ ئۈستىگە شىزاڭنىڭ  لاماسى دالاي ۋە..........قاتارلىق دۆلەتنى پارچىلاشقا پۇرسەت كۈتىۋاتقان گۇماشتىلا ھەر قاچان دۆلىتىمىز ئىچىدە چوڭ تىپتىكى بىر ئۇرۇشنىڭ بولۇشىنى تۆت ككزى بىلەن كۈتۈپ كىلىۋاتىدۇ .   بۇ نۇقتىنى چۈشىپ يەتكەن دۆلىتىمىز ئۆزىنىڭ ھازىرقىدەك دىياۋيۇ ئارىلى بولمسۇن ياكى تەيۋەن مەسىلىسىگە ئوخشاش ئۇرۇشقا سورەپ كىردىغان مەسىلىسدە  زورمۇ زور ئۇرۇشتىن ساقىلنىپ قىلىش يولدا داۋاملىق ئىزچىل مىڭىشنى خالايدۇ دەپ قاراشقا بولىدۇ .

شەرقى دىڭىز بۇيلىرىدا:

دۆلىتىمىزنىڭ دىياۋيۈ ۋە خۇاڭيەن ئاراللىرىغا بولغان نارازىلىق كەيپىياتىنىڭ كۈنسىرى يۇقىرلىشى بىلەن باشلانغان جۇڭخۇا ئوغۇل قىزلىرىنىڭ دۆلەت ئىچىدىكى ياپۇن ماللىرىغا ئۆچمەنلىك قىلىش ۋە ساياھەتكە چىقىشنى ئەمەلدىن قالدۇرۇش قاتارلىق ئۇسۇللار بىلەن ۋەتەنپەرۋەرلىكىنى ئىپادىلىشى، ،ئامىركىنىڭ ئاسىيا سەپىرىدە ھە دىگەندىلا دۆلىتىمىزنىڭ ئەتىراپىدىكى خۇشنىللىرى بىلەن ھەمكارلىق دىپلۇماتىيەسىنى كۈچەيتىشى،تىنىچ ئوكياننى ئايلىنىش ۋە ئارالنى قايتۇرۋىلىش نامىدىكى ھەربىي مانىۋىرلىرى دۆلىتىمىزنى ھەقىقەتەنمۇ ئەنسىز قىلىۋەتتى.ئىنسانىيەت پەيدا بولغاندىن كىيىنلا دۆلىتىمىزنىڭ بىر ئايرىلماس مۇقىم زىمىنلىرى بولۇپ تۇرغان جايلار نۆۋەتتە ئىلگىرى دۆلىتىمىزنىڭ تۇزىنى يەپ ئەقىلگە كىرىپ قالغان ھارام تاماقلىرى تەرىپىدىن بىسىۋىلىنىش خەۋىپىگە دۇچ كەلمەكتە.

ئوتتۇرا ئاسىيادا:

ئامرىكا پىرزىنىتى ئوباما (ئامرىكا گوۋۇيۈەنى) دۆلەت ئىشلىرى مەھكىمىسىنىڭ كاتىبى ھىلارى كىلىنتون بىلەن دونى ئوتتۇرا ئاسىياغا تىكىتى،ئوتتۇرا ئاسىيانى ئىگىلەش مىنىڭچە قانداقتۇر ئوبامانىڭ سىياسى ئەندىزىسى بولماستىن بەلكى ئىلگىركى سابىق پىرزىنىت جۇرجى كىچىك بوشتىن قالغان خىزمەت ‹‹مىراسى››ھىسابلىنىدۇ.سابىق پىرزىنىت جورجى كىچىك بوش بىزگە نامەلۇم بولغان ئۇرۇشخۇمار تەشكىلاتنىڭ تەشەببۇسى بىلەن 11- سىنتەبىر تىرۇرلۇق ھۇجۇمىنى پىلانلىدى،بۇ پىلان ئافىغانىستاننى ئىگىلەش ئارقىلىق كونا رەقىبى روسىيەنىڭ ئوتتۇرا ئاسىيادىكى تەسىرىنى ئاجىزلىتىش بىلەن بىرگە دۆلىتىمىزنىڭ تەرەققىياتىنى چەكلەشنى مەخسەد قىلىدۇ. ئەزەلدىن ئامىرىكىغا ئۆچمەنلىك قىلمايدىغان ۋە ئايرۇپىلان بۇلاپ دۇنيا سۇدا تەشكىلاتىغا ھۇجۇم قىلغۇدەك مادارى يۇق تالبان ۋە بازا تەشكىلاتىغا قارا قۇيۇق قالپاق كەيدۈرۈپ بۇ ۋەقەنىڭ پىلانلىغۇچىسى دەپ نام بەردى،ۋە بۇ نامنىڭ خەلىقئارادا بازارلىشىشىغا كاپالەتلىك قىلىش ئۈچۈن بازا تەشكىلاتى ۋە تالبان تەشكىلاتىغا ئەزا بولغان ئاسىي ئەزالىرىنى قۇتۇرتىپ بۇ ئىشقا مەسئۇل بولىدىغانلىقى تۇغۇرسىدا جار سالدۇرۇش ئارقىلىق مەخسىدىگە يەتتى.بۇ ۋەقە بولۇپ ئۇزاق ئۆتمەي كىچىك بوش دۆلەت ئىچىدە ئامرىكىلىقلارنىڭ ھۇجۇم قىلىش سادالىرىنى تازا ياڭرىتىپ ئاندىن افىغانىستانغا ھۇجۇم قۇزغىدى،ئىراقتىن قايىتقان ئامرىكىنىڭ نەچچە تۈمەن ئەسكىرى ئوتتۇرا ئاسىيادىكى ھەربى بازىلارغا جىددىي يۆتكەلدى.بۇ خىل ئۇرۇش ۋەھىمىسى  دۆلىتىمىزگەم جۈملىدىن شىنجاڭغا تەھدىت ئىلىپ كەلدى.ئامىرىكىنىڭ ئافىغانىستان ۋە پاكىستاندا قانات يايدۇرغان تىرۇرلۇققا قارشى ئۇرۇش تەننەرخى ئىشىپ بىرىۋاتقان ئەھۋالدا دۆلىتىمىزدىن قەشقەر بىلەن ئافىغانىستان چىگىرلىنىدىغان دۇنياغا داڭلىق كارىدۇر،ۋاخان كارىدۇرىنى ئىچىۋىتىشنى تەلەپ قىلىپ كەلدى،بۇ كارىدۇر ئىچىۋىتىلسە ئامرىكا باشچىلىقىدىكى ئىتىپاقداش دۆلەتلەر ئارمىيەلىرى شىنجاڭ زىمىنىدىن پايدىلىنىپ ئافىغانىستان ۋە پاكىستان قاتارلىق دۆلەتلەرگە ھەربى لاۋازىمەتلىكلەرنى ئىلىپ كىرىش ئىدى،بۇنداق بۇلغاندا ئامرىكىنىڭ لاۋازىمەتلىك تۇشۇش تەننەرخى زۇر دەرىجىدە ئازلايدۇ،لىكىن بىيجىڭ بۇ سۇئالغا ھازىرغىچە ئىنىق بىر نىمە دىمىدى،مەخسەد بولسا مىنىڭچە ئافىغانىستان ۋە پاكىستاندىكى ئۇرۇش تەسىرى دۆلەت ئىچىگە تارقىلىپ مۇقىمسىزلىق ئامىلى يۈز بىرىشتىن ئەندىشە قىلىدۇ.ئوباما تەخىتكە چىقىپ 2014-يىلى ئافىغانىستان ۋە پاكىستاندىن ئەسكەر چىكىندۈرۈشنى باشلايدىغانلىقىنى ئىلان قىلدى.لىكىن يۈزى قىلىن ئامرىكا بىر زىمىنغا پۇتىنى قۇيىۋالدىمۇ،بولدى،بىزەڭلىك قىلىپ ئۇ يەردىن چىقماي تۇرىۋالىدۇ.نۆۋەتتە ئامرىكا دۆلىتىمىزگە خۇشنا ئۆزبىكىستان،قازاقىستان،مۇڭغۇلىيە قاتارلىق دۆلەتلەردىكى ھەربىي بازىلىرىنى تىخىچە ئىشلىتىۋاتىدۇ.خەۋەرلەردە ئىيتىلىشچە ماناس ھەربىي بازىسى قاتارلىق ئامرىكىنىڭ ئوتتۇرا ئاسىيادىكى ھاۋا ئارمىيە بازلىرى ئايرۇپىلان بىلەن ماڭغاندا دۆلىتىمىزگە بىر نەچچە سائەتلىكلا كىلىدىغان زىمىندا ئىكەن.

شەرقى جەنۇب تەرەپتە:
ئامرىكا ئاسىيادىكى ئىستىراتىگىيەلىك شىرىكى ھىسابلايدىغان ھىندىستان ،ۋە باشقا ئىتىپاقداشلىرى بىلەن بەلكىم بىز بىلمىگەن قىلتاقلارنى تۈزىۋاتىدۇ دەپ قارايمەن.ئامرىكا شەرقى جەنۇبى ئاسىيا ئەللىرى ئىتىپاقى يىغىنىدا قاتناشقاندا ئۆزلىرىنى ،تىنىچ ئوكيان دۆلىتى دەپ ئاتاپ بۇ يىغىنغا قاتنىشىش سالاھىيتىدە قاتنىشىۋاتىدۇ.

ئوتتۇرا شەرىقتە:
سۈرىيە ۋەزىيتى ۋە ئىران يادىرو كىرزىسى سەۋەبلىك بولىۋاتقان جىددىيچىلىكلەر ئوتتۇرا شەرىقتىكى دۆلەتلەرنى خىلىلا سەگىتتى،سەئۇدى،قاتار،بەھرەيىن،ئەزەربەيجان قاتارلىق دۆلەتلەر ئ.امرىكىدىن بەس بەستە قۇرال ياراغ سىتىۋىلىشنى تىزلەتمەكتە.
شۇنىڭغا ئەگىشىپ سۈرىيەدىكى ئىچكى ئۇرۇش قانلىق قىرغىنچىلىققا ئايلىنىپ ھاكىمىيەت بىشىدىكى بەشىر ئاساد بىلەن ئۆكتىچىلەر ئوتتۇرسىدا قانلىق تۇقۇنۇشلار بولۇپ ئۆتتى،رسىيە ئولتۇرالمايلا قالدى،چۈنكى روسىيەنىڭ ئوتتۇرا شەرىقتىكى سۈرىيەدىلا ھەربىي بازىسى بار.شۇڭا سۈرىيە قۇلدىن كەتسە روسىيەنىڭ ئوتتۇرا شەرىقتە سايىسىمۇ قالمايدۇ دىگەن گەپ.
ئىران يادىرو كىرزىسى بولسا ئىسىرائىلىيە باشتىن ئاياق تىركىشىۋاتقان مەسىلىدۇر،ئىسىرائىلىيە ئىراننىڭ يادىرۇ قۇرال ياسىشى ۋە تەھدىتى ئالدىدا ئوتقا چۈشكەن تۇخۇدەك تىرىلاپ قالغاندا ياۋروپا ئىتىپاقى ۋە ئاكىسى ئامرىكىغا يىلىنىپ باردى.لىكىن ئامرىكا ياۋرۇپا دۆلەتلىرى ۋە ئىسىرائىلىيەنىڭ ھەربى ھۇجۇم قۇزغاش تەكلىپىنى رەت قىلىۋەتتى.

مىنىڭچە ئوباما سۈرىيە كىرزىسى ۋە ئىران يادىرۇ مەسىلىسىدە دانا تەدبىر چىقارمىسا ئالغان پايدىسىدىن تارتقان چىقىمى ئىشىپ كىتىدۇ.سۈرىيە ۋەزىيتىدە كەسكىنلەشكەن ۋاقىتتا بۇ ئۇرۇش قاينىمىغا تۈركىيە كىرىپ قىلىشتىن تەستە ساقلىنىۋاتىدۇ.سۈرىيە يىقىندا ئىتىپ چۈشۈرۋەتكەن تۈركىيەنىڭ جەڭ ئايرۇپىلانىنى روسلار قىلغان دىگەن گۇمان ئوتتۇرغا قۇيۇلماقتا. ئوباما سۈرىيەدىكى مۇستەبىت ھاكىمىيەت بىلەن قارىشلىشىۋاتقان ئۆكتىچىلەرنى قۇرال بىلەن تەمىنلەيمىز دەپ ۋەدە بەردى يۇ،ھازىرغىچە ئەمىللەشمىدى،بۇ سۈرىيەدىكى ئۆكتىچىلەرنى ئۆمۈدسىزلەندۈردى.نۆۋەتتىكى ئوتتۇرا شەرىق ۋەزىيتىدە سۈرىيەنى قۇللاۋاتقان ئىران،ئىراننى قۇللاۋاتقان روسىيە مەۋجۇت ئامىللاردۇر.شۇڭا ئوباما سۈرىيەگە تىگىش قىلسا ئىراننىڭ بۇنىڭدىن خاتىرجەمسىزلىنىپ،ھامان كۈنىمىز مۇشۇنداقلا بولىدىكەن دىگەن گۇمان بىلەن جاننى ئالقانغا ئىلىپ قۇيۇپ ئۇرۇش قۇزغىشىدىن ئەندىشە قىلىدۇ،بۇنداق ئۇرۇش ھامان ئامرىكىغا پايدىسىز،ھەم كۇنا رەقىبى روسىيەنى چۆچىتىپ قۇيىدۇ.شۇڭا ئاۋۋال ئىراننى مۇللاق ئاتقۇزسا سۈرىيە ۋەزىيتى ئۆزلىكىدىن تىنجىيدۇ.روسىيە بولسا قارا قۇلىنى يىغىشتۇرىدۇ.شۇڭا ئىراننى مۆلچەرلەۋاتقان بولىشى مۇمكىن.


يۇقارقى بايانلاردىن شۇنى كۇرىۋالىمىزكى ئامىرىكا دۆلىتىمىزنى قۇرشاش سىياسىتىنى بۇشاشتۇرماستىن تىخىمۇ قاتتىق ئىجىرا قىلىۋاتىدۇ دەپ قارايمىز.

دىمەك  چوڭ دۆلەتلەرنىڭ ئۆز ئارا ھەربىي ھۇجۇم قۇزغاش ئىھىتىمالى يۇق،ئەمما كىچىك دۆلەتلەرنى پىچكا ئورنىدا قۇربانلىق قىلىش نىيىتى بار.

ئامىرىكا ياپۇنىيەنى ۋە فىلىپىننى كۈشكۈرتىپ دۆلىتىمىزگە قارشى سادالارنى ياڭرىتىۋاتىدۇ،ئەمما دۆلىتىمىز ئۆزىنى قاتتىق تۇتىۋالدى.ئوتتۇرا شەرىقتە ئىسىرائىلىيە ئىرانغا ھۇجۇم قۇزغايمىز دەپ توۋلاپ يۈىيدۇ،لىكىن ئوباما رەت قىلىپ تۇرىدۇ.بۇ ئاقىۋەت5تە ئامرىكىنىڭ ئىككىلا تەرەپتىن بىرلا ۋاقىتتا ھۇجۇم قىلىش نىيىتى بارلىقى بىلدۈرەمدۇ،ياكى ئامرىكا شۇنداق بىر كۈننى كۈتۈپ تۇرامدۇ......
