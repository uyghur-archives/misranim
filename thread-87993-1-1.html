ئەسسالامۇ ئەلەيكۇم مىسرانىم مۇنبىرىنىڭ ئەزىزلىرى ، ھەممىڭلارنىڭ تېنىگە سالامەتلىك ئىشلىرىغا چەكسىز ئۇتۇقلار يار بولسۇن

كەمىنە چىمبۇلاق بۇ تېمىنى بۇنداق پاساھەتلىك باشلىشىمدا مۇنداق سەۋەب بار ، 10 نەچچە قېتىملىق ساھىبخان مەيداندىكى مۇسابىقىنى تورداشلارنىڭ ھوزۇرىغا كەڭ سۇنۇش ئۈچۈن پۈتۈن كۈچىمىزنىڭ يېتىشىچە تىرىشچانلىق كۆرسىتىپ ھەر بىر مەيدان مۇسابىقىنى ئەڭ تېز سۈرئەتتە ھوزۇرىڭلارغا سۇنۇشتا كېچىلەپ فىلىم ئىشلەپ مانا ئەڭ ئاخىرقى مۇسابىقىنىمۇ ئوڭۇشلۇق تۈگىتىۋالدۇق .
گەرچە بۇ جەرياندا نۇرغۇن يېتەرسىزلىكلەر بولسىمۇ  يوقلىقتىن بارلىققا ،  ئاددىيلىقتىن مۇرەككەپلىككە كەلگەن مۇسابىقە فىلىملىرىنى سۈرەتكە ئېلىش ۋە تارتىش خىزمىتىمىزنىڭ نەتىجىسىدە كۆپچىلىكنىڭ نۇرۇنلىغان ياخشى باھالىرى ۋە كۆڭۈل سۆزلىرىگە ئېرىشتۇق . دەرۋەقە بىزگە ھىچقانداق ئىقتىسادي كىرىم كىرمەيدىغان بولسىمۇ ئەمما بىز پۇتبۇل يەنى دىيارىمىز پۇتبۇلچىلىقنىڭ كەلگۈسى ئۈچۈن موھىم خىزمەتلىرىمىزنى تاشلاپ مۇسابىقە ئۈچۈن كۈچ چىقاردۇق .
بۇ يەردە بۇ گەپلەرنى دەپ ئولتۇرۇشۇمدىكى مەقسەت ، بۇندىن كېيىن يەنە پۇرسەت بولسا دىيارىمىز پۇتبۇلچىلىقى ئۈچۈن بارلىقىمىزنى ئاتايمىز ، ھەمدە كۆپچىلىكنىڭمۇ پۇتبۇلدىن ئىبارەت ئىرادىنى سىنايدىغان ، كىشىگە ئىشەنچ ئاتا قىلالايدىغان ، دۇنيادىكى ئەڭ ئۇلۇغ بۇ تەنھەرىكەتنىڭ مىللىتىمىز ئارىسىدا تېخىمۇ ئوموملىشىشى ۋە ئەۋلادلىرىمىزنىڭمۇ مۇشۇ ساھەدە چېنىقىپ ئىرادىلىك ، ۋىجدانلىق ياشلار بولۇپ چىقىشى ئۈچۈن سىزنىڭمۇ قولىڭىزدىن كېلىشىچە كۈچ چىقىرىشىشىڭىزنى ، پەرزەنتلىرىڭىزنىڭ پۇتبۇل قىزغىنلىقىغا سوغۇق سۇ سەپمەي قوللاپ بېرىشىڭىزنى ئۈمىد قىلىمەن !
!
