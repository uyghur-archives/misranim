    سىز مۇھىتنى ئۆزگەرتەلمەيسىز، بىراق ئۆزىڭىزنى ئۆزگەرتەلەيسىز.

    ياش بولغىنىمىز ئۈچۈن بېشىمىزدىن بەزىبىر سەرگۈزەشتىلەرنى ئۆتكۈزىمىز. مەسىلەن، مۇھەببەت، دوستلۇق دېگەندەك. ھېچقانداق ئادەم ھەربىر كۈنىنى مەڭگۈ خۇشال-خۇرام ۋە بەختلىك ئۆتكۈزۈپ بولالمايدۇ. ھېچقانداق ئادەم ئۆزىنىڭ قەيسەرلىكى ۋە ئاجىزلىقىغا بىمالال ھالدا يۈزلىنەلمەيدۇ. سىزنى تاۋلايدىغىنى كەچمىش بىلەن ئازاب-ئوقۇبەت. سىزنى بەختلىك قىلىدىغىنى كەڭ قورساقلىق بىلەن مېھرىبانلىق، سىزنى خاتىرجەم قىلىدىغىنى چۈشىنىش ۋە ئىشەنچ.

    سىز ئەمەلىيەتنى ئۆزگەرتەلمەيسىز، بىراق پوزىتسىيەرىڭىزنى ئۆزگەرتەلەيسىز.

    نېمە ئۈچۈنكىن، بەزى نەرسىلەرنى زادىلا ئۆزگەرىتكىلى بولمايدۇ. بەلكىم، ھەقىقى غايىمىزنى تاپالمىغىنىمز ئۈچۈندۇر. بەلكىم، يىراقتىكى كەم-كۇتىسىز گۈزەللىكنى قوغلاشقىنىمىز ئۈچۈندۇر. بىز ئىلگىرى ماختانغان ئىدۇق، خام-خىياللارنى قىلدۇق، تار دائىرىدىكى نىشانىمىز ئۈچۈن كۈرەش قىلدۇق. ئارقىمىزغا بۇرۇلۇپ قارىغان ۋاقتىمىزدا، نۇرغۇن ئىشلارنىڭ مەنىسىزلىكىنى ھېس قىلىمىز.

    سىز ئۆتمۈشنى ئۆزگەرتەلمەيسىز، بىراق ھازىرنى ئۆزگەرتەلەيسىز.

    ئۆتۈپ كەتكىنى كېتىۋەرسۇن، كەلگۈسى تېخىمۇ ياخشى بولىدۇ، چۈنكى زۆرۈر بولمىغان يۈكنى چۆرۈۋەتكەندە، تۇرمۇش ئاندىن تېخىمۇ گۈزەل بولىدۇ. كىشىلىك ھايات تولىمۇ قىسقا، بىزنىڭ ئۇنى ياخشى ئۆتكۈزمەسلىككە يەنە نېمە ئاساسىمىز بار؟ نۇرغۇن ئىشلار سىزنىڭ قىلىشىڭىزنى كۈتۈپ تۇرۇۋاتىدۇ، ئىنتايىن مۇھىم ئادەملەر سىزنىڭ قەدىرلىشىڭىزنى كۈتۈپ تۇرۇۋاتىدۇ، كەينىڭىزگە قارىماڭ، ئالدىڭىزدىكى دۇنيا تېخىمۇ گۈزەل.

    سىز ئۇلارنى كونتىرول قىلالمايسىز، بىراق ئۆزىڭىزنى باشقۇرالايسىز.

    يول ھەرقانچە ئەگرى-توقاي ۋە تىك بولسىمۇ، ئۆزىڭىزنىڭ يولىنى مېڭىڭ. بىراق چۇقۇم ئالغا قاراپ مېڭىڭ.ھېچكىمنىڭ سىزنىڭ تەقدىرىڭىزنى كونتىرول قىلىش ھوقۇقى يوق، قىيىنچىلىق ھەرقانچە ئېغىر، جاراھەت ھەددى-ھىساپسىز بولسىمۇ، ھەرگىز ئىككىلەنمەڭ ھەم پۇشايمان قىلماڭ. چۈنكى، ئۆتمۈشنى ئەسلىگىنىڭىزدە، سىزنىڭ كۆرسىتىپ بېرەلەيدىغىنىڭىز ئەنە شۇ ئازاپلىق ۋە كۆز يېشى تامچىلىغان يول، سىز ئاۋازىڭىزنى قۇيۇۋېتىپ مەغرۇرانە توۋلىيالايسىز. قاراڭلار، مانا بۇ مەن ئۆزەم ماڭغان يول.

    سىز ئەتىنى بىلەلمەيسىز، بىراق بۈگۈننى تۇتۇپ تۇرالايسىز.

    ‹‹ئورمانلىقتا بىر ئاچال يول بار، مەن قەدەم ئىزى ئازراق بولغان ئاشۇ يولدا مېڭىشنى خالايمەن، شۇندىلا مېنىڭ پۈتۈن ھاياتىم پۈتۈنلەي باشقىچە بولىدۇ››. بىر يولدا ماڭىدىغان ئادەم كۆپ بولۇۋەرسە، ھامان پاتقاق بولۇپ كېتىدۇ. نېمە ئۈچۈن يەنە بىر يولنى ئالماشتۇرۇپ مېڭىپ باقمايمىز؟ شۇندىلا بەلكىم باشقىچە بىر نەتىجە بېرىپ قېلىشى مۇمكىن. ئۆزىڭىزنىڭ بۈگۈنىنى ياخشى تۇتۇپ تۇرالىسىڭىزلا، ئەتىڭىز جەزمەن تېخىمۇ ياخشى بولىدۇ.

    سىز ھاۋارايىنى ئىدارە قىلالمايسىز، بىراق كەيپىياتىڭىزنى تەڭشىيەلەيسىز.

    سىز تۇرمۇشقا كۈلۈپ باقسىڭىز تۇرمۇشمۇ سىزگە قاراپ كۈلۈمسىرەيدۇ. بىز كۆڭلىمىزنى ئەمدى بوغۇۋالمايلى، ئەمدى ئۇ ئەركىنلىككە چىقسۇن. ئەركىنلىككە ئېرىشكەن قەلبلەر پەرۋاز قىلىپ، گۈزەل قۇياش نۇرىنى كۈتۈۋالسۇن. سىز چىراينى تاللىماڭ، بىراق سىز ئاشۇ كۈلۈمسىرەشلىرىڭىز بىلەن مىڭلىغان، ئونمىڭلىغان كىشىلەر بىلەن ئۇچرىشىپ تۇرالايسىز. ئۇلاردىن ھاياتىڭىزدا ھەقىقىي سىزنى بىلىدىغان ھەمراھىڭىزنى تاپالايسىز، ھەقىقىي ئىشەنچىلىك دوستلىرىڭىزنى تاپالايسىز، مانا بۇ بەخت. مەيلى ھەرقانداق ۋاقىتتا بولمىسۇن، مەيلى سىز نىمىلا قىلماڭ، مەيلى قانداق مۇھىتقا دۇچ كەلمەڭ، سىز شۇنى بىلىڭكى، شۇنداق بىر ئادەم بار، ئۇ سىزنىڭ ئۆيگە قايتىشىڭىزنى ھەر ۋاقىت ساقلاپ تۇرىدۇ. شۇنداق بىر ئادەم باركى، ئۇ سىز بىلەن بىرگە سىزنىڭ خۇشاللىقىڭىز ۋە ئازابىڭىزدىن تەڭ بەھرىمەن بولالايدۇ، سىزنىڭ قەيسەرلىكىڭىز ۋە ئاجىزلىقىڭىزنى كۆرۈپ تۇرالايدۇ. مانا بۇ كۆڭۈلنىڭ تەسەللىيسى، شۇنداق تۇرۇپ سىز يەنە بەختلىك ئەمەسمۇ؟
