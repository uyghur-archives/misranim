                                                        بىنا قىلىپ مۇنبەرنى،
                                                        ئات قويدىڭىز <<مىسرانىم>>.
                                                        ئۈمۈدۈم شۇ مۇنبەرنى،
                                                        گۈللەندۈرۈڭ مىسرانىم.
                                                      
                                                                     ھۆكۈمران
                                                       باشقۇرغۇچى ھۆكۈمران،
                                                       ئىسمى سىرلىق ئاڭلىنار.
                                                       ئادىل تۇتساڭ مۇنبەرنى،
                                                       نامىڭ دىلدا داڭلىنار.
                                                                   ئۆل
                                                      <<ئۆل>> دەپ ئاڭلاپ ئىسمىڭنى،
                                                      توۋا دىدىم ئىسمىڭغا.
                                                      سىنى قاغاش نىيتىم يوق،
                                                      دەرت كەلمىسۇن جىسمىڭغا.
                                                                   نۆل
                                                     ئىسىم يوقتەك جاھاندا ،
                                                     قويۇۋاپلا بىر نۆلنى.
                                                     يوقاپ كەتلە مۇنبەردىن،
                                                     تاپلىمۇ يا بىر چۆلنى
                                                                  كەتىرەن
                                                     ئىسمىڭىزنى ئوقۇسام،
                                                     ھەيران بولۇپ قالىمەن.
                                                     گۈل تۇتسام گەر دوسلارغا،
                                                     كەترەن گۈلى ئالىمەن.
                                                                 كەلگۈندى
                                                     قەيەرلەردىن كەلگەنسەن،
                                                    مىسرانىمغا كەلگۈندى.
                                                    ئۇزۇن بولدى يەرلەشتىڭ،
                                                    ئەمدى دىمەي<<كەلگۈندى>>
                                                                بوغالتىر
                                                    كەسپىڭ بىلەن ئىسمىڭنى،
                                                    بىر ئاتاپسەن بوغالتىر.
                                                    مىڭنى يۈز دەپ كۆرسەتمە!
                                                    چۈنكى سەن بىر<<بوغالتىر>>.
                                                                غوجىدار
                                                    غوجىدار دەپ تولىسام،
                                                    ھە دىمەيسەن غوجىدار.
                                                   نىمىشقا دەپ سورىسام،
                                                   يۇرتۇڭ ئىكەن<<غوجىدار>>
                                                               بايخىنىم
                                                   باي خىنىمنىڭ ئەي دوسلار،
                                                   قانچىلىكتۇر بايلىقى.
                                                  كونا ئەزا مۇنبەرگە،
                                                  1000سوم چىقماس بايلىقى.

                                                 <<ئىلى>>دىكى ئۇرۇشتا
                                                  قۇربان بولغان<<رىزۋانگۈل>>

                                                  ئەسكە سالار ئۆتمۈشنى.

                                                  ساقلاپتىكەن يەتمىشنى.

                                                  ئالتۇن مەن دەپ داڭلايسەن.

                                                  زەنجىركەبى باغلايسەن.
