　　　　دۇنيادىكى ئەڭ كۈچلۈك سىرلىق تەشكىلات فرې مىسىن ھەم ئۇلارنىڭ زور كۆلەملىك ئېرىق تازىلاش پىلانى
　　　
　　　فرى مېسىن  تەشكىلاتى دۇنيادىكى ئەڭ زور ،ئەڭ سىرلىق ئەڭ كۈچلۈك مەخپىي تەشكىلات غەرب ئەللىرىدىكى  ھەر ساھەدىكى ئەڭ سەرخىل كىشىلەر(ئېلىم-پەن ئارقىلىق دۇنيانى كونترول قىلغىلى بولىدىغانلىقىغا شەكسىز ئىشىندىغان كىشىلەر)  ئۇنىڭغا ئەزا بولغان بولۇپ  ئۇلار بۇ تەشكىلات ئارقىلىق يېقىنقى ئۈچ يۈز  يىلنىڭ ياقى دۇنيادا يۈز بەرگەن ھەرقايسى چوڭ ۋەقەلەرگە قول تىقىپ كەلمەكتە .مەسلەن ، 18-ئەسردىكى ئەنگىلىيەدىكى سىياسىي ئۆزگىرىش،ئامېرىكا دۆلىتىنىڭ قۇرۇلۇشى،فىرانسىيە ئاقارتىش ھەرىكىتى،ناپالىئون ھاكىمىيىتىنىڭ ئاغدۇرۇلۇشى،ئىتالىيە مۇستەقىللىق ھەرىكىتى،ئىككى قېتىملىق دۇنيا ئۇرۇشى،1948-يىلىدىكى ئىسرائىلىيە دۆلىتىنىڭ قۇرۇلىشى،11-سىنتەبىر ۋەقەسى،ئېراق،.ئافغانىستان ئۇرۇشى،جوڭگۇدىكى سارىس تارقىلىشچان زۇكىمى،2008-يىلىدىكى پۇل-مۇئامىلە كىرزىسى دېگەندەك.فرې مىسىننىڭ ئىش-پائالىيىتىنى ۋاراقلىغاندا پىلان ئىچىدىكى پىلان،سۈيقەست ئىچىدىكى سۈيقەستتىن كىشنىڭ  ئەقلى لال بولۇپ قالىدۇ. فرې مىسىننىڭ سىياسىي پائالىيىتىنى ئېچىپ تاشلاش دۇنيا يېقىن زامان تارىخىنى قايتا يېزىشقا مەجبۇر قىلىدۇ.
　　　فرې مىسىن ئىلگىرى  ساناقلىق ئۇيغۇرچە مەنبەلەردە خەنزۇچە 共济会 نىڭ مەنە تەرجىسى بويىچە «ئورتاق ياردەم تەشكىلاتى» دەپ ئېلىنغان،ئەمەلىيەتتە فرې مىسىننىڭ ئىنگىلىزچە ئاتىلىشى freemason بۇنىڭ تەرجىمىسى ئەركىن تاشچى ،بۇ تېمىدا تەشكىلات ئىسىمى ئاھاڭ تەرجىسى بويىچە ئېلىندى.
　　　فرې مىسىننىڭ  ئالاھىدە، غەيرىي دىنىي ئىتقادى (كاتولىك دىنى،پراۋوسلاۋىيە دىنى ،خىرىستىئان دىنى)ئىشىندىغان خىرىستوس بىلەن پەقەت ئوخشىمايدۇ.فرې مىسىن ئىشىنىدىغىنى يەھۇدىي دىنى،شەيتان دىنى(دۇرۇىدىزىم)
　　　
　　　تەشكىلاتنىڭ كېلىپ چىقىشى: فرې مىسىننىڭ ئىككى تارىخىي مەنبەسى بار بىرىنچىسى:قەدىمىي ئوتتۇرا شەرق، «فرى مېسىن ئاساسىي قائىدىلىرى»(ئاڭلاشلارغا قارىغاندا 1701-يىلى يېزىلغان،1723-يىلى نەشر قىلىنغان)نىڭ بىرىنچى قىسمى «تارىخ»تا تەشكىلات تارىخىغا بەرگەن چۈشەندۈرۈشتىن قارىغاندا،تەۋرات قىسىلىرىدە بايان قىلىنغان توپان بالاسىدىن كېيىن دۇنيادىكى ھەممە كىشى بىرلا خىل تىل ،بىرلا خىل ئاھاڭدا سۆزلەيدىغان بولغان،نوھنىڭ ئەۋلادلىرى ئۈزلۈكسىز كۆپەيگەن،پۈتكۈل دۇنياغا تارقالغان.
　　　كېيىن  ئۇلار بىر شەھەر ھەم  مۇنار قۇرماقچى بولغان ،ئۇلار  ئۆز-ئارا ئىتتىپاقلىشىپ ئىشلەش نەتىجىسىدە ئۇلار قۇرغان بۇ مۇنار ئاسمانغا تاقىشىپ تەڭرىنى چۆچۈتىۋەتكەن.تەڭرى ئىنسانلارنىڭ شۆھرەتپەرەسلىكى ھەم  ھاكاۋۇرلىقىدىن غەزەپلەنگەن،بۇ كىشىلەرنى جازالىماقچى بولغان،ئۇ :ئەگەر ئىنسانلار راستىنلا بىرلىشىپ ئاسمانغا تاقىشىدىغان مۇنار تىكلىيەلىسە كېيىن ئۇلار قانداق ئىشنى قىلالمايدۇ،چوقۇم ئامال تېىپپ ئۇلارنى توختۇتۇش كېرەك دەپ قاراپ ئىنسانلارغا ئارىسىغا يوشۇرۇن كېلىپ ئۇلارنىڭ تىلىنى قالايمىقانلاشتۇرۇۋەتكەن.ئىنسانلار شۇنىڭ بىلەن ھەرتەرەپكە تارقىلىپ كەتكەن،مۇنار تىكلەش ئىشىمۇ يېرىم يولدا توختاپ قالغان.
　　　بۇ كىشىلەرنىڭ ئارىسىدا پەقەت تاشچىلارلا ئىنسانلار ئىلگرى ئېرىشكەن بىلىملەرنى،تەبىئىي پەن ھەم گېئومتىرىيە سىرىنى ساقلاپ قالغان،بۇ بىلىملەرگە ئاساسەن ئۇلار ئىنسانلارنىڭ پەقەت تەڭرىنىڭ «مۇكەممەل بولمىغان كۆپەيتمە»سى ئىكەنلىكىنى ئېچىپ تاشلىغان.تاشچىلار ئەگەر ئۆزى تىرىشسا،ئىنسانلارنىڭ روھى ھەم تېنىدىكى كەمچىللىكلەرنى يەڭگىلى بولىدىغانلىقىنى بايقىغان.
　　　تاشچىلار بۇ ئۇلۇغ بىلىمنى ئاشكارىلىماسلىققا قەسەمياد قىلغان،ئۇلار مەخپىي تەشكىلات بولۇپ شەرتلىك بەلگە،مەخپىيەتلىكلەر ئارقىلىق ئۆز-ئارا تونۇشقان،ھەمدە تەشكىلات ئىچىدە دەرىجە،ئىش تەقسىماتى بېكىتكەن.بۇ «ئەركىن تاشچىلار»ئېرۇسالىمدا سۇلايمان پادىشاھنىڭ ئوردىسىنى ياسىغان،ئۇلار قەدىمىي گىرىتسىيەدە «قۇرۇلۇش تەشكىلاتى»دەپ ئاتالغان،ئۇلار ھەم ئۇلۇغ قۇرۇلۇشلارغا چوقۇنغان،ئاسترنومىيە ھەم ئاناتومىيەگە پىششىق بولغان،ئۆز-ئارا «بۇرادەر»دىيشىكەن.

　　　يەنى ئاساسىي لىنىيە ئېراننى غەرب دۆلەتلىرى بىلەن توقۇنۇشۇشقا سۆرەپ كىرىش،ئاندىن جوڭگۇغا ياردەم بېرىش.بۇ غىدىقلاش ئارقىلىق مەيلى جوڭگۇ ۋە ياكى ئېران بولسۇن مەلۇم خىلدىكى يادرو قورال ئىشلىتىدۇ،بۇ تەرەپتىن گۇمانلىنشقا بولمايدۇ،چۈنكى ئېراندا مول مىقداردىكى نىفىت بار،ئۇ جوڭگۇنىڭ ھاياتلىق مەنبەسى.ھەم شۇنداقلا جوڭگۇنىڭ دۇنيادىكى تەسىر كۈچىنى يەنىمۇ كۈچەيتىشىنىڭ ئاساسى.
　　　ئاخباراتچىنىڭ  ئېيتىشىچە بۇ ئۇرۇشنىڭ ئەڭ ئاخىرقى مەقسىتى دۇنيا نوپۇسىنىڭ 50％—90%نى ئازايتىش،دۇنيا نوپۇسىنى 500 مىليونغا چۈشۈرۈش .ئۇ يىغنىدا قاتناشقۇچىلارنىڭ بۇساننى ئېنىق دېگەنلىكىنى ئاڭلىغان.
　　　بۇ قورقۇنچلۇق ئەھۋال پىلانلانغىلى كۆپ يىل بولدى.
　　　5) 2008-يىلى ئامېرىكىدا يۈز بەرگەن پۇل-مۇئامىلە كىرزىسى ھەرگىزمۇ تاسادىپىيلىق ئەمەس،ئۇ پىلانلىق قىلىنغان ئىش.بۇ قېتىملىق كىرزىس،دۇنيا ئىقتىسادىي بايلىقىنى مەركەزلىك كونترول قىلىش ئۈچۈن. بىز 2008-يىلىدىن باشلاپ كۆرگەن ئىشلارنى ئۇلار 2005-يىلى 6-ئايدىلا پىلانلاپ بولغان.

　　　فىرې مىسىن ئۆتكەن نەچچە ئون يىل ئىچىدە قىلغانلىرىنىڭ ھەممىىسىنى مەخپىي پىلانلىغان. قايسى «مۇۋاپىق»نامزات دۇنيا مەدەنىيىتىگە ۋارىسلىق  قىلىدۇ-ئاق تەنلىكلەر ،بۇ نېمە ئۈچۈن بۇ تۈرنىڭ  ئانگلو ساكسون پىلانى دەپ ئاتىلىدىغانلىقىنى چۈشەندۈرۈپ بېرىدۇ.
　　　ھاۋا،سۇ،ئېنىرگىيە،تۇپراق بۇ ئىنسانلارنىڭ ھاياتىدىكى تۆت ئاساسىي ئامىل.غەربنىڭ كەلگۈسىشۇناسلىرى يەرشارىنىڭ ئەڭ كۆپ كۆتۈرۈشچانلىقىنىڭ ئون مىليارد ئىكەنلىكىنى،يەرشارىنىڭ ئون مىليارددىن ئېشىپ كەتكەن نوپۇس ،مۇشۇ ئاساسىي ساندىكى ھەر يىلى كۆپىيدىغان نوپۇسقا بەرداشلىق بېرەلمەيدىغانلىقىنى بىلدۈرگەن. ئىنسانلار ئۈچۈن ئېيتقاندا 21-ئەسرنىڭ ئالدىنقى 50يىلى ئىنسانىيەت تارىخىدىكى چوڭ بۇرۇلۇش نۇقتىسى.ھازىر ھەريىلى 100مىليونلۇق سۈرئەت بىلەن كۆپىيىۋاتىدۇ
　　　فرې مىسىنر ھەر 115000يىلدا يەرشارىدا چوڭ گىئولوگىيىلىك ھادىسە يۈز بېرىدىغانلىقىغا ئىشنىدۇ. بىز يەرشارىنىڭ چوڭ گىئولوگىيىلىك ھادىسىلەر يۈز بېرىدىغان دەۋرىگە كىردۇق.گىئولوگلار كىشىلەرگە-بۇخىل گىئولوگىيىلىك ھادىسىلەرنىڭ ھەر 11500يىلدا يۈز بېرىدىغانلىقىنى دەپ بەردى.ھازىر ئۆتكەن قېتىم يۈز بەرگەن ۋاقتقا دەل 11500 يىل بولدى 2012-يىلى ئېنىق يۈز بەرمىسىمۇ لېكىن كەلگۈسى 10يىل ئىچىدە يۈز بېرىشى مۇمكىن
　　　كۆپ  يىل ئىلگىرى ئۇلار دېنۋىر شەھىرىنىڭ ئەتراپىدىكى تاغ ئاستىدا زور يەر ئاستى قەلئەسى ياسىغان .بۇنى ياسىغان پۇل ئۈچ پاناما قانىلىنى ياسىغىلى بولىدۇ. ئۇنىڭ كۆلىمى 27 قەۋەتگە باراۋەر بولۇپ بەش يۇلتۇزلۇق مېھمانخانىغا ئوخشايدۇ،سۇ يېمەكلىك تۇرمۇش بۇيۇملىرى تەلتۆكۈس،كۆڭۈل ئېچىش ئىشخانا دېگەندەكلەرنىڭ بىرسىمۇ كەم ئەمەس،تاشقى دۇنيادىن ئايرىلىپ ئالتە يىل تۇرۇشقا بولىدۇ.فرې مىسىن تەشكىلاتى ئەزالىرى،غەرب دۆلەتلىرىنىڭ رەھبەرلىرى،پۇلدارلار مۇشۇ يەر ئاستى ئوردىغا كىرىۋالىدۇ.گىئولوگىيىلىك ئاپەت ئۇرۇش ئاپەت ئاخىرلاشقاندىن كېيىن ،ئۇلار ئۇ يەردىن چىقىىپ ،نورۋىگىيەدىكى سوغۇق ئامباردا ساقلانغان ئۇرۇقلارنى قايتىدىن ئۆستۈرۈپ يېڭى ھاياتنى باشلايدۇ.
　　　ھەممە ئادەم شىمالىي ياۋروپادىكى ئۇرۇق ئامبىرىنى بىلىدۇ-بۇبىر ئاشكارا خەۋەر-دۇنيادىكى بارلىق ئۆسۈملۈكلەر ھەم ھايۋانلارنىڭ ئۇرۇقى،نورۋىگىيەنىڭ شىمالىدىكى ئامبارغا چوڭقۇر كۆمۈلگەن.غەرب نۇرغۇنلىغان ئالدىنى ئېلىش تەدبىرلىرىنى قوللاندى، دۇنيا بايلىقىنى ئاپەتتىن كېينكى 500 مىليون كىشنىڭ ئىشلىتىشىگە يېتىدۇ
　　　
　　　
　　　بۇ يېڭى دۇنيا باياناتنىڭ نۇقتىلىق يېرى دۇنيا ئاھالىسى 500 مىليونغا ئازايتىش،لېكىن ھازىر دۇنيادىكى نوپۇس 7مىليارد،بۇ بىر زور پىلان،دىمەك،6مىليارد 500مىليونغا يېقىن ئادەم تىرىك قالالمايدۇ،50%نى تازىلاش پەقەت دەسلەپكى قەدەم
　　　بۇنىڭ ئۈچۈن ئاۋۋال نوپۇسى ھەممىدىن كۆپ جوڭگۇ مىللەتلىرى تازلىنىشى كېرەك.چۈنكى ئۇلار  يڭى دۇنياغا «بىز  ۋارىسلىق قىلىمىز»،«ئۇلار ئەمەس».دەپ قارايدۇ. غايەت زور نوپۇس ئاساسى بولغانلىقى ئۈچۈن،جوڭگۇ ئىقتىسادى ھەم سىياسىتىنى پارچىلاش بىر .ئۇزۇن ھەم ئازابلىق جەريان.فرېى مىسىن ئىككى قول تەييارلىق كۆردى،بىرىنچىسى بۇ تۈزۈلمىنى ئۆز ئىچىدىن پارچىلاش،ئەگەر بۇ كار قىلمىسا كەلگۈسىدە ھەربىي جەھەتتىن پارچىلاش
　　　. سارىسنى ئويلاپ بېقىڭ (كېسەل بولغان ھەم ئۆلۈپ كەتكەن كىشىلەر قايسى ئېرىقتىن)،يەنە 2007-يىلىدىكى قارا سەيشەنبە(جوڭگۇدا ئىسلاھات ئىشكنى ئېچىۋىتىشتىن بۇيان يارىتىلغان بايلىقنىڭ سەككىزدىن بىرى غايىپ بولدى)بۇلارنىڭ ھەممىسى پەقەت سىناپ كۆرۈپ بېقىش،دۈشمەنلەر ئۈچۈن ئېيتقاندا پەقەت ھۆكۈمەت پارچىلانسىلا ھەممىنىڭ  پارچىلىنىشى تېز بولىدۇ. بىل رىيان ھەم جېسسىي ۋىتۇرا فرې مىسىننىڭ ئىچكى قىسمىنى زىيارەت قىلىش ھەمدە ئۇزۇن مەزگىل ئىز قوغلاپ تەكشۈرۈش ئارقىلىق ئۇلارنىڭ نوپۇسنى تازىلاش ئۇسۇلىنى ئاشكارىلىدى :بۇلار ۋىروس ،بىئولوگيىلىك ئۇسۇل ھەمدە زەھەرلىك يېمەكلىك ،ئۇرۇش ئاپىتى
　　　جوڭگۇنىڭ 1مىليارد 300مىليون نوپۇسىنى يوقۇتۇش،بۇ قەدەم يۈرگۈزۈلۈۋاتقىلى ناھايىتى كۆپ يىل بولدى.بۇنىڭدا ئاساسلىق جوڭگۇنىڭ دېھقانچىلىقى ھەم يېمەكلىك سانائىتىنى كونترول قىلىش،جوڭگۇلۇقلارنىڭ يېمەكلىكىنى زەھەرلەش،جوڭگۇدا كۆپلەپ كۆرۈلۈۋاتقان زەھەرلىك يېمەكلىك،زەھەرلىك ئىچىملىك ھادىسىلىرى ئەمەلىيەتتە بۇ پىلاننىڭ ئىپادىلىنىشى،بۇ ئاساسلىق جوڭگۇلۇقلارنىڭ تەن ساپاسىنى تۆۋەنلىتىش، ،ئۆلۈش نىسبىتىنى ئاشۇرۇپ،ئۆمرىنى قىسقارتىش ئۈچۈن.بىرنەچچە يىل بۇرۇنقى سەنلۇ سۈت پاراشوكى ۋەقەسى بۇلتۇرقى زەھەرلىك سۈت پاراشوكى ۋەقەسىدىن كېين جوڭگۇلۇقلار دۆلەت ئىچىدە ياسالغان سۈت مەھسۇلاتلىرىغا ئىشەنمەس بولۇپ كەتتى،بۇ ۋاقتتا سىرلىق ئاسپارتامى قوشۇلغان زەھەرلىك سۈت پاراشوكلىرى ئاللىقاچان دۆلەت دەرۋازىسىنىڭ سىرتىدا ساقلاپ تۇراتتى،ئۇلار جوڭگۇ سۈتچىلىكى كىرزسقا چۆكۈپ قالغان ۋاقتدىن پايدىلىنىپ جوڭگۇلۇق ئىستىمالچىلار تەرىپدىن بۇلاڭ-تالاڭ قىلىندى.
　　　بۇ خىل سۈت پاراشوكىنى ئىشلەتكەندە بالىلارنىڭ يېتىلىشى تولۇق بولماسلىق،كېسەلگە قارشى تۇرۇش كۈچى ئاجىز بولۇش ،ئەۋلاد قالدۇرالماسلىق قاتارلىقلار ئالامەتلەر كۆرۈلىدۇ.
　　　2009-يىلى  H1N1 كەلدى،  دۇنيا سەھىيە تەشكىلاتى  بۇنىڭ  خەتىرى ناھايىتى كۈچلۈك دەپ تەشۋىق قىلدى.رەئىس چىن فىڭفۇجىن جوڭگۇنىڭ بۇ ئىشقا ئالاھىدە كۆڭۈل بۆلىشىنى ئەسكەرتتى،جوڭگۇ سەھىيە مىنسىتىرلىكى ھەمدە مەركىزىي تېلېۋىزىيە ئىستانسى توختىماستىن جىددىي كەيپىيات شەكىللەندۈردى،«ئاقكۆڭۈل»دۇنيا سەھىيە تەشكىلاتى جوڭگۇغا كەڭ كۆلەملىك ۋاكسىنا ئەملەش تەكلىپى بەردى  جوڭگۇلۇقلارنىڭ ئالداش ئۈچۈن ئوباما يەنە بىر مەيدان ۋاكسىنا ئەملەش ئويۇنى ئوينىدى.شۇنىڭ بىلەن جوڭگۇدىكى بارلىق داۋالاش ئورگانلىرىغا بۇ ۋاكسىنا ھەقسىز تارقىتىپ بېرىلدى،ئوقۇغۇچى،ھەربىي،دۆلەت بايرىمى مۇھاپىزەتچىلىرىغا ئومۇميۈزلۈك ۋاكسىنا ئۇرۇلدى.جەمىيەتنىڭ باشقا ساھەلىرىدىكى كىشىلەرمۇ بەس-بەستە ۋاكسىنا ئۇرۇش قوشۇنىغا قېتىلدى،ئاخىرىدا پۈتۈن مەملىكەتتە تەخمىنەن 70مىليون كىشىگە ۋاكسىنى ئۇرۇلدى.بۇ نەتىجە فرې مىسىننى ناھايىتى خۇشال قىلىۋەتتى،ئۇلار  چىن فىڭفۇجىن ھەم جوڭگۇ ھۆكۈمىتىنى تەقدىرلىدى.شۇنىڭ H1N1بىلەن غايىپ بولدى.
　　　بۇلتۇر جوڭگۇ سەھىيە مىنسىتىرلىكى دۇنيا سەھىيە تەشكىلاتىنىڭ قۇترۇتىشى ئاستىدا جوڭگۇدىكى يۈز مىليون ئۆسمۈرگە قارىتىلغان قىزىلنىڭ ئالدىنى ئېلىش ۋاكسىنى ئەملەش ئېلىپ باردى.سەھىيە مىنىستىرلىكى  بۇنى دۇنيا سەھىيە تەشكىلاتىغا بەرگەن ۋەدىسىنى ئەمەلگە ئاشۇرۇش ئۈچۈن دەپ چۈشەندۈردى،ۋاكسىننىڭ كەلگەن مەنبەسى توغرۇلۇق دېگەن گەپلىرى زىددىيەتلىك چىقىپ قالدى ھەمدە كىشىلەرنىڭ گۇمانىنى قوزغىدى.لېكىن ئەپسۇسلىنارلىقى ھۆكۈمەتنىڭ كۈچى بەك كۈچلۈك،سەھىيە تەشكىلاتىنىڭ فرې مىسىننىڭ بۇيرۇقىنى  ئورۇندىشى بەكلا كەسكىن بولۇپ كەتتى،يەنىلا 50مىليون ئۆسمۈرگە گۇمانلىق قىزىل ۋاكسىنىشى ئەملەندى،بۇ ئەمەلىيەتتە پۈتۈن جوڭگۇلۇقلار ئۈچۈن تولدۇرۋالغۇسىز زور يوقىتىش .
　　　
　　　ئۇ ئاللىقاچان جوڭگۇنىڭ ھەرقايسى ساھەلىرىگە سېڭىپ كېردى،ئۇلارنىڭ مەقسىتى دۇنيا نوپۇسىنىڭ ئۈچتىن ئىككى قىسىمىنى يوقۇتۇش ئۇلارنىڭ نەزىرىدە كۆپلىگەن كىشىلەر پەقەت يەرشارىدىكى ئاز بولغان بايلىقلارنى ئىسراپ قىلىۋاتىدۇ.بۇ كىشىلەر (ئۈچتىن ئىككى قسىم)ئەخلەت داشقال.
　　　(بەلكىم نۇرغۇن كىشىلەرنىڭ سوئالىغا جاۋاب مۇشۇ يەردە تېپىلىشى مۇمكىن،نېمە ئۈچۈن ئافرىقا رايونى شۇنچىۋالا ياردەمنى قۇبۇل قىلىپ تۇرۇپ يەنىلا نامراتلىق ھەم كېسەلدىن نېرى بولالمايدۇ؟ئەگەر شۇ ياردەم بەرگۈچى غەرب دۆلەتلەر ئۇلار كونتروللىقىدىكى تەشكىلاتلارنىڭ ھەقىقەتەن ئۇ يەردىكى كىشىلەرگە ياردەم بېرىشى نىيىتى بولغان بولسا بەلكىم ئەھۋال ھازىرقىدەك ئېغىر بولماسلىقى مۇمكىن ئىدى،بەلكىم ئافرىقىدا ئېرىق تازلاش پىلانى ئۇزۇندىن بۇيان ئېلىپ بېرىلىۋاتقان بولۇشى مۇمكىن،بىل گەيتىس،بافىت ئەمەلىيەتتە دۇنياغا ياردەم بېرىش ئەمەس بەلكى ئېرىق تازلاش پىلانىنىڭ ئاۋانگارتلىرى)
　　　جەنۇبىي ئامېرىكا،ئافرىقا ھەم ئاسيادىكى باشقا دۆلەتلەرنىڭ تەقدىرى بۇ پىلاندا ئاشكارىلانمىغان.مۆلچەرلىنىشىچە،بۇ دۆلەتلەر ۋاقتىنچە ساقلاپ قېلىنشى مۇمكىن،لېكىن ئۇلار ئۇزۇن مەزگىل  مۇستەقىل مەۋجۇد بولۇپ تۇرالمايدۇ.
　　　
　　　ئومۇمەن، كەلگۈسىدە يۈز بېرىدىغان ئىشلار،ئېران كەلگۈسى 18ئاي ئىچىدە (2011-يىلىنىڭ ئوتتۇرىدىن باشلاپ ھېسابلىغاندا)ئىچىدە ئىسرائىلىيەنىڭ ھۇجۇمىغا ئۇچرىشى مۇمكىن.جوڭگۇ ئېرانغا ياردەم بېرىشى ئارقىلىق ئۆزىنىڭ نىفىت مەنپەئەتىنى قوغدىماقچى بولىدۇ.ئېران ياكى جوڭگۇ ئېغۋارگەرچىلىككە ئۇچراپ بىرىنچى بولۇپ يادرو قورال ئىشلەتكەن دۆلەت بولۇپ قالىدۇ. ،لېكىن ئۇلارنىڭ ئىككىنچى قېتىم غەربكە ھۇجۇم قىلغۇدەك كۈچى يوق.شۇنىڭ ئۈچۈن غەرب دەسلەپتە يادرو قورال ئىشلەتمىسىمۇ بولىدۇ،پەقەت بۇ ئىككى دۆلەتنى تېرىكتۈرۈش ئۇلارنىڭ ئاۋۋال قول شىلتىشقا مەجبۇر قىلىشى كېرەك.ئوتتۇرا شەرقنىڭ كۆپ قىسىم رايونلىرى مۇشۇ سەۋەبلىك پۈتۈنلەي ۋەيران بولىدۇ.قىسقىغىنا ۋاقتتا نەچچە مىليون كىشى ئۆلىدۇ.ئاندىن ئۇرۇش بىر مەزگىل توختاپ قالىدۇ.بۇ ۋاقتتا غەرب دۆلەتلىرى كىشلەرگە قاتتىق ھەربىي باشقۇرۇش ئېلىپ بارىدۇ..ئۇنىڭدىن كېيىن جوڭگۇ بىئولوگىيىلىك قورالنىڭ ھۇجۇمىغا ئۇچرايدۇ.ئۇ يۇقۇملۇق زۇكامغا ئوخشايدىغان نەرسە بولىدۇ.ئۇ پۈتۈن دۇنيادا تارقايدۇ...ئاندىن ئۈچىنچى دۇنيا ئۇرۇشى پارتلايدۇ.
　　　ئامېرىكا ئالەم قاتنىشى ئىدارىسىنىڭ «كۆك نۇر پىلانى»نى ئاڭلىغانمۇ؟
　　　(بۇمۇ فرې مىىسننىڭ رەزىل پىلانلىرىنىڭ بىرقىسمى،مەن مۇھىم دەپ قارىغاچقا تەرجىمە قىلدىم)
　　　كانادا كۋېبېك ئۆلكىسىدىكى داڭلىق مۇخبىر سېرگې مونتاس ئۇ 1990-يىلى «كۆك نۇر پىلانى»نى ئاشكارىلىغان.لېكىن  كېيىن يەنە تەكشۈرۈشكە قاتناشقان بىر مۇخبىر بىلەن ئارقا-ئارقىدىن يۈرەك كېسىلى بىلەن ئۆلۈپ كەتكەن.لېكىن ئىككىلىسىنىڭ ئىلگىرى يۈرەك كېسەللىك تارىخى يوق ئىدى.
　　　ئەڭ كۆپ تارقىلىپ يۈرگەن قاراش،ئامېرىكىدىكى بىرقىسىم فرې مىسىن ئەزالىرى 3D لازىر تەسۋىر چۈشۈرۈش تېخنىكسى ئارقىلىق ،ئاسماندا ھەرقايسى ئەل رەھبەرلىرىنىڭ سۈرىتىنى پەيدا قىلىپ ،خەلق ئارىسىدا ئەنسىزلىك،قورقۇش پەيدا قىلىدۇ.تۆۋەندىكى ۋىدىئودىن ئۇلارنىڭ ھازىرقى تېخنىكىسنىڭ قانچىلىك ئېغىرلىقىنى بىلەلەيسىز

　　　بىرقىتىملىق يەر تەۋرەشتە  ئارخولوگلار يېڭى بىر خارابىنى بايقاپ،دۇنيادىكى ئاساسلىق دىنلارنىڭ ھەممىسىنىڭ بۇرمىلانغانلىقىنى جاكارلايدۇ.كىشىلەرنىڭ دىنغا بولغان ئارىسالدىلىقىنى قوزغايدۇ،ئۇنىڭدىن كېيىن توساتتىنلا بىر كۈنى،دۇنيانىڭ ھەرقايسى رايونلىرىدا ئىلاھنىڭ سۈرىتى كۆرۈنىدۇ،شەرقىي ئاسيادا ساكيامونى،ئوتتۇرشەرقتە ئاللاھ،ياۋروپادا ئەيسا...ھەتتا كومپىيۇتىر ،تېلېۋىزورلار مۇ مۇشۇنىڭ بىلەن تولۇپ كېتىدۇ.
　　　
　　　(كۆك نۇر پىلانى ھەم ئالاقىدار پىلانلارنىڭ مەقسىتى ئۇلارنىڭ نەزىرىدىكى ئەخلەت خەلققە بولۇپمۇ دىنغا ئىشىنىدىغان خەلقنىڭ دىنى ئىتقادىنى سۇسلاشتۇرۇش،ئاۋۋال شۇ كىشىلەرنىڭ دىتىغا ياقىدىغان نەرسىلەرنى پەيدا قىلىدۇ،ئاندىن ئۇنىڭ ساختىلىقىنى ئاشكارىلاپ ئۇلارنىڭ ئوسالغا ئەھۋالغا چۈشۈرۈپ قويىدۇ،ئۇلارنىڭ ئىتقادىنى تەۋرىتىدۇ،بۇنداق ئۇسۇلنى ئىلگىرى ئاڭلىغان،لېكىن كۆك نۇر پىلانى تېخىمۇ ئىلغار،ئەنۇ ئاشۇ ئاتالمىش «مۆجىزىلەر»ئارقىلىق سىزگە نېمە ئىش قىلىش كېرەكلىكىنى ئۆگىتىدۇ،دۈشمەنلەر بىز ئويلىغاندىنمۇ ھېيلىگەر بىز ئويلىغاندىنمۇ كۈچلۈك،ئەزەلدىن كېلىش مەنبەسى ئېنىق بولمىغان «مۆجىزە»لەرگە ئىشىنىپ قېلىشتىن ئۆزۈمنى تارتاتتىم.مەن ئويلايمەن ،بىزنىڭ ئىتقادىمىز قانداقتۇر مۆجىزىلەرگە تايىنىپ قالماسلىقى كېرەك،ئۇ مۆجىزىلەر بولمىسىمۇ  ئاللاھنىڭ ھېكمەتلىرى تولۇپ ياتىدۇ،شۇ  توغرۇلۇق ئويلىنىش كېرەك مانا مۇشنىڭ ئۆزى مۆجىزە.كۆك نۇر پىلانى ماڭا ئۆتكەندە تارقىلىپ يۈرگەن ھەرەمدە كۆرۈنگەن پەرىشتىنى ئەسلىتىپ قويدى،،ئۇ ئىش ئۆتۈپ كەتتى،مەككە ئاللاھنىڭ مۇقەددەس يېرى ،ئۇنداق مۆجىزە كۆرۈلۈش ئېھتىماللىقىنى چەتكىمۇ قاقمايمەن،لېكىن ئەمدى ئىككىنچى بىر يەردە شۇنداق مۆجىزە كۆرۈنۈپ قالسا،ھەرگىز ئىشەنمەيمەن.كۆپچىلىككە دەيدىغىنىم،كۆكنۇر پىلانى پەقەت فرې مىسىن پىلانىنىڭ ئاشكارىلانغان قىسمى ئۇنىڭدىن باشقا يەنە قانداق پىلانلار بار...شۇنىڭ ئۈچۈن بۇنىڭدىن كېيىن يەنە قانداقتۇر ئاتالمىش «مۆجىزە»لەر كۆرۈنسە سەگەك بولايلى،ئۇنىڭ كەينىدە دۈشمەنلەرنىڭ قارا قولى بولۇش ئىھتىماللىقىنى ئويلايلى. بىزنىڭ ئىتقاد-ئىشەنچىمىزنىڭ مۇشۇنداق سېپى يوق نەرسىلەرگە تايىنىپ قېلىشىنىڭ ئالدىنى ئالايلى.
　　　كەلگۈسىدە كۆك نۇر پىلانى ئەمەلىيلىشىپ قالسا،بەلكىم ئاسماندا پەرىشتە ياكى بىرەر پەيغەمبەر كۆرۈنۈشى مۇمكىن،لېكىن تېخىمۇ يامىنى   لېكىن شىنجاڭ شەرقىي ئاسياغا يېقىن بولغاندىن كېيىن بەلكىم ئاسماندا ساكيامونى كۆرۈنۈپ قېلىشىمۇ مۇمكىن......ئەگەر شۇنداق بولۇپ قالسا بىزنىڭ ئارىمىزدىكى «مۆجىزە»گە ئامراق كىشىلەر،شۇنىڭغىمۇ ئىشىنىپ قېلىپ قالمىسىلا بولاتتى..
　　　
　　　فرې مىسىننى تونۇشتۇرۇش  ئاساسەن تۈگىدى،.شەخسەن ئۆزۈم فرې مىسىن ھەم  «ئانگول ساكسون پىلانى»نى بىلگەندىن كېيىن كاللامدىكى ئۇزۇندى بۇيان يېشىلمەي كېلىۋاتقان بەزى نەرسىلەرنىڭ جاۋابىنى تېپىۋالغاندەك بولۇپ قالدىم،.
　　　ئېرىق تازىلاش پىلانىنىڭ مەقسىتى كەلگۈسىدە يۈز بېرىدىغان ئاپەت ئۈچۈن دىيىلسە مېنى دېگەندەك قايىل قىلالمايدۇ،مېنىڭچە ئۇ دۇنيا نوپۇسىنىڭ ھەددىن ئارتۇق ئېشىپ كېتىشىنىڭ ئالدىنى ئېلىش ئۈچۈن بولۇشى مۇمكىن،بۇ مەزمۇننى بىز ئىلگىرى تولۇقسىز جۇغراپىيە كىتابتا ئۆگەنگەن،ئېسىمدە قىلىشچە يەرشارى بەش مىلياردتىن ئارتۇق نوپۇسنى كۆتۈرەلمەيدۇ دېيىلگەنغۇ دەيمەن،ھەممەيلەن دېگۈدەك شۇنى ئوقۇپ ھەدەپ قويۇپ يېتەرلىك دىققەت قىلماپتۇق،لېكىن دۇنيانىڭ ئەڭ سەرخىل كىشىلىرى بۇنىڭ تەييارلىقىنى قىلىپ بوپتۇ.مۇنداق ئويلىساق،ئاق تەنلىكلەردە ئەزەلدىن ئېرىق ئەۋزەللىكى مەۋجۇد،ئۇلارنىڭ يەرشارىنىڭ غېمىنى يەپ يەرشارىنىڭ گۈزەللىكىدىن بايلىقىدىن پايدىلىنىشقا  مۇناسىپ كېشىلەر ئۆزى دەپ ھېس قىلىشى مۇمكىن.يەنە بىر مەسىلە فرې مىسىنىڭ كاتتىباشلىرى يەھۇدىيلا ئۇلارنىڭ ئوتتۇرا شەرققە شۆلگەي ئېقىتىپ يۈرگىنىنى ھەممە بىلىدۇ،يەنە كېلىپ ئوتتۇرا شەرق دۇنيادىكى بايلىق ئەڭ مول جاي،بەلكىم بۇنىڭدا مۇشۇنداق سەۋەب مۇناسىۋىتى بولۇشى مۇمكىن.
　　　بەلكىم بۇ نېمە ئۈچۈن 2012-يىللىق ئاپەت توغرىسىدا سۆز-چۆچەكلەرنىڭ بىراقلا كۆپىيىپ كەتكەنلكىنى چۈشەندۈرۈپ بېرىشى مۇمكىن،ھازىر كىشىلەر گەرچە بۇيىل قىيامەت قايىم بولىدىغانلىقىغا ئىشىنىپمۇ كەتمىسىمۇ لېكىن مەلۇم بىر ئاپەت ياكى بىرنەچچە خىل ئاپەت يۈز بېرىش ،چوڭ ئۆزگىرىش يۈز بېرىدىغانلىقىنى ئېڭىدە قۇبۇل قىلىپ بولدى،بۇ فرې مىىسن بۇيىل ياكى يېقىن كەلگۈسىدە دۇنيادا چوڭ ئۆزگەرتىش ئېلىپ بېرىشى،بۇنى خەلقنىڭ ئىديىسىدىن ئۆتكۈزۈشى  ئۈچۈن پايدىلىق.
　　　 يەنە باشقا نۇقتدىن قارىغاندا راستىنلا شۇنداق گىئولوگىيىلىك ئاپەت يۈز بېرىشىمۇ مۇمكىن،ئۇلار بۇدۇنيادا ئىنتايىن ئاز ئادەمنىڭ ھايات قالالايدىغانلىقىنى بىلىپ يېتىپ ئۆزىنى شۇ كىشىلەر دەپ قارىغان،ھەم شۇنىڭدىن بۇرۇن دۇنيانى ئۆز خالىغىنى بويىچە تازىلىماقچى بولغان بولۇشى مۇمكىن.
　　　ئومۇمەن مەن فرې مىسىننى پۈتۈنلەي ئىنكار قىلمايمەن ھەم شەرتسىز ئىشىنىپمۇ كەتمەيمەن،«دۇنيانىڭ يېڭى تەرتىپى »دېگەن كىتابنىڭ ئاپتورى ناھايىتى توغرا ئېيتقان «فرې مىسىنىڭ سىرلىق خاراكتىرى بىلەن ئېيتقاندا ،ئۇ توغرىسىدا تارقالغان گەپ-سۆزلەرنىڭ چىنلىقىنىڭ ئىسپاتلاش ياكى ئىنكار قىلىش ھەممىسى ئىنتايىن مۈشكۈل ئىش» بۇ ئۇچۇلارنىڭ ھەممىسى تور ئارقىلىق تارقىلىۋاتىدۇ،بەلكىم فرې مىسىن ئۆزى مۇشۇنداق خەۋەرلەرنى تارقىتىش ئارقىلىق بىزنىڭ دىققىتىمىزنى  ئۆزىنىڭ ھەقىقىي مۇددىئاسى ھەم ھەرىكىتىدىن بۇراۋاتقان بولۇشى ۋە ياكى ئۇلارنىڭ سرلىق پەردىسىنى ئېچىشقا تەقەززا بولغان كىشىلەر بەزى شەخىس ھەم نەرسىلەردىكى ئورتاقلىقلارنى تېپىپ چىقىپ ئۇنىڭغا قارا چاپلاۋاتقان بولۇشىمۇ مۇمكىن.، مۇنداقچە ئېيتقاندا مەن  بۇ  سۈيقەستلىق تەشكىلاتنىڭ چىنلىققا ئۇيغۇن ئىكەنلىكىنى توغرا تاپتىم لېكىن، يەنە بۇنداق ئېچىنىشلىق  ئەمەلىيەتنى قۇبۇل قىلىشتىن ئۆزۈمنى تارتىۋاتىمەن شۇنىڭ ئۈچۈن ئۆزۈمگە يەنە  مەلۇم مۇمكىنچىلىك قالدۇرۇپ قويۇشنى لايىق تاپتىم،دۇنياۋى ۋەقەلەر كەينىدىن فرې مىسىننىڭ قولىنىڭ بارلىقىنى ئىزدەيمەن،مەن ئويلايمەن ئەگەر يۇقىرىدا بايان قىلىنغان ھەممىسى راست بولسا مېنىڭ دۈشمىنىمنىڭ كىم ئىكەنلىكىنى بىلمەي ئۆتۈپ كەتكىم يوق.
　　　بۇ ئەقىللىق كىشىلەر ئۇزۇندىن بۇيان ئۆزىنىڭ كۈچ-قۇدرىتىگە تايىنىپ نېمنىڭ چوتىنى سوققان بولسا شۇنىڭغا دەل ئېرىشىپ كەلدى،لېكىن ئەگەر ئۇلار ھەقىقەتەن پۈتۈن دۇنيا ئاھالىسنىڭ 90%دىن ئارتۇقىنى قىرىپ تاشلاشنى پىلانلىغان بولسا راستنلا ئېغىر خاتالاشقان بولىدۇ،بۇ دېگەن خۇداغا قارشى چىققانلىق بۇ كىشىلەر ئاددى،ئاجىز بولسىمۇ ھەممىسى ئاللاھنىڭ بەندىسى ...ئاللاھ ھەرگىز بەندىلىرىنىڭ بۇنداق قىرغىن بولۇپ كېتىشىگە قاراپ تۇرمايدۇ،بۇ ئۆزىنى بىلمەي قالغان كىشىلەرنىڭ بۇ نۆۋەت يېڭىلىدىغانلىقىغا ئىشەنچىم كامىل
　　　مەن فرې مىسىنگە قارشى تۇرىمەن لېكىن مېنىڭ قارشى تۇرۇش كومىېتىتىغا ئادەم يىغىۋاتقىنىم يوق ئەگەر  يۇقىرىقىلارغا ئىشەنمىسىڭز،ئىنكار قىلماقچى بولسىڭىز ئەگەر چامىڭىز يەتسە قانچە ئىلمى،توغرا ھالدا ئىنكار قىلالىسىڭىز مەن  شۇنچە خۇشال بولىمەن،لېكىن ئىسىڭىزدە بولسۇن،نىشانىڭىز مەن ئەمەس فرې مىسىن.،ئەگەر ئۇنداق قىلالمىسڭىز مەن يەنە بۇ فرې مىسىن توغرۇلۇق ئەستايىدىل ئويلىنىشىڭىزنى ،كەلگۈسى توغرۇلۇق مەلۇم پىلانىڭىز بولۇشىنى ئۈمىد قىلىمەن
　　　ئاندىن يەنە بىر مەسىلە  كىشلەرنىڭ كاللىسىدا ئاسانلا پەيدا بولىدۇ،فرې مىسىن شۇنچە كۈچلۈك بولسا نېمىشقا بۇ سىرلا ئاشكارىلىنىپ كېتىدۇ،بۇ توغرۇلۇق مەنمۇ ئويلىنپ قالغان..ھازىر قارىسام فرې مىسىن ئەزالىرى ئىنتايىن ئەقىللىق كۈچلۈك  بولسىمۇ لېكىن دۇنيانىڭ باشقا ئاھالىلىرى بەك دۆتمۇ ئەمەس،ئۇلار ھەرھالدا ئۆزىنىڭ ئەتراپىدا شۇنداق بىر قارا كۈچنىڭ بارلىقىنى ،بىلىپ تۇرغىدەك كاللىمۇ بار.
　　　ئەمدى،  ھازىرقى ھېسياتنى ئىپادىلەپ بېرىدىغان بىر ناخشا سوۋغا قىلاي،ناخشىنىڭ ئېيتقۇچىسى مايكول جىكسون دۇنيادىكى ئاز ساندىكى فرې مىسىنغا ئاشكارا قارشى چىققان كىشىلەرنىڭ بىرى. نۇرغۇن كىشىلەر مايكول جىكسونغا قەست قىلغان فرې مىسىن  دەپ قارايدۇ، بۇ كىشىلەرنىڭ ئۇ ئۆلۈپ كەتكەندىن ئۇنىڭ ئۆلۈم سەۋەبىنى چۈشەندۈرىدىغان جاي تاپالماي قىلىپ قويغان گېپى ئەمەس،مايكول جىكسون 2003-يىلى يولۇققان دېلودا فرې مىسىننىڭ قولى بار،بۇنى مەن 2006-يىلى ئاڭلىغان،ئويلاپ باقسام شۇ ۋاقتتا تۇنجى قېتىم فرې مىسىننىڭ ئىسمىنى ئاڭلاپتىكەنمەن،ئومۇمەن مايكول جىكسون ئاخىرقى ھاياتىدا ۋەيران بولۇشىدا فرې مىسىن ئاساسىي سەۋەبچى
