نۇرغۇن دوستلار نىڭ كىملىك بىجىرىش توغرىسىدا بىلمەيدىغان جايلىرى بولىشى مۇمكىن شۇنىڭ ئۈچۈن مەن دوستلارغا بۇ تېمىنى يېزىش ئارقىلىق ئازراق چۈشەنچە بېرىپ ئۆتمەكچى
ھازىر كىملىك بىجىرىش ئەڭ تۆۋەن ياش چەكلىمىسى 6 ياش .6 ياشلىق بالىلارنىڭ كىملىكىنىڭ ئىناۋەتلىك ۋاقتى 5 يىل بولىدۇ ،تاكى 16 ياشتىن يۇقىرى ئۆسمۈرلەرگىچە ھەممىسىنىڭ كىملىكىنىڭ ئىناۋەتلىك ۋاقتى 5 يىل بولىدۇ ،16 ياشتىن ي25 ياشقىچە بولغانلارنىڭ كىملىكىنىڭ ئىناۋەتلىك ۋاقتى 10 يىللىق 26 ياشتىن 45 ياشقىچە بولغانلارنىڭ  كىملىكىنىڭ ئىناۋەتلىك ۋاقتى 20 يىل بولىدۇ ،45 ياشتىن يۇقىرلارنىڭ ئۇزۇن مۇددەتلىك بولىدۇ . يۈتۈپ كەتكەن بولسا ،يۈتۈپ كەتكەن كىملىكنى بىجىرگىلى 2 يىل بولغان بولسا چىراي شەكلىڭىزدە ئانچە ئۆزگىرىش بولمىغان بولسا قايتا رەسىمگە چۈشۈشنىڭ ھاجىتى يوق 2 يىلدىن يۇقىرى بولغانلار قايتىدىن رەسىمگە چۈشۈشكە توغرا كېلىدۇ بۇرۇن كىملىك يۈتۈپ كەتسە چوقۇم گېزىتخانىغا بېرىپ ئېلان بېرىپ ،شۇ ئېلاندىن بىر پارچە ئېلىپ ساقچىخانىغا بېرىپ مەسئۇل ساقچىغا كۆرسىتىشكە توغرا كېلەتتى ، ھازىر يەنى 2012-يىلى 4- ئاينىڭ 20 - كۈندىن كىيىن باشلاپ پۈتۈن شىنجاڭ رايونىنىڭ نوپۇس كەسپىنى بىجىرىش سېستىمىسىنىڭ نەشىرىنى يوقىرى كۆتۈرۈپ يېڭى نەشىرىنى ئىشلەتكەندىن باشلاپ  ، كىملىك يۈتۈپ كەتسە گېزىت ئېلانى بېرىشنىڭ ئورنى يوق ،بىۋاستە تۇرۇشلۇق ئورۇندىكى ساقچىخانىغا بېرىپ كىملىكىڭىزنى يۈتەنلىكىنى ھەم قايتىدىن بىرنى تۇلۇقلەپ بىجىرمەكچى ئىكەنلىكىڭىزنى ئېيتىپ ،رەسىمگە چۈشۈپ كىملىك نۇمۇرىڭىزنى دەپ بەرسىڭىز ۋە 40 يۈەن تۆلىسىڭىز تېز بولسا 1 يېرىم ئايدا بولمىسا 3 ئايدا كىملىك قولىڭىزغا تېگىدۇ ، 5 ئايدا تېگىدۇ دىگەن بىكار گەپ ، كىملىك بىجەرسىڭىز جامائەت خەۋىپسىزلىك ئىدارىسىنىڭ بىر تۇتاش بىجىرىش ۋاقتى 3 ئاي بولىدۇ ، ھەرئاينىڭ 20 - كۈنىدىن 25-كۈنىگىچە بولغان ۋاقىتتا يېڭى بىجىرىلگەن كىملىك ھەرقايسى ساقچىخانىلارغا تاپشۇرۇپ بېرىلىدۇ ، ھەم ساقچىخانىنىڭ مۇناسىۋەتلىك خادىملىرىمۇ (نوپۇس بۆلۈمىدىكى خادىم) سىزگە 26-چىسىلاغىچە سىزگە تېلفۇن قىلىپ كىملىكىڭىزنى ئېلىۋېلىشىڭىزنى ئۇقتۇرۇش قىلىدۇ . يەنە بىر ئىش ، كىملىككە جىددى ئېھتىياجلىق بولسىڭىز ،كىملىكنى قايتا تۇلۇقلاپ بېرىشنى ئېيتىپ رەسىنگە چۈشۈپ 40 يۈەننى تاپشۇرغاندىن كىيىن قولىڭىزغا بىر دانە تالۇن بېرىدۇ (كىملىك چىققۇچە كىملىك ئورنىدا ئىشلىتىسىز) شۇ تالوننى ئېلىپ ئۆيىڭىزگە يېقىنراق پوچتىخانا ياكى تېز يەتكۈزۈش مۇلازىمەت ئورۇنلىرىغا بېرىپ تېز كىملىك (快证)مۇلازىمىتى بىجىرىمەن دىسىڭىز 80يۈەندىن 100 يۈەنگىچە مۇلازىمەت ھەققى تاپشۇرىسىز ، كىملىكىڭىز بىر ھەپتىدە قولىڭىزغا تېگىدۇ . دېققەت قىلىدىغان مۇھىم ئىش شۇكى ساقچىخانىدىن بەرگەن تالوننىڭ تېز كىملىك كەسپىنى بىجىرىش ۋاقتى پەقەت 4 كۈنلا ! شۇنىڭ ئۈچۈن ئادەتتە قولىڭىزغا تالۇننى ئېلىپلا ئۇدۇل پوچتىخانا ياكى تېز يەتكۈزۈش مۇلازىمەت ئورۇنلىرىغا بېرىپ تېز كىملىك مۇلازىمىتىنى بىجىرىۋېلىڭ بولمىسا    3 ئاي ساقلاشقا توغرا كېلىدۇ . تېز كىملىك كەسپىنى ساقچىخانىدا بىجىرمەيدۇ .قېنى ئەمدى كىملىكى يۈتۈپ كەتكەن دوستلىرىمىز خېلى چۈشەندىڭىز ھەقاچان ؟

قېرىنداشلار مەلۇم كىشى ئۈلۈپ كەتسە شۇ ئادەمنىڭ ئۈلۈپ ك ...

بىر ئادەم ئۆلۈپ كەتسە  ، ساقچىخانا بۇنى بىلسە دەرھال ئۆچۈرمەيدۇ  .
ئۆچۈرۈشنىڭ شەرتى  : ئۆلۈپ كەتكۈچى دوختۇرخانىدا جان ئۈزگەن بولسا ، دوختۇرخانا ئۆلۈپ كەتكەنلىك ئىسپاتى چىقىرىدۇ  ،  ئەگەر ئۆلۈپ كەتكۈچى دوختۇرخانا سىرتىدا جان ئۈزگەن بولسا ، شۇ ئۆلۈپ كەتكۈچىنىڭ  نوپۇسى تۇرۇشلۇق ئاھالە كومىتىت  ، كەنىت  كومىتىتى ، تۈەن مەيدان   ....  ئىسپات چىقىرىدۇ   . ساقچىخانا شۇ ئىسپاتقا ئاساسەن ، ئۆلۈپ كەتكۈچىنىڭ نوپۇسىنى ئۆچۈرۈدۇ  .
ساخچىخانا بۇنىڭغا قارىتا  ئۆلگەنلىك ئىسپاتى چىقىرىپ  ،  ئۆلۈگۈچىنىڭ ئائىلە تاۋاباتىغا بىرىدۇ .
مانا  بۇ  ئۆلگەنلىك   ئىسپاتى .
ساقچىخانا  ئۆلگۈچى ئۈچۈن كىملىك بىجىرمەيدۇ  . پەقەت ئۆلگەنلىك ئىسپاتى بىجىرىدۇ  .
ئۆلگۈچىنىڭ ئائىلە تاۋاباتى ، ئۆلگەنلىك ئىسپاتىغا ئاساسەن ئۆلگۈچىنىڭ ھايات ۋاختىدا بىر تەرەپ قىلىپ بولالمىغان ئىشلىرىنى  ۋاكالىتەن بىر تەرەپ قىلىدۇ  .
ۋاكالىتەن بىر تەرەپ قىلىشتا  ، ھۆكۈمەت گۇۋالىق خېتى بولۇش كېرەك  .

جان تەندە ئامانەت  ، ھەر كىشى  ئۆزىنىڭ ئىشلىرىنى پۇختا بىجىرىشى كېرەك .  ئىشلەتكەن مەخپى نومۇرنى خاتىرىگە يېزىپ قالدۇرۇپ قويۇش لازىم  .  ئائىلە ئەزالىرى بەزى ئىشلارنى بىر تەرەپ قىلىشتا پايدىلىنىدۇ  .
قۇرئان ۋە ھەدىسلەردە ئىسپاتلاندىكى : بىر ئادەم ئۆلسە ، ئۇ بۇ دۇنيادا قىلغان ياخشى بولسۇن يامان بولسۇن ، بارلىق چوڭ – كىچىك ئىشلىرى ئۈچۈن ھىساب – كىتاب قىلنىدۇ   . ياخشى ئىش مۇكاپاتلىنىدۇ ، ئەسكى ئىش جازالىنىدۇ ، ھىساۋاتنىڭ بىرىنچى سەھنىسى قەبرىدە بولۇدۇ  .
