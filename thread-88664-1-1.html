يۈرىكىم ئاغرىدى ،مۇجۇپ ئاغرىۋاتىدۇ ....


          كەچمىشىم : كۆرگەندىن كىيىن دوسلارنىڭ سەمىمىي تەكلىپ بىرىشىنى ئۈمىد قىلىمەن ،قەلەم قۇۋۋىتىڭلارنى ئايىمىغايسىلەر ،رەھمەت :

           دوستۇم بىلەن تونۇشقىلى 7يىل بولدى ،شۇنداق يىقىن ئۆتەتتۇق ،نۇرغۇن كىشىلەر تىخى بىزنى "ئىككىڭلار ئاچا-سىڭىلمۇ ؟"دەپ سورايتتى ،يىقىنلىقىمىزدىن بىر كۈن تىلىفۇن سۆزلىشىپ مۇڭداشمىساق كۆرگىمىز شۇنداق كىپ كىتىدىغان .ھەتتا كىم بالدۇر توي قىلسا يەنە بىرىمىز چوقۇم قولداش بولىمىز دەپ ۋەدىلەشكەن تىخى .تەقدىرنىڭ چاقى چۆرگىلەپ تۇرغاچقىمۇ نۆۋەت مەن بىلەن ئۇنى ،ھەم مىنىڭ بىلەن مۇناسىۋەتلىك يەنە بىر مەن ئۈچۈن مۇھىم كىشى ئۈچىمىزنى سىنىدى :شۇنداق بىر كۈنلەردە تونۇشقىنىمىزغا 3يىل بولغان ،2يىل مۇھەببەتلىشىپ يۈرگەن يىگىتىم تۇيۇقسىز ئۇچۇر يوللاپ ،جىددىي ئىش بارلىقىنى،ئۆزى بىلەن دەرھال ئالاقىلىشىشىمنى ئىيىتتى ،(ئۇ يىگىت بىلەن خىزمەت ئورنىمىز نىڭ بىر شەھەردە بولماسلىقى سەۋەپلىك نۇرغۇن ئىگىز -پەسلىكلەردىن كىيىن ،مەن ئۆزۈمىنى تارتقان ،گەرچە ئۇنىڭغا شۇنداق كۆيسەممۇ ئۇنى ئائىلىسىدىكىلەر ئالدىدا تەڭلىكتە قالدۇرغۇم كەلمىگەن ،ۋاز كىچىشنى ئۆزۇم تاللىغان دىسەكمۇ بولىدۇ ) ئۇنىڭ تىلىفۇن نومۇرىنى قارا تىزىملىككە تارتىۋەتكەنلىكىم ئۈچۈن ئۇنىڭغا تىلىفۇن قايتۇردۇم ،ئۇ مەندىن ئەشۇ 7يىللىق يىقىنىمنى،ئۇنىڭ بىلەن قانچىلىك دەرىجىدە ئىكەنلىكىمنى  سۈرۈشتە قىلىپ ئىزدەپتۇ ھەمدە دوستۇم بىلەن توي قىلىشنى ئويلىشىپ بولغانلىقىنى  ئىيىتتى ،بىشىمدىن تۈتۈن چىكەتتى ،نىمە دەي ؟ئەمدى ئۇنىڭغا ............
ئۆزۇمنى قاتتىق تۇتۇۋىلىپ قاچاندىن باشلاپ تۇنۇشقانلىقىنى سورىدىم ،ئۇ بولغان ئەھۋاللارنى باشتىن ئاخرى قىسقىغىنە سۆزلەپ بەردى،ھەمدە ئۇنىڭ بىلەن توي قىلسا بولىدىغان ياكى بولمايدىغانلىقنى سورىدى (قىزنىڭ ئەخلەق -پەزىلىتنى جەزىملەشتۈرۈش مەقسىتىدە ).مەن توي قىلسا بولىدىغانلىقىنى،ئۇلارغا قولداش بولۇپ تويىنى قىزىتىپ بىرىدىغانلىقىمنى  دىدىم ،ئۇ ماڭا "بىلىمەن ،ئىچىڭ ئىيچىشتى ،لىكىن مەنمۇ توي قىلمىسام بولمايدۇ ،چىدىغىن "دەيدۇ ،گەپلىرىنى ئاڭلاپ تۇرۇپكىنى قويماي شۇنداق تۇرىۋىرىپتىمەن ،ئۇنىڭ ئەينى ۋاقىتلاردا كىچە -كىچىلەپ ماڭا يالۋۇرىشلىرىنى ،توختىماي تىلىفۇن قىلىپ سىغىنغانلىىقىنى ئىيىتقان شۇ كىشى ئىكەنلىكىگە  ئىشەنگۈم كەلمىدى .دوستۇم بىلەن چ چ دا بىر بىرىمىزنىڭ بارلىقىنى بىلىشسەكمۇ بۇرۇنقىغا ئوخشاش چاقچاقلاشماي ،پەرۋامۇ قىلىشمىدۇق ،ئۇنىڭ نىمە دىيىشىنى بىلەلمىگەنلىكىنى ھىس قىلىپ ئاخىرى ئۆزۈم ئاكتىپلىق بىلەن خەت يازدىم ،ئۇنىڭدىن مۇناسىۋىتىنىڭ قاچاندىن باشلانغانلىقىنى سورىدىم ،ئۇ دەررۇ ماڭا تىلىفۇن قىلىپ ،ئىكىسىنىڭ ئۈندىداردا تونۇشقانلىقىنى ،ئىككى ئايمۇ بولمىغانلىقىنى ئىيىتىتى.يىگىتىم ماڭا (ئائىلىسىدىكىلەرنىڭ تونۇشتۇرغانلىقىنى دىگەن ).مەن دوستۇمدىن قانداق قىلىدىغانلىقىنى سورىدىم ،ئۇنىڭ گەپلىرىدىن تويغا قوشۇلغۇسى بارلىقى چىقىپ تۇراتتى ،باشقا بىر نەرسە دىمەي توي قىلساڭ بولىدۇ ،دىدىميۇ ئۇنىڭ بىلەن توي قىلماسلىقى كىرەكىلىكىنى ئىغىزىمدىن چىقىرالمىدىم .(شۇ كۈنگە قەدەر يىگىتىمنىڭ سەمىمىي ئەمەسلىكىنى ،نۇرغۇن ئەيىپىنىڭ بارلىقىنى، پەقەت يۈرەكتىن كۆيگەنلىكىم سەۋەپلىك كۆزۈمىنىڭ يۇمۇلۇپ قالغىنىنى،بوپتۇ دەۋەتكىنىمنى قەلبىمدە ئايان ئىدى ،)دوستۇمنىڭ بەزى گەپلىرىنى ئاڭلىغاندىن كىيىن ئۇنىڭ يەنە توي قىلىش ئالدىدىمۇ  ئۇنىڭ ھىسياتى بىلەن ئوينىشۋاتقانلىقىنى بىلىپ يەتتىم،بىر كىچە ئويلىنىش ئارقىلىق دوستۇمغا بەزى گەپلەرنى ئەسكەرتىشنى توغرا تاپتىم ،ۋە ئەتىسى چۈشلۈك تاماق ۋاقتىدا ئۆزۇم دىىيىشكە تىگىشلىك دەپ قارىغان گەپلەرنى دىدىم ،شۇ جەرياندا دوستۇمغا ئۇنىڭدىن نەچچە قىتىملاپ تىلىفۇن كەلدى ،دوستۇم يا مىنىڭ ئالدىمدا ئۇنىڭ تىلىفۇنىنى ئالالماي شۇنداق بىسەرەمجان بولۇپ كىتىپ باراتتى ،كۆڭلۈمدە ئۇنىڭ ئۈچۈن شۇنداق بىئارام بولىۋاتاتتىم ،دوستۇم ماڭا "ئۇسىزگە  نىمىلەرنى دىمىسۇن ئىشەنمەڭ ،ماڭا چوقۇم ئىشىنىڭ ،ھەرگىز يىنىۋالماڭ "دەپ ئۇچۇر قىپتۇ دىدى . مەن تىخمۇ ئاچچقىتن تولغاندىم ،دوستۇمنىڭ قانداق ئويلىغانلىقىنى سورىسام گىپىنىڭ ئورانىدىن كىمگە ئىشەنسە بولىدىغانلىقى بىلەلمەيۋاتقانلىقى ،ئىككىلىنىۋاتقىنى بىلىنىپ تۇراتتى ،شۇندىلا يەنە گەپ قىلىشىمنىڭ ئارتۇقچە ئىكەنلىكىنى تونۇپ يەتتىم ،ۋە ئۆينى سەن تۇتىسەن ئاداش ،پەقەت سەن ئۈچۈن دىدىم ،ئۇنىڭ قانداقلىقىنى مەن تىخىمۇ بىلىۋاتىمەن ،قالغىنىنى ئۆزۈڭ قارار قىل دەپ گىپىمنى توختاتتىم ،بىز تاماق يەپ بولغاندىن كىيىن پەسكە چۈشتۇق ،يىگىت ئۇنى بىز تاماق يىگەن تاماقخانىنىڭ ئالدىغا ماشىنىسى بىلەن ئالغىلى كەپتۇ ،ئۇ ماشىنىدىن چۈشمىدى ،دەرھال قاراپ ئۇنىڭ ماشىنىسى ئىكەنلىكىنى تونىدۇم ،دوستۇم ئۇنىڭ بىلەن كۆرۈشۈپ بولۇپ ،ئۇنىڭ بىزنى سودا سارايغا ئاپىرىپ قويىدىغانلىقىنى ،بىللە ئايلىنىپ كىلىشىمىزنى ئىيىتتى ،شۇندىلا دوستۇمنىڭ قارارىنى بىلىپ بولدۇم .ۋە ئۆيدە ئىشىمنىڭ بارلىقىنى دەپ خوش دەپلا كىتىپ قالدىم ،يولدا قانداق ماڭغىنىمنىمۇ بىلمىدىم ،مەن يولدا مىڭىۋاتامدىمەن ياكى يول مىنى مىڭىۋاتامدۇ ھىس قىلىپ بولالمايتتىم .
         شۇ كۈندىن باشلاپ دوستۇم بىلەن تا ئون كۈنگىچە يا تىلفۇنلاشمىدۇق ،چ چ دا بار بولساقمۇ خەت يىزىشمىدۇق ،شۇ كۈنلەردە ئۆزۈمنىڭ نىمە ئىشلارنى قىلىۋاتقىنىمنىمۇ بىلمەيدىغان دەرىجىدە خىزمەتنى شۇنداق ئىشلەپ كەتتىم ،پەقەت كەچ بولۇپ يوتقانغا كىرگىنىمدىلا ھەممە ئىش كۆز ئالدىمدا نامايەن بولاتتى ،دوستۇم بىلەن كۈلۈشكەن ،ئوينىغان چاغلار ،ئالىي مەكتەپتىكى شوخلۇقلار ،ياتاقلىرىمىزدىكى  مەنىلىك كۈنلەر  نامايەن بولاتتى ؛ئۇنىڭ (يىگىتىم )بىلەن بارغان جايلار ،ئۇنىڭ كىچە -كىچىلەپ  توختىماي پاراڭ قىلىپ بىرىشلىرى ،بەزىدە قىلىغان بالىلارچە قىلىقلىرى ،قىيداشلىرى ،يەنە چىدىماي ئەتىسىلا تىلىفۇن قىلىشلىرى ......ھەممىسى قايتا -قايتا نامايەن بولاتتى ،ئىچىمنى بىر ئاچچىق تۇيغۇ قىينايتتى ،"يۈرىكىم ئاغرىدى ،مۇجۇلۇپ ئاغرىدى "دىگەننىڭ مەنىسىنى چوڭقۇر چۈشەنگەن ئىدىم .
           دوستۇم 11كۈن بولغاندا تىلفۇن قىلىپ مەندىن ئەھۋال سورىدى ،مەن شۇ "ياخشى ،ئىشلەۋاتىمەن "دىن باشقا گەپنى قىلمىدىم ،ئۇنىڭغا گەپ قىلغۇن كەلمىگەنىدى ،ئاغزىمنى بىر نىمە دىيىشكە زورلىساممۇ ئاغزىمدىن چىقمىدى ،قانداق قىلىمەن ؟  ئەلۋەتتە ،بۇ مىنىڭ ئىختىيارىم ئەمەس ئىدى .......
            كۈنلەر شۇ تەرىقىدە ئۆتىۋەردى ،ئىككىسىنىڭ مىجەزىنىڭ پەقەتلا كىلىشمەيدىغانلىقى ،ئىككىسىنىڭ بۇ تويىنىڭ بولمايدىغانلىقى ماڭا بەش قولدەك ئايان ئىدى ،ئىچىمدە دوستۇمغا قاتتىق ئىچىندىم ،ئۇنىڭ يەتتە يىللىق مىنىڭدەك دوستىنىڭ سۆزلىرىگە شەكلەنگەنلىكىگە جەھلىم قاتاتتى ،يەتتە يىللىق شۇ بىچچارىلىكتە قالغان ئىشەنچىسىز دوسلۇقىمغا ئىچىمدە قاتتىق ئەپسۇسلىناتتىم.شۇ ۋاقىتلارداتونۇشقىلى  ئىككى ئاي بولغان كىشىنىڭ ئۇنى مەندىن تارتىپ كەتكەنلىكىگە پەقەتلە ئىشەنگۈم كەلمەيتتى .
             ئارىدىن يگىرمە نەچچە كۈن ئۆتۈپ دوستۇمنىڭ باشقا بىرى بىلەن توي قىلىشنى دىيىشكەنلىكىنى بىلدىم ،يۈرىكىم تىخىمۇ زەردىلەندى ،دوستۇمغا تىخىمۇ ئاچچىقىم كەلدى ،ئاشۇنى دەپ مۇناسىۋىتىمىزگە (دوسلۇق ئەينىكىمىزنى چۇل چۇل قىلىپ يۈرگىنىڭگە لايىق توي قىلىپلا ۋەتسەڭ بولماسمىدى )قاتتق دەز كەتگۈزگىنىڭگە چۇشلۇق ئىش قىلسان بولماسمىدى دەپ قايناپ قالاتتىم .........تۇرۇپ يەنە سىنىڭ شۇنىڭ قولىغا چۈشۈپ قالمىغىنىڭغا  قەلبىمدىكى يەنە شۇ  دوسلۇق مىھرى ئالقىش بىرىۋاتاتتى ........
             سىنىڭ قايتا -قايتا خاپۇلىما ،ئۇنىڭ بىلەن توي قىلمىدىمغۇ ؟دىيىشلىرىڭ يەنە مىنى قايتىشىمغا زورلايتتى ،قەلبىمنىڭ "ئۇنىڭدەك بىرسى ئۈچۈن سىلەرنىڭ دوسلۇقىڭلارنىڭ ئاخرلىشىشىنى خالىمايۋاتىمەن،بولدى ئەسلىڭلارغا قايتىپ قىلىڭلار "دەپ نىدا قىلىشى مىنى ئويلاندۇراتتى ،مىنىڭمۇ شۇنداقلا ئاخرلىشىشىغا كۆزۈم قىيىمايتتى ،چۈنكى ئارىمىزدا نۇرغۇن ئىسسىق -سوغۇقلار ئۆتكەن ،سىنىڭ دەردلىرىڭ ،مىنىڭ تىڭىرقاشلىرىم بىر -بىرىمىزگە ھەمرا بولغان ئىدى .
             مانا ھازىر شۇ  ئىشلارغىمۇ بىر دەمدە ئۈچ ئايدىن ئىشىپتۇ ،لىكىن كۆرۈشكەن قىتىمىمىز 3قىتىمغىمۇ يەتمەيدۇ ،سىنىڭ شۇ كۈنلەردىن بىرى مەن بىلەن قانچىلاپ  مۇڭداشقۇڭ كەلگەنلىكىنى بىر "ئاللاھ "بىلىدۇ ، بىر سەن بىلىسەن ،بىر مىنىڭ كۆڭلۈم تۇيىيىدۇ ،  لىكىن مىنىڭ سەن پارچە قىلىغان بۇ يۈرىگىم سىنى كۆرسەملا ئاشۇ نامەرتنى يادىمغا سىلىۋاتىدۇ ،ئۇنىڭ بىلەن يۈرگەن ،ماڭا تۇنجى قىتىم چىن مۇھەببەتنى ھىس قىلدۇرغان يەنە شۇ نامەرتنى ئىسىمگە سىلىۋاتىدۇ؟............

          تورداش دوسلۇرۇم ،مىنىڭ كەچمىشىم مۇشۇ يەردە ئاخىرلاشتى ،مەن زادى قانداق قىلسام مۇشۇ مەسىلەمنى ياخشى ھەل قىلالايمەن ؟ ماڭا يول كۆرسەتسەڭلار ...........
