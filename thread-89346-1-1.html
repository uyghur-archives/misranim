يۇرتۇم قاراقاش ئاپتۇنۇم رايۇنىمىزنىڭ جەنۇبىغا جايلاشقان خوتەن ۋىلايىتىنىڭ بىرىنچى چوڭ ناھىيەسى،نۇپۇس سانى 600000  مىڭ ئەتىراپىدا 2009 يىلدىكى نۇپۇسى 550 مىڭ ئىدى،نۇپۇسنىڭ 89 پىسەنتى ئۇيغۇرلار بۇلۇپ ئىسلام دىنىغا ئىتقات قىلىمىز، ھەم  ھازىرغا قەدەر يىتىپ كەلگەن مىللىيچە چېلىشىش،ئوغلاق تارتىشىش،دارۋازلىق سەنئىتى،قەدىمى خوتەن قەغىزىنى ياساش .ئەتلەس ۋە گىلەم ماكانىدۇر،
يۇرتۇمدا يەنە تۇلۇق ئوتتۇرا .تۇلۇقسىز ئوتتۇرا بۇلۇپ 16 يېزا بازاردا كەڭ تارقالغان مەكتەپلەر قۇرۇلغان،يەنە ناھىيە مەركىزىگە ئۈرۈمچىدىكى دۆڭكۆۋرۈك سودا مەركىزىدىن ئىككى ھەسسە ئارتۇق، كۈلىمى ۋە سۈپىتىدە ئېشىپ چۈشكىدەك ھەم قۇيۇق مىللى پۇراققا ئىگە مىللەتلەر كۇچىسى بولسا،
يەنە يان تەرپىدە يىڭىدىن قۇرۇلغان خەلىق مەيدانى بۇ قەدىم دىيارنىڭ ھۆسنىگە - ھۆسۈن قۇشۇپ كەلمەكتە.يەنە كەينى-كەينىدىن سېلىنغان بىر-بىرىدىن ئىگىز سودا - سارايلار قۇرۇلدى،ھەم تۇنجى بۇلۇپ قاراقاش سارىيى تىجارەت باشلىدى.
دەيمەنكى يۇرتۇم ھەر ھالدا زاماننىڭ ئارقىدا ئەمەس.بازارلىرىمىز بار ھەركۈنى شۇنداق قىززىيدۇ دەڭە،ئالما ،شاپتۇل بىھى، ئەنجۈر،قىزىل ئۈزۈم ئاق بېغىش مۇناقا سەيۋە كىشمىشى ئۈزۈلمەيدۇ مىۋىلەر بازارلاردىن ياز قىشى،بازارلاردا باققاللار ئاندىن تولا يىراق يېقىندىن تۈرلۈك تۈمەن ماللىرى بىلەن بازارلارنى ئاۋاتلاشتۇرسا،ئىش - ئەمگەكتىن بوشىغان دېھقانلار ھەم ئىشچى خىزمەتچىللىرمىز بازارنى شۇنداق قىززىتىۋېتىدۇ،
ھۈنەرۋەنلەر ئاندىن تولا نىمىگە ئىھتىياجلىق بولسىڭىز ئىزدىسىڭىزلا تېپىلىدۇ ،
يەرلىمىز شۇنداق مۇنبەت سۇلۇق يەرلەر شال بىلەن تولسا ، باشقىلىرى چىلان.ياڭاق ،ئالما .ئۈرۈك ، قوغۇن تاۋۇز بىلەن تەبىئەتنىڭ گۈزەللىكىنى بېيىتىپ كەلمەكتە، بۇغداي-قوناق شۇنداق ئوخشايدۇ،بىزدە ئۇنداق زاۋۇت ئۇنى ئىشلىتىدىغان قائىدە يوق ،پەقەتلا يىرى يوقلار ھەم چوڭ تىز تاماقخانىلار خالتىلىق ئۇن ئىشلىتىدۇ ، باشقا كىشىلىرىمىز يىل بۇيى ئۈزىمىز تېرىغان ئاشلىقنى ئۇزۇقلۇق قىلىمىز ،
قاراقاشنىڭ شەھەر ئەتىراپىدا سەي كۆكتات بەكمۇ كۆپ چىقسا ،يىڭى يەر كاۋاك يېزىلىرىدا ئۈزۈم شۇنداق كۆپ ،يىلدا نەچچە توننا ئۈزۈم ساتىدۇ ھەر ئۆيلۈك دېھقان،يەنە جاھانباغ،داراسكال يېزىللىرىدا شال ئېتىزلىرى بەك كۆپ يەرلىرى ئاساسەن شال بىلەن تولغان ،
يەنە يۇرتۇمنىڭ چوڭ جەمە ئالدىدا تۈمۈرچىلىك شۇنداق تەرەققى قىلغان ، قوللىرى چىۋەر ئەقىللىق ھۈنەرۋەنلىرىمىز چىرايلىق نەقىشلىرى بىلەن تۈمۈردىن تۇرمۇشقا لازىمەتلىكلەرنى ياساپ كىلىۋاتىدۇ،يەنە بىر تەرپىدە ياغاچچى ئۇستىلىرمىز تۈمۈرچىلەردىن قېلىشمۇغىدەك ماھارىتى بىلەن بارزارنى تېخىمۇ ئاۋاتلاشتۇرماقتا،
ھازىر يۇرتىمىزدا قاتناش قۇلايلىق ،يوللارغا ماي ياتقۇزۇلدى ، ھەر كۇچا دوقمۇشلاردا ماگزىنلار ئېچىلدى ،دېھقانلىرىمىز بېيىپ ئانچە - مۇنچە ماشىنىلاشتى ، ئون دېھقاننىڭ بەشىدە ماششىنا بەشىدە مۇتسىكىلىتتىن بىر ئىككىدىن بار ، ئىشەك ھارۋىلار پەقەت يىراق يېزىلاردا ئانچە - مۇنچە ئىشلىتىلىدۇ ،يېزا ئىگىلىك ئىشلىرىمۇ ھەممىسى دىگۈدەك ماشىنىلاشتى ، ئالۋان سىلىقمۇ ئاز قالدى ، قىسقىسى دېھقانلىرمىز خۇشال ،
جەمىيەت تىنىچ،ھەممىمىز خاتىرجەم ياشاۋاتىمىز ، ئامانلىققا كاپالەتلىك قىلىش ئۈچۈن كەنىتلەرگە ساقچى ئۇرۇنلاشتى ،
پۇخرالىرىمىزمۇ ئاڭلىق ئۇرۇش جىدەل يوق دىيەرلىك ، بىرسى جىدەللەشسە تۇشمۇ-تۇشتىن ئاجىرتىپ تەسەللى بىرىشكە ئادەتلەنگەنلىكتىن كىم بولمىسۇن ئاجىرتىپ قۇيىمىز ،يىگىتلىرمىز شۇنداق تىرىشچان ھەممىسى دىگىدەك ھۈنەر كەسىپ ئىگىلەپ تۇرمۇشىنى قامداپ كىلىۋاتىدۇ ، تىجارەتچىللىرمىز شۇنداق ئاق كۈڭۈل ، ئەگەر نەرسە سېتىۋالماقچى بولسىڭىز كىيىم بولسا . كىيىپ سىناپ كۈرۈڭ دەپ شارائىت ھازىرلىسا ،يىمەك ئىچمەك بولسا سىز كىلىشىڭىزگە ئاۋال تېتىپ كۈرۈشىڭىزنى سورايدۇ ،ھەرگىز تارتىنجاقلىق قىلمايدۇ ، بىرسى - بىرسىگە ئىچى تارلىق قىلىشمايدۇ ،
تاماقخانىلىرمىز بەك پاكىز ، كۈتكۈچىلەر شۇنداق مۇلاھىم ھەم قىزغىن ، ئۈرۈمچىدىكىدەك تاماققا ئاۋال پۇل دەپ كىيىن چاي ئەكىلىدىغان قائىدىمىز يوق ، سىز كىرىشىڭىزگە سۇغۇق سۇ كەلتۈرەيمۇ چاي ئىچەمسىز دەيدۇ ، ئاندىن نىمە تاماق يەيسىزكىن دەيدۇ ،
ناۋاييلارمۇ ھەم شۇنداق نان يىمەكچى بولسىڭىز قىززىق بولسا سۇغا چۈكۈرۈپ ناننىڭ ئۇلىدا شور تۇز بولسا ئېلىۋەتكەندىن كىيىن قېنى ئالسىلا دەپ ئەدەپ بىلەن مۇئامىلە قىلىدۇ ، يەنە سىز بىر نەچچە قېتىم نان ئالسىڭىز يېنىڭىزدا پۇل بولمىسىمۇ نىسىيگە بىرىۋېرىدۇ ، ئەگەر كۆپرەك كىرەك بولسا ئۈيىڭىزگىچە ئاپىرىپ بىرىپ ھىساباتنى ئاي ئاخىردا قىلسىڭىزمۇ بۇلىدۇ،
دېھقانلىرمىز شۇنداق مېھماندوس مەيلى سىز دوس بۇلۇڭ ياكى ناتۇنۇش بۇلۇڭ تاماق ۋاختىدا بولسا ئاۋال سىزگە كەلتۈرىدۇ ، ئەگەر تاماق ۋاختى بولمىسىمۇ بىر پىيالە چاي ياكى سۇغۇق سۇ بىلەن مەزلە كەلتۈرۈپ ھاردۇق سورايدۇ ، ھەرگىزمۇ قىزغىنىش قورۇنۇش ئەھۋالى مەۋجۇت ئەمەس.
مانا بۇ قاراقاش مىنىڭ ماكانىم ئايلىنىپ - ئايلىنىپ ھېرىپ كىتىمەن شۇنداق كەڭرى ،
ئەر - ئايال تىرىشچان پۇل تېپىشقا ماھىر بۇلۇپ ئىش-ئەمگەكتە بىرگە قاتنىشىدۇ ، بۇرۇن پاختا تېرىپ يوتقان تىكىپ ساتقان بولسا ھازىر يوتقان تۇرى مېھمان كۆرپىسى تىكىپ سېتىۋاتىدۇ ، ئەرلەرمۇ ھەم كەسىپكە ماھىر بۇرۇن پاختا ئېلىپ ساتاتتى، ھازىر تۈمۈر يەنى كىرەكسىز تۈمۈر باھاسى ياخشى دەپ شۇنۇڭغا كىرىشىپ كەتتى ، دىمەك قاراقاش خەلقى ئىنتايىن تىرىشچان 10 نەچچە يىلنىڭ ئالدىدا ئاقسۇغا چىقىپ پاختا تىرىش باشلانغاندىن كىيىن ھازىرغىچە داۋاملىشىۋاتىدۇ ، ھەم تۇرغۇن ئەمگەك كۈچى ئىشقا ئۇرۇنلاشتى ، بېيىغانلار كىيىنكىلەرنى يىتەكلەپ ھاياتلىق ئۈچۈن كۈرەش قىلىپ كىلىۋاتىمىز ،
خالىسىڭىز كىلىڭ باغچىلار ئاندىن تۇلا چىمەنزار ھەم ئەزىمەتلەر ئارام گاھى مىڭ يىللىق قەدىمى چىنار سەيلىگاھى ، قەدىمى سۇ تۈگمىنى ساياھەت مەركىزى، قەغەزچىلىك كۇچىسى، شاخلىق پىئودال قورۇسى قاتارلىق مەنزىرگاھلىرمىز ، ئۆزگىچە مىللى پۇراققا ھەم قەدىمى تۈسكە ئىگە ،
مانا بۇ قاراقاش مىنىڭ دىيارىم،يازغانلىرىم تېخى يۈزنىڭ بىرى ،
