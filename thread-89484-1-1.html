                                     چايخانىدىكى قىممەت پاراڭلار

                                 قاپاقتىن-  باراڭ قىممەت.
                                        چايخانىدا- پاراڭ قىممەت.
                                       گومۇشتىن - ساراڭ قىممەت.
                                        تۇرۇڭدىن-تاراڭ قىممەت.
                                       نىمە قىممەت-قاراڭ قىممەت.

                                        ماشىنىدىن- ماي قىممەت.
                                        ئاسمان كەڭرى-ئاي قىممەت.
                                        شەھەلەردە-  جاي قىممەت.
                                        ئىگىز تاغدا- لاي قىممەت.
                                         كەمبىغەلدىن -  باي قىممەت.

                                          مايدىن- گاز  قىممەت.
                                          پاختىدىن-ماز  قىممەت.
                                           ناخشىدىن -ساز قىممەت.
                                           سەتىراچقا-تاز قىممەت.
                                            قارا قىشتا -ياز قىممەت.

                                           بۇغدايدىن- سامان قىممەت.
                                            ياخشىدىن-يامان قىممەت.
                                           سۆرەمدىن-ساپان قىممەت.
                                            ئالقاندىن-تاپان قىممەت.
                                             ئىشتاندىن-چاپان قىممەت.

                                              ھاياتقا-مازا قىممەت.
                                               يازودا-غىزا قىممەت.
                                               پۇلى يوققا-بازا قىممەت.
                                                ھايۋانغا-ئىزا قىممەت.
                                                يولداشقا-نىزا قىممەت.

                                                قوتازدىن- قوي قىممەت.
                                                قېرىغاندا-توي قىممەت.
                                                كىيىمدىن-بوي قىممەت.
                                                موچەندىن-كوي قىممەت.
                                                  ئەر-ئايالغا-ھوي قىممەت.

                                                  يايلاقتىن-تاغ قىممەت.
                                                   قۇملۇقتىن-باغ قىممەت.
                                                   جىگەردىن-ياغ قىممەت.
                                                   قېلىن يۈزدە-داغ قىممەت.
                                                  تۇمۇچۇقتىن-زاغ قىممەت.
                                                
                                                    بۇرۇتتىن-ساقال قىممەت.
                                                     تەمسىلدىن-ماقال قىممەت.
                                                    گۇرۇچتىن-پاخال قىممەت.
                                                     چىنىدىن-ساپال قىممەت.
                                                     دوقمۇشتىن-ئاچال قىممەت.
                              
                                                    يەليىمىشتىن-سەي قىممەت.
                                                    رىستۇراندا-مەي قىممەت.
                                                    گىتتارچىغا-نەي قىممەت.
                                                    گاچىلارغا-پەي قىممەت.
                                                    سۇ كەچمىسە-خەي قىممەت.
