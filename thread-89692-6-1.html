بۇرادەر سىز ئادەتتە نىمىش قىلسىز ؟ جۇڭگۇدا ئۇيغۇر دىن باشقا مۇسۇلمان يوق دەپ قارامسىز ؟ ئىچكىردە يىمەكلىك ئىشلەپ چىقارغۇچىلار سىزنىڭچە قالايمىقان نەرسىلارنى پەرىق قىلالمايدىغان شۇنچە ھاماقەتمۇ ؟ ئىشنى شۇنداق باشلىمىسا بۇيەردە سىزنىڭ نىمىڭىز بارتى ؟ زاۋۇت بارمۇ ؟ تىخنىك بارمۇ ؟ پۇل بارمۇ ؟ پۇل ؟ خەخنىڭ ئىشىدىن قۇسۇر ئىزلەيدىغان قاپاق كاللىمىزنى ئىشلىتىپ يىراققا قارايلى ، بىكاردىن بىكار خەخنى ۋالاخشىپ تىللىماي ئۇلارغا جانابى ئاللادىن ئىنساپ تىلەيلى دوسلار . ئىشلىرغا ئوڭۇشلۇق تىلەيلى . تەرەققىياتىمىزغا ئاستىرىتتىن پۇتلا كاشاڭ بولۋاتقان مۇناپىقلارغا پۇرسەت يارتىپ بەرمەيلى . نىڭشادا ، تاڭشەندە ، گەنسۇنىڭ لىڭشا دىگەن يەرلەدە ............ سىلىنغان مەسجىت نىڭ مۇنارى دىيارىمىزدىكىدىن ئىككى ھەسسە چىقىدىكەن ، دىنى مەكتەپلىرىنى سىزنى قايىل قىلىدۇ . بۇلار تىخى بىزنىڭ مۇسۇلماندارچىلىقىمىزدىن گۇمان قىلدىكەن . كاللىسى ئوچۇقلىرى ھىسداشلىق قىلدىكەن . مۇسۇلمانچە دىگەن ھەرقانداق زاۋۇتنى تۇڭگانلار نازارەت قىلدىكەن ، كىچىككىنە گۇمان بولسا دەيدىغان يەرگە بىرىپ دەپ ھەل قىلدىكەن . بىزچۇ ؟ قىلالمايمىز . ئازراق ئىشقىلغانلارنى دوزاخقا بۇيرۇيمىز . دورامچىلىق قىلدى، ئوخشۇتۇپ ياسىدى ، دەپسىز  . مۇستەقىل بىرنەرسىنى سىز ياساپ بازارغا سىلىپ بىقىڭ ، ھۈكىمەتنىڭ ئاشۇ ئايىغى چىقماس يىمەكلىك رەسمىيەتلىرىنى ئۆتەپ بىقىڭ ، شۇچاغدا بىلسىز . سىز يەكلەۋاتقان ئاشۇ كارخانچىلار پۇتۈن رەسمىيەتلىرى تولۇق زاۋۇتنى تىپىپ ، مەھسۇلاتنى نازارەت قىلىپ ، قايتا قايتا تەكشۇرۇپ ئوزى ۋە باللىرى يەپ بىقىپ ئاندىن بازارغا سالىدۇ . سىزدە بار ئىمان شۇلاردىمۇ باردۇ ؟ بۇرۇن ھازىر بارنەرسىلەرنىڭ ھەممىسى دىگىدەك بارتى . بارا بارا يىمىدىقمۇ ؟ ئىخلاس ، ئارمان ، ئامىنە .... قاتارلىق شىركەتلەر بۇرۇن ئىچكىردە ياستىپ چىقىپ ئەقىللىق خەلقىمىزنىڭ قوللىشى بىلەن ، پۇل تاپتى ، ئەقىل تاپتى ، تىخنىك ئىگەللىدى ، ئاخىردا ئۆزى مۇستەقىل زاۋۇت قۇرۇپ ئىچكىردىن كىردىغان شۇخىل مالنىڭ ئالدىنى توستى ئەمەسمۇ . قايسى بالا تۇغۇلۇپلا يۇگرەيدۇ ؟ بىلسىڭىز  . بۇ كارخانچىلار پۇل تاپسۇن . ئەقىل تاپسۇن . ئىچكىردە يۇرۇپ ، ئۇيغۇرچە بولمىسا شىنجاڭلىقلارنىڭ قوبۇل قىلمايدىغانلىغىنى ئاشۇ چىشىنى بىلەپ تۇرغانلارغا بىلدۇرسۇن . ئۇيغۇرچە يىزىقسىز ، ئۇيغۇرچە ماركىسىز مەھسۇلاتلارنى دىيارىمىزدا قارشى ئىلىنمايدىغانلىغىنى دۇنياغا جاكارلىسۇن . بىز پەقەت مەھسۇلاتنى ياسىغان شىركەت ئۇيغۇرنىڭمۇ ئەمەسمۇ ؟ دەپ سورايلى . قوللايلى . ناماز ئوقۇش ئۈچۇن مەسجىت كىرەك . ئەڭ بولمىغاندا جەينىماز كىرەك ، ئۇيغۇرنىڭ قولىدا چىقىدىغان يىمەكلىككە پۇل كىرەك ، يول كىرەك ، تىخنىك كىرەك ، ئۈسكىنە كىرەك ، شارت شارت تەستىق سالىدىغان باشلىق كىرەك ، باشلىققا مىدىر سىدىر قىلمايدىغان ھوقوق كىرەك ، بۇنىڭغا سىزنىڭ ، مىنىڭ ، خەلقىمىزنىڭ مەدىتى ، قوللىشى كىرەك . شۇندىلا كارخانچىلار ئىچىكىرگە بارماي ، دورامچلىق قىلماي سىزنى رازى قىلىدۇ . بۇنۇڭغا جەريان ۋە ئازراق ۋاقىت كىرەك ، شۇڭغىچە ، بىزنىڭ ئىچكىرگە كىرىپ دوس دۇشمەندىن پەرىقلىنىشىمىزگە ئاساس سىىلۋاتقان لارنى يوقۇتالمايۋاتقانلارنىڭ دىپىگە ئۇسۇل ئوينىماي چىداپ تۇرۇڭ .
