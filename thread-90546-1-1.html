_دادا مېنى ئۆيلەپ قوي.
مېنىڭ ئالدىمدىلا ئۇنىڭ دادىسىغا دېگەن،گىپى يەنە شۇ بولدى.بېچارە دادا مېنىڭ ئالدىمدا سەل ئوڭايسىزلاندى،بولغاي-،بىردەم تۇرۋالغاندىن كېيىن.
-ئوغلۇم،مېنىڭچە بولسا سېنى ھازىرلا ئۆيلەپ قويغۇم بار.لىكىن ئائېلىمىزنىڭ شارائىتى ئۆزۈڭگىمۇ ئايان.ئىككى ئىنىڭ ئوقىۋاتىدۇ.ئاچاڭنى يېقىندىلا تالالىق قىلدۇق.بېچارە ئاناڭ سەيپۇڭلۇق ھۈنىرىنى قىلىپ،ئانچە-مۇنچە،ئۇنىڭ-بۇنىڭ كېيىملىرىنى تىكىپ،ئۆينىڭ ئۇششاق-چۈششەك خىراجەتلىرىدىن چىقىۋاتىدۇ.مېنىڭمۇ موزدوزلۇق ھۈنىرىمنى قىلىپ تاپقان ئازغىنە پۇلۇم،ئاشۇ ئىككى ئىنىڭنى ئوقۇتۇشقا ئاران يېتىۋاتىدۇ.ئاچاڭنىڭمۇ تويىنى ئۇرۇق-تۇغقانلاردىن،قەرز ئېلىپ دېگەندەك قىلۋالدۇق.ئەمدى سېنى؟...بوپتۇ تويۇڭنىمۇ قىلدۇق دىگەندىمۇ،بىر ئۆينىڭ مورىسىدىن ئىس چىقىرىش ئۇنچە ئاسان ئەمەس،ئوغلۇم!
سېنى ئالىي مەكتەپكىچە ئوقۇتۇشقا قۇربىتىمىز يەتمىدى.يا ئۆزۈڭ بىرەر ھۈنەرنىڭ پېشىنى تۇتمۇدۇڭ.يا بىرەر تېجارەتنىڭ،ئېپىنى قىلمىدىڭ.بىزدىن كېيىن قالساڭ جىنىڭنى قانداق جان ئېتىسەن؟بىز ساڭا بىر ئۆمۈر باقىۋەندە،-ئەمەس،ئوغلۇم!
ئۇ دادىسىغا بىر نېمە دىمەكچى بولىۋىدى،قولىدىن تارتىپ دادىسى بىلەن خوشلۇشۇپ،ئۇنى تالاغا ئېلىپ ماڭدىم.ئۇ ئۇ تالاغا چىقىپلا سۆزلەپ كەتتى.
-قارا،ئاداش دادامغا ئۆيلەپ قويۇشنىڭ گىپىنى قىلساملا مۇشۇنداق سۆزلەپ كېتىدۇ...
ئۇ ئۆزىنىڭ قىلۋاتقانلىرىنى توغرا دەپ ئويلىسا كېرەك.دادىسىدىن رەنجىپ قايناپلا كەتتى.بىز پاراڭ بىلەن مەھەلىنىڭ چېتىدىكى قار ياغاچنىڭ ئاستىغا ئورۇنلاشقان،ئورۇندۇققا كىلىپ ئولتۇردۇق.ئۇ توختايدىغاندەك قىلمايتى.
-ئاداش،توختا!كۆڭلۈڭگە كەلسىمۇ مەن ئىككى ئېغىز گەپ قىلاي.داداڭنىڭ گەپلىرىنىڭ نەرى خاتا؟دېمىسىمۇ ئۆيۈڭنىڭ ئەھۋالى ساڭا بەش قولدەك ئايان.ئىككى ئىنىڭنى ئوقۇتۇشقا ئاز پۇل كەتمەيدۇ.داداڭ سىنى ئالىي مەكتەپتەئوقۇتالمىغان،بىلەن ئىككى ئىنىڭنى بولسىمۇ ئوقۇتۇش ئارزۇسى بادۇر؟-بەلكىم.مۇشۇنىڭ ھەممىسىگە پۇل كېتىدۇ.ئاداش!ئاچاڭنىمۇ بىر-مۇنچە پۇل خەجلەپ يىقىندىلا تالالىق قىلدى.ئەمدى سېنى قايسى پۇل بىلەن ئۆيلەيدۇ؟ئۆزۈڭمۇ بىلىسەن.ھازىر بىزنىڭ غۇلجىنىڭ شارائىتىدا ئاز دېگەندىمۇ،توي قىلىش ئۈچۈن،40-50مىڭ يۈەن بولمىسا توي قىلغىلى بولمايدۇ.ئاتا-ئاناڭ بۇ پۇلنى نەدىن تاپىدۇ؟ئاتا-ئاناڭنى ئۇنداق قېيناپ،خىجالەت قىلما.بىزمۇ ئوغۇل-بالا ئەركەك بولغاندىكىن پۇل تېپىپ،ئاتا-ئانىمىزنىڭ ئالدىغا قويۇپ.ئاپا-دادا مانا مەن ئۆيلەنگۈدەك پۇل تېپىپتىمەن،پالانچىنىڭ قىزىنى ئالاي دېگەن.بولسا ئەلچىلەرنى ئەۋەتىپ تويۇمنى قىلىپ قويساڭلا دىسەك.بىزگىمۇ،ئاتا-ئانىمىزغىمۇ ياخشى ئەمەسمۇ؟بىزنى بېقىپ چوڭ قىلغۇچە ئۇلار نېمە كۈنلەرنى كۆرمەيدۇ؟ئۇلارغا قاچانغۇچە ئېغىرىمىزنى سالىمىز؟داداڭنىڭ دىگىنىدەك سەن يا بىرەر ھۈنەرنىڭ پېشىنى تۇتمۇدۇڭ.داداڭ بىر ئۇبدان موزدۇزلۇق ھۈنىرىمنى ئۈگىنىۋال دىسە،ھازىر كېم ئاياق تىككۈزۈپ كىيىدۇ؟بازاردا تەييار ئاياقتىن جىق نەرسە بولمىسا دەپ،داداڭنىڭ ھۈنىرىنى ياراتمىدىڭ.كىچىك داداڭنىڭ ئاشخانىسىغا،ھۈنەر ئۈگەنسۇن.دەپ،ئاپىرىپ قويسا تۆت كۈن،بولا-بولماي ماڭا سەي ئاخلاتتى،ئوچاققا ئوت قالاتتى،ئوچاقنىڭ كۈلىنى ئالغۇزدى،بۇستا بىشىم ئاغرىپ كەتتى.دەپ،تاۋى نازۇكلىق قىلىپ،ئۇيەردىن چىقىۋالدىڭ.ئەمسە بىشىڭ ئاغىرمايدىغان،ئوت قالاپ،كۈل ئالمايدىغان،ساتىراشلىق ھۈنىرىنى ئۈگەن دەپ.داداڭ يالئۇرۇپ دېگەندەك ئاۋۇ ئالىم ئاكىنىڭ ساتىراشخانىسىغا،ئەكىرىپ قويدى.ئۇيەردىمۇ بىر-نەچچە كۈن ئۆتە-ئۆتمەي،خەقنىڭ بېشىنى يۇيۇپ،ساقىلىنى يۇمشۇتۇپ،كۈندە 20-30 لۆڭگە يۇيۇپ،قولۇم سوپۇن پۇراپ كۆڭلۈم ئېلىشىپ كەتتى دەپ،ئۇ يەردىنمۇ چىقىۋالدىڭ.ئاۋۇ ئاغىنىمىز شۆھرەت بىر ئۇبدان ياخشى كۆڭلى بىلەن،ئاداش مەن بىلەن بىللە تېلفۇنچىلىق قىل دىسە،ئۇنىمۇ بىرە ئايدەك قىلىپ بېقىپ،پايدىسى ئازكەن،جاپا تارتقانغا تۇشلۇق، ئىش بولمايدىكەن دەپ ئۇنىمۇ قىلمىدىڭ.ساڭا زادى نېمە يارايدىكىن؟ھىچ بىلمىدىم.ئاداش!كىچىك ئىشنى ياراتمىساڭ،چوڭىنى قىلالمىساڭ.ھەي.......
ئۇ تاماكىدىن بىرنى ئىلىپ تۇتاشتۇرۇپ،قاتتىق بىرنى شورىدىدە.
-شۇ،ئاداش مىنىڭ قولۇمدىن ھېچ ئىش كەلمەيدىغان ئوخشايدۇ.
-ئۇنداق دېمە،ئاداش ھەممە ئادەم ئانىسىنىڭ قوسىقىدىن،تۇغۇلۇپلا ھۈنەرۋەن،تېجارەتچى،ئالىم بولۇپ كەتمەيدۇ.ھەممە ئىش تېرىشقاندا،زېرىكتىم-تېرىكتىم دېمەي داۋاملاشتۇرغاندا.ئاندىن زورىيىدۇ،كىچىكتىن چوڭىيىدۇ.ئۇنداق ئۆزۈڭنى تاشلىۋەتمە.ئاڭلىسام كۈندە دېگىدەك مەيلى ئېتسۇن،ياكى ئېتمىسۇن،قەيەردە توي-سورۇن بولسا شۇ يەردىن كەلمەيدىكەنسەن.سىنى باشقىلار نېمە دەيدىكەن،بىلەمسەن؟ئىرشات تۆمۈر باغاق دەيدىكەن.بۇ لەقەم سەن ئۈچۈن ياخشىما؟دادىللىرىنىڭ بېشىنى ئاغرىتىپ،مېنى ئۆيلەپ قوي.دەۋاتقان ئاۋۇ ئاق نانچى،نوچىلارنى كۆرىۋاتىسە.ئۆيلۈنۈپ ئالدى،بىرە يىل.كەينى بىر-ئىككى ئاي ئۆي تۇتۇپ ئاجىرىشىپ كېتىپ بارمامدۇ.سېنىڭمۇ شۇلاردەك بولغۇڭ بارمۇ؟يا.قولىمىزدا بىرەر ھۈنىرىمىزنىڭ تايىنى يوق.يا تېجارەتنى ئەپلەشتۈرەلمەيمىز.خوتۇننى ئىلىۋىلىپ يەنە شۇ ئاتا-ئانىللىرىمىزغا ئېغىرىمىزنى سالغان.ئۇلار بىزنى باققاننى،ئاز دەپ ئەمدى خوتۇنىمىزنىمۇ باقامدۇ؟بۇنداق قىلىپ يۈرسەك قولىمىزدا خوتۇن تۇرامدۇ؟ئاداش.ئاۋۇ مەھەلىمىزدىكى پەرھات تەلۋىنى بىلىسە.ئۇ ئۆيدە بىر ئوغۇل بولغاچقىمۇ.ئۆيدىكىللىرى ئۇنىڭغا بەك ئامراق.نىمە دىسە شۇنى قىلىپ بىرىدۇ.خوتۇن ئالىمەن دېدى.تويىنى يامان ئەمەس قىلىپ ئۇنى ئۆيلەپ قويدى.پىراۋا ئىلىپ،ماشىنا ھەيدەشنى ئۈگىنىمەن دىۋىدى،پىراۋىنىمۇ ئالدى.پىراۋىسى چىقىپ بولغۇچە،دادىسىغا ماڭا ماشىنا ئېلىپ بەر دەپتۇ.دادىسى
-ئوغلۇم پىراۋاڭ يېڭى چىقتى بىر مەزگىل ئۆتسۇن.تويۇڭنى قىلىمىز دەپ خېلى چىقىمدا بولدۇق.سىڭىللىرىڭ ئوقىۋاتىدۇ.ھازىرچە ساڭا ماشىنا ئېلىپ بەرگۈدەك پۇل يوق.
دىسە،ئاۋۇ تەلۋە نېمە دەپتۇ.بىلەمسە؟ماڭا ماشىنا ئېلىپ بەرگۈدەك پۇلۇڭ يوق ئادەم مېنى نىمە دەپ تاپقان دەپتۇ.بىچارە دادىغا بۇ گەپلەر ئېغىر كەلدى،بولغاي.بىر-نەچچە كۈن ساقسىز بولۇپ قاپتۇ.ھەي...ھېلىمۇ بىچارە ئاتا-ئانىلار كۆتۈرۈشلۈككەن.ئۆيگە ئەكىرگەن بىرە كويچەن پۇلىمىزنىڭ تايىنى يوق.ھېچ ئىش قىلماي،لاغايلاپ ئۆيگە كىرىپ،قاچا بىكالاپ يۈرسەكمۇ بىزنى ھېچقاچان يەپ كەتتىڭ،ئىچىپ كەتتىڭ دېمەي،بىچارە ئانىللىرىمىز بىكا قىلغان قاچىمىزنى،كىر بولغان كېيمىمىزنى يۇيۇپ،باشقىلارنىڭ ئالدىدا بالامنىڭ بوينى قىسىلىپ،قالمىسۇن دەپ يېنىڭدا خەجلىگىدەك پۇلۇڭ بارمۇ؟دەپ،يانچۇقلىرىمىزغا پۇل سېلىپ قويىۋاتقان.بىچارە دادىللىرىمىز ئوغلۇمنىڭ كىيىملىرى مەينەت بولۇپ،كەتمىسۇن.ئاغىنلىرى بىلەن بارىدىغان يەرلىرىگە بارالماي قالمىسۇن.دەپ ھويلا-ئاراملارنىڭ،ئېغىر-يىنىك ئىشلىرىنىڭ ھەممىنى ئۆزى قىلۋاتقان.نىمە ئۈچۈن؟شۇنداق قىلسىمۇ ئۇلارنىڭ تارتىۋاتقان جاپالىرىنى ئۇنتۇپ،ئۇلارغا ئېغىرىمىزنى سىلىپ.ئۇلاردىن رەنجىپ يۈرىمىز؟نېمىشكە ئۇلارنىڭ قەدرىگە يەتمەيمىز؟بىزمۇ خوتۇن ئىلىپ،بىر كۈنلىرى بالىلىق بولغاندا.ئاتا-ئانا بولۇشنىڭ نەقەدەر جاپالىق ئىكەنلىكىنى ھېس قىلغان ۋاقتىمىزدا ئۇلارنىڭ قەدرىگە يېتىشىمىز مۇمكىن.لېكىن ئۇ چاغدا بىر-قىسىملىرىمىز تولىمۇ كېچىككەن بولىمىز.چۈنكى بىز ئەمدى قەدرىگە يەتمەكچى بولغان ئاتا-ئانىللىرىمىز ئاللىبۇرۇن بىز بىلەن خوشلۇشۇپ ئۇ دۇنياغا كەتكەن بولىدۇ.ئاداش!
ئۇنىڭغا گەپلىرىم،تەسىر قىلدى بولغاي. يانچۇقىدىن يەنە بىر تال تاماكىنى ئىلىپ،ئوت يىقىپ بېشىنى ساڭگىلاتقان پېتى،تاماكىسىنى چېكىپ ئۈندىمەي ئۇزۇن ئولتۇرۇپ كەتتى....
