بۇ يۇمۇرنى ئاق بۈركۈت تورىدا يازغان، دىلمۇرات بولسا ئاق بۈركۈت تورىنىڭ قۇرغۇچىسى.

مۇنبىرىمىزدىكى باپكار، دونا ئەپەندى، دىلمۇراتلاركەينى-كەينىدىن توينىمۇ قىلىۋالدى. تويغا بارالىغانلار توينىڭ پولوسىغا ، خۇشاللىقىغا جۆر بولدى، ماڭا ئوخشاش خىزمەتتىن يىل بويى بىكار بولالمايدىغان بىچارىلەر مانا مۇشۇنداق چەتتە سىڭايان قاراپ قالدى
ئەمدىكى گەپ دىلمۇراتنىڭ تويدىن ئۇزۇن ئۆتمەي بېشىدىن ئۆتكەن سەرگۈزەشتىسىدە بولىدۇ. بۇنى ئالدىنقى كۈنى دىلۇرات چ چ دا پاراڭلىشىپ ئولتۇرۇپ دەپ بەرگەن گەرچە ھېچكىممگە دېمەڭ دەپ ئەسكەەەرتكەن بولسىمۇ شۇنداق دېگۈم كېلىپ كەتتى.
تويلار تۈگەپ قېيىن ئانىسى دىلمۇراتقا بىر ماشىنا سوۋغا قىپتۇ. ئەمدى دىلمۇراتنىڭ خۇشلۇقىنى دېمەيلا قويىلى ئەل ئۆمرىدە باغچىدىكى ئويۇنچۇق ماشىنىدىن باشقا ماشىنىنىڭ رولىنى تۇتۇپ باقمىغان دۆلەت كادىرى ئۈچۈن بۇنىڭدىن ئارتۇق نېمە ھايات شادلىقى بار دەيسىلەر؟
-ھوي دىلمۇراتجان بۇيەرگە كېلىڭ،
قاق سەھەردە كومپىيۇتېرنى ئېچىپ يەنە شۇ تۈگىمەس مۇنبەرنىىڭ ئىشى بىلەن قالغان دىلمۇراتنى قېيىن ئانىسى چاقىردى
-قاچاندا قارىسا كومپىيۇتېرنڭ ئالدىدىن قوپالمايدىكەنسىز، ئاكىڭىز يېڭى ماشىنا ئېلىپ بۇرۇنقى ماشىنىسىنى ئىشلەتمەيدىغان بولدى. سىز ئەمدى ئاشۇ سانتانى ھەيدەپ ئاز-تولا تاكسىيچىلىق قىلىڭ، ھېلى بىكا قىزىمنى ئاچلىقتىن ئۆلتۈرۈپ قويماڭ يەنە
دىلمۇراتنىڭ كۆزلىرىگە ھاياجاندا ئىسسىق ياش كەلدى، مىڭ رەھمەتلەر بىلەن قېيىنئانىسىنىڭ قولىدىن ماشىنىنىڭ ئاچقۇچىنى ئالدى
ئەمدى ماشىنىغا كەلسەك ماشىنىدا سو يوق يېپيېڭى ماشىنا، پەقەت شۇ بىر تەرەپ بۇرۇن بىر قېتىم يۈك ماشىنىسى بىلەن سۈركۈلشۈپ ئۇزۇن تارتۇقتىن بىرى قالغان، بوراندا دەرەخ ئۆرۈلۈپ ئۈۈستى ئازراق مىجىلىپ كېتىپتىكەن شۇ. ھە راست، ئۈستى ئازراق تېشىلىپ قالغان چېغى، سۇ ئۆتىدۇ دەيدۇ. شۇنداق يامغۇر يېغىپ قالسا تىمىلداپ ئاقىدىكەن، چېلەكنى قويۇپ قويۇپ ھەيدەۋەرسە بولىدىكەن، بەكبولمىسا يامغۇر ياغقاندا ھەيبدىمەس  شۇ ماشىنىنى.
يەنە بىر كۈنى سەھەردە قېيىنئانىسى دىلمۇراتنى چاقىردى،
-بالام، بۈگۈن .ايالىڭىز بىلەن مېنى شاياردىكى سىڭلىمنىڭ ئۆيىگە بىر ئاپىرىپ كېلىڭە
قېيىنئانىنىڭ گېپىگە ياق، مۇنبەرگە چىقاتتىم دېگىلى بولاتتتىمۇ؟
دىلمۇرات قېيىن ئانىسى بىلەن ئايالى چىققۇچە ماشىنىنى ھۈرتۈپ، يۇيۇپ، تازلاپ پارقىرىتىۋەتتى.
-ئاۋۋال بىزنى بازارغا  بىر ئاپىرىڭە، سىڭلىمغا سوۋغاتلىق ئېلىۋالاي
قېيىن ئانىسى ماشىنىغا چىقىپلا پەرمان بەردى،
ۋاقتىڭلارنى ئالماسلىق ئۈچۈن تۆۋەندىكى بايانلارنى قىسقارتتىم
ئوت ئالدۇرالماي بىردەم ھەپىلىشىش   3 مىنۇت
كاپوتنى ئېچىپ ئىچىدىكى لاقا-لۇقىلارغا قاراپ بىر ھازا ھاڭۋىقىپ تۇرۇش     10مىنۇت
ئەتۋىركىنى ئېلىپ ئۇيەر-بۇيەرنى كوچىلاپ بېقىش15 مىنۇت
ئەسلىدە ماشىنىنىڭ بېنزىنىنىڭ تۈگىگەنلىكى تۇيۇقسىز يادىغا يېتىپ يېقىن يەردىكى ماي قاچىلاش پونكىتىغا ماشىنىنى ئىتتىرىپ ئاپىرىپماي قاچىلاش   45مىنۇت
شۇنداق قېيىنچىلىقلار بىلەن ماشىنا ئوت ئېلىپ بازارغا ماڭدى.
ئانا-بالا ئىككىسى بازارغا كىرىپ كەتتى، دىلمۇرات ماشىنىدىكى 3قانالدىن باشقاقانالنى تارتالمايدىغان رادىئونى ئېچىپ قويۇپ ئەرلەر ساغلاملىقى ھەققىدىكى لېكسىيەنى ئاڭلاۋاتقىنىدا يان تەرەپتىن سېمىز، سەت بىر ئايال چىقىپ كەلدى
-خەلق دوختۇرخانىسى تەرەپكە 15كويغا ئاپىرامسىز؟
ئۆ\زۈڭلار بىلىسىلەر، خوتۇن خەق بازارغا كىرىپ كەتسە يېرىم كۈن ئۆتمەي چىقمايدۇ. دىلمۇرات كۆڭلىدە بېنزىننىڭ پۇلىنى بولسىمۇ تاپاي دەپ ئويلاپ بۇ كىرانى تارتىشقا قوشۇلدى
نەدىن بىلسۇن، كېلىشمەسلىكنىڭ مۇشۇ  خوتۇندىن باشلىنىدىغانلىقىنى، ئەسلى زابوي ئايالكەنتۇق، ماشىنىغا چىقىپلا ماشىنىنىڭ ئىچىنى ھاراقنىڭ ھىدى بىلەن توشقۇزىۋەتتى، بىچارە دىلمۇرات ئەتىگەن ماشىنا يۇيغان لاتىسىنى بۇرنىغا تارتىپ ماشىنىنى ھەيدىدى. ھېلىقى ئايال 200 مېتىر مېڭىپلا ئۇيقۇغا كەتتى.
بارىدىغان يەرگىمۇ كەلدى، دىلمۇرات ھېلىقى خوتۇننى ئويغىتالماي ئاۋارە
-كەدۇق ھەدە، چۈشۈڭ ماشىنىدىن
دېۋىدەك ئۇخلاۋاتقان ئايال قىمىرلاپ قويامدىغان، دىلمۇرات تۈرتۈپ باقىدۇ جىم، قىچىقپلاپ باقىدۇ جىم، چىمداپ باقىدۇ جىم
ئاخىرى بولماي 15كوينىڭ مەرىدىن كېچىپ ئۇ ئايالنى سۆرەپ چۈشۈپ دەرۋازا ئالدىغا ياتقۇزۇپ قويۇپ قېيىن ئانىسىبىلەنئا يالىنىڭ ئالدىغا ماڭدى،
شايار يولى تەرەپكىمۇ ماڭدى، يولدا كېتىۋىتىپ دىلمۇرات ئىختىيارسىز يان تەرەپكە قارىۋىدى ئاياللارنىڭ بىر پاي توپلىيىنى كۆرۈپ قالدى. ھىم، ئەمدى ئۆلىدىغان بولدۇم دەپ ئويلىدى ئۇ. ئەگەر ئۇنى ئايىلى ياكى قېيىنئانىسى ككرۈپلا قالسا قىيامەتتىنمۇ قورقۇنۇچلۇق ئىشلارنىڭ بولىدىغانلىقى ئۇنىڭغا ئايان
كاللىسى ماركودەك ئىشلەيدىغان دىلمۇرات توپلەينى ئەپچىللىك بىلەن ئېلىپ ئاستاغىنا ئىشىكنى ئېچىپ تاشلىۋەتتى، بۇنى ئەلۋەتتە يېرىم كۈندىن ئارتۇق بازار چۆرگىلەپ ھېرىپ ئۇيقۇغا كەتكەن قېيىنئانا، ۋە ئېرىدىن يوشۇرۇن كىملەرگىدۇر قىسقا ئۇچۇر يوللاش بىلەن ئاۋارە يېڭى كېلىن سەزمىدى.
ماشىنا يولدا داراقشىپ كېتىۋاتقىنىدا دىلمۇرات ئارقىنى كۆرىدىغان ئەينەككە شۇنداق قارىۋىدى كەينىدىن قارا كىيا ئاپتومبىل ھەدەپ سىگنال بېسىپ كېلىۋاتاتتى، شۇنداق زەن سېلىپ  قاراپ  شوپۇرنىڭ دېرىزىدىن چىقىپ تۇرغان قولىدا ھېلىقى بالاتەككۈر توپلەينىڭ بارلىقىنى كۆردى.
بەلكىم ئەزرائىل كەلسىمۇ بۇنچىلىكك ھودۇقماستى دىلمۇرات، شۇنىڭ بىلەن ماينى قۇلاققىچە دەسسەپ قاچتى، سۈرئەت كۆرسەتكۈچ 80گە ۋاششىدە چىقتى، ھېلىقى كىيا يەنە تاپبېسىپ كېلىۋاتاتتى، 10گە
ىقتى، 120گىمۇ چىقتى ماشىنا دېگەن تاراقشىۋاتىدۇ، قېيىنئانىسىنىڭ تىلى گەپكە كەلمەي
-ھوي...با..لام..سىز..بىزنى..سوقۇپ..ئۆلتۈرەي..دەمسىز؟
-ۋاي..ئاپا..مەن..ئەمەس..كاربىراتورغا..ماي..جىق..كەتكەن..ئوخشايدۇ.
ھېلىقىى كىيانى ئاخىرى بىر يەردە قالدۇرۇپ قويدى، شايارغىمۇ ەلدى، ئالدى كاپوتنى ئېچىپ شۇنداق قارىۋىدى چىققان ئىس-تۈتەكنىڭ جىقلىقىدا قايسى ماتور، قايسى نېمە ھېچنىمىنى ئىلغا قىلالمىدى،
قېيىنئانا بىلەن ئايالى تې\ىىچە ماشىنىدىن چۈشمەي ئولتۇراتتى
دىلمۇرات كېلىپ:
-جېنىم ئانا،كەلدۇق چۈشسىلە
-ۋاي بالام، بايا ماڭغاندا پۇتۇم ئىسسىپ كېتىدۇ دەپ توپلىيىمنى سېلىۋەتكەن، بىر پېيى يوق تۇرىدا؟
دىلمۇرات بىچارە نۇمۇس، تىل-دەشنامغا ئۆلدىغۇ لېكىن. ئىچىگە سىغدۇرالماي بۇ گەپلەرنىڭ ھەممىسىنى ماڭا دەدپ بەردى، مېنىڭ ئىچىمدە گەپ تۇرمىغاچ سىلەرگە دەپ بەردىم. مېنىڭ دېگەنلىكىمنى دىلمۇراتقا دەپ قويماڭلا، بولمىسسا مەندىن تېخىمۇ بەك رەنجىپ قالىدۇ
