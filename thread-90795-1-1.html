    «ھوزۇر»ۋە«چوغلۇق»دا «سەمرىۋاتقان» ئاياللار  


  شايار ناھىيەسىنىڭ يېزا-بازارلىرىدا مۇخبىرلىق قىلىش جەريانىدا ئۆز كۈچىگە تاينىپ ئىگىلىك تىكلەپ،ئۆزىگىمۇ،ئۆزگىگىمۇ نەپ يەتكۈزگەن نۇرغۇن تىپلارنى ئۇچىراتتىم.ئۇلارنىڭ ئىچىدىكى «ھوزۇر»ۋە«چوغلۇق»ماركىسىدا مەھسۇلات ئىشلەپچىقىرىپ تونۇلىۋاتقان ئىككى ئايال كارخانىچىنىڭ ئىگىلىك تىكلەش جەريانى ۋە ئۇلارنىڭ مەھسۇلاتلىرىنىڭ بازار ئەھۋالى ئالاھىدە قىزىقىشىمنى قوزغىدى.
   

ئاينىساخانغا قوي يۇڭى پايدا قىلدى


     رىىماتىزىم كېسىلى سەۋەبىدىن ئىش- ھەركەتتە خېلىدىن بۇيان، قىينىلىپ كېلىۋاتقان ئاينىساخا كېرەم 2006- يىلى ناھىيەلىك پاختا- چىگىت شىركىتىدىكى ئىش ئورنىدىن قالدۇرۇلدى. بۇنىڭ بىلەن ئۇنىڭ ئانچە باياشات بولمىغان ئائىلىسى ئوقۇتقۇچى ئېرى مەمتېلى سەيدۇلنىڭ مائاشىغىلا قاراشلىق بولۇپ قالدى.شۇنىڭدىن كېيىن ئاينىساخان كېرەمنىڭ كۈنلىرى بىرەر كەسىپ بىلەن شوغۇللىنىپ، ئائىلىسىنىڭ كىرىمىنى كۆپەيتىپ، ئىككى ئوغلىنى  ئالىي مەكتەپتە ئوقۇش ئارزۇسىىغا يەتكۈزۈش ئىدى. مەمتېلى سەيدۇل ئۇنى باشقا ئىشلاردىن غەم يېمەي، سالامەتلىكىنى ئويلىشىنى تاپىلاپ كەلگەن بولسىمۇ، ئۇ ئېرىنىڭ يۈكىنى يەڭگىللىتىش توغرىسىدىكى ئويلىنىش- ئىزدىنىشنى توختاتمىدى. 2006- يىلى ئۇ ناھىيە بازىرىدىكى بىر تىۋىپقا كېسىلىنى كۆرسەتكەندە، ئۇ ئاينىساخان كېرەمگە داۋالىنىشنى ئۈزۈپ قويماسلىق بىلەن بىللە، پۇت- قولىنى ئىسسىق ساقلاش كېرەكلىكىنى، بۇنىڭ ئۈچۈن قوي يۇڭىدىن يوتقان- كۆرپە قىلدۇرۇپ يېتىش مەسلىھەتىنى بەردى، ئۇنىڭ دېيىشىچە، قوي يۇڭى بەدەندىكى ھۆللىكنى چىقىرىپ تاشلاپ، ئىممۇنتېت كۈچىنى ئاشۇرۇپ، بەدەن سوۋۇپ كېتىش،بۆرەك ئاجىزلاش، بۆرەككە سۇ چۈشۈش، ئاياللاردا كۆپ كۆرۈلىدىغان سۇلۇق ئىششىق، سۆڭەك ئۆسۈۋېلىش، سوغوقتىن بولغان كۆپ تەرلەش، ماغدۇرسىزلىنىش قاتارلىقلارغا مەنپەئەت قىلىدىكەن.ئاينىساخان كېرەم تىۋىپنىڭ مەسلىھەتىنى ئېرى مەمتىلى سەيدۇلغا  دېگەن ئىكەن، كۈيۈچان بۇ ئەر شۇ كۈنىلا ئۇنىڭغا بەش كىلوگرام قوي يۇڭى ئەكىلىپ بېرىپتۇ.ئۇ بۇ قوي يۇڭىنى يۇيۇپ  قۇرۇتۇپ، ساۋاپ، يوتقان- كۆرپە تىكىپ، تىۋىپنىڭ دېگىنىنى سىناپ كۆرۈپتۇ. ئۈچ ئاي ئۆتكەندە ئۇنىڭ كەم ماغدۇرلۇق، ھاۋا سوۋىغاندا پۇت- قولى ئاغرىيدىغان ئالامەتلەتلەردە رۇشەن ئۆزگۈرۈش بوپتۇ.ئاينىساخان كېرەمنىڭ رىماتىزىم كېسىلىنىڭ قوي يۇڭىدىن تىكىلگەن يوتقان- كۆرپىدىن خېلىلا شىپاھ تاپقانلىقىنى ئاڭلىغان خۇلۇم- قوشنىلار ئۇنىڭغا قوي يۇڭىدىن تىكىلگەن يوتقان- كۆرپە بۇيرىتىدىغان بوپتۇ. بۇ ئىشتىن ئىلھانلانغان ئاينىساخان كېرەم 2009- يىلى قوي يۇڭى سېلىپ تىككەن 20 جۇپ يوتقان- كۆرپە تىزلا ئوبدان پۇل بوپتۇ. بۇ تىجارەتتىن كۆزى ئېچىلغان ئاينىساخان كېرەم ئويلىغىنىنى ئېرىگە ئېيتىپ، ئۇنىڭ قوللىشىغا ئېرىشىپتۇ. شۇنىڭ بىلەن مەمتېلى سەلەي بانكىدىن 80 مىڭ يۈەن پۇل قەرز ئېلىپ، بۇ پۇلغا قوي يۇڭى ئېتىش- يۇيۇش ماشىنىسى، قوي يۇڭىنى كۈيە چۈشۈشتىن ساخلايدىغان مەخسۇس سوپۇن پارشۇكى سېتىۋېلىپ ئەكەلتۈرۈپ بېرىپتۇ. بۇنىڭ بىلەن ئاينىسخان كېرەم ئېرىنىڭ ياردىمىدە  سېتىۋېلىنغان قوي يۇڭىنى پاكىزە يۇيۇپ ۋە ئېتىپ، يوتقان- كۆرپە، يۆگەك، ياستۇق، جىلىتكە تىكىپ، ناھىيە بازىرىدىكى «كۈمۈش كۆۋرۈك» خەلقئارا كاتتا بازىرىدىن دوكان ئىجارە ئېلىپ بازارغا ساپتۇ. بۇ جەرياندا ئۇ ئۆز مەھسۇلاتلىرىغا « ھوزۇر» ماركىسىنى ئەنگە ئالدۇرۇپ ۋە تەشۋىقات ۋەرەقىسى، ئىسىم كاتوچكىسى باستۇرۇپ، تارقىتىپتۇ. بۇنىڭ بىلەن ئۇنىڭ مەھسۇلاتلىرى شاياردىكىلەرنىڭ تۇرمۇش بويۇمىدىن يۇرت ئاتلايدىغان سوۋغاتلىق بويۇمغا ئايلىنىپتۇ. ئاينىساخاننىڭ ئەھۋال تونۇشتۇرىشى مۇشۇ يەرگە كەلگەندە،
      __ 100 كىلوگرام قوي يۇڭىنى يۇيۇپ، يوتقان-كۆرپە تىككىدەك ھالەتكە كەلتۈرگەندە ئۇنىڭ يېرىمى خوراپ كېتىدىكەن- دېدى مەمتېلى سەلەي گەپ قىستۇرۇپ. - بۇلتۇر بەش توننا قوي يۇڭى سېتىۋېلىپ، ئۇنىڭدىن 1000 يوتقان- كۆرپە تىكىپ ساتقان ئىدۇق. بۇ يىل ئون توننا قوي يۇڭى سېتىۋېلىپ، 2000 گە يېقىن يوتقان- كۆپە تىكىپ، بازارغا سالدۇق، ھازىر ئۇنىڭ خېلى بىر قىسمى سېتىلىپ بولدى. يوتقان- كۆرپىلەرنىڭ يۇڭ سۈپىتى ۋە مىقتارىغا ساختىلىق قىلمىدۇق. شۇڭا ھازىر خېرىدارلىرىمىزنىڭ بەزىلىرى تېلېفۇن بىلەن مال زاكاس قىلىدۇ. دېگەن ۋاقىتتا بۇيرۇتقان مېلىنى يەتكۈزۈپ بېرىمىز.بۇ جەرياندا  تۆۋەن تۇرمۇش كاپالىتىدىن بەھرىمان بولىۋاتقان شەھەر ئاھالىسىدىن تۆت كىشىنى يىل بويى ئۆزىمىزگە ھەمكارلاشتۇردۇق. ھەر يىلى 5- ئايدىن، 8- ئاينىڭ ئاخىرىغىچە 15 كىشىنى ئىشلىتىمىز. بانكا قەرزىنىڭ  يېرىمىنى قايتۇرۇپ  بولدۇق، گەپنىڭ پوس كاللىسىنى دېسەم، قوي يۇڭى ئاينىساخانغا پايدا قىلدى.  ئىككى بالىمىزنىمۇ ئالىي مەكتەپتە خاتىرجەم ئوقۇتىۋاتىمىز،
      بىز دۇكاندىكى قوي يۇڭىدىن ئىشلەنگەن مەھسۇلاتلارنىڭ ئىچى- تېشىنى كۆردۇق. يوتقان- كۆپىلەرنىڭ يۇڭى تورغا ئېلىنغان بولۇپ، بۇ تور يۇڭ ئەستەرلىنىپ، پاختا رەخت بىلەن قاپلانغان بولغاچقا، يېپىنىپ ياتقان كىشىگە ھەقىقەتەن ھوزۇر بېرىدىغانلىقى بىلىنىپ تۇراتتى.

«چوغلۇق»تا كۆكلەۋاتقان مېھرىگۈل


  
       بىر ئائىلىدىن بىر ئادەم  ئىش ئورنىدىن قالسا، شۇ جەمەتتىكىلەرنىڭ ئىش ئورنىدىن قالغۇچىغا مەنىۋى جەھەتتىن تەسەللى بېرىپ، تىرىكچىلىك يولى كۆرسىتىپ بېرىشى،مەبلەغ جەھەتتىن ياردەم بېرىشى  تەس كەلمەيدۇ. ئەمما مۇستەققىل ئائىلىسى بار بولسىمۇ ، بىر جەمەتتىن ئۈچ كىشى  بىراقلا ئىش ئورنىدىن قالسا  بۇ جەمەتتىكىلەرنىڭ ماددىي ۋە مەنىۋىي تۇرمۇشىغا كېلىدىغان بېسىمنى باشقىلارنىڭ تەسسەۋۋۇر قىلمىغى تەس.
شايار ناھىيەسىنىڭ شايار بازىرى جەنۇبىي خەلق يولى مەھەللىسىدە «چوغلۇق» رەشىلىيە كارخانىسىنىڭ خوجايىنى مىھرىگۈل مۇھەممەد ، ئىنىسى راھمانجان ھەم ئۇنىڭ ئايالى مارىگۈل قاۋۇل 2005- يىلى كارخانا ئىسلاھاتىنىڭ شامىلىدا ناھىيەلىك پاختا- چىگىت شىركىتىدىن ئىش ئورنىدىن قېلىپ، بىز يۇقىرىدا تىلغا ئېلىپ ئۆتكەن قىسمەتكە دۇچار بولغان بىر جەمەت كىشىلىرىدۇر.
         بۇ بىر جەمەت كىشىلىرى ئىش ئورنىدىن قالغاندىن كېيىن، ئۆز ئالدىغا تىرىكچىلىك قىلىشقا باشلىغان بولسىمۇ، مېھرىگۈلنىڭ ئىنىسى راھمان مۇھەممەد ۋە ئۇنىڭ خوتۇنىنىڭ تىجارەت ياكى ھۈنەر- كەسىپتىن خەۋىرى بولمىغاچقا ئىشنى نەدىن باشلاشنى بىلەلمەي يۈردى. مىھرىگۈل كىچىكىدە ئانىسىدىن رەشىلىيە ئىشلەشنى ئۈگەنگەن بولغاچقا، ئىش ئورنىدىن قالدۇرۇلغاندىن كېيىن، ئانىسى بىلەن يوتقان كىرلىكى، ياستۇق قېپىغا رەشىلىيە ئىشلەپ، سېتىپ كىرىم قىلماقچى بولغان بولسىمۇ، قول  بىلەن ئىشلەنگەن رەشىلىيەنىڭ سۈپىتى ياخشى بولسىمۇ، ئىش سۈرئىتى ئاستا بولغاچقا، ئەجىرگە يارىشا كىرىم قىلالمىدى. شۇنداق بولسىمۇ ئۇ قىلىۋاتقان ئىشىدىن ۋاز كەچمەي، ھەر قېتىم ئاپىسى بىلەن ئۈرۈمچىگە مالغا بارغاندا رەشىلىيە ئىشلەش ماشىنىسى توغرىسىدا تىڭ- تىڭلاپ يۈردى. 2010- يىلىغا كەلگەندە، ئېنىق ئۇچۇرغا ئېرىشتى-دە، ئورۇق- تۇغقانلىرىنىڭ ياردىمىدە ئېرىشكەن 80 مىڭ يۈەن قەرز پۇلنىڭ 70 مىڭ يۈەنىگە 63 يىڭنىلىك رەشىلىيە ئىشلەش ماشىنىسى سېتىۋېلىپ، زاۋۇت تەرەپنىڭ ياردىمىدە ماشىنىنى ئىككى ئايدىن كېيىن ئىشقا كىرىشتۇردى. بۇ ماشىنىدا ئىشلىنىدىغان يوتقان كىرلىكى، ياستۇق قېپى ،داستىخاننىڭ گۈل نۇسقىرى  كومپىيوتىردا لايھەلىنىدىغان بولغاچقا، ئۆزى يېڭى گۈل نۇسقىسىنى لايھىلىيەلمەي، ماشىنىدا كۆرۈلگەن كاشىلىلارنى  زاۋۇت تەرەپ تېخنىك ئەۋەتكىچە ئوڭشىيالماي، ساخلاپ تۇرۇشقا مەجبۇر بولغانلىقتىن، ئىشلىرى بىرمەزگىل يۈرۈشمىدى. شۇنىڭ بىلەن ئۇ ئىنىسى راھمانجاننى زاۋۇتقا گۈل نۇسقىلىرىنى كومپىيوتىردا لايھەلەش، رەشىلىيە ماشىنىسىدا كۆرۈلگەن ئاددىي كاشىلىلارنى ئوڭشاشنى ئۈگىنىپ كېلىشكە ئەۋەتىپ، بۇ مەسىلىنى ھەل قىلدى. شۇنىڭدىن كېيىن ئۇ ئىنىسىنى تېخنىكىغا، ئىنىسىنىڭ ئايالىنى سۈپەتكە مەسئۇل قىلىپ، مەھەللىسىدىكى تۆۋەن تۇرمۇش كاپالىتى پۇلى ئالىدىغان ئائىلىلەردىكى يەتتە ئايالنى كارخانىسىغا ياردەملەشتۈرۈپ، شاياردا ماشىنىدا رەشىلىيە ئىشلەيدىغان بىردىن بىر رەشىلىيە كارخانىسىنى ۋۇجۇتقا چىقاردى.
        مىھرىگۈل ئىلگىرى بىر ئايدا ئىككى، ئۈچ يوتقان كىرلىكىگە رەشىلىيە ئىشلىيەلىگەن بولسا بۇلتۇردىن بويان، بۇ ماشىنىنىڭ خاسىيىتىدە ھەر ئىككى يېرىم سائەتتە يەتتە يوتقان كىرلىكىگە رەشىلىيە ئىشلەپ چىقىرالايدىغان بوپتۇ. مىھرىگۈلنىڭ كارخانىسىدا ئىشلەپچىقىرىلغان كارۋات  بويۇملىرىنىڭ رەختىنىڭ سۈپىتى ياخشى، گۈل نۇسقىلىرى ئۆزگىچە بولغاچقا، ئۇنىڭ مەھسۇلاتلىرى شاياردىلا بازار تېپىپ قالماي، توقسۇ ناھىيەسى، ئاقسۇ شەھىرىدىكى كارۋات بويۇملىرىنى توپ ساتىدىغان سودىگەرلەر تېلېفۈندا زاكاس قىلىدىغان مەھسۇلاتقا ئايلىنىپتۇ.
         ئىش ئورنىدىن قالدۇرۇلۇشتىن ئىلگىرى شىركەتتە مالىيە خىزمىتى بىلەن شوغۇللىنىدىغان مېھرىگۈل ھازىر خېلى پىشقان كارخانىچىغا ئايلىنىپتۇ. سەۋەبى، ئىنىسى ۋە ئۇنىڭ خوتۇنىغا تېخنىك قاتارىدا ئىش ھەققى بېرىدىكەن، ئۆزىمۇ مۇقىم ئىش ھەققى ئېلىپ، ئاشقان پايدىنىڭ بىر قىسىمىنى ئايلانما مەبلەغ قىلىپ، ئاشقان قىسمى بىلەن قەرز قايتۇرۇپ، ھازىرغا قەدەر قەرزىنى كۆپ قىسمىنى قايتۇرۇپ بوپتۇ. رەشىلىيە ئىشلەيدىغان ئىشچىلارغا قىلغان ئىشىنىڭ سۈپىتى ۋە مىقتارىغا  قاراپ ھەق بېرىپ، ئۇلارنىڭ مەسئۇليەتچانلىقى ۋە ئاكتىپلىقىنى قوزغاپتۇ. شۇڭا ھەر ئايدا مىھرىگۈل بازارنى تەمىنلەيدىغان رەشىلىيە ئىشلەنگەن كارۋات بويۇملىرى 800 دىن كۆپ ئىكەن.
       مىھرىگۈل مۇنداق دېدى: مەن ئىش ئورنىدىن قالغان بولساممۇ، تىرىشىپ بۈگۈنكى ھالەتكە يەتتىم.يىلدا20-30مىڭ يۈەن كىرىم قىلىمەن. ئۈزۈمنىڭ جەمىئىيەتكە يۈك بولۇپ قالمىغانلىقىمدىن پەخىرلىنىمەن. مەھسۇلاتىمغا «چوغلۇق» دەپ ماركا ئالدىم. مەقسىدىم كارخانامنى ئۆچمەس «چوغ»قا ئايلاندۇرۇپ، ئىللىقلىققا تەشنا كىشىلەرگە «چوغ» ھارارىتى بېرىش.
