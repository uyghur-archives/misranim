بۇ تېمىنى كۆرۈپ ئۈرۈمچى دۆڭكۆۋرۈكتىكى بىزنىڭ كىشىلىرىمىز قىلىۋاتقان شۇنداق بىر قىمار ئۇسۇلى ئېسىمگە كېلىپ قالدى، يەنى بىر كىشى تەخسىنىڭ ئۈستىگە 4-5 تال قوناخنى چېچىپ سىز نەچچە تاللىقىن پەرىقلەندۈرەلىگىدەك سۈرئەتتە كاردون قەغەز بىلەن يېپىۋالىدىكەن،ئاندىن،-نەچچە تال ، كىم ئاچقۇزىدۇ،مۇنچە كويغا ، دەپ پۇلنىڭ سانىنى دەيدىكەن ، ئەتراپتىكىلەر دەسلەپ ئانچە جۈرئەت قىلالمايدىكەن ، لىكىن ئەتراپتا 3-4 شېرىكى بولغاچقا ئۇلار بىر ئىككى قېتىم پۇل قويۇپ ‹‹ئۇتۇش››ئارقىلىق ئەتراپتىكى ساددا كىشىلەرنى قىزىقتۇرىدىكەن.  مەن كۆرگەندە دەل شۇنداق بولدى ، يەنى  20 ياشتىن ئەمدى ئاشقان ئىككى بالا كېلىپ بىر ئاز قاراپ تۇرغاندىن كېيىن ‹‹ ئۇتقۇچىلارنىڭ ›› ئۇتۇقىغا قىزىقتىمۇ بىرسى دەرھال پۇل قويدى ، لىكىن ئۇ 10 مىنۇتقا يەتمىگەن ئەھۋالدە نەق 1500 يۈەننى شۇلارغا تاشلاپ بەردى ، مېنىڭ قىياسىمچە بۇ بالىلار باشقا يۇرتتىن بىرەر ئىش ئىزدەپ ئۈرۈمچىگە كەلگەندەك قىلاتتى، يېنىدا بار پۇلى شۇ بولسا كېرەك ،- ئاكا يۈز كويومنى قايتۇرۇپ بېرىڭ پۇلۇم قالمىدى دېدى ، لىكىن ئۇلاردا نە ئىنساپ ، نە ئىمان بولسۇن ، بولمىسا شۇ ئىشلارنى قىلارمىدى ، ئالدىراش نەرسىلىرىنى يىغىپ كېتىپ قالدى . ئۇ بالا چىرايىدا قان قالمىغان ھالدا قانداقتۇر خىيال ئىلكىدە تۇرۇپ قالدى ، ھەمرايىنىڭ ، -ماڭە ئاداش كېتەيلى ، دېيىشى بىلەن ئېغىر قەدەملىرى بىلەن كېتىپ قالدى. بىرەر يامان ئىشنى كۆرسەڭلەر ھەركىتىڭلار بىلەن توسۇڭلار، ئۇنۇڭغا قۇربۇڭلار يەتمىسە تىلىڭلار بىلەن توسۇڭلار، ئۇمۇ مۆمكۈن بولمىسا دىلىڭلار بىلەن توسۇڭلار ، دېيىلگەندەك ماڭا نە ئۇلارغا قارشىلىق ۋە ياكى تىلىم ئارقىلىق قارشىلىق قىلىش مۇمكىن ئەمەس ئىدى ، بىراق كۆڭلۈمدە بىزدىكى بۇ ئۆزىنى بىر ئۇيغۇر ، مۇسۇلمان سانايدىغان شۇ مۇناپىق كىشىلىرىمىزگە مىڭ لەنەت ئوقۇدۇم ھەم ئۈزۈمنى ئاچچىق بىر يىغا تۇتۇپ كەتتى ، كىمگە ، نىمىگە يىغلاي؟! شۇ مۇناپىق قويمىچىلارنىڭ نە ئىنساپ ، نە ئىمان بولمىغان قارا قەلبىگە لەنەت ئوقۇپ يىغلايمۇ ۋەياكى شۇ چاغدا تالاي قويۇلۇپ كەتكەن كىشىلىرىمىزنىڭ ساددىلىقى ، نادانلىقى، پۇلغا بولغان تويماسلىقىغا يىغلايمۇ بىلەلمەيلە قالدىم. مەنمۇ قانداقتۇر بىر قىممەتلىك نەرسەمنى يۈتتۈرۈپ قويغاندەك ئاچچىق بولۇپ يولۇمغا كېتىپ قالدىم.  شۇ ئىشلارنى  ئويلاپ چىقىشقا ئىشلىگەن كاللىسىنى ئىلىمگە، بىلىمگە ئىشلەتسە ، ھېچ بولمىسا يامانلىق قىلماسلىققا ئىشلەتسە قانچىلىك ياخشى بولار؟ يەنە بىر ئىچىم ئاچچىق بولغان يىرى باشقا مىللەت كىشىلىرى شۇ ئىشنى بىرھازا كۆرگەندىن كىيىن كىنايىلىك كۈلۈپ قويۇپ كېتىپ قالىدىكەن ، خۇددى -ھەي نادان خەلق، سەن مۇشۇ نادانلىقىڭ بىلەن ئارقىدا قالغانسەن دەۋاتقاندەك...   دۆڭكۆۋرۈكتەك ھەرخىل مەدەنىيەتكە ۋەكىللىك قىلىدىغان كىشىلەر ئاياغ باسىدىغان موشۇنداق ئورۇندا بىزدىكى بۇ خىل كىشىنى ئېچىندۇرىدىغان ئەھۋالنىڭ يۈز بېرىشى ھەقىقەتەن ئەپسۇسلىنارلىق .
ئاللاھ ھەممىمىزگە ئىنساپ ئاتا قىلسۇن ! مىللىتىمىزنى پاك ئىمانلىق ، ئىناق مىللەت قىلسۇن ئامىن!!
