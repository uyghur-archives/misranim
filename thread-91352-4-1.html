تۇخۇمدىن تۈك ئۈندۈرۈشمۇ؟ھەسەت قىلىشمۇ؟كۆرەلمەسلىكمۇ؟ئۆزىنى ھەممىنى بىلگۈچى ھىساپلاشمۇ؟ ھەرقانداق بىر ئىشقا يامان تەرەپتىن چىقىپ ھۆكۈم قىلىش ئاقىلانىلىق ئەمەس!
دىيارىمىزدا ئۆز قېرىنداشلىرىمىز ئىچىدە، چېكىدىن ئاشقان ئەخلاقسىز قىلمىشلار بولۇپ تۇرۇدۇ،ئاشنا ئويناش،رەڭلىك بەزمە،زەھەر چېكىش،ئىپپىتىنى يۇلتۇزلۇق مېھمانخانىلاردا ئاشكارە سېتىش،ئوغرى-قاراقچىلىق،ئالدامچىلىق،جازانىخورلۇق،كىشىنىڭ ھەققىنى يېيىش،شەيتان ئويۇنلىرىغا مىليون پۇل خەجلەپ سورۇن ھازىرلاش........دەپ تۈگەتكىلى بولمايدۇ.ئۇنداقتا ئۇيغۇر يۇرتىنى ۋە ئۇيغۇرلارنى مۇسۇلماندارچىلىقنىڭ سىرتىغا چىقىرۋېتەمدۇق،دىيارىمىزدا بولۇنىۋاتقان ساناپ تۈگەتكۈسىز،ياخشى ئىش ئەمەللەر،ساۋاپلىق ئىشلار،ياخشى ئادەملەرنى،ئىناق جەمئىيىتىمىزنى ئاشۇ ئەخلاقسىز ئىشلار يۈزىسدىن يوققا چىقىرامدۇق؟ھەممىمىز ئوغرى-قاراقچى،بەتقىلىقمۇ؟
بەزى كىشىلەر شۇنداقكى،باشقىلارنىڭ ئازراق ئارتۇقچىلىغىنى كۆرۈشكە شۇنچە چىدىمايدۇ،باشقىلارنىڭ ساۋاپلىق ياخشى ئىشلىرىنى شەكىلۋازلىق،مەلۇم مەنپەتنى چىقىش قىلغان دەپ تۇرۇۋالىدۇ،تىرناق ئاستىدىن كىر ئىزدەيدۇ،بۇ خىل ئادەملەرنىڭ ھىچقاچان ئۆز ئالدىغا مۇستەقىل تەپەككۈرى بولمايدۇ،ئەمما ھەقىقەتنى ئاستىن ئۈستۈن قىلىش ئۈچۈن،بۇلۇڭ پۇشقاقتىن پىكىر توپلايدۇ،بىرنەرسە دېسەڭ ئاشۇ ئەسكى-تۈسكىلىرىنى ئالدىڭغا يايىدۇ،تازنىڭ نېمىسى بار تۆمۈر تاغىقى بار،توردىكى ئىنكارچىلارنىڭ نېمىسى ۋار؟ ۋىكىپىېدىسى بار!
بىلمىگەننى بىلدىم ، كۆرمىگەننى كۆردۇم، ئاڭلىمىغاننى ئاڭلىدىم دېمە!
