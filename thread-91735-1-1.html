     ئەسسالامۇ ئەلەيكۇم ئەزىز قېرىنداشلېرىم ئاللاھ سىلەرگە ئامانلىق ئاتا قىلسۇن!:

مەن توردىشىمىز kukhelal نىڭ"بىشەم ئاياللارغا تاياق دورا "تېمىدىكى يوللانمىسىنى ھەم ئۇ تېمىغا يوللانغان ئىنكاسلارنى  كۆزدىن كەچۈرگەندىن كېيىن تۆۋەندىكى ھىكمەتلىك سۆزلەرنىمۇ  يوللاپ كويۇشنى لايىق كۆردۈم ...........


           (  لايىق تاللاش ئالدىدا تۇرغان ئوغۇل قېرىنداشلېرىمغا: مەن بەزى بىر ئوغۇل قېرىنداشلېرىمنىڭ كۆڭلىدىكى يوشۇرۇن  سۇئاللېرىغا تۆۋەندىكى  شائىېر، يازغۇچى  ئالىملېرىمىزنىڭ ھىكمەتلېرىدىن جاۋاپ چىقىپ قالا دەپ ئويلىدىم ..بەزى قېرىنداشلېرىم ماڭا ئۆيلەنمەسلىكنى قارار قىلغانلىقىنىمۇ دىگەنتى ھەرگىز ئۇنداق دىمەڭ قېرىندىشىم! )

                                                      **************************

      قۇياش بولمىسا، گۈل ئېچىلمايدۇ: مۇھەببەت بولمىسا ، بەخىت بولمايدۇ؛ ئاياللار بومىسا، مۇھەببەتمۇ بولمايدۇ.ئانا بولمىسا شائىرمۇ بولمايدۇ، باتۇرمۇ بولمايدۇ.

                                                                                           _گوركى

ئۆزۈڭگە << ئۆيلىنەيمۇ؟ ئۆيلەنمەيمۇ؟>> دەپ سۇئال قويما، ياخشىسى ئالدىڭغا << ياخشى ياشىشىم كېرەك>> دىگەن بىرلا مەخسەتنى قوي. ئەنە شۇ چاغدا ، ۋاقىت-سائىتى كېلىپ شۇنداق بىر ۋەزىيەت تۇغۇلىدۇكى. ئۆيلەنمەسلىكنىڭ ئىلاجى قالمايدۇ.

                                                                                         _ لېف. تولستوي

شۇنداق ئادەملەرمۇ باركى، ئۇلار مۇھەببەت بوستانىنى ئارلاپ، ئۆيلىنىشكە قەدەم قويمايدۇ ۋە بۇ پەيىز بېغىنىڭ گۈلزارىدىن گۈللەر ئۈزمەيدۇ، ئەكسىچە دائىم غەم يەپ ، ئۆيىدە ئۆزلېرى يالغۇز - تەنھا  ئولتۇرۇدۇ. بەزى بىر كەم ئەقىل بۇرنىنىڭ ئۇچىنىلا كۆرۈدىغان كىشىلەر بولسا، پۇلۇم يىتىشمەيدۇ دەپ ياكى  خۇتۇن ئالسا ئەپلەپ باقالماسلىقتىن ئەنسىرەپ گۈزەل ۋە بەختىيار ئاياللار ۋىسالىدىن يۈز ئۆرۈيدۇ. ئەمما مەلۈمكى، ئۆيلىنىش يولىدا ۋە ئۇنىڭ سەرف-خەرجى يولىدا قانچىلىك نەرسە ئىشلىتىلسۇن، كىيىنچىلىك بۇلار تولدۇرۋېلىنىدۇ. ئەمما بۇ يولنى بېسىپ ئۆتۈشكە تۆۋەندىكىلەرگە ئەمەل قىلماق كېرەك؛ شۇنداق قىزغا ئەلچى كىرگۈزۈلسۇنكى ، بۇ قىزنىڭ مىھرى ئىسسىق، يۈزىدىن ئالىيجاناپلىق ئىپپەت نامايەن بولسۇن، پاكلىق ۋە پاكىزلىق پەردىسى ئارقىسىدا تەربىيىلەنگەن، چىرايلىق ئىش ۋە ئادەتلەرگە ئۈگۈتۈلگەن بولسۇن! بۇنداق قىز ئەڭ ياخشى تارتۇق بولۇپ، ياشلىق ۋە ھايات داستىخىنىنىڭ لەززىتىنى مول قىلىدۇ.

                                                                                             _  بەرخۇردار ئىبن مەھمۇد

           خۇتۇن ئېلىشتا خۇتۇننىڭ مال-مۈلكىدىن، پۇلىدىن بىر نەرسە كېلىشنى ئىزدىمە ۋە خۇتۇننىڭ تولىمۇ گۈزىلىنى ئىزدىمە. ئالغان خۇتۇنىڭ شۇنداق بولسۇنكى، خۇتۇنلۇققا، ئۆي تۇتۇشقا لاياقىتى بولسۇن، ئۆي ئىشلېرىنى بىلىدىغان،  ئۆزىنىڭ ھالال يولدىشىنى سۆيگۈچى، ھايالىق، نومۇسلۇق،ئۆزىنى ھارامدىن ساقلىغۇچى بولسۇن.يەنە ئۆز مەرتىۋەڭدىن يۇقۇرى، ئۆزۈڭدىن باي خۇتۇننى ئالما. خۇتۇنىڭ ئالدىدا تۆۋەن ئورۇندا بولمىغايسەن.

             خۇتۇنىڭ بالاغەتكە يەتكەن ئەقىللىق بولۇشى لازىم. ئۇ ئۆي تۇتۇش يوللېرىنى بىلسۇن. بەك ياش ، بويىغا يىتەر-يەتمىگەن كىچىك قىزنى ئالما. چۈنكى مەرت ئادەملەر خۇتۇننى ئائىلىسىنى ياخشىلاش ئۈچۈن ئالىدۇ. پەقەت جىمائ ئۈچۈنلا ئالمايدۇ. ئالغان خۇتۇنىڭ ئاتا-ئانىسىنىڭ، مەكتەپنىڭ توغرا تەربىيەسىنى ئالغان، ئۆي تۇتۇش تەربىيەسىنى ئۆگەنگەن بولسۇن. ئۇنداق خۇتۇن قولۇڭغا كىرەر بولسا، ئۇنى ئېلىشتا بىپارۋالىق قىلما.

                                                                                                     _ قابۇس


                                                           ئەگەر ئۆيلىنىشنى خالىساڭ ئۆزۈڭ،

                                                           تاللىۋال خىلىنى، ئىتتىك قىل كۆزۈڭ.


                                                                                *****

                                                           تىگى ياخشى بولسۇن تۇخۇم ۋە ئۇرۇغ،

                                                          ئۇياتچان تەقۋا ئۆزى بەك ئېرىغ.


                                                                                *****

                                                          تىرىش، ئۆي قىزىنى ئال، قول تەگمىگەن،

                                                          سېنىڭدىن بۆلەك ئەر يۈزىنى كۆرمىگەن.


                                                                                *****

                                                          سېنىلا سۆيۈپ باشقىنى بىلمىگەن،

                                                          يارامسىز ، قىلىقسىز ئىشلارنى قىلمىغاي.


                                                                                *****

                                                           خۇتۇن ئالساڭ ئالغىن، ئۆزۈڭدىن تۈۋەن،

                                                           ئىسىلما ئىگىزگە  ئەسىر بولۇرسەن.


                                                                               *****

                                                           خۇتۇن ئالساڭ ئالغىن، تۈۋەن، توغرىنى،

                                                           سۆيۇنچ بىرلە ئۆتكەي  ھاياتىڭ كۈنى.


                                                                               *****

                                                           چىرايلىق دىمە، خۇلقى ئۆزنى دە،

                                                           كى خۇلقى ياخشى بولسا، يورۇتقاي سېنى.  


                                                                               *****

                                                           گۈزەل خۇينى ئىستە، ئاختارما جامال.

                                                          خۇيى ياخشى بولسا، تاپار ئۇ كامال.


                                                                              *****

                                                          ئايا باي خۇتۇنغا ئىنتىلگۈچى جان،

                                                          ئەسىر بولما ئاڭا، ئى ئالىم ئىنسان.


                                                                             *****

                                                          ئۇ مالغا سۈيەتىپ، تىلىنى ئۇزاتۇر،

                                                          ئۇنىڭ رەيى سېنى ھەر كويغا سالۇر.


                                                                            *****

                                                         ئەي ئەسلى بۈيۈككە ئىنتىلگۈچى سەن،

                                                         ئۈمرىڭنى خار ئەتمە بۇنىڭلىق بىلەن.


                                                                             *****

                                                         تىلەك بايلىق بولسا، بېرۇر مال يىغىپ،

                                                         بىيىتقاي ئۇ سېنى قىزىل يۈز قىلىپ.


                                                                             *****

                                                         دۇرۇس خۇيلۇق بولسا، گۈزەللىك كىلۇر،

                                                         خۇتۇننىڭ كۆركى ئەخلاق، بىلىملىك بىلۇر.
