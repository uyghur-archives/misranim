مىنىڭچە بۇنداق تۈگىمەس قائىدە يۇسۇنلار خەنزو يولداشلىرىمىزنىلا ئەمەس ، قېرىنداشلىرىمىزنىمۇ بىزار قىلدىغۇ دەيمەن ؟
   بىز ئۇيغۇرلار قائىدە يۇسۇنلۇق خەلق . ھۈكۈمەت خىزمەتچىلىرى دۈشەنبىدىن جۈمەگىچە ھۈكۈمەت ئۈچۈن ئىشلەيمىز . شەنبە يەكشەنبىلەر كەلسە بۇ كۈنلەر ئۆزىمىزگە تەۋە ئەمەس . تۈگىمەس توي تۈكۈن ، نېزىر چىراق ۋە ھەرخىل ناملاردىكى ئولتۇرۇشلار بىلەن بەك ئالدىراش بۇلۇپ كىتىۋاتىمىزمۇ قانداق ؟ مەن قارىسام ھىچقانداق بىر ئادەم مۇشۇنداق تۈگىمەس توي تۈكۈنلەردىن خوشال بۇلۇپ كەتمىدى ، توي تۈكۈنلەرنىڭ كۆپلىكىدىن قاھشىغىنى قاھشىغان . مىنىڭچە بىز مۇشۇنداق قائىدە يۇسۇنلىرىمىزنى ئازلاتساق بۇلاتتى . كېسەللەرنى يوقلايلى ! لېكىن بارسىمۇ بۇلىدىغان ، بارمىسمۇ بۇلىدىغان يەرلەرگە بارمىساق دەيمەن .
...............................................................................................
نىيىتىڭىز توغرا بولسىمۇ ،لىكىن بىر قىسىم پىكىرلىرىڭىزدە بىر تەرەپلىمىلىك بار. بولۇپمۇ قىزىل خەتلەردە
ئادەم تۇغۇلغان ئىكەن چوڭ بولۇپ توي قىلىدۇ. ئۇ ئۆلىدۇ، تېرىكلەر نەزىرىسىنى قىلىدۇ. شۇڭابۇ ئىككىسى ھەرگىز تۈگۈمەيدۇ ياكى ئازلىمايدۇ. بۇ قائىدە يوسۇنلىرىمىزنى ئازايتىش مۈمكىن ئەمەس. ھەتتا بۇ ئۇيغۇرلارغىلا ئەمەس پۈتكۈل ئىنسانىيەتكە ئورتاق ئىش. كېسەل يوقلاش چوڭ ساۋاپلىق ئىش. (ھەدىستىن كۆرۈۋىلىڭ)، يوقلىماي بولمايدۇ. دەرۋەقە 10يىلدىن بېرى قەشقەر شەھرىدە(قەشقەردە ئەمەس) نەزىر قىلىشپەسلەپ، يەنە بىر ئاز پەيدا بولۇپ تۇرىۋاتىدۇ، ئىختىيار قىلغانلار ئەرلەرگەئانچە مۇنچە قىلىۋاتىدۇ. قارىمايغا بارسام نەزىر قىلدى، ئاياللار كەلمىدى، سەھەردىلار ئەرلەر بىلەن چىرايلىق تۈگۈدى، ھېچقانداق ئىشقا پۇتلاشمىدى.  قالغان يۇرتلىرىمىزدا قانداق بىلمىدىم.
مەن قارىسام ھىچقانداق بىر ئادەم مۇشۇنداق تۈگىمەس توي تۈكۈنلەردىن خوشال بۇلۇپ كەتمىدى ، توي تۈكۈنلەرنىڭ كۆپلىكىدىن قاھشىغىنى قاھشىغان. گەپنى ئۇنداق "ھېچقانداق بىر ئادەم "دەپ كېسىپلا دەۋەتمەڭ، خوش بولىدىغانلارمۇ،  قاخشايدىغانلار بولىدۇ، لىكىن ھەممە ئادەم ئۇنداق بولىشى ناتايىن. پەقەت بولمىسا، ئاتا- ئانا خوش بولىدۇ.
10 مىليۇن ئۇيغۇرنىڭ قانچىلىكى ھۆكۈمەت خىزمەتچىسى؟ ئازراق خىزمەتچىنى دەپ دادىمىز ئۆلسە، نەزىرىسىنى قىلماي، ئاچىمىز، قەينى ئاتىمىز ئاغرىسا يوقلىماي يۈرەمدۇق؟ يۇرتقا توي بەرمەي، قىز- يىگىتلەر غىپپىدە بىر-بىرىنىڭ ئۆيىگە كىرىۋالسا بولارمۇ؟ مېنىڭچە (ئەگەر سىز توي قىلمىغان بولسىڭىز) شۇنداق قىلالامسىز؟ توي قىلغان بولسىڭىز بالىلىرىڭىزنىڭ تويىنى قىلماي قىزىڭىزنى چىقىرىپ بېرەلەمسىز، كېلىن خېنىمنى ئۆيگە ئەكىرىۋالالامسىز.
دەرۋەقە ئورۇنسىز، چاي، بەزمە- باراۋەتلەرنى ئازايتساق، يوقاتساق بولىدۇ. ئېھتىمال سىز مۇشۇ نۇقتىنى تەكىتلەۋاتقان بولىشىڭىز مۈمكىن. بەزى ئىشلارنى ئۇنداق ئاسان چەكلىۋەتكىلى بولمايدۇ
خەنلەرنىڭمۇ توي، ئۆلۈم ئۇزۇتىشىدا نۇرغۇن ئىشلىرى بار، بىزنىڭ كاللىمىزدىن ئۆتمەيدۇ. بىر خەن بىلەن پراڭلىشىپ قالدىم، ئۇ دەيدۇ: بىز تەرپلەردە ئاتا- ئانىمىز ئاغرىپ قالسا دوختۇر پۇلىدىن قورقمايمىز، ئۆلۈپ قالسا گۈەنسەي پۇلىدىن قورقىمىز دەيدۇ. نېمە ئۈچۈن؟ قىممەت.
بۇ يەردىكى خەنلەرنىڭ تۈزۈك ئۇرۇق تۇققانلىرى يوق. ئۆزى يەپ، ئۆزى چىقىرىدۇ، لىكىن بىز ئوخشىمايمىز.  بىرەر يات تائىپىدىكىلەرنىڭ گېپىنى دەستەك قىلىۋالمايلى.
يەنە بىرى بىر يېقىنىمنىڭ بالىسى يىللىقى 14500يۈەنلىك بىر ئالىي مەكتەپكە ئۆتۈپ قاپتىكەن. پۇلنى غەملىيەلمەي يۈرگەندە، ئۇرۇق- تۇققانلار، دوستلار ئىككى تۈركۈم بولۇپ ئۆيىگە باردۇق، بىرەر قاچىدىن لەغمەن يېيىشتۇق.  مۇبارەكلىدۇق، ياخشى تىلەكلەرنى تىلىدۇق، بالىغا ئىلھام بەردۇق، ئاتىغىنىمىزنى تاشلىدۇق. كىيىن بىلسەم يېرىمدىن كۆپرەك قىيىنچىلىق ھەل بوپتۇ، ھەممىمىز خوش بولدۇق.
