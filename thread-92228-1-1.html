پەرزەنت تەربىيسىنى قانچە بالدۇر باشلىسىڭىز  شۇنچە يۇقىرى ئۈنۈمگە ئېرىشىسىز


ھەر قانداق پەرزەنت ئوخشاش تۇغۇلىدۇ، لېكىن تەربىيلىنىش مۇھىتى ئوخشىمايدۇ، بالىنىڭ تالانتلىق بولىشى دەل تەلىم- تەربىيىنىڭ مەھسۇلىدۇر. پەرزەنت تەربىيىسىنىڭ باشلىنىش ۋاقتى ئۇنىڭ ئەقلىي قابىلىيىتىنىڭ قانچىلىك دەرىجىدە ئېچىلىشىنى بەلگىلەيدۇ. ئادەمنىڭ پۈتۈن ئۆمرىدىكى تەلىم-تەربىيەنىڭ ئاساسى ۋە ئاچقۇچى ـــ ئۆسمۈرلۈك دەۋردىن باشلىنىدۇ .پەرزەنت ـــ بۇ دۇنيادىكى سىزگە ئاتا قىلىنغان ئەڭ ياخشى سوۋغات ۋە بايلىق.

ئۇلۇغ مۇتەپپەككۇر، ئالىم يۈسۈپ خاس ھاجىپ بوۋىمىز بىزگە قالدۇرۇپ كەتكەن ئالەمشۇمۇل ئەسىرى «قۇتادغۇبىلىك» دە پەرزەنت تەربىيىسىنىڭ مۇھىملىقى توغرىسىدا شۇنداق قۇرلارنى قالدۇرغان ئىكەن.



« بالام، بۈگۈن سەندىنلا غەم يەۋاتىمەن،. بالىغا ئاتىنىڭ ئەمگىكى سىڭگەن بولسا ئوغلىنىڭ مىجەز-خۇلقى ئوبدان يېتىشىپ چىقىدۇ. ئاتا ئوغلىغا قاتتىق قول بولسا، يۈزى يورۇق بولىدۇ، ئەگەر قاتتىق قول بولمىسا ئوغلىنىڭ ۋاقتى زايە كېتىدۇ. ئوغلى بار كىشى زېرەك بولسا، ئۇنى دائىم قاتتىق تۇتۇشى كېرەك. ئوغۇل-قىزغا پەزىلەت،   بىلىم ئۆگىتىش لازىمكى، شۇ پەزىلەت بىلەن خۇي-پەيلى گۈزەل بولسۇن.»

                                 -«قۇتادغۇبىلىك»   يۈسۈپ خاس ھاجىپ



تەلىم تەربىيىنىڭ ئاساسى - گۆدەك چېغىدا بالىغا ئېتقاد، غايە، ساغلام تۇرمۇش ئادىتى، پاك قەلب ۋە جۇشقۇنلۇقنى يېتىلدۈرۈش باسقۇچى بۇلۇپ، بۇ باسقۇچ ناھايىتى مۇھىم.

ئاتا-ئانا پەرزەنت تەربىيىسىدە ناھايىتى زىل ھەم يۇقىرى تەلەپ قويۇشى، ئۇنى ئۆز مەيلىگە قويۇپ بېرىپ، كىيىن ئىلىمسىز ھەم يارىماس بولۇپ چوڭ بولغاندا، بالىسىدىن ئاغرىنىپ دەرت ئېيتىپ يۈرۈشنىڭ ناھايىتى ئەخمىقانە ئىش ئىكەلىكىنى، ئەگەر بالىنىڭ ياخشى تەربىيلىنىپ، ياراملىق ئادەم بولۇپ چىقسا، ئاتا-ئانا ئۈچۈن ھاياتىدىكى ئەڭ چوڭ نىممەت ئىكەنلىكى، ئەگەر ئۇ بىر كارغا كەلمەس، ئىلىمسىزلەردىن بولۇپ قالسا، ئاتا-ئانا ئۈچۈن ئەڭ چوڭ ئاپەت بولىدىغانلىقى ھەر بىرىمىزگە ئېنىق.

0 ياشتىن 8 ياشقىچە بولغان مەزگىل بالىنىڭ تەرەققىيات جەريانىدىكى ھەل قىلغۇچ مەزگىل، ئادەمنىڭ يۇشۇرۇن قابىلىيىتىنى يۇقىرى سۈپەتتە ئېچىش، تالانتىنى بايقاش ۋە جارى قىلدۇرۇش، ئەڭ ياخشى ئۆگىنىش ۋە تۇرمۇش ئادىتىنى يىتىلدۈرىشتىكى ئەڭ ھالقىلىق مەزگىل .

بالا تۇغۇلغاندىن كىيىنكى 4 يىللىق تەرەققىياتىنىڭ دەرىجىسى ۋە مۇھىملىقى بالىنىڭ پۈتۈن ئۆمرىدىكى ھەرقانداق ۋاقىتتىكىدىنمۇ بەك مۇھىم. ئەگەر بىز ھاياتلىقنىڭ ئۆزگۈرۈشى ،ھاياتلىقنىڭ ماسلىشىشى ،تاشقى دۇنيانى بويسۇندۇرۇش ۋە ئېرىشكەن نەتىجىلىرىمىزدىن قارىساق ،ئادەمنىڭ 0 ياشتىن 4 ياشقىچە بولغان مەزگىلدىكى ئەقلى قابىلىيىتى ئەمەلىيەتتە 4 ياشتىن يۇقىرى بولغان تاكى ئۆلگىچە بولغان ئارىلىقتىكى ھەرقانداق مەزگىلىنىڭ ئومۇمى يىغنىدىسىدىنمۇ مۇھىم ھەم چوڭ بولىدۇ ،شۇڭلاشقا بۇ 4 يىل مەزگىلى بىر ئادەمنىڭ تەلىم- تەربىيە مەزگىلىدىكى بىر ئۆمرى دەپ قارىساق بولىدۇ، 4 يىلدىن كېيىن تاكى ئۆلگىچە بولغان مەزگىلنى بىز كىينكى يېرىم ئۆمرى دەپ قارايمىز..ئۇنىڭدىن باشقا بۇ مەزگىلدە بالىنڭ ھەرىكەتكە بولغان ئېھتىياجى تاماققا بولغان ئېھتىياجىدىنمۇ كۈچلۈك بولىدۇ.

ئامېرىكا گېلوندىمان تەتقىقات ئورنىنىڭ تەجرىبە نەتىجىسى:تەجرىبە ئارقىلىق ئۇلار ئادەتتىكىدەك 2000 بالىنى تۆۋەندىكىدەك ئىقتىدارلارغا ئىگە تالانتلىق بالا قىلىپ تەربىيلەپ چىققان .

1. 4000 خىلدىن ئارتۇق شەيئىنى تونۇيالىغان .

2.  ئىككى ياكى ئىككىدىن ئارتۇق تىللار بىلەن، 4000 دىن ئارتۇق سۆزلەرنى ئوقۇيالىغان.

3.  ئىككى ياكى ئىككىدىن ئارتۇق تىلدىكى بەلگىلىك جۈملىلەرنى ،دىئالوگلارنى چۈشىنەلىگەن .

4.  كىتاپنى بىمالال ئوقۇيالىغان .

5.  ئاددى ھېسابلاشلارنى قىلالىغان .

6.  تەبىئەت دۇنياسىدىكى نۇرغۇنلىغان ھايۋانلارنى، ئۆسۈملۈكلەرنى ،گۈل-گىياھ، ئوت-چۆپ، دەل-دەرەخلەرنى،كۆكتاتلارنى، ھاشارەتلەرنى،تونىيالىغان.

7.  نۇرغۇنلىغان تەبىئەت ھادىسلىرى ۋە تۇرمۇشتكى ئىشلارنى دەپ بېرەلىگەن.

8.  خەت يازالىغان .

9.  نۇرغۇنلىغان قىزىقىشلىرى بولغان .( مەسىلەن: رەسىم سىزىش، گىمناستىكا ئويناش، سۇ ئۈزۈش قاتارلقلار.)

10. كۈچلۈك بىلىم ئېلش ئارزۇسى شەكىللەنگەن ،ئۆگىنىش قىلىش تۇرمۇشتىكى ئەڭ كۆڭۈللۈك ئويۇنىغا ئايلانغان.

11.   جانلىق-تېتىك بولغان شەخسىي خاراكتىرگە ۋە ساغلام، نورمال بولغان بەدەنگە ئىگە بولغان.

12.    ئەقلى تەپەككۇر كۆرۈلۈشكە باشلىغان.

بۇ بالىلارنىڭ 6 ياشقا كىرگەندىكى نەتىجىسى ئادەمنى تېخىمۇ ھەيران قالدۇرىدۇ، ئۇلارنىڭ ئىگىلىگەن تۇرمۇش بىلىملىرى ھەتتا ئاتا-ئانىلىرىدىنمۇ ئېشىپ كەتكەن .ئوقۇغان كىتاپلىرى،باشقا 16 ياشلىق بالىلارنىڭكىدىنمۇ جىق بولغان ھەتتا بەزى بالىلار ئۆزىنىڭ ئەسەرلىرىنمۇ ئېلان قىلغان.

ئادەم ساپال قاچىغا ئوخشايدۇ، كىچىك ۋاقتىدا ئۇنىڭ ئۆمۈرلۈك ھالىتى شەكىللىنىدۇ. بالىلىق مەزگىل ساپال قاچا ياسايدىغان سېغىز لايغا ئوخشايدۇ، قانداق تەربىيلەنسە، ئۇنىڭدا شۇنداق ھالەت شەكىللىنىدۇ. شۇنىڭ ئۈچۈن پەررزەنت تەربىيسىنىڭ ئاچقۇچلۇق مەزگىلىنى چىڭ تۇتۇپ، ھازىرلا تۇتۇش قىلىش، ئۇنى ئەڭ ياخشى تەربىيە مۇھىتى بىلەن تەمىنلەش، ھازىرقى زامان بالىلىرىنىڭ ئېرىشىشقا تىگىشىلىق ئاساسىي شارائىتى. شۇڭا بالا تەربىيىسىنى چوقۇم بالدۇرراق باشلاش كېرەك. قانچە بالدۇر باشلانسا، ئۈنۈمى شۇنچە ياخشى ھەم كۆرۈنەرلىك بولىدۇ، بالا تېخىمۇ مۇكەممەل ئادەم بولۇپ يېتىلىدۇ. چۈنكى بالىڭىزنىڭ ھازىرقى دەۋرى دەل سىزنىڭ ئويىڭىزدىكىدەك ياسىۋالغىلى بولىدىغان «سېغىزلاي» دەۋرى.

ئۇنىڭدىن سىرت، پەرزەنتىڭىزنىڭ ئەقلىي قابىلىيىتنىڭ ئومۇمىي ساپاسىنى يۇقىرى كۆتۈرەي دىسىڭىز، كۆپ تەرەپلىمە قىزىقىشىنى يېتىلدۈرۈشكە سەل قارىماڭ. شۇنى ئەسكەرتىش كېرەككى، ئۇ ئۆگىنىشتە ھەر قانچە ئىجدىھاد قىلسىمۇ، نېرۋىسى زەخمىلىنىپ كەتمەيدۇ. بوۋىمىز يۈسۈپ خاس ھاجىپ بىزگە بىلىم ئېلىشنىڭ ۋە بىلمنىڭ قەدرىنىڭ نەقەدەر مۇھىملىقىنى تەكىتلەپ شۇنداق قۇرلارنى قالدۇرغان ئىكەن.



«بىلىملىكنىڭ سۆزى ھەقىقەتەن سۆيۈملۈك جانغا ئوخشاشتۇر. تەلۋە (نادان) بىلىمنىڭ قەدرىنى نەدىن بىلسۇن؟ قەيەردە بىلىم تېپىلسا، بىلىملىك بېرىپ ئالىدۇ. تەمكىنلىك، ئالىيجاناپلىق ئەقىلنىڭ ئىشى. ئەقىللىقنىڭ سۆزىگە ئەمەل قىلسا، پايدىسى كۆپتۇر. بىلىملىكنىڭ سۆزى سىزگە كۆزدۇر.»     

  -«قۇتادغۇبىلىك»   يۈسۈپ خاس ھاجىپ



پەرزەنتىنى تەربىيلەش ئۈچۈن، ئاتا-ئانا ئالدى بىلەن ئۆزى ئەقىل-ئىدراكلىق بولۇشى، كۆپ تەرەپلىمە بىلىمگە ئىگە بولىشى كېرەك. ئەگەر سىز بۇ شارائىتلارنى ھازىلىمىغان بولسىڭىز، غەم قىلماڭ، داۋاملىق ئوقۇڭ، بۇنىڭغىمۇ ئامال بار.

بالىڭىزنىڭ خوشاللىقلىرى ئىچىدە ئەڭ مۇھىمى كىتاب ئوقۇش، بىراق بالىڭىزنىڭ ئوقۇيدىغان كىتابلىرنى تاللاشتا ئالاھىدە دېققەت قىلىش كېرەك. بىر ئادەمنىڭ قانداق كىتابقا ئامراق بولۇشى، ھەمىشە ئۇنىڭ تۇنجى قېتىم قانداق كىتاب ئوقۇغانلىقىغا قاراپ بەلگىلىنىدۇ، شۇنداقلا بالىلىق مەزگىلىدە ئوقۇغان كىتاب ھەمىشە بىر ئادەمنىڭ پۈتكۈل ئۆمرىگە تەسىر كۆرسىتىدۇ. كىچىك بالىلار باشقىلارنىڭ ھېكايە ئېيتىپ بېرىشىگە ئامراق كېلىدۇ. بۇ بالىنىڭ بىلىمىنى موللاشتۇرۇپلا قالماستىن، يەنە بالىنىڭ تېخىمۇ كۆپ كىتاب ئوقۇشىغا كۆۋرۈك بولالايدۇ. ئەگەردە ئۇنىڭغا ھەمىشە زوق-شوخ بىلەن ھىسىياتنى تولۇق ئىپادىلەپ ، ئوبرازلىق ۋە جانلىق تىل بىلەن پېرسوناژلارنىڭ ھەجۋىي رەسىمىلىرى بىلەن بىرگە مۇبالىغىلەشتۈرۈپ سۆزلەپ بەرگەندە، بالىنىڭ قىزىقىشىنى زور دەرىجىدە قوزغىغىلى، ۋە ئۇنىڭ سىز بىلەن بىرگە قوللىرىنى پۇلاڭلىتىپ، سىزنىڭ كۆزىڭىزگە تەلمۈرۈپ، زور قىزىقىش ۋە ھاياجانلىق ھىس تۇيغۇلار بىلەن تەلمۈرۈپ ئولتۇرغانلىقىنى بايقايسىز...ئەگەر كىتاپ ئۆزى سۆزلەپ كەتسىچۇ؟ ئەلۋەتتە، بالىڭىز كىتابتىن كۆزىنى ئۈزمەيدىغان بولىدۇ-دە، كىتابقا تەسەۋۋۇر قىلالمىغۇدەك دەرىجىدە قىززىقىپ، كىتابتىن ئايرىلالمايدۇ. چۈنكى كىتابنى ئۇ ئەڭ ياخشى ھەمراھىم دەپ بىلىدۇ. ئەلۋەتتە شۇ ئوخشاش بىر ھېكايىنى ئوبرازلىق كارتون رەسىملىرى بىلەن چۈچۈك ئاۋازلىق ئانا تىلدىن باشقا يەنە بىر قانچە تىللاردا ئوخشاشلا ھىسىياتلىق «سۆزلىيەلەيدىغان » كىتابنى بىز بارلىققا كەلتۈردۇق. بۇ دەل تەقەززالىق بىلەن كۈتكەن سۈمەييە «قوش تىل ئەڭگۈشتەرى»- بۇلبۇلگويا دۇر. بۇلبۇلگوياغا قىززىقسىڭىز ماقالىنى داۋامىلىق ئوقۇڭ.

ياپونىيەلىك ئالىم پوچۇن مېڭچىڭ 2 ياشتىن 12 ياشقىچە بولغان 200 بالىغا قارىتا ئەقلى ئىقتىدار ئېچىش تەجىربىسى ئېلىپ بارغان. بۇنىڭدا 4 ياشتىن كىچىك بالىلارنىڭ ئەقلى ئىقتىدارنىڭ يۇقىرى كۆتىرىلىشى نىسبىتى % 20 ئەتراپىدا بولغان، 4 ياشتىن 5 ياشقىچە بولغان بالىلارنىڭ %17 ئەتراپىدا بولغان، 6 ياشتىن چوڭ بالىلارنىڭ %15 ئەتىراپىدا بولغان .ئادەتتە نورمال ئادەمنىڭ ئەقلى قۇۋۋەت كويفېنسىنتى 90 دىن 110 ئەتراپىغىچە بولىدۇ .بۇ تەجرىبە "ئەقلى قۇۋۋەت پەقەتلا ئىرسىيەت خاراكتىرلىك بولىدۇ" دېگەن سەپسەتىنى بۇزۇپ تاشلىدى .ئالىملار دۇنيادىكى مۇزىكا، رەسساملىق، ئەدەبىيات، چەتئەل تىلى، تەنتەربىيە، پەن-تېخنىكا قاتارلىق ساھەلەردىكى 66 نەپەر ئۆسمۈر تالانتلىقلارنى تەكشۈرۈش ئارقىلىق شۇنى بايقىدىكى ئۆسمۈر تالانتلىقلار تۇغۇلۇپلا شۈنداق تالانتلىق بولۇپ تۇغۇلغان ئەمەس، بەلكى ئۇلار تەلىيىگە بالىلىق دەۋرىدە مائارىپ تەربىيسىگە ئالاھىدە ئەھمىيەت بىرىدىغان ئائىلىدە تۇغۇلغانلىقى ئۈچۈن، بالىلىق دەۋرىدىلا سېستىمىلىق مائارىپ تەربىيىسىنى قوبۇل قىلغانلىقى ئۈچۈندۇر. ھەرقانداق بىر بالا تۇغۇلغاندىلا غايەت زور يوشۇرۇن ئەقلى قۇۋۋەت بايلىقى بىلەن تۇغۇلىدۇ، ئۇنىڭ قانچىلىك قىسمىنى ئىشلىتەلەيدىغان بولۇش، بالىلىق دەۋىردىكى ئەقلى- قۇۋۋەت قابىليىتىنى ئېچىش تەلىم-تەربىيىسىگە باغلىق.

ئۇلۇغ ئالىم ئېينىشتىيىن ئەقلى قۇۋىتىنىڭ پەقەت %7 تىنىلا ئشلىتەلىگەن ئىكەن. ئادەتىكى كىشىلەر پۈتۈن ئۆمرىدە پەقەت %3 گىچىلا ئشلىتىدىكەن.قالغان %90 دىن ئارتۇق قىسمى، بىزنىڭ بالىلىق دەۋىردىكى تەربىيە ئارقىلىق ئېچىشىمىزغا باغلىق.خەلقىمىزدە مۇنداق بىر مەشھۇر ھىكمەتلىك سۆز بار: « بالىلىقتىكى ئۆگىنىش تاشقا مۆھۈر باسقاندەك، قېرىلىقتىكى ئۆگىنىش قۇمغا سۇ سىڭگەندەك ».شۇنىڭ ئۈچۈن دەيمىزكى پەرزەنت تەربىيىسى جەمىيەتنىڭ ئاساسى، بالىلىق مەزگىل يېتىلىشنىڭ ئانىسى.

ھەرخىل ئويۇن ئويناشمۇ ئەڭ ياخشى چارىلەردىن بىرى. بالا كىچىڭ چېغىدا ئويۇنغا قېنىپ چوڭ بولۇشى كېرەك. قىزىقارلىق كارتون رەسىملەر، كۆڭۈللۈك ئويۇن، يېقىملىق بالىلار ئاۋازى، بوۋاقلار ۋە ئۆسمۈرلەرنىڭ ئەقلى قابىلىيتىنى ئاچىدۇ. دەل شۇ ئويۇنلاردا نۇرغۇن بىلىملەر يوشۇرنغان بولىدۇ. تەربىيىدە ھەرگىزمۇ قاچىلاپ قويۇش شەكلىدىكى ئۆگىتىش ئۇسۇلىنى قوللانماڭ. چۈنكى بۇ خۇددىي دەل-دەرەخلەرنىڭ يوپۇرماقلىرىغا سۇ سەپكەندەك ئىش بولۇپ، سۇ ھەرگىز دەرەخنىڭ يىلتىزىغا يېتىپ بارالمايدۇ. ئۇنداقتا، دەرەخ قانداقمۇ يېتەرلىك نەملىككە ئېرىشەلىسۇن؟ بۇ خىل تەلىم-تەلىم تەربىيە خۇددىي ئائىلىدە باققان بىر ھايۋاننىڭ ئاغزىغا كۆپچىلىك بەس- بەس بىلەن بىر نەرسە تىقىۋېرىپ، ئاخىرىدا ئۇنىڭ ئاغزىنى تولدۇرۋەتكەن بىلەن ئوخشاش. ھەرخىل قىزىقارلىق ئويۇنلار ئارقىلىق پەرزەنتىڭىزنىڭ ئومۇمىي قىزىقىشىنى يېتىلدۈرۈشنى ئۇنتۇماڭ. بالىڭىز سىزنىڭ دائىم كىتاۋىي سۆزلەر ئارقىلىق، چوڭلارغا چۈشۈرگەندەك بۇيرۇق تەلەپپۇزىدىكى پەرمانلىرىڭىزنى بەجا كەلتۈرۈپ كۆنۈپ قالمىسۇن. بۇنداق بولغاندا ئۇنىڭ بالىلىق قىزىقىشى بارا- بارا ئۆلۈپ ھېچنىمىگە قىززىقمايدىغان بولۇپ قالىدۇ. سىز بالىڭىز چوڭ بولغاندا، «مېنىڭ بالام ھېچنىمىگە قىززىقمايدۇ، بەك جىمغۇر، ئادەم خالمايدۇ، تالاغىمۇ چىقمايدۇ» دەپ زارلانسىڭىز كېچىككەن بولىسىز. ئۇنىڭ 1-2 ياش چاغلىرىدىكى ئەمدى ئەقىل كىرىۋاتقان مەزگىلىدە سىزدىن چۈچۈك تىللاردا سورىغان سوئاللىرىغا نېمە دەپ جاۋاپ بەرگەنلىكىڭىزنى ئوبدان ئەسلەپ بېقىڭ... «بۇ بالام نېمىنى كۆرسە سوراۋاتقان، يوقىلاڭ گەپلەرنى قىلىپ، ئادەمنى بېزار قىلىپ...» دىگەنمۇ ياكى، ھەر بىر سوئالىغا ئەستايىدىللىق بىلەن جاۋاپ بەرگەنمۇ؟ ھەتتا ئۇ سورىغان بەزەن سوئاللارغا جاۋاپ تاپالمىغاندا، يالغاندىنلا بىر گەپلەرنى توقۇپ بالىڭىزنىڭ سەبىي قەلبىنى ئالدىغانمۇ ياكى «بالام، بۇ سوئالىڭ ھەقىقەتەن كىشىنى ئويغا سالىدىكەن، مەنمۇ بىلمەيدىكەنمەن، سەن ھەقىقەتەن چوڭ بوپسەن،-ھە! بىز بىرلىكتە ئىزدەپ تاپايلى» دەپ ئۇنى باشلاپ كۈتۈپخانا ياكى ئالاقىدار توربەتلەردىن شۇ توغرىسىدا ماتېريال ئىزدىگەنمۇ؟

بىز يۇقىرىدا تەكىتلەۋاتقان ئويۇن ھەرگىزمۇ بالىنىڭ تەن-سالامەتلىكى ۋە ئەقلى قابىلىيىتىنىڭ نورمال تەرەققىياتىغا زىيانلىق ئويۇنلار ئەمەس بەلكى قىزىقارلىق كارتون رەسىملەر،ئىنچىكىلىك بىلەن لاھىيەلەنگەن كۆڭۈللۈك ئويۇنلار، يېقىملىق بالىلار ئاۋازى بولۇپ بوۋاقلار ۋە ئۆسمۈرلەرنىڭ ئەقلى قابىلىيتىنى زور دەرىجىدە ئاچىدىغان مەنىلىك ئويۇنلارنى كۆرسىتىدۇ.

سۈمەييە «قوش تىل ئەڭگۈشتەرى» خاسيەتلىك قەلىمى ۋە يۈرۈشلۈك كىتابلىرى پەرزەنتىڭىزنىڭ يۇشۇرۇن قابىلىيىتىنى ئېچىش ئەقلى تەپەككۇر قىلىش ئادىتىنى يىتىلدۈرۈش ،تەبىئەت دۇنياسى، تۇرمۇشنى بەلگىلىك چۈشىنىش، ئاددى ماتېماتىكىلىق بىلىملەر،خەنزۇ تىلى، ئېنگىلىز تىلى ئاساسى بىلىملىرىنى ئويۇن ئىچىدە ئۈگۈنۈشىگە زور ياردەم بىرەلەيدۇ.

ئانا تىلىمىز ئۇيغۇر تىلىنى ئۆگىنىشتە ئەڭ ياخشى ياردەمچى بولالايدۇ، بارلىق كىتاب قىزىقارلىق ئويۇن، رەڭدار كارتون رەسم ،يېقىملىق مۇزىكا، چۈچۈك بالىلار ئاۋازى، ئەمىلى ھايۋاناتلارنىڭ ۋە جىسىملارنىڭ ئاۋازى بىلەن بىرلەشكەن بولۇپ، پەرزەنتىڭىزدە ئويۇن ئىچىدە بىلىم ئىگىلەپ، ئۆگۈنۈشنى تۇرمۇشتىكى ئەڭ ياخشى ئويۇن دەپ قارايدىغان ياخشى ئادەتنى ۋە كۆپ تەرەپلىمە قىزىقىشنى يېتلدۈرگىلى بولىدۇ. ئەلۋەتتە، سىزمۇ پەرزەنتىڭىزگە كۆپرەك ھەمرا بولۇپ بېرىڭ .

تۇغۇت بۆلۈمىدە ئوخشاش تۇغۇلغان بوۋاقلاردىن نېمە ئۈچۈن شۇنچە ئوخشىمىغان پەرزەنتلەر تەربىيلىنىپ چىقىدۇ؟ دەل ئۇ تۇغۇلۇپ ئۆسكەن ،تەربىيلەنگەن مۇھىت ۋە تەربىيلىنىش ۋاقتىنىڭ بالدۇر كىيىنلىكى شۇ بوۋاقنىڭ ئەقلىي قابىليىتىنىڭ ئېچىلىش دەرىجىسىنى بەلگىلىگەن. يەنى، ئۇنىڭ «سېغىزلاي دەۋرى» ئەھمىيەتلىك تەربىيىنى قوبۇل قىلغان.ئۇنىڭدىن سىرت كۆپ تەرەپلىمە قىزىقىشىنى يېتىلدۈرۈشكە سەل قارىماي، كىچىڭ چېغىدىن باشلاپلا ئىنچىكلىك بىلەن لاھىيەلەنگەن مەنىلىك ئويۇنلار بىلەن تەربىيلەنگەن ۋە تەربىيىدە ھەرگىزمۇ قاچىلاپ قويۇش شەكلىدىكى ئۆگىتىش ئۇسۇلىنى قوللانماي ئۇنىڭ قىزىقىشىغا ھۆرمەت قىلغان. شۇنىڭغا ئوخشاش ئامىللار ئارقىلىق ھېچ كىچىكىدە ھېچ بىر تەربىيە كۆرمىگەن «تاشلاندۇق بالىلار» بىلەن ئەتراپلىق تەربىيلەنگەن «تەلەيلىك بالىلار» تەربىيلىنىش مۇھىتى ۋە ئاتا-ئانا ساپاسىنىڭ ئوخشاشماسلىق تۈپەيلىدىن چوڭ بولغاندا بەزىلىرى «بالا-قازا» بولسا، بەزىلىرى ئاتا ئۈچۈن ئەڭ چوڭ «نېممەت» بولغان.ئەلۋەتتە ئەتراپىمىزدا شۇنىڭغا ئوخشاش مىساللارنى كۆپلەپ ئۇچىرتىشىمىز مۇمكىن.

مىڭ ئاڭلىغاندىن بىر كۆرگەن ئەلا!

سۈمەييە «قوش تىل ئەڭگۈشتەرى» يۈرۈشلۇك ئاۋازلىق كىتابلىرى، ئۇيغۇر تىلى ئېلىپبەسى، بالىلار ناخشىسىدىن خەنزۇچە خەت ئۆگىنىش 1-2 –قىسىم، ھايۋاناتلار دۇنياسىدا پىنيىن ئۆگىنىش، ھايۋاناتلاردۇنياسىدا ئىنگلىز تىلى ئۆگىنىش، ھايۋاناتلار دۇنياسىدىكى ماتېماتىكا، تەبىئەتنى كۈزىتىش، تۇرمۇشتكى بىلىملەر، ئەقلى تەپەككۈرى قىلىش ئادىتىنى يىتىلدۈرۈش، يۇشۇرۇن قابىلىيەتنى ئېچىش 1-2-3-4-5 - قىسىم، قاتارلىق 14 دانە كىتاب ۋە بىر دانە 34 بەتلىك ئاسما كالېندارلار توپلىمىدىن تۈزۈلگەن بولۇپ 0 ياشتىن 17 ياشقىچە بولغان بالىلارغا ناھايىتى ماس كېلىدۇ. قىزىقارلىق كارتون رەسىملەر، كۆڭۈللۈك ئويۇن، يېقىملىق بالىلار ئاۋازى، بالىلارنىڭ ئەقلى قابىلىيتىنى ئېچىش بىلەن بىرلىكتە مىللەتنىڭ كەلگۈسى تەرەققىياتى ۋە ئانا تىلنىڭ گۈللەپ ياشنىشى ئۈچۈن ھەر بىر ئائىلىدە كەم بولسا بولمايدىغان ياخشى قورال ھېسابلىنىدۇ.

پەرزەنتلىرىڭىزگە كۆپرەك ھەمراھ بولۇش بىلەن بىرگە ئۇنىڭغا چىن قەلبىڭىزدىن غەمخورلۇق قىلىشىڭىز، تېلىۋۇزۇر كۆرگەندىن ۋە مەنىسىز قاتار-چايلاردىن كۆپ كۆڭۈللۈك.

«قوش تىل ئەڭگۈشتەرى» خاسيەتلىك قەلىمى ۋە يۈرۈشلۈك كىتابلىرى پەرزەنتىڭىزنىڭ يۇشۇرۇن قابىلىيىتىنى ئېچىش ئەقلى تەپەككۇر قىلىش ئادىتىنى يىتىلدۈرۈش ،تەبىئەت دۇنياسى، تۇرمۇشنى بەلگىلىك چۈشىنىش، ئاددى ماتېماتىكىلىق بىلىملەر، خەنزۇ تىلى ۋە ئېنگىلىز تىلى ئاساسى بىلىملىرىنى ئويۇن ئىچىدە ئۈگۈنۈشىگە زور دەرىجىدە ياردەم بىرەلەيدۇ.

سۈمەييە «قوش تىل ئەڭگۈشتەرى» ئاتا-ئانىلار بىلەن بىرگە ئۆزىنىڭ ئەقلىي قابىلىيىتنى ئېچىش يۈرۈشلۈك ئاۋازلىق كىتابلىرى ۋە كىيىنكى تەتقىقات مېۋىلىرى ئارقىلىق مىللىتىمىزنىڭ كەلگۈسى بولغان ئوماق پەرزەنتلىرىمىزنىڭ گۈزەل ئەتىسىنى يارىتىش ئۈچۈن پۈتۈن كۈچى بىلەن تىرىشىدۇ، ھەم بالىلىرىمىزغا شۇنداق بالىلىق ياردەم بېرەلەيدىغانلىقىدىن سۆيۈنىدۇ، شۇنداقلا بۇنى شېركەتنىڭ تۈپ مىزانى قىلىدۇ.

مەن «بالىنى ئەڭ ياخشىسى قوساقتىكى ھامىلە ۋاقتىدىن باشلاپ تەربىيىلەشنى باشلىسا، بولسا تېخى قوساقتا پەيدا بولماي تۇرۇپ تەربىيەلەشنى باشلىسا تېخىمۇ ياخشى بولىدۇ» دەپ قارايتتىم. چۈنكى مەن غەزەللي(1058) [خۇراسان]، ئىبنى ھەجەر ئەسقالانى(1442-1372) [مىسىر قاھىرە] نىڭ  «ئەخلاقنامە» (ھىدايەتنامە) دېگەن كىتابىدىن مۇنداق قۇرلارنى ئوقۇغان ئىدىم.
مۇسۇلمانلارنىڭ پەرزەنت تەربىيىسى قايسى نۇقتىدىن باشلىنىدۇ؟ ناماز ئوقۇش يېشىدىنمۇ؟ ياكى خەتمە قىلىش، قۇلاق تېشىش يېشىدىنمۇ؟
قۇرئان كەرىمدە: «ئۆز ۋاقتىدا ئىمراننىڭ ئايالى ئېيتتى: (پەرۋەردىگارىم! مەن قورسىقىمدىكى پەرزەنتىمنى( دۇنيا ئىشلىرىدىن) ئازاد قىلىنغان ھالدا چوقۇم سېنىڭ خىزمىتىڭگە ئاتىدىم. (ئۇنى) قوبۇل قىلغىن. سەن ھەقىقەتەن( دۇئايىمنى) ئاڭلاپ تۇرغۇچىسەن. ( نىيىتىمنى) بىلىپ تۇرغۇچىسەن. ئۇ قىز تۇغقان ۋاقتىدا: ‹ئى پەرۋەردىگارىم! قىز تۇغدۇم› دېدى. ئۇنىڭ نېمە تۇغقانلىقىنى ئاللا ئوبدان بىلىدۇ. (ئۇ تەلەپ قىلغان) ئوغۇل (ئاتا قىلىنغان) قىزغا ئوخشاش ئەمەس. (بەلكى) بۇ قىز ئارتۇقتۇر. ئىمراننىڭ ئايالى ئېيتتى:«مەن ئۇنى مەريەم ئاتىدىم، مەن ھەقىقەتەن ئۇنىڭ ئۈچۈن، ئۇنىڭ ئەۋلاتلىرى ئۈچۈن ساڭا سېغىنىپ، قوغلاندى شەيتاندىن پاناھلىق تىلەيمەن.» دەپ بايان قىلىنغان.(سۈرە ئال ئىمران35-36ئايەتلەر)
يەنە سۈرە مەريەمدە:« ئۇ ئۆز ۋاقتىدا پەرۋەردىگارىغا پەس ئاۋازدا مۇناجات قىلدى، ئۇ ئېيتتى:« پەرۋەردىگارىم! مەن كۈچ-قۇۋۋىتىمدىن كەتتىم. چېچىم ئاقاردى. پەرۋەردىگارىم! ساڭا دۇئا قىلىپ ئۈمىدسىز بولۇپ قالغىنىم يوق. مەن تاغىلىرىمنىڭ بالىلىرىنىڭ ئورنۇمنى باسالماسلىقىدىن (يەنى، دىننى قولدىن بېرىپ قويۇشلىرىدىن) ھەقىقەتەن ئەنسىرەيمەن. منېىڭ ئايالىم تۇغماستۇر. (پەزلىڭدىن) ماڭا بىر (ياخشى) پەرزەنت ئاتا قىلغىن. ئۇ ماڭا ۋارس بولغاي ۋە يەئقۇپ ئەۋلادىغا ۋارس بولغاي. پەرۋەردىگارىم، ئۇنى سېنىڭ رازىلىقىڭغا ئېرىشىدىغان قىلغىن.» (سۈرە مەريەم2-5-ئايەتلەر) دەپ بايان قىلىنغان.
