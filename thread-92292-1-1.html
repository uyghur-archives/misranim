  مەزكۇر تېما قۇتادغۇبىلىك تورىدا ئېلان قىلىنغان شۇ ناملىق خەت بۇلۇپ، كىشىنى ئويلاندۇرىدىغان، قايىل قىلارلىق تېما ئىكەن، مەنچە بۇ تېما ھەر بىر سىتودېنتلىرىمىزنىڭ بىر ئوقۇپ بېقىشىغا ئەرزىگۈدەك، تەپسىلاتى تۆۋەندە

ئەسسالاممۇ ئەلەيكۇم:
   ئاتا، ئاپا سىلەرگە سالىدىغان بىر مەسلىھەتىم بار ئىدى. خىزمەت ۋاقتىڭلار ۋە مىنىڭ دەرىس ۋاقىتلىرىمنىڭ ئۇدۇل كېلىپ قېلىپ تېلفۇن پارىڭىمىزنىڭ ئۇزاققا بارماسلىقىدىن ئەنسىرەپ خەت ئارقىلىق ئالاقىلىشىشنى لايىق تاپتىم.

   مەن تۇللۇق ئوتتۇردا ئوقۇۋاتقان ۋاقىتلىرىمدا پەقەت ياخشى بىر ئالىي مەكتەپكە كىرىۋېلىپ شۇ ئارقىلىق يول تېپىشنى ئويلايتىم. ئۇ چاغدا ئالىي مەكتەپ دېگەن مېنىڭ نەزىرىمدە ئاجايىپ بىر گۈزەل ئىلىم-مەرىپەت ماكانى ئىدى. ئۇنى شۇنداق تەسسەۋۋۇر قىلاتتىمكى« ھەرگىزمۇ تۇللۇق ئوتتۇردىكىگە ئوخشاش ھەممە دەرىسلەرنى ئادەمگە زورلاپ ئۆتمەيدۇ، ئوقۇغۇچى قايسى ئىلىمنى ئۆگەنگۈسى كەلسە شۇ ئىلىمنى ئۆگىنىدۇ، ئوقۇتقۇچىلارنىڭ بىلىم دائىرىسى كەڭ بىز بىلەن ھەمسۆھبەتتە بولالايدۇ، ئوقۇغۇچىلىرى ھەممىسى تاللىنىپ بارغان بولغاچقا ۋاقىتنى بېكارغا تاينى يوق ئىشلارغا سەرىپ قىلىدغان، تۇللۇق ئوتتۇردىكىدەك ئوينايدىغانلار يوق، ھەممىمىز رىقابەتلىشىپ ئىلىم-پەننىڭ چوقىسىغا چىقىش ئۈچۈن تېنىمسىز تىرىشىمىز، شۇنداق قىلىپ تۆت يىل ئۆگەنسەك قولىمىزدىن ھەرقانداق چوڭ ئىشلار كېلىدىغان دەرىجىدە قابىلىيەتلىك بولۇپ چىقىمىز، قولدا ھۈنەر بولغاندىنكېيىن ھىچكىمگە يىلىنىپ يالۋۇرۇپ ياشىمايمىز. . . . ئالىي مەكتەپ بولسا مىنىڭ بۇ ئارزۇ-ئارمانلىرىمنى ئەمەلگە ئاشۇرىدىغان جاي»دەپ ئويلاپ قاپتىكەنمەن.

  نىڭشىيادا بىر يىل تەييارلىقتا ئوقۇۋاتقان چاغلاردا بۇ كۆز قارىشىم پۈتۈنلەي ئۆزگىرىپ كەتتى. ئالىي مەكتەپ كۆزۈمگە جەمىئيەتنىڭ ئەڭ قاراڭغۇ زۇلمەتلىك جايلىرىدەك كۆرۈنۈپ كەتتى. شۇنداق ساپ پاك بالىلار جىمى يامان ئىشلارنى ئالىي مەكتەپكە كېلىپ قىلىشنى ئۆگىنىدىكەن. شۇنداق ئاددى-ساددا بالىلار ئالىي مەكتەپكە كەلگندىن باشلاپلا ھەشەمەت قوغلىشىدىغان، ئابرۇي پەرەس، يۈز-تاپساملا دەيدىغان كېبىرلىك بوپ كېتىدىكەن. پاك قىزلار بۇزۇلۇپ ھاراق سورۇنلىرىغا ئۆزىنى ئاتىدىغان بوپ كېتىدىكەن. ئۈچ يىل تۇللۇق ئوتتۇردا بېسىم ئىچىدە ياشاپ ئۇنىڭغا چىدىيالمىغان بالىلار ئالىي مەكتەپكە چىقىپلا بارىلىق بېسىملاردىن خالاس بولۇپ ئۆزىنى ھارام ئىشلارغا ھاياتنىڭ قىسقىغىنە خۇشاللىقلىرىغا ئاتىدىكەن. ھاراق دىگەننى ھاۋا دەپ نەپەس نىڭ ئورنىغا ھاراقنى دەسسىتىدىكەن. «ئۆگىنىش»؟؟؟ ھاھاھا ئالىي مەكتەپتە ئۆگىنىش نېمە قىلسۇن؟ئىلىم-مەرىپەت نېمە قىلسۇن؟ قېنى كىم ئاشۇ سورۇن-ئولتۇرۇشلار، ئۇچۇرشىشلار، باغچا ۋە چىرايلىق ساياھەت ئورۇنلىرىنىڭ مەرىدىن كېچىپ ئۆزىنىڭ گۈزەل ياشلىق باھارىنى ئارمىكىنىڭ نوبىل مۇكاپاتىغا ئېرىشكەن ئالىملىرىمۇ تەستە ئىشلەيدىغان ياكى ئىشلىيەلمەيدىغان قىلچە ئەھمىيەتسىز يۇقۇرى تەرىجىلىك ماتىماتكا سۇئاللىرىنى ئىشلەشكە، بەرىبرى مەۋسۇم ئاخىرىدا ئوقۇتقۇچى ئىمتىھانغا چۈشىدىغان جايلارنى كۆرسىتىپ بېرىدىغان، شۇ چاغدا ئازراقلا يادىلاپ قويسا تولۇق نۇمۇر ئالغىلى بولىدىغان دەرىسلەرنى يادىلاشقا، پارتىيە ئەزالىرى يىغىنى، ئىتتىپاق ئەزالىرى يىغىنى دېگەندەك قۇرۇق يىغىنلارنى ئاڭلاشقا، (ئالىي مەكتەپتە ۋاقىت ئىسراپ قىلىشتىن باشقا ھېچقانداق ئەھمىيىتى يوق يىغىن دوكىلاتلار بەك جېق)كىتابلىرىنى ئۆمۈچۈك تورى قاپلاپ كەتكەن 10~20 يىل بۇرۇن يېزىلغان كونا ئىدىيە، قالاق ئىلىملەر خاتىرلەنگەن قىرائەتخانىلارغا كىرىشكە، ئاللا ۋە ئاخىرەت يولىدا ئىبادەت(يالغۇز نامازنىلا كۆرسەتمەيدۇ!ئاللاھنىڭ رىزالىقى ئۈچۈن قىلىنغان بارىلىق ياخشى ئەمەللەر)قىلىش. . . . يولىدا «ئىسراپ»قىلىشقا كىم ئۇنايدۇ؟؟ئۇناۋاتقانلارنىڭ ساننى بىر تال گۈرۈچ دېسەك ئۇنىمايۋاتقانلار بىر غايەت زور ئاشلىق ئامبىرى. مەن تۇللۇق ئوتتۇردا ئاساسەن مەكتەپتە بولغاچقا جەمئيەتنى تازا چشەنمەپتىكەنمەن. ئالىي مەكتەپ بولسا جۇڭگو جەمئىيتىنىڭ بىر كىچىكلىتىلگەن كۆرۈنىشىكەن. نىڭشىيادا ئاشۇ ئەھۋاللارنى كۆرگەن ۋاقتىمدا ئۆزەمگە ھە قاچان تىينجىن ئۇنۋېرستىتى ئۇنداق ئەمەستۇ دەپ تەسسەللىي بەردىم. بۇ يەرگە كېلىپ بىلدىمكى ھەتتا چىڭخۇا، بىېيجىڭ ئۇنۋېرستلىرىدىمۇ ئەھۋال پۈتۈنلەي ئوخشايدىكەن.

جۇڭگودا ئالىي مەكتەپتە ئوقۇپ ئوقۇغۇچىلارنىڭ ئېرىشىدىغىنى بىر ھەقىقىي ئەسقاتىدىغان ھۈنەر كەسىپ بولماستىن بەلكى قورۇق دېپلومدىن ئىبارەت بىر كېنىشكا بولىدىكەن خالاس، مەيلى بىز جۇڭگودا ئاسپىرانىتلىقتا ئوقۇيلى، مەيلى دوكتۇرلۇقتا ئوقۇيلى يەنىلا ئىلىم-پەن ساھەسىدىكى كۆزلىگەن مەقسىدىمىزگە يېتەلمەيدىكەنمىز. چۈنكى جۇڭگودا ھەقىقىي ئالىي مائارىپ تېخى شەكىنلەنمەپتۇ. ئالىي مەكتەپ پەقەت باللارنىڭ قىزغىنلىقىنى سوۋۇتىدىغان، ئۇلارغا ساختىپەزلىكنى، خۇشامەتچىلىكنى ئۆگىتىدىغان ئومۇمەن جۇڭگودىن ئىبارەت بۇ غەلىتە دۆلەتتە قانداق ھارام يوللار ئارقىلىق باشقىلارغا ئورەك كولاپ ياشاشنى ئۆگىتىدىغان بىر رەزىل جەمئىيەت ئىكەن. . .

جوڭگۇ مائارىپىنىڭ نەقەدەر قورقۇنۇچلۇق ئىكەنلىكىنى بەلكىم سىلەر تېخى ھېس قىلمىغان بولىشىڭلار مۇمكىن!بۇ خىل مائارىپ باللارنىڭ يېڭىلىق يارىتىش ئىقتىدارىنى پۈتۈنلەي يوقىتىپ، باللارنى ماشىنا ئادەمگە ياكى قۇلغا ئايلاندۇرۇپ قويىدىكەن، بالىلار تەبىيلا ئوقۇتقۇچى بىر دېسە بىر، ئىككى دېسە ئىككى دېيىشنى ئۆگىنىدىكەن ۋە تەپپەككۇر قىلىشنى ئەسلا ئويلاپ كۆرمەيدىكەن. بىزنىڭ مەكتەپتەپتىكى خەنزۇلار ئاساسەن 650 ئەتىراپىدا نومۇر ئېلىپ كەلگەن باللار، لىكىن ئۇلار بىلەن پاراڭلىشىپ باقساق ئۇلارنىڭ قورسىقىدا ھېچنىمە يوقلۇقىنى، ئۇلارنىڭ تولىمۇ گۆدەك ئىكەنلىكىنى، ئەتىراپىدا نېمە ئىش بولىۋاتقانىقى بىلەن كارى يوق سۆزسىز ئىتائەتمەن دۆت قۇل ئىكەنلىكىنى بايقايدىكەنمىز. ئۇلارنىڭ بۇ ھالىغا قاراپ «توۋا ئەينى ۋاقىتتىكى ساۋساۋ، گۈەنيۈ، جۇگېلىاڭلار. . . چىققان مىللەت مۇشىمىدۇ »دەپ ئويلاپ قالىمەن. ئۇلارنىڭ مائارىپى تۈگىشىپتۇ!
ئۇنىڭ ئۈستىگە مېنىڭمۇ شەيتىنىم كۈچلۈك بەزىدە تۈزگەن پىلانلىرىم دوسىت تارتىشىپ بۇزۇلۇپ كېتىدۇ، قايتا-قايتا پىلان تۈزۈپ كەتتىم. بىراق ئادەم ئۆزى تۇرىۋاتقان مۇھىتنىڭ تەسىرىگە ھامان ئۇچىرايدىكەن. بۇ خۇددى بىزنىڭ ئۆينىڭ يەرئاستى ئۆيىدە تاشلاپ، تۆكۈۋېتىلمەي خەنزۇ باشلىقلارغا ئېلىپ بېرىشقا ساقلاپ قويۇلغان ھاراقلار بولغىنىغا ئوخشايدۇ.
ئەمدى ئاساسىي مەقسەتكە كەلسەك قۇدرەت ياقۇپمۇ تۇللۇق ئوتتۇرنى پۈتتۈرگىنىگە 10 يىل بوپتۇ. بۇ جەرياندا ئۇ نۇرغۇن ئىشلارنى باشتىن ئۆتكۈزدى. مانا ھازىر 30 ياشقا كىرمەي تۇرۇپلا ئۆزىنىڭ ئىگىلىگىنى تىكلىدى. مەنمۇ ئەمدى مۇشۇ يولنى بويلاپ ئۆزەمنىڭ چىقىش يولۇمنى تاپماقچى. ھازىرقى ئوتتەك قىزغىنلىقىم بۇ مۇھىتتا سۇسلىشىپ كېتىۋاتىدۇ. ئۆگىنىش سۈرئىتىممۇ بەك ئاستا بولۇپ قالدى. شۇڭا بىر يىل ئوقۇشتىن ۋاقتىنچە توختاپ ئۈرۈمچىگە قايتىپ بىر يىل ئۆزەمنى قۇدرەت تاپقۇزايمىكىن دەيمەن. ئويلاپ باقسام مەن ئۆگىنىشكە تېگىشلىك ئەقەللىي بىلىملەرمۇ بەك كۆپكەن. مەسىلەن :نۇتۇق قابىلىتىمنى ئاشۇرۇش، بىرەر ئۇيغۇرچە ساز چېلىشنى ئۆگنىش، خېتىمنى ئوڭشاش، ئېڭگىلىزچىنى ياخشى ئۆگىنىش ھەم دىن جەھەتتىن ئۆزەمدىكى كام بىلىملەرنى تولۇقلاش. . . دېگەندەك.

ئەمدى ھەرقانداق بىر ئاتا-ئانا بالامغا ياخشى بولسىكەن دەيدۇ. سىلەر تا ھازىرغىچە تەلىپىمنى رەت قىلىپ باقمىدىڭلار. ئۆزەمنىڭمۇ ئويلىغان يېرىم بار. ئشقىلىپ ۋاقىتنى بېكارغا ئۆتكۈزىۋەتكىم كەلمىدى. بۇ يەردە ئوقىۋەرسەم پىلانىم كەينىگە سۆرىلىۋەرگەچكە ئۆزەمگە بولغان ئىشەنچىم ئاجىزلاپ كېتىدىكەن. قىزغىنلىقىمدىن ئايرىلىپ قالغىم يوق!ئىقتىدارىمنى بۇ يەردىكى بېكار ئىشلارغا سەرىپ قىلىپ تۈگىتىۋەتسەم مېنى ئەقىل بىلەن ياراتقان ئاللاھقا قانداق يۈز كېلەلەيمەن؟سىلەرگە قانداق يۈز كېلەلەيمەن؟بەلكىم ئويلاندىغانسىلەر«بالام مەكتەپنى پۈتتۈرۈپ كەلسە ھۆكۆمەتكە بىر خىزمەتكە ئالدۇرۇپ قويساق شۇ مائاش بىلەن بىزنى بېقىپ بىر ئۆمۈر پىنسىيەگە چىقىپ كەتسە. . . »دەپ. بىراق ئاخىرەتتە ئاللاھ ساڭا بەرگەن ئەقىلنى نېمىگە ئىشلەتتىڭ دېسە نېمە دەيمەن؟«باشلىقنىڭ ئۆيىگە پارا ئەكىرىپ ئاران تاپقان خىزمەتكە شۈكرى قىلىپ جىم يۈردۈم» دەمدىم؟؟؟پەيغەمبەر ئۈممەتلىرى ئالغا ئىنتىلىشچان غايىلىك، تىرىشان، چىداملىق، قۇدرەتلىك بولىشى كېرەك!شۇڭىلاشقىمۇ پەيغەمبەر ئەلەيسىلام دەۋرىدە ئەرەب دۇنياسى ئەڭ تەرەققىي قىلغان زىمىنغا ئايلانغان. مېنىڭ مەكتەپتىن توختىشىم ھەرگىزمۇ ئەمدى ئالىي مەكتەپكە بارمايمەن دېگىنىم ئەمەس! بىر يىل ئىچىدە مەنمۇ چوقۇم ئەينى ۋاقىتتا قۇدىرەت ياقۇپ بەرگەن ئمتىھانلاردىن چوقۇم ئۆتۈشكە ھەم يۇقۇرى نومۇر ئېلىپ ھەقسىز ئوقۇشنى قولغا كەلتۈشكە سىلەرگە ۋەدە بېرىمەن!قارارىمنى ئويلىنىپ بېقىشىڭلارنى سوراپ
