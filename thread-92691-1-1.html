     بۇ بىر رېئاللىق ، مەلۇم تۇنۇشۇمنىڭ ئەمەلىي ئەھۋالى ، ئۇنىڭ رۇخسىتىنى ئالغاندىن كېيىن بۇ يەرگە ئېلىپ كەلدىم .             
             مۇھەببەتتىكى ئىككىلىنىش
   ئۇنىڭ يېشى ھازىرقى دەۋرگە نىسبەتەن چوڭ ھېسابلانمايتتى ،  " قىزبالا ۋاقتىدا توي قىلمىسا ، بارلىق گۇناھ ئاتا - ئانىغا ئارتىلىدۇ ." دېيىلگەندەك ، ئۇ توي قىلىشنى تاللىغانىدى . گەرچە ئۇ ئەمدى 20 نىڭ قارىسىنى ئالغان بولسىمۇ ئەمما ئۇنىڭ سەرگۈزەشتىلىرىنى بەلكى خېلى ياشتىكى ئاتا - ئانىلىرىمىزمۇ بېشىدىن ئۆتكۈزۈپ باقمىغان . ئۇنىڭ تاللىشى پەقەت توي قىلىش ، چۈنكى ئۇدۇنيادا ئاتا -ئانىنىڭ ئورنىنىنىڭ جەننەتتە بۇلىشىمۇ شۇ پەرزەنتىنىڭ ھەق يولىدا تەربىيلەنگەن ياكى تەربىيىلەنمىگەنلىكىدە .
   ئۇ ئەنە شۇ خىياللار بىلەن كۈنىنى كىتاپ ئوقۇش بىلەن ئۆتكۈزەتتى ، كۈنلەرنىڭ بىرىدە ئۇ ئۈندىداردا بىرى بىلەن تۇنۇشۇپ قالدى ، قارشى تەرەپ تولىمۇ مۇلايىم ئىدى . نەتىجىدە ئۇلار تۇنۇش چىقىپ قېلىشتى ، پەقەت ئادەتتىكىدەك تەرەپپال كۈرۈشۈپ بېقىشقان ئىكەن . ئالاقە بارغانسېرى قۇيۇقلاشتى ،بارا - بارا ئۈندىداردىن تىلفۇن ئالاقىسىگە ئۆتتى ، ئاندىن كۆرۈشۈشلەر بولدى .قىز ئۇنى ئويلىسىلا قەلبىدە ئۇنىڭغا نىسبەتەن بىر خىل سېزىم پەيدا بۇلاتتى ، چۈنكى ئۇ يىگىتتىكى ھەربىر مىجەز - خۇلق قىزنىڭ ئارزۇسىدىكى ئادەم ئىدى .  بىر كۈنى ئۇلار يەنە ئۈندىداردا تۈگىمەس پاراڭنى باشلىۋەتتى . يىگىت تۇيۇقسىز بىر مۇھىم گېپى بارلىقىنى ئېيتتى _ دە ، قىزنىڭ يۈرەكلىرى رىتىمسىز سوقۇشقا باشلاپ ، تۇراقلاشقۇچە يىگىت سۆزىنى قىلىپ بولدى :
  _ ماڭا ياتلىق بۇلۇشنى خالامسىز ؟
تۇيۇقسىز بۇ گەپنى ئاڭلىغان قىز قاندق كەيپىياتتا بۇلىشىنى بىلەلمەي قالدى . ئۇلار ئۇزاققىچە پاراڭلاشتى . قىز كۆپ ئويلىنىشلار بىلەن ، ھەرقايسى تەرەپلەرنى كۆزدە تۇتقان ئاساستا يىگىتنى قۇبۇل قىلدى . چوڭلارمۇ بىلىشتى ، قىشنىڭ چىقىشى بىلەن توي قىلىشقىمۇ پۈتۈشۈپ بولغانىدى .
   بىر كۈنى كەچ قىز يىگىتنىڭ رەسىمىنى كۆرۈش ئۈچۈن ئۇنىڭ چ چ سىنى ئاچتى _ دە ، كۆزىگە ئىشەنمىگەن ھالدا بىرى بىلەن پاراڭلاشقان پاراڭ خاتىرىسىگە كۆز يۈگۈرتتى ، گەپلەردىن ئقارشى تەرەپ يىگىتنىڭ بۇرۇنقى سۆيگەن ئادىمى ئىدى، قىزنىڭ يۈرىكى ئاغرىدى ، ئازابلاندى ، ياش تۆكتى ، چۈنكى ئۇلار پات يېقىندا توي قىلماقچى ئەمەسمۇ ؟؟؟ قىز بۇ توغرىسىدا يىگىت بىلەن پاراڭلاشتى . ھەقىقىي ئازاپ مانا مۇشۇ يەردىن يەنى يىگىتنىڭ ئۇنىڭ بىلەن بولغان مۇناسىۋىتىنى چۈشەندۈرۈش بىلەن باشلاندى . ئۇ قىز يىگىتنىڭ بۇرۇنقى سۆيگىنى ئىكەن ، ئەينى چاغدا قىز باشقىنى ياخشى كۆرگەنلىكى ئۈچۈن بۇ يىگىتكە ماقۇل بولمىغان ، يىگىت ھەرقانچە قىلىپمۇ ئۇنى ماقۇل كەلتۈرەلمىگەن . ئاندىن ھەممىنى يېڭىدىن باشلىغان . ھازىر ئۇ قىز ئۆزى ياخشى كۆرگەن ئادىمىدىن ئايرىلىپتۇ ، ئەينى ۋاقىتتا بۇ يىگىتكە ماقۇل بولمىغانلىقىدىن پۇشايمان قىلىپتۇ ، مانا بۇلار بىلەن يىگىتنىڭ قەلبىدىكى ئۆچۈرۈلمەكچى بولغان ئوت قايتا تۇتۇشتى بولغاي ، ئۆزىنىڭ تويىنىڭ پۈتۈشۈپ بولغانلىقىنى ئۇنتىغان ھالدا ئۇنىڭغا قايتا كۆڭۈل ئىزھار قىلىپتۇ ئەپسۇس ، يەنىلا رەت قىلىنىپتۇ ، چۈنكى ئۇمۇ يىگىتنىڭ قىز دوستى بارلىقىدىن خەۋەردار ئىكەن . يىگىت چۈشۈندۈرگەچ شۇنى دەپ سالدىكى : مەن ئۇنى ئۇنتۇپ كېتەلمىگەن ، سىزنى چۈشەنگەندىن كېيىن سىزنى قولدىن بېرىپ قويماي دەپ ياخشى كۆرمەي تۇرۇپ كۆڭۈل ئىزھار قىلغان ، ۋاقىتنىڭ ئۆتۈشى بىلەن ياخشى كۆرۈپ كېتەلەيمەن دەپ ئويلىغان ........... ھازىر سىزگە يۈز كېلەلمەيدىغان ئىشنى قىلدىم ، ئەمما سىزنى ئالداش نىيىتىم يوق _________
   بۇچۈشەندۈرۈشلەر بىلەن ئۇنىڭ يۈرىكى زېدە بولدى ، ھەقىقەتەن يىگىت تىغ بىلەن ئۇنىڭ يۈرىكىنى زېدە قىلدى . قىز ئۇنىڭ ئۈچۈن قىلغان ئەقىدىسىگە ، ئۇنىڭغا ئۈمۈد باغلاپ ئولتۇرغانلىقىغا ، ئۇنىڭغا ئىشەنگەنلىكىگە ئېچىنىپ ياش تۆكتى ، بىر بالا دەپ ئاتا- ئانىسىنىڭ ئەتىۋارلاپ ، ئەركىلىتىشىدە چوڭ بولغان بۇ قىز قانچىلىك ئىش قىلالايدۇ دەيسىز ؟ شۇڭا ئۇ كېيىنكى تۇرمۇش ئۈچۈن ، يىگىتنىڭ ئۆيدىكىلىرى ئالدىدا يۈزىنىڭ تۆۋەن بۇلۇپ قالماسلىقى ئۈچۈن ئۇ ھەممىنى ئۈگۈنىۋاتقان ئىدى ، ئۇ شۇنىڭ ئۈچۈن ياش تۆكتى .گەرچە يىگىت يەنىلا ئۇنى تاللىغان بولسىمۇ ، ئۇ يەنىلا ئازابلىنىۋاتاتتى ، چۈنكى ھەرقانداق بىر ئادەم ھەقىقىي بەخىتكە ئېرىشىشنى خالايدۇ . قىز يىگىتنىڭ تاللىشىدىن يەنىلا گۇمان قىلىدۇ ، نېمىشقا دەمسىلەر ؟ چۈنكى ئوغۇللار دوست _ بۇرادەرلىرى ئالدىدىكى ئابرويىنى يۇقىرى قىلىش ئۈچۈن ئىنتىلىدۇ ، ئەگەر بۇ ئىشى دوستلىرى ئالدىدا ئاشكارىلانسا ، بارلىق يېقىنلىرىنىڭ ئۆزىگە بولغان كۆز قارىشى ئۆزگىرىدۇ ، شۇڭا يۈز ساقلاپ قېلىش ئۈچۈنمۇ ئۇنى تاللاش ئېھتىماللىقىنى يوق دېگىلى بولمايدۇ ، ئەلۋەتتە . يىگىت ھەقىقەتەن ياخشى ئىدى ، مەن ئويلايمەن بەلكى بۇ ئىشلار ھايات مۇساپىسىدىكى ئەگرى -توقايلىقلاردۇر ، ئاللاھنىڭ ئۇلارغا سالغان سىنىقىدۇر . ئادەم مۇكەممەل بولمايدۇ ، خاتالىقنى تۇنۇش ئەڭ توغرا تاللاش ،ئەمما تۇرمۇشتىكى بەزى خاتالىقلارنى كەچۈرۈپ پۇرسەت بېرىشكە بولىدۇ ئەمما بەزىلىرىنى ئۇنداق قىلىشقا بولمايدۇ .
   قىز يەنە ئۇنى ئويلىماقتا ، كەچۈرەيمۇ ، كەچۈرمەيمۇ دەپ ئىككىلەنمەكتە ، زادى قانداق قىلسا بولار؟


    بەلكى بەزى قېرىنداشلىرىم بۇيازمىنى بىمەنىلىك دەپ چۈشىنىپ ئۆز خالىغىنىچە ئىنكاس قايتۇرىشى مۇمكىن ، مەن بۇنى كۆپچىلىكنىڭ مۇھەببەتنىڭ ھەرخىل قىيىنچىلىقلىرىغا بولغان مۇئامىلىسىنى بىلىش بىلەن بىللە يۇقاردا سۆزلەنگەن ئۇ قىزغا ياردىمىنىڭ بۇلۇپ قىلىشىنى ھەم مۇھەببەتنىڭ ئەگرى - توقايلىقلىرىدا قىينىلىپ مېڭىۋاتقان ياش قىزلىرىمىزنىڭ ئېھتىيات قىلىشقا تېگىشىلىك ئىشلىرىدىن بىرى ئىكەنلىكىنى ۋە ئوغۇللىرىمىزنىڭمۇ مۇھەببەتكە ئەستايىدىل مۇئامىلە قىلىپ ، كۆز ئالدىنىلا كۆرمەي كەلگۈسىنى كۆپ ئويلىشى كېرەكلىكىنى ئاساس مەقسەت قىلغان ئاساستا يوللىدىم . ( ئىبرەت ئۈچۈن )

مۇھەببەت دىگەن بۇ ئۇقۇمنىڭ لوغەت مەنىسى ئاجايىپ شۇ سەۋەپ ھەرقانداق بىر ئادەم دەسلەپ ئۇنىڭغا قىزقىدۇ،كىيىن ئۇنىڭدىن زىرىكىدۇ،ئاخردا ئۆزمۇھەبىتىنى ئۈزى ۋەيران قىلىپ بولاپ ئۆزىچە قارشى تەرەپتىن رەنجىيدۇ،ياكى ئۈزىنىڭ خاتالىغنى تۇنۇيدۇ دە ئۈزىدىن رەنجىيدۇ،ئاشۇ ئازغۇنلۇقنى دەپ ھەممىنى ئۇنتۇيدۇ،ئۇنداق ۋاقىتتا نە ياراتقۇچى،قىينىلىپ بېقىپ چوڭ قىلغان ئاتا-ئانىسى،كىرەك بولسا جېنىنىمۇ ئايمايدىغان قېرىنداشلىرى،دوسلىرنىمۇ ئۇنۇتقان ھالدا بېشىچىلاپ تېگى يوق ھاڭغا سەكرەيدۇ،ئەلۋەتتە بۇنى ئۈزىمۇ بىلمەي قالىدۇ. بىلگەندە بولسا كىچىككەن بولىدۇ،شۇ نوختا ئېنىقكى ۋاقىت كەينىگە يانمايدۇ.
