                                                 داڭلىق كىشىلەرنىڭ ھەزىللىرى



                                             1  .    ئاياللار دەل بۇنىڭ ئەكسىچە


         بىسمارك گىرمانىيەنىڭ 19 - ئەسىردىكى مەشھۇر سىياسونى ، ئۇ بىر كۈنى سانكت پېتېر بۇرگدا بىر خانىم بىلەن تانسا ئويناۋىتىپ ئۇ خانىمنى :
           _ سىز پەرىشتىدەك چىرايلىق ئىكەنسىز  _ دەپ ماختاپتۇ  
           _ دىپلوماتلارنىڭ سۆزىگە ھەرگىز ئىشەنگىلىى بولمايدۇ _ دەپ جاۋاپ بېرىپتۇ خانىم ،
          _ نىمىشقا ؟
          _ بۇنىڭ سورۇغۇدەك نىمىسى بار دەيسىز ، دىپلوماتلارنىڭ < بولىدۇ > دىگىنى < مۇمكىن > دىگىنى ، <مۇمكىن > دىگنى < بولمايدۇ > دىگىنى ، ناۋادا ئۇلارنىڭ ئاغزىدىن < بولمايدۇ > دىگەن سۆز چىقسا بۇنداق كىشى دىپلومات بولالمايدۇ ،
          _ خانىم دىگىنىڭىز تامامەن توغرا ، بۇ بەلكىم كەسپىمىزنىڭ ئالاھىدىلىكى بولسا كىرەك ، لىكىن سىلەر ئاياللار دەل بۇنىڭ ئەكسىنچە
         _ قانداق دەيسىز ؟
       _ بۇنىڭمۇ سورۇغۇدەك نىمىسى بار دەيسىز ، ئاياللارنىڭ < بولمايدۇ > دىگىنى < مۇمكىن > دىگىنى ،<مۇمكىن> دىگىنى < بولىدۇ > دىگىنى ، ناۋادا ئاغزىدىن ھە دەپ بولغۇچە < بولىدۇ > دىگەن گەپ چىقىپ قالسا بۇنداق ئايالنى ھەقىقى ئايال دىگىلى بولمايدۇ ،


                                          2  .   ئىككى يۈزلىمىچىلىك

         سۆۋىت ئىتتىپاقى كومپارتيسىنىڭ باش شۇجىسى خرششېف  ب د ت نى < سۆۋىت ئىتتىپاقىنىڭ ئىشلىرىغا توسقۇنلۇق قىلدى > دەپ ، ئەيىپلەپ ئۇزاق ئۆتمەيلا  ب د ت نىڭ باش كاتىپىنى دۆلەت زىياپىتىگە ئالاھىدە تەكلىپ قىپتۇ ھەم مىھماننى ناھايىتى قىزغىن كۈتىۋاپتۇ ، بىرەيلەن ئۇنىڭدىن :
       _ ئۇلارنى شۇنچە ئەيىپلەپ تۇرۇپ يەنە نىمىشقا بۇنچە قىزغىن كۈتۈپ كىتىسىز _دەپ سورىغان ئىكەن ، خرششېف :
       _ سىز كاۋكاز رايونىدىكى خەلىقنىڭ ئادىتىنى تازا بىلمەيدىغان ئوخشايسىز ، بۇ يەردىكىلەردە : < دۈشمەن ئۆيىڭىزگە مىھماندارچىلىققا كەلسە ، راسا كۈتۈڭ ، بوسۇغىڭىزدىن چىقىپ كىتىشىگە گىلىغا پىچاق ئۇرسىڭىز بولىدۇ > دىگەن گەپ بار _ دەپ جاۋاپ بېرىپتۇ .


                                             لىكولىن ھەزىللىرى


           ئامرىكىنىڭ 16 - قىتىملىق زوڭتوڭى لىنكولىن مەكتەپتە ئوقۇۋاتقان مەزگىلىدە بىر كۈنى مۇئەللىمى ئۇنىڭدىن سوراپتۇ :
         _ سىز تەسرەك بىر سۇئالغا جاۋاپ بېرەمسىز ياكى ئاسانراق ئىككى سۇئالغىمۇ ؟
         _ تەسرەك بىر سۇئالغا جاۋاپ بېرەي  _ دەپتۇ لىنكولىن
         _ ئەمىسە دەپ بىقىڭە ، تۇخۇم قانداق پەيدا بولغان ؟
        _ توخۇ تۇققان
        _ توخۇچۇ ؟
        _ مۇئەللىم سۇئالىڭىز ئىككى بوپ كەتتى .


                            *                *                   *                 *                     *

        لىنكولىن ئادۇۋكاتلىق قىلىپ يۈرگەن كۈنلىرىدە شەھەرگە پىيادە مىڭىپتۇ ، ئۇ ئارقىسىدىن بىر پىكاپ كەلگەنلىكىنى كۆرۈپ پىكاپنى توختۇتۇپ شوپۇرغا :
          _ پەلتۇيۇمنى شەھەرگە ئالغاچ كىتەمسىز ؟ _ دەپتۇ
         _  ئالغاچ كىتەي  _ دەپتۇ شوپۇر _ لىكىن ئۇنى سىزگە قانداق تاپشۇرۇپ بېرمەن ؟
          _  خاتىرجەم بولۇڭ  _ دەپتۇ لىنكولىن كۈلۈپ تۇرۇپ _پەلتۇنىڭ ئىچىدە مەن بار .


                  *                     *                      *                     *                            *

     لىنكولىن بىر قىتىم نۇتۇق سۆزلاۋاتسا ياردەمچىسى ئۇنىڭغا بىر باغاقچە بېرىپتۇ  ، باغاقچىگە < ئەخمەق > دەپ يىزىلغان ئىكەن ، لىنكولىن بىرسىنىڭ قالايمىقانچىلىق چىقارماقچى بولىۋاتقانلىقىنى بىلىپ  كۈلۈپ تۇرۇپ :
      _ خانىملار ، ئەپەندىىلەر  ، ئىلگىرى دايىم   ئىمزاسى ئۇنتۇلۇپ قالغان باغاقچىلارنى تاپشۇرۋالاتتىم ، لىكىن ئىمزاسىلا قويۇلغان ، مەزمۇنى ئۇنتۇلۇپ قالغان مۇنداق باغاقچىنى تۇنجى قىتىم تاپشۇرۋىلىشىم ،

                              مەنبە   :  شىنجاڭ قانۇنچىلىق گىزىتى
