        خەلقئارا جەمئيەتنىڭ كۈچىمۇ ئەرافاتنى ئەزرائىلنىڭ چاڭگىلدىن قۇتۇلدۇرۇپ قالغان

ئەرافاتنىڭ مەخسۇس ئايروپىلانىغا خەلق ئاۋىئاتسىيسىنىڭ بەلگىسى چۈشۈرۈلگەن بۇلۇپ ھاۋادا ئۇرۇش قىلىش ئىقتىدارىغا ئىگە ئەمەس.ئۇ نەچچە قېتىملىق ئۇچۇش جەرياندا ئىسرائىلىيەنىڭ كۈرەشچى ئايروپىلانلىرىغا يۇلۇققان،شۇنداقلا ئىسرائىليەنىڭ كۈرەشچى ئايروپىلانلىرىنىڭ ئىز قۇغلىشىغا ئۇچىرغان بۇلسىمۇ،ئەمما ئۇ يەنىلا بىخەتەر ھالدا بارماقچى بۇلغان نىشانغا بارالىغان.
چۈنكى ئىسرائىليەنىڭ ئەرافاتقا ھاۋادا ھۇجۇم قىلالماسلىقىدا كۇڭلىدە ئىككىلىنىش ۋە دەرگۇمان بار.1973-يىلى 2-ئاينىڭ مەلۇم بىر كۈنى لىۋىيىنىڭ بىر يۇلۇچىلار ئايروپىلانى دېڭىز قۇلتۇقى رايوندىن ئۇچۇپ ئۆتۈۋېتىپ سىناي يېرىم ئارلىنىڭ ھاۋا بۇشلىقىغا كەلگەندە تۇيۇقسىز كەلگەن بىر باشقۇرۇلىدىغان بومبا ئايروپىلانغا تېگىدۇ.ئايروپىلاندىكى 100 نەچچە نەپەر يولۈچى ۋە ئايروپىلان خادىملىرىنىڭ ھەممىسى قازا قىلىدۇ.
ۋەقەدىن كېيىن،تەكشۈرۈش ئارقىلىق بۇ قېتىملىق ۋەقەنى ئىسرائىلىيە ھاۋا ئارمىيسىنىڭ كۈرەشچى ئايروپىلانلىرى كەلتۈرۈپ چىقارغانلىقى ئاشكارلىندۇ.
ئەسلىدە ‹‹ مۇسادى ›› نىڭ ئىگلىگەن ئىنتايىن مەخپى ئاخپاراتىدا ئەرافاتنىڭ مۇشۇ قىتىملىق يولۈچىلار ئايروپىلانىدا ئىكەنلىكى مۇقەررەرلەشتۈرۈلگەنىدى.
شۇ ئىشتىن كېيىن ئامېرىكا ئىسرائىليەنى ھىمايىسىگە ئېلىش پوزىتسىيسىدە بۇلغان بۇلسىمۇ،ئەمما خەلقئارا جامائەتچىلىكنىڭ كۈچلۈك ئەيىبلىشى ئاستىدا ئىسرائىليە بېشىنى كۆتۈرەلمەي قالىدۇ.
شۇنىڭدىن كېيىن ئىسرائىليە كۈرەشچى ئايروپىلانلىرى يولۈچىلار ئايروپىلاندىن ھېچقانچە پەرقى بۇلمىغان ئەرافاتنىڭ مەخسۇس ئايروپىلانىغا ھۇجۇم قىلىشقا جۇرئەت قىلالماس بۇلۇپ قالىدۇ.

                                  يۈكسەك ھوشيارلىقنى ساقلاش

ئۇزاققا سوزۇلغان ئۇرۇش داۋامىدا ئەرافات ئۆزىنى قانداق نىقابلاشنى ئۆگىنۋالغان بولۇپ،دۈشمەننى ئالداش ئۈچۈن ئۇ دائىم قېرى ئادەمنىڭ سىياقىدا ياسىنۋالىدۇ.
بىر قېتىم ئەرافات ئولتۇرغان ئاممىۋى قاتناش ئاپتۇبۇسى يېرىم يۇلدا ئىسرائىليە بىخەتەرلىك ساقلاش قىسىملىرىنىڭ تەكشۈرۈشكە يولۇقىدۇ.ئىسرائىليە ئەسكەرلىرى گۇمانلىق دەپ قارىغان ھەربىر يولۈچىنى ناھايتى تەپسىلىي تەكشۈردۇ.ئاپتۇبۇستا پادىچى سىياقىدىكى بىر قېرى ئادەم ئولتۇراتتى.ئۇ ئۇستىگە كونا بىر پەرەنجە كىيۋالغان بولۇپ،پەرەنجىنىڭ نەچچە يېرىگە ياماق سېلىنغانىدى.ئىسرائىليە ئەسكەرلىرى ئۇنىڭغا بىر سەپسېلىپ قارىۋەتكەندىن كېيىن ئۇنى ئۆتكۈزۈۋىتىدۇ.بۇ ئەسكەر بۇ كۆرۈمسىز قېرىنىڭ ئۆزلىرى قوغلاپ يۈرگەن نامى مەشھۇر ئەرافات ئىكەنلىكىنى نەدىن بىلسۇن ؟!
دۈشمەننىڭ يوشۇرۇن قەستىدىن ساقلىنىش ئۈچۈن ئەرافاتنىڭ قوغدىغۇچىلىرى ئىنتايىن مەخپى بىخەتەرلىك تەدبىرلىرىنى بەلگىلىگەنىدى.ئۇنىڭ ھەركىتى ئىنتايىن تۇراقسىز بولۇپ،كېچىسى قونىدىغان تۇرالغۇسىنى دائىم يۆتكەپ بىر جايدا مۇقىم تۇرمايتى،ئۇنىڭ قوغدىغۇچىلىرى ئەرافات ئۈچۈن ئون نەچچە دانە ئوقتىن مۇداپىئەلىندىغان ماشىنا تەييارلىغان بولۇپ ئۇنىڭ ھەر ۋاقىت ئىشلىتىشىگە تەييارلىناتتى.
ئەرافات بىرەر يەرگە زىيارەتكە بېرىشنىڭ ئالدىدا ھەرگىزمۇ يولغا چىقىش ۋاقتىنى ۋە بارىدىغان نىشاننى ئاشكارلىمايتى،ھەتتا ئۇنىڭ ئايالى سۇخامۇ بەزىدە ئۇنىڭ ھەركىتىگە ئېنىق بىر نەرسە دېيەلمەيتى.ئەرافات ‹‹ مېنىڭ بۈگۈن كېچە نەدە ئۇخلايدىغانلىقىمنى ھېچكىم بىلمەيدۇ،مەن ماشىنىغا ئولتۇرغاندىن كېيىن ئاندىن شوپۇرغا نەگە باردىغانلىقىمنى ئېيتىمەن.ئوچقۇچىمۇ بەزىدە ئايروپىلان ھاۋاغا كۆتۈرۈلگەندىن كېيىن ئاندىن نەگە باردېغانلىقىمنى بىلىدۇ ›› دەيدۇ.1982-يىلى ئىسرائىليە لىۋانغا تاجاۋۇز قىلغان ۋاقىتتا ئەرافات بىرەر ئۆيدە مۇقىم تۇرمىغان بولۇپ،ئەكىسچە ماشىنا ياكى كۆۋرۈك ئاستى ئۇنىڭ ئەڭ ياخشى تۇرالغۇسى ئىدى.1982-يىلى 8-ئاينىڭ مەلۇم بىر كۈنى تۈن نېسپىدە ئەرافات بىر بىنانىڭ 14-قەۋىتىدە يىغىن ئېچىۋاتاتتى.ئۇنىڭ ياردەمچىسى ئۇنىڭغا ئىتالىيە كونسۇلىنىڭ ئۇنىڭ بىلەن كۆرۈشمەكچى ئىكەنلىكىنى ئېيتىدۇ.ئەرافات بۇنىڭدىن تۇيۇقسىز بىر خەتەرنى ھېس قىلىدۇ-دە،دەرھاللا تۇرۇشلۇق ئورنىنى ئۆزگەرتىدۇ،بارلىق خادىملار بىنادىن چېكىنىپ چىقىدۇ.ئۇلار ئەمدىلا يولنىڭ قارشى تەرىپىدىكى بىر بىنا ئېچىگە كىرگەندە ئۆزلىرى تېخى نەچچە مىنۇت ئىلگىرلا ئايرىلغان بىنا تۇيۇقسىز دەھشەتلىك پارتىلاپ كېتىدۇ.
بۇ قېتىملىق پارتىلاشتا 250 ئادەم قازا قىلىدۇ.ئەمما ئەرافات ۋە ئۇنىڭ كەسىپداشلىرى ھېچقانداق زىيان-زەخمەتكە ئۇچىرمايدۇ.
ئەرافاتنىڭ خەتەرلىك ھايات مۇساپىسىدە،ئىسرائىليەنىڭ ئىنچىكىلىك بىلەن پىلانلانغان بىر قېتىملىق بومباردىماندىن ئامان قېلىشى جۇڭگونىڭ ‹‹ زېرەك ئادەم ئالدىن كۆرەر بولۇر ›› دېگەن ھېكىمەتلىك سۆزنى ئىسپاتلاپ بەردى.
مەن ئۇلارغا ھېسدشلىق قىلمايمەن،بەلكىم ئۇلارنىڭ ھاياتىغا ئېچىنمەن !

بەخىسىزلىك ئىچىدىكى بەخىت:ئەرافات يەنە بىر قېتىم خەتەردىن قۇتۇلۇپ قالدى.

ئەرافاتنىڭ بۇ قېتىملىق خەتەردىن قۇتۇلۇپ قېلىشى پۈتۈنلەي تاسادپىيلىق.چۈنكى،ئالدىنقى كۈنى ئاخشىمى ئەرافات تۇنىسقا كەلگەندىن كېيىن ئەسلى پىلان بۇيىچە پەلەستىن ئازادلىق تەشكىلاتنىڭ باش شىتابىدىكى تۇرالغۇسىغا قايىتماقچى ئىدى.ئەمما ئەرافات يېرىم يولدا تۇيۇقسىزلا ئالدى بىلەن پەلەستىننىڭ تۇنۇستىكى ئىش بېجىرىش ئورنىغا بېرىشنى قارار قىلدى.شۇ كۈنى ئاخشىمى ئەرافات بۇ ئىش بېجىرىش ئورنىدا تۈن نىسپىگىچە خىزمەت قىلىدۇ،ئەمما ئۇ كېچىلەپ بولسىمۇ پەلەستىن ئازادلىق تەشكىلاتىنىڭ باش شىتابىغا قايتىپ ئارام ئېلىشنى ئۇيلايتى،ماشىنا ئەمدىلا قۇزغىلاي دەپ تۇرۇشىغا ئۇنىڭ ياردەمچىسى دەرھال بىر تەرەپ قىلمىسا بولمايدىغان بىر تۈركۈم مۇھىم ھۆججەتنىڭ بارلىقىنى ئېيتتى.ئەرافات ماشىنىدىن چۈشۈپ ئىش بىجىرىش ئورنىدىكى ئىشخانىغا كېرىپ خېلى بىر ۋاقىتقىچە خىزمەت قىلغاندىن كېيىن ئىش بېجىرىش ئورنىدا قونۇپ قالدى.
2-كۈنى ئەتىگەندە ئەرافات پەلەستىن ئازادلىق تەشكىلاتىنىڭ باش شىتابىغا كەلگەندە،بۇ يەر ئاللىبۇرۇن خارابىلىككە ئايلانغانىدى.خىزمەت بىناسى ۋە ئۇنىڭ تۇرالغۇسىنىڭ ئىزى كۆرۈنميتى.خىش-كېسەك ئاستىدىن جەسەتلەر كولاپ ئېلىنۋاتاتتى.ئەگەر ھېلىقى مۇھىم ھۆججەت بۇلمىغان بۇلسا ئەرافات بەلكىم ھايات قالمىغان بۇلاتتى.
