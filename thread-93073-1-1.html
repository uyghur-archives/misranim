تۈنۈگۈن بىر دوستۇم بىلەن بازار ئايلىنىپ قالدىم ، ئاندىن تاماق يىگەچ   بىر دەم پاراڭلىشىپ قالدۇق ، ئۇ ماڭا ئالدىنقى كۈنى نىمە ئىشلارنى قىلغانلىقىنى سۈزلەپ بەردى ، گەپلىرىنى ئاڭلاپ ئۇنىڭغا تولىمۇ ئاچچقىم كەلدى ، ئاندىن ئۇنىڭغا بىر مۇنچە تەربىيە قىلدىم ،
ئالدىنقى كۈنى دوستۇم ئىشغانىسىدا ئىشلەۋاتسا تونىشىدىغان بىر قىزدىن تۇيۇقسىز تېلىغون كەلگەنلىكىنى ھەم ئۇنىڭ بىلەن كۈرىشىدغان ئىشى بارلىقىنى ئېيتىپتۇ ، دوستۇم ئۇنىڭ گىپىنى ئاڭلاپ بەكلا خوشال بوپتۇ ، چۈنكى ئۇقىز رەڭلىىك سورۇن ئوينايدىغان قىز بولغاچقا دوستۇم بۈگۈن كېچە تازا پەيزى قىلىدىغان بولدۇم دەپ ئىشىنى تاشلاپلا قېچىپتۇ ، ئاندىن ئىككىسى بىللە تاماق يەپ بولۇپ ئۇدۇللا مەلۇم  مېمانغانىغا قاراپ مېڭىپتۇ ،
مېھمانغانىنىڭ ئالدىغا كەلگەندە دوستۇمنىڭ ئايالىنىڭ يېقىن بىر تۇققىنى ئۇلارنى كۈرۈپ قاپتۇ ، ھەم دوستۇم بىلەن سالاملىشىپ يولىغا مېڭىپتۇ ، دوستۇم ئەمدى چاتاق چىقىدىغان بولدى دەپ ئەنسىرەپ تۇرىشىغا ئايالىدىىن تېلىغون كېلىپ قاپتۇ ، ھەم ئۇنىڭدىن نەدە ئىكەنلىكى ، نىمە ئىش قىلۋاتقانلىقنى سوراپ دەرھال ئۈيگە قايتىپ كېلىش كىرەكلىكىنى ئىيتىپتۇ ، دوستۇم ئۇ قىزغا باشقا كۈنى كۈرىشىشكە كېلىشىپ يولغا سېلىپ قويۇپ ئۈيىگە قايتىپتۇ ،
ئۈيىگە قايتىپ كەلگەندىن كىيىن بىر مەيدان قاتتىق جەڭ باشلىنىپ كېتىپتۇ ، ئايالى بولسا ئاپامنىككىگە كېتىمەن دەپ يىغلاپ قاغشاپ كېتىپتۇ ، دوستۇم بولسا كەچ بولغاندا ئۈيىگە كېتىپ قالسا چوڭلارنىڭ گۇمان قلىشىدىن ئەنسىرەپ بىر مۇنچە يالۋۇرۇپ ئايالىنى تۇتۇپ قاپتۇ ، ھەم خاتالىقىنى تونۇپتۇ  ، بۇندىن كىيىن بۇنداق ئىشلارنىڭ قايتا سادىر بولمايدىغانلقى توغۇرسىدا ۋەدە بېرىپتۇ ،
ئۇ كۈنى دوستۇم ئىدارىسىدە نۈۋەتچىلىك قىلاتتىكەندۇق ، ئايالى سەن نۈۋەتچىلىك قىلمەن دەپ ئالداپ ئۈيگە كەلمەي سىرتتا باشقا خوتۇنلار بىلەن ئويناپ يۈرۈپسەن دەپ تازا جىدەل قىلىپتۇ ، جىدەل بېسىلىشىغا باشلىقىدىن تېلىفون كەپتۇ ، ئاندىن ئىدارىسىغا قايتىپ كەپتۇ ، باشلىقدىنمۇ بىر مۇنچە تىل ئاڭلاپتۇ ،
ئۇنىڭ گەپلىرىنى ئاڭلاپ بولۇپ ئۇنىڭغا سىرتتا قالايماقان باشقىلار بىلەن غەيرى مۇناسىۋەتتە بولماسلىقنى ، ئايالىغا ، بالىسغا كۈڭۈل بۈلۈپ چىرايلىق ئۈتىشى توغۇرلۇق تەربىيە قىلدىم ،
دوستۇمنىڭ دىيشىچە ھېلىقى قىز دوستۇم بىلەن ياش پەرقى چوڭ بولمىغان توي قىلمىغان قىز ئىكەن ،
نىمىشقا بەزى بىر كىشىلىرىمىز ئىللىق ئائىلە قۇرۇپ بولغاندىن كىيىن  ئايالىغا كۈڭۈل بۈلمەي سىرتتا بولمىغۇر ئىشلارنى قىلىپ يۈرىدىغاندۇ ؟ ئايالى ، بالىلىرى ئۇلارنىڭ بەختى ئەمەسمۇ ؟ ئائىلىدە  ھۈرمەتكە سازاۋەر ياخشى ئەر بولۇپ ئۈتسە ئۇنىدىنمۇ ياخشى ئىش  بولامدۇ ؟ بىر قىسىم قىزلامۇ ئۈزىنىڭ ئىززەت ھۈرمىتىنى يەرگە ئۇرۇپ باشقلار بىلەن  نورمال بولمىغان  مۇناسىۋەتتە بولۇپ يۈرىشى ئادەمنى تولىمۇ ئويغا سالىدۇ  ، باشقىلارنىڭ ئائىلىسىنى بۇزۇش ئۇلار ئۈچۈن شۇنداق ياخشى ئىشمۇ ؟ ئۇنداق قىزلار ئۈزىنڭ كەلگۈسى بەختىنى ئويلىمامدىغاندۇ ؟
دوستۇمنىڭ بايانلىرىدەك ئىشلار جەمىيتىمىزدە خېلىلا بار ، ھەممىمىز ئىللىق ئائىلمىزنى ياخشى باشقۇرۇپ بالىلىرىمىزنىڭ بەختى ئۈچۈن ناتوغرا ئىشلاردىن يىراق تۇرايلى .
