ئۆلۈشكۈن دوختۇرخاندا ئىشلەيدىغان بۇرادىرىم بىلەن ئولتۇرۇشقا بارىدىغان بولۇپ ئۇنىڭ ئىشتىن چۈشۈشىگە ئۈلگۈرۈپ ئۇنى ئالغىلى دوختۇرخانغا باردىم . مەن ئۇ ئىشلەيدىغان بۆلۈمگە كىرسەم كەيپىيات بىر ئاز جىددى ئىكەن.شۇنداقلا بىر قانچە كىشى بىمارنىڭ ئائىلە تەۋەلىكى بولسا كىرەك يىغلاپ ھازا ئىچىۋاتاتتى ،شۇئەسنادا ئىچىدىكى بۆلۈمدىن بىر سىستىرا چىقىپ بىمارنىڭ يىقىن تۇققىنى قايسىڭلا؟ دەپ سورىدى بۇ سۇئالنىڭ سورىلشى بىلەن يىغا زارا ئەۋجىگە چىقىشقا باشلىدى.سىستىرا دەرھال سىلەر خۇشال بولۇشۇڭلار كىرەك تىنقى توختاپ بولغان بىمار ھۇشىغا كەلدى بىۋاستە يىقىن تۇققانلاردىن بىرسى كىرىپ ھەمرا بولساڭلا بولىدۇ دىدى. بۇ گەپنى ئاڭلىغان مەنمۇ ھەم ھەيران بولۇپ دوستۇم ئىشتىن چۈشۈپ چىققاندىن كىيىن بۇ قانداق ئىش ئىكەنلىكىنى سورىدىم .دوستۇم ئۇ بىمارنىڭ ئائىلسىدىكىلەرنىڭ ئۇ بىمارنى دوختۇرخانغا ئىلىپ كەلگەن ۋاقىتتا تىنقى توختاپ بولغانلىقىنى بىراق دوختۇرلارنىڭ يۈرەك قوزغاتقۇچ دەزمال ئىشلەتكەندىن كىيىن ھۇشىغا كەلگەنلىكىنى بۇنداق ئىشنىڭ يىلدا بىرنەچچە قىتىم دوختۇرخانلاردا ئاز ساندا يۈز بىرىپ تۇرىدىغانلىقىنى ئىيتىپ بەردى .دوستۇمنىڭ بۇ سۆزى بىلەن بۇنىڭدىن 10 يىل بۇرۇنقى يەكەندىن قەشقەرگە ئاپتۇبۇستا كىلىۋاتقۇچە بىرەيلەننىڭ ئۆزى بىلەن بىللە يان ئورۇندۇقتا ئولتۇرغان سەپەرداشقا سۆزلەپ بەرگەن مۇنۇ ھىكايىسى قۇلىقىمغا قايتا ئاڭلانغاندەك بولدى،ئۇ سەپەرداشنىڭ خوشنىسى چۈش پىشنلەر بىلەن تۈگەپ كىتىپتۇ ئۆيىدىكىلەر دەرھال تۇتۇش قىلىپ بۇ ئادەمنى ناماز ئەسىرگە ئۈلگۈرتۈپ نامىزىنى چۈشۈرۈپ تىزلىكتە يەرلىكتە قويىۋىتىپتۇ،ئەتىسى ئۆيىدىكىلەر تۇپراق بىشغا چىقىپ خەتمە قۇرئان ئوقۇۋاتسا باللىرىم مىنى چىقىرىۋىلىڭلار دىگەن ئاۋاز ئاڭلىنىپتۇ بۇلار دەماللىققا قورقۇپ كەتكەن بولسىمۇ دەرھال ئىسىنى يىغىپ ھۇشىغا كەلگەن دادىسىنى گۆردىن چىقىرۋاپتۇ.شۇ چاغدا ئاپتۇبۇستا ئولتۇرغان ھەممەيلەن بۇ سەپەرداشتىن بۇ ئىشقا ھەيران بولۇپ يىنش يىنشلاپ بۇ ئىشنىڭ جەريانىنى سورىۋالغان ئىدۇق.يىقىندىن بۇيانمۇ تورلاردا، چەتئەللەردىكى ئىچكىردىكى مۇشۇنىڭغا ئوخشاش ئەھۋاللار پات پات  يىزىلىپ تۇرۇۋاتىدۇ مەن ئويلاپ قالدىم ئىسلام دىنىمىزدا ئادەم ئۆلگەندىن كىيىن تىز يەرلىكتە قويۇش تەشەببۇس قىلىندۇ .بىراق ھىلقى بىمار دوختۇرخانغا ئىلىپ كىلىنمەي تىز يەرلىكتە قويىۋىتىلگەن بولسا  قانداق بولارئىدى ھە............؟ بۇ ئىش ئىسلام دىنمىزنىڭ بۇ خىل تەشەببۇسى بىلەن زىت كىلەمدۇ قانداق؟ مىيىتنى چوقۇم تىز يەرلىكتە قويۇۋىتىش كىرەكمۇ ؟ ياكى ئەڭ ئاخىرقى قۇتقۇزۇش تەدبىرلىرنى قىلىپ بىقىش كىرەكمۇ؟ تورداشلار مۇنازىرە قىلىپ باقساق؟ بۇ ئىنتايىن سەل قاراشقا بولمايدىغان بىر ئىش ئىكەن ئۆلۈم ھەممىمىزنىڭ بىشىدا بار ئىش شۇڭا باشقۇرغۇچىلارنىڭ بۇ تىمىغا ئەستايىدىل قاراپ تەستىقلاپ قويۇشىنى تەۋسىيە قىلىمەن.

ئادەم جان ئۈزۈش ۋاختىدا قول - پۇت تىرناقلىرى ساغىرىدۇ، چىرايى ساغىرىدۇ ، ئالقان نىڭ دۈمبىسى تەرەپتىكى قان تومۇرلار چاپلىشىپ كېتىدۇ  يەنى قان تۇمۇرنى كۆرگىلى بولمايدۇ .
ئادەم بەدىنىنىڭ تۆۋەنكى قىسمى پۇت بارماقلىرىدىن باشلاپ مۇزلاشقا باشلايدۇ . تەدرىجى پۇت ، تىز ، يوتا، بىلەك ، بەل قىسمى  مۇزلاشقا باشلايدۇ . بۇ ۋاقىتتا جان بىرىش ئازابىنى تارتىدۇ ، جان بىرىش بەك قىيىن بولۇدۇ  . ئاخىردا نەپەستىن قالىدۇ .
نەپەستىن قالغان - قالمىغانلىقىنى بىلىشتە بويۇن قان تومۇرىنى تۇتۇش ، بۇرنىغا قولىنى تەككۈزۈش ئارقىلىق بىلگىلى بولۇدۇ . جان چىقىپ بولغاندا پۈتۈن بەدەن ناھايىتى سوغۇق بولۇدۇ  .
مەن دادام ، قېينى ئاتام ، قېينى ئانامنىڭ نەپەس ئۈزگەن ۋاقىتلىرىدا يېنىدا بار ئىدىم  ، بۇنداق ئەھۋاللارنى كۆرۈپ تۇردۇم .
نەپەس ئۈزگەندىن كېيىن يەرلىكتە قويۇشقا تەيرارلىق قىلىمىز  ، قانچا تېز بولسا شۇنچە ياخشى ، بۇ ئىسلام دىنىنىڭ ئىلغار تەشەببۇسىنىڭ بىرىدۇر  .
شۇڭلاشقا ھەر بىر ئىشىمىزدا دىنىمىزغا ھۆرمەت قىلىشىمىز كېرەك .

تولۇقلىما مەزمۇن (2012-12-8 11:32 PM):
بىھۇش ئادەمنى بارلىق تەدبىرنى قوللۇنۇپ قۇتۇلدۇرۇش كېرەك .
قۇرئان ۋە ھەدىسلەردە ئىسپاتلاندىكى : بىر ئادەم ئۆلسە ، ئۇ بۇ دۇنيادا قىلغان ياخشى بولسۇن يامان بولسۇن ، بارلىق چوڭ – كىچىك ئىشلىرى ئۈچۈن ھىساب – كىتاب قىلنىدۇ   . ياخشى ئىش مۇكاپاتلىنىدۇ ، ئەسكى ئىش جازالىنىدۇ ، ھىساۋاتنىڭ بىرىنچى سەھنىسى قەبرىدە بولۇدۇ  .

مىيىتنى يۇيۇپ - تاراپ، نامىزىنى چۈشۈرۈپ، يېقىن تۇققانلار ۋىدالىشىپ، يىغا - زار قىلىشىپ، جىنازىغا سېلىپ يەرلىككە ئېلىپ بارغىچە بولغان ئارلىقتا ھەر قانداق <يالغان ئۆلۈك> مۇ ياتقان يېرىدىن دەسىت تۇرۇپ كېتەلىشى مۆمكىن. شۇڭا بۇ جەھەتتە ئۇنداق غەم قىلىپ كېتىش ئارتۇقچە دەپ قارايمەن.

كۆپچىلىك يەنە شۇنىڭغىمۇ دىققەت قىلىشى كىرەككى، ھازىر <ئۆلۈم بوپتۇ> دىگەن خەۋەرنى ئاڭلاپ، پالاقشىپ بارغانلار (ئۇرۇق - تۇققانلار ۋە بىرقىسىم دوسىت - بۇرادەرلەردىن باشقا) ئۆلۈمنىڭ ھۆرمىتىنى ئازراقمۇ قىلمايدىغان بولۇپ كېتىشتى، قازا ئىگىسىنىڭ تۇققانلىرى بىلەن تىنىچلىق سورىشىپ بولغاندىن كىيىنلا، سەل نېرىراق يەرگە بېرىشىپ، يىگەن - ئىچكەننى پەش قىلىشىپ چىرايىنى كالچايتىشىپ كۈلۈشىۋاتقان، چاقچاقلىشىۋاتقان، مۇئاشنىڭ قانچىلىك ئۆسكەنلىكىنى، باشلىقىنىڭ ئۆزىنى قانچىلىك ئەتىۋارلايدىغانلىقىنى ھەتتا دىياۋيۈداۋ مەسىلىسى، ئوباما...قاتارلىقلار ئۈستىدە گەپ سېتىشىۋاتقانلىقىنى كۆرەلەيمىز، شۇنىمۇ كېسىپ ئېيتالايمەنكى، ئەشۇلارنىڭ يېرىمدىن كۆپرەكى ھەتتا قوڭ- پوڭنىمۇ يۇماي تەرەتسىز كىلىشىدۇ. بۇ قىلىقلارنىڭ ھەممىسى مىيىتقا بولغان ھۆرمەتسىزلىك، قازا قىلغۇچىنىڭ ئائىلىسىدىكىلەرگە قىلىنغان ھاقارەت، ئەڭ مۇھىمى چوڭ گۇناھ ھېساپلىنىدۇ، بۇ قىلىق بولۇپمۇ زىيالىلار ئارىسىدا ئىنتايىن ئېغىر!  مۇشۇ سەۋەپلەر تۈپەيلىدىن مىيىت گەرچە يۇيۇلۇپ بولغان بولسىمۇ ئەشۇ تاھەرەتسىز كىشىلەر سەۋەپلىك <مەينەت> پېتى دەپىنە قىلىنىپ كېتىۋاتىدۇ!!!
