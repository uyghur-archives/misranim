  ئەسسالامۇ ئەلەيكۇم مىسرانىم مۇنبىرىدىكى مېھنەتسىز تەر تۆكۈۋاتقان جاپاكەش مۆھتىرەم باشقۇرغۇچىلار ! ئالدى بىلەن سىلەرگە ئوتلۇق سالام يوللايمەن . قىلىۋاتقان ئىشىڭلارغا ئۇتۇق تېنىڭلارغا سالامەتلىك مەڭگۈ يار بولسۇن !  

  مۇمكىن بولسا مىسرانىم مۇنبىرى ھازىرلاپ بەرگەن مۇكاپاتلىق ئەسەرلەر سەھىپىسىگە ئاتاپ يوللۇدۇم . بىزلەرگە بۇ پۇرسەتنى يارىتىپ بەرگىنىڭلارغا كۆپ رەھمەت              
  
              بىزنىڭ پەرزەنتلىرىمىز

   بۇ تىمىنى يېزىۋېتىپ تۇرۇپ-تۇرۇپ ئويلاپ قالدىم . بۇ ھەقتە ئۆزۈمگە كۆپلىگەن سۇئاللارنى قويدۇم . كىشىلەرگە ھېچ پايدىسى زۆرۈريىتى بولمىغان بىر نېمىلەرنى يېزىۋاتىمەنمۇ ياكى خەلقىمگە يارىغۇدەك تېتىملىق ئەمەلىي تۇرمۇشنىڭ بىرەر بۆلۈكىنى يۇرۇتۇپ بېرىۋاتىمەنمۇ؟! مۇنبەرنى ئاۋات قىلىش ۋەياكى ئەسىرىمنىڭ نادىر تىمىلار بۇلۇپ چىقىپ مۇكاپاتلىق ئەسەرلەر سەھنىسىدىن ئۇرۇن ئالسىكەن دېگەن مەقسەتتە يېزىۋاتىمەنمۇ  ياكى بولمىسا مۇشۇ ئاددى بايانلىرىم ئارقىلىق ئەمەلىي تۇرمۇشتىكى بەزىبىر ئۆزگەرتىشكە تېگىشلىك نۇقسانلارنى تۈزۈتۈپ خەلقىمگە مېھنەتسىز سۇنىۋاتىمەنمۇ؟!  مەندىن بۇرۇن باشقىلار ئېيتىپ بولغان قۇرۇپ پەتىۋالارنى بىلجىرلاۋاتىمەنمۇ ياكى باشقىلار بايقىغان ئەمما ئېيتىشقا قۇدرىتى يەتمىگەن بۇ ئەھۋاللارنى بايان قىلىۋاتىمەنمۇ؟!......
ئەنە شۇنداق چىگىش خىياللار ئىلكىدە قەلىمىم قەغەز ئۈستىدە بىردە توختاپ ئاللىقانداق خىياللارغا پاتسا ، بىردە ئىتتىك-ئىتتىك قەدەملىرى بىلەن يىنىك تەۋرىمەكتە .  ئۇشبۇ بايانلارنى يېزىۋېتىپ بەزىدە يۈرەك سۆزلۈرىنىڭ كەينىگە كۆپ چىكىتنى قۇيۇپ قۇيۇپ خېلى بىر ۋاقىتلارغىچە ھاياجان ئىلكىدە تۇرۇپ كەتسە ، يەنە بەزى ۋەقەلىكنى سۈرەتلەۋېتىپ قورسىقىغا قاچىلىۋالغان قاپقارا سىياھ يېتىشمەي قۇرلار سۇس چىقىپ قالاتتى .
   بىز ئۇيغۇرلاردا ئالماس ماتا زورلاپ ساتا دېگەن گەپ بار . بازار ئىگىلىكى تەرەققىي قىلغان بۈگۈنكى كۈندە كوچا-كويلارغا قايناق بازارلاردىن تارتىپ بۇلۇڭ-پۇچقاقلارغىچە قارايدىغان بولساق ، بىر قىسىم قالاق سودىگەرلەرنى بايقايمىز . يەنە كىلىپ كۆپۈنچىسى قۇرامىغا يەتمىگەن كىچىك بالىلار . بۇ كىچىك بالا سودىگەرلەرنى زادى نېمە دېسەك بۇلار؟! كۆپچىلىككە تېخىمۇ چۈشۈنىشلىك قىلىپ ئېيتقاندا مەن  باغچىلاردا كوچا-كويلاردا ئاۋات بازار خالىي  ئورۇنلاردا ئادەمنى بىزار قىلىپ زورلاپ ساتارمەن يۈزى قېلىن كىچىك بالىلاردىن بەكمۇ بىزارمەن . ئۇلارنىڭ ساتقان نېمىسىنىڭ تايىنى يوق ، يەنە كىلىپ بازارغا چىققاندا ئۇچرىغانلىكى ئادەملەرگە ئاڭلىماققا چىرايلىق ئاڭلىنىدىغان گەپلىرى بىلەن ئادەمگە پوقتەك چاپلىشىپ بازار تەرتىپىنى ئېغىر دەرىجىدە بۇزغۇنچىلىققا ئۇچرىتىش بىلەن بىرگە ، كىشىلەرنىڭ تۇرمۇشىغىمۇ كۆپلىگەن يامان تەسىرلەرنى ئېلىپ كەلدى .
  ھەممىمىزگە مەلۇملۇق ( بەلكىم بىلمەيدىغان كۆرۈپ باقمىغانلار كۆپ ئەمەسقۇ دەيمەن ) بەزىدە باغچىلارنى ئايلىنىپ سەيلە قىلىۋاتقىنىڭىزدا يەردىن ئۈنگەندەكلا بىر قانچە كىچىك بالىلار پەيدا بولىدۇ . ئۇلار ئالدىڭىزغا كېلىپ قولىدىكى سېغىز   تازىلىق قەغىزى دېگەندەك نەرسىلەرنى كۆرسىتىپ سېتىۋېلىشىڭىزنى ئۆتۈنىدۇ . ئالمايمەن دېسىڭىز بىر ئاڭلىماققا چىرايلىق ئاڭلىنىدىغان گەپلەرنى قىلىپ سىزگە زورلايدۇ، بەزىلىرى ھەتتا يالغان گەپلەرنى تۇقۇپ ئىچىڭىزنى ئاغرىتىش بەدىلىگە قولىدىكى نەرسىلىرىنى سېتىۋېلىشىڭىزنى ئۆتۈنۈپ يېنىڭىزدىن كەتمەي قېلىنلىق قىلىپ كۆزىڭىزگە قاراپ بىزىرىپ تۇرىشىدۇ . ئالمايمەن دېدىڭىزمۇ قۇتۇلمىقىڭىز تەس ، قۇتۇلدۇم دېگەن بىلەنمۇ  كېتىۋاتقىنىڭىزدا ئۇ كىچىك ساتارمەنلەرنىڭ يۈرۈكىڭىزنى ئۆرتەيدىغان سېسىق بىزەڭ گەپلىرى قۇلىقىڭىزغا ئاڭلىنىپ تۇرىدۇ .  ساتقان سېغىزلىرىنىڭ بەزىلىرى سۈپىتى ئۆزگەرگەن ناچار مەھسۇلات ئىكەنلىكىنى دېمەيلا قۇياي ، ئۇلار ھەتتا چىشى يوق بوۋاي-مومايلارغىمۇ بىچارە قىياپەتتە كۆرۈنۈپ يېلىنىپ بىرەر تال سېغىز  سېتىۋېلىشىنى ئۆتىنىدۇ . ئۇلار ئالمايمەن دېسە تاس قالىدۇ زورلاپ يۈرۈپ پېشىگە ئېسىلىپ يانچۇقىدىن پۇلىنى ئۆزى ئېلىۋالغىلى . بۇنىڭ كۈپ-كۈندۈزدە بولىۋاتقان بۇلاڭچىلىقتىن نېمە پەرقى بولسۇن . ھەتتا ياقا يۇرتلۇق يۇلۇچى ساياھەتچىلەرنى ئارام خۇدا بىرەر جايدا تۇرغىلىمۇ قويمايدۇ . بىرىنىڭ سېغىزىنى ئەپ قويدۇڭمۇ يەنە باشقىلارمۇ بىر-بىرلەپ ئۆزىنىڭكىدىنمۇ بىرەر تال ئېلىپ قۇيىشىنى تەلەپ قىلىپ تۇرىۋالىدۇ . چۈنكى ئۇلار ئاممىۋى سورۇنلاردا ھەممە ئادەملەرنىڭ ئۆز ئابرويىغا كۆپ دىققەت قىلىدىغىنىنى ، شۇ سەۋەپ سىزنىڭ باشقىلار تەرىپىدىن بىرەر تال سېغىز ئالالمايدىغان ھالغا چۈشۈپ قاپتۇ دەپ مەسخىرە قىلىشىدىن ئەنسىرەپمۇ بولسا بىرەر تال ئەپ قويىدىغىنىڭىزنى ياخشى بىلىدۇ .
   مەن مىسال ئۈچۈن ئەمەلىي بېشىمدىن ئۆتكەن بىر قانچە ئىشلارنى سۆزلەپ بېرەي : ئەڭ تەرەققىي قىلغان مەركىزى شەھەر ئۈرۈمچىنىڭ چوڭ كۆۋرۈك سودا بازىرى ئەتراپىدىكى خالتا كوچىلاردا بىر قانچە كىچىك بالىلارنىڭ قولىغا تازىلىق قەغىزى سېغىز دېگەندەك نەرسىلەرنى كۆتۈرۈۋېلىپ ئادەملەر ئارىسىدا ئۇياق-بۇياقلارغا چېپىپ يۈرگەنلىكىنى بەلكىم سىلەرمۇ كۆرگەن بۇلىشىڭلار مۇمكىن . بەزىدە ئۇ ئاتالمىش كىچىك زومىگەرلەرنىڭ ئادەمنى قانچىلىك بىزار قىلىدىغىنى سىلەرگىمۇ مەلۇم . بىچارە قىياپەتكە كىرىۋېلىپ زورلاپ ساتقان نەرسىلىرىنى ئالماي كەتكىنىڭىزدە ئۇلارنىڭ مۇشتۇمدەك كىچىك تۇرۇپ قىلغان سېسىق گېپىدىن قارغاشلىرىدىن ھوشىڭىزدىن كەتكىلى تاسلا قالىسىز .  بىرەر تال ئەپ قويمىساڭ دىۋانە بۇلۇپ كېتەرسەن دېيىشلىرىچۇ  تېخى .
   يازلىق تەتىلنىڭ ئالدىنقى ئايلىرىدا ئازراق ئىشلار بىلەن قەشقەرگە ساياھەتكە بېرىپ قالدىم . مەدەنىيەت يۇرتى بولغان ئەزىزانە قەشقەرنىڭ قەدىمىيلىكى ۋە ھەرقايسى جەھەتتىكى مىللىي پۇراققا ئىگە گۈزەللىكى ماڭا بەكلا چوڭقۇر تەسىر قىلدى . ( تىمىدىن چەتنەپ كەتمەسلىك ئۈچۈن ئەزىزانە قەشقەرنىڭ گۈزەللىكى ھەقتە پەقەت مۇشىنچىلىك سۆزلەي )  بىر كۈنى شۇ يەرلىك بىر قانچە دوستلۇرۇم بىلەن تۈمەن دەرياسى سوزۇلۇپ ياتقان قەشقەر شەھرىنىڭ ئاۋات بازارلىرىنى، قەدىمىي مىللىي خاسلىققا ئىگە ئۆي ئىمارەتلەر بىلەن توشۇپ كەتكەن خالتا كوچىلىرىنى ، خەلق مەيدانى خەلق باغچىسىنى ئايلاندۇق .  باغچا ئىچىدىكى بىر ئورۇندۇقتا دوستۇم بىلەن سوغۇق ئىچىملىك ئېچكەچ پاراڭلىشىپ ئولتۇراتتۇق ، بىردەمدىلا ئەتراپىمىزغا سەككىز-ئون ياشلار چاممىسىدىكى كىچىك بالىلار يىغىلدى . ئۇلارنىڭ قوللۇرىدا بىر قانچىدىن سېغىز تۇراتتى . مەن ئىلگىرىمۇ باشقا شەھەرلەردە بۇ خىل ئەھۋالنى ئۇچراتقىنىم ئۈچۈن بىر قاراپلا ئۇلارنىڭ مەقسىتىنىڭ نېمە ئىكەنلىكىنى بىلىپ بولدۇم ، شۇنداقتىمۇ يۇۋاشلىقچە ئۇلارنىڭ سېغىزلىرىنى گەپ سۆزسىزلا سېتىۋېلىشنى خالىمىدىم . چۈنكى مەن ئەزەلدىن سېغىز چاينىمايتتىم دوستۇممۇ ماڭا ئوخشاشلا ئىدى . ئۇلار ئالدىمىزغا كېلىپ قولىدىكى سېغىزنى ئۇزاتقىنىچە
___ ئەپ قويىڭە ، دېگىنىچە قۇچىقىمغىلا تۆكىۋەتكىلى تاس قالدى . ئاچچىقىم كەلسىمۇ ئۇ بېشىدا چىرايلىق ئەپ قويىڭە دېگەنلىكى ئۈچۈن ئۇنىڭغا جاۋابەن :
  __ خاپا بولما ئۇكام ، مەن ئەزەلدىنى سېغىز چايناشنى ياقتۇرمايمەن . باشقا ياقلارغا بېرىپ قاراپ باقساڭ بوپتىكەن . دېدىم .ئۇ خۇددى سۆزۈمنى ئاڭلىمىغاندەك ھېچ بىر كېتىدىغاندەك ئەمەس . ئالۋالقىدىنمۇ بەكرەك زورلاپ تۇرىۋالدى ، ئۇنىڭغىچە ياندا قاراپ تۇرغان ئىككى-ئۈچ شېرىكىمۇ تەڭلا ئالدىمىزغا كېلىپ  ئەپ قويۇڭ دېگىنىچە قاراپ تۇرۇۋېلىشتى . ئاچچىقىم تازا بىر كەلدىيۇ ، لېكىن يەنە چىرايلىقچە رەت قىلدىم . سۆزۈمنىڭ ئۇرامىدىن ئۇلار مېنىڭ ياقا يۇرتلۇق ئىكەنلىكىمنى بىلىۋالدىمۇ قانداق بىردىنلا يۇپۇرۇلۇپ كېلىپ قولىدىكى سېغىزنى ماڭا تەڭلىگىنىچە تاسلا قالدى ئۇرۇندۇقتىن ئىتتىرىپ چۈشۈرىۋەتكىلى . ئاچچىقىمدا ئورنۇمدىن ئىتتىك تۇرۇپ كەتتىم . تىللاي دېسەم يېنىمدا دوستۇم سەت تۇراتتى . كھچىتىغا بىر تەستەك سالاي دەپمۇ ئويلىدىم ، لېكىن چوڭ بۇلۇپ قالغان ئادەم بىر كىچىك بالىنى ئۇرسا ياخشى بولمايدىغانلىقى ماڭا ئايان ئىدى . يەنە كىلىپ ئۆلۈپ قالسىمۇ تاپىسىنى پۇل قىلىپ زورلاپ ساتىدىغان بۇ كۆكەمىلەرنىڭ جېدىلىگە ئارىلىشىپ قېلىشنى خالىمايتتىم .
ھەي، كونىلار يەكچەشمىنىڭ يۇرتىغا بارساڭ بىر كۆزۈڭنى قىس دەپتىكەن . بىرەر تال ئەپ قويمىسام ئۇلارنىڭ زادى ئارام بەرمەيدىغانلىقىغا كۆزۈم يېتىپ ئامالسىز ئۇنىڭغا ماڭا ئىككىنى بەر دەپ پورتمالىمنى ئاقتۇردۇم . ئۇ نەيرەڭلىرىنىڭ كارغا كەلگەنلىكىدىن خۇشاللىقىنى يۇشۇرالماستىن مەنسىتمىگەندەك سەت ھىجىيىپ قۇيۇپ سېغىزدىن ئىككىنى ماڭا سۇندى . پورتمالىمدا پارچە پۇل بولمىغانلىقى ئۈچۈن ئۇنىڭغا بەش يۈۋەنلىك پۇلدىن بىرنى بەردىم . ئۇنىڭ قولىدىن سېغىزنى ئېلىۋېتىپ : ئاشقان پۇلۇمنى قايتۇرۇپ بەر ، دېدىم . ئۇ ماڭا لاپپىدە بىر قاراپ قۇيۇپ ئۈندىمىدى. ئارقىدىنلا بۇرۇلۇپ كېتىشكە تەمشەلدى
__ ھەي !  دەپ ۋارقىرىدىم يۇقۇرى ئاۋازدا ئاچچىق بىلەن . ئۇ كېتىۋاتقاچ ئېرەنسىزلىك بىلەن بۇرۇلۇپ ماڭا قاراپ
__ ھە ، نېمە گەپ ؟  قايلىغاندا سېغىز ئاز  كەپ قالغان ئوخشىمامدا ، ماڭا رەھمەت ئېيتىشنى ئۇنتۇپ قالغانلىرى ئېسىلىرىگە كەپ قالدىمۇ-يە ، . دېۋىدى ئەرۋاھىم ئۇچۇپ:
  ___ كۆتە ۋا گـېپىڭنى ، ئاشقان پۇلۇمنى قايتۇرماي نەگە ماڭماقچىسەن ؟ . دېدىممەن ئۇنىڭغا يېقىنلاپ كېلىپ ، ئۇ ماڭا پۇتۇمدىن بېشىمغىچە قايتا بىر  مەنسىتمىگەندەك قاراپ قۇيۇپ :
  ___ ئاكا ، سىلى كىچىك چاغلىرىدا باشلانغۇچ مەكتەپنىڭ تېمىدىن ئارتىلىپ قېچىپ چىققانمۇ نىمە ؟ . ئۇنىڭ گېپىنى تازا ئاڭقىرالماي بىر ئاز تۇرۇپ قاپتىمەن . ئۇنىڭغىچە ياندىكى ئېغىزىدىن سۈت پۇراپ تۇرىدىغان كىچىك بىر شېرىكى توپىغا مېلەنگەن قولىنىڭ بارماقلىرىنى چىقىرىپ تۇرۇپ :
  ___ سىلى بايا سېغىزدىن ئىككىنى بىكارغىلا ئالدىممىكىن دېگەنما ؟ سېغىزنىڭ بىرى ئىككى يۈۋەن ئىككىسى قۇشۇلۇپ تۆت يۈۋەن دېگەن گەپ، يېنىمىزدا پارچە پۇل يوقكەن شۇنداق قىلىپ بىزگە تېگىدىغىنى بەش يۈۋەن . خاپۇلاشماي كەم قالغان بىر يۈۋەنگە سېغىزنى بىرەر كۈن ئوشۇق چاينىۋالسىلا . قارىساممۇ بىر موچەننىڭ ئۈستىدە تىك تۇرۇدىغاندەك كۆرۈنىلا .
  ئۇلار شۇنداق دېگىنىچە كېتىپ قېلىشتى . مەن بولسام بازاردىن بىر يۈۋەنگە سېتىۋالغىلى بولىدىغان سېغىزدىن ئىككىنى قولۇمدا تۇتقۇنىمچە تۇرۇپ قاپتىمەن . دەردىمنى ئىچىمگە يۇتۇپ چوڭقۇر بىر ئۇھ تارتقىنىمچە ئۈن-تۈنسىز  باغچىدىن چىكەتمەك بۇلۇپ سىرتقا قاراپ ماڭدىم ، كېتىۋاتقاچ نەچچە چامدا بىر شۇنداق ئەھۋاللارغا يۇلۇقتۇم .
  چۈشلۈك تاماقتىمۇ چىۋىن يەپ سالغاندەك گېلىمدىن تۈزۈك غىزا ئۆتمىدى . شۇ يەرلىك بىر قانچە دوستۇم ماڭا كۈلۈپ قاراپ قۇيۇپ :
  ___ بولدى قىل ، ئاداش ئىچىڭگە بەك ئەپ كەتمە . بۇ ئادالەتسىزلىككە تۇنجى قېتىم يۇلۇققان ئوخشىمامسەن؟ بۇنداق ئىشلار دائىم دېگۈدەك بۇلۇپ تۇرىدۇ ئۇلارغىمۇ باشقا ئامال يوق ، شۇ ئىشقا كۆنۈپ پىشىپ كەتكەن . ياخشىسى سەن بايا ئۇلارغا قاتتىق تەگمەپسەن ، بولمىسا ئارقىسىدىن يۈرىۋاتقان  چوڭ خوجايىنلاردىن تاياق يەپ كۆزۈڭدىكى ياشقا قۇشۇلۇپ ئاغزى بۇرنۇڭمۇ تازا بىر نەمدىلىپ چىققۇدەكسەن ئۇ يەردىن .
بۇ ئىشلانى ئاستا-ئاستا چۈشۈنۈپ قالغان ئىدىم ، ئىچىمدە ئاچچىق بىر تۇيغۇ تىپىرلىماقتا ئىدى . بىردە ئۇ كىچىك زومىگەرلەردىنى ئاغرىنسام بىردە ئۇلارنىڭ كىچىككىنە تۇرۇپ چىرىپ كەتكەن تەقدىرىدىن ئېچىناتتىم . قايتىش سەپىرىمدە شۇ يەرلىك دوستلۇرۇم ھەرخىل سوۋغا ئوتلۇق تېلەكلىرى بىلەن مېنى ئۇزۇتۇپ قويدى . ئۆزۈم پۇشقۇنۇپ كېتىۋاتقان پويىزنىڭ دەرىزىسىدىن بارا-بارا يېراقلاۋاتقان قەشقەر شەھرىگە قاراپ چوڭقۇر خىياللارغا پاتتىم . قەلبىمنى بىر خىل ئايرىلىش ئازابى ئۆرتىمەكتە ئىدى . يېراقلارغا قاراپ ئۆزۈمنىڭ شۇ ئىش سەۋەپلىك ئەزىزانە قەشقەرگە ۋە ئاق كۆڭۈل مېھماندوست خەلقىگە بولغان مۇھەببىتىمنىڭ تەسىرگە ئۇچرىماسلىقىنى ئۈمۈد قىلىپ ، چوڭقۇر سېغىنىش ۋە مۇھەببەت تۇيغۇسى بىلەن بۇ يۇرتتىن ئايرىلدىم .
   خەلقى ئاق كۆڭۈل خۇش چاقچاق يۇمۇرىستىك تۇرمۇشقا باي ياپيېشىل تاغلار قوينىدىكى گۈزەل ماكان غولجا شەھرى ھەممىمىزگە تونۇشلۇق . شۇ شەھەرنىڭ ئەڭ گۈزەل قايناق ساياھەت رايۇنى بولغان ، سۇلىرى ھەيۋەت بىلەن ئېقىۋاتقان ئىلى دەريا بويلىرىدا سۇنىڭ مۇڭلۇق شىردىرلاشلىرىدىن قىرغاقلارغا ئۇرۇلۇپ يىنىك چايقىلىشلىرىدىن ۋە غولجا خەلقىنىڭ خۇش چاقچاق يۇمۇرىستىك سۆزلىرىدىن قەلبىڭىز قانداقتۇر ھاياجان شاتلىقىغا چۆمگىنىڭىزدە ، سىزگە ئەگىشىۋېلىپ زورلاپ سېغىز سېتىۋاتقان كىچىك زومىگەرلەر كۆڭلىڭىزنى دىلىغۇل قىلماي قالمايدۇ . ئۇلارمۇ بېشىدا چىرايلىق سۆزلەر بىلەن بىرەر تال سېغىز ئېلىپ قۇيىشىڭىزنى ئۆتىنىدۇ ، ئالمىسىڭىز زورلاپ كەينىڭىزگە كىرىۋالىدۇ ، ھەتتا سىزنى بىرەر جايدا خاتىرجەم تۇرۇپ تاماشا قىلغىلىمۇ قويمايدۇ . ئاۋات بازارلىرىدىن بىرى بولغان خەنزۇ بازىرى ئەتراپىدىكى كوچىلاردا ۋە قىزىل بايراق يان تەرىپىگە جايلاشقان خەلق مەيدانى ئەتراپىدىمۇ شۇخىل ئەھۋال مەۋجۇت . بۇ يەرلەردىمۇ يەنە ئاللىقانداق كىشىنى بىزار قىلىدىغان ھېكايىلەر بۇلۇپ ئۆتتى بۇنىسى ماڭا نامەلۇم .
  يەنە دىيارىمىز شىنجاڭنىڭ ھەرقايسى شەھەرلىرىنى بىر ئارىلاپ كۆرۈش ئىستىكىم بار ئىدى ، ئەمما پۇرسەت بولماي قالدى . مەن يۇقارقى كۈندۈلۈك تۇرمىشىمىزدا ئۇچرىتىپ تۇرىدىغان ئاددى ۋەقەلەر ئارقىلىق ، بەزى يۇرتلارنىڭ شەخسەن ئىسمىنى ئاتاپ تۇرۇپ تەنقىت قىلماقچى ئەمەسمەن . چۈنكى مېنىڭ ئۇنچىلىك ھۇقۇقۇم يوق ، پەقەتلا مېنىڭ مېھنەتسىز خەلقىم ئۈچۈن ياخشى ياماننى ئىلغاپ بىر ئىش قىلىشىمىزنى تەكىتلەش ھۇقۇقۇم بار . مېنىڭلا ئەمەس ۋىژدان ئىگىسى بولغان ھەربىرىمىزنىڭمۇ ھۇقۇقىمىز بار مەجبۇريىتىمىز بار . رايونىمىز شىنجاڭ ھەربىر قېرىنداشلىرىمىزنىڭ ئورتاق ئانا يۇرتى ، كەلگۈسىنىڭ ئەۋلادلىرى بولغان كىچىك پەرزەنتلەر مۇشۇنداق تۈگۈشۈپ يولدىن چىقىپ كىتىۋاتسا ، ھەتتا بەزىبىر چوڭلارمۇ مەسئۇلىيەتسىزلىك بىلەن كىچىك بالىلارنى تەشكىللەپ ئەشۇنداق قارىسا ساتقان نەرسىسى قانۇنلۇق ئەمما ئىنسانىي ئەخلاقىمىزغا خىلاپ بولغان ، سودا قىلىش ئۇسۇلى مۇئامىلىسى دۇرۇس بولمىغان ئىشلارغا سېلىۋاتسا بىز يەنە بۇ ھەقتە ئويلانماي تۇرالايمىزمۇ؟!... ئەزىز ئۇيغۇر قېرىنداشلىرىم قېنى ئويلاپ باقايلى بۇ زادى كىمنىڭ مەسئۇليىتى .
  بىلىمەن بىر قىسىم ئاز ساندىكى پەرزەنتلەر  باققۇچىسى يوق ئائىلە شارائىتى ناچار ئاتا-ئانىسىدىن يېتىم قالغانلىقى ئۈچۈن بازارلاردا قانۇنلۇق يايمىلارنى ئېچىپ ، بازار قائىدە تەرتىپلىرىگە ۋە سودا ئەخلاقىغا بويسۇنغان ھالدا سەرەڭگە ، كىر يۇيۇش پاراشوكى ، دېھقانچىلىقتا ئىشلىتىدىغان ھاشارەت دورىلىرى دېگەندەك نەرسىلەرنى سېتىۋاتىدۇ بىزمۇ ئۆز ئىختىيارلىقىمىز بىلەن ئېلىۋاتىمىز . ئەمما ئۇلار ھېچقاچان يولدىن ئۆتىۋاتقان كىشىلەرنىڭ ئالدىنى توسۇپ نەرسىلىرىنى زورلاپ ساتمايدۇ ، ھەرگىزمۇ سېغىز سېتىۋاتقان كىچىك زومىگەر بالىلارغا ئوخشاش ئالمايمەن دېسىمۇ زورلاپ يۈرۈپ ساتمايدۇ ،ئۆزىنىڭ تېگىشلىك رىسقىنى تېپىپ يەيدۇ . ئەلۋەتتە ، بىزمۇ ئىنساندە . ھەرقانچە بولۇپ كەتسەكمۇ ئىنساۋىمىزنى يۇقاتقىنىمىز يوق . ئۇنداق بالىلارنى ئۇچرىتىپ قالساق سېتىۋاتقان نەرسىلىرىنى باھا تالاشماي سېتىۋالىمىز .
   بىزدە ئۇيغۇرلار يازغۇچىسى ئەختەم ئۆمەر يازغان ( مېنىڭچىمۇ ئەسەر ئاپتورى خاتا ئەمەسقۇ دەيمەن ) بېلەيچى بوۋاي دېگەن ھېكايە بار . ھېكايىنى بىردىنبىر بايان قىلىپ ئولتۇرمىساممۇ ھەممىمىزگە تونۇشلۇققۇ دەيمەن . ھېكايىدە بوۋاي مىڭ بىر مۇشاقاتتا قان تەرى بەدىلىگە ياساپ چىققان تاش بېلەينى بازارغا ساتقىلى ئەپ چىققىنىدا ، ئاپتور ئۇچراپ قالىدۇ . ئۇ بىر ھازا بوۋاي بىلەن پاراڭلىشىپ كۆرگەندىن كېيىن بوۋاينىڭ قىلىۋاتقان ئىشىنىڭ نەقەدەر جاپالىق ئىكەنلىكىنى بىلىپ ، ئۇنىڭغا ئىچ ئاغرىتىپ بېلەيدىن بىر قانچىنى ئارتۇق سېتىۋالماقچى بولىدۇ . ئەمما ساددا ئاق كۆڭۈل بېلەيچى بوۋاي ئالىيجاناپلىق بىلەن ئۇنىڭغا پەقەت بىرنىلا سېتىپ بېرىدۇ . چۈنكى ئۇ ئاپتورنىڭ ئىچ ئاغرىتىشى بەدىلىگە بېلەينى ئوشۇق سېتىپ جىپ پۇل تاپسام دەپ ئويلىمايدۇ . ئەمەلىيەتتىمۇ بىر ئائىلىدە بىر بېلەي بولسا نەچچە ئون يىل ئىشلىتىشكە يېتىدۇ . بىر ئادەمنىڭ بېلەينى ئوشۇق سېتىۋېلىشى باشقىلارغا نىسپەتەن ئىچ ئاغرىتىش ھېساپلانسا ، بوۋايغا نىسپەتەن ساتقان نەرسىنىڭ سېتىۋالغۇچى تەرىپىدىن ئىشلىتىش قىممىتى بولمىسا ساتقان نەرسە قەدىرسىز بولۇپ قالىدۇ . شۇنداق ئىكەن بىر بوۋاي باشقىلارنىڭ ئىچ ئاغرىتىشى بەدىلىگە قىلمىغان سودىنى بۈگۈنكى ياشلىرىمىز شوغۇللانساق ھەتتا شۇنى بىردىنبىر چىقىش يولى قىلىۋالساق سېلەرچە مۇۋاپىق بولارمۇ؟! يەنە بەزىلەر تۇرۇپ ئۇ بىر ھېكايە تۇرسا ئەمەلىي تۇرمۇشتا نېمە قىممىتى بولسۇن دېيىشى مۇمكىن ، ھەي ئۇيغۇر قېرىنداشلىرىم! ئەدىبىي ئەسەرلەرنى بىر ئەسەرلا خالاس دەپ قۇيۇپ ئەمەلىي تۇرمۇشقا تەتبىقلىمىساق رىئالىقتىكى خاتالىقلارنى ئىنسانىي ئەخلاقىمىزغا زىت بولغان تەرەپلەرنى تۈزەتمىسەك بەدئىي ئەسەرلىرىمىزنى لاپ ئۇرۇپ سۆزلەشنىڭ نېمە ئەھمىيىتى بولسۇن . يەنە كىلىپ بۇ مىسرانىم مۇنبىرىدە قۇرۇق ئەخلاقىي پەزىلىتىمىزنى ساقلاپ يۈرگىنىمىزنىڭ نېمە ئەھمىيىتى . شۇ بەدئىي ئەسەرلەرمۇ شۇ تۇرمۇشتىن كېلىپ چىققانغۇ؟!
