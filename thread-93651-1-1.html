ياخشى خەۋەرنى ئۆز ۋاقتىدا يەتكۈزۈشكە ئالدىرىغانلىقنىڭ ئۆزى بىر ياخشىلىق. ئۇنىڭ ئۈستىگە، بۇ ئۇشبۇ خەۋەر ئۇيغۇر مەدەنىيىتىگە مۇناسىۋەتلىك زور خوشاللىق ھەم كۈتۈلمىگەن يېڭىلىق بولغاچقا، ئوقۇرمەنلىرىمنى كېچىكتۈرمەستىن ۋاقىپ قىلىشقا ئالدىرىدىم. شۇنداق قىلىپ، گەپ ئىسپانىيەدە بولۇپ ئۆتكەن فولكلورلۇق ئانسامبىللارنىڭ خەلىقئارا فېستىۋالىدا بىرىنچى ئورۇننى ئىگىلەپ، مىللىتىمىز شەنىنى كۆتەرگەن «ئىرادە» ئانسامبلى ھەققىدە. ئىسپانىيەدە فولكلورلۇق ئانسامبللارنىڭ خەلىقئارا دەرىجىدىكى فېستىۋاللىرىنى ئۆتكۈزۈش ئەنئەنىسى بۇرۇندىن قېلىپلاشقان بولۇپ، ئۇ ھەر يىلى ئەمەلىيەتتىن ئۆز ئىپادىسىنى تېپىپ كەلمەكتە. بۇنىڭدىن بىر ئاز ۋاقىت ئىلگىرى ئەر-ئايال پىداكارلار دىلشات ناسىروۋ بىلەن سۈرىيەم ھەزىموۋا ئىسپانىيەنىڭ قازاقستان جۇمھۇرىيىتىدىكى ئەلچىخانىسىنىڭ ۋەكىللىرى بىلەن ئالاقىلىشىپ، بۇ چوڭ سىناققا قاتنىشىش رەسمىيەتچىلىكلىرىدىن ئۇچۇر ئالغان ئىدى. ئاندىن ئۇنىڭ شەرتلىرىنى ئورۇنلاش ئۈچۈن ئالمۇتا — ئاستانە ئارىلىغىدا ھەرىكەت قىلىدۇ. مەلۇم بولۇشىچە، ئاساسلىق ھۆججەتلەر ئانسامبىل ۋە ئۇنىڭ رەپەرتۇارى توغرىلىق مەلۇمات ھەم ئۇلارنى تەستىقلەيدىغان ۋىيدىئوماتىرياللاردىن ئىبارەت بولۇپ، ئۇلارنى ئالدىن-ئالا، يەنى بەش ئاي ئىلگىرى يەتكۈزۈپ بېرىش كېرەك ئىكەن. بۇلارنىڭ ھەممىسى ئۆز ۋاقتىدا بېجىرىلىدۇ. ئاخىرى، فېستىۋالنىڭ ئېچىلىشىغا ئون كۈن قالغاندا، ئېنىقىراقى، ئاۋغۇست ئېيىنىڭ يىگىرمىلىرى تەكلىپنامە كېلىدۇ. …30-ئاۋغۇست. تاڭ سەھەردە ئالمۇتا — ئىستامبۇل — ۋالىنسىيە رەيسى بىلەن بەجايىكى ھەيۋەتلىك ئاققۇشلاردەك تاڭ شەپىقىدە كۈمۈشتەك يالتىراپ، ئايروپىلان يەردىن كۆككە كۆتىرىلىپ، ئۇچۇپ بارىدۇ. ئۇنىڭ ئىچىدە پانفىلوۋ ناھىيەسى ھاكىمىنىڭ ئورۇنباسارى ئامانتاي رائەۋ، ئىسىم-شەرىپى يۇرت دائىرىسىدىن ئالقىپ، بىپايان جۇمھۇرىيىتىمىزگە تونۇلۇۋاتقان دىلشات ناسىروۋ بىلەن سۈرىيەم ھەزىموۋا باشلىغان «ئىرادە» ئانسامبلى بار. ئۇلار شۇ كۈنى فېستىۋال شەھىرى بولغان — سان-خاۋئەراغا يېتىپ كەلدى. بۇ يەردە ئۇلارنى شەھەر ھۆكۈمىتىدىكى رەسمىي ۋەكىللەر، فېستىۋالنى ئۆتكۈزۈشكە مەسئۇل شەخسلەر ۋە سەنئەت مۇخلىسلىرى سەمىمىي كۈتۈۋالدى. ئۇلارنىڭ ئارىسىدا سەنئەتشۇناس تاتيانا لارگىنا ئىسىملىق رۇس ئايالى بولۇپ، ئىسپانىيا پۇخرالىقىغا ئۆتكەن ئىكەن. مانا شۇنىڭ تەرجىمانلىغىدا ئۆز ئارا تونۇشۇشلار باشلىنىپ، دەسلەپكى دەقىقىلەردىكى ياتسىنىش، ئەيمىنىشلەر دوستانە مۇئامىلە-مۇناسىۋەتلەرگە ئۇلىشىپ كېتىدۇ… سان-خاۋئەرا دۇنياۋى سەنئەت بايقاش-كۆرۈكلىرى ئۆتكۈزۈلگىنى بىلەن، ئۆلكىلىك دەرىجىدىكى شەھەر ئىكەن. شەھەر ئاھالىسى «قازاقىستان» دەپ ئاڭلىغىنى بولمىسا، جۇمھۇرىيىتىمىز توغرىسىدا تەپسىلىي مەلۇماتلارغا ئىگە ئەمەس ئوخشايدۇ.  بىز ئۇيغۇرلار ھەققىدىغۇ تامامەن ئاڭلىماپتۇ. ئۇنىڭ ئۈستىگە موشۇ كۈنگىچە بۇ ئەنئەنىۋىي فېستىۋالغا بىرمۇ ئۇيغۇر ئانسامبلى قاتناشمىغان ئىكەن. موشۇ بوشلۇقنى تولدۇرۇپ، ئامانتاي ئابدۇقادىر ئوغلى، دىلشات ھۈسەيىن ئوغلى بىلەن سۈرىيەم تۇرغان قىزى ئۇلارغا جۇمھۇرىيىتىمىز ھەققىدە ھەرتەرەپلىمە ئاخبارات-چۈشەنچە بېرىدۇ. شۇنىڭ بىلەن بىر قاتاردا، 30-ئاۋغۇست كۈنىنىڭ بىز ئۈچۈن ئالاھىدە ئەھمىيەتكە ئىگە ئىكەنلىگىنى ئېيتىدۇ. سەۋەبى، موشۇ كۈنى قازاقستان كونستىتۇسىياسى قوبۇل قىلىنىپ، ئۇنىڭدا ئېلىمىزدا ئىستىقامەت قىلىۋاتقان 130دىن ئوشۇق مىللەتنىڭ ھەق-ھوقۇقى تولۇق كاپالەتلەندۈرۈلگەنلىكىنى چۈشەندۈرىدۇ. مەركىزىي ئاسىيادىكى قەدىمىي ئۇيغۇر خەلقىنىڭ قازاقىستاندىكى ۋەكىللىرىنىڭمۇ ئۆز سەنئىتى بىلەن مەدەنىيىتىنى تەرەققىي ئەتكۈزۈۋاتقانلىغى، ئۇشبۇ فېستىۋالغا ئىشتراك قىلىش ئۈچۈن «ئىرادە» ئانسامبلىنىڭ قاتنىشىۋاتقانلىغى ئەنە شۇ تەڭ ھوقۇقلۇقنىڭ يارقىن مىسالى ئەكەنلىگىنى قەيت قىلىدۇ. بۇنىڭدىن قانائەت ھاسىل قىلغان ساھىپخانلار مېھمانلارنى قىزغىن تەبرىكلەپ، كونستىتۇسىيامىز كۈنى شەرىپىگە دەستىخان يېيىپ، ئالاھىدە ھۆرمەت كۆرسىتىدۇ. 1-سېنتەبىر كۈنى 24-خەلىقاارا فولكلورلۇق فېستىۋال تەنتەنىلىك رەۋىشتە باشلاندى. ھاياجانلىق، شۇنىڭ بىلەن بىللە، ئەندىشىلىك دەقىقىلەر. نوۋىتى كېلىپ، بىزنىڭ قارا كۆزلىرىمىز سەھنىگە كۆتىرىلدى. ئەيتماقچى، ئىسپانىيەدە مۇنداق كونسەرت-تاماشىلارنى مەدەنىيەت ئۆيلىرى ياكى سەنئەت سارايلىرىغا قارىغاندا، كۆپىرەك ئوچۇق ئاسمان ئاستىدا ئۆتكۈزۈش ئەنئەنىسى قېلىپلاشقان ئىكەن. ئۇ كۈنى سىمىلداپ يامغۇر يېغىپ تۇراتتى. «بۇ ئاللاھنىڭ بىزگە تۆكۈلۈۋاتقان رەھمىتى، بالىلىرىم. ئاقىۋىتى خەيىرلىك بولغاي!» موشۇنداق ئىزگۈ تىلەكلەر بىلەن كىچىكىدىن ئىمان-ئېتىقادلىق سۈرىيەم تۇرغان قىزى ئالتۇن قەپەزدە ئاسراپ بېقىۋاتقان ئالتۇن قۇشلىرىنى ياۋروپا ئاسمىنىغا ئۇچىرىۋەتتى. …ئالدى بىلەن سەلتەنەتلىك ھەم ھەيۋەتلىك مۇقام ياڭرىدى. بىردە مۇڭ-دەرت، بىردە ئارزۇ-ئارمان، بىردە كاج پەلەككە باش ئەگمىگەن ئىسىيانكارلىق، بىردە ھايات ئىشقىدىن لەززەتلىنىپ، بەھىرلىنىش سېزىملىرى يۇغىرىلىپ كەتكەن ئۇلۇغۋار ناۋا داستان-نەغمىلەر ھەممىنى تەۋرەتتى. سەھنىدە جۈپ-جۈپى بىلەن پەيدا بولغان ئوغۇل-قىزلىرىمىز ئۇيغۇر ئۇسسۇللىرىنىڭ ئىنسان قەلبىنى سېھىىرلاۋالغۇچى تىلسىم كۈچ-قۇدرىتىنى ھەقدادىغا يەتكۈزدى. مەيدان سۇ سەپكەندەك جىمىپ، تاماشىبىنلار تىڭىرقاپ قېلىشتى. ھەممىنىڭ دىققەت-ئېتىبارى پەقەتلا سەھنىدە. ئۇ يەردە بولسا، «ئىرادە» ئونلىغان ئەسىرلەرنى ئارقىدا قالدۇرۇپ، بىزنىڭ دەۋرىمىزگە يېتىپ كەلگەن ئۇلۇغ سەنئەتنىڭ ھەققانىي ۋارىسى ھەم مىراسخورى ئىكەنلىگىنى نامايىش قىلماقتا. بىر چاغدا بىر-بىرىنى مەزمۇن-ماھىيەت جەھەتتىن تولۇقلاپ، پۈتۈن مەيداننى مەھلىيا قىلىۋالغان نەغمە-ناۋا توختىدى. بۇ فېستىۋالنىڭ شەرتىگە مۇۋاپىق ئاجرىتىلغان قىرىق مىنۇتنىڭ ئۆتۈپ كەتكەنلىكى ئىدى. بىراق تاماشىبىنلارنىڭ گۈلدۈراس چاۋاكلىرى بىلەن «براۋو!»، «بىست!» (بىزچە «جۈپ بولسۇن» دىگىنى) دەپ توۋلاشلىرى توختىمىدى. ئاخىرى ئادىل باھالىغۇچىلارنىڭ قارارى بىلەن يەنە ئون مىنۇت قوشۇپ بېرىلدى. دەم ئېلىش كۈنلىرى بولۇپ قالغانلىقتىن، فېستىۋالنىڭ يەكۈنى ئارىدىن ئىككى كۈن ئۆتۈپ چىقىرىلدى. نەتىجىدە «ئىرادە» بىرىنچى ئورۇننى ئېگىلىدى. بۇ ھەقتە سان-خاۋئەرا شەھىرىنىڭ شەھەر باشلىقى (ئەپسۇسكى، تىل بىلمىگەنلىكتىن، ئىسىم-شەرىپى ھىچ كىمنىڭ يادىدا قالماپتۇ) ئۆزى ئېلان قىلىپ، 1- دەرىجىلىك دىپلوم بىلەن شۇنىڭغا مۇۋاپىق ستاتۇەتكىنى دىلشات ناسىروۋ ھەم سۈرىيەم ھەزىموۋاغا قىزغىن چاۋاكلار ساداسى ئاستىدا تەنتەنىلىك شارائىتتا تاپشۇردى. «ئاكا، دەل شۇ دەقىقىلەردە بىزنىڭ بالىلىرىمىزلا ئەمەس، سۈرىيەم ئىككىمىزمۇ ئۆزىمىزنى تۇتالماي قالدۇق…» دېگەن دىلشاتنىڭ سۆزىگە ئۇلاپ: «يىراق ياركەنتتە تۇرۇپ، مۇنداق خۇش خەۋەرنى ئاڭلاپ، بىزمۇ شۇنداق ھالەتنى باشتىن كەچۈردۇق» دېدىم. ئىسپانىيەدە ئاۋغۇست-سېنتەبىر ئايلىرى سەنئەت بايرىمىنىڭ مەۋسۈمى سۈپىتىدە ئۆتىدىكەن. بۇ ئىلگىرىدىن كېلىۋاتقان ئەنئەنە بولۇپ، بۈگۈنكى دۇنياۋىي بوھران شارائىتىدىمۇ داۋاملىشىۋېتىپتۇ. سان-خاۋئەرادىن كېيىن «ئىرادە»نى بىر قاتار شەھەرلەرگە تەكلىپ قىلدى. مەسىلەن، مۇەسترا شەھىرىدە بولۇپ ئۆتكەن سەنئەت بايرىمىگە پەرۇ سەنئەتچىلىرى بىلەن بىللە قاتنىشىپ، ئەللىك مىنۇتلۇق كونسېرت كۆرسەتتى. ئالباسىتو، مەراندا شەھەرلىرىدە ئۆتكەن بايقاشلارغىمۇ ئىشتراك قىلدى. برازىلىيە، ئۆزبېكستان ۋە باشقا دۆلەتلەردىن كەلگەن سەنئەتچىلەر بىلەن مۇسابىقىغا چۈشۈپ، تاماشىبىنلارنىڭ ئىللىق ئىخلاس-ئېھتىرامىغا بۆلەندى. مەراندا شەھىرىدە تەلەپلەرگە بىنائەن ئۈچ كۈن ئۇدا كونسېرت قويغانلىغى، ئالباسىتو شەھىرىدىكى ئىككىنچى قەتىملىق كونسەرتىغا 3500دىن ئوشۇق ئادەمنىڭ قاتناشقانلىغى، ئۇمۇمەن، ھەر بىر كونسېرتتا 1000 — 1500دىن ئوشۇق تاماشىبىننىڭ بولغانلىغى ۋە نىھايەت، بارغان يەرىدە1- دەرىجىلىك دىپلوم ھەم خاتىرە سوۋغىلىرىنىڭ تاپشۇرۇلغانلىغى ئانسامبىل سەپىرىنىڭ قانچىلىك دەرىجىدە ئۇتۇقلۇق ئۆتكەنلىگىدىن دىرەك بېرىدۇ. يەنە بىر قىززىق يېرى، ئۇ ئەلدە ھەر بىر شەھەردە بايرام ئۆتكۈزۈلىدىغان مەيدانغا بىر ئاز قالغاندا، ناخشا-ئۇسسۇللار بىلەن پىيادە مېڭىپ، كىرىپ بېرىش رەسمىيىتى بار ئىكەن. ئانسامبىلنىڭ بەدىئىي رەھبىرى پولات ھەزىموۋنىڭ ئېيتىپ بېرىشىچە، ئۇلار كارناي، سۈنەيلەرنىڭ جور بولۇشىدا دولان ئۇسۇلىنى ئىجرا قىلىپ ماڭغاندا، كوچىنىڭ ئىككى قاسنىغىدىكى ئادەملەر بىزنىڭ قارا كۆزلىرىمىزنى ئەگىشىپ مېڭىپتۇ. ئۇمۇمەن، «ئىرادە» ئانسامبلى 1 — 10-سېنتەبىر كۈنلىرى ئارىلىقىدا يەتتە قېتىم كونسېرت قويۇپتۇ ۋە ئۇلارنىڭ ھەممىسى ئىنتايىن ئۇتۇقلۇق ئۆتۈپتۇ. ئانسامبلنىڭ بۇ سەپىرى پەقەت كونسېرت قويۇش بىلەنلا ئۆتتى دېسەك، ئازلىق قىلىدۇ. بىزنىڭ يۇتداشلىرىمىز ھەر بىر كونسېرت باشلىنىشتىن ئىلگىرى قازاقىستان جۇمھۇرىيىتى توغرۇلۇق، ئۇيغۇر خەلقىنىڭ تارىخى، ئۇنىڭ سەنئىتى، ئون ئىككى مۇقام، مىللىي چالغۇ ئەسۋاپلىرىمىز ھەققىدە ئىمكانىيىتىنىڭ يېتىشىچە ئەتراپلىق چۈشەنچە بېرىپ، مەملىكىتىمىزنى، مىللىتىمىزنى تونۇشتۇرۇپتۇ. شۇنىڭ بىلەن بىر قاتاردا، بارغان شەھەرلەر بايراملىرىغا ئاستانەدىكى بايتەرەك مونۇمەنتىنىڭ ماكەتىنى، دومبىرا بىلەن بىزنىڭ ئەسۋاپلىرىمىزنى سوۋغا قىپتۇ. ئۆز نوۋىتىدە ئۇلارمۇ قېرىنداشلىرىمىزنى قۇرۇق قول قايتۇرماپتۇ. …14-سېنتەبىر، جۈمە كۈنى. ئۇلۇق «يىپەك يولى»نىڭ بويىدا، ئۆسەكنىڭ ئەڭ بىر مەنزىرىلىك جايىدا كۈتۈۋاتىمىز. ناھىيەلىك كېڭەشنىڭ كاتىبى تىلەئۇبەرگەن ئوسپانوۋ، ياركەنت شەھىرىنىڭ ھاكىمى باقيت نۇرەلوۋ، ناھىيەلىك ئۇيغۇر مەدەنىيەت مەركىزىنىڭ رەئىسى رەھىمجان توختاخۇنوۋ ۋە شەھەردىكى بىر قاتار مەكتەپلەرنىڭ مۇدىرلىرى بىلەن ئوقۇغۇچىلىرى بار بۇ توپنىڭ سېپى بارا-بارا ئۇلغىيىپ، تاقىتى تاق بولماقتا. بىر كەمدە كۈتكەن ئادەملىرىمىز ئولتۇرغان ماشىنىلارنىڭ قارىسى كۆرۈنۈپ، توپتىن يۈز مېتىرچە نېرى يەرگە كېلىپ توختىدى. قازاقىستان جۇمھۇرىيىتىنىڭ دۆلەت بايرىغىنى تىك تۇتۇپ، قەددىنى رۇسلىغان ھالدا ئەنە بىزنىڭ ئەتىگەندىن بېرى بىتاقەت كۈتكەن ئەزىمەتلىرىمىز كېلىۋاتىدۇ. سەپ ئالدىدا دىلشات ناسىروۋ بىلەن سۈرىيەم ھەزىموۋا، ئۇلارنى ئەگىشىپ سەپ تۈزگەن «ئىرادە» ئانسامبلىنىڭ ئوغۇل-قىزلىرى. كارناي، سۈنەي ۋە ناغرىلارنىڭ ھەيۋەتلىك ساداسى ئاستىدا قۇچاقلىشىپ، كۆرۈشۈپ كەتتۇق. مەكتەپ ئوقۇغۇچىلىرى ئۇلارغا قۇچاق-قۇچاق گۈلدەستىلەرنى تەغدىم قىلدى. ئامانتاي رائەۋ بىلەن تىلەئۇبەرگەن ئوسپانوۋ ئۇلارنى ياۋروپادىكى زور مۇۋاپىقىيىتى بىلەن سەمىمىي تەبرىكلىدى. شۇنداقلا باقيت نۇرەلوۋ، رەھىمجان توختاخۇنوۋ، رىزايدىن ئەيسائەۋ، سەيىت ۋەلىئە ۋمۇ سۆز سۆزلەپ، بۇ مۇۋاپپەقىيەتنىڭ «ئىرادە» ئانسامبلىنىڭ كەلگۈسى يۈكسەك پەرۋازى ئۈچۈن باشلانما بولۇپ قېلىشىغا تىلەك بىلدۈردى. شۇنىڭدىن كېيىن شەھەرگە قاراپ يول ئالغان ماشىنىلار سېپى «ساز» كافىخانىسىنىڭ ئالدىغا كېلىپ توختىدى. بۇ يەردىمۇ كارناي، سۈنەي بىلەن ناغرا ساداسى ياڭراپ، يوللىرىغا تەلمۈرتكەن ئارزۇلۇق مېھمانلار تەنتەنىلىك ھالدا كۈتۈۋېلىندى. مول تەييارلانغان كەڭ دەستىخان ئەتراپىدا چىن يۈرەكتىن ئىزگۈ تىلەكلەر ئىزھار قىلىندى. رەھىمجان توختاخۇنوۋنىڭ قىسقىچە  سۆزىدىن كېيىن ئامانتاي رائەۋ سۆزگە چىقتى. ناھىيە ھاكىمىنىڭ ئۈزۈرلۈك سەۋەپلەر بىلەن بۈگۈنكى تەنتەنىلىك ئۇچرىشىشقا قاتنىشالمايۋاتقانلىغىنى ئېيتىپ، «ئىرادە»نى ناھىيە ئاھالىسى نامىدىن تەبرىكلىدى. ئانسامبىل بىلەن بىللە ئىسپانىيەگە بېرىپ، ئۇنتۇلماس تەسىراتلار ئېلىپ كەلگەنلىگىنى قەيت قىلدى. سۆز نوۋىتى دىلشات ناسىروۋقا كېلىپ، ئۇ مونۇلارنى قەيت قىلدى: — بىزنى مۇشۇنداق قوللاپ-قۇۋەتلەپ، ھېسداشلىق بىلدۈرۈۋاتقىنىڭلار ئۈچۈن چەكسىز مىننەتدارمىز. قازاقىستاندا ئىستىقامەت قىلىۋاتقانلىغىمىزغا پەخىرلىنىشىمىز كېرەك. «ئىرادە»نىڭ تەشكىل قىلىنىپ، موشۇ دەرىجىگە يېتىشى بۇ بىزگە يارىتىلىۋاتقان شارائىت بىلەن كۆرسىتىلىۋاتقان غەمخورلۇقنىڭ نەتىجىسى. ئانسامبل ئالدىدا تېخى كۆپ ئىشلار تۇرىدۇ. ئىسپانىيەدىكى كۈچ سىنىشىشقا قاتنىشىشىمىز ئەنە شۇلارنىڭ باشلانمىسى. ئويلايمەنكى، كۆڭۈلگە پۈككەن ئارزۇ-ئارمانلىرىمىز ئەمەلگە ئاشىدۇ. چۈنكى بۇنىڭ ئۈچۈن ئېلىمىزدا ھەممە شەرت-شارائىتلار بار.  بىزنى ھەر قاچان قوللاپ-قۇۋەتلەپ كېلىۋاتقان ناھىيەلىك ھاكىمىيەتكە سەمىمىي مىننەتدارلىقىمىزنى ئىزھار قىلماقچىمەن. ھەتتا ئەشۇ يىراق ئىسپانىيەدە يۈرگىنىمىزدىمۇ ماخاببات سەدۇاقاس ئوغلى بىگەلدىئە ۋ بىزنى ئۇنتۇماستىن، ھەر كۈنى دېگىدەك تېلىفون ئارقىلىق خەۋەرلىشىپ تۇردى.ئەھۋالىمىزنى سوراپ، روھىي جەھەتتىن يار-يۆلەك بولدى. فېستىۋالدا غالىب چىقىپ، قازاقىستاننىڭ تېخىمۇ كەڭ تونۇلۇشىغا ئازراق بولسىمۇ ھەسسىمىزنى قوشقانلىغىمىزنى بەختىمىز دەپ ھېساپلايمەن، — دېدى سۈرىيەم ھەزىموۋا ئۆز سۆزىدە. تەنتەنىدە شۇنداقلا نانەسىمە ئەركىن، ئابدۇرەشىت ھوشۇروۋ، يالقۇنجان تاھىروۋلار سۆزگە چىقىپ، موشۇنداق بىر ئىجادىي كوللىكتىپنى تەشكىللەپ، بارلىق شەرت-شارائىتلارنى يارىتىپ بېرىۋاتقانلىغى ئۈچۈن دىلشات ھۈسەيىن ئوغلى بىلەن سۈرىيەم تۇرغان قىزىغا ئانسامبل ئەزالىرى نامىدىن تەشەككۈر بىلدۈردى. ئۇچرىشىش ئانسامبىلنىڭ ھەممىدە ئاجايىپ ياخشى تەسىراتلارنى قالدۇرغان كونسېرتى بىلەن يەكۈنلەندى.
ئىلاۋە: ماقالە نەشىرگە تەييارلىنىۋاتقان پەيتتە، تەھرىراتىمىزغا پانفىلوۋ ناھىيەسىنىڭ ھاكىمى ماخاببات سەدۇاقاس ئوغلى بىگەلدىئە ۋ «ئىرادە» ئانسامبلىنىڭ كوللەكتىۋىنى قوبۇل قىلىپ، ئىسپانىيەدە ئۆتكەن فېستىۋالدا بىرىنچى ئورۇننى ئىگىلىگەنلىكى مۇناسىۋىتى بىلەن تەبرىكلىگەنلىكى توغرىلىق خەۋەر كەلدى. ھاكىم ئۇلارغا ئەڭ ياخشى تىلەك-ئىستەكلىرىنى ئىزھار قىلىپ، تېخىمۇ چوڭ ئۇتۇقلارنى قولغا كەلتۈرۈشىگە تىلەكداشلىق بىلدۈرۈپتۇ. سۈرەتلەردە: سەھنىدە — «ئىرادە». 2. «ئىرادە»نى  يۇرتداشلارشۇنداق داغدۇغىلىق كۈتۈۋالدى.




ئەسكەرتىش:ئەھميەتلىك تىما بولغاچقا تېھران بىلوگىدىن يۆتكەپ كىلندى.
