                                                     تىلەمچىلەرگە باشقا يول يوقمۇ؟؟........

                                                        ئابدىلتىپ ئابدىخالىق
                              

دۇنيا ئىقتىسادى بۈگۈنكىدەك تەرەققىي قىلغان ،ئىقتىسادنىڭ تەڭپۇڭلىشىش قەدىمى كۈندىن-كۈنگە تىزلىشىۋاتقان ، خەلقىمىزنىڭ ئىقتىساد-ئىستىمال سەۋىيەسى بۈگۈنكىدەك ئەلالىشىۋاتقان مۇشۇ ۋاقىتلاردىمۇ بىز يەنىلا بازارلاردا،يول-يوللاردا،دوختۇرخانىلاردا،ھەم مەسجىدلىرىمىزدىمۇ ئۆزىنى بىچارە كۆرسىتىش ئۈچۈن قىلمىغىنى قالماي،يىغلاپ ھالى قالماي ،ۋاقىراۋۋىرىپ قۇلىقىمىزنى ئاغرىتىۋىتىدىغان،ئازادە ماڭغىلى قويماي يوللاردا مەينەتچىلىككە پۇچىلىنىپ ياتقان، قىلىنلىق بىلەن ئادەمگە يىپىشىۋىلىپ ،ئاتالمىش ( سەدىقە ) تىلەيدىغان ھاجەتمەن ،تىلەمچى ،سايىل ،ئابداللارنى ئۇچرىتىپ تۇرىمىز.ھەمدە رازىلىقىمىز بىلەن ياكى رازىلىقسىز زورى بىلەن پارچ پۇللارنى شۇلارغا بىرىۋاتىمىز،بەزىسىگە ئىچ ئاغرىتىۋاتىمىز، بەزىلىرىنىڭ قىلغان ساختىلىقلىىدىن مىيقىمىزدا كۈلگەچ ئىچىنىپ قالىمىز .ئەجەبا شۇلار جۈملىدىن شۇ قىرىنداشلىرىمىز راستىنلا شۇنچىلىك قىيىنچىلىقتا قالغانمىدۇ ،دەپ ئويلاپ كىتىمىز .زادى شۇلار تىلەمچىلىك قىلمىسا جېنىنى باقالمامدۇ ؟ ياكى ئۇلار پۇل ئۈچۈنلا شۇنداق قىلىۋاتامدۇ ؟بۇ ئىشلار ھەممىمىزگە ئايان .لىكىن بۇ يەردە مەن ئوتتۇرىغا قويماقچى بولغان مەسىلە : ئۇلارغا بۇنىڭدىن باشقا چىقىش يولى يوقمىدۇ؟

سۆزلەپ مۇشۇ يەرگە كەلگەندە مەن ئۆز كۈزۈم بىلەن كۆرگەن مۇنۇ ئىشنى ئوتتۇرغا قويماقچى؛
مۇشۇ يىلى يازدا خوتەن شەھىرىمىزدە ئۆتكۈزۈلگەن ،( قاشتېشى بايرىمى) ۋە ( قولدا تۇقۇلغان قوي يۇڭى گىلىمى يەرمەنكىسى)مەزگىلىدە يەرمەنكىنى كۈرۈپ كىلىش ئۈچۈن سىڭلىم بىلەن ئىككىمىز قاراقاشتىن ئاتايىتەن بارغانىدۇق.يەرمەنكىدىكى ئىسىل قاشتىشى ، رەڭدار گىلەملەر، ئالاھىدە نەپىس ئىشلەنگەن قول ھۈنەر بۇيۇملىرىنى كۈرىۋىتىپ مەن ئويلىمىغان بىر ئىشقا ئۇچرىدىم .يەرمەنكە خەلىق مەيدانىغا ئۇيۇشتۇرۇلغان بولغاچقا ،خەلىق مەيدانىنى بىر ئادەم دىڭىزى دىسەكمۇ بۇلاتتى .چەتئەل،ئىچكىرى ئۆلكىلەردىن ،شىنجاڭنىڭ ھەر قايسى جايلىرىدىن كەلگەن ،سودىگەر، ساياھەتچىلەر بەك كۆپ ئىدى .مەن خەلىق مەيدانىنىڭ يان تەرىپىدىكى كەشتىچىلىك بۇيۇملىرىنى كۈرۈپ كىلىش ئۈچۈن مىڭىۋاتسام بىر مۇنچە كىشىلەر چەمبەرسىمان ئورىۋالغان يەرگە كۈزۈم چۈشتى ،ۋە نىمە ئىش بولغانلىقىنى ئۇقۇپ بېقىش ئۈچۈن ئۇ يەرگە باردىم .توپنىڭ يىنىغا كىلىشىمگىلا ئاۋازىدىن ئۇيغۇر ئەمەس لىكىن ئۇيغۇرچىنى خېلى يامان ئەمەس سۆزلەيدىغان،  سېرىق چاچلىرىنى  مىللىتىمىزنىڭ دوپپىسىغا يۇشۇرۇشقا ئۈلگۈرگەن بىر چەتئەللىك ئايالنىڭ گىپى خىيالىمنى بۆلدى .
ۋاي  شۇنداك ۋىجدانلىك ياكشى ئادەمكىنا بۇ .ئۇزەم ئىشلەپ يەي دەيدۇ     مۇشۇنداكلا جىك بوسا ھە.            مەن توپنىڭ ئىچىگە سۇقۇنۇپ كىرىپ قارىسام توپنىڭ ئۇتتۇرىسىدا ئوتتۇز ياش ئەتراپىدىكى ،كىيىملىرى رەتسىز كىيىلگەن،چاچ ساقاللىرى ئۈسۈپ كەتكەن ، تەن قۇرۇلۇشى نورمال ئەمەس (يەنى بەلكىم كىچىك چېغىدا بالىلار پارالىچى بۇلۇپ ساقايغان بولسا كېرەك)بىر ئادەم كىچىككىنە يايمىغا ،ئادەتتە كۆپ ئىشلىتىدىغان يانچۇق قەغىزىدىن رەتلىك تىزىپ ئۇلتاغانىكەن .ئۇنىڭ ئۈزى ساق بولمىغان ئەھۋال ئاستىدىمۇ يەنە مۇشۇنچىلىك تىجارەت قىلالىغىنىغا ھەيران قالغان ياكى ئۇنىڭغا زوقى كەلگەن بولسا كېرەك ،ھەر مىللەتتىن نۇرغۇن كىشىلەر ئۇنى ئورىۋالغانىكەن.شۇنداق تۇرۇشۇمغا ھېلىقى چەتئەللىك ئايال يەنە سۆز  باشلىدى ؛ ھەي لورا  سەن كۆدۇڭمۇ .بايام ئۇ مەن پۇلنى قايتۇرماڭ دىسەممۇ ئۇ  ئۇنىمايدىكەن تېخى ،بىزنىڭ دۆلەتتىمۇ شۇندا بوسا ھە ،ئۇ شۇنداق دىگەچ قەغەزدىن ساناپ ئوننى ئالدى . ئاندىن كۈلۈمسىرگىنىچە ئۇ كىشىگە قاراپ ئېلىڭە ماۋۇ ئون كوينى، دەپلا باشقا كىشىلەرگە قاراپ مۇراجىئەت تەلەپپۇزىدا .دوسلار مەن روسىيەدىن كەلدىم ،بۇ ئادەم مىنى بەكلا تەسىر لەندۈردى .ئۇ  ئۆز ..ئۆز..ئايال توختاپ قالدى .مەن ئۇنىڭ گىپنى تاپالمايۋاتقانلىقىنى سىزىپ دەرھال دىدىم. ئۆز نىسىۋىسىنى دىمەكچىما ؟ ئايال خۇشاللىقىدىن سەكرىگىنىچە   ، ھە .ھە    شۇنداك دىمەكچىمەن .ئايال سۆزىنى قايتا باشلىدى ؛ئۆزنىسىۋىسىنى ئۆزى تىپىپ يىسەم دەيدىكەن .قەغەز ھەممىمىزگە لازىم بۇلىدىغان نەرسە. باشقا دۇكاندىن سېتىۋالغۇچە مۇنۇ ئەپەندىدىن سېتىۋالساڭلا دىمەكچى. ئايالنىڭ گىپى تۈگە- تۈگىمەي ھەممە ئادەم ئىككى- ئۈچتىن قەغەز ئېلىپ پۇل بىرىشكە باشلىدى .مەنمۇ قۇلۇمدىكى ئىككى قەغەزنى تۇتقىنىمچە خىيال قىلىپ كىتىپتىمەن .
مانا ئۇيغۇر ،مانا ئۇيغۇر  ،ئاددىيلىق ئىچىدىن ئۇلۇغلۇق تاپالايدىغان ئۇيغۇر دىگەن مانا مۇشۇنداق .قەلبىم باشقىچە بىر خىل ھاياجانغا تۇلۇپ ،شۇ قەغەز ساتقۇچىغا ،ياق ،ياق قەغەزچى ئەمەس ،ئۇ بىر سودىگەر  ،ھەقىقى سودىگەرگە قارىغىنىمچە كۆزلىرىمدىن ئىسسىق ياشلىرىم مەڭزىمگە سىرغىغىنىنىمۇ  تۇيماي قاپتىمەن ،تۇيۇقسىز سىڭلىمنىڭ بىلىكىمنى تارتىشىدىن ئىسىمگە كەلدىم . ئاكا ،ئاكا سىنى چاقىرۋاتىدۇ .مەن قارىسام ھېلىقى روسىيەلىك ئايال ماڭا قاراپ تۇراتتى .مەن ھاپىلا -شۇپىلا يىشىمنى سۇرتكىنىمچە ئۇنىڭغا سۇئال نەزىرىدە تىكىلدىم .رەھمەت سىزگە ، مەن ئۇيغۇر تىلىنى ياخشى كۈرىمەن لېكىن تازا دىگۈدەك ئۈگىنىپ بۇلاممىدىم.ياردىمىڭىزگە يەنە بىر قىتىم رەھمەت  ئېيتىمەن.  ئۇ  مىنى گەپ قىلدۇرمايلا رەھمەت ئېيتىپ كەتتى  .مەنمۇ ئېغىز ئاچتىم .ۋاي  بولدى ،مەن سىزگە رەھمەت ئېيتسام بۇلىدۇ تېخى . ئۇ دىدى  نىمىشكا  ئەمدى . چۇنكى سىز ماڭا ئانا تىلىمنىڭ نەقەدەر گۈزەل ھەم قەدىرلىك ئىكەنلىكىنى تۇنۇتتىڭىز ئەمەسمۇ . ئاڭغىچە يەنە بىر سېرىق چاچلىق ئەر ئۇلارنى چاقىردى بولغاي ، ئۇ ماڭا مەن شىنجاڭدا كۆرگەن بۇ ئىشنى چۇقۇم دوسلىرىمغا يەتكۈزىمەن  ھەم سىزنىمۇ ئۇنۇتمايمەن  خەير خوش مەن كەتمىسەم بولمايدۇ دىدى  .  مەنمۇ ئۇنىڭ بىلەن خوشلىشىپ ئۆيگە قايتىش ئۈچۈن تاكسىغا چىقىپ ماڭدىم .لېكىن خىيالىم يەنىلا شۇ كىشىدە ئىدى .
شۇنداق ھېس قىلىپ قالدىم . مەسجىدتە جۈمە نامىزىنىڭ پەرىزىنى ئۇقۇپ بۇلا بولماي تۇرساق ، خۇدايىم بېرە . يوقلاپ قۇياڭلا دەپ سۈننەت نامازنىمۇ جايىدا ئۇقۇغىلى قويمايدىغان شۇ ئاياللار ،ئانام كىسەل ئاتام كىسەل دەپ يۇز قىزارتىپ ئولتۇرىدىقان شۇ ئەرلەر ،ئۇغۇللارمۇ شۇنچىلىك كاللىسىنى ئىشلەتمەي ئادىمىيلىك ۋىجدان ،غۇرۇرىنى بىر ياققا قايرىپ قۇيۇپ تۈۋەنچىلىك بىلەن نان تىپىپ يىيىشنى تاللىشى ئەقىلگە مۇۋاپىقمۇ ؟
ئاز تاپسىمۇ  ھالاللىق بىلەن ئەجىر -مىھنىەت قىلىپ نان تىپىپ يىگەن شۇ  كىشىنى شۇنچە  كىشىلەر ھۆرمەتلەپ ،قانچىلىغان كىشى ھېسداشلىق قىلىپ نەرسە -كىرەكلىرىنى سىتىۋالدىيۇ،ئۇ تىلەمچىلەر نىمىگە ئېرىشىۋاتىدۇ.سەت گەپ ،مەينەت يەر ، ھارام رىزىقلارغا ئىگە بۇلىۋاتىدۇ  شۇ.ئۇلار مۇشۇ قىلمىشى ئۈچۈن قىيامەت كۈنى نىمە كۈنلەرنى كۈرەر(ھاجەتمەن ئەمەسلەرنى دىمەكچى).
ساق بولسىمۇ تىلەمچىلىك قىلىدىغان ،مېيىپ بولسىمۇ تىجارەت قىلىشنى   توغرا تىپىش بىزگە شۇنى ئىسپاتلاپ بەردىكى ئادەمنىڭ ۋىجدان غۇرۇرىنى ساقلىشى ،بەش ئەزاسىنىڭ ساق بۇلۇشىغا مۇناسىۋەتلىك ئەمەس بەلكى روھىيتىنىڭ ساغلاملىقى ئىكەن. مۇشۇ مۇناسۋەت بىلەن روھىيتى ساغلام بارلىق قىرىنداشلىرىمنىڭ بۇندىن كىيىن تېخمۇ جۇلالىنىپ ھايات يوللىرىنى نۇرلاندۇرشىنى .خۇدايىم بېرە گە كۇنۇپ كەتكەن شۇ كىشىلىرىمىزنىمۇ يىنىغا تارتىپ بىرە تىجارەت قىلىشقا (ھەممىسىنى قەغەزچى بۇلۇڭ دىگىنىم يوق جۇمۇ)رىغبەتلەندۇرشىنى ئۈمىد قىلىمەن.
مەن ئىشىنىمەن .بىز ھارامدىن ساقلىنىپ .چىن ئىخلاسىمىز بىلەن ئىگىمىز ،ياراتقۇچىمىز،مىھرىبان ،شەپقەتلىك ﷲ تىن يۇرتىمىزنىڭ  بۇندىن كىيىن تىخىمۇ گۈزەل ،باياشات بۇلۇشىنى تىلەيدىغانلا بولساق ئنشائاللا ئاشۇنداق تىلەمچىلەرنىڭ قەلبىگە  ﷲ ئىنساپ ۋە ئىمان ئاتا قىلغۇسى..........
   مەن ئويلايمەن ...تىلەمچىلەر يوق دۇنيا ئادەمگە تېخىمۇ گۈزەل تۇيۇلغۇسى ..................

ئاخىرىدا بارلىق مىللەت سۈيەر قىرىنداشلىرىمغا ۋە شۇ قەغەز ساتقۇچى سودىگەرگە تۆۋەندىكى شىئېرىمنى سۇندۇم .ياقتۇرۇشۇڭلارنى ئۈمىد قىلىمەن .رەھمەت....

                                               بۇرچىمىزنى ھەر دائىم بىلىپ ياشايمىز.
                                               غۇرۇر ھەم ۋىجداننى ساتماي ياشايمىز.
                                               كۆڭلىمىز پاك بىزنىڭ قايماي ياشايمىز.
