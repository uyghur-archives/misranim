                                                                    بولمىسۇن


                                 
                                                         ئابدىلتىپ ئابدىخالىق
                     
                                                   يۈرەكلىرىڭ لەختە قان بولسا
                                                  كۆزلىرىڭ ئەسلا ياشقا تولمىسۇن.
                                               ئاھ ئۇرۇپ كەتمىگىن ھەسرەتلەرگە
                                              ياشلىقىڭ توزغاقتەك توزغان بولمىسۇن.
                    
                                              يۈرەكلىرىڭ قان-زەرداپ بولسا
                                                ھاراق تاماكا داۋا بولمىسۇن.
                                        بىردەملىك ھەۋەسنىڭ كەينىگە كىرىپ
                                       ئۆمۈر بېغېڭ مەڭگۈلۈك خازان بولمىسۇن.
                 
                                          ياشسەن ئنشائاللا ياشايسەن ئۇزاق
                                       تەر تۆكمىگەن نىئمەت سىنىڭ بولمىسۇن.
                                          ھاياتلىق يولىدا سىناقلار ھەم كۆپ
                                    ساڭا شەيتان ۋەسۋەسىسى يىقىن بولمىسۇن.
               
                                            مەغلۇب بۇلۇپ قالساڭ ناۋادا
                                          ئۈمىد سەندىن يىراق بولمىسۇن.
                                      ۋە ياكى شان شەرەپ نۇر چاچسا ساڭا
                                     مەغرۇرلۇق ساڭا ھەرگىز يىقىن بولمىسۇن

               
                                                  مۇھەببەتلىك كۈنلىرىڭ بۇلار
                                           يۈرەكلىرىڭ ھاياجاندا تىپچەكلەپ سۇقار.
                                           سۆي ھەم كۆي ئۇنىڭغا جىنىڭ تەسەددۇق،
                                           ۋە لېكىن بىردىن ئۆزگە يارىڭ بولمىسۇن.
               
                                           گۈل ئچىدە گۈل بار،تاللاپ ئۈزگىن سەن.
                                        ئۈزگەن گۈلنى سۇلدۇرماي قەدىرلىگىن سەن .      
                                            لېكىن گۈلنىڭ قەدرىنى بىلمەي ئۈزۈپ،
                                            سەن خۇشال گۈل لېكىن خار بولمىسۇن.
                  
                                       ھەر قەدەمنى ئويلاپ باسقىن تەمكىنلىك بىلەن،
                                         قەدەملىرىڭ خاتا بېسىلىپ زارىڭ بولمىسۇن.
                                            سەن ئىنسان ھەر دائىم ئۈزۈڭنى بىلگىن،
                                       ئەيىش-ئىشرەت مەئىشەتكە رايىڭ بولمىسۇن.

            
                                      ھەر دائىم كۆك ئاسماندەك قەلبىڭ بولسۇن پاك،
                                     كۈڭۈلنىڭ كەينىگە كىرىپ يۈرەگىڭنى قىلما چاك.
                                        كىرپىكىڭنى ئىگىپ بولماس ھەر دائىما لاك.
                                 ياشلىقىڭنىڭ ھەممىسى پەرداز بىلەن گۈزەل بولمىسۇن.
                
                                    ياشلىقىڭنىڭ قىممىتى بىلىم بىلەن نۇر چاچار ،
                                   خۇشپۇراق چاچمىغان گۈلدىن كىپىنەكمۇ قاچار.
                                 ئۈگەنگىن ھەم ئۈگەتكىن ياشلىقىڭ ساڭا بەرداش،
                                دىل قاراڭغۇلۇقى يامان قەلىب كۈزۈڭ كور بولمىسۇن.

            


                                 مەنبە      ؛      ئىجادىيتىم ......سەتئسىل
قوچقارنىڭ كەينىگە يانغىنى قورققىنى ئەمەس.
