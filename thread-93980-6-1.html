xahrohنىڭ بىرقانچە تىمىلارغا يازغان بىرقىسىم ئىنكاسلىرىنى يىغىپ قويدۇم كۆرۈپ بىقىڭلار.

قازاقلار ئۆزلىرىنى تىلەمچىسى يوق مىللەت دەپ پەخىرلىنىدىكەن ،
نىسپەتلەشتۈرۈپ كۆرسە ،دۇنيا بويىچە تىلەمچى ئەڭ جىق مىللەت بىز ئۇيغۇرلارمىكىن
دەيمەن ،،،،

شۇنى دەيمەن ،ئىسىت مېنىڭ ئۇيغۇر بوپ قالغىنىم ،،نىمىشقىمۇ ئۇيغۇر بوپ قالغاندىمەن ،،،
شاھ ئەۋلادىمۇ ماتىتەيگە بوزەك بۇلامدىكەن ؟شاھ ئەۋلادىمۇ چىڭگىزخانغا بوزەك بۇلامدىكەن ؟
شاھ ئەۋلادىمۇ ئۇرخۇن دەريا بويلىرىدا قىرغىزلاردىن قورقۇپ قېچىپ يۈرەمدىكەن ؟شاھ ئەۋلادىمۇ
زاڭزۇلاردىن ئاپاق خوجا دىگەن نامنى تىلەپ مۇڭغۇللارنى باشلاپ كىرىپ ئۆز قىېرىنداشلىرىنى
قىرغۇزامدىكەن ؟؟؟؟؟يوقالسۇن بۇنداق قۇرۇق ماختاشلار ،،،،ئىشقىلىپ يەنا بىر تۇغۇلۇش پۇرسىتى
بولسا ئىدى ،ئاللاھتىن ھەرگىز ئۇيغۇر قىلىپ ياراتماسلىقنى تىلەر ئىدىم ،،،،،

سۆيۈملۈك شىۋىتلار ،،،،،،ئۆزىنىڭ دىنىنى  باشقىلارغا تاڭماقچى بولغانلىقى ئۈچۈن
بىزنىڭ تىللىشىمىزغا ئۇچرىدى ،لېكىن شۇ ۋاقىتتىكى مەكتەپ يۈزى كۆرمىگەن قارا قورساق
خەلقىمىز ئۈچۈن يېڭلىقلارنى ئېلىپ كەلدى ،دوختۇرخانا قۇرۇپ كېسەللەرنى داۋالاپ ساقايتى ،فوتو سۈرەت ۋە سىنئالغۇلار بىلەن شۇ ۋاقىتتىكى ئۇيغۇرلار تۇرمىشىدىن خاتىرە قالدۇرۇپ بىزگە يەتكۈزدى،ئەجىبا قەشقەردىكى سوپا ئىشانلار ،موللاملارنىڭ قولىدىن مۇشۇنداق ئىش كېلەمتى ؟

خېتى ئەجەپ چىرايلىقكەن ،،،،بۇ ئۇيغۇر يېڭى يېزىقىمۇ قانداق ؟

مۇكاپاتلىق پائالىيەتكە ئاتاپ يازغانسىز ،،،،ئىشەنچىم كاملىكى، بولمىدى دىگەندىمۇ
بىردانە توك چەينىكىگە ئېرىشكۈدەكسىز ،،،،،

مۇسۇلمان دەپ يەتكۈچە ،جار سالىسەن  مىللىتىم ،
مۇسۇلمانغا يات ئىشلار سەندە كۆپتۇر مىللىتىم ،
يانچۇقىنى كولايسەن ،رايون ئاتلاپ ھېيىقماي ،
ياغلىق قاپاق ئاقارماس ،قانداق قىلاي مىللىتىم !

سەبىيلەرنى ساتىدۇ ،ئاتا-ئانا ۋەيرانە ،
ئۇيغۇر دىگەن مۇشۇنداق ،دەپ قالىدۇ كۆرگەنلەر ،
ماختاشلارنى سېغىندىم ،تىللاشلاردىن زېرىكتىم ، ،
ئەيىپلەرگە باي شۇنچە ھەيرانىڭمەن ئۇيغۇرۇم ،

!باشقىلار بىزنى ئوغرى يانچۇقچى دىسە بىزنىڭ بېرىدىغان جاۋابىمىز:«ھەممىنى بىر
تاياقتا ھەيدەيدۇيەي ...» دەپ كېتىمىز ،،،،مېنىڭچە خەنزۇلارنىڭ بىزنى شۇنداق
دىگىنى توغرا ،،،ئۇلارنىڭ كۆرگىنى شۇ ئىچكىرىدە يانچۇقچىلىق قىلىۋاتقان ،
ئاق سېتىۋاتقان ئۇيغۇرلار بولغاندىكىن شۇنداق دېمەي نېمە دەيدۇ ،
بىز جېنىمىزدا بىر تەرەپلىمىلىك بىلەن ئاتۇشلۇقلارنى پىخسىق،غۇجىلىلىقلارنى
ئۇنداق ،تۇرپانلىقلارنى مۇنداق دەيمىز ،لېكىن ئەمەلىيەتتە ھەممىسى
ئۇنداقمۇ ئەمەسقۇ ،شۇنڭىغا
ئوخشاش قائىدە ،،،،،،ھەيرانمەن ،جۇڭگۇدا نۇپۇسى بىزدىن جىق بولغان
ئاز سانلىق مىللەتلەردىن زاڭزۇ ،مۇڭغۇللار بار ،،،نىمىشقا ئۇلارنى
ئوغرى يانچۇقچى دىمەيدۇ،نىمىشقا بىزدە ئۆز پۇتىغا پالتا چاپىدىغان
ھارامتاماق مۇناپىقلار بۇنچە كۆپ چىقىدۇ ؟؟؟
ئىچكىىرىدە ئاشۇنداق ئوغۇرلۇق قىلغانلارنى ئۇرۇرپتۇ چاناپتۇ دىسە
