ھاۋا كۈندىن كۈنگە سوۋۇپ بىزنىڭ تاكسى تىجارىتىمىز خېلى ياخشى بولۇپ كېتۋاتاتتى، بۈگۈنمۇ بامدات نامىزىمنى ئوقۇپ بولۇپلا كۈندىكىدەك كوچىغا ئاتلاندىم،ماشىنىنى قىزىتقاچ ئاستا كېتۋېتىپ قارىسام ئاتەش مېھمانخانىسىنىڭ ئالدىدا 3-4چە ئادەم قولىنى كۆتۈرۈپ تاكسى توسۋاتاتتى، تېخى ئەمدىلا يۇرۇق چۈشۋاتقان ۋاقىت بولغاچقا ماشىنىنى ئۇلارنىڭ يېنىدا توختۇتۇپ ئاندىن ئۇلارنىڭ ئىككى جۈپ ئويغۇر قىز-يىگىتى ئىكەنلىكىنى بىلەلىدىم.

-بىزنى xxئوتتۇرا مەكتىپىگە ئاپىرىپ قويامسىز؟.-دىدى يىگىتلەردىن بىرى.  
-40 يۇەن بەرسەڭلار ئاپىرىپ قوياي.-دىدىم.ئۇ مەكتەپنىڭ ئارلىقى20كىلومىتىردىن ئاشاتتى.   
-يېنىمىزدا 30يۈەنلا پۇل قاپتۇ ئاكا ياردەم قىلىپ ئاپىرىپ قويسىڭىز بوپتىكەن،دەرىسكە ئۈلگۈرۈپ بارمىساق بولمايتتى.قاراپ باقسام ئۇنىڭ سۆزلىرى راستەك قىلاتتى.   
-چىقىڭلار ئەمسە ئاپىرىپ قوياي.-دىدىم مەن ئانچە خوشياقماي.ئۇلار ماشىنىغا چىقىشى بىلەنلا ماشىنىنى قوزغاتقاچ سورۇدۇم؛   
-سىلەر ئوقۇغۇچىما؟  
-ھەئە،تۇلۇقسىز 3-يىللىقتا ئوقۇيمىز.  
-بۇ ئاتماس تاڭدا بۇ يەردە نىمە قىلىسىلەر؟   
-ئاخشام بىر دوستىمىزنىڭ تۇغۇلغان كۈنىگە چىققان....-دىدى بىرى سەل دۇدۇقلاپ.شۇ چاغدا ئارسىدىكى بىر قىزنىڭ تېلىفۇنى سايراپ گەپنى ئۈزۈپ قويدى.  
-ھە مانا ماشىنىدا كېتۋاتىمىز،مۇئەللىم سوراپ قالسا مېنى ئاخشام ئۆيىدە بىر ئىشى چىقىپ قېلىپ ئۆيىگە كەتكەنتى دەپ قويۇپ تۇر،ھە راست ئالىيەنى خوشنا مەھەللىدىكى ئاچىسىنىڭ ئۆيىگە كەتتى دىگىن،بىزمۇ 15مىنوتقىچە بېرىپ بولىمىز،خوش. '  
توۋۋا بۇلار مەكتەپتە مۇشۇنداق ئوقۇمدىكىنە؟،ئەينەكتىن بۇ قىزلارغا سەپ سېلىپ قاراشقا باشلىدىم،بۇلار ئاغزىدىن تېخى ئانىسىنىڭ سۈتى كەتمىگەن ئەمدىلا 15-14ياشلارنىڭ قارىسىنى ئالغان كىچىك بالىلار ئىدى.ھەيرانلىقتا يەنە سوراشقا باشلىدىم؛
-سىلەر مۇشۇنداق ئوينايدىغان پۇللارنى ئاتا-ئاناڭلار بېرەمدا؟ ,
-ۋاي ياق ئاكا،ئۇلاردىن ئۇنچىلىك پۇل چىقاتتىمۇ،ئۆزىمىز ئىشلەپ تاپىمىز.- دىدى بايىقى يىگىت. 
-سىلەر مەكتەپتە ئوقۇپ يۈرۈپ قانداق ئىشلەيسىلەر؟  
-بىز ھازىر ياتاقلىق مەكتەپتە بولغاچقا ئۆيگە ھەپتىدە بىرلا قېتىم بارىمىز،ھەپتىنىڭ بىرىنچى كۈنى مەكتەپكە كېلىپلا2-كۈنى دەرىسكە كىرمەي ئىشلەيمىز،بىر كۈن ئىشلىسەكلا150يۈەن تاپالايمىز،مۇئەللىملەر سورىسا ئۆيدە ئىش چىقىپ قالدى دەيمىز،شۇنداق بىر نەچچە كۈن ئىشلىسەكلا 600-700يۇەن تېپىپ مۇشۇنداق بىر ئاخشام مابەتلىرىمىز بىلەن ئويناپ كېلەلەيمىز.-دىدى يېنىدىكى قىزلارغا قاراپ كۈلۈپ.  
-مەكتەپ بۇنىڭغا ھېچقانداق گەپ قىلمامدا؟  
-ئوغۇرلۇقچە چىقىمىز،بۇرۇن ئانچە دىققەت قىلمايتتى،لېكىن ئالدىنقى ئايدا تۇلۇقسىز 2-يىللىقتىكى 2قىز سىرتقا چىقىپ ھاراق ئىچىپ مەس بولۇپ كىرىپ مەكتەپ ياتىقىنى بىر ئاخشام مالىماتاڭ قىلغاچقا ھازىر بەك چىڭ تۇتىۋاتىدۇ،لېكىن يالغان ئېيتىپ چىقىلىۋىرىمىز.....   
ئۇلارنى مەكتەپ دەرۋازىسى ئالدىغا چۈشۈرۈپ قويۇپ كەينىمگە ياندىم،كاللام ئاجايىپ بىر ھالەتكە كېلىپ قالغان ئىدى.  
-ئاھ ،كەلگۈسىمىزنىڭ ئىگىلىرى بولغان بالىلار، ئاتا-ئاناڭلارنىڭ سىلەردىن كۈتكەن ئۈمۈدى شۇمىدى؟ خەلقىمىزنىڭ كۈتكەن ئۈمۈدى شۇمىتى؟مۇئەللىملەرنىڭ..........؟   
تېخى ئەمدىلا تۇلۇقسىز ئوقۇۋاتقان بالىلىرىمىز،تېخى قىزلىرىمىز ھاراق ئىچىپ،مېھمانخانىلاردا يېتىپ يۈرسە،ئاتا-ئانالىرىمىز ھېچنىمىدىن بىخەۋەر باللىرىمىز ئوقۇۋاتىدۇ دەپ ئولتۇرىۋەرسە، بۇ مىللەت،بۇ ئەۋلاتلارنىڭ كەلگۈسى نىمە بولۇپ كىتەر؟ ئابدۇكېرىمكا،ئەجەبا بۇنىڭغا ھېچقانداق ئامال يوقمۇ؟

ئەسسىلامۇ ئەلەيكۇم   ھۆرمەتلىك قىرىندىشىم ناھايتى ئەھمىيەتلىك تىما يوللاپسىز بۇ مەسىلە ئىنتايىن ئىغىر ۋەھىمە پەيدا قىلىدىغان بىر خىل ۋىرۇس شەكىلدىكى كىسەللىك بولۇپ ھەر بىر ئەقىل ئىگىسنى ئويلاندۇرىدۇ
بۇنىڭدا بىرىنجىدىن  جەمئىيتىمىزدىكى بىرقىسىم سەمىيەتسىز ، ۋىجدانسىز ، پەسكەش ئوقۇتقۇچىلارنىڭ ئادەم تەربىيلىشىدە ( بىر قىسىم ۋىجدانى ئويغاق ئوقۇتقۇچىللىرىمىزغا تىگىپ كەتمىسۇن ) ئىككىنچىدىن  نىجىس  ئەخلاقسىز ، بىلىمسىز  ئاتا - ئانىلارنىڭ ئىغىر بەكمۇ ئىغىر مەسئۇلىيتى
ئوقۇتقۇچىلارنىڭ ئوقۇغۇچىلار ئالدىدىكى مەسئۇلىيتى بەكمۇ ئىغىر بولسىمۇ بۇنى سەزمەستىن ساپا مائارىپىغا چىتىۋىلىپ ۋاقىت ئۆتكۈزۈپ دەرىس ئۆتۈش ، ئۆزىنىڭ ئوقۇغۇچىللىرى ئالدىدا ئوقۇتقۇچىلىق سالاھىيتىنى ھەم ئىناۋىتىنى ساقلىماي ھەزلەك كىيىنىش ، پاراڭ سىلىش ، كالتە كۇسار نوگاي شىم كىيىملەرنى كىيىپ ئۆزىنىڭ ئوقۇغۇچىللىرى ھەتتا ئۆزىنىڭ قىزىدەك ، ئىنىسىدەك ئوقۇغۇچىلار ئالدىدا ھاياسىز ، ئەخلەقسىز ئىشلارنى قىلىش ، قىسقىسى كۆرسىمۇ كۆرمەسكە سىلىش ، جۈپلىشىپ قالسىمۇ مەنمۇ مۇشۇنداق ئوقۇتقۇچىغۇ دەپ تەربىيە ئىشلىمەسلىكتىن بولماقتا  مەن ئوقۇتقۇچىلارنى بەكمۇ ھۆرمەتلەيمەن بىراق بىر قىتىم ئوقۇتقۇچىلار ھەققىدە تىما يوللىسام ھەممىسى   بىز ئوقۇغۇچىلارنى چىڭ باشقۇرساق دادىمىز تىللايدۇ  ئانىمىز مائاشىمىزنى تۇتۇۋالىدۇ دىگەندەك  باھانە سۆزلەر بىلەن مۇنازىرە قىلدى ............. دىمەك ئوقۇتقۇچىلارنىڭ مەسئۇلىيتى بەكمۇ ئىغىر
ئەمدى  ئاتا - ئانىلارچۇ .................... ھەي ئاتا  - ئانىلار  قىزلىرىڭىز ۋەيران بولماقتا پەرزەنتىڭىز ۋەيران بولماقتا   نېمە كارىڭىز دەمسىز ؟ سىز بىر ئانىمۇ بۈگۈن ئاخشام شۇ قاتار چايغا بارماي پەرزەنتىڭىز بىلەن 10 مىنۇت پىكىر ئالماشتۇرۇپ ئەخلاق ئىسلام دىنى ياكى بىر دىن ياكى قۇتادغۇبىلكتىن بىر پارچە سۆز مىسال ئىلىپ تەربىيە قىلغان بولسىڭىز  بەلكىم قىزىڭىزنىڭ قىزلىق ئىپپىتى جايىدا تۇرغان بولار ئىدى
ھەم ئەخلاقسىز تەربىيسىز بىر قىز  جەمىيەتتە ئازلىغان بولاتتى  شەرمى ھايا ئازراق بولسىمۇ ئازلىغان بولاتتى  
ھەي دادا  .......... سىز بۈگۈن شۇ باشلىقىڭىزنى خوشامەتكە مېھمانغا چاقىرمىغان بولسىڭىز  ياكى  ئاشنىڭىز چاقىرغان جايغا بارمىغان بولسىڭىز ئوغلىڭىز سىزنى دورىمىغان شۇ ئىپلاس پەسكەش ئىشلارنى قىلمىغان بولاتتى  دادىلىق ھەيۋىڭىز بىلەن پەرزەنتىڭىزنى ئەخلاقلىق ، ئىنساپ دىيانەتلىك ئىمانلىق  ، ئەركەكلىك مىجەز سىڭڭەن بىر پەرزەنىت تەربىيلىگەن بولاتتىڭىز  ھازىر ئوغلىڭىز  ....................... قىزىڭىز  ............. ھەي    ئەمدى ئابدۇكېرىم ئەپەندىممۇ  ئىتوت ئارقىلىق قامچىلاپمۇ باقار بىراق  گەپ  بۇ يەردە ئەمەس
