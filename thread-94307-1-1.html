سېغىنىشنىڭ بۇنچىلىك تاتلىق ئىكەنلىكىنى بالدۇرراق بىلگەن بولسام ، ھاياتنىڭ بۇنچىلىك گۈزەل ئىكەنلىكىنى بالدۇرراق ھىس قىلغان بولسام ،بۇنچىلىك ئۆزۈمنى بىھۇدە قىينىمىغان ،بىھۇدە كۆز ياش

تۆكمىگەن بۇلار ئىدىم .



تاڭ سەھەر ،قارنىڭ غىرىچ -غىرىچ ئاۋازىغا ئەگىشىپ يىنىك مېڭىۋاتىمەن ..ئەتراپ شۇنچىلىك سۈزۈك ئىدىكى ھەممە ماڭا شۇنچىلىك گۈزەل تۇيۇلماقتا..قەدىمىم بارغانچە تىزلىشىپ سىەن تۇرغان شۇ ئۇپۇققا قاراپ كىتىۋاتىمەن ،سەن گۇيا قەلبىمدىكى سۇزۇك بىر تامچە شەبنەمگە ئوخشاش مىنىڭ قاغجىراپ كەتكەن قەلبىمگە ئىللىقلىق ۋە نەمخۇش ئاتا قىلۋاتىسەن بارا- بارا سىڭىپ كىتۋاتقىنىڭنى سەزگىنىمدە پۈتۈن جىسمىم ئاللىقانداقتۇر بىرخىل سىزىم ئىچىدە تاتلىققىنا تولغىنىدۇ.مەن شۇتاپ سىنىڭ كۆز ئالدىمدا پەيدا بۇلىشىڭغا شۇنچىلىك تەقەززار بولدۇمكى نازۇك ھىس تۇيغۇلىرىمنىمۇ قىلچىلىك باشقۇرۇشقا ئامالسىزمەن .قەلبىم خۇددى گۇڭگا لىرىكىدەك سىنىڭ يۈرىكىڭگە بۈسۈپ كىرمەكتە. سەن بولساڭ خىيالىمدىكى ئاشۇ ئارمانسىز دەقىقىلەرنىڭ تىزراق ۋىسال تېپىشى ئۈچۈن ھەر كۈنى ھەر سائەت ھەر دەقىقە قۇلۇغۇمغا ئاستاغىنە شىۋىرلاۋاتىسەن ... دەل شۇتاپ پۈتۈن ۋۇجۇدۇم ئوتتەك قىززىپ بىرخىل ھارارەت مىنى قىينايدۇ ...قىينايدۇ...ئەشۇ بىغەم ئۇخلاۋاتقان بارلىق مەۋجۇداتلارمۇ بىزنىڭ شىرىن سۆيگۈمىزگە ئەسرا بۇلۇپ ئۇيقۇلىرى يىشىلىپ كەتسىمۇ يەنىلا بىزگە دەخلى قىلماستىن ئاستاغىنە كۆزلىرىنى يۇمۇپ بىزگە بەخىت تىلىمەكتە...جېنىم سېنى سېغىندىم ...كۆزلىرىمدە لىققىدە ياش،،،يۈرىگىمدە ساڭا بولغان سېغىنىشنىڭ چېقىنلىرى ۋالىلدىماقتا.بىلىپ تۇرىمەن ئىككىلىمىز دەل شۇ خىل ھىسيات ئىچىدە تۇرۇپتىمىز....لەۋلىرىمنى شەبنەمنىڭ سۈيىدە يۇيۇپ ،قەلبىمنى ۋاپادارلىق تەختىدە بىخلاندۇرۇپ،يۈرىگىمنى بىرپارچە ئاپئاق گۈل قىلىپ ساڭا سۇنغۇم بار......



نە ھاجەت؟ئۇشبۇ دەمدە ئۇنىڭ ئالىدىغىنى سەن سەرخۇش ئىبادىتىڭدە سېغىنىپ كەلگەن قۇپقۇرۇق يوقلۇق تۇرسا...-ــــــــــــ بىلگىنكى شۇ ھاجەتلىك بولمىغان بولسىدى بۇ ئالەم تېخىمۇ گۈزەل بولغان ،قىلچىلىكمۇ ئۇرۇش تالاش بولمىغان بولار ئىدى ....شۇ ھاجەت سېنى مېنى ھەمىگە تەقەززار قىلماقتا...ئەمما بىلسەم ھەممە نەرسە تەل بۇلۇپ كەتكەن كۈندىنمۇ زىرىكىدۇ كىشى شۇ ھاجەتلىكمۇ شۇنچىلىك خۇشال ھىسلارنى ئەپ كىلۋاتىدۇ.




يىراقتىن ساڭا سىنىڭ سايەڭگە ئەگىشىپ مېڭىۋاتقىنىمدا مەن شۇنچىلىك خۇشال بۇلىمەن ....بۇ ھاجەت مىنى بەخىتلىك قىلۋاتىدۇ..مەن ئىرادىدىن پۈتۈلگەن سۇلماس قانىتىمنى كەڭرى يېيىپ سامادا ئەركىن ئۇچۇۋاتىمەن ...بۇ ھەم مىنىڭ ئارزۇيۇم ئىدى .ئەركىن نەپەس ئالغان شۇ سامادىن مەن شۇنچىلىك رازى ،،،
گەرچە داۋاملىق مۇشۇنداق ئۇچۇش نىسىپ بولمىسىمۇ؟ گەرچە مەڭگۈ داۋام قىلىشقا غەيرىتىم كەمچىل بولسىمۇ مەن رازى ....چۈنكى ئارمانلىرىم مەندىن ئۈمۈتسىزلەنگەن مۇشۇ دەقىقىدە سەن ماڭا قايتىدىن غەيرەت ۋە خۇشاللىق ئاتا قىلدىڭ.شۇتاپ خاتىرەم قۇپقۇرۇق ھەممىنى قايتىدىن باشلاشقا تىرىشىۋاتىمەن ،ھېچ نەرسە ھېچ نەرسە ماڭا تۇسالغۇ بۇلالمايدۇ.
سەن ئويغاتقان ئاشۇ تاتلىق ھىسلار ئاياللىق لاتاپىتىمگە قايتىدىن تۈس بىرىپ مىنى تېخىمۇ جەزبىدار قىلىۋەتتى ...كۆر جېنىم ،مېنى تىزراق كۆرگىن ...يۈرىگىمدە بىر سەن ئۈچۈن يېزىلىۋاتقان

پۈتمەس تۈگىمەس داستانلارنى ئۇقۇپ باققىن ... يۈرىگىمدىكى ئەشۇ شىرىن ئازاپلارغا تەڭ جور بولغىنىڭدىن چۇقۇم رازى بۇلىسەن ....


قۇلاق تۈۋىمدە شۇنداق مۇڭ بىر ناخشا ،ئۇ مۇھەببەت ناخشىسى،سەن شۇ ناخشىلىرىڭغا مىنىمۇ ھەم قېتىۋالدىڭ.مەن سۆيگۈنىڭ قۇلى بۇلۇپ ياشاۋاتقىنىمدا سەن ئىككىلەنمەستىن مېنى بۇ ئاززاپتىن قۇتۇلدۇرۋالدىڭ.ھەتتا ئۆز سويگۈڭنىڭ مىھىرلىك بەرگخلىرىدىن تىلتۇمار ئېسىپ بوينۇمغا تاقاپ قۇيىۋاتىسەن مەن بەكمۇ بەخىتلىك شۇ تاپ،سىنىڭ شۇ سۆزلىرىڭ ئىسىمدە ...مىنى كۈتۈپ تۇرغانمىدىڭ جانىم ....ھەر ھەقىقەت مەن بىر سىنىلا كۈتۈپ تۇرغان ئوخشايمەن .كۈلپەت بىلەن ياشلانغان كۆزلىرىم كۈلگىنىدە قانچىلىك گۈزەل بوپ كەتكىنىنى سىزىۋاتىسەن چۇقۇم .ھەيرانمەن سەن مىنى ھېچبىر كۈچىمەيلا ئۆزۈڭگە يانداش قىلىۋالدىڭ.
يەر زىمىننى چاڭ تۇزان قاپلاپ،مۇشكۈل تاغ داۋانلار مىىنىڭ يۇلۇمنى تۇسۇپ تۇرسىمۇ مەن شۇنچىلىك كۈچلۈك ھازىر مەندىكى بۇ كۈچ غەيرەتنى سەن ئاتا قىلدىڭ .
چاڭقىغان دىللىرىمغا زەمزەم سۈيىدەك ئارامبەخىش ھۇزۇر ئاتا قىلۋاتىسەن ،بۇ مىسكىن خەستە كۆڭلۈمنى شات خۇراملىققا چومدۈرىۋاتىسەن ...مەن بەكمۇ خۇشال شۇتاپ. كۆزلىرىمدىن چاچرىغان سۇيگۈ ۋولقانلىرىنىڭ يۈرىگىڭنى كۆيدۈرۈپ كۈل قىلۋەتكۈسىلا بار .بوينۇمغا سەن ئېسىپ قويغان مۇھەببەت تىلتۇمارى مىنى بالا قازادىن يامان كۆزلەردىن ساقلىغۇسى ....كۆزۈڭنى يۇم جانىم بىرلىكتە چۈشلەر پەرۋازىدە سەيلە قىلايلى!



چۈشلەردىن چىقىۋاتىمەن .چۈنكى چۈشلىرىم مانا رىياللىققا ئايلاندى .يىراقتىكى ئىزگۈ چاقىرىقلارغا ئەگىشىپ مىنى ھەممە ھەممە خۇشال قىلماقتا .دىمىقىمغا ئۇرۇلغان شۇ خۇشپۈراق ھەر كۈنى دەرىزەمدىن سەن ئەۋەتكەن شۇ خۇش پۇراق .شۇ ھىدلارغا ئەگىشىپ مىنىڭ كەلگىنىمگە ئىشەنگىن .بۇخىل تۇيغۇ بەك گۈزەل بەكمۇ ........................
بۈگۇن نىمىشقىدۇر كۆڭلۈم شۇنچىلىك يېرىم ، نىمىلەرنىدۇ ئويلايمەن. ئەمما زادى ئىزلىگەن شۇ

نەرسەمنى تاپالمايمەن .ماڭا نىمە بولغاندۇ ؟ھەتتا ئۇزۈممۇ بىلمەيمەن ...... كۆزلىرىم يىغلايدۇ...يۈرگۈم

ئۇندىنمۇ ھەم.... كىتىۋاتىمەن.. تۇلىمۇ ئىزگۈ خىيال ئىچىدە ئۈنسىز ......شۇندا سەن كەلدىڭ ئالدىمغا
تۇيۇقسىز خۇددى يەردىن ئۈنگەندەكلا ،،،<<ساڭا نىمە بولدى پەرىشتەم يىغلاپسەنغۇ؟؟؟؟ >>
<< ياق مەن نەدە يىغلاپتىمەن، كۇزۇمگە قۇم كىرىپ كەتتى شۇ>> ئاستاغىنە كۆزۈمنى ئۇۋىلىدىمغۇ دەيمەن ھەقاچان ،<<بوپتۇ يىغلىۋال كۆزۇم ، سەن كۆزۈم ئەمەسمۇ شۇنداق يىغلاڭغۇ...ئۇششۇق....بوپتۇ كوپرەك يىغلىۋال >> سىنىڭ شۇنداق دىگىنىڭچە مىنى باغرىڭغاباسقىنىڭ ئىسىمدە .......ئەسلى مەن چۇش كۆرگەن ئىكەنمەنغۇ؟؟؟؟ كۆزلىرىممۇ نەم شۇنداق يەنىلا شۇپېتى ....مىنى سەن يىغلاڭغۇ دەيسەن بەزىدە ئۇششۇق ،بەزىدە كۆزۈم دىسەڭ،
بەزىدە گۈلۈم ...ئەمما مەن يەنىلا سىنىڭ مىنى تاتلىقىم دەپ ئەركىلىتىشىڭنى بەكمۇ ياقتۇراتتىم ....



.سىنى شۇنچىلىك بەك سېغىندىم...دۇنيادا كۈتۈشدىنمۇ ئارتۇق ئېغىر ئازاپلىق كۈن بولمىسا كىرەك.
چۈنكى سېغىنىش ئىچىدىكى جۇدالىق ئادەمنى شۇنچىلىك قىينايدۇكى ،بەزىدەھەتتائۈزۈڭنىمۇ تاپالماي قالىسەن ..دۇنيادا كۈتۈشتىنمۇ ئارتۇق يەنە بەخىتمۇ بولمىسا كىرەك ...چۇنكى مەن سىنى سۈيىمەن ...چۇنكى سەن يەنىلا بىرلا مىنىڭ...... تۇيۇقسىز قەلبىم يەنە ئىللىپ قالدى...خۇددى سەن يېنىمدا باردەك.....
ئىسىڭدڭمۇ ئىككىمىز ئۇزاقتىن ئۇزاق مۇڭداشقان ئەشۇ كېچە
قاچانمۇ بۇ كۈنلەر ئاخىرلىشار، قاچانمۇ ئىككىمىز نىڭ ۋىسال پەيتلىرى يىتىپ كىلەر.
