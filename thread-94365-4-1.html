ھەقىقەتەن بەزى ئەزالىرىمىز ئالدىراپلا ، دارۋىنىزىمغا ھەم شۇنداقلا تېما ئىگىسىگە ھۇجۇمنى باشلاپتۇ. تورداشلارنىڭ دارۋىنىزىمنى قۇبۇل قىلماسلىقى تەبىئىي، لېكىن تېما ئىگىسىگە ھۇجۇمنى ئۇنداق تېز باشلىمىساق.
تورداشلارنىڭ دارۋىنىزىمنى ياقتۇرماسلىقىنىڭ سەۋەبى پەرىزىمچە جانلىقنى ئىلاھ ياراتقان نەزىرىيىسىنى دارۋىن ئىنكار قىلغان دەپ قارايدۇ. ئەمەلىيەتتە ئۇنداق ئەمەس
ئەمەلىيەتتە جانلىقنى ئىلاھ ياراتقان نەزىرىيىسىنى تۇنجى بولۇپ ئىنكار قىلغۇچى لامارك، دارۋىندىن بۇرۇن ياشىغان ئالىم. دارۋىن ئۆزىنىڭ تەدىرجى تەرەققىيات نەزىرىيىسىنى كېيىن لامارك ياراتقان نەزىرىيە بويىچە ئۇنىڭ بەزىبىر نوقتىلىرىنى قۇبۇل قىلىش، بەزى قىسىمىنى ئىنكار قىلىش، بەزى قىسىمىنى ئەلالاشتۇرۇش ئاساسىدا ئوتتۇرىغا قويغان.
لاماركنىڭ نەزىرىيىسىنىڭ يادروسى: ئېرىشمە ئېرىسيەت، يەنى ئىشلەتكەن ئەزالار تەرەققىي قىلىدۇ، ئىشلەتمىگەن ئەزالار رودىمېنتلىشىدۇ. دارۋىننىڭ نەزىرىيىسىنىڭ يادروسى بولسا ماسلاشقانلار ياشايدۇ، ماسلىشالمىغانلار ھالاك بولىدۇ.تۆۋەندىكى مىسال بويىچە ئىككى نەزىرىيەنى چۈشىنىپ باقساق:
مەسىلەن: قەدىمكى زاماندا زىراپىلەرنىڭ بوينى قىسقا ئىدى، ئۇنداقتا ھازىرقى زىراپە قانداق بارلىققا كەلدى؟
لامارك قارىشى بويىچە ئېيتقاندا: ئەينى چاغدا زىراپىلەرنىڭ ھەممىسى قىسقا بويۇن ئىدى، كېيىن تەبىئىي مۇھىت ناچارلىشىپ زىراپىلەر ئېگىز دەرەختىن يوپۇرماق ئېلىپ يېيىشكە مەجبۇرى بولدى، شۇنىڭ بىلەن زىراپىلەرنىڭ ئېگىزگە ئىنتىلىشى جەريانىدا زىراپىلەرنىڭ بوينى تەرەققىي قىلىپ ئۇزىراپ، بۇ مۇقىم ئېرىسىيەتكە ئايلىنىپ، ھازىرقى ئۇزۇن بويۇنلۇق زىراپىگە ئايلاندى دېيىلىدۇ.
دارۋىن نەزىرىيىسى بۇنىڭغا ئوخشىمايدۇ: دارۋىن نەزىرىيىسى بويىچە چۈشەندۈرۈلگەندە: قەدىمكى زاماندا ھەقىقەتەن زىراپىلەرنىڭ بوينى قىسقا ئىدى، لېكىن ئەزەلدىنلا گېنلىرىنىڭ توساتتىن ئۆزگىرىشى سەۋەبىدىن پەيدا بولغان ئۇزۇن بويۇنلۇق زىراپىلەرمۇ بار ئىدى، لېكىن زىراپەتوپىدا ئۇلارنىڭ ئىگەللەيدىغان سانى ئاز بولغاچقا ئۇلارنىڭ ئەركىن چېتىشىپ ئۇزۇن بويۇنلۇق زىراپە ئەۋلادلىرىنى قالدۇرۇش نىسبىتى تۆۋەن ئىدى. كېيىن مۇھىت ئۆزگىرىشى سەۋەبىدىن زىراپىنىڭ ئوزۇقلۇق شارائىتى ناچارلاشتى. شۇنىڭ بىلەن زىراپىلەر پەقەت ئېگىز دەرەخلەردىن يوپۇرماق ئېلىپ يېيىشكە مەجبۇرىي بولدى. شۇنىڭ بىلەن زور كۆپ قىسىمنى ئىگەللەيدىغان قىسقا بۇيۇن زىراپىلەر ھالاك بولدى، ئۇزۇن بويۇنلۇقلىرىنىڭ ئۆزگىرىشى شۇ مۇھىتقا نىسبەتەن ‹‹پايدىلىق ئۆزگىرىش››بولغاچقا ھايات قالدى. دېمەك زىراپە توپىدا ناھايتى ئاز ساننى ئىگەللەيدىغان ئۇزۇن بويۇن زىراپە قالدى،ئەسلىدىنلا زىراپە توپىغا تەۋە بولغان زىمىن كەڭرى سان ئاز بولغاچقا ئۇزۇن بويۇن زىراپىلەر يېتەرلىك ماكان ۋە بايلىق شارائىتىغا ئېرىشتى، شۇنىڭ بىلەن زىيادە كۆپىيىش ئارقىلىق قىسقا ۋاقىت ئىچىدە زىراپە تۈر توپىنىڭ سانىنى ئەسلىگە كەلتۈردى، ئۇلارنىڭ ئەجدادىدا ئۇزۇن بويۇن بولۇشنىڭ گېنى بولغاچقا بۇ گېنلار ئەۋلادىغا ئىرىسيەت قېلىپ ئەۋلادلىرىنىڭ ھەممىسى ئۇزۇن بويۇنلۇق زىراپە بولدى، ئاخىرىدا قىسقا بويۇنلۇق زىراپە مۇھىت تەرىپىدىن تولۇق ھالاك قىلىنىپ دۇنيادا پەقەت ئۇزۇن بويۇنلۇق زىراپە قالدى، شۇڭلاشقا ھازىرقى ئۇزۇن بويۇنلۇق زىراپە قالدى دېيىلىدۇ

مېنىڭچە دارۋىن ياراتقان نەزىرىيىنىڭ ئاساسى مۇشۇ شۇ، لېكىن ئادەم مايمۇندىن يارالغان دېگەن قاراشنى دارۋىننىڭ ئوتتۇرىغا قويغان-قويمىغانلىقىنى بىلمەيدىكەنمەن. مېنىڭچە دارۋىن نەزىرىيىسىگە ۋارسلىق قىلغۇچىلار كېيىن ئوتتۇرىغا قويغان بولۇشى مۇمكىن.
شۇڭا دارۋىننىڭ نەزىرىيىسى كىچىككىنە ئادەم مايمۇندىن ئۆزگەرگەن دېگەندىنلا ئىبارەت ئەمەس،كۆپ مەزمۇننى ئۆز ئىچىگە ئالغان، كەڭ دائىرىلىك نەرسە بۇ يەردە دارۋىننىڭ ھەممە نەزىرىيىسىنى تولۇق قۇبۇل قىلىڭ دېگەنلىك ئەمەس، تەدىرجى تەرەققىياتنىڭ ماھىيىتىنى پەقەت بىر چۈشىنىپ قويۇشقا بولىدۇ

يەنە بىر دەيدىغىنىم: ئاتىزىمچى دېگەن نېمە؟ ھەر قانداق بىر دىننى ئىنكار قىلىدىغان ئادەمنى ئاتىزىمچى دەيمىز، قانداقتۇر ئىسلام دىنىنى ئىنكار قىلىپ، باشقا دىنغا ئىشىنىدىغان ئادەمنى ئاتىزىمچى دېمەيمىز، دارۋىنننىڭ ئاتىزىمچى ياكى ئەمەسلىكى ھەققىدە ئېنىق قارار چىقارغىلى بولمايدۇ. چۈنكى ئىمان، ئېتقاد دېگەن ئادەمنىڭ دىلىدا بولىدۇ، ئەينى ۋاقىتتا ياشىغان ئادەمنىڭ دىلىنى مۇنبەردىكىلەرنىڭ قانداقسىگە شۇنچە ئاسان چۈشىنىۋالغانلىقىنى بىلمىدىم

ئىشقىلىپ بىز شۇنداق مۇھىت شۇنداق ئىدولوگىيەدە ئۆسۈپ چوڭ بولۇپتۇقكى دىيىشكە ئىغىزدىن چىقىرىشقا ھەتتا ئويلاشقىمۇ بولمايدۇ دىگەن گەپلەرنى پىكىرلەرنى ئاڭلالا قالساق قاتتىق قورقۇنۇش ساررسىمىگە چۈشۈمىز ۋە ئالدى كەينىنى ئويلىماي دەررۇ زەربە بىرىشكە، بىرىنجى كالتەك بىلەنلا ئۇجۇقتۇرۋىتىشكە ئۇرىنىمىز، ئۇنداقلارغا بۇرۇن يەھۇدا، كاپىر دەپ قالپاق تەييارتى، ھازىر ماسون دەپ يىڭى مودا قالپاق چىقىپتۇ، لىكىن باشقا يەرلەردە ھەر خىل ئىقىم، ھەر خىل ئىدىيە، ھەر خىل دىنلارنىڭ ۋەكىللىرى بىر يەرگە كىىلىپ سىنىڭ قانچىلىك خاتا مىنىڭ قانچىلىك توغرا مەن دەي سەن ئاڭلاپ تۇر دەپ ئەركىن ئازادە ئىلمى ، دوستانە پاراڭىشالايدىكەن. بۇ ئەنئەنە بىزنىڭ بوۋىمىز ئىمپراتور ئەكبەر شاھىنشاھتىن باشلانغان. مىنىڭچە ئۇنداق ئىلمى، تەمكىن دوستانە ئىلىپ بىرىلغان مۇھاكىمىلەر ئۇنچىلا  قورقۇنۇچلۇق قوبۇل قىلغۇسىز ئەمەس، ھىچنىمىنى بىلمەي چۈشەنمەي تۇرۇپ ئىغىزنى ، كاللىنى پىچەتلىۋەتلىك قورقۇنۇشلۇق، مەسىلەن بۇ تىمىدىكى ت بەت ئىنكاسلىدا نۇرغۇن نۇرغۇن بىلىملەر باركەن، خەلقىمىزمۇ باشقا تىللارغا خىلى پىشىۋىتىپتۇ، ئىشقىلىپ بۇ تىمىدا بۇرۇنقىدەك تۇيۇقسىز ئۇجۇقتۇرپ قالپاق كىيگۈزۈپ تون پىچىپ دوزاخ بىلەن تەھدىت سالىدىغان ئەھۋاللار كۆرۈلمىدى، ھەم تىمىغا ئىنكاسلىغا تورداشلارنىڭ قىزىقىشىمۇ كۈچلۈك، بۇ ياخشى ئەھۋال، ھەرگىزمۇ كۇفرىنى، ئاسىلىقنى كۆرگەندە قارشى تۇرماي ئەدىپىنى بەرمەي جىم تۇرۇۋالغانلىق ئەمەس. ئۇنىڭ ئۈستىگە بۇ تىمىغا كىرگەن بىرە يەھۇدا، خىرىستىيان ياكى دىنسىز بايقالمىدى،
