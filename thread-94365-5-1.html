بۇ يەردە بىر ئاكىمىزنىڭ بۇ تىمىغا مۇناسىۋەتلىكرەك ئىلىپ بارغان بىر سۆھبىتى باركەن، كۆرۈپ باقساڭلا بولىدۇ. سۆھبەتتە ئوتتۇرىغا قويۇلغىنى ئالدىنقى يىلىغۇ دەيمەن دۇنيا ئىلىم پەن ساھەسىدىكى ئۈچ چوڭ نەتىجىنىڭ بىرى دەپ ئىلان قىلىنغان قۇشلارنىڭ ئەجدادى توغرىلىق دكلىتىمىزدىن چىققان تەتقىقات ماقالىسى، بۇ بىزنىڭ شۆھرەت مۇتەللىپ ماقالىسىنى ئىلان قىلغان تەبىئەت ژورنىلىدا ئىلان قىلىنغان، ۋە تەدرىجى تەرەققىيات نەزەرىيەسىدىكى ھازىرغىچە كام قىلىۋاتقان بىر بوشلۇقنى تولدۇردى دەپ باھا بىرىلگەن، قىسقىچە مەزمۇنى بولسا بىرى ، دىنازاۋۇرلا بىز ھازىرغىچە پەەرەز قىلغاندەك نەسلى قۇرۇپ كەتكەن ئەمەس ، يەنە بىرى دىنازاۋۇرلار نەچچە 10 مىليون يىللىق مۇھىتقا ماسلىشىش ۋە تەبىئى تاللىنىشنىڭ نەتىجىسىدە ناھايىتىمۇ چوڭ بولغانتۆۋەن دەرىجىلىك سوغۇق قانلىق ئۆمۈلۈگۈچى ھايۋاندىن ھازىر بىز كۆرۈپ تۇرغان تەن قۇرۇلۇشى كىچىك، ھاۋادا ئۇچۇشقا ماسلاشقان، يۇقۇرى دەرىجىلىك قۇشلار سىنىپىغا ئايلىنىپ كەتكەن، بۇ يەردە دىققەت قىلىشقا ئەززىيدىغىنى بۇ جەريان نەچچە ئون مىليون يىللاردا ئاندىن تاماملانغان،
بۇ فىلىمگە قىزىققۇچىلار بولسا بۇ ئۇيغۇر تىلىدا ئىلىپ بىرىلغان سۆھبەتنى ئۇلۇنىشى

مەن دەۋاتىمەن يولداشلار، تەدرىجى تەرەققىيات بىلەن يارىلىش قىسسىسى بىر بىرىگە زىت ئەمەس، ئەكسىچە بىر بىرىگە جىپسىلىشىپ ماس كىلىدۇ،
ھەممە شەيئى، ھەر قانداق ھادىسە دەسلەپ باردىن يوق بولۇپ يوقتىن تەجرىجى تەرەققى قىلىپ ، گۈللىنىپ، زۇۋاللىققا يۈز تۇتۇپ ئاخىرى يوقىلىدۇ،
يۇقۇرىدا دىگەن بارلىق شەيئى ۋە ھادىسىلەرنى ئۆز ئىچىگە ئالغان كائىنات يەنى ئالەممۇ، ئالەمنىڭ پەرپا بولىشى،  تەدرىجى كىڭىيىپ زورىيىشى، ئەڭ ئاخىرى بىرىپ يوقىلىشى بۇ تاق يۆلۈنۈشلشك جەريان دىندىمۇ پەننىدىمۇ قەيىت قىلىنغان.ئەمدى يارىلىش قىسسىسى بىلەن پەندىكى بۇ ئىككى جەرياننىڭ ماس كىلىشىگە قاراپ باقايلى

مەسىلەن يارىلىش قىسسىسىدە ئاللا ئەڭ ئاۋال ئالەمنى ياراتتى ،ئاندىن نۇرنى ياراتتى، ئاندىن ھاياتلىق پائالىيىتى ئۈچۈن زۆرۈر بولغان نۇر، ھاۋا، سۇ تاش تۇپراقنى ياراتتى، ئاندىم مەخلۇقاتلارنى ياراتتى، ئەڭ ئاخىرىدا ھەممە جانلىقتىن ئەزەز قىلىپ ئادەمنى ياراتتى، يارىلىش قىسسسىدە بۇ جەريان 6 كۈندە تاماملانغان دىيىلىدۇ، لىكىن گەپ بىز ئىنسانلار ئۈچۈن بۇ 6 كۈن ۋاقىت زادى قانچىلىق ۋاقىت؟نەچچە مىليارت يلمۇ؟ بۇنى ئىپادىلەش قىيىن ، چۈنكى ئاللانىڭ دەرگاھىدا زىمىندا ئىپادىلەنگەن ۋاقىت ۋە زاماننىڭ ئەھمىيىتى يوق، ماكان ۋە ۋاقىت ئۆلچىمى پەقەت ئاللا تەرىپىدىن ئىنسانغا بىرىلگەن نىمەت
لىكىن ھىلى بىز دەپ ئۆتكەن كائىنات ئالەمنى ئاللا تەرىپىدىن يوقتىن بار قىلىشى، ھاياتلىق پائالىيىتىگە زۆرۈر بولغان نۇر، ھاۋا، سۇ، تۇپراق، تاغۇ تاش قاتارلىق شەرتلەرنىڭ ھازىرلىنىشى بۇمۇ يارىلىش قىسسىسىدىكى تەرتىپ بويىچە زىمىندا يۈز بەرگەن ، ئوخشاش تەرتىپتە تەدرىجى يۈز بەرگەن، مەسىلەن بىڭ باڭ نەزىرىيەسىدە ئېيتىلغاندەك ئەسلى يوق بولغان كايىنات، ياكى بۈگۈن بىز ياشاۋاتقان ئالەم قارا ئۆڭكۈرنىڭ پارتلىشى بىلەن پەيدا بولغان ۋە شۇنىڭدىن بۇيان ئۈزلۈكسىز كىڭەيمەكتە، ئۇنىڭدىن بۇرۇنقى ئالەن خۇددى ئاللا ئالەمنى يارىتىشتىن بۇرۇنقى يوقلۇق ھالىتىدە ئىدى، ھەتتا نۇر ، ھاۋا ، سۇ ، زامان، ھەتتا ۋاقىت دىگەن نەرسىلەرمۇ مەۋقۇت ئەمەس ئىدى، ۋاقىتنىمۇ ئاللا يارىتىپ ئىنسانغا نىمەت قىلىپ بەرگەن ، شۇنىڭ ئۈچۈن ۋاقىتنىمۇ ئاللانىڭ نىمىتى، ئامانىتى دەيمىز، ۋقىتنى قەدىرلەيلى دوستلار، ۋاقىت مۇشۇنداق كونكىرىت ئۇقۇم.

ئەمىسە ئالەمنىڭ يوقۇلىشىچۇ؟
پەندە ئالەمنىڭ يوقتڭن بار بولۇپ ، توختىماي كىڭىيىپ ئاخىرى بىرىپ قارا ئۆڭكۈر تەرىپىدىن سۈمۈرۈلۈپ يەنە ئەسلىدىكى يوقلۇق ھالىتىگە قايتىشى بۇ بىر جەريان، دىنىمىزدا بۇنىڭغا جىپسىلىشىدىغىنى، بىزنى، بارلىق مەخلۇقات مەۋجۇداتلىنى ، بارلىق شەيئى ھادىسىلەرنى ، ئىنسان ئۆلچىمىدىكى ۋاقىتنى، زاماننى ئۆز ئىچىگە ئالغان ئالەممۇ ئاخىرى يوقىلىدۇ، بۇنىڭغا دىنىمىزدا ناھايىتى چىرايلىق قىلىپ زامانە ئاخىر دەپ ئىسىم قويۇلغان، يەنى ئاللا يوقلۇقتىن ياراتقان ئالەمنىڭ ، پۇتۇن نەرسىنىڭ زامان ۋە ماكاننىڭ ئاخىرلىشىش نۇقتىسى ، يەنى كايىناتنىڭ يۆگۈلۈپ قارا ئۆڭكۈر تەرىپىدىن سۈمۈرۈلىشى، ئۇ چاغدا تەكشىلىكتىكى ئالەم بىر تال ئىنچىكە سىزىق سۈپىتىدە قارا ئۆڭكۈردىن ئۆتىدۇ ، بۇ قاراشنى خافكىن ئوتتۇرىغا قويغان.
ئەمەلىيەتتە يارىلىش قىسسى ۋە پەندە ئوتتۇرىغا قويۇلغان بۇ ئوخشاش جەريان بىر بىرىگە بەك جىپسىلىشىدۇ، دىمەك زامانە ئاخىرغا، جقيامەتكە ئككى تەرەپ بىر بىرىنى شۇنداق ئىشەنمىگەنلىك يەنە ب بىر تەرەپتىن پەنگە ئىشەنمىگەنلىك بولىدۇ، بۇ ئىككى تەرەپ بىر بىرىنى شۇنداق قوللايدۇ

150 مىليون يىل بۇرۇنقى دىنازاۋۇر دەۋرىدە ئەگەر ئىت كالا توخو مۇشۈك يولۋاس شىر دىگەندەك سۈت ئەمگۈچى ھايۋانلارمۇ يەر يۈزىدە بولغان بولسا ، ھەتتا سۈت ئەمگۈچىلەردىنمۇ تۆۋەن دەرىجىلىك بولغان قۇشلار سىنىپى دىنازاۋۇر زامانىدا يەر شارىدا بولغان بولسا نەچچە يۈز يىللىق ئارخىلوگىيەلىك تەكشۇرۇشلەر بۇنى بايقاپ قىزىلمىلىرىنى تاشقا ئايلانغان قىتىشمىلىرىنى تىپىپ بولىتى، ئەپسۇس بۇنداق ئەھۋال بايقالمىدى ، ھازىرقى زامان ئادىمىنىڭ تاشقا ئايلانغان قىتىشمىلىرىنى ئەڭ كۆپ بولغاندىمۇ ئىككى يۈز مىڭ يىلغا ئاپىرىشقا بولىدۇ، يەنە تىخى ئۇ قىتىشمىلىنىڭ تۈزۈلىشى بىز ھازىرقى ئىنسانلاردىن زور پەرىقلىدۇ.
ئارخىلوگىيەدە جانلىقلارنىڭ يەر قاتلاملىرىدا تاشقا ئايلىنىپ ساقلىنىپ قىلىشىنىڭ بىر دىن بىر شەرتى شۇ جانلىقتا قاتتىق سۆڭەك بولىشى كىرەك ، شۇنىڭ ئۈچۈن بىز بىلىقلار سىنىپىدىن بۇرۇن پەيدا بولغان ھايۋالارنىڭ تاز قىىتىشمىلىرىنى تاپالمايىز، چۈنكى قاتتىق سۆڭەك بولۇش خوردىلىق ياكى ئومۇرتقۇلۇق ھايۋالاردىن باشلىنىدۇ، مەسىلەن بىلىقلار سىنىپى، ئارخىلوگىيە ئىلمىدا شۇنداقكىن، جانلىق قانچە ئىپتىدايى، قانچە تۆۋەن دەرىجىلىك بولسا شۇنچە قەدىمى يەر قاتلاملىرىدىن تىپىلىدۇ بۇنىڭ يىلىنى كاربون ئىزوتوپلۇق تەكشۈرۈش دىگەندەك ئىلمى ئۇسۇللار بىلەن يەخمىنى ھىساپلاپ چىققىلى بولىدۇ، بۇ تەدرىجى تەرەققىيات شۇنداق زەنجىرسىمان، تاق يۆلۈنۈشلۈۈككىن ، شۇنداق تەرىپلىككىن ئىپتىدائى، تەدىمى تاش قاتلاملىرىدىن تىپىلغان جالىقتىن يۇقۇرى دەرىجىلىك بولغان جانلىق شۇ تاش قاتلىمىدا تىپىلمايدۇ، پەقەت ئۇنىڭدىن تۆۋەن دەرىجىلىگىنىڭ تىپىلىشى مۇمكىن،
مەسىلەن بىزنىڭ قۇمۇل تۇرپان گۇچۇڭ جىمسار قاتارلىق رايونلىرىمىزدىكى 70 مىليوندىن 150 مىليون يىللغىچە بولغان قەسىمى تاز قاتلاملىرىدىن دىنازاۋۇر ۋە تۇخۇملىرى كۆپلەپ بايقىلىۋاتىدۇ، لىكىن دىنازاۋۇر بايقالغان شۇ تاش قاتلاملىرىدا دىنازاۋۇرلار بىلەن بىللە ئۇچار قۇش ، ئات كالا، ئىت مۈشۈك، بورە يولۋاس دىگەندەك دىنازاۋۇرلاردىنمۇ ئالى دەرىجىلىك بولغان قۇشلار سىنىپى ۋە سۈت ئەمگۈچىلەر سىنىوىغا تەۋە بىرە جانلىقنىڭ ھازىرغىچە تىپىلغىنى يوق، بىز تىپىلىپ قالارمۇ دەپمۇ خام خىيال قىلمايمىز، چۈنكى دىنازاۋۇر ياشىغان ئوتتۇرا ھايات ئىراسىنىڭ يورا دەۋرىدە ئادەم ئەمەس ئۇچار قۇش ۋە ھەر قانداق سۈت ئەمگۈچى ھايۋان تىخى يوقتى، بۇ قاراش بۇ تەرتىپ دارۋېننىڭ تەدرىجى تەرەققىيات تەلىماتىنىڭ مۇھىم بىر تەركىۋى قىسمى، ھازىرغىچە ئاغدۇرۇلۇپ كەتمەستىن بەلكى بارغانسىرى ئىسپاتلىنىپ بىيىپ بىرىۋاتىدۇ، ھەم ئاخىلوگىيەلىك تەكشۇرۇشلەرگە ئۇسۇل ۋە ئىدىيە تەرەپتىن قوشقان تۆھپىسى ئىنتايىن زور.
بۇ تەرەپ يەنە دارۋېن تەلىماتىنىڭ مۇھىم قىسمى ئەمەس، دارۋېن تەلىماتىنىڭ مۇھىم قىسمى تەبىئى تاللىنىش ۋە مۇھىتقا ماسلىشىش داۋامىدىكى تۈرلەرنىڭ تۆۋەن دەرىجىدىن يۇقۇرى دەرىجىگە ئاددىلىقتىن مۇرەككەپلىككە ئۈزلۈكسىز  تاق يۆلۈنۈشلۈك تەرەققى قىلىشى.

لىكىن ھازىرقى كلون تىخنىكىسىدا ئەركەك تەنىچىدىن ئەركەك تەنىچىنىلا، چىشى تەنىچىدىن چىشى تەنىچىنىلا يىتىشتۈرگىلى بولىدۇ، يارىلىش قىسسىسىدا كۆرسىتىلگەندەك ئەركەكتىن چىشىنى، چىشىدىن ئەركەكنى تىخىچە يىتىشتۈرۈپ باققىنى يوق، لىكىن كىتاپلىرىمىزدا بۇ ھىكمەت توغرىلىق بىشارەت بىرىلگەنكەن بۇ كەلگۈسىدە ئەمەلگە ئاشىدۇ.
ھەتتا ئاللا ھەر قانداق كىسەلنى داۋاسى بىلەن ئەۋەتكەن، ئىنسان ئىزدىنىپ داۋاسىنى تىپىپ كەلگەن ، دىمەككى ئەيدىزنىڭمۇ داۋاسى تىپىلىدۇ، چۈنكى داۋاسىنىمۇ بىللە ئەۋەتكەن
شۇنىڭ ئۈچۈن ھە دىسىلا قاتتىق ئۆتۈكلۈك قىلماي ئىلىم مەرىپەتكە يۈرۈش قىلايلى دوسلا، بىلمىگەننى بىلەيلى، چۇنكى ئاللانىڭ يەنە نۇرغۇب ھىكمەتلىرى بىشارەتلىرى بىلەن بىزنىڭ ئىچىشىمىزنى كۈتۈپ تۇرۇپتۇ، نىمە دىگەن ساۋاپلىق ئىش ھە.
