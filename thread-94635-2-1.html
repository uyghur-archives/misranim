بۇنىڭدىن شۇنى كۆرۈپ يەتتىمكى شىنجاڭغا ئەيدىز ئەكىرگەن مەينەتلەردىن يۇقۇملانغان ئەقىلسىز ئىشەكلەرنىڭ مىللىتىمدىن بولغانلىقى مىنىڭ غەزىپىمنى تاشۇرماقتا . دىمەك بىز جوڭگودىكى رايونلار ئىچىدە ئەيدىز ئەك كۆپ مىللەت ، باشقا چوشقىلارغا قارىغاندا بىرىجى بولالمىساقمۇ ھەرھالدا ئالدىنقى قاتارنى قولدىن بەرمەپتىمىز ، ئاشۇ ناپاك ، زەيپالارنىڭ بۇ دۇنيادا خار ئۇ ئالەمدە زار بولۇپ كەتكەي .

  باشقىلارنىڭ مىنى تاش يۈرەككەن دىمەسلىكىنى ئۈمىت قىلىمەن ، يوقارقىلار ئۆزى تاپقان ، بىلىپ تۇرۇپ ئۆلۈمگە ئۆزىنى ئاتقان نائەھلىلەر ، ئۇلارغا ھىسداشلىق قىلغۇم يوق . چۈنكى مىنىڭ تاغاممۇ ئاق چىكىپ ئائىلىسىنى ۋەيران قىلدى ، ھاياتىنىمۇ نابۇت قىلدى . ئۇمۇ زەھەرنىڭ قانداق نەرسە ئىكەنلىكىنى بىلەتتى ، لىكىن يەنىلا يانمىدى ، شۇڭا بۇنداق ئەقىلىسىزلار ئىچ ئاغرىتىش ئارتۇقچە ئىش خالاس .
   ئەيدىزنىڭ ئۇۋىسى : ھەممەيلەنگە مەلۇم زەھەرلىك چىكىملىك ، دىبا ، شوپاڭلار ، ئەجىبا ئۆزى خالاپ ئۆزى ئىزدەپ سوراپ بۇ نىجىس كىسەلنى تاپقانلار ئىچ ئاغرىتىشقا ،كۆڭۈل بۆلىشكە تىگىشلىكمۇ ؟

  ئاشۇ ھىسداشلىقىمىزغا تەلمۈرۈۋاتقان نۇرغۇنلىغان نارسىدە ، ئاجىز ، غىرىپ ،غۇرۋالار ئەتراپىمىز پەرۋانىدەك يۈرمەكتە ، شۇ خەيرىخالىقنى شۇلارغا بەرسەك ، بۇ لەنەتكۈرنى بالدۇرراق بولسىمۇ ئۇ ئالەمگە يوللىۋەتسەك ، چۈنكى بۇلار جەمىئىيەت ، ئائىلە يۈك ۋابالار .

شىنجاڭدا ئۇيغۇر ئەڭ كۆپ بولغاندىكىن، ئەلۋەتتە ئاۋۇ پوتەي ئېگىز چىقىپتۇ، بىرىنچى بولۇپ شىنجاڭدىن بايقالغان ئەيدىز قايسى مىللەتتتىن ئىدى؟ بۇ يەردە ئەجەپ بۇنى دېمەپتۇ؟ ئېتىقاد سۇسلىشىپ، كاپىرلىققا يۈزلەنەەندە ، بۇ نىجىس كېسەلنى بىزگە ئىبرەت قىلىپ بەرگەندۇ ئاللا، بۇنى كۆرۈپ چۆچۈپ توۋا قىلسۇن دەپ شۇندان جىق بەرگەندۇ، ئەلۋەتتە ئاللا بەندىلىرىگە زۇلۇم قىلمايدۇ، يۇقۇردىكى بىر توردىشىمىز ئېيتقاندەك، كەيپنى بىلىپ تۇرۇپ قىلىپ، كېسەلنى بىلىپ تۇرۇپ يۇقتۇرۇپ ياكى يۇقتۇرىۋېلىپ ، ئۆزىگە زۇلۇم قىلىپ غېمىدە يوق يۈرگەنلەرگە بەرگەن كېسەل بۇ.
