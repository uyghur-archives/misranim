يوللىغان ۋاقتى 2012-12-26 09:56:01 PM |ھەممە قەۋەتنى كۆرۈش |
بۇ ھايات، بۇ تىڭىرقاش، بۇ قىسمەتلەر مېنى ھەقىقەتەن ھارغۇزۇپ قويدى، كىمدىن، نېمىدىن ئاغرىنىشىمنى پەقەت بىلەلمىدىم، گەرچە كىچىكىمدىن ئائىلەمدە ھېچنەرسەم كەم ئەمەس، دېگىنىم دېگەن بولۇپ چوڭ بولغان بولساممۇ، لېكىن مۇھەببەتتىن ئىبارەت بۇ تېگى يوق دېڭىزدا يا چۆكۈپ كەتمىدىم، يا بىرەر قۇرۇقلۇققا چىقالماي لەيلەپ قالدىم، تۇرمۇشۇمدىكى بۇ ئىككى ئادەم يەنە مەن ھەم مېنى سۆيىدىغان ئىنىسى، مېنى ياخشى كۆرىدىغان ئاكىسىدىن ئىبارەت بۇ قېرىنداشلارنى شۇنداق ۋەيران قىلدىمكى، ئۇلارنىڭ بىرى ئۆز سۆيگۈسىنى، ئازابلىرىنى، قايغۇسىنى ئۆزى بىلەن ئېلىپ بىزدىن مەڭگۈلۈك ئايرىلدى، يەنە بىرى بولسا گەرچە گۇناھلار ماكانى بولغان، كىشىگە خۇشاللىقتىن كۆپرەك ئازاب، كۈلكىدىن كۆپرەك يىغا ئاتا قىلىدىغان بۇ رەھىمسىز دۇنيادا زادى كىمگە نەپرەتلىنىشىنى بىلەلمىگەن ئاساستا قالدى،
مەن شۇ ياتقان پېتى خېلى كۈنلەرگىچە پەقەت كۆزۈمنى ئېچىپلا ياتتىم، ئۆزۈممۇ بىلمەيمەن، خۇددىي گەپ قىلىشنى ئۇنتۇپ كەتكەن كىشىدەك بىر ئېغىز گەپ قىلماي شۇ ئەتراپىمدىكى كىشلەرگە قاراپ ياتاتتىم، ياكى بولمىسا كۆڭلۈمدىكى كىشىمنى، بەگزادىمنى دەخلىسىز، توسالغۇسىز كۆرەلەيدىغان، سىردىشالايدىغان چۈشلىرىم قوينىغا غەرق بولۇپ ئۇيقۇغا كېتەتتىم، ھېچكىممۇ ماڭا زادى كىمنىڭ ھايات، كىمنىڭ ئۆلگەنلىكى توغرىسىدا ئېغىز ئاچمىدى،
مېنىڭمۇ نېمىشقىدۇر بىلگۈم يوق ئىدى، چۈنكى ئەگەر بۇ ئاق ياغلىقنىڭ ئىگىسى بەگزاد بولۇپ قالسا بۇ ھايات شۇ دەقىقىدىلا مەن ئۈچۈن تامام، ئەگەردە سەردار بولسا تېخىمۇ بولمايدۇ، گەرچە ئۇ بەگزاد ئىككىمىزنىڭ شۇنچە يىل ئازاب چېكىشىگە سەۋەبچى بولغان، ئۇنىڭغا نەپرىتىم بولسىمۇ، لېكىن مەن ھېچقاچان ئۇنىڭ ئۆلۈپ ماڭا ھەم بەگزادقا، ئىشقىلىپ بۇ ئىشلارنى چۈشەندۈرۈشكە تېگىشلىك بولغان بارلىق كىشلەرگە چېچقانداق ئىشنى چاندۈرماي «ئۆلۈم» بىلەن ئەڭ ئاخىرقى جاۋابىنى بېرىشىنى تامامەن خالىمايمەن.
    - ھەدە، گەپ قىلە، نېمىشقا گەپ قىلمايسەن؟
    خىياللىرىمنى شۇ جايىدىلا توختىتىپ سىڭلىم دىلرۇس ماڭا گەپ قىلدى.
    ئۇنىڭغا نۇرسىز، ئاللىقاچان ياشلىرى قۇرۇپ كەتكەن كۆزلىرىمنى مەقسەتسىزلا ئۇنىڭغا تىكتىم.
    - بەگزاد قېنى؟
    دىلرۇس گەپنىڭ ئورنىغا ماڭا ياش بىلەن جاۋاب قايتۇردى.
    - نېمىشقا يىغلايسەن؟ ئۇ ئۆلدىما؟ ياق، دىلرۇس، ماڭا ھەرگىز «ھەئە» دەب جاۋاب بەرمە سىڭلىم، ئەمدى مەن كۆتۈرەلمەيمەن، مەن سەندىن ئۆتۈنەي، ئاھ خۇدا، ياق، ياق!!!
    - جېنىم ھەدە، ئۆزۈڭنى تۇتۇۋال، بەگزاد ساق، لېكىن سەردار ئاكام ئۇ ئۆلۈۋالدى.
    - نېمە، يالغان، ئۇ نېمىشقا ئۆلۈۋالىدىكەن، ھەممە ئىشنى ئۆزى قىلىپ نېمە ھەققى باركەن ئۆلۈۋالىدىغانغا؟ مېنىڭ ئۇنىڭدىن سورايدىغان نۇرغۇن سوئاللىرىم بار تۇرسا، ئۇ مۇشۇنداقلا كەتسە قانداق بولىدۇ؟
    بۇ يىغىلار، نېمە ئۈچۈن، كىم ئۈچۈن ئېقىۋاتىدۇ؟ سەرداردىن تېخى ئېلىپ بولالمىغان ئۆچلىرىمگە ئىچىم ئاغرىپمۇ ياكى بۇ لەنەتكە تولغان سۆيگۈگە ئېچىنىپمۇ، سەردارنىڭ ھېچنىمە چۈشەندۈرمەي مۇشۇنداقلا قارا يەر تېگىگە كىرىۋالغانلىقى ئۈچۈنمۇ؟ ھېچ بىلمىدىم.
    - يىغلىما ھەدە، ئۇ كۈنى سەردار ئاكام بەگزادنى ئۇرماقچى بولغاندا سەن ئۆزۈڭ بىلەن توسۇۋاپسەن، ئۇ خىرۇستال سېنىڭ بېشىڭغا تېگىپتۇ، بۇ ئەھۋالدىن سەردار ئاكام قورقتىمۇ ئىشقىلىپ قالغانلار سېنىڭ بىلەن بولۇپ كەتكەندە ئۇ يۈگۈرگەن پېتى چىقىپ ماشىنىنى ھەيدەپ كېتىپتۇ، ئىككى كۈندىن كېيىن بىزگە ئۆلۈم خەۋىرى كەلدى، قاتناش ھادىسىسى، لېكىن ئۇنىڭ يېنىدىن چىققان ساڭا يازغان خەت ھەم شۇ يەردىكى كۆرگەن كىشىلەرنىڭ ئېيتىشىچە تورمۇزنىمۇ تۇتماي ئۇدۇل بېرىپ ماشىنىنى ئۇرغانلىقىدىن ئۇنىڭ ئۆزىنى قەستەن ئۆلۈۋالغانلىقىنى بىلدۇق، ھەدە، بەگزاد بەك تۈگىشىپ كەتتى، ئۇ سەن ھوشسىز ياتقاندا بىر نەچچە قېتىم كەلدى، تېخى بايىلا ماڭا تېلېفون قىپتىكەن، ھېلىغىچە كېلەرمىكىن. -  سىڭلىم ماڭا سەردارنىڭ خېتىنى بېرىپ قويۇپ چىقىپ كەتتى.
    «سالام ئەقىدە، بۇ خەتنى قولىڭىزغا ئېلىپ ئوقۇغان ۋاقتىڭىزدا بەلكىم مەن ئاللىقاچان بۇ رەزىل دۇنيادىن كەتكەن بولىمەن، ئەمدى بىلسەم مەن مۇھەببەتنىڭ نېمە ئىكەنلىكىنى بىلمەيدىكەنمەن، پەقەت يىگانە كۆڭلۈمنىڭ بىر سىزگە تەشنا ئىكەنلىكىنى بىلىمەن، سۆيگۈنىڭ نېملىكىنى بىلمەيدىكەنمەن، پەقەت سىزدىن ئايرىلالمايدىغانلىقىمنى، ئايرىلساملا سىزنى سېغىنىدىغانلىقىمنى بىلىمەن، ئەقىدە، سىز بىلەن توي قىلغاندىكى كۈنلەردە سىزدىن بەكرەك ئازابلانغان ئادەم مەن، چۈنكى سىز مېنى پەقەت ياخشى كۆرمەيسىز، توي كېچىسى ئەڭ نازۇك، ئەڭ چىرايلىق غۇنچىلىرىڭىز مەن تەرەپتىن ئۈزۈلۈۋاتقاندىمۇ سىزنىڭ مېنىڭ ئۆز ئېنىم بەگزادنى ئويلاۋاتقانلىقىڭىزنى، يۈرىكىڭىزنىڭ يەنىلا شۇنىڭدا ئىكەنلىكىنى بىلەتتىم، ماڭا تەۋەسى پەقەت شۇ قىلچە ھارارەت بولمىغان تەن، ھەم چىڭ يۇمۇلغان كۆزلەردىن سىرغىپ چۈشكەن ياشلار بولدى، لېكىن شۈكرى قىلدىم، چۈنكى ئەنە شۇنداق پەسلىكلەر بەدىلىگە كەلگەن پەس سۆيگۈگە ئاران ئېرىشەتتىم.
    بەگزاد، ئويلىمىغان يەردىن قايتىپ كەلدى، ئازابلىرىم تېخىمۇ ئازايدى، لېكىن بەگزادنىڭ مەن يوق ۋاقىتلاردا سىز بىلەن پاراڭلىشىپ، ئىككىڭلارنىڭ ‹دىدارلىشىۋېلىشىڭلار› نى مەن كۆرمەسكە، بىلمەسكە سېلىپ يۈردۈم، چۈنكى مۇشۇ ئارقىلىق ئۇنىڭغا ھەم سىزگە بولغان قەرزىمنى ئازراق بولسىمۇ قايتۇرالىدىم دەپ ئويلايتىم، يۈرىكىم ئاغرىسىمۇ، سىزنىڭ قانچىلىك دەرىجىدە شەرمىي - ھايالىقلىقىڭىز ماڭا تامامەن ئايان ئىدى، گەرچە ياخشى كۆرىدىغان ئادىمىڭىز بەگزاد بولسىمۇ، لېكىن ھەرگىزمۇ ماڭا ئاسىيلىق قىلمايتىڭىز، شۇنىڭ ئۈچۈن مەن ئەنە شۇنداق ئىشلارنى بىلمەسكە سېلىپ يۈرىۋەردىم، نېمە دېگەن پەس، نېمە دېگەن خارلىقتا قالغان سۆيگۈ - ھە؟ مانا مۇشۇنداق سۆيدۈم سىزنى، ئۆزۈمنىڭ قانچىلىك دەرىجىدە رەزىللىشىپ كېتىپ بارغانلىقىمنى ھېس قىلغان بولساممۇ، لېكىن سىزدىن ئايرىلىپ قالماسلىقنى، بەگزادنىڭ ئاھۇ - زارلىرىغا، سىزنىڭ يۈرەك زەرداپلىرىڭىزغا، سىلەرنىڭ سۆيگۈ ئۇۋالىڭلارغا تېگىشتىم، چۈنكى مەن بۇ يولغا ماڭغاندىلا شۇنداق بولۇشىغا، ھەتتا مەڭگۈ شۇنداق بولۇشىغا يول قويغان، لېكىن مانا بۈگۈنكى كۈنلۈككە كەلگەندە يەر شارى مەن ئۈچۈن تەتۈر، پۈتۈنلەي تەتۈر چۆرگىلىدى، خۇددى يانار تاغ پارتلىغاندەك پۈتۈن ئىشلىرىم ئاشكارىلاندى.
    لېكىن مەن ھەرگىزمۇ ئۆزۈمنى سىلەرنىڭ ئالدىڭلاردا گۇناھكار ھېسابلىمايمەن، كەچۈرۈممۇ سورىمايمەن، چۈنكى سىزگە ئاشىق بولۇشىم ھېچقاچان گۇناھ ئىش ئەمەس، بەگزادقا ئوخشاشلا سىزنى مەنمۇ ياخشى كۆردۈم، ئۆزۈمنىڭ، مەڭگۈ ئۆزۈمنىڭ قىلغۇسى كەلدى، بەزى ئاشىقلارغا ئاللاھ ئاسان يوللار ئارقىلىق مەشۇقىنىڭ ۋىسالىغا يەتكۈزىدىكەن، بەزى مەندەك ئاشىقلارغا نۇرغۇن نەرسىلەردىن ۋاز كېچىش بەدىلىگە مەشۇقىنىڭ ۋەسلىگە يەتكۈزىدىكەن، مانا مەن شۇنداق ۋاسىتە تاللىماي سىزنىڭ ۋىسالىڭىزغا يەتتىم. خەير! سىلەرنىڭ ئالدىڭلارغا گۇناھكارلارچە بېرىپ كەچۈرۈم سورىغۇم يوق، ياكى بولمىسا بۇ دۇنيادا تىرىك ياشاپ بەگزاد ئىككىڭلارنىڭ قايغۇلاردىن كېيىن تاپقان بەختىڭلارنى كۆرگۈممۇ يوق، ھامىنى ئېرىشىدىغىنىم يەنە شۇ «نەپرەت»، شۇنىڭ ئۈچۈن ئۆلۈمدىن ئىبارەت بۇ قايتماس يولغا مېڭىپ سىزنىڭ ماڭا يالغاندىن بولسىمۇ بەرگەن شۇ ئاتالمىش ‹مۇھەببىتىڭىز› نى ئۆزۈم بىلەن مەڭگۈگە ئېلىپ كەتتىم.
    سىزنى مەڭگۈ سۆيۈپ: سەردار»
    ئۇ ئۆلۈپ كەتتى، ھەممىنى ئۆزى بىلەن ئېلىپ ئەمەس، بەلكى ھەممە ھىجراننى ماڭا، بەگزادقا تاشلاپ، بىزنى بۇ ئاخىرى يوق يولغا باشلاپ قويۇپ كەتتى. سەردار ئۈچۈن ھاز ئېچىپمۇ يىغلىمىدىم، ئۇنىڭغا ئېچىنىپمۇ يىغلىمىدىم ھەم ئۇنى كەچۈرۈشنىمۇ ئويلاپ باقمىدىم، چۈنكى سەردار ھېچقاچان مەندىن ئۇنى كەچۈرۈشۈمنى، ياكى ئۇنىڭغا ئېچىنىشىمنى تەلەپ قىلىپ باقمىدى. دۇنيا بەگزاد ئىككىمىزگە، مۇھەببەت بەگزاد ئىككىمىزگە قالدى، لېكىن بۇ لەنەتلىك سۆيگۈنى ھەر ئىككىلىمىز داۋاملاشتۇرالمايتۇق، مېنىڭ تەخىر قىلالماي، ئۆز سۆيگۈمگە ئەمەل قىلالماي، بەگزادنىڭ ئۆتكۈزمىگەن خاتالىقنىڭ جازاسىنى تارتىپ كەچۈرۈم سوراشنىڭ خاتا يولىنى تاللىغانلىقىنىڭ بەدىلىگە كېلىپ چىققان بۇ مۇھەببەتنى قانداقمۇ داۋاملاشتۇرالايمىز؟
    شۇنىڭ بىلەن سەردارنىڭ قىرىق نەزىرسى توشۇپ، ئىشلار بىر قۇر بېسىققاندىن كېيىن مەنمۇ، بەگزادمۇ بۇ بىزگە ئاجايىپ قىسمەتلەرنى ئاتا قىلغان شەھەرنى تاشلاپ ئايرىم - ئايرىم يەرلەرگە كەتتۇق.
    سەردار ئۆز سۆيگۈسىنى ئۆزى بىلەن بىللە قارا يەر تېگىگە ئىلىپ كەتتى، بەگزاد ئىككىمىز بولساق ۋىسال تاپالمىغان مۇھەببىتىىمىزنى يەنە بىر قېتىم قەلبىمىزنىڭ ئەڭ چوڭقۇر، پىنھان يەرلىرىگە كۆمگەن ھالدا تونۇشقاندىن تارتىپ تاكى بۈگۈنگىچە بولغان ھەربىر كۈنىمىزنىڭ شاھىدى بولغان بۇ شەھەردىن ئۆزىمىز بىلەن بىللە ئېلىپ كېيىنكى ھايات يولىمىزنى ئايرىغان ھالدا ئايرىم يەرلەرگە ئايرىلىپ كەتتۇق.
خاتىمە

    سېنى قەلب تۆرەمنىڭ ئەزىز مېھمىنى قىلىش ئۈچۈن يۈرەك قېنىمنى سەرپ قىلىپ، ساڭا ئاتاپ داغدام بىر يول ياسىغان ئىدىم، لېكىن سەن مەن ياسىغان بۇ يولدىن ماڭماستىن، بەلكى دەرەخلەر ئارىسىغا غازاڭلار تۆكۈلگەن چىغىر يولغا قاراپ يۈرۈپ كەتتىڭ.
    مەن ئۆزۈمنى تولىمۇ بىچارە، بەختسىز چاغلىغىنىمچە كەينىڭدىن قاراپ تۇرۇپ قالدىم، سەن ئالدىڭغا بىر نەچچە قەدەم ماڭا - ماڭمايلا پۇتۇڭنى چاڭگاللىغىنىڭچە ئولتۇرۇپ قالدىڭ، تاپىنىڭ قىپقىزىل قانغا بويالغان ئىدى، سەن دىققەت قىلماستىن باشقىلار غازاڭلار ئارىسىغا تاشلاپ قويغان ئەينەك پارچىسىگە دەسسىۋالغان ئىدىڭ. سەن تولىمۇ بىچارە ھالدا كەينىڭگە بۇرۇلۇپ ماڭا قارىدىڭ، كۆزىمىز قايتىدىن ئۇچرىشىپ قالدى، كۆزۈڭ چوڭقۇر پۇشايمان ۋە ئازاب يېشىغا تولغان ئىدى، توساتتىن ئۆزۈمنىڭ ساڭا قارىغاندا تولىمۇ بەختلىك ئىكەنلىكىمنى ھېس قىلدىم.
    تامام.
