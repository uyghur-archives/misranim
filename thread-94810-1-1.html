ماۋزېدۇڭنىڭ تۇنجى مۇھەببەتلەشكەن قىز دوستىنىڭ بۇرۇن ئۆلۇپ كىتىش سىرى يېشىلدى(رەسىم)

ئاپتۇرى:لى فېڭشىيەن

  

   ماۋزېدوڭنىڭ بىرىنجى خانىمىنىڭ ئىسمى لوداشىيۇبۇلۇپ،توي قىلغاندىن كىيىن بىللە تۇرۇپ باقمىغان ھەم بالىسىمۇ يوق.ماۋزېدوڭ شاۋ شانچوڭدىن ئايرىلغاندىن كىيىنكى تۇنجى ئاشىقى تاۋ سىيوڭ خانىم.

  تاۋ سىيوڭ،شۇ چاغدا 16ياش بۇلۇپ،چاڭشادىكى سودىگەرلەر ئۇيۇشمىسىنىڭ باشلىقىنىڭ قىزى،خۇنەننىڭ شىياڭتەن دىگەن يىرىدىن،مۇتىۋەرنىڭ قىزى،مىھرىبان،ئاقكۇڭۇل،كىلىپ چىقىشى ياخشى،بىلىملىك كىشىلەرنىڭ سالىمىنى ئىلىك ئالىدىغان ئىسىل پەزىلەتلىك قىزبۇلۇپ،ھەقىقەتەن ئۇ كۆپچىلىككە قىلچىمۇ بوينى قاتتىقلىق قىلىپ،تەكەببۇرلۇق ۋە يامانلىق قىلمايتى.خۇنەن چاڭشادىكى بىرىنجى دارىلمۇئەللىمىندە ماۋزېدوڭ بىلەن ساۋاقداش ھەم يۇرتداش ئىدى،شۇ چاغدا«چاڭشادا بىرىنجى ئايال ئىختىساس»دىگەن چىرايلىق نام بىرىلگەن.

تاۋ سىيۇڭ1916-يىلى ئىمتىھاب بىرىپ،جۇجىيەنفەن ئەپەندىم ئاچقان جونەن قىزلار ئوتتۇرا مەكتىپىنىڭ دارىلمۇئەللىمن2-سىنىپىغا قۇبۇل قىلىنغان،ھەم داڭلىق ئايال ئىنقىلاپچى شىياڭ جىڭيۇ بىلەن ساۋاقداش بۇلۇپ ئۇقۇغان.شىياڭ جىڭيۇ،تاۋيى ۋە جى چاڭلار«جو نەندىكى ئۇچ تالانت ئگىسى»دەپ ئاتالغان.ئۇقۇش پۇتتۇرگەندىن كىيىن،شىياڭ جىڭيۇ يۇرتى شۈپۇغا قايتىپ مەكتەپ قۇرغان،ھەمدە ئۇ داۋاملىق جونەن قىزلار مەكتىپى بىلەن قۇيۇق ئالاقىنى ساقلاپ كەلگەن.

  20-ئەسىرنىڭ20-يىللىرىنىڭ بېشىدا،تاۋ سىيوڭمۇ چاڭشادىكى مائارىپ ساھەسىدىكى زاتلار بىلەن ۋەزىيەت توغرىسىدا پىكىرلىشىپ تۇرغان،خۇنەن ئۇقۇغۇچىلار بىرلەشىسى ۋە خۇنەندىكى ھەر ساھەسەدىكىلەر بىلەن بىرلەشمىلەردە بولغان،ئۇ مۇئاۋىن جەمئىيەت باشلىقى بىلەن تەڭ ئىدى،شۇ چاغدا ماۋزېدوڭ پەقەت كېڭەش ئەزاسى ئىدى.تاۋ چاڭشادىكى داڭلىق گۈزەل،بويى ئىنتايىن ئىگىز،يۇقۇرى تالانتلىق بۇلۇپ.ئۇقۇش پۇتتۇرگەندىن كىيىن مەكتەپتە قىلىپ خىزمەت قىلغان.ئۇ بىر يېڭى تىپتىكى ئايال،ئىدىيىسى كەسكىن،مائارىپ ئارقىلىق دۆلەتنى قۇتقۇزۇشنى تەشەببۇسچىسى قىلغۇچى.

  تاۋ سىيوڭ بولسا ماۋزىدوڭنىڭ تۇنجى مۇھاببەتلەشكەن ئاشىقى.يېڭى ئىلمىي جەمئىيەتنىڭ پائالىيەتلىرىدە،ماۋزېدوڭ مەردانىلىقى جۇش ئۇرۇپ تۇردىغان سالاپىتى ۋە ئىش بىجىرگەندە ئىنچىلىك بىلەن كەسكىن بىجىردىغان خراكتىرگە.بىر تۇركۇم ئايال ئۇقۇتقۇچىلار ئاپرىن ئىيىتقان.بۇنىڭ ئىچىدە،شىياڭتەندىكى يۇرۇتلىقىمۇ بار ئىدى،تاۋيى تىخىمۇ ئۇنىڭغا توختىماي كۇڭۇل ئىزھار قىلغان.يېڭى ئىلمىي جەمئىيەتتىكى پائالىيەتتە،ئۇ دائىم مەقسەتلىك ھالداماۋزېدۇڭغا يىقىنلىشاتتى،شۇنىڭ بىلەن بىرگە ئۇنى قوغلىشاتتى،شۇڭلاشقا قۇيۇق مۇناسىۋەت جەريانىدا ئۇلار ئوتتۇرسىدا مىھرى-مۇھەببەت پەيدا بۇلۇشقا باشلىغان.

  ماۋزېدوڭ بىلەن تاۋيى شىياڭتەنلىك بىر يۇرۇتلۇق بۇلۇپلا قالماستىن،بەلكى ئىنتىلىشىمۇ ئوخشاش ئىدى،بىراق ئائىلە ئىقتىسادى جەھەتتىكى پەرق ئىنتايىن چوڭ ئىدى.تاۋ ئائىلىسى شىياڭتەندىكى نامدار باي ئائىلەرنىڭ بىرى،تاۋيى بىلەن ماۋزېدوڭنىڭ مۇناسىۋىتى شۇئان ئائىلە باشلىقلىرىنىڭ كۈچلۈك قارشىلىقىغا ئۇچرىدى.نەتىجىدە،قىز تەرەپنىڭ ئائىلىسىنىڭ قارشىلىقى تۇپەيلىدىن،ماۋزېدوڭ بىلەن تاۋيىنىڭ ئەمدىلا بىخلىنىۋاتقان مۇھەببىتى ئىنتايىن تىز ئۇشتۇمتۇت توختاپ قىلىپ،ئاخىرغىچە داۋاملاشتۇۇرالماي قالدى.

ئەمەلىيەتتە،سىياسىي قاراشنىڭ ئوخشىماسلىقى ئىككەيلەننىڭ ئايرىلىپ كىتىشنىڭ مۇھىم ئامىللىرىنىڭ بىرى.

  ئىككەيلەن ئايرىلىپ كەتكەندىن كىيىن،بۇرۇنقى دوستلىقىنى ساقلاپ قالغان.ماۋزېدوڭ مەدەنىيەت ئۇيۇشمىسى قۇرغاندا،تاۋيى ئاساسلىق مەبلەغ سالغۇچىلارنىڭ بىرى ئىدى.ئۇ چاغدا،تاۋسىيوڭ بىلەن شىياڭ جىڭيۇ،ئوخشاش،ئىدىيىسى ئۇچۇق،كەسكىن قىزلار بۇلۇپلا قالماستىىن،يەنە يېڭى ئىلمىي جەمئىيەتنىڭ كۆزگە كۈرۇنگەن ئايال باتۇرلىرى ئىدى.ماۋزېدۇڭ مەدەنىيەت ئۇيۇشمىسى تەسىس قىلغان چاغدا ،ئۇ ئۇچەيلەن نىڭ ئىچىدىكى 30كۇمۇش تەڭگە چىقىرىپ ياردەم قىلغۇچىلەرنىڭ بىرى.يېڭى ئىلمىي جەمئىيەتنىڭ دەسلەپكى باسقۇچلىرىدا كۆپ قېتىملىق مۇنازىرىدە ،ئۇ «مائارىپ ئارقىلىق دۆلەتنى قۇتقۇزۇش»نى تەشەببۇس قىلغان.كىيىن،ئۇ ماۋزېدۇڭنىڭ «ئۆزگەرگەن جۇڭگۇ ۋە دۇنيا»شۇئارىنى قوللىغان.

  تەكشۇرۇپ يەكۈنلەنلەشلەرگە ئاساسلانغاندا،ماۋزېدۇڭ دەسلەپ،تاۋسىيوڭغا قارتا بۇ مەزگىلدىكى مۇھەببەتنى ئىنتايىن قەدىرلىگەن،نۇرغۇن ئادەملەرنىڭ مۇناسىۋەتلىك ئەسلىمىللىرى بۇ تەرىپىنى ئىسپاتلايدۇ.شۇ چاغدىكى«كۈيئۇغۇلنى تەبرىكلەپ،دوستتىن ئايرىلما»ناخشىسى بولسا 1922-يىلى تاۋسىيوڭغا بىر-بىرلەپ يازغان مۇھاببەت ناخشىسنى،ئادەتتىكىچىە دەپ قارىغان،«كۇيئوغۇلنى تەبرىكلەپ.دوستتىن ئارىلما»ناخشسىنى بولسا ماۋزېدوڭ ياڭ كەيخۇيغا سوۋغا قىلغان،شۇ چاغدا ئۇ ياڭ كەيخۇي بىلەن توي قىلغىنىغا بىر يىلدىن ئاشقان،شۇ سەۋەپتىن ياڭ بىلەن ئۇنىڭ ئوتتۇرسىدا بىر مەيدان زىددىيەت تۇغۇلغان،بىراق تەخمىنەن مۆلچەرلىسەك ماۋزېدوڭ بىلەن تاۋسىيوڭنىڭ بۇ قېتىمقى كونا مۇھاببىتىنىڭ قايتا تېرىلگەن بولسىمۇ بىراق، ھەقىقى ئۇزگىرىش ھاسىل قىلالمىدى،چۇنكى ياڭ كىيىن خەت قالدۇرۇپ،ئۇنىڭ،ماۋزېدۇڭ بىلەن ئوتتۇرىدىكى«ئۇقۇشماسلىقنى تۇگۇتىش»نى ئېيىتقان.....

   ماۋزېدوڭ بىلەن تاۋسىيوڭ ئايرىلىپ كەتكەندىن كىيىن،خۇنەن ئۆلكىلىك ئۇقۇغۇچىلار ئۇيۇشمىسىنىڭ رەئىسى پېڭ خۇاڭزېڭ تاۋيىنى قىزغىن قوغلاشقان،بىراق رەت قىلشقا ئۇچرىغان.تاۋيى ئۇمۇر بۇيى ياتلىق بولمىغان،كىيىن شاڭخەيگە بارغان،ئۇزىنىڭ ھەممە كۇچىنى قىزلار تەربىيلەشكەئاتىغان،شاڭخەي،نەنجىڭ،چاڭشا قاتارلىق يەرلەردە قىزلار مەكتىپىنى قۇرغان،دىڭ لېڭ قاتارلىق بىر تۇركۇم ئايال شاگىرىتلارنى تەربىيلەپ چىققان.1932-يىللىرىنىڭ باشلىرىدا كىسەل بىلەن چاڭشادا ئالەمدىن

ئۆتكەن،شۇ چاغدا 37ياش ئىدى......
