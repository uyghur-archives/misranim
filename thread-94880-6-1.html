بۇنىڭ بىلەن ئۇ توربەت ياكى باشقىلار بىلەن نىمە مۇناسىۋىتى ؟نىمىنى ئەرىز قىلىسىز ؟سزمۇ رەسىم ياكى باشقا ئەسەر يوللىغاندا ئۇيغۇر ئىزاھاتىنى  توغرا بەرسىڭىز ئۇ ئىشلار يوق.يەنە بىر ئىش سىز شۇ ئىزدەش تورىدىن ئۇيغۇر پالانى دىگەننى كۆپ قىتىم ئىزدىگەندە ،چىكىلىش سانى ئاشقاندا ئىزدەش ماتورى سىزگە ھەتتا پالانى ،ئۇ ،بۇ دىگەننىمۇ سىزگە كۆرسىتىپ بىرىدۇ .مەسىلەن قىشلىق كىيىم دىگەن ئىسىمدە سىز قىشلىق ئىزھاتى برىلگەن نەرسلەرمۇ ،كىيىم دىگەن ئىزاھاتى بىرىلگەنلىرىمۇ سىزگە چىقىرىپ بىرىدۇ .بىزخەقمۇ غىت قىسمايمىز ،چىرايلىق بىر قىزنىڭ رەسىمىنى ئۇيغۇر ئىزاھاتى بىلەن تورغا چىقىرىپ قۇيىمىز ،ئاندىن ئۇنىڭ ئەسلى ئىسمى بىلەن ئىزدىسە بىرمۇنچە ئىشلار چىقىۋاتقان باشقا مىللەتلەر كۆرسە ھە بۇ ھىلىقى ئۇيغۇركەن دەپ  كەينىگە ئۇيغۇر پالانى دەپ 'ئىزاھات بىرىپ تورغا يوللاۋاتقان .قاراڭلار  يەنە بىر ئاھىدە سەۋەبمۇ بار بىزدە ئوتتۇرا ئاسىيا ،روسىيە ،ياۋروپا بەدەن ،چىراي قاش ،چاچ ئالاھىدىلىكى گەۋدىلىنىپ تۇرىدۇ ،شۇڭىمۇ باشقا مىللەتلەر بىزنى شۇ ياۋروپالىقلارغا ئوخشاش ئىدىيە ئازادلىقى ،بەدەن ئازادلىقى بار دەپ قايرايدۇ .شۇڭا باشتا ائۆزىمىز توغرا ئىش قىلايلى .
