  كۆپ خۇتۇنلۇق بۇلۇش ۋە ئۇيغۇر ئائىلە ئەخلاقى


    يېقىندىن بۇيان تۇردا كۆپ خۇتۇنلۇق بۇلۇش تۇغۇرسىدا مۇنازىرە  ئاۋۇپ كەتتى . بۇ خىل ھادىسە جەمىيەتتە بىر خىل ناتۇغرا بۇلغان ئىجدىمائى كەيپىياتنى پەيدا قىلدى.گەرچە كۆپ خۇتۇنلۇق بۇلۇش دىننى نۇختىدىن ئېلىپ ئېتقاندا تۇغرا بۇلمىسمۇ ،ئەممادىننىڭ بۇ گۇزەل ئەخلاقىنى تۇغرا چۇشنەلمىگەن بەزى بىرئەرلەرئۇزنى چۇشكۇنلۇككە باشلاپ ماڭدى . بىزنىڭ مۇقۇقەددەس دىنمىز بىر قىسسىم بىلىم قۇرۇلمىسى ئاجىز بۇلغان دىننى پەتىۋاچىلارنىڭ پەتىۋاىسى تۇپەيلى ،گۇزەل ئەخلاققا يىتەكلەيدىغان بۇ دىن بىر قىسسىم كىشلەرنىڭ پايدىلنىش قۇرالىغا ئايلىنىپ قالدى. ئەجىبا ئۇلار ئۇزىنىڭ بىر كۇنى ئاللاغا جاۋاپ بىردىغانلىقىنى ئۇيلىمامدىغاندۇ.‹‹نەپسىڭلارغا ئەگەشمەڭلار .شەيتان سىلەرگە ھەقىقەتەن ئاشكارە دۇشمەن ›› دەيدۇ قۇرئان كەرىمدە.
مۇقەددەس ئىسلام دىنى ئەرلەرگىلا ئەمەس ،ئاياللارغىمۇ ھۇقۇق بەرگەن .  ئىسلام دىنىدا ھەر قانداق بىر كىشگە زۇلۇم قىلىش قەتئى چەكلىنىدۇ . ئاللادىن قالسا بۇ دىندا ئادەم ئەڭ ئۇلۇغ .شۇنداق بۇلغان ئىكەن مەيلى ئايال بۇلسۇن ياكى ئەر بۇلسۇن ۋە ياكى باي-نامىرات ،چۇڭ-كىچىك كىم بۇلۇشىدىن قەتئى نەزەر ئۇلارغا زۇلۇم قىلىشقا بۇلمايدۇ.‹‹ كىمكى زۇلۇم قىلدىكان ياكى دىندا يۇق ئىشلارنى دىن نامىدىن قىلدىكەن ،ئۇ چۇقۇم ئازغۇنلارددىن بۇلدۇ .بۇ ئۇنىڭ ئۇزىگە قايتۇرلىدۇ››
ئاياللارمۇ دىن يۇلغا قۇيغان چەك ئىچىدەئۇزىنىڭ ھۇقۇقىدىن پايدىلانسا ئەلۋەتتە بۇلدۇ.شۇڭا ئەرلەر ئىككىنجى بىرخۇتۇننى ئېلىشى تۇغرا كەلسە ئۇزنىڭ ھازىر نىكايىدا بۇلغان ئايالىنىڭ ماقۇلىقىدىن ئۇتۇشى كىرىك . ئەگەر ئايالى قۇشالمىسا چىرايلىق نەسىيەت قىلىش كىرىك .ئۇنۇڭغا مۇقەددەس ئىسلامنىڭ شۇ تۇغۇرسىدىكى ئايەت-ھەدىسلەرنى دەلىل كەلتۇرۇپ ئۇنىڭ ماقۇللىقىنى  ئېلىش كىرەك . ئايەت-ھەدىسلەرنىڭ مەزمۇننى ئۇز نەپسى-خائىشقا ئەگىشىپ مەنىسىنى ئاشۇرۇش ۋە ياكى كەمەيتىۋېتىشكە قەتئى بۇلمايدۇ. ‹‹مۇشۇنداق ھادىسە بەك كۇپ ئۇچۇرۋاتىدۇ ››.ھەر قانداق بىر زۇراۋانلىق ۋاستىسى ۋە تەھتىت سېلىشقا قەتئى بۇلمايدۇ .  ئەگەر ئايال قۇشالسا ،تۇي مۇراسىمى ھەم تۇيلۇق بۇيۇمنى بىرىنجى ئايالىڭىزنىڭكىگە ئۇخشاش قىلشىڭىز كىرەك . ئەگەر ئككىنجى ئايالىڭىز ئاددى بۇلۇشنى ئىلتىماس قىلسا ئۇنى رازى قىلغان ئاساستا تۇي قىلسىڭىز بۇلدۇ .چۇقۇم بىرىنجى ۋە ئىككىجى ئايالىڭىز رازى بۇلۇشى كىرىك .ھەم ئۇلارنىڭ ئاتا -ئانىسى ئەگەر ئاتا-ئانىسى بۇلمىسا ئۇلارغا باش -پانا بۇلغۇدەك بىرەر ئۇرۇغ -تۇققىنى بۇ تۇيغا قۇشۇلىشى كىرەك.ھەم تۇي ئۇستىدە ئىمكان بار تۇرۇشى كىرەك . ھەم ئىككى ئايال بىر ئائىلىدە تۇرۇشى كىرەك . ھەم ئۇلارغا ئىمكاننىڭ بارىچە ئۇخشاش مۇئامىلىدە بۇلۇش كىرەك.ئەگەر ئۇلار ئايرىم تۇرۇشنى ئۇيلاشقان بۇلسا ئۇلارغا ئۇخشاش بۇلغان تۇرالغۇ ھازىرلاپ بىرلىشى كىرەك ھەم ئۇلارنىڭ يېنىدا تۇرۇش  ۋاختىنى ئادىل بۇلۇشىغا كاپالەتلىك قىلىش كىرەك .بۇ جەرياندا ئاياللارنىڭ ھەققى تۇلۇق ئادا قىلنىشى كىرەك.يۇقارقى ئىشلارنىڭ بەزىللىردە ئادىل بۇلالمىساڭلار ئاياللىرىڭلارنىڭ رازىلىقىنى ئېلشىڭلارھەم ئاللانىڭ رازىلىقىنى تىلشىڭلار كىرەك.بۇ جەرياندا ئاياللار ئۇتكۇزگەن گۇناھلار ئەر بۇلغۇچىنىڭ زىممىسىگە بۇلدۇ .بۇمەسىلنى ئەرلەر ياخشى ئۇيلىشى كىرەك.ئەرلەر ئايالنىڭ رازىلىقىسىز ئۇغۇرلىقچە خۇتۇن ئالسا بۇ دىندا بىردەك ئىتراپ قىلىنمايدۇ . ھەم تۇي بۇلغۇچى ئايالنىڭ ئاتا -ئانىسى بۇ تۇيغا رازىلىق بەرمىسە ئۇخشاشلا بۇ تۇي ئىتراپ قىلىنمايدۇ.دىن ئىتراپ قىلمىغان ئەر-خۇتۇنلۇق بىردەك زىنا بۇلدۇ.ھەم بۇ ئۇزگىگە ھەم ئۇزۇگە زۇلۇم قىلغانلىق بۇلدۇ.ئەگەرنىكاھىدىكى  ئايال يۆلدىشىنىڭ ئىككىنجى خۇتۇن ئېلشىغا قەتئى  قۇشۇلمىسا ئەر بۇلغۇچى سەۋىر قىلشى ياكى ئايالغا زۇلۇم قىلماي ئايالنىڭ ھەققىنى بىرىپ چڭرايلىقچە ئاجىرشىپ كىتىشى كىرەك .بۇ دىننىڭ ھۇكمى.

ئەمدى يەنە بىر مەسىلىگە كەلسەك ،ھازىر ئايالدىن ھەم جامائەتتىن يۇشۇرۇن خۇتۇن ئۇستىگە خۇتۇن ئېلىش ئاۋۇپ كەتتى.يەنە كىلىپ ئۇ ئاياللارمۇ ئەر ئۇستىگە ئەر تۇتۇۋاتىدۇ .ئىككىنجى خۇتۇن بارلىق بەھىرمان بۇلدىغان دۇلەتتىن بەھىرمان بۇلۋاتىدۇ .ماششىنا، ئۆي،ئالتۇن -كۇمۇش،قىممەت باھالىق كىيىم -كىچەك ،ياخشى يەپ -ئىچىش ئىككىنجى خۇتۇننىڭ .ئەمما جاپانى تەڭ تاتقان ئۆيدىكى خۇتۇنغا جاپا بىلەن تاپا تەنە بار .ئۇ ياخشى نەرسىلەردىن ئۆيدىكى خۇتۇنغا ئايدا ،يىلدا بىر قېتىم  با  يا  يۇق .  بۇ ھادىسە نۇرغۇن ئائىلىلەرنى ئەخلاق كىرىزىسىغا دۇچار قىلدى .نۇرغۇن ئەرلەرنىڭ تۇغرا ھەل قىلالماسىق تۇپەيلى ئۇغۇرلۇقچە قىلغان بۇ بىمەنە ئىشى ھازىر جەمىيەتتە بەزى بىر يىڭىدا ئازىراق پول تاپقان ئەخمەق-نادانلارنىڭ ئارسىدا ئۇمۇمىلىشىقا باشلىدى.
ئاخىرى بۇ ئەرلەر قىلغان بىمەنە ئىش نۇرغۇن بالىنى يىتىملىق كۇچۇسىغا تاشلاپ قۇيدىى .نۇرغۇن ئاياللار ئائىلىسىنى بۇزۇپ بۇ ئاتالمىش بەخىت يۇلنى تاللىۋالدى. شۇنداق قىلىپ بۇ خاتا يۇل ئۇيغۇرلارنىڭ ئائىلە ئەخلاقىنى پايخان قىلدى .ھەم بۇنۇڭغا تۇلگەن بەدەل ھەقىقەتەن بەك كۆپ بۇپكەتتى .مەسلەن .يىتىم بۇلغان بالا.تۇل قالغان ئايال.بۇزۇلىۋاتقان پۇل خۇمار قىزلار. بۇزۇلغان ئەقىدە . تۇكۇلگەن ياش......قاتارلىق ھادىسلەردىن شۇنى كۇرۋېلىشقا بۇلدۇ.

ئۇيغۇرلارنىڭ ئائىلە ئەخلاقى ئىسلام ئەقىدىسى بىلەن سۇغۇرلوپ ئۇزنىڭ ئائىلىدىكى رۇلنى تۇلۇق جارى قىلدۇرۇپ كەلگەن . بۇ ئەخلاق ئائىلىنى مەزمۇت تۇتۇپ تۇرالىغان .شۇ ئەخلاق بۇلغاچقا ئائىلىدە ئەر-ئاياللار بىر -بىرسىنى ھۇرمەت قىلىپ كەلگەن . بىر -بىرسىنىڭ ئائىلىدىكى ئۇرنىنى ھۇرمەتلىگەن .ئائىلىنىڭ يۇكنى ئۇرتاق زىممىسىگە ئالغان .ئەڭ ئادىل ۋە مۇۋاپىق بۇلغان ئائىلە ئىش تەخسىماتى پەيدا بۇلغان .شۇڭلاشقا بۇ ئۇيغۇرلارنىڭ ئائىلە ئەخلاقىنى بارلىق ئۇيغۇرلار بىردەك قۇغدىشىمىز كىرەك.
ئەجىبا نۇرغۇن كىشلەر زاماندىن قاقشايدۇ.زاماندا نىمە گۇنا .بىز ئەڭ ئاۋال ئۇزىمىزنى سۇيشىمىز،ئۇزىمىزنى ھۇرمەتلىشىمىز كىرەك. بىز ئۇزىمىزنىڭ ئادەتلىرىمىزنى ئۇزىمىز ھۇرمەتلىمىگەچكە باشقىلارمۇ بىزنىڭ ئادەتلىرىمىزنى ئىتىراپ قىلمايۋاتىدۇ.
بىز شۇڭا زاماندىن قاقشىماي قايتىدىن ئۇزلىكىمىزگە قايتىشىمىز ھەم ئۇزىمىزنىڭ ھەر بىر ئادەتلىرىمىزنى ئىمكان بار ئۇزىمىز ئىشلىتىشىمىز كىرەك.

خۇلاسە
بىز تېخې خۇتۇن داۋاسىغا چۇشكىلى بالدۇر .ئۇيغۇرلار ئۇيلىشىدىغان مەسلە بەك كۆپ .خۇتۇن داۋاسى مەسىلىسىدىن مۇھىم مەسلە خېلى بار. ئاۋال پۇتۇن ئۇيغۇرلار بىردەك ئالدىمىزدىكى ئەخلاق مەسىلىسنى ھەل  قىلشىمىز كىرەك.ئەخلاق ۋە ئەقىدە ھازىر پۇتۇن ئۇيغۇرلار بىردەك كۇڭۇل بۆلمىسە بۇلمايدىغان تەخىرىسىز مەسىلگە ئايلاندى. شۇڭا بارلىق ئاتىلار بىردەك ئائىلىگە مەسئۇل بۇلىشى . ئۇغۇل -قىزلىرغا مەسئۇل بۇلشى ھەم ئەقىدە ۋە ئەخلاققا يات ھەرقانداق بىر مەسىلە قايسى ئائىلىدە كۇرىلىشىدىن قەتئى نەزەر ،ئۇنىڭغا چۇقۇم ئەستايىدىل مۇئامىلە قىلنىشى كىرەك .ئەقىدە ۋە ئەخلاقنى قۇغداشنى ھەر بىر ئەدەم چۇقۇم بۇرۇچ تۇيغۇسى قىلىش كىرەك . مانا بۇ بىز كۇڭۇل بۇلدىغان ئەڭ مۇھىم مەسلە .
خۇتۇن داۋاسىغا كەلسەك چۇقۇم دىن ئىتراپ قىلدىغان نىكاھلىق ئائىلە قۇرۇڭ . ئۇغۇرلۇقچە خۇتۇن ئېلىشتىن ھەزەر ئەيلەڭ .
كىم كۆپ خۇتۇن ئالسا نۇچى ئەمەس بەلكى بەخىت بۇلغان .ئىمان بۇلغان .بەركەت بۇلغان .ئاللا رازى بۇلدىغان ،ئەل-جامائەت ئىتراپ قىلدىغان ،باللارنى ياخشى تەلىم -تەربىيەگە ئىگە قىلالىغان ئەر  ئەڭ  نۇچى . شۇندا ئائىلە ئەڭ بەخىتلىك ئائىلە .

نىكاھ كۈچكە ئىگە بولمايدۇ. ئۇنداقتا سىز سىزنى باشقۇرۇۋاتقان ھۆكۈمەتنىڭ مۇناسىۋەتلىك ئورگىنىدىن رەسمىيەت ئالماي توي قىلسىىڭىز ئۇ شەرتلەرگە تولۇق رىئايە قىلالامسىز؟؟؟ 3- ئەجەبا ، ئىككىنچى خوتۇن ئېلىشتا كاپىرنىڭ رەسمىيىتىنى ئالساق بولمايدۇ، دېگۈچىلەر بىرىنچى خوتۇنىنى كىمنىڭ رەسمىيىتى بىلەن ئالغان بولغىيدى؟ يانچۇقىدىكى كىملىك كارتىسى، تىجارەت كىنىشكىسى ۋە ھاكازالارنى بېجىرگەندە ئۇلارنىڭ ئىككىنچى خوتۇن ئالىدىغاندا تۇتۇۋالغان «تەقۋالىق غۇرۇرى» تېخى «پىشىپ يېتىلمىگەن» بولغىيمىدى؟4- بىر كىشىنىڭ بويىغا يەتكەن بىر قىزى ياكى بىر ھەمشىرىسى بولسا، خوتۇنى بار، بىرقانچە بالىسىمۇ بار، 30، 40 ياش ئەتراپىدا بىر «تاشچى ھاجىم» قىزلىرىنى ياكى ھەمشىرىلىرىنى ماڭا ئىككىنچى خوتۇنلۇققا بەرسىلەر، مەخپىي ئالىمەن، خوتۇن-بالىلىرىم ئۇقمايدۇ، دەپ خېرىدار بولۇپ كەلسە، ئەگەر ئۇ كىشىنىڭ ئەقىل-ھوشى جايىدا بولسا، پىترىتى بۇزۇلمىغان بولسا ۋە پۇلنىڭ بەندىسى بولۇپ كەتمىگەن بولسا، ئۇ تەكلىپكە ماقۇل دەرمۇ؟؟؟ئەلجاۋاب: ئەسلا ماقۇل دېمەيدۇ. دېمەككى، بۇ ئىشنى ئەقىل ئىنكار قىلىدۇ، ۋىجدان قوبۇل قىلمايدۇ، ئىنسانىي زوق تۈكۈرۈپ تاشلايدۇ، شۇنداقلا ئىسلامىي ئاڭ ئۇ ئىشتىن  يىرگىنىدۇ، پىترەت نەپرەت قىلىدۇ. 5- ئىككىنچى خوتۇن ئېلىشتا «بىز كاپىرغا ئىتائەت قىلمايمىز» دەيدىغان «مۇجاھىد» لار نېمىشقا ھەرەمگە بېرىش ئۈچۈن شۇ كاپىرنىڭ پاسپورتىنى ئېلىشقا ئۇنچىۋالا كۈچ چىقىرىدۇ؟؟؟ نېمە ئۈچۈن مۇناسىۋەتلىك كۇنسۇلدىن ۋىزا ئالىدۇ؟ نېمە ئۈچۈن ئىنسانلار بەلگىلىگەن ماكاندا، ئىنسانلار بەلگىلىگەن زاماندا، ئىنسانلار ياسىغان قانتاش ۋاسىتىلىرىغا چىقىپ مەككىگە بارىدۇ. نېمىشقا ئۆيىدىن چىقىپلا خۇسۇسىي ماشىنىسىغا ئولتۇرۇپ ئۇدۇللا ھەرەم تەرەپكە ماڭمايدۇ؟دېمەك، ئۇنداقلار ئىسلامنى ھايۋان نەپسىنى قاندۇرۇش، ئىناۋەت ۋە ئېتىبارىنى يۈكسەلدۈرۈش، يۈز-ئابرۇيىنى ئۆستۈرۈش… ئۈچۈن ئىشلىتىشكە تىرىشىۋاتىدۇ. ئىشنىڭ ئەسلى مۇشۇنداق بولغانلىقى ئۈچۈن ئۇلارغا بەش ۋاخ ناماز ئوقۇش ئېغىر كېلىدۇ، لېكىن ھەرەمگە بېرىپ «ھاجىم» بولۇشقا پۈتۈن كۈچى بىلەن تىرىشىدۇ. ئىككىلىسى ئوخشاش پەرزغۇ!خۇددى مۇشۇنىڭغا ئوخشاش ئۇنداقلارنىڭ پېقىر-مىسكىنلەرگە، مىللەتنىڭ ماددىي ۋە مەنىۋى مەدەنىيەتلىرىنى ساقلاپ قېلىشقا، دىيارىمىزدا ئىلىم- مەرىپەتنىڭ نۇرىنى يورۇتۇشقا پۇل بەرگۈسى كەلمەيدۇ، لېكىن «ئاللاھ رىزاسى ئۈچۈن» پېقىر قىز-چوكانلارنى نىكاھىغا ئېلىپ قويىدۇ. ئاللاھنىڭ دىنى بۇنداقلارنىڭ چۈشەنچە ۋە قىلمىشلىرىدىن ئادا-جۇدادۇر.
