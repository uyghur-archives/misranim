بۇ كۈيلەر سېنىڭ ئۆچۈن



   ھېچكىم ھېچكىمنى ئالدىيالمايدۇ ، پەقەت بىز ئۆز - ئۆزىمىزنى ئالدايمىز.. ھەئە، مەن جاراھەتتىن قورقمايدىغان ، مەغلۇبلۇققا باش ئەگمەيدىغان خۇددى سىلەرگە ئوخشاشلا .. ئېتىمنى ھاۋاغا قاراپ چاپتۇرىدىغان ، قۇيۇنغا ئەگىشىدىغان ، شامال بىلەن بەسلىشىدىغان ، جاھانغا پاتمايدىغان بىر بالا ئېدىم! نۇرغۇن قىزلارغا قانىتىنى كەرگەن ، يارىلاندۇرغان ھەم يارىلانغان يارا ئېدىم ؛ شۇ قېتىم مەن ناھايىتى ئېغىر بىر مەغلۇبىيەتكە ئۇچرىدىم ، دېرىزىدىن قاراپ تۇراتتىم ، مەيدەمنى قار دۆۋىلىرىگە ياقاتتىم ، شۇنچىلىك مۇڭلۇق ناخشىلارنى ئېيتاتتىمكى .. ئۇنى تۇنجى قېتىم شۇ ۋاقىتتا ئۇچراتقان ئېدىم...


مېنىڭ سۆيگىنىم


( ھېكايە )


ئۇنىڭ ئىسمىنى پەسىل ، قۇشلارنىڭ پەسلى ، ئۇ مېنىڭ سۆيگىنىم .
مەن ئۇنىڭ بىلەن تۇنجى قېتىم QQ دا سۆزلەشكەن ۋاقتىمدا ، باشقا ئىشلارنى ئويلىغان ، ئەمما مېنىڭ سۆيگىنىم بولىدۇ دەپ ئويلىمىغان . ئۇنىڭ ئەتلەس كۆينەك كىيىپ چۈشكەن رەسىمىنى ، ئۇ ماڭا بوشلۇقىنىڭ مەخپى نۇمۇرىنى ئېيتىپ بەرگەن كۈنى كۆردۈم .
_ چىرايلىق قىزكەن ، لېكىن كۆزى سەل ئۆلۈكمۇ قانداق ؟.. ،- بۇ مېنىڭ ئۇنىڭغا بولغان تۇنجى تەسىراتىم . كۆزلىرى چوڭ ئېدى ، لېكىن قۇيرىقى قىيىپ قويغاندەك .. سەل قىيغاق .. ھەم ئۇخلاپ قالغاندەك ..ئىشقىلىپ ، مەن ھېچقاچان خاتالىشىپ باقمىغان ، دېمەك ئۇ ئۆلۈك كۆزلۈك قىز ، ھەرگىز خاتا باھا بەرمىدىم ، دەپ ئويلايتتىم ، لېكىن كېيىنكى كۈنلەردە ئۇنىڭ ماڭا ئەڭ چوڭقۇر تەسىر قالدۇرغىنىمۇ ئۇنىڭ كۆزى .. ھېلىقى كېچە يىغلىغان ئېدى..  مەن ئۇ قەدەر سۈزۈك ھەم چوڭقۇر نۇرلۇق كۆزلەرنى ئەزەلدىن كۆرۈپ باقمىغان ئېدىم ، ئۇنى شەھلا كۆز دېيىش كېرەكمۇ ؟ ياق ، مەن شەھلا كۆزلەرنى كۆرگەن ، ئۇنىڭ كۆزى ھەرگىزمۇ شەھلا كۆزلەرنىڭكىدەك ئۇنداق ئېزىتقۇ ئەمەس .. ئۇ كۆز يوقلا بىر كۆز ئىشقىلىپ ، مەن كۆرگەن ئېدىم..
  كېيىن بىز ناھايىتى يېقىن ، خۇددى ياخشى كۆرۈشكەنلەردەكلا بوپ قالدۇق ؛ ھېچكىم بىز ياخشى كۆرۈشۈپ باقايلى ، دەپ ئېيتىپ باقمىغان ، بۇ ناھايىتى تەبىلا بىر ياخشى كۆرۈشكەنگە ئوخشاپ قالدى . كېيىن بىز پاراڭلاشتۇق ، ئۇنىڭ ئاۋازىنى ئاڭلاپ ئۆز - ئۆزۈمدىن ناھايىتى نۇمۇس قىلدىم ، قانداق قىلىقسىز گەپلەرنى قىلغانمەن ھە ؟ ۋۇ كىچىك جىن ، ئۇنىڭ ئاۋازى ھەرگىزمۇ 21 ياشقا كىرگەن قىزنىڭ ئاۋازى ئەمەس ! ئەزبىرايى خۇدا ، بەئەينى 11 ياشلىق قىزنىڭ ئاۋازى ئىدى . دەرھال سۆزلىگۈچنى ئۈزىۋەتتىم ، ئىچىمدە ناھايىتى خاپا بولدۇم .
_ ئاكا ، نېمىشقا ئۈزىسەن ؟ نېمىشقا گەپ قىلمايسەن ؟ نېمىنى خاتا قىلدىم مەن ؟ ،- ئۇ بەكلا خاپا ئېدى ، مەن ئۇنىڭغا ئۇنچىلىك پەرۋا قىپمۇ كەتمەي :
_ شۇمتەك ، ئاكاڭ بىلەن ئويناشقىنىڭما بۇ ؟ ماڭ بېرىپ ئاداشلىرىڭ بىلەن ئوينا ، 21 ياشمىش تېخى شۇمتەك ، تاياق قېپى ..
_ ھازىرنىڭ ئۆزىدە ئالمايدىغان بولساڭ ساڭا جىنىمنى تىكىمەن !
_ نېمەڭنى تىكسەڭمۇ ئالمايمەن شۇمتەك ، ماڭە بېرىپ شار ئوينامسەن ، تاش تەمەك ئوينامسەن ، شۇنى ئوينا!
    ئۇ تولىمۇ غۇرۇرى كۈچلۈك ، نەزىرى يۇقىرى بىر قىز ئېدى ، بەلكىم پۈتۈن ھاياتىدا تۇنجى قېتىم ئۇنى رەت قىلغان ئادەم مەن بولسام كېرەك ، قانچىلىك جېنىغا تەگكىنىنى ئۆزى بىلىدۇ ، بەلكىم ئاچچىقتىن بىر بۇردا بىر بۇردا بوپ كەتكەندۇ ، رەت قىلسام قىلسام ئېلىۋەرگەن ئېدى . ئاخىر بولماي سۆزلىگۈچنى قۇبۇل قىلدىم ، ئۇ ئاچچىقتا بۇغۇلغىنىدىم تىرەن تىرەن نەپەس ئالاتتى :
_ بىلىپ قوي ئاكا ، كىمىڭنى كۆزگە ئىلمايسەن ، مەن يەنە ئۈچ ئايدىن كېيىن 22 ياشقا كىرىمەن ، پالان پانى مەكتەپنىڭ پۇستانى كەسپىدە ئوقۇيمەن دەپ ساڭا دېمىدىمما ؟ سالاھىيەت كېنىشكامنى كۆرەمسەنيا ؟
    ئۇھ ، يۈرىكىمەي .. ئۇنىڭ ئاۋازى بايىقىدىن خېلىلا ياخشى چىقىۋاتاتتى ، ھەر ھالدا 22 ياشلىق قىزنىڭكىگە ئوخشىمىسىمۇ .. مەن كۈلۈپ تۇرۇپ ئۇنى دائىمقىدەك ئەركىلەتتىم ، تەسەللى بەردىم . تېلىفون نۇمۇرىنى ئېيتىپ بەرگىلىمۇ نىكەم ، مەن بىرمۇ قېتىم ئۇنىڭغا تېلىفون قىلىپ باققان ، ۋە ياكى ئۆزۈمنىڭ نۇمۇرىنىمۇ دەپ بەرگەنمۇ ئەمەس ئېدىم ، ئۇ مۇشۇ باھانىدە ھەممىنى يىغۇشتۇرۇپ ھارغىچە قاقشىۋالدى.
   بىر كۈنى ئۇ مەخسۇس مەن ئۆچۈن كامىرا سېتىۋالغىنىنى دېدى ، ئۇ بىر نورۇز بايرىمى كۈنى ئېدى ؛ چاچرىلىرىنى قىرىق تال ئۆرىگەن ( ئەمەليەتتە قانچە تال ئۆرىگىنىنى بىلمەيمەن ، بەلكىم يىگىرمە تال ئۆرىگەندۇ ، ھە ئۆزىنىڭ يېشىغا كەلتۈرۈپ 21 تال ئۆرىگەندۇ ، مەيلى قانچە تال ئۆرىسە، ئىشقىلىپ ئۇ ئۇزۇن ئۆتمەيلا 22 ياشقا كىردى ) ئاپپاق كوپتىسى شۇنچىلىك ياراشقان ، بېشىدا گىلەم دوپپا ، يۈزلىرى سۇس قىزىرىپ تۇرغان ناھايىتى ئوماق بىر قىز ئېدى .
   _ ھى  ھى ھى ھى ، ياخشىمۇ سەن ئوغرى چاشقان ،- ئۇنى  كۆرۈپ سۈيۈنىشىمنى ھەرگىز باسالمىدىم ، - ھەي ، بارمىقىڭنى چىشلىمە ، نېرى ئاپار.. ھەي قىز ، سەن قانچە ياش سەن زادى  ؟ نېمىشقا تىنماي بارمىقىڭنى چىشلەيسەن ؟ ،- ئۇ ھىلىدىن ھېلىغا بارمىقىنى ئېغىزىغا سۇناتتى ، بەلكىم بۇ ئۇنىڭ خىجىل بولغىنىدۇ لېكىن مەن ماڭا ئەركىلىدى دەپ چۈشەندىم . قىلىقلىرى بەئەينى ‹ قىزىل راۋاقتىكى چۈش › تىكى لىن دەييۈنىڭ ئۆزى ئېدى . ماڭا بۇ قىلىقلىرى بەك دىگۈدەك يېقىپ كەتمىدى .. ئىشقىلىپ ، نېمىلا بولسۇن ...ئەسلىدە خۇددى بىر ياۋايى بۆرىدەك ئۇنىڭ نازۇك قامىتىگە كۆزلىرىمنىڭ يېغى ئېقىپ كەتكۈچە قاراي دېگەن ، لېكىن پۈتۈن ئەسۇ يادىمنى ئۇنىڭ كوپتىسىنىڭ تۈگمىلىرى تارتىۋالدى . ئۇنىڭ تۈگمىلىرى ئاپپاق .. ياق ناۋاترەڭدە!جۇلالىنىپ تۇرغان ، خۇددى سەدەپتەكلا .. ياق ، يامغۇر تامچىلىرىغا ئوخشايتتى . گويا قايسى بىر ئۇستا ھۈنەرۋەن يامغۇر تامچىلىرىنى تۇتىۋېلىپ ئۇنىڭ كوپتىسىغا قاداپ قويغاندەكلا ...
   _ كىيمىڭ نېمانچە ئېسىل ؟
   _ ھە ؟ ،- ئۇ چۈشەنمىگەندەك قىزىرىپ تۇرىۋەردى .
   _ تۈگمەڭنى دەيمەن .. تۈگمەڭ يامغۇر تامچىلىرىغا ئوخشايدىكەن .
ئۇ تۈگمىسىنى ئويناپ ۋىلىقلاپ كۈلىۋەتتى دە ، ماڭا پۈتۈن ئىشقى بىلەن تىكىلدى ...
...
كېيىن بىز رەسمى سىرداشلارغا ئايلاندۇق ، بۇنىڭغا بىر ئايمۇ ۋاقىت كەتمىدى ، مەيلى كەتمىسە ، بىر كۆرۈپ كۆڭلىدە قالىدىغانلارمۇ بولغانغۇ ! ئۇ ماڭا تاكى ئېسىنى بىلگىنىدىن تارتىپ ھازىرغىچە بولغان جىمى ھاياتىنى ( ئەلۋەتتە قالدى قاتتىلىرىمۇ كۆپتۇ ) سۆزلەپ بولغاندەك قىلدى ، بىز خۇددى تۇنۇشقىلى بىر ئاي ئەمەس ، بەلكى ئېغى يوق ئىشتان كىيىدىغان زامانىمىزدىن تارتىپلا بىللە ئويناپ چوڭ بولغانلارغا بەكرەك ئوخشايتتۇق ، ھەم گۇمانمۇ قىلىشمايتتۇق . بۇرۇن ئۇنى پەقەتلا ياقتۇرمىغان بولساممۇ ، ھازىر ھەقىقەتەن ياقتۇرغان ئېدىم ، ئۇ بولمىسا ئىچىمنى بىر نەرسە تاتىلايتتى . ئۇنىڭ ماڭا بولغان ئىشەنچىنى ھەرگىزمۇ يەردە قويمىغىيدىم .. دەپ تىلەپ كېتەتتىم . مەن يالغان گەپنى قۇمدەك سۆزلەپ  باققان ، ئەمما ھېچكىمنىڭ ئىشەنچىنى يەردە قويۇپ باقمىغان ...
   ئىچىمدە دائىم ، خۇدا سېنى ماڭا يولۇقتۇرۇپ پاناھىدا ساقلاپتۇ ، باشقا بىرىگە ، ئالدامچىغا ئۇچرىغان بولساڭ قانداقمۇ قىلار ئېدىڭ ؟ .. دەپ خىيال قىلاتتىم دە ئۇنى تېخىمۇ بەكرەك قوغدىشىم ، قەدىرلىشىم ، ئاسرىشىم كېرەكلىكىنى ھىس قىلاتتىم .
   بىر كۈنى قاملاشمىغان ئىش بولدى ، ئۇ توغرا كەلمىگەنلا بىر ۋاقىتتا ‹ دەرسىم بار › دەپ قالدى ، مەن باشتا كۆپ ئويلىمىدىم ، لېكىن بىرى تېلفون قىلىپ باق دىگەندەكلا تېلىفون قىلدىم . ئەتراپى بەكلا قىيا چىيا بولمىسىمۇ ، ئىشقىلىپ تىنچ ئەمەس ئېدى ، دەرھاللا ئىچىمگە جىن كىردى :
   _ سەن دەرستە ئەمەسكەنسەن !
   ئۇ بىر نېمە دەپ ئېغىز ئېچىپ بولغۇچە يېنىدىن بىر ئەرنىڭ ئاۋازى ئاڭلاندى :
   _ كىمكەن ؟ ئەكىلىڭە تېلىفوننى ...
   ئۇنىڭغىچە تېلىفون قۇيۇلۇپ كەتتى ، ‹ غۇژژىدە › ئاچچىقىم كەلدى . قارا بۇنى ، مۇشۇ يېشىمدا مۇشۇ ماڭقىغا ئەخمەق بولدۇممۇ ؟!ئۇلاپلا يەنە تېلفون قىلدىم ، ئۇ ئەر يەنىلا يېنىدا .. تۈزۈك گەپلىشە گەپلەشمەي تېلىفون يەنە قويۇلۇپ كەتتى ، شۇ ۋاقىتتا غەزەپلەنگەن شىرنىڭ ئۆزىگە ئايلانغان ئېدىم ، زادى بۇ قىزلارنىڭ ئىچىدە ئوبدىنى يوقمۇ؟!! دەقىقە ئۆتمەيلا ئۇ ئۆزى تېلىفون قايتۇردى ، جان جەھلىم بىلەن ھۆكىرىدىم :
   _  ئاۋۇ يېنىڭدىكى لاتاغىلاپ كىمۇ ؟!
   _ ئاكا .. بۇ ..
   _ بېرە تېلىفوننى !! كىم ئۇ ئىتنىڭ پۇشتى ؟!.. ،_ ئۇنىڭغا گەپ قىلىشقىمۇ پۇرسەت بەرمەي دىگۈدەك ھاياجانلىناتتىم ، ئازابلىناتتىم ، ئەسەبىلىشەتتىم ..
   _ ۋاي مېنىڭ ساۋاقداشلىرىم يوقمۇ ! ،- ئۇمۇ جان - جەھلى بىلەن ۋاقىرىدى ،- مەن سىنىپتا ، سىنىپتا قىزلارمۇ ئوغۇللارمۇ بار ئاشۇنداق !!

   _ ماقۇل
   _ ئەمسە پۇشايمان قىلما !

   _ بېرىپ بولدۇڭمۇ ئاچا ؟
   _ ھەئە .

بەلكىم بۇ شۇ قىز شۇدۇ ، يا بەلكىم باشقا دوستلىرى بىلەنمۇ شۇنداق دىيىشىدىغاندۇ .. ئەلۋەتتە بۇلار مۇھىم ئەمەس . مۇھىمى بۇ قىزنىڭ ئاغزىنى قانداق ئاچۇرۇش ئېدى ، سەل دىققەت قىلمىسام چاناتتى ..

   _ مۆمىنجاننى دەيمەن ، - دېدىم مەن ئوغۇرلۇق قىلىۋاتقاندەك يۈرىكىم پوكۇلداپ ، دىمىسىمۇ ئوغۇرلۇقتەكلا بىر ئىش قىلىۋاتاتتىم . مۆمىنجان دېگەن ئۇنىڭ يۈرگىنىنىڭ ئىسمى ئېدى ..
   _ ۋوي ، يەنە نېمە بولدى ؟ پويىزغا قول تۇتۇشۇپ خۇشال چۈشۈپ كېتىۋاتقان ئەمەسمىدىڭلار ؟ يەنە ئۇرۇشۇپ قالدىڭلارما ؟

   _ يا.. ياق ، ئۇرۇشۇپ قالمىدۇق ، لېكىن ئا.. ئارافاتنى بىلىسەنغۇ ، شۇ تېلىفون قىلىپ بەك خاپا سېلىۋەتتى .

   ئېھ، قىزلار.. بۇ قىزلار.. نېمىشقا تاۋۇز تېرىمىغان قىزلار؟ نېمىشقا ئەر تاۋۇزلار دومىلاپ چۈشىدىغان تاۋۇز تېرىمىغان ؟ چېكىپ چېكىپ تازا خىللىۋالسا ھە ؟! كومپىيۇتېرنى يۇلۇپ ئېلىپ يەرگە غەزەپ بىلەن ئۇرۇپ شۇنچىلىك قاتتىق ، ھەم ئىچىنىشلىق ھۆكىرىۋەتتىم .
