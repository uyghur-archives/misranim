ئالىيە توپتۇغرا تۆت كۈن ھۇشسىز يېتىپ ئاخىر ھۇشىغا كىلىپ ئۈيىگەقايتتى
خالىدە دوختۇرخانىدىن چىققاندىن بىرى گەپ قىلىماس بۇلۇپ قالغان قىزىغا قاراپ ئازابلاندى ،قىزىنىڭ بۇنداق بوپ قېلىشنى ئاللانىڭ ئۈزىنى جازالىغىنى دەپ بىلدى ،ئۆز – ئۈزىدىن نەپرەتلەندى ، ئائىلىسىدىن  ئىرىدىن كۆڭلى قالدى ، ھەتتا قىزىنى دىمىسە ئۈلىۋالغۇسى بارئىدى ،ئەلۋەتتە ئۇنىڭ بۇ ئائىلىدىن بۇرۇنلا كۆڭلى قالغان ،ئۇ شۇنداق بۇزۇق ئايالمىدى ،ياق ئۇ ئىلگىرى ئىرىنى چىن دىلىدىن سۈيەتتى ھەتتا جېنىدىنمۇ ئەزىز بىلەتتى ،ئەكرەم ئىككىسى ئالى مەكتەپتە تۇنۇشقان ،شۇ كۈڭۈللۈك مەكتەپ ھاياتىنى بىللە ئۆتكۈزۈپ ئۇنتۇلغۇسىز ئەسلىمىلەرنى قالدۇرغان ،ياشلىقنىڭ مۇھەببەتنىڭ پەيزىنى سۈرگەن ،ئەكرەم خالىدەنى ئۆلگۈدەك ياخشى كۈرەتتى ،ھەر قېتىم ئۇنىڭ قاپقارا چۇڭ قۇي كۆزلىرىگە ، ئالمدىن سۈزۈك يۈزلىرىگە ،قاپقارا ئەگىم قاشلىرىغا ، تال چىۋىقتەك بەللىرىگە قاراپ گۇيا پەرىزاتنى كۈرىۋاتقاندەك ھىسياتتا بولاتتى ،مۇشنداق گۈزەل ئەخلاقلىق ،تىرىشچاق ئەقىللىق ،پەرىشتە سۈپەت ئاق كۈڭۈل قىزغا يۇلۇققىنىنى ئۈچۈن ئۈزىنى بۇ دۇنيادىكى بىردىنبىر بەخىتلىك ئىنسان ھىساپلايتتى ،دىمسىمۇ خالىدە ھەقىقەتەن ئەخلاقلىق ،ئاق كۈڭۈل بۇلۇپ بىر كۆرگەن كىشىنى مەھلىيا قىلىپ كىىچىلىرى چۈشەتكۈدەك چىرايلىق قىز ئىدى ،ئۇغۇل ساۋاقداشلىرىنىغۇ دىمەيلا قۇيايلى ھەتتا قىز ساۋاقداشلىرىمۇ < ھەي ،ئەگەر ئۇغۇل بالا بۇلغان بۇلسام بۇلاپ كىىتپ بولسىمۇ  چۇقۇم سىنى ئالاتتىم >دىيىشپ كىتتەتتى ،شۇنداق قىلىپ  ئۇقۇش پۈتتۈرگەندىن كىيىن ئۇ نۇرغۇن يىگىتلەرنى داغدا قۇيۇپ ئەكرەمگە ياتلىق بولدى،تويدىن كىيىن ئۇلار تېخىمۇ ئامراقلىشىپ كەتتى ،بىر – بىرىنى قەدىرلەپ ، ھۆرمەتلەپ كىشىنىڭ ھەۋىسىنى قۇزغۇغىدەك دەرىجىدە ئىناق بەخىتلىك ئۆتتى ئەمما شىرىن خاتىرجەم كۈنلەر ئۇزۇنغا بارمىدى ، خالىدە قىزىغا ئېغىر ئاياق بۇلۇپ ئۈچ ئاي بولغاندا ئەكرەمنىڭ خىزمىتى ئۈيىدىن 90كىلۇمىتىر يىراقتىكى بىر يېزىغا يۆتكەلدى ، قاتناش قۇلايسىز بولغاچقا ئەكرەم ئۇيەردە يېتىپ ئىشلەشكە مەجبۇر بولدى ،ئۇ ماڭىدىغان چاغدا ئامراق ئايالىدىن ئايرىلىشقا كۈزى قىيماي ھەتتا كۆز يېشىمۇ قىلدى ، ئەكرەم ئۈيىگە ھەپتىدە بىر كىلەتتى ، ھەر كەلگىنىدە خالىدە ئەكرەمنىڭ ئەڭ ئامراق تاماقلىرىنى ئىتىپ بىرىپ كۈتىۋالاتتى ،چىنىدەك پاكىز تازىلانغان ئۆي ،ئۇخشۇتۇپ ئىتىلگەن مىزىلىك تاماق ، ۋاپادار كۈيۈمچان چىرايلىق خۇتۇن ، يەنە تۇغۇلۇش ئالدىدا تۇرغان بالىسى ئەكرەمنى ئۈزىگە ماگىنىتتەك تارتىپ تۇرغاچقا ئۇ جۈمەنىڭ كىلشىنى تۆت كۈزى بىلەن كۈتەتتى ، بىر ئاماللارنى قىلىپ ئۈيىگە بىر مىنۇت بولسىمۇ بالدۇرراق كىتىشكە ئالدىراييتتى ،
ئەكرەم ھەر قىتىم مېڭىش ئالدىدا خالىدەنى چوڭقۇر مىھرى بىلەن ئاۋايلاپ قۇچاقلاپ تۇرۇپ <مىنىڭ پەرىشتەم ، سىز ئاللاھنىڭ ماڭا قىلغان ئەڭ چوڭ ھىممىتى ،سىزنى ئاللاغا تاپشۇردۇم ،مەن ۋە بالىمىز ئۈچۈن چۇقۇم ئۈزىڭىزنى ئاسراڭ >دىگەندەك مىھىرلىك سۆزلەر بىلەن خۇتىنىغا بولغان مۇھەببىتىنى ئىپادىلەيتتى ،خالىدەمۇ كۈزىگە ياش ئېلىپ قىيمىغان  ھالدا ئۇنى ئۇزۇتۇپ قۇياتتى ،ئەمما ئاخرقى ھىساپتا بارلىق تىرشچانلىقلار يەنىلا نۆل بولدى ،ئەكرەم بارغانچە ئۆيگە ئىكىكى ئۈچ ھەپتىدە بىر كىلىدىغان بولدى ،ھەتتا بەزىدە ئايلاپ ئۆيگە قايتمايتتى ،نىمىشقا ؟ چۈنكى ئۇ ئازدى ،تالاغا قارىدى ،چىرايلىق كۈيۈمچان ئايالىغا  يۈزسىزلىك قىلدى تۇغۇلۇش ئالدىكى بالىسىنڭ يۈزىگە قارا سۈركى،ئائىلىسىدىن ئايرىلىپ شۇ يىراق جاپالىق رايۇندا ئىشلەش ئۇنىڭغا ھەقىقەتەن ئېغىر كەلدى ، ئىشلەپ چارچىغاندا مۇزدەك سۇغۇق ياتاققا قايتىش ، ئۈچ ۋاق ئاشخانىنىڭ تەمسىز تاماقلىرىنى يىيىش ،ئۇزۇن كىچىلەرنى يالغۇز ئۆتكۈزۈشكە ئەكرەم تۇلىمۇ تەستە كۆندى ،ئۇ دائىم ياتىغدا يالغۇز ئئولتۇرۇپ خالىدە بىلەن بىللە ئۆتكۈزگەن ياشلىق ھاياتىنى ، بەخىتلىق تۇرمۇشىنى ، يەنە تۇغۇلۇش ئالدىدىكى بالىسىنى ئۇيلاپ كۆڭلى يېرىم بولاتتى ،< ئەگەر قاتناش قۇلايلىق بولغان بولسا ھە ؟! گاچچىدە بىرېىپ خالىدەنى بىر كۈرۈپ كەلسەم ،ئۇنىڭ قۇسىغىنى سىلاپ تۇرۇپ بالام بىلەن قانغىچە مۇڭدىشىۋالسام ،نىمە دىگەن ياخشى بۇلاتتى >دىگەنلەرنى ئۇيلايتتى ،شۇنداق كۈنلەرنىڭ بىرىدە ئەكرەمنىڭ خىزمەددىشى ئالىم بىرنەچچە دوسلىرىنى باشلاپ كاۋاپ ، ھاراق ۋەباشقا بىرمۇنچە يىمەكلىكلەرنى ئېلىپ رۇخسەتسىزلا ئەكرەمنىڭ ياتىغىغا رەڭلىك ئۇلتۇرۇش قىلىش ئۈچۈن كىرىپ كەلدى، بۇ يازمىنى ئاخىرىدا   يولتۇز تەھرىرلىگەن. ۋاقتى  2013-1-5 10:34 PM
