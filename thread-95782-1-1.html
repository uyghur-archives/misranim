       مەن بىر دوستۇمنىڭ ئۆيىگە بىرىپ كىتاپتىن مۇنۇ ھېكايىنى كۆرۈپ قالدىم ئۇقۇپ باقسام يامان ئەمەس ئىكەن ، شۇڭا تورداشلىرىم بىلەن تەڭ بەھىرلىنەيلى دەپ مۇنبەرگە ئىلىپ كەلدىم ... كۆرۈپ بىقىڭ ....

                           ياخشىلىق ھامان يەردە قالمايدۇ   


       بىر مۇزدۇز قارا قىشنىڭ بىر كۈنى يايمىسنى يىغىشتۇرۇپ ئۆيىگە قايىتماقچى بولغىندا ، بىر بالىنىڭ بىر بولۇڭدا يىغلامسىراپ ئولتۇرغانلىقىنى كۆرۈپ قاپتۇ .بالا بەكلا توڭلاپ كەتكەچكە ،موزدۇز ئۇنى ئۆيىگە ئىلىپ كېتىپتۇ ،بىراق ئويلىمىغان يەردىن موزدۇزنىڭ خوتۇنى ئۇنى كېچىچە تىللاپ  چىقىپتۇ .  - بۇنداق بالىنى كىم بېقىۋالىدۇ ؟ سىرىتتا بۇنداق بالىلارئىنتايىن كۆپ ، ئۇنىڭ ئۈستىگەبىزنىڭ كۈنىمىز ئاران تۇرسا ،بۇ بالىنى قانداقمۇ بىقىپ بۇلالايمىز ؟  - دەپتۇ خوتۇنى چالۋاقاپ .  ئۇلار شۇنداق كۆپ تالاش -تارتىش قىلىشقان بولسىمۇ ، موزدۇز يەنىلا بۇ بالىنى ئۆيىدە قوندۇرۋاپتۇ ، ئەتىسى موزدۇز بالنىڭ ئاتا-ئانىسنى سۈرۈشتە قىپتۇ .كۈنلەر شۇ تەرىقىدە ، ئىككى ئاي ئۆتۈپ كەتكەن بولسىمۇ ، بالىنىڭ ئاتا ئانىسى تېپىلماپتۇ . موزدۇزنىڭ خوتۇنىمۇ بارغانسىرى بۇ بالىغا ئامراق بولۇپ قاپتۇ ھەمدە بۇ بالىنى بېقىۋىلىش ئويىغا كەپتۇ .ئەمما ، موزدۇز بۇ بالىنىڭ ئاتا - ئانىسنى  ئىزدەشنى توختۇتۇپ قويماپتۇ .بىر كۈنى بۇ بالىنىڭ ئاتا - ئانىسى بۇ خەۋەرنى ئۇقۇپ ، ئۇنى ئىزدەپ كەپتۇ ۋە رەخمەت دەپمۇ قويماي ، بالىنى ئىلىپ كىتىپتۇ . كىيىن موزدۇز ئۆيىدىكىلەر بىلەن بۇ شەھەردىن كۈچۈپ كىتىپتۇ ،ئۇ ھىلىقى بالىنىڭ بىر كۈنلەردە ئۆزلىرىنى يوقلاپ كىلىشنى ئۈمۈد قىپتۇ ، بىراق بۇ بالا شۇ كەتكەنچە قايتا كەلمەپتۇ . ئارىدىن ئون نەچچە يىل ئۆتۈپ ، بىر يىگىتنىڭ توردا مەخسۇس ئادەم ئىزدەش ئاساس قىلىنغان بىر تور بېكتىنى قۇرغانلىقى ھەققىدە خەۋەر تارقىلىپتۇ . قىزقارلىقى ، ھىلىقى موزدۇزنىڭ ئىسمى بۇ تور بېكەتنىڭ ئىسمى قىلىنغان ئىكەن . بۇ تور بېكتىگە كىرسىلا كۆزگە چېلقىدىغان يەرگە بۇ تور بېكەت قۇرغۇچىسى چىقىرىپ قويغان ئادەم ئىزدەش ئېلانىنى كۆرگىلى بولىدىكەن ، ئۇنىڭ ئىزدىمەكچى بولغىنى دەل كۆپ يىللار ئىلگىرى ئۇنىڭغا غەمخورلۇق قىلغان ۋە ياردەم بەرگەن ھىلقى موزدۇز ئىكەن . موزدۇز بۇنىڭدىن خەۋەر تېپىپ ،ئاخىرى ھېلىقى بالا بىلەن كۆرۈشۈپتۇ . بالا ئۇنىڭغا ئۆزىنىڭ كاتتا باي بولۇپ كەتكەنلىكىنى ئېيتىپتۇ ھەمدە ئۇنىڭغا بولغان رەھمىتىنى بىلدۈرۈش ئۈچۈن ، بۇ موزدۇزغا كاتتا داچىدىن بىرنى سوۋغا قىپتۇ .
          ياخشىلىق ھامان يەردە قالمايدۇ ،شۇڭا قولىڭىزدىن كىلىشچە ياردەم قىلىڭ ،  
ھايات مانا مۇشۇنداق تەسىرلىك ھېكايىلەر بىلەن ھەقىقى قىممىتىنى تاپالايدۇ .
                                - ئامانگۈل مۇھەممەت تەرجىمىسى .

                                    يۇمۇرلار  

                               مىراس دەۋاسى

    بىر ئەر سەكراتقا چۈشۈپ قالغاندا ،خوتۇنى ۋە بالا- جاقا، ئۇرۇق - تۇققانلىرىنى يىغىپ ، مىراسنى كۆزىنىڭ ئۇچۇق ۋاقتىدا بۆلىۋەتمەكچى بوپتۇ . شۇ ئەسنادا بىر ئەر ئۇلارنىڭ ئارىسدا پەيدا بولۇپ قاپتۇ . - سىز كىم ؟ دەپ سوراپتۇ سەكراتتىكى ئەر بوش ئاۋازدا، - بۇ يەردىكىلەرنىڭ ھەممىسى ئۇرۇق - تۇققاندارچىلىق مۇناسىۋىتى بارلار . - مەن خوتۇنىڭىزنىڭ بولغۇسى ئىرى ، شۇڭا مىنىڭمۇ سىلەر بىلەن تۇققاندارچىلىق مۇناسىۋىتىم بار، - دەپتۇ ھىلىقى ئەر .

                              ئەپچىل جاۋاپ

مۇئەللىم ؛ ئاچلىق ئېلان قىلىش دېگەن نىمە ؟
ئۇقۇغۇچى ؛كۆپرەك تاماق يېيىش ئۈچۈن ۋاقتىنچە تاماق يېمەسلىك ھەركىتى .     

                            پەرىق
      ئۇقۇتقۇچى دەرىس ۋاقتىدا ئۇقۇغۇچىلارغا ؛ << ھەممىمىزئوخشاش، سىلەر بىلەن مەنمۇ ئوخشاش .....>> دەپ تازا سۆزلىگىلى تۇرۇپتۇ .  
    - مۇئەللىم ، سىز ھەرگىز بىزگە ئوخشىمايسىز ، دەپتۇ بىر ئۇقۇغۇچى .
- ئۇنداقتا ، دەپ باقە ، سەن بىلەن مىنىڭ نىمە پەرقىم ؟  - دەپ سوراپتۇ .
- سىز دىگەن ئىمتاھان ۋاقتىدا تۆت ئەتراپقا قارىيالايسىز ، مەن ئىككى يىنىمغىمۇ قارىيالمايمەن ....

             مەنبە ؛  ئامانلىق ژۇرنىلىدىن ئىلىندى .

ياخشىلىق ھامان يەردە قالمايدۇ ،شۇڭا قولىڭىزدىن كىلىشچە ياردەم قىلىڭ ،  
ھايات مانا مۇشۇنداق تەسىرلىك ھېكايىلەر بىلەن ھەقىقى قىممىتىنى تاپالايدۇ .
ياخشى بولساڭ ئېشىڭنى يەيسەن ، يامان بولساڭ بېشىڭنى يەيسەن .
ياخشىلىق قىل قولۇڭدىن كەلسە  .

قۇرئان ۋە ھەدىسلەردە ئىسپاتلاندىكى : بىر ئادەم ئۆلسە ، ئۇ بۇ دۇنيادا قىلغان ياخشى بولسۇن يامان بولسۇن ، بارلىق چوڭ – كىچىك ئىشلىرى ئۈچۈن ھىساب – كىتاب قىلنىدۇ   . ياخشى ئىش مۇكاپاتلىنىدۇ ، ئەسكى ئىش جازالىنىدۇ ، ھىساۋاتنىڭ بىرىنچى سەھنىسى قەبرىدە بولۇدۇ  .

دوستۇڭ ھاراقكەش بولسا يىيىشىڭ مۇش  .

غەيۋەتنىڭ ياخشىسى يوق  .

قاننى قان بىلەن يۇغىلى بولماس .

كىشىگە قىلساڭ يامانلىق ، تاپالمايسەن ئامانلىق .

نىيىتى ياماننىڭ قازىنى تۆشۈك .

ئورىنى كىم كولىسا ئۆزى چۈشۈدۇ .

يامان قالار ئۇياتقا ، ياخشى يېتەر مۇرادقا .

قۇرئان ۋە ھەدىسلەردە ئىسپاتلاندىكى : بىر ئادەم ئۆلسە ، ئۇ بۇ دۇنيادا قىلغان ياخشى بولسۇن يامان بولسۇن ، بارلىق چوڭ – كىچىك ئىشلىرى ئۈچۈن ھىساب – كىتاب قىلنىدۇ   . ياخشى ئىش مۇكاپاتلىنىدۇ ، ئەسكى ئىش جازالىنىدۇ ، ھىساۋاتنىڭ بىرىنچى سەھنىسى قەبرىدە بولۇدۇ  .
