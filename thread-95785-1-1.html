                                    مىنى ئۇنداق كارى يوق تاشلىماڭلار باللىرىم ..                          
                                        <بىر ياشانغان بوۋاينىڭ تىلىدىن>.  
                                                            <1>
                                    كەتتى بەكمۇ ئۇزاققا قايتماس ياشلىق چاغلىرىم .
                                   يىگىت ئىدىم قېرىدىم ئاپئاق ساقال چاشلىرىم
                                   زەئىپلەشتى مەن بىلەن ئوپكە جىگەر تاللىرىم .
                                   بۇرۇنقىدەك شاش ئەمەس ئوركىشلىمەس قانلىرىم .
                                   تاشلىۋەتمەڭ مەن مىھمان .ئۇزاق ئەمەس چاغلىرىم .
                                  كارىڭىز يوق مەن بىلەن تاشلىماڭلا باللىرىم.
                                                          2.
                                  بالاڭ بولدى ئەي ئوغلۇم ئاتاڭ بارمۇ يادىڭدا؟
                                  بىر تال سامسا ئالمىدىڭ ئاقچاڭ يوقمۇ يانىڭدا.
                                  ئاناڭ كىتىپ ياشقىنە سىنى باققان باغرىمدا .
                                  ئاق توقاچنى سەن يىگەن مەن تويۇنغان زاغرىمدا.
                                  ئوتۇن سېتىپ گوش بەرگەن ئۇن قالمىسا تاغرىمدا.
                                  مىنى باخماي ئانداقلا تاشلىماڭلا باللىرىم.
                                                       3.
                                  مەن چوڭلانى باققانتىم كەتكۇزمىەگن يېنىمدىن.
                                  دۇئاسىنى ئالغانتىم ئەزىز كورۇپ جېنىمدىن.
                                  بۇغداي ئېشىم بولمىسا بال تامدۇرۇپ تىلىمدىن .
                                  نەسىھەتنى ئاڭلايتىم ياش ئېقىتىپ كۇزۇمدىن.
                                  مۇمكىن بولسا جېنىمدا قۇتقازسام دەپ ئۇلۇمدىن.
                                  مىنى باخماي ئانداقلا تاشلىماڭلا باللىرىم.
                                                    4.
                                  كۇندە ماراپ تۇڭلۇكتىن ئەزرائىلنى ساخلايمەن .
                                  تەرەت تۇتماق تەس ئىكەن تەننى قانداق پاكلايمەن؟
                                  سۇ ئالالماي قاپاققا قېرىلىقتىن قاخشايمەن.
                                  بالام يېقىپ بىرەر دەپ ئوچاققا ئوت ياقمايمەن.
                                  كۇندە توڭلاپ ياتىمەن سوقتىن كىمگە داتلايمەن؟
                                  مىنى باخماي ئانداقلا تاشلىماڭلا باللىرىم.
                                               5.
                                  مىنىڭ بىلەن كارىڭ يوق يازدا كەتتىڭ سەيلىگە.
                                  چىقالمىسام ھويلامدىن مىنى قويۇپ مەيلىگە.
                                   يالغۇزلۇقتا مەجنۇن بوپ تەشنا بولدۇم لەيلىگە.
                                  چۇشەنمىدىڭ ئاتاڭنى يەتمەي خۇيى پەيلىگە.
                                  كىيىمىم كىر نىگىردەك ئوخشاپ  قالدىم تەلۋىگە.
                                  مىنى باخماي ئانداقلا تاشلىماڭلا باللىرىم.
                                               6.
                                  سىزمۇ مەندەك قېرارسىز سىزگە مۇھتاج بولارسىز.
                                  مەندەك تۇل بوپ قالسىڭىز دەرتنى كىمگە تۇكەرسىز
                                   جۇپ بولمىسا ھەم دەرتلىك كىمگە يىغلاپ كۇلەرسىز.
                                   مىنى ياخشى باخسىڭىز قېرغاندا كۇرەرسىز
                                   مەن خوش بولسام سىزمۇ ھەم قېرغاندا بىلەرسىز.
                                   مىنى باخماي ئانداقلا تاشلىماڭلا باللىرىم
                                                      . 7 .
                                    قۇرئان ئوقۇڭ بېشىمدا تەسەللىگە تەشنامەن.
                                    قايتارمەن مەن مىھماندىن ئولۇم بىلەن ئاشنامەن .
                                    ئوغرى كىرسە ھوجرىغا قارانچۇقمەن قاشامەن.
                                    دەپ قالۇرسىز ۋاي ئىسسىت باقماپتىمەن باشتا مەن
                                    خۇدا بەرسە جەننەتنى باياشاتتا ياشامەن
                                    كارىڭلا يوق بىز بىلەن تاشلىماڭلا باللىرىم.

بەزىلەرنىڭ تېخى نۇمۇس قىلماي ياشانغان ئاتا-ئانىسىنىڭ ھۆكۈمەت بەرگەن ئازغىنە تۇرمۇش پۇلىنى بىرىپ تۇرۇڭ، ئەتىلا قايتۇرۇپ بىرەي دەپ ئېلىۋېلىپ قايتۇرۇپ بەرمىگىننى، ئاتا-ئانىلىرى ئامالسىز: مەن پالانى نەرسە ئالاي دىگەنتىم بولسا بەرگەن بولسىڭىز دەپ سورىسا يائاللا، ئالىدىغىنىڭىز مۇشۇ ئىككى تال پۇلغۇ دەپ ئېتىپ دوق قىلىپ بەرگىننى ئاڭلاپ بەك غەزىپىم كەلدى،
مۇۇشۇ جەمىيەتتىكى مېھىىبان ئانىلار، مېھىرىبان ئاتىلار ۋە ئۇنىڭدىن باشقا ياخشى نىيەتلىك سخاۋەتلىك قوللارمۇ ياردەمنى بالىارغىلا قىلىدىكەن، مۇمكىن بولسا ياشانغانلارغىمۇ ياردەم قىلىدىغانلار چىقسا دىگەن ئۈمىتتىمەن
ئوغرى گۇمانخور، يالغانچى قەسەمچى كىلىدۇ.
