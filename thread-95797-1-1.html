  ھۆرمەتلىك تورداشلار : بۇ يەردە سىلەرنىڭ بىر ئىش ئۈچۈن ئەسكەرتمەكچىمەن . بۇ ھازىر ھەممەيلەن ئومۇمىيۈزلۈك ئۇچرايدىغان ئەھۋال شۇڭا ھەممىڭلارنى مەن ئۇچرىغان ئەھۋالغا ئۇچراپ قالمىسۇن دەپ ئويلايمەن .
  ھازىر ئۆي زىننەتلەش بىردە ئەڭ ئەۋج ئالغان بىر ئىش بۇلۇپ قالدى. بۇنىڭدىن ئۈچ ئاي بۇرۇن بىزمۇ ئۆي زىننەتلەش ئۈچۈن تۇتۇشقان ئىدۇق. مىنىڭ ھەدەم بىر تۇنۇشىنىڭ تۇنۇشىنى ئېلىپ كەلگەن. ئۇ ئادەم ئۇلارنىڭ شۇنچىلىك ئىشەنچلىك شۇنچىلىك ياخشى قىلىدۇ دەو ھەم ئۆزى ئۈستىدە تۇرۇپ ئۆينىڭ زىننەتلىشىپ بەرمەكچى بولغان. بىز بىر يېرىم ئايغا توختاملىشىپ 45 مىڭ كويغا پۈتۈشكەن. بىز پۈتكىچە قىش كىرىپ قالىدۇ دەپ دەرىزە ۋە پارلارنى ئالماشتۇرماسلىقنى تەلەپ قىلغان. ئۆيىمىز 60 كىۋادىرات مېتىرلىقلا ئۆي ئىدى.
    شۇنداق قىلىپ ئۇلار ئىشنى باشلىغان. بىراق بىر يېرىم ئايدا ئەمەس ئۈچ ئاي كەتتى. ۋاقىت كەتسىلا مەيلىغۇ، ئۇلار بىزنى جاندىن تويغۇزۇپ ئۆلىۋېلىشقا مەجبۇر قىلغىلى تاسلا قالدى. ئىشقىلىپ قاشاڭ بولا شۇنداق بولا . دادام ياشانغان ئادەم بولغاچقا ئۇنى تۇققانلارنىڭ ئۆيىگە ئاپىرىپ قۇيۇپ ئۆي پۈتكەندە ئەكىلىۋالماقچى بولغان. ئىشقىلىپ ئاپام ئىككىمىزنى ئاجىز قىز ئاياللار دەپ بوزەك قىلدىمۇ بىلمىدىم، بىر كۈن ئىشلەيدۇ، ئون كۈن يوق ، بەش كۈن چىداپ ئىشلەپ قالسا 20 كۈن يوق ، تىلفون قىلسا زىنھار ئالمايدۇ.ئىشقىلىپ ئاچچىقتىن كېسىلىپ قالاي دەپ قالدىم . قىلىڭلار دىگەن ئىشنىڭ بىرنىمۇ قىلمىدى. مەن ئۇلارنىڭ چالا-پۇچۇقلىرىنى دىسەم ئون بەت يازسام بولىدۇ. ئەڭ ياخشىسى دىمەي.
   ئەڭ ئاخىرىدا قانداق قىلدى دىسەڭلەريېرىم ئىشنى تاشلاپ كېتىپ قالدى. بىز باشقىلارنى ئەكىلىپ قالغىنىنى قىلغۇزماقچى بولدۇق ، باشقىلار ئۆيگە كىرىپ شۇنچىلىك ئەپسۇسلىنىپ كەتتى. پۈتۈن ئورنىتىلغان يەرلەر خاتا، يا ئۇ يەرلەرنى چېقىشنىڭ ئورنى يوق. ئىشكاپقا سىرنى بىرلا قېتىم بەرگەن. ماتېرياللىرى شۇنچىلىك ناچاركەن. ئىسىت ... دىيىشىپ كەتتى ھەممەيلەن.  
    بىز ھەممىمىز ئۇيغۇر ،مۇسۇلمان بولغاندىن كېيىن دەپ ئۇلارغا شۇنچىلىك ئىشىنىپ كېتىپتىمىز، پۇلنىڭ ھەممىنى بېرىپ بولغان . ئاخىرىدا بەش مىڭنى ئېلىپ قالسام ماتىريال يوق دەپ بىر ھەپتە ئىش تاشلاپ يېتىۋالغاننىڭ ياقى بوپتۇ دەپ پۇلنى بەرگەن. ئاپاممۇ ئۇلارنىڭ ئالدىدا تولا يىغلاپ يالۋۇرۇپ كەتتى. ئىشقىلىپ  ئۆمرۈمدە بۇنداق چارچاپ بۇنچىلىك ئاچچىق يۇتۇپ باقماپتىكەنمەن. بىر كۈنى ئەتتىگەندە تېخى يۈرۈگۈمگە سانجىق تۇتۇپ قېلىپ يېرىم سائەتتەك قوپالماي قالدىم .
     ئۇلار ئاساسلىق ئۈرۈمچى، غۇلجا تەرەپلەردە ئىش قىلىدىغان ئۆي زىننەتلەتلىگۈچىلەر. شېرىك بولغان ئىككەيلەن بار ئىكەن ، بىرىنىڭ ئىسمى مەۋلان 25-26 ياشلاردا بار. غۇلجىلىك، يەنە بىرىنىڭ ئىسمى باقى ھاجى 27-28 ياشلاردا بار ئاتۇشلۇق. بۇلار  ئەسسالامۇ ئەلەيكۇم  بىلەن ھە مەقۇل  دىگەن گەپنى ئاغزىدىن چۈشۈرمەيدىغان خەقكەن. توۋا....ئىشقىلىپ بۇ ئادەملەرگە ئىشىنىپ ئۆيۈڭلارنى تاپشۇرۇپ سالماڭلار، بولمىسا پۇلۇڭلارغا ،سالامەتلىكىڭلارغا زور دەرىجىدە زىيان سېلىۋالىسىلەر .
      بىر دوستۇمغا دەردىمنى تۈكۈپ بەرسەم ماڭا مۇشۇنداق قىلىپ كۆپچىلىكنى ئاگاھلاندۇرۇپ قۇيۇشنى دىدى، شۇڭا بۇ يازمىنى يازدىم. كۆپچىلىك بۇ كىشىلەردىن قاتتىق ھەزەر ئەيلەڭلار.
