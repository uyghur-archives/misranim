                                                مسرانىمدا يىگىتلەرگە قۇشاق كەتتى

    (قاراپ باقسام مىسرانىمدا ساپلا قىزلا قۇشاققا قىتىلىپتۇ،شۇڭا مەن يىگىتلەرنىمۇ بىر قۇشاققا قىتىپ قۇياي دېدىم .بۇپتىمۇ قىزلار؟)


                                              مىسرانىمدىن ئۈچۈپتۇ،

                                              يىگىتلەرنىڭ ئىزلىرى.

                                              قورۇقتىمۇ قىزلادىن؟

                                              يوق بۇپ كەتتى سۆزلىرى.


                                              مىسرانىمدا يىگىتلەر،

                                              بەكمۇ مەغرور سۆزلەيتتى.

                                              ئىككى گەپنىڭ بىرىدە،

                                              قىز ئوۋلاشنى كۆزلەيتتى.


                                             جەڭ ئېلان قىپ قىزلاغا،

                                             جەڭ تۈگۈمەي قىچىپتۇ.

                                             ھەر قايسىسى ھەر ياقتىن،

                                             مۈكۈشكە جاي ئىزدەپتۇ.


                                             يىگىتلەرنىڭ قىزلانى ،

                                             كەلتۈرۈشكە ئىپى كۆپ.

                                             يانچۇغىدا پۇلى يوق،

                                             شېرىن-شىكەر گىپى كۆپ.


                                            مىسرانىمغا قىزلارنىڭ ،

                                            سەرخىللىرى يىغىلغان.

                                            ھەر بىرى بىر قىزىلگۈل،

                                            باھار پەسلى ئىچىلغان.


                                           ئالدىماق بىز قىزلانى،

                                           ئاسان ئەمەس بىلىڭلا،

                                           ئېغىزدىكى لاپ گەپكە،

                                           گول بولمايمىز زادىلا.


                                          قۇرۇق گەپنى تۇلا قىپ،

                                          يىگىتلەر بەك ھېرىپتۇ.

                                          كەلتۈرەلمەي قىزلانى،

                                          بىشى بەكمۇ قىتىپتۇ.


                                          ئاكا-ئۇكا باش قۇشۇپ،

                                          يىڭى پىلان تۈزۈپتۇ.

                                          كىلىشكەن بىر يىگىتنى،

                                          رول ئېلىشقا تاللاپتۇ.


                                         تاللاپ چىققان يىگىتنى،

                                         بۇلار راسا ياساپتۇ.

                                         داڭلىق بىر ماشىنىنى،

                                         ئارىيەت ئېپ راسلاپتۇ.


                                        شۇنداق قىلىپ ئۇلارنىڭ،

                                        تەييارلىقى پۈتۈپتۇ.

                                        رولغا چىققان ئۇ يىگىت،

                                        ئالدىمىزغا كىلىپتۇ.


                                        بىزنىڭ قىزلار ئەقىللىق،

                                        ئالدانماپتۇ ئۇنىڭغا.

                                        يىگىتلەر بەك تېركىپتۇ،

                                        يىتەلمىگەچ نىشانغا.


                                       شۇنداق قىلىپ يىگىتلەر،

                                       قايتىدىن باش قۇشۇپتۇ.

                                       ئالداش ئۈچۈن قىزلانى،

                                       قايتا پىلان تۈزۈپتۇ.


                                       بىراق تۈزگەن پىلانى،

                                       بىرەر باشقا چىقماپتۇ.

                                       شۇڭا ئۇشبۇ يىگىتلەر،

                                       مىسرانىمدىن قىچىپتۇ.


                                      شۇنداق قىلىپ يىگىتلەر،

                                      قىزلارغا ئەل بۇلۇپتۇ.

                                      ئاق بايراقنى چىقىرىپ،

                                      باشلىرىغا تاقاپتۇ.


                                     شۇندىن باشلاپ مىسرانىم ،

                                     بىز قىزلارغا قىلىپتۇ.

                                     مىسرانىمدا قىزلانىڭ،

                                     شاد نەغمىسى  ياڭراپتۇ.


                                     چىدالماپتۇ يىگىتلەر،

                                     ئاڭلاپ ئۇشبۇ نەغمىنى.

                                     قۇشۇلماق بۇپ بەزمىگە،

                                     قولغا ئاپتۇ گىتارنى.


                                     ئەمما قىزلار ئۇلارغا،

                                     پەقەت قاراپ قويماپتۇ.

                                     يىگىتلەرنىڭ ناخشىسى،

                                     يا قۇلاققا ياقماپتۇ.


                                     "ئەي يىگىتلەر ناخشىنى،

                                      ئېيتامسىلەر مۇشۇنداق؟

                                      كەيپىياتنى بۇزدۇڭلار،

                                      ئولتۇرغانتۇق بىز خۇشۋاق."


                                      ئاڭلاپ قىزلار سۆزىنى،

                                      يىگىتلەر بەك تېركىپتۇ.

                                      "خەپ"دەپ جىلە بۇلۇشۇپ،

                                      بەزمىدىن ھەم قىچىپتۇ.


                                     خاپا بولماڭ قېرىنداش،

                                     مۇنبەردىكى دوست-سىرداش.

                                     چاقچاق قىلدىم سىزلەرگە،

                                     سىزمۇ جۆر بۇلۇڭ ئاداش.


                                    قۇشاق-چاقچاق كۈلكىدە،

                                    ياشىرىدۇ يۈرەكلەر.

                                    شاد كۈلكىگە تۇلىدۇ،

                                    مىسرانىمدا كۈڭۈللەر.


                                    سىزمۇ كىلىڭ بىرلىكتە،

                                    مۇشائېرە ئوينايلى.

                                    مەن قۇشاقنى باشلىدىم،

                                    داۋامى سىزگە قالدى.
