مۇنا بىز ساۋاغداش بۇلۇپ بىر سىنىپتا ئۇقۇغىلى3  يىل بوپتۇ تەغدىرنىڭ ئورۇنلاشتۇرىشىمۇ بىلمىدىم ئالىي مەكتەپتىمۇ  بىر سىنىپقا بىر كەسىپتە ئۇقۇپ قالدۇق تۇللۇق ئوتتۇردىن باشلاپ بىر سىنىپتا ئۇقۇپتىمىز  سىنىپتا كىرگەندىن باشلاپلا  سەن نىمىشكىكى  كۈزۈمگە  ئىللىق كۆرۈنگەن ئىدىڭ  شۇ چاغلارنى ئەسلىسەم بەك كۈلگۈم كېلىدۇ سەن تۇرمۇش ھەييىتى  ئىدىڭ  مەن بولسام ئۆگۈنۈش ھەييىتى  سەن چۈشۈنەمىگەن ئىشلىيەمىگەن تاپشۇرۇقلارنى مېنڭ يېنىمغا ئەكىلىپ مەندىن سورايتىڭ ياكى كۈچۈرەتتىڭ تازلىق دىگەندە بولسا سەن مىنى ئاسان  تىز تۈگەيدىغان تازلىققا بۆلۈپ قۇياتتىڭ ىىەن ھەر قېتىم شۇنداق قىلاتتىڭ باشقىلار بۇنى كۇرۇپ ئىچىدە بىر نىمىلەرنى دەپ ئىككىمزىنىڭ ئارقىسىدىن غەيۋەت قىلاتتى ، ھەرسائەت دەرىس ئارلىقىدا مەن ناخشا تىڭشايتىم سىنىپتىكى قىزلارنىڭ ھەممىسى  مىنى بەك دەردى بار دەيتى سەن بولساڭ ئوغۇرلۇقچە قارايتىڭ شۇنداق قاسرىسام ئىتتىك ئالدىڭغا قارىۋالاتتىڭ بىر كۇنى سەن كارۇدۇردا تۇرسەم يېنىمغا كەلدىڭ دە ماڭا چاقچاق قىلدىڭ مىنىڭ ئۇ چاغدا كەيپىياتىم  بولغاچقا  ئاچچىقىمدا  سىنى سىلكىشلەپ نېرى تۇرە  دەپ قاپىقىمنى تۈرۈپ ساپتىمەن  سەن شۇ ئىشتىن باشلاپلا ماڭا تازا گەپ قىلمايدىغان بۇلۇپ قالدىڭ .
مۇنا ھەش پەش  دىگۈچە 3 يىل ئۆتۈپ ئالىي مەكتەپكە باردۇق  سەن دوختۇرلۇقنى تاللىدىڭ مەن بولسام كومپىيوتېر كېيىن مەلۇم سەۋەبلەر تۇپەيلى مەن تاللىغان كەسىپنى تاللىدىڭ ئالىي مەكتەپ دىگەن يەر ئادەمنى راسلا بەك ئۆزگەرتىدىكەن بۇۇرن تەنتەربىيە ئىشتىنىمۇ كەيمەيدىغان بىر قىز ئەمدىلىكتە  تار پادىچىلار ئىشىتىنى كىيىپ يۈرەتتىڭ ياغلىقىڭنىمۇ مودا بىرخىل چىگىشنى ئۆگۈنۋالغان بۇلۇپ بۇۇرنقى تەبى يېزا قىزىدەك يۈرۇشلۈرۈڭ پۇتۇنلەي ئۆزگەرگەن ئىدى( بۇنىڭ سەۋەبى مەن بولۇپ قالغان ئىكەنمەن چۇنكى مەن ئارلاشقان قىز دەپ شۇنداق مودا يۇرەتتى ساڭا مەن دوست بولغاندىكىن ئەگەر شۇنداق يۇرسىڭىز مەن سىزنى تىللايمەن دىگن ئىدىم شۇڭا قەسقەن قىلغان ئىكەنسەن ، مەن ئۇنى بىلمەي ساڭا نىمشىقا بۇنداق كىيىنىسى زئۆزگىرىپ كەتتىڭىز دىسەم سەن ماڭا قاقشاپ  ئاچىڭىزنى باشقۇرۇڭ بېرىپ مىنى باشقۇرماي دىگەن ئىدىڭ شۇ ئاچچىقىمدا ساڭاڭ 3 كۇنگىچە ەپ قىلمىغان ئىدىم دوسلىرىڭغا سەن ئەمدى ئۇ ماڭاڭ گەپ قىلمايدىغان بولدى ژىەىنىڭدە ئۇلار چۇقۇم گەپ قىلىدۇ دەپتىكەن مەن ەەپ قىلدىم بەك قىزىقارلىق بولغان ئىدى  ) ، بىزنى بايرام قۇيۇپ بەردى بىز بىر يۇرتلۇق باللا بىرەگە قايتتۇق سەن بىلەن ھازىرمۇ بۇرۇنقىدەك ئىدۇق گەپ قىلسام بەك گەپ قىلاتىتىڭ بوممىسا مەن گەپ قىلمىسام گەپمۇ قىلمايتىڭ دائىم دەرىس ۋاقتىدا  ماڭا قاراپ تۇراتتىڭ مەن سىزىپ ئولتۇراتتىم مەنمۇ بەزىدە ئىختىيار سىز ھالدا ساڭا قارايتىم سەن دەرھال ئالدىڭغا  قارىۋالاتتىڭ  ياكى كىتابقا قارىۋالاتتىڭ سىنى  قەتتى چۇشۇەنىلى بولمايدىغان قىز دەپ ئويلايتىم چۇنكى مىنى ياخشى كۇرەمسەن كۆرمەمسەن ئۇنى بىللەمگەن ئىدىم  بايراملىق ئۆيگە قايتاشىدا  مەن ئارلاشقان قىز دوستۇم ماڭ اتېلفۇن قىلىپ بىرگە قايتىشنى دىدى مىنىڭ ئۇنىڭ بىلەن ئارلىشۋاتقانلىقىمنى ئاز ساندىكى دوسلىرىملا ئۇقاتتى . (چۇنكى مەن ئاشكارلاشنى خالىمايتىم ئۇنى ياخشى كۆرمەيتىم )  شۇڭا مەن سىنىڭ كۈرۈپ قېلىشىڭدىن ئەنسىرەپ ئۆيگە بىر سائەت بالدۇر ماڭدىم ھېلىمۇ ياخشى ئۇچۇرشۇپ  قالمىدۇ بىراق ئاپتىبۇستىن  چۈشۈپ يېزىدىكى ئۆيگە ماڭىدىغان كىچىك ماشىنىدا ئۇچۇرشۇپ قالدۇق سەن ماشىنىغا چىقىپماڭا  بىر قاراپلا  گەپمۇ قىلماي خۇيلۇنۇپ ماڭا قاراپمۇ قويماي ئولتۇراتىىڭ مەن نمشىقا ئۇنداق قىلغىنىڭنى ئويلاپ يەتتىم سەن مىنى قىز دوستۇمدىن كۇللىگگن ئىدىڭ مەن ئۇ چاغدا سىنى مىنى ياخشى كۇرۇدىكەن دەپ ئويلاپ بىر خۇشال بولسام بىر ئازاپلىناتتىم سىنى قانداق بوپكەتكەندۇ دەپ ئويلايتىم ماشىنا توختىدى سەن تىز سۈرئەتتە چۇشۇتۈڭ د ەتىز  مېڭىپ ئۆيۇڭ تەرەپكە يۇرۇپ كەتتىڭ  مەنمۇ قىز دوستۇم بىلەن خوشلىشىپ ئۆيگە كىرىپ كەتتىم  ئۆيگە كىرىپ سىنىلا ئويلايتىم قانداق  چۇشۇنۇپ قالغاندۇ ئەمدى قانداق بۇلار  ئۇ  ماڭا  تېخىمۇ ئۆچ بولىدىغان بولدى دەپ ئويلايتىم ھەي................
بايرامدىن يېنىپ مەكتەپكە باردۇق  مەكتەپكە بېرىپ  2 كۇندىن كېيىن ياتاقدىشىمدىن سىنىڭ باشقا بىرى بىلەن ئارلىشۋاتقىنىڭنى ئۇنىڭ بۇۇرنلا ساڭا تەلەپ قويغىنى بىراق سەن قۇۇشلماي كەلگىنىڭنى بىراق   بايرامدىن يېنىپ كىلىپلا ئۇنى ئىزدەپ قۇشۇلغىنڭنى ماڭا دەپ بەردى مەنئۇ گەپلەرنى ئاڭلاپ  ھەيران قالدىم ۋە كۆڭلۇم بىر قىسىمىلا  بۇلۇپ قالدى ئۇ ماڭاڭ قېيداپ  شۇنداق قىلغانمىدۇ ؟
  مىنىڭ ئۇ قىز بىلەن نىمىشقا ئارلاشقىنىمنى سەن ئۇقامتىڭ ئۇ قىز مىنى ياخشى كۇرۇپ قالغىنى ئۆزى دىگەن ئىدى شۇڭا رەت قىللامىدىم  دەپ ساڭاڭ  دىگىم كېلەتتى بىراق بۇ  بىر  ساراڭلىق ئىدى  ھا  ھا  ھا ......     
(  مەن دۆتلۈك قىلىپ ئۇ  قىزنى ئازاپلىماي دەپ ئويلاپتىمەن  ئۆزىمۇ ياخشى قىز بولغاندىكىن دەپ ئويلاپتىمەن ئەمدى پۇشايمان قىلىپ ئايرىلپ كىتەي دىىدم يۇ نەتىجىدە يىغلاپ ماڭا تېلفۇن قىلىپ تۇرۇۋاسا يەنە شۇ باغرى يۇمشاقلىق قىلىپ ياخشى كۆمىسەممۇ قايتا يۇۋرىپتىمەن ئەمدى پۇشايمان ) سەن ھەممىنى ئۇقاتتىڭ بۇرۇندىنلا سىنىپتىمۇ مىنى ياخشى كۈرۈپ قالغان قىزلاركۆپ ئىدى بۇمەكتەپكە كىپمۇ يەنەشۇنداق بولدى سىنىڭ ئەڭ يېقىن دوسلۇرىڭدىن بىرنەچسسىنى مىنى ياخشى كۇرۇپ قالغانلىقىنى ماڭا ھەرخىل ۋاستىلەر ئارقىلىق دىدى ( مەن قىز دوستۇم بىلەن ئاخىر ئايرىلىپ كەتكەن ئىدىم ئۇنى ئۇلار بىلىپ بوپتىكەن )  مەن ئۇلارنىڭ ھەممىسىگە كۆڭلۇمدە ئادىمىم بار دىگەن مەنىدە گەپلەرنى قىلدىم ئۇلار چۇشەندى بەزىلىرى قاتىق ئازاپلانغانلىقىنى مۇڭداشقۇدا  ئۇچۇردا  رەسىم ياكى دا ھەرخىل سۆزلەرئارقىلىق ئىپادىلەشتى  مەنمۇ كۆمىگەن كىشى بۇلۇۋالدىم چۇنكى ئۇ مىنى ياخشى كۈرۈدىكەن ئازاپلىنىپ قالمىسۇن دەپ يۈرۈپ قويغاننىڭ پايدىسى يوق ئىكەن كېيىن ئۇنى ۋە ئۆزىڭىزنى تېخىمۇ چىڭ ئازاپلايدىكەنسىز  شۇڭا ئەڭ ياخشىسى ئۇنداق قىلماي دەپ ئويلىىدم (سۆيگەننى سۆيۈپ قوي كۆڭلى قالمىسۇن ، بىر سۆيۈشمۇ غەنىمەت مەيلى قانسۇن قانمىسۇن . بۇ ناخشا تۈپتىن خاتتا بۇنداق قىلىشقا بولمايدۇ دەپ ئويلىدىم دە   بىر كۇنى مەن كانكا تېيىلىش مەيدانىنىڭ گېپىنى ئاڭلاپ قالدىم بۇ مەكتەپكە كىلىپ  ئاياق ئالغىلى بازاغا  چىقىقپتىمەن بوممىسا ھېچيەرنى ئۇقمايتىم ئۇيەرگە بۇرۇنمۇ كىرىپ باققان ئىدىم ئۆزەمنىڭمۇ تېيىلىش ئايىغى بولغاچقا تېيىلىشنى ئۇاقتتىم كىچىك ۋاقتىمدا ئۆگەنگەن   شۇڭا ئۇ يەرگە بارغۇم كەلدى بىللە بارىلىى دىسەم ماقۇل دىدى مەن دوسلۇرىغا  دىگەن ئىدىم سىنىڭمۇ بىللا ئاپىرپىپتۇ مەن ساڭا ئۆگەتتىم سەن قۇلۇمنى تۇتۇشقا جۈرئەت قىللامىدىڭ سەن ئاخىر قۇلۇمنىڭ ئۇچىنى چىڭ تۇتقان ئىدىڭ مەن سىنى خىجىل بۇلۇىۋاتدۇ دەپ ئويلاپ قۇلۇڭنى سىلىكىپ تارتىپ چاپىنىمنىڭ يېڭىنى تۇتۇقۇزۇپ قويدۇم سىنىڭ چىرايىڭ بىردىنلا تاتىرىپ كەتتى مەن  يەنە پۇشايمان قىلدىم   مەن يەنە خاتتا ئىش قىلغان ئىدىم
سەن كەينى كەينىدىن يىقىلدىڭ مەن سىنى چىڭ چۇشۇپ كەتمىسۇن دەپ سىنى چىڭ تۇتۇۋېلىپ ئۆزەم يىقىلىپ چۈشكەن ئىدىم نەچچە قېتىم شۇنداقب ولدى ئاخىرىدا  تىزىم  چىڭ تىكەتكەنلىكتىن ئىششىپ كىتىپتۇ مەن بىردەم ئارام ئالايلى دەپ ئورۇندۇق بار يەگە ماڭدۇق ئىنتايىن تىز تېيىلىپ مېڭىۋاتقان ىبىرسى ئىككىمزىنى سۇقۇۋاتتى مەن چاققانلىق بىلەن سىنى تۇتۇۋالدىم بىراق سەن يەنىلا يىقىپ كەتتىڭ مەن سىنىڭ ئاستىڭدا ياتاتتىم  سەن مىنىڭ  ئۈتۈمگە چۈشكەن بۇلۇپ ھېچ يرىڭ  بىرنىمە بولمماتۇ بىراق ئىككىمىز  ئۈسسۈشۈپ كىتىپتىىمىز  يۈزۈپنى ئىسسىق بىر ھارارەت  سىيپاپ ئۆتۈپ كەتتى   يۈزۈمىز چاپلىشىپ قالغاندەك ھېس قىلدىم  ئورنۇمدىن تەستە تۇردۇم كوچا ئابتىبۇسغا چىقىپ  مەكتەپكە قايتتۇق سەن ماڭڭا  يۇشۇرۇنچە توختىماي قارايتىڭ پۇتۇم شۇنداق ئاغىرىپ كەتكەن بولسىمۇ قەستەنگە تۈز  ماڭاتتىم بىراق يەنىلە بىرە بىر ئا؟خساپ سەن پۇتىڭىز قانراقراق دەپ سورىغاندا مەن ھېنىمە بوممىدى دىسەم  ئاساۋاتىسىزغۇ ئەمسە  دىگەن ئىدىڭ مەن گەپ قىلالمىدىم. مەكتەپكە  قايتتۇق سەنمۇ مەنمۇ ئۆز ياتىقىمىزغا كىرىپ كەتتۇق باشقا دوسلىرىمىزمۇ كىرىپ كەتتى . ئاتىسى سىنىپقا   چىقتۇق چۈشتىن  كېيىن دەرسى يوق ئىدى ئىختىيارى  ئۆگىنىش قىلدۇق سەن ئارلاشقان ئۇ بالا يېنىڭغا بېرىپ ئولتۇرۇپ ساڭا قانداق چاچاق قىلسۇن كۈلكۈلۈك ئىش قىلسۇن سە نقەتتى كۈلمەيتىڭ ماڭا قارايتىڭ مەنمۇ ئىختىيارسىز  ساڭا قاراپ قۇياتتىم ئىككەڭلىنى كۈرۈپ ىر ئاچچىقىم كىلەتتى بىر كۈلگۈم توۋا....
سەن مىنى كۈرۈپلا  قەستەن كۇلگەندەك قىياپەتكە كىرىۋالاتتىڭ  ھەم بەك خۇشال ئويناۋاتقاندەك كۈرۈنەتتىڭ مەن ئىچىمدە ئۇ ماڭا قەستەن كۆرسۈتۈپ قۇيۇش ئۈچۈن مۇشۇنداق قىلىۋېتامدىغاندۇ دەپ ئويلىدىم كېيىن ئاڭلىسام دوسلىرىڭنىڭ ھەممىسى مىنى سىنى ياخشى كۈرۈدىكەن دەپ يۈرۈپتۇ  سەنمۇئۇنى ئۇقۇپسەن بىر كۈنى بىر دوستۇڭ ماڭا ئۇچۇر يوللاپ سىنىڭ ئۇ بالىنى ياخشى كۆرمەيدىغانلىقىڭنى دىدى مەن نىمەئۈچۈن ئارلىشىپتۇ ئەمسە دىسەم ئۇماڭا سەۋەبىنى دەپ بەرمىدى مەن كۆڭلۈمدە ئۇ مىنىڭ ئاچچقىمنى كەلتۇرۇش ئۈچۈن ئارلاشقان چېغىدەپ ئويلىدىم  ئۇنىڭغا كېيىن ئايرىلىپ كىتىلى دەسپەن ئۇماڭا شۇنداق دىدى بىز يان يان ياتاقتا ياتقاچقا ئۇ ياتاققا كىرىپ تۇراتتى ئىككەڭلىنڭ ئىشىنى ئاڭلىسام كۆڭلۇم بىر قىسىمىلا بۇلاتتى بۈگۈن يەنە كانكا تېيىلىش مەيدەنىغا باردۇق ئۇغۇل دوستۇڭ بارما دىگەن بولسىمۇ سەن ئوغۇرلۇقچە  مېڭىۋەردىڭ  مەنب ۇگۇن سەن بلەن يەنە بىرگەئوينىدىم  سەن قۇلۇمى ئۇزاتسام  قۇلۇڭنى بەرگەن ئىدىڭ شۇ ۋاقىتتا شۇنداق بىر ھېسىيات ئۆزەممۇ بىلمەيدىغان بىرخىل ھېسىيات مىنى غەلىتە قىلىپ قويدى ئىككىمىز قلتۇتۇشۇپلا قۇلۇمىز ئايرىلىپ كەتتى سەن مېڭىپ كەتتىڭ ئۆزەڭنى توختىتاماي  توۋا دەيمەن باشقا قىز ساۋاقداشلىرىم بىلەن قول تۇشۇپ ئوينايلى دىگەندە قۇلۇنى تۇتقان بىراق ئۇنداق بۇلۇپ باقمىغان ئىدىم .
مەكتەپكە  قايتتۇق  بىرنەچچە  ھەپتىدىن كېيىن  ھېلىقى دوستۇڭ ئاخشىمى يەنە ئۇچۇر يوللاپ مەندىن سىنى ياخشى كۈرۈدىغان كۈرمەيدىغانلىقىنى سورىدى مەن ياخشى كۈرۈمەن دىگەن مەنىدە گەپ قىلدىم ئۇ مېنى ساقلاشقا دەۋەت قىلدى ئۇ ماڭا  سىز ساقلاڭ ئۇ سىزنىڭ كۆڭلىڭىزنى چۇقۇم چۈشۈنۈپ قالىدۇ بىر كۇنى دىدى ( ماڭا بۇ گەپ خۇددى ئىككەڭلار بىر يەدە تۇرۇپ مىنى سىناۋاتقاندەك  تۇيۇلغان ئىدى ) مۇنا  تەتىلمۇ قويۇپ بەردى ناھىيىگە كەلگىچە بىر ماشىنا كەلدۇق ئۇ كەينىگە پات پات قاراپ قۇياتتى مەنمۇ ئۇنىڭ  كەينىدىن قاراپ قۇياتتىم  ماشىنىدىن چۈشۈپ مەن دوستۇم بىلەن ئانامنىڭ ئۆيىگە ماڭدىم ئۇ دوستى بىلەن قالدى مەن ئاانمنىڭ ئكيىگەن كىرىپ ئاندىن يېزىغا ماڭىدىغان ماىنىغا چىقىدىغان بىكەتكە  بېرىش ئۈچۈن تاكسىغا چىقتىم  بىكەتكە  كەلدۇق مەن تاكسىدىن چۈشۈپ تۇرسام ئۇ  ماڭاڭ قاراپ  مەنىلىك  كۈلۈپ قويدى دە ماشىنىغا چىقىپ كەتتى مەن ياخشى بولدى بىر ماشىنىغا چىقىدىغان بولدۇم دەپ ئويلىغان ئىدىم ئەپۇسۇ ئۇ ماشىنا تۇشۇپ قاپتۇ دوستۇم بىلەن ئىككىمزى كەينىدىكى ماشىنىغا چىقتۇق  ئۆيگە كەلگەندىن بېرى ئۇنى كۆرگۈم كىلىدىغان  مەكتەپكە  بارغۇم كىلىپلا  تۇرىدۇ ئۈچ-تۆتكۈن بولدى  مىجەزىممۇ  يوق ھەي   توۋا  كىتابلاردا  دىگەن مۇھەببەت  سېغىنىش  ئازابى  شۇمىدۇ ...........ھا  ھا  ھا
