يېقىندا بىر دوستۇم مىنى يوقلاپ كەلگەندە ماڭا مۇنداق بىر ھېكايە سۆزلەپ بەردى. ئۇنىڭ دىيىشىچە، ئۇنىڭ ئاكىسى ئىككى يىل ئىلگىرى بۆرەك زەئىپلىشىش كېسىلى بىلەن دوختۇرخانىدا يىتىپ قالغان بولۇپ، جىددىي بۆرەك ئالماشتۇرۇشقا توغرا كەلگەن. ئۇنىڭ بەش قېرىندىشى بولۇپ، ئۇنىڭ ئىچىدە پەقەت ۋە پەقەت بىرسىنىڭلا قان تېپى توغرا كەلگەچكە، ھەممە ئۈمۈد پەقەت شۇ بىر قېرىنداشتىلا قالغان ھەم ھەممەيلەن شۇنداق خۇشال بولۇشقان. چۈنكى ماس كېلىدىغان بۆرەك تاپماق ئۈنچە ئاسان ئەمەس، يەنە كىلىپ، باشقىلارنىڭكىنى نۇرغۇن پۇل خەجلەپ سېتىۋىلىشقا توغرا كىلەتتى. بۆرىكى ماس كەلگىنى دوستۇمنىڭ ھەدىسى بولۇپ، ئۇ ھەدىسى ئاللىبۇرۇن توي قىلىپ ئىككى بالىسى بار ئىدى. ئۇمۇ ئۆز ئاكىسىنىڭ ھاياتىنى قۇتقۇزۇپ قالالايدىغانلىقىدىن خۇشاللانغان ھەم ئىككىلەنمەيلا بىر بۆرىكىنى ئاكىسىغا تەقدىم قىلىشقا تەييارلانغان ئىدى. ئەپسۇسكى ئۇنىڭ يولدىشى بۇ خىل ئىنسانىي مېھرى-مۇھەببەتكە تولغان بىر ئىشنى قوللاش ئۇياقتا تۇرسۇن، ئەكسىچە دوستۇمنىڭ ھەدىسىگە: «سەن بىر بۆرىكىڭدىن ئايرىلىپ چوقۇم كىيىن كېسەلمەن بولۇپ قالىسەن، ئۇ چاغدا مەن سىنى بېقىشقا چولام يوق، ئاناڭنىڭ ئۆيىگە بىرىپ يات. سەن مۇشۇ سەۋەپتىن پۈتۈن ئائىلىمىزنى بەختسىز قىلىسەن» دەپ، ئۇنى بار كۈچى بىلەن توسىغان. ئاخېرىدا، ئۇ ئەر ئاتا-ئانىسىنىڭ ۋە خىزمەتداشلىرىنىڭ نەسىھىيتى بىلەن ئامالسىز بۇ ئىشقا قوشۇلغان. شۇنداق قىلىپ دوستۇمنىڭ ئاكىسىنىڭ ھاياتى قۇتقۇزۇپ قىلىنغان. ھازىر دوستۇمنىڭ ھەدىسى بىر بۆرىكى بىلەن ساغلام تۇرمۇش كەچۈرمەكتە.

بۇ ئاددىي ھېكايە ئۇيغۇر جەمئىيتىمىزدىكى بىر ھەقىقىي يۈز بەرگەن ۋەقە بولۇپ، مۇشۇنىڭغا ئوخشاش نۇرغۇن ھېكايىلەر بولۇپ ئۆتتى ياكى داۋاملىشىۋاتماقتا. بۇ يەردە كىشىنىڭ دېققىتىنى قوزغايدىغان سۇئال: ئەگەر بۇ ئىش شۇ ئەرنىڭ قېرىندىشىدا يۈز بەرگەن بولسىا، بۇ ئەر ئايالىنىڭ توسقۇنلۇقىغا ئۇچرارمىدى؟ ھەم ئايالىنىڭ بۇ ئىشقا ئارىلىشىش ھوقوقى بولارمىدى؟ مېنىڭچە ئۇنداق بولىشى ناتايىن ئىدى. گەرچە دىنىمىزدا ۋە مىللىي ئۆرپ-ئادەتلىرىمىزدە، ئەر ۋە ئاياللارنىڭ باراۋەرلىكى تەكىتلىنىدىغان بولسىمۇ، لىكىن جەمئىيتىمىزدىكى چوڭ-كىچىك نۇرغۇن ئىشلاردا يەنىلا ئاياللارنىڭ ھوقوقلىرى مەلۇم نۇقتىدىن ئېيتقاندا كاپالەتكە ئىگە ئەمەس. نۇرغۇن ۋاقىتلاردا، بىز ئەرلەرنىڭ ئاياللىرىنى ھەتتا يىگىتلەرنىڭ قىزلارنى خۇددى ئۆز مۈلۈكىدەك كۆرۈپ، ئۇلارنىڭ ھوقوقلىرىنى خالىغانچە دەپسەندە قىلىۋاتقىنىنى ئۇچرىتىمىز ۋە ئاڭلاپ تۇرىمىز. ئەلۋەتتە، بۇ دىگەنلىك ھەممىلا ئادەم شۇنداق دىگەنلىك ئەمەس، ۋە ياكى ئەر ۋە ئاياللار ھەر بىر ئىشتا چوقۇم تەڭ تۇرىشى كىرەك دىگەنلىكمۇ ئەمەس. بەلكى مەن بۇ يەردە تەكىتلىمەكچى بولغىنىم، بىر قىسىم ئەرلىرىمىزدىكى ھەددېدىن زىيادە چوڭچىلىق قىلمىشلىرى ۋە ئاياللارنىڭ يولدىشى ۋە يولدىشىنىڭ ئۇرۇق تۇغقانلىرى تەرىپىدىن تەڭسىز مۇئامىلىگە ئۇچرىشى، ئادالەتسىزلىك، غەيۋەتخورلۇق ۋە زۇلۇمدىن بەختسىز كىشىلەرگە ئايلىنىشىدىن ئىبارەت بولغان بىر رىئاللىقتۇر. بۇ توغرىدىكى مىساللار جەمئىيتىمىزدە قەدەمدە بىر ئۇچراپ تۇرىدۇ.  جەمئىيتىمىزدە ئەۋج ئالغان قىزلارنىڭ پەزىلىتىنى ئەمەس بەلكى مال-دۇنياسىنى ھەممىدىن چوڭ بىلىدىغان، كىلىنلەردىن قۇسۇر ئىزدەپ ئەر-ئاياللارنىڭ نىكاھ مۇناسىۋىتىگە تەسىر يىتىدىغان ئىشلار بۇنىڭ مىساللىرى بولالايدۇ. بۇ توغۇرلۇق چوڭ ھەجىمدىكى بىر ماقالە يېزىش مۇمكىن بولسىمۇ، مەن يۇقرىدىكى قىسقا مىسال ئارقىلىق تورداشلارنىڭ بۇ ھەقتىكى كۆز قاراشلىرىنى ئوتتۇرىغا قويۇشىنى ئۈمۈد قىلىمەن.

ئەسكەرتىش: بۇ تېمىنىڭ مەقسىتى ھەرگىزمۇ جەمئىيتىمىزدىكى ئەرلەرنى قامچىلاش بولماستىن، بەلكى يۇقرىدىكى ھېكايىدە ئېيتىلغاندەك نامۇۋاپىق قىلمىشلارنى قامچىلاپ، ساغلام بولغان ئەر-ئاياللىق مۇناسىۋىتىنى ساقلاشقا تۈرتكە بولىدىغان كۆز قاراشلارنى تارتىپ چىقىشتىن ئىبارەت. تورداشلارنىڭ مۇلاھېزىلىرى ۋە ئىنكاسلىرىدا ئېغىر-بىسىق، ئىلمىي ھەم ئەپۇچان بولۇشىنى سورايمەن.

بۆرىكىنى باشقىلارغا بىرىشتە ،ئاۋال ئۆزىنىڭ يەنە بىر بۆرىكى ساغلام بولىشىنى ئالدىنقى شەرىت قىلىش كېرەك ،ئاندىن بورەكنى قوبۇل قىلىش ئوبىكتى ياردەم بىرىشكە ئەرزىمدۇ دىگەندىمۇ نۇختىلىق ئويلىشىش كېرەك،بىر تۇققان قىرىندىشىم دەپلا بۆرىكىمنى بىرىمەن دىسە بولمايدۇ،ناۋادا ئۇ كىشى  كۈتۈنۈشنى بىلمەيدىغان ،ئۆزىنىڭكىنىلا راس قىلىدىغان جاھىل ئادەم بولسا ،ئۆزىنىڭ كۆپ يىللىق ساغلام بولمىغان تۇرمۇس ئادىتى ھەققىدە قىلچىلىك ئويلانمىغان ۋە ئويغانمىغان بولسا، مىنىڭ شەخسى پىكرىمچە بولغاندا ياردەم بىرىشكە  ئەسلا ئەرزىمەيدۇ،بولمىسا بەدىنىدە يەكلەش ئالامەتلىرى كۆرۈلمىدى دىگەن تەقدىردىمۇ،ناچار تۇرمۇش ئادىتى تۈپەيلىدىن يەنە بىر نەچچە يىلدىن كىيىن قايتا بۆرەك زەئىپلىشىش كىسىلىگە گىرىپتار بولىشى مۇمكىن. ئادەم خاتالىق ئۆتكۈزسە،جازاسىنى كۆپىنچە باشقىلار تاتىدۇ،بولمىسا ئىنسان خاتالىقنى ئازراق ئۆتكۈزگەن بولاتتى!
غەرىپچە تىبابەتتە،ئادەم بىر بۆرەككە تايىنىپ نورمال تۇرمۇش كەچۈرەلەيدۇ دەپ قارىلىدۇ،ئەمما قايسى كىتاپتا قاچان كۆرگىنىم ئىسىمدە قالماپتۇ،ئەنئەنىۋى جۇڭيى تىبابىتى نەزىرىيىسىگە ئاساسلانغاندا سول تەرەپتىكى بۆرەك ئۆرلىتىشكە،ئوڭ تەرەپتىكى بۆرەك تۆۋەنلىتىشكە مەسئۇل دەپ قارىلىدۇ،شۇڭلاشقا بىر بۆرەك بىلەن قالغاندا كۈتۈنىشكە ،جۈملىدىن بۆرەكنى ئاسراشقا بەكرەك ئەھمىيەت بىرىش كېرەك. بۇ ۋەقەگە چىتىشلىق ئاكا ۋە سىڭىلنىڭ تېنىنىڭ سالامەت، ئائىلىسىنىڭ ئىناق ئىتتپاق بولۇشىغا تىلەكداشمەن.مىنىڭ بۆرەكنى ئاسراش تەدبىرلىرى دىگەن ماقالەم سىلەرگە ئازدۇر كۆپتۇر پايدىسى بولار دېگەن ئۈمۈدتىمەن.
