                                                       سەنئەت ۋە بىز

                                                          (تۇمارىس)

                                                 ۋاراقلىساق تارىخ بىتىنى،
                                                 ئىزىمىز ئەجەپمۇ ھەيۋەت.
                                                 ھالا كىلىپ بۈگۈنكى كۈنگە،
                                                دەسمايىمىز پەقەتلا سەنئەت.

                                               بىز ياراتقان ئىدۇق دۇنيادا،
                                               بىر سانالغان مەدەنىيەتنى.
                                               بۈگۈن لىكىن ئاتىلىپ قالدۇق،
                                              ناخشا-ئۇسسۇل سەنئەت مىللىتى.

                                              ئەسلەپ بىقىڭ ئەجدادلىرىمىز،
                                              مىراس قىلغان قۇتادغۇبىلىگ.
                                              ئەمما بۈگۈن ناخشا-ئۇسسۇلغا،
                                             مەپتۇن بولدۇق دىلدىن شۇنچىلىك.

                                             مەھمۇت بۇۋام ھەمدە نەۋائى،
                                             دەۋىر سۈرۈپ ئۈتكەن دىيار بۇ.
                                             ئۈچتىمۇ يا مەرىپەت نۇرى،
                                             دىلنى ياكى غەپلەت باستىمۇ؟

                                            قاراپ باقسام ئۇزاق ئۈتمۈشكە،
                                            بىز ئاجايىپ بىر مىللەت ئىدۇق.
                                            ھالا بۈگۈن ناخشا-ئۇسسۇلدا،
                                            ئەڭ تالانتلىق مىللەت ئاتالدۇق.

                                           بولدى يىتەر ناخشا-ئۇسسۇلغا،
                                           مەپتۇن بۇلۇپ ئۈتكەن كۈنىمىز.
                                           كۆزنى ئېچىپ قارايلى بىر رەت،
                                           ئۆزگىلەردىن كەمكىن نەرىمىز؟

                                           سەنئەت ئەمەس بارلىقىمىز،
                                           ئۈگۈنەيلى ئىلىم-مەرىپەت.
                                           ئاتالمايلى ئەمدى دۇنيادا،
                                           سەنئەت ئۈچۈن يارالغان مىللەت.

                                          بارلىقىمىز بولمىسۇن سەنئەت،
                                          يۈكسەلسۇن تىز ئىلىم-مەرىپەت.
                                         چولپان بۇلۇپ چاقنىسۇن مەڭگۈ،
                                        "ئۇيغۇر" نامى ئەبەدىلئەبەد.
