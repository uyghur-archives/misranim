مىسرانىم مۇكاپاتلىق ئەسەرلەر پائالىيتىگە  
                       

                                   زامانىمىزدىكى قىيامەت
                                 
                                       نىجات رۇزىمەممەت ( نۇرلان )      
                                       

                           
                                      قالمىسا زىمىندا ئەگەر ئادالەت   ،
                                      ئۇيۇلسا تارىخقا ساختا ھەقىقەت ،
                                      بولمىسا قەلىبتە ھېچبىر ھىدايەت ،
                                      يۇقالسا زىمىندىن مىھرى-مۇھاببەت ،
                                      رەزىللىك ئۈستىگە ياغسۇن قىيامەت .
                          
                                       مەلئۇنلار خارلىسا بۇ ئەزىز تەننى ،
                                       نادانلىق قاپلىسا بۇ بۈيۈك ئەلنى ،
                                       بالاڭمۇ خار قىلسا ئۆز ئانا تىلنى ،
                                       ئايالدەك قىلىپ قويسا نەۋقىران ئەرنى،
                                       شۇملۇقلار شەررىگە كەلسۇن قىيامەت .
                           
                                       ئەخلاقنىڭ ھۆرمىتى نۆلگە تەڭ بولسا ،
                                       ئىمان ھەم ھايامۇ دىلدىن يۇقالسا ،
                                       مەنپەئەت ئۇرۇشى قاتتىق يامىرىسا ،
                                       كىشلەردە قىلچىلىك ھۆرمەت قالمىسا ،
                                       ئىنسانلار قەلبىگە بەرسۇن قىيامەت .
                           
                                       نادانلىق زىمىندا سەل بولۇپ ئاقسا ،
                                       ھورۇنلۇق ھەممىگە ياغ كەبى ياقسا ،
                                       غەيۋەتلەر ھەممىنى توختىماي چاقسا ،
                                       باتۇرلۇق بارچىدىن مەڭگۈلۈك تانسا ،
                                       غەپلەتلەر يۈزىگە چاقسۇن قىيامەت .
                           
                                       كۆزلەردە ئۈمۈدلەر يوقالغان چاغدا ،
                                       قورقۇنۇش قەلىبنى چىرمىغان چاغدا ،
                                       ۋىجداننى ئالتۇنغا سېتىشقان چاغدا ،
                                       باشقىغا ئورەكلەر قېزىشقان چاغدا ،
                                       جاھالەت كۆزلەرنى كۆمسۇن قىيامەت .
                             
                                       ئىشرەتلەر بەزمىسى يەتسە پەلەككە ،
                                       شەيتانلار ئارزۇسى ئاشسا تىلەككە ،
                                       ئۆستەڭلەر ھەممىسى ئاقسا سەلەپكە ،
                                       دوستۇڭنىڭ ئوقياسى تەگسە يۈرەككە ،
                                       زۇلۇملار قوينىغا ئالسۇن قىيامەت .
                                
                                       ‹‹ياشلىق›› دەپ شەيتاننىڭ يولىغا باقسا ،
                                       ‹‹قاشلىق›› دەپ  پوقنىمۇ قېشىغا باسسا ،
                                       ‹‹شاشلىق››دەپ جېنىنى ھوشۇققا قاتسا ،
                                       ‹‹باشلىق›› دەپ يۈزىگە پۇتىنى ياقسا ،
                                       ۋىجداننى قامچىغا باسسۇن قىيامەت .
                              
                                        ھەق سۆزلەر بىزلەردىن بولسا خىجالەت ،
                                        قۇرقۇنۇش ئەل ئىچى تاپسا ئىناۋەت ،
                                        ئىمانغا قەلىبلەر قىلسا خىيانەت ،
                                        ئىنسانلار ئارسىغا ئۈنسە ئاداۋەت ،
                                        ئىنساننى دوۋزاققا سۈرسۇن قىيامەت .
