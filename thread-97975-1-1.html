ئادەمنىڭ باش قىسمى ئۇرۇلۇپ كەتسە،ئەستە تۇتۇش قابىليىتى تەسىرگە ئۇچرامدۇ

(تەرجىمە قىلغۇچى: تىلماچ)

ئىنسانلار شۇنىڭ ئۇچۇن يەر شارنىڭ ھۆكۇمرانىغا ئايلىنالىدىكى،چۇنكى ئۇلارنىڭ چوڭ مېڭىسى بىر قەدەر تەرەققە قىلغان.گەرچە بۇركۈت ئادەمدىن يىراقنى كۆرەلىسىمۇ،ئىت سېسىق پۇراققا ئادەمدىن سەزگۇر بولسىمۇ،ئات ئادەمدىن تېز چاپسىمۇ،ئەمما ئۇلارنى ئادەم بىلەن سېلىشتۇرغىلى بولمايدۇ.ئادەمنى يەيدىغان يولۋاس،ھىلىگەر تۇلكىمۇ ئادەمگە يۇۋاشلىق بىلەن بوي سۇنىدۇ.ئىنسانلارنىڭ تۇيغۇ،سەزگۇ،ھىسيات،ئىرادە،خاتىرە،تەھلىل قىلىش،ھۆكۇم قىلىش ۋە ئەمگەك قىلىش ئىقتىدارىغا ئىگە بۇلۇشىدىكى سەۋەپ شۇكى،ئۇلاردا چوڭ مېڭىنىڭ بولغانلىقىدا.

ئادەمنىڭ چوڭ مېڭىسىنى مۇستەھكەم باش سۆڭەك قوغداپ تۇرىدۇ.ئادەتتە زەخمىگە ئۇچرىمايدۇ.ناۋادا كۈتۇلمىگەن ئەۋال يۈز بىرىپ،باش قىسمى ئۇرۇلۇش زەربىسىگە ئۇچرىسا،نۇرغۇن ئاۋارىچىلىق پەيدا بۇلىدۇ.سىرىتقى كۆرۇنۇشتە ھېچقانداق زەخمە يوقتەك كۆرۇنگىنى بىلەن،ئەمما ئۇرۇلۇشنىڭ زەربە دولقۇنى سۆڭەكتىن ئۆتۇپ مېڭىگە كىرىپ كېتىدۇ-دە،چوڭ مېڭە پوسىتلاق قەۋىتىنىڭ ئۇچۇرلارنى نۇرمال يەتكۇزۈش ئىقتىدارنى ۋاختىنچە ئۇزۇپ قويىدۇ.ئۇرۇلۇش زەربىسگە ئۇچرىغاندا،كۆزلىرىدىن ئۇچقۇن كۆتىرىلىپ،كۆز ئالدى قاراڭغۇلىشىدۇ.شۇنۇڭ بىلەن بىرگە ئەس-ھۇشىنى يوقىتىدۇ.قىسقا ۋاقىت ئىچىدە ئەس-ھۇشىغا كەلگەندىن كىيىن،قان بېسىم،تۇمۇر سوقۇش ۋە نەپەس ئېلىش قاتارلىق ھاياتلىق كۆرسەتكۇچىللىرىدە ھىېچقانداق ئۆزگىرىش بولمىغاندەك.پەقەت ئۇرۇلۇش زەربىسىگە ئۇچرىغاندىكى ئەھۋال ئېسىدە قالمايدۇ.ئۇرۇلۇش زەربىسىزگە ئۇچراش ۋاختى قانچە يېقىنلاشقانسېرى،خاترسى شۇنچە ناچارلىشىدۇ.دوختۇرلار بۇ خىل ھادىسىنى تەتۇر يۆنۇلۇشلىك ئۇنتۇغاقىلىق كىسىلى دەپ ئاتايدۇ.بۇ خىل كىسەللىك ھالىتىدە گەرچە ئاستا-ئاستا ئەسلىگە كىلىش بولسىمۇ،ئەمما ئەينى ۋاقىتتىكى ئۇرۇلۇش زەربىسىنىڭ غىدىقلىشى تۇپەيلىدىن،ھەمىشە كېسەللىك ئاسارىتى (مېڭە زەخمىلەنگەندىن كىيىنكى نېرۋا فۇنكىسيە كىسىلى) قېلىپ قالىدۇ.بۇ خىل كېسەللىكنىڭ ئالامەتلىرى ئادەتتە باش ئاغرىش،باش قېيىش،ئاۋازدىن قورقۇش،ئۇيقۇسىزلىق،چارچاش،ئەستە تۇتۇش ۋە چۇشۇنىش قابىليىتى تۆۋەنلەش قاتارلىق ئالامەتلار ئارقىلىق ئىپادىلنىدۇ.

چوڭ مېڭە شۇنداق مۇھىم بولغان ئىكەن، ئۇنداقتا ئۇنى قانداق قوغدىشىمىز كېرەك؟ ئالۋەتتە،بىز پاسسىپ ئۇسۇللارنى،مەسلەن:چوڭ مېڭىمىز ئۇرۇلۇپ كېتىشتىن قورقۇپ،ھېچقانداق پائالىيەتلەرگە قاتناشماسلىقتەك ئۇسۇللارنى قوللانماسلىقىمىز كېرەك.ئۇنداقتا قانداق قىلىشىمىز كېرەك؟ ئاكتىپ ئۇسۇل شۇكى،ھەر بىر ئادەم بەزى تەنتەربىيە پائالىيەتلىرىگە قاتنىششى،مەسىلەن:بەدەن گىمناستىكىسى،توپ تۇرلىرى قاتارلىق قاتارلىق ھەركەتلەر.مۇشۇنداق قىلغاندا بەدەن ۋە مۇسكۇللارنىڭ جانلىقلىغىنى ئاشۇرغىلى ھەم جىددى ئەھۋالغا تاقابىل تۇرۇش قابىليىتىنى كۇچەيتىپ،ئاسانلا مېڭە سىلكىنىش يۇز بىرىشتىن ساقلانغىلى بۇلىدۇ.

ئادەمنىڭ مۇشۇ كاللىسى راسلا ئاجايىپ جۇمۇ،
مەنمۇ بىر ئىشنى دەپ بېرەي، ئالىملار مۇنداق بىر مەسىلىنى تەتقىق قىلىپتۇ، يەنى ئادەمنىڭ بېشى ئۈزۈلگەندىن ھامانلا سېزىم يوقامدۇ ياكى دەرھال يوقامدۇ؟ بۇنى بىلىش ئۈچۈن ئۇلار ئۆلۈمگە ھۆكۈم قىلىنغان بىر جىنايەتچى بىلەن كېلىشىپ، جىنايەتچى ئۇلار بىلەن ماسلىشىدىغان، ئالىملار بۇنىڭ بەدىلىگە ئۇنىڭ ئائىلىسىگە نۇرغۇن پۇل بېرىدىغان كېلىشىپتۇ، جەريان : جىنايەتچى كاللىسى كېسىلگەن ھامان بىر تەتقىقاتچى ئۇنىڭ ئىسمىنى چاقىرىدۇ، ئەگەر جىنايەتچى ھېس قىلسا كۆزىنى ئاچىدۇ
نەتىجە: جىنايەتچى ئۆزىنىڭ ئىسمىنى چاقىرغان ئاۋازنى ئاڭلاپ كۆزىنى ئېچىپ ئىنكاس قايتۇرغان، لېكىن ۋاقتى 3سېكۇنتقىمۇ يەتمىگەن، دېمەك كاللا ئۈزۈلگەن ھامان سېزىم شۇ ھامانلا يوقىلىپ كەتمەيدۇ
