ئادەملەر نىمە ئۇچۇن سېرىق تەنلىك،ئاق تەنلىك،قوڭۇر تەنلىك دەپ ئايرىلىدۇ

(تەرجىمە قىلغۇچى: تىلماچ)

پۇتۇن دۇنيادا ياشاۋاتقان پۇتكۇل ئىنسانلار بىئولوگىيە ئىلمىدە بىرلا ماددىغا تەۋە بۇلۇپ ،ئۇلارنىڭ ئەجدادى بىر.ئەمما ئولتۇراقلاشقان رايۇن ۋە مۇھىتنىڭ ئوخشىماسلىقى تۇپەيلى،ئادەم بەدىنى تەبئى مۇھىتقا ماسلىشىش ئۇچۇن،ئوخشاش بولمىغان قىياپەت ۋە بەدەن شەكلى ششەكىللەنگەن.

چاچ شەكلى،چاچ رەڭگى،تېرە رەڭگى،كۆز رەڭگى،يۈز شەكلى،باش شەكلى،قان تىپى يەنە باش سۆڭىگى قاتارلىق ئىرسىيەت ئاھىدىلىگىگە ئاساسەن،ئىنسانلارنى بەش چوڭ تۈرگە بۆلۇشكە بۇلىدۇ:

ئاق تەنلىك(كافكازلار)،سېرىق تەنلىك(موڭغۇللار)،قوڭۇر تەنلىك(مالايالار)،قارا تەنلىك(نىگىرولار)،قىزىل تەنلىك(ئامرىكىلىقلار)

بۇگۇنكى دەۋىردىكى ئىنسانشۇناسلار بەدەن سۇپەت ھالىتىدىكى ئورتاق ئىرسىيەتكە  ئاساسەن،تىرىسىنىڭ رەڭگىنى ئاساس قىلىپ،ئىنسانلارنى سېرىق تەنلىك،قارا تەنلىك ۋە ئاق تەنلىك دەپ ئۇچ چوڭ تۇرگە بۆلىدۇ.بۇ ئۇچ چوڭ تۇردىكىلەرنىڭ تېرىسىنىڭ رەڭگىنىڭ ئوخشىماسلىغىدىكى سەۋەپ،تۇرنىڭ ئىرسىيەت ئامىلىدىن باشقا،ئاساسلىق قارا پىگمىنىتنىڭ تېرىدىكى تارقىلىش ئورنى ئوخشىمايدۇ.سېرىق تەنلىكلەردىكى قارا پىگمىنىت ئاساسلىق تېرىسىنىڭ ئاستىنقى قاتلىمىغا تارقالغان بۇلۇپ،تۇقۇلما ۋە قان تۇمۇر قاتلىمىدىن ئۆتۇپ تېرىنى سېرىق كۆرسىتىدۇ.قارا تەنلىكلەردىكى قارا پىگمىنىتنىڭ مىقدارى كۆپ بۇلۇپلا قالماي،بەلكى تېرىنىڭ ئاستىنقى قاتلىمىنىڭ ئۇستىدىكى سىپىنا قەۋىتى بىلەن دانچە قەۋىتىگە تارقالغان .شۇڭا تېرىسى قارا كۆرىنىدۇ.ئاق تەنلىكلەردىكى قارا پىگمىنىت بىلەن سېرىق تەلىكلەردىكى قارا پىگمىنىت ئوخشاش،ئەمما ئازراق بۇلۇپ،تېرىسى ئاق كۆرىنىدۇ.

سېرىق تەنلىكلەر ــ موڭغۇللار ۋە ئاسىيا- ئامرىكىلىقلار دەپمۇ ئاتىلىدۇ.بۇلار شەرقى ئاسىيا،شىمالى ئاسىيا،ئوتتۇرا ئاسىيا،شىمالى قۇتۇپ ۋە ئامرىكا قىتئەسى قاتارلىق جايلارغا تارقالغان  بۇلۇپ،جۇڭگۇلۇقلار،مۇڭغۇللار،چاۋشەنلىكلەر،ياپۇنلۇقلار،ھىندىيانلاي ۋە ئېسكىمولارنى ئۆز ئىچىگە ئالىدۇ.ئۇلارنىڭ ئاساسلىق بەدەن ئالاھىدىلىكى تېرىسىنڭ رەڭگى  سېرىق ياكى ئاق؛چاچ شەكلى تۇز بۇلۇپ،رەڭگى قارا؛كۆزىنىڭ رەڭگى قېنىق بۇلۇپ،ياڭاق سۆڭىكى ئىگىز.

ئاق تەنلىكلەرـــ ياۋرۇپالىقلار،ياۋرۇ-ئاسىيالىقلار ۋە كافكازلار دەپمۇ ئاتىلىدۇ.بۇلار ياۋرۇپا،شىمال ئافرىقا ۋە ئاسىيادىكى تۇركىيە،ئىران،ئىراق،ئافخانىستان،پاكىستان،بىنگال ۋە ھىندىستان قاتارلىق جايلارغا تارقالغان.ئۇلارنىڭ تېرىسنىڭ رەڭگى سۇس؛چېچىنىڭ رەڭگى سېرىق ياكى قارامتۇل بۇلۇپ،چاچ شەكلى كۆپىنچە دولقۇنسىمان ياكى تۈز كۆرىنىدۇ؛كۆزىنىڭ رەڭگى كۆك ياكى كۈلرەڭ بۇلۇپ،بەدەن تۈكچىللىرى ۋە ساقال-بۇرۇتلرى تەرەققى قىلغان.

قارا تەنلىكلەر ــــ نىگىرولار،ئىكۋاتۇرلىقلار دەپمۇ ئاتىلىدۇ.ئۇلار ئاساسەن ئافرىقا،ئوكيانىيە،ھىندىستاننىڭ جەنۇبى،سىرلانكا،مېلانىسىيە،كالىمانتان قاتارلىق جايلارغا تارقالغان.بۇلارنىڭ بەدەن ئالاھىاىلىگى تېرىسى كۆپىنچە قارا ياكى قارامتۇل كۆرىنىدۇ؛چاچ پاسۇنى بۇدرە ياكى دولقۇنسىمان بۇلۇپ،چېچىنىڭ رەڭگى توق قارا،كۆزىنىڭ رەڭگى قارامتۇل كىلىدۇ.
