                                  ياتاق يىتىشمەي     
                                  (ھېكايە)                                             
                            ئەختەر
                                                                 1  
ئەسقەر سائىتىگە قارىغاچ توختىماي سايرىۋاتقان تېلىفۇنىنى زەردە بىلەن بېسىۋەتتى، ئۇنىڭ خۇسىسىي شوپىرى كەينىگە قاراپ خىقىراق ئاۋازدا كۈلدى:                                   
-مانا، مانا ئەسقەركا يۇرۇتقا كەپ بولغىچە تومۇچىقلىرىڭىز ئىزدىگىلى تۇردى، باياتتىن ياقى ئالاھەزەل يىگىرمە قېتىم تېلىفۇن قىلغاندۇ بىرەرسىنى بولسىمۇ ئالمىدىڭىز، خاپا بوپقالمىسۇن يەنە بىرەر ئېغىز بولسىمۇ تىنىپ قۇيۇڭ ئەمدى...                                             
-سىنى ئاناڭ قۇچقاچ پۇقىدا ئېغىزلاندۇرغانمىكىن دەيمەن تۇلا كاسىلدىماي ماشىناڭنى ھەيدەشنى بىلە...                                                                     
-نەگە ھەيدەيمەن...                                                                  
-ئاپلا،سىنى قانداق قىلىۋەتسەم بۇلا، ئۆيگە ھەيدىمەي نەگە ھەيدەيىتتىڭ...              
-ئاچچىقلانمىڭە ئاكا، ئەمدى دەيمىنا، ھەر قېتىم تىجارەت بىلەن قايتىپ كەلگىنىڭىزدە داجىئۇ دىيەن بىڭگەنلەردە بىرەر كۈن قۇنىۋېتىپ ئەتىسى ئاندىن ئۆيگە كىرەتتىڭىز ئەمەسمۇ، شۇڭا سورىدىم...                                                                           
- توغرا دەيسەن ئەسلىدە ھاردۇقۇمنى ئوبدان چىقىرىپ قايىتسام بۇلاتتى، لېكىن ھازىرقى خۇتۇنلارغا جىن تەگدىمۇ قانداق نەدىكى ئۇششاق شۇملارنى دورايدىغان بوپقاپتۇ ھېلىقى نىمىتى؟ ئاناڭنى ئەجەپ ئىسىمگە كەلمەيدىيا، ھە ئاشىق- مەشۇقلار بايرىمى دەمدا؟ بۈگۈن شۇنداق بىر كۈن ئوخشايدۇ، ھەدەڭ بايا تېلىفۇن قىپتۇ، ‹‹ ئەسقەرجان سىز ئۆيگە پەيدا بولمىغىلى بىر ئاي بوپقالدى، ئۇنىڭ ئۈستىگە بۈگۈن ئاشىق- مەشۇقلار بايرىمى، بۈگۈن نەدە بولسىڭىزمۇ ئالدىمغا جەزمەن ئۈنۈپ بولمىسىڭىز بولمايدۇ،سوۋغات ئەكىلىشنى ئۇنتىماڭ،ئىشقىلىپ بۈگۈن ئۆيگە تىزدىن پەيدا بولمىسىڭىز بۇندىن كېيىن ئۆيگە دەسسىگۈچى بولماڭ›› دەيدۇ، ئاناڭنى بۇنداق نەس باسقان بايرام نەدىن پەيدا بولغان دەيمەن ھېرىپ كەلگەندە بىر ھاردۇقۇمنى چىقىرارمەن دىسەم ھازازۇل خۇتۇن ئەجەپ ئىچىمنى سىقتى...       –ئاكا، ھەدەمدىمۇ گۇناھ يوق، ئويلاپ بېقىڭە، سىز ئۆيگە كىرمىگىلى بىر ئايچە بوپقالدى، ئۇنىڭ ئۈستىگە ئۆيگە ئۇزاقتا بىر كىرىدىكەنسىز، كىرگەندىكىن ئۇزاق تۇرماي بېشىڭىزنى تىقىپ قويۇپلا يەنە سىرىتقا سوكىلدايدىكەنسىز، بۇنداق قىلسىڭىز ھەرقانداق خۇتۇنكىشىنىڭ ئاچچىقى كىلىدۇ-دە...                                                                          
-ۋاي، توڭ سويما، سەن تېخى ماڭا سۆزلەپكەتتىڭما، تۇلا كاسىلدىماي ئالدىڭغا قاراپ ماشىناڭنى ھەيدىگىنە، نىمىلا دىگەن بىلەن مەن سەندىن نەچچە ئىشتاننى كۆپ يىرىتقان...   
شوپۇر مۇغەمبەرلەرچە كۆلۈپ قۇيۇپ غىڭشىپ ناخشا ئېيتىشقا باشلىدى، ئەتىراپقا قاراڭغۇ چۈشكەن بولۇپ شەھەر چىراغلىرى ئەتىراپنى يۇرۇتۇپ تۇراتتى، كەچ بولسىمۇ ئادەملەر كۇچىلاردا مىغىلداپ تۇراتتى، يول ياقىسىدا ئەتىرگۈل كۈتۈرۋالغان ياش قىزلار ئەتىراپتا مېڭىشىۋاتقان ياش قىز- يىگىتلەرنىڭ ئالدىغا يۈگۈرۈپ بېرىپ گۈللىرىنى تىزراق سېتىشقا ئالدىرايىتتى، يول ياقىسىدىكى ياتاقلارنىڭ ئالدىغا قىز- يىگىتلەر توپلىشىۋالغان بولۇپ ئەسقەر بۇ مەنزىرلەرنى كۆرۈپ سۆزلەپ كەتتى:                                                                 
-يا پاناھ، جاھان نىمە بوپكىتىۋاتىدۇ؟ توۋا قىلدىم، سەمەت قارىغىنە ساپلا ئۇششاق بالىلار... بۈگۈن ياتاق خۇجايىنلىرى تازا پۇل تاپىدىكەندە...                                      
ئەسقەر شۇنداق دەپ تۇرىشىغا تېلىفۇنى تۇيۇقسىز يەنە جىرىڭشىدى...                     
-ئاپلا، بۇ كۆك پىتقا نىمە دەپ يۇلۇقتىم ئەمدى، يەنە تېلىفۇن قىلدى...- ئەسقەر تېلىفۇننى ئالدى، قارشى تەرەپتىن زىل بىر ئاۋاز كەلدى:                                            
-ۋەي جېنىمما، باياتتىن بىرى تېلىفۇن قىلسام ئالمايسىزغۇ؟                              
-ھە، ئاپپاق، خاپا بولماڭ، مۇھىم ئىش بىلەن بولۇپ كىتىپ ئاڭلىماپتىمەن، ھە نەدە بوپكەتتىڭىز؟                                                                          
-نەدە بۇلاتتىم، ئىش يوق ۋاڭبادا ئولتۇردۇم شۇ ناخشا تىڭشاپ، بۈگۈن ئاشىق- مەشۇقلار بايرىمىمىش، شۇڭا سىزنى سېغىنىپ تېلىفۇن قىلغان، قانداق كۆرۈشكىڭىز يوقما مەن بىلەن؟
-ۋاي بولمامدىغان، لېكىن بۈگۈنچە، كۈرىشەلمەيدىغاندەك قىلىمىز...                     
-شۇنداقمۇ؟ ئەمىسە بۇندىن كېيىن بىراقلا كۆرۈشمەي قۇيايلى خوش...                  
-ۋاي، ئاپپاق، خاپا بولمىڭە توختاپ تۇرمامسىز، مۇنداق قىلايلى، مەن ئاۋۋال ئۆيگە بېشىمنى بىر تىقىپ قۇياي ئاندىن، نوچى ياتاقتىن بىرنى ئېلىپ... ھى.. ھى...                  
-كۆرۈشسەك ھازىر بولسۇن بولمىسا مەن ساقلاپ بۇلالمايمەن...                           
-ھە، ئەمىسە مۇنداق قىلايلى، مەن ئالدى بىلەن ياتاقتىن بىرنى زاكاز قىلىۋېتىپ ئاندىن سىزگە تېلىفۇن قىلاي، بىرەر سائەت ناخشا ئاڭلاپ ئولتۇغاچ تۇرۇڭ قانداق؟                 
-مەيلى، بىر سائەتلا جۇمۇ، ھە راست، بۈگۈن ئاشىق- مەشۇقلار بايرىمى جۇمۇ، سوۋغاتنى ئۇنتۇپ قالماسسىز؟                                                                    
-ۋاي، ئاپپاق، خاتىرجەم بۇلىڭە...                                                   
ئەسقەر تېلىفۇننى قۇيۇپ ئۆزىچە غۇدۇڭشىپ كەتتى:                                       
-ھۇ پاسكىنا قۇيىمەنلا دېگەن، كىچىكلىكىنى دېمىسەم كۆتەر قاسقىنىڭنى دەيىتتىم، ھە ئۇكام ئالدىمىزدىكى داجىئۇ دىيەننىڭ ئالدىغا ھەيدىگىنە...                                    
سەمەت ماشىنىنى چىراغلىرى ھەرخىل رەڭدە يېنىپ تۇرغان بىر بىنانىڭ ئالدىدا سىلىق توختاتتى، ئەسقەر ماشىنىدىن ئالدىراپ چۈشۈپ كەتتى، سەمەت ئۇلاپ ئىككى تاماكنى چىكىپ بولغاندا ئەسقەر ماشىنىغا ئاغىزىنى بۇزۇپ چىقتى:                                         
-ۋاي ئاناڭنى ياتاق قالماپتۇ، ھەقىچان ياتاقلارنى ئۇششاق شۇملار ئىگەللەپ بولغان گەپ، ئەمدى ‹‹ ئوتلۇق›› قا ھەيدىگىنە...                                                  
‹‹ ئوتلۇق›› تىن ئالاھەزەل يېرىم سائەتتىن كېيىن چىققان ئەسقەر يەنە ئاغىزىنى قۇيىۋەتتى:
-ئاناڭنى بۇ يەردىمۇ ياتاق يوق دەيدا، بۇ ماڭقىلار بۈگۈن ياتاقنىڭ ھەممىسىنى ئىگەللەپ بولغانمۇ دەيمەن، مۇلازىمەتچىلىرى بىلەن سۇقۇشۇپ قالغىلى قىل قالدىم،ھەممىسى ئاشۇ نەس باسقۇ بايرامنىڭ كاساپىتىغۇ دەيمەن، توۋا قىلدىم بورۇن بۇنداق ئىش توي قىپ بولغاندا بۇلىدىغان، بۇ ماڭقىلار توي قىلمايلا ئەجەپ ئەر-خۇتۇندەك... ئۇكام ئەمدى ‹‹ ئۇچار بۈركۈت›› كە ھەيدىگىنە...                                                         
ئۇلار پۈتۈن شەھەرنى ئالا قويماي ئاختۇرۇپمۇ مۇۋاپىق بىر ياتاق تاپالمايۋاتاتتى، ئەسقەر توختىماي ھېلىقى ئاشنىسىغا يالۋۇرماقتا ئىدى:                                          
-ئاپپاق، بىردەم تەخىر قىلسىڭىزچۇ، شۇ تاپتا ياتاق تاپالماي پىقىراپ يۈرىيمەن، يەنە يېرىم سائەت مۆھلەت بىرىڭ بۇلامدۇ؟...                                                   
سەمەت رولغا بېشىنى قويۇپ مۈگىدەپ قالغانىدى، بىر چاغدا ئەسقەر كالچاراپ كۈلگىنىچە ماشىنىنىڭ دەرىزىسنى چەكتى، سەمەت خوش ياقمىغاندەك بېشىنى كۆتەردى:              
-ھە، ئاخىرى بىر ياتاق چىقتى، ئۇكام سىنى يەنە بىر ئىشقا ئاۋارە قىلىدىغان بولدۇم، خاپا بولماي ئۆيگە بېرىپ ھەدەڭگە بۇ نەرسىنى ئاپىرىپ بەرگەچ، ‹‹ ئەسقەركامنىڭ مۇھىم ئىشى چىقىپ قاپتۇ، ئەتە ئەتتىگەن پەيدا بۇلىدىكەن ›› دەپقۇيە، ھى... ھى... ئۇكام ساڭا تاپشۇردۇم ئەمىسە...                                                                 
ئەسقەر شۇنداق دەپ سەل يىراقتا تۇرغان مودا كىيىنىۋالغان ياش ئاشنىسى بىلەن قولتۇقلىشىپ دېگۈدەك چىراغلىرى پالىلداپ تۇرغان چوڭ مىھمانخانىغا كېرىپ كەتتى، سەمەت قۇلىدىكى بولۇتكىغا قاراپ ئېغىر خۇرسىنىپ قويۇپ ماشىنىنى چوڭ يولغا بۇرىدى...                  
                                                           2
سەمەت ئەسقەرنىڭ ئۆيىگە بارغاندا ئەسقەرنىڭ ئايالى مۇرا چىراي بولۇپ ئولتۇرغانىكەن، ھېلىقى سوۋغاتنى بىرىۋىدى ئايال سوۋغاتنى زەردە بىلەن يەرگە ئېتىپ ئۈزىنى كىرىسلوغا تاشلاپ يىغلاپ كەتتى، سەمەت ئايالغا تەسەللى بەرگەن بولسىمۇ لېكىن ئايال يىغىدىن توختىمىدى، سەمەت ئاخىرى ئامال قىلالماي بېشىنى چايقاپ قويۇپ سىرىتقا ماڭدى، شۇ تاپتا سائەت ئىككى بوپكەتكەن بولۇپ سەمەت ماشىنىنىڭ سۈرئىتىنى تېخىمۇ تىزلەتتى، ئۇ ئۆيىگە كەلگىلى ئاز قالغاندا ئەسقەردىن تېلىفۇن كەلدى:                                                      
-ئۇكام، چاتاق چىقتى، بىزنىڭ پەزىلەت بۈگۈن ئۆيگە قايىتماپتۇ، تىز ئالدىمغا كەل قىزىمنى تاپايلى، تىز بولغىن... ئاپلا، قىز باققاننىڭ سازايىمۇ بۇ ئەمدى...                        
ئۇيقىسىزلىق ۋە ھارغىنلىقتىن ئۆيىگە بورۇنراق قايىتىشنى ئويلىغان سەمەت ئاچچىقىدا رولغا بىرنى قويدىدە ماشىنىنى كەينىگە بۇرىدى، ۋە ئۆز-ئۈزىگە ‹‹ مانا بۇ ئۆتنە جاھان، سەن قىزىڭدەك قىزلارغا قىپ قويغان، مانا ئەمدى قىزىڭ ساڭا كۆرسىتىشكە باشلاپتۇ››                     
سەمەت ئۇيقۇسنى قاچۇرۇش ئۈچۈن يۇقۇرى ئاۋازدا ناخشا قۇيۋەتتى، يول بۇيىدىكى كېچىلىك چىراغلار ئاللا قاچان ئۆچكەن بولسىمۇ مېمانخانا- ياتاقخانىلارنىڭ ئالا يېشىل چىراغلىرى ئەتىراپنى يۇرۇتۇپ ،توختىماي‹‹ مېھمان›› چىللاۋاتقاندەك قىلاتتى...
