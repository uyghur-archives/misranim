ئەسسەلامۇ ئەلەيكۇم مىسرانىم ئەھلى سىلەر بىلەن بۇ تىمام ئارقىلىق دىدارلاشقانلىقىمدىن خۇشالمەن. لىكىن يۈركۈم ئېچىشۋاتىدۇ، ئەپسۇسلىنىۋاتقان ھالدا قەلەم تەۋرىتىۋاتىمەن . لېكىن مىللىتىمنىڭ ئالدىدىكى قەرزىمنى ئادا قىلىش مەقسىتىدە يەنىلا سىلەرنىڭ ھوزۇرىڭلارغا بۇ مۇھاكىمىنى سۇندۇم . پىكىر بېرىشىڭلارنى سورايمەن.  تۈنۈگۈن تور ئارلاۋىتىپ تۇيۇقسىز باشقا قېرىنداش مىللەتلەرنىڭ تۈركلەر توغرىسدىكى مۇلاھىزىسىنى ئۇچرىتىپ  قالدىم . (بۇ خەنزۇچە تور بەت ئىدى) مەن قىزىقىش ئېلىكىدە ئۇلارنىڭ مۇلاھىزىسىگە ئىشتىراك قىلدىم. لىكىن شۇنداق پۇشايمان قىلدىمكى ، بىر كېچە ئىچىمنى پۇشۇرۇپ ئۇخلاتمىغان بۇ <<ئەخلەت>>مۇلاھىزدىن يىرگەندىم.  نېمىشقا ئاچىغلىندىغاندۇر دەمسىلەر؟ سەۋەبىنى "ئاتالمىش مۇلاھىزىچىلەرنىڭ " مۇنۇ ئىنكاسلىرىدىن تىڭشاڭلار. بەيدۇغا تۈرك دەپ باسقىنىمنى بىلمەن (خەنزۇچە) بىر نەچچە تور بەت كۆزنەكلىرى ئېچىلدى . تاللاپ بىرسىگە كىردىم. ماۋزۇسىدا تۈرك ئۇيغۇرلىرى دەپ ، بىر قىسقا فىلىم قۇيۇلدى . كۆرگەن بولسىڭىز ئېدىڭىز ،ئاغزىڭىزنى ئېچىپ قاراپلا قالغان بولاتتىڭىز (خۇددى ماڭا ئوخشاش) كۆرنۈشلەر شۇنداق ھەيۋەتلىك ، ئېسىل ئېدى. قەدىمكى ئۇيغۇر چەۋاندازلىرنىڭ ئات چاپتۇرۇپ، جەڭ قىلۋاتقان كۆرنۇشلىرى مېنى مەمنۇن قىلدى. پەخىرلەندۇردى.(بەلكىم بۇ فىلىمنى تۈركىيە،ۋە ياكى باشقا چەتئەل دۆلەتلىرى ئىشلىگەن بولشى مۇمكىن) ئۇچ مىنۇتتىن كىيىن فىلىم تۈگىدى. ئىكراننى ئاستىغا سىرىدىم. شۇھامان مىنى ئاباياتىندىن بىرى ئازاپلاۋاتقان"ئاتالمىش مۇلاھىزچىلەرنىڭ" پىكرى كۆزۈمگە چېلىقتى . بۇ ئۇلارنىڭ بۇ فىلىمگە بەرگەن بولمىغۇر سەپسەتلىرى ئىدى.  بىرىنجى بولۇپ ئىنكاس يازغان "مۇلاھىزىچى" (كېيىنكى ئىنكاسلاردىن ئۇنىڭ قازاقلىقىنى بىلدىم) مۇنداق يېزىپتۇ: ئۇيغۇرلار تۈركلەر ئەمەس ، تۈرك بولغان بولسا، ھازىرقىدەك ئېشەك مىنىپ يۈرمەيتى. دەپتۇ.               "ئېشەك" ئاغزىمدىن بوشقىنە لىكىن ئاچچىق شۇ بىر ئېغىز گەپ چىقتى . گويا ئۇ مېنى ئېشەك دەپ تىللاۋاتقاندەك بولدۇم. شۇھامان كاللامدا ئېشەك بىلەن ئات توغرىسدىكى بۇ ئوخشىمغان ئىككى خىل ھايۋان توغىرىسىدىكى مۇلاھىزە چىرمىۋالدى. شۇ ھامان ئۇنىڭ ئېيتقانلىرىغا رەددىيە بەرمەكچى بولدۇم ،شۇنچە ئۇرۇنغان بولساممۇ ، خەنزۇتىلى سەۋىيەمنىڭ چەكلىكلىكى، ئۇ تور بەتكە ئەزا بولۇشنىڭ قىيىنلىقى تۈپەيلى ئاچچىق خۇرسىنىش ھاقارەتلىك سۈكۈت ئىچىدە، قاراپ تېڭىرقاپ قېلىشتىن باشقا ئىشنى قىلالمىدىم. كاللامدىكى ئويلار چەكسىز خىيال دېڭىزىغا غەرىق بولدى. ئارقىدىنلا يادىمغا داڭلىق تۈركشۇناس گۇننار يارىڭنىڭ(1907-2002) قەشقەرگە قايتا سەپەر ناملىق كىتابى نامايان بولدى. بۇ ئەسەردە ئاساسلىق قىلىپ ئىككى نۇقتا سۆزلەنگەن. بىرى 20-ئەسىرنىڭ 20~30-يىللىردىكى ئۇيغۇرلارنىڭ ئوتتۇرا ئەسىر ئۇسلۇبىدىكى بازارلىرى، يەنە بىرى بولسا ئېشەكتىن ئېبارەت بۇ ھايۋانىڭ ، ئۇيغۇرلارنىڭ تۇرمۇشىدىكى ئورنى، قاتناش قورالى، ئېشلەپچىقرىش پايدىلنىش قىممىتىنىڭ يۇقىرىلىقىدەك نۇقتىلار ئېنچىكە تەسۋىرلەنگەن.  لىكىن بىز ، ئېشەكتىن ئېبارەت بۇ بىر ھايۋاننىڭ ئىشلىتىشچانلىقى، قوللىنىشچانلىقى تۇپەيلىلا، باشقىلارنىڭ بىزنىڭ تۇرك قان سىستىمىسغا تەۋە ئىكەنلىكىمىزنى ئىنكار قىلشى خاتا ئەلۋەتتە. ھازىرقى زامان شىنجاڭغا نەزەر سالدىغان بولساق، ئۆزگىرىشلەر چوڭ تەرەققىيات سۈرئىتى تېز تۇرسا، قانداقمۇ "ئاتالمىش ئۇ "مۇلاھىزىچىلەرنىڭ" ئېيتقىنىدەك، چارۋىچىلىق قىلىپ ،ئوۋ ئوۋلاپ ،ئات مىنىپ يۈرسەك ھازىرقى دەۋېر تەلپىگە ماسلىشالايمىز. ئېلىم پەن ئارقىلىق ئۆزىمىزنى كۈچلەندۈرۈپ ،تەرەققى قىلالايمىز. يەنە كىلىپ تارىخ بەتلىرىنى ۋاراقلايدىغان بولساق ،ئۇيغۇرلارنىڭ "ئات ئۈستى مەدەنىيتى" ئىنتايىن شانلىق ۋە ئۇزۇن ئىدى. زەپەر قۇچۇپ ئات ئىزلىرنى ، پۈتكۈل ئاسىياغا قالدۇرغان.سىلەر بەلكىم بۇ تېمىنى ئوقۇپ بۇ بولمىغۇ سۆزلەپ، ئېشەك كېرەكمۇ ئات دەپ ، نېمە دېمەكچىدۇر دەپ قالغانلار بولشىڭلار مۇمكىن، لېكىن مەن سىلەر گە بۇنى تاللاڭلار دېمەكچى ئەمەسمەن.  ئېشەكنى ئېلىپ ئېيتساق، ئۇ نەزىرىمىزدە، يۇۋاش، قانائەتچان، جاپاغا چىداملىق، كۆتۈرۈشچان ھايۋان . ئات بولسا، يۈگۈرۈك، جەسۇر، باتۇرلۇقنىڭ سىمۋولى بولغان ھايۋان. لىكىن بىزباشقىلارنىڭ بۇ ئېككى خىل ھايۋاننىڭ ،ئارقا كۆرىنىشىدىن، تارىخنى، مەدەنىيەتنى ،ۋە باشقا سەپسەتلەرنى، توقۇشىغا يول قويماسلىقىمىز كېرەك.  بىر تونۇشۇم مەندىن ، بۇ تېمامنى ئوقۇپ ، ھازىرقى ئۇيغۇرلارغا ئېشەك كېرەكمۇ ئاتمۇ دەپ سورىدى ، مەن ئىككىلىسى كېرەك ئەمەس تەرەققىيات ، ئىتپاقلىق كېرەك دەپ جاۋاپ بەردىم. (بۇ پەقەت شەخسى پىكرىم)  توغرا، ئۇيغۇرلارغا ھازىر بەك كۆپ نەرسە كىرەك.  بۇنى ۋاقىت مۇناسىۋىتى بىلەن دەپ يۇرمەي، دىمىسەممۇ ، بۇ كۆڭلۈڭلەرگە ئايان .  ئەگەر سىلەر شۇ "مۇلاھىزچىلەرنىڭ" دىگىنىنى ھەقىقى كۆرۇپ باقماقچى بولساڭلار ، ئىنكاس شەكىلدە يوللاپ بەرسەم بولدۇ.(ناۋادا خەنزۇتىلى سەۋيىسى يۇقىرراق بولغانلار) سۆزۇمنى يىغىنچاقلىسام، بىز ئۇيغۇرلار قەدىمدىن مەدەنىيەتلىك، زەپەر قۇچقان باتۇر خەلىق. شۇڭا بىز ئەۋلادلار ئەجدادلىرمىزنىڭ كىملىكىنى بىلشىمىز، ئۇلارغا ھۆرمەت قىلشىمىز ، تارىخنى بىلشىمىز ۋە بىيتىشىمىز كېرەك. ئۆزىمىزنى كۈچلەندۈرۈپ،خەلقىئارادا، دۇنيا كۇچلۈك مىللەتلەر قاتارىدىن ئورۇن ئېلىشقا تىرىشىشىمىز  كېرەك.                                     بۇ يازمىنى ئاخىرىدا   ئەلشاھ تەھرىرلىگەن. ۋاقتى  2013-2-11 02:44 PM

ئۆتمۈشىمىز گەرچە نۇرغۇن قېتىم غۇربەتچىكىك ۋە جاپادا ئۆتكەن بولسىمۇ ، قەيسەرلىك بىلەن نۇرغۇن شانلىق ئىز قالدۇرغان خەق بىز،  بەزى «تۈرۈك» قېرىنداش مىللەتلەر شۇ ۋەجىدىن كۆرەلمەيدۇ، ئىچىدىن بىزگە ئۆچمەنلىك قىلىدۇ ، ھەتتا ھازىرقى ئەھۋالىمىزنى زاڭلىق قىلغۇسى كىلىپلا تۇرىدۇ. ئەمىليەتتە بۇنداق پىسخىك ھالەت ئۆزىنىڭ ھالى ئۆزىگە ئايان ئىكەنلىكىنى بىلدۈرۈپ تۇرۇپتۇ.

بۇ يەردە قازاق ئانداق قىرغىز ئانداق دەپ تىللاشنىڭ ئورنى يوق، چۈنكى ئۇلارنىڭ ئىچىدىمۇ قېرىنداشلىق مىھرى بارلار نۇرغۇن.لىكىن يەنە بىر مۇھىم نەرسىنى تەكىتلەپ قۇياي، قېرىنداش تۇققان مىللەت دەپ مىھىر كۈتىدىغان ئىشنى قەتتى قىلىشقا بولمايدۇ، بۇنىڭ كوچىدىكى تىلەمچىلىك بىلەن ھىچقانداق پەرقى يوق، ھەممە ئادەم ھەممە قەۋم ئۆز كۈچىگە تايانغاندىلا قۇدرەت تاپقىلى بولىدۇ.

كىيىنكى ئابزاستىكى سۆزلىرىڭىز بەكمۇ يارىدى، زادى شۇنداق قىلىش كىرەك.
================
مەن تارىختىكى بىر قانچە مىللەتلەرنىڭ ئەھۋالىنى قىسقىچە سۆزلەي دىگەن، لېكىن ئىتتىپاقلىققا تەسىر يەتمىسۇن، (تارىختا ئىلگىرى دۇنيانى سوراپ ھازىر خار بۇلىۋاتقان، ئىلگىرى خار بۇلۇپ ھازىر كۈچىيىۋاتقان مىللەتلەر كۆپقۇ) ئات ئايلىنىپ ئوقۇرنى تاپىدۇ، ئاينىڭ 15ى ئايدىڭ، 15ى قاراڭغۇ، يەر شارىمۇ چۆرگىلەپ تۇرىدىغۇ. بىزمۇ ئۇلۇغ جۇڭگو كوممۇنىستىك پارتىيىسىنىڭ دانا رەھبەرلىكىدە ئۇچقاندەك تەرەققى قىلىمىز.
=================
تېما ئىگىسى خاپا بولماڭ، بىز يەنىلا بىز، بىر ئىككى ياچىۋەكنىڭ ئۇنداق دىگىنىگە شۇنداق بۇلۇپ قالامتى. ئۇنداق گەپلەرگە قۇلاق سالماي، ئورتاق تىرىشساقلا بۇلىدىغۇ.
ئەيىپ- نوقسانلارنى ئاۋۋال ئۆزىڭىزدىن ئىزلەڭ، غەيۋەت قىلماڭ، ئاچچىقىم كىلىپ قالىدۇ
