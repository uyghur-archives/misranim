يەتتە قىزلىرىم مازىرى ئۇچ تۇرپان ناھىيە بازىرىنىڭ مۆلجەر تاغ باغرىنىڭ شىمالى يان باغرىغا جايلاشقان بۇلۇپ ، يەتتە قىزنىڭ قەبرىسى ياغاچ رېشاتكا ئىچىگە ئورۇنلاشتۇرۇلغان . تارىخى مەنبەلەردىن قارىغاندا چىڭ سۇلالىسىنىڭ پادىشاھى چىيەنلۇڭنىڭ 30 - يىللىرى ( مىلادى 1765 - يىلى ) زۇلۇمغا قارشى ئۇچتۇرپاندا يۇز بەرگەن ئاكا - ئۇكا رەخمەتۇللا ۋە ئەسمەتۇللا باشچىلىقىدىكى دېھقانلار قوزغىلىڭى ( جىگدە يېغىلىقى ) قاتناشچىلىرىدىن ئايال پالۋان مايسىخان باشلىق يەتتە قىز نەۋكەرنىڭ مەقبەرەسىدۇر .

ئۇچتۇرپان دېھقانلىرى رەھمىتۇللا ۋە ئەسمىتۇللا بەگلەرنىڭ رەھبەرلىكىدە يىرىم يىلدىن كۆپرەك; ۋاقىت چىڭ سۇلالىسى قوشۇنلىرى بىلەن ئىلىشىپ ئاخىرى مەغلۇپ بولىدۇ ۋە قوزغىلاڭغا قاتناشقان ئىنقىلابچىلارنىڭ ئۇچتىن بىرى قىرغىن قىلىنىدۇ ، 10 مىڭدەك كىشى ئىلىغا سۈرگۈن قىلىنىدۇ ، بارلىق بالايى ئاپەت يۇرتتا قالغان قىزلارنىڭ بېشىغا كېلىدۇ . ئۇچتۇرپان خان ئامبىلى سۇچىڭ ۋە ھاكىمبەگ ئابدۇللا ۋەكىللىكىدىكى مانجۇ ھۇكۇمرانلىرىنىڭ ئۇيغۇر خەلقى ئۇستىدىن دەھشەتلىك يۇرگۇزۋاتقان زۇلۇمغا چىدىمىغان بىر تۇركۇم ئەر - ئايال ئاخىرى رەخمەتۇللا بەگنىڭ سىڭلىسى مايسىخاننىڭ ئەتراپىغا يىغىلىپ تاغ ، جىلغا ، ئۇرمانلىقلاردا يۇشۇرۇن چىڭ سۇلالىسى قوشۇنلىرىغا داۋاملىق زەربە بېرىدۇ . مانا بۇ تارىختىكى  مايسىخان جىڭى ئېيتىلىشچە بۇقىتىمقى جەڭ ئۇچ ئاي داۋام قىلىپ ، مايسىخان بىلەن غۇنچە قىز باشچىلىقىدىكى يەتتە نەپەر ئايال دۇشمەننىڭ قۇرشاۋىدا قىلىپ ئاخىرى تاغدىن ئۆزىنى ئېتىپ مەرتلەرچە قۇربان بىلىدۇ . كىيىن كىشىلەر بۇ يەتتە نەپەر قەھرىمان قىزلىرىمىزنىڭ خاتىرىسى ئۇچۇن مۇلجەر تىغىنىڭ غەربىي شىمالىغا يەتتە قىز مەقبەرىسىنى ياساتقان ئىكەن .

1942 - يىلى ئۇچتۇرپانغا ھاكىم بۇلۇپ تۇرغان لۇجى ( ئەخمەتجان قاسىمى قاتارلىقلار بىلەن بىرلىكتە ئايرۇپىلان ۋەقەسىدە قۇربان بۇلغان ) يازغان  ئۇچتۇرپاندىكى ئاسارى - ئەتىقىلەر ۋە جۇڭگۇ - رۇسيە چىگرىسى تۇغرىسىدا ئىزاھات  دىگەن ماقالىسىدە :  يەتتە قىزىم مەقبەرىسى مۇسۇلمانچە ياسالغان ، ئۇستى يىپىقلىق ، بۇ مەقبەرنى ئۇيغۇرچە يەتتە قىزلىرىم دەپ ئاتايدۇ ، مەقبەرەنى بىر شەيىخ ساقلايدۇ ، ھازىرغا قەدەر ئۇيغۇر كىشلىرى كىلىپ مال ئۇلتۇرۇپ نەزىر بېرىدۇ . ئاياللار دۇئا تىلاۋەت قىلىپ يىغا - زار  قىلىشىدۇ ، بۇ ئۇچتۇرپاندىكى مۇھىم ئاسەر ئەتىقىلەرنىڭ بىرى  دەپ كۆرسەتكەن .دىمەك يەتتە قىزلىرىم مەقبەرىسى 20 - ئەسىردىن بۇيان ئىزچىل تاۋاپكار ۋە زىيارەتگاھ بۇلۇپ كەلمەكتە ، بۇ تاۋاپكار ناھىيە دەرىجىلىك مۇھىم قۇغدىلىدىغان مەدەنىيەت يادىكار ئورنى قىلىپ بىكىتىلگەن .

يەتتە قىزنىڭ ئىسمى : مايسىخان ، غۇنچە قىز ، دىلبەر قىز ، زۇھرە گۇل ، قەمبەرنىسا ، گۇلئايىم ، ئەختەرنىسا .

يەتتە قىزنىڭ ئىچىدە ھەممىدىن نۇچىسى يەنە مايسىخان . قەلەمىي ۋە ئەلەمىي ئىشلارغا ماھىرلىقتا ئۇنىڭ ئالدىغا ھىچكىم يەتمەيدۇ . قايماقتەك سۇزلەپ ئاڭلىغۇچىلارنىڭ ئىچ - باغرىنى مۇمىيادەك يۇمشىتىۋىتىدىغان غۇنچە قىز ئۇلار ئىچىدىكى ئەڭ زۇۋاندازلىرىدىن ، چىچەنلىك ۋە ھازىر جاۋاپلىقتا ئۇ ھەممىنى بىسىپ چۈشىدۇ . كۇپ ۋاقتىنى گۇزەل خىياللارنى سۇرۇش بىلەن ئۇتكۇزىدىغان ئۇيچان قىز دىلبەرنىڭ شائىرلىق ماھارىتىگە ھەممەيلەن تاڭ قالىدۇ . ئۇ ئۇيغا پاتقان چاغلىرىدا يىگىتلەرنىڭ كۇزىگە باشقىچە ئۇماق كۇرىنىپ كىتىدۇ .چىرايىدىن تاڭدىكى چۇلپاندەك نۇر تۇكۇلۇپ تۇردىغان زۇھرە گۇل تەققى - تۇرقى تۇپتۇز ئۇسكەن شەمشاتنى ئەسلىتىدىغان زىلۋا بۇيلۇق قەمبەرنىسا . بۇلاقتەك كۇزلىرى يىگىتلەرنى چۇكتۇرۇۋېتىشكە ئاران تۇرىدىغان شەھلا كۇزلۇك گۇل ئايىم ، دىللارنى چاڭ كەلتۇرۇپ ناخشا تۇۋلاپ تايچاقتەك قىيغىتىپ ئۇيناشقا ئادەتلەنگەن بۇيى ئۇسسىمۇ بالىلىق مىجەزى ئۇسمىگەن ئەركە قىز ئەختەرنىسالارنىڭمۇ ھەر بىرىنىڭ ئۇز ئارتۇقچىلىقلىرى بار .



مەنبە: سەئىدىيە گۈزەل سەنئەت تورى.

زوقلانغۇدەك ئۇيغۇر پەخرى يەتتە قىزلىرىم،
باتۇرلۇقنىڭ نۇرلۇق تەختى يەتتە قىزلىرىم.   
  ئۆزى ئۈچۈن مەڭگۈ ئۆلمەس ئابىدە تىكلەپ،
بىزگە غۇرۇر بەرگەن ئەسلى يەتتە قىزلىرىم.
باھادىرلىق مىدال ئاسقان يەتتە قىزلىرىم،
جەسۇرلىقى تىلدا داستان يەتتە قىزلىرىم.
ئۇيغۇرنىڭ ئاياللىرىمۇ «باتۇر »دىگۈزگەن،   
رەقىبىنى قىلغان چاشقان يەتتە قىزلىرىم .
زۇلۇم ئۈچۈن جەڭلەرئۇرغان يەتتە قىزلىرىم،     
شىھىت بولۇپ پارلاپ تۇرغان يەتتە قىزلىرىم.   
ئادالەت ھۆرلۈك يولىدا جان پىدا ئەيلەپ،
باتۇرلۇقتىن ئۈلگە قۇرغان يەتتە قىزلىرىم.
