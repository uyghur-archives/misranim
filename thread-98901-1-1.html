مىسرانىم مۇنبىرىدىكى بارلىق دوسلار ياخشىمۇسىلەر قۇتلۇق كۈنلەردە يەنە سىلەر بىلەن مىسرانىمدا كۈرۈشكىنىمدىن ئىنتايىن خۇشالمەن .
ئالدىنقى قېتىم مەن ئۇزۇندىن بىرى ئۇيلاپ دوسلار بىلەن پىكىرلىشىش مەقسىتىدە يۇللىغان بالا تەربىيىلەشنى ئۈگىتىش مەكتىپى قۇرۇش پىلانى غا ياخشى پىكىر تەكلىپ بەرگىنىڭلاردىن بەك مەمنۇن بولدۇم .
بىراق بۇ تىمىنى مىنىڭ يۇللىشىم ھەرگىزمۇ مىنىڭ مۇشۇ ئىشنى قىلماقچى بۇلغىنىم ئەمەس بەلكى دوسلارنىۇ بۇ پىلاننى كۈرۈپ قىلالىسا باشقا سودا تۈرىگە مەبلەغ سېلىپ پايدا تاپالماي زىيان تارتىپ ئۈزىنىمۇ ئۈزگە مەبلەغ سالغۇچىنىمۇ ئازاپقا دۇچار قىلغاننى ئاز دەپ مىللەت يۇرت ئالدىدا زىيانلىق نەرسىنى تارقىتىپ (بەزى سودىگەرلىرىمىز قىلىۋاتقان ئىشلار بەكلا ناچار بۇلۇپ بۇ ئىنتايىن ئاز سالماقنى ئىگەللەيدۇ .شۇنداقتىمۇ تەكىتلەش مۇھىم دەپ ئۇيلىدىم.)خەقنى ئالدىغاندىن كۈرە بۇ يېڭى ئۇچۇرنى كۈرۈپ پايدىلىنىپ قالسا شۇ سەۋەپلىك بۇلۇپ بۇنداق ئۇرۇن،مەكتەپ ،جەمئىيەت ،ئىلىمي تەشكىلات،ئۈن سىن بۇيۇملىرى ئۇرۇنلىرى،كىتاپخانا ،نەشرىيات ھەم ئىقتىسادىي قىممەت يارىتىپ ئۈزىنى مەنپەئەتلەندۈرسە ھەم خەلقنى تۇغرا يۇلغا يىتەكلەپ جەمئىيەت مەسىلىلىرىنى تۈزىسە ،ھىچ بۇلمىسا يۇل باشلاپ بەرسە كاللامدا قۇرۇق پىلان پېتى قالغاندىن مىڭ ھەسسە ياخشى دەپ قاراپ بىلىمسىزلىكىمگە قارىماي ئۇتتۇرىغا قۇيغان ئىدىم  .
https://uyghur-archive.com/misranim/thread-98707-1-1.html
يۇقارقى ئۇرۇنغا كىرسىڭىز مەن يۇللىغان پىلاننى كۈرىسىز .
بىراق بەزى دوسلار بۇ پىلاندا پەقەت مەكتەپ قۇرۇش پىلانىلا ئۇتتۇرىغا قۇيۇلۇپتۇ دەپ ئىنكاس يېزىپتۇ ئەمەلىيەتتە مىنىڭ ئۇتتۇرىغا قۇيغىنىم ھەرگىزمۇ مەكتەپنىڭلا ئىشى ئەمەس .
مەن تىمىنى يېزىش جەريانىدا سەۋەنلىك سادىر قىلدىممۇ ؟بىلمىدىم خاتا چۈشەنچىنى قايتا تۈزىتىش نۇقتىسىدا تۇلۇقلاشقا تىگىشلىك نۇقتىلارنى تۇلۇقلاپ يۇللىدىم .
يەنى پىلانىم 4تۈرلۈك ئۇسۇلدا ئەمەلگە ئاشۇرۇلۇش ئىمكانىيىتى بار دەپ قارايمەن .
1.بالا تەربىيىسىنى ياخشىلاش مەقسىتىدىكى ئىلمىي جەمئىيەت (خاراكتىرى ئىلمىي جەمئىيەت )
ئەزا قۇبۇل قىلىدۇ .ئەزالىق پۇلى ئالىدۇ ئاندىن ئەزالىق پۇلىغا داڭلىق پىسخۇلۇگ مائارىپچى، بالا تەربىيىچىسى ،مۇتەخەسسىلەرنى تەكلىپ قىلىپ لىكسىيە ئۇرۇنلاشتۇرىدۇ .قەرەللىك قەرەلسىز ئەزالىرىنى يىغىپ ئەزالىرىغا قانداق بالا تەربىيىلەش مەسىلىلىرى بۇيىچە پىكىر ئالماشتۇرىدۇ .يېڭىلىقلارنى يەتكۈزىدۇ .ئەگەر شارائىتى يار بەرسە پىلاستىنكا چىقىرىپ قەرەللىك تارقىتىدۇ .
2.بالىلار تەربىيىسى مەسىلىلىرى تەشكىلاتى
بۇ ئۇرۇن مەلۇم ئەزالار قۇللىغۇچىلار ئارقىلىق پايدىلىق بىلىملەرنى تۈرلۈك ئۇرۇنلاردا ئۇتتۇرىغا قۇيۇپ ۋە ئۇنى كەڭ تەشۋىق قىلىپ ئۈز خاھىشى يۈلىنىشى بۇيىچە ،پايدىلىق ئۇچۇرلارنى خەنزۇچە ۋە باشقا تىللاردىن تەرجىمە قىلىپ تارقىتىدۇ ،ئۇمۇلاشتۇرىدۇ .ھەرخىل لىكسىيەلەرنى ئۇرۇنلاشتۇرىدۇ .
3.بالىلار تەربىيىسى پايدىلىق بىلىملىرى تەشۋىق كىتاپچىلار (لىنتىلار)
مورتۇمغا ئۇخشاش نەشىر ئۇسۇلىنى ساقلىغان نەشىر گە ئۇيۇشتۇرغۇچىلار چىقىپ
مەخسۇس بالىلار تەربىيىسىگە قارىتىلغان  نەشىر بۇيۇملىرىنى نەشىرگە تەييارلاپ ياكى ئۇرۇنلاشتۇرۇپ مەبلەغ چىقىرىپ نەشىر قىلدۇرىدۇ .ۋە مەلۇم مەنپەئەت بۇيىچە پايدىلىنىدۇ.
ئىزاھات
مەھمۇت قەشقەرىنىڭ قىسىملىق تەربىيە تىپىنى ئالغان لىنتىسى ئىنتايىن ياخشى بازار تاپتى. بۇ نىمىنى چۈشەندۈرىدۇ ؟ بۇ ئۇيغۇر مەدەنىيەت ساھەسىدە تەربىيىۋىي قىممىتى يۇقىرى بالىلار تەربىيىسىدە مۇھىم سالماقنى ئىگەلەيدىغان بۇيۇملارغا بولغان ئىھتىياجنىڭ ئىنتايىن يۇقىرى ئىكەنلىكىنى بىراق بۇنداق بۇيۇملارنىڭ بەك كەمچىللىكىنى كۈرسىتىپ بىرىدۇ.(بۇلۇپمۇ لىنتىلار)
4.بالا تەربىيىلەشنى ئۈگىتىش مەكتىپى
بۇ مەكتەپ تۈرلۈك پايدىلىق كىتاپ پايدىلىق بىلىملەرنى يىغىپ ياكى رەتلەپ ۋە ياكى تەرجىمە قىلىپ ئاھالىلەر كۈپرەك ئۇرۇنلاشقان ئۇرۇنلاردا ئۇتتۇرىغا چىقىپ  داڭلىق ئۇستازلارنى تەكلىپ قىلىپ كۈنلۈك ،ھەپتىلىك ئايلىك ،دەرىس ئۇرۇنلاشتۇرۇپ ياكى قەرەللىك قەرەلسىز لىكسىيەلەرنى ئۇرۇنلاشتۇرۇپ لىكسىيەلەرنى تارقىتىپ مەسىلىلەرنى تۈزۈتۈش ئۇسۇللىرىنى ۋاقتىدا ئۇتتۇرىغا قۇيۇپ ساقلانغان مەسىلىلەرنى ھەل قىلىدىغان مەكتەپ بۇلىدۇ بۇ خىل مەكتەپنىڭ ئىشىنىمىزكى كىيىنلىكى ئىقتىسادىي قىممەت يارىتىش كۈچى،ياراتقان مەنىۋىي قىممىتى ئىنتايىن كۈچلۈك بۇلىدۇ .
ئەپسۇس تاھازىرگىچە بۇنداق پائالىيەت ياكى جەمئىيەت تەشكىلات ۋە مەكتەپ قۇرۇلمىدى. بۇ ئويلىنىشقا تىگىشلىك ئىش ئەلبەتتە.
ئىقتىساد نۇقتىسىدا ئىنتايىن بازار بوشلۇقى بار تۈر
خالىساڭلار 1602050073ئارقىلىق پاراڭلاششاقمۇ بۇلىدۇ .
بۇ پىلانلار مىنىڭ كاللامدا قالغاندىن بىرەر بايراق مەرىپەتپەرۋەر كىشى ئوتتۇرىغا چىقىپ قىلىپ باقسا پىلان كەچكىچە پىلان پېتى قالمىسا دەپ ئۇيلايمەن .
تۇلۇقلاشقا تىگىشلىك نۇقتىلار بار دەپ قارىساڭلار  ياكى چۈشىنەلمەسلىك مەسىلىلىرى كۈرۈلسە ئىنكاس شەكلىدە ئۇتتۇرىغا قۇيساڭلارمۇ بۇلىدۇ .بىلىمسىزلىك كەم ئەقىللىك تۈپەيلى يىتەرسىزلىك مانا مەن دەپ چىقىپ تۇرىدۇ تۈزىتىشكە پۇرسەت ۋە ياردىمى بۇلۇپ قالسا ئەجەپ ئەمەس
