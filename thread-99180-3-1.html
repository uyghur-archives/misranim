بۇنداق يەرلەرگە كىردىغانلارنىڭ كوپرەكى يېزىدىن كەلگەن  دىھقانلار  ۋە باشقا يۇرتتىن شەھەر ئاتلاپ قىمار ئوينايدىغانلار ئىكەن دىققەت قىلسام  
بىز ئۇيغۇرلارنىڭ پۇل تېپىشى شۇنداق تەس تۇرسا نىمىشقا بۇنداق پۇلننى قالايمىقان بۇزۇپ چاچىدىغاندۇ بۇلار قىمار ئوينىغىسى كەلسە 5-6قىمارۋاز بىريەرگە كىلىپ ئۇشۇق ئاتسىمۇ مەيلى ئەمەسمۇ ئۇتتۇرغان پۇل باشقا مىللەتككە كەتكەندىن ياخشاراق

توۋا دەيمەن باشقا جايلاردا تاقىلدىغانلىرى تاقىلىپ بوپتۇ  نىمىشقا كۇچادا تاقىيالمايدۇ
كۇچادا سىياسەت باشقىمۇ يا  ؟
سىياسەت نىمىشقا بىر جاي بىلەن بىر جايدا ئوخشىمايدۇ  يەرلىك سىياسەتنى يۇقۇردىن باشقۇرمامدۇ دەيمەن
مەسلەن كۇچادا ئۆتكەن يىللاردىن كونا ئۆيلەرنى چېقىش ئەۋچ ئېلىپ كەتتى  ئۆينى چاققاندە 700-800 يۇئەندىن چاقتتى  ئۆزلىرنىڭ ئۆيلىرنى ساتقاندا  يۇقىرى باھادە ساتتى ھازىر كۇچادا ئۆيلەرنىڭ باھاسسى ئۈچ مىڭ نەچچەيۇز يۇئەن بولىۋاتىدۇ
ئۇندىن باشقا يەنە شوپۇرلارنىڭ  يۇك يۇشۇش  يۇلۇچى تۇشۇش ئىشقا چىقىش گۇۋاھنامىسنى ‹شاڭگاڭجېڭ›نى يىلدا بىر تەكشۇرتۇش يولغا قويدى تەكشۇرتۇش ھەققىمۇ ئەرزان ئەمەسكەن باششقا جايلاردا ئاڭلىسام بۇنداق ئىش يوق ئىكەن 6 يىلدا بىر تەكشۇرتىدىكەن  
ئىشقىلىپ بىزنىڭ بۇ كۇچا  ئاجايىپلا بىر يەر

شەكلى ئۆزگەرگەن ئۇيۇنچۇقخانىلار قۇرۇلىۋىدى،ئۆيلىرىنى ساتقان ياكى ھۆكۈمەت تەرەپتىن قانۇنلۇق چىقىلغان يەرنىڭ ئىگىللرى بولمىش ئاددىي كاللىغا ئىگە بىر قىسىم ئۇيغۇرلىرىمىز،بولۇپمۇ يىزىدىن كەلگەن ساددا دىھقانلىرىمىز قولدىكى ئىككى توك توك پۇلغا مەستانە بولۇشۇپ،ئۆزىچە مەغرۇر بولۇپ قىمارخانىغا چىپىشتى،بۇنداق ئورۇندا ھىلى ئۇرۇش جىدەل،بەزىدە ئۇتۇپ بەزىدە ئۇتتۇرۇپ قۇيۇپ گاھ گاداي قەلەندەر،گاھ شاھ سۈپەت مەغرۇر بولۇشۇپ  ئاخىر ئۇتقىنىدىن ئۇتتۇرغىنى كۆپىيىپ نەچچە ئائىلە ۋەيران،نەچچە ياشلار ساقچىدا،ياكى قەرىز تۆلەش بىلەن ئۆتىۋاتىدۇ.پات-پات مەدىنىيەت ئورۇنلىرى تەكشۈرۈپ تۇرسىمۇ لىكىن پۈتۈنلەي چەكلەش ھۇقۇقى ۋە پۇخرالارنى توغرا يولغا باشلاش ئىمكانىيتى بولمىغاچقا پەقەتلا ۋاقىتلىق توختۇتۇپ قۇيۇشقا مەجبۇر بولدى.مەدىنىيەت ئورۇنلىرىغىمۇ ئامال يۇقتە،خەقنى بۇ يەرگە كىرمە،ئوينىما دىيەلمەيدۇ.لىكىن ئەقلى يۇق،ئىغىزدا شەھەر ئالىدىغان ياشلار توپى زورىيىپ ئاخىر جەمىيەت تەرەققىياتىغا پايدىسىز ئەھۋاللار كىلىپ چىقتى.بەزى ناھىيە ۋە شەھەرلەردە بۇنداق شەكلى ئۆزگەرگەن قىمارخانىلار تاقالغان بولسىمۇ لىكىن بەزى جايلار ھامان مەۋجۇت بولۇپ تۇرماقتا.ساقچى باشقۇرمامدىغاندۇ دەيدىكەن خەقلەر،ساقچىدا پۇخرانى نىمىشقا بۇ يەرگە كىرىپ ئوينايسەن،بۇ بىر جىنايەت دەپ ئەكىرىپ سولاپ قۇيالمايدۇ،گەرچە ئۇيۇن ئۇيناش پۇخرانىڭ ھۇقۇقى بولسىمۇ لىكىن بىر ئىشنى قىلسا ۋايغا يەتكۈزمىگۈچە بولدى قىلمايدىغان بۇ ياشلار توپى ئاخىر قاق سەنەم بولۇپ بۇرنى قۇلاققىچە تۇيغاندىمۇ يەنە شۇنداق يەرنى ماكان تۇتۇپ يۈرگەنلەرمۇ بولدى.ئۇلارغۇ مەيلى،قىزلارنىڭ ۋە كىچىچە ئۇيۇن ئۇيناپ يۈردىغان قىزلارنىڭمۇ پەيدا بولىشى ئادەمنى ھەقىقەتەن ئويغا سالماي تۇرالمايدۇ.مەن جىددى بىر ىش بىلەن بىراۋنى ئىزدىمەكچى بولۇپ  تىلپۇن ئۇرسام پالانى جايدىكى بىر ئۇيۇنخانىدا ئىكەنلىكىدىن خەۋەر تاپتىم دە باردىم،كۆرگىنىم شۇ سىلەر ئىيتقاندەك ،ئۈرۈمچى ۋاقتى كىچە سائەت 12 لەردىمۇ نۇرغۇن ياشلار ۋە ئەرخوتۇنلارمىكىن دەپمۇ قالدىم،قىزلارمۇ باركەن.ئۇ ئادەمنى تىپىپ زۆرۈر ئىش بارلىقىنى ،تىزدىن بىجىرىش كىرەكلىنى ئيسام يىرىم سائەت تۇرۇپ تۇرساڭ دەيدۇ.ماقۇل دەپ تۇرۇپ قاپتىمەن،ئالاھەزەل ئون مىنۇتلاردا بىر ئورانغان،بىر موداراق ئىككى قىز كىرىپ كەلدى دە باش ئورۇنغا بىرىپ يۈز يۈەنلىك پۇلدىن ئۈچ تۆتنى بىرىپ ھەممىسىگە تەڭگە ئىلىپ بىر يەرگە بىرىپ ئۇ تەڭگىلەرنى تاشلاشقا باشلىدى،تاراق تۇرۇق،قىيداش،تۇرۇپ تۇرۇپ قوللىرى بىلەن ئويناۋاتقان ماشىنىسىنى مۇشىتلاپمۇ قۇياتتى،ئاخىر ئورانغان قىزغا تىلپۇن كەلدى بولغاي،بىر نىمىلەرنى دەپ تىلپۇننى قۇيۇپ يىنىدىكى ھەمراھىغا:ئاپام تىلپۇن قىپتۇ قايتامدۇق دىسە ھەمراھى:نىمانداق تولا تىلپۇن قىلىدۇ،بىز يا يۈتۈپ كەتمىسەك،يا يىتىپ قالمىساق دەپ قۇيۇپ يەنە تەڭگە ئالغىلى مىڭىپ خۇجايىنغا بىر نىمىلەرنى دىۋىدى خۇجايىنى خۇشال ھالدا بىرمۇنچە تەڭگىنى بىرىۋەتتى.بىردەمدىلا تەڭگىسىنى تۈگىتىۋەتكەن بۇ قىزلار خۇجايىنغا يالۋۇرۇپ يىنىدا پۇل قالمىغانلىقىنى ،بەش يۈەن بەرسە كىرا قىلىپ قايتىدىغانلىقىنى ئىيتتى،خۇجايىن موداراق قىزنىڭ قۇلىقىغا بىر نىمىلەرنى دىۋىدى ئۇ قىز تىلپۇن نۇمۇرىنى ئىيتىپ بەردى بولغاي خۇجايىنى يىنىدىن 50 يۈەننى چىقىرىپ قول ئىلىشىپ خۇشلاشتى.ئىشقىلىپ بۇ مەنزىرىگە سەل غەزىپىم تىشىپ قالدى،ئاڭغىچە مەن ساقلاۋاتقان ئادەمنىڭمۇ ئۇيۇنى تۈگەپ قالدى،بىرلىكتە يىنىپ چىقتۇق،كۆرگەنلىرىمنى سۇرىسام :ئاداش،بۇنداق قىزلارغا ئىشەنگىلى بولمايدۇ،ئورانغان بىلەن مەقسەت باشقا،بۇ خۇجايىننىڭمۇ بۇنداق قىزدىن نەچچە تۇنۇشى بار.دىمەسمۇ.كاللامغا سۇ قۇيۋەتكەندەك بولۇپ قالدىم.سەنمۇ بۇنداق يەرگە كىرىپ ئۇيۇن ئۇينىغۇچە ئۈيۈڭگە بىرىپ خۇتۇنۇڭ بىلەن ئۇينىساڭچۇ دىسەم:قىمار نىڭ خۇمارى يامانكەن،سىنىڭ قىرىق خۇتۇنۇڭنىڭ كۆڭلىمۇ مۇشۇ نىجىسچىلىك بولسىتى كاشكى ،ئۇينىماي دەيمەن،ئۆگىنىپ قاپتىمەن،دەيدۇ.ھازىر ئۇنىڭ باشقىلارغا تۆلەيدىغان قەرزى ئونمىڭدىن ئارتۇق،ئايالى يامانلاقتا،خۇدايىم بەرگەن ئىككى ئوغۇلنىڭ چوڭى مەكتەپ يىشىغا يەتتى،ئىچىم ئاغىرىپ قالىدۇ.ئىشقىلىپ بۇنداق يەرنى تاقىۋەتسە ياخشى بولىدۇ،بۇنداق يەر كىشىلەرنىڭ ئىشلەپ چىقىرىش ۋە جەمىيەتنىڭ تەرەققىياتىغا ئىنتايىن پاسسىپ تەسىر كۆرسىتىدۇ.ئۇنى بۇنى دەيمىز،ھۆكۈمەت بۇنداق يەرنى تاقىۋەتكىنى بىلەن قارتا ئۇيناپ بىر بىرىنى سۇرىۋىتىدىغان ،توخۇ سۇقۇشتۇرۇپ بىر بىرى بىلەن قىمار ئۇينايدىغان ئەھۋاللارمۇ يامان جىق.ئىشقىلىپ بۇنداق كىشىلەرگە قاراپ مىللەتنىڭ كىيىنكى تەغدىرىگە ئىچىنىش بىلەن قاراپ قالىمەن.


يەنە بىرى،مۇشۇ ئىچكىردىن چىققان قانۇنسىز ۋە قانۇنلۇق پۇخرالار شىنجاڭدەك يىرى كەڭ،مۇھىتى ساپ،تەبىئىي جۇغراپىيەسى ئۆزگىچە،خۇاشىيا مەدىنىيتىدىن تۈپتىن پەرىقلىنىدىغان  مەدىنىيەتكە ئىگە زىمىنغا چىققاندىن كىيىن پاختا تىرىپ،قۇرلۇش ئورۇنلىرىدا ئىشلەپ،ئەخلەت تىرىپ يۈرۈپ نۇرغۇن زىمىن ۋە تىرىلغۇ يەرگە ئىگە بولدى،ئەمما بىزلەرچۇ؟يەرلىرىنى سىتىپ تاپقان پۇلىغا قىمار ئۇيناپ ياكى خۇتۇن يەڭگۈشلەپ ئويۇن تاماششاغا چۆكۈپ كەتتى،ئاقىۋەتتە بىر ئوبدان خۇتۇن ئىرىدىن خىتىنى ئىلىپ تۇرمۇشى غۇربەتچىلىكتە قالغاچقا ئامالسىز بەزى بىر كىشىلەرنىڭ تەلەپلىرىنى قۇبۇل قىلىپمۇ يۈرۋاتىدۇ.شۇ خەقمۇ مۇشۇ نىجىس قىمارخانىغا تىكىۋەتكەن پۇلىغا ئۆيدىكى خۇتۇنىغا كىيىم،ياشىنىپ قالغان ئاتىسىغا گۆش،بالىلىرىغا ئىلغار ئۇقۇش قۇراللرى سىتىۋالسىچۇ؟مۇشۇنداق كىشىلەر ئوينادىغاننى ئۇيناپ قەرىزدار بولۇپ خەقتىن پۇل پۈتمىگەندە بولسا ئاتا ئانىسىدىن ئالىدۇ،ئۇلارنى قاخشال قىلغاندا خەقتىن سۇرايدۇ،ئۇلاردىن پۈتمىسە:بىز ئۇيغۇرچۇ مۇشۇداق خەق دەپ مىللەتتىن چۈشىدۇ،خۇمسى سولامچىلار.دائىم ھۆكۈمەت،ھۆكۈمەت دەپلا زارلايدىغان قايسى زامان بۇ،خەقمۇ ئاۋۇ ئىككى مۈرىدىكى قاپاق ئەمەس كاللا بولغاندىكىن بۇنداق يەرنىڭ قانداقلىقىنى بىلگەندىكىن،بىلمىسە بىرەر قىتىم كىرىپمۇ بىلمەمدىكەن دەيمەن،كىرمىسە بىرى بۇينىدىن سىرىتماق سىلىپ سۆرەپ ئۇينىمىساڭ ئۆلتۈرىمەن دىمىگەنكىن نىمە بار ئۇ يەردە.قارىسىڭىز جۇمەدە،ئىككى ھىيىتتا نامازدا يۈرگەنئاخشىمىسى قارىسىڭىز تاماكا دىگەننى ***نى شۇرىغاندەك شۇراپ ئولتۇرۇپ كەتكەن خۇمسى گۇيلار.
