  شىنجاڭ ئۇنۋېرسىتېتى يۇمشاق دېتاللار ئىنستېتوتىنىڭ ئوقۇغۇچىسى ئۆتكۈر ئەركىن 2012-يىلى 11-ئاينىڭ 23-كۈنى جۇڭگو كۆچمە خەۋەرلىشىش شىركىتىنىڭ ئويۇشتۇرۇشى بىلەن گۇاڭجۇدا ئۆتكۈزۈلگەن «جۇڭگو كۆچمە خەۋەرلىشىش  MM (Mobile Marketنىڭ قىسقارتىلمىسى) مىليونلىغان ياشلىرىنىڭ ئىگىلىك تىكلەش مۇسابىسىقىسى»دە 2012-يىللىق چېمپىيون بولدى. يېقىندا شىنجاڭ تېلۋىزىيە ئىستانسى 5-قانىلىنىڭ ‹‹مەدەنىيەت بوستانى›› پروگراممىسىدا شىنجاڭ ئۇنۋىرىستىتى يۇمشاق دېتال ئىنىستىتۇتىنىڭ ئوقوغۇچىسى ئۆتكۈر ئەركىننىڭ ئەمگەك مىۋىسى ۋە شان-شەرىپى توغرىسىدا شىنجاڭ ئۇنۋىرىستىتى يۇمشاق دېتال ئىنىستىتۇتىنىڭ مەسئۇل ئوقۇتقۇچىلىرى ۋە  بۇ شان-شەرەپنىڭ ئىگىسى ئۆتكۈر ئەركىن قاتارلىقلار بىلەن ‹‹ياشلار ۋە ئىزدىنىش روھى›› تېمىسىدا مەخسۇس سۆھبەت ئۇيۇشتۇرۇلۇپتۇ. تەپسىلاتىنى تۆۋەندىكى ئورۇندىن كۆرۈپ چىققايسىلەر:

تۆۋەندە ئۆتكۈر ئەركىن ۋە ئۇنىڭ مۇۋاپىقىيىتى توغرىسىدا قۇتيار مۇنبىرىدىن كۆچۈرۈلگەن مەزمۇن:


ئۆتكۈر ئەركىن يۇمشاق دېتال ساھەسىدە ئۈچ تۈردە مەملىكەت بويىچە مۇكاپاتلىنىپ، شىنجاڭ خەلقىگە شان-شەرەپ كەلتۈردى


     شىنجاڭ ئىنىۋېرسىتېتى يۇمشاق دېتاللار ئىنستېتوتىنىڭ ئوقۇغۇچىسى ئۆتكۈر ئەركىن 2012-يىلى 11-ئاينىڭ 23-كۈنى جۇڭگو كۆچمە خەۋەرلىشىش شىركىتىنىڭ ئويۇشتۇرۇشى بىلەن گۇاڭجۇدا ئۆتكۈزۈلگەن «جۇڭگو كۆچمە خەۋەرلىشىش  MM (Mobile Marketنىڭ قىسقارتىلمىسى) مىليونلىغان ياشلىرىنىڭ ئىگىلىك تىكلەش مۇسابىسىقىسى»دە 2012-يىللىق چېمپىيون بولدى. ئۇنى ئېلىمىز بويىچە چىمپىيونلۇققا ئېرىشتۈرگەن يۇمشاق دېتالى، ئۇيغۇرچە تور كۆرگۈ 2.0  ئىدى.
(بۇ دېتالنى چۈشۈرۈش مەنزىلى:http://mm.10086.cn/1007/300002641714.html?p=&o=a&jid=-110600258)     بۇ قېتىملىق مۇسابىقىگە ئەسلىدە مەملىكەت بويىچە بىر مىليون 600 مىڭ كىشى ئۆزى ئىشلىگەن يۇمشاق دېتاللىرىنى قاتناشتۇرغان بولۇپ، دەسلەپكى قەدەمدە ئۇلاردىن ئالدىنقى 30 پارچە ئەسەر تاللانغانىدى. شۇ ئەسىرى تاللانغان 30 كىشى مۇسابىقىگە چاقىرتىلىپ، قاتنىشىش شەرىپىگە ئېرىشەتتى. گەرچە، بۇ مۇكاپات تۈرى 2010-يىلىلا تەسسىس قىلىنغان (مۇكاپات تۈرىنىڭ نىشانى مېنىڭ ھېس قىلىشىمچە: ياشلارنى ئىشسىز قالماي يۇمشاق دېتال ئېچىش ئارقىلىق ئىگىلىك تىكلەش يولىغا يېتەكلەش، شۇنداقلا جۇڭگو كۆچمە خەۋەرلىشىش تەرەققىياتىنى ئىلگىرى سۈرۈشكە پايدىلىق بىر تۈركۈم يوشۇرۇن كۈچلەرنى ئۆز ئەتىراپىغا توپلاش ئارقىلىق كۆچمە خەۋەرلىشىش ئىشلىرىنى ئىلگىرى سۈرۈش ئاساسىي مەقسەت قىلىنغان. شۇڭا، مۇكاپاتقا قاتنىشىدىغانلار گەرچە چەكلەنمىسىمۇ كۆپىنچىسى ستۇدېنتلار ياكى ئالىي مەكتەب ئوقۇپ ئىش كۈتىۋاتقان كەڭ ياشلار ھەمدە يۇمشاق دېتال ئېچىش شىركەتلىرىنى قۇرۇپ، مۇشۇ ساھادە نەچچە يىللاپ ئىقتىسادىي ۋە ئىجتىمائىي ئۈنۈم يارتقانلار، يېڭىلىق ياراتقانلار) بولسىمۇ، ئالدىنقى ئىككى قېتىمدا شىنجاڭدىن بىرمۇ كىشىنىڭ ئەسىرى كىرەلمەي ئاق پېتى تاشلىنىپ قالغانىدى. بۇ يىل ئۆتكۈر ئەركىن شىنجاڭغا نىسبەتەن بۇ بوشلۇقنى تۇنجى بولۇپ تولدۇرغان، مۇنداقچە ئېيتقاندا شىنجاڭدىن بۇ مۇسابىقىگە قاتنىشىش سالاھىيىتىگە ئېرىشكەن بىردىن-بىر كىشى بولۇپ قالغان ئىدى.     شۇ كۈنى ئۇنىڭغا تېلېفوندا، ئۆزىنىڭ ئالدىنقى قاتاردىكى 30نىڭ بىرىگە كىرگەنلىكى، ئەتە ئىلمىي دوكلات بېرىدىغانلىقى، كېلىپ-كېتىدىغان ئايروپىلان بېلىتىنىلا ئەمەس، ياتاق ۋە باشقا چىقىملارغىچە جۇڭگو كۆچمە خەۋەرلىشىش شىركىتىنىڭ ئۆز ئۈستىگە ئېلىپ ئالدىن ئورۇنلاشتۇرۇپ قويغانلىقى ئۇقتۇرۇلغانىدى.
     بۇ خەۋەردىن ئۆتكۈر ئەركىن ھەم خۇش بولدى، ھەم جىددىيلەشتى. نېمىلا قىلمىسۇن، ۋاقىت ئۇنىڭغا يېتىشمەيتتى. شۇ ۋەجىدىن شۇ كۈنى قويغان-تۇتقىنىنى بىلمەيلا قالدى.
     ئۇنىڭ ئويلىغىنى، ئۆزى ئىشلىگەن ئىككى تۈردىكى يۇمشاق دېتال ھەققىدە ئىلمىي دوكلات يېزىش، PowerPoint شەكلىدە ئۆلچەملىك تەخلىۋېلىش، ئۇندىن باشقا شىنجاڭدىكى ئۇيغۇر خەلقىنىڭ بىر پەرزەنتى سۈپىتىدە قاتناشقىنى ئۈچۈن ئۆز خەلقىنىڭ ئوبرازىغا ۋەكىللىك قىلىدىغان دوپپىغا ئوخشاش مىللىي پاسوندىكى كىيىملەرنىمۇ تەخلىۋېلىشى، سەپەر تەييارلىقىنى قىلىۋېلىشى كېرەك ئىدى. ئەمما، بارى-يوق يېرىم كۈنلۈك ۋاقىت ھېچنەرسىگە يەتمەيتتى. شۇڭا، ئۇ «مېنى نەتىجە ئېرىشەلمىسۇن دېگەندەك، مۇنداقچىلا چاقىرىپ قويغان ئوخشايدۇ. ھەقىقىي چاقىرغان ئادەم، مۇدەتتىن بىر نەچچە كۈن بۇرۇن ئۇقتۇرۇپ قويغان بولسا ياخشى بولماسمىدى» دەپ پۇچىلاندى ھەمدە بۇ قېتىملىق مۇسابىقىدىن سەل ئۈمىدسىزلەنگەنمۇ بولدى. ئەمما، نېمىلا بولمىسۇن بۇنداق پۇرسەت ئاسان قولغا كەلمەيتتى.
     شۇ كۈنى ئۇ كېچىچە تۈزۈك ئۇخلىيالمىدى. ئۇنىڭ ئۈستىگە، شۇ كۈنلەردە قېلىن قار ياققانىدى. قار ياغسا، يولدا كېتىدىغان ۋاقىتقا ئالدىن ھۆددە قىلىپ بولمايتتى. شۇڭا، ئۇ  ئۇيقىسى قانمىغان بولسىمۇ ئۈرۈمچى ۋاقتى سائەت 4 تىلا ئورنىدىن قوپۇپ، ئايرۇدۇرۇمغا قاراپ يولغا چىقتى.
     ئۇ ئالدىرىغان بولسىمۇ ئايرۇپىلان ئالدىرىمىدى. ئۇچۇش ۋاقتى كەلگەن بولسىمۇ ھاۋا ناچار بولغاچقا ئۇچۇش كېچىكتۈرۈلدى، ئايرۇپىلان نەچچە رەت ھاۋاغا كۆتۈرۈلۈپ باققان بولسىمۇ ئوڭۇشلۇق ئۇچالماي يەنە ئايرۇدۇرۇمغا يېنىپ قوندى. شۇنداق قىلىپ، ئۇچۇش ۋاقتى توپ-توغرا تۆت سائەتلەپ ئارقىغا سۈرۈلگەنىدى. بۇنىڭ بىلەن، ئۇنىڭ بەزى چالا تەييارلىقىمنى بېرىپلا مۇكەممەل قىلىۋالارمەن، دېگەن نىيىتىمۇ سۇغا چىلىشىدىغاندەك ئىدى. تەلىيىگە يارىشا، ئۇ گۇاڭجۇغا چۈشكەن كۈنى ئەمەس، ئەتىسى دوكلات بېرىشكە ئورۇنلاشتۇرۇلغانىدى. بۇ دېگەنلىك، يېرىم ھەل قىلغۇچ مۇسابىگە قاتنىشىش دېگەنلىك ئىدى.
     ئەتىسى (11-ئاينىڭ 23-كۈنى) يېرىم ھەل قىلغۇچ مۇسبىقىگە قاتنىشىشقا توغرا كەلدى. ھەر ھالدا، ھېرىپ-چارچىغان بولسىمۇ ھەممىنى ئۇنتۇپ، چالا قالغان ئىلمىي دوكلات تەييارلىقلىرىنى ئاز-تولا قىلىپ ئۈلگەرگەنىدى. ئەپسۇس، ئەسىرى قاتناشتۇرۇلغان، مۇسابىقىگە كىرگەن ئىچكىرى ئۆلكىدىن كەلگەن باشقا ياشلار بىر-بىرىدىن خىل، سەۋىيىسى يۇقىرى، تەييارلىغان ئىلمىي دوكلادىمۇ سۇ كىرمىگۈدەك پۇختا ئىدى. ئۇلارنىڭ كۆپىنچىسى، يانفون ئويۇنچۇقلىرىنى لايىلىگەنلەر بولۇپ، رىقابەتچىلىرىنىڭ ھەيران قالارلىق يۇقىرى پروگرمما سەۋىيىسى ئۇنى تەمتىرىتىپ قويغانىدى.
     23-چىسلادىكى يېرىم ھەل قىلغۇچ مۇسبىقە نەتىجىسىنى ئېلان قىلىش ۋاقتى ئاخىر يېتىپ كەلدى. كىم بىرىنچى بولدىكىن؟ ھەممىنىڭ قۇلىقى دىڭ، يۈرەكلىرى ھاياجانلىق گۈپۈلدەپ سوقاتتى... سەھنىدە مۇكاپاتلانغۇچىلار تىزىملىكىنى ئېلان قىلغۇچى: «بۇ قېتىملىق يېرىم ھەل قىلغۇچ مۇسابىقىدە بىرىنچى بولغان كىشى...» دەپ جاراڭلىق ئاۋازىنى سوزۇپ بىردىن توختاپ قالدى. ئۆز كۆزىگە ئىشەنمىگەندەك ياكى خاتا ئوقۇپ سالغاندەك، كۆز ئەينەك تاقىغان كۆزىگە تىزىملىكنى يېقىن تۇتۇپ بىر ھازا تۇرۇپ كەتكەندىن كېيىن، ئاندىن: «گەرچە، شىنجاڭدىن كەلگەن بولسىمۇ... نەتىجىسى ئالاھىدە ياخشى بولغان ئوقۇغۇچى ئەركىن!» دەپ جاكارلىدى (ئۇلار ئۆتكۈر ئەركىن دېسە، ئەركىن دېگىنى ئۇنىڭ ئىسمى دەپ خاتا ھالدا دادىسىنىڭ ئىسمى بويىچە ئاتاپ قالغانىدى).
     ئىش بۇنىڭلىق بىلەن ئاخىرلاشمىغانىدى. ئالدىنقى 30 دىن، يېرىم ھەل قىلغۇچ مۇسبىقە ئارقىلىق ئالدىنقى 10 تاللىۋېلىنىپ، قالغان 20 سى قايتۇرۇلدى. ئەمدى مۇشۇ 10 كۈچلۈك رىقابەتچىلەر ئەتىدىكى ھەل قىلغۇچ مۇسابىقىدە ئۈستۈنلۈك تالىشاتتى."
مۇتەخەسسىلەر، ئۇستازلار ئۇنى ئەتىكى سەھنىدە قانداق ئۈستۈنلۈك تالىشىش، قانداق غەلىبە قازىنىش ھەققىدە كۈچەپ يېتەكچىلىك قىلدى، كۆپ ئەجىر سىڭدۈردى. شۇڭا، ئۆتكۈر ئەركىننىڭ ئاتا-ئانىسىدىن تېلېفون كەلسىمۇ، ئۇلار تېلېفوننى ئالماسلىققا دەۋەت قىلاتتى. ئاتا-ئانىسىمۇ نېمە ئىشلارنىڭ بولىۋاتقىنىدىن ئۇدۇللۇق خەۋەردار بولۇش ئۈچۈن كۆزى تۆت بولسىمۇ، بۇنداق ئۇچۇردىن خەۋەردار بولۇشمۇ تەسكە چۈشەتتى.
     2012-يىلى 11-ئاينىڭ 24-كۈنى كەسكىن ئېلىشىش بولدى. ئۆتكۈر ئەركىننىڭ دادىسى ئوغلىنىڭ بىرىنچى بولۇشىغا تازا ئىشەنچ قىلالمايتتى، ئۇنىڭچە بولغاندا، مۇشۇ يېرىم ھەل قىلغۇچ مۇسابىقىگە كىرەلىگىنىمۇ ئوغلۇم ئۈچۈن مۆجىزىدەك چوڭ بىر ئىش بولدى، دەپ قارايتتى. شۇڭا، ئوغلىنىڭ بۇ نەتىجىسى يېتەرلىك، ئۇ چىمپىيۇن بولالمىسىمۇ كۆزى ئېچىلىپ، تېخىمۇ تىرىشىش نىيىتىگە كەلسىلا شۇ يېتەرلىك، دەپ ئويلايتتى.

     24-چىسلادىكى ئېلىشىشنىڭ نەتىجىسى ئېلان قىلىنغاندا، ئۆتكۈر ئەركىنلا ئەمەس، ئۇنى ئاڭلىغان ئاتا-ئانىسىنىڭمۇ بىر ئىشىنىپ، بىر ئىشەنگۈسى كەلمەي قالدى. چۈنكى، بۇ خەۋەرنى ئۆتكۈر ئەركىننىڭ گۇاڭجۇدىكى ستۇدېنت دوستلىرى ئۇنىڭ ئاتا-ئانىسىغا تېلېفوندا قىسقىلا قىلىپ يەتكۈزگەنىدى. شۇنداق بولغاچقا، دوستلىرى بىلمەي ئالدىنقى كۈنىدىكى يېرىم ھەل قىلغۇچ مۇسابىقىدە بىرىنچى بولغىنىنى، مەملىكەت بويىچە بىرىنچى بولدى دەپ ئويلاپ قېلىپ خاتا يەتكۈزۈپ قالغاندەك تۇيغۇمۇ بەرگەنىدى. راست-يالغانلىقىنى ئۆز ئېغىزىدىن ئاڭلاش ئۈچۈن دادىسى نەچچە رەت تېلېفون بەردى، ئۇچۇر قىلدى، سادا يوق. يەنە باشقىلاردىن ئۇقۇشۇپ بېقىپ، راستلىقىغا ئارانلا ئىشەندى-يۇ، تۇرۇپلا يەنە گۇمانلانغۇسى كېلەتتى: شۇنچە كۈچلۈك رىقابەتچىلىرى تۇرغان يەردە، ئۇنىڭغا نۆۋەت تېگىشى مۇمكىنمۇ؟ دېگەننىلا ئويلايتتى دادىسى.
كەچكە يېقىن تېلېىفونى ئاخىرى ئۇلاندى. بۇ ۋاقىتتا، دادىسى ئوغلىنىڭ شان-شەرىپىدىن تېخىمۇ ھاڭ-تاڭ بولۇپ تۇرۇپ قالدى. چۈنكى، ئۆتكۈر بىرلا ۋاقىتتا ئۈچ تۈردە مەملىكەت بويىچە مۇكاپاتقا ئېرىشكەن ئىدى.
     بۇ ئۈچ تۈر تۆۋەندىكىدەك ئىدى:
     1. «جۇڭگو كۆچمە خەۋەرلىشىش  MM (Mobile Marketنىڭ قىسقارتىلمىسى) مىليونلىغان ياشلىرىنىڭ ئىگىلىك تىكلەش مۇسابىسىقىسى»دە 2012-يىللىق چېمپىيون.» (中国移动“MM百万青年创业大赛” 2012年冠军)
      2. «مەملىكەت بويىچە MM مىليونلىغان ياشلىرىنىڭ ئىگىلىك تىكلەش مۇسابىسىقىسىدە ئۈچ يىلدىن بۇيان مۇكاپاتلانغان ئەسەرلەرنى خۇلاسىلەپ ئومۇمىي باھالاشتا تاللانغان 10 داڭدار زات» (نىڭ بىرىنچىسى) بولۇپ مۇكاپاتلانغان. (全国“MM百万青年创业大赛三年总评十大风云人物” 一等奖)
      3. ئۆتكۈر ئەركىننىڭ مۇكاپاتقا يەنە بىر ئەسىرىمۇ تاللانغانىدى. بۇ ئەسەر، مەھسۇلاتنىڭ جىڭ-ساختىلىقىنى تەكشۈرۈش سىستېمىسى بولۇپ، «جۇڭگو كۆچمە خەۋەرلىشىش  مىليونلىغان ياشلىرىنىڭ ئىگىلىك تىكلەش مۇسابىسىقىسىدە ئالىي مۇكاپات»قا ئېرىشكەنىدى. (中国移动“MM百万青年创业大赛” 优秀奖)     بۇ مۇكاپاتتا، ئۆتكۈر ئەركىنگە مۇكاپات لوڭقىلىرى تەقدىم قىلىندى. ئۇلار ئۆتكۈر ئەركىنگە، 2012-يىلى 12-ئاينىڭ 5-كۈنى دۇنيا بويىچە نوپۇزلۇق پروگرامىچىلار چوڭ يىغىلىشى بولىدىغانلىقىنى، مۇكاپاتىنىڭ شۇ ۋاقىتتا ئاشۇ جايدا تارقىتىپ بېرىلىدىغانلىقىنى، مۇكاپات سوممىسىنىڭ 240 مىڭ يۈەن ئىكەنلىكىنى ئېيتتى.
     ئاڭلاشلارغا قارىغاندا، جۇڭگو كۆچمە خەۋەرلىشىش شىركىتى ئۆتكۈر ئەركىن ئىشلىگەن مەھسۇلاتنى قانۇنلۇق ھالدا نەشر قىلىپ ھەقلىق يۇسۇندا تارقىتىدىكەن.
