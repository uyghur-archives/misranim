بۇگۇن ئىشلىرىمنى بالدۇر تۇگۇتۇپ، دۇستۇمنىڭ سىڭلىسىنىڭ تۇيىغا ماڭدىم. ساھىپخان بۇلغانلىقىم ئۇچۇن ئۇلار ئۆزەم ئۇيدە  بىر قۇللۇق كۇڭۇل قۇيۇپ قىلغان گىرىملىرىمنى ياراتماي، مىنى پەردازخانىغا مەجبۇرى ئاپىرىپ، چاچلىرىمنى ئاجايىپ پاسۇندا ياساپ ،يۇزلىرىمنى چاڭچىلىكەش، مايمۇن سۇنھۇكۇڭدەك قىلىۋەتتى. ئۇنى ئازدەپ ھەرخىل پاقىراق چاچ قاتۇرغۇچلاربىلەن، ئەزەلدىنلا جىق، ئۇزۇن  چىچىمنى تاياقتەك قاتۇرۇپ، تىخمۇ جىقايتىپ، كاللامنى سىۋەت شەكلىگە كىرگۇزۇپ قۇيدى. مەن ئادەتتە بۇ چاچلارنى يىغۇشتۇرالماي، يا بۇغىۋالاتتىم ،ياكى ئاددىلا تۇگىۋالاتتىم .ھەي...ھەشەمەتچىلىك!
تۇيدىكى ۋاراڭ-چۇرۇڭ ،قالايمىقانچىلىق جىدەلنى بىر دىمەيلا قۇيۇڭ!
كەيپىياتىم تۇلىمۇ ناچار، تازلىق ئۇيىگە قۇلۇمنى يۇغىلى كىرىپ ئەينەككە قارساملا جىلە بۇلاتتىم، ھىلىمۇ ياخشى قىز تۇي بۇلغاچ بۇرۇنلا بىكار بۇلدۇق، يىگىت تەرەپمۇ قىزنى ۋاختىدا يۇتكەپ كەتتى. ھەرنىمە بۇلسا بالدۇراق دىممنى ئالىدىغان بۇلدۇم شۇكرى!
مەن شۇ خىياللار بىلەن ۋاراڭ-چۇرۇڭغا تۇلغان رىستۇران،  ئادەملەر تۇپىدىن ئايرىلىپ، سىرىتتىكى مۇزدەك ھاۋادىن قىنىپ نەپەس ئىلىپ، يۇل بۇيىدا ماشىنا تۇسقاچ ئاستا كىتىۋىتىپ ، ھەممە ئادەمنىڭ ماڭا مەسخىرىلىك قاراۋاتقىنىنى سىزىپ، نۇمۇس كۇچۇمدە ئالدىمغا كەلگەن بىر ‹قارا تاكسى›غا ئاران ئۇلگۇردۇم،
شۇپۇر ياشلا بالا بۇلۇپ، مىنىڭ بىشىمدىن ئايىقىمغا مەسخىرىلىك قاراپ ،
-ھەدە تۇيغا بارامسىز؟ يا  تۇيدىن ياندىڭىزمۇ؟ نەگە ئاپىرىپ قۇيمەن؟
-ھەئە ، تۇيدىن ياندىم ،ئىغىر كۇرمەي يىقىن ئەتراپتا بىرە ساتىراچخانا بۇلسا ئاپىرىپ قۇيسىڭىز!
شۇپۇر ماڭا غەلىتە قاراپ،
-گىرىمىلىرىڭىز كۇڭۇلدىكىدەك بۇلماپتۇدە ؟باشقىدىن گىرىم قىلدۇرۇپ يىڭى سورۇندا يىگىتلەرنىڭ كۇزىدىن ئوت چىقىرۋىتەي دىگەن ئۇخشىمامسىز؟
ئۇنىڭ ئىككى بىسلىق سۇزلىرى جىنىمغا تەگدى.تۇرۇپلا شۇپۇرغا ‹غۇژژ›دە ئاچچىغىم كەلگەن بۇلسىمۇ، مەنزىلىمگە سالامەت يىتىۋىلىش ئۇچۇن يالغان كۇلۇپ،
-ياقەي، چىرايىمدىكى بۇ بۇياقلارنى. ما سىۋەتتەك كاللامنى يۇدۇرۇپ قايتمىسام ،ئاپام مىنى بۇسۇغىدىنمۇ ئاتلاتمايدۇ،
شۇندىلا شۇپۇر سەل ئۇيلىنىۋىلىپ.
-كۇڭلىڭىزگە كەلمىسۇن، تۇي گىرىمى بۇلسىمۇ بۇنچىۋالا قىنىق بۇلمايدۇ، چىرايىڭىز ،چىچىڭىز، كىيىمىڭىز بىلەن پەقەت ماس كەلمەپتۇ قەتئى ئىچىلماپسىز،
-مەنمۇ شۇنداق ئۇيلىغان، شۇڭا كۇچىدا مىڭىشتىن نۇمۇس قىلىپ ماشىنىڭىزغا ئاران ئۇلگۇردۇم،
شۇپۇرئەمدى ماڭا ئەدەپ بىلەن مۇئامىلە قىلىشقا باشلىدى.
-قارسام كىيىنىشىڭىز شۇنچە سىپتا. ئىلمى. يارىشىملىق. ئەمما.....
‹قارا بۇسۇلتەكنىڭ كىچىككىنە تۇرۇپ ئۇستى -بىشىمغا سەپ- سىلىپ  ماڭا باھا بىرىپ بۇلغىنىنى،›
مەن ئىچىمدە شۇلارنى ئۇيلاپ.
-بىر تۇيغا ساھىپخان ئىدىم ،ئۇلار مەجبۇرى مۇشۇنداق قىلىپ قۇيدى .بۇلمىسىغۇ ئەتتىگەن  ئۇزەم ئۇيۇمدىن كىيىمگە ماسلاشتۇرۇپ جابدۇنۇپ چىققان ،‹مۇراسىم كىيىمى›نىمۇ مەجبۇرلىغان ،كەيگىم كەلمىدى.
-ئۇنداق ھەشەمەتچىلىكنىڭ نىمە ھاجىتى، تۇيى بۇلغانلار بەخىتلىك بۇلسا بۇلمىدىمۇ؟
-شۇنى دىمەمسىز ؟ تۇينى مەن قىلمىسام يا،
-ھازىر قەرىز ئىلىپ تۇي قىلدىغانلارمۇ بەك كۇپىيىپ كەتتى. تۇي سىلىقىنىڭ ئىغىرلىغىدىن قىز-ئۇغۇللار بەختىنى تاپالماي ،تەمتىرەپ يۇرۇپ كۇپۇنچىسى كۇچىغا چىقىپ بۇزۇلۇپ كىتىۋاتىدۇ، ئاتا-ئانىلار نىمىنى ئۇيلايدىغاندۇ زادى؟
-بەلكىم بۇ ئۇلار ئۇچۇن زۇر شان-شەرەپ ،يۇز-ئابرۇي بۇلسا كىرەك،
-ھەي... شۇ قەرىزئىلىپ تۇي قىلغانلار،تۇيدىن كىيىن بۇسۇغىسىدىن كەتمەي قەرىز سۇيلەپ كىلدىغان قەرىز ئىگىلىرىنىڭ  كۇپلىگىدىن بار بىساتىنى سىتىپ، جىدەللىشىپ، ئۇزۇن ئۇي تۇتالماي ئاجرىشىپ، شۇنچە كۇپ رىيازەت چىكىپ ،مىڭ بىرجاپادا قۇرىغان بەخىتلىك ئائىلە تۇۋرىكىنىڭ ھۇلىنى كۇلاپ ۋەيران قىلىپ، بەخىتسىزلەرگە  ئايلىنىپ، بىر ئۇمۇر دىلى ئازاريەپ ، ھىجران ئازابى تارتىپ ، بۇنىڭغا پۇشايمان دۇرىسى ئىزلەپ يۇردىكەن،
-راس دەيسىز، مىنىڭمۇ ئىدىيەمدىن ئۇتمەيدۇ،
-سىز تۇي قىلغانمۇ؟
- نىمە ؟كۇزىڭىزگە قىزكۇرۇنۇپ كەتمىگەندىمەن ئۇكام؟ بۇركۇتتەك بىر ئۇغلۇم بار،
-  ھە ،مۇنداق دەڭ ،كىچىكلا تۇي قىپتىكەنسىزدە؟
-ھەئە،
-بۇلدى كەلدۇق، بۇ دۇكان يامان ئەمەس ،ئەرزان ھەم خىلى پاكىز ،قايىتسىڭىز تىلىفۇن قىلىڭ، سىز بىكار بۇلۇپ بۇلغىچە قاراڭغۇ چۇشۇپ كىتىدۇ، ماشىنىمۇ ئىتتىك كەلمەيدۇ ،سۇغۇقتا ئاۋارە بۇلسىز،
-بۇلىدۇ، رەھمەت،
مەن ماشىنىنىڭ پۇلىنى بىرىپ، شۇپۇر كۇرسەتكەن دۇكانغا كىردىم ،دۇكاندا مەندىن باشقا 3 ئادەم بۇلۇپ، ئۇتتۇرياشلىق بىر ئەر چېچىنى ياسىتىۋىتىپتۇ ، پەرىزىمچە 15 ياشلار ئەتراپىدا بۇلسا كىرەك، بىر قىز دىرىزىگە يۇلۇنۇپ ئۇرە تۇرۇپتۇ، يەنە بىرى ساتىراچ قىز ئىكەن.
بىردەمدىن كىيىن ساتىراچ قىزنىڭ قۇلى بىكار بۇلدى. مەن ھىلىقى كىچىك قىزغا قاردىم ،ئۇ كىملەرنىدۇ ساقلاۋاتقاندەك غىمىدا يۇق تۇراتتى.
ساتىراچ مىنىڭدىن سۇرىدى.
-سىزگە قانداق مۇلازىمەت كىرەك؟
-چىچىمنى يۇيۇپ ئەسلى ھالىتىگە ئەكەپ قۇيسىڭىز،
-قىنى بۇياققا كىلىڭ، ئەسلدە ئۇيدە ئۇزىڭىز يۇسىڭىزمۇ بۇلىدۇ!
- چىچىمدا چاچ قىسقۇچتىن بىرە يۇزى بارمىكىن، ئىچىم سىقىلىپ كىتىدىكەن، چىچىم بەك جىق ، ھەم ئۇزۇن  باشقۇرۇپ بۇلالمايمەن ،كىسىۋىتەي دىسەم كۇڭلۇم ئۇنمايدۇ ،ئۇزۇن چاچقا  ئۇزەممۇ ئامراق كۇنۇپ قاپتىمەن ،
ساتىراچ خىلى ئۇزۇن ھەپىلىشىپ چاچ قىسقۇچلارنى ئىلىپ بۇلۇپ، چىچىمنى پاكىز يۇيۇپ چاچ قۇرۇتقۇچ بىلەن قۇرۇتىۋاتاتتى.
كۇزۇم ئالدىمدىكى ئۇزەمنىڭ ئەكسى چۇشۇپ تۇرغان كۇرنۇش ئەينەككە چۇشۇش بىلەن چۇچۇپ كەتتىم،
چۇنكى ئاغزىدىن تىخى ئانا سۇتىنىڭ ھىدى كەتمىگەن ھىلىقى كىچىك قىز، بىرتال تاماكىنى تۇتاشتۇرۇپ خىيالىدا يۇق خۇددى كۇنا بەڭگىلەردەك ئىچىگە كۇچەپ تارتىپ، ئاغزىدىن يۇمۇلاق ھالقا سىمان، كۇكۇش تۇتۇندىن چەمبىرەك ھاسىل قىلىپ چىقىرىۋاتاتتى.
مەن ھەيرا بۇلۇپ ئۇنىڭدىن كۇزۇمنى ئۇزمەي چىكەتكىدەك قاراپ ئۇلتۇراتتىم،
قىز تاماكىسىنى يىرىملاشتۇرمايلا 60 ياشلاردىن ھالقىغان بىر قىرى ئەركىشى. ھاسىرغىنىچە ئاران مىڭىپ ئىشىكتىن كىردى.
-ساتىراچ ئۇستام ،ئەتە نەۋرەمنىڭ تۇيى ئىدى. چىچىمنى ئۇزەم ئۇيدە بۇيىسام ،پىشانە، يۇز، قۇلاقلىرىمغا بۇياق يۇقۇپ قالدى .ھىچقانداق قىلىپ چىقىرالمىدىم، بۇنىڭ بىرە چارىسى بارمۇ؟
بۇۋاي بىشىدىكى نەچچە تال شالاڭ تۇكنى قاپ-قارا رەڭدە بۇيىغان بۇلۇپ، دىمىسىمۇ بۇياق ھەممە يىرىگە سۇگۇشۇپ بەئەينى ھايۋاناتلار باغچىسىدىكى قىرى مۇشۇك ئىيىققىلا ئۇخشاپ، شۇنداق كۇلكىلىك بۇلۇپ قالغان،
-بۇنىڭ باشقا ئامالى يۇق، تاماكىنىڭ كۇلىنى سۇرۇپ ئىرتىۋەتسىڭىز، شۇھامانلا  چىقىپ كىتىدۇ ،بۇ شالاڭ، ئازغىنە چاچلىرىڭىزنى جاپا تارتىپ ئۇيدە بۇيغىچە، يىنىمغا كەلگەن بۇلسىڭىز ھەقسىز بۇياپ قۇياتتىم ئەمەسمۇ؟
-سىزنى ئاۋارە قىلماي دىدىم، مەسلەن باشقا يەردە بۇياتسام نەچچە پۇلغا بۇيايدۇ؟
-جىق ئالسا 10 كۇي ئالارمىكىن!
- 10 كۇي ؟يەنىلا ئۇيدە بۇيىغان ياخشى. 3كۇيغىلا بۇيىۋالدىم مانا، ھا...ھا...ھا...
بۇۋاي ئۇزىدىن مەننۇن، رازى بۇلغاندەك ھۇزۇرلۇنۇپ كۇلدى.
چىرايلىرىنى قۇرۇقلار قاپلىغان، يۇزلىرىنى قىرىلىق دىغى باسقان يۇرىگى كىتىپ ئۇپكىسى قالغان بۇ ئادەمگە چاچ بۇياشنى نىمە قۇيغاندۇ ئەمدى؟  نەۋرىسىنىڭ تۇيى بۇلسا ئاق چاچ بىلەن سالاپەتلىك كىرسە ئۇنى بىرەسى  مەرىكە زالىنىڭ ئىشىگىدىن قۇغلىۋەتمەستى! چۇڭ سۇپەت ھۇرمىتى بىلەن ئۇلتۇراتتى. مانا ئەمدى بىشىغا قارسا قازاننىڭ كۇيىسىدەك، چىرايىغا قارسا دۇنيا خەرىتىسىدەك ،كىشلەرنىڭ راسا كۇلكىسىگە قالمىسىلا بۇلاتتىغۇ؟
مەن شۇلارنى خىيال قىلۋاتقاندا ،بۇۋاينىڭ كۇزى ھىلىقى كىچىك بەڭگى قىزغا چۇشتى.
مەن ئىچىمدە ،‹بۇۋاي بۇقىزغا نەسىھەت قىلىپ ،بىر بۇلسا راسا خىجىل قىلىدۇ ،يەنە بىرى بۇلسا يىغلىتىۋىتىدۇ ،تۇيە كىچىك تۇرۇپ كەيىپكە ئۇگەنگەنگە، خۇپ بۇلدى تازا. › دەپ ئۇيلاپ تۇرشۇمغا بۇۋاي، مىنىڭ خىيالىمنىڭ دەل ئەكسىچە گەپ قىلدى .ئۇنىڭ سۇزىنى ئاڭلاپ قۇلاقلىرىمغا ئىشەنمىگەندەك بەكلا ھەيران قالدىم،
-قىزىم، چىكىۋاتقان تاماكىڭىزنىڭ كۇلىنى قۇلۇمغا قىقىپ بىرەمسىز؟
بۇۋاي قىزنىڭ ماقۇللىغىنى ئالمايلا ،ئىككى قۇلىنى خۇددى كۇچىدىكى تىلەمچىلەردەك جۇپلەپ، ئالقىنىنى ئىچىپ قىزنىڭ ئالدىغا بىرىپ، قۇلىنى سۇزۇپ تەڭلەپ تۇردى.
قىز باشتا سەل ھۇدۇقتى. كىيىن مىيىقىدا كۇلۇپ قۇيۇپ ،چىكىۋاتقان تاماكىسىنىڭ كۇلىنى بۇۋاينىڭ قۇلىغا قاقتى.
مەن ئەينەكتىن بۇلارنى كۇرۇپ ئۇلتۇردۇم، تاماكا چىكىلىپ بۇلدى.
بۇۋاي قۇلىدىكى كۇلنى خۇددى ئەتىۋارلىق نەرسىنى ئاسرىغاندەك ئاۋايلاپقىنا، يەنە بىر قۇلىنىڭ ئالقىنىغا تۇكۇپ، قىز يۇلۇنۇپ تۇرغان دەرىزە تەكچىسىگە ،ئاغزىنى ئىچىپ قۇيۇپ قۇيغان تاماكىدىن بىرتال سۇغۇرۇپ قىزغا تەڭلىدى.
قىز بۇۋايغا بىر قاراپ قۇيۇپ ئۇنىمۇ تۇتاشتۇرۇپ چەكتى.
شۇنداق قىلىپ قىز كەينى-كەينىدىن3 تال تاماكىنى چىكىپ تۇگەتتى.
بۇۋاي 4-تاماكىنى سۇنغاندا قىز،
-بۇلدى. گىلىم ئىچىشىپ كەتتى. مەيدەم قاتتىق بۇلىۋالدى. ئەمدى چەكمەيمەن،
-قىزىم بۇ كۇل يەتمەيدۇ،  يەنە بىرتال چىكىڭ،
-چەككىڭىز بۇلسا ئۇزىڭىز چىكىڭ ،ئەنە تاماكا دىرىزىدە تۇرۇپتۇ،
-مەن تاماكا چەكمەيمەن، مىنىڭ ‹چىگۇيەن› كىسىلىم بار،
-چەكمەيمەن دىدىمغۇ، نىمە زۇرلاۋىرىسىز؟ ھىلمۇ 3 تال چىكىپ بۇلدۇم، نىمانداق قىلىدۇ،
قىز بۇۋايغا قاراپ ئالايدى.
-بۇياقنى جىق يىرىمگە سۇركىۋاپتىمەن، ئەتە تۇيدا سەت تۇردۇ،
- نەۋرىڭىزنىڭ تۇيى بۇلسا مىنىڭ نىمە چاتىتىغىم، قىزىقكىنا،
-قىزىم، مەن ئۇچۇن يەنە بىرتال تاماكا چىكىڭ...
بۇۋاينىڭ گىپىڭ ئاخىرى تۇگىمەيلا، چىچىمنى تاراۋاتقان ساتىراچنىڭ قۇلىنى ئىتتىرۋىتىپ، ئاچچىغىم بىلەن ئۇرنۇمدىن دەس تۇردۇم،
-بۇۋا ،چوپ-چوڭ ئادەم،بىر ئەركەك تۇرۇپ،قىرىغاندا مۇشۇ يىشىڭىزدا، كىچىككىنە بىر قىزنى تاماكا چىكىشتىن تۇسماقتا يۇق، ئۇزىڭىزنىڭ مەنپەئەتى ئۇچۇن ئۇنىڭغا تاماكا سۇنۇپ، ئۇنى يەنە چىكىشكە زۇرلاۋاتامسىز؟ ناۋادا ئۇسىزنىڭ قىزىڭىز، ياكى نەۋرىڭىز بۇلسىمۇ شۇنداق قىلامتىڭىز؟
-مىنىڭ قىزلىرىم-نەۋرىلىرىم بۇنداق سەت ئىشنى. ناچار قىلىقنى قىلمايدۇ،
-ئەمسە سىزنىڭ ھازىر قىلىۋاتقىنىڭىز چىرايلىق ئىش ئىكەندە ؟بۇمۇ سىزنىڭ نەۋرىڭىزدەك قىزئىكەنغۇ؟
بۇۋاي گۇيا مىنى يەۋىتىدىغاندەك ماڭا سەتلا قاراپ  بىرنى ئالىيىپ قۇيۇپ، ئالقىنىدىكى تاماكىنىڭ كۇلىنى ئاۋايلاپ كۇتۇرۇپ، ساتىراچخانىدىن ئىنجىقلىغىنىچە ئاستا قەدەملەر بىلەن چىقىپ كەتتى.
شۇ ئارلىقتا بالام تىلىفۇن قىلىپ قالدى. مەن تىلفۇننى سۇزلۇشۇپ بۇلغىچە ھىلىقى كىچىك بەڭگى قىز مەخسىدىمنى بىلىپ قالدىمۇ تىزلا غايىپ بۇلدى. بۇلمىسا ئۇنىڭمۇ ئەدىۋىنى ئۇخشۇتۇپ بىربەرمەكچى ئىدىم،
ساتىراچ بىزنىڭ تىلىمىزنى بىلمىسىمۇ نىمە دىيىشكىنىمىزنى ئاز-تۇلا چۇشەندى بۇلغاي ،بىشىنى چايقاپ ئاچچىق كۇلۇپ قۇيدى.بۇ سىرلىق كۇلكىنىڭ ئىچىگە نۇرغۇن سۇئاللارنىڭ يۇشۇرۇنغانلىقى ھەممىمىزگە ئايان! بۇلسا كىرەك.
گۇزەللىككە ئىنتىلىش بۇ بىر ياخشى ئىش. ئەمما ھەر كىشى ئۇزىنىڭ يىشىغا ماس ھالدا، گۇزەللىك قارشىنى تۇرغۇزۇش ، يىڭىلىقنى قۇبۇل قىلىش  ئىڭىنى يىتىلدۇرۇش كىرەك ،قارغۇلارچە قۇغلاشقان گۇزەللىك قارشى ،مۇدا ئىقىمى. ئۇچقاندەك قۇبۇل قىلنىۋاتقان غەرىپ مەدەنىيىتى. سىزنى، بىزنى، كىشلەرنىڭ كۇلكە ئۇبىكتىغا ئايلاندۇرۇپ قۇيشىدا گەپ يۇق .
مەن چىچىمنى ئاندا-مۇندا بۇغۇپلا، ساتىراچنىڭ ھەققىنى بىرىۋىتىپ سىرىتقا چىقتىم،
يۇلدا كىتىۋاتىمەن، سىرىتتىكى سۇغۇق ماڭا قىلچە تەسىر قىلمىغاندەك، ئاچچىق خىياللار ئىلكىدە يۇرۇگۇمنى مۇجۇپ كىلىۋاتقان، ئۇزەم يۇلۇققان، كۇرگەن ،ئاڭلىغانلىرىمغا جاۋاپ ئىزلەپ كىتىۋاتىمەن،
بۇنىڭ جاۋابى زادى نەدە؟
قۇلاق تۇۋۇمدە يەنىلا شۇ بۇۋاينىڭ بۇغۇق خىقىراق ئاۋازدا سۇزلىگەن سۇزى. ئەكىس- سادا قايتۇرۇپ تەكرار ئاڭلانماقتا،
-قىزىم، مەن ئۇچۇن يەنە بىرتال تاماكا چىكىڭ!..............
تۇگىدى.

ئى قىز بەكمۇ ئېشىپ كىتىپتۇ ،ئۇيغۇر قىزى بولغانكىن ئادەم قىلدىغان قىلىقنى قىلسا بولمامدۇ ،ھايۋاندىنمۇ ئۆتە نەرسە ئىكەن ئۇ ....
ئى بوۋايچۇ تېخى ئۆزىنىڭ نەۋرىسىدەك قىزغا تەربىيە قىلىشتىن تۈگۈل ئۇنى چىكىشكە زورلىغىنىچۇ ،ئازىر ئادەملەردە  نىمانداق ۋىجدان دىگەن نەرسە قالمىغاندۇ ؟؟؟؟

ئۆيكەن يىلى نەۋرە باللار يىغلىش قىلغان ئىدۇق ،تاماقلىرىمىزنى يىيىشىپ بولۇپ چاقچاق قىلىشپ يولدا كىتىپ باراتتۇق ،ئالدىمىزدىكى مەيغانىنىڭ ئالدىدا ياغلىق چىگىپ ،ئۇزۇن يوپكىلارنى كىيىپ تۇرغان بىر قانچە قىزلارنى كۆرۈپ قالدۇق ،ھەمدە بىر بىرىمىزگە قارىشىپ تۇرۇپلا قالدۇق ،ئۇلار ھازىريو يوللاردا ئۇچۇرتۇپ تۇردىغان مودا روپاشلار  ئىدى ،بىزنى ھەممىدىن ھەيران قالدۇرغىنى ھەممىسىنىڭ قولىدا بىردىن تاماكا تۇراتتى ،بۇنى كۆرگەن نەۋرە ئاكام (ئۇ بىزنى ئادەتتە ئاتا _ ئانىمىزدىنمۇ بەكرەك باشقۇراتتى ،ئۇنىڭ ھەر بىر قېتىم بىزگە بەرگەن تەربىيەلىرىنى ئاڭلىغىنىمىزدا يۈرەكلىرىمىز ئېرىپلا كىتەتتى )بىر قىزنىڭ قولىدىن تارتىپ تاماكىسىنى دەششەت بىلەن تاشلىۋەتتە ،ئۇ قىزلارمۇ نېمىش بولغانلىقىنى بىلەلمەي قېلىشتى بولغاي ھاڭۋېقىپلا تۇرۇپ قالدى ،بۇ ئەسنادا نەۋرە ئاكام ئۇلارغا تەربىيە قىلىشقا باشلىدى ،ئۇ قىزلارنىڭ قولىقىغا تەربىيە كىردىغاندەك ئەمەس ئىدى ،ھەمدە بىر قىز تىلغۇننى ئېلىپ بىللە ئېلىپ كىرگەن يىەىتىگە تىلغۇن قىلدى ،يىگىتتى ئۇ مەيغانىنىڭ ئىچىدى ،مەس ھالاتتە كۆزىنى چەكچەتكىنىچە چىقىپ كەلدى ،بۇنى كۆرۈپ تېخىمۇ خەزىپى تاشقان نەۋرە ئاكام ئۆزىنى تۇتۋالدىغاندەك ئەمەس ئىدى ،ئۇنىڭ مۇشتلىېىنى تۈگىشنىنى كۆرۈپ سەل ئەنسىرەپ قالدۇق ،ئۇرۇش چىقىشنى خالىمايمىزدە ئەلۋەتتە ،شۇڭا بىز بىرلىشىپ نەۋرە ئاكامنى كىتىشكە دەۋەت قىلىپ سورەپ ئېلىپ ماڭدۇق ،ئۇ ئاچچىقتىن ‹‹ھەي ! ئاشۇ ياغلىىقا چۇشلۇق ئىش قىلساڭلار بولمامدۇ ››دېدى ،ئۇنى ئىىتتىرىپ ئاران دىگەندە بۇ يەردىن ئايرىلدۇق ،لېكىن ھەممىمىزنىڭ كۆڭلىدە ئاشۇ قىزلارغا لەنەت ئوقۇدۇق ،ياغلىققا چۇشلۇق ئىش قىلسا بولمامدىغاندۇ ،ئۇيغۇرلارنىڭ يۈزىنى مۇشۇنداقلار تۈكىدىغۇ زادى دىگەنلەرنى ئويلۇدۇق .
