                                                        سورىساڭ

                                                      ئاينى ئاسماندىن سورا،
                                                      لاينى ساماندىن سورا.
                                                      تۇتقان يوللىرىڭ قانداق،
                                                      غورۇر-ۋىجداندىن سورا.

                                                      ھاياتلىق يولى نېمە؟
                                                      كىشىلىك ئۇلى نېمە؟
                                                      ئۆمۈرنىڭ گۈلى نېمە؟
                                                      بېغى غازاندىن سورا.

                                                      جاپاكەش مېھنىتىنى،
                                                      بۇردا نان قىممىتىنى.
                                                      مىسكىنلىك قىسمىتىنى،
                                                      جۇت ھەم بوراندىن سورا.

                                                     قۇشلارنىڭ ئۆيى نەدۇر؟
                                                     كۆللەرنىڭ سۈيى نەدۇر؟
                                                     غېرىپلىق كۈيى نەدۇر؟
                                                     ئەھلى ھىجراندىن سورا.
     
                                                     مۇشەققەتنى، جەبىرنى،
                                                     شۈكۈركۈكنى،سەبىرنى.
                                                     داۋراڭسىز مىڭ ئەجىرنى،
                                                     قىردا دېھقاندىن سورا.

                                                     قەلىبتىكى ئاقلىقنى،
                                                     خاتىرجەملىك،ساقلىقنى.
                                                     ھالاللىقنى پاكلىقنى،
                                                     ئەھلى ئىماندىن سورا.

                                                     بالا كىمدۇر،چالا كىم؟
                                                     ئاتا كىمدۇر ئانا كىم؟
                                                     بۇ دۇنيادا ۋاپا كىم،
                                                     يېتىم بولغاندىن سورا.

                                                     ئاشىقلىقنىڭ دەردىنى،
                                                     يۈرەكتىكى ئەرزىنى.
                                                     تەنھالىقنىڭ ۋەزنىنى،
                                                     بەختى پايخاندىن سورا.

                                                     راھەتنى-پاراغەتنى،
                                                     بەخـىت ۋە سائادەتنى.
                                                     ئاداققى ئادالەتنى،
                                                     زاتى رەھماندىن سورا.

                                                     داننى خاماندىن سورا،
                                                     جاننى جاناندىن سورا.
                                                     يېتەي دېسەڭ ۋىسالغا،
                                                     پەقەت سۇبھاندىن سورا.




                                                        يادىمغا يەتسىڭىز

      
                                             ۋۇجۇدۇم مىھىردىن ئىللىپ كېتىدۇ،
                                             ئەگەر سىز يادىمغا كەلسىڭىز جېنىم.
                                             تۇيغۇمدا قار- مۇزلار ئېرىپ كېتىدۇ،
                                             ئەگەر سىز يادىمغا يەتسىڭىز جېنىم.

                                             ئالدىمغا يېيىلار ئىللىق بىر باھار،
                                             قۇياش ھەم سېخى بىر دوستقا ئايلىنار.
                                             ئىچ-باغرىم ئەڭ گۈزەل كۈيلەرگە تولار،
                                             ئەگەر سىز يادىمغا يەتسىڭىز جېنىم.

                                             سىز ماڭا ئارمانسىز،ماڭا تېپىشماق،
                                             گاھ شېرىن تۇيغۇمسىز، گاھ شېرىن قىيناق.
                                             تىترىمەي بۇ يۈرەك تۇرىدۇ قانداق،
                                             ئەگەر سىز يادىمغا يەتسىڭىز جېنىم.

                                             دەيدىكەن ھاياتلىق يوللىرى داۋان،
                                             بۇندا كۆپ ئاشىقنىڭ رەڭلىرى سامان،
                                             بىراق مەن ھەممىنى ئۇنتۇيمەن ھامان،
                                             ئەگەر سىز يادىمغا يەتسىڭىز جانان.



                                                        بۇ كېچە


     
                                             كۆزلىرىم يۇلتۇزدەك يېنىپ تۇرىدۇ،
                                             بۇ كېچە مەن يەنە ئۇخلىيالمايمەن.
                                             يۈرەككە بىر ئەلەم  پېتىپ تۇرىدۇ،
                                             بۇ كېچە مەن يەنە ئۇخلىيالمايمەن.

                                             قوشنىلار چىرىغى ئۆچىدۇ بىر-بىر،
                                             كۈيلەرنى تىمتاسلىق يۇتىدۇ ئاخىر.
                                             ئازابقا ئايلىنىپ شۇ قەدەر ئېغىر،
                                             بۇ كېچە مەن يەنە ئۇخلىيالمايمەن.

                                             ئاخىرقى بىر تال شام يىغلار ئېسەدەپ،
                                             تورۇسلارباغرىمدەك تۇرىدۇ تىترەپ.
                                             ھەسرەتلەر قوينىدا چۆكۈپ گاھ لەيلەپ،
                                             بۇ كېچە مەن يەنە ئۇخلىيالمايمەن.

                                             ھېچكىمنىڭ يادىغا كەلمەيمەن شۇ تاپ،
                                             غېرىبلىق ئاقماقتا مەڭزىمنى بويلاپ.
                                             سۈبىھنى كۈتكەنچە سېنىلا ئويلاپ،
                                             بۇ كېچە مەن يەنە ئۇخلىيالمايمەن.



                                                           قۇشقاچ


                                                 ئالدىمدىكى كىچىك قۇشقاچ،
                                                 سەن نېمانچە ماڭا ئوخشاش.
                                                 مەن نېمانچە ساڭا ئوخشاش،
                                                 ئالدىمدىكى كىچىك قۇشقاچ.

                                                 قىيالاردىن يېقىلدىڭمۇ؟
                                                 يالغۇز قويۇپ كېتىلدىڭمۇ؟
                                                 ھىجران بولۇپ ئېچىلدىڭمۇ؟
                                                 ئالدىمدىكى كىچىك قۇشقاچ.

                                                 بىناۋاسەن،سايرىمايسەن،
                                                 بىچارەمەن،يايرىمايمەن.
                                                 يۈرەك يىغلار پىغان بىلەن،
                                                 ئالدىمدىكى كىچىك قۇشقاچ.

                                                 سەن بىر قۇشسەن،مەن بىر ئىنسان،
                                                 سەن بىر مېھمان مەن بىر مېھمان.
                                                 بىز دۇنياغا ئامانەت جان،
                                                 ئالدىمدىكى كىچىك قۇشقاچ.

                                                 ئالقىنىمدا تۇرغىنى چوغ،
                                                 پەرۋازىڭغا تەككىنى ئوق.
                                                 ئىككىمىزگە ۋاپادار يوق،
                                                 ئالدىمدىكى كىچىك قۇشقاچ.

                                                سەن مۇساپىر،مەن بىر غېرىپ،
                                                ھەسرىتىمىز تۇرار يېنىپ.
                                                ھېچكىم بىزنى سۆيمەس كېلىپ،
                                                ئالدىمدىكى كىچىك قۇشقاچ.

                                                يات كۆرمىگىن كەل قېشىمغا،
                                                قونغىن يالغۇز بۇ بېشىمغا.
                                                سەنمۇ تەنھا، مەنمۇ تەنھا،
                                                ئالدىمدىكى كىچىك قۇشقاچ.

                                                پەلەكلەرگە قىلاي نالە،
                                                سەن دەردىڭنى بايان ئەيلە.
                                                خۇدا يۆلەك بىچارىگە،
                                                ئالدىمدىكى كىچىك قۇشقاچ.



                                                       ئاتا


                                                        
                                          كۆزلىرىمدىن مارجان تۆكۈلەر،
                                          يۈزلىرىمدىن ھىجران تۆكۈلەر.
                                          كەتكىنىڭنى ئويلىسام ئاتا،
                                          يۈرۈگۈمدىن ۋولقان تۆكۈلەر.

                                          خانىداندىن سېنى تاپالماي،
                                          يۈزلىرىڭگە يۈزۈم ياقالماي.
                                          جېنىم ئاتا سەنسىز ياتالماي،
                                          كۈيلىرىمدىن پىغان تۆكۈلەر.

                                          بالىلىرىم كىمگە ئەركىلەر،
                                          كىم ئۇلارغا ئات بولۇپ بېرەر.
                                          ئاتا بىزگە كىم سەندەك كۆيەر،
                                          ئويلىرىمدىن غازان تۆكۈلەر.

                                         كۆزلىرىمدىن ئارمان تۆكۈلەر،
                                         كۈيلىرىمدىن پىغان تۆكۈلەر.
                                         دىلغا سىغماس ھەسرىتىم ئاتا،
                                         يۈرۈگۈمدىن ۋولقان تۆكۈلەر.....
