                                 ھىجىران پەسلى   
                                       
                                        ئالىمجان تۇرغۇن (سەرسانى )


                                  كىرىسەن چۈشۈمگە ھەر كۈنى شۇنداق ،
                                  چىھرىڭدىن كۈلكىلەر ياغدۇرۇپ جېنىم  .
                                  خۇشاللىق قەلبىمدە ياسايدۇ دولقۇن  ،
                                  سۆيۈپ مەن لىۋىڭگە تۆمەنمىڭ قېتىم  .

                                  قوغلايسەن قەلبىمدىن ھىجىران چۆلىنى ،
                                  ئايىغىڭ يەتكەن يەر بۇلۇپ گۈلۈستان  .
                                  باھارنىڭ قەدىمى كېلەر سەن بىلەن  ،
                                  شۇندىلا ئۆچەككە كىرىپ زىمىستان  .

                                  ھىجىراننىڭ جىنلىرى قېچىشار ھەر يان  ،
                                  سەن كېلىپ خانەمگە ياققاندا گۈلخان  .
                                   كىشەندىن قۇتۇلۇپ سەكرەيدۇ كۈلكە  ،
                                  نۇرلىنارب چىرايىم  بولغان  بزەپىران   .

                                  بەخىتتە  يېپىلار   تەقدىر   سوتىمۇ ،
                                  سەن گۇۋاھ بولغاندا سۆيگۈمگە كېلىپ .
                                  سېغىنىش ،  ئازابلار چۈشەر  سولاققا ،
                                  سەن كېلىپ ھىجىرانغا يازغاندا رىتسىپ .

                                  يىگانە  قالدۇرۇپ   كەتمىگىن  جېنىم  ،
                                  سېغىنىش  سولاقتىن كەتمىسۇن  قېچىپ .
                                  بەخىتنىڭ  كارۋىنى ماڭسۇن  مەڭگۈلۈك ،
                                  ئويغۇنۇپ كەتمەي مەن چۈشۈمدىن كېچىپ ...


                                 تۇيۇقسىز  چىقىپلا  قالدىڭ  ئالدىمغا ،
                                 كۆزۈڭگە  تىكىلىپ  قارىدىم   جېنىم  .
                                 قارچۇقۇم  ئىچىدە  قېتىپ  قالسىكەن ،
                                 ئاھ جېنىم سىنىڭكى گۈزەل قامىتىڭ .

                                 خۇشپۇراق  تارىلىپ  تۇرىدۇ  سەندىن ،  
                                 ئالدىڭدىن قانچە رەت ئۆتسەم ھەر كۈنى .
                                 سەندىمۇ   كىيىكتەك  ئىپار  بارمىدۇ ،   
                                 ياكى  گۈل  سۈيىدە  باققانمۇ  سىنى  . ،

                                     ************************

                                 ئۇچۇپ بېرىپ كۆرەي دىسەم يار سىنى ،
                                 ئارلىق  يىراق ،  مىنىڭ  نىمە  ئامالىم .
                                 ماڭا تونۇش ئاشۇ خۇشبۇي ھېدىڭنى ،
                                 پۇراي  دىسەم   يىراقتىسەن  جانانىم  .

                                 رەسىمىڭگە  قاراپ  ئۇزاق  ئولتۇرۇپ ،
                                 خىيالىمدا  مۇڭدىشىمەن  سەن  بىلەن .
                                 شۇم تەقدىردە ئايرىلىپ مەن سىنىڭدىن ،
                                 بىلمەپتىمەن  ، مۇشەقىتى  كۆپ  ئىكەن .

                                             مەنبە : ئۆزۈم
