ئەقىدە قولىغا ئالغان خالتىنى خۇددى بىرىنىڭ يۈرگىنى كۆتۈرگەندەك مەيدىسىگە چىڭ تاڭغىنىچە دىلشاتنىڭ ئۆيىدىن ئايرىلدى ، ئۇ خۇددى ئارقىسىغا قارىسىلا دىلشاتنى كۆردىغاندەك ، ئۇ ئۆزىگە قاراپ كۈلۈمسىرەۋاتقاندەك تۇيغۇدا كەينىگە قايرىلىپ قاراشقا پېتىنالمىدى ، خۇددى بىراۋ ئۇنىڭغا قولىنى پۇلاڭلىتىپ قاراۋاتقاندەك ھېسياتتا ئىدى ، ئۇ ئارقىسىغا قارىدى ، بىراق ھېچنىمىنى كۆرەلمىدى ، ئۇنىڭ كۆز ئالدىنى يامغۇردەك تۆكۈلۈۋاتقان ياشلىرى توسۇپ تۇراتتى . ياشلىرى توسمىغان تەقدىرىدىمۇ ئۆزىنىڭ بۇ شىرىن ئەمما ئازاپلىق خىيالىنىڭ ئەمەليەتكە ئايلانمايدىغىنى ئېنىقلا ئىدى ، بىراق ئەقىدە ئەنە شۇنداق خىياللىرى بىلەن ئۆزىنى بەزلىمەكچىمىدۇ ياكى ئۆزىنى ئۇنىڭ ئۆلۈپ كەتمىگىنىگە ئىشەندۈرمەكچىمىدۇ ؟
ئەقىدە ئۆيىگە كېلىپلا ياتاق ئۆيىگە كىرىپ كەتتى ، ئەقىدەنىڭ ئانىسى خاسيەت ئاچا خۇددى قىزىنىڭ شۇنداق قىلىدىغىنى بىلىدىغاندەكلا ئۆزىنىڭ ئۆيىگە كىرىپ كەتتى . قىزىنىڭ ھازىر تەسەللىيگە مۇھتاج ئەمەس بەلكى رىئالىقنى دادىللىق بىلەن قۇبۇل قىلالايدىغان ئەقلىي كۈچكە ئىگە ئىكەنلىكىنى بۇ ئائىلە ئايالى ئېنىق بىلەتتى . ئاتىسىدىن كىچىكلا يېتىم قالغان بۇ قىزغا ھەم ئاتا ھەم ئانا بولغان مۇشۇ 15 يىلدىن بۇيان بويۇن قىسمىغان ، ياش تۆكمىگەن بۇ قىز بۇ قېتىم ئانا ئالدىدا قانچلىك ياشىنىڭ تۆكۈلگىنى ، ئاننىنىڭ باغرىنى قانچلىك تىلغىننى ھېس قىلالمايتتى . بەلكىم ئۇ بۇرۇنقى كىچىكلىكىنى ئۇنتۇپ كەتكەندۇ ، بىراق ئۇ ئانا ئالدىدا مەڭگۈ بالا ، مەڭگۈ گۈدەك ئىدى .
ياتاق ئۆيىنى ئىچىدىن تاقىغان ئەقىدە ئۆزىنى بېسىۋالغان ھالدا ئالدىدىكى خالتىنى ئاچتى . بىر خاتىرە ، بىر دىيۇملۇق رەسسىمدىن بىر پارچە ، چوڭايتىلغان رەسسىمدىن بىر پارچە ، چاچ بوغقۇچتىن بىر تال ، بىر نەچچە تال پارچە ۋاراق ( ئەقىدە يازغان لىرىكا بولسا كېرەك ) ، قېپقىزىل رەڭدە بويالغان يۈرەك شەكلى  ..... ئەقىدە كۆرگەنلىرىگە ئىشنىش ياكى ئىشەنمەسلىك ھەققىدە كۆپ ئويلاندى  ئەڭ ئاددىيسى دىلشات چاخچاق قىلىپ ئۇنىڭدىن سوراپ ئېلىۋالغان ، ئەقىدەنىڭ ئېلۋېتىلگەن تىرنىقىمۇ تۇراتتى ، دىلشاتقا شۇ ۋاقىتتا ‹‹ نىممىمۇ قىلارسەن بۇ مەينەت تىرناقنى ؟ ›› دىگىنىدە دىلشات كۈلۈپ تۇرۇپ ‹‹ قانداق قىلاتتىم ، ساقلاپ قويۇپ تويۇڭدا ياكى بولمىسا شۇنداق چوڭراق كۈنلەرنىڭ بىرىدە قايتۇرۇپ بېرىمەن ، شۇ ۋاقىتتا قانداق قىلىسەن ؟ ›› دىگەن ئىدى ، ئەقىدە بولسا كۈلگىنىچە ‹‹ ساراڭ  ، تويۇمغا شۇ مەينەت تىرنىقىمنى كۆتۈرۈپ بارامسەن ؟ ›› دەپ كۈلۈشۈپ كەتكەن ئىدى . بىراق ئەقىدە بۇ تىرناقنى ئۇنتۇپ قالغان ئىدى . ئەمدىلىكتە بولسا ئىچىدە كۈلمەكتە ، شۇ كىچىككىنە چاخچاق دەپ ئويلىغان دىلشاتنىڭ ھەر بىر سۆزلىرىنىڭ قەسەم بولۇپ يۈرگىگە ئۇيۇپ كېتىدىغىنىنى ئۇ شۇنچە كېچكىپ چۈشۈنۈپ يەتكەن ئىدى . چاچ بوغقىچنىمۇ پويىزدا دىلشاتنىڭ يۈتتۈرۋەتتىم دىگەن باھانىسى بىلەن بۈگۈنگە كەلگەندە بۇ خالتىدىن چىقىشىنى ئۇ خىيالىغىمۇ كەلتۈرمىگەن ئىدى . ئاۋۇ پارچە ۋاراقتىكى لىرىكىلارنى ئوقۇپ ئۆزىمۇ شىرىن ھېسلار ، قورقۇنۇچ ئىچىدە قېلىپ قالغىننى ئويلاپمۇ كۆرمىگەن ئىدى . ئىشنى قىلىپ دىلشات مەيلى چاخچاق قىلىپ ئېيتقان سۆزلىرى بولسۇن ياكى ئۇنى ئالدىغانلىرى بولسۇن ھەممىنىڭ بۈگۈنكى مۇشۇ كۈنگە كېلىپ ئاشكارلىنىدىغىننى ئەقىدە ئويلاپمۇ باقمىغان ئىدى . ئاشۇ چاخچاقلىرىنىڭ قەسەم كەبى ئويۇلغان يۈرەكنىڭ ئەمدى سوقمايدىغىننى ئەقىدە چۈشۈنۈپ يەتكىنىدە ئۆزىنى يالغۇز ھېس قىلىپ قالدى ، بىراق بۇ يالغۇزلۇقنىڭ دىلشاتقا قانچىلىك ھەمراھ بولغىننى ئۇ بىر ئۆمۈر چۈشەنمەسلىكى مۇمكىن ئىدى .
ئەقىدە دىلشاتنىڭ خاتىرىسىنى قولىغا ئالدى ، ئىككىلىنىش ئىلكىدە خاتىرىنى شۇنچە تەستە ئاچتى ، چۈنكى بۇ خاتىرىنىڭ ئىچىدە ئەقىدە ئۆزىمۇ چۈشەنمەيدىغان يەنە بىر دىلشاتنىڭ ياشايدىغىننى ، ئۆزىمۇ بىلمەيدىغان بىراۋلارنىڭ بۇ خاتىرىگە ئىز سالغىننى بىلىۋالغاندەك ئىدى .

‹‹ مېنى يۆلەنچۈك قىلما ، تاغ كەبى سېنى يۆلەپ تۇرالمايمەن ، مېنى كۆچەت قىلما ، قارغاي كەبى يىلتىز تارتالمايمەن ، مېنى پەقەتلا شامال قىل ، سەن مۇھتاج بولغاندا ساڭا ئۆزۈمنىڭ مەۋجۇتلۇقىنى ھېس قىلدۇرىمەن ، سېنى يەڭگىل يەلپۈشلىرىم بىلەن مەدەت بېرىمەن ! ››
خاتىرىنىڭ بىرىنجى بېتىگە يېزىلغان سۆزلەردىن ئەقىدە گاڭگىراپلا قالدى ، دىلشاتنىڭ ئەنە شۇ ئۈمۈدسىزلىك ئىچىدە ئۆزىگە ئۈمۈد بېرىپ يول كۆرسىتىپ كەلگەن شۇ كۈچنىڭ زادى نىمە تەرپىدىن بېرىلگىنىگە ھەيران بولماقتا ، ئۆز كۆڭلىدە شۇنچىلىك  بەردەم ، ئىچىگە سايا تاشلىغىنىدا كۆزلىرىدىن ھەممىنى كۆرەلەيدىغان بۇ يىگىتنىڭ زادى نىمە سەۋەپتىن ئۆزىگە شۇنچە چوڭ ئۈمۈدلەرنى ئېلىپ كېلەلىگىننى ئۇ خىيالىدىمۇ ھېس قىلماقچى بولاتتى . بىراق بۇ سۆزلەرنى كۆرگەندىن كىيىن ئۇ دىلشاتنىڭ ئۇنچە بەردەم ئەمەس ئىكەنلىكىنى ھېس قىلماقتا ئىدى .
‹‹ ماڭا ئىچ ئاغرىتما ! مەن شۇ ھالغا چۈشۈپ قالغان تەقدىرىدىمۇ ماڭا ئىچ ئاغرىتما ! چۈنكى ئۆزۈم سۆيگەن ، ئۆزۈم ياخشى كۆرگەن كىشىلەر تەرپىدىن ئىچ ئاغرىتىلىشنىڭ ماڭا قانچلىك خورلۇق ئېلىپ كېلدىغىننى ھېس قىلالمايسەن . ماڭا ئاغرىتماقچى بولغان شۇ ئىچىڭنى كوچىدىكى دىۋانىلارغا ، كۆڭلىڭگە مەن كەبى ئىز سالالمايۋاتقان ئاشۇ بىچارىلەرگە ئاغرىتىۋال ! ››
ئەقىدە تېخىمۇ بەكرەك گاڭگىرىدى ، ئۇ بۇ سۆزنى بۇرۇنمۇ دىگەن ، كۆزلىرىنى ئەقىدەنىڭ كۆزىگە مىختەك قاداپ تۇرۇپ دىگەن ئىدى ، توغرا شۇ ۋاقىتتا دىلشاتنىڭ كۆزىدىن ياش پىلدىرلاپ قالغان بولسىمۇ شۇنچىلىك بەردەم ھالدا ئۆزىگە ئىشەنگەن ھالدا دىگەن ئىدى .
خاتىرىنىڭ ئۈچۈنجى بېتىدىن باشلاپ دىلشاتنىڭ كۈندىلىك خاتىرىسى ئىدى ، كۆپۈنچىسى دىلشاتنىڭ ئۆزىنى بەزلەش ، ئۆزىنى گوللاش ئىكەنلىكىنى كۆرگەن ئەقىدە كۆزلىرىگە يەنە بىر قېتىم ئىشىنەلمىدى . ئىشىنىشكە ئۇ راستىنلا ئامالسىز ئىدى ، چۈنكى ئالدىدا پۈتۈن ، كۈلۈپ يۈرگەن دىلشاتنىڭ ئەمەليەتتە بولسا ئىچى تۈتۈنگە قان ياشقا تولغان شۇنچە ئاجىز ، خۇددى پورلىشىپ كەتكەن قېرى سۆگەتكە ئوخشاپ قالغىننى ئۇ مانا بۈگۈنكى كۈنگە كەلگەنگە ھېس قىلدى .
خاتىرىدىكى ھەممىنى ئوقۇپ گاھىدا يىغلاپ ، گاھىدا كۈلۈپ كېتىۋاتقان ئەقىدەنىڭ كۆزلىرىدە بىر خىل ئىشەنمەسلىك چىقىپ تۇراتتى ، بىراق ئەمەليەت ، كۆ ئالدىدىكى ئېقى قەغەز ، قارىسى سىياھ بولغان شۇ خەتلەرنى ئۇ يادىلاۋتقاندەك شۇنچە ئىنچىكىلىك بىلەن ئوقۇۋاتاتتى ، خۇددى ئوقۇغۇچىسىنىڭ يۈز نۇمۇرلۇق قەغىزىدىن قۇسۇر ئىزدەۋاتقان ئوقۇتقۇچىدەك ھەر بىر سۆزنى شۇنچە ئەستايىدىللىق بىلەن ئوقۇۋاتاتتى .
‹‹ خاتىرىنى يا ئوقۇيسەن ، يا ئوقۇمايسەن ! شۇ بىر نەچچە كۈنلىك ھالىتىمنى ئويلاشقىمۇ جۈرئەت قىلالمايمەن ، بەلكىم تەقدىرنىڭ ماڭا قىلغان چاخچىقى ئەنە شۇنداق بولسا كېرەك . بىراق شۇ ئېسىڭدە بولسۇن ، بۇنىڭ ھەممىنى ئۆزۈم خالاپ قىلغان ، سەندىن شۇنداق نەرسىلەرنى تەلەپ قىلدىم ، چۈنكى سېنىڭ ماڭا مەڭگۈلۈك ئۆچ بولۇشۇڭ ئۈچۈن ، سېنى يىغلاتتىم ، بۇندىن كىيىن چىرايىمغا كۈلۈپ تۇرۇپ قارىيالىشىڭ ئۈچۈن ، سەندىن ئەپۇ سورىمايمەن ، ساڭا چىن دىلىمدىن چىقىرىپ بەخىتمۇ تىلىيەلمەيمەن ، بۇ خاتىرىنى ئوقۇۋاتقان ۋاقتىڭدا مەن بەلكىم بىراۋلارنىڭ تۇرمىشىغا يانداشقان ياكى بۇ دۇنيادىن خوشلاشقان بولىمەن ، سەن ھامىنى بۇ خاتىرىنى ئوقۇيسەن ، شۇ ۋاقىتتا مېنى سەن قىلچىلىك چۈشەنمەيدىغىنىڭنى ھېس قىلىسەن . ››
ئەقىدە بېشىنى كۆتۈرۈپ دېرىزە سىرتىغا قارىدى ، سىرىت ئاللىقاچان قارا لىباسىنى ئارتىپ ئۆز ئۇيقۇسۇنى ئۇخلاش ئۈچۈن مۈكۈنمەكتە ئىدى ، يىراقتىن يۇپۇرماقلارنىڭ شىرىلداشلىرى ، شامالنىڭ دېرىزىنى دەممۇ دەم چېكىشى ئەقىدەنى باشقا خىياللارغا سۆرىمەكتە ئىدى .


داۋامى ئىنكاس شەكلىدە يوللىنىدۇ .
مۇنبەرداشلارنىڭ توغرا چۈشۈنۈشىنى ئۈمۈد قىلىمەن ، ھېكايىنىڭ ئاساسىي قۇرۇلمىسى تولۇق بولغان بىلەن بېيتىدىغان مەزمۇنغا سەل كۈچگىلى سېلىۋاتىدۇ ، ھەم بۈگۈن يۈز بەرگەن بەزى ئىشلار سەۋەبلىك دىققىتىمنى تازا مەركەزلەشتۈرەلمەيۋاتىمەن ، ئەمما ھېكايىنى چوقۇم سىلەرنىڭ كۆڭلۈڭلارغا ياققىدەك ئاخىرلاشتۇرغىچە تىرشىمەن . دوسلارنىڭ ، مېنى قوللاپ قۇۋەتلىگەن مۇنبەرداشلارغا رەھمەت ئېيىتمەن !
