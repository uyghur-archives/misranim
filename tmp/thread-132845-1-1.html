                                    ئولتۇرۇپ مەن خىلۋەت بۇ جايدا،
                                    ئەتراپىمغا نەزەر تاشلىدىم.

                                    خىيال چىرماپ ئالدى جىسمىمنى،
                                   بايانىمنى ئەمدى باشلىدىم.

                                  جىمجىت خىيال سۈرۈپ ئولتۇرۇپ،
                                  كەچمىشىمنى بىر رەت ئەسلىدىم.
                                  ھاياجانغا تولدى يۈرۈكۈم،
                                   ئىختىيارسىز كۆزۈم ياشلىدىم.

                                    بالىلىقىمنى ئەسلىسەم شۇنداق،
                                    شوخ ئىكەنمەن ، ئاجايىپ كەپسىز.
                                   ئۆتەر ئىكەن كۈنلۈرۈم ئويناپ،
                                   دەرىت-ئازاپ يوق، شۇنچىلىك غەمسىز.

                                   ئوينىشاتتۇق بىر توپ بالىلار،
                                   تال چىۋىقنى ئات ئىتىپ مىنىپ.
                                   ئۇندىن ئارتۇپ بەخىت يوق ئىكەن،
                                   ئوينىۋالساق قانغۇچە كۈلۈپ.

                                 شۇنداق تاتلىق ئىكەن بالىلىق،
                                 قانماس ئىكەن ھېچكىم مىھرىگە.
                                  ئەسلىگەندە سەبى چاغلارنى،
                                  گۈل قۇناركەن ئىنسان چىھرىگە.

                                 ۋاقىت گۇيا ئاتقان ئوق كەبى،
                                 ئېلىپ كەتتى بالىلىقىمنى.
                                 تېخى ئويناپ قانمايلا تۇرۇپ،
                               كۆتۈپ قالدىم بۇ ياشلىقىمنى.

                               ئاڭلىدىم مەن بەزى ياشلاردىن،
                               ياشلىقىڭدا ئوينا راسا دەپ.
                               ۋىجدان مىنى تارتىپ يېنىغا،
                               دەۋەت قىلدى، مەغرۇر ياشا دەپ.

                               تۇنۇپ يەتتىم نۇرغۇن ئىشلارنى،
                              قەدەم قۇيۇپ ياشلىق سىپىگە.
                               يېزىپ ماڭدىم ھەر بىر ئىشىمنى،
                              ھاياتىمنىڭ تارىخ بىتىگە.

                            ھايات دىگەن تۆت كۈنلۈك پەقەت،
                           شۇڭا بىكار ئۆتمىسۇن ياشلىق.
                           ئۆمۈر دىگەن خۇشال ئۆتمەمدۇ؟
                           ئەقىل ئاڭا بولسا گەر باشلىق.
