مۇشۇ رامىزاننىڭ خاسىيتىگە مەن راسىتلا قايىل بولۇپ قالدىم جۇمۇ...ئاجايىپ ئىش.نىيەت قىلىپ روزا تۇتساق پىژغىرىم ئىسسق كۈنلەردىمۇ ئىغىزىمىز ئازىراق بەتتام بولۇپ قالمىغاندا باشقا ئالاھىدە ئىشلار  يۈز بەرمەيدىكەن.بۇ ئوتتۇز كۈنلۈك روزىنىڭ مۇنداق تەرىپىنىمۇ ھىس قىلدىم.يەنى ،ھازىر دۇنيانىڭ بەزى جايلىرىدا مۇستەققىلىق،ۋەتەننىڭ شان شەرىپى ئۈچۈن كۈرەش بولىۋاتىدۇ.مەسلەن قېرىم مۇستەققىل بولىمىز دەپ،چىچىنىيە مۇستەققىل بولىمىز دەپ،پاكىستان،ئافغانىستان ئىراق ۋە سۈرىيە قاتارلىق رايۇنلاردا بولسا دۆلەتنىڭ شان شەرىپى ئۈچۈن ئۇرۇشلار بولىۋاتىدۇ.ئۇرۇش رايونىدىكى خەلىقلەر تۇرمۇشتا قىيىنچىلىقلار تارتىدۇ.ئەلۋەتتە بىر رايوندا ئۇرۇش خەۋىپى كۈچىيدىكەن ئۇ جاي ۋە شۇ جايغا يىقىن خۇشنا رايۇنلاردا ئاشلىق ئىچىملىك سۇ ۋە تببىي دورىلار قىممەت ياكى كاسات بولىدۇ.ئاچلىقتىن ئۆلۈپ كىتىدىغانلار كۆرۈلىدۇ.بىراق روزا تۇتۇپ يۈررۈپ كۆنۈپ كەتكەن مۇسۇلمان قەۋىمى ئۈچۈن بىر قانچە كۈندە بىرەر ۋاق تاماق يىيىشتىن ئۇلارنىڭ سالامەتلىكىنى ناچارلاشتۇرۋىتىش ناتايىن بىر ئىش ئىكەن.ئەمما ئوتتۇز كۈن ئەمەس،يىرىم كۈن قۇرساققا بىر نىمە يىمىسە چىدىيالمايدىغان ئادەملەر بولسا ئىشى چاتاق بولىدىكەن...باشقا مەقسەتتە دىمىدىم جۇمۇ،ئوتتۇرا شەرىق ئەللىرى مۇسۇلمانلارنى تەشكىل قىلغان ئەرەپ رايونى،بۇ رايۇندا ئۇرۇش بولىۋاتىدۇ.بۇ ئۇرۇش رايۇنىدا بىر پۇخرانى ئاچلىقتىن ئۆلۈپتۇ دەپ ئاڭلاپ باقمىدۇق.ئەلۋەتتە شۇنداق بولغانلار بولىشى مۇمكىن.(قايسى بىر ۋاقىتتا بىر خەۋەردە:سۈرىيەنىڭ ھەلەب رايۇنىدا كىشىلەر ئاچ قالغانلىقتىن شۇ يەرلىك مەسچىدنىڭ ئىمامى كىشىلەرنى مۆشۈك گۆشىنى يىسە بولىدىغانلىقىنى،ئەمما بۇ گۆشنىڭ پەقەت جاننى ساقلاش ئۈچۈن يىيىلگىنى ئۈچۈن گۇناھ بولمايدىغانلىقىنى خەۋەر قىلىپتىكەن.شۇ چاغدا بۇ ئىمامنىڭ گىپىنىڭ ئاساسى بارلىقىنى ھىس قىلغانتىم.دىمەكچى بولغىنىم جان ساقلاش زۆرۈر،بۇ ئالدىنقى شەرىت،لىكىن شارائىت بولمىغان جايدا روزا تۇتۇپ كۆندۈرگەن ئادەم بىلەن پەقەتلا روزا تۇتۇپ باقمىغان ئادەملەر بىرەر ئىككى كۈنلۈك ئاچلىق مەزگىلىدە بىلىنىشى مۇمكىن.

رامىزان مەزگىلدىكى چەكلەش خارەكتىرلىك قائىدىلەر بۇ تاماكىغا بولغان يەكلەشنى كۈچەيتىدۇ.رامىزان تۇتقان ئادەم كىچىسى ھەم ئىپتاردىمۇ بۇنى ئىغىزغا ئالمىسا ھەپتە ئىچىدە پۇراش سىزىمى ئەسلىگە كىلىدۇ.ئىككى ھەپتىدىن بىر ئايغىچە چىرايى سۈزۈلىدۇ.(تاماكا چەككۈچىنىڭ چىرايى بىر ئاز تاتىراڭغۇ كىلىدۇ.)قۇرساق ئاچىدۇ.ئىشتاھا كۈچىيىدۇ.بەدەندىكى ئۆزگىرىشنى ئىنىق ھىس قىلغىلى بولىدۇ.پۇت قوللىرىڭىزغا كۈچ كىرىۋاتقاندەك ھىس قىلىدىكەنسىز.بارغانچە تاماكا چەككەن كىشىدىن بىزارلىق ھىس قىلىسىز.پۇراقتىن سەسكىنىسىز....ئىشقىلىپ تاماكىدىن قۇتۇلساق مۇشۇنداق ئالامەتلەر يۈز بىرىدۇ.
ئۆزىڭىز ياخشى كۆرگەن ئادەمنىڭ ئېللىتى خېسلەت بېلىنسە،ئۆزىڭىز يامان كۆرگەن ئادەمنىڭ خېسلىتى ئېللەت بېلىنىدۇ.
