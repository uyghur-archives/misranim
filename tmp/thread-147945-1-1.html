        بەزىدە ئۆزۈمگە ، قېرىنداشلىرىمغا ۋە ئۇلارنىڭ تۇرمۇشىغا قاراپ ،چوڭقۇر خىياللارغا چۆكۈپ كېتىمەن.تەغدىر ئۈستىدە بېشىم تاشتەك قاتقۇچە قايتا-قايتا ئويلاپ كېتىمەن.بۈگۈنمۇ مېنى شۇنداق ئويغا سالىدىغان بىر ئىش يۈزبەردى.مەن يەنە خىياللار ئىلكىدە،تەپەككۇر قۇشلىرىم قارا ئاسماندا ئۇچۇشقا باشلىدى.
            ئاچام توي قىلغىلى تۆت يىل بولۇپ قالدى.دەسلەپكى ئىككى يىل ئۇلار شۇنداق خاتىرجەم،تىنىچ-ئامان،بەختلىك تۇرمۇش كەچۈردى.كۈلكە ئاۋازى ئۇلارنىڭ ئۆيىدىن ئايرىلمايتتى.مانا ھازىر ئۇلارمۇ ئىككى ئوغۇل پەرزەنتلىك بولۇپ قالغان ئىدى.بالىلارنىڭ چوڭى ئۈچ ياشقا،كىچىكى بىر ياشقا كىرىپ قالدى.بىلمىدىم،قاچانلاردا تەغدىرنىڭ قارا بۇلۇتى ئۇلارنىڭ بېشى ئۈستىگە كېلىپ توختىدىكىن!؟قېينى ئاكام ئۆيىنى قامداش،بالىلىرىنى بېقىش ،قىسقىسى ئۆزىنىڭ ئەرلىك ھەم ئاتىلىق مەسئۇلىيىتىنى ئادا قىلىش ئۈچۈن توختىماي ئىشلەيتتى.قاچانلاردىدۇر بىر ۋاقىتلاردا بېقىنىم ئاغرىۋاتىدۇ دەپ يۈرەتتى.بىراق ئۆز ۋاقتىدا  بۇ ئاغرىقلارنى پەقەت خىزمەتتە ھېرىپ-چارچىغانلىقتىن بولسا كېرەك دەپ ئۆزىمۇ ئۇنچە بەك كۆڭۈل بۆلۈپ كەتمىگەن ئىدى.بىراق ئاغرىق بارغانچە كۈچىيىۋاتاتتى.
       بىر كۈنى ئاخشىمى قېينى ئاكام ئىشتىن چۈشۈپ سارغايان پېتى ئۆيگە كىرىپتۇ.ۋە شۇ يېتىپ ئەتىسى ئورنىدىن تۇرالماس بولۇپ قاپتۇ.ئاچام ئۆيدە ئارام ئالدۇرۇپ،دورا قىلىپ بىر ئىككى كۈندىن كېيىن ئاكام ئاۋايلاپ ماڭالايدىغان بولدى.كېيىن ئاكامنى دوختۇرخانىغا ئاپاردۇق.تەكشۈرۈش نەتىجىسىدە ئاكامنىڭ بەل قىسمىدا ئۆسمە ئۆسۈپ قالغانلىقى چىقتى.شۇنىڭ بىلەن ئۇرۇق-تۇققانلار يىغىش قىلىپ ئىككى-ئۈچ تۈمەندەك پۇل بىلەن ئاكامنى ئوپراتسىيە قىلدۇردۇق.ئوپراتسىيە ئوڭۇشلۇق بولدى.بىراق ئاكامنىڭ كېسىلى خېلى ئېغىر بولغانلىقى ئۈچۈنمۇ ئوپراتىسىيەدىن كېيىن بەكلا ئاجىزلاپ كەتكەن ئىدى.دوختۇر ئالتە ئايدىن بىر يىلغىچە ئېغىر ئىش قىلماسلىقنى،ئېغىر نەرسىلەرنى كۆتۈرمەسلىكنى ئېيتتى.مانا كۈنلەر ئۆتىۋاتىدۇ،تۈۋرۈكى ئاجىز بولغان ئۆينىڭ ئۆرلۈپ چۈشۈش ئېھتىماللىقى يۇقۇرى بولغىنىدەك ئاچاملارنىڭ تۇرمۇشىمۇ بارا-بارا تەسلەشتى.ئاغرىقچان يولدىشى ھەم يەنە ئىككى بالىنى بېقىش ئاچامنىڭ زىممىسىگە يۈكلەندى.ئەلۋەتتە بەزىدە  ئاكاممۇ چىداپ تۇرالماي،ئۆزىگە ئۆزى ئىچى پۇشۇپ ئاچامغا ياردەملىشىپ كېتەتتى.ئاچام،بىچارە ئاچام جاپا چېكىۋاتاتتى،بەكلا جاپادا قالغان ئىدى.گاھى كۆزى ياش پېتى نۇرغۇن خىياللارغا چۆكۈپ كېتەتتى.ئاچامنىڭ چوڭ بالىسى ھەنىم بولسا ھېچقانداق كېسە ل بولماي ئااسانلا چوڭ بولۇپ قالدى،ئۈچ ياشقا كىرىپ قالغاچقىمۇ ئىشقىلىپ يېشىغا ماس كەلمىگەن ھالدا خېلى ئىشلاردا ئاچامنى ئاۋارە قىلمايتتى.بىراق ئۇلارنىڭ كىچىك بالىسى بەكلا ئاغرىقچان بولۇپ قالدى.
   ئاكام ئاغرىقچان بولۇپ قالغاندىن كېيىن خىزمىتىدىن ئايرىلغان ئىدى.ئاكامنى داۋالىتىش ھەم ئۆينىڭ ئۇششاق-چۈششەك خىراجەتلىرى بولۇپ ئۇلارنىڭ ئۆيىنى خېلى ھالسىرىتىپ قويغان ئىدى. كىچىك بالىسىنى دوختۇرغا ئاپىرىش ئۈچۈن ئاچام بىلەن قېينى ئاكام ئۇنىڭ بۇنىڭدىن قەرز ئالدى.بىراق بالىنىڭ كېسىلى بەك ئۆزگۈرىشچان ئىدى،تۇرۇپلا ياخشى بولۇپ قالاتتى ،تۇرۇپلا يەنە بەك ئېغىرلاپ كېتەتتى.كۈنلەر ئۆتۈۋاتىدۇ،تۈنلەر ئۇيقۇسىز.نەچچە نەچچە قېتىم كەچ كىرىشى بىلەن ئاچام بىلەن ئاكام ئىككىسى ئالدىراپلا دوختۇرخانىغا كېتىپ قالغان ئىدى.چوڭ ئوغلى ئويناپ كىرىپ ئاتىسى بىلەن ئاپىسىنىڭ يوقلىقىنى كۆرۈپ بولۇشىغا يىغلاپ،قاچانلاردىدۇر ھېرىپ چارچاپ ئۇخلاپ قالاتتى.ئۇخلىسىمۇ قاتتىق كارۋات ئۈستىدىلا ياكى داق يەردىلا ئۇخلاپ قالاتتى.بىزنىڭ ئۆي بىلەن ئاچاملارنىڭ ئۆينىڭ ئارلىقى خېلى بار ئىدى. شۇڭا بىز بۇنداق ئىشلاردىن  بىخەۋەر قالاتتۇق.بەزىدە ئاشۇ ئۈچ ياشلىق ئۇكامغا بەكلا ئىچىم ئاغرىيتتى،ئاغرىقچان بولۇپ قېلىپ يۈرۈكى ئازابتىن ئۆرتۈنىۋاتقان،ئاتىلىق مەجبۇرىيىتىنى ئادا قىلالماي پۇچىلىنىۋاتقان ئاكامغا،بىر ئائىلىنىڭ بارلىق يۈكىنى ئۈستىگە ئېلىپ،كۆزلىرى ياش پېتى ئۇ ئىش بۇ ئىش بىلەن يۈرگەن ئاچامغا بەكلا ئىچىم ئاغرىيتتى.ئىچىم پىچاق بىلەن تىلغاندەك ئېچىشاتتى.ئۆزۈمنىڭ ئوقۇش يېشىدا بولۇپ قالغانلىقىمغا،ئاشۇ ئاجىزەلەرگە ئازراقمۇ ياردەم قىلالمايۋاتقانلىقىمغا تولىمۇ،تولىمۇ پۇشايمان قىلاتتىم.
    ئەمدى مېنى بۈگۈن مۇشۇ ئىشلارنى قايتا ئويلاشقا سېلىپ ،خىياللار دېڭىزىغا غەرق قىلغان ئىشقا كەلسەم.بۇ يىل بىزنىڭ  تۇققانلار  شەھەر ئەتراپىدىكى بىر جايدا ئاشخانا ئاچقان ئىدى.مۇشۇ ئاشخانىغا كېرەكلىك بەزى جابدۇقلارنى ئاچاملارنىڭ ھاۋالىسى بىلەن ئۇلارنىڭ ئۆيىدىن ئېلىپ كېلىپ ئىشلەتكەن ئىدۇق.ھازىر ئۇ جابدۇقلارغا ئانچا ئېھتىياجلىق بولمىغاچقا شۇ ئۆي جابدۇقلىرىنى ئاچاملارنىڭكىگە ئاپىرىۋەتمەكچى بولدۇق.بۇ ئىشقا مەن مەسئۇل بولغان ئىدىم.جابدۇقلارنى ماشىنىغا بېسىپ ئاچاملارنىڭكىگە كەلدۇق.بىر بالا يۈزلىرى توپا-چاڭ،كىيىملىرى مايلىشىپ قارداپ كەتكەن،چاچلىرى توپا،كىچىك پۇتلىرى لاي -پاتقاق ئارىسىدا قالغان بىر بالا ئاچاملارنىڭ ئۆيى ئالدىدا  تۇرۇپ،چەتتەرەكتە ئويناۋاتقان باشقا بالىلارغا قاراپ تۇراتتى.ئەنە شۇ بالا،مەن ئۇنى تونىۋالدىم ئۇ دەل مېنىڭ ئاچامنىڭ چوڭ ئوغلى ئىدى،بۇدۇنياغا تۇغۇلۇپ كۈننىڭ سېرىقىنى كۆرۈشكە تېخى مۇۋەپپەق بولالمىغان،ئاتا-ئانىسىغا تەلمۈرۈپ تۇرغان بۇ بالا دەل ئاچامنىڭ ئوغلى ئىدى.مەن ماشىنىدىن سەكرەپ چۈشتۈمدە يۈگۈرۈپ بېرىپ باشقىلار ئۆزلۈرىنى قاچۇرۇپ تۇرغان شۇ بالىنى باغرىمغا باستىم.ئاچچىق بىر يىغا كەلدى،ئۆزۈمنى ئاران تۇتىۋالدىم.بىراق نەچچە تامچە كۆز يېشىم ئىختىيارسىز ئېقىپ كەتتى.ئىچ- باغرىم پىژ-پىژ قىلىپ ئېچىشاتتى.
  ئىككى كۈن ئىلگىرى ئاچامنىڭ كىچىك ئوغلى ئاغرىپ قېلىپ جىددىيلا شەھەرلىك دوختۇرخانىغا ئاپىرىلىپ ياتاققا ئېلىندى.ئاچام بىلەن ئاكام چوڭ بالىسىنى بىر قوشنىسىغا تاپشۇرۇپلا ئالدىراپ كېتىپ قالغان ئىكەن.بالا بىر ۋاقىتلاردا ئاتىسى،ئاپىسىنىڭ يوقلۇقىنى بىلىپ يەنە ياشلىرى قۇرغىچە يىغلاپتۇ.بەلكىم ئاتا-ئانىسىنى كېلىپ قالىدۇ دېگەن ئۈمۈدتە بولسا كېرەك،قوشنىلارنىڭ ئۆيىگە كىرگىلى ئۇنىماي سىرتتىلا،ئۆزىنىڭ ئۆيى ئەتراپىدا يۈرۈپتۇ.قوشنىلارمۇ ئالدىراش بولغاچقا بالىنىڭ دېرىكىنى قىلىپ كېتەلمەپتۇ.بىچارە ئۇكام  ،ئۆيۈدىكىلەرنى ئىزدەپ تاپالماي مەھەللىدىكى بىر پاكار سوقما تام ئۈستىدە ئۇخلاپ قاپتۇ.چىڭقى چۈش ۋاقتى،شۇ ئىسسىقتا قوسىقىغا ھېچنېمە يېمىگەن بىچارە ئۇكام ئاتا-ئانىسى كېلىپ ئۆزىنى كۆرمەي ئۆتۈپ كەتمىسۇن دېگەندەك،شۇ قاتتىق تام ئۈستىدىلا ئۇخلاپ قاپتۇ.مانا بۈگۈن ئۇ مېنى كۆرۈپلا يىغلاپ كەتتى.كىچىك بالا دېگەن شۇنداق بولىدىكەن.ئۇ بىردەمدىلا ھەممىنى ئۇنتۇپ،خاتىرجەم ھالدا مېنىڭ ئەتراپىمدا غەمسىز كۈلۈپ ئويناشقا باشلىدى.گەرچە ئۇنىڭ كۈلۈپ ئوينىغانلىقىنى كۆرسەممۇ ئىچىم تېخىمۇ ئاچچىق بولۇپ كەتتى.ئاشۇ بىچارا ئۇكام،ئاتا-ئانىسىنىڭ يوقلۇقىنىمۇ ئۇنتۇپ،مەندەك بىر قېرىندىشىنى كۆرۈپلا ھەممىدىن خاتىرجەم ھالدا ئويناپ يۈرەتتى.ئەلۋەتتە ئۇنىڭ بۇ خۇشاللىقى دەل ئۇنىڭ ئۆزلىرىنىڭ بېشى ئۈستىدە يۈرگەن قارا بۇلۇتنى كۆرمىگەنلىكىدە،ئۇ ھېچنېمىنى بىلمەيدۇ،ئۇ ئاتىسىنىڭ كېسەل سەۋەبىدىن ئاجىزلاپ كېتىۋاتقانلىقىنى،ئۇكىسىنىڭ دوختۇرخانىدا كېسەل كارۋىتىدا ياتقانلىقىنى،ئاپىسىنىڭ قاتتىق ئازاب چېكىۋاتقانلىقىنى بىلمەيدۇ،ئۇ بۇلارنى ھېس قىلىپ يېتىشكە تېخى كىچىك،ئۇ غەمسىز،ئۇنىڭغا پەقەت ئاتىسى بىلەن ئاپىسى يېنىدا بولۇپ بەرسىلا بولدى،مانا بۇ ئۇنىڭ ئۈچۈن ئەڭ چوڭ بەخت.ئۇ ئويناۋاتىدۇ،مېنى مەركەز قىلىپ ئەتراپىمدا چۆرگىلەپ ئويناۋاتىدۇ.كۆزۈم ياش پېتى يالغان كۈلۈپ ئۇنى ئەركىلەتتىم،يۈزىگە سۆيۈپ قويدۇم. ماشىنىدىكى نەرسىلەرنى چۈشۈرۈپ بولۇپ قايتماقچى بولدۇق.مەنمۇ ئىشلەۋاتقان بولغاچقا تېزراق ماڭمىسام بولمايتتى.
   ماشىنىغا چىقتىم،ماشىنا ئىشىكىنىڭ يېپىلغان ئاۋازىنى ئاڭلاپ،ئۇ دەرھاللا مېنى چاقىرىپ يىغلاشقا باشلىدى.:<ئاپامنىڭ يېنىغا چېتىمەن،ئاپامنىڭ يېنىغا ئەچىتىڭ!!!>.ئۇنىڭ يىغا ئارلاش ئاۋازى،چۈچۈك تىل بىلەن ئېيتقان نالىسى يۈرۈگۈمنى پىژ قىلىپ كۆيدۈردى.ئۇكامنىڭ ئاپىسى يەنى مېنىڭ ئاچام بۈگۈن كېلىپ ئۇكامنىمۇ ئۆزلىرىنىڭ يېنىغا ئېلىپ كەتمەكچى ئىدى.ئاچاممۇ يولغا چىقىپ بولغان ئىدى.ئۇكام چۈشەنسۇن،چۈشەنمىسۇ بۇ ئەھۋاللارنىڭ ھەممىسىنى ئۇنىڭغا دېدىم.ئۇ ھە دەپ بېشىنى لىڭشىتاتتى.ماشىنىغا چىقتىم،ماشىنا قوزغالدى.بىراق ئۇكاممۇ ماشىنا بىلەن تەڭلا قوزغالدى.شوپۇر ئالدىرايتتى،مەنمۇ ھەم ئالدىرايتتىم.ئۇ كىچىك پۇتلىرى بىلەن تېز يۈگۈرەيتتى.يىغا ئاۋازى ماشىنىنىڭ ئاۋازىنى بېسىپ چۈشكەن ئىدى.<ئاپامنىڭ يېنىغا چىتىمەن،ئاپامنىڭ يېنىغا ئەچىتىڭ!>.ئۇ توختايدىغاندەك ئەمەس. مەن چىدىمىدىم،پەقەتلا چىدالمىدىم.ماشىنىنى توختاتتىم ،يۈگۈرۈپ بېرىپ ئۇكامنى قايتا باغرىمغا باستىم.ئەمدى ئۈنۈمنى،ياشلىرىمنىمۇ كونتىرول قىلالمىدىم.مەنمۇ ئۇنىڭغا ئوخشاش ئۈن سېلىپ يىغلاپ كەتتىم.شۇ تاپ مەن تەڭقىسلىقتا قالغان ئىدىم،ئىش ئورنۇمغا تېزراق ماڭمىسام بولمايتتى،بىراق ئۇكامنىڭ يىغىسىغا چىداپ تۇرالمايتتىم.ئۇنىڭ<ئاتام -ئاپام يوق بوسا بوپتۇ،سىز بىلەن چىتەي،سىز بىلەن چىتىمەن،مېنى ئەچىتىڭ،مېنى ئەچەتسىڭىز يىجلىمايمەن>دېگەن چۈچۈك تىللىرى مېنى تېخىمۇ ئىككىلەندۈرەتتى.ئۇنى ئەكەتسەم بولمايتتى،بىراق ئۇنىڭ يىغىسىغا چىدالمايتتىم.ئاخىرى ئۇنى گەپ بىلەن گوللاپ بولالماي،دۇكانغا ئەكىرىپ جىق نەرسىلەرنى ئېلىپ بەردىم.مەنمۇ  ئامالسىزئۇنى ئاچاملارنىڭ شۇ قوشنىسىنىڭ ئۆيىگە ئەكىرىپ قويۇپ ماڭدىم.ماشىنا قوزغالدى،بىچارە ئۇكام يەنە يىغلىغان پېتى يۈگۈرۈپ چىقتى.ماشىنىنىڭ ئەينىكىدىن ئۇنى كۆردۈم.ئۇ يەرگە چېچىلىۋاتقان مەن دۇكاندىن ئېلىپ بەرگەن نەرسىلەرگىمۇ قارىماي يۈگۈرەيتتى.ئۇ تۇيۇقسىز شېغىل يولدا يىقىلىپ چۈشتى،يۈرۈگۈم جىغ قىلىپ قالدى.كۆردۈم ئاچاملارنىڭ قوشنىسى چىقتى،مەن سەل ئارام تاپقاندەك ئالدىمغا قارىۋالدىم.ئارقامغا قايتا قارىمىدىم،بەلكىم مەن بەك تاش يۈرەكتۇرمەن.لېكىن كۆزۈمدىن ياش توختىماي ئاقاتتى.بىردەمدىن كېيىن ئاچام ئۇنى ئېلىپ كەتكەنلىكى توغۇرلۇق مېنى خەۋەرلەندۈرۈپ قويدى.شۇندىلا كۆڭلۈم ئارامىغا چۈشتى.ئۇھ دېدىم،ئۇھ...
